372     private static class ImageFileFilter extends FileFilter {
373 
374         @Override
375         public boolean accept(File f) {
376             String filename = f.getName();
377             if (f.isDirectory()) {
378                 return true;
379             }
380             if (filename != null) {
381                 return filename.endsWith(".jpg") || filename.endsWith(".jpeg")
382                         || filename.endsWith(".png") || filename.endsWith(".bmp");
383             }
384             return false;
385         }
386 
387         @Override
388         public String getDescription() {
389             return "*.png | *.bmp |*.jpg | *.jpeg";
390         }
391     }

3384     public static void setProxyInformation(Connection connection) {
3385         ProxyType configProxyType = Connection.ProxyType.valueByText(getCachedValue(KEY_PROXY_TYPE, "None"));
3386         if (configProxyType == null) {
3387             return;
3388         }
3389 
3390         connection.setProxyType(configProxyType);
3391         if (configProxyType != ProxyType.NONE) {
3392             String host = getCachedValue(KEY_PROXY_ADDRESS, "");
3393             String port = getCachedValue(KEY_PROXY_PORT, "");
3394             if (!host.isEmpty() && !port.isEmpty()) {
3395                 connection.setProxyHost(host);
3396                 connection.setProxyPort(Integer.parseInt(port));
3397                 String username = getCachedValue(KEY_PROXY_USERNAME, "");
3398                 connection.setProxyUsername(username);
3399                 if (getCachedValue(KEY_PROXY_REMEMBER, "false").equals("true")) {
3400                     String password = getCachedValue(KEY_PROXY_PSWD, "");
3401                     connection.setProxyPassword(password);
3402                 }
3403             } else {
3404                 logger.warn("host or\\and port are empty: host=" + host + ", port=" + port);
3405             }
3406         }

67      public void showDialog(MageDialogState mageDialogState) {
68  
69          // window settings
70          MageFrame.getDesktop().remove(this);
71          if (this.isModal()) {
72              MageFrame.getDesktop().add(this, JLayeredPane.MODAL_LAYER);
73          } else {
74              MageFrame.getDesktop().add(this, JLayeredPane.PALETTE_LAYER);
75          }
76          if (mageDialogState != null) mageDialogState.setStateToDialog(this);
77          else this.makeWindowCentered();
78  
79          showDownloadControls(false); // call to change window size
80  
81          this.setVisible(true);
82      }

133     private void enableActionControl(boolean enable, Component comp) {
134         if (enable) {
135             // restore last enable state
136             comp.setEnabled(actionsControlStates.getOrDefault(comp, true));
137         } else {
138             // save enable state and disable it
139             actionsControlStates.putIfAbsent(comp, comp.isEnabled());
140             comp.setEnabled(false);
141         }
142     }

166     public void showDownloadControls(boolean needToShow) {
167         // 2 modes:
168         //  - only message;
169         //  - message + download controls and buttons        
170         this.panelGlobal.setVisible(true);
171         this.buttonStop.setVisible(!needToShow); // stop button only for loading mode
172         this.tabsList.setVisible(needToShow);
173         this.panelCommands.setVisible(needToShow);
174 
175         // auto-size form
176         if (needToShow) {
177             this.setWindowSize(this.sizeModeMessageAndControls.width, this.sizeModeMessageAndControls.height);
178         } else {
179             this.setWindowSize(this.sizeModeMessageOnly.width, this.sizeModeMessageOnly.height);
180         }
181         this.makeWindowCentered();
182         //this.setLocationRelativeTo(null); // center screen //FIX
183 
184         // icons on tabs left side
185         setTabTitle(0, "Standard download", "/buttons/card_panel.png");
186         setTabTitle(1, "Custom download", "/buttons/list_panel.png");
187 
188         // TODO: add manual mode as tab
189         this.tabsList.getTabComponentAt(1).setEnabled(false);
190         this.tabsList.setEnabledAt(1, false);
191     }

114     private PermanentView createPermanentCard(Game game, UUID controllerId, String code, String cardNumber, int powerBoosted, int toughnessBoosted, int damage, boolean tapped, boolean transform, List<Ability> extraAbilities) {
115         CardInfo cardInfo = CardRepository.instance.findCard(code, cardNumber, false);
116         ExpansionInfo setInfo = ExpansionRepository.instance.getSetByCode(code);
117         CardSetInfo testSet = new CardSetInfo(cardInfo.getName(), setInfo.getCode(), cardNumber, cardInfo.getRarity(),
118                 new CardGraphicInfo(cardInfo.getFrameStyle(), cardInfo.usesVariousArt()));
119         Card newCard = CardImpl.createCard(cardInfo.getClassName(), testSet);
120 
121         Set<Card> cardsList = new HashSet<>();
122         cardsList.add(newCard);
123         game.loadCards(cardsList, controllerId);
124 
125         Card permCard = CardUtil.getDefaultCardSideForBattlefield(game, newCard);
126         if (extraAbilities != null) {
127             extraAbilities.forEach(ability -> permCard.addAbility(ability));
128         }
129 
130         // meld card must be a special class, see CardUtils.putCardOntoBattlefieldWithEffects
131         PermanentCard permanent;
132         if (permCard instanceof MeldCard) {
133             permanent = new PermanentMeld(permCard, controllerId, game);
134         } else {
135             permanent = new PermanentCard(permCard, controllerId, game);
136         }
137 
138         if (transform) {
139             // need direct transform call to keep other side info (original)
140             TransformAbility.transformPermanent(permanent, permCard.getSecondCardFace(), game, null);
141         }
142 
143         if (damage > 0) permanent.damage(damage, controllerId, null, game);
144         if (powerBoosted > 0) permanent.getPower().setBoostedValue(powerBoosted);
145         if (toughnessBoosted > 0) permanent.getToughness().setBoostedValue(toughnessBoosted);
146         permanent.removeSummoningSickness();
147         permanent.setTapped(tapped);
148         PermanentView cardView = new PermanentView(permanent, permCard, controllerId, game);
149 
150         return cardView;
151     }

153     private CardView createFaceDownCard(Game game, UUID controllerId, String code, String cardNumber, boolean isMorphed, boolean isManifested, boolean tapped) {
154         CardInfo cardInfo = CardRepository.instance.findCard(code, cardNumber, false);
155         ExpansionInfo setInfo = ExpansionRepository.instance.getSetByCode(code);
156         CardSetInfo testSet = new CardSetInfo(cardInfo.getName(), setInfo.getCode(), cardNumber, cardInfo.getRarity(),
157                 new CardGraphicInfo(cardInfo.getFrameStyle(), cardInfo.usesVariousArt()));
158         Card newCard = CardImpl.createCard(cardInfo.getClassName(), testSet);
159 
160         Set<Card> cardsList = new HashSet<>();
161         cardsList.add(newCard);
162         game.loadCards(cardsList, controllerId);
163 
164         Card permCard = CardUtil.getDefaultCardSideForBattlefield(game, newCard);
165 
166         PermanentCard perm = new PermanentCard(permCard, controllerId, game);
167         perm.setFaceDown(true, game);
168         perm.setMorphed(isMorphed);
169         perm.setManifested(isManifested);
170         perm.removeSummoningSickness();
171         perm.setTapped(tapped);
172         if (perm.isTransformable()) {
173             perm.setTransformed(true);
174         }
175         PermanentView cardView = new PermanentView(perm, permCard, controllerId, game);
176         cardView.setInViewerOnly(false); // must false for face down
177         return cardView;
178     }

365     private void handleCardClick(Event event) {
366         MageCard panel = (MageCard) event.getComponent();
367         if (event.getEventType() == ClientEventType.CARD_DOUBLE_CLICK) {
368             // card tap
369             if (panel.getMainPanel() instanceof CardPanel) {
370                 CardPanel main = (CardPanel) panel.getMainPanel();
371                 if (main.getGameCard() instanceof PermanentView) {
372                     // new settings must be as a new copy -- it would activate the animations
373                     PermanentView oldPermanent = (PermanentView) main.getGameCard();
374                     PermanentView newPermament = new PermanentView(
375                             (Permanent) oldPermanent.getOriginalCard(),
376                             game.getCard(oldPermanent.getOriginalCard().getId()),
377                             UUID.randomUUID(),
378                             game
379                     );
380                     newPermament.overrideTapped(!oldPermanent.isTapped());
381                     main.update(newPermament);
382                 }
383             }
384         } else {
385             // card choose
386             panel.setChoosable(!panel.isChoosable());
387         }
388         cardsPanel.redraw();
389     }

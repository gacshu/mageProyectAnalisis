75      public void showDialog(UUID roomId) {
76          this.roomId = roomId;
77          if (!lastSessionId.equals(SessionHandler.getSessionId())) {
78              lastSessionId = SessionHandler.getSessionId();
79              this.player1Panel.setPlayerName(SessionHandler.getUserName());
80              this.player1Panel.showLevel(false); // no computer
81              cbTournamentType.setModel(new DefaultComboBoxModel(SessionHandler.getTournamentTypes().toArray()));
82  
83              cbGameType.setModel(new DefaultComboBoxModel(SessionHandler.getTournamentGameTypes().toArray()));
84              cbDeckType.setModel(new DefaultComboBoxModel(SessionHandler.getDeckTypes()));
85  
86              cbTimeLimit.setModel(new DefaultComboBoxModel(MatchTimeLimit.values()));
87              cbSkillLevel.setModel(new DefaultComboBoxModel(SkillLevel.values()));
88              cbDraftCube.setModel(new DefaultComboBoxModel(SessionHandler.getDraftCubes()));
89              cbDraftTiming.setModel(new DefaultComboBoxModel(Arrays.stream(TimingOption.values())
90                      .filter(o -> !o.equals(TimingOption.NONE))
91                      .toArray())
92              );
93              // update player types
94              int i = 2;
95              for (TournamentPlayerPanel tournamentPlayerPanel : players) {
96                  tournamentPlayerPanel.init(i++);
97              }
98              cbAllowSpectators.setSelected(true);
99              cbPlaneChase.setSelected(false);
100             this.setModal(true);
101             this.setLocation(150, 100);
102         }
103 
104         onLoadSettings(0);
105 
106         this.setVisible(true);

675     private void btnOkActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnOkActionPerformed
676 
677         // get settings
678         TournamentOptions tOptions = getTournamentOptions();
679 
680         // CHECKS
681         TournamentTypeView tournamentType = (TournamentTypeView) cbTournamentType.getSelectedItem();
682         if (tournamentType.isRandom() || tournamentType.isRichMan()) {
683             if (tOptions.getLimitedOptions().getSetCodes().size() < tournamentType.getNumBoosters()) {
684                 JOptionPane.showMessageDialog(
685                         MageFrame.getDesktop(),
686                         String.format("Warning, you must select %d packs for the pool", tournamentType.getNumBoosters()),
687                         "Warning",
688                         JOptionPane.WARNING_MESSAGE
689                 );
690                 return;
691             }
692         }
693         if (tournamentType.isDraft() && tOptions.getLimitedOptions() instanceof DraftOptions) {
694             DraftOptions draftOptions = (DraftOptions) tOptions.getLimitedOptions();
695             if (draftOptions.getTiming() == TimingOption.NONE) {
696                 JOptionPane.showMessageDialog(MageFrame.getDesktop(), "Warning, you must select draft timing option", "Warning", JOptionPane.WARNING_MESSAGE);
697                 return;
698             }
699         }
700 
701         // save last settings
702         onSaveSettings(0, tOptions);
703 
704         // run
705         table = SessionHandler.createTournamentTable(roomId, tOptions);
706         if (table == null) {
707             // message must be send by server!
708             return;
709         }
710 
711         // join AI
712         for (TournamentPlayerPanel player : players) {
713             if (player.getPlayerType().getSelectedItem() != PlayerType.HUMAN) {
714                 if (!player.joinTournamentTable(roomId, table.getTableId(), DeckImporter.importDeckFromFile(this.player1Panel.getDeckFile(), true))) {
715                     // error message must be send by sever
716                     SessionHandler.removeTable(roomId, table.getTableId());
717                     table = null;
718                     return;
719                 }
720             }
721         }
722 
723         // join itself
724         if (SessionHandler.joinTournamentTable(
725                 roomId,
726                 table.getTableId(),
727                 this.player1Panel.getPlayerName(),
728                 PlayerType.HUMAN, 1,
729                 DeckImporter.importDeckFromFile(this.player1Panel.getDeckFile(), true),
730                 tOptions.getPassword())) {
731             // all fine, can close create dialog (join dialog will be opened after feedback from server)
732             this.hideDialog();
733             return;
734         }
735 
736         JOptionPane.showMessageDialog(MageFrame.getDesktop(), "Error joining tournament.", "Error", JOptionPane.ERROR_MESSAGE);
737         SessionHandler.removeTable(roomId, table.getTableId());
738         table = null;
739     }//GEN-LAST:event_btnOkActionPerformed

747     private void updateNumSeats() {
748         // int numPlayers = (Integer) this.spnNumPlayers.getValue();
749         int numSeats = (Integer) this.spnNumSeats.getValue();
750 
751         if (numSeats > 2) {
752             TournamentTypeView tournamentType = (TournamentTypeView) cbTournamentType.getSelectedItem();
753             if (numSeats >= tournamentType.getMinPlayers()) {
754                 createPlayers(numSeats - 1);
755                 spnNumPlayers.setValue(numSeats);
756             } else {
757                 numSeats = tournamentType.getMinPlayers();
758                 createPlayers(numSeats - 1);
759                 spnNumPlayers.setValue(numSeats);
760                 spnNumSeats.setValue(numSeats);
761             }
762             spnNumWins.setValue(1);
763         }
764     }

791     protected String playerLoadDeck() {
792         if (fcSelectDeck == null) {
793             fcSelectDeck = new JFileChooser();
794             fcSelectDeck.setAcceptAllFileFilterUsed(false);
795             fcSelectDeck.addChoosableFileFilter(new DeckFileFilter("dck", "XMage's deck files (*.dck)"));
796         }
797         String lastFolder = MageFrame.getPreferences().get("lastDeckFolder", "");
798         if (!lastFolder.isEmpty()) {
799             fcSelectDeck.setCurrentDirectory(new File(lastFolder));
800         }
801         int ret = fcSelectDeck.showDialog(this, "Select Deck");
802         if (ret == JFileChooser.APPROVE_OPTION) {
803             File file = fcSelectDeck.getSelectedFile();
804             return (file.getPath());
805         }
806         return "";
807     }

63      public void showDialog() {
64          String serverAddress = MagePreferences.getServerAddressWithDefault(ClientDefaultSettings.serverName);
65          this.txtServer.setText(serverAddress);
66          this.txtPort.setText(Integer.toString(MagePreferences.getServerPortWithDefault(ClientDefaultSettings.port)));
67          this.txtUserName.setText(MagePreferences.getUserName(serverAddress));
68          this.txtPassword.setText(MagePreferences.getPassword(serverAddress));
69          this.chkAutoConnect.setSelected(Boolean.parseBoolean(MageFrame.getPreferences().get(KEY_CONNECT_AUTO_CONNECT, "false")));
70          this.chkForceUpdateDB.setSelected(false); // has always to be set manually to force comparison
71  
72          String selectedFlag = MageFrame.getPreferences().get(KEY_CONNECT_FLAG, "world");
73          // set the selected country/flag
74          for (int i = 0; i < cbFlag.getItemCount(); i++) {
75              String[] name = (String[]) cbFlag.getItemAt(i);
76              if (name[1].equals(selectedFlag)) {
77                  cbFlag.setSelectedIndex(i);
78                  break;
79              }
80          }
81          this.setModal(true);
82          this.setLocation(50, 50);
83          this.setVisible(true);
84      }

86      private void saveSettings() {
87          String serverAddress = txtServer.getText().trim();
88          MagePreferences.setServerAddress(serverAddress);
89          MagePreferences.setServerPort(Integer.parseInt(txtPort.getText().trim()));
90          MagePreferences.setUserName(serverAddress, txtUserName.getText().trim());
91          MagePreferences.setPassword(serverAddress, String.valueOf(txtPassword.getPassword()).trim());
92          MageFrame.getPreferences().put(KEY_CONNECT_AUTO_CONNECT, Boolean.toString(chkAutoConnect.isSelected()));
93  
94          // last settings for reconnect
95          MagePreferences.saveLastServer();
96      }

109     private void chooseAndSetServerSettingsFromOther() {
110         BufferedReader in = null;
111         Writer output = null;
112         try {
113             String serverUrl = PreferencesDialog.getCachedValue(KEY_CONNECTION_URL_SERVER_LIST, "http://xmage.de/files/server-list.txt");
114             if (serverUrl.contains("xmage.info/files/")) {
115                 serverUrl = serverUrl.replace("xmage.info/files/", "xmage.de/files/"); // replace old URL if still saved
116                 PreferencesDialog.saveValue(KEY_CONNECTION_URL_SERVER_LIST, serverUrl);
117             }
118             URL serverListURL = new URL(serverUrl);
119 
120             Connection.ProxyType configProxyType = Connection.ProxyType.valueByText(PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_TYPE, "None"));
121             Proxy p = null;
122             Proxy.Type type = Proxy.Type.DIRECT;
123             switch (configProxyType) {
124                 case HTTP:
125                     type = Proxy.Type.HTTP;
126                     break;
127                 case SOCKS:
128                     type = Proxy.Type.SOCKS;
129                     break;
130                 case NONE:
131                 default:
132                     p = Proxy.NO_PROXY;
133                     break;
134             }
135 
136             if (p == null || !p.equals(Proxy.NO_PROXY)) {
137                 try {
138                     String address = PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_ADDRESS, "");
139                     Integer port = Integer.parseInt(PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_PORT, "80"));
140                     p = new Proxy(type, new InetSocketAddress(address, port));
141                 } catch (Exception ex) {
142                     throw new RuntimeException("Gui_DownloadPictures : error 1 - " + ex);
143                 }
144             }
145 
146             if (p == null) {
147                 JOptionPane.showMessageDialog(null, "Couldn't configure Proxy object!", "Error", JOptionPane.ERROR_MESSAGE);
148                 return;
149             }
150 
151             boolean URLNotFound = false;
152             try {
153                 in = new BufferedReader(new InputStreamReader(serverListURL.openConnection(p).getInputStream()));
154             } catch (SocketTimeoutException | FileNotFoundException | UnknownHostException ex) {
155                 logger.warn("Could not read serverlist from: " + serverListURL.toString());
156                 File f = new File("serverlist.txt");
157                 if (f.exists() && !f.isDirectory()) {
158                     logger.info("Using buffered serverlist: serverlist.txt");
159                     URLNotFound = true;
160                     in = new BufferedReader(new FileReader("serverlist.txt"));
161                 }
162             }
163             List<String> servers = new ArrayList<>();
164             if (in != null) {
165 
166                 if (!URLNotFound) {
167                     // write serverlist to be able to read if URL is not available
168                     File file = new File("serverlist.txt");
169                     if (file.exists() && !file.isDirectory()) {
170                         file.delete();
171                     }
172                     output = new BufferedWriter(new FileWriter(file));
173                 }
174 
175                 String inputLine;
176                 while ((inputLine = in.readLine()) != null) {
177                     logger.debug("Found server: " + inputLine);
178                     servers.add(inputLine);
179                     if (output != null) {
180                         output.append(inputLine).append('\n');
181 
182                     }
183                 }
184             }
185             if (servers.isEmpty()) {
186                 JOptionPane.showMessageDialog(null, "Couldn't find any server.");
187                 return;
188             }
189 
190             String selectedServer = (String) JOptionPane.showInputDialog(null,
191                     "Choose XMage Public Server:", "Input",
192                     JOptionPane.INFORMATION_MESSAGE, null, servers.toArray(),
193                     servers.get(0));
194             if (selectedServer != null) {
195                 String[] params = selectedServer.split(":");
196                 if (params.length == 3) {
197                     setServerSettings(params[1], params[2], true);
198                 } else {
199                     JOptionPane.showMessageDialog(null, "Wrong server data format.");
200                 }
201             }
202         } catch (Exception ex) {
203             logger.error(ex, ex);
204         } finally {
205             StreamUtils.closeQuietly(in);
206             StreamUtils.closeQuietly(output);
207         }
208     }

625     private void btnConnectActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnConnectActionPerformed
626 
627         if (txtServer.getText().trim().isEmpty()) {
628             JOptionPane.showMessageDialog(rootPane, "Please provide a server address");
629             return;
630         }
631         if (txtPort.getText().trim().isEmpty()) {
632             JOptionPane.showMessageDialog(rootPane, "Please provide a port number");
633             return;
634         }
635         if (txtUserName.getText().isEmpty()) {
636             JOptionPane.showMessageDialog(rootPane, "Please provide a user name");
637             return;
638         }
639         // txtPassword is not checked here, because authentication might be disabled by the server config.
640         if (Integer.parseInt(txtPort.getText()) < 1 || Integer.parseInt(txtPort.getText()) > 65535) {
641             JOptionPane.showMessageDialog(rootPane, "Invalid port number");
642             txtPort.setText(Integer.toString(MagePreferences.getServerPortWithDefault(ClientDefaultSettings.port)));
643             return;
644         }
645 
646         char[] input = new char[0];
647         try {
648             connection = new Connection();
649             connection.setHost(this.txtServer.getText().trim());
650             connection.setPort(Integer.parseInt(this.txtPort.getText().trim()));
651             connection.setUsername(this.txtUserName.getText().trim());
652             connection.setPassword(String.valueOf(this.txtPassword.getPassword()).trim());
653             connection.setForceDBComparison(this.chkForceUpdateDB.isSelected() || RepositoryUtil.isDatabaseEmpty());
654             String allMAC = "";
655             try {
656                 allMAC = Connection.getMAC();
657             } catch (SocketException ex) {
658             }
659             connection.setUserIdStr(System.getProperty("user.name") + ":" + System.getProperty("os.name") + ":" + MagePreferences.getUserNames() + ":" + allMAC);
660             MageFrame.getPreferences().put(KEY_CONNECT_FLAG, ((CountryItemEditor) cbFlag.getEditor()).getImageItem());
661             PreferencesDialog.setProxyInformation(connection);
662 
663             // pref settings
664             MageFrame.getInstance().setUserPrefsToConnection(connection);
665 
666             logger.debug("connecting: " + connection.getProxyType() + ' ' + connection.getProxyHost() + ' ' + connection.getProxyPort());
667             task = new ConnectTask();
668             task.execute();
669         } finally {
670             Arrays.fill(input, '0');
671         }
672 
673     }//GEN-LAST:event_btnConnectActionPerformed

681     private class ConnectTask extends SwingWorker<Boolean, Void> {
682 
683         private boolean result = false;
684         private String lastConnectError = "";
685 
686         private static final int CONNECTION_TIMEOUT_MS = 2100;
687 
688         @Override
689         protected Boolean doInBackground() throws Exception {
690             lblStatus.setText("Connecting...");
691             setConnectButtonsState(false);
692             result = MageFrame.connect(connection);
693             lastConnectError = SessionHandler.getLastConnectError();
694             return result;
695         }
696 
697         @Override
698         protected void done() {
699             try {
700                 get(CONNECTION_TIMEOUT_MS, TimeUnit.MILLISECONDS);
701                 if (result) {
702                     lblStatus.setText("");
703                     connected();
704                     MageFrame.getInstance().prepareAndShowTablesPane();
705                 } else {
706                     lblStatus.setText("Could not connect: " + lastConnectError);
707                 }
708             } catch (InterruptedException | ExecutionException ex) {
709                 logger.fatal("Update Players Task error", ex);
710             } catch (CancellationException ex) {
711                 logger.info("Connect: canceled");
712                 lblStatus.setText("Connect was canceled");
713             } catch (TimeoutException ex) {
714                 logger.fatal("Connection timeout: ", ex);
715             } finally {
716                 MageFrame.stopConnecting();
717                 setConnectButtonsState(true);
718             }
719         }
720     }

847     private void doFastFlagSearch() {
848         Choice choice = new ChoiceImpl(false);
849 
850         // collect data from country combobox String[name][code]
851         Map<String, String> choiceItems = new LinkedHashMap<>();
852         DefaultComboBoxModel flagModel = (DefaultComboBoxModel) cbFlag.getModel();
853         String[] flagItem;
854 
855         for (int i = 0; i < flagModel.getSize(); i++) {
856             flagItem = (String[]) flagModel.getElementAt(i);
857             choiceItems.put(flagItem[1], flagItem[0]);
858         }
859 
860         choice.setKeyChoices(choiceItems);
861         choice.setMessage("Select your country");
862 
863         // current selection value restore
864         String needSelectValue = null;
865         flagItem = (String[]) flagModel.getSelectedItem();
866         if (flagItem != null) {
867             needSelectValue = flagItem[1];
868         }
869 
870         // ask for new value
871         PickChoiceDialog dlg = new PickChoiceDialog();
872         dlg.setWindowSize(300, 500);
873         dlg.showDialog(choice, needSelectValue);
874         if (choice.isChosen()) {
875             flagItem = new String[2];
876             flagItem[0] = choice.getChoiceValue();
877             flagItem[1] = choice.getChoiceKey();
878             flagModel.setSelectedItem(flagItem);
879         }
880     }

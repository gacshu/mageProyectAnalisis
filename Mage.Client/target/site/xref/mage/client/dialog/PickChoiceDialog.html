44      public void showDialog(Choice choice, String startSelectionValue, UUID objectId, MageDialogState mageDialogState, BigCard bigCard) {
45          this.choice = choice;
46          this.bigCard = bigCard;
47          this.gameId = objectId;
48  
49          setLabelText(this.labelMessage, choice.getMessage());
50          setLabelText(this.labelSubMessage, choice.getSubMessage());
51          btCancel.setEnabled(!choice.isRequired());
52  
53          // special choice (example: auto-choose answer next time)
54          cbSpecial.setVisible(choice.isSpecialEnabled());
55          cbSpecial.setText(choice.getSpecialText());
56          cbSpecial.setToolTipText(choice.getSpecialHint());
57  
58          // 2 modes: string or key-values
59          // store data in allItems for inremental filtering
60          // http://logicbig.com/tutorials/core-java-tutorial/swing/list-filter/
61          this.allItems.clear();
62          if (choice.isKeyChoice()) {
63              for (Map.Entry<String, String> entry : choice.getKeyChoices().entrySet()) {
64                  this.allItems.add(new KeyValueItem(entry.getKey(), entry.getValue(), choice.getHintType()));
65              }
66          } else {
67              for (String value : choice.getChoices()) {
68                  this.allItems.add(new KeyValueItem(value, value, choice.getHintType()));
69              }
70          }
71  
72          // sorting
73          if (choice.isSortEnabled()) {
74              this.allItems.sort((o1, o2) -> {
75                  Integer n1 = choice.getSortData().get(o1.getKey());
76                  Integer n2 = choice.getSortData().get(o2.getKey());
77                  return Integer.compare(n1, n2);
78              });
79          }
80  
81          // search
82          if (choice.isSearchEnabled()) {
83              panelSearch.setVisible(true);
84              this.editSearch.setText(choice.getSearchText());
85          } else {
86              panelSearch.setVisible(false);
87              this.editSearch.setText("");
88          }
89  
90          // listeners for inremental filtering
91          editSearch.getDocument().addDocumentListener(new DocumentListener() {
92              @Override
93              public void insertUpdate(DocumentEvent e) {
94                  choice.setSearchText(editSearch.getText());
95                  loadData();
96              }
97  
98              @Override
99              public void removeUpdate(DocumentEvent e) {
100                 choice.setSearchText(editSearch.getText());
101                 loadData();
102             }
103 
104             @Override
105             public void changedUpdate(DocumentEvent e) {
106                 choice.setSearchText(editSearch.getText());
107                 loadData();
108             }
109         });
110 
111         // listeners for select up and down without edit focus lost
112         editSearch.addKeyListener(new KeyListener() {
113             @Override
114             public void keyTyped(KeyEvent e) {
115                 //System.out.println("types");
116             }
117 
118             @Override
119             public void keyPressed(KeyEvent e) {
120                 if (e.getKeyCode() == KeyEvent.VK_UP) {
121                     doPrevSelect();
122                 } else if (e.getKeyCode() == KeyEvent.VK_DOWN) {
123                     doNextSelect();
124                 }
125             }
126 
127             @Override
128             public void keyReleased(KeyEvent e) {
129                 //System.out.println("released");
130             }
131         });
132 
133         // listeners double click
134         // you can't use mouse wheel to switch hint type, cause wheel move a scrollbar
135         listChoices.addMouseListener(new MouseAdapter() {
136 
137             @Override
138             public void mouseClicked(MouseEvent e) {
139                 if (e.getClickCount() == 2) {
140                     doChoose();
141                 }
142             }
143 
144             @Override
145             public void mouseExited(MouseEvent e) {
146                 choiceHintHide();
147             }
148         });
149 
150         listChoices.addMouseMotionListener(new MouseMotionAdapter() {
151 
152             @Override
153             public void mouseMoved(MouseEvent e) {
154                 // hint show
155                 JList listSource = (JList) e.getSource();
156 
157                 // workaround to raise on real element, not empty space
158                 int index = -1;
159                 Rectangle r = listSource.getCellBounds(0, listSource.getLastVisibleIndex());
160                 if (r != null && r.contains(e.getPoint())) {
161                     index = listSource.locationToIndex(e.getPoint());
162                 }
163 
164                 if (index > -1) {
165                     choiceHintShow(index);
166                 } else {
167                     choiceHintHide();
168                 }
169             }
170         });
171 
172         // listeners for ESC close
173         if (!choice.isRequired()) {
174             String cancelName = "cancel";
175             InputMap inputMap = getRootPane().getInputMap(JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
176             inputMap.put(KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), cancelName);
177             ActionMap actionMap = getRootPane().getActionMap();
178             actionMap.put(cancelName, new AbstractAction() {
179                 public void actionPerformed(ActionEvent e) {
180                     doCancel();
181                 }
182             });
183         }
184 
185         // window settings
186         MageFrame.getDesktop().remove(this);
187         if (this.isModal()) {
188             MageFrame.getDesktop().add(this, JLayeredPane.MODAL_LAYER);
189         } else {
190             MageFrame.getDesktop().add(this, JLayeredPane.PALETTE_LAYER);
191         }
192         if (mageDialogState != null) {
193             mageDialogState.setStateToDialog(this);
194         } else {
195             this.makeWindowCentered();
196         }
197 
198         // final load
199         loadData();
200 
201         // start selection
202         if (startSelectionValue != null) {
203             int selectIndex = -1;
204             for (int i = 0; i < this.listChoices.getModel().getSize(); i++) {
205                 KeyValueItem listItem = (KeyValueItem) this.listChoices.getModel().getElementAt(i);
206                 if (listItem.getKey().equals(startSelectionValue)) {
207                     selectIndex = i;
208                     break;
209                 }
210             }
211 
212             if (selectIndex >= 0) {
213                 this.listChoices.setSelectedIndex(selectIndex);
214                 this.listChoices.ensureIndexIsVisible(selectIndex);
215             }
216         }
217 
218         this.setVisible(true);
219     }

221     private void choiceHintShow(int modelIndex) {
222 
223         switch (choice.getHintType()) {
224             case CARD:
225             case CARD_DUNGEON: {
226                 // as popup card
227                 if (lastModelIndex != modelIndex) {
228                     // new card
229                     KeyValueItem item = (KeyValueItem) listChoices.getModel().getElementAt(modelIndex);
230                     String cardName = item.getValue();
231 
232                     if (choice.getHintType() == ChoiceHintType.CARD) {
233                         cardInfo.init(cardName, this.bigCard, this.gameId);
234                     } else if (choice.getHintType() == ChoiceHintType.CARD_DUNGEON) {
235                         CardView cardView = new CardView(new DungeonView(Dungeon.createDungeon(cardName)));
236                         cardInfo.init(cardView, this.bigCard, this.gameId);
237                     }
238 
239                     cardInfo.onMouseEntered(MouseInfo.getPointerInfo().getLocation());
240                 } else {
241                     // old card
242                     cardInfo.onMouseMoved(MouseInfo.getPointerInfo().getLocation());
243                 }
244                 lastModelIndex = modelIndex;
245                 break;
246             }
247 
248             default:
249             case TEXT: {
250                 // as popup text
251                 if (lastModelIndex != modelIndex) {
252                     // new hint
253                     listChoices.setToolTipText(null);
254                     KeyValueItem item = (KeyValueItem) listChoices.getModel().getElementAt(modelIndex);
255                     listChoices.setToolTipText(item.getValue());
256                 }
257                 lastModelIndex = modelIndex;
258                 break;
259             }
260         }
261     }
262 
263     private void choiceHintHide() {
264         switch (choice.getHintType()) {
265             case CARD:
266             case CARD_DUNGEON: {
267                 // as popup card
268                 cardInfo.onMouseExited();
269                 break;
270             }
271 
272             default:
273             case TEXT: {
274                 // as popup text
275                 listChoices.setToolTipText(null);
276                 break;
277             }
278         }
279 
280         lastModelIndex = -1;
281     }

293     private void loadData() {
294         // load data to datamodel after filter or on startup
295         String filter = choice.getSearchText();
296         if (filter == null) {
297             filter = "";
298         }
299         filter = filter.toLowerCase(Locale.ENGLISH);
300 
301         this.dataModel.clear();
302         for (KeyValueItem item : this.allItems) {
303             if (!choice.isSearchEnabled() || item.getValue().toLowerCase(Locale.ENGLISH).contains(filter)) {
304                 this.dataModel.addElement(item);
305             }
306         }
307     }

319     private void doNextSelect() {
320         int newSel = this.listChoices.getSelectedIndex() + 1;
321         int maxSel = this.listChoices.getModel().getSize() - 1;
322         if (newSel <= maxSel) {
323             this.listChoices.setSelectedIndex(newSel);
324             this.listChoices.ensureIndexIsVisible(newSel);
325         }
326     }
327 
328     private void doPrevSelect() {
329         int newSel = this.listChoices.getSelectedIndex() - 1;
330         if (newSel >= 0) {
331             this.listChoices.setSelectedIndex(newSel);
332             this.listChoices.ensureIndexIsVisible(newSel);
333         }
334     }

357     public boolean setChoice() {
358         KeyValueItem item = (KeyValueItem) this.listChoices.getSelectedValue();
359         boolean isSpecial = choice.isSpecialEnabled() && cbSpecial.isSelected();
360 
361         // auto select one item (after incemental filtering)
362         if ((item == null) && (this.listChoices.getModel().getSize() == 1)) {
363             this.listChoices.setSelectedIndex(0);
364             item = (KeyValueItem) this.listChoices.getSelectedValue();
365         }
366 
367         if (item != null) {
368             if (choice.isKeyChoice()) {
369                 choice.setChoiceByKey(item.getKey(), isSpecial);
370             } else {
371                 choice.setChoice(item.getKey(), isSpecial);
372             }
373             return true;
374         } else {
375             // special choice can be empty
376             if (choice.isSpecialEnabled() && choice.isSpecialCanBeEmpty()) {
377                 if (choice.isKeyChoice()) {
378                     choice.setChoiceByKey(null, isSpecial);
379                 } else {
380                     choice.setChoice(null, isSpecial);
381                 }
382                 return true;
383             }
384 
385             // nothing to choose
386             choice.clearChoice();
387             return false;
388         }
389     }

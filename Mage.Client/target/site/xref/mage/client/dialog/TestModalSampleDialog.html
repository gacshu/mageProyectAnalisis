33      public void showDialog() {
34          this.setModal(useModal);
35          getRootPane().setDefaultButton(buttonCancel);
36  
37          // windows settings
38          MageFrame.getDesktop().remove(this);
39          if (this.isModal()) {
40              MageFrame.getDesktop().add(this, JLayeredPane.MODAL_LAYER);
41          } else {
42              MageFrame.getDesktop().add(this, JLayeredPane.PALETTE_LAYER);
43          }
44          this.makeWindowCentered();
45          Point p = this.getLocation();
46          p.x = p.x + offsetX;
47          p.y = p.y + offsetY;
48          this.setLocation(p);
49  
50          // Close on "ESC"
51          registerKeyboardAction(e -> onCancel(), KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
52  
53          this.setVisible(true);
54      }
55  
56      private void onCancel() {
57          if (useHideDialog) {
58              this.hideDialog();
59          } else {
60              this.removeDialog();
61          }
62      }

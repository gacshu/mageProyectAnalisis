1   package mage.client.components.ability;

3   import org.jdesktop.swingx.painter.AbstractPainter;
4   
5   import java.awt.*;
6   import java.awt.geom.Area;
7   import java.awt.geom.Ellipse2D;
8   import java.awt.geom.RoundRectangle2D;
9   
10  /**
11   * Background painter.
12   *
13   * @author nantuko
14   */
15  public class BackgroundPainter extends AbstractPainter {
16  
17      private final Color bgColor = Color.black;
18  
19      static final float BACKGROUND_ALPHA = 0.6f;
20  
21      public BackgroundPainter() {
22          super();
23          setAntialiasing(true);
24      }
25  
26      @Override
27      protected void doPaint(Graphics2D g2, Object o, int i, int i1) {
28          float alpha = BACKGROUND_ALPHA;
29          Component c = (Component)o;
30          Composite composite = g2.getComposite();
31          if (composite instanceof AlphaComposite) {
32              alpha *= ((AlphaComposite) composite).getAlpha();
33          }
34  
35          g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, alpha));
36          g2.setColor(bgColor);
37  
38          RoundRectangle2D rect = new RoundRectangle2D.Double(0, 0, c.getWidth() - 1, c.getHeight() - 1, 24, 24);
39          g2.fill(rect);
40  
41          Ellipse2D ellipse = new Ellipse2D.Double(-c.getWidth(),
42                  c.getHeight() / 3.0, c.getWidth() * 3.0,
43                  c.getHeight() * 2.0);
44  
45          Area area = new Area(new Rectangle(0, 0, c.getWidth(), c.getHeight()));
46          area.subtract(new Area(ellipse));
47          area.intersect(new Area(rect));
48  
49          alpha = 0.1f;
50          if (composite instanceof AlphaComposite) {
51              alpha *= ((AlphaComposite) composite).getAlpha();
52          }
53          g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, alpha));
54          g2.setColor(new Color(1.0f, 1.0f, 1.0f));
55          g2.fill(area);
56          g2.setComposite(composite);
57      }
58  }

1   package mage.client.components;
2   
3   import java.awt.Color;
4   import javax.swing.JEditorPane;
5   import javax.swing.SwingUtilities;
6   import mage.client.util.GUISizeHelper;
7   import org.mage.card.arcane.ManaSymbols;
8   import org.mage.card.arcane.UI;
9   
10  /**
11   * Component for displaying text in mage. Supports drawing mana symbols.
12   *
13   * @author nantuko
14   */
15  public class MageTextArea extends JEditorPane {
16      private String currentText;
17      private int currentPanelWidth;
18  
19      public MageTextArea() {
20          UI.setHTMLEditorKit(this);
21          setEditable(false);
22          setBackground(new Color(0, 0, 0, 0)); // transparent background
23          setFocusable(false);
24      }
25  
26      @Override
27      public void setText(String text) {
28          setText(text, 0);
29      }
30  
31      public void setText(String text, final int panelWidth) {
32          if (text == null) {
33              return;
34          }
35  
36          if(text.equals(currentText) && panelWidth == currentPanelWidth)
37              return;
38  
39          currentText = text;
40          currentPanelWidth = panelWidth;
41  
42          final StringBuilder buffer = new StringBuilder(512);
43          // Dialog is a java logical font family, so it should work on all systems
44          buffer.append("<html><body style='font-family:Dialog;font-size:");
45          buffer.append(GUISizeHelper.gameDialogAreaFontSizeBig);
46          buffer.append("pt;margin:3px 3px 3px 3px;color: #FFFFFF'><b><center>");
47  
48          // Don't know what it does (easy italc?) but it bugs with multiple #HTML color codes (LevelX2)
49          //text = text.replaceAll("#([^#]+)#", "<i>$1</i>");
50          //text = text.replaceAll("\\s*//\\s*", "<hr width='50%'>");
51          text = text.replace("\r\n", "<div style='font-size:5pt'></div>");
52  
53          final String basicText = ManaSymbols.replaceSymbolsWithHTML(text, ManaSymbols.Type.DIALOG);
54          if (!text.isEmpty()) {
55              buffer.append(basicText);
56          }
57  
58          buffer.append("</b></center></body></html>");
59  
60          SwingUtilities.invokeLater(() -> {
61              String promptText = buffer.toString();
62              MageTextArea.super.setText(promptText);
63              // in case the text don't fit in the panel a tooltip with the text is added
64              if (panelWidth > 0 && MageTextArea.this.getPreferredSize().getWidth() > panelWidth) {
65                  String tooltip = "<html><center><body style='font-family:Dialog;font-size:"
66                          + GUISizeHelper.gameDialogAreaFontSizeBig
67                          + ";color: #FFFFFF'><p width='500'>" + basicText + "</p></body></html>";
68                  MageTextArea.super.setToolTipText(tooltip);
69              } else {
70                  MageTextArea.super.setToolTipText(null);
71              }
72              setCaretPosition(0);
73          });
74      }
75  }

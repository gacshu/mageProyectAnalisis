65      public ChoiceDialog(DlgParams params, String title) {
66          super(params);
67          this.params = params;
68          this.gameId = params.gameId;
69          this.title = title;
70  
71          cards = params.getCards();
72          isOptional = params.isOptional();
73          isCancelStopsPlaying = params.isCancelStopsPlaying();
74          isChooseAbility = params.isChooseAbility();
75  
76          in_a_row = 5;
77          rows = 2;
78  
79          // calculate max pages
80          maxPages = cards.size() / (in_a_row * rows);
81          if (cards.size() % (in_a_row * rows) != 0) {
82              maxPages++;
83          }
84  
85          initialize();
86      }

117     private void displayCards(CardsView cards, UUID gameId, BigCard bigCard) {
118         if (cards.isEmpty()) {
119             return;
120         }
121 
122         java.util.List<Component> toRemove = new ArrayList<>();
123         for (int i = getComponentCount() - 1; i > 0; i--) {
124             Component o = getComponent(i);
125             if (o instanceof MageCard) {
126                 toRemove.add(o);
127             }
128         }
129         for (Component aToRemove : toRemove) {
130             remove(aToRemove);
131         }
132 
133         java.util.List<CardView> cardList = new ArrayList<>(cards.values());
134 
135         int width = SettingsManager.instance.getCardSize().width;
136         int height = SettingsManager.instance.getCardSize().height;
137 
138         int dx = 0;
139         int dy = 30;
140 
141 
142         int startIndex = (page - 1) * (in_a_row * rows);
143         int countPerPage = (in_a_row * rows);
144         int j = 0;
145 
146         for (int i = startIndex; i < cards.size() && i < startIndex + countPerPage; i++) {
147             if (i > startIndex && i % in_a_row == 0) { // next row
148                 j++;
149                 dx = 0;
150             }
151 
152             CardView card = cardList.get(i);
153             MageCard cardImg = Plugins.instance.getMageCard(card, bigCard, new CardIconRenderSettings(), getCardDimension(), gameId, true, true, PreferencesDialog.getRenderMode(), true);
154             cardImg.setCardContainerRef(this);
155             cardImg.update(card);
156             cardImg.setCardBounds(dx, dy + j * (height + 30), width, height);
157             this.add(cardImg);
158 
159             dx += (width + 20);
160         }
161 
162         repaint();
163     }

181     private HoverButton getJButtonPrevPage() {
182         if (jButtonPrevPage == null) {
183             jButtonPrevPage = new HoverButton("", ImageManagerImpl.instance.getDlgPrevButtonImage(),
184                     ImageManagerImpl.instance.getDlgActivePrevButtonImage(),
185                     ImageManagerImpl.instance.getDlgPrevButtonImage(),
186                     new Rectangle(60, 60));
187             int w = getDlgParams().rect.width - 75;
188             int h = getDlgParams().rect.height - 90;
189             jButtonPrevPage.setBounds(new Rectangle(w / 2 - 125, h - 50, 60, 60));
190             jButtonPrevPage.setVisible(false);
191 
192             jButtonPrevPage.setObserver(() -> {
193                 if (page == 1) {
194                     return;
195                 }
196 
197                 AudioManager.playPrevPage();
198 
199                 page--;
200                 getJButtonPrevPage().setVisible(false);
201                 getJButtonOK().setVisible(false);
202                 getJButtonNextPage().setVisible(false);
203                 revalidate();
204                 displayCards(params.getCards(), params.gameId, params.bigCard);
205                 if (page != 1) {
206                     getJButtonPrevPage().setVisible(true);
207                 }
208                 getJButtonOK().setVisible(true);
209                 getJButtonNextPage().setVisible(true);
210             });
211         }
212         return jButtonPrevPage;
213     }

220     private HoverButton getJButtonNextPage() {
221         if (jButtonNextPage == null) {
222             jButtonNextPage = new HoverButton("", ImageManagerImpl.instance.getDlgNextButtonImage(),
223                     ImageManagerImpl.instance.getDlgActiveNextButtonImage(),
224                     ImageManagerImpl.instance.getDlgNextButtonImage(),
225                     new Rectangle(60, 60));
226             int w = getDlgParams().rect.width - 75;
227             int h = getDlgParams().rect.height - 90;
228             jButtonNextPage.setBounds(new Rectangle(w / 2 + 45, h - 50, 60, 60));
229 
230             jButtonNextPage.setVisible(maxPages > 1);
231 
232 
233             jButtonNextPage.setObserver(new Command() {
234                 private static final long serialVersionUID = -3174360416099554104L;
235 
236                 public void execute() {
237                     if (page == maxPages) {
238                         return;
239                     }
240 
241                     AudioManager.playNextPage();
242 
243                     page++;
244                     getJButtonPrevPage().setVisible(false);
245                     getJButtonOK().setVisible(false);
246                     getJButtonNextPage().setVisible(false);
247                     revalidate();
248                     displayCards(params.getCards(), params.gameId, params.bigCard);
249                     getJButtonPrevPage().setVisible(true);
250                     getJButtonOK().setVisible(true);
251                     if (page != maxPages) {
252                         getJButtonNextPage().setVisible(true);
253                     }
254                 }
255             });
256         }
257         return jButtonNextPage;
258     }

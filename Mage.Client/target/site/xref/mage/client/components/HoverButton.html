107     @Override
108     public void paintComponent(Graphics g) {
109         Graphics2D g2d = (Graphics2D) g;
110         if (isEnabled()) {
111             if (isHovered || textAlwaysVisible) {
112                 if (isHovered) {
113                     wasHovered = true;
114                     setCenterColor(Color.YELLOW);
115                 }
116                 g.drawImage(hoverImage, 0, 0, imageSize.width, imageSize.height, this);
117                 if (text != null) {
118                     if (textColor != null) {
119                         g2d.setColor(textColor);
120                     }
121                     if (useMiniFont) {
122                         g2d.setFont(textFontMini);
123                     } else {
124                         g2d.setFont(textFont);
125                     }
126                     textOffsetX = calculateOffset(g2d);
127                     g2d.drawString(text, textOffsetX, textOffsetY);
128                 }
129             } else {
130                 if (wasHovered) {
131                     wasHovered = false;
132                     setCenterColor(origCenterTextColor);
133                 }
134                 g.drawImage(image, 0, 0, imageSize.width, imageSize.height, this);
135             }
136             if (isSelected) {
137                 if (selectedImage != null) {
138                     g.drawImage(selectedImage, 0, 0, imageSize.width, imageSize.height, this);
139                 } else {
140                     System.err.println("No selectedImage for button.");
141                 }
142             }
143         } else {
144             g.drawImage(disabledImage, 0, 0, imageSize.width, imageSize.height, this);
145         }
146         if (topText != null) {
147             if (useMiniFont) {
148                 g2d.setFont(textFontMini);
149             } else {
150                 g2d.setFont(textFont);
151             }
152             topTextOffsetX = calculateOffsetForTop(g2d, topText);
153             g2d.setColor(textBGColor);
154             g2d.drawString(topText, topTextOffsetX + 1, 14);
155             g2d.setColor(textColor);
156             g2d.drawString(topText, topTextOffsetX, 13);
157         }
158         if (topTextImage != null) {
159             g.drawImage(topTextImage, 4, 3, this);
160         }
161         if (topTextImageRight != null) {
162             g.drawImage(topTextImageRight, this.getWidth() - 20, 3, this);
163         }
164 
165         if (centerText != null) {
166             g2d.setColor(centerTextColor);
167             int fontSize = 40;
168             int val = Integer.parseInt(centerText);
169             if (val > 9999) {
170                 fontSize = 24;
171             } else if (val > 999) {
172                 fontSize = 28;
173             } else if (val > 99) {
174                 fontSize = 34;
175             }
176             drawCenteredStringWOutline(g2d, centerText, centerTextArea, new Font("Arial", Font.BOLD, fontSize));
177         }
178         g2d.setColor(textColor);
179         if (overlayImage != null) {
180             g.drawImage(overlayImage, (imageSize.width - overlayImageSize.width) / 2, 10, this);
181         } else if (set != null) {
182             // draw only if it is not current tab
183             if (!drawSet) {
184                 g2d.setFont(textSetFontBoldMini);
185                 g2d.drawString(set, 5, 25);
186             }
187         }
188 
189         if (drawSet && set != null) {
190             g2d.setFont(textSetFontBold);
191             int w = (int) (getWidth() / 2.0);
192             int h = (int) (getHeight() / 2.0);
193             int dy = overlayImage == null ? 15 : 25;
194             g2d.translate(w + 5, h + dy);
195             g2d.rotate(-Math.PI / 2.0);
196             g2d.drawString(set, 0, 0);
197         }
198     }

204     private int calculateOffset(Graphics2D g2d) {
205         if (textOffsetX == -1) { // calculate once
206             FontRenderContext frc = g2d.getFontRenderContext();
207             int textWidth = (int) textFont.getStringBounds(text, frc).getWidth();
208             if (textWidth > buttonSize.width) {
209                 g2d.setFont(textFontMini);
210                 useMiniFont = true;
211                 frc = g2d.getFontRenderContext();
212                 textWidth = (int) textFontMini.getStringBounds(text, frc).getWidth();
213             }
214             if (alignTextLeft) {
215                 textOffsetX = 0;
216             } else {
217                 textOffsetX = (imageSize.width - textWidth) / 2;
218             }
219         }
220         return textOffsetX;
221     }
222 
223     private int calculateOffsetForTop(Graphics2D g2d, String text) {
224         if (topTextOffsetX == -1) { // calculate once
225             FontRenderContext frc = g2d.getFontRenderContext();
226             int textWidth = (int) textFont.getStringBounds(text, frc).getWidth();
227             int neededImageWidth = (topTextImage == null ? 0 : topTextImage.getWidth(this));
228             int availableXWidth = imageSize.width - neededImageWidth;
229             topTextOffsetX = (availableXWidth - textWidth) / 2 + neededImageWidth;
230         }
231         return topTextOffsetX;
232     }

366     /**
367      * Draw a String centered in the middle of a Rectangle.
368      *
369      * @param g The Graphics instance.
370      * @param text The String to draw.
371      * @param rect The Rectangle to center the text in.
372      * @param font
373      */
374     public void drawCenteredStringWOutline(Graphics2D g, String text, Rectangle rect, Font font) {
375         // Get the FontMetrics
376         FontMetrics metrics = g.getFontMetrics(font);
377         // Determine the X coordinate for the text
378         int x = rect.x + (rect.width - metrics.stringWidth(text)) / 2;
379         // Determine the Y coordinate for the text (note we add the ascent, as in java 2d 0 is top of the screen)
380         int y = rect.y + ((rect.height - metrics.getHeight()) / 2) + metrics.getAscent();
381         // Set the font
382         g.setFont(font);
383 
384         GlyphVector gv = font.createGlyphVector(g.getFontRenderContext(), text);
385         g.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
386                 RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
387         g.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
388                 RenderingHints.VALUE_FRACTIONALMETRICS_ON);
389         g.drawGlyphVector(gv, x, y);
390 
391         g.translate(x - 1, y - 1);
392         for (int i = 0; i < text.length(); i++) {
393             g.setColor(Color.BLACK);
394             g.draw(gv.getGlyphOutline(i));
395         }
396         g.translate(-x + 1, -y + 1);
397 
398     }

400     public void gainLifeDisplay() {
401         if (faderGainLife == null && doGainFade) {
402             doGainFade = false;
403             faderGainLife = new Timer(50, new ActionListener() {
404                 public void actionPerformed(ActionEvent ae) {
405                     gainX++;
406                     int alpha = Math.max(250 - gainX, 200);
407                     setCenterColor(new Color(2 * gainX, 210, 255, alpha));
408                     repaint();
409                     if (gainX >= 100) {
410                         setCenterColor(new Color(200, 210, 0, 200));
411                         gainX = 100;
412 
413                         if (faderGainLife != null) {
414                             faderGainLife.stop();
415                             faderGainLife.setRepeats(false);
416                             faderGainLife.setDelay(50000);
417                         }
418                     }
419                 }
420             });
421             gainX = 0;
422             faderGainLife.setInitialDelay(25);
423             faderGainLife.setRepeats(true);
424             faderGainLife.start();
425         }
426     }
427 
428     public void loseLifeDisplay() {
429         if (faderLoseLife == null && doLoseFade) {
430             doLoseFade = false;
431             faderLoseLife = new Timer(50, new ActionListener() {
432                 public void actionPerformed(ActionEvent ae) {
433                     loseX++;
434                     int alpha = Math.max(250 - loseX, 200);
435                     setCenterColor(new Color(250 - loseX / 2, 130 + loseX, 0, alpha));
436                     repaint();
437                     if (loseX >= 100) {
438                         setCenterColor(new Color(200, 210, 0, 200));
439                         loseX = 100;
440                         stopLifeDisplay();
441 
442                         if (faderLoseLife != null) {
443                             faderLoseLife.stop();
444                             faderLoseLife.setRepeats(false);
445                             faderLoseLife.setDelay(50000);
446                         }
447                     }
448                 }
449             });
450             loseX = 0;
451             faderLoseLife.setInitialDelay(25);
452             faderLoseLife.setRepeats(true);
453             faderLoseLife.start();
454         }
455     }
456 
457     public void stopLifeDisplay() {
458 
459         if (faderGainLife != null && gainX >= 100) {
460             faderGainLife.stop();
461             faderGainLife = null;
462         }
463         doGainFade = true;
464         if (faderLoseLife != null && loseX >= 100) {
465             faderLoseLife.stop();
466             faderLoseLife = null;
467         }
468         doLoseFade = true;
469     }

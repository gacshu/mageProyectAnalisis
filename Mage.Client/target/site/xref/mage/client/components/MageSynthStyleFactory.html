3   import javax.swing.JComponent;
4   import javax.swing.plaf.synth.Region;
5   import javax.swing.plaf.synth.SynthStyle;
6   import javax.swing.plaf.synth.SynthStyleFactory;
7   
8   /**
9    * Class makes {@link JInternalFrame} translucent background possible.
10   * This class provides fix that makes setOpaque(false) and setBackgroundColor(any color) working, 
11   * especially for Nimbus LAF that has great problems with it.
12   * 
13   * @version 0.1 16.11.2010
14   * @author nantuko
15   */
16  public class MageSynthStyleFactory extends SynthStyleFactory {
17      private final SynthStyleFactory wrappedFactory;
18  
19      public MageSynthStyleFactory(SynthStyleFactory factory) {
20          this.wrappedFactory = factory;
21      }
22  
23      public SynthStyle getStyle(JComponent c, Region id) {
24          SynthStyle s = wrappedFactory.getStyle(c, id);
25          if (id == Region.INTERNAL_FRAME) {
26              s = new TranslucentSynthSytle(s);
27          }
28          return s;
29      }
30  }

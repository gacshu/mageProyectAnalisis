3   import java.util.ArrayList;
4   import java.util.List;

6   /**
7    * @author JayDi85

9   public enum CardLanguage {
10  
11      ENGLISH("en", "English"),
12      SPANISH("es", "Spanish"),
13      FRENCH("fr", "French"),
14      GERMAN("de", "German"),
15      ITALIAN("it", "Italian"),
16      PORTUGUESE("pt", "Portuguese"),
17      JAPANESE("jp", "Japanese"),
18      KOREAN("ko", "Korean"),
19      RUSSIAN("ru", "Russian"),
20      CHINES_SIMPLE("cns", "Chinese Simplified"),
21      CHINES_TRADITION("cnt", "Chinese Traditional");
22  
23      private final String code;
24      private final String text;
25  
26      CardLanguage(String code, String text) {
27          this.code = code;
28          this.text = text;
29      }
30  
31      @Override
32      public String toString() {
33          return code;
34      }
35  
36      public String getCode() {
37          return code;
38      }
39  
40      public String getText() {
41          return text;
42      }
43  
44      public static String[] toList() {
45          List<String> res = new ArrayList<>();
46          for (CardLanguage l : values()) {
47              res.add(l.toString());
48          }
49          return res.toArray(new String[0]);
50      }
51  
52      public static CardLanguage valueByText(String text) {
53          for (CardLanguage type : values()) {
54              if (type.text.equals(text)) {
55                  return type;
56              }
57          }
58          return CardLanguage.ENGLISH;
59      }
60  
61      public static CardLanguage valueByCode(String code) {
62          for (CardLanguage type : values()) {
63              if (type.code.equals(code)) {
64                  return type;
65              }
66          }
67          return CardLanguage.ENGLISH;
68      }
69  }

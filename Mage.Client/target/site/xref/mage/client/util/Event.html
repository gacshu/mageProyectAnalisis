1   package mage.client.util;
2   
3   import java.awt.*;
4   import java.awt.event.MouseEvent;
5   import java.io.Serializable;
6   
7   /**
8    * @author BetaSteward_at_googlemail.com, JayDi85
9    */
10  public class Event implements Serializable {
11  
12      private final Object source; // can be null for non card events like popup from panel
13      private final Component component; // component that raise the event (example: MageCard)
14      private final ClientEventType eventType;
15      private final int number;
16      private final int xPos;
17      private final int yPos;
18  
19      private final MouseEvent mouseEvent;
20      // force alt mouse state for fake mouse event (example: card's selector can add card to deck by panel's button instead double click)
21      private final boolean mouseAltDown;
22  
23      public Event(Object source, ClientEventType eventType) {
24          this(source, eventType, 0);
25      }
26  
27      public Event(Object source, ClientEventType eventType, int number) {
28          this(source, eventType, number, 0, 0, null);
29      }
30  
31      public Event(Object source, ClientEventType eventType, int number, int xPos, int yPos, Component component) {
32          this(source, eventType, number, xPos, yPos, component, null, false);
33      }
34  
35      public Event(Object source, ClientEventType eventType, int number, int xPos, int yPos, Component component,
36                   MouseEvent mouseEvent, boolean mouseAltDown) {
37          this.source = source;
38          this.eventType = eventType;
39          this.number = number;
40          this.xPos = xPos;
41          this.yPos = yPos;
42          this.component = component;
43          this.mouseEvent = mouseEvent;
44          this.mouseAltDown = mouseAltDown;
45      }
46  
47      public Object getSource() {
48          return source;
49      }
50  
51      public ClientEventType getEventType() {
52          return eventType;
53      }
54  
55      public int getNumber() {
56          return number;
57      }
58  
59      public int getxPos() {
60          return xPos;
61      }
62  
63      public int getyPos() {
64          return yPos;
65      }
66  
67      public Component getComponent() {
68          return component;
69      }
70  
71      public MouseEvent getMouseEvent() {
72          if (this.mouseEvent == null) {
73              // GUI must catch a correct events, so look at stack trace and find a problem code with event generation
74              throw new IllegalArgumentException("Error, found empty mouse event.");
75          }
76  
77          return mouseEvent;
78      }
79  
80      public boolean isMouseAltDown() {
81          if (mouseEvent != null) {
82              return mouseEvent.isAltDown();
83          } else {
84              return mouseAltDown;
85          }
86      }
87  }

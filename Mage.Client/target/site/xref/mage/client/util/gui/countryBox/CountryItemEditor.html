2   package mage.client.util.gui.countryBox;
3   
4   import java.awt.*;
5   import javax.swing.*;
6   import javax.swing.border.EmptyBorder;
7   import javax.swing.plaf.basic.BasicComboBoxEditor;
8   
9   
10  /**
11   * Editor for JComboBox
12   * @author wwww.codejava.net, JayDi85
13   *
14   */
15  public class CountryItemEditor extends BasicComboBoxEditor {
16      private final JPanel panel = new JPanel();
17      private final JLabel labelItem = new JLabel();
18      private String[] editValue = new String[2];
19       
20      public CountryItemEditor() {
21          panel.setLayout(new BoxLayout(panel, BoxLayout.X_AXIS));
22          panel.setBackground(new Color(0, 100,190, 255));
23  
24          panel.add(labelItem);
25          labelItem.setAlignmentX(Component.LEFT_ALIGNMENT);
26          labelItem.setMaximumSize(new Dimension(Short.MAX_VALUE, Short.MAX_VALUE));
27          labelItem.setBorder(new EmptyBorder(0, 5, 0, 0));
28          labelItem.setOpaque(false);
29          labelItem.setForeground(Color.WHITE);
30  
31          editValue = null;
32      }
33       
34      @Override
35      public Component getEditorComponent() {
36          return this.panel;
37      }
38       
39      @Override
40      public Object getItem() {
41          return this.editValue;
42      }
43       
44      public String getImageItem() {
45          return this.editValue[1];
46      }
47  
48      @Override
49      public void setItem(Object item) {
50          if (!(item instanceof String[])) {
51              return;
52          }
53  
54          String[] newItem = (String[])item;
55  
56          editValue = new String[2];
57          editValue[0] = newItem[0];
58          editValue[1] = newItem[1];
59  
60          labelItem.setText(editValue[0]);
61          labelItem.setIcon(new ImageIcon(getClass().getResource("/flags/"+ editValue[1] + ".png")));
62      }  
63  }

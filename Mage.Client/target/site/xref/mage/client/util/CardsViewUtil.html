16      public static CardsView convertSimple(SimpleCardsView view) {
17          CardsView cards = new CardsView();
18  
19          for (SimpleCardView simple : view.values()) {
20              CardInfo cardInfo = CardRepository.instance.findCard(simple.getExpansionSetCode(), simple.getCardNumber());
21              Card card = cardInfo != null ? cardInfo.getMockCard() : null;
22              if (card != null) {
23                  cards.put(simple.getId(), new CardView(card, simple));
24              }
25          }
26  
27          return cards;
28      }
29  
30      public static CardsView convertSimple(SimpleCardsView view, Map<String, Card> loadedCards) {
31          CardsView cards = new CardsView();
32  
33          for (SimpleCardView simple : view.values()) {
34              String key = simple.getExpansionSetCode() + '_' + simple.getCardNumber();
35              Card card = loadedCards.get(key);
36              if (card == null) {
37                  CardInfo cardInfo = CardRepository.instance.findCard(simple.getExpansionSetCode(), simple.getCardNumber());
38                  card = cardInfo != null ? cardInfo.getMockCard() : null;
39                  loadedCards.put(key, card);
40              }
41              if (card != null) {
42                  cards.put(simple.getId(), new CardView(card, simple));
43              }
44          }
45  
46          return cards;
47      }
48  
49      public static CardsView convertCommandObject(List<CommandObjectView> view) {
50          CardsView cards = new CardsView();
51          for (CommandObjectView commandObject : view) {
52              CardView cardView;
53              if (commandObject instanceof EmblemView) {
54                  cardView = new CardView((EmblemView) commandObject);
55              } else if (commandObject instanceof DungeonView) {
56                  cardView = new CardView((DungeonView) commandObject);
57              } else if (commandObject instanceof PlaneView) {
58                  cardView = new CardView((PlaneView) commandObject);
59              } else if (commandObject instanceof CommanderView) {
60                  cardView = (CommanderView) commandObject;
61              } else {
62                  continue;
63              }
64              cards.put(commandObject.getId(), cardView);
65          }
66          return cards;
67      }

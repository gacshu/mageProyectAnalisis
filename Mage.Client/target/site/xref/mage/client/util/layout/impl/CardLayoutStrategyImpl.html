23      private static final int ATTACHMENTS_OFFSET_ALL_X = 12; // offset to shift all attachments
24      private static final int ATTACHMENTS_OFFSET_SINGLE_Y = 12;
25      private static final int ATTACHMENTS_MAX_COLUMNS = 10; // after 10 attachments no more offsets (will be "hide")
26  
27      final class AttachmentLayoutInfos {
28  
29          private int columns;
30          private int attachments;
31  
32          public AttachmentLayoutInfos(int columns, int attachments) {
33              this.columns = columns;
34              this.attachments = attachments;
35          }
36  
37          public int getColumns() {
38              return columns;
39          }
40  
41          public int getAttachments() {
42              return attachments;
43          }
44  
45          public void increaseAttachments() {
46              attachments++;
47          }
48  
49          public void increaseColumns() {
50              columns++;
51          }
52      }

54      @Override
55      public void doLayout(BattlefieldPanel battlefieldPanel, int battlefieldWidth) {
56          Map<UUID, MageCard> cards = battlefieldPanel.getPermanentPanels();
57          JLayeredPane mainPanel = battlefieldPanel.getMainPanel();
58  
59          // does the basic layout of rows and colums
60          int height = Plugins.instance.sortPermanents(battlefieldPanel.getUiComponentsList(), cards, battlefieldPanel.isTopPanelBattlefield());
61  
62          mainPanel.setPreferredSize(new Dimension(battlefieldWidth - 30, height));
63  
64          for (PermanentView permanent : battlefieldPanel.getBattlefield().values()) {
65              if (permanent.getAttachments() != null && !permanent.isAttachedTo()) { // Layout only permanents that are not attached to other permanents itself
66                  groupAttachments(battlefieldPanel, mainPanel, cards, permanent);
67              }
68          }
69      }
70  
71      private void groupAttachments(BattlefieldPanel battlefieldPanel, JLayeredPane mainPanel, Map<UUID, MageCard> cards, PermanentView permanentWithAttachmentsView) {
72          MageCard cardWithAttachments = cards.get(permanentWithAttachmentsView.getId());
73          if (cardWithAttachments == null) {
74              return;
75          }
76          // calculate how many vertical columns are needed and number of attachements
77          AttachmentLayoutInfos attachmentLayoutInfos = calculateNeededNumberOfVerticalColumns(0, battlefieldPanel, cards, permanentWithAttachmentsView);
78          // group by columns
79          layoutAttachements(
80                  cardWithAttachments.getCardLocation().getCardX(),
81                  attachmentLayoutInfos.getColumns(),
82                  attachmentLayoutInfos.getAttachments(),
83                  permanentWithAttachmentsView,
84                  cards,
85                  battlefieldPanel,
86                  mainPanel,
87                  cardWithAttachments.getCardLocation().getCardBounds()
88          );
89          mainPanel.setComponentZOrder(cardWithAttachments, 0);
90  
91      }
92  
93      private void layoutAttachements(int startingCardX,
94              int maxColumnLevels,
95              int ZOrder,
96              PermanentView permanentWithAttachmentsView,
97              Map<UUID, MageCard> cards,
98              BattlefieldPanel battlefieldPanel,
99              JLayeredPane mainPanel,
100             Rectangle lastAttachmentRect
101     ) {
102         // put attachments to the next level (take lastAttachmentRect and apply offsets)
103         MageCard cardWithAttachments = cards.get(permanentWithAttachmentsView.getId());
104         if (cardWithAttachments == null) {
105             return;
106         }
107         int col = getVerticalCul(permanentWithAttachmentsView, battlefieldPanel); // from right to left [2][1][0]
108         int currentAttachmentCol = col + 1;
109         cardWithAttachments.getLinks().clear();
110         int verticalIndex = permanentWithAttachmentsView.getAttachments().size();
111         for (UUID attachmentId : permanentWithAttachmentsView.getAttachments()) {
112             // put child attachments of the attachment
113             PermanentView attachedPermanentView = battlefieldPanel.getBattlefield().get(attachmentId);
114             if (attachedPermanentView != null && attachedPermanentView.getAttachments() != null && !attachedPermanentView.getAttachments().isEmpty()) {
115                 layoutAttachements(startingCardX, maxColumnLevels, ZOrder, attachedPermanentView, cards, battlefieldPanel, mainPanel, lastAttachmentRect);
116             }
117 
118             // put attachment
119             MageCard attachedCard = cards.get(attachmentId);
120             if (attachedCard != null) {
121                 // x position
122                 Point point = new Point();
123                 point.setLocation(startingCardX + (maxColumnLevels - currentAttachmentCol) * Math.max(cardWithAttachments.getCardLocation().getCardWidth() / ATTACHMENTS_MAX_COLUMNS, ATTACHMENTS_OFFSET_ALL_X), lastAttachmentRect.getY());
124                 lastAttachmentRect.setLocation(point);
125 
126                 // set position first to the same as of the permanent it is attached to
127                 attachedCard.setCardLocation(lastAttachmentRect.x, lastAttachmentRect.y);
128 
129                 // y position
130                 cardWithAttachments.getLinks().add(attachedCard);
131                 int dyOffset = Math.max(cardWithAttachments.getCardLocation().getCardHeight() / ATTACHMENTS_MAX_COLUMNS, ATTACHMENTS_OFFSET_SINGLE_Y);
132                 if (verticalIndex == 1) {
133                     lastAttachmentRect.translate(Math.max(cardWithAttachments.getCardLocation().getCardWidth() / ATTACHMENTS_MAX_COLUMNS, ATTACHMENTS_OFFSET_ALL_X), dyOffset);
134                 } else {
135                     lastAttachmentRect.translate(0, dyOffset);
136                 }
137                 cardWithAttachments.setCardLocation(lastAttachmentRect.x, lastAttachmentRect.y);
138 
139                 battlefieldPanel.moveToFront(attachedCard);
140                 battlefieldPanel.moveToFront(cardWithAttachments);
141                 mainPanel.setComponentZOrder(attachedCard, ZOrder--);
142                 verticalIndex--;
143             }
144         }
145     }

147     private AttachmentLayoutInfos calculateNeededNumberOfVerticalColumns(int currentCol, BattlefieldPanel battlefieldPanel, Map<UUID, MageCard> cards, PermanentView permanentWithAttachmentsView) {
148         int maxCol = ++currentCol;
149         int attachments = 0;
150         for (UUID attachmentId : permanentWithAttachmentsView.getAttachments()) {
151             PermanentView attachedPermanent = battlefieldPanel.getBattlefield().get(attachmentId);
152             if (attachedPermanent != null) {
153                 attachments++;
154                 if (attachedPermanent.getAttachments() != null && !attachedPermanent.getAttachments().isEmpty()) {
155                     AttachmentLayoutInfos attachmentLayoutInfos = calculateNeededNumberOfVerticalColumns(currentCol, battlefieldPanel, cards, attachedPermanent);
156                     if (attachmentLayoutInfos.getColumns() > maxCol) {
157                         maxCol = attachmentLayoutInfos.getColumns();
158                         attachments += attachmentLayoutInfos.getAttachments();
159                     }
160                 }
161             }
162         }
163         return new AttachmentLayoutInfos(maxCol, attachments);
164     }
165 
166     // The root permanent is col 0. An attachment attached to the root is col 1. And an attachement attached to the first attachment is col 2. etc.
167     private int getVerticalCul(PermanentView permanentView, BattlefieldPanel battlefieldPanel) {
168         int cul = 0;
169         if (permanentView.isAttachedTo()) {
170             PermanentView attachedToPermanent = battlefieldPanel.getBattlefield().get(permanentView.getAttachedTo());
171             if (attachedToPermanent != null) {
172                 cul = getVerticalCul(attachedToPermanent, battlefieldPanel);
173             }
174             cul++;
175         }
176         return cul;
177     }

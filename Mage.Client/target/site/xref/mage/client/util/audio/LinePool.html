3   import java.util.ArrayDeque;
4   import java.util.HashSet;
5   import java.util.LinkedList;
6   import java.util.Queue;
7   import java.util.Set;
8   import java.util.Timer;
9   import java.util.TimerTask;

11  import javax.sound.sampled.AudioFormat;
12  import javax.sound.sampled.AudioSystem;
13  import javax.sound.sampled.DataLine;
14  import javax.sound.sampled.LineEvent.Type;
15  import javax.sound.sampled.LineUnavailableException;
16  import javax.sound.sampled.Mixer;
17  import javax.sound.sampled.SourceDataLine;
18  
19  import org.apache.log4j.Logger;
20  
21  import mage.utils.ThreadUtils;
22  
23  public class LinePool {
24  
25      private final org.apache.log4j.Logger logger = Logger.getLogger(LinePool.class);
26      private static final int LINE_CLEANUP_INTERVAL = 30000;
27  
28      private final Queue<SourceDataLine> freeLines = new ArrayDeque<>();
29      private final Queue<SourceDataLine> activeLines = new ArrayDeque<>();
30      private final Set<SourceDataLine> busyLines = new HashSet<>();
31      private final LinkedList<MageClip> queue = new LinkedList<>();
32  
33      /*
34       * Initially all the lines are in the freeLines pool. When a sound plays, one line is being selected randomly from
35       * the activeLines and then, if it's empty, from the freeLines pool and used to play the sound. The line is moved to
36       * busyLines. When a sound stops, the line is moved to activeLines if it contains <= elements than alwaysActive
37       * parameter, else it's moved to the freeLines pool. Every 30 seconds the lines in the freeLines pool are closed
38       * from the timer thread to prevent deadlocks in PulseAudio internals.
39       */
40  
41      private final Mixer mixer;
42      private final int alwaysActive;
43  
44      public LinePool() {
45          this(new AudioFormat(22050, 16, 1, true, false), 4, 1);
46      }
47  
48      public LinePool(AudioFormat audioFormat, int size, int alwaysActive) {
49          this.alwaysActive = alwaysActive;
50          mixer = AudioSystem.getMixer(null);
51          DataLine.Info lineInfo = new DataLine.Info(SourceDataLine.class, audioFormat);
52          for (int i = 0; i < size; i++) {
53              try {
54                  SourceDataLine line = (SourceDataLine) mixer.getLine(lineInfo);
55                  freeLines.add(line);
56              } catch (LineUnavailableException e) {
57                  logger.warn("Failed to get line from mixer", e);
58              }
59          }
60          new Timer("Line cleanup", true).scheduleAtFixedRate(new TimerTask() {
61              @Override
62              public void run() {
63                  synchronized (LinePool.this) {
64                      for (SourceDataLine sourceDataLine : freeLines) {
65                          if (sourceDataLine.isOpen()) {
66                              sourceDataLine.close();
67                              logger.debug("Closed line " + sourceDataLine);
68                          }
69                      }
70                  }
71              }
72          }, LINE_CLEANUP_INTERVAL, LINE_CLEANUP_INTERVAL);
73      }
74  
75      private synchronized SourceDataLine borrowLine() {
76          SourceDataLine line = activeLines.poll();
77          if (line == null) {
78              line = freeLines.poll();
79          }
80          if (line != null) {
81              busyLines.add(line);
82          }
83          return line;
84      }
85  
86      private synchronized void returnLine(SourceDataLine line) {
87          busyLines.remove(line);
88          if (activeLines.size() < alwaysActive) {
89              activeLines.add(line);
90          } else {
91              freeLines.add(line);
92          }
93      }
94  
95      public void playSound(final MageClip mageClip) {
96          final SourceDataLine line;
97          synchronized (LinePool.this) {
98              logger.debug("Playing: " + mageClip.getFilename());
99              logLineStats();
100             line = borrowLine();
101             if (line == null) {
102                 // no lines available, queue sound to play it when a line is available
103                 queue.add(mageClip);
104                 logger.debug("Sound queued: " + mageClip.getFilename());
105                 return;
106             }
107             logLineStats();
108         }
109         ThreadUtils.threadPool.submit(() -> {
110             synchronized (LinePool.this) {
111                 try {
112                     if (!line.isOpen()) {
113                         line.open();
114                         line.addLineListener(event -> {
115                             logger.debug("Event: " + event);
116                             if (event.getType() != Type.STOP) {
117                                 return;
118                             }
119                             synchronized (LinePool.this) {
120                                 logger.debug("Before stop on line " + line);
121                                 logLineStats();
122                                 returnLine(line);
123                                 logger.debug("After stop on line " + line);
124                                 logLineStats();
125                                 MageClip queuedSound = queue.poll();
126                                 if (queuedSound != null) {
127                                     logger.debug("Playing queued sound " + queuedSound);
128                                     playSound(queuedSound);
129                                 }
130                             }
131                         });
132                     }
133                     line.start();
134                 } catch (LineUnavailableException e) {
135                     logger.warn("Failed to open line", e);
136                 }
137             }
138             byte[] buffer = mageClip.getBuffer();
139             logger.debug("Before write to line " + line);
140             line.write(buffer, 0, buffer.length);
141             line.drain();
142             line.stop();
143             logger.debug("Line completed: " + line);
144         });
145     }
146 
147     private void logLineStats() {
148         logger.debug(String.format("Free lines: %d; Active: %d; Busy: %d",
149                 freeLines.size(), activeLines.size(), busyLines.size()
150         ));
151     }
152 }

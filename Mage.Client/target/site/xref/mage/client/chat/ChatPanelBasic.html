170     private boolean containsSwearing(String message, String level) {
171 
172         if (level.equals("0")) {
173             return false;
174         }
175         message = '.' + message + '.';
176 
177         message = message.toLowerCase(Locale.ENGLISH);
178         message = message.replaceAll("[a@]([s5][s5]+)", "axyx");
179         message = message.replaceAll("b.([t\\+][t\\+]+)", "buxyx");
180         message = message.replaceAll("(.)(\\1{1,})", "$1");
181         message = message.replaceAll("[@]", "a");
182         message = message.replaceAll("[il]", "1");
183         message = message.replaceAll("[e]", "3");
184         message = message.replaceAll("[0]", "o");
185         message = message.replaceAll("[5z]", "s");
186         message = message.replaceAll("\\W", ".");
187         message = message.replaceAll("(.)(\\1{1,})", "$1");
188         message = message.replaceAll("\\.", "");
189 
190         Matcher matchPattern = profanityPattern.matcher(message);
191         if (matchPattern.find()) {
192             return true;
193         }
194 
195         if (level.equals("2")) {
196             message = message.replaceAll("\\.", "");
197             message = '.' + message + '.';
198             matchPattern = profanity2Pattern.matcher(message);
199             return matchPattern.find();
200         }
201         return false;
202     }

206     /**
207      * Display message in the chat. Use different colors for timestamp, username
208      * and message.
209      *
210      * @param username    message sender
211      * @param message     message itself
212      * @param time        timestamp
213      * @param turnInfo    game turn info, can be null for non game messages
214      * @param messageType
215      * @param color       Preferred color. Not used.
216      */
217     public void receiveMessage(String username, String message, Date time, String turnInfo, MessageType messageType, MessageColor color) {
218         StringBuilder text = new StringBuilder();
219         if (time != null) {
220             text.append(getColoredText(TIMESTAMP_COLOR, timeFormatter.format(time) + getTurnInfoPart(turnInfo) + ": "));
221         }
222         String userColor;
223         String textColor;
224         String userSeparator = " ";
225         switch (messageType) {
226             case STATUS: // a message to all chat user
227                 textColor = STATUS_COLOR;
228                 userColor = STATUS_COLOR;
229                 break;
230             case USER_INFO: // a personal message
231                 textColor = USER_INFO_COLOR;
232                 userColor = USER_INFO_COLOR;
233                 break;
234             default:
235                 userColor = SessionHandler.getUserName().equals(username) ? MY_COLOR : OPPONENT_COLOR;
236                 textColor = MESSAGE_COLOR;
237                 userSeparator = ": ";
238         }
239         if (color == MessageColor.ORANGE) {
240             textColor = "Orange";
241         }
242         if (color == MessageColor.YELLOW) {
243             textColor = "Yellow";
244         }
245         if (messageType == MessageType.WHISPER_FROM) {
246             if (username.equalsIgnoreCase(SessionHandler.getUserName())) {
247                 if (message.toLowerCase(Locale.ENGLISH).startsWith("profanity 0")) {
248                     PreferencesDialog.saveValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0");
249                 } else if (message.toLowerCase(Locale.ENGLISH).startsWith("profanity 1")) {
250                     PreferencesDialog.saveValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "1");
251                 } else if (message.toLowerCase(Locale.ENGLISH).startsWith("profanity 2")) {
252                     PreferencesDialog.saveValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "2");
253                 }
254             }
255             username = "Whisper from " + username;
256         }
257         if (messageType == MessageType.WHISPER_TO) {
258             username = "Whisper to " + username;
259         }
260 
261         Matcher matchPattern = cardNamePattern.matcher(message);
262         String messageToTest = message;
263         while (matchPattern.find()) {
264             messageToTest = message.replaceFirst("<font bgcolor=orange.*?</font>", "");
265         }
266 
267         if (messageType == MessageType.USER_INFO || messageType == MessageType.GAME || messageType == MessageType.STATUS
268                 || PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0").equals("0")
269                 || !PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0").equals("0") && !containsSwearing(messageToTest, PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0"))) {
270             if (username != null && !username.isEmpty()) {
271                 text.append(getColoredText(userColor, username + userSeparator));
272             }
273             text.append(getColoredText(textColor, ManaSymbols.replaceSymbolsWithHTML(message, ManaSymbols.Type.CHAT)));
274             this.txtConversation.append(text.toString());
275         } else if (PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0").equals("1")) {
276             if (username != null && !username.isEmpty()) {
277                 text.append(getColoredText("black", username + userSeparator));
278             }
279             text.append(getColoredText(textColor, ManaSymbols.replaceSymbolsWithHTML("<font color=black size=-2>" + message + "</font> <font size=-2>Profanity detected.  Type: <font color=green>/w " + SessionHandler.getUserName() + " profanity 0</font>' to turn the filter off</font></font>", ManaSymbols.Type.CHAT)));
280             this.txtConversation.append(text.toString());
281         } else if (PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_USE_PROFANITY_FILTER, "0").equals("2")) {
282             text.append(getColoredText(textColor, ManaSymbols.replaceSymbolsWithHTML("<font color=black size=-2>" + username + ": Profanity detected.  To make it less strict, type: </font> <font color=green size=-2>/w " + SessionHandler.getUserName() + " profanity 1</font>", ManaSymbols.Type.CHAT)));
283             this.txtConversation.append(text.toString());
284         }
285     }

398     public void handleKeyTyped(java.awt.event.KeyEvent evt) {
399         if (evt.getKeyChar() == KeyEvent.VK_ENTER) {
400             if (parentChatRef != null) {
401                 SessionHandler.sendChatMessage(parentChatRef.chatId, this.txtMessage.getText());
402             } else {
403                 SessionHandler.sendChatMessage(chatId, this.txtMessage.getText());
404             }
405             this.txtMessage.setText("");
406             this.txtMessage.repaint();
407         }
408     }

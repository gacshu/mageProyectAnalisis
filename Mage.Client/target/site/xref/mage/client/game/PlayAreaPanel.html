60      public PlayAreaPanel(PlayerView player, BigCard bigCard, UUID gameId, int priorityTime, GamePanel gamePanel, PlayAreaPanelOptions options) {
61          this.gamePanel = gamePanel;
62          this.options = options;
63          initComponents();
64          setOpaque(false);

66  
67          popupMenu = new JPopupMenu();
68          if (options.isPlayer) {
69              addPopupMenuPlayer(player.getUserData().isAllowRequestHandToAll());
70          } else {
71              addPopupMenuWatcher();
72          }
73          this.add(popupMenu);
74          setGUISize();
75

77          update(null, player, null);

80      public void CleanUp() {
81          battlefieldPanel.cleanUp();
82          playerPanel.cleanUp();
83  
84          for (ActionListener al : btnCheat.getActionListeners()) {
85              btnCheat.removeActionListener(al);
86          }
87  
88          // Taken form : https://community.oracle.com/thread/2183145
89          // removed the internal focus of a popupMenu data to allow GC before another popup menu is selected
90          for (ChangeListener listener : MenuSelectionManager.defaultManager().getChangeListeners()) {
91              if (listener.getClass().getName().contains("MenuKeyboardHelper")) {
92                  try {
93                      Field field = listener.getClass().getDeclaredField("menuInputMap");
94                      field.setAccessible(true);
95                      field.set(listener, null);
96                  } catch (Exception e) {
97                      // ignored
98                  }
99                  break;
100             }
101         }
102 
103         for (MouseListener ml : battlefieldPanel.getMainPanel().getMouseListeners()) {
104             battlefieldPanel.getMainPanel().removeMouseListener(ml);
105         }
106         popupMenu.getUI().uninstallUI(this);
107

512     public final void init(PlayerView player, BigCard bigCard, UUID gameId, int priorityTime) {
513         this.playerPanel.init(gameId, player.getPlayerId(), player.getControlled(), bigCard, priorityTime);
514         this.battlefieldPanel.init(gameId, bigCard);

520     public final void update(GameView game, PlayerView player, Set<UUID> possibleTargets) {
521         this.playerPanel.update(game, player, possibleTargets);
522         this.battlefieldPanel.update(player.getBattlefield());
523         if (this.allowViewHandCardsMenuItem != null) {
524             this.allowViewHandCardsMenuItem.setSelected(player.getUserData().isAllowRequestHandToAll());
525         }

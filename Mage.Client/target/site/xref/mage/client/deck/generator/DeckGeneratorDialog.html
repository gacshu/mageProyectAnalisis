283         c.ipadx = 0;

288         c.gridx = 2;
289         c.gridwidth = 1;

321     public String saveDeck(Deck deck) {
322         try {
323             // Random directory through the system property to avoid random numeric string attached to temp files.
324             String tempDir = System.getProperty("java.io.tmpdir");
325             // Generated deck has a nice unique name which corresponds to the timestamp at which it was created.
326             String deckName = "Generated-Deck-" + dateFormat.format(new Date());
327             File tmp = new File(tempDir + File.separator + deckName + ".dck");
328             tmp.getParentFile().mkdirs();
329             tmp.createNewFile();
330             deck.setName(deckName);
331             XMAGE.getExporter().writeDeck(tmp.getAbsolutePath(), deck.getDeckCardLists());
332             cleanUp();
333             return tmp.getAbsolutePath();
334         } catch (Exception e) {
335             MageFrame.getInstance().showError("Couldn't generate deck. Try again.");
336         }
337         return null;
338     }

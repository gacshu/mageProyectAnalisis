42      protected void playerLoadDeck() {
43          String lastFolder = MageFrame.getPreferences().get("lastDeckFolder", "");
44          if (!lastFolder.isEmpty()) {
45              fcSelectDeck.setCurrentDirectory(new File(lastFolder));
46          }
47          int ret = fcSelectDeck.showDialog(this, "Select Deck");
48          if (ret == JFileChooser.APPROVE_OPTION) {
49              File file = fcSelectDeck.getSelectedFile();
50              this.txtPlayerDeck.setText(file.getPath());
51              try {
52                  MageFrame.getPreferences().put("lastDeckFolder", file.getCanonicalPath());
53              } catch (IOException ex) {
54              }
55          }
56          fcSelectDeck.setSelectedFile(null);
57      }
58  
59      protected void generateDeck() {
60          String path = DeckGenerator.generateDeck();
61          if (path != null) {
62              this.txtPlayerDeck.setText(path);
63              MageFrame.getPreferences().put("defaultDeckPath", path);
64          }
65      }

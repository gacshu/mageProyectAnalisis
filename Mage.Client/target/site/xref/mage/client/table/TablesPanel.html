273     private int[] parseSeatsInfo(String info) {
274         String[] valsList = info.split("/");
275         int[] res = {0, 0};

283     public static int randomizeTimout(int minTimout) {
284         // randomize timeouts to fix calls waves -- slow server can creates queue and moves all clients to same call window
285         int increase = (int) (minTimout * REFRESH_TIMEOUTS_INCREASE_FACTOR);
286         return minTimout + RandomUtil.nextInt(increase);

542     private void addTableSelectListener(JTable table) {
543         // https://stackoverflow.com/a/26142800/1276632
544 
545         // save last selection
546         table.getSelectionModel().addListSelectionListener(new ListSelectionListener() {
547             @Override
548             public void valueChanged(ListSelectionEvent e) {
549                 int modelRow = TablesUtil.getSelectedModelRow(table);
550                 if (modelRow != -1) {
551                     // needs only selected
552                     String rowId = TablesUtil.getSearchIdFromTable(table, modelRow);
553                     tablesLastSelection.put(table, rowId);
554                 }
555             }
556         });
557 
558         // restore selection
559         table.getModel().addTableModelListener(new TableModelListener() {
560             @Override
561             public void tableChanged(TableModelEvent e) {
562                 SwingUtilities.invokeLater(new Runnable() {
563                     @Override
564                     public void run() {
565                         String lastRowID = tablesLastSelection.get(table);
566                         int needModelRow = TablesUtil.findTableRowFromSearchId(table.getModel(), lastRowID);
567                         int needViewRow = TablesUtil.getViewRowFromModel(table, needModelRow);
568                         if (needViewRow != -1) {
569                             table.clearSelection();
570                             table.addRowSelectionInterval(needViewRow, needViewRow);
571                         }
572                     }
573                 });
574             }
575         });

714     public void startUpdateTasks(boolean refreshImmediately) {
715         if (SessionHandler.getSession() != null) {
716             // active tables and server messages
717             if (updateTablesTask == null || updateTablesTask.isDone() || refreshImmediately) {
718                 if (updateTablesTask != null) updateTablesTask.cancel(true);
719                 updateTablesTask = new UpdateTablesTask(roomId, this);
720                 updateTablesTask.execute();
721             }
722 
723             // finished tables
724             if (this.btnStateFinished.isSelected()) {
725                 if (updateMatchesTask == null || updateMatchesTask.isDone() || refreshImmediately) {
726                     if (updateMatchesTask != null) updateMatchesTask.cancel(true);
727                     updateMatchesTask = new UpdateMatchesTask(roomId, this);
728                     updateMatchesTask.execute();
729                 }
730             } else {
731                 if (updateMatchesTask != null) updateMatchesTask.cancel(true);
732             }
733 
734             // players list
735             if (updatePlayersTask == null || updatePlayersTask.isDone() || refreshImmediately) {
736                 if (updatePlayersTask != null) updatePlayersTask.cancel(true);
737                 updatePlayersTask = new UpdatePlayersTask(roomId, this.chatPanelMain);
738                 updatePlayersTask.execute();
739             }
740         }

852         java.util.List<RowFilter<Object, Object>> typeFilterList = new ArrayList<>();

854             typeFilterList.add(RowFilter.regexFilter("Two|Commander|Free|Tiny|Momir", TablesTableModel.COLUMN_GAME_TYPE));
855         }

857             typeFilterList.add(RowFilter.regexFilter("Constructed", TablesTableModel.COLUMN_GAME_TYPE));
858         }

860             typeFilterList.add(RowFilter.regexFilter("Booster|Sealed|Jumpstart", TablesTableModel.COLUMN_GAME_TYPE));
861         }
862 
863         // format
864         java.util.List<RowFilter<Object, Object>> formatFilterList = new ArrayList<>();
865         if (btnFormatBlock.isSelected()) {
866             formatFilterList.add(RowFilter.regexFilter("^Constructed.*Block", TablesTableModel.COLUMN_DECK_TYPE));
867         }
868         if (btnFormatStandard.isSelected()) {
869             formatFilterList.add(RowFilter.regexFilter("^Constructed - Standard", TablesTableModel.COLUMN_DECK_TYPE));
870         }
871         if (btnFormatModern.isSelected()) {
872             formatFilterList.add(RowFilter.regexFilter("^Constructed - Modern", TablesTableModel.COLUMN_DECK_TYPE));
873         }
874         if (btnFormatPioneer.isSelected()) {
875             formatFilterList.add(RowFilter.regexFilter("^Constructed - Pioneer", TablesTableModel.COLUMN_DECK_TYPE));
876         }
877         if (btnFormatLegacy.isSelected()) {
878             formatFilterList.add(RowFilter.regexFilter("^Constructed - Legacy", TablesTableModel.COLUMN_DECK_TYPE));
879         }
880         if (btnFormatVintage.isSelected()) {
881             formatFilterList.add(RowFilter.regexFilter("^Constructed - Vintage", TablesTableModel.COLUMN_DECK_TYPE));
882         }
883         if (btnFormatPremodern.isSelected()) {
884             formatFilterList.add(RowFilter.regexFilter("^Constructed - Premodern", TablesTableModel.COLUMN_DECK_TYPE));
885         }
886         if (btnFormatCommander.isSelected()) {
887             formatFilterList.add(RowFilter.regexFilter("^Commander|^Duel Commander|^Centurion Commander|^Penny Dreadful Commander|^Freeform Commander|^Freeform Unlimited Commander|^MTGO 1v1 Commander|^Duel Brawl|^Brawl", TablesTableModel.COLUMN_DECK_TYPE));
888         }
889         if (btnFormatTinyLeader.isSelected()) {
890             formatFilterList.add(RowFilter.regexFilter("^Tiny", TablesTableModel.COLUMN_DECK_TYPE));
891         }
892         if (btnFormatOathbreaker.isSelected()) {
893             formatFilterList.add(RowFilter.regexFilter("^Oathbreaker", TablesTableModel.COLUMN_DECK_TYPE));
894         }
895         if (btnFormatLimited.isSelected()) {
896             formatFilterList.add(RowFilter.regexFilter("^Limited", TablesTableModel.COLUMN_DECK_TYPE));
897         }
898         if (btnFormatOther.isSelected()) {
899             formatFilterList.add(RowFilter.regexFilter("^Momir Basic|^Constructed - Pauper|^Constructed - Frontier|^Constructed - Extended|^Constructed - Eternal|^Constructed - Historical|^Constructed - Super|^Constructed - Freeform|^Australian Highlander|^European Highlander|^Canadian Highlander|^Constructed - Old|^Constructed - Historic", TablesTableModel.COLUMN_DECK_TYPE));
900         }
901 
902         // skill
903         java.util.List<RowFilter<Object, Object>> skillFilterList = new ArrayList<>();
904         if (btnSkillBeginner.isSelected()) {
905             skillFilterList.add(RowFilter.regexFilter(this.tableModel.getSkillLevelAsCode(SkillLevel.BEGINNER, true), TablesTableModel.COLUMN_SKILL));
906         }
907         if (btnSkillCasual.isSelected()) {
908             skillFilterList.add(RowFilter.regexFilter(this.tableModel.getSkillLevelAsCode(SkillLevel.CASUAL, true), TablesTableModel.COLUMN_SKILL));
909         }
910         if (btnSkillSerious.isSelected()) {
911             skillFilterList.add(RowFilter.regexFilter(this.tableModel.getSkillLevelAsCode(SkillLevel.SERIOUS, true), TablesTableModel.COLUMN_SKILL));
912         }
913 
914         String ratedMark = TablesTableModel.RATED_VALUE_YES;
915         java.util.List<RowFilter<Object, Object>> ratingFilterList = new ArrayList<>();
916         if (btnRated.isSelected()) {
917             // yes word
918             ratingFilterList.add(RowFilter.regexFilter("^" + ratedMark, TablesTableModel.COLUMN_RATING));
919         }
920         if (btnUnrated.isSelected()) {
921             // not yes word, see https://stackoverflow.com/a/406408/1276632
922             ratingFilterList.add(RowFilter.regexFilter("^((?!" + ratedMark + ").)*$", TablesTableModel.COLUMN_RATING));
923         }
924 
925         // Password
926         String passwordMark = TablesTableModel.PASSWORD_VALUE_YES;
927         java.util.List<RowFilter<Object, Object>> passwordFilterList = new ArrayList<>();
928         if (btnPassword.isSelected()) {
929             // yes
930             passwordFilterList.add(RowFilter.regexFilter("^" + passwordMark, TablesTableModel.COLUMN_PASSWORD));
931         }
932         if (btnOpen.isSelected()) {
933             // no
934             passwordFilterList.add(RowFilter.regexFilter("^((?!" + passwordMark + ").)*$", TablesTableModel.COLUMN_PASSWORD));
935         }
936 
937         // Hide games of ignored players
938         java.util.List<RowFilter<Object, Object>> ignoreListFilterList = new ArrayList<>();
939         String serverAddress = SessionHandler.getSession().getServerHostname().orElse("");
940         final Set<String> ignoreListCopy = IgnoreList.getIgnoredUsers(serverAddress);
941         if (!ignoreListCopy.isEmpty()) {
942             ignoreListFilterList.add(new RowFilter<Object, Object>() {
943                 @Override
944                 public boolean include(Entry<? extends Object, ? extends Object> entry) {
945                     final String owner = entry.getStringValue(TablesTableModel.COLUMN_OWNER);
946                     return !ignoreListCopy.contains(owner);
947                 }
948             });
949         }
950 
951         if (stateFilterList.isEmpty() || typeFilterList.isEmpty() || formatFilterList.isEmpty()
952                 || skillFilterList.isEmpty() || ratingFilterList.isEmpty()
953                 || passwordFilterList.isEmpty()) { // no selection
954             activeTablesSorter.setRowFilter(RowFilter.regexFilter("Nothing", TablesTableModel.COLUMN_SKILL));
955         } else {
956             java.util.List<RowFilter<Object, Object>> filterList = new ArrayList<>();
957 
958             if (stateFilterList.size() > 1) {
959                 filterList.add(RowFilter.orFilter(stateFilterList));
960             } else if (stateFilterList.size() == 1) {
961                 filterList.addAll(stateFilterList);
962             }
963 
964             if (typeFilterList.size() > 1) {
965                 filterList.add(RowFilter.orFilter(typeFilterList));
966             } else if (typeFilterList.size() == 1) {
967                 filterList.addAll(typeFilterList);
968             }
969 
970             if (formatFilterList.size() > 1) {
971                 filterList.add(RowFilter.orFilter(formatFilterList));
972             } else if (formatFilterList.size() == 1) {
973                 filterList.addAll(formatFilterList);
974             }
975 
976             if (skillFilterList.size() > 1) {
977                 filterList.add(RowFilter.orFilter(skillFilterList));
978             } else if (skillFilterList.size() == 1) {
979                 filterList.addAll(skillFilterList);
980             }
981 
982             if (ratingFilterList.size() > 1) {
983                 filterList.add(RowFilter.orFilter(ratingFilterList));
984             } else if (ratingFilterList.size() == 1) {
985                 filterList.addAll(ratingFilterList);
986             }
987 
988             if (passwordFilterList.size() > 1) {
989                 filterList.add(RowFilter.orFilter(passwordFilterList));
990             } else if (passwordFilterList.size() == 1) {
991                 filterList.addAll(passwordFilterList);
992             }
993 
994             if (ignoreListFilterList.size() > 1) {
995                 filterList.add(RowFilter.orFilter(ignoreListFilterList));
996             } else if (ignoreListFilterList.size() == 1) {
997                 filterList.addAll(ignoreListFilterList);
998             }
999 
1000             if (filterList.size() == 1) {
1001                 activeTablesSorter.setRowFilter(filterList.get(0));
1002             } else {
1003                 activeTablesSorter.setRowFilter(RowFilter.andFilter(filterList));
1004             }
1005         }

1680     private void createTestGame(String gameName, String gameType, boolean useMonteCarloAI) {
1681         TableView table;
1682         try {
1683             String testDeckFile = "test.dck";
1684             File f = new File(testDeckFile);
1685             if (!f.exists()) {
1686                 // default test deck
1687                 testDeckFile = DeckUtil.writeTextToTempFile(""
1688                         + "5 Swamp" + System.lineSeparator()
1689                         + "5 Forest" + System.lineSeparator()
1690                         + "5 Island" + System.lineSeparator()
1691                         + "5 Mountain" + System.lineSeparator()
1692                         + "5 Plains");
1693             }
1694             DeckCardLists testDeck = DeckImporter.importDeckFromFile(testDeckFile, false);
1695 
1696             PlayerType aiType = useMonteCarloAI ? PlayerType.COMPUTER_MONTE_CARLO : PlayerType.COMPUTER_MAD;
1697             int numSeats = gameName.contains("2") ? 2 : 4;
1698             boolean multiPlayer = numSeats > 2;
1699 
1700             MatchOptions options = new MatchOptions(gameName, gameType, multiPlayer, numSeats);
1701             options.getPlayerTypes().add(PlayerType.HUMAN);
1702             options.getPlayerTypes().add(aiType);
1703             for (int i=2 ; i < numSeats ; i++) {
1704                 options.getPlayerTypes().add(aiType);
1705             }
1706             options.setDeckType("Variant Magic - Freeform Commander");
1707             options.setAttackOption(MultiplayerAttackOption.MULTIPLE);
1708             options.setRange(RangeOfInfluence.ONE);
1709             options.setWinsNeeded(2);
1710             options.setMatchTimeLimit(MatchTimeLimit.NONE);
1711             options.setFreeMulligans(2);
1712             options.setSkillLevel(SkillLevel.CASUAL);
1713             options.setRollbackTurnsAllowed(true);
1714             options.setQuitRatio(100);
1715             options.setMinimumRating(0);
1716             String serverAddress = SessionHandler.getSession().getServerHostname().orElse("");
1717             options.setBannedUsers(IgnoreList.getIgnoredUsers(serverAddress));
1718             table = SessionHandler.createTable(roomId, options);
1719 
1720             SessionHandler.joinTable(roomId, table.getTableId(), "Human", PlayerType.HUMAN, 1, testDeck, "");
1721             SessionHandler.joinTable(roomId, table.getTableId(), "Computer", aiType, 1, testDeck, "");
1722             for (int i=2 ; i < numSeats ; i++) {
1723                 SessionHandler.joinTable(roomId, table.getTableId(), "Computer" + i, aiType, 1, testDeck, "");
1724             }
1725             SessionHandler.startMatch(roomId, table.getTableId());
1726         } catch (HeadlessException ex) {
1727             handleError(ex);
1728         }

1864         }

1875             count = 0;
1876             isFirstRun = false;
1877             panel.reloadServerMessages();
1878         }

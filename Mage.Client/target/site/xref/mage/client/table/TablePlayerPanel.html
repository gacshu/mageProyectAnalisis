30      public void init(int playerNum, PlayerType playerType) {
31          cbPlayerType.setModel(new DefaultComboBoxModel(SessionHandler.getPlayerTypes()));
32          this.lblPlayerNum.setText("Player " + playerNum);
33          if (ClientDefaultSettings.otherPlayerIndex != null) {
34              Integer index = Integer.parseInt(ClientDefaultSettings.otherPlayerIndex);
35              if (index >= cbPlayerType.getItemCount()) {
36                  cbPlayerType.setSelectedIndex(cbPlayerType.getItemCount() - 1);
37              } else {
38                  cbPlayerType.setSelectedIndex(index);
39              }
40          }
41          if (playerType != null) {
42              this.cbPlayerType.setSelectedItem(playerType);
43          }
44      }
45  
46      public boolean joinTable(UUID roomId, UUID tableId) throws IOException, ClassNotFoundException {
47          if (this.cbPlayerType.getSelectedItem() != PlayerType.HUMAN) {
48              return SessionHandler.joinTable(roomId, tableId, this.newPlayerPanel.getPlayerName(), (PlayerType) this.cbPlayerType.getSelectedItem(), this.newPlayerPanel.getLevel(), DeckImporter.importDeckFromFile(this.newPlayerPanel.getDeckFile(), true), "");
49          }
50          return true;
51      }
52  
53      public PlayerType getPlayerType() {
54          return PlayerType.getByDescription(this.cbPlayerType.getSelectedItem().toString());
55      }

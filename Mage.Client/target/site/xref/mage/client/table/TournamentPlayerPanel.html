34      public boolean joinTournamentTable(UUID roomId, UUID tableId, DeckCardLists deckCardLists) {
35          if (this.cbPlayerType.getSelectedItem() != PlayerType.HUMAN) {
36              return SessionHandler.joinTournamentTable(
37                      roomId,
38                      tableId,
39                      this.txtPlayerName.getText(),
40                      (PlayerType) this.cbPlayerType.getSelectedItem(),
41                      (Integer) spnLevel.getValue(),
42                      deckCardLists,
43                      "");
44          }
45          return true;
46      }

91      @Override
92      public void loadCards(CardsView showCards, SortSetting sortSetting, BigCard bigCard, UUID gameId, boolean merge) {
93          if (this.sortSetting == null) {
94              this.sortSetting = sortSetting;
95          }
96          this.bigCard = bigCard;
97          this.gameId = gameId;
98          int landCount = 0;
99          int creatureCount = 0;
100         int instantCount = 0;
101         int sorceryCount = 0;
102         int enchantmentCount = 0;
103         int artifactCount = 0;
104         if (!merge) {
105             this.clearCards();
106             for (CardView card : showCards.values()) {
107                 addCard(card, bigCard, gameId);
108             }
109         } else {
110             for (CardView card : showCards.values()) {
111                 if (!cards.containsKey(card.getId())) {
112                     addCard(card, bigCard, gameId);
113                 }
114                 if (updateCountsCallback != null) {
115                     if (card.isLand()) {
116                         landCount++;
117                     }
118                     if (card.isCreature()) {
119                         creatureCount++;
120                     }
121                     if (card.isInstant()) {
122                         instantCount++;
123                     }
124                     if (card.isSorcery()) {
125                         sorceryCount++;
126                     }
127                     if (card.isEnchantment()) {
128                         enchantmentCount++;
129                     }
130                     if (card.isArtifact()) {
131                         artifactCount++;
132                     }
133                 }
134             }
135 
136             // no easy logic for merge :)
137             for (Iterator<Entry<UUID, CardView>> i = cards.entrySet().iterator(); i.hasNext(); ) {
138                 Entry<UUID, CardView> entry = i.next();
139                 if (!showCards.containsKey(entry.getKey())) {
140                     i.remove();
141                     if (displayNoCopies) {
142                         String key = entry.getValue().getName() + entry.getValue().getExpansionSetCode() + entry.getValue().getCardNumber();
143                         if (cardsNoCopies.containsKey(key)) {
144                             Integer count = cardsNoCopies.get(key);
145                             count--;
146                             if (count > 0) {
147                                 cardsNoCopies.put(key, count);
148                             } else {
149                                 cardsNoCopies.remove(key);
150                             }
151                             for (int j = 0; j < view.size(); j++) {
152                                 CardView cv = view.get(j);
153                                 if (cv.getId().equals(entry.getValue().getId())) {
154                                     if (count > 0) {
155                                         // replace by another card with the same name+setCode
156                                         String key1 = cv.getName() + cv.getExpansionSetCode() + cv.getCardNumber();
157                                         for (CardView cardView : cards.values()) {
158                                             String key2 = cardView.getName() + cardView.getExpansionSetCode() + cardView.getCardNumber();
159                                             if (key1.equals(key2)) {
160                                                 view.set(j, cardView);
161                                                 break;
162                                             }
163                                         }
164                                     } else {
165                                         view.remove(j);
166                                     }
167                                     break;
168                                 }
169                             }
170                         }
171                     } else {
172                         view.removeIf(cardView -> cardView.getId().equals(entry.getKey()));
173                     }
174                 }
175             }
176 
177             if (updateCountsCallback != null) {
178                 updateCountsCallback.update(cards.size(), creatureCount, landCount, sorceryCount, instantCount, enchantmentCount, artifactCount);
179             }
180         }
181 
182         sort(this.sortSetting.getSortIndex(), this.sortSetting.isAscending());
183         drawCards(sortSetting);
184     }

216     private Object getColumn(Object obj, int column) {
217         CardView c = (CardView) obj;
218         switch (column) {
219             case 0:
220                 if (displayNoCopies) {
221                     String key = c.getName() + c.getExpansionSetCode() + c.getCardNumber();
222                     Integer count = cardsNoCopies.get(key);
223                     return count != null ? count : "";
224                 }
225                 return "";
226             case 1:
227                 return c.getDisplayFullName(); // show full name in deck editor table, e.g. adventure with spell name
228             case 2:
229                 // new svg images version
230                 return ManaSymbols.getClearManaCost(c.getManaCostStr());
231                 /*
232                 // old html images version
233                 String manaCost = "";
234                 for (String m : c.getManaCost()) {
235                     manaCost += m;
236                 }
237                 String castingCost = UI.getDisplayManaCost(manaCost);
238                 castingCost = ManaSymbols.replaceSymbolsWithHTML(castingCost, ManaSymbols.Type.TABLE);
239                 return "<html>" + castingCost + "</html>";
240                 return castingCost;
241                 */
242             case 3:
243                 return c.getColorText();
244             case 4:
245                 return c.getTypeText();
246             case 5:
247                 return c.isCreature() ? c.getPower() + '/'
248                         + c.getToughness() : "-";
249             case 6:
250                 return c.getRarity() == null ? "" : c.getRarity().toString();
251             case 7:
252                 return c.getExpansionSetCode();
253             case 8:
254                 return c.getCardNumber();
255             case 9:
256                 return RateCard.rateCard(c.getOriginalCard(), null);
257             case 10:
258                 return ManaSymbols.getClearManaCost(c.getColorIdentityStr());
259             default:
260                 return "error";
261         }
262     }

264     private void addCard(CardView card, BigCard bigCard, UUID gameId) {
265         if (cardDimension == null) {
266             cardDimension = new Dimension(ClientDefaultSettings.dimensions.getFrameWidth(),
267                     ClientDefaultSettings.dimensions.getFrameHeight());
268         }
269         cards.put(card.getId(), card);
270 
271         if (displayNoCopies) {
272             String key = card.getName() + card.getExpansionSetCode() + card.getCardNumber();
273             Integer count = 1;
274             if (cardsNoCopies.containsKey(key)) {
275                 count = cardsNoCopies.get(key) + 1;
276             } else {
277                 view.add(card);
278             }
279             cardsNoCopies.put(key, count);
280         } else {
281             view.add(card);
282         }
283     }

323     public void addListeners(final JTable table) {
324         // updates card detail, listens to any key strokes
325 
326         table.addKeyListener(new KeyListener() {
327             @Override
328             public void keyPressed(KeyEvent ev) {
329             }
330 
331             @Override
332             public void keyTyped(KeyEvent ev) {
333             }
334 
335             @Override
336             public void keyReleased(KeyEvent ev) {
337                 int row = table.getSelectedRow();
338                 if (row != -1) {
339                     showImage(row);
340                 }
341             }
342         });
343 
344         // updates card detail, listens to any mouse clicks
345         table.addMouseListener(new MouseAdapter() {
346             @Override
347             public void mousePressed(MouseEvent e) {
348                 int row = table.getSelectedRow();
349                 if (row != -1) {
350                     showImage(row);
351                 }
352             }
353         });
354 
355         // sorts
356         MouseListener mouse = new MouseAdapter() {
357             @Override
358             public void mousePressed(MouseEvent e) {
359                 TableColumnModel columnModel = table.getColumnModel();
360                 int viewColumn = columnModel.getColumnIndexAtX(e.getX());
361                 int column = table.convertColumnIndexToModel(viewColumn);
362 
363                 if (column != -1) {
364                     // sort ascending
365                     boolean asc = true;
366                     if (recentSortedColumn == column) {
367                         asc = !recentAscending;
368                     }
369                     sortSetting.setSortIndex(column);
370                     sortSetting.setAscending(asc);
371                     sort(column, asc);
372                     fireTableDataChanged();
373                 }
374             }
375         };
376         table.getTableHeader().addMouseListener(mouse);
377     }

379     private void showImage(int row) {
380         CardView card = view.get(row);
381         if (!card.getId().equals(bigCard.getCardId())) {
382             if (!MageFrame.isLite()) {
383                 Image image = Plugins.instance.getOriginalImage(card);
384                 if (image instanceof BufferedImage) {
385                     // XXX: scaled to fit width
386                     bigCard.setCard(card.getId(), EnlargeMode.NORMAL, image, new ArrayList<>(), false);
387                 } else {
388                     drawCardText(card);
389                 }
390             } else {
391                 drawCardText(card);
392             }
393         }
394     }

396     private void drawCardText(CardView card) {
397         JXPanel panel = GuiDisplayUtil.getDescription(card, bigCard.getWidth(), bigCard.getHeight());
398         panel.setVisible(true);
399         bigCard.hideTextComponent();
400         bigCard.addJXPanel(card.getId(), panel);
401     }

407     public boolean sort(int column, boolean ascending) {
408         // used by addCard() to resort the cards
409         recentSortedColumn = column;
410         recentAscending = ascending;
411 
412         MageCardComparator sorter = new MageCardComparator(column, ascending);
413         view.sort(sorter);
414 
415         fireTableDataChanged();
416 
417         return true;
418     }

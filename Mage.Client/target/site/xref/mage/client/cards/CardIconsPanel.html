72      public CardIconsPanel(CardIconPosition position, CardIconOrder order, CardIconColor color, int cellsVisibleCount, int iconSizePercent, List<CardIcon> icons, Rectangle startingCardSize) {
73          super(null);
74          this.position = (position != null ? position : DEFAULT_POSITION);
75          this.color = (color != null ? color : DEFAULT_COLOR);
76          this.iconSizePercent = iconSizePercent;
77          this.icons = icons;
78  
79          // corners have only one icon with center order
80          if (this.position.getMaxIconsAmount() == 1) {
81              this.order = CardIconOrder.START;
82              this.cellsOffset = 0;
83              this.cellsMaxCount = 1;
84          } else {
85              this.order = order != null ? order : DEFAULT_ORDER;
86              this.cellsOffset = 1;
87              this.cellsMaxCount = 7;
88          }
89          int maxIcons = Math.max(1, Math.min(this.cellsMaxCount, cellsVisibleCount)); // must be in [1..cells];
90          this.cellsVisibleCount = Math.min(maxIcons, this.position.getMaxIconsAmount());
91  
92          this.setVisible(false);
93          this.setOpaque(false);
94          if (DebugUtil.GUI_CARD_ICONS_DRAW_PANEL_BORDER) {
95              this.setBorder(BorderFactory.createLineBorder(Color.red));
96          }
97  
98          this.updateSizes(startingCardSize);
99      }

101     public void updateSizes(Rectangle cardSize) {
102         this.calcSizes(cardSize);
103 
104         // panel uses GridLayout with gaps, grid is static size, so the sizes structure:
105         // [gap + icon + gap + icon + ... gap]
106 
107         // corner icons must be same sizes as max possible on left/right (for a more beautiful look)
108         int panelFullSize = this.halfSize * 2 + cardSize.height;
109         int panelIconSize = (panelFullSize - (7 + 1) * this.iconsGap) / 7;
110         int cornerHalfSize = Math.min(panelIconSize, this.halfSize * 2) / 2; // real icons can be limited by height or width
111 
112         // move panel to the inner (for a more beautiful look)
113         // 2/3 keep inside and 1/3 keep outside
114         // panels already centered by halfSize, so use "- this.halfSize"
115         int panelOffset = Math.round(this.halfSize * 2 * KEEP_ICON_IN_CARD_INSIDE_PERCENT / 100f) - this.halfSize;
116 
117         Rectangle panelRect;
118         Point panelTranslate;
119         switch (this.calcedPosition) {
120             case TOP:
121                 panelRect = new Rectangle(cardSize.x - this.halfSize, cardSize.y - this.halfSize, cardSize.width + this.halfSize * 2, this.halfSize * 2);
122                 panelTranslate = new Point(0, panelOffset);
123                 this.setLayout(new GridLayout(1, this.calcedCellsMaxCount, iconsGap, 0));
124                 break;
125             case LEFT:
126                 panelRect = new Rectangle(cardSize.x - this.halfSize, cardSize.y - this.halfSize, this.halfSize * 2, cardSize.height + this.halfSize * 2);
127                 panelTranslate = new Point(panelOffset, 0);
128                 this.setLayout(new GridLayout(this.calcedCellsMaxCount, 1, 0, iconsGap));
129                 break;
130             case RIGHT:
131                 panelRect = new Rectangle(cardSize.x + cardSize.width - this.halfSize, cardSize.y - this.halfSize, this.halfSize * 2, cardSize.height + this.halfSize * 2);
132                 panelTranslate = new Point(-panelOffset, 0);
133                 this.setLayout(new GridLayout(this.calcedCellsMaxCount, 1, 0, iconsGap));
134                 break;
135             case BOTTOM:
136                 panelRect = new Rectangle(cardSize.x - this.halfSize, cardSize.y + cardSize.height - this.halfSize, cardSize.width + this.halfSize * 2, this.halfSize * 2);
137                 panelTranslate = new Point(0, -panelOffset);
138                 this.setLayout(new GridLayout(1, this.calcedCellsMaxCount, iconsGap, 0));
139                 break;
140             case CORNER_TOP_LEFT:
141                 panelRect = new Rectangle(cardSize.x - cornerHalfSize, cardSize.y - cornerHalfSize, cornerHalfSize * 2, cornerHalfSize * 2);
142                 panelTranslate = new Point(panelOffset, panelOffset);
143                 this.setLayout(new GridLayout(1, 1, 0, 0));
144                 break;
145             case CORNER_TOP_RIGHT:
146                 panelRect = new Rectangle(cardSize.x + cardSize.width - cornerHalfSize, cardSize.y - cornerHalfSize, cornerHalfSize * 2, cornerHalfSize * 2);
147                 panelTranslate = new Point(-panelOffset, panelOffset);
148                 this.setLayout(new GridLayout(1, 1, 0, 0));
149                 break;
150             case CORNER_BOTTOM_LEFT:
151                 panelRect = new Rectangle(cardSize.x - cornerHalfSize, cardSize.y + cardSize.height - cornerHalfSize, cornerHalfSize * 2, cornerHalfSize * 2);
152                 panelTranslate = new Point(panelOffset, -panelOffset);
153                 this.setLayout(new GridLayout(1, 1, 0, 0));
154                 break;
155             case CORNER_BOTTOM_RIGHT:
156                 panelRect = new Rectangle(cardSize.x + cardSize.width - cornerHalfSize, cardSize.y + cardSize.height - cornerHalfSize, cornerHalfSize * 2, cornerHalfSize * 2);
157                 panelTranslate = new Point(-panelOffset, -panelOffset);
158                 this.setLayout(new GridLayout(1, 1, 0, 0));
159                 break;
160             default:
161                 throw new IllegalArgumentException("Card icons do not support position " + this.calcedPosition);
162         }
163         panelRect.translate(panelTranslate.x, panelTranslate.y);
164         this.setBounds(panelRect);
165 
166         // reload icons for new size
167         this.updateIcons();
168     }

174     public void updateIcons(List<CardIcon> newIcons) {
175         this.removeAll();
176         if (newIcons != null) {
177             this.icons.clear();
178             this.icons.addAll(newIcons);
179         }
180 
181         // auto-hide panel on empty icons
182         if (this.icons.isEmpty()) {
183             this.setVisible(false);
184             return;
185         } else {
186             this.setVisible(true);
187         }
188 
189         int actualMaxVisibleCount = Math.min(this.calcedCellsVisibleCount, this.calcedCellsMaxCount - this.calcedCellsOffset * 2); // preserve offset cells
190 
191         List<Component> visibleComponents = new ArrayList<>();
192         List<Component> combinedComponents = new ArrayList<>();
193         List<Component> orderedComponents = new ArrayList<>();
194 
195         // structure:
196         // * icons panel - control the icons size and offsets in card;
197         // * grid panel - control the icons order and position (put empty panel for empty space)
198         // * grid's cell - control one icon
199         // * label - stretched icon image that occupy all cell's space
200         Map<Component, CardIcon> cardLinks = new HashMap<>();
201         this.icons.stream()
202                 .sorted(CardIconComparator.instance)
203                 .forEach(icon -> {
204                     Component iconComponent = createIconComponent(icon);
205                     if (iconComponent != null) {
206                         visibleComponents.add(iconComponent);
207                         cardLinks.put(iconComponent, icon);
208                     }
209                 });
210 
211         // OPTIMIZE visible components (if card contains too much icons then combine it in one "...")
212         if (visibleComponents.size() > actualMaxVisibleCount) {
213             while (visibleComponents.size() > actualMaxVisibleCount - 1) {
214                 // combined must contains minimum 2 elements
215                 combinedComponents.add(visibleComponents.remove(visibleComponents.size() - 1));
216             }
217             String combinedHint = combinedComponents
218                     .stream()
219                     .map(cardLinks::get)
220                     .filter(Objects::nonNull)
221                     .sorted(CardIconComparator.instance)
222                     .map(CardIcon::getCombinedInfo)
223                     .collect(Collectors.joining("<br>"));
224             CardIcon combinedIcon = new CombinedCountIcon(combinedComponents.size(), combinedHint);
225             Component combinedComponent = createIconComponent(combinedIcon);
226             if (combinedComponent != null) {
227                 visibleComponents.add(combinedComponent);
228             }
229         }
230 
231         // add offsets to the start of the list
232         if (this.calcedOrder == CardIconOrder.START || this.calcedOrder == CardIconOrder.END) {
233             for (int i = 0; i < this.calcedCellsOffset; i++) {
234                 JPanel panel = new JPanel(null);
235                 panel.setOpaque(false);
236                 visibleComponents.add(0, panel);
237             }
238         }
239 
240         // fill components list to max (grid can't put elements to cells, so must fill all)
241         while (visibleComponents.size() < this.calcedCellsMaxCount) {
242             JPanel panel = new JPanel(null);
243             panel.setOpaque(false);
244             visibleComponents.add(panel);
245         }
246 
247         // ORDER visible components
248         // icons sort order example with CENTER order:
249         // 1: [1]
250         // 3: [2 1 3]
251         // 5: [4 2 1 3 5]
252         // 7: [6 4 2 1 3 5 7]
253         //
254         // icons sort order example with START order (END order is same but reversed):
255         // 1: [1]
256         // 3: [1 2 3]
257         // 5: [1 2 3 4 5]
258         // 7: [1 2 3 4 5 6 7]
259         if (this.calcedOrder == CardIconOrder.CENTER) {
260             // CENTER
261             if (this.calcedCellsMaxCount == 1) {
262                 Arrays.asList(1).forEach(i -> orderedComponents.add(visibleComponents.get(i - 1)));
263             } else if (this.calcedCellsMaxCount == 3) {
264                 Arrays.asList(2, 1, 3).forEach(i -> orderedComponents.add(visibleComponents.get(i - 1)));
265             } else if (this.calcedCellsMaxCount == 5) {
266                 Arrays.asList(4, 2, 1, 3, 5).forEach(i -> orderedComponents.add(visibleComponents.get(i - 1)));
267             } else if (this.calcedCellsMaxCount == 7) {
268                 Arrays.asList(6, 4, 2, 1, 3, 5, 7).forEach(i -> orderedComponents.add(visibleComponents.get(i - 1)));
269             } else {
270                 throw new IllegalArgumentException("Card icons do not support max size as " + this.calcedCellsMaxCount);
271             }
272         } else if (this.calcedOrder == CardIconOrder.START) {
273             // START
274             orderedComponents.addAll(visibleComponents);
275         } else if (this.calcedOrder == CardIconOrder.END) {
276             // END
277             orderedComponents.addAll(visibleComponents);
278             Collections.reverse(orderedComponents);
279         } else {
280             throw new IllegalArgumentException("Card icons do not support order type " + this.calcedOrder);
281         }
282 
283         // ADD real components to the grid
284         orderedComponents.forEach(this::add);
285     }

287     private Component createIconComponent(CardIcon icon) {
288         if (!SvgUtils.haveSvgSupport()) {
289             return null;
290         }
291 
292         // direct call
293         //BufferedImage iconImage = ImageManagerImpl.instance.getCardIcon(icon.getIconType().getResourceName(), this.halfSize * 2);
294 
295         // cached call
296         BufferedImage iconImageCached = ImageCache.getCardIconImage(
297                 icon.getIconType().getResourceName(),
298                 this.halfSize * 2,
299                 color.toString()
300         ).getImage();
301 
302         if (iconImageCached != null && this.font != null) {
303             BufferedImage iconImageWithText = ImageManagerImpl.deepCopy(iconImageCached); // must copy cached value before modify
304 
305             // text
306             JLabel label = new JLabel();
307             label.setToolTipText("<html>" + ManaSymbols.replaceSymbolsWithHTML(icon.getHint(), ManaSymbols.Type.CARD_ICON_HINT));
308             if (!icon.getText().isEmpty()) {
309                 Graphics2D g2d = iconImageWithText.createGraphics();
310                 g2d.setColor(PreferencesDialog.getCurrentTheme().getCardIconsTextColor(this.color));
311                 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
312                 Rectangle rect = CardRendererUtils.reduceRect(new Rectangle(0, 0, iconImageWithText.getWidth(), iconImageWithText.getHeight()), 0.8f);
313                 CardRendererUtils.drawCenteredText(g2d, icon.getText(), rect, this.font, true);
314                 g2d.dispose();
315             }
316 
317             // the stretch icon can occupy all space (full grid's cell)
318             StretchIcon s = new StretchIcon(iconImageWithText, true);
319             label.setIcon(s);
320             label.setIconTextGap(0);
321             if (DebugUtil.GUI_CARD_ICONS_DRAW_ICON_BORDER) {
322                 label.setBorder(BorderFactory.createLineBorder(Color.green));
323             }
324             return label;
325         }
326         return null;
327     }

329     private void calcSizes(Rectangle cardSize) {
330         // small mode takes 20% of card's sizes diff
331         boolean smallMode = false;
332         int maxW = GUISizeHelper.battlefieldCardMaxDimension.width;
333         int minW = GUISizeHelper.battlefieldCardMinDimension.width;
334         if (minW > maxW) {
335             // on wrong settings
336             maxW = GUISizeHelper.battlefieldCardMinDimension.width;
337             minW = GUISizeHelper.battlefieldCardMaxDimension.width;
338         }
339         // cardSize.width < 120 - disable small mode on too big cards
340         if (cardSize.width < MAXIMUM_CARD_WIDTH_FOR_ICONS_SMALL_MODE && (cardSize.width < minW + (maxW - minW) * 0.2f)) {
341             smallMode = true;
342         }
343 
344         // auto-sizeable icons (smaller for small card, normal for big)
345         this.calcedSizeSizePercent = this.iconSizePercent;
346         if (smallMode) {
347             this.calcedSizeSizePercent = Math.round(this.calcedSizeSizePercent * 1.5f);
348         }
349 
350         // auto-amount for icons (less for small, normal for big)
351         this.calcedCellsMaxCount = this.cellsMaxCount;
352         this.calcedCellsVisibleCount = this.cellsVisibleCount;
353         this.calcedCellsOffset = this.cellsOffset;
354         this.calcedPosition = this.position;
355         this.calcedOrder = this.order;
356         if (smallMode) {
357             this.calcedCellsMaxCount = Math.min(5, this.calcedCellsMaxCount);
358             this.calcedCellsVisibleCount = Math.min(1, this.calcedCellsVisibleCount);
359             this.calcedCellsOffset = Math.min(1, this.calcedCellsOffset);
360             // change order of multi-icons and ignore corners (make icons it centered)
361             if (this.calcedPosition.getMaxIconsAmount() > 1) {
362                 this.calcedOrder = CardIconOrder.CENTER;
363             }
364         }
365 
366         // REAL SIZES
367 
368         // auto-sizeable gaps (use test render form to find best values)
369         this.iconsGap = Math.floorDiv(cardSize.width, 100) * 2;
370 
371         // icons intersect the card like mtg arena
372         this.halfSize = Math.max(MINIMUM_ICON_SIZE / 2, Math.round(cardSize.width / 100.0f * this.calcedSizeSizePercent / 2.0f));
373         this.font = new Font("Arial", Font.PLAIN + Font.BOLD, Math.round(this.halfSize * 1.5f));
374     }

12  import java.util.Map;

14  import java.util.regex.Pattern;

16  import javax.swing.JComponent;

18  public class ManaBarChart extends JComponent {
19  
20      Map<String, Integer> pips_at_cmcs = new HashMap<>();
21  
22      ManaBarChart() {
23      }
24  
25      ManaBarChart(Map<String, Integer> pips_at_cmcs) {
26          this.pips_at_cmcs = pips_at_cmcs;
27      }
28  
29      @Override
30      public Dimension getPreferredSize() {
31          Dimension preferred = super.getPreferredSize();
32          Dimension minimum = getMinimumSize();
33          Dimension maximum = getMaximumSize();
34          preferred.width = Math.min(Math.max(preferred.width, minimum.width), maximum.width);
35          preferred.height = Math.min(Math.max(preferred.height, minimum.height), maximum.height);
36          return preferred;
37      }
38  
39      @Override
40      public void paint(Graphics g) {
41          drawBar((Graphics2D) g, getBounds());
42      }
43  
44      void drawBar(Graphics2D g, Rectangle area) {
45          Pattern regex = Pattern.compile("^([0-9]+)##(.)}");
46          Map<Integer, Integer> totals_at_cmcs = new HashMap<>();
47          int max_num_pips = 0;
48          int max_cmc = 0;
49  
50          for (String key : pips_at_cmcs.keySet()) {
51              Matcher regexMatcher = regex.matcher(key);
52              int num_pips = pips_at_cmcs.get(key);
53              while (regexMatcher.find()) {
54                  String cmc = regexMatcher.group(1);
55                  int cmc_num = Integer.parseInt(cmc);
56                  if (max_cmc < cmc_num) {
57                      max_cmc = cmc_num;
58                  }
59  
60                  int total_at_cmc = 0;
61                  if (totals_at_cmcs.get(cmc_num) != null) {
62                      total_at_cmc = totals_at_cmcs.get(cmc_num);
63                  }
64                  totals_at_cmcs.put(cmc_num, total_at_cmc + num_pips);
65                  if (max_num_pips < total_at_cmc + num_pips) {
66                      max_num_pips = total_at_cmc + num_pips;
67                  }
68              }
69          }
70  
71          if (max_num_pips <= 0) {
72              max_num_pips = 1;
73          }
74          int height_factor = 200 / max_num_pips;
75          int width_factor = 200 / (max_cmc + 2);
76          if (width_factor > 20) {
77              width_factor = 20;
78          }
79          if (width_factor < 11) {
80              width_factor = 11;
81          }
82  
83          g.setColor(new Color(130, 130, 130));
84          for (int i = 0; i < max_num_pips; i++) {
85              if (max_num_pips > 10) {
86                  if (i % 10 == 0) {
87                      g.drawLine(0, 200 - 1 - i * height_factor, 400, 200 - 1 - i * height_factor);
88                  } else if (i % 10 == 5) {
89                      Stroke dashed = new BasicStroke(1, BasicStroke.CAP_BUTT, BasicStroke.JOIN_BEVEL, 0, new float[]{5}, 0);
90                      Stroke oldstroke = g.getStroke();
91                      g.setStroke(dashed);
92                      g.drawLine(0, 200 - 1 - i * height_factor, 400, 200 - 1 - i * height_factor);
93                      g.setStroke(oldstroke);
94                  }
95              } else if (i % 2 == 0) {
96                  g.drawLine(0, 200 - 1 - i * height_factor, 400, 200 - 1 - i * height_factor);
97              } else if (i % 2 == 1) {
98                  Stroke dashed = new BasicStroke(1, BasicStroke.CAP_BUTT, BasicStroke.JOIN_BEVEL, 0, new float[]{5}, 0);
99                  Stroke oldstroke = g.getStroke();
100                 g.setStroke(dashed);
101                 g.drawLine(0, 200 - 1 - i * height_factor, 400, 200 - 1 - i * height_factor);
102                 g.setStroke(oldstroke);
103             }
104         }
105 
106         for (int i = 0; i < 17; i++) {
107             if (i % 5 == 0) {
108                 g.drawLine(width_factor * i, 200, width_factor * i, 190);
109             } else {
110                 g.drawLine(width_factor * i, 200, width_factor * i, 195);
111             }
112 
113         }
114 
115         Map<Integer, Integer> running_totals_at_cmcs = new HashMap<>();
116         for (String key : pips_at_cmcs.keySet()) {
117             Matcher regexMatcher = regex.matcher(key);
118             int num_pips = pips_at_cmcs.get(key);
119             while (regexMatcher.find()) {
120                 String cmc = regexMatcher.group(1);
121                 int cmc_num = Integer.parseInt(cmc);
122                 String color = regexMatcher.group(2);
123 
124                 int total_at_cmc = 0;
125                 if (running_totals_at_cmcs.get(cmc_num) != null) {
126                     total_at_cmc = running_totals_at_cmcs.get(cmc_num);
127                 }
128 
129                 if (color.equalsIgnoreCase("w")) {
130                     g.setColor(Color.WHITE);
131                 }
132                 if (color.equalsIgnoreCase("u")) {
133                     g.setColor(Color.BLUE);
134                 }
135                 if (color.equalsIgnoreCase("b")) {
136                     g.setColor(Color.BLACK);
137                 }
138                 if (color.equalsIgnoreCase("r")) {
139                     g.setColor(Color.RED);
140                 }
141                 if (color.equalsIgnoreCase("g")) {
142                     g.setColor(Color.GREEN);
143                 }
144                 if (color.equalsIgnoreCase("c")) {
145                     g.setColor(Color.DARK_GRAY);
146                 }
147                 g.fill(new Rectangle2D.Double(cmc_num * width_factor, 200 - 1 - total_at_cmc - num_pips * height_factor, 10, num_pips * height_factor));
148                 running_totals_at_cmcs.put(cmc_num, total_at_cmc + num_pips * height_factor);
149             }
150         }
151 
152     }

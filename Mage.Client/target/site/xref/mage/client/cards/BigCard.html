75      public void setCard(UUID cardId, EnlargeMode enlargeMode, Image image, List<String> strings, boolean rotate) {
76          if (rotate && getWidth() > getHeight()) {
77              image = TransformedImageCache.getRotatedResizedImage((BufferedImage)image, getHeight(), getWidth(), Math.toRadians(90.0));
78          } else {
79              image = TransformedImageCache.getResizedImage((BufferedImage)image, getWidth(), getHeight());
80          }
81  
82          if (this.cardId == null || enlargeMode != this.enlargeMode || !this.cardId.equals(cardId)) {
83              if (this.panel != null) {
84                  remove(this.panel);
85              }
86              this.cardId = cardId;
87              this.enlargeMode = enlargeMode;
88              bigImage = image;
89              synchronized (this) {
90                  source = null;
91                  hue = 0.000f;
92              } 
93              StringBuilder displayedText = new StringBuilder();
94              for (String textLine: strings) {
95                  if (textLine != null && !textLine.replace(".", "").trim().isEmpty()) {
96                      displayedText.append("<p style='margin: 2px'>").append(textLine).append("</p>");
97                  }                
98              }        
99              this.text.setText(displayedText.toString());
100             repaint();
101         }

3   import java.awt.Color;
4   import java.awt.Dimension;
5   import java.awt.Graphics;
6   import java.awt.Graphics2D;
7   import java.awt.Rectangle;
8   import java.io.Serializable;
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  import javax.swing.JComponent;
13  
14  class Slice implements Serializable {
15  
16      double value;
17      Color color;
18  
19      public Slice(double value, Color color) {
20          this.value = value;
21          this.color = color;
22      }
23  }
24  
25  public class ManaPieChart extends JComponent {
26  
27     private List<Slice> slices = new ArrayList<>();
28  
29      ManaPieChart() {
30      }
31  
32      ManaPieChart(Integer w, Integer u, Integer b, Integer r, Integer g, Integer c) {
33          if (w != null && w > 0) {
34              slices.add(new Slice(w, Color.WHITE));
35          }
36          if (u != null && u > 0) {
37              slices.add(new Slice(u, Color.BLUE));
38          }
39          if (b != null && b > 0) {
40              slices.add(new Slice(b, Color.BLACK));
41          }
42          if (r != null && r > 0) {
43              slices.add(new Slice(r, Color.RED));
44          }
45          if (g != null && g > 0) {
46              slices.add(new Slice(g, Color.GREEN));
47          }
48          if (c != null && c > 0) {
49              slices.add(new Slice(c, Color.LIGHT_GRAY));
50          }
51      }
52  
53      @Override
54      public Dimension getPreferredSize() {
55          Dimension preferred = super.getPreferredSize();
56          Dimension minimum = getMinimumSize();
57          Dimension maximum = getMaximumSize();
58          preferred.width = Math.min(Math.max(preferred.width, minimum.width), maximum.width);
59          preferred.height = Math.min(Math.max(preferred.height, minimum.height), maximum.height);
60          return preferred;
61      }
62  
63      @Override
64      public void paint(Graphics g) {
65          drawPie((Graphics2D) g, getBounds(), slices.toArray(new Slice[slices.size()]));
66      }
67  
68      void drawPie(Graphics2D g, Rectangle area, Slice[] slices) {
69          double total = 0.0D;
70          for (int i = 0; i < slices.length; i++) {
71              total += slices[i].value;
72          }
73          
74          if (total == 0.0D) {
75              return; //there are no slices or no slices with a value > 0, stop here
76          }
77  
78          double curValue = 0.0D;
79          int startAngle = 0;
80          int lastAngle = 0;
81          for (int i = 0; i < slices.length; i++) {
82              startAngle = lastAngle;
83              int arcAngle = (int) (slices[i].value * 360 / total);
84  
85              g.setColor(slices[i].color);
86              g.fillArc(area.x, area.y, area.width - 20, area.height - 20, startAngle, arcAngle);
87              curValue += slices[i].value;
88              lastAngle += arcAngle;
89          }
90      }
91  }

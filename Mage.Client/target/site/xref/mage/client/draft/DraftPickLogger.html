3   import java.io.File;
4   import java.io.IOException;
5   import java.nio.file.Files;
6   import java.nio.file.Path;
7   import java.nio.file.StandardOpenOption;
8   import java.text.SimpleDateFormat;
9   import java.util.Date;
10  import java.util.UUID;
11  
12  import org.apache.log4j.Logger;
13  
14  import mage.view.DraftView;
15  
16  public class DraftPickLogger {
17  
18    private static final Logger LOGGER = Logger.getLogger(DraftPickLogger.class);
19  
20    private final Path logPath;
21    private final boolean logging;
22    private boolean headerWritten = false;
23  
24    public DraftPickLogger(File directory, String logFilename) {
25      this.logging = true;
26      if (!directory.exists()) {
27        directory.mkdirs();
28      }
29      this.logPath = new File(directory, logFilename).toPath();
30      try {
31        Files.write(logPath, new byte[0], StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING);
32      } catch (IOException ex) {
33        LOGGER.error(null, ex);
34      }
35    }
36  
37    public DraftPickLogger() {
38      this.logging = false;
39      this.logPath = null;
40    }
41  
42    public void updateDraft(UUID draftId, DraftView draftView) {
43      if (headerWritten) {
44        return;
45      }
46      headerWritten = true;
47      Date now = new Date();
48      SimpleDateFormat formatter = new SimpleDateFormat("M/d/yyyy h:mm:ss a");
49      StringBuilder buffer = new StringBuilder()
50        .append("Event #: ").append(draftId).append("\n")
51        .append("Time:    ").append(formatter.format(now)).append('\n');
52      buffer.append("Players:\n");
53      for (String player : draftView.getPlayers()) {
54        buffer.append("    ").append(player).append('\n');
55      }
56      buffer.append('\n');
57      appendToDraftLog(buffer.toString());
58    }
59  
60    public void logPick(String setCode, int packNo, int pickNo, String pick, String[] currentBooster) {
61      StringBuilder b = new StringBuilder();
62      if (pickNo == 1) {
63        b.append("------ ").append(setCode).append(" ------\n\n");
64      }
65      b.append("Pack ").append(packNo).append(" pick ").append(pickNo).append(":\n");
66      for (String name : currentBooster) {
67        b.append(pick.equals(name) ? "--> " : "    ");
68        b.append(name);
69        b.append('\n');
70      }
71      b.append('\n');
72      appendToDraftLog(b.toString());
73    }
74  
75    private void appendToDraftLog(String data) {
76      if (logging) {
77        try {
78          Files.write(logPath, data.getBytes(), StandardOpenOption.APPEND);
79        } catch (IOException ex) {
80          LOGGER.error(null, ex);
81        }
82      }
83    }
84  
85  }

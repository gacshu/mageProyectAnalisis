23  /**
24   * This class stores ALL card images in a cache with soft values. this means
25   * that the images may be garbage collected when they are not needed any more,
26   * but will be kept as long as possible.
27   * <p>
28   * Key format: "[cardname]#[setname]#[type]#[collectorID]#[image size]#[additional data]"
29   *
30   * <li>#Normal: request for unrotated image</li>
31   * <li>#Tapped: request for rotated image</li>
32   * <li>#Cropped: request for cropped image that is used for Shandalar like card
33   * look</li>
34   * </ul>
35   *
36   * @author JayDi85
37   */
38  public final class ImageCache {
39  
40      private static final Logger LOGGER = Logger.getLogger(ImageCache.class);
41  
42      private static final SoftValuesLoadingCache<String, ImageCacheData> IMAGE_CACHE; // cards and tokens
43      private static final SoftValuesLoadingCache<String, ImageCacheData> FACE_IMAGE_CACHE;
44      private static final SoftValuesLoadingCache<String, ImageCacheData> CARD_ICONS_CACHE;
45  
46      /**
47       * Common pattern for keys. See ImageCache.getKey for structure info
48       */
49      private static final Pattern KEY_PATTERN = Pattern.compile("(.*)#(.*)#(.*)#(.*)#(.*)");
50      private static final Pattern CARD_ICON_KEY_PATTERN = Pattern.compile("(.*)#(.*)#(.*)");
51  
52      static {
53          // softValues() = Specifies that each value (not key) stored in the map should be wrapped in a SoftReference
54          // (by default, strong references are used). Softly-referenced objects will be garbage-collected in a
55          // globally least-recently-used manner, in response to memory demand.
56          IMAGE_CACHE = SoftValuesLoadingCache.from(key -> {
57              try {
58                  boolean usesVariousArt = false;
59                  if (key.matches(".*#usesVariousArt.*")) {
60                      usesVariousArt = true;
61                      key = key.replace("#usesVariousArt", "");
62                  }
63                  Matcher m = KEY_PATTERN.matcher(key);
64  
65                  if (m.matches()) {
66                      String name = m.group(1);
67                      String setCode = m.group(2);
68                      Integer type = Integer.parseInt(m.group(3));
69                      String collectorId = m.group(4);
70                      if (collectorId.equals("null")) {
71                          collectorId = "0";
72                      }
73  
74                      CardDownloadData info = new CardDownloadData(name, setCode, collectorId, usesVariousArt, type);
75  
76                      boolean cardback = false;
77                      String path;
78                      if (collectorId.isEmpty() || "0".equals(collectorId)) {
79                          // TOKEN
80                          // can be a normal token or a token from card (see embalm ability)
81                          info.setToken(true);
82                          TFile tokenFile;
83  
84                          // try normal token
85                          path = CardImageUtils.buildImagePathToCardOrToken(info);
86                          tokenFile = getTFile(path);
87  
88                          // try token from card
89                          if (tokenFile == null || !tokenFile.exists()) {
90                              CardDownloadData tempInfo = new CardDownloadData(info);
91                              tempInfo.setToken(false);
92                              path = CardImageUtils.buildImagePathToCardOrToken(info);
93                              tokenFile = getTFile(path);
94                          }
95  
96                          // try unknown token image
97                          if (tokenFile == null || !tokenFile.exists()) {
98                              // TODO: replace empty token by other default card, not cardback
99                              cardback = true;
100                             path = CardImageUtils.buildImagePathToDefault(DirectLinksForDownload.cardbackFilename);
101                         }
102                     } else {
103                         // CARD
104                         path = CardImageUtils.buildImagePathToCardOrToken(info);
105                     }
106 
107                     TFile file = getTFile(path);
108                     if (file == null) {
109                         return new ImageCacheData(path, null);
110                     }
111 
112                     if (cardback) {
113                         // TODO: is there any different in images styles? Cardback must be from scryfall, not wizards
114                         // need cardback image
115                         BufferedImage image = loadImage(file);
116                         image = getRoundCorner(image);
117                         return new ImageCacheData(path, image);
118                     } else {
119                         // need normal card image
120                         BufferedImage image = loadImage(file);
121                         image = getWizardsCard(image);
122                         image = getRoundCorner(image);
123                         return new ImageCacheData(path, image);
124                     }
125                 } else {
126                     throw new RuntimeException(
127                             "Requested image doesn't fit the requirement for key (<cardname>#<setname>#<collectorID>): " + key);
128                 }
129             } catch (Exception ex) {
130                 if (ex instanceof ComputationException) {
131                     throw (ComputationException) ex;
132                 } else {
133                     throw new ComputationException(ex);
134                 }
135             }
136         });
137 
138         FACE_IMAGE_CACHE = SoftValuesLoadingCache.from(key -> {
139             try {
140                 Matcher m = KEY_PATTERN.matcher(key);
141 
142                 if (m.matches()) {
143                     String name = m.group(1);
144                     String setCode = m.group(2);
145                     // skip type
146                     // skip collectorId
147 
148                     String path = CardImageUtils.generateFaceImagePath(name, setCode);
149                     TFile file = getTFile(path);
150                     if (file == null) {
151                         return new ImageCacheData(path, null);
152                     }
153 
154                     BufferedImage image = loadImage(file);
155                     return new ImageCacheData(path, image);
156                 } else {
157                     throw new RuntimeException(
158                             "Requested face image doesn't fit the requirement for key (<cardname>#<artid>#: " + key);
159                 }
160             } catch (Exception ex) {
161                 if (ex instanceof ComputationException) {
162                     throw (ComputationException) ex;
163                 } else {
164                     throw new ComputationException(ex);
165                 }
166             }
167         });
168 
169         CARD_ICONS_CACHE = SoftValuesLoadingCache.from(key -> {
170             try {
171                 Matcher m = CARD_ICON_KEY_PATTERN.matcher(key);
172 
173                 if (m.matches()) {
174                     int cardSize = Integer.parseInt(m.group(1));
175                     String resourceName = m.group(2);
176                     CardIconColor cardIconColor = CardIconColor.valueOf(m.group(3));
177                     BufferedImage image = ImageManagerImpl.instance.getCardIcon(resourceName, cardSize, cardIconColor);
178                     return new ImageCacheData(resourceName, image);
179                 } else {
180                     throw new RuntimeException("Wrong card icons image key format: " + key);
181                 }
182             } catch (Exception ex) {
183                 if (ex instanceof ComputationException) {
184                     throw (ComputationException) ex;
185                 } else {
186                     throw new ComputationException(ex);
187                 }
188             }
189         });
190     }

232     public static BufferedImage getRoundCorner(BufferedImage image) {
233         if (image != null) {
234             BufferedImage cornerImage = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.TYPE_INT_ARGB);
235 
236             // corner
237             float ROUNDED_CORNER_SIZE = 0.11f; // see CardPanelRenderModeImage
238             int cornerSizeBorder = Math.max(4, Math.round(image.getWidth() * ROUNDED_CORNER_SIZE));
239 
240             // corner mask
241             Graphics2D gg = cornerImage.createGraphics();
242             gg.setComposite(AlphaComposite.Src);
243             gg.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
244             gg.setColor(Color.white);
245             gg.fill(new RoundRectangle2D.Float(0, 0, cornerImage.getWidth(), cornerImage.getHeight(), cornerSizeBorder, cornerSizeBorder));
246 
247             // image draw to buffer
248             gg.setComposite(AlphaComposite.SrcAtop);
249             gg.drawImage(image, 0, 0, null);
250             gg.dispose();
251 
252             return cornerImage;
253         } else {
254             return image;
255         }
256     }

283     /**
284      * Returns the Image corresponding to the key
285      */
286     private static ImageCacheData getImage(String key) {
287         try {
288             ImageCacheData data = IMAGE_CACHE.getOrNull(key);
289             return data != null ? data : new ImageCacheData("ERROR: key - " + key, null);
290         } catch (ComputationException ex) {
291             // too low memory
292             if (ex.getCause() instanceof NullPointerException) {
293                 return new ImageCacheData("ERROR: low memory?", null);
294             }
295             LOGGER.error(ex, ex);
296             return new ImageCacheData("ERROR: see logs", null);
297         }
298     }

337     /**
338      * Generate key for images cache (it must contain all info to search and load image from the disk)
339      *
340      * @param card
341      * @param cardName  - can be alternative name
342      * @param imageSize - size info, 0 to use original image (with max size)
343      */
344     private static String getKey(CardView card, String cardName, int imageSize) {
345         return (card.isToken() ? cardName.replace(" Token", "") : cardName)
346                 + '#' + card.getExpansionSetCode()
347                 + '#' + card.getType()
348                 + '#' + card.getCardNumber()
349                 + '#' + imageSize
350                 + (card.getUsesVariousArt() ? "#usesVariousArt" : "");
351     }

364     /**
365      * Load image from file
366      *
367      * @param file file to load image from
368      * @return {@link BufferedImage}
369      */
370     public static BufferedImage loadImage(TFile file) {
371         if (file == null) {
372             return null;
373         }
374         if (!file.exists()) {
375             LOGGER.debug("File does not exist: " + file.toString());
376             return null;
377         }
378         BufferedImage image = null;
379         try {
380             try (TFileInputStream inputStream = new TFileInputStream(file)) {
381                 image = ImageIO.read(inputStream);
382             }
383         } catch (Exception e) {
384             LOGGER.error(e.getMessage(), e);
385         }
386 
387         return image;
388     }

390     /**
391      * Returns an image scaled to the size given
392      *
393      * @param original
394      * @return
395      */
396     public static BufferedImage getNormalSizeImage(BufferedImage original) {
397         if (original == null) {
398             return null;
399         }
400 
401         int srcWidth = original.getWidth();
402         int srcHeight = original.getHeight();
403 
404         int tgtWidth = Constants.CARD_SIZE_FULL.width;
405         int tgtHeight = Constants.CARD_SIZE_FULL.height;
406 
407         if (srcWidth == tgtWidth && srcHeight == tgtHeight) {
408             return original;
409         }
410 
411         return TransformedImageCache.getResizedImage(original, tgtWidth, tgtHeight);
412     }

414     /**
415      * Returns the image appropriate to display the card in the picture panel
416      *
417      * @param card
418      * @param width
419      * @param height
420      * @return
421      */
422     public static ImageCacheData getImage(CardView card, int width, int height) {
423         String key = getKey(card, card.getName(), width);
424         ImageCacheData data = getImage(key);
425         if (data.getImage() == null) {
426             LOGGER.debug("Image doesn't exists in the cache: " + key);
427             return data;
428         }
429 
430         double scale = Math.min((double) width / data.getImage().getWidth(), (double) height / data.getImage().getHeight());
431         if (scale >= 1) {
432             return data;
433         }
434 
435         BufferedImage newImage = TransformedImageCache.getResizedImage(data.getImage(), (int) (data.getImage().getWidth() * scale), (int) (data.getImage().getHeight() * scale));
436         data.setImage(newImage);
437         return data;
438     }

457     /**
458      * Returns the image appropriate to display for a card in a picture panel,
459      * but only it was ALREADY LOADED. That is, the call is immediate and will
460      * not block on file IO.
461      *
462      * @param card
463      * @param width
464      * @param height
465      * @return
466      */
467     public static ImageCacheData tryGetImage(CardView card, int width, int height) {
468         String key = getKey(card, card.getName(), width);
469         ImageCacheData data = tryGetImage(key);
470         if (data.getImage() == null) {
471             LOGGER.debug(key + " not found");
472             return data;
473         }
474 
475         double scale = Math.min((double) width / data.getImage().getWidth(), (double) height / data.getImage().getHeight());
476         if (scale >= 1) {
477             return data;
478         }
479 
480         BufferedImage newImage = TransformedImageCache.getResizedImage(data.getImage(), (int) (data.getImage().getWidth() * scale), (int) (data.getImage().getHeight() * scale));
481         data.setImage(newImage);
482         return data;
483     }

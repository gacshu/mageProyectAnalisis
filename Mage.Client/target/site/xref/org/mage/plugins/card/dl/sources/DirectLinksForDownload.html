1   package org.mage.plugins.card.dl.sources;
2   
3   import mage.client.constants.Constants;
4   import org.mage.plugins.card.dl.DownloadJob;
5   
6   import java.io.File;
7   import java.util.*;
8   
9   import static org.mage.plugins.card.dl.DownloadJob.fromURL;
10  import static org.mage.plugins.card.dl.DownloadJob.toFile;
11  import static org.mage.plugins.card.utils.CardImageUtils.getImagesDir;
12  
13  /**
14   * Used when we need to point to direct links to download resources from.
15   *
16   * @author noxx
17   */
18  public class DirectLinksForDownload implements Iterable<DownloadJob> {
19  
20      private static final Map<String, String> directLinks = new LinkedHashMap<>();
21  
22      public static final String cardbackFilename = "cardback.jpg";
23      public static final String foretellFilename = "foretell.jpg";
24  
25      static {
26          directLinks.put(cardbackFilename, "https://upload.wikimedia.org/wikipedia/en/a/aa/Magic_the_gathering-card_back.jpg");
27          directLinks.put(foretellFilename, "https://api.scryfall.com/cards/tkhm/23/en?format=image");
28      }
29  
30      private final File outDir;
31  
32      public DirectLinksForDownload() {
33          this.outDir = new File(getImagesDir() + Constants.RESOURCE_PATH_DEFAULT_IMAGES);
34          if (!outDir.exists()) {
35              outDir.mkdirs();
36          }
37      }
38  
39      @Override
40      public Iterator<DownloadJob> iterator() {
41          List<DownloadJob> jobs = new ArrayList<>();
42  
43          for (Map.Entry<String, String> url : directLinks.entrySet()) {
44              File dst = new File(outDir, url.getKey());
45              jobs.add(new DownloadJob(url.getKey(), fromURL(url.getValue()), toFile(dst)));
46          }
47          return jobs.iterator();
48      }
49  }

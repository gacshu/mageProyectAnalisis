14  /**
15   * Downloader GUI to control and show progress
16   *
17   * @author Clemens Koza
18   */
19  public class DownloadGui extends JPanel {
20      private static final long serialVersionUID = -7346572382493844327L;
21  
22      private final Downloader downloader;
23      private final DownloadListener listener = new DownloadListener();
24      private final BoundedRangeModel progressModel = new DefaultBoundedRangeModel(0, 0, 0, 0);
25      private final JProgressBar progressBar = new JProgressBar(progressModel);
26  
27      private final Map<DownloadJob, DownloadPanel> jobPanels = new HashMap<>();
28      private final JPanel basicPanel = new JPanel();
29  
30      public DownloadGui(Downloader downloader) {
31          super(new BorderLayout());
32          this.downloader = downloader;
33          downloader.addPropertyChangeListener(listener);
34  
35          JPanel p = new JPanel(new BorderLayout());
36          p.setBorder(BorderFactory.createTitledBorder("Progress:"));
37          p.add(progressBar);
38          JButton closeButton = new JButton("X");
39          closeButton.addActionListener(e -> {
40              this.downloader.cleanup();
41          });
42          p.add(closeButton, BorderLayout.EAST);
43          add(p, BorderLayout.NORTH);
44  
45          basicPanel.setLayout(new BoxLayout(basicPanel, BoxLayout.Y_AXIS));
46          JScrollPane pane = new JScrollPane(basicPanel);
47          pane.setPreferredSize(new Dimension(500, 300));
48          add(pane);
49          for (int i = 0; i < downloader.getJobs().size(); i++) {
50              addJob(i, downloader.getJobs().get(i));
51          }
52      }
53  
54      public Downloader getDownloader() {
55          return downloader;
56      }

58      private class DownloadListener implements PropertyChangeListener {
59  
60          @Override
61          public void propertyChange(PropertyChangeEvent evt) {
62              String name = evt.getPropertyName();
63              if (evt.getSource() instanceof DownloadJob) {
64                  // one job changes
65                  DownloadPanel panel = jobPanels.get(evt.getSource());
66                  switch (name) {
67                      case "state":
68                          if (evt.getOldValue() == State.FINISHED || evt.getOldValue() == State.ABORTED) {
69                              // started
70                              changeProgress(-1, 0);
71                          }
72                          if (evt.getNewValue() == State.FINISHED || evt.getOldValue() == State.ABORTED) {
73                              // finished
74                              changeProgress(+1, 0);
75                          }
76                          if (panel != null) {
77                              panel.setVisible(panel.getJob().getState() != State.FINISHED);
78                              panel.revalidate();
79                          }
80                          break;
81                      case "message":
82                          if (panel != null) {
83                              JProgressBar bar = panel.getBar();
84                              String message = panel.getJob().getMessage();
85                              bar.setStringPainted(message != null);
86                              bar.setString(message);
87                          }
88                          break;
89                  }
90              } else if (evt.getSource() == downloader) {
91                  // all jobs changes (add/delete)
92                  if ("jobs".equals(name)) {
93                      IndexedPropertyChangeEvent ev = (IndexedPropertyChangeEvent) evt;
94                      int index = ev.getIndex();
95  
96                      DownloadJob oldValue = (DownloadJob) ev.getOldValue();
97                      if (oldValue != null) {
98                          removeJob(index, oldValue);
99                      }
100 
101                     DownloadJob newValue = (DownloadJob) ev.getNewValue();
102                     if (newValue != null) {
103                         addJob(index, newValue);
104                     }
105                 }
106             }
107         }
108     }
109 
110     private synchronized void addJob(int index, DownloadJob job) {
111         job.addPropertyChangeListener(listener);
112         changeProgress(0, +1);
113         DownloadPanel p = new DownloadPanel(job);
114         jobPanels.put(job, p);
115         basicPanel.add(p, index);
116         basicPanel.revalidate();
117     }
118 
119     private synchronized void removeJob(int index, DownloadJob job) {
120         assert jobPanels.get(job) == basicPanel.getComponent(index);
121         job.removePropertyChangeListener(listener);
122         changeProgress(0, -1);
123         jobPanels.remove(job);
124         basicPanel.remove(index);
125         basicPanel.revalidate();
126     }
127 
128     private synchronized void changeProgress(int progress, int total) {
129         progress += progressModel.getValue();
130         total += progressModel.getMaximum();
131         progressModel.setMaximum(total);
132         progressModel.setValue(progress);
133         this.progressBar.setStringPainted(true);
134         this.progressBar.setString(progress + "/" + total);
135     }
136 
137     private class DownloadPanel extends JPanel {
138         private static final long serialVersionUID = 1187986738303477168L;
139 
140         private final DownloadJob job;
141         private final JProgressBar bar;
142 
143         DownloadPanel(DownloadJob job) {
144             super(new BorderLayout());
145             this.job = job;
146 
147             setBorder(BorderFactory.createTitledBorder(job.getName()));
148             add(bar = new JProgressBar(job.getProgress()));
149             JButton b = new JButton("X");
150             b.addActionListener(e -> {
151                 switch (this.job.getState()) {
152                     case NEW:
153                     case PREPARING:
154                     case WORKING:
155                         this.job.setState(State.ABORTED);
156                 }
157             });
158             add(b, BorderLayout.EAST);
159 
160             if (job.getState() == State.FINISHED | job.getState() == State.ABORTED) {
161                 changeProgress(+1, 0);
162             }
163             setVisible(job.getState() != State.FINISHED);
164 
165             String message = job.getMessage();
166             bar.setStringPainted(message != null);
167             bar.setString(message);
168 
169             Dimension d = getPreferredSize();
170             d.width = Integer.MAX_VALUE;
171             setMaximumSize(d);
172         }
173 
174         DownloadJob getJob() {
175             return job;
176         }
177 
178         JProgressBar getBar() {
179             return bar;
180         }
181     }

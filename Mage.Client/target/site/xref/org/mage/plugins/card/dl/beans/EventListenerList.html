10  /**
11   * The class EventListenerList.
12   *
13   * @version V0.0 08.04.2010
14   * @author Clemens Koza
15   */
16  public class EventListenerList extends javax.swing.event.EventListenerList {
17  
18      private static final long serialVersionUID = -7545754245081842909L;
19  
20      /**
21       * Returns an iterable over all listeners for the specified classes. the
22       * listener classes are in the specified order. for every class, listeners
23       * are in the reverse order of registering. A listener contained multiple
24       * times (for a single or multiple classes) is only returned the first time
25       * it occurs.
26       *
27       * @param <T>
28       * @param listenerClass
29       * @return
30       */
31      public <T extends EventListener> Iterable<T> getIterable(final Class<? extends T>... listenerClass) {
32          //transform class -> iterable
33          List<Iterable<T>> l = Lists.transform(asList(listenerClass), new ClassToIterableFunction<>());
34  
35          //compose iterable (use an arraylist to memoize the function's results)
36          final Iterable<T> it = Iterables.concat(new ArrayList<>(l));
37  
38          //transform to singleton iterators
39          return () -> new SingletonIterator<>(it.iterator());
40      }
41  
42      /**
43       * Returns an iterator over all listeners for the specified classes. the
44       * listener classes are in the specified order. for every class, listeners
45       * are in the reverse order of registering. A listener contained multiple
46       * times (for a single or multiple classes) is only returned the first time
47       * it occurs.
48       *
49       * @param <T>
50       * @param listenerClass
51       * @return
52       */
53      public <T extends EventListener> Iterator<T> getIterator(Class<T>... listenerClass) {
54          return getIterable(listenerClass).iterator();
55      }
56  
57      /**
58       * Iterates backwards over the listeners registered for a class by using the
59       * original array. The Listener runs backwards, just as listener
60       * notification usually works.
61       */
62      private class ListenerIterator<T> extends AbstractIterator<T> {
63  
64          private final Class<? extends T> listenerClass;
65          private final Object[] listeners = listenerList;
66          private int index = listeners.length;
67  
68          private ListenerIterator(Class<? extends T> listenerClass) {
69              this.listenerClass = listenerClass;
70          }
71  
72          @Override
73          @SuppressWarnings("unchecked")
74          protected T computeNext() {
75              for (index -= 2; index >= 0; index -= 2) {
76                  if (listenerClass == listeners[index]) {
77                      return (T) listeners[index + 1];
78                  }
79              }
80              return endOfData();
81          }
82      }
83  
84      /**
85       * Transforms a class to the associated listener iterator
86       */
87      private class ClassToIterableFunction<T> implements Function<Class<? extends T>, Iterable<T>> {
88  
89          public Iterable<T> apply(final Class<? extends T> from) {
90              return () -> new ListenerIterator<>(from);
91          }
92      }
93  
94      /**
95       * Filters the delegate iterator so that every but the first occurrence of
96       * every element is ignored.
97       */
98      private static class SingletonIterator<T> extends AbstractIterator<T> {
99  
100         private final Iterator<T> it;
101         private final HashSet<T> previous = new HashSet<>();
102 
103         public SingletonIterator(Iterator<T> it) {
104             this.it = it;
105         }
106 
107         @Override
108         protected T computeNext() {
109             while (it.hasNext()) {
110                 T next = it.next();
111                 if (previous.add(next)) {
112                     return next;
113                 }
114             }
115             return endOfData();
116         }
117     }

292     private Object[] getSetsForCurrentImageSource() {
293         // Set the available sets to the combo box
294         List<String> supportedSets = selectedSource.getSupportedSets();
295         List<String> setNames = new ArrayList<>();
296 
297         // multiple sets selection
298         if (selectedSource.isCardSource()) {
299             setNames.add(ALL_IMAGES);
300             setNames.add(ALL_MODERN_IMAGES);
301             setNames.add(ALL_STANDARD_IMAGES);
302         }
303         if (selectedSource.isTokenSource()) {
304             setNames.add(ALL_TOKENS);
305         }
306 
307         // single set selection
308         Collection<ExpansionSet> dbSets = Sets.getInstance().values();
309         Collection<String> comboSets = dbSets.stream()
310                 .filter(exp -> supportedSets.contains(exp.getCode()))
311                 .sorted(Comparator.comparing(ExpansionSet::getReleaseDate).reversed())
312                 .map(this::getSetNameWithYear)
313                 .collect(Collectors.toList());
314         setNames.addAll(comboSets);
315 
316         if (setNames.isEmpty()) {
317             logger.error("Source " + selectedSource.getSourceName() + " creates no selectable items.");
318             setNames.add("not available");
319         }
320         return setNames.toArray(new String[0]);
321     }

323     private void reloadCardsToDownload(String selectedItem) {
324         // find selected sets
325         selectedSets.clear();
326         boolean onlyTokens = false;
327         List<String> formatSets;
328         List<String> sourceSets = selectedSource.getSupportedSets();
329         switch (selectedItem) {
330 
331             case ALL_IMAGES:
332                 selectedSets.addAll(selectedSource.getSupportedSets());
333                 break;
334 
335             case ALL_STANDARD_IMAGES:
336                 formatSets = ConstructedFormats.getSetsByFormat(ConstructedFormats.STANDARD);
337                 formatSets.stream()
338                         .filter(sourceSets::contains)
339                         .forEachOrdered(selectedSets::add);
340                 break;
341 
342             case ALL_MODERN_IMAGES:
343                 formatSets = ConstructedFormats.getSetsByFormat(ConstructedFormats.MODERN);
344                 formatSets.stream()
345                         .filter(sourceSets::contains)
346                         .forEachOrdered(selectedSets::add);
347                 break;
348 
349             case ALL_TOKENS:
350                 selectedSets.addAll(selectedSource.getSupportedSets());
351                 onlyTokens = true;
352                 break;
353 
354             default:
355                 // selects one set
356                 ExpansionSet selectedExp = findSetByNameWithYear(selectedItem);
357                 if (selectedExp != null) {
358                     selectedSets.add(selectedExp.getCode());
359                 }
360                 break;
361         }
362 
363         // find missing cards to download
364         cardsDownloadQueue.clear();
365         int numberTokenImagesAvailable = 0;
366         int numberCardImagesAvailable = 0;
367         for (CardDownloadData data : cardsMissing) {
368             if (data.isToken()) {
369                 if (selectedSource.isTokenSource()
370                         && selectedSource.isTokenImageProvided(data.getSet(), data.getName(), data.getType())
371                         && selectedSets.contains(data.getSet())) {
372                     numberTokenImagesAvailable++;
373                     cardsDownloadQueue.add(data);
374                 }
375             } else {
376                 if (!onlyTokens
377                         && selectedSource.isCardSource()
378                         && selectedSource.isCardImageProvided(data.getSet(), data.getName())
379                         && selectedSets.contains(data.getSet())) {
380                     numberCardImagesAvailable++;
381                     cardsDownloadQueue.add(data);
382                 }
383             }
384         }
385         updateProgressText(numberCardImagesAvailable, numberTokenImagesAvailable);
386     }

404     private void updateProgressText(int cardCount, int tokenCount) {
405         missingTokensCount = 0;
406         for (CardDownloadData card : cardsMissing) {
407             if (card.isToken()) {
408                 missingTokensCount++;
409             }
410         }
411         missingCardsCount = cardsMissing.size() - missingTokensCount;
412 
413         uiDialog.setCurrentInfo("Missing: " + missingCardsCount + " card images / " + missingTokensCount + " token images");
414         int imageSum = cardCount + tokenCount;
415         float mb = (imageSum * selectedSource.getAverageSize()) / 1024;
416         updateProgressMessage(String.format(
417                 cardIndex == imageSum
418                         ? "%d of %d (%d cards/%d tokens) image downloads finished! Please close!"
419                         : "%d of %d (%d cards/%d tokens) image downloads finished! Please wait! [%.1f Mb]",
420                 0, imageSum, cardCount, tokenCount, mb
421         ), 0, imageSum);
422     }

429     private static List<CardDownloadData> prepareMissingCards(List<CardInfo> allCards, boolean redownloadMode) {
430 
431         // get filter for Standard Type 2 cards
432         Set<String> type2SetsFilter = new HashSet<>();
433         List<String> constructedFormats = ConstructedFormats.getSetsByFormat(ConstructedFormats.STANDARD);
434         if (constructedFormats != null && !constructedFormats.isEmpty()) {
435             type2SetsFilter.addAll(constructedFormats);
436         } else {
437             logger.warn("No formats defined. Try connecting to a server first!");
438         }
439 
440         // prepare checking list
441         List<CardDownloadData> allCardsUrls = Collections.synchronizedList(new ArrayList<>());
442         try {
443             allCards.parallelStream().forEach(card -> {
444                 if (!card.getCardNumber().isEmpty() && !"0".equals(card.getCardNumber()) && !card.getSetCode().isEmpty()) {
445                     String cardName = card.getName();
446                     boolean isType2 = type2SetsFilter.contains(card.getSetCode());
447                     CardDownloadData url = new CardDownloadData(cardName, card.getSetCode(), card.getCardNumber(), card.usesVariousArt(), 0, false, card.isDoubleFaced(), card.isNightCard());
448 
449                     // variations must have diff file names with additional postfix
450                     if (url.getUsesVariousArt()) {
451                         url.setDownloadName(createDownloadName(card));
452                     }
453 
454                     url.setFlipCard(card.isFlipCard());
455                     url.setSplitCard(card.isSplitCard());
456                     url.setType2(isType2);
457 
458                     // main side
459                     allCardsUrls.add(url);
460 
461                     // second side
462                     // xmage doesn't search night cards by default, so add it and other types manually
463                     if (card.isDoubleFaced()) {
464                         if (card.getSecondSideName() == null || card.getSecondSideName().trim().isEmpty()) {
465                             throw new IllegalStateException("Second side card can't have empty name.");
466                         }
467 
468                         CardInfo secondSideCard = CardRepository.instance.findCardWithPreferredSetAndNumber(card.getSecondSideName(), card.getSetCode(), card.getCardNumber());
469                         if (secondSideCard == null) {
470                             throw new IllegalStateException("Can''t find second side card in database: " + card.getSecondSideName());
471                         }
472 
473                         url = new CardDownloadData(
474                                 card.getSecondSideName(),
475                                 card.getSetCode(),
476                                 secondSideCard.getCardNumber(),
477                                 card.usesVariousArt(),
478                                 0, false, card.isDoubleFaced(), true);
479                         url.setType2(isType2);
480                         allCardsUrls.add(url);
481                     }
482                     if (card.isFlipCard()) {
483                         if (card.getFlipCardName() == null || card.getFlipCardName().trim().isEmpty()) {
484                             throw new IllegalStateException("Flipped card can't have empty name.");
485                         }
486                         CardDownloadData cardDownloadData = new CardDownloadData(
487                                 card.getFlipCardName(),
488                                 card.getSetCode(),
489                                 card.getCardNumber(),
490                                 card.usesVariousArt(),
491                                 0, false, card.isDoubleFaced(), card.isNightCard());
492                         cardDownloadData.setFlipCard(true);
493                         cardDownloadData.setFlippedSide(true);
494                         cardDownloadData.setType2(isType2);
495                         allCardsUrls.add(cardDownloadData);
496                     }
497                     if (card.getMeldsToCardName() != null) {
498                         if (card.getMeldsToCardName().trim().isEmpty()) {
499                             throw new IllegalStateException("MeldsToCardName can't be empty in " + card.getName());
500                         }
501 
502                         CardInfo meldsToCard = CardRepository.instance.findCardWithPreferredSetAndNumber(card.getMeldsToCardName(), card.getSetCode(), card.getCardNumber());
503                         if (meldsToCard == null) {
504                             throw new IllegalStateException("Can''t find meldsToCard in database: " + card.getMeldsToCardName());
505                         }
506 
507                         // meld cards are normal cards from the set, so no needs to set two faces/sides here
508                         url = new CardDownloadData(
509                                 card.getMeldsToCardName(),
510                                 card.getSetCode(),
511                                 meldsToCard.getCardNumber(),
512                                 card.usesVariousArt(),
513                                 0, false, false, false);
514                         url.setType2(isType2);
515                         allCardsUrls.add(url);
516                     }
517                     if (card.isModalDoubleFacesCard()) {
518                         if (card.getModalDoubleFacesSecondSideName() == null || card.getModalDoubleFacesSecondSideName().trim().isEmpty()) {
519                             throw new IllegalStateException("MDF card can't have empty name.");
520                         }
521                         CardDownloadData cardDownloadData = new CardDownloadData(
522                                 card.getModalDoubleFacesSecondSideName(),
523                                 card.getSetCode(),
524                                 card.getCardNumber(),
525                                 card.usesVariousArt(),
526                                 0, false, true, true);
527                         cardDownloadData.setType2(isType2);
528                         allCardsUrls.add(cardDownloadData);
529                     }
530                 } else if (card.getCardNumber().isEmpty() || "0".equals(card.getCardNumber())) {
531                     logger.error("Card has no collector ID and won't be sent to client: " + card.getName());
532                 } else if (card.getSetCode().isEmpty()) {
533                     logger.error("Card has no set name and won't be sent to client:" + card.getName());
534                 } else {
535                     logger.info("Card was not selected: " + card.getName());
536                 }
537             });
538 
539             allCardsUrls.addAll(getTokenCardUrls());
540         } catch (Exception e) {
541             logger.error(e);
542         }
543 
544         // find missing files
545         List<CardDownloadData> cardsToDownload = Collections.synchronizedList(new ArrayList<>());
546         allCardsUrls.parallelStream().forEach(card -> {
547             if (redownloadMode) {
548                 // need all cards
549                 cardsToDownload.add(card);
550             } else {
551                 // need missing cards
552                 File file = new TFile(CardImageUtils.buildImagePathToCardOrToken(card));
553                 if (!file.exists()) {
554                     cardsToDownload.add(card);
555                 }
556             }
557         });
558 
559         return Collections.synchronizedList(new ArrayList<>(cardsToDownload));
560     }

562     public static List<CardDownloadData> getTokenCardUrls() throws RuntimeException {
563         List<CardDownloadData> list = new ArrayList<>();
564         InputStream in = DownloadPicturesService.class.getClassLoader().getResourceAsStream("card-pictures-tok.txt");
565 
566         if (in == null) {
567             logger.error("resources input stream is null");
568             return list;
569         }
570 
571         try (InputStreamReader input = new InputStreamReader(in);
572              BufferedReader reader = new BufferedReader(input)) {
573 
574             String line = reader.readLine();
575             while (line != null) {
576                 line = line.trim();
577                 if (line.startsWith("|")) { // new format
578                     String[] params = line.split("\\|", -1);
579                     if (params.length >= 5) {
580                         int type = 0;
581                         if (params[4] != null && !params[4].isEmpty()) {
582                             type = Integer.parseInt(params[4].trim()); // token number for same names
583                         }
584                         String fileName = "";
585                         if (params.length > 5 && params[5] != null && !params[5].isEmpty()) {
586                             fileName = params[5].trim();
587                         }
588                         String tokenClassName = "";
589                         if (params.length > 7 && params[6] != null && !params[6].isEmpty()) {
590                             tokenClassName = params[6].trim();
591                         }
592 
593                         if (params[1].toLowerCase(Locale.ENGLISH).equals("generate") && params[2].startsWith("TOK:")) {
594                             String set = params[2].substring(4);
595                             CardDownloadData card = new CardDownloadData(params[3], set, "0", false, type,true);
596                             card.setTokenClassName(tokenClassName);
597                             list.add(card);
598                         } else if (params[1].toLowerCase(Locale.ENGLISH).equals("generate") && params[2].startsWith("EMBLEM:")) {
599                             String set = params[2].substring(7);
600                             CardDownloadData card = new CardDownloadData("Emblem " + params[3], set, "0", false, type, true, fileName);
601                             card.setTokenClassName(tokenClassName);
602                             list.add(card);
603                         } else if (params[1].toLowerCase(Locale.ENGLISH).equals("generate") && params[2].startsWith("EMBLEM-:")) {
604                             String set = params[2].substring(8);
605                             CardDownloadData card = new CardDownloadData(params[3] + " Emblem", set, "0", false, type,true, fileName);
606                             card.setTokenClassName(tokenClassName);
607                             list.add(card);
608                         } else if (params[1].toLowerCase(Locale.ENGLISH).equals("generate") && params[2].startsWith("EMBLEM!:")) {
609                             String set = params[2].substring(8);
610                             CardDownloadData card = new CardDownloadData(params[3], set, "0", false, type, true, fileName);
611                             card.setTokenClassName(tokenClassName);
612                             list.add(card);
613                         } else if (params[1].toLowerCase(Locale.ENGLISH).equals("generate") && params[2].startsWith("PLANE:")) {
614                             String set = params[2].substring(6);
615                             CardDownloadData card = new CardDownloadData(params[3], set, "0", false, type, true, fileName);
616                             card.setTokenClassName(tokenClassName);

650     @Override
651     public void run() {
652         this.cardIndex = 0;
653         this.resetErrorCount();
654 
655         File base = new File(getImagesDir());
656         if (!base.exists()) {
657             base.mkdir();
658         }
659 
660         Connection.ProxyType configProxyType = Connection.ProxyType.valueByText(PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_TYPE, "None"));
661         Proxy.Type type = Proxy.Type.DIRECT;
662         switch (configProxyType) {
663             case HTTP:
664                 type = Proxy.Type.HTTP;
665                 break;
666             case SOCKS:
667                 type = Proxy.Type.SOCKS;
668                 break;
669             case NONE:
670             default:
671                 proxy = Proxy.NO_PROXY;
672                 break;
673         }
674 
675         if (type != Proxy.Type.DIRECT) {
676             try {
677                 String address = PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_ADDRESS, "");
678                 Integer port = Integer.parseInt(PreferencesDialog.getCachedValue(PreferencesDialog.KEY_PROXY_PORT, "80"));
679                 proxy = new Proxy(type, new InetSocketAddress(address, port));
680             } catch (Exception ex) {
681                 throw new RuntimeException("Gui_DownloadPicturesService : error 1 - " + ex);
682             }
683         }
684 
685         int downloadThreadsAmount = Integer.parseInt((String) uiDialog.getDownloadThreadsCombo().getSelectedItem());
686 
687         if (proxy != null) {
688             logger.info("Started download of " + cardsDownloadQueue.size() + " images"
689                     + " from source: " + selectedSource.getSourceName()
690                     + ", language: " + selectedSource.getCurrentLanguage().getCode()
691                     + ", threads: " + downloadThreadsAmount);
692             updateProgressMessage("Preparing download list...");
693             if (selectedSource.prepareDownloadList(this, cardsDownloadQueue)) {
694                 update(0, cardsDownloadQueue.size());
695                 ExecutorService executor = Executors.newFixedThreadPool(downloadThreadsAmount);
696                 for (int i = 0; i < cardsDownloadQueue.size() && !this.isNeedCancel(); i++) {
697                     try {
698                         CardDownloadData card = cardsDownloadQueue.get(i);
699 
700                         logger.debug("Downloading image: " + card.getName() + " (" + card.getSet() + ')');
701 
702                         CardImageUrls urls;
703                         if (card.isToken()) {
704                             if (!"0".equals(card.getCollectorId())) {
705                                 continue;
706                             }
707                             urls = selectedSource.generateTokenUrl(card);
708                         } else {
709                             urls = selectedSource.generateCardUrl(card);
710                         }
711 
712                         if (urls == null) {
713                             String imageRef = selectedSource.getNextHttpImageUrl();
714                             String fileName = selectedSource.getFileForHttpImage(imageRef);
715                             if (imageRef != null && fileName != null) {
716                                 imageRef = selectedSource.getSourceName() + imageRef;
717                                 try {
718                                     card.setToken(selectedSource.isTokenSource());
719                                     Runnable task = new DownloadTask(card, imageRef, fileName, selectedSource.getTotalImages());
720                                     executor.execute(task);
721                                 } catch (Exception ex) {
722                                 }
723                             } else if (selectedSource.getTotalImages() == -1) {
724                                 logger.info("Image not available on " + selectedSource.getSourceName() + ": " + card.getName() + " (" + card.getSet() + ')');
725                                 synchronized (sync) {
726                                     update(cardIndex + 1, cardsDownloadQueue.size());
727                                 }
728                             }
729                         } else {
730                             Runnable task = new DownloadTask(card, urls, cardsDownloadQueue.size());
731                             executor.execute(task);
732                         }
733                     } catch (Exception ex) {
734                         logger.error(ex, ex);
735                     }
736                 }
737 
738                 executor.shutdown();
739                 while (!executor.isTerminated()) {
740                     try {
741                         TimeUnit.SECONDS.sleep(1);
742                     } catch (InterruptedException ie) {
743                     }
744                 }
745             }
746         }
747 
748         try {
749             TVFS.umount();
750         } catch (FsSyncException e) {
751             logger.fatal("Couldn't unmount zip files", e);
752             JOptionPane.showMessageDialog(null, "Couldn't unmount zip files", "Error", JOptionPane.ERROR_MESSAGE);
753         } finally {
754             //
755         }
756 
757         // stop
758         reloadCardsToDownload(uiDialog.getSetsCombo().getSelectedItem().toString());
759         enableDialogButtons();
760 
761         // reset images cache
762         ImageCache.clearCache();
763     }

772         private final CardDownloadData card;
773         private final CardImageUrls urls;

778         DownloadTask(CardDownloadData card, CardImageUrls urls, int count) {
779             this.card = card;
780             this.urls = urls;
781             this.count = count;
782             this.actualFilename = "";
783             useSpecifiedPaths = false;

786         DownloadTask(CardDownloadData card, String baseUrl, String actualFilename, int count) {
787             this.card = card;
788             this.urls = new CardImageUrls(baseUrl, null);
789             this.count = count;
790             this.actualFilename = actualFilename;
791             useSpecifiedPaths = true;

794         @Override
795         public void run() {
796             if (DownloadPicturesService.getInstance().isNeedCancel()) {
797                 synchronized (sync) {
798                     update(cardIndex + 1, count);
799                 }
800                 return;
801             }
802 
803             TFile fileTempImage;
804             TFile destFile;
805             try {
806 
807                 if (card == null) {
808                     synchronized (sync) {
809                         update(cardIndex + 1, count);
810                     }
811                     return;
812                 }
813 
814                 // gen temp file (download to images folder)
815                 String tempPath = getImagesDir() + File.separator + "downloading" + File.separator;
816                 if (useSpecifiedPaths) {
817                     fileTempImage = new TFile(tempPath + actualFilename + "-" + card.hashCode() + ".jpg");
818                 } else {
819                     fileTempImage = new TFile(tempPath + CardImageUtils.prepareCardNameForFile(card.getName()) + "-" + card.hashCode() + ".jpg");
820                 }
821                 TFile parentFile = fileTempImage.getParentFile();
822                 if (parentFile != null) {
823                     if (!parentFile.exists()) {
824                         parentFile.mkdirs();
825                     }
826                 }
827 
828                 // gen dest file name
829                 if (useSpecifiedPaths) {
830                     if (card.isToken()) {
831                         destFile = new TFile(CardImageUtils.buildImagePathToSet(card) + actualFilename + ".jpg");
832                     } else {
833                         destFile = new TFile(CardImageUtils.buildImagePathToTokens() + actualFilename + ".jpg");
834                     }
835                 } else {
836                     destFile = new TFile(CardImageUtils.buildImagePathToCardOrToken(card));
837                 }
838 
839                 // FILE already exists (in zip or in dir)
840                 // don't use, images can be re-downloaded
841                 /*
842                 if (destFile.exists()) {
843                     synchronized (sync) {
844                         update(cardIndex + 1, count);
845                     }
846                     return;
847                 }
848                 */
849 
850                 // check zip access
851                 TFile testArchive = destFile.getTopLevelArchive();
852                 if (testArchive != null && testArchive.exists()) {
853                     try {
854                         testArchive.list();
855                     } catch (Exception e) {
856                         logger.error("Error reading archive, it's can be corrupted. Try to delete it: " + testArchive.toString());
857 
858                         synchronized (sync) {
859                             update(cardIndex + 1, count);
860                         }
861                         return;
862                     }
863                 }
864 
865                 // can download images from many alternative urls
866                 List<String> downloadUrls;
867                 if (this.urls != null) {
868                     downloadUrls = this.urls.getDownloadList();
869                 } else {
870                     downloadUrls = new ArrayList<>();
871                 }
872 
873                 boolean isDownloadOK = false;
874                 URLConnection httpConn = null;
875                 List<String> errorsList = new ArrayList<>();
876                 for (String currentUrl : downloadUrls) {
877                     URL url = new URL(currentUrl);
878 
879                     // on download cancel need to stop
880                     if (DownloadPicturesService.getInstance().isNeedCancel()) {
881                         return;
882                     }
883 
884                     // download
885                     selectedSource.doPause(url.getPath());
886 
887                     httpConn = url.openConnection(proxy);
888                     if (httpConn != null) {
889 
890                         httpConn.setRequestProperty("User-Agent", "Mozilla/5.0 (Macintosh; U; Intel Mac OS X 10.4; en-US; rv:1.9.2.2) Gecko/20100316 Firefox/3.6.2");
891                         try {
892                             httpConn.connect();
893                         } catch (SocketException e) {
894                             incErrorCount();
895                             errorsList.add("Wrong image URL or java app is not allowed to use network. Check your firewall or proxy settings. Error: " + e.getMessage() + ". Image URL: " + url.toString());
896                             break;
897                         } catch (UnknownHostException e) {
898                             incErrorCount();
899                             errorsList.add("Unknown site. Check your DNS settings. Error: " + e.getMessage() + ". Image URL: " + url.toString());
900                             break;
901                         }
902                         int responseCode = ((HttpURLConnection) httpConn).getResponseCode();
903 
904                         // check result
905                         if (responseCode != 200) {
906                             // show errors only on full fail (all urls were not work)
907                             String info = String.format("Image download failed for %s - %s%s, http code: %d, url: %s",
908                                     card.getSet(),
909                                     card.getName(),
910                                     (card.getDownloadName().equals(card.getName()) ? "" : ", cardDownloadName: " + card.getDownloadName()),
911                                     responseCode,
912                                     url
913                             );
914                             errorsList.add(info);
915 
916                             if (logger.isDebugEnabled()) {
917                                 // Shows the returned html from the request to the web server
918                                 logger.debug("Returned HTML ERROR:\n" + convertStreamToString(((HttpURLConnection) httpConn).getErrorStream()));
919                             }
920 
921                             // go to next try
922                             continue;
923                         } else {
924                             // all fine
925                             isDownloadOK = true;
926                             break;
927                         }
928                     }
929                 }
930 
931                 // can save result
932                 if (isDownloadOK && httpConn != null) {
933                     // save data to temp
934                     try (InputStream in = new BufferedInputStream(httpConn.getInputStream());
935                          OutputStream tfileout = new TFileOutputStream(fileTempImage);
936                          OutputStream out = new BufferedOutputStream(tfileout)) {
937                         byte[] buf = new byte[1024];
938                         int len;
939                         while ((len = in.read(buf)) != -1) {
940                             // user cancelled
941                             if (DownloadPicturesService.getInstance().isNeedCancel()) {
942                                 // stop download, save current state and exit
943                                 TFile archive = destFile.getTopLevelArchive();
944                                 ///* not need to unmout/close - it's auto action
945                                 if (archive != null && archive.exists()) {
946                                     logger.info("User canceled download. Closing archive file: " + destFile.toString());
947                                     try {
948                                         TVFS.umount(archive);
949                                     } catch (Exception e) {
950                                         logger.error("Can't close archive file: " + e.getMessage(), e);
951                                     }
952                                 }
953                                 try {
954                                     TFile.rm(fileTempImage);
955                                 } catch (Exception e) {
956                                     logger.error("Can't delete temp file: " + e.getMessage(), e);
957                                 }
958                                 return;
959                             }
960                             out.write(buf, 0, len);
961                         }
962                     }
963                     // TODO: add two faces card correction? (WTF)
964                     // SAVE final data
965                     if (fileTempImage.exists()) {
966                         if (!destFile.getParentFile().exists()) {
967                             destFile.getParentFile().mkdirs();
968                         }
969                         new TFile(fileTempImage).cp_rp(destFile);
970                         try {
971                             TFile.rm(fileTempImage);
972                         } catch (Exception e) {
973                             logger.error("Can't delete temp file: " + e.getMessage(), e);
974                         }
975 
976                     }
977                 } else {
978                     // download errors
979                     for (String err : errorsList) {
980                         logger.warn(err);
981                     }
982                 }
983 
984             } catch (AccessDeniedException e) {
985                 incErrorCount();
986                 logger.error("Can't access to files: " + card.getName() + "(" + card.getSet() + "). Try rebooting your system to remove the file lock.");
987             } catch (Exception e) {
988                 incErrorCount();
989                 logger.error("Unknown error: " + e.getMessage(), e);
990             } finally {
991             }
992 
993             synchronized (sync) {
994                 update(cardIndex + 1, count);
995             }
996         }

999     private void update(int lastCardIndex, int needDownloadCount) {
1000         this.cardIndex = lastCardIndex;
1001 
1002         if (cardIndex < needDownloadCount) {
1003             // downloading
1004             float mb = ((needDownloadCount - lastCardIndex) * selectedSource.getAverageSize()) / 1024;
1005             updateProgressMessage(String.format("%d of %d image downloads finished! Please wait! [%.1f Mb]",
1006                     lastCardIndex, needDownloadCount, mb), lastCardIndex, needDownloadCount);
1007         } else {
1008             // finished
1009             List<CardDownloadData> downloadedCards = Collections.synchronizedList(new ArrayList<>());
1010             DownloadPicturesService.this.cardsMissing.parallelStream().forEach(cardDownloadData -> {
1011                 TFile file = new TFile(CardImageUtils.buildImagePathToCardOrToken(cardDownloadData));
1012                 if (file.exists()) {
1013                     downloadedCards.add(cardDownloadData);
1014                 }
1015             });
1016 
1017             // remove all downloaded cards, missing must be remains
1018             this.cardsDownloadQueue.removeAll(downloadedCards);
1019             this.cardsMissing.removeAll(downloadedCards);
1020 
1021             if (this.cardsDownloadQueue.isEmpty()) {
1022                 // stop download
1023                 updateProgressMessage("0 images remaining. Please close.");
1024             } else {
1025                 // try download again
1026             }
1027 
1028             enableDialogButtons();
1029         }

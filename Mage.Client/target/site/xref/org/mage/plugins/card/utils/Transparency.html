10  import java.awt.image.ImageFilter;
11  import java.awt.image.ImageProducer;
12  import java.awt.image.RGBImageFilter;
13  
14  public final class Transparency {
15      public static Image makeColorTransparent(Image im, final Color color) {
16          ImageFilter filter = new RGBImageFilter() {
17              // the color we are looking for... Alpha bits are set to opaque
18              public final int markerRGB = color.getRGB() | 0xFF000000;
19  
20              @Override
21              public int filterRGB(int x, int y, int rgb) {
22                  if ((rgb | 0xFF000000) == markerRGB) {
23                      // Mark the alpha bits as zero - transparent
24                      return 0x00FFFFFF & rgb;
25                  } else {
26                      // nothing to do
27                      return rgb;
28                  }
29              }
30          };
31  
32          ImageProducer ip = new FilteredImageSource(im.getSource(), filter);
33          return Toolkit.getDefaultToolkit().createImage(ip);
34      }
35  
36      public static BufferedImage makeImageTranslucent(BufferedImage source,
37              double alpha) {
38          BufferedImage target = new BufferedImage(source.getWidth(), source
39                  .getHeight(), java.awt.Transparency.TRANSLUCENT);
40          // Get the images graphics
41          Graphics2D g = target.createGraphics();
42          // Set the Graphics composite to Alpha
43          g.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
44                  (float) alpha));
45          // Draw the image into the prepared reciver image
46          g.drawImage(source, null, 0, 0);
47          // let go of all system resources in this Graphics
48          g.dispose();
49          // Return the image
50          return target;
51      }
52  }

15  public class CompoundProperties extends AbstractProperties {
16      private final List<Properties> delegates;
17  
18      public CompoundProperties(Properties... delegates) {
19          this.delegates = asList(delegates);
20          Collections.reverse(this.delegates);
21      }
22  
23      public CompoundProperties(List<Properties> delegates) {
24          this.delegates = new ArrayList<>(delegates);
25          Collections.reverse(this.delegates);
26      }
27  
28      public <T> Property<T> property(String name, Property<T> property) {
29          for(Properties p:delegates)
30              property = p.property(name, property);
31          return property;
32      }
33  
34      public <E> List<E> list(String name, List<E> list) {
35          for(Properties p:delegates)
36              list = p.list(name, list);
37          return list;
38      }
39  
40      public <E> Set<E> set(String name, Set<E> set) {
41          for(Properties p:delegates)
42              set = p.set(name, set);
43          return set;
44      }
45  
46      public <K, V> Map<K, V> map(String name, Map<K, V> map) {
47          for(Properties p:delegates)
48              map = p.map(name, map);
49          return map;
50      }
51  }

291     private Map<String, String> getSetLinks(String cardSet) {
292         Map<String, String> setLinks = new HashMap<>();
293         try {
294             String setNames = setsAliases.get(cardSet.toLowerCase(Locale.ENGLISH));
295             Set<String> aliasesStart = new HashSet<>();
296             if (cardNameAliasesStart.containsKey(cardSet)) {
297                 aliasesStart.addAll(cardNameAliasesStart.get(cardSet));
298             }
299             if (setNames == null) {
300                 setNames = cardSet.toLowerCase(Locale.ENGLISH);
301             }
302             Preferences prefs = MageFrame.getPreferences();
303             Connection.ProxyType proxyType = Connection.ProxyType.valueByText(prefs.get("proxyType", "None"));
304             for (String setName : setNames.split("\\^")) {
305                 String URLSetName = CardUtil.urlEncode(setName);
306                 String baseUrl = "http://mythicspoiler.com/" + URLSetName + '/';
307 
308                 Map<String, String> pageLinks = getSetLinksFromPage(cardSet, aliasesStart, prefs, proxyType, baseUrl, baseUrl);
309                 setLinks.putAll(pageLinks);
310 
311                 // try to download images for double-faced cards
312                 try {
313                     String doubleFacedUrl = baseUrl + "dfc.html";
314                     pageLinks = getSetLinksFromPage(cardSet, aliasesStart, prefs, proxyType, baseUrl, doubleFacedUrl);
315                     setLinks.putAll(pageLinks);
316                 } catch (Exception ex) {
317                     // that's ok if we cannot download double-faced cards for some sets
318                 }
319             }
320 
321         } catch (IOException ex) {
322             System.out.println("Exception when parsing the mythicspoiler page: " + ex.getMessage());
323         }
324         return setLinks;
325     }

327     private Map<String, String> getSetLinksFromPage(String cardSet, Set<String> aliasesStart, Preferences prefs,
328                                                     ProxyType proxyType, String baseUrl, String pageUrl) throws IOException {
329         Map<String, String> pageLinks = new HashMap<>();
330 
331         String urlDocument;
332         Document doc;
333         if (proxyType == ProxyType.NONE) {
334             urlDocument = pageUrl;
335             doc = Jsoup.connect(urlDocument).get();
336         } else {
337             String proxyServer = prefs.get("proxyAddress", "");
338             int proxyPort = Integer.parseInt(prefs.get("proxyPort", "0"));
339             URL url = new URL(pageUrl);
340             Proxy proxy = new Proxy(Proxy.Type.HTTP, new InetSocketAddress(proxyServer, proxyPort));
341             HttpURLConnection uc = (HttpURLConnection) url.openConnection(proxy);
342 
343             uc.connect();
344 
345             String line;
346             StringBuffer tmp = new StringBuffer();
347             BufferedReader in = new BufferedReader(new InputStreamReader(uc.getInputStream()));
348             while ((line = in.readLine()) != null) {
349                 tmp.append(line);
350             }
351             doc = Jsoup.parse(String.valueOf(tmp));
352         }
353 
354         Elements cardsImages = doc.select("img[src^=cards/]"); // starts with cards/
355         if (!aliasesStart.isEmpty()) {
356             for (String text : aliasesStart) {
357                 cardsImages.addAll(doc.select("img[src^=" + text + ']'));
358             }
359         }
360 
361         for (Element cardsImage : cardsImages) {
362             String cardLink = cardsImage.attr("src");
363             String cardName = null;
364             if (cardLink.startsWith("cards/") && cardLink.endsWith(".jpg")) {
365                 cardName = cardLink.substring(6, cardLink.length() - 4);
366             } else if (aliasesStart.contains(cardLink)) {
367                 cardName = cardLink.substring(0, cardLink.length() - 4);
368             }
369             if (cardName != null && !cardName.isEmpty()) {
370                 if (cardNameAliases.containsKey(cardSet + '-' + cardName)) {
371                     cardName = cardNameAliases.get(cardSet + '-' + cardName);
372                 } else if (cardName.endsWith("1") || cardName.endsWith("2") || cardName.endsWith("3") || cardName.endsWith("4") || cardName.endsWith("5")) {
373                     cardName = cardName.substring(0, cardName.length() - 1);
374                 } else if (cardName.endsWith("promo")) {
375                     cardName = cardName.substring(0, cardName.length() - 5);
376                 }
377                 pageLinks.put(cardName, baseUrl + cardLink);
378             }
379         }
380         Map<String, String> linksToAdd = manualLinks.get(cardSet);
381         if (linksToAdd != null) {
382             for (Map.Entry<String, String> link : linksToAdd.entrySet()) {
383                 pageLinks.put(link.getKey(), baseUrl + "cards/" + link.getValue() + ".jpg");
384             }
385         }
386         return pageLinks;
387     }

394     @Override
395     public CardImageUrls generateCardUrl(CardDownloadData card) throws Exception {
396         String collectorId = card.getCollectorId();
397         String cardSet = card.getSet();
398         if (collectorId == null || cardSet == null) {
399             throw new Exception("Wrong parameters for image: collector id: " + collectorId + ", card set: " + cardSet);
400         }
401         if (card.isFlippedSide()) { //doesn't support rotated images
402             return null;
403         }
404         Map<String, String> setLinks = sets.computeIfAbsent(cardSet, k -> getSetLinks(cardSet));
405         String searchName = card.getDownloadName().toLowerCase(Locale.ENGLISH)
406                 .replaceAll(" ", "")
407                 .replaceAll("\\.", "")
408                 .replaceAll("&", "and")
409                 .replaceAll("-", "")
410                 .replaceAll("'", "")
411                 .replaceAll(",", "")
412                 .replaceAll("/", "");
413         String link = setLinks.get(searchName);
414         return new CardImageUrls(link);
415     }

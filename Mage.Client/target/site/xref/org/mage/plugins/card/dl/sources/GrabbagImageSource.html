50      @Override
51      public CardImageUrls generateCardUrl(CardDownloadData card) throws Exception {
52          if (singleLinks == null) {
53              setupLinks();
54          }
55  
56          // multiple images card use special a,b,c,d versions
57          String url;
58          String postfix = card.getCollectorIdPostfix();
59          if (postfix.isEmpty()) {
60              url = singleLinks.get(card.getSet() + "/" + card.getName());
61          } else {
62              url = singleLinks.get(card.getSet() + "/" + card.getName() + "-" + postfix);
63          }
64  
65          if (url != null) {
66              return new CardImageUrls(getSourceName(card, url) + url);
67          }
68          return null;
69      }

505     private String getSourceName(CardDownloadData card, String httpImageUrl) {
506         if (card.getSet().equals("MTG")) {
507             return "http://static.starcitygames.com/sales/cardscans/";
508         } else if (card.getSet().equals("SWS")) {
509             return "https://i.imgur.com/";
510         } else {
511             return "http://magiccards.info/scans/en/";

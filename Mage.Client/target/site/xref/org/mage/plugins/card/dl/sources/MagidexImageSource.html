234     @Override
235     public CardImageUrls generateCardUrl(CardDownloadData card) throws Exception {
236         String cardDownloadName = card.getDownloadName().toLowerCase(Locale.ENGLISH);
237         String cardSet = card.getSet();
238 
239         if (cardDownloadName == null || cardSet == null) {
240             throw new Exception("Wrong parameters for image: cardDownloadName: " + cardDownloadName + ",card set: " + cardSet);
241         }
242 
243         if (card.isSplitCard()) {
244             cardDownloadName = cardDownloadName.replaceAll(" // ", "");
245         }
246 
247         // This will properly escape the url
248         URI uri = new URI("http", "magidex.com", "/extstatic/card/" + formatSetName(cardSet) + '/' + cardDownloadName + ".jpg", null, null);
249         return new CardImageUrls(uri.toASCIIString());
250     }
251 
252     private String formatSetName(String setName) {
253         if (setNameReplacement.containsKey(setName)) {
254             setName = setNameReplacement.get(setName);
255         }
256         return setName;
257     }

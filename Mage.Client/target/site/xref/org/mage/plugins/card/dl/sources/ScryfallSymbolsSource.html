60      private void parseData(String sourcePath) {
61  
62          String sourceData = "";
63          try {
64              sourceData = new String(Files.readAllBytes(Paths.get(sourcePath)));
65          } catch (IOException e) {
66              LOGGER.error("Can't open file to parse data: " + sourcePath + " , reason: " + e.getMessage());
67          }
68  
69          // gen symbols list
70          List<String> allMageSymbols = new ArrayList<>(Arrays.asList(SYMBOLS_LIST));
71          for (Integer i = SYMBOLS_NUMBER_START; i <= SYMBOLS_NUMBER_END; i++) {
72              allMageSymbols.add(String.valueOf(SYMBOLS_NUMBER_START + i));
73          }
74  
75          Map<String, String> foundedData = new HashMap<>();
76  
77          // search raw data
78          sourceData = sourceData.replaceAll(".card-symbol", "\n.card-symbol"); // css as one line, but need multiline
79          Pattern regex = Pattern.compile(REGEXP_MANA_PATTERN);
80          Matcher regexMatcher = regex.matcher(sourceData);
81          while (regexMatcher.find()) {
82              String symbolCode = regexMatcher.group(1).trim();
83              String symbolData = regexMatcher.group(2).trim().replace(" ", "").replaceAll("\n", ""); // decoder need only wrapped text as one line
84  
85              foundedData.put(symbolCode, symbolData);
86          }
87  
88          // dirs maker
89          File dir = getSymbolFileNameAsSVG("W").getParentFile();
90          if (!dir.exists()) {
91              dir.mkdirs();
92          }
93  
94          // decode and save data (only if not exist)
95          for (String needCode : allMageSymbols) {
96  
97              String searchCode = needCode.replace("/", "");
98  
99              if (!foundedData.containsKey(searchCode)) {
100                 LOGGER.warn("Can't found symbol code from scryfall: " + searchCode);
101                 continue;
102             }
103 
104             File destFile = getSymbolFileNameAsSVG(searchCode);
105             if (destFile.exists() && (destFile.length() > 0)) {
106                 continue;
107             }
108             try(FileOutputStream stream  = new FileOutputStream(destFile)) {
109                 // base64 transform
110                 String data64 = foundedData.get(searchCode);
111                 Base64.Decoder dec = Base64.getDecoder();
112                 byte[] fileData = dec.decode(data64);
113 
114                 stream.write(fileData);
115 
116                 LOGGER.info("New svg symbol downloaded: " + needCode);
117             } catch (Exception e) {
118                 LOGGER.error("Can't decode svg icon and save to file: " + destFile.getPath() + ", reason: " + e.getMessage());
119             }
120         }
121     }

124     private class ScryfallSymbolsDownloadJob extends DownloadJob {
125 
126         private String cssUrl = ""; // need to find url from colors page https://scryfall.com/docs/api/colors
127 
128         // listener for data parse after download complete
129         private class ScryfallDownloadOnFinishedListener implements PropertyChangeListener {
130             private String downloadedFile;
131 
132             public ScryfallDownloadOnFinishedListener(String ADestFile) {
133                 this.downloadedFile = ADestFile;
134             }
135 
136             @Override
137             public void propertyChange(PropertyChangeEvent evt) {
138 
139                 if (!evt.getPropertyName().equals(STATE_PROP_NAME)) {
140                     throw new IllegalArgumentException("Unknown download property " + evt.getPropertyName());
141                 }
142 
143                 if (evt.getNewValue() != State.FINISHED) {
144                     return;
145                 }
146 
147                 // parse data and save to dest
148                 parseData(this.downloadedFile);
149             }
150         }
151 
152         @Override
153         public void onPreparing() throws Exception {
154             this.cssUrl = "";
155 
156             org.jsoup.nodes.Document doc = CardImageUtils.downloadHtmlDocument(CSS_SOURCE_URL);
157             org.jsoup.select.Elements cssList = doc.select(CSS_SOURCE_SELECTOR);
158             if (cssList.size() == 1) {
159                 this.cssUrl = cssList.first().attr("href");
160             }
161 
162             if (this.cssUrl.isEmpty()) {
163                 throw new IllegalStateException("Can't find stylesheet url from scryfall colors page.");
164             } else {
165                 this.setSource(fromURL(this.cssUrl));
166             }
167         }
168 
169         private String destFile = "";
170 
171         public ScryfallSymbolsDownloadJob() {
172             // download init
173             super("Scryfall symbols source", fromURL(""), toFile(DOWNLOAD_TEMP_FILE)); // url setup on preparing stage
174             this.destFile = DOWNLOAD_TEMP_FILE;
175             this.addPropertyChangeListener(STATE_PROP_NAME, new ScryfallDownloadOnFinishedListener(this.destFile));
176 
177             // clear dest file (always download new data)
178             File file = new File(this.destFile);
179             if (file.exists()) {
180                 file.delete();
181             }
182         }
183     }
184 
185     private DownloadJob generateDownloadJob() {
186         return new ScryfallSymbolsDownloadJob();
187     }

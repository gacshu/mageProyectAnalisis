6   import javax.swing.*;
7   import javax.swing.border.EmptyBorder;
8   import javax.swing.table.DefaultTableCellRenderer;
9   import java.awt.*;
10  import java.awt.image.BufferedImage;
11  import java.util.StringTokenizer;
12  
13  /**
14   * @author JayDi85
15   */
16  public final class ManaSymbolsCellRenderer extends DefaultTableCellRenderer {
17  
18      // base panel to render
19      private final JPanel renderPanel = new JPanel();
20  
21      @Override
22      public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected,
23                                                     boolean hasFocus, int row, int column) {
24  
25          // get table text cell settings
26          DefaultTableCellRenderer baseRenderer = (DefaultTableCellRenderer) table.getDefaultRenderer(String.class);
27          JLabel baseComp = (JLabel) baseRenderer.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
28  
29          // apply settings to mana panel from parent
30          renderPanel.setOpaque(baseComp.isOpaque());
31          renderPanel.setForeground(CardRendererUtils.copyColor(baseComp.getForeground()));
32          renderPanel.setBackground(CardRendererUtils.copyColor(baseComp.getBackground()));
33          renderPanel.setBorder(baseComp.getBorder());
34  
35          // icons size with margin
36          int symbolWidth = GUISizeHelper.symbolTableSize;
37          int symbolHorizontalMargin = 2;
38  
39          // create each mana symbol as child label
40          String manaCost = (String) value;
41          renderPanel.removeAll();
42          renderPanel.setLayout(new BoxLayout(renderPanel, BoxLayout.X_AXIS));
43          if (manaCost != null) {
44              StringTokenizer tok = new StringTokenizer(manaCost, " ");
45              while (tok.hasMoreTokens()) {
46                  String symbol = tok.nextToken();
47  
48                  JLabel symbolLabel = new JLabel();
49                  symbolLabel.setFont(GUISizeHelper.tableFont);
50                  symbolLabel.setBorder(new EmptyBorder(0, symbolHorizontalMargin, 0, 0));
51  
52                  BufferedImage image = ManaSymbols.getSizedManaSymbol(symbol, symbolWidth);
53                  if (image != null) {
54                      // icon
55                      symbolLabel.setIcon(new ImageIcon(image));
56                  } else {
57                      // text (mana symbols withoiut brackets like R U * B)
58                      // TODO: add auto-size text from ManaSymbols.draw
59                      if (symbol.equals(CardInfo.SPLIT_MANA_SEPARATOR_SHORT)) {
60                          symbolLabel.setText(CardInfo.SPLIT_MANA_SEPARATOR_RENDER);
61                      } else {
62                          symbolLabel.setText("{" + symbol + "}");
63                      }
64                      symbolLabel.setOpaque(false);
65                      symbolLabel.setForeground(baseComp.getForeground());
66                      symbolLabel.setBackground(baseComp.getBackground());
67                  }
68  
69                  renderPanel.add(symbolLabel);
70              }
71          }
72  
73          return renderPanel;
74      }
75  }

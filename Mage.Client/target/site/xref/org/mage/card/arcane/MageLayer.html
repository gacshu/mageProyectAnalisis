66      public MageLayer(MageCard mainPanel, CardIconRenderSettings iconsRender) {
67          this.mainPanel = mainPanel;
68          this.mainPanel.setTopPanelRef(this);
69  
70          // component structure (border layout):
71          //  - main container: JLayeredPane (center)
72          //    * layer with card: custom size, border layout
73          //      - main panel + spaces
74          //    * layer with icons: custom size
75          //    * layer with debug drawing
76  
77          // component
78          this.setLayout(new BorderLayout());
79          this.setOpaque(false);
80  
81          // main container
82          this.mainContainer = new JLayeredPane();
83          this.mainContainer.setOpaque(false);
84          this.add(this.mainContainer, BorderLayout.CENTER);
85  
86          // card layer
87          this.mainLayerCard = new JPanel(new BorderLayout());
88          this.mainLayerCard.setOpaque(false);
89          this.mainContainer.add(this.mainLayerCard, (Integer) 0);
90          // main panel + spaces
91          this.mainLayerCard.add(this.mainPanel, BorderLayout.CENTER);
92          this.initEmptySpaces();
93  
94          // icons layer
95          this.mainLayerIcons = new JPanel(null);
96          this.mainLayerIcons.setOpaque(false);
97          this.mainContainer.add(this.mainLayerIcons, (Integer) 10);
98  
99          // debug layer
100         if (DebugUtil.GUI_CARD_DRAW_MOUSE_CONTAINS_BOUNDS) {
101             this.mainLayerDebug = new JPanel(null);
102             this.mainLayerDebug.setOpaque(false);
103             this.mainLayerDebug.setBorder(BorderFactory.createLineBorder(Color.MAGENTA));
104             this.mainContainer.add(this.mainLayerDebug, (Integer) 20);
105         }
106 
107         // init icons panels, real icons and sizes will added by setCardBounds and setSizes
108         this.iconsRender = iconsRender;
109         this.initCardIconsPanels();
110 
111         // Warning, you must ignore outer/draw spaces, use getCardLocation to find a real component and card size/position
112         // If you inherits a MageLayer then you must implements contains(int x, int y) for correct mouse events
113 
114         if (DebugUtil.GUI_CARD_DRAW_OUTER_BORDER) {
115             this.setBorder(BorderFactory.createLineBorder(Color.red));
116         }
117         if (DebugUtil.GUI_CARD_DRAW_INNER_BORDER) {
118             this.mainPanel.setBorder(BorderFactory.createLineBorder(Color.green));
119         }
120     }

140     private void initCardIconsPanels() {
141         this.iconsDebugPanel = null;
142         this.iconsAbilitiesPanel = null;
143         this.iconsPlayablePanel = null;
144         this.iconsCommanderPanel = null;
145 
146         if (this.iconsRender.isDebugMode()) {
147             // DEBUG MODE -- only one debug panel
148             this.iconsDebugPanel = CardIconsPanelFactory.createDebugPanel(this.iconsRender);
149             this.iconsPanels.add(this.iconsDebugPanel);
150         } else {
151             // NORMAL mode -- multiple panels
152             if (PreferencesDialog.getRenderIconsForAbilities()) {
153                 // abilities
154                 this.iconsAbilitiesPanel = CardIconsPanelFactory.createAbilitiesPanel();
155                 this.iconsPanels.add(this.iconsAbilitiesPanel);
156 
157                 // commander
158                 this.iconsCommanderPanel = CardIconsPanelFactory.createCommanderPanel();
159                 this.iconsPanels.add(this.iconsCommanderPanel);
160             }
161             // playable amount
162             if (PreferencesDialog.getRenderIconsForPlayable()) {
163                 this.iconsPlayablePanel = CardIconsPanelFactory.createPlayablePanel();
164                 this.iconsPanels.add(this.iconsPlayablePanel);
165             }
166         }
167 
168         this.iconsPanels.forEach(panel -> this.mainLayerIcons.add(panel));
169     }

247     @Override
248     public void setCardBounds(int x, int y, int width, int height) {
249         // base idea: child layers should not know about parent layer
250         //
251         // render logic:
252         // * scale the current layer to fit additional elemenst like icons
253         // * draw child layer with new sizes
254         //
255         // animation logic (maybe it can be change in the future):
256         // * animation implemented as g2d graphic context scale in Paint() method
257         // * all layers and elements must be moved as one object
258         // * only the main panel (child) can do a calcs for the animation (so send parent sizes to recalc it)
259 
260         // if (this.getTopPanelRef() == this && this.getOriginal().getName().equals("Kathari Remnant")) { // for debug only
261         if (this.getTopPanelRef() == this) { // TODO: is it support multi layer drawing?
262 
263             // scale inner card and create space for additional drawing like icons
264             MageCardSpace innerSpace = getAdditionalSpaces(width, height);
265 
266             // extra space for animation and other drawing
267             // WTF, I'm tired with render calcs, so make BIG draw spaces for any needs
268             MageCardSpace outerSpace = new MageCardSpace(width * 2, width * 2, height * 2, height * 2);
269             //MageCardSpace outerSpace = new MageCardSpace(50, 30, 150, 20);
270             this.lastOuterSpace = outerSpace;
271 
272             // construct new spaces (outer + inner)
273             MageCardSpace fullSpace = MageCardSpace.combine(innerSpace, outerSpace).withDebugColor(innerSpace.getDebugColor());
274             this.setEmptySpaces(fullSpace);
275             //noinspection deprecation - it's ok to use inner setBounds here
276             this.setBounds(x - outerSpace.getLeft(), y - outerSpace.getTop(), width + outerSpace.getWidth(), height + outerSpace.getHeight());
277             mainPanel.setCardBounds(x + innerSpace.getLeft(), y + innerSpace.getTop(), width - innerSpace.getWidth(), height - innerSpace.getHeight());
278         } else {
279             this.setEmptySpaces(0, 0, 0, 0);
280             //noinspection deprecation - it's ok to use inner setBounds here
281             this.setBounds(x, y, width, height);
282             mainPanel.setCardBounds(x, y, width, height);
283         }
284 
285         MageCardLocation location = this.getCardLocation();
286 
287         // panel sizes
288         this.mainLayerCard.setBounds(0, 0, location.getComponentWidth(), location.getComponentHeight());
289         this.mainLayerIcons.setBounds(0, 0, location.getComponentWidth(), location.getComponentHeight());
290 
291         // icons sizes
292         Rectangle cardSize = new Rectangle(location.getCardRelativeX(), location.getCardRelativeY(), location.getCardWidth(), location.getCardHeight());
293         iconsPanels.forEach(panel -> {
294             panel.updateSizes(cardSize);
295         });
296     }

307     private void updateCardIcons(CardView card) {
308         Map<CardIconsPanel, List<CardIcon>> newIcons = new HashMap<>();
309         this.iconsPanels.forEach(panel -> newIcons.put(panel, new ArrayList<>()));
310 
311         List<CardIcon> allIcons = new ArrayList<>();
312         // main icons
313         allIcons.addAll(card.getCardIcons());
314         // playable icons
315         if (card.getPlayableStats().getPlayableImportantAmount() > 0) {
316             allIcons.add(new PlayableCountIcon(card.getPlayableStats()));
317         }
318 
319         // create panels
320         allIcons.forEach(cardIcon -> {
321             CardIconCategory category = cardIcon.getIconType().getCategory();
322             // debug panel must take all icons (position depends on render settings)
323             if (iconsDebugPanel != null) {
324                 newIcons.get(iconsDebugPanel).add(cardIcon);
325             }
326 
327             // playable panel (bottom left corner)
328             if (iconsPlayablePanel != null && category == CardIconCategory.PLAYABLE_COUNT) {
329                 newIcons.get(iconsPlayablePanel).add(cardIcon);
330             }
331 
332             // abilities panel (left side)
333             if (iconsAbilitiesPanel != null && category == CardIconCategory.ABILITY) {
334                 newIcons.get(iconsAbilitiesPanel).add(cardIcon);
335             }
336 
337             // commander panel (top center)
338             if (iconsCommanderPanel != null && category == CardIconCategory.COMMANDER) {
339                 newIcons.get(iconsCommanderPanel).add(cardIcon);
340             }
341         });
342         this.iconsPanels.forEach(panel -> panel.updateIcons(newIcons.get(panel)));
343     }

538     @Override
539     public void paint(Graphics g) {
540         // inner card panel can decide about transform/scale settings (example: tapped),
541         // so the top parent layer must be scaled too
542 
543         // real card can be put on any parent layer's position (see outer spaces), so the render logic here:
544         // * find real card sizes (without outer spaces)
545         // * calc new sizes for the animation/rotation
546         // * apply outer spaces to calculated sizes
547         // TODO: is it support multi layers?
548         int offsetX = this.getOuterSpace().getLeft();
549         int offsetY = this.getOuterSpace().getTop();
550         int extraWidth = this.getOuterSpace().getWidth();
551         int extraHeight = this.getOuterSpace().getHeight();
552         Rectangle componentRect = this.getCardLocation().getComponentBounds();
553         MageCardAnimationSettings settings = getAnimationSettings(
554                 offsetX,
555                 offsetY,
556                 componentRect.width - extraWidth,
557                 componentRect.height - extraHeight
558         );
559 
560         if (!settings.isVisible()) {
561             return;
562         }
563 
564         if (!isValid() || !mainPanel.isValid()) {
565             mainPanel.validate();
566             super.validate();
567         }
568 
569         Graphics2D g2d = (Graphics2D) g;
570         settings.doTransforms(g2d);
571 
572         super.paint(g);
573     }

575     @Override
576     public boolean contains(int x, int y) {
577         // TODO: is it work with multi layer?
578         // Mouse coords checking to find a child component under the mouse (example: show card hint on mouse over or button click)
579         // Swing uses relative coords here (0,0 is component's top left corner)
580         // WARNING, when you fail a parent coord check then all childs goes to ignore (example: top layer miss check,
581         // then no card panel get it and no card hints on mouse over)
582         MageCardLocation needLocation = this.getCardLocation();
583 
584         // TODO: added contains support for icons hint
585         //   implement idea: use custom "contains" methods for all components structure: from top layer to icon label
586         //   another implement idea: save last AffineTransforms from paint method, translate it to current component and check coords (most accurate method)
587 
588         // extra size for icons (workaround to fix tooltips over icons)
589         Rectangle iconsOffset = new Rectangle(0, 0);
590         if (this.iconsPanels.stream().anyMatch(Component::isVisible)) {
591             CardIconsPanel samplePanel = this.iconsPanels.stream().findFirst().get();
592             iconsOffset.x = -samplePanel.getHalfSize();
593             iconsOffset.y = -samplePanel.getHalfSize();
594             iconsOffset.height = samplePanel.getHalfSize();
595             iconsOffset.width = samplePanel.getHalfSize();
596         }
597 
598         Rectangle normalRect = new Rectangle(
599                 needLocation.getCardRelativeX() + iconsOffset.x,
600                 needLocation.getCardRelativeY() + iconsOffset.y,
601                 needLocation.getCardWidth() + iconsOffset.width,
602                 needLocation.getCardHeight() + iconsOffset.height
603         );
604         Rectangle animatedRect = animateCoords(this, normalRect);
605 
606         // debug draw just for color info, real draw will be transformed/animated with card, so you can look at draw rect
607         if (DebugUtil.GUI_CARD_DRAW_MOUSE_CONTAINS_BOUNDS) {
608             this.mainLayerDebug.setBounds(animatedRect.x, animatedRect.y, animatedRect.width, animatedRect.height);
609             if (animatedRect.contains(x, y)) {
610                 this.mainLayerDebug.setBorder(BorderFactory.createLineBorder(Color.green));
611             } else {
612                 this.mainLayerDebug.setBorder(BorderFactory.createLineBorder(Color.MAGENTA));
613             }
614         }
615 
616         return animatedRect.contains(x, y);
617     }

619     public static Rectangle animateCoords(MageCard card, Rectangle normalRect) {
620         int needX = normalRect.x;
621         int needY = normalRect.y;
622         int needW = normalRect.width;
623         int needH = normalRect.height;
624         int cx = needX;
625         int cy = needY;
626         int cw = needW;
627         int ch = needH;
628 
629         if (card.isTapped()) {
630             // TODO: add rotate support for non 90 angles in the future
631             // rotate by 90 only
632             // example before:
633             // * coord: 50, 150
634             // * size: 126 x 176
635             // example after:
636             // * coord: 50, 150 + 176 - 126
637             // * size: 176 x 126
638             cx = needX;
639             cy = needY + needH - needW;
640             cw = needH;
641             ch = needW;
642         }
643 
644         return new Rectangle(cx, cy, cw, ch);
645     }

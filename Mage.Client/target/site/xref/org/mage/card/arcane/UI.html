76      public static ImageIcon getImageIcon (String path) {
77          InputStream stream = null;
78          try {
79              stream = UI.class.getResourceAsStream(path);
80              if (stream == null && new File(path).exists()) {
81                  stream = new FileInputStream(path);
82              }
83              if (stream == null) {
84                  throw new RuntimeException("Image not found: " + path);
85              }
86              byte[] data = new byte[stream.available()];
87              stream.read(data);
88              return new ImageIcon(data);
89          } catch (IOException ex) {
90              throw new RuntimeException("Error reading image: " + path);
91          } finally {
92              StreamUtils.closeQuietly(stream);
93          }
94      }

96      public static void setHTMLEditorKit (JEditorPane editorPane) {
97          editorPane.getDocument().putProperty("imageCache", imageCache); // Read internally by ImageView, but never written.
98  
99          editorPane.setEditorKit(new HTMLEditorKit() {
100             private static final long serialVersionUID = -54602188235105448L;
101 
102             @Override
103             public ViewFactory getViewFactory () {
104                 return new HTMLFactory() {
105                     @Override
106                     public View create(Element elem) {
107                         Object o = elem.getAttributes().getAttribute(StyleConstants.NameAttribute);
108                         if (o instanceof HTML.Tag) {
109                             HTML.Tag kind = (HTML.Tag) o;
110                             if (Objects.equals(kind, HTML.Tag.IMG)) {
111                                 return new ImageView(elem) {
112                                     @Override
113                                     public URL getImageURL() {
114                                         URL url = super.getImageURL();
115                                         URI uri = null;
116                                         try {
117                                             uri = url.toURI();
118                                         } catch (URISyntaxException ex) {
119                                         }
120                                         // Put an image into the cache to be read by other ImageView methods.
121                                         if (uri != null && imageCache.get(uri) == null) {
122                                             imageCache.put(uri, Toolkit.getDefaultToolkit().createImage(url));
123                                         }
124                                         return url;
125                                     }
126                                 };
127                             }
128                         }
129                         return super.create(elem);
130                     }
131                 };
132             }
133         });
134     }

136     public static void setVerticalScrollingView (JScrollPane scrollPane, final Component view) {
137         final JViewport viewport = new JViewport();
138         viewport.setLayout(new ViewportLayout() {
139             private static final long serialVersionUID = 7701568740313788935L;
140             @Override
141             public void layoutContainer (Container parent) {
142                 viewport.setViewPosition(new Point(0, 0));
143                 Dimension viewportSize = viewport.getSize();
144                 int width = viewportSize.width;
145                 int height = Math.max(view.getPreferredSize().height, viewportSize.height);
146                 viewport.setViewSize(new Dimension(width, height));
147             }
148         });
149         viewport.setView(view);
150         scrollPane.setHorizontalScrollBarPolicy(ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
151         scrollPane.setViewport(viewport);
152     }

154     public static String getDisplayManaCost (String manaCost) {
155         // A pipe in the cost means "process left of the pipe as the card color, but display right of the pipe as the cost".
156         int pipePosition = manaCost.indexOf("{|}");
157         if (pipePosition != -1) {
158             manaCost = manaCost.substring(pipePosition + 3);
159         }
160         return manaCost;
161     }

168         if (EventQueue.isDispatchThread()) {
169             runnable.run();
170             return;
171         }
172         try {
173             EventQueue.invokeAndWait(runnable);
174         } catch (InterruptedException ex) {
175         } catch (InvocationTargetException ex) {
176             throw new RuntimeException(ex);
177         }
178     }

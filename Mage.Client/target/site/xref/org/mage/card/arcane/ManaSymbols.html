106         // you can add bigger size for better quality
107         Map<String, BufferedImage> pngImages = manaImages.get(50);

109 
110             File pngPath = new File(getResourceSymbolsPath(ResourceSymbolSize.PNG));
111             if (!pngPath.exists()) {
112                 pngPath.mkdirs();

114 
115             for (String symbol : symbols) {
116                 try {
117                     BufferedImage image = pngImages.get(symbol);
118                     if (image != null) {
119                         File newFile = new File(pngPath.getPath() + File.separator + symbol + ".png");
120                         ImageIO.write(image, "png", newFile);
121                     }
122                 } catch (Exception e) {

134         }

164                     } else {

166                     }
167                 } catch (Exception e) {
168                 }
169             }
170

172             try {

174                 if (!file.exists()) {
175                     file.mkdirs();
176                 }
177                 String pathRoot = getResourceSetsPath(ResourceSetSize.SMALL) + set;

287     private static boolean loadSymbolImages(int size) {
288         // load all symbols to cache
289         // priority: SVG -> GIF
290         // gif remain for backward compatibility
291 
292         AtomicIntegerArray iconErrors = new AtomicIntegerArray(2); // 0 - svg, 1 - gif
293 
294         AtomicBoolean fileErrors = new AtomicBoolean(false);
295         Map<String, BufferedImage> sizedSymbols = new ConcurrentHashMap<>();
296         IntStream.range(0, symbols.length).parallel().forEach(i -> {
297             String symbol = symbols[i];
298             BufferedImage image = null;
299             File file;
300 
301             // svg
302             if (SvgUtils.haveSvgSupport()) {
303                 file = getSymbolFileNameAsSVG(symbol);
304                 if (file.exists()) {
305                     try {
306                         InputStream fileStream = new FileInputStream(file);
307                         image = loadSymbolAsSVG(fileStream, file.getPath(), size, size);
308                     } catch (FileNotFoundException e) {
309                         // it's ok to hide error
310                     }
311                 }
312             }
313 
314             // gif
315             if (image == null) {
316 
317                 iconErrors.incrementAndGet(0); // svg fail
318 
319                 file = getSymbolFileNameAsGIF(symbol, size);
320                 if (file.exists()) {
321                     image = loadSymbolAsGIF(file, size, size);
322                 }
323             }
324 
325             // save
326             if (image != null) {
327                 sizedSymbols.put(symbol, image);
328             } else {
329                 iconErrors.incrementAndGet(1); // gif fail
330                 fileErrors.set(true);
331             }
332         });
333 
334         // total errors
335         String errorInfo = "";
336         if (iconErrors.get(0) > 0) {
337             errorInfo += "SVG fails - " + iconErrors.get(0);
338         }
339         if (iconErrors.get(1) > 0) {
340             if (!errorInfo.isEmpty()) {
341                 errorInfo += ", ";
342             }
343             errorInfo += "GIF fails - " + iconErrors.get(1);
344         }
345 
346         if (!errorInfo.isEmpty()) {
347             logger.warn("Symbols can't be load for size " + size + ": " + errorInfo);
348         }
349 
350         manaImages.put(size, sizedSymbols);
351         return !fileErrors.get();
352     }

354     private static void renameSymbols(String path) {
355         File file = new File(path);
356         if (!file.exists()) {
357             return;
358         }
359 
360         final PathMatcher matcher = FileSystems.getDefault().getPathMatcher("glob:**/*.jpg");
361         try {
362             Files.walkFileTree(Paths.get(path), new SimpleFileVisitor<Path>() {
363                 @Override
364                 public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
365                     if (matcher.matches(file)) {
366                         Path gifPath = file.resolveSibling(file.getFileName().toString().replaceAll("\\.jpg$", ".gif"));
367                         Files.move(file, gifPath, StandardCopyOption.REPLACE_EXISTING);
368                     }
369                     return FileVisitResult.CONTINUE;
370                 }
371             });
372         } catch (IOException e) {
373             logger.error("Couldn't rename mana symbols on " + path, e);
374         }
375     }

398                 break;

605      * @param value

608      */
609     public static synchronized String replaceSymbolsWithHTML(String value, Type type) {
610 
611         // mana cost to HTML images (urls to files)
612         // do not use it for new code - try to suppotr svg render
613         int symbolSize;
614         switch (type) {
615             case TABLE:
616                 symbolSize = GUISizeHelper.symbolTableSize;
617                 break;
618             case CHAT:
619                 symbolSize = GUISizeHelper.symbolChatSize;
620                 break;
621             case DIALOG:
622                 symbolSize = GUISizeHelper.symbolDialogSize;
623                 break;
624             case TOOLTIP:
625                 symbolSize = GUISizeHelper.symbolTooltipSize;
626                 break;
627             case CARD_ICON_HINT:
628             default:
629                 symbolSize = 11;
630                 break;
631         }
632 
633         // auto size
634         ResourceSymbolSize needSize = null;
635         if (symbolSize <= 15) {
636             needSize = ResourceSymbolSize.SMALL;
637         } else if (symbolSize <= 25) {
638             needSize = ResourceSymbolSize.MEDIUM;
639         } else {
640             needSize = ResourceSymbolSize.LARGE;
641         }
642 
643         // replace every {symbol} to <img> link
644         // ignore data backup
645         String replaced = value
646                 .replace("{this}", "|this|");
647 
648         // not need to add different images (width and height do the work)
649         // use best png size (generated on startup) TODO: add reload images after update
650         String htmlImagesPath = getResourceSymbolsPath(ResourceSymbolSize.PNG);
651         htmlImagesPath = htmlImagesPath
652                 .replace("$", "@S@"); // paths with $ will rise error, need escape that
653 
654         replaced = replaced.replace(CardInfo.SPLIT_MANA_SEPARATOR_FULL, CardInfo.SPLIT_MANA_SEPARATOR_RENDER);
655         replaced = REPLACE_SYMBOLS_PATTERN.matcher(replaced).replaceAll(
656                 "<img src='" + filePathToUrl(htmlImagesPath) + "$1$2$3" + ".png' alt='$1$2$3' width="
657                         + symbolSize + " height=" + symbolSize + '>');
658 
659         // replace hint icons
660         if (replaced.contains(HintUtils.HINT_ICON_GOOD)) {
661             replaced = replaced.replace(HintUtils.HINT_ICON_GOOD, GuiDisplayUtil.getHintIconHtml("good", symbolSize) + "&nbsp;");
662         }
663         if (replaced.contains(HintUtils.HINT_ICON_BAD)) {
664             replaced = replaced.replace(HintUtils.HINT_ICON_BAD, GuiDisplayUtil.getHintIconHtml("bad", symbolSize) + "&nbsp;");
665         }
666         if (replaced.contains(HintUtils.HINT_ICON_RESTRICT)) {
667             replaced = replaced.replace(HintUtils.HINT_ICON_RESTRICT, GuiDisplayUtil.getHintIconHtml("restrict", symbolSize) + "&nbsp;");
668         }
669         if (replaced.contains(HintUtils.HINT_ICON_REQUIRE)) {
670             replaced = replaced.replace(HintUtils.HINT_ICON_REQUIRE, GuiDisplayUtil.getHintIconHtml("require", symbolSize) + "&nbsp;");
671         }
672         if (replaced.contains(HintUtils.HINT_ICON_DUNGEON_ROOM_CURRENT)) {
673             replaced = replaced.replace(HintUtils.HINT_ICON_DUNGEON_ROOM_CURRENT, GuiDisplayUtil.getHintIconHtml("arrow-right-square-fill-green", symbolSize) + "&nbsp;");
674         }
675         if (replaced.contains(HintUtils.HINT_ICON_DUNGEON_ROOM_NEXT)) {
676             replaced = replaced.replace(HintUtils.HINT_ICON_DUNGEON_ROOM_NEXT, GuiDisplayUtil.getHintIconHtml("arrow-down-right-square fill-yellow", symbolSize) + "&nbsp;");
677         }
678 
679         // ignored data restore
680         replaced = replaced
681                 .replace("|this|", "{this}")
682                 .replace("@S@", "$");
683 
684         return replaced;
685     }

687     public static String replaceSetCodeWithHTML(String set, String rarity, int size) {
688         if (setImagesExist.containsKey(set)) {
689             int factor = size / 15 + 1;
690             Integer width = setImagesExist.get(set).width * factor;
691             Integer height = setImagesExist.get(set).height * factor;
692             return "<img src='" + filePathToUrl(getResourceSetsPath(ResourceSetSize.SMALL)) + set + '-' + rarity + ".png' alt='" + rarity + "' height='" + height + "' width='" + width + "' >";
693         } else {
694             return set;
695         }
696     }

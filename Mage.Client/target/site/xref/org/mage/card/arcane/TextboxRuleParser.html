4   import mage.view.CardView;
5   import org.apache.log4j.Logger;

7   import java.awt.*;
8   import java.util.ArrayDeque;
9   import java.util.ArrayList;
10  import java.util.Deque;
11  import java.util.List;
12  import java.util.regex.Matcher;
13  import java.util.regex.Pattern;

15  /**
16   * @author StravantUser
17   */
18  public final class TextboxRuleParser {
19  
20      private static final Logger LOGGER = Logger.getLogger(TextboxRuleParser.class);
21  
22      private static final Pattern BasicManaAbility = Pattern.compile("\\{T\\}: Add \\{(\\w)\\}\\.");
23      private static final Pattern LevelAbilityPattern = Pattern.compile("Level (\\d+)-?(\\d*)(\\+?)");
24      private static final Pattern LoyaltyAbilityPattern = Pattern.compile("^(\\+|\\-)(\\d+|X): ");
25      private static final Pattern SimpleKeywordPattern = Pattern.compile("^(\\w+( \\w+)?)\\s*(\\([^\\)]*\\))?\\s*$");
26      private static final Pattern FontColorValuePattern = Pattern.compile("color\\s*=\\s*[\"'](\\w+)[\"']");
27  
28      // Parse a given rule (given as a string) into a TextboxRule, replacing
29      // symbol annotations, italics, etc, parsing out information such as
30      // if the ability is a loyalty ability, and returning an TextboxRule
31      // representing that information, which can be used to render the rule in
32      // the textbox of a card.
33      public static TextboxRule parse(CardView source, String rule, String cardNameToUse) {
34          // List of regions to apply
35          List<TextboxRule.AttributeRegion> regions = new ArrayList<>();
36  
37          // Leveler / loyalty / basic
38          boolean isLeveler = false;
39          int levelFrom = 0;
40          int levelTo = 0;
41  
42          boolean isLoyalty = false;
43          int loyaltyChange = 0;
44  
45          boolean isBasicMana = false;
46          String basicManaSymbol = "";
47  
48          // Parse the attributedString contents
49          int index = 0;
50          int outputIndex = 0;
51  
52          // Is it a simple keyword ability?
53          {
54              Matcher simpleKeywordMatch = SimpleKeywordPattern.matcher(rule);
55              if (simpleKeywordMatch.find()) {
56                  return new TextboxKeywordRule(simpleKeywordMatch.group(1), regions);
57              }
58          }
59  
60          // Is it a basic mana ability?
61          {
62              Matcher basicManaMatcher = BasicManaAbility.matcher(rule);
63              if (basicManaMatcher.find()) {
64                  isBasicMana = true;
65                  basicManaSymbol = basicManaMatcher.group(1);
66              }
67          }
68  
69          // Check if it's a loyalty ability. Must be right at the start of the rule
70          {
71              Matcher loyaltyMatch = LoyaltyAbilityPattern.matcher(rule);
72              if (loyaltyMatch.find()) {
73                  // Get the loyalty change
74                  if (loyaltyMatch.group(2).equals("X")) {
75                      loyaltyChange = TextboxLoyaltyRule.MINUS_X;
76                  } else {
77                      loyaltyChange = Integer.parseInt(loyaltyMatch.group(2));
78                      if (loyaltyMatch.group(1).equals("-")) {
79                          loyaltyChange = -loyaltyChange;
80                      }
81                  }
82                  isLoyalty = true;
83  
84                  // Go past the match
85                  index = loyaltyMatch.group().length();
86              }
87          }
88  
89          Deque<Integer> openingStack = new ArrayDeque<>();
90          StringBuilder build = new StringBuilder();
91          while (index < rule.length()) {
92              int initialIndex = index;
93              char ch = rule.charAt(index);
94              switch (ch) {
95                  case '{': {
96                      // Handling for `{this}`
97                      int closeIndex = rule.indexOf('}', index);
98                      if (closeIndex == -1) {
99                          // Malformed input, nothing to do
100                         ++index;
101                         ++outputIndex;
102                         build.append(ch);
103                     } else {
104                         String contents = rule.substring(index + 1, closeIndex);
105                         if (contents.equals("this") || contents.equals("source")) {
106                             // Replace {this} with the card's name
107                             build.append(cardNameToUse);
108                             index += contents.length() + 2;
109                             outputIndex += cardNameToUse.length();
110                         } else {
111                             Image symbol = ManaSymbols.getSizedManaSymbol(contents.replace("/", ""), 10);
112                             if (symbol != null) {
113                                 // Mana or other inline symbol
114                                 build.append('#');
115                                 regions.add(new TextboxRule.EmbeddedSymbol(contents, outputIndex));
116                                 ++outputIndex;
117                                 index = closeIndex + 1;
118                             } else {
119                                 // Bad entry
120                                 if (contents.equals(CardInfo.SPLIT_MANA_SEPARATOR_FULL)) {
121                                     build.append(CardInfo.SPLIT_MANA_SEPARATOR_RENDER);
122                                     outputIndex += contents.length() - CardInfo.SPLIT_MANA_SEPARATOR_RENDER.length();
123                                 } else {
124                                     build.append('{');
125                                     build.append(contents);
126                                     build.append('}');
127                                     outputIndex += (contents.length() + 2);
128                                 }
129                                 index = closeIndex + 1;
130                             }
131                         }
132                     }
133                     break;
134                 }
135                 case '&':
136                     // Handling for `&mdash;`
137                     if (rule.startsWith("&mdash;", index)) {
138                         build.append('—');
139                         index += 7;
140                         ++outputIndex;
141                     } else if (rule.startsWith("&bull", index)) {
142                         build.append('•');
143                         index += 5;
144                         ++outputIndex;
145                     } else {
146                         LOGGER.error("Bad &...; sequence `" + rule.substring(index + 1, index + 10) + "` in rule.");
147                         build.append('&');
148                         ++index;
149                         ++outputIndex;
150                     }
151                     break;
152                 case '<': {
153                     // Handling for `<i>` and `<br/>`
154                     int closeIndex = rule.indexOf('>', index);
155                     if (closeIndex != -1) {
156                         // Is a tag
157                         String tag = rule.substring(index + 1, closeIndex);
158                         if (tag.charAt(tag.length() - 1) == '/') {
159                             // Pure closing tag (like <br/>)
160                             if (tag.equals("br/")) {
161                                 build.append('\n');
162                                 ++outputIndex;
163                             } else {
164                                 // Unknown
165                                 build.append('<').append(tag).append('>');
166                                 outputIndex += (tag.length() + 2);
167                             }
168                         } else if (tag.charAt(0) == '/') {
169                             // Opening index for the tag
170                             int openingIndex;
171                             if (openingStack.isEmpty()) {
172                                 // Malformed input, just make an empty interval
173                                 openingIndex = outputIndex;
174                             } else {
175                                 openingIndex = openingStack.pop();
176                             }
177 
178                             // What tag is it?
179                             switch (tag) {
180                                 case "/i":
181                                     // Italics
182                                     regions.add(new TextboxRule.ItalicRegion(openingIndex, outputIndex));
183                                     break;
184                                 case "/b":
185                                     // Bold, see if it's a level ability
186                                     String content = build.substring(openingIndex);
187                                     Matcher levelMatch = LevelAbilityPattern.matcher(content);
188                                     if (levelMatch.find()) {
189                                         try {
190                                             levelFrom = Integer.parseInt(levelMatch.group(1));
191                                             if (!levelMatch.group(2).isEmpty()) {
192                                                 levelTo = Integer.parseInt(levelMatch.group(2));
193                                             }
194                                             if (!levelMatch.group(3).isEmpty()) {
195                                                 levelTo = TextboxLevelRule.AND_HIGHER;
196                                             }
197                                             isLeveler = true;
198                                         } catch (Exception e) {
199                                             LOGGER.error("Bad leveler levels in rule `" + rule + "`.");
200                                         }
201                                     }
202                                     break;
203                                 case "/font":
204                                     // Font it is an additional info of a card
205                                     // lets make it blue like it is in tooltip
206                                     regions.add(new TextboxRule.ColorRegion(openingIndex, outputIndex, Color.BLUE));
207                                     break;
208                                 default:
209                                     // Unknown
210                                     build.append('<').append(tag).append('>');
211                                     outputIndex += (tag.length() + 2);
212                                     break;
213                             }
214                         } else // Is it a <br> tag special case? [Why can't it have a closing `/`... =( ]
215                         {
216                             if (tag.equals("br")) {
217                                 build.append('\n');
218                                 ++outputIndex;
219                             } else {
220                                 // Opening tag
221                                 openingStack.push(outputIndex);
222                             }
223                         }
224                         // Skip characters
225                         index = closeIndex + 1;
226                     } else {
227                         // Malformed tag
228                         build.append('<');
229                         ++outputIndex;
230                         ++index;
231                     }
232                     break;
233                 }
234                 default:
235                     // Normal character
236                     ++index;
237                     ++outputIndex;
238                     build.append(ch);
239                     break;
240             }
241             if (outputIndex != build.length()) {
242                 // Somehow our parsing code output symbols but didn't update the output index correspondingly
243                 LOGGER.error("The human is dead; mismatch! Failed on rule: `" + rule + "` due to not updating outputIndex properly.");
244 
245                 // Bail out
246                 build = new StringBuilder(rule);
247                 regions.clear();
248                 break;
249             }
250             if (index == initialIndex) {
251                 // Somehow our parsing failed to consume the
252                 LOGGER.error("Failed on rule `" + rule + "` due to not consuming a character.");
253 
254                 // Bail out
255                 build = new StringBuilder(rule);
256                 regions.clear();
257                 break;
258             }
259         }
260 
261         // Build and return the rule
262         rule = build.toString();
263         if (isLoyalty) {
264             return new TextboxLoyaltyRule(rule, regions, loyaltyChange);
265         } else if (isLeveler) {
266             return new TextboxLevelRule(rule, regions, levelFrom, levelTo);
267         } else if (isBasicMana) {
268             return new TextboxBasicManaRule(rule, regions, basicManaSymbol);
269         } else {
270             return new TextboxRule(rule, regions);
271         }
272     }
273 }

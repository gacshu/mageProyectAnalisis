39      public void startMultiGames() throws Exception {
40          for (int i = 0; i < GAME_START_COUNT; i++) {
41              logger.info("Starting game");
42              startGame();
43          }
44      }
45  
46      private void startGame() throws Exception {
47          frame = null;
48          Thread.setDefaultUncaughtExceptionHandler((t, e) -> logger.fatal(null, e));
49          SwingUtilities.invokeLater(() -> {
50              synchronized (sync) {
51                  try {
52                      frame = new MageFrame();
53                  } catch (Throwable e) {
54                      Assert.fail("Can't start client app");
55                  }
56                  frame.setVisible(true);
57                  sync.notifyAll();
58              }
59          });
60  
61          synchronized (sync) {
62              if (frame == null) {
63                  sync.wait();
64              }
65              ui = MageFrame.getUI();
66              ui.doClick(MageComponents.TABLES_MENU_BUTTON);
67              ui.doClick(MageComponents.NEW_GAME_BUTTON);
68              ui.doClick(MageComponents.NEW_TABLE_OK_BUTTON, 500);
69              ui.doClick(MageComponents.TABLE_WAITING_START_BUTTON);
70          }
71  
72          sleep(3000);
73          frame.setVisible(false);
74      }
75  
76      private void sleep(int ms) {
77          try {
78              TimeUnit.MILLISECONDS.sleep(ms);
79          } catch (Exception e) {
80              e.printStackTrace();
81          }
82      }

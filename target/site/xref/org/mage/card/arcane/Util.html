25      public static final ThreadPoolExecutor threadPool; // uses for card images loading
26      static private int threadCount;
27  
28      static {
29          threadPool = new ThreadPoolExecutor(4, 4, 0L, TimeUnit.MILLISECONDS, new LinkedBlockingQueue(), new ThreadFactory() {
30              @Override
31              public Thread newThread(Runnable runnable) {
32                  threadCount++;
33                  Thread thread = new Thread(runnable, "Util" + threadCount);
34                  thread.setDaemon(true);
35                  return thread;
36              }
37          });
38          threadPool.prestartAllCoreThreads();
39      }

41      public static void broadcast(byte[] data, int port) throws IOException {
42          DatagramSocket socket = null;
43          try {
44              socket = new DatagramSocket();
45              broadcast(socket, data, port, NetworkInterface.getNetworkInterfaces());
46          } finally {
47              StreamUtils.closeQuietly(socket);
48          }
49      }
50  
51      private static void broadcast(DatagramSocket socket, byte[] data, int port, Enumeration<NetworkInterface> ifaces)
52              throws IOException {
53          for (NetworkInterface iface : Collections.list(ifaces)) {
54              for (InetAddress address : Collections.list(iface.getInetAddresses())) {
55                  if (!address.isSiteLocalAddress()) {
56                      continue;
57                  }
58                  // Java 1.5 doesn't support getting the subnet mask, so try the two most common.
59                  byte[] ip = address.getAddress();
60                  ip[3] = -1; // 255.255.255.0
61                  socket.send(new DatagramPacket(data, data.length, InetAddress.getByAddress(ip), port));
62                  ip[2] = -1; // 255.255.0.0
63                  socket.send(new DatagramPacket(data, data.length, InetAddress.getByAddress(ip), port));
64              }
65          }
66      }

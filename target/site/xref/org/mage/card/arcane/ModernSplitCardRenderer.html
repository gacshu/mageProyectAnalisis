20      private static class HalfCardProps {
21  
22          int x, y, w, h, cw, ch;
23  
24          String name;
25          String typeLineString;
26          String manaCostString;
27          ObjectColor color;
28          List<TextboxRule> rules = new ArrayList<>();
29          List<TextboxRule> keywords = new ArrayList<>();
30      }

32      private static final List<CardType> ONLY_LAND_TYPE = Arrays.asList(CardType.LAND);
33  
34      // Right and left halves of the card content
35      private HalfCardProps rightHalf = new HalfCardProps();
36      private HalfCardProps leftHalf = new HalfCardProps();
37  
38      // Where and how big is the divider between the card halves
39      private int dividerAt;
40      private int dividerSize;
41  
42      // Is fuse / aftermath
43      private boolean isFuse = false;
44      private boolean isAftermath = false;

46      public ModernSplitCardRenderer(CardView view) {
47          super(view);
48  
49          rightHalf.manaCostString = ManaSymbols.getClearManaCost(cardView.getRightSplitCostsStr());
50          leftHalf.manaCostString = ManaSymbols.getClearManaCost(cardView.getLeftSplitCostsStr());
51  
52          rightHalf.color = new ObjectColor(cardView.getRightSplitCostsStr());
53          leftHalf.color = new ObjectColor(cardView.getLeftSplitCostsStr());
54  
55          parseRules(view.getRightSplitRules(), rightHalf.keywords, rightHalf.rules);
56          parseRules(view.getLeftSplitRules(), leftHalf.keywords, leftHalf.rules);
57  
58          rightHalf.typeLineString = cardView.getRightSplitTypeLine();
59          leftHalf.typeLineString = cardView.getLeftSplitTypeLine();
60  
61          rightHalf.name = cardView.getRightSplitName();
62          leftHalf.name = cardView.getLeftSplitName();
63  
64          isFuse = view.getRules().stream().anyMatch(rule -> rule.contains("Fuse"));
65          isAftermath = view.getRightSplitRules().stream().anyMatch(rule -> rule.contains("Aftermath"));
66  
67          // It's easier for rendering to swap the card halves here because for aftermath cards
68          // they "rotate" in opposite directions making consquence and normal split cards
69          // have the "right" vs "left" as the top half.
70          if (!isAftermath()) {
71              HalfCardProps tmp = leftHalf;
72              leftHalf = rightHalf;
73              rightHalf = tmp;
74          }
75      }
76  
77      private boolean isAftermath() {
78          return isAftermath;
79      }
80  
81      private boolean isFuse() {
82          return isFuse;
83      }

85      @Override
86      protected void layout(int cardWidth, int cardHeight) {
87          // Pass to parent
88          super.layout(cardWidth, cardHeight);
89  
90          // Decide size of divider
91          if (isAftermath()) {
92              dividerSize = borderWidth;
93              dividerAt = (int) (cardHeight * 0.54);
94          } else {
95              int availHeight = cardHeight - totalContentInset - 3 * borderWidth;
96              dividerSize = borderWidth * 2;
97              dividerAt = (int) (totalContentInset + availHeight * 0.5 - borderWidth);
98          }
99  
100         // Decide size of each halves box
101         rightHalf.x = leftHalf.x = totalContentInset;
102         rightHalf.w = leftHalf.w = cardWidth - 2 * totalContentInset;
103         leftHalf.y = totalContentInset;
104         leftHalf.h = dividerAt - totalContentInset;
105         rightHalf.y = dividerAt + dividerSize;
106         rightHalf.h = cardHeight - rightHalf.y - borderWidth * 3;
107 
108         // Content width / height (Exchanged from width / height if the card part is rotated)
109         if (isAftermath()) {
110             leftHalf.cw = leftHalf.w;
111             leftHalf.ch = leftHalf.h;
112         } else {
113             leftHalf.cw = leftHalf.h;
114             leftHalf.ch = leftHalf.w;
115         }
116         rightHalf.cw = rightHalf.h;
117         rightHalf.ch = rightHalf.w;
118 
119         // Fuse space
120         if (isFuse()) {
121             rightHalf.ch -= boxHeight;
122             leftHalf.ch -= boxHeight;
123         }
124     }

126     @Override
127     protected void drawBackground(Graphics2D g) {
128         if (cardView.isFaceDown()) {
129             drawCardBack(g);
130         } else {
131             { // Left half background (top of the card)
132                 // Set texture to paint the left with
133                 g.setPaint(getBackgroundPaint(leftHalf.color, cardView.getCardTypes(), cardView.getSubTypes()));
134 
135                 // Draw main part (most of card)
136                 g.fillRoundRect(
137                         borderWidth, borderWidth,
138                         cardWidth - 2 * borderWidth, leftHalf.h + contentInset - borderWidth - 2 * cornerRadius + (cornerRadius - 1),
139                         cornerRadius - 1, cornerRadius - 1);
140 
141                 // Draw the M15 rounded "swoosh" at the bottom
142                 g.fillRoundRect(
143                         borderWidth, dividerAt - borderWidth - 4 * cornerRadius,
144                         cardWidth - 2 * borderWidth, cornerRadius * 4,
145                         cornerRadius * 2, cornerRadius * 2);
146 
147                 // Draw the cutout into the "swoosh" for the textbox to lie over
148                 g.fillRect(
149                         borderWidth + contentInset, dividerAt - 2 * borderWidth,
150                         cardWidth - borderWidth * 2 - contentInset * 2, borderWidth * 2);
151             }
152 
153             { // Right half background (bottom half of the card)
154                 // Set texture to paint the right with
155                 g.setPaint(getBackgroundPaint(rightHalf.color, cardView.getCardTypes(), cardView.getSubTypes()));
156 
157                 // Draw the M15 rounded "swoosh"es at the top and bottom
158                 g.fillRoundRect(
159                         borderWidth, dividerAt + dividerSize + borderWidth,
160                         cardWidth - 2 * borderWidth, rightHalf.h - 2 * borderWidth,
161                         cornerRadius * 2, cornerRadius * 2);
162 
163                 // Draw the cutout into the "swoosh" for the textbox to lie over
164                 g.fillRect(
165                         borderWidth + contentInset, dividerAt + dividerSize,
166                         cardWidth - borderWidth * 2 - contentInset * 2, rightHalf.h);
167             }
168         }
169     }

171     @Override
172     protected void drawArt(Graphics2D g) {
173         if (artImage != null && !cardView.isFaceDown()) {
174             if (isAftermath()) {
175                 Rectangle2D topRect = ArtRect.AFTERMATH_TOP.rect;
176                 int topLineY = (int) (leftHalf.ch * TYPE_LINE_Y_FRAC);
177                 drawArtIntoRect(g,
178                         leftHalf.x, leftHalf.y + boxHeight, leftHalf.cw, topLineY - boxHeight,
179                         topRect, false);
180 
181                 Rectangle2D bottomRect = ArtRect.AFTERMATH_BOTTOM.rect;
182                 int bottomLineY = (rightHalf.ch - boxHeight) / 2;
183                 drawArtIntoRect(g,
184                         rightHalf.x + rightHalf.w - bottomLineY, rightHalf.y, bottomLineY - boxHeight, rightHalf.h,
185                         bottomRect, false);
186 
187             } else {
188                 // NOTE: Art rects are reversed here, that is on purpose because we swap the left / right half
189                 // of split cards for rendering for consistency between aftermath and normal split
190                 Rectangle2D topRect = ArtRect.SPLIT_RIGHT.rect;
191                 int topLineY = (int) (leftHalf.ch * TYPE_LINE_Y_FRAC);
192                 drawArtIntoRect(g,
193                         leftHalf.x + boxHeight, leftHalf.y, topLineY - boxHeight, leftHalf.h,
194                         topRect, false);
195 
196                 Rectangle2D bottomRect = ArtRect.SPLIT_LEFT.rect;
197                 int bottomLineY = (int) (rightHalf.ch * TYPE_LINE_Y_FRAC);
198                 drawArtIntoRect(g,
199                         rightHalf.x + boxHeight, rightHalf.y, bottomLineY - boxHeight, rightHalf.h,
200                         bottomRect, false);
201             }
202         }
203     }

205     protected void drawSplitHalfFrame(Graphics2D g, CardPanelAttributes attribs, HalfCardProps half, int typeLineY) {
206         // Get the border paint
207         Color boxColor = getBoxColor(half.color, cardView.getCardTypes(), attribs.isTransformed);
208         Paint textboxPaint = getTextboxPaint(half.color, cardView.getCardTypes(), cardWidth, false);
209         Paint borderPaint = getBorderPaint(half.color, cardView.getCardTypes(), cardWidth);
210 
211         // Draw main frame
212         g.setPaint(borderPaint);
213         g.drawRect(
214                 0, 0,
215                 half.cw - 1, half.ch - 1);
216 
217         // Background of textbox
218         g.setPaint(textboxPaint);
219         g.fillRect(
220                 1, typeLineY,
221                 half.cw - 2, half.ch - typeLineY - 1);
222 
223         // Draw the name line box
224         CardRendererUtils.drawRoundedBox(g,
225                 -borderWidth, 0,
226                 half.cw + 2 * borderWidth, boxHeight,
227                 contentInset,
228                 borderPaint, boxColor);
229 
230         // Draw the type line box
231         CardRendererUtils.drawRoundedBox(g,
232                 -borderWidth, typeLineY,
233                 half.cw + 2 * borderWidth, boxHeight - 4,
234                 contentInset,
235                 borderPaint, boxColor);
236 
237         // Draw the name line
238         drawNameLine(g, attribs, half.name, half.manaCostString,
239                 0, 0,
240                 half.cw, boxHeight);
241 
242         // Draw the type line
243         drawTypeLine(g, attribs, half.typeLineString,
244                 0, typeLineY,
245                 half.cw, boxHeight - 4, true);
246 
247         // Draw the textbox rules
248         drawRulesText(g, half.keywords, half.rules,
249                 2, typeLineY + boxHeight + 2 - 4,
250                 half.cw - 4, half.ch - typeLineY - boxHeight, false);
251     }

253     private Graphics2D getUnmodifiedHalfContext(Graphics2D g) {
254         Graphics2D g2 = (Graphics2D) g.create();
255         g2.translate(leftHalf.x, leftHalf.y);
256         return g2;
257     }
258 
259     private Graphics2D getAftermathHalfContext(Graphics2D g) {
260         Graphics2D g2 = (Graphics2D) g.create();
261         g2.translate(rightHalf.x, rightHalf.y);
262         g2.rotate(Math.PI / 2);
263         g2.translate(0, -rightHalf.w);
264         return g2;
265     }
266 
267     private Graphics2D getLeftHalfContext(Graphics2D g) {
268         Graphics2D g2 = (Graphics2D) g.create();
269         g2.translate(leftHalf.x, leftHalf.y);
270         g2.rotate(-Math.PI / 2);
271         g2.translate(-leftHalf.cw, 0);
272         return g2;
273     }
274 
275     private Graphics2D getRightHalfContext(Graphics2D g) {
276         Graphics2D g2 = (Graphics2D) g.create();
277         g2.translate(rightHalf.x, rightHalf.y);
278         g2.rotate(-Math.PI / 2);
279         g2.translate(-rightHalf.cw, 0);
280         return g2;
281     }

283     @Override
284     protected void drawFrame(Graphics2D g, CardPanelAttributes attribs, BufferedImage image, boolean lessOpaqueRulesTextBox) {
285         if (isAftermath()) {
286             drawSplitHalfFrame(getUnmodifiedHalfContext(g), attribs, leftHalf, (int) (leftHalf.ch * TYPE_LINE_Y_FRAC));
287             drawSplitHalfFrame(getAftermathHalfContext(g), attribs, rightHalf, (rightHalf.ch - boxHeight) / 2);
288         } else {
289             drawSplitHalfFrame(getLeftHalfContext(g), attribs, leftHalf, (int) (leftHalf.ch * TYPE_LINE_Y_FRAC));
290             drawSplitHalfFrame(getRightHalfContext(g), attribs, rightHalf, (int) (rightHalf.ch * TYPE_LINE_Y_FRAC));
291             if (isFuse()) {
292                 Graphics2D g2 = getRightHalfContext(g);
293                 int totalFuseBoxWidth = rightHalf.cw * 2 + 2 * borderWidth + dividerSize;
294                 Paint boxColor = getTextboxPaint(cardView.getColor(), ONLY_LAND_TYPE, totalFuseBoxWidth, false);
295                 Paint borderPaint = getBorderPaint(cardView.getColor(), ONLY_LAND_TYPE, totalFuseBoxWidth);
296                 CardRendererUtils.drawRoundedBox(g2,
297                         -borderWidth, rightHalf.ch,
298                         totalFuseBoxWidth, boxHeight,
299                         contentInset,
300                         borderPaint, boxColor);
301                 drawNameLine(g2, attribs, "Fuse (You may cast both halves from your hand)", "",
302                         0, rightHalf.ch,
303                         totalFuseBoxWidth - 2 * borderWidth, boxHeight);
304             }
305         }

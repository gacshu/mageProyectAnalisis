51      public static void prepareCss(String cssFileName, String cssAdditionalSettings, Boolean forceToCreateCss) {
52          // css must be created all the time, so ignore svg check here
53          //if (!SvgUtils.haveSvgSupport())
54  
55          File cssFile = new File(SvgUtils.getSvgTempFile(cssFileName));
56          if (forceToCreateCss || !cssFile.exists()) {
57  
58              // Rendering hints can't be set programatically, so
59              // we override defaults with a temporary stylesheet.
60              // These defaults emphasize quality and precision, and
61              // are more similar to the defaults of other SVG viewers.
62              // SVG documents can still override these defaults.
63              String css = "svg {"
64                      + CSS_BASE_SETTINGS
65                      + cssAdditionalSettings
66                      + "}";
67              FileWriter w = null;
68              try {
69                  cssFile.getParentFile().mkdirs();
70                  cssFile.createNewFile();
71                  w = new FileWriter(cssFile);
72                  w.write(css);
73              } catch (Throwable e) {
74                  logger.error("Can't create css file for svg: " + cssFile.toPath().toAbsolutePath().toString(), e);
75              } finally {
76                  StreamUtils.closeQuietly(w);
77              }
78          }
79      }

81      /**
82       * Load svg file content as image
83       *
84       * @param svgFile               content (can be resource or file)
85       * @param svgInfo               info to show in logs or errors
86       * @param cssFileName           css settings
87       * @param cssAdditionalSettings additional css settings (warning, if you change additional settings then css file must be re-created)
88       * @param resizeToWidth         image size
89       * @param resizeToHeight        image size
90       * @param useShadow             draw image with shadow (not implemented)
91       * @return can return null on error (some linux systems can have compatibility problem with different java/svg libs)
92       * @throws IOException
93       */
94      public static BufferedImage loadSVG(InputStream svgFile, String svgInfo,
95                                          String cssFileName, String cssAdditionalSettings,
96                                          int resizeToWidth, int resizeToHeight, boolean useShadow) throws IOException {
97          if (svgFile == null) {
98              throw new IllegalArgumentException("Empty svg data or unknown file");
99          }
100 
101         // load SVG image
102         // base loader code: https://stackoverflow.com/questions/11435671/how-to-get-a-buffererimage-from-a-svg
103         // resize code: https://vibranttechie.wordpress.com/2015/05/15/svg-loading-to-javafx-stage-and-auto-scaling-when-stage-resize/
104         useShadow = false; // TODO: implement shadow drawing
105         if (useShadow && ((resizeToWidth <= 0) || (resizeToHeight <= 0))) {
106             throw new IllegalArgumentException("Must use non zero sizes for shadow");
107         }
108 
109         final BufferedImage[] imagePointer = new BufferedImage[1];
110 
111         // css settings for svg
112         SvgUtils.prepareCss(cssFileName, cssAdditionalSettings, false);
113         File cssFile = new File(SvgUtils.getSvgTempFile(cssFileName));
114 
115         TranscodingHints transcoderHints = new TranscodingHints();
116 
117         // resize
118         int shadowX = 0;
119         int shadowY = 0;
120         if (useShadow) {
121             // shadow size (16px image: 1px left, 2px bottom)
122             shadowX = 1 * Math.round(1f / 16f * resizeToWidth);
123             shadowY = 2 * Math.round(1f / 16f * resizeToHeight);
124             resizeToWidth = resizeToWidth - shadowX;
125             resizeToHeight = resizeToHeight - shadowY;
126         }
127 
128         if (resizeToWidth > 0) {
129             transcoderHints.put(ImageTranscoder.KEY_WIDTH, (float) resizeToWidth); //your image width
130         }
131         if (resizeToHeight > 0) {
132             transcoderHints.put(ImageTranscoder.KEY_HEIGHT, (float) resizeToHeight); //your image height
133         }
134 
135         transcoderHints.put(ImageTranscoder.KEY_XML_PARSER_VALIDATING, Boolean.FALSE);
136         transcoderHints.put(ImageTranscoder.KEY_DOM_IMPLEMENTATION,
137                 SVGDOMImplementation.getDOMImplementation());
138         transcoderHints.put(ImageTranscoder.KEY_DOCUMENT_ELEMENT_NAMESPACE_URI,
139                 SVGConstants.SVG_NAMESPACE_URI);
140         transcoderHints.put(ImageTranscoder.KEY_DOCUMENT_ELEMENT, "svg");
141         transcoderHints.put(ImageTranscoder.KEY_USER_STYLESHEET_URI, cssFile.toURI().toString());
142 
143         try {
144             TranscoderInput input = new TranscoderInput(svgFile);
145             ImageTranscoder t = new ImageTranscoder() {
146 
147                 @Override
148                 public BufferedImage createImage(int w, int h) {
149                     return new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB);
150                 }
151 
152                 @Override
153                 public void writeImage(BufferedImage image, TranscoderOutput out) {
154                     imagePointer[0] = image;
155                 }
156             };
157             t.setTranscodingHints(transcoderHints);
158             t.transcode(input, null);
159         } catch (Exception e) {
160             throw new IOException("Can't load svg file: " + svgInfo + " , reason: " + e.getMessage());
161         }
162 
163         BufferedImage originImage = imagePointer[0];
164 
165         if (useShadow && (originImage.getWidth() > 0)) {
166             // draw shadow
167             // origin image was reduces in sizes to fit shadow
168             // see https://stackoverflow.com/a/40833715/1276632
169 
170             // a filter which converts all colors except 0 to black
171             ImageProducer prod = new FilteredImageSource(originImage.getSource(), new RGBImageFilter() {
172                 @Override
173                 public int filterRGB(int x, int y, int rgb) {
174                     if (rgb == 0) {
175                         return 0;
176                     } else {
177                         return 0xff000000;
178                     }
179                 }
180             });
181             // create whe black image
182             Image shadow = Toolkit.getDefaultToolkit().createImage(prod);
183             // result
184             BufferedImage result = new BufferedImage(originImage.getWidth() + shadowX, originImage.getHeight() + shadowY, originImage.getType());
185             Graphics2D g = (Graphics2D) result.getGraphics();
186             // draw shadow with offset (left bottom)
187             g.drawImage(shadow, -1 * shadowX, shadowY, null);
188             // draw original image
189             g.drawImage(originImage, 0, 0, null);
190             return result;
191         } else {
192             // return origin image without shadow
193             return originImage;
194         }
195 
196         /*
197         BufferedImage base = GraphicsUtilities.createCompatibleTranslucentImage(w, h);
198         Graphics2D g2 = base.createGraphics();
199         g2.setColor(Color.WHITE);
200         g2.fillRoundRect(0, 0, image.getWidth(), image.getHeight(), 10, 10);
201         g2.dispose();
202 
203         ShadowRenderer renderer = new ShadowRenderer(shadowSize, 0.5f,
204                 Color.GRAY);
205         return renderer.createShadow(base);
206          */
207         //imagePointer[0];
208     }

210     /**
211      * Check if the current system support svg (some linux systems can have compatibility problems due to different java/svg libs)
212      * <p>
213      * Call it on app's start
214      *
215      * @return true on support, also save result for haveSvgSupport
216      */
217     public static boolean checkSvgSupport() {
218         // usa sample icon for svg support testing
219         // direct call, no needs in cache
220         BufferedImage sampleImage = ImageManagerImpl.instance.getCardIcon(FlyingAbilityIcon.instance.getIconType().getResourceName(), 32, CardIconColor.DEFAULT);
221         haveSvgSupport = (sampleImage != null && sampleImage.getWidth() > 0);
222         if (!haveSvgSupport) {
223             logger.warn("WARNING, your system doesn't support svg images, so card icons will be disabled. Please, make a bug report in the github.");
224         }
225         return haveSvgSupport;
226     }

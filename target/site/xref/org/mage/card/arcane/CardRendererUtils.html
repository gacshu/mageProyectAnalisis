35      /**
36       * Convert an abstract image, whose underlying implementation may or may not
37       * be a BufferedImage into a BufferedImage by creating one and coping the
38       * contents if it is not, and simply up-casting if it is.
39       *
40       * @param img The image to convert
41       * @return The converted image
42       */
43      public static BufferedImage toBufferedImage(Image img) {
44          // Null? No conversion to do
45          if (img == null) {
46              return null;
47          }
48  
49          // Already a buffered image?
50          if (img instanceof BufferedImage) {
51              return (BufferedImage) img;
52          }
53  
54          // Create a buffered image with transparency
55          BufferedImage bimage = new BufferedImage(img.getWidth(null), img.getHeight(null), BufferedImage.TYPE_INT_ARGB);
56  
57          // Draw the image on to the buffered image
58          Graphics2D bGr = bimage.createGraphics();
59          bGr.drawImage(img, 0, 0, null);
60          bGr.dispose();
61  
62          // Return the buffered image
63          return bimage;
64      }

66      public static Color abitbrighter(Color c) {
67          int r = c.getRed();
68          int g = c.getGreen();
69          int b = c.getBlue();
70          int alpha = c.getAlpha();
71  
72          int plus_r = (255 - r) / 2;
73          int plus_g = (255 - g) / 2;
74          int plus_b = (255 - b) / 2;
75  
76          return new Color(r + plus_r,
77                  g + plus_g,
78                  b + plus_b,
79                  alpha);
80      }

82      public static Color abitdarker(Color c) {
83          int r = c.getRed();
84          int g = c.getGreen();
85          int b = c.getBlue();
86          int alpha = c.getAlpha();
87  
88          int plus_r = Math.min(255 - r, r) / 2;
89          int plus_g = Math.min(255 - g, g) / 2;
90          int plus_b = Math.min(255 - b, b) / 2;
91  
92          return new Color(r - plus_r,
93                  g - plus_g,
94                  b - plus_b,
95                  alpha);
96      }

99      // Used on various card parts.
100     public static void drawRoundedBox(Graphics2D g, int x, int y, int w, int h, int bevel, Paint border, Paint fill) {
101         g.setColor(new Color(0, 0, 0, 150));
102         g.drawOval(x - 1, y - 1, bevel * 2, h);
103         g.setPaint(border);
104         g.fillOval(x, y, bevel * 2 - 1, h - 2);
105         g.drawOval(x + w - bevel * 2, y, bevel * 2 - 1, h - 1);
106         g.drawOval(x + 1, y + 1, bevel * 2 - 3, h - 3);
107         g.drawOval(x + 1 + w - bevel * 2, y + 1, bevel * 2 - 3, h - 3);
108         g.drawRect(x + bevel, y, w - 2 * bevel, h - 1);
109         g.drawRect(x + 1 + bevel, y + 1, w - 2 * bevel - 2, h - 3);
110         g.setPaint(fill);
111         g.fillOval(x + 2, y + 2, bevel * 2 - 4, h - 4);
112         g.fillOval(x + 2 + w - bevel * 2, y + 2, bevel * 2 - 4, h - 4);
113         g.fillRect(x + bevel, y + 2, w - 2 * bevel, h - 4);
114         g.setPaint(fill);
115         g.setColor(abitbrighter(g.getColor()));
116         g.drawLine(x + 1 + bevel, y + 1, x + 1 + bevel + w - 2 * bevel - 2, y + 1);
117         g.setPaint(fill);
118         g.setColor(abitdarker(g.getColor()));
119         g.drawLine(x + 1 + bevel, y + h - 2, x + 1 + bevel + w - 2 * bevel - 2, y + h - 2);
120     }

151     // Get the width of a mana cost rendered with ManaSymbols.draw
152     public static int getManaCostWidth(String manaCost, int symbolSize) {
153         int width = 0;
154         manaCost = manaCost.replace("\\", "");
155         StringTokenizer tok = new StringTokenizer(manaCost, " ");
156         while (tok.hasMoreTokens()) {
157             tok.nextToken();
158             width += symbolSize;
159         }
160         return width;
161     }

163     // Abbreviate a piece of rules text, making substitutions to decrease its
164     // length. Also abbreviate reminder text.
165     private static final Pattern abbreviationPattern;
166     private static final Map<String, String> abbreviations = new HashMap<>();
167     private static final Pattern killReminderTextPattern;
168 
169     static {
170         // Available abbreviations
171         abbreviations.put("enters the battlefield", "ETB");
172         abbreviations.put("less than", "<");
173         abbreviations.put("greater than", ">");
174 
175         // Compile into regex
176         String patternString = "(";
177         Iterator<String> it = abbreviations.keySet().iterator();
178         while (it.hasNext()) {
179             patternString += it.next();
180             if (it.hasNext()) {
181                 patternString += "|";
182             }
183         }
184         patternString += ")";
185         abbreviationPattern = Pattern.compile(patternString);
186 
187         // Reminder text killing
188         killReminderTextPattern = Pattern.compile("\\([^\\)]*\\)");
189     }
190 
191     public static String abbreviateRule(String rule) {
192         StringBuffer build = new StringBuffer();
193         Matcher match = abbreviationPattern.matcher(rule);
194         while (match.find()) {
195             match.appendReplacement(build, abbreviations.get(match.group(1)));
196         }
197         match.appendTail(build);
198         return build.toString();
199     }

201     public static String killReminderText(String rule) {
202         return killReminderTextPattern.matcher(rule).replaceAll("")
203                 .replaceAll("<i>", "")
204                 .replaceAll("</i>", "");
205     }

215     public static String getCardLifeWithDamage(CardView cardView) {
216         // life with damage
217         String originLife = cardView.getToughness();
218         if (cardView instanceof PermanentView) {
219             int damage = ((PermanentView) cardView).getDamage();
220             int life;
221             try {
222                 life = Integer.parseInt(originLife);
223                 originLife = String.valueOf(Math.max(0, life - damage));
224             } catch (NumberFormatException e) {
225                 //
226             }
227         }
228         return originLife;
229     }

239     public static Color getCardTextColor(MageInt value, boolean drawAsDamaged, Color defaultColor, boolean textLight) {
240         if (drawAsDamaged) {
241             return textLight ? CARD_TEXT_COLOR_BAD_LIGHT : CARD_TEXT_COLOR_BAD_DARK;
242         }
243 
244         // boost colorizing
245         if (value != null) {
246             int currentValue = value.getValue();
247             int baseValue = value.getModifiedBaseValue();
248             if (currentValue < baseValue) {
249                 return textLight ? CARD_TEXT_COLOR_BAD_LIGHT : CARD_TEXT_COLOR_BAD_DARK;
250             } else if (currentValue > baseValue) {
251                 return textLight ? CARD_TEXT_COLOR_GOOD_LIGHT : CARD_TEXT_COLOR_GOOD_DARK;
252             } else {
253                 return defaultColor;
254             }
255         }
256 
257         return defaultColor;
258     }

260     /**
261      * Reduce rect by percent (add empty space from all sides and keep rect position)
262      * Example usage: reduce rect to fit auto-size text
263      *
264      * @param rect
265      * @param reduceFactor
266      * @return
267      */
268     public static Rectangle reduceRect(Rectangle rect, float reduceFactor) {
269         float newWidth = rect.width * reduceFactor;
270         float newHeight = rect.height * reduceFactor;
271         int offsetX = Math.round((rect.width - newWidth) / 2f);
272         int offsetY = Math.round((rect.height - newHeight) / 2f);
273         return new Rectangle(rect.x + offsetX, rect.y + offsetY, Math.round(newWidth), Math.round(newHeight));
274     }

276     /**
277      * Draw a String centered in the middle of a rectangle.
278      *
279      * @param g2d             The graphics instance
280      * @param text            The string to draw
281      * @param rect            The rectangle to center the text in
282      * @param font
283      * @param isAutoScaleFont if the text is too big then it will scale a font to fit it in the rect
284      */
285     public static void drawCenteredText(Graphics2D g2d, String text, Rectangle rect, Font font, boolean isAutoScaleFont) {
286         if (DebugUtil.GUI_RENDER_CENTERED_TEXT_DRAW_DEBUG_LINES) {
287             g2d.drawLine(rect.x, rect.y + rect.height / 2, rect.x + rect.width, rect.y + rect.height / 2);
288             g2d.drawLine(rect.x + rect.width / 2, rect.y, rect.x + rect.width / 2, rect.y + rect.height);
289         }
290 
291         // https://stackoverflow.com/a/23730104/1276632
292         Font affectedFont = font;
293         if (isAutoScaleFont) {
294             affectedFont = scaleFont(g2d, text, rect, font);
295         }
296 
297         g2d.setFont(affectedFont);
298         FontRenderContext frc = g2d.getFontRenderContext();
299         GlyphVector gv = affectedFont.createGlyphVector(frc, text);
300         Rectangle2D box = gv.getVisualBounds();
301         float offsetX = (float) (((rect.getWidth() - box.getWidth()) / 2d) + (-box.getX()));
302         float offsetY = (float) (((rect.getHeight() - box.getHeight()) / 2d) + (-box.getY()));
303 
304         g2d.drawString(text, rect.x + offsetX, rect.y + offsetY);
305     }

307     /**
308      * Auto scale font to fit current text inside the rect (e.g. decrease font size for too big text)
309      *
310      * @param g2d  graphics context
311      * @param text text to draw
312      * @param font base font
313      * @param rect the bounds for fitting the string
314      * @return a scaled font
315      */
316     private static Font scaleFont(Graphics2D g2d, String text, Rectangle rect, Font font) {
317         // https://stackoverflow.com/a/876266/1276632
318         FontRenderContext frc = g2d.getFontRenderContext();
319 
320         double needWidth = rect.getWidth();
321         double needHeight = rect.getHeight();
322 
323         float fontMinSize = 1f;
324         float fontMaxSize = 1000f;
325         Font scaledFont = font;
326         float scaledFontSize = scaledFont.getSize();
327 
328         while (fontMaxSize - fontMinSize > 1f) {
329             scaledFont = scaledFont.deriveFont(scaledFontSize);
330 
331             TextLayout layout = new TextLayout(text, scaledFont, frc);
332             float currentWidth = layout.getVisibleAdvance();
333             LineMetrics metrics = scaledFont.getLineMetrics(text, frc);
334             float currentHeight = metrics.getHeight();
335 
336             if ((currentWidth > needWidth) || (currentHeight > needHeight)) {
337                 fontMaxSize = scaledFontSize;
338             } else {
339                 fontMinSize = scaledFontSize;
340             }
341             scaledFontSize = (fontMinSize + fontMaxSize) / 2f;
342         }
343 
344         return scaledFont.deriveFont((float) Math.floor(scaledFontSize));
345     }

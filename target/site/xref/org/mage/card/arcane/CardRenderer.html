21  /**
22   * @author stravant@gmail.com
23   * <p>
24   * Common base class for card renderers for each card frame / card type.
25   * <p>
26   * Follows the template method pattern to implement a new renderer, implement
27   * the following methods (they are called in the following order):
28   * <p>
29   * * drawBorder() Draws the outermost border of the card, white border or black
30   * border
31   * <p>
32   * * drawBackground() Draws the background texture / color of the card
33   * <p>
34   * * drawArt() Draws the card's art
35   * <p>
36   * * drawFrame() Draws the card frame (over the art and background)
37   * <p>
38   * * drawOverlays() Draws summoning sickness and possible other overlays
39   * <p>
40   * * drawCounters() Draws counters on the card, such as +1/+1 and -1/-1
41   * counters
42   * <p>
43   * Predefined methods that the implementations can use:
44   * <p>
45   * * drawRules(font, bounding box)
46   * <p>
47   * * drawNameLine(font, bounding box)
48   * <p>
49   * * drawTypeLine(font, bounding box)
50   */
51  public abstract class CardRenderer {

113     // The Construtor
114     // The constructor should prepare all of the things that it can
115     // without knowing the dimensions that the card will be rendered at.
116     // Then, the CardRenderer can be called on multiple times to render the
117     // card at various sizes (for instance, during animation)
118     public CardRenderer(CardView card) {
119         // Set base parameters
120         this.cardView = card;
121 
122         if (card.getArtRect() == ArtRect.SPLIT_FUSED) {
123             parseRules(card.getLeftSplitRules(), textboxKeywords, textboxRules);
124             parseRules(card.getRightSplitRules(), textboxKeywords, textboxRules);
125         } else {
126             parseRules(card.getRules(), textboxKeywords, textboxRules);
127         }
128     }

130     protected void parseRules(List<String> stringRules, List<TextboxRule> keywords, List<TextboxRule> rules) {
131         // Translate the textbox text and remove card hints
132         for (String rule : stringRules) {
133             // remove all card hints
134             if (rule.equals(HintUtils.HINT_START_MARK)) {
135                 break;
136             }
137 
138             // workaround to use real split card names
139             String realCardName = cardView.getDisplayName();
140             if (cardView.isSplitCard()) {
141                 for (String partRule : cardView.getLeftSplitRules()) {
142                     if (partRule.equals(rule)) {
143                         realCardName = cardView.getLeftSplitName();
144                         break;
145                     }
146                 }
147                 for (String partRule : cardView.getRightSplitRules()) {
148                     if (partRule.equals(rule)) {
149                         realCardName = cardView.getRightSplitName();
150                         break;
151                     }
152                 }
153             }
154 
155             // Kill reminder text
156             if (PreferencesDialog.getCachedValue(PreferencesDialog.KEY_CARD_RENDERING_REMINDER_TEXT, "false").equals("false")) {
157                 rule = CardRendererUtils.killReminderText(rule).trim();
158             }
159 
160             if (!rule.isEmpty()) {
161                 TextboxRule tbRule = TextboxRuleParser.parse(cardView, rule, realCardName);
162                 if (tbRule.type == TextboxRuleType.SIMPLE_KEYWORD) {
163                     keywords.add(tbRule);
164                 } else if (tbRule.text.isEmpty()) {
165                     // Nothing to do, rule is empty
166                 } else {
167                     rules.add(tbRule);
168                 }
169             }
170         }
171     }

173     private static int getBorderWidth(int cardWidth) {
174         return (int) Math.max(
175                 BORDER_WIDTH_MIN,
176                 BORDER_WIDTH_FRAC * cardWidth);
177     }

179     // Layout operation
180     // Calculate common layout metrics that will be used by several
181     // of the operations in the template method.
182     protected void layout(int cardWidth, int cardHeight) {
183         // Store the dimensions for the template methods to use
184         this.cardWidth = cardWidth;
185         this.cardHeight = cardHeight;
186 
187         // Corner radius and border width
188         cornerRadius = (int) Math.max(
189                 CORNER_RADIUS_MIN,
190                 CORNER_RADIUS_FRAC * cardWidth);
191 
192         borderWidth = getBorderWidth(cardWidth);
193     }

195     /**
196      * How far does a card have to be spaced down from a rendered card to show
197      * it's entire name line? This function is a bit of a hack, as different
198      * card faces need slightly different spacing, but we need it in a static
199      * context so that spacing is consistent in GY / deck views etc.
200      *
201      * @param cardWidth
202      * @return
203      */
204     public static int getCardTopHeight(int cardWidth) {
205         // Constants copied over from ModernCardRenderer and tweaked
206         float BOX_HEIGHT_FRAC = 0.065f; // x cardHeight
207         int BOX_HEIGHT_MIN = 16;
208         int boxHeight = (int) Math.max(
209                 BOX_HEIGHT_MIN,
210                 BOX_HEIGHT_FRAC * cardWidth * 1.4f);
211         int borderWidth = getBorderWidth(cardWidth);
212         return 2 * borderWidth + boxHeight;
213     }

215     // The Draw Method
216     // The draw method takes the information caculated by the constructor
217     // and uses it to draw to a concrete size of card and graphics.
218     public void draw(Graphics2D g, CardPanelAttributes attribs, BufferedImage image) {
219 
220         // Pre template method layout, to calculate shared layout info
221         layout(attribs.cardWidth, attribs.cardHeight);
222         isSelected = attribs.isSelected;
223         isChoosable = attribs.isChoosable;
224 
225         // Call the template methods
226         drawBorder(g);
227         drawBackground(g);
228         lessOpaqueRulesTextBox = false;
229         drawArt(g);
230         drawFrame(g, attribs, image, lessOpaqueRulesTextBox);
231         if (!cardView.isAbility()) {
232             drawOverlays(g);
233             drawCounters(g);
234         }
235     }

256     // Draw summoning sickness overlay, and possibly other overlays
257     protected void drawOverlays(Graphics2D g) {
258         if (cardView.isCreature() && cardView instanceof PermanentView) {
259             if (((PermanentView) cardView).hasSummoningSickness()) {
260                 int x1 = (int) (0.2 * cardWidth);
261                 int x2 = (int) (0.8 * cardWidth);
262                 int y1 = (int) (0.2 * cardHeight);
263                 int y2 = (int) (0.8 * cardHeight);
264                 int[] xPoints = {
265                         x1, x2, x1, x2
266                 };
267                 int[] yPoints = {
268                         y1, y1, y2, y2
269                 };
270                 g.setColor(new Color(255, 255, 255, 200));
271                 g.setStroke(new BasicStroke(7));
272                 g.drawPolygon(xPoints, yPoints, 4);
273                 g.setColor(new Color(0, 0, 0, 200));
274                 g.setStroke(new BasicStroke(5));
275                 g.drawPolygon(xPoints, yPoints, 4);
276                 g.setStroke(new BasicStroke(1));
277                 int[] xPoints2 = {
278                         x1, x2, cardWidth / 2
279                 };
280                 int[] yPoints2 = {
281                         y1, y1, cardHeight / 2
282                 };
283                 g.setColor(new Color(0, 0, 0, 100));
284                 g.fillPolygon(xPoints2, yPoints2, 3);
285             }
286         }

289     protected void drawArtIntoRect(Graphics2D g, int x, int y, int w, int h, Rectangle2D artRect, boolean shouldPreserveAspect) {
290         // Perform a process to make sure that the art is scaled uniformly to fill the frame, cutting
291         // off the minimum amount necessary to make it completely fill the frame without "squashing" it.
292         double fullCardImgWidth = artImage.getWidth();
293         double fullCardImgHeight = artImage.getHeight();
294         double artWidth = artRect.getWidth() * fullCardImgWidth;
295         double artHeight = artRect.getHeight() * fullCardImgHeight;
296         double targetWidth = w;
297         double targetHeight = h;
298         double targetAspect = targetWidth / targetHeight;
299         if (!shouldPreserveAspect) {
300             // No adjustment to art
301         } else if (targetAspect * artHeight < artWidth) {
302             // Trim off some width
303             artWidth = targetAspect * artHeight;
304         } else {
305             // Trim off some height
306             artHeight = artWidth / targetAspect;
307         }
308         try {
309             BufferedImage subImg
310                     = artImage.getSubimage(
311                     (int) (artRect.getX() * fullCardImgWidth), (int) (artRect.getY() * fullCardImgHeight),
312                     (int) artWidth, (int) artHeight);
313             g.drawImage(subImg,
314                     x, y,
315                     (int) targetWidth, (int) targetHeight,
316                     null);
317         } catch (RasterFormatException e) {
318             // At very small card sizes we may encounter a problem with rounding error making the rect not fit
319         }
320     }

369     // Draw +1/+1 and other counters
370     protected void drawCounters(Graphics2D g) {
371         int xPos = (int) (0.65 * cardWidth);
372         int yPos = (int) (0.15 * cardHeight);
373         if (cardView.getCounters() != null) {
374             for (CounterView v : cardView.getCounters()) {
375                 // Don't render loyalty, we do that in the bottom corner
376                 if (!v.getName().equals("loyalty")) {
377                     Polygon p;
378                     switch (v.getName()) {
379                         case "+1/+1":
380                             p = PLUS_COUNTER_POLY;
381                             break;
382                         case "-1/-1":
383                             p = MINUS_COUNTER_POLY;
384                             break;
385                         case "time":
386                             p = TIME_COUNTER_POLY;
387                             break;
388                         default:
389                             p = OTHER_COUNTER_POLY;
390                             break;
391                     }
392                     double scale = (0.1 * 0.25 * cardWidth);
393                     Graphics2D g2 = (Graphics2D) g.create();
394                     g2.translate(xPos, yPos);
395                     g2.scale(scale, scale);
396                     g2.setColor(Color.white);
397                     g2.fillPolygon(p);
398                     g2.setColor(Color.black);
399                     g2.drawPolygon(p);
400                     g2.setFont(new Font("Arial", Font.BOLD, 7));
401                     String cstr = String.valueOf(v.getCount());
402                     int strW = g2.getFontMetrics().stringWidth(cstr);
403                     g2.drawString(cstr, 5 - strW / 2, 8);
404                     g2.dispose();
405                     yPos += ((int) (0.30 * cardWidth));
406                 }
407             }
408         }
409     }

457     private Color getRarityColor() {
458         if (cardView.getRarity() != null) {
459             switch (cardView.getRarity()) {
460                 case RARE:
461                     return new Color(255, 191, 0);
462                 case UNCOMMON:
463                     return new Color(192, 192, 192);
464                 case MYTHIC:
465                     return new Color(213, 51, 11);
466                 case SPECIAL:
467                     return new Color(204, 0, 255);
468                 case BONUS:
469                     return new Color(129, 228, 228);
470                 case COMMON:
471                 default:
472                     return Color.black;
473             }
474         } else {
475             // tokens
476             return Color.black;
477         }
478     }

480     // Get a string representing the type line
481     protected String getCardTypeLine() {
482         if (cardView.isAbility()) {
483             if (cardView.getAbilityType() == AbilityType.TRIGGERED) {
484                 return "Triggered Ability";
485             } else if (cardView.getAbilityType() == AbilityType.ACTIVATED) {
486                 return "Activated Ability";
487             } else if (cardView.getAbilityType() == null) {
488                 // TODO: Triggered abilities waiting to be put onto the stack have abilityType = null. Figure out why
489                 return "Triggered Ability";
490             } else {
491                 return "??? Ability";
492             }
493         } else {
494             StringBuilder sbType = new StringBuilder();
495             String spType = getCardSuperTypeLine();
496             String subType = getCardSubTypeLine();
497             if (spType.equalsIgnoreCase("")) {
498                 sbType.append(subType);
499             } else {
500                 sbType.append(spType);
501                 if (!subType.equalsIgnoreCase("")) {
502                     sbType.append("- ");
503                     sbType.append(subType);
504                 }
505             }
506 
507             return sbType.toString();
508         }
509     }

511     protected String getCardSuperTypeLine() {
512         StringBuilder spType = new StringBuilder();
513         for (SuperType superType : cardView.getSuperTypes()) {
514             spType.append(superType).append(' ');
515         }
516         for (CardType cardType : cardView.getCardTypes()) {
517             spType.append(cardType.toString()).append(' ');
518         }
519         return spType.toString();
520     }

522     protected String getCardSubTypeLine() {
523         StringBuilder subType = new StringBuilder();
524 
525         if (!cardView.getSubTypes().isEmpty()) {
526             for (SubType sType : cardView.getSubTypes()) {
527                 subType.append(sType).append(' ');
528             }
529         }
530         return subType.toString();
531     }

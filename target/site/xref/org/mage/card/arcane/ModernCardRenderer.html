306     @Override
307     protected void drawBackground(Graphics2D g) {
308         // Draw background, in 3 parts
309 
310         if (cardView.isFaceDown()) {
311             // Just draw a brown rectangle
312             drawCardBack(g);
313         } else {
314             if (cardView.getFrameStyle() == FrameStyle.UST_FULL_ART_BASIC) {
315                 return;
316             }
317 
318             boolean isExped = false;
319             if (cardView.getExpansionSetCode().equals("EXP")) {
320                 isExped = true;
321             }
322             BufferedImage bg = getBackgroundImage(cardView.getColor(), cardView.getCardTypes(), cardView.getSubTypes(), isExped);
323             if (bg == null) {
324                 return;
325             }
326             int bgw = bg.getWidth();
327             int bgh = bg.getHeight();
328 
329             // Draw main part (most of card)
330             RoundRectangle2D rr = new RoundRectangle2D.Double(borderWidth, borderWidth,
331                     cardWidth - borderWidth * 2, cardHeight - borderWidth * 4 - cornerRadius * 2,
332                     cornerRadius - 1, cornerRadius - 1);
333             Area a = new Area(rr);
334 
335             RoundRectangle2D rr2 = new RoundRectangle2D.Double(borderWidth, cardHeight - borderWidth * 4 - cornerRadius * 4,
336                     cardWidth - borderWidth * 2, cornerRadius * 4,
337                     cornerRadius * 2, cornerRadius * 2);
338             a.add(new Area(rr2));
339 
340             // Draw the M15 rounded "swoosh" at the bottom
341             Rectangle r = new Rectangle(borderWidth + contentInset, cardHeight - borderWidth * 5, cardWidth - borderWidth * 2 - contentInset * 2, borderWidth * 2);
342             a.add(new Area(r));
343             g.setClip(a);
344             g.drawImage(bg, 0, 0, cardWidth, cardHeight, 0, 0, bgw, bgh, BOX_BLUE, null);
345             g.setClip(null);
346         }

363             rect = new Rectangle2D.Float(.079f, .11f, .84f, .63f);
364         } else {
365             rect = ArtRect.NORMAL.rect;
366         }
367         return rect;

409     @Override
410     protected void drawArt(Graphics2D g) {
411         if ((artImage != null || faceArtImage != null) && !cardView.isFaceDown()) {
412 
413             boolean useFaceArt = false;
414             if (faceArtImage != null && !isZendikarFullArtLand()) {
415                 useFaceArt = true;
416             }
417 
418             // Invention rendering, art fills the entire frame
419             if (useInventionFrame()) {
420                 useFaceArt = false;
421                 drawArtIntoRect(g,
422                         borderWidth, borderWidth,
423                         cardWidth - 2 * borderWidth, cardHeight - 2 * borderWidth,
424                         getArtRect(), false);
425             }
426 
427             boolean shouldPreserveAspect = true;
428             Rectangle2D sourceRect = getArtRect();
429 
430             if (cardView.getMageObjectType() == MageObjectType.SPELL) {
431                 useFaceArt = false;
432                 ArtRect rect = cardView.getArtRect();
433                 if (rect == ArtRect.SPLIT_FUSED) {
434                     // Special handling for fused, draw the art from both halves stacked on top of one and other
435                     // each filling half of the art rect
436                     drawArtIntoRect(g,
437                             totalContentInset + 1, totalContentInset + boxHeight,
438                             contentWidth - 2, (typeLineY - totalContentInset - boxHeight) / 2,
439                             ArtRect.SPLIT_LEFT.rect, useInventionFrame());
440                     drawArtIntoRect(g,
441                             totalContentInset + 1, totalContentInset + boxHeight + (typeLineY - totalContentInset - boxHeight) / 2,
442                             contentWidth - 2, (typeLineY - totalContentInset - boxHeight) / 2,
443                             ArtRect.SPLIT_RIGHT.rect, useInventionFrame());
444                     return;
445                 } else if (rect != ArtRect.NORMAL) {
446                     sourceRect = rect.rect;
447                     shouldPreserveAspect = false;
448                 }
449             }
450 
451             // Normal drawing of art from a source part of the card frame into the rect
452             if (useFaceArt) {
453                 int alternate_height = cardHeight - boxHeight * 2 - totalContentInset;
454                 drawFaceArtIntoRect(g,
455                         totalContentInset + 1, totalContentInset + boxHeight,
456                         contentWidth - 2, typeLineY - totalContentInset - boxHeight,
457                         alternate_height,
458                         sourceRect, shouldPreserveAspect);
459             } else if (!isZendikarFullArtLand()) {
460                 drawArtIntoRect(g,
461                         totalContentInset + 1, totalContentInset + boxHeight,
462                         contentWidth - 2, typeLineY - totalContentInset - boxHeight,
463                         sourceRect, shouldPreserveAspect);
464             }
465         }

485         boolean isZenUst = isZendikarFullArtLand() || isUnstableFullArtLand();
486         boolean isOriginalDual = isOriginalDualLand();
487 
488         // Draw the main card content border
489         g.setPaint(borderPaint);

492             g.drawImage(FRAME_INVENTION, 3, 3, cardWidth - 6, cardHeight - 6, null);

498                     totalContentInset, totalContentInset,

500         }
501

503         if (useInventionFrame()) {

523                     g.fillRect(totalContentInset + 1, typeLineY + boxHeight + 1, contentWidth - 2, total_height_of_box);
524                 }
525                 if (frameColors.getColorCount() >= 3) {
526                     g.fillRect(totalContentInset + 1, typeLineY + boxHeight + 1, contentWidth - 2, total_height_of_box);
527                 }
528                 if (frameColors.getColorCount() == 2) {
529                     if (isOriginalDual) {
530                         g.setPaint(getSpiralLandTextboxColor(twoColors.get(0), twoColors.get(1), true));
531 
532                         // Horizontal bars
533                         g.fillRect(totalContentInset + 1                     , typeLineY + boxHeight + 1                                           , contentWidth - 2                      , height_of_spiral);
534                         g.fillRect(totalContentInset + 1 + 2*height_of_spiral, typeLineY + boxHeight + 1 + 2*height_of_spiral                      , contentWidth - 2 - 4*height_of_spiral , height_of_spiral);
535                         g.fillRect(totalContentInset + 1 + 4*height_of_spiral, typeLineY + boxHeight + 1 + 4*height_of_spiral                      , contentWidth - 2 - 8*height_of_spiral , height_of_spiral);
536                         g.fillRect(totalContentInset + 1 + 6*height_of_spiral, typeLineY + boxHeight + 1 + 6*height_of_spiral                      , contentWidth - 2 - 12*height_of_spiral, height_of_spiral);
537

819     public void drawUSTCurves(Graphics2D g2, BufferedImage image, int x, int y, int x2, int y2,
820                               int topxdelta, int endydelta,
821                               Color boxColor, Paint paint) {
822         BufferedImage artToUse = artImage;
823 
824         int srcW = x2;
825         int srcH = y2;
826         if (artToUse != null) {
827             srcW = artToUse.getWidth();
828             srcH = artToUse.getHeight();
829         }
830 
831         g2.setPaint(paint);
832 
833         // Dimensions:  534 height, 384 width, 34 offset at top, 41 offset at bottom.  Curve at bottom right is from an ellipse: 245 high, 196 wide, with center offset from
834         // right side by 36  (so top left is at: (width - 159, height - 41 -196)  center at: 41+127 = width - 36, height - 168)
835         int scan_width = 384;
836         int scan_height = 534;
837         int scan_ew = 196;
838         int scan_eh = 254;
839         int offset_ew = 159;
840         int offset_eh = 41;
841         int middle_ew = 52;
842         int middle_eh = 26;
843 
844         // Bottom left side arc
845         int ex = (offset_ew - scan_ew) * x2 / scan_width;
846         int ey = y2 - (offset_eh + scan_eh) * y2 / scan_height;
847         int bot_ey = y2 - offset_eh * y2 / scan_height;
848         int ew = scan_ew * x2 / scan_width;
849         int eh = scan_eh * y2 / scan_height;
850         int end_curve_ex = ex + ew / 2;
851 
852         Arc2D arc = new Arc2D.Double(ex, ey, ew, eh, 180, 90, Arc2D.OPEN);
853 
854         // Bottom right side arc
855         ex = x2 - offset_ew * x2 / scan_width;
856         ey = y2 - (offset_eh + scan_eh) * y2 / scan_height;
857         bot_ey = y2 - offset_eh * y2 / scan_height;
858         Arc2D arc2 = new Arc2D.Double(ex, ey, ew, eh, 270, 90, Arc2D.OPEN);
859 
860         // Middle bump.. 52x26
861         int mid_ex = x2 / 2 - middle_ew * x2 / (scan_width * 2);
862         int mid_ey = bot_ey - middle_eh * y2 / (scan_height * 2);
863         int end_mid_ex = x2 / 2 + middle_ew * x2 / (scan_width * 2);
864 
865         Arc2D arc3 = new Arc2D.Double(mid_ex, mid_ey, middle_ew * x2 / scan_width, middle_eh * y2 / scan_height, 180, -180, Arc2D.OPEN);
866 
867         Path2D.Double curve = new Path2D.Double();
868         curve.moveTo(0, 0);
869         curve.lineTo(0, bot_ey);
870         curve.append(arc, true);
871         curve.lineTo(mid_ex, bot_ey);
872         curve.append(arc3, true);
873         curve.lineTo(x2 - ew / 2, bot_ey);
874         curve.append(arc2, true);
875         curve.lineTo(x2, 0);
876         curve.lineTo(0, 0);
877 
878         g2.setClip(curve);
879         if (artToUse != null) {
880             artToUse = artImage.getSubimage(0, 0, srcW, srcH);
881             g2.drawImage(artToUse, 0, 0, x2, y2, null);
882         }
883 
884         g2.setClip(null);
885         g2.setStroke(new BasicStroke(3));
886         g2.draw(arc);
887         g2.draw(new Rectangle(end_curve_ex, bot_ey, mid_ex - end_curve_ex, 0));
888         g2.draw(arc3);
889         g2.draw(new Rectangle(end_mid_ex, bot_ey, mid_ex - end_curve_ex, 0));
890         g2.draw(arc2);
891         g2.setStroke(new BasicStroke(1));
892         g2.setColor(boxColor);
893     }

1204     protected RuleLayout layoutRules(Graphics2D g, List<TextboxRule> rules, int w, int h, int fontSize) {

1206         Font font = new Font("Arial", Font.PLAIN, fontSize);

1225         layout.attributedRules = attributedRules;

1233     protected void drawRulesText(Graphics2D g, List<TextboxRule> keywords, List<TextboxRule> rules, int x, int y, int w, int h, boolean forceRules) {
1234         // Gather all rules to render
1235         List<TextboxRule> allRules = new ArrayList<>(rules);
1236 
1237         // Add the keyword rule if there are any keywords
1238         if (!keywords.isEmpty()) {
1239             String keywordRulesString = getKeywordRulesString(keywords);
1240             TextboxRule keywordsRule = new TextboxRule(keywordRulesString, new ArrayList<>());
1241             allRules.add(0, keywordsRule);
1242         }
1243 
1244         if (isUnstableFullArtLand()) {
1245             return;
1246         }
1247 
1248         // Basic mana draw mana symbol in textbox (for basic lands)
1249         if (!forceRules && (allRules.size() == 1 && (allRules.get(0) instanceof TextboxBasicManaRule) && cardView.isLand() || isZendikarFullArtLand())) {
1250             if (!isZendikarFullArtLand()) {
1251                 drawBasicManaTextbox(g, x, y, w, h, ((TextboxBasicManaRule) allRules.get(0)).getBasicManaSymbol());
1252                 return;
1253             } else // Big circle in the middle for Zendikar lands
1254                 if (allRules.size() == 1) {
1255                     // Size of mana symbol = 9/4 * h, 3/4h above line
1256                     if (allRules.get(0) instanceof TextboxBasicManaRule) {
1257                         drawBasicManaSymbol(g, x + w / 2 - 9 * h / 8 + 1, y - 3 * h / 4, 9 * h / 4, 9 * h / 4, ((TextboxBasicManaRule) allRules.get(0)).getBasicManaSymbol());
1258                     } else {
1259                         drawBasicManaSymbol(g, x + w / 2 - h - h / 8, y - 3 * h / 4, 9 * h / 4, 9 * h / 4, cardView.getFrameColor().toString());
1260                     }
1261                     return;
1262                 } else {
1263                     if (allRules.size() > 1) {
1264                         drawBasicManaSymbol(g, x + w / 2 - h - h / 8, y - 3 * h / 4, 9 * h / 4, 9 * h / 4, cardView.getFrameColor().toString());
1265                     }
1266                     return;
1267                 }
1268         }
1269 
1270         // Go through possible font sizes in descending order to find the best fit
1271         RuleLayout bestLayout = null;
1272         for (int fontSize : RULES_TEXT_FONT_SIZES) {
1273             bestLayout = layoutRules(g, allRules, w, h, fontSize);
1274 
1275             // Stop, we found a good fit
1276             if (bestLayout.fits) {
1277                 break;
1278             }
1279         }
1280 
1281         // Nothing to draw
1282         if (bestLayout == null) {
1283             return;
1284         }
1285 
1286         // Do we have room for additional padding between the parts of text?
1287         // If so, calculate the padding based on how much space was left over
1288         int padding;
1289         if (bestLayout.fits) {
1290             padding = (int) (((float) bestLayout.remainingHeight) / (1 + allRules.size()));
1291         } else {
1292             // When the text doesn't fit to begin with there's no room for padding
1293             padding = 0;
1294         }
1295 
1296         // Do the actual draw
1297         loyaltyAbilityColorToggle = false;
1298         g.setColor(Color.black);
1299         int curY = y + padding;
1300         for (int i = 0; i < bestLayout.attributedRules.size(); ++i) {
1301             AttributedString attributedRule = bestLayout.attributedRules.get(i);
1302             TextboxRule rule = allRules.get(i);
1303             int adv = drawSingleRule(g, attributedRule, rule, x, curY, w, h, true);
1304             curY += adv + padding;
1305             h -= adv;
1306             if (h < 0) {
1307                 break;
1308             }
1309         }
1310     }

1345     // Draw a single rule and returns the amount vertically advanced by, but
1346     // only if doDraw is true. If doDraw is false, just returns the vertical
1347     // advance if the rule were to be drawn.
1348     private int drawSingleRule(Graphics2D g, AttributedString text, TextboxRule rule, int x, int y, int w, int h, boolean doDraw) {
1349         // Inset, in case we are a leveler or loyalty ability
1350         int inset = 0;
1351         if (rule != null && rule.type == TextboxRuleType.LOYALTY) {
1352             inset = cardWidth / 12;
1353         }
1354         int availWidth = w - inset;
1355         if (availWidth < 0) {
1356             return 0;
1357         }
1358 
1359         FontRenderContext frc = g.getFontRenderContext();
1360         AttributedCharacterIterator textIter = text.getIterator();
1361         LineBreakMeasurer measure = new LineBreakMeasurer(textIter, frc);
1362         float yPos = y;
1363         float remain = h;
1364         AttributedCharacterIterator newLineCheck = text.getIterator();
1365         while (measure.getPosition() < textIter.getEndIndex()) {
1366             // Advance iterator to next line break
1367             newLineCheck.setIndex(measure.getPosition());
1368             char ch;
1369             while ((ch = newLineCheck.next()) != CharacterIterator.DONE) {
1370                 if (ch == '\n') {
1371                     break;
1372                 }
1373             }
1374 
1375             // Get the text layout
1376             TextLayout layout = measure.nextLayout(availWidth, newLineCheck.getIndex(), false);
1377             float ascent = layout.getAscent();
1378             yPos += ascent;
1379             remain -= ascent;
1380             if (remain < 0) {
1381                 break;
1382             }
1383             if (doDraw) {
1384                 g.setColor(Color.black);
1385                 layout.draw(g, x + inset, yPos);
1386             }
1387             yPos += layout.getDescent() + layout.getLeading() - 2;
1388         }
1389 
1390         // Advance
1391         int advance = ((int) Math.ceil(yPos)) - y;
1392 
1393         // Is it a loyalty ability?
1394         if (rule != null && rule.type == TextboxRuleType.LOYALTY) {
1395             TextboxLoyaltyRule loyaltyRule = (TextboxLoyaltyRule) rule;
1396             Polygon symbol;
1397             int symbolWidth = (x + inset) - borderWidth - 4;
1398             int symbolHeight = (int) (0.7f * symbolWidth);
1399             if (symbolHeight > advance) {
1400                 advance = symbolHeight;
1401             }
1402             int symbolX = x - borderWidth;
1403             int symbolY = y + (advance - symbolHeight) / 2;
1404             if (doDraw) {
1405                 if (loyaltyRule.loyaltyChange < 0 || loyaltyRule.loyaltyChange == TextboxLoyaltyRule.MINUS_X) {
1406                     symbol = new Polygon(
1407                             new int[]{
1408                                     symbolX,
1409                                     symbolX + symbolWidth,
1410                                     symbolX + symbolWidth,
1411                                     symbolX + symbolWidth / 2,
1412                                     symbolX,},
1413                             new int[]{
1414                                     symbolY,
1415                                     symbolY,
1416                                     symbolY + symbolHeight - 3,
1417                                     symbolY + symbolHeight + 3,
1418                                     symbolY + symbolHeight - 3,},
1419                             5);
1420                 } else if (loyaltyRule.loyaltyChange > 0) {
1421                     symbol = new Polygon(
1422                             new int[]{
1423                                     symbolX,
1424                                     symbolX + symbolWidth / 2,
1425                                     symbolX + symbolWidth,
1426                                     symbolX + symbolWidth,
1427                                     symbolX,},
1428                             new int[]{
1429                                     symbolY + 3,
1430                                     symbolY - 3,
1431                                     symbolY + 3,
1432                                     symbolY + symbolHeight,
1433                                     symbolY + symbolHeight,},
1434                             5);
1435                 } else {
1436                     symbol = new Polygon(
1437                             new int[]{
1438                                     symbolX,
1439                                     symbolX + symbolWidth,
1440                                     symbolX + symbolWidth,
1441                                     symbolX,},
1442                             new int[]{
1443                                     symbolY,
1444                                     symbolY,
1445                                     symbolY + symbolHeight,
1446                                     symbolY + symbolHeight,},
1447                             4);
1448                 }
1449                 g.setColor(new Color(0, 0, 0, 128));
1450                 g.fillRect(x + 2, y + advance + 1, w - 2, 1);
1451                 g.setColor(Color.black);
1452                 g.fillPolygon(symbol);
1453                 g.setColor(new Color(200, 200, 200));
1454                 g.setStroke(new BasicStroke(2));
1455                 g.drawPolygon(symbol);
1456                 g.setStroke(new BasicStroke(1));
1457                 g.setColor(Color.white);
1458                 g.setFont(boxTextFont);
1459                 String loyaltyString = loyaltyRule.getChangeString();
1460                 int textWidth = g.getFontMetrics().stringWidth(loyaltyString);
1461                 g.drawString(loyaltyString,
1462                         symbolX + (symbolWidth - textWidth) / 2,
1463                         symbolY + symbolHeight - (symbolHeight - boxTextHeight) / 2);
1464 
1465                 advance += 3;
1466                 loyaltyAbilityColorToggle = !loyaltyAbilityColorToggle;
1467             }
1468         }
1469 
1470         return advance;
1471     }

1715         boolean red = color.isRed() || secondColor.isRed();

1717 
1718         if (white && green) {
1719             return firstOne ? LAND_SPIRAL_TEXTBOX_WHITE : LAND_SPIRAL_TEXTBOX_GREEN;

1722             return firstOne ? LAND_SPIRAL_TEXTBOX_WHITE : LAND_SPIRAL_TEXTBOX_BLUE;

1724         if (black && white) {

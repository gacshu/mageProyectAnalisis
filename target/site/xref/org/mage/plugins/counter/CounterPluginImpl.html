1   package org.mage.plugins.counter;
2   
3   import mage.interfaces.PluginException;
4   import mage.interfaces.plugin.CounterPlugin;
5   import net.xeoh.plugins.base.annotations.PluginImplementation;
6   import net.xeoh.plugins.base.annotations.events.Init;
7   import net.xeoh.plugins.base.annotations.events.PluginLoaded;
8   import net.xeoh.plugins.base.annotations.meta.Author;
9   import org.apache.log4j.Logger;
10  
11  import java.io.*;
12  
13  /**
14   * Implementation of {@link CounterPlugin}.<br/>
15   * Stores data in data folder. 
16   * 
17   * @version 0.1 14.11.2010 Initial Version
18   * @author nantuko
19   */
20  @PluginImplementation
21  @Author(name = "nantuko")
22  public class CounterPluginImpl implements CounterPlugin {
23  
24      private static final String PLUGIN_DATA_FOLDER_PATH = "plugins" + File.separator + "plugin.data" + File.separator + "counters";
25  
26      private static final String DATA_STORAGE_FILE = "counters";
27  
28      private static final Logger log = Logger.getLogger(CounterPluginImpl.class);
29  
30      private boolean isLoaded = false;
31  
32      @Init
33      public void init() {
34          File dataFolder = new File(PLUGIN_DATA_FOLDER_PATH);
35          if (!dataFolder.exists()) {
36              dataFolder.mkdirs();
37              if (!dataFolder.exists()) {
38                  throw new RuntimeException("CounterPluginImpl: Couldn't create folders: " + PLUGIN_DATA_FOLDER_PATH);
39              }
40          }
41          File data = new File(PLUGIN_DATA_FOLDER_PATH + File.separator + DATA_STORAGE_FILE);
42          if (!data.exists()) {
43              try {
44                  data.getParentFile().mkdirs();
45                  data.createNewFile();
46              } catch (IOException e) {
47                  log.error(e.getMessage(), e);
48                  throw new RuntimeException("Couldn't create data file for counter plugin: " + e.getMessage());
49              }
50          }
51  
52          this.isLoaded = true;
53      }
54  
55      @PluginLoaded
56      public void newPlugin(CounterPlugin plugin) {
57          log.info(plugin.toString() + " has been loaded.");
58      }
59  
60      @Override
61      public String toString() {
62          return "[Game counter plugin, version 0.1]";
63      }
64  
65      @Override
66      public void addGamePlayed() throws PluginException {
67          if (!isLoaded) return;
68          File data = new File(PLUGIN_DATA_FOLDER_PATH + File.separator + DATA_STORAGE_FILE);
69          if (data.exists()) {
70              int prev = 0;
71  
72              try(ObjectInputStream ois = new ObjectInputStream(new FileInputStream(data))) {
73                  Object o = ois.readObject();
74                  CounterBean c;
75                  if (o instanceof CounterBean) {
76                      c = (CounterBean)o;
77                      prev = c.getGamesPlayed();
78                  }
79              } catch (EOFException e) {
80                  // do nothing
81              } catch (IOException e) {
82                  throw new PluginException(e);
83              } catch (ClassNotFoundException e) {
84                  throw new PluginException(e);
85              }
86  
87              try(ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(data))) {
88                  synchronized (this) {
89                      CounterBean c = new CounterBean();
90                      c.setGamesPlayed(prev+1);
91                      oos.writeObject(c);
92                      oos.close();
93                  }
94              } catch (IOException e) {
95                  throw new PluginException(e);
96              }
97          } else {
98              log.error("Counter plugin: data file doesn't exist, please restart plugin.");
99          }
100     }
101 
102     @Override
103     public int getGamePlayed() throws PluginException {
104         if (!isLoaded) return -1;
105         File data = new File(PLUGIN_DATA_FOLDER_PATH + File.separator + DATA_STORAGE_FILE);
106         if (!data.exists()) {
107             return 0;
108         }
109         if (data.exists()) {
110             try(ObjectInputStream ois = new ObjectInputStream(new FileInputStream(data))) {
111                 synchronized (this) {
112                     Object o = ois.readObject();
113                     CounterBean c = null;
114                     if (o instanceof CounterBean) {
115                         c = (CounterBean)o;
116                     }
117                     ois.close();
118                     return c == null ? 0 : c.getGamesPlayed();
119                 }
120             } catch (EOFException e) {
121                 return 0;
122             } catch (IOException e) {
123                 throw new PluginException(e);
124             } catch (ClassNotFoundException e) {
125                 throw new PluginException(e);
126             }
127         } else {
128             log.error("Counter plugin: data file doesn't exist, please restart plugin.");
129             return 0;
130         }
131     }

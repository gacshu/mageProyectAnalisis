51      private CardImageUrls innerGenerateURL(CardDownloadData card, boolean isToken) {
52          String prepared = preparedUrls.getOrDefault(card, null);
53          if (prepared != null) {
54              return new CardImageUrls(prepared, null);
55          }
56  
57          String defaultCode = CardLanguage.ENGLISH.getCode();
58          String localizedCode = languageAliases.getOrDefault(this.getCurrentLanguage(), defaultCode);
59          // loc example: https://api.scryfall.com/cards/xln/121/ru?format=image
60  
61          // WARNING, some cards haven't direct images and uses random GUID:
62          // As example: Raging Ravine - https://scryfall.com/card/uma/249/raging-ravine
63          // https://img.scryfall.com/cards/large/front/5/4/54f41726-e0bb-4154-a2db-4b68b50f5032.jpg
64          String baseUrl = null;
65          String alternativeUrl = null;
66  
67          // TOKENS TRY
68  
69          // tokens support only direct links
70          if (isToken) {
71              baseUrl = ScryfallImageSupportTokens.findTokenLink(card.getSet(), card.getName(), card.getType());
72              alternativeUrl = null;
73          }
74  
75          // CARDS TRY
76  
77          // direct links to images via hardcoded API path
78          // used for cards with non-ASCII collector numbers or another use cases
79          if (baseUrl == null) {
80              String link = ScryfallImageSupportCards.findDirectDownloadLink(card.getSet(), card.getName(), card.getCollectorId());
81              if (link != null) {
82                  if (ScryfallImageSupportCards.isApiLink(link)) {
83                      // api
84                      baseUrl = link + localizedCode + "?format=image";
85                      alternativeUrl = link + defaultCode + "?format=image";
86                      // workaround to use cards without english images (some promos or special cards)
87                      if (Objects.equals(baseUrl, alternativeUrl) && baseUrl.endsWith("/en?format=image")) {
88                          alternativeUrl = alternativeUrl.replace("/en?format=image", "?format=image");
89                      }
90                  } else {
91                      // image
92                      baseUrl = link;
93                  }
94              }
95          }
96  
97          // double faced cards (modal double faces cards too)
98          if (card.isTwoFacedCard()) {
99              if (card.isSecondSide()) {
100                 // back face - must be prepared before
101                 logger.warn("Can't find back face info in prepared list "
102                         + card.getName() + " (" + card.getSet() + ") #" + card.getCollectorId());
103                 return new CardImageUrls(null, null);
104             } else {
105                 // front face - can be downloaded normally as basic card
106             }
107         }
108 
109         // basic cards by api call (redirect to img link)
110         // example: https://api.scryfall.com/cards/xln/121/en?format=image
111         if (baseUrl == null) {
112             baseUrl = String.format("https://api.scryfall.com/cards/%s/%s/%s?format=image",
113                     formatSetName(card.getSet(), isToken),
114                     card.getCollectorId(),
115                     localizedCode);
116             alternativeUrl = String.format("https://api.scryfall.com/cards/%s/%s/%s?format=image",
117                     formatSetName(card.getSet(), isToken),
118                     card.getCollectorId(),
119                     defaultCode);
120             // workaround to use cards without english images (some promos or special cards)
121             // bug: https://github.com/magefree/mage/issues/6829
122             // example: Mysterious Egg from IKO https://api.scryfall.com/cards/iko/385/?format=image
123             if (Objects.equals(baseUrl, alternativeUrl)) {
124                 // without loc code scryfall must return first available image
125                 alternativeUrl = String.format("https://api.scryfall.com/cards/%s/%s/?format=image",
126                         formatSetName(card.getSet(), isToken),
127                         card.getCollectorId());
128             }
129         }
130 
131         return new CardImageUrls(baseUrl, alternativeUrl);
132     }

134     private String getFaceImageUrl(Proxy proxy, CardDownloadData card, boolean isToken) throws Exception {
135         final String defaultCode = CardLanguage.ENGLISH.getCode();
136         final String localizedCode = languageAliases.getOrDefault(this.getCurrentLanguage(), defaultCode);
137 
138         List<String> needUrls = new ArrayList<>();
139         int needFaceIndex = card.isSecondSide() ? 1 : 0;
140 
141         String apiUrl = ScryfallImageSupportCards.findDirectDownloadLink(card.getSet(), card.getName(), card.getCollectorId());
142         if (apiUrl != null) {
143             // BY DIRECT URL
144             // direct links via hardcoded API path. Used for cards with non-ASCII collector numbers
145             if (localizedCode.equals(defaultCode)) {
146                 // english only, so can use workaround without loc param (scryfall download first available card)
147                 // workaround to use cards without english images (some promos or special cards)
148                 needUrls.add(apiUrl);
149             } else {
150                 // localize, must use loc params
151                 needUrls.add(apiUrl + localizedCode);
152                 needUrls.add(apiUrl + defaultCode);
153             }
154         } else {
155             // BY CARD NUMBER
156             // localized and default
157             needUrls.add(String.format("https://api.scryfall.com/cards/%s/%s/%s",
158                     formatSetName(card.getSet(), isToken),
159                     card.getCollectorId(),
160                     localizedCode));
161             if (!localizedCode.equals(defaultCode)) {
162                 needUrls.add(String.format("https://api.scryfall.com/cards/%s/%s/%s",
163                         formatSetName(card.getSet(), isToken),
164                         card.getCollectorId(),
165                         defaultCode));
166             }
167         }
168 
169         InputStream jsonStream = null;
170         String jsonUrl = null;
171         for (String currentUrl : needUrls) {
172             // connect to Scryfall API
173             URL cardUrl = new URL(currentUrl);
174             URLConnection request = (proxy == null ? cardUrl.openConnection() : cardUrl.openConnection(proxy));
175             request.connect();
176             try {
177                 jsonStream = (InputStream) request.getContent();
178                 jsonUrl = currentUrl;
179                 // found good url, can stop
180                 break;
181             } catch (FileNotFoundException e) {
182                 // localized image doesn't exists, try next url
183             }
184         }
185 
186         if (jsonStream == null) {
187             throw new MageException("Couldn't find card's JSON data on the server");
188         }
189 
190         // OK, found card data, parse it
191         JsonObject jsonCard = JsonParser.parseReader(new InputStreamReader(jsonStream)).getAsJsonObject();
192         JsonArray jsonFaces = JsonUtil.getAsArray(jsonCard, "card_faces");
193         if (jsonFaces == null) {
194             throw new MageException("Couldn't find card_faces in card's JSON data: " + jsonUrl);
195         }
196         if (jsonFaces.size() < needFaceIndex + 1) {
197             throw new MageException("card_faces doesn't contains face index in card's JSON data: " + jsonUrl);
198         }
199 
200         JsonObject jsonFace = jsonFaces.get(needFaceIndex).getAsJsonObject();
201         JsonObject jsonImages = JsonUtil.getAsObject(jsonFace, "image_uris");
202         if (jsonImages == null) {
203             throw new MageException("Couldn't find image_uris in card's JSON data: " + jsonUrl);
204         }
205 
206         return JsonUtil.getAsString(jsonImages, "large");
207     }

209     @Override
210     public boolean prepareDownloadList(DownloadServiceInfo downloadServiceInfo, List<CardDownloadData> downloadList) {
211         // prepare download list example (
212         Proxy proxy = downloadServiceInfo.getProxy();
213 
214         preparedUrls.clear();
215 
216         // prepare stats
217         int needPrepareCount = 0;
218         int currentPrepareCount = 0;
219         for (CardDownloadData card : downloadList) {
220             if (card.isTwoFacedCard() && card.isSecondSide()) {
221                 needPrepareCount++;
222             }
223         }
224         updatePrepareStats(downloadServiceInfo, needPrepareCount, currentPrepareCount);
225 
226         for (CardDownloadData card : downloadList) {
227             // need cancel
228             if (downloadServiceInfo.isNeedCancel()) {
229                 return false;
230             }
231 
232             // prepare the back face URL
233             if (card.isTwoFacedCard() && card.isSecondSide()) {
234                 currentPrepareCount++;
235                 try {
236                     String url = getFaceImageUrl(proxy, card, card.isToken());
237                     preparedUrls.put(card, url);
238                 } catch (Exception e) {
239                     logger.warn("Failed to prepare image URL (back face) for " + card.getName() + " (" + card.getSet() + ") #"
240                             + card.getCollectorId() + ", Error Message: " + e.getMessage());
241                     downloadServiceInfo.incErrorCount();
242                 }
243 
244                 updatePrepareStats(downloadServiceInfo, needPrepareCount, currentPrepareCount);
245             }
246         }
247 
248         return true;
249     }

251     private void updatePrepareStats(DownloadServiceInfo service, int need, int current) {
252         synchronized (service.getSync()) {
253             service.updateProgressMessage(
254                     String.format("Preparing download list... %d of %d", current, need),
255                     current,
256                     need
257             );
258         }
259     }

327     private String formatSetName(String setName, boolean isToken) {
328         if (isToken) {
329             // tokens uses xmage codes to find direct link download
330             return setName.toLowerCase(Locale.ENGLISH);
331         } else {
332             return ScryfallImageSupportCards.findScryfallSetCode(setName);
333         }
334     }

336     @Override
337     public List<String> getSupportedSets() {
338 
339         // cards
340         List<String> supportedSetsCopy = new ArrayList<>(ScryfallImageSupportCards.getSupportedSets());
341 
342         // tokens
343         for (String code : ScryfallImageSupportTokens.getSupportedSets().keySet()) {
344             if (!supportedSetsCopy.contains(code)) {
345                 supportedSetsCopy.add(code);
346             }
347         }
348 
349         return supportedSetsCopy;
350     }

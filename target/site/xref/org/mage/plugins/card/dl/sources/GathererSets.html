174     // checks for wrong card settings and support (easy to control what all good)
175     private static final HashMap<String, CheckResult> setsToDownload = new HashMap<>();
176 
177     private void CheckSearchResult(String searchCode, ExpansionSet foundedExp, boolean canDownloadTask,
178                                    boolean haveCommon, boolean haveUncommon, boolean haveRare, boolean haveMyth) {
179         // duplicated in settings
180         CheckResult res = setsToDownload.get(searchCode);
181 
182         if (res != null) {
183             logger.error(String.format("Symbols: found duplicate code: %s", searchCode));
184         } else {
185             res = new CheckResult(searchCode, foundedExp, haveCommon, haveUncommon, haveRare, haveMyth);
186             setsToDownload.put(searchCode, res);
187         }
188 
189         // not found
190         if (foundedExp == null) {
191             logger.error(String.format("Symbols: can't find set by code: %s", searchCode));
192             return;
193         }
194 
195         // checks for founded sets only
196         // too early to download
197         if (!canDownloadTask) {
198             Calendar c = Calendar.getInstance();
199             c.setTime(foundedExp.getReleaseDate());
200             c.add(Calendar.DATE, -1 * DAYS_BEFORE_RELEASE_TO_DOWNLOAD);
201             logger.warn(String.format("Symbols: early to download: %s (%s), available after %s",
202                     searchCode, foundedExp.getName(), c.getTime()));
203         }
204     }

206     private void AnalyseSearchResult() {
207         // analyze supported sets and show wrong settings
208         Date startedDate = new Date();
209 
210         for (ExpansionSet set : Sets.getInstance().values()) {
211 
212             CheckResult res = setsToDownload.get(set.getCode());
213 
214             // 1. not configured at all
215             if (res == null) {
216                 logger.warn(String.format("Symbols: set is not configured: %s (%s)", set.getCode(), set.getName()));
217                 continue; // can't do other checks
218             }
219 
220             if (logger.isDebugEnabled()) {
221                 // 2. missing rarity icon:
222                 // WARNING, need too much time (60+ secs), only for debug mode
223                 ///*
224                 if (!set.getCardsByRarity(Rarity.COMMON).isEmpty() && !res.haveCommon) {
225                     logger.error(String.format("Symbols: set have common cards, but don't download icon: %s (%s)", set.getCode(), set.getName()));
226                 }
227                 if (!set.getCardsByRarity(Rarity.UNCOMMON).isEmpty() && !res.haveUncommon) {
228                     logger.error(String.format("Symbols: set have uncommon cards, but don't download icon: %s (%s)", set.getCode(), set.getName()));
229                 }
230                 if (!set.getCardsByRarity(Rarity.RARE).isEmpty() && !res.haveRare) {
231                     logger.error(String.format("Symbols: set have rare cards, but don't download icon: %s (%s)", set.getCode(), set.getName()));
232                 }
233                 if (!set.getCardsByRarity(Rarity.MYTHIC).isEmpty() && !res.haveMyth) {
234                     logger.error(String.format("Symbols: set have mythic cards, but don't download icon: %s (%s)", set.getCode(), set.getName()));
235                 }
236                 //*/
237 
238                 // 3. info: sets with alternative numbers
239                 for (ExpansionSet.SetCardInfo card : set.getSetCardInfo()) {
240                     if (String.valueOf(card.getCardNumberAsInt()).length() != card.getCardNumber().length()) {
241                         logger.info(String.format("Symbols: set have alternative card but do not config to it: %s (%s)", set.getCode(), set.getName()));
242                         break;
243                     }
244                 }
245 
246                 // 4. info: sets with missing cards for boosters
247                 if (set.getMaxCardNumberInBooster() != Integer.MAX_VALUE) {
248                     for (ExpansionSet.SetCardInfo card : set.getSetCardInfo()) {
249                         if (card.getCardNumberAsInt() > set.getMaxCardNumberInBooster()) {
250                             if (card.getRarity() == Rarity.LAND) {
251                                 logger.info(String.format("Symbols: set's booster have land above max card number: %s (%s), %s - %s", set.getCode(), set.getName(), card.getCardNumber(), card.getName()));
252                             } else {
253                                 logger.info(String.format("Symbols: set's booster missing nonland card:: %s (%s), %s - %s", set.getCode(), set.getName(), card.getCardNumber(), card.getName()));
254                             }
255                         }
256                     }
257                 }
258             }
259         }
260 
261         Date endedDate = new Date();
262         long secs = (endedDate.getTime() - startedDate.getTime()) / 1000;
263         logger.debug(String.format("Symbols: check completed after %d seconds", secs));
264     }

266     @Override
267     public Iterator<DownloadJob> iterator() {
268         Calendar c = Calendar.getInstance();
269         c.setTime(new Date());
270         c.add(Calendar.DATE, DAYS_BEFORE_RELEASE_TO_DOWNLOAD);
271         Date compareDate = c.getTime();
272         List<DownloadJob> jobs = new ArrayList<>();
273         boolean canDownload;
274 
275         setsToDownload.clear();
276 
277         for (String symbol : symbolsBasic) {
278             ExpansionSet exp = Sets.findSet(symbol);
279             canDownload = false;
280             if (exp != null && exp.getReleaseDate().before(compareDate)) {
281                 canDownload = true;
282                 jobs.add(generateDownloadJob(symbol, "C", "C"));
283                 jobs.add(generateDownloadJob(symbol, "U", "U"));
284                 jobs.add(generateDownloadJob(symbol, "R", "R"));
285             }
286             CheckSearchResult(symbol, exp, canDownload, true, true, true, false);
287         }
288 
289         for (String symbol : symbolsBasicWithMyth) {
290             ExpansionSet exp = Sets.findSet(symbol);
291             canDownload = false;
292             if (exp != null && exp.getReleaseDate().before(compareDate)) {
293                 canDownload = true;
294                 jobs.add(generateDownloadJob(symbol, "C", "C"));
295                 jobs.add(generateDownloadJob(symbol, "U", "U"));
296                 jobs.add(generateDownloadJob(symbol, "R", "R"));
297                 jobs.add(generateDownloadJob(symbol, "M", "M"));
298             }
299             CheckSearchResult(symbol, exp, canDownload, true, true, true, true);
300         }
301 
302         for (String symbol : symbolsOnlyMyth) {
303             ExpansionSet exp = Sets.findSet(symbol);
304             canDownload = false;
305             if (exp != null && exp.getReleaseDate().before(compareDate)) {
306                 canDownload = true;
307                 jobs.add(generateDownloadJob(symbol, "M", "M"));
308             }
309             CheckSearchResult(symbol, exp, canDownload, false, false, false, true);
310         }
311 
312         for (String symbol : symbolsOnlySpecial) {
313             ExpansionSet exp = Sets.findSet(symbol);
314             canDownload = false;
315             if (exp != null && exp.getReleaseDate().before(compareDate)) {
316                 canDownload = true;
317                 jobs.add(generateDownloadJob(symbol, "M", "S"));
318             }
319             CheckSearchResult(symbol, exp, canDownload, false, false, false, true);
320         }
321 
322         // check wrong settings
323         AnalyseSearchResult();
324 
325         return jobs.iterator();
326     }

328     private DownloadJob generateDownloadJob(String set, String rarity, String urlRarity) {
329         File dst = new File(outDir, set + '-' + rarity + ".jpg");
330         if (codeReplacements.containsKey(set)) {
331             set = codeReplacements.get(set);
332         }
333         String url = "https://gatherer.wizards.com/Handlers/Image.ashx?type=symbol&set=" + set + "&size=small&rarity=" + urlRarity;
334         return new DownloadJob(set + '-' + rarity, fromURL(url), toFile(dst));
335     }

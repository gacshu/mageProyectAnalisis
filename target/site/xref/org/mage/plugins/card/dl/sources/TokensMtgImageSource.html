70      private String getEmblemName(String originalName) {
71  
72          for (SubType subType : SubType.getPlaneswalkerTypes()) {
73              if (originalName.toLowerCase(Locale.ENGLISH).contains(subType.toString().toLowerCase(Locale.ENGLISH))) {
74                  return subType.getDescription() + " Emblem";
75              }
76          }
77          return null;

80      @Override
81      public CardImageUrls generateTokenUrl(CardDownloadData card) throws IOException {
82          String name = card.getName();
83          String set = card.getSet();
84          int type = card.getType();
85  
86          // handle emblems
87          if (name.toLowerCase(Locale.ENGLISH).contains("emblem")) {
88              name = getEmblemName(name);
89          }
90  
91          // we should replace some set names
92          if (SET_NAMES_REPLACEMENT.containsKey(set.toLowerCase(Locale.ENGLISH))) {
93              set = SET_NAMES_REPLACEMENT.get(set.toLowerCase(Locale.ENGLISH));
94          }
95  
96          // Image URL contains token number
97          // e.g. https://tokens.mtg.onl/tokens/ORI_010-Thopter.jpg -- token number 010
98          // We don't know these numbers, but we can take them from a file
99          // with tokens information that can be downloaded from the site.
100         if (tokensData.isEmpty()) {
101             LOGGER.info("Source " + getSourceName() + " provides no token data.");
102             return null;
103         }
104 
105         String key = set + "/" + name;
106         List<TokenData> list = tokensData.get(key);
107         if (list == null) {
108             LOGGER.warn("Could not find data for token " + name + ", set " + set + ".");
109             return null;
110         }
111 
112         TokenData tokenData;
113         if (type == 0) {
114             tokenData = list.get(0);
115         } else {
116             if (type > list.size()) {
117                 LOGGER.warn("Not enough images variants for token with type number " + type + ", name " + name + ", set " + set + '.');
118                 return null;
119             }
120             tokenData = list.get(card.getType() - 1);
121         }
122 
123         String url = "https://tokens.mtg.onl/tokens/" + tokenData.getExpansionSetCode().trim() + '_'
124                 + tokenData.getNumber().trim() + '-' + tokenData.getName().trim() + ".jpg";
125         url = url.replace(' ', '-');
126         return new CardImageUrls(url);
127     }

184     private HashMap<String, List<TokenData>> getTokensData() throws IOException {
185         synchronized (tokensDataSync) {
186             if (tokensData == null) {
187                 DownloadPicturesService.getInstance().updateGlobalMessage("Find tokens data...");
188                 tokensData = new HashMap<>();
189 
190                 // get tokens data from resource file
191                 try (InputStream inputStream = this.getClass().getResourceAsStream("/tokens-mtg-onl-list.csv")) {
192                     List<TokenData> fileTokensData = parseTokensData(inputStream);
193                     for (TokenData tokenData : fileTokensData) {
194                         String key = tokenData.getExpansionSetCode() + "/" + tokenData.getName();
195                         List<TokenData> list = tokensData.get(key);
196                         if (list == null) {
197                             list = new ArrayList<>();
198                             tokensData.put(key, list);
199                             supportedSets.add(tokenData.getExpansionSetCode());
200                             LOGGER.debug("Added key: " + key);
201                         }
202                         list.add(tokenData);
203                     }
204                 } catch (Exception exception) {
205                     LOGGER.warn("Failed to get tokens description from resource file tokens-mtg-onl-list.csv", exception);
206                 }
207 
208                 String urlString = "http://tokens.mtg.onl/data/SetsWithTokens.csv";
209                 Proxy proxy = CardImageUtils.getProxyFromPreferences();
210                 URLConnection conn = proxy == null ? new URL(urlString).openConnection() : new URL(urlString).openConnection(proxy);
211 
212                 // description on site may contain new information
213                 // try to add it
214                 try (InputStream inputStream = conn.getURL().openStream()) {
215                     List<TokenData> siteTokensData = parseTokensData(inputStream);
216                     for (TokenData siteData : siteTokensData) {
217                         // logger.info("TOK: " + siteData.getExpansionSetCode() + "/" + siteData.getName());
218                         String key = siteData.getExpansionSetCode() + "/" + siteData.getName();
219                         supportedSets.add(siteData.getExpansionSetCode());
220                         List<TokenData> list = tokensData.get(key);
221                         if (list == null) {
222                             list = new ArrayList<>();
223                             tokensData.put(key, list);
224                             list.add(siteData);
225                         } else {
226                             boolean newToken = true;
227                             for (TokenData tokenData : list) {
228                                 if (siteData.getNumber().equals(tokenData.number)) {
229                                     newToken = false;
230                                     break;
231                                 }
232                             }
233                             if (newToken) {
234                                 list.add(siteData);
235                             }
236                         }
237                     }
238                     DownloadPicturesService.getInstance().updateGlobalMessage("");
239                     DownloadPicturesService.getInstance().showDownloadControls(true);
240                 } catch (Exception ex) {
241                     LOGGER.warn("Failed to get tokens description from tokens.mtg.onl", ex);
242                     DownloadPicturesService.getInstance().updateGlobalMessage(ex.getMessage());
243                 }
244             }
245         }
246 
247         return tokensData;
248     }

250     private List<TokenData> parseTokensData(InputStream inputStream) throws IOException {
251         List<TokenData> newTokensData = new ArrayList<>();
252 
253         try (InputStreamReader inputReader = new InputStreamReader(inputStream, "Cp1252");
254              BufferedReader reader = new BufferedReader(inputReader)) {
255             // we have to specify encoding to read special comma
256 
257             String header = reader.readLine(); // skip header
258             String line = reader.readLine();
259             // states
260             // 0 - wait set name
261             // 1 - wait cards
262             // 2 - process cards
263             int state = 0;
264             String set = null;
265             while (line != null) {
266                 if (line.trim().isEmpty()) {
267                     if (state == 2) {
268                         state = 0;
269                     }
270                 } else if (state == 0) {
271                     set = line.substring(0, 3);
272                     state = 1;
273                 } else {
274                     if (state == 1) {
275                         state = 2;
276                     }
277                     String[] split = line.split(",");
278                     // replace special comma for cards like 'Ashaya‚ the Awoken World'
279                     String name = split[0].replace('‚', ',').replace("â€š", ",");
280                     String number = split[1];
281                     TokenData token = new TokenData(name, number, set);
282                     newTokensData.add(token);
283                 }
284 
285                 line = reader.readLine();
286             }
287         }
288 
289         return newTokensData;
290     }

4   import mage.MageObject;
5   import mage.abilities.effects.Effect;
6   import mage.constants.AbilityType;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  
11  /**
12   *
13   * @author BetaSteward_at_googlemail.com
14   */
15  public abstract class StaticAbility extends AbilityImpl {
16  
17      protected StaticAbility(AbilityType abilityType, Zone zone) {
18          super(abilityType, zone);
19      }
20  
21      public StaticAbility(Zone zone, Effect effect) {
22          super(AbilityType.STATIC, zone);
23          if (effect != null) {
24              this.addEffect(effect);
25          }
26      }
27  
28      @Override
29      public boolean isInUseableZone(Game game, MageObject source, GameEvent event) {
30          if (game.getShortLivingLKI(getSourceId(), zone)) {
31              return true; // maybe this can be a problem if effects removed the ability from the object
32          }
33          if (game.getPermanentEntering(getSourceId()) != null && zone == Zone.BATTLEFIELD) {
34              return true; // abilities of permanents entering battlefield are countes as on battlefield
35          }
36          return super.isInUseableZone(game, source, event);
37      }
38  
39      public StaticAbility(StaticAbility ability) {
40          super(ability);
41      }
42  }

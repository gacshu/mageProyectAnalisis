1   package mage.abilities.hint.common;
2   
3   import mage.MageInt;
4   import mage.MageObject;
5   import mage.abilities.Ability;
6   import mage.abilities.hint.Hint;
7   import mage.filter.StaticFilters;
8   import mage.game.Game;
9   
10  import java.util.List;
11  import java.util.Objects;
12  import java.util.stream.Collectors;
13  
14  /**
15   * @author TheElk801
16   */
17  public enum CovenHint implements Hint {
18      instance;
19  
20      @Override
21      public String getText(Game game, Ability ability) {
22          List<String> powers = game
23                  .getBattlefield()
24                  .getActivePermanents(
25                          StaticFilters.FILTER_CONTROLLED_CREATURE,
26                          ability.getControllerId(), ability, game
27                  )
28                  .stream()
29                  .filter(Objects::nonNull)
30                  .map(MageObject::getPower)
31                  .mapToInt(MageInt::getValue)
32                  .distinct()
33                  .sorted()
34                  .mapToObj(String::valueOf)
35                  .collect(Collectors.toList());
36          return "Different powers among creatures you control: " + powers.size()
37                  + (powers.size() > 0 ? " (" + String.join(", ", powers) + ')' : "");
38      }
39  
40      @Override
41      public Hint copy() {
42          return this;
43      }
44  }

2   package mage.abilities.common;
3   
4   import mage.abilities.TriggeredAbilityImpl;
5   import mage.abilities.effects.Effect;
6   import mage.constants.Zone;
7   import mage.game.Game;
8   import mage.game.events.GameEvent;
9   import mage.game.permanent.Permanent;
10  import mage.game.permanent.PermanentToken;
11  import mage.target.targetpointer.FixedTarget;
12  
13  public class OpponentSacrificesNonTokenPermanentTriggeredAbility extends TriggeredAbilityImpl {
14  
15      public OpponentSacrificesNonTokenPermanentTriggeredAbility(Effect effect) {
16          super(Zone.BATTLEFIELD, effect, false);
17          setTriggerPhrase("Whenever an opponent sacrifices a nontoken permanent, ");
18      }
19  
20      public OpponentSacrificesNonTokenPermanentTriggeredAbility(final OpponentSacrificesNonTokenPermanentTriggeredAbility ability) {
21          super(ability);
22      }
23  
24      @Override
25      public boolean checkEventType(GameEvent event, Game game) {
26          return event.getType() == GameEvent.EventType.SACRIFICED_PERMANENT;
27      }
28  
29      @Override
30      public boolean checkTrigger(GameEvent event, Game game) {
31          if (game.getPlayer(getControllerId()).hasOpponent(event.getPlayerId(), game)) {
32              Permanent permanent = (Permanent) game.getLastKnownInformation(event.getTargetId(), Zone.BATTLEFIELD);
33              if (permanent != null && !(permanent instanceof PermanentToken)) {
34                  getEffects().setTargetPointer(new FixedTarget(event.getTargetId(), game.getState().getZoneChangeCounter(event.getTargetId())));
35                  return true;
36              }
37          }
38          return false;
39      }
40  
41      @Override
42      public OpponentSacrificesNonTokenPermanentTriggeredAbility copy() {
43          return new OpponentSacrificesNonTokenPermanentTriggeredAbility(this);
44      }
45  
46  }

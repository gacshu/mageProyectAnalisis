17  public class BecomesTargetControlledPermanentTriggeredAbility extends TriggeredAbilityImpl {
18  
19      public BecomesTargetControlledPermanentTriggeredAbility(Effect effect, boolean optional) {
20          super(Zone.BATTLEFIELD, effect, optional);
21          setTriggerPhrase("Whenever a permanent you control becomes the target of a spell or ability an opponent controls, ");
22      }
23  
24      private BecomesTargetControlledPermanentTriggeredAbility(final BecomesTargetControlledPermanentTriggeredAbility ability) {
25          super(ability);
26      }
27  
28      @Override
29      public BecomesTargetControlledPermanentTriggeredAbility copy() {
30          return new BecomesTargetControlledPermanentTriggeredAbility(this);
31      }
32  
33      @Override
34      public boolean checkEventType(GameEvent event, Game game) {
35          return event.getType() == GameEvent.EventType.TARGETED;
36      }
37  
38      @Override
39      public boolean checkTrigger(GameEvent event, Game game) {
40          StackObject sourceObject = game.getStack().getStackObject(event.getSourceId());
41          if (sourceObject == null) {
42              return false;
43          }
44          Player targetter = game.getPlayer(event.getPlayerId());
45          if (targetter == null || !targetter.hasOpponent(this.controllerId, game)) {
46              return false;
47          }
48          Permanent permanent = game.getPermanentOrLKIBattlefield(event.getTargetId());
49          if (permanent == null || !permanent.isControlledBy(this.getControllerId())) {
50              return false;
51          }
52          // If a spell or ability an opponent controls targets a single permanent you control more than once,
53          // Battle Mammoth’s triggered ability will trigger only once.
54          // However, if a spell or ability an opponent controls targets multiple permanents you control,
55          // Battle Mammoth’s triggered ability will trigger once for each of those permanents.
56  
57          // targetMap - key - targetId, value - Set of stackObject Ids
58          Map<UUID, Set<UUID>> targetMap = (Map<UUID, Set<UUID>>) game.getState().getValue("targetMap" + this.id);
59          if (targetMap == null) {
60              targetMap = new HashMap<>();
61          }
62          Set<UUID> sourceObjects = targetMap.get(event.getTargetId());
63          if (sourceObjects == null) {
64              sourceObjects = new HashSet<>();
65          }
66          if (!sourceObjects.add(sourceObject.getId())) {
67              return false;
68          }
69          targetMap.put(event.getTargetId(), sourceObjects);
70          game.getState().setValue("targetMap" + this.id, targetMap);
71          return true;
72      }
73  }

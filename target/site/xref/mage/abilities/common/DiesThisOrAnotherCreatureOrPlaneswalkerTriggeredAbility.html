2   package mage.abilities.common;
3   
4   import mage.MageObject;
5   import mage.abilities.TriggeredAbilityImpl;
6   import mage.abilities.effects.Effect;
7   import mage.constants.Zone;
8   import mage.filter.common.FilterCreatureOrPlaneswalkerPermanent;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  import mage.game.events.ZoneChangeEvent;
12  
13  /**
14   * @author noxx
15   */
16  public class DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility extends TriggeredAbilityImpl {
17  
18      protected FilterCreatureOrPlaneswalkerPermanent filter;
19  
20      public DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility(Effect effect, boolean optional) {
21          this(effect, optional, new FilterCreatureOrPlaneswalkerPermanent());
22      }
23  
24      public DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility(Effect effect, boolean optional, FilterCreatureOrPlaneswalkerPermanent filter) {
25          super(Zone.ALL, effect, optional); // Needs "ALL" if the source itself should trigger or multiple (incl. source go to grave)
26          this.filter = filter;
27          setTriggerPhrase("Whenever {this} or another " + filter.getMessage() + " dies, ");
28      }
29  
30      public DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility(DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility ability) {
31          super(ability);
32          this.filter = ability.filter;
33      }
34  
35      @Override
36      public DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility copy() {
37          return new DiesThisOrAnotherCreatureOrPlaneswalkerTriggeredAbility(this);
38      }
39  
40      @Override
41      public boolean checkEventType(GameEvent event, Game game) {
42          return event.getType() == GameEvent.EventType.ZONE_CHANGE;
43      }
44      
45      @Override
46      public boolean checkTrigger(GameEvent event, Game game) {
47          ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
48          if (zEvent.isDiesEvent()) {
49              if (zEvent.getTarget() != null) {
50                  if (zEvent.getTarget().getId().equals(this.getSourceId())) {
51                      return true;
52                  } else {
53                      if (filter.match(zEvent.getTarget(), getControllerId(), this, game)) {
54                          return true;
55                      }
56                  }
57              }
58          }
59          return false;
60      }
61  
62      @Override
63      public boolean isInUseableZone(Game game, MageObject source, GameEvent event) {
64          return TriggeredAbilityImpl.isInUseableZoneDiesTrigger(this, event, game);
65      }
66  }

3   package mage.abilities.common;
4   
5   import mage.abilities.TriggeredAbilityImpl;
6   import mage.abilities.effects.Effect;
7   import mage.game.events.GameEvent;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  import mage.game.stack.Spell;
11  
12  /**
13   * @author jeffwadsworth
14   */
15  public class BecomesTargetControllerSpellTriggeredAbility extends TriggeredAbilityImpl {
16  
17      public BecomesTargetControllerSpellTriggeredAbility(Effect effect, boolean optional) {
18          super(Zone.BATTLEFIELD, effect, optional);
19          setTriggerPhrase("When you become the target of a spell, ");
20      }
21  
22      public BecomesTargetControllerSpellTriggeredAbility(final BecomesTargetControllerSpellTriggeredAbility ability) {
23          super(ability);
24      }
25  
26      @Override
27      public BecomesTargetControllerSpellTriggeredAbility copy() {
28          return new BecomesTargetControllerSpellTriggeredAbility(this);
29      }
30  
31      @Override
32      public boolean checkEventType(GameEvent event, Game game) {
33          return event.getType() == GameEvent.EventType.TARGETED;
34      }
35  
36      @Override
37      public boolean checkTrigger(GameEvent event, Game game) {
38          if (event.getTargetId().equals(controllerId)) {
39              if (game.getObject(event.getSourceId()) instanceof Spell) {
40                  return true;
41              }
42          }
43          return false;
44      }
45  }

5   import mage.abilities.effects.Effect;

8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.game.permanent.Permanent;
11  import mage.game.stack.Spell;
12  import mage.players.ManaPoolItem;
13  import mage.players.Player;
14  
15  import java.util.UUID;
16  
17  /**
18   * @author TheElk801
19   */
20  public class ManaSpentDelayedTriggeredAbility extends DelayedTriggeredAbility {
21  
22      private final FilterSpell filter;
23  
24      public ManaSpentDelayedTriggeredAbility(Effect effect, FilterSpell filter) {
25          super(effect, Duration.Custom, true, false);
26          this.filter = filter;
27          setTriggerPhrase("When you spend this mana to cast " + filter.getMessage() + ", ");
28      }
29  
30      private ManaSpentDelayedTriggeredAbility(final ManaSpentDelayedTriggeredAbility ability) {
31          super(ability);
32          this.filter = ability.filter;
33      }
34  
35      @Override
36      public ManaSpentDelayedTriggeredAbility copy() {
37          return new ManaSpentDelayedTriggeredAbility(this);
38      }
39  
40      @Override
41      public boolean checkEventType(GameEvent event, Game game) {
42          return event.getType() == GameEvent.EventType.MANA_PAID;
43      }
44  
45      @Override
46      public boolean checkTrigger(GameEvent event, Game game) {
47          if (!getSourceId().equals(event.getSourceId())) {
48              return false;
49          }
50          Permanent sourcePermanent = getSourcePermanentOrLKI(game);
51          if (sourcePermanent == null
52                  || sourcePermanent
53                  .getAbilities(game)
54                  .stream()
55                  .map(Ability::getOriginalId)
56                  .map(UUID::toString)
57                  .noneMatch(event.getData()::equals)) {
58              return false;
59          }
60          Spell spell = game.getStack().getSpell(event.getTargetId());
61          return spell != null && filter.match(spell, spell.getControllerId(), this, game);
62      }
63  
64      @Override
65      public boolean isInactive(Game game) {
66          if (super.isInactive(game)) {
67              return true;
68          }
69          // must remove effect on empty mana pool to fix accumulate bug
70          // if no mana in pool then it can be discarded
71          Player player = game.getPlayer(this.getControllerId());
72          return player == null
73                  || player
74                  .getManaPool()
75                  .getManaItems()
76                  .stream()
77                  .map(ManaPoolItem::getSourceId)
78                  .noneMatch(getSourceId()::equals);
79      }
80  }

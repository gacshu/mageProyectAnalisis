66  class AnimateDeadReplaceAbilityEffect extends ContinuousEffectImpl implements SourceEffect {
67  
68      private final boolean becomesAura;
69      private Ability newAbility;
70      private TargetCreaturePermanent newTarget;
71  
72      public AnimateDeadReplaceAbilityEffect(boolean becomesAura) {
73          super(Duration.Custom, Outcome.AddAbility);
74          this.becomesAura = becomesAura;
75          staticText = (becomesAura ? "it becomes an Aura with" :
76              "it loses \"enchant creature card in a graveyard\" and gains"
77          ) + " \"enchant creature put onto the battlefield with {this}.\"";
78          if (becomesAura) {
79              dependencyTypes.add(DependencyType.AuraAddingRemoving);
80          }
81      }
82  
83      private AnimateDeadReplaceAbilityEffect(final AnimateDeadReplaceAbilityEffect effect) {
84          super(effect);
85          this.becomesAura = effect.becomesAura;
86          this.newAbility = effect.newAbility;
87          this.newTarget = effect.newTarget;
88      }
89  
90      @Override
91      public AnimateDeadReplaceAbilityEffect copy() {
92          return new AnimateDeadReplaceAbilityEffect(this);
93      }
94  
95      @Override
96      public void init(Ability source, Game game) {
97          super.init(source, game);
98          affectedObjectList.add(new MageObjectReference(source.getSourceId(), game));
99  
100         FilterCreaturePermanent filter = new FilterCreaturePermanent("creature put onto the battlefield with {this}");
101         filter.add(new AnimateDeadPredicate(source.getSourceId()));
102         newTarget = new TargetCreaturePermanent(filter);
103         newAbility = new EnchantAbility(newTarget);
104     }
105 
106     @Override
107     public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
108         Permanent permanent = affectedObjectList.get(0).getPermanent(game);
109         if (permanent == null) {
110             discard();
111             return true;
112         }
113         switch (layer) {
114             case TypeChangingEffects_4:
115                 if (becomesAura) {
116                     permanent.addSubType(game, SubType.AURA);
117                 }
118                 break;
119             case AbilityAddingRemovingEffects_6:
120                 if (!becomesAura) {
121                     List<Ability> toRemove = new ArrayList<>();
122                     for (Ability ability : permanent.getAbilities(game)) {
123                         if (ability instanceof EnchantAbility &&
124                                 ability.getRule().equals("Enchant creature card in a graveyard")) {
125                             toRemove.add(ability);
126                         }
127                     }
128                     permanent.removeAbilities(toRemove, source.getSourceId(), game);
129                 }
130                 permanent.addAbility(newAbility, source.getSourceId(), game);
131                 permanent.getSpellAbility().getTargets().clear();
132                 permanent.getSpellAbility().getTargets().add(newTarget);
133         }
134         return true;
135     }
136 
137     @Override
138     public boolean apply(Game game, Ability source) {
139         return false;
140     }
141 
142     @Override
143     public boolean hasLayer(Layer layer) {
144         return Layer.AbilityAddingRemovingEffects_6 == layer || becomesAura && Layer.TypeChangingEffects_4 == layer;
145     }
146 }

148 class AnimateDeadPutOntoBattlefieldEffect extends OneShotEffect {
149 
150     private final boolean becomesAura;
151     private final boolean tapped;
152 
153     public AnimateDeadPutOntoBattlefieldEffect(boolean becomesAura, boolean tapped) {
154         super(Outcome.PutCreatureInPlay);
155         this.becomesAura = becomesAura;
156         this.tapped = tapped;
157         StringBuilder sb = new StringBuilder(becomesAura || tapped ? "put " : "return ");
158         sb.append(becomesAura ? "target creature card from a graveyard" : "enchanted creature card");
159         sb.append(becomesAura || tapped ? " onto" : " to");
160         sb.append(" the battlefield ");
161         if (tapped) {
162             sb.append("tapped ");
163         }
164         sb.append("under your control and attach {this} to it. When {this} leaves the battlefield, that creature's controller sacrifices it");
165         staticText = sb.toString();
166     }
167 
168     private AnimateDeadPutOntoBattlefieldEffect(final AnimateDeadPutOntoBattlefieldEffect effect) {
169         super(effect);
170         this.becomesAura = effect.becomesAura;
171         this.tapped = effect.tapped;
172     }
173 
174     @Override
175     public AnimateDeadPutOntoBattlefieldEffect copy() {
176         return new AnimateDeadPutOntoBattlefieldEffect(this);
177     }
178 
179     @Override
180     public boolean apply(Game game, Ability source) {
181         Player player = game.getPlayer(source.getControllerId());
182         Permanent enchantment = source.getSourcePermanentIfItStillExists(game);
183         if (player == null || enchantment == null) {
184             return false;
185         }
186         Card card = game.getCard(becomesAura ? getTargetPointer().getFirst(game, source) : enchantment.getAttachedTo());
187         // If this ability was copied or made to trigger an additional time, the card might no longer be in the graveyard
188         // See https://github.com/magefree/mage/issues/8253
189         if (card == null || game.getState().getZone(card.getId()) != Zone.GRAVEYARD) {
190             return true;
191         }
192         // Put card onto the battlefield under your control...
193         player.moveCards(card, Zone.BATTLEFIELD, source, game, tapped, false, false, null);
194         game.getState().processAction(game);
195 
196         Permanent creature = game.getPermanent(CardUtil.getDefaultCardSideForBattlefield(game, card).getId());
197         if (creature == null) {
198             return true;
199         }
200         // ...and attach {this} to it
201         creature.addAttachment(enchantment.getId(), source, game);
202         // When {this} leaves the battlefield, that creature's controller sacrifices it
203         game.addDelayedTriggeredAbility(new AnimateDeadDelayedTriggeredAbility(new FixedTarget(creature, game)), source);
204         return true;
205     }
206 }

218     }

222         return new AnimateDeadDelayedTriggeredAbility(this);
223     }
224

236 class AnimateDeadPredicate implements Predicate<Permanent> {
237 
238     private final UUID sourceId;
239 
240     public AnimateDeadPredicate(UUID sourceId) {
241         this.sourceId = sourceId;
242     }
243 
244     @Override
245     public boolean apply(Permanent input, Game game) {
246         AnimateDeadWatcher watcher = game.getState().getWatcher(AnimateDeadWatcher.class, sourceId);
247         return watcher != null && watcher.check(input);
248     }
249 
250     @Override
251     public String toString() {
252         return "AnimateDeadPredicate(" + sourceId + ')';
253     }

256 class AnimateDeadWatcher extends Watcher {
257 
258     private final Set<UUID> putBySource = new HashSet<>();
259 
260     public AnimateDeadWatcher() {
261         super(WatcherScope.CARD);
262     }
263 
264     @Override
265     public void watch(GameEvent event, Game game) {
266         switch (event.getType()) {
267             case ZONE_CHANGE:
268                 ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
269                 if (sourceId.equals(zEvent.getTargetId())) {
270                     // clear all data when source enchantment changes zones
271                     putBySource.clear();
272                     return;
273                 }
274                 if (zEvent.getToZone() != Zone.BATTLEFIELD) {
275                     return;
276                 }
277                 if (sourceId.equals(zEvent.getSourceId())) {
278                     putBySource.add(event.getTargetId());
279                 } else {
280                     putBySource.remove(event.getTargetId());
281                 }
282                 return;
283             case BEGINNING_PHASE_PRE:
284                 if (game.getTurnNum() == 1) {
285                     putBySource.clear();
286                 }
287         }
288     }
289 
290     public boolean check(Permanent permanent) {
291         return putBySource.contains(permanent.getId());
292     }

20  /**
21   * Each colored mana symbol (e.g. {U}) in the mana costs of permanents you
22   * control counts toward your devotion to that color.
23   *
24   * @author LevelX2
25   */
26  public enum DevotionCount implements DynamicValue {
27      W(ColoredManaSymbol.W),
28      U(ColoredManaSymbol.U),
29      B(ColoredManaSymbol.B),
30      R(ColoredManaSymbol.R),
31      G(ColoredManaSymbol.G),
32      WU(ColoredManaSymbol.W, ColoredManaSymbol.U),
33      WB(ColoredManaSymbol.W, ColoredManaSymbol.B),
34      UB(ColoredManaSymbol.U, ColoredManaSymbol.B),
35      UR(ColoredManaSymbol.U, ColoredManaSymbol.R),
36      BR(ColoredManaSymbol.B, ColoredManaSymbol.R),
37      BG(ColoredManaSymbol.B, ColoredManaSymbol.G),
38      RG(ColoredManaSymbol.R, ColoredManaSymbol.G),
39      RW(ColoredManaSymbol.R, ColoredManaSymbol.W),
40      GW(ColoredManaSymbol.G, ColoredManaSymbol.W),
41      GU(ColoredManaSymbol.G, ColoredManaSymbol.U);
42  
43      private final ArrayList<ColoredManaSymbol> devotionColors = new ArrayList<>();
44      private final String message;
45      private final String reminder;
46      private final Hint hint;
47  
48      DevotionCount(ColoredManaSymbol... devotionColor) {
49          this.devotionColors.addAll(Arrays.asList(devotionColor));
50          this.message = "your devotion to "
51                  + String.join(" and ", devotionColors.stream()
52                          .map(ColoredManaSymbol::getColorName)
53                          .collect(Collectors.toList()));
54  
55          this.reminder = "<i>(Each "
56                  + String.join(" and ", devotionColors.stream()
57                          .map(s -> "{" + s + "}")
58                          .collect(Collectors.toList()))
59                  + " in the mana costs of permanents you control counts toward "
60                  + this.message
61                  + ".)</i>";
62  
63          this.hint = new ValueHint(this.message.replace("your d", "D"), this);
64      }
65  
66      @Override
67      public int calculate(Game game, Ability sourceAbility, Effect effect) {
68          int devotion = game.getBattlefield()
69                  .getAllActivePermanents(sourceAbility.getControllerId())
70                  .stream()
71                  .map(MageObject::getManaCost)
72                  .flatMap(Collection::stream)
73                  .mapToInt(this::checkCost)
74                  .sum();
75          int countIncrease = game.getBattlefield()
76                  .getAllActivePermanents(sourceAbility.getControllerId())
77                  .stream()
78                  .map(permanent -> permanent.getAbilities(game))
79                  .flatMap(Collection::stream)
80                  .filter(IncreaseDevotionAbility.class::isInstance)
81                  .mapToInt(x -> 1)
82                  .sum();
83          return devotion + countIncrease;
84      }
85  
86      private int checkCost(ManaCost manaCost) {
87          return devotionColors.stream().anyMatch(manaCost::containsColor) ? 1 : 0;
88      }
89  
90      @Override
91      public DevotionCount copy() {
92          return this;
93      }
94  
95      @Override
96      public String toString() {
97          return "X";
98      }
99  
100     @Override
101     public String getMessage() {
102         return message;
103     }
104 
105     public String getReminder() {
106         return reminder;
107     }
108 
109     public Hint getHint() {
110         return hint;
111     }

113     public static final class IncreaseDevotionAbility extends SimpleStaticAbility {
114 
115         public IncreaseDevotionAbility() {
116             super(new InfoEffect("Your devotion to each color and each combination of colors is increased by one."));
117         }
118 
119         private IncreaseDevotionAbility(final IncreaseDevotionAbility ability) {
120             super(ability);
121         }
122 
123         @Override
124         public IncreaseDevotionAbility copy() {
125             return new IncreaseDevotionAbility(this);
126         }
127     }

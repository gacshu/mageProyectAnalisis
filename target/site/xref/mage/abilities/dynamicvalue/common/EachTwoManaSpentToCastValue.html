1   package mage.abilities.dynamicvalue.common;
2   
3   import mage.Mana;
4   import mage.abilities.Ability;
5   import mage.abilities.dynamicvalue.DynamicValue;
6   import mage.abilities.effects.Effect;
7   import mage.constants.AbilityType;
8   import mage.constants.ColoredManaSymbol;
9   import mage.game.Game;
10  import mage.watchers.common.ManaSpentToCastWatcher;
11  
12  /**
13   * @author TheElk801
14   */
15  public enum EachTwoManaSpentToCastValue implements DynamicValue {
16      WHITE(ColoredManaSymbol.W),
17      BLUE(ColoredManaSymbol.U),
18      BLACK(ColoredManaSymbol.B),
19      RED(ColoredManaSymbol.R),
20      GREEN(ColoredManaSymbol.G);
21      private final ColoredManaSymbol coloredManaSymbol;
22  
23      EachTwoManaSpentToCastValue(ColoredManaSymbol manaSymbol) {
24          this.coloredManaSymbol = manaSymbol;
25      }
26  
27      @Override
28      public int calculate(Game game, Ability sourceAbility, Effect effect) {
29          if (sourceAbility.getAbilityType() == AbilityType.SPELL) {
30              return sourceAbility.getManaCostsToPay().getUsedManaToPay().getColor(coloredManaSymbol) / 2;
31          }
32          Mana payment = game
33                  .getState()
34                  .getWatcher(ManaSpentToCastWatcher.class)
35                  .getLastManaPayment(sourceAbility.getSourceId());
36          if (payment == null) {
37              return 0;
38          }
39          return payment.getColor(coloredManaSymbol) / 2;
40      }
41  
42      @Override
43      public EachTwoManaSpentToCastValue copy() {
44          return this;
45      }
46  
47      @Override
48      public String getMessage() {
49          return "for each {" + this.coloredManaSymbol + "}{" + this.coloredManaSymbol + "} spent to cast it";
50      }
51  }

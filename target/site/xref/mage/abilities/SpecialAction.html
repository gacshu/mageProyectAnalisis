1   package mage.abilities;

3   import mage.abilities.costs.mana.AlternateManaPaymentAbility;
4   import mage.abilities.costs.mana.ManaCost;
5   import mage.abilities.mana.ManaOptions;
6   import mage.constants.AbilityType;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.stack.Spell;
10  import mage.game.stack.StackObject;
11  
12  import java.util.UUID;
13  
14  /**
15   * @author BetaSteward_at_googlemail.com, JayDi85
16   */
17  public abstract class SpecialAction extends ActivatedAbilityImpl {
18  
19      private final AlternateManaPaymentAbility manaAbility; // mana actions generates on every pay cycle, no need to copy it
20      protected ManaCost unpaidMana;
21  
22      public SpecialAction() {
23          this(Zone.ALL);
24      }
25  
26      public SpecialAction(Zone zone) {
27          this(zone, null);
28      }
29  
30      public SpecialAction(Zone zone, AlternateManaPaymentAbility manaAbility) {
31          super(AbilityType.SPECIAL_ACTION, zone);
32          this.usesStack = false;
33          this.manaAbility = manaAbility;
34      }
35  
36      public SpecialAction(final SpecialAction action) {
37          super(action);
38          this.unpaidMana = action.unpaidMana;
39          this.manaAbility = action.manaAbility;
40      }
41  
42      public boolean isManaAction() {
43          return manaAbility != null;
44      }
45  
46      public void setUnpaidMana(ManaCost manaCost) {
47          this.unpaidMana = manaCost;
48      }
49  
50      public ManaCost getUnpaidMana() {
51          return unpaidMana;
52      }
53  
54      public ManaOptions getManaOptions(Ability source, Game game, ManaCost unpaid) {
55          if (manaAbility != null) {
56              return manaAbility.getManaOptions(source, game, unpaid);
57          }
58          return null;
59      }
60  
61      @Override
62      public ActivationStatus canActivate(UUID playerId, Game game) {
63          if (isManaAction()) {
64              // limit play mana abilities by steps
65              int currentStepOrder = 0;
66              if (!game.getStack().isEmpty()) {
67                  StackObject stackObject = game.getStack().getFirst();
68                  if (stackObject instanceof Spell) {
69                      currentStepOrder = ((Spell) stackObject).getCurrentActivatingManaAbilitiesStep().getStepOrder();
70                  }
71              }
72              if (currentStepOrder > manaAbility.useOnActivationManaAbilityStep().getStepOrder()) {
73                  return ActivationStatus.getFalse();
74              }
75          }
76          return super.canActivate(playerId, game);
77      }
78  }

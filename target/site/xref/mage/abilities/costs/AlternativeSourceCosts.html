6   /**
7    * Interface for abilities that add alternative costs to the source.
8    * <p>
9    * Example of such additional source costs: {@link mage.abilities.keyword.KickerAbility}
10   *
11   * @author LevelX2
12   */
13  public interface AlternativeSourceCosts {
14  
15      /**
16       * Ask the player if they want to use the alternative costs
17       *
18       * @param ability ability the alternative cost is activated for
19       * @param game
20       * @return
21       */
22      boolean askToActivateAlternativeCosts(Ability ability, Game game);
23  
24      /**
25       * Is the alternative spell cost currently available
26       *
27       * @param source spell ability the alternative costs can be paid for
28       * @param game
29       * @return
30       */
31      boolean isAvailable(Ability source, Game game);
32  
33      /**
34       * Was the alternative cost activated
35       *
36       * @param game
37       * @param source
38       * @return
39       */
40      boolean isActivated(Ability source, Game game);
41  
42      /**
43       * Suffix string to use for game log
44       *
45       * @param game
46       * @return
47       */
48      String getCastMessageSuffix(Game game);
49  
50      void resetCost();
51  }

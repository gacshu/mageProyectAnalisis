7   import mage.abilities.effects.common.ChooseSecretOpponentEffect;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  import mage.players.Player;
11  
12  import java.util.UUID;
13  
14  /**
15   * @author LevelX2
16   * @author credman0
17   */
18  public class RevealSecretOpponentCost extends CostImpl {
19  
20      public RevealSecretOpponentCost() {
21          this.text = "Reveal the player you chose";
22      }
23  
24      public RevealSecretOpponentCost(final RevealSecretOpponentCost cost) {
25          super(cost);
26      }
27  
28      @Override
29      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
30          return controllerId != null
31                  && controllerId.equals(ChooseSecretOpponentEffect.getSecretOwner(source, game))
32                  && ChooseSecretOpponentEffect.getChosenPlayer(source, game) != null;
33      }
34  
35      @Override
36      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
37          if (controllerId == null || !controllerId.equals(ChooseSecretOpponentEffect.getSecretOwner(source, game))) {
38              return false;
39          }
40          UUID opponentId = ChooseSecretOpponentEffect.getChosenPlayer(source, game);
41          if (opponentId == null) {
42              return paid;
43          }
44          ChooseSecretOpponentEffect.setSecretOwner(null, source, game); // because only once, the value is set to null
45          Player controller = game.getPlayer(controllerId);
46          Player opponent = game.getPlayer(opponentId);
47          MageObject sourceObject = game.getObject(source);
48          if (controller != null && opponent != null && sourceObject != null) {
49              if (sourceObject instanceof Permanent) {
50                  ((Permanent) sourceObject).addInfo(ChooseSecretOpponentEffect.SECRET_OPPONENT, null, game);
51              }
52              game.informPlayers(sourceObject.getLogName() + ": " + controller.getLogName() + " reveals the secretly chosen opponent " + opponent.getLogName());
53          }
54          paid = true;
55          return paid;
56      }
57  
58      @Override
59      public RevealSecretOpponentCost copy() {
60          return new RevealSecretOpponentCost(this);
61      }
62  }

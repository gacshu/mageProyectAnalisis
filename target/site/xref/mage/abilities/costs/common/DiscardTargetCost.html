21  public class DiscardTargetCost extends CostImpl {
22  
23      List<Card> cards = new ArrayList<>();
24      protected boolean randomDiscard;
25  
26      public DiscardTargetCost(TargetCardInHand target) {
27          this(target, false);
28      }
29  
30      public DiscardTargetCost(TargetCardInHand target, boolean randomDiscard) {
31          this.addTarget(target);
32          this.randomDiscard = randomDiscard;
33          this.text = "discard " + target.getDescription();
34      }
35  
36      public DiscardTargetCost(DiscardTargetCost cost) {
37          super(cost);
38          this.cards.addAll(cost.cards);
39          this.randomDiscard = cost.randomDiscard;
40      }
41  
42      @Override
43      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
44          this.cards.clear();
45          this.targets.clearChosen();
46          Player player = game.getPlayer(controllerId);
47          if (player == null) {
48              return false;
49          }
50          int amount = this.getTargets().get(0).getNumberOfTargets();
51          if (randomDiscard) {
52              this.cards.addAll(player.discard(amount, true, true, source, game).getCards(game));
53          } else if (targets.choose(Outcome.Discard, controllerId, source.getSourceId(), source, game)) {
54              Cards toDiscard = new CardsImpl();
55              toDiscard.addAll(targets.get(0).getTargets());
56              Cards discarded = player.discard(toDiscard, true, source, game);
57              if (!discarded.isEmpty()) {
58                  cards.addAll(discarded.getCards(game));
59              }
60          }
61          paid = cards.size() >= amount;
62          return paid;
63      }
64  
65      @Override
66      public void clearPaid() {
67          super.clearPaid();
68          this.cards.clear();
69          this.targets.clearChosen();
70      }
71  
72      @Override
73      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
74          return targets.canChoose(controllerId, source, game);
75      }
76  
77      @Override
78      public DiscardTargetCost copy() {
79          return new DiscardTargetCost(this);
80      }
81  
82      public List<Card> getCards() {
83          return cards;
84      }
85  }

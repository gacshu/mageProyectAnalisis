26      @Override

28          Cost cost = ManaUtil.createManaCost(amount, game, ability, null);

30      }
31  
32      @Override
33      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
34          Cost cost = ManaUtil.createManaCost(amount, game, ability, null);
35          paid = cost.pay(ability, game, source, controllerId, noMana);
36          return paid;
37      }
38

40      public DynamicValueGenericManaCost copy() {

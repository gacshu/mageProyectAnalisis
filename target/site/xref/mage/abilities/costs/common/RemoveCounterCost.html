43      public RemoveCounterCost(Target target, CounterType counterTypeToRemove, int countersToRemove) {
44          this.target = target;
45          this.counterTypeToRemove = counterTypeToRemove;
46          this.countersToRemove = countersToRemove;

48          this.text = setText();
49      }

51      public RemoveCounterCost(final RemoveCounterCost cost) {
52          super(cost);
53          this.target = cost.target.copy();
54          this.countersToRemove = cost.countersToRemove;
55          this.counterTypeToRemove = cost.counterTypeToRemove;
56      }

58      // TODO: TayamLuminousEnigmaCost can be simplified
59      @Override
60      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
61          paid = false;
62          int countersRemoved = 0;
63          Player controller = game.getPlayer(controllerId);
64          if (controller == null) {
65              return false;
66          }
67          if (countersToRemove == 0) { // Can happen when used for X costs where X = 0;
68              paid = true;
69              return paid;
70          }
71          target.clearChosen();
72  
73          Outcome outcome;
74          if (target instanceof TargetPermanent) {
75              outcome = Outcome.UnboostCreature;
76          } else if (target instanceof TargetCard) {  // For Mari, the Killing Quill
77              outcome = Outcome.Neutral;
78          } else {
79              throw new RuntimeException(
80                      "Wrong target type provided for RemoveCounterCost. Provided " + target.getClass() + ". " +
81                              "From ability " + ability);
82          }
83  
84          if (!target.choose(outcome, controllerId, source.getSourceId(), source, game)) {
85              return paid;
86          }
87          for (UUID targetId : target.getTargets()) {
88              Card targetObject;
89              if (target instanceof TargetPermanent) {
90                  targetObject = game.getPermanent(targetId);
91              } else {  // For Mari, the Killing Quill
92                  targetObject = game.getCard(targetId);
93              }
94  
95              if (targetObject == null
96                      || targetObject.getCounters(game).isEmpty()
97                      || !(counterTypeToRemove == null || targetObject.getCounters(game).containsKey(counterTypeToRemove))) {
98                  continue;
99              }
100             String counterName = null;
101 
102             if (counterTypeToRemove != null) {  // Counter type specified
103                 counterName = counterTypeToRemove.getName();
104             } else if (targetObject.getCounters(game).size() == 1) {  // Only one counter of creature
105                 for (Counter counter : targetObject.getCounters(game).values()) {
106                     if (counter.getCount() > 0) {
107                         counterName = counter.getName();
108                     }
109                 }
110             } else {  // Multiple counters, player much choose which type to remove from
111                 Choice choice = new ChoiceImpl(true);
112                 Set<String> choices = new LinkedHashSet<>();
113                 for (Counter counter : targetObject.getCounters(game).values()) {
114                     if (targetObject.getCounters(game).getCount(counter.getName()) > 0) {
115                         choices.add(counter.getName());
116                     }
117                 }
118                 choice.setChoices(choices);
119                 choice.setMessage("Choose a counter to remove from " + targetObject.getLogName());
120                 if (!controller.choose(Outcome.UnboostCreature, choice, game)) {
121                     return false;
122                 }
123                 counterName = choice.getChoice();
124             }
125 
126             if (counterName != null && !counterName.isEmpty()) {
127                 int countersLeft = countersToRemove - countersRemoved;
128                 int countersOnPermanent = targetObject.getCounters(game).getCount(counterName);
129                 int numberOfCountersSelected = 1;
130                 if (countersLeft > 1 && countersOnPermanent > 1) {
131                     numberOfCountersSelected = controller.getAmount(1, Math.min(countersLeft, countersOnPermanent),
132                             "Remove how many counters from " + targetObject.getIdName(), game);
133                 }
134                 targetObject.removeCounters(counterName, numberOfCountersSelected, source, game);
135                 countersRemoved += numberOfCountersSelected;
136                 if (!game.isSimulation()) {
137                     game.informPlayers(controller.getLogName() +
138                             " removes " + (numberOfCountersSelected == 1 ? "a" : numberOfCountersSelected) + ' ' +
139                             counterName + (numberOfCountersSelected == 1 ? " counter from " : " counters from ") +
140                             targetObject.getName());
141                 }
142                 if (countersRemoved == countersToRemove) {
143                     this.paid = true;
144                     break;
145                 }
146             }
147         }
148 
149         return paid;
150     }

152     @Override
153     public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
154         return target.canChoose(controllerId, source, game);
155     }

157     private String setText() {
158         StringBuilder sb = new StringBuilder("remove ");
159         if (counterTypeToRemove != null) {
160             sb.append(CardUtil.numberToText(countersToRemove, counterTypeToRemove.getArticle()));
161             sb.append(' ');
162             sb.append(counterTypeToRemove.getName());
163         } else {
164             sb.append(CardUtil.numberToText(countersToRemove, "a"));
165         }
166         sb.append(countersToRemove > 1 ? " counters from " : " counter from ");
167         if (target.getMaxNumberOfTargets() > 1) {
168             sb.append(target.getTargetName());
169         } else {
170             sb.append(CardUtil.addArticle(target.getTargetName()));
171         }
172         return sb.toString();
173     }

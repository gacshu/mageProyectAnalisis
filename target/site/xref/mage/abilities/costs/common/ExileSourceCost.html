1   package mage.abilities.costs.common;

3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.costs.Cost;
6   import mage.abilities.costs.CostImpl;
7   import mage.cards.Card;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  import mage.players.Player;
11  
12  import java.util.UUID;
13  
14  /**
15   * @author BetaSteward_at_googlemail.com
16   */
17  public class ExileSourceCost extends CostImpl {
18  
19      /**
20       * @param toUniqueExileZone moves the card to a source object dependant
21       *                          unique exile zone, so another effect of the same source object (e.g.
22       *                          Deadeye Navigator) can identify the card
23       */
24      public ExileSourceCost() {
25          this.text = "exile {this}";
26      }
27  
28      public ExileSourceCost(ExileSourceCost cost) {
29          super(cost);
30      }
31  
32      @Override
33      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
34          MageObject sourceObject = ability.getSourceObject(game);
35          Player controller = game.getPlayer(controllerId);
36          if (controller == null || !(sourceObject instanceof Card)) {
37              return paid;
38          }
39          controller.moveCards((Card) sourceObject, Zone.EXILED, source, game);
40          // 117.11. The actions performed when paying a cost may be modified by effects.
41          // Even if they are, meaning the actions that are performed don't match the actions
42          // that are called for, the cost has still been paid.
43          // so return state here is not important because the user indended to exile the target anyway
44          paid = true;
45          return paid;
46      }
47  
48      @Override
49      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
50          return source.getSourceObjectIfItStillExists(game) instanceof Card;
51      }
52  
53      @Override
54      public ExileSourceCost copy() {
55          return new ExileSourceCost(this);
56      }
57  }

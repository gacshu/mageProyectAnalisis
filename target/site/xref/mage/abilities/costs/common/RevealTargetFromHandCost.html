19   *
20   * @author jeffwadsworth

22  public class RevealTargetFromHandCost extends CostImpl {
23  
24      public int manaValues = 0;
25      protected int numberCardsRevealed = 0;
26      protected List<Card> revealedCards;
27  
28      public RevealTargetFromHandCost(TargetCardInHand target) {
29          this.addTarget(target);
30          this.text = (target.getNumberOfTargets() == 0 ? "you may reveal " : "reveal ") + target.getDescription();
31          this.revealedCards = new ArrayList<>();
32      }
33  
34      public RevealTargetFromHandCost(final RevealTargetFromHandCost cost) {
35          super(cost);
36          this.manaValues = cost.manaValues;
37          this.numberCardsRevealed = cost.numberCardsRevealed;
38          this.revealedCards = new ArrayList<>(cost.revealedCards);
39      }
40  
41      @Override
42      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
43          if (targets.choose(Outcome.Benefit, controllerId, source.getSourceId(), source, game)) {
44              manaValues = 0;
45              numberCardsRevealed = 0;
46              Player player = game.getPlayer(controllerId);
47              Cards cards = new CardsImpl();
48              for (UUID targetId : targets.get(0).getTargets()) {
49                  Card card = player.getHand().get(targetId, game);
50                  if (card != null) {
51                      manaValues += card.getManaValue();
52                      numberCardsRevealed++;
53                      cards.add(card);
54                      revealedCards.add(card);
55                  }
56              }
57              if (numberCardsRevealed > 0) {
58                  MageObject baseObject = game.getBaseObject(source.getSourceId());
59                  player.revealCards(baseObject == null ? "card cost" : baseObject.getIdName(), cards, game);
60              }
61              if (targets.get(0).getNumberOfTargets() <= numberCardsRevealed) {
62                  paid = true; // e.g. for optional additional costs.  example: Dragonlord's Prerogative also true if 0 cards shown
63                  return paid;
64              }
65          }
66          paid = false;
67          return paid;
68      }
69  
70      public int getConvertedCosts() {
71          return manaValues;
72      }
73  
74      public int getNumberRevealedCards() {
75          return numberCardsRevealed;
76      }
77  
78      public List<Card> getRevealedCards() {
79          return revealedCards;
80      }
81  
82      @Override
83      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
84          return targets.canChoose(controllerId, source, game);
85      }
86  
87      @Override
88      public RevealTargetFromHandCost copy() {
89          return new RevealTargetFromHandCost(this);
90      }
91  }

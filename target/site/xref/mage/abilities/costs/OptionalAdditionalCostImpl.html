3   import mage.abilities.costs.mana.VariableManaCost;

5   /**
6    * @author LevelX2
7    */
8   public class OptionalAdditionalCostImpl extends CostsImpl<Cost> implements OptionalAdditionalCost {
9   
10      protected String name;
11      protected String reminderText;
12      protected String delimiter;
13  
14      protected boolean activated;
15      protected int activatedCounter;
16      protected boolean repeatable;
17  
18      public OptionalAdditionalCostImpl(String name, String reminderText, Cost cost) {
19          this(name, " ", reminderText, cost);
20      }
21  
22      public OptionalAdditionalCostImpl(String name, String delimiter, String reminderText, Cost cost) {
23          this.activated = false;
24          this.name = name;
25          this.delimiter = delimiter;
26          this.reminderText = "<i>(" + reminderText + ")</i>";
27          this.activatedCounter = 0;
28          this.add(cost);
29      }
30  
31      public OptionalAdditionalCostImpl(final OptionalAdditionalCostImpl cost) {
32          super(cost);
33          this.name = cost.name;
34          this.reminderText = cost.reminderText;
35          this.delimiter = cost.delimiter;
36          this.activated = cost.activated;
37          this.activatedCounter = cost.activatedCounter;
38          this.repeatable = cost.repeatable;
39      }
40  
41      @Override
42      public String getName() {
43          return this.name;
44      }
45  
46      /**
47       * Returns the complete text for the addional cost or if onlyCost is true
48       * only the pure text fore the included native cost
49       *
50       * @param onlyCost
51       * @return
52       */
53      @Override
54      public String getText(boolean onlyCost) {
55          if (onlyCost) {
56              return getText();
57          } else {
58              return name + delimiter + getText();
59          }
60      }
61  
62      /**
63       * Returns a reminder text, if the cost has one
64       *
65       * @return
66       */
67      @Override
68      public String getReminderText() {
69          String replace = "";
70          if (reminderText != null && !reminderText.isEmpty()) {
71              replace = reminderText.replace("{cost}", this.getText(true));
72          }
73          return replace;
74      }
75  
76      @Override
77      public void setReminderText(String reminderText) {
78          this.reminderText = reminderText;
79      }
80  
81      @Override
82      public void setMinimumCost(int minimumCost) {
83          for (VariableCost cost : this.getVariableCosts()) {
84              if (cost instanceof VariableManaCost) {
85                  ((VariableManaCost) cost).setMinX(minimumCost);
86              }
87          }
88      }
89  
90      /**
91       * Returns a text suffix for the game log, that can be added to the cast
92       * message.
93       *
94       * @param position - if there are multiple costs, it's the postion the cost
95       *                 is set (starting with 0)
96       * @return
97       */
98      @Override
99      public String getCastSuffixMessage(int position) {
100         StringBuilder sb = new StringBuilder();
101         if (isActivated() && (!isRepeatable() || getActivateCount() > 0)) {
102             sb.append(position > 0 ? " and " : "").append(" with ");
103             if (isRepeatable()) {
104                 sb.append(getActivateCount()).append(getActivateCount() > 1 ? " times " : " time ");
105             }
106             sb.append(name);
107         }
108         return sb.toString();
109     }
110 
111     /**
112      * If the player intends to pay the cost, the cost will be activated
113      */
114     @Override
115     public void activate() {
116         activated = true;
117         ++activatedCounter;
118     }
119 
120     /**
121      * Reset the activate and count information
122      */
123     @Override
124     public void reset() {
125         activated = false;
126         activatedCounter = 0;
127     }
128 
129     /**
130      * Set if the cost be multiple times activated
131      *
132      * @param repeatable
133      */
134     @Override
135     public void setRepeatable(boolean repeatable) {
136         this.repeatable = repeatable;
137     }
138 
139     /**
140      * Can the cost be multiple times activated
141      *
142      * @return
143      */
144     @Override
145     public boolean isRepeatable() {
146         return repeatable;
147     }
148 
149     /**
150      * Returns if the cost was activated
151      *
152      * @return
153      */
154     @Override
155     public boolean isActivated() {
156         return activated;
157     }
158 
159     /**
160      * Returns the number of times the cost was activated
161      *
162      * @return
163      */
164     @Override
165     public int getActivateCount() {
166         return activatedCounter;
167     }
168 
169     @Override
170     public void setCostType(VariableCostType costType) {
171         this.getVariableCosts().forEach(cost -> {
172             cost.setCostType(costType);
173         });
174     }
175 
176     @Override
177     public OptionalAdditionalCostImpl copy() {
178         return new OptionalAdditionalCostImpl(this);
179     }
180 }

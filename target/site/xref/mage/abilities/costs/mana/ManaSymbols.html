3   package mage.abilities.costs.mana;
4   
5   import mage.ManaSymbol;
6   import mage.constants.ColoredManaSymbol;
7   
8   import java.util.ArrayList;
9   import java.util.EnumMap;
10  import java.util.Map;
11  
12  /**
13   * Represents the mana symbols on a card.
14   *
15   * @author noxx
16   */
17  public class ManaSymbols extends ArrayList<ManaSymbol> {
18  
19      private static final Map<ColoredManaSymbol, ManaSymbol> coloredManaMap = new EnumMap<ColoredManaSymbol, ManaSymbol>(ColoredManaSymbol.class) {{
20          put(ColoredManaSymbol.W, ManaSymbol.W);
21          put(ColoredManaSymbol.U, ManaSymbol.U);
22          put(ColoredManaSymbol.B, ManaSymbol.B);
23          put(ColoredManaSymbol.R, ManaSymbol.R);
24          put(ColoredManaSymbol.G, ManaSymbol.G);
25      }};
26  
27      /**
28       * Contains all possible hybrid mana costs (each represents different hybrid mana symbol)
29       * We'll use it for converting from hybrid mana cost to hybrid mana symbol.
30       */
31      private static final Map<ManaSymbol, HybridManaCost> hybridManaMap = new EnumMap<ManaSymbol, HybridManaCost>(ManaSymbol.class) {{
32  
33      /**
34       *  Build map of all possible hybrid mana symbols assigning corresponding instance of hybrid mana cost.
35       */
36          put(ManaSymbol.HYBRID_BG, new HybridManaCost(ColoredManaSymbol.B, ColoredManaSymbol.G));
37          put(ManaSymbol.HYBRID_BR, new HybridManaCost(ColoredManaSymbol.B, ColoredManaSymbol.R));
38          put(ManaSymbol.HYBRID_GU, new HybridManaCost(ColoredManaSymbol.G, ColoredManaSymbol.U));
39          put(ManaSymbol.HYBRID_GW, new HybridManaCost(ColoredManaSymbol.G, ColoredManaSymbol.W));
40          put(ManaSymbol.HYBRID_RG, new HybridManaCost(ColoredManaSymbol.R, ColoredManaSymbol.G));
41          put(ManaSymbol.HYBRID_RW, new HybridManaCost(ColoredManaSymbol.R, ColoredManaSymbol.W));
42          put(ManaSymbol.HYBRID_UB, new HybridManaCost(ColoredManaSymbol.U, ColoredManaSymbol.B));
43          put(ManaSymbol.HYBRID_UR, new HybridManaCost(ColoredManaSymbol.U, ColoredManaSymbol.R));
44          put(ManaSymbol.HYBRID_WB, new HybridManaCost(ColoredManaSymbol.W, ColoredManaSymbol.B));
45          put(ManaSymbol.HYBRID_WU, new HybridManaCost(ColoredManaSymbol.W, ColoredManaSymbol.U));
46      }};
47  
48      /**
49       * Extracts mana symbols from {@link ManaCost} using {@link mage.ManaSymbol} as a base class for the mana symbols.
50       *
51       * @param manaCost
52       * @return
53       */
54      public static ManaSymbols buildFromManaCost(ManaCost manaCost) {
55  
56          ManaSymbols manaSymbols = new ManaSymbols();
57  
58          if (manaCost instanceof ManaCostsImpl) {
59              ManaCostsImpl manaCosts = (ManaCostsImpl)manaCost;
60              for (int i = 0; i < manaCosts.size(); i++) {
61                  ManaCost mc = (ManaCost) manaCosts.get(i);
62                  if (mc instanceof ColoredManaCost) {
63                      for (Map.Entry<ColoredManaSymbol, ManaSymbol> entry : coloredManaMap.entrySet()) {
64                          if (mc.containsColor(entry.getKey())) {
65                              manaSymbols.add(entry.getValue());
66                              break;
67                          }
68                      }
69                  } else if (mc instanceof HybridManaCost) {
70                      for (Map.Entry<ManaSymbol, HybridManaCost> entry : hybridManaMap.entrySet()) {
71                          if (compareHybridCosts((HybridManaCost) mc, entry.getValue())) {
72                              manaSymbols.add(entry.getKey());
73                              break;
74                          }
75                      }
76                  }
77              }
78          }
79  
80          return manaSymbols;
81      }
82  
83      /**
84       * Compare two instance of hybrid mana cost.
85       *
86       * @param h1
87       * @param h2
88       *
89       * @return
90       */
91      private static boolean compareHybridCosts(HybridManaCost h1, HybridManaCost h2) {
92          return h1.getMana1() == h2.getMana1() && h1.getMana2() == h2.getMana2()
93              || h1.getMana1() == h2.getMana2() && h1.getMana2() == h2.getMana1();
94      }
95  }

7   /**
8    * Interface for abilities that allow the player to pay mana costs of a spell in alternate ways.
9    * For the payment SpecialActions are used.
10   * <p>
11   * Example of such an alternate payment ability: {@link mage.abilities.keyword.DelveAbility}
12   *
13   * @author LevelX2, JayDi85
14   */
15  public interface AlternateManaPaymentAbility {
16      /**
17       * Adds the special action to the state, that allows the user to do the alternate mana payment
18       *
19       * @param source
20       * @param game
21       * @param unpaid unapaid mana costs of the spell
22       */
23      void addSpecialAction(Ability source, Game game, ManaCost unpaid);
24  
25      /**
26       * All possible mana payments that can make that ability (uses to find playable abilities)
27       *
28       * @param source
29       * @param game
30       * @param unpaid
31       * @return
32       */
33      ManaOptions getManaOptions(Ability source, Game game, ManaCost unpaid);
34  
35      /**
36       * Mana payment step where you can use it
37       *
38       * @return
39       */
40      ActivationManaAbilityStep useOnActivationManaAbilityStep();
41  }

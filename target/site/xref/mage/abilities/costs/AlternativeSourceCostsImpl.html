14  import java.util.Iterator;

16  /**
17   * @author TheElk801
18   */
19  public abstract class AlternativeSourceCostsImpl extends StaticAbility implements AlternativeSourceCosts {
20  
21      protected final AlternativeCost alternativeCost;
22      protected final String reminderText;
23      private int zoneChangeCounter = 0;
24  
25      protected AlternativeSourceCostsImpl(String name, String reminderText, String manaString) {
26          this(name, reminderText, new ManaCostsImpl<>(manaString));
27      }
28  
29      protected AlternativeSourceCostsImpl(String name, String reminderText, Cost cost) {
30          super(Zone.ALL, null);
31          this.name = name;
32          this.reminderText = reminderText;
33          this.alternativeCost = new AlternativeCostImpl<>(name, reminderText, cost);
34      }
35  
36      protected AlternativeSourceCostsImpl(final AlternativeSourceCostsImpl ability) {
37          super(ability);
38          this.alternativeCost = ability.alternativeCost.copy();
39          this.reminderText = ability.reminderText;
40          this.zoneChangeCounter = ability.zoneChangeCounter;
41      }
42  
43      @Override
44      public boolean askToActivateAlternativeCosts(Ability ability, Game game) {
45          if (ability instanceof SpellAbility) {
46              handleActivatingAlternativeCosts(ability, game);
47          }
48          return isActivated(ability, game);
49      }
50  
51      protected boolean handleActivatingAlternativeCosts(Ability ability, Game game) {
52          Player player = game.getPlayer(ability.getControllerId());
53          if (player == null) {
54              return false;
55          }
56          this.resetCost();
57          if (!alternativeCost.canPay(ability, this, player.getId(), game)
58                  || !player.chooseUse(Outcome.Benefit, "Cast this for its " + this.name + " cost? (" + alternativeCost.getText(true) + ')', ability, game)) {
59              return false;
60          }
61          alternativeCost.activate();
62          if (zoneChangeCounter == 0) {
63              Card card = game.getCard(getSourceId());
64              if (card != null) {
65                  zoneChangeCounter = card.getZoneChangeCounter(game);
66              } else {
67                  throw new IllegalArgumentException("source card not found");
68              }
69          }
70          ability.getManaCostsToPay().clear();
71          ability.getCosts().clear();
72          for (Iterator<Cost> it = ((Costs<Cost>) alternativeCost).iterator(); it.hasNext(); ) {
73              Cost cost = it.next();
74              if (cost instanceof ManaCost) {
75                  ability.getManaCostsToPay().add((ManaCost) cost.copy());
76              } else {
77                  ability.getCosts().add(cost.copy());
78              }
79          }
80          return true;
81      }
82  
83      @Override
84      public boolean isActivated(Ability ability, Game game) {
85          Card card = game.getCard(sourceId);
86          if (card != null && card.getZoneChangeCounter(game) <= zoneChangeCounter + 1) {
87              return alternativeCost.isActivated(game);
88          }
89          return false;
90      }
91  
92      @Override
93      public Costs<Cost> getCosts() {
94          return (Costs<Cost>) alternativeCost;
95      }
96  
97      @Override
98      public String getRule() {
99          return alternativeCost.getText(false) + ' ' + alternativeCost.getReminderText();
100     }
101 
102     @Override
103     public void resetCost() {
104         alternativeCost.reset();
105         this.zoneChangeCounter = 0;
106     }
107 
108     @Override
109     public boolean isAvailable(Ability source, Game game) {
110         return true;
111     }
112 
113     public String getCastMessageSuffix(Game game) {
114         return alternativeCost.getCastSuffixMessage(0);
115     }
116 }

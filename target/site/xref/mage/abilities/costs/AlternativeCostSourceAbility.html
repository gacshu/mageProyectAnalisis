90      @Override
91      public void addCost(Cost cost) {
92          AlternativeCost alternativeCost = convertToAlternativeCost(cost);
93          if (alternativeCost != null) {
94              this.alternateCosts.add(alternativeCost);
95          }
96      }

98      private AlternativeCost convertToAlternativeCost(Cost cost) {
99          //return cost != null ? new AlternativeCost2Impl(null, cost.getText(), cost) : null;
100         return cost != null ? new AlternativeCostImpl(null, "", cost) : null;
101     }

110         boolean conditionApplies = condition == null || condition.apply(game, source);
111         boolean filterApplies = filter == null || filter.match(game.getCard(source.getSourceId()), game);

116     @Override
117     public boolean askToActivateAlternativeCosts(Ability ability, Game game) {
118         if (ability != null && AbilityType.SPELL == ability.getAbilityType()) {
119             if (filter != null) {
120                 Card card = game.getCard(ability.getSourceId());
121                 if (!filter.match(card, ability.getControllerId(), ability, game)) {
122                     return false;
123                 }
124             }
125             Player player = game.getPlayer(ability.getControllerId());
126             if (player != null) {
127                 Costs<AlternativeCost> alternativeCostsToCheck;
128                 if (dynamicCost != null) {
129                     alternativeCostsToCheck = new CostsImpl<>();
130                     alternativeCostsToCheck.add(convertToAlternativeCost(dynamicCost.getCost(ability, game)));
131                 } else {
132                     alternativeCostsToCheck = this.alternateCosts;
133                 }
134 
135                 String costChoiceText;
136                 if (dynamicCost != null) {
137                     costChoiceText = dynamicCost.getText(ability, game);
138                 } else {
139                     costChoiceText = alternativeCostsToCheck.isEmpty() ? "Cast without paying its mana cost?" : "Pay alternative costs? (" + alternativeCostsToCheck.getText() + ')';
140                 }
141                 if (alternativeCostsToCheck.canPay(ability, ability, ability.getControllerId(), game)
142                         && player.chooseUse(Outcome.Benefit, costChoiceText, this, game)) {
143                     if (ability instanceof SpellAbility) {
144                         ability.getManaCostsToPay().removeIf(VariableCost.class::isInstance);
145                         CardUtil.reduceCost((SpellAbility) ability, ability.getManaCosts());
146 
147                     } else {
148                         ability.getManaCostsToPay().clear();
149                     }
150                     if (!onlyMana) {
151                         ability.getCosts().clear();
152                     }
153                     for (AlternativeCost alternateCost : alternativeCostsToCheck) {
154                         alternateCost.activate();
155                         for (Iterator it = ((Costs) alternateCost).iterator(); it.hasNext(); ) {
156                             Cost costDetailed = (Cost) it.next();
157                             if (costDetailed instanceof ManaCost) {
158                                 ability.getManaCostsToPay().add((ManaCost) costDetailed.copy());
159                             } else if (costDetailed != null) {
160                                 ability.getCosts().add(costDetailed.copy());
161                             }
162                         }
163                     }
164 
165                     // save activated status
166                     game.getState().setValue(getActivatedKey(ability), Boolean.TRUE);
167                 } else {
168                     return false;
169                 }
170             } else {
171                 return false;
172             }
173         }
174         return isActivated(ability, game);
175     }

177     private String getActivatedKey(Ability source) {
178         return getActivatedKey(this.getOriginalId(), source.getSourceId(), source.getSourceObjectZoneChangeCounter());
179     }

181     private static String getActivatedKey(UUID alternativeCostOriginalId, UUID sourceId, int sourceZCC) {
182         // can't use sourceId cause copied cards are different...
183         // TODO: enable sourceId after copy card fix (it must copy cards with all related game state values)
184         return ALTERNATIVE_COST_ACTIVATION_KEY + "_" + alternativeCostOriginalId + "_" /*+ sourceId + "_"*/ + sourceZCC;
185     }

187     /**
188      * Search activated status of alternative cost.
189      * <p>
190      * If you need it on resolve then use current ZCC (on stack)
191      * If you need it on battlefield then use previous ZCC (-1)
192      *
193      * @param game
194      * @param source
195      * @param alternativeCostOriginalId you must save originalId on card's creation
196      * @param searchPrevZCC             true on battlefield, false on stack
197      * @return
198      */
199     public static boolean getActivatedStatus(Game game, Ability source, UUID alternativeCostOriginalId, boolean searchPrevZCC) {
200         String key = getActivatedKey(
201                 alternativeCostOriginalId,
202                 source.getSourceId(),
203                 source.getSourceObjectZoneChangeCounter() + (searchPrevZCC ? -1 : 0)
204         );
205         Boolean status = (Boolean) game.getState().getValue(key);
206         return status != null && status;
207     }

209     @Override
210     public boolean isActivated(Ability source, Game game) {
211         Costs<AlternativeCost> alternativeCostsToCheck;
212         if (dynamicCost != null) {
213             alternativeCostsToCheck = new CostsImpl<>();
214             alternativeCostsToCheck.add(convertToAlternativeCost(dynamicCost.getCost(source, game)));
215         } else {
216             alternativeCostsToCheck = this.alternateCosts;
217         }
218         for (AlternativeCost cost : alternativeCostsToCheck) {
219             if (cost.isActivated(game)) {
220                 return true;
221             }
222         }
223         return false;
224     }

236     @Override
237     public String getRule() {
238         if (rule != null) {
239             return rule;
240         }
241         StringBuilder sb = new StringBuilder();
242         if (condition != null) {
243             sb.append(condition.toString());
244             if (alternateCosts.size() > 1) {
245                 sb.append(", rather than pay this spell's mana cost, ");
246             } else {
247                 sb.append(", you may ");
248             }
249         } else {
250             sb.append("You may ");
251         }
252         int numberCosts = 0;
253         String remarkText = "";
254         sb.append(CardUtil.concatWithAnd(alternateCosts
255                 .stream()
256                 .map(cost -> cost.getCost() instanceof ManaCost
257                         ? "pay " + cost.getText(true)
258                         : cost.getText(true))
259                 .map(CardUtil::getTextWithFirstCharLowerCase)
260                 .collect(Collectors.toList())));
261         if (condition == null || alternateCosts.size() == 1) {
262             sb.append(" rather than pay this spell's mana cost");
263         } else if (alternateCosts.isEmpty()) {
264             sb.append("cast this spell without paying its mana cost");
265         }
266         sb.append('.');
267         if (numberCosts == 1 && remarkText != null) {
268             sb.append(' ').append(remarkText);
269         }
270         return sb.toString();
271     }

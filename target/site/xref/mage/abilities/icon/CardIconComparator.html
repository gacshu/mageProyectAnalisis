8   public enum CardIconComparator implements Comparator<CardIcon> {
9       instance;
10  
11      @Override
12      public int compare(CardIcon a, CardIcon b) {
13          // by icon type
14          int res = Integer.compare(a.getIconType().getSortOrder(), b.getIconType().getSortOrder());
15  
16          // by text
17          if (res == 0) {
18              res = String.CASE_INSENSITIVE_ORDER.compare(a.getCombinedInfo(), b.getCombinedInfo());
19          }
20  
21          return res;
22      }
23  }

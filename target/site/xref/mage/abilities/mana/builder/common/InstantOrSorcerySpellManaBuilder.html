19  public class InstantOrSorcerySpellManaBuilder extends ConditionalManaBuilder {
20  
21      @Override
22      public ConditionalMana build(Object... options) {
23          return new InstantOrSorceryCastConditionalMana(this.mana);
24      }
25  
26      @Override
27      public String getRule() {
28          return "Spend this mana only to cast an instant or sorcery spell";
29      }
30  }
31  
32  class InstantOrSorceryCastConditionalMana extends ConditionalMana {
33  
34      public InstantOrSorceryCastConditionalMana(Mana mana) {
35          super(mana);
36          staticText = "Spend this mana only to cast an instant or sorcery spell";
37          addCondition(new InstantOrSorceryCastManaCondition());
38      }
39  }
40  
41  class InstantOrSorceryCastManaCondition extends ManaCondition implements Condition {
42  
43      @Override
44      public boolean apply(Game game, Ability source) {
45          if (source instanceof SpellAbility) {
46              MageObject object = game.getObject(source);
47              return object != null && object.isInstantOrSorcery(game);
48          }
49          return false;
50      }
51  
52      @Override
53      public boolean apply(Game game, Ability source, UUID originalId, Cost costsToPay) {
54          return apply(game, source);
55      }
56  }

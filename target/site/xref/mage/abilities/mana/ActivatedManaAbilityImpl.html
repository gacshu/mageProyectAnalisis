18  public abstract class ActivatedManaAbilityImpl extends ActivatedAbilityImpl implements ManaAbility {
19  
20      protected List<Mana> netMana = new ArrayList<>();
21      protected boolean undoPossible;
22      protected boolean poolDependant;
23  
24      public ActivatedManaAbilityImpl(Zone zone, ManaEffect effect, Cost cost) {
25          super(AbilityType.MANA, zone);
26          this.usesStack = false;
27          this.undoPossible = true;
28          if (effect != null) {
29              this.addEffect(effect);
30          }
31          if (cost != null) {
32              this.addCost(cost);
33          }
34      }
35  
36      public ActivatedManaAbilityImpl(final ActivatedManaAbilityImpl ability) {
37          super(ability);
38          this.netMana.addAll(ability.netMana);
39          this.undoPossible = ability.undoPossible;
40          this.poolDependant = ability.poolDependant;
41      }
42  
43      @Override
44      public ActivationStatus canActivate(UUID playerId, Game game) {
45          // check if player is in the process of playing spell costs and they are no longer allowed to use
46          // activated mana abilities (e.g. because they started to use improvise or convoke)
47          if (!game.getStack().isEmpty()) {
48              StackObject stackObject = game.getStack().getFirst();
49              if (stackObject instanceof Spell) {
50                  switch (((Spell) stackObject).getCurrentActivatingManaAbilitiesStep()) {
51                      case BEFORE:
52                      case NORMAL:
53                          break;
54                      case AFTER:
55                          return ActivationStatus.getFalse();
56                  }
57              }
58          }
59  
60          return super.canActivate(playerId, game);
61      }
62  
63      /**
64       * Used to check the possible mana production to determine which spells
65       * and/or abilities can be used. (player.getPlayable()).
66       * <p>
67       * Also used for deck's card mana cycle with game = null (score system, etc)
68       *
69       * @param game
70       * @return
71       */
72      @Override
73      public List<Mana> getNetMana(Game game) {
74          if (netMana.isEmpty() || (game != null && game.inCheckPlayableState())) {
75              List<Mana> dynamicNetMana = new ArrayList<>();
76              for (Effect effect : getEffects()) {
77                  if (effect instanceof ManaEffect) {
78                      List<Mana> effectNetMana = ((ManaEffect) effect).getNetMana(game, this);
79                      if (effectNetMana != null) {
80                          dynamicNetMana.addAll(effectNetMana);
81                      }
82                  }
83              }
84              return dynamicNetMana;
85          }
86          List<Mana> netManaCopy = new ArrayList<>();
87          for (Mana mana : netMana) {
88              netManaCopy.add(mana.copy());
89          }
90          return netManaCopy;
91      }
92  
93      @Override
94      public List<Mana> getNetMana(Game game, Mana possibleManaInPool) {
95          if (isPoolDependant()) {
96              List<Mana> poolDependantNetMana = new ArrayList<>();
97              for (Effect effect : getEffects()) {
98                  if (effect instanceof ManaEffect) {
99                      List<Mana> effectNetMana = ((ManaEffect) effect).getNetMana(game, possibleManaInPool, this);
100                     if (effectNetMana != null) {
101                         poolDependantNetMana.addAll(effectNetMana);
102                     }
103                 }
104             }
105             return poolDependantNetMana;            
106         }
107         return getNetMana(game);
108     }
109 
110     @Override
111     public Set<ManaType> getProducableManaTypes(Game game) {
112         Set<ManaType> manaTypes = new HashSet<>();
113         for (Effect effect : getEffects()) {
114             if (effect instanceof ManaEffect) {
115                 manaTypes.addAll(((ManaEffect) effect).getProducableManaTypes(game, this));
116             }
117         }
118         return manaTypes;
119     }
120 
121     /**
122      * Used to check if the ability itself defines mana types it can produce.
123      *
124      * @return
125      */
126     @Override
127     public boolean definesMana(Game game) {
128         return !getNetMana(game).isEmpty();
129     }
130 
131     /**
132      * Is it allowed to undo the mana creation. It's e.g. not allowed if some
133      * game revealing information is related (like reveal the top card of the
134      * library)
135      * <p>
136      * TODO: it helps with single mana activate for mana pool, but will not work
137      * while activates on paying for casting (e.g. user can cheats to see next
138      * draw card)
139      *
140      * @return
141      */
142     public boolean isUndoPossible() {
143         return undoPossible;
144     }
145 
146     public void setUndoPossible(boolean undoPossible) {
147         this.undoPossible = undoPossible;
148     }
149 
150     @Override
151     public boolean isPoolDependant() {
152         return poolDependant;
153     }
154 
155     @Override
156     public ActivatedManaAbilityImpl setPoolDependant(boolean poolDependant) {
157         this.poolDependant = poolDependant;
158         return this;
159     }
160 
161 }

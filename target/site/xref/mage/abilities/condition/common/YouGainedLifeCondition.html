1   package mage.abilities.condition.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.condition.IntCompareCondition;
5   import mage.constants.ComparisonType;
6   import mage.game.Game;
7   import mage.watchers.common.PlayerGainedLifeWatcher;
8   
9   /**
10   * Created by IGOUDT on 5-4-2017.
11   */
12  public class YouGainedLifeCondition extends IntCompareCondition {
13  
14      public YouGainedLifeCondition(ComparisonType type, int value) {
15          super(type, value);
16      }
17  
18      @Override
19      protected int getInputValue(Game game, Ability source) {
20          PlayerGainedLifeWatcher watcher = game.getState().getWatcher(PlayerGainedLifeWatcher.class);
21          return watcher == null ? 0 : watcher.getLifeGained(source.getControllerId());
22      }
23  
24      @Override
25      public String toString() {
26          return "if you gained " + (value == 0 ? "" : (value + 1) + " or more ") + "life this turn";
27      }
28  }

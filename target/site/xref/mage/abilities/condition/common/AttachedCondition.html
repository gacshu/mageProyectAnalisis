3   import mage.abilities.Ability;
4   import mage.abilities.condition.Condition;
5   import mage.game.Game;
6   import mage.game.permanent.Permanent;

8   /**
9    *
10   * @author Quercitron
11   */
12  public enum AttachedCondition implements Condition {
13  
14     instance;
15  
16      @Override
17      public boolean apply(Game game, Ability source) {
18          Permanent attachment = game.getPermanent(source.getSourceId());
19          if (attachment == null || attachment.getAttachedTo() == null) {
20              return false;
21          }
22          Permanent attachedTo = game.getPermanent(attachment.getAttachedTo());
23          return (attachedTo != null);
24      }
25  }

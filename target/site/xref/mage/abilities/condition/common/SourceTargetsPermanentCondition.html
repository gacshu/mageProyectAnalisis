1   package mage.abilities.condition.common;

3   import mage.abilities.Ability;
4   import mage.abilities.condition.Condition;
5   import mage.filter.FilterPermanent;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   import mage.game.stack.StackObject;
9   import mage.target.Target;
10  
11  import java.util.Iterator;
12  
13  /**
14   * @author TheElk801
15   */
16  public class SourceTargetsPermanentCondition implements Condition {
17  
18      private final FilterPermanent filter;
19  
20      public SourceTargetsPermanentCondition(FilterPermanent filter) {
21          this.filter = filter;
22      }
23  
24      @Override
25      public boolean apply(Game game, Ability source) {
26          StackObject sourceSpell = game.getStack().getStackObject(source.getSourceId());
27          if (sourceSpell == null) {
28              return false;
29          }
30          Iterator<Target> targets = sourceSpell.getStackAbility().getTargets().iterator();
31          while (targets.hasNext()) {
32              Permanent permanent = game.getPermanentOrLKIBattlefield(targets.next().getFirstTarget());
33              if (filter.match(permanent, source.getControllerId(), source, game)) {
34                  return true;
35              }
36          }
37          return false;
38      }
39  
40      @Override
41      public String toString() {
42          return "it targets " + filter.getMessage();
43      }
44  
45  }

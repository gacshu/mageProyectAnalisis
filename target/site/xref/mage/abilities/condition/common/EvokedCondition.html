3   package mage.abilities.condition.common;
4   
5   import mage.abilities.Ability;
6   import mage.abilities.condition.Condition;
7   import mage.abilities.keyword.EvokeAbility;
8   import mage.cards.Card;
9   import mage.game.Game;
10  
11  /**
12   *  Checks if a the spell was cast with the alternate evoke costs
13   *
14   * @author LevelX2
15   */
16  
17  public enum EvokedCondition implements Condition {
18  
19      instance;
20  
21      @Override
22      public boolean apply(Game game, Ability source) {
23          Card card = game.getCard(source.getSourceId());
24          if (card != null) {
25              return card.getAbilities(game).stream()
26                      .filter(EvokeAbility.class::isInstance)
27                      .anyMatch(evoke -> ((EvokeAbility) evoke).isActivated(source, game));
28          }
29          return false;
30      }
31  }

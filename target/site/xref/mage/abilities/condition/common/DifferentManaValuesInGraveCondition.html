1   package mage.abilities.condition.common;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.condition.Condition;
6   import mage.game.Game;
7   import mage.players.Player;
8   import mage.util.CardUtil;
9   
10  /**
11   * @author TheElk801
12   */
13  public enum DifferentManaValuesInGraveCondition implements Condition {
14      FIVE(5);
15      private final int amount;
16  
17      DifferentManaValuesInGraveCondition(int amount) {
18          this.amount = amount;
19      }
20  
21      @Override
22      public boolean apply(Game game, Ability source) {
23          Player player = game.getPlayer(source.getControllerId());
24          return player != null
25                  && player
26                  .getGraveyard()
27                  .getCards(game)
28                  .stream()
29                  .mapToInt(MageObject::getManaValue)
30                  .distinct()
31                  .count() >= amount;
32      }
33  
34      @Override
35      public String toString() {
36          return "there are " + CardUtil.numberToText(amount) + " or more mana values among cards in your graveyard";
37      }
38  }

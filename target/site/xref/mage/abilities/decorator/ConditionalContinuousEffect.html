15  /**
16   * Adds condition to {@link ContinuousEffect}. Acts as decorator.
17   *
18   * @author nantuko
19   */
20  public class ConditionalContinuousEffect extends ContinuousEffectImpl {
21  
22      protected ContinuousEffect effect;
23      protected ContinuousEffect otherwiseEffect;
24      protected Condition baseCondition;
25      protected Condition condition;
26      protected boolean initDone = false;
27  
28      public ConditionalContinuousEffect(ContinuousEffect effect, Condition condition, String text) {
29          this(effect, null, condition, text);
30      }
31  
32      /**
33       * Only use this if both effects have the same layers (TODO: add tests to search it?)
34       *
35       * @param effect
36       * @param otherwiseEffect
37       * @param condition
38       * @param text
39       */
40      public ConditionalContinuousEffect(ContinuousEffect effect, ContinuousEffect otherwiseEffect, Condition condition, String text) {
41          super(effect.getDuration(), effect.getLayer(), effect.getSublayer(), effect.getOutcome());
42          this.effect = effect;
43          this.otherwiseEffect = otherwiseEffect;
44          this.baseCondition = condition;
45          this.staticText = text;
46  
47          // checks for compatibility
48          EffectType needType = EffectType.CONTINUOUS;
49          if (effect.getEffectType() != needType) {
50              throw new IllegalArgumentException("ConditionalContinuousEffect supports only " + needType + " but found " + effect.getEffectType().toString());
51          }
52          if (otherwiseEffect != null && otherwiseEffect.getEffectType() != needType) {
53              throw new IllegalArgumentException("ConditionalContinuousEffect supports only " + needType.toString() + " but found " + effect.getEffectType().toString());
54          }
55          if (otherwiseEffect != null && effect.getEffectType() != otherwiseEffect.getEffectType()) {
56              throw new IllegalArgumentException("ConditionalContinuousEffect must be same but found " + effect.getEffectType().toString() + " and " + otherwiseEffect.getEffectType().toString());
57          }
58      }
59  
60      public ConditionalContinuousEffect(final ConditionalContinuousEffect effect) {
61          super(effect);
62          this.effect = effect.effect.copy();
63          if (effect.otherwiseEffect != null) {
64              this.otherwiseEffect = effect.otherwiseEffect.copy();
65          }
66          this.condition = effect.condition;
67          this.baseCondition = effect.baseCondition;
68          this.initDone = effect.initDone;
69      }
70  
71      @Override
72      public boolean isDiscarded() {
73          return this.discarded || effect.isDiscarded() || (otherwiseEffect != null && otherwiseEffect.isDiscarded());
74      }
75  
76      @Override
77      public void init(Ability source, Game game) {
78          super.init(source, game);
79          if (baseCondition instanceof LockedInCondition) {
80              condition = new FixedCondition(((LockedInCondition) baseCondition).getBaseCondition().apply(game, source));
81          } else {
82              condition = baseCondition;
83          }
84          effect.setTargetPointer(this.targetPointer);
85          effect.init(source, game);
86          if (otherwiseEffect != null) {
87              otherwiseEffect.setTargetPointer(this.targetPointer);
88              otherwiseEffect.init(source, game);
89          }
90          initDone = true;
91      }
92  
93      @Override
94      public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
95          if (!initDone) { // if simpleStaticAbility, init won't be called
96              init(source, game);
97          }
98          boolean conditionState = condition.apply(game, source);
99          if (conditionState) {
100             return effect.apply(layer, sublayer, source, game);
101         } else if (otherwiseEffect != null) {
102             return otherwiseEffect.apply(layer, sublayer, source, game);
103         }
104         if (!conditionState && effect.getDuration() == Duration.OneUse) {
105             used = true;
106         }
107         switch (effect.getDuration()) {
108             case OneUse:
109                 used = true;
110                 break;
111             case Custom:
112             case WhileControlled:
113                 this.discard();
114         }
115         return false;
116     }
117 
118     @Override
119     public boolean apply(Game game, Ability source) {
120         if (condition == null && baseCondition != null) {
121             condition = baseCondition;
122         }
123         boolean conditionState = condition != null && condition.apply(game, source);
124         if (conditionState) {
125             effect.setTargetPointer(this.targetPointer);
126             return effect.apply(game, source);
127         } else if (otherwiseEffect != null) {
128             otherwiseEffect.setTargetPointer(this.targetPointer);
129             return otherwiseEffect.apply(game, source);
130         }
131         switch (effect.getDuration()) {
132             case OneUse:
133                 used = true;
134                 break;
135             case Custom:
136             case WhileControlled:
137                 this.discard();
138         }
139         return false;
140     }
141 
142     @Override
143     public String getText(Mode mode) {
144         if ((staticText == null || staticText.isEmpty()) && this.effect != null) { // usefull for conditional night/day card abilities
145             return effect.getText(mode);
146         }
147         return staticText;
148     }
149 
150     @Override
151     public boolean hasLayer(Layer layer) {
152         return effect.hasLayer(layer);
153     }
154 
155     @Override
156     public ConditionalContinuousEffect copy() {
157         return new ConditionalContinuousEffect(this);
158     }
159 
160     @Override
161     public EnumSet<DependencyType> getDependencyTypes() {
162         if (effect != null) {
163             return effect.getDependencyTypes();
164         }
165         return super.getDependencyTypes();
166     }
167 
168     @Override
169     public EnumSet<DependencyType> getDependedToTypes() {
170         if (effect != null) {
171             return effect.getDependedToTypes();
172         }
173         return super.getDependedToTypes();
174     }
175 
176     @Override
177     public Set<UUID> isDependentTo(List<ContinuousEffect> allEffectsInLayer) {
178         if (effect != null) {
179             return effect.isDependentTo(allEffectsInLayer);
180         }
181         return super.isDependentTo(allEffectsInLayer);
182     }
183 
184     /**
185      * Return all effects list, for tests only
186      *
187      * @return
188      */
189     public List<ContinuousEffect> getAllEffects() {
190         List<ContinuousEffect> res = new ArrayList<>();
191         if (this.effect != null) res.add(this.effect);
192         if (this.otherwiseEffect != null) res.add(this.otherwiseEffect);
193         return res;
194     }
195 
196     @Override
197     public Condition getCondition() {
198         return condition == null ? baseCondition : condition;
199     }
200 }

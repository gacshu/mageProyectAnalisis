167     @Override
168     public void init(Ability source, Game game, UUID activePlayerId) {
169         targetPointer.init(game, source);
170         // 20210112 - 611.2c
171         // 611.2c  If a continuous effect generated by the resolution of a spell or ability modifies the
172         // characteristics or changes the controller of any objects, the set of objects it affects is
173         // determined when that continuous effect begins. After that point, the set won’t change.
174         // (Note that this works differently than a continuous effect from a static ability.)
175         // A continuous effect generated by the resolution of a spell or ability that doesn’t
176         // modify the characteristics or change the controller of any objects modifies the
177         // rules of the game, so it can affect objects that weren’t affected when that continuous
178         // effect began.If a single continuous effect has parts that modify the characteristics or
179         // changes the controller of any objects and other parts that don’t, the set of objects
180         // each part applies to is determined independently.
181         if (AbilityType.STATIC != source.getAbilityType()) {
182             if (layer != null) {
183                 switch (layer) {
184                     case CopyEffects_1:
185                     case ControlChangingEffects_2:
186                     case TextChangingEffects_3:
187                     case TypeChangingEffects_4:
188                     case ColorChangingEffects_5:
189                     case AbilityAddingRemovingEffects_6:
190                     case PTChangingEffects_7:
191                         this.affectedObjectsSet = true;
192                 }
193             } else if (hasLayer(Layer.CopyEffects_1)
194                     || hasLayer(Layer.ControlChangingEffects_2)
195                     || hasLayer(Layer.TextChangingEffects_3)
196                     || hasLayer(Layer.TypeChangingEffects_4)
197                     || hasLayer(Layer.ColorChangingEffects_5)
198                     || hasLayer(Layer.AbilityAddingRemovingEffects_6)
199                     || hasLayer(Layer.PTChangingEffects_7)) {
200                 this.affectedObjectsSet = true;
201             }
202         }
203         setStartingControllerAndTurnNum(game, source.getControllerId(), activePlayerId);
204     }

231     @Override
232     public boolean isInactive(Ability source, Game game) {
233         // YOUR turn checks
234         // until end of turn - must be checked on cleanup step, see rules 514.2
235         // other must checked here (active and leave players), see rules 800.4
236         switch (duration) {
237             case UntilYourNextTurn:
238             case UntilEndOfYourNextTurn:
239             case UntilYourNextEndStep:
240                 break;
241             default:
242                 return false;
243         }
244 
245         // cheat engine put cards without play and calls direct applyEffects with clean -- need to ignore it
246         if (game.getActivePlayerId() == null) {
247             return false;
248         }
249 
250         boolean canDelete;
251         Player player = game.getPlayer(startingControllerId);
252 
253         // discard on start of turn for leaved player
254         // 800.4i When a player leaves the game, any continuous effects with durations that last until that player's next turn
255         // or until a specific point in that turn will last until that turn would have begun.
256         // They neither expire immediately nor last indefinitely.
257         switch (duration) {
258             case UntilYourNextTurn:
259             case UntilEndOfYourNextTurn:
260                 canDelete = player == null || (!player.isInGame() && player.hasReachedNextTurnAfterLeaving());
261                 break;
262             default:
263                 canDelete = false;
264         }
265 
266         if (canDelete) {
267             return true;
268         }
269 
270         // discard on another conditions (start of your turn)
271         switch (duration) {
272             case UntilYourNextTurn:
273                 if (player != null && player.isInGame()) {
274                     return this.isYourNextTurn(game);
275                 }
276                 break;
277             case UntilYourNextEndStep:
278                 if (player != null && player.isInGame()) {
279                     return this.isYourNextEndStep(game);
280                 }
281         }
282 
283         return canDelete;
284     }

325     @Override
326     public Set<UUID> isDependentTo(List<ContinuousEffect> allEffectsInLayer) {
327         Set<UUID> dependentToEffects = new HashSet<>();
328         if (dependendToTypes != null) {
329             for (ContinuousEffect effect : allEffectsInLayer) {
330                 if (!effect.getId().equals(this.getId())) {
331                     for (DependencyType dependencyType : effect.getDependencyTypes()) {
332                         if (dependendToTypes.contains(dependencyType)) {
333                             dependentToEffects.add(effect.getId());
334                             break;
335                         }
336                     }
337                 }
338             }
339         }
340         return dependentToEffects;
341     }

396     private void generateGainAbilityDependenciesFromAbility(CompoundAbility compoundAbility) {
397         if (compoundAbility == null) {
398             return;
399         }
400         for (Ability ability : compoundAbility) {
401             generateGainAbilityDependenciesFromAbility(ability);
402         }
403     }
404 
405     private void generateGainAbilityDependenciesFromAbility(Ability ability) {
406         if (ability == null) {
407             return;
408         }
409 
410         // 1. "Is all type" ability (changeling)
411         // make dependency
412         if (ability instanceof ChangelingAbility) {
413             this.addDependencyType(DependencyType.AddingCreatureType);
414         }
415     }
416 
417     private void generateGainAbilityDependenciesFromFilter(Filter filter) {
418         if (filter == null) {
419             return;
420         }
421 
422         // 1. "Is all type" ability (changeling)
423         // wait dependency
424         // extraPredicates from some filters is player related, you don't need it here
425         List<Predicate> list = new ArrayList<>();
426         Predicates.collectAllComponents(filter.getPredicates(), list);
427         if (list.stream().anyMatch(SubType.SubTypePredicate.class::isInstance)) {
428             this.addDependedToType(DependencyType.AddingCreatureType);
429         }
430     }

432     public boolean canLookAtNextTopLibraryCard(Game game) {
433         // If the top card of your library changes while you’re casting a spell, playing a land, or activating an ability,
434         // you can’t look at the new top card until you finish doing so. This means that if you cast the top card of
435         // your library, you can’t look at the next one until you’re done paying for that spell. (2019-05-03)
436         if (!game.getStack().isEmpty()) {
437             StackObject stackObject = game.getStack().getFirst();
438             return !(stackObject instanceof Spell)
439                     || !Zone.LIBRARY.equals(((Spell) stackObject).getFromZone())
440                     || stackObject.getStackAbility().getManaCostsToPay().isPaid(); // mana payment finished
441         }
442         return true;
443     }

1   package mage.abilities.effects;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.common.CreateTokenCopyTargetEffect;
5   import mage.constants.Outcome;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   import mage.target.targetpointer.FixedTarget;
9   
10  /**
11   * Created by glerman on 20/6/15.
12   */
13  public class CreateTokenCopySourceEffect extends OneShotEffect {
14  
15      private final int number;
16      private final boolean tapped;
17  
18      public CreateTokenCopySourceEffect() {
19          this(1);
20      }
21  
22      public CreateTokenCopySourceEffect(int copies) {
23          this(copies, false);
24      }
25  
26      public CreateTokenCopySourceEffect(int copies, boolean tapped) {
27          super(Outcome.PutCreatureInPlay);
28          this.number = copies;
29          this.tapped = tapped;
30          staticText = "create a " + (tapped ? "tapped " : "") + "token that's a copy of {this}";
31      }
32  
33      public CreateTokenCopySourceEffect(final CreateTokenCopySourceEffect effect) {
34          super(effect);
35          this.number = effect.number;
36          this.tapped = effect.tapped;
37      }
38  
39      @Override
40      public boolean apply(Game game, Ability source) {
41          Permanent permanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
42          if (permanent == null) {
43              return false;
44          }
45          CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect(
46                  source.getControllerId(), null, false, number, tapped, false
47          );
48          effect.setTargetPointer(new FixedTarget(source.getSourceId(), game));
49          return effect.apply(game, source);
50      }
51  
52      @Override
53      public CreateTokenCopySourceEffect copy() {
54          return new CreateTokenCopySourceEffect(this);
55      }
56  }

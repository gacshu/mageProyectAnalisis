1   package mage.abilities.effects.mana;
2   
3   import mage.ConditionalMana;
4   import mage.Mana;
5   import mage.abilities.Ability;
6   import mage.abilities.mana.builder.ConditionalManaBuilder;
7   import mage.choices.ManaChoice;
8   import mage.game.Game;
9   import mage.players.Player;
10  
11  import java.util.ArrayList;
12  import java.util.List;
13  
14  /**
15   * @author jeffwadsworth
16   */
17  public class AddConditionalManaOfTwoDifferentColorsEffect extends ManaEffect {
18  
19      private final ConditionalManaBuilder manaBuilder;
20  
21      public AddConditionalManaOfTwoDifferentColorsEffect(ConditionalManaBuilder manaBuilder) {
22          super();
23          this.manaBuilder = manaBuilder;
24          staticText = "Add two mana of different colors. " + manaBuilder.getRule();
25      }
26  
27      private AddConditionalManaOfTwoDifferentColorsEffect(final AddConditionalManaOfTwoDifferentColorsEffect effect) {
28          super(effect);
29          this.manaBuilder = effect.manaBuilder;
30      }
31  
32      @Override
33      public List<Mana> getNetMana(Game game, Ability source) {
34          List<Mana> netMana = new ArrayList<>();
35          netMana.add(Mana.AnyMana(2));
36          return netMana;
37      }
38  
39      @Override
40      public Mana produceMana(Game game, Ability source) {
41          if (game != null) {
42              Player player = getPlayer(game, source);
43              Mana mana = new ConditionalMana(manaBuilder.setMana(
44                      ManaChoice.chooseTwoDifferentColors(
45                              player, game), source, game).build());
46              return mana;
47          }
48          return new Mana();
49      }
50  
51      @Override
52      public AddConditionalManaOfTwoDifferentColorsEffect copy() {
53          return new AddConditionalManaOfTwoDifferentColorsEffect(this);
54      }
55  }

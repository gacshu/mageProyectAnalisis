24      // the effectAbilityMap holds for each effect all abilities that are connected (used) with this effect
25      private final Map<UUID, Set<Ability>> effectAbilityMap = new HashMap<>();
26  
27      public ContinuousEffectsList() {
28      }
29  
30      public ContinuousEffectsList(final ContinuousEffectsList<T> effects) {
31          this.ensureCapacity(effects.size());
32          for (ContinuousEffect cost : effects) {
33              this.add((T) cost.copy());
34          }
35          for (Map.Entry<UUID, Set<Ability>> entry : effects.effectAbilityMap.entrySet()) {
36              Set<Ability> newSet = new HashSet<>();
37              for (Ability ability : entry.getValue()) {
38                  newSet.add(ability.copy());
39              }
40              effectAbilityMap.put(entry.getKey(), newSet);
41          }
42      }
43  
44      public ContinuousEffectsList<T> copy() {
45          return new ContinuousEffectsList<>(this);
46      }

48      public void removeEndOfTurnEffects(Game game) {
49          // calls every turn on cleanup step (only end of turn duration)
50          // rules 514.2
51          for (Iterator<T> i = this.iterator(); i.hasNext(); ) {
52              T entry = i.next();
53              boolean canRemove;
54              switch (entry.getDuration()) {
55                  case EndOfTurn:
56                      canRemove = true;
57                      break;
58                  case UntilEndOfYourNextTurn:
59                      canRemove = entry.isYourNextTurn(game);
60                      break;
61                  case UntilYourNextEndStep:
62                      canRemove = entry.isYourNextEndStep(game);
63                      break;
64                  default:
65                      canRemove = false;
66              }
67              if (canRemove) {
68                  i.remove();
69                  effectAbilityMap.remove(entry.getId());
70              }
71          }
72      }
73  
74      public void removeEndOfCombatEffects() {
75          for (Iterator<T> i = this.iterator(); i.hasNext(); ) {
76              T entry = i.next();
77              if (entry.getDuration() == Duration.EndOfCombat) {
78                  i.remove();
79                  effectAbilityMap.remove(entry.getId());
80              }
81          }
82      }

84      public void removeInactiveEffects(Game game) {
85          for (Iterator<T> i = this.iterator(); i.hasNext(); ) {
86              T entry = i.next();
87              if (isInactive(entry, game)) {
88                  i.remove();
89                  effectAbilityMap.remove(entry.getId());
90              }
91          }
92      }

94      private boolean isInactive(T effect, Game game) {
95          // ends all inactive effects -- calls on player leave or apply new effect
96          if (game.getState().isGameOver()) {
97              // no need to remove effects after end -- users and tests must see last game state
98              return false;
99          }
100 
101         /*
102         800.4a  When a player leaves the game, all objects (see rule 109) owned by that player leave the game and any effects
103         which give that player control of any objects or players end. Then, if that player controlled any objects on the stack
104         not represented by cards, those objects cease to exist. Then, if there are any objects still controlled by that player,
105         those objects are exiled. This is not a state-based action. It happens as soon as the player leaves the game.
106         If the player who left the game had priority at the time they left, priority passes to the next player in turn
107         order who’s still in the game.
108          */
109         // objects removes doing in player.leave() call... effects removes is here
110         Set<Ability> set = effectAbilityMap.get(effect.getId());
111         if (set == null) {
112             logger.debug("No abilities for effect found: " + effect.toString());
113             return false;
114         }
115         Iterator<Ability> it = set.iterator();
116         while (it.hasNext()) {
117             Ability ability = it.next();
118             if (ability == null) {
119                 it.remove();
120             } else if (ability instanceof MageSingleton) {
121                 return false;
122             } else if (effect.isDiscarded()) {
123                 it.remove();
124             } else {
125                 // 800.4k  When a player leaves the game, any continuous effects with durations that last until that
126                 // player’s next turn or until a specific point in that turn will last until that turn would have begun.
127                 // They neither expire immediately nor last indefinitely.
128                 MageObject object = game.getObject(ability.getSourceId());
129                 boolean isObjectInGame = ability.getSourceId() == null || object != null; // Commander effects have no sourceId
130                 boolean hasOwnerLeftGame = false;
131                 if (object instanceof Card) {
132                     Player owner = game.getPlayer(((Card) object).getOwnerId());
133                     hasOwnerLeftGame = !owner.isInGame();
134                 }
135 
136                 switch (effect.getDuration()) {
137                     //
138                     case WhileOnBattlefield:
139                     case WhileInGraveyard:
140                     case WhileOnStack:
141                     case EndOfStep:
142                     case EndOfCombat:
143                     case EndOfGame:
144                         // if the related source object does no longer exist in game - the effect has to be removed
145                         if (hasOwnerLeftGame || !isObjectInGame) {
146                             it.remove();
147                         }
148                         break;
149                     case OneUse:
150                         if (hasOwnerLeftGame || effect.isUsed()) {
151                             it.remove();
152                         }
153                         break;
154                     case Custom:
155                     case UntilYourNextTurn:
156                     case UntilEndOfYourNextTurn:
157                     case UntilYourNextEndStep:
158                         // until your turn effects continue until real turn reached, their used it's own inactive method
159                         // 514.2 Second, the following actions happen simultaneously: all damage marked on permanents
160                         // (including phased-out permanents) is removed and all "until end of turn" and "this turn" effects end.
161                         // This turn-based action doesn’t use the stack.
162                         // custom effects must process it's own inactive method (override)
163                         // custom effects may not end, if the source permanent of the effect has left the game
164                         // 800.4a (only any effects which give that player control of any objects or players end)
165                         if (effect.isInactive(ability, game)) {
166                             it.remove();
167                         }
168                         break;
169                     case EndOfTurn:
170                         // end of turn discards on cleanup steps
171                         // 514.2
172                         break;
173                     case UntilSourceLeavesBattlefield:
174                         if (hasOwnerLeftGame || game.getState().getZone(ability.getSourceId()) != Zone.BATTLEFIELD) {
175                             it.remove();
176                         }
177                         break;
178                     case WhileControlled:
179                         Permanent permanent = ability.getSourcePermanentIfItStillExists(game);
180                         if (hasOwnerLeftGame || permanent == null || !permanent.isControlledBy(ability.getControllerId())) {
181                             it.remove();
182                         }
183                         break;
184                     default:
185                         throw new IllegalStateException("Effects gets unknown duration " + effect.getDuration() + ", effect: " + effect.toString());
186                 }
187             }
188         }
189         return set.isEmpty();
190     }

192     /**
193      * Adds an effect and its connected ability to the list. For each effect
194      * will be stored, which abilities are connected to the effect. So an effect
195      * can be connected to multiple abilities.
196      *
197      * @param effect - effect to add
198      * @param source - connected ability
199      */
200     public void addEffect(T effect, Ability source) {
201         Set<Ability> set = effectAbilityMap.computeIfAbsent(effect.getId(), x -> new HashSet<>());
202         if (set.stream()
203                 .filter(ability -> ability.getSourceId().equals(source.getSourceId()))
204                 .map(Ability::getId)
205                 .anyMatch(source.getId()::equals)) {
206             return;
207         }
208         set.add(source);
209         this.add(effect);
210     }

212     public Set<Ability> getAbility(UUID effectId) {
213         return effectAbilityMap.computeIfAbsent(effectId, x -> new HashSet<>());
214     }

216     public void removeTemporaryEffects() {
217         for (Iterator<T> i = this.iterator(); i.hasNext(); ) {
218             T entry = i.next();
219             if (entry.isTemporary()) {
220                 i.remove();
221                 effectAbilityMap.remove(entry.getId());
222             }
223         }
224     }

226     @Override
227     public void clear() {
228         super.clear();
229         effectAbilityMap.clear();
230     }
231 
232     @Override
233     public boolean contains(Object object) {
234         if (!(object instanceof ContinuousEffect)) {
235             return false;
236         }
237 
238         // search by id
239         ContinuousEffect need = (ContinuousEffect) object;
240         for (Iterator<T> iterator = this.iterator(); iterator.hasNext(); ) {
241             T test = iterator.next();
242             if (need.equals(test)) {
243                 return true;
244             }
245             if (need.getId().equals(test.getId())) {
246                 return true;
247             }
248         }
249         return false;
250     }

11  public abstract class WhileConditionContinuousEffect extends ContinuousEffectImpl {
12      protected Condition condition;
13  
14      public WhileConditionContinuousEffect(Duration duration, Layer layer, SubLayer sublayer, Condition condition, Outcome outcome) {
15          super(duration, outcome);
16          this.condition = condition;
17          this.layer = layer;
18          this.sublayer = sublayer;
19      }
20  
21      public WhileConditionContinuousEffect(final WhileConditionContinuousEffect effect) {
22          super(effect);
23          this.condition = effect.condition;
24      }
25  
26      @Override
27      public boolean apply(Game game, Ability source) {
28          if (condition.apply(game, source)) {
29              return applyEffect(game, source);
30          }
31          return false;
32      }
33  
34      protected abstract boolean applyEffect(Game game, Ability source);
35  }

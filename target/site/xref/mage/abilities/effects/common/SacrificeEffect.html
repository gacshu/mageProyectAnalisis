22  public class SacrificeEffect extends OneShotEffect {
23  
24      private FilterPermanent filter;
25      private String preText;
26      private DynamicValue count;
27  
28      public SacrificeEffect(FilterPermanent filter, int count, String preText) {
29          this(filter, StaticValue.get(count), preText);
30      }
31  
32      public SacrificeEffect(FilterPermanent filter, DynamicValue count, String preText) {
33          super(Outcome.Sacrifice);
34          this.filter = filter;
35          this.count = count;
36          this.preText = preText;
37          setText();
38      }
39  
40      public SacrificeEffect(final SacrificeEffect effect) {
41          super(effect);
42          this.filter = effect.filter;
43          this.count = effect.count;
44          this.preText = effect.preText;
45      }
46  
47      @Override
48      public boolean apply(Game game, Ability source) {
49          boolean applied = false;
50          for (UUID playerId : targetPointer.getTargets(game, source)) {
51              Player player = game.getPlayer(playerId);
52              if (player != null) {
53                  FilterPermanent newFilter = filter.copy(); // filter can be static, so it's important to copy here
54                  newFilter.add(new ControllerIdPredicate(player.getId()));
55                  int amount = count.calculate(game, source, this);
56                  int realCount = game.getBattlefield().countAll(newFilter, player.getId(), game);
57                  amount = Math.min(amount, realCount);
58                  Target target = new TargetPermanent(amount, amount, newFilter, true);
59                  if (amount > 0 && target.canChoose(player.getId(), source, game)) {
60                      while (!target.isChosen()
61                              && target.canChoose(player.getId(), source, game)
62                              && player.canRespond()) {
63                          player.chooseTarget(Outcome.Sacrifice, target, source, game);
64                      }
65                      for (int idx = 0; idx < target.getTargets().size(); idx++) {
66                          Permanent permanent = game.getPermanent(target.getTargets().get(idx));
67                          if (permanent != null
68                                  && permanent.sacrifice(source, game)) {
69                              applied = true;
70                          }
71                      }
72                  }
73              }
74          }
75          return applied;
76      }
77  
78      public void setAmount(DynamicValue amount) {
79          this.count = amount;
80      }
81  
82      @Override
83      public SacrificeEffect copy() {
84          return new SacrificeEffect(this);
85      }
86  
87      private void setText() {
88          StringBuilder sb = new StringBuilder();
89          if (preText != null) {
90              sb.append(preText);
91          }
92          if (preText != null && (preText.endsWith("player") || preText.endsWith("opponent"))) {
93              sb.append(" sacrifices ");
94          } else {
95              if (preText == null || preText.isEmpty()) {
96                  sb.append("sacrifice ");
97              } else {
98                  sb.append(" sacrifice ");
99              }
100         }
101         if (count.toString().equals("1")) {
102             sb.append(CardUtil.addArticle(filter.getMessage()));
103         } else {
104             sb.append(CardUtil.numberToText(count.toString(), "a"));
105             sb.append(filter.getMessage());
106         }
107         staticText = sb.toString();
108     }
109 }

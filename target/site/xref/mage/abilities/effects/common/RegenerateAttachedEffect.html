1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.OneShotEffect;
5   import mage.constants.AttachmentType;
6   import mage.constants.Outcome;
7   import mage.game.Game;
8   import mage.game.permanent.Permanent;
9   import mage.target.targetpointer.FixedTarget;
10  
11  import java.util.Locale;
12  
13  /**
14   * @author jeff
15   */
16  public class RegenerateAttachedEffect extends OneShotEffect {
17  
18      protected AttachmentType attachmentType;
19  
20      public RegenerateAttachedEffect(AttachmentType attachmentType) {
21          super(Outcome.Regenerate);
22          this.attachmentType = attachmentType;
23          this.setText();
24      }
25  
26      public RegenerateAttachedEffect(final RegenerateAttachedEffect effect) {
27          super(effect);
28          this.attachmentType = effect.attachmentType;
29      }
30  
31      @Override
32      public boolean apply(Game game, Ability source) {
33          // must use lki cause attachment can be sacrificed to activate regen
34          Permanent attachment = game.getPermanentOrLKIBattlefield(source.getSourceId());
35          if (attachment == null) {
36              return false;
37          }
38          Permanent permanent = game.getPermanent(attachment.getAttachedTo());
39          if (permanent == null) {
40              return false;
41          }
42  
43          RegenerateTargetEffect regenEffect = new RegenerateTargetEffect();
44          regenEffect.setTargetPointer(new FixedTarget(permanent, game));
45          game.addEffect(regenEffect, source);
46          return true;
47      }
48  
49      @Override
50      public RegenerateAttachedEffect copy() {
51          return new RegenerateAttachedEffect(this);
52      }
53  
54      private void setText() {
55          staticText = "Regenerate " + attachmentType.verb().toLowerCase(Locale.ENGLISH) + " creature";
56      }
57  }

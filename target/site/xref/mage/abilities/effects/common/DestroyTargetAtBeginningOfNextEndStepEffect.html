2   package mage.abilities.effects.common;
3   
4   import mage.abilities.Ability;
5   import mage.abilities.common.delayed.AtTheBeginOfNextEndStepDelayedTriggeredAbility;
6   import mage.abilities.effects.OneShotEffect;
7   import mage.constants.Outcome;
8   import mage.game.Game;
9   import mage.target.targetpointer.FixedTarget;
10  
11  /**
12   *
13   * @author LoneFox
14   */
15  public class DestroyTargetAtBeginningOfNextEndStepEffect extends OneShotEffect {
16  
17      public DestroyTargetAtBeginningOfNextEndStepEffect() {
18          super(Outcome.Detriment);
19          this.staticText = "Destroy that creature at the beginning of the next end step";
20      }
21  
22      public DestroyTargetAtBeginningOfNextEndStepEffect(final DestroyTargetAtBeginningOfNextEndStepEffect effect) {
23          super(effect);
24      }
25  
26      @Override
27      public DestroyTargetAtBeginningOfNextEndStepEffect copy() {
28          return new DestroyTargetAtBeginningOfNextEndStepEffect(this);
29      }
30  
31      @Override
32      public boolean apply(Game game, Ability source) {
33          DestroyTargetEffect effect = new DestroyTargetEffect();
34          effect.setTargetPointer(new FixedTarget(source.getFirstTarget(), game));
35          AtTheBeginOfNextEndStepDelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(effect);
36          game.addDelayedTriggeredAbility(delayedAbility, source);
37          return true;
38      }
39  }

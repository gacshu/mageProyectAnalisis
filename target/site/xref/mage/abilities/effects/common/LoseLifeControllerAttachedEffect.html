2   package mage.abilities.effects.common;
3   
4   import mage.abilities.Ability;
5   import mage.abilities.dynamicvalue.DynamicValue;
6   import mage.abilities.dynamicvalue.common.StaticValue;
7   import mage.abilities.effects.OneShotEffect;
8   import mage.constants.Outcome;
9   import mage.constants.Zone;
10  import mage.game.Game;
11  import mage.game.permanent.Permanent;
12  import mage.players.Player;
13  
14  
15  public class LoseLifeControllerAttachedEffect extends OneShotEffect {
16  
17      protected DynamicValue amount;
18  
19      public LoseLifeControllerAttachedEffect(int amount) {
20          this(StaticValue.get(amount));
21      }
22  
23      public LoseLifeControllerAttachedEffect(DynamicValue amount) {
24          super(Outcome.LoseLife);
25          this.amount = amount;
26          setText();
27      }
28  
29      public LoseLifeControllerAttachedEffect(final LoseLifeControllerAttachedEffect effect) {
30          super(effect);
31          this.amount = effect.amount.copy();
32      }
33  
34      @Override
35      public LoseLifeControllerAttachedEffect copy() {
36          return new LoseLifeControllerAttachedEffect(this);
37      }
38  
39      @Override
40      public boolean apply(Game game, Ability source) {
41          Permanent enchantment = game.getPermanent(source.getSourceId());
42          if (enchantment == null) {
43              enchantment = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
44          }
45          if (enchantment != null && enchantment.getAttachedTo() != null) {
46              Permanent creature = game.getPermanent(enchantment.getAttachedTo());
47              if (creature == null) {
48                  creature = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
49              }
50              if (creature != null) {
51                  Player player = game.getPlayer(creature.getControllerId());
52                  if (player != null) {
53                      player.loseLife(amount.calculate(game, source, this), game, source, false);
54                      return true;
55                  }
56              }
57          }
58          return false;
59      }
60  
61      private void setText() {
62          StringBuilder sb = new StringBuilder();
63          sb.append("its controller loses ").append(amount.toString()).append(" life");
64          String message = amount.getMessage();
65          if (!message.isEmpty()) {
66              sb.append(" for each ");
67              sb.append(message);
68          }
69          staticText = sb.toString();
70      }
71  }

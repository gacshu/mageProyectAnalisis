29  public class MeldEffect extends OneShotEffect {
30  
31      private final String meldWithName;
32      private final String meldIntoName;
33      private final boolean attacking;
34  
35      public MeldEffect(String meldWithName, String meldIntoName) {
36          this(meldWithName, meldIntoName, false);
37      }
38  
39      public MeldEffect(String meldWithName, String meldIntoName, boolean attacking) {
40          super(Outcome.Benefit);
41          this.meldWithName = meldWithName;
42          this.meldIntoName = meldIntoName;
43          this.attacking = attacking;
44      }
45  
46      public MeldEffect(final MeldEffect effect) {
47          super(effect);
48          this.meldWithName = effect.meldWithName;
49          this.meldIntoName = effect.meldIntoName;
50          this.attacking = effect.attacking;
51      }
52  
53      @Override
54      public MeldEffect copy() {
55          return new MeldEffect(this);
56      }
57  
58      @Override
59      public boolean apply(Game game, Ability source) {
60          Player controller = game.getPlayer(source.getControllerId());
61          Permanent sourcePermanent = source.getSourcePermanentIfItStillExists(game);
62          if (controller == null
63                  || sourcePermanent == null
64                  || !sourcePermanent.isControlledBy(controller.getId())
65                  || (attacking && !sourcePermanent.isAttacking())
66                  || !sourcePermanent.isOwnedBy(controller.getId())) {
67              return false;
68          }
69          // Find the two permanents to meld.
70          FilterPermanent filter = new FilterControlledPermanent("permanent named " + meldWithName);
71          filter.add(new NamePredicate(meldWithName));
72          filter.add(TargetController.YOU.getOwnerPredicate());
73          if (attacking) {
74              filter.add(AttackingPredicate.instance);
75          }
76          if (!game.getBattlefield().contains(filter, source, game, 1)) {
77              return false;
78          }
79          TargetPermanent target = new TargetPermanent(filter);
80          target.setNotTarget(true);
81          controller.choose(outcome, target, source, game);
82  
83          Permanent meldWithPermanent = game.getPermanent(target.getFirstTarget());
84          if (sourcePermanent == null || meldWithPermanent == null) {
85              return false;
86          }
87          Cards cards = new CardsImpl(sourcePermanent);
88          cards.add(meldWithPermanent);
89          controller.moveCards(cards, Zone.EXILED, source, game);
90          // Create the meld card and move it to the battlefield.
91          Card sourceCard = cards.get(sourcePermanent.getId(), game);
92          Card meldWithCard = cards.get(meldWithPermanent.getId(), game);
93          if (sourceCard == null
94                  || meldWithCard == null
95                  || !sourceCard.meldsWith(meldWithCard)
96                  || !meldWithCard.meldsWith(sourceCard)) {
97              return true;
98          }
99          List<CardInfo> cardInfoList = CardRepository.instance.findCards(
100                 new CardCriteria()
101                         .name(meldIntoName)
102                         .setCodes(sourceCard.getExpansionSetCode())
103                         .nightCard(true)
104         );
105         if (cardInfoList.isEmpty()) {
106             return false;
107         }
108         MeldCard meldCard = (MeldCard) cardInfoList.get(0).getCard().copy();
109         meldCard.setOwnerId(controller.getId());
110         meldCard.setTopHalfCard(meldWithCard, game);
111         meldCard.setBottomHalfCard(sourceCard, game);
112         meldCard.setMelded(true, game);
113         game.addMeldCard(meldCard.getId(), meldCard);
114         game.getState().addCard(meldCard);
115         meldCard.setZone(Zone.EXILED, game);
116         controller.moveCards(meldCard, Zone.BATTLEFIELD, source, game, attacking, false, false, null);
117         if (attacking) {
118             game.getCombat().addAttackingCreature(meldCard.getId(), game);
119         }
120         return true;
121     }
122 }

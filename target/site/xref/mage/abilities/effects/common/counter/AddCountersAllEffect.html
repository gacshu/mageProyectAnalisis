1   package mage.abilities.effects.common.counter;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.constants.Outcome;
7   import mage.counters.Counter;
8   import mage.filter.FilterPermanent;
9   import mage.game.Game;
10  import mage.game.permanent.Permanent;
11  import mage.players.Player;
12  
13  /**
14   * @author North
15   */
16  public class AddCountersAllEffect extends OneShotEffect {
17  
18      private final Counter counter;
19      private final FilterPermanent filter;
20  
21      public AddCountersAllEffect(Counter counter, FilterPermanent filter) {
22          super(Outcome.Benefit);
23          this.counter = counter;
24          this.filter = filter;
25          staticText = "put " + counter.getDescription() + " on each " + filter.getMessage();
26      }
27  
28      public AddCountersAllEffect(final AddCountersAllEffect effect) {
29          super(effect);
30          this.counter = effect.counter.copy();
31          this.filter = effect.filter.copy();
32      }
33  
34      @Override
35      public boolean apply(Game game, Ability source) {
36          Player controller = game.getPlayer(source.getControllerId());
37          MageObject sourceObject = game.getObject(source);
38          if (controller != null && sourceObject != null) {
39              if (counter != null) {
40                  for (Permanent permanent : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
41                      permanent.addCounters(counter.copy(), source.getControllerId(), source, game);
42                      if (!game.isSimulation()) {
43                          game.informPlayers(sourceObject.getLogName() + ": " + controller.getLogName() + " puts " + counter.getCount() + ' ' + counter.getName()
44                                  + " counter on " + permanent.getLogName());
45                      }
46                  }
47              }
48              return true;
49          }
50          return false;
51      }
52  
53      @Override
54      public AddCountersAllEffect copy() {
55          return new AddCountersAllEffect(this);
56      }
57  }

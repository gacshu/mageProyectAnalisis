23  public class AddCountersSourceEffect extends OneShotEffect {
24  
25      private Counter counter;
26      private boolean informPlayers;
27      private DynamicValue amount;
28      private boolean putOnCard;
29  
30      public AddCountersSourceEffect(Counter counter) {
31          this(counter, false);
32      }
33  
34      public AddCountersSourceEffect(Counter counter, boolean informPlayers) {
35          this(counter, StaticValue.get(0), informPlayers);
36      }
37  
38      public AddCountersSourceEffect(Counter counter, DynamicValue amount, boolean informPlayers) {
39          this(counter, amount, informPlayers, false);
40      }
41  
42      /**
43       * @param counter
44       * @param amount        this amount will be added to the counter instances
45       * @param informPlayers
46       * @param putOnCard     - counters have to be put on a card instead of a
47       *                      permanent
48       */
49      public AddCountersSourceEffect(Counter counter, DynamicValue amount, boolean informPlayers, boolean putOnCard) {
50          super(Outcome.Benefit);
51          this.counter = counter.copy();
52          this.informPlayers = informPlayers;
53          this.amount = amount;
54          this.putOnCard = putOnCard;
55          staticText = CardUtil.getAddRemoveCountersText(amount, counter, "{this}", true);
56      }
57  
58      public AddCountersSourceEffect(final AddCountersSourceEffect effect) {
59          super(effect);
60          if (effect.counter != null) {
61              this.counter = effect.counter.copy();
62          }
63          this.informPlayers = effect.informPlayers;
64          this.amount = effect.amount;
65          this.putOnCard = effect.putOnCard;
66      }
67  
68      @Override
69      public boolean apply(Game game, Ability source) {
70          if (counter == null) {
71              return false;
72          }
73  
74          Player controller = game.getPlayer(source.getControllerId());
75          if (controller == null) {
76              return false;
77          }
78  
79          if (putOnCard) {
80              Card card = game.getCard(source.getSourceId());
81              if (card == null) {
82                  return false;
83              }
84  
85              Counter newCounter = counter.copy();
86              int countersToAdd = amount.calculate(game, source, this);
87              if (countersToAdd > 0 && newCounter.getCount() == 1) {
88                  countersToAdd--;
89              }
90              newCounter.add(countersToAdd);
91              List<UUID> appliedEffects = (ArrayList<UUID>) this.getValue("appliedEffects");
92              card.addCounters(newCounter, source.getControllerId(), source, game, appliedEffects);
93              if (informPlayers && !game.isSimulation()) {
94                  Player player = game.getPlayer(source.getControllerId());
95                  if (player != null) {
96                      game.informPlayers(player.getLogName() + " puts " + newCounter.getCount() + ' ' + newCounter.getName() + " counter on " + card.getLogName());
97                  }
98              }
99              return true;
100         } else {
101             Permanent permanent = game.getPermanent(source.getSourceId());
102             if (permanent == null && source.getAbilityType() == AbilityType.STATIC) {
103                 permanent = game.getPermanentEntering(source.getSourceId());
104             }
105             if (permanent == null) {
106                 return false;
107             }
108 
109             if ((source.getSourceObjectZoneChangeCounter() == 0 // from static ability
110                     || source.getSourceObjectZoneChangeCounter() == permanent.getZoneChangeCounter(game))) { // prevent to add counters to later source objects
111                 Counter newCounter = counter.copy();
112                 int countersToAdd = amount.calculate(game, source, this);
113                 if (amount instanceof StaticValue || countersToAdd > 0) {
114                     if (countersToAdd > 0 && newCounter.getCount() == 1) {
115                         countersToAdd--;
116                     }
117                     newCounter.add(countersToAdd);
118                     int before = permanent.getCounters(game).getCount(newCounter.getName());
119                     List<UUID> appliedEffects = (ArrayList<UUID>) this.getValue("appliedEffects");
120                     permanent.addCounters(newCounter, source.getControllerId(), source, game, appliedEffects); // if used from a replacement effect, the basic event determines if an effect was already applied to an event
121                     if (informPlayers && !game.isSimulation()) {
122                         int amountAdded = permanent.getCounters(game).getCount(newCounter.getName()) - before;
123                         Player player = game.getPlayer(source.getControllerId());
124                         if (player != null) {
125                             game.informPlayers(player.getLogName() + " puts " + amountAdded + ' ' + newCounter.getName() + " counter on " + permanent.getLogName());
126                         }
127                     }
128                 }
129             }
130         }
131         return true;
132     }
133 
134     @Override
135     public AddCountersSourceEffect copy() {
136         return new AddCountersSourceEffect(this);
137     }
138 }

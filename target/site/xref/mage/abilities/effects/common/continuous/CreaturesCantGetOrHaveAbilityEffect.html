17   *
18   * @author LevelX2

20  public class CreaturesCantGetOrHaveAbilityEffect extends ContinuousEffectImpl {
21  
22      private final Ability ability;
23      private final FilterCreaturePermanent filter;
24  
25      public CreaturesCantGetOrHaveAbilityEffect(Ability ability, Duration duration) {
26          this(ability, duration, StaticFilters.FILTER_OPPONENTS_PERMANENT_CREATURES);
27      }
28  
29      public CreaturesCantGetOrHaveAbilityEffect(Ability ability, Duration duration, FilterCreaturePermanent filter) {
30          super(duration, Outcome.Detriment);
31          this.ability = ability;
32          this.filter = filter;
33          staticText = filter.getMessage() + " lose " + ability.getRule() + " and can't have or gain " + ability.getRule();
34          addDependedToType(DependencyType.AddingAbility);
35      }
36  
37      public CreaturesCantGetOrHaveAbilityEffect(final CreaturesCantGetOrHaveAbilityEffect effect) {
38          super(effect);
39          this.ability = effect.ability;
40          this.filter = effect.filter;
41      }
42  
43      @Override
44      public CreaturesCantGetOrHaveAbilityEffect copy() {
45          return new CreaturesCantGetOrHaveAbilityEffect(this);
46      }
47  
48      @Override
49      public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
50          Player controller = game.getPlayer(source.getControllerId());
51          if (controller != null) {
52              for (Permanent permanent : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
53                  if (permanent != null) {
54                      permanent.removeAbility(ability, source.getSourceId(), game);
55                  }
56              }
57              return true;
58          }
59          return false;
60  
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          return false;
66      }
67  
68      @Override
69      public boolean hasLayer(Layer layer) {
70          return layer == Layer.AbilityAddingRemovingEffects_6;
71      }
72  }

11  import java.util.ArrayList;
12  import java.util.List;
13  import java.util.Locale;
14  
15  /**
16   * @author emerald000
17   */
18  public class AddCardTypeSourceEffect extends ContinuousEffectImpl {
19  
20      private final List<CardType> addedCardTypes = new ArrayList<>();
21  
22      public AddCardTypeSourceEffect(Duration duration, CardType... addedCardType) {
23          super(duration, Layer.TypeChangingEffects_4, SubLayer.NA, Outcome.Benefit);
24          for (CardType cardType : addedCardType) {
25              this.addedCardTypes.add(cardType);
26              if (cardType == CardType.ENCHANTMENT) {
27                  dependencyTypes.add(DependencyType.EnchantmentAddingRemoving);
28              } else if (cardType == CardType.ARTIFACT) {
29                  dependencyTypes.add(DependencyType.ArtifactAddingRemoving);
30              }
31          }
32      }
33  
34      public AddCardTypeSourceEffect(final AddCardTypeSourceEffect effect) {
35          super(effect);
36          this.addedCardTypes.addAll(effect.addedCardTypes);
37      }
38  
39      @Override
40      public void init(Ability source, Game game) {
41          super.init(source, game);
42          affectedObjectList.add(new MageObjectReference(source.getSourceId(), game.getState().getZoneChangeCounter(source.getSourceId()), game));
43      }
44  
45      @Override
46      public boolean apply(Game game, Ability source) {
47          Permanent permanent = game.getPermanent(source.getSourceId());
48          if (permanent != null && affectedObjectList.contains(new MageObjectReference(permanent, game))) {
49              for (CardType cardType : addedCardTypes) {
50                  permanent.addCardType(game, cardType);
51              }
52              return true;
53          } else if (this.getDuration() == Duration.Custom) {
54              this.discard();
55          }
56          return false;
57      }
58  
59      @Override
60      public AddCardTypeSourceEffect copy() {
61          return new AddCardTypeSourceEffect(this);
62      }
63  
64      @Override
65      public String getText(Mode mode) {
66          if (staticText != null && !staticText.isEmpty()) {
67              return staticText;
68          }
69          StringBuilder sb = new StringBuilder();
70          sb.append("{this} becomes ");
71          boolean article = false;
72          for (CardType cardType : addedCardTypes) {
73              if (!article) {
74                  if (cardType.toString().startsWith("A") || cardType.toString().startsWith("E")) {
75                      sb.append("an ");
76                  } else {
77                      sb.append("a ");
78                  }
79                  article = true;
80              }
81              sb.append(cardType.toString().toLowerCase(Locale.ENGLISH)).append(" ");
82          }
83          sb.append("in addition to its other types ").append(this.getDuration().toString());
84          return sb.toString();
85      }
86  }

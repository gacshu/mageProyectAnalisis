3   package mage.abilities.effects.common.continuous;
4   
5   import mage.ObjectColor;
6   import mage.abilities.Ability;
7   import mage.abilities.effects.ContinuousEffectImpl;
8   import mage.constants.*;
9   import mage.game.Game;
10  import mage.game.permanent.Permanent;
11  
12  /**
13   * @author noxx
14   */
15  public class AddCardColorAttachedEffect extends ContinuousEffectImpl {
16  
17      private ObjectColor addedColor;
18      private AttachmentType attachmentType;
19  
20      public AddCardColorAttachedEffect(ObjectColor addedColor, Duration duration, AttachmentType attachmentType) {
21          super(duration, Layer.TypeChangingEffects_4, SubLayer.NA, Outcome.Benefit);
22          this.addedColor = addedColor;
23          this.attachmentType = attachmentType;
24          setText();
25      }
26  
27      public AddCardColorAttachedEffect(final AddCardColorAttachedEffect effect) {
28          super(effect);
29          this.addedColor = effect.addedColor;
30          this.attachmentType = effect.attachmentType;
31      }
32  
33      @Override
34      public boolean apply(Game game, Ability source) {
35          Permanent equipment = game.getPermanent(source.getSourceId());
36          if (equipment != null && equipment.getAttachedTo() != null) {
37              Permanent target = game.getPermanent(equipment.getAttachedTo());
38              if (target != null) {
39                  if (addedColor.isBlack())
40                      target.getColor(game).setBlack(true);
41                  if (addedColor.isBlue())
42                      target.getColor(game).setBlue(true);
43                  if (addedColor.isWhite())
44                      target.getColor(game).setWhite(true);
45                  if (addedColor.isGreen())
46                      target.getColor(game).setGreen(true);
47                  if (addedColor.isRed())
48                      target.getColor(game).setRed(true);
49              }
50          }
51          return true;
52      }
53  
54      @Override
55      public AddCardColorAttachedEffect copy() {
56          return new AddCardColorAttachedEffect(this);
57      }
58  
59      private void setText() {
60          StringBuilder sb = new StringBuilder();
61          sb.append(attachmentType.verb());
62          sb.append(" creature is a ").append(addedColor.getDescription()).append(" in addition to its colors");
63          staticText = sb.toString();
64      }
65  }

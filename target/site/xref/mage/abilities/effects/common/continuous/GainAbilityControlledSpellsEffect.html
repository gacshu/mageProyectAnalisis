14  /**
15   * @author Styxo
16   */
17  public class GainAbilityControlledSpellsEffect extends ContinuousEffectImpl {
18  
19      private final Ability ability;
20      private final FilterCard filter;
21  
22      public GainAbilityControlledSpellsEffect(Ability ability, FilterCard filter) {
23          super(Duration.WhileOnBattlefield, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
24          this.ability = ability;
25          this.filter = filter;
26          staticText = filter.getMessage() + " you cast have " + ability.getRule() + '.';
27      }
28  
29      public GainAbilityControlledSpellsEffect(final GainAbilityControlledSpellsEffect effect) {
30          super(effect);
31          this.ability = effect.ability;
32          this.filter = effect.filter;
33      }
34  
35      @Override
36      public GainAbilityControlledSpellsEffect copy() {
37          return new GainAbilityControlledSpellsEffect(this);
38      }
39  
40      @Override
41      public boolean apply(Game game, Ability source) {
42          Player player = game.getPlayer(source.getControllerId());
43          Permanent permanent = game.getPermanent(source.getSourceId());
44          if (player == null || permanent == null) {
45              return false;
46          }
47  
48          for (Card card : game.getExile().getAllCards(game)) {
49              if (card.isOwnedBy(source.getControllerId())
50                      && filter.match(card, game)) {
51                  game.getState().addOtherAbility(card, ability);
52              }
53          }
54          for (Card card : player.getLibrary().getCards(game)) {
55              if (filter.match(card, game)) {
56                  game.getState().addOtherAbility(card, ability);
57              }
58          }
59          for (Card card : player.getHand().getCards(game)) {
60              if (filter.match(card, game)) {
61                  game.getState().addOtherAbility(card, ability);
62              }
63          }
64          for (Card card : player.getGraveyard().getCards(game)) {
65              if (filter.match(card, game)) {
66                  game.getState().addOtherAbility(card, ability);
67              }
68          }
69  
70          // workaround to gain cost reduction abilities to commanders before cast (make it playable)
71          game.getCommanderCardsFromCommandZone(player, CommanderCardType.ANY)
72                  .stream()
73                  .filter(card -> filter.match(card, game))
74                  .forEach(card -> {
75                      game.getState().addOtherAbility(card, ability);
76                  });
77  
78          for (StackObject stackObject : game.getStack()) {
79              // only spells cast, so no copies of spells
80              if ((stackObject instanceof Spell)
81                      && !stackObject.isCopy()
82                      && stackObject.isControlledBy(source.getControllerId())) {
83                  Card card = game.getCard(stackObject.getSourceId());
84                  if (filter.match(card, game)) {
85                      game.getState().addOtherAbility(card, ability);
86                      return true;
87                  }
88              }
89          }
90          return false; // TODO: Why is this returning false?
91      }
92  }

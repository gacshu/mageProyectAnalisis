1   package mage.abilities.effects.common.continuous;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.Mode;
5   import mage.abilities.effects.ContinuousEffectImpl;
6   import mage.constants.*;
7   import mage.filter.FilterPermanent;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  
11  public class CreaturesBecomeOtherTypeEffect extends ContinuousEffectImpl {
12  
13      protected final FilterPermanent filter;
14      private final SubType subType;
15  
16      public CreaturesBecomeOtherTypeEffect(FilterPermanent filter, SubType subType, Duration duration) {
17          super(duration, Outcome.Neutral);
18          this.filter = filter;
19          this.subType = subType;
20  
21          this.dependendToTypes.add(DependencyType.BecomeCreature);  // Opalescence and Starfield of Nyx
22          this.staticText = this.filter.getMessage() + " is " + this.subType.getIndefiniteArticle()
23                  + " " + this.subType.toString() + " in addition to its other types";
24      }
25  
26      protected CreaturesBecomeOtherTypeEffect(final CreaturesBecomeOtherTypeEffect effect) {
27          super(effect);
28          this.filter = effect.filter;
29          this.subType = effect.subType;
30      }
31  
32      @Override
33      public boolean apply(Game game, Ability source) {
34          return false;
35      }
36  
37      @Override
38      public CreaturesBecomeOtherTypeEffect copy() {
39          return new CreaturesBecomeOtherTypeEffect(this);
40      }
41  
42      @Override
43      public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
44          if (layer == Layer.TypeChangingEffects_4) {
45              for (Permanent object : game.getBattlefield().getActivePermanents(this.filter, source.getControllerId(), game)) {
46                  object.addSubType(game, this.subType);
47              }
48          }
49  
50          return true;
51      }
52  
53      @Override
54      public boolean hasLayer(Layer layer) {
55          return layer == Layer.TypeChangingEffects_4;
56      }
57  }

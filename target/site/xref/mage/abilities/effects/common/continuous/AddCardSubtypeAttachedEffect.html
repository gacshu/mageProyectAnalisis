3   package mage.abilities.effects.common.continuous;
4   
5   import mage.abilities.Ability;
6   import mage.abilities.effects.ContinuousEffectImpl;
7   import mage.constants.*;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  
11  /**
12   * @author nantuko
13   */
14  public class AddCardSubtypeAttachedEffect extends ContinuousEffectImpl {
15      private SubType addedSubtype;
16      private AttachmentType attachmentType;
17  
18      public AddCardSubtypeAttachedEffect(SubType addedSubtype, Duration duration, AttachmentType attachmentType) {
19          super(duration, Layer.TypeChangingEffects_4, SubLayer.NA, Outcome.Benefit);
20          this.addedSubtype = addedSubtype;
21          this.attachmentType = attachmentType;
22          setText();
23      }
24  
25      public AddCardSubtypeAttachedEffect(final AddCardSubtypeAttachedEffect effect) {
26          super(effect);
27          this.addedSubtype = effect.addedSubtype;
28          this.attachmentType = effect.attachmentType;
29      }
30  
31      @Override
32      public boolean apply(Game game, Ability source) {
33          Permanent equipment = game.getPermanent(source.getSourceId());
34          if (equipment != null && equipment.getAttachedTo() != null) {
35              Permanent target = game.getPermanent(equipment.getAttachedTo());
36              if (target != null)
37                  target.addSubType(game, addedSubtype);
38          }
39          return true;
40      }
41  
42      @Override
43      public AddCardSubtypeAttachedEffect copy() {
44          return new AddCardSubtypeAttachedEffect(this);
45      }
46  
47      private void setText() {
48          StringBuilder sb = new StringBuilder();
49  
50          sb.append(attachmentType.verb());
51          sb.append(" creature becomes ").append(addedSubtype).append(" in addition to its other types"); //TODO add attacked card type detection
52          staticText = sb.toString();
53      }
54  }

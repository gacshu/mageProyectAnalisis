29  public class SacrificeOpponentsUnlessPayEffect extends OneShotEffect {
30  
31      protected Cost cost;
32      protected FilterPermanent filter;
33  
34      public SacrificeOpponentsUnlessPayEffect(int genericManaCost) {
35          this(new GenericManaCost(genericManaCost), StaticFilters.FILTER_PERMANENT);
36      }
37  
38      public SacrificeOpponentsUnlessPayEffect(Cost cost, FilterPermanent filter) {
39          super(Outcome.Sacrifice);
40          this.cost = cost;
41          this.filter = filter;
42          this.staticText = "each opponent sacrifices " + CardUtil.addArticle(filter.getMessage()) + " unless they " + CardUtil.addCostVerb(cost.getText());
43      }
44  
45      public SacrificeOpponentsUnlessPayEffect(final SacrificeOpponentsUnlessPayEffect effect) {
46          super(effect);
47          if (effect.cost != null) {
48              this.cost = effect.cost.copy();
49          }
50  
51          if (effect.filter != null) {
52              this.filter = effect.filter.copy();
53          }
54      }
55  
56      @Override
57      public SacrificeOpponentsUnlessPayEffect copy() {
58          return new SacrificeOpponentsUnlessPayEffect(this);
59      }
60  
61      @Override
62      public boolean apply(Game game, Ability source) {
63          List<UUID> permsToSacrifice = new ArrayList<>();
64          filter.add(TargetController.YOU.getControllerPredicate());
65  
66          for (UUID playerId : game.getOpponents(source.getControllerId())) {
67              Player player = game.getPlayer(playerId);
68  
69              if (player != null) {
70                  Cost costToPay = cost.copy();
71                  String costValueMessage = costToPay.getText();
72                  String message = ((costToPay instanceof ManaCost) ? "Pay " : "") + costValueMessage + '?';
73  
74                  costToPay.clearPaid();
75                  if (!(player.chooseUse(Outcome.Benefit, message, source, game)
76                          && costToPay.pay(source, game, source, player.getId(), false, null))) {
77                      game.informPlayers(player.getLogName() + " chooses not to pay " + costValueMessage + " to prevent the sacrifice effect");
78  
79                      int numTargets = Math.min(1, game.getBattlefield().countAll(filter, player.getId(), game));
80                      if (numTargets > 0) {
81                          TargetPermanent target = new TargetPermanent(numTargets, numTargets, filter, true);
82  
83                          if (target.canChoose(player.getId(), source, game)) {
84                              player.chooseTarget(Outcome.Sacrifice, target, source, game);
85                              permsToSacrifice.addAll(target.getTargets());
86                          }
87                      }
88                  } else {
89                      game.informPlayers(player.getLogName() + " chooses to pay " + costValueMessage + " to prevent the sacrifice effect");
90                  }
91              }
92          }
93  
94          for (UUID permID : permsToSacrifice) {
95              Permanent permanent = game.getPermanent(permID);
96  
97              if (permanent != null) {
98                  permanent.sacrifice(source, game);
99              }
100         }
101 
102         return true;
103     }
104 }

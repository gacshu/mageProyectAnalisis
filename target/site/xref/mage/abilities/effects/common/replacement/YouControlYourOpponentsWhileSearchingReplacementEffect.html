1   package mage.abilities.effects.common.replacement;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.ReplacementEffectImpl;
5   import mage.constants.Duration;
6   import mage.constants.Outcome;
7   import mage.game.Game;
8   import mage.game.events.GameEvent;
9   import mage.game.events.SearchLibraryEvent;
10  import mage.players.Player;
11  
12  /**
13   * @author JayDi85
14   */
15  public class YouControlYourOpponentsWhileSearchingReplacementEffect extends ReplacementEffectImpl {
16  
17      public YouControlYourOpponentsWhileSearchingReplacementEffect() {
18          super(Duration.WhileOnBattlefield, Outcome.Benefit);
19          staticText = "You control your opponents while they're searching their libraries";
20      }
21  
22      YouControlYourOpponentsWhileSearchingReplacementEffect(final YouControlYourOpponentsWhileSearchingReplacementEffect effect) {
23          super(effect);
24      }
25  
26      @Override
27      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
28          SearchLibraryEvent se = (SearchLibraryEvent) event;
29          se.setSearchingControllerId(source.getControllerId());
30          return false;
31      }
32  
33      @Override
34      public boolean checksEventType(GameEvent event, Game game) {
35          return event.getType() == GameEvent.EventType.SEARCH_LIBRARY;
36      }
37  
38      @Override
39      public boolean applies(GameEvent event, Ability source, Game game) {
40          Player controller = game.getPlayer(source.getControllerId());
41          return controller != null
42                  && game.isOpponent(controller, event.getPlayerId())
43                  // verify that the controller of the ability is searching their library
44                  && event.getTargetId().equals(event.getPlayerId());
45      }
46  
47      @Override
48      public YouControlYourOpponentsWhileSearchingReplacementEffect copy() {
49          return new YouControlYourOpponentsWhileSearchingReplacementEffect(this);
50      }
51  }

1   package mage.abilities.effects.common;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.Mode;
6   import mage.abilities.effects.Effect;
7   import mage.abilities.effects.Effects;
8   import mage.abilities.effects.OneShotEffect;
9   import mage.constants.Outcome;
10  import mage.game.Game;
11  import mage.players.Player;
12  
13  /**
14   * @author spjspj
15   */
16  public class RollDiceEffect extends OneShotEffect {
17  
18      protected Effects executingEffects = new Effects();
19      protected int numSides;
20  
21      public RollDiceEffect(Effect effect, int numSides) {
22          this(effect, Outcome.Neutral, numSides);
23      }
24  
25      public RollDiceEffect(Effect effect, Outcome outcome, int numSides) {
26          super(outcome);
27          addEffect(effect);
28          this.numSides = numSides;
29      }
30  
31      public RollDiceEffect(final RollDiceEffect effect) {
32          super(effect);
33          this.executingEffects = effect.executingEffects.copy();
34          this.numSides = effect.numSides;
35      }
36  
37      public void addEffect(Effect effect) {
38          if (effect != null) {
39              executingEffects.add(effect);
40          }
41      }
42  
43      @Override
44      public boolean apply(Game game, Ability source) {
45          Player controller = game.getPlayer(source.getControllerId());
46          MageObject mageObject = game.getObject(source);
47          if (controller != null && mageObject != null) {
48              controller.rollDice(outcome, source, game, numSides);
49              return true;
50          }
51          return false;
52      }
53  
54      @Override
55      public String getText(Mode mode) {
56          if (!staticText.isEmpty()) {
57              return staticText;
58          }
59          return "Roll a " + numSides + " sided die";
60      }
61  
62      @Override
63      public RollDiceEffect copy() {
64          return new RollDiceEffect(this);
65      }
66  }

2   package mage.abilities.effects.common;
3   
4   import mage.abilities.Ability;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.cards.Card;
7   import mage.constants.Outcome;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  import mage.players.Player;
11  
12  /**
13   *
14   * @author LevelX2
15   */
16  public class ReturnToBattlefieldUnderOwnerControlSourceEffect extends OneShotEffect {
17  
18      private boolean tapped;
19      private boolean attacking;
20      private int zoneChangeCounter;
21  
22      public ReturnToBattlefieldUnderOwnerControlSourceEffect() {
23          this(false);
24      }
25  
26      public ReturnToBattlefieldUnderOwnerControlSourceEffect(boolean tapped) {
27          this(tapped, -1);
28      }
29  
30      public ReturnToBattlefieldUnderOwnerControlSourceEffect(boolean tapped, int zoneChangeCounter) {
31          this(tapped, false, zoneChangeCounter);
32      }
33  
34      public ReturnToBattlefieldUnderOwnerControlSourceEffect(boolean tapped, boolean attacking, int zoneChangeCounter) {
35          super(Outcome.Benefit);
36          this.tapped = tapped;
37          this.attacking = attacking;
38          this.zoneChangeCounter = zoneChangeCounter;
39          staticText = "return that card to the battlefield"
40                  + (tapped ? " tapped" : "")
41                  + (attacking ? " attacking" : "")
42                  + " under its owner's control";
43      }
44  
45      public ReturnToBattlefieldUnderOwnerControlSourceEffect(final ReturnToBattlefieldUnderOwnerControlSourceEffect effect) {
46          super(effect);
47          this.tapped = effect.tapped;
48          this.zoneChangeCounter = effect.zoneChangeCounter;
49          this.attacking = effect.attacking;
50      }
51  
52      @Override
53      public ReturnToBattlefieldUnderOwnerControlSourceEffect copy() {
54          return new ReturnToBattlefieldUnderOwnerControlSourceEffect(this);
55      }
56  
57      @Override
58      public boolean apply(Game game, Ability source) {
59          Player controller = game.getPlayer(source.getControllerId());
60          Card card = game.getCard(source.getSourceId());
61          if (controller != null && card != null) {
62              // return only from public zones
63              switch (game.getState().getZone(card.getId())) {
64                  case EXILED:
65                  case COMMAND:
66                  case GRAVEYARD:
67                      if (zoneChangeCounter < 0 || game.getState().getZoneChangeCounter(card.getId()) == zoneChangeCounter) {
68  
69                          if (controller.moveCards(card, Zone.BATTLEFIELD, source, game, tapped, false, true, null)) {
70                              if (attacking) {
71                                  game.getCombat().addAttackingCreature(card.getId(), game);
72                              }
73                          }
74                      }
75                      break;
76              }
77              return true;
78          }
79          return false;
80      }
81  }

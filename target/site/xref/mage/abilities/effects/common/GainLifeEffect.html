3   package mage.abilities.effects.common;
4   
5   import mage.abilities.Ability;
6   import mage.abilities.Mode;
7   import mage.abilities.dynamicvalue.DynamicValue;
8   import mage.abilities.dynamicvalue.common.StaticValue;
9   import mage.abilities.effects.OneShotEffect;
10  import mage.constants.Outcome;
11  import mage.game.Game;
12  import mage.players.Player;
13  
14  /**
15   * @author BetaSteward_at_googlemail.com
16   */
17  public class GainLifeEffect extends OneShotEffect {
18  
19      private DynamicValue life;
20  
21      public GainLifeEffect(int life) {
22          this(StaticValue.get(life));
23      }
24  
25      public GainLifeEffect(DynamicValue life) {
26          super(Outcome.GainLife);
27          this.life = life;
28      }
29  
30      public GainLifeEffect(DynamicValue life, String rule) {
31          super(Outcome.GainLife);
32          this.life = life;
33          staticText = rule;
34      }
35  
36      public GainLifeEffect(final GainLifeEffect effect) {
37          super(effect);
38          this.life = effect.life.copy();
39      }
40  
41      @Override
42      public GainLifeEffect copy() {
43          return new GainLifeEffect(this);
44      }
45  
46      @Override
47      public boolean apply(Game game, Ability source) {
48          Player player = game.getPlayer(source.getControllerId());
49          if (player != null) {
50              player.gainLife(life.calculate(game, source, this), game, source);
51          }
52          return true;
53      }
54  
55      @Override
56      public String getText(Mode mode) {
57          if (staticText != null && !staticText.isEmpty()) {
58              return staticText;
59          }
60          StringBuilder sb = new StringBuilder();
61          String message = life.getMessage();
62          sb.append("you gain ");
63          if (message.isEmpty() || !life.toString().equals("1")) {
64              sb.append(life).append(' ');
65          }
66          sb.append("life");
67          if (!message.isEmpty()) {
68              sb.append(life.toString().equals("1") ? " equal to the number of " : " for each ");
69              sb.append(message);
70          }
71          return sb.toString();
72      }
73  }

25  /**
26   * "Put a {filter} card from {zone 1} or {zone 2} onto the battlefield.
27   *
28   * @author TheElk801, Alex-Vasile
29   */
30  public class PutCardFromOneOfTwoZonesOntoBattlefieldEffect extends OneShotEffect {
31  
32      private final FilterCard filterCard;
33      private final boolean tapped;
34      private final Effect effectToApplyOnPermanent;
35      private final Zone zone1;
36      private final Zone zone2;
37  
38      public PutCardFromOneOfTwoZonesOntoBattlefieldEffect(FilterCard filterCard) {
39          this(filterCard, false);
40      }
41  
42      public PutCardFromOneOfTwoZonesOntoBattlefieldEffect(FilterCard filterCard, boolean tapped) {
43          this(filterCard, tapped, null);
44      }
45  
46      /**
47       *
48       * @param filterCard                Filter used to filter which cards are valid choices. (no default)
49       * @param tapped                    If the permanent should enter the battlefield tapped (default is False)
50       * @param effectToApplyOnPermanent  An effect to apply to the permanent after it enters (default null)
51       *                                  See "Swift Warkite" or "Nissa of Shadowed Boughs".
52       * @param zone1                     The first zone to pick from (default of HAND)
53       * @param zone2                     The second zone to pick from (defualt of GRAVEYARD)
54       */
55      public PutCardFromOneOfTwoZonesOntoBattlefieldEffect(FilterCard filterCard, boolean tapped, Effect effectToApplyOnPermanent, Zone zone1, Zone zone2) {
56          super(filterCard instanceof FilterCreatureCard ? Outcome.PutCreatureInPlay : Outcome.PutCardInPlay);
57          this.filterCard = filterCard;
58          this.tapped = tapped;
59          this.effectToApplyOnPermanent = effectToApplyOnPermanent;
60          this.zone1 = zone1;
61          this.zone2 = zone2;
62      }
63  
64      public PutCardFromOneOfTwoZonesOntoBattlefieldEffect(FilterCard filterCard, boolean tapped, Effect effectToApplyOnPermanent) {
65          this(filterCard, tapped, effectToApplyOnPermanent, Zone.HAND, Zone.GRAVEYARD);
66      }
67  
68      public PutCardFromOneOfTwoZonesOntoBattlefieldEffect(FilterCard filterCard, Zone zone1, Zone zone2) {
69          this(filterCard, false, null, zone1, zone2);
70      }
71  
72      private PutCardFromOneOfTwoZonesOntoBattlefieldEffect(final PutCardFromOneOfTwoZonesOntoBattlefieldEffect effect) {
73          super(effect);
74          this.filterCard = effect.filterCard;
75          this.tapped = effect.tapped;
76          this.zone1 = effect.zone1;
77          this.zone2 = effect.zone2;
78          if (effect.effectToApplyOnPermanent != null) {
79              this.effectToApplyOnPermanent = effect.effectToApplyOnPermanent.copy();
80          } else {
81              this.effectToApplyOnPermanent = null;
82          }
83      }
84  
85      @Override
86      public boolean apply(Game game, Ability source) {
87          Player controller = game.getPlayer(source.getControllerId());
88          if (controller == null) {
89              return false;
90          }
91  
92          Cards cardsInZone1 = getCardsFromZone(game, controller, zone1);
93          Cards cardsInZone2 = getCardsFromZone(game, controller, zone2);
94  
95          boolean cardsAvailableInZone1 = cardsInZone1.count(filterCard, game) > 0;
96          boolean cardsAvailableInZone2 = cardsInZone2.count(filterCard, game) > 0;
97          if (!cardsAvailableInZone1 && !cardsAvailableInZone2) {
98              return false;
99          }
100 
101         boolean choose1stZone;
102         if (cardsAvailableInZone1 && cardsAvailableInZone2) {
103             choose1stZone = controller.chooseUse(outcome, "Where do you want to chose the card from?",
104                     null, zone1.name(), zone2.name(), source, game);
105         } else {
106             choose1stZone = cardsAvailableInZone1;
107         }
108 
109         Zone zone = choose1stZone ? zone1 : zone2;
110         Cards cards = choose1stZone ? cardsInZone1 : cardsInZone2;
111         TargetCard targetCard;
112 
113         switch (zone) {
114             case HAND:
115                 targetCard = new TargetCardInHand(filterCard);
116                 break;
117             case GRAVEYARD:
118                 targetCard = new TargetCardInGraveyard(filterCard);
119                 break;
120             case COMMAND:
121                 targetCard = new TargetCardInCommandZone(filterCard);
122                 break;
123             default:
124                 return false;
125         }
126         controller.choose(outcome, cards, targetCard, game);
127         Card card = game.getCard(targetCard.getFirstTarget());
128         if (card == null || !controller.moveCards(card, Zone.BATTLEFIELD, source, game, tapped, false, false, null)) {
129             return false;
130         }
131 
132         if (effectToApplyOnPermanent != null) {
133             effectToApplyOnPermanent.setTargetPointer(new FixedTarget(card.getId()));
134             effectToApplyOnPermanent.apply(game, source);
135         }
136         return true;
137     }
138 
139     private static Cards getCardsFromZone(Game game, Player player, Zone zone) {
140         switch (zone) {
141             case HAND:
142                 return player.getHand();
143             case COMMAND:
144                 return new CardsImpl(game.getCommanderCardsFromCommandZone(player, CommanderCardType.ANY));
145             case GRAVEYARD:
146                 return player.getGraveyard();
147             default:
148                 return new CardsImpl();
149         }
150     }
151 
152     @Override
153     public PutCardFromOneOfTwoZonesOntoBattlefieldEffect copy() {
154         return new PutCardFromOneOfTwoZonesOntoBattlefieldEffect(this);
155     }
156 
157     @Override
158     public String getText(Mode mode) {
159         return "you may put " + CardUtil.addArticle(this.filterCard.getMessage()) +
160                 " from your hand or graveyard onto the battlefield" +
161                 (this.tapped ? " tapped" : "") +
162                 (effectToApplyOnPermanent == null ? "" : ". " + effectToApplyOnPermanent.getText(mode));
163     }
164 }

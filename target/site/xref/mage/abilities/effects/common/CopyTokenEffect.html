1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.ContinuousEffectImpl;
5   import mage.constants.*;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   import mage.game.permanent.token.Token;
9   
10  public class CopyTokenEffect extends ContinuousEffectImpl {
11  
12      protected Token token;
13  
14      public CopyTokenEffect(Token token) {
15          super(Duration.WhileOnBattlefield, Layer.CopyEffects_1, SubLayer.CopyEffects_1a, Outcome.BecomeCreature);
16          this.token = token.copy();
17          staticText = "You may have {this} enter the battlefield as a copy of " + token.getDescription() + " on the battlefield";
18      }
19  
20      public CopyTokenEffect(final CopyTokenEffect effect) {
21          super(effect);
22          this.token = effect.token.copy();
23      }
24  
25      @Override
26      public boolean apply(Game game, Ability source) {
27          Permanent permanent = game.getPermanent(source.getSourceId());
28          permanent.setName(token.getName());
29          permanent.getColor(game).setColor(token.getColor(game));
30          permanent.removeAllCardTypes(game);
31          for (CardType type : token.getCardType(game)) {
32              permanent.addCardType(game, type);
33          }
34          permanent.removeAllSubTypes(game);
35          permanent.copySubTypesFrom(game, token);
36          permanent.getSuperType().clear();
37          for (SuperType type : token.getSuperType()) {
38              permanent.addSuperType(type);
39          }
40          permanent.getAbilities().clear();
41          for (Ability ability : token.getAbilities()) {
42              permanent.addAbility(ability, source.getSourceId(), game);
43          }
44          permanent.getPower().setModifiedBaseValue(token.getPower().getModifiedBaseValue());
45          permanent.getToughness().setModifiedBaseValue(token.getToughness().getModifiedBaseValue());
46          //permanent.getLoyalty().setBoostedValue(card.getLoyalty().getValue());
47  
48          return true;
49  
50      }
51  
52      @Override
53      public CopyTokenEffect copy() {
54          return new CopyTokenEffect(this);
55      }
56  
57  }

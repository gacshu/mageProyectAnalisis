16  /**
17   * FAQ 2013/01/11
18   * <p>
19   * 702.98. Evolve
20   * <p>
21   * 702.98a Evolve is a triggered ability. "Evolve" means "Whenever a creature
22   * enters the battlefield under your control, if that creature's power is
23   * greater than this creature's power and/or that creature's toughness is
24   * greater than this creature's toughness, put a +1/+1 counter on this
25   * creature."
26   * <p>
27   * 702.98b If a creature has multiple instances of evolve, each triggers
28   * separately
29   * <p>
30   * Rulings
31   * <p>
32   * When comparing the stats of the two creatures, you always compare power to
33   * power and toughness to toughness. Whenever a creature enters the battlefield
34   * under your control, check its power and toughness against the power and
35   * toughness of the creature with evolve. If neither stat of the new creature is
36   * greater, evolve won't trigger at all. For example, if you control a 2/3
37   * creature with evolve and a 2/2 creature enters the battlefield under your
38   * control, you won't have the opportunity to cast a spell like Giant Growth to
39   * make the 2/2 creature large enough to cause evolve to trigger. If evolve
40   * triggers, the stat comparison will happen again when the ability tries to
41   * resolve. If neither stat of the new creature is greater, the ability will do
42   * nothing. If the creature that entered the battlefield leaves the battlefield
43   * before evolve tries to resolve, use its last known power and toughness to
44   * compare the stats. If a creature enters the battlefield with +1/+1 counters
45   * on it, consider those counters when determining if evolve will trigger. For
46   * example, a 1/1 creature that enters the battlefield with two +1/+1 counters
47   * on it will cause the evolve ability of a 2/2 creature to trigger. If multiple
48   * creatures enter the battlefield at the same time, evolve may trigger multiple
49   * times, although the stat comparison will take place each time one of those
50   * abilities tries to resolve. For example, if you control a 2/2 creature with
51   * evolve and two 3/3 creatures enter the battlefield, evolve will trigger
52   * twice. The first ability will resolve and put a +1/+1 counter on the creature
53   * with evolve. When the second ability tries to resolve, neither the power nor
54   * the toughness of the new creature is greater than that of the creature with
55   * evolve, so that ability does nothing. When comparing the stats as the evolve
56   * ability resolves, it's possible that the stat that's greater changes from
57   * power to toughness or vice versa. If this happens, the ability will still
58   * resolve and you'll put a +1/+1 counter on the creature with evolve. For
59   * example, if you control a 2/2 creature with evolve and a 1/3 creature enters
60   * the battlefield under your control, it toughness is greater so evolve will
61   * trigger. In response, the 1/3 creature gets +2/-2. When the evolve trigger
62   * tries to resolve, its power is greater. You'll put a +1/+1 counter on the
63   * creature with evolve.
64   *
65   * @author LevelX2
66   */
67  public class EvolveAbility extends TriggeredAbilityImpl {
68  
69      public EvolveAbility() {
70          super(Zone.BATTLEFIELD, new EvolveEffect());
71      }
72  
73      public EvolveAbility(EvolveAbility ability) {
74          super(ability);
75      }
76  
77      @Override
78      public boolean checkEventType(GameEvent event, Game game) {
79          return event.getType() == GameEvent.EventType.ENTERS_THE_BATTLEFIELD;
80      }
81  
82      @Override
83      public boolean checkTrigger(GameEvent event, Game game) {
84          if (!event.getTargetId().equals(this.getSourceId())) {
85              Permanent triggeringCreature = ((EntersTheBattlefieldEvent) event).getTarget();
86              if (triggeringCreature != null
87                      && triggeringCreature.isCreature(game)
88                      && triggeringCreature.isControlledBy(this.controllerId)) {
89                  Permanent sourceCreature = game.getPermanent(sourceId);
90                  if (sourceCreature != null && isPowerOrThoughnessGreater(sourceCreature, triggeringCreature)) {
91                      this.getEffects().get(0).setTargetPointer(new FixedTarget(event.getTargetId(), game));
92                      return true;
93                  }
94              }
95          }
96          return false;
97      }
98  
99      @Override
100     public boolean checkInterveningIfClause(Game game) {
101         return true;
102     }
103 
104     public static boolean isPowerOrThoughnessGreater(Permanent sourceCreature, Permanent newCreature) {
105         if (newCreature.getPower().getValue() > sourceCreature.getPower().getValue()) {
106             return true;
107         }
108         return newCreature.getToughness().getValue() > sourceCreature.getToughness().getValue();
109     }
110 
111     @Override
112     public String getRule() {
113         return "Evolve <i>(Whenever a creature enters the battlefield under your control, if that creature has greater power or toughness than this creature, put a +1/+1 counter on this creature.)</i>";
114     }
115 
116     @Override
117     public EvolveAbility copy() {
118         return new EvolveAbility(this);
119     }
120 }
121 
122 class EvolveEffect extends OneShotEffect {
123 
124     public EvolveEffect() {
125         super(Outcome.BoostCreature);
126     }
127 
128     public EvolveEffect(final EvolveEffect effect) {
129         super(effect);
130     }
131 
132     @Override
133     public EvolveEffect copy() {
134         return new EvolveEffect(this);
135     }
136 
137     @Override
138     public boolean apply(Game game, Ability source) {
139         Permanent triggeringCreature = game.getPermanentOrLKIBattlefield(getTargetPointer().getFirst(game, source));
140         if (triggeringCreature != null) {
141             Permanent sourceCreature = game.getPermanent(source.getSourceId());
142             if (sourceCreature != null && EvolveAbility.isPowerOrThoughnessGreater(sourceCreature, triggeringCreature)) {
143                 sourceCreature.addCounters(CounterType.P1P1.createInstance(), source.getControllerId(), source, game);
144                 game.fireEvent(GameEvent.getEvent(GameEvent.EventType.EVOLVED_CREATURE, sourceCreature.getId(), source, source.getControllerId()));
145             }
146             return true;
147         }
148         return false;
149     }
150 }

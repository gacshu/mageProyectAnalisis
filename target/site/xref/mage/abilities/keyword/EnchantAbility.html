2   package mage.abilities.keyword;
3   
4   import mage.constants.Zone;
5   import mage.abilities.StaticAbility;
6   import mage.target.Target;
7   
8   /**
9    *
10   * @author BetaSteward_at_googlemail.com
11   * @author North
12   */
13  public class EnchantAbility extends StaticAbility {
14  
15      protected String targetName;
16  
17      public EnchantAbility(Target target) {
18          super(Zone.BATTLEFIELD, null);
19          this.targetName = target.getTargetName();
20      }
21  
22      public EnchantAbility(final EnchantAbility ability) {
23          super(ability);
24          this.targetName = ability.targetName;
25      }
26  
27      @Override
28      public EnchantAbility copy() {
29          return new EnchantAbility(this);
30      }
31  
32      @Override
33      public String getRule() {
34          StringBuilder sb = new StringBuilder();
35          sb.append("Enchant ").append(targetName);
36          if (!this.getEffects().isEmpty()) {
37              sb.append(". ").append(super.getRule());
38          }
39          return sb.toString();
40      }
41  }

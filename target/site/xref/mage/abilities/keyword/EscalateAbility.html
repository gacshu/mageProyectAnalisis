19  public class EscalateAbility extends SimpleStaticAbility {
20  
21      public EscalateAbility(Cost cost) {
22          super(Zone.ALL, new EscalateEffect(cost));
23          this.setRuleAtTheTop(true);
24      }
25  
26      public EscalateAbility(final EscalateAbility ability) {
27          super(ability);
28      }
29  
30      @Override
31      public EscalateAbility copy() {
32          return new EscalateAbility(this);
33      }
34  }
35  
36  class EscalateEffect extends CostModificationEffectImpl {
37  
38      private final Cost cost;
39  
40      EscalateEffect(Cost cost) {
41          super(Duration.WhileOnBattlefield, Outcome.Detriment, CostModificationType.INCREASE_COST);
42          this.cost = cost;
43          this.staticText = "Escalate " + cost.getText() + " <i>(Pay this cost for each mode chosen beyond the first.)</i>";
44      }
45  
46      EscalateEffect(final EscalateEffect effect) {
47          super(effect);
48          this.cost = effect.cost.copy();
49      }
50  
51      @Override
52      public boolean apply(Game game, Ability source, Ability abilityToModify) {
53          int numCosts = abilityToModify.getModes().getSelectedModes().size() - 1;
54          for (int i = 0; i < numCosts; i++) {
55              abilityToModify.addCost(cost.copy());
56          }
57          return true;
58      }
59  
60      @Override
61      public boolean applies(Ability abilityToModify, Ability source, Game game) {
62          if (abilityToModify.getSourceId().equals(source.getSourceId()) && (abilityToModify instanceof SpellAbility)) {
63              return abilityToModify.getModes().getSelectedModes().size() > 1;
64          }
65          return false;
66      }
67  
68      @Override
69      public EscalateEffect copy() {
70          return new EscalateEffect(this);
71      }
72  }

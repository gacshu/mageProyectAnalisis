1   package mage.abilities.keyword;
2   
3   import java.util.UUID;
4   import mage.abilities.common.BecomesBlockedSourceTriggeredAbility;
5   import mage.abilities.effects.common.LoseLifeTargetEffect;
6   import mage.game.Game;
7   import mage.game.events.GameEvent;
8   import mage.target.targetpointer.FixedTarget;
9   
10  public class AfflictAbility extends BecomesBlockedSourceTriggeredAbility {
11  
12      private final int lifeLoss;
13  
14      @Override
15      public AfflictAbility copy() {
16          return new AfflictAbility(this);
17      }
18  
19      public AfflictAbility(int amount) {
20          super(new LoseLifeTargetEffect(amount)
21                  .setText("Afflict " + amount + " <i>(Whenever this creature becomes blocked, defending player loses " + amount + " life.)</i>"), false);
22          lifeLoss = amount;
23      }
24  
25      @Override
26      public boolean checkTrigger(GameEvent event, Game game) {
27          if (super.checkTrigger(event, game)) {
28              UUID defenderId = game.getCombat().getDefendingPlayerId(getSourceId(), game);
29              if (defenderId != null) {
30                  this.getEffects().setTargetPointer(new FixedTarget(defenderId));
31                  return true;
32              }
33          }
34          return false;
35      }
36  
37      public AfflictAbility(final AfflictAbility afflictAbility) {
38          super(afflictAbility);
39          lifeLoss = afflictAbility.lifeLoss;
40      }
41  
42      @Override
43      public String getRule() {
44          return "Afflict " + lifeLoss + " <i>(Whenever this creature becomes blocked, defending player loses " + lifeLoss + " life.)</i>";
45      }
46  }

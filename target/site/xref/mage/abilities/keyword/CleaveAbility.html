1   package mage.abilities.keyword;

3   import mage.abilities.SpellAbility;
4   import mage.abilities.costs.mana.ManaCostsImpl;
5   import mage.abilities.effects.Effect;
6   import mage.cards.Card;
7   import mage.constants.SpellAbilityType;
8   import mage.constants.TimingRule;
9   
10  /**
11   * @author TheElk801
12   */
13  public class CleaveAbility extends SpellAbility {
14  
15      public CleaveAbility(Card card, Effect effect, String manaString) {
16          super(new ManaCostsImpl<>(manaString), card.getName() + " with cleave");
17          this.spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
18          this.addEffect(effect);
19          this.setRuleAtTheTop(true);
20          this.timing = (card.isSorcery(null) ? TimingRule.SORCERY : TimingRule.INSTANT);
21      }
22  
23      public CleaveAbility(final CleaveAbility ability) {
24          super(ability);
25      }
26  
27      @Override
28      public CleaveAbility copy() {
29          return new CleaveAbility(this);
30      }
31  
32      @Override
33      public String getRule(boolean all) {
34          return getRule();
35      }
36  
37      @Override
38      public String getRule() {
39          return "Cleave " + getManaCostsToPay().getText() + " <i>(You may cast this spell for its cleave cost. If you do, remove the words in square brackets.)</i>";
40      }
41  }

1   package mage.abilities.keyword;
2   
3   import mage.abilities.common.SimpleStaticAbility;
4   import mage.abilities.dynamicvalue.common.PermanentsOnBattlefieldCount;
5   import mage.abilities.effects.common.AffinityEffect;
6   import mage.abilities.hint.ValueHint;
7   import mage.constants.SubType;
8   import mage.constants.Zone;
9   import mage.filter.common.FilterControlledPermanent;
10  
11  /**
12   * @author LevelX2
13   */
14  
15  public class AffinityForLandTypeAbility extends SimpleStaticAbility {
16      private final String rulesText;
17  
18      public AffinityForLandTypeAbility(SubType landType, String pluralName) {
19          super(Zone.ALL, null);
20          rulesText = "Affinity for " + pluralName + " <i>(This spell costs {1} less to cast for each " + landType + " you control.)</i>";
21          setRuleAtTheTop(true);
22  
23          FilterControlledPermanent filter = new FilterControlledPermanent(landType);
24          addEffect(new AffinityEffect(filter));
25          addHint(new ValueHint(pluralName + " you control", new PermanentsOnBattlefieldCount(filter)));
26      }
27  
28      public AffinityForLandTypeAbility(final AffinityForLandTypeAbility ability) {
29          super(ability);
30          this.rulesText = ability.rulesText;
31      }
32  
33      @Override
34      public SimpleStaticAbility copy() {
35          return new AffinityForLandTypeAbility(this);
36      }
37  
38      @Override
39      public String getRule() {
40          return rulesText;
41      }
42  }

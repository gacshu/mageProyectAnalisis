21  /**
22   * 702.36. Morph
23   * <p>
24   * 702.36a Morph is a static ability that functions in any zone from which you
25   * could play the card its on, and the morph effect works any time the card is
26   * face down. "Morph [cost]" means "You may cast this card as a 2/2 face-down
27   * creature, with no text, no name, no subtypes, and no mana cost by paying {3}
28   * rather than paying its mana cost." (See rule 707, "Face-Down Spells and
29   * Permanents.")
30   * <p>
31   * 702.36b To cast a card using its morph ability, turn it face down. It becomes
32   * a 2/2 face-down creature card, with no text, no name, no subtypes, and no
33   * mana cost. Any effects or prohibitions that would apply to casting a card
34   * with these characteristics (and not the face-up cards characteristics) are
35   * applied to casting this card. These values are the copiable values of that
36   * object's characteristics. (See rule 613, "Interaction of Continuous Effects,"
37   * and rule 706, "Copying Objects.") Put it onto the stack (as a face-down spell
38   * with the same characteristics), and pay {3} rather than pay its mana cost.
39   * This follows the rules for paying alternative costs. You can use morph to
40   * cast a card from any zone from which you could normally play it. When the
41   * spell resolves, it enters the battlefield with the same characteristics the
42   * spell had. The morph effect applies to the face-down object wherever it is,
43   * and it ends when the permanent is turned face up. #
44   * <p>
45   * 702.36c You can't cast a card face down if it doesn't have morph.
46   * <p>
47   * 702.36d If you have priority, you may turn a face-down permanent you control
48   * face up. This is a special action; it doesn't use the stack (see rule 115).
49   * To do this, show all players what the permanents morph cost would be if it
50   * were face up, pay that cost, then turn the permanent face up. (If the
51   * permanent wouldn't have a morph cost if it were face up, it cant be turned
52   * face up this way.) The morph effect on it ends, and it regains its normal
53   * characteristics. Any abilities relating to the permanent entering the
54   * battlefield dont trigger when its turned face up and dont have any effect,
55   * because the permanent has already entered the battlefield.
56   * <p>
57   * 702.36e See rule 707, "Face-Down Spells and Permanents," for more information
58   * on how to cast cards with morph.
59   *
60   * @author LevelX2
61   */
62  public class MorphAbility extends AlternativeSourceCostsImpl {
63  
64      protected static final String ABILITY_KEYWORD = "Morph";
65      protected static final String ABILITY_KEYWORD_MEGA = "Megamorph";
66      protected static final String REMINDER_TEXT = "You may cast this card face down as a "
67              + "2/2 creature for {3}. Turn it face up any time for its morph cost.";
68      protected static final String REMINDER_TEXT_MEGA = "You may cast this card face down "
69              + "as a 2/2 creature for {3}. Turn it face up any time for its megamorph "
70              + "cost and put a +1/+1 counter on it.";
71      protected Costs<Cost> morphCosts;
72      // needed to check activation status, if card changes zone after casting it
73      private final boolean megamorph;
74  
75      public MorphAbility(Cost morphCost) {
76          this(morphCost, false);
77      }
78  
79      public MorphAbility(Cost morphCost, boolean megamorph) {
80          super(megamorph ? ABILITY_KEYWORD_MEGA : ABILITY_KEYWORD, megamorph ? REMINDER_TEXT_MEGA : REMINDER_TEXT, new GenericManaCost(3));
81          this.morphCosts = new CostsImpl<>();
82          this.morphCosts.add(morphCost);
83          this.megamorph = megamorph;
84          this.setWorksFaceDown(true);
85          Ability ability = new SimpleStaticAbility(new BecomesFaceDownCreatureEffect(
86                  morphCosts, (megamorph ? FaceDownType.MEGAMORPHED : FaceDownType.MORPHED)));
87          ability.setWorksFaceDown(true);
88          ability.setRuleVisible(false);
89          addSubAbility(ability);
90      }
91  
92      public MorphAbility(final MorphAbility ability) {
93          super(ability);
94          this.morphCosts = ability.morphCosts; // can't be changed
95          this.megamorph = ability.megamorph;
96      }
97  
98      @Override
99      public MorphAbility copy() {
100         return new MorphAbility(this);
101     }
102 
103     @Override
104     public boolean askToActivateAlternativeCosts(Ability ability, Game game) {
105         switch (ability.getAbilityType()) {
106             case SPELL:
107                 Spell spell = game.getStack().getSpell(ability.getId());
108                 if (spell != null) {
109                     spell.setFaceDown(true, game);
110                     if (handleActivatingAlternativeCosts(ability, game)) {
111                         game.getState().setValue("MorphAbility" + ability.getSourceId(), "activated");
112                         spell.getColor(game).setColor(null);
113                         game.getState().getCreateMageObjectAttribute(spell.getCard(), game).getSubtype().clear();
114                     } else {
115                         spell.setFaceDown(false, game);
116                     }
117                 }
118                 break;
119             case PLAY_LAND:
120                 handleActivatingAlternativeCosts(ability, game);
121         }
122         return isActivated(ability, game);
123     }
124 
125     public Costs<Cost> getMorphCosts() {
126         return morphCosts;
127     }
128 
129     @Override
130     public String getRule() {
131         boolean isMana = morphCosts.get(0) instanceof ManaCost;
132         return alternativeCost.getName() + (isMana ? " " : "&mdash;") +
133                 morphCosts.getText() + (isMana ? ' ' : ". ") + alternativeCost.getReminderText();
134     }
135 
136     public static void setPermanentToFaceDownCreature(MageObject mageObject, Game game) {
137         mageObject.getPower().setModifiedBaseValue(2);
138         mageObject.getToughness().setModifiedBaseValue(2);
139         mageObject.getAbilities().clear();
140         mageObject.getColor(game).setColor(new ObjectColor());
141         mageObject.setName("");
142         mageObject.removeAllCardTypes(game);
143         mageObject.addCardType(game, CardType.CREATURE);
144         mageObject.removeAllSubTypes(game);
145         mageObject.getSuperType().clear();
146         mageObject.getManaCost().clear();
147         if (mageObject instanceof Permanent) {
148             ((Permanent) mageObject).setExpansionSetCode("");
149             ((Permanent) mageObject).setRarity(Rarity.SPECIAL);
150         }
151     }

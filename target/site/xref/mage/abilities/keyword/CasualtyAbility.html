8   import mage.abilities.costs.common.SacrificeTargetCost;
9   import mage.abilities.effects.common.CopySourceSpellEffect;
10  import mage.constants.ComparisonType;
11  import mage.constants.Outcome;
12  import mage.constants.Zone;
13  import mage.filter.common.FilterControlledCreaturePermanent;
14  import mage.filter.common.FilterControlledPermanent;
15  import mage.filter.predicate.mageobject.PowerPredicate;
16  import mage.game.Game;
17  import mage.players.Player;
18  import mage.target.common.TargetControlledPermanent;
19  
20  /**
21   * @author TheElk801, Alex-Vasile
22   */
23  public class CasualtyAbility extends StaticAbility implements OptionalAdditionalSourceCosts {
24  
25      private static final String keywordText = "Casualty";
26      private final String promptString;
27      private final String rule;
28  
29      protected OptionalAdditionalCost additionalCost;
30  
31      private static TargetControlledPermanent makeFilter(int number) {
32          FilterControlledPermanent filter = new FilterControlledCreaturePermanent(
33                  "creature with power " + number + " or greater"
34          );
35          filter.add(new PowerPredicate(ComparisonType.MORE_THAN, number - 1));
36          return new TargetControlledPermanent(1, 1, filter, true);
37      }
38  
39      public CasualtyAbility(int number) {
40          super(Zone.STACK, null);
41          String reminderText = "As you cast this spell, you may sacrifice a creature with power " +
42                  number + " or greater. When you do, copy this spell.";
43          this.additionalCost = new OptionalAdditionalCostImpl(keywordText, reminderText, new SacrificeTargetCost(makeFilter(number)));
44          this.additionalCost.setRepeatable(false);
45          this.promptString = "Sacrifice a creature with power " + number + " or greater?";
46          this.rule = additionalCost.getName() + ' ' + number + ' ' + additionalCost.getReminderText();
47          this.setRuleAtTheTop(true);
48      }
49  
50      private CasualtyAbility(final CasualtyAbility ability) {
51          super(ability);
52          this.additionalCost = ability.additionalCost;
53          this.promptString = ability.promptString;
54          this.rule = ability.rule;
55      }
56  
57      public void resetCasualty() {
58          if (additionalCost != null) {
59              additionalCost.reset();
60          }
61      }
62  
63      @Override
64      public CasualtyAbility copy() {
65          return new CasualtyAbility(this);
66      }
67  
68      @Override
69      public void addOptionalAdditionalCosts(Ability ability, Game game) {
70          if (!(ability instanceof SpellAbility)) {
71              return;
72          }
73  
74          Player player = game.getPlayer(ability.getControllerId());
75          if (player == null) {
76              return;
77          }
78  
79          this.resetCasualty();
80          boolean canPay = additionalCost.canPay(ability, this, ability.getControllerId(), game);
81          if (!canPay || !player.chooseUse(Outcome.Sacrifice, promptString, ability, game)) {
82              return;
83          }
84  
85          additionalCost.activate();
86          for (Cost cost : ((Costs<Cost>) additionalCost)) {
87              ability.getCosts().add(cost.copy());
88          }
89          game.fireReflexiveTriggeredAbility(new ReflexiveTriggeredAbility(
90                  new CopySourceSpellEffect(), false, "when you do, copy this spell"
91          ), ability);
92      }
93  
94      @Override
95      public String getCastMessageSuffix() {
96          return additionalCost == null ? "" : additionalCost.getCastSuffixMessage(0);
97      }
98  
99      @Override
100     public String getRule() {
101         return rule;
102     }
103 }

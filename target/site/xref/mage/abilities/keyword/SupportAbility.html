2   package mage.abilities.keyword;
3   
4   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
5   import mage.abilities.effects.keyword.SupportEffect;
6   import mage.cards.Card;
7   import mage.filter.common.FilterCreaturePermanent;
8   import mage.filter.predicate.mageobject.AnotherPredicate;
9   import mage.target.common.TargetCreaturePermanent;
10  
11  /**
12   * 701.32. Support
13   *
14   * 701.32a “Support N” on a permanent means “Put a +1/+1 counter on each of up
15   * to N other target creatures.” “Support N” on an instant or sorcery spell
16   * means “Put a +1/+1 counter on each of up to N target creatures.”
17   *
18   * @author LevelX2
19   */
20  public class SupportAbility extends EntersBattlefieldTriggeredAbility {
21  
22      /*
23       * For enchantments, the text should not include the word "other".
24       * The otherPermanent choice removes the word "other" from rule text creation.
25       */
26      public SupportAbility(Card card, int amount, boolean otherPermanent) {
27          super(new SupportEffect(card, amount, otherPermanent));
28          if (!card.isInstantOrSorcery()) {
29              FilterCreaturePermanent filter = new FilterCreaturePermanent("creatures");
30              if (card.isCreature()) {
31                  filter.add(AnotherPredicate.instance);
32                  filter.setMessage("other target creatures");
33              }
34              addTarget(new TargetCreaturePermanent(0, amount, filter, false));
35          }
36      }
37  
38  
39      public SupportAbility(Card card, int amount){ this( card, amount, true); }
40  
41  
42      public SupportAbility(final SupportAbility ability) {
43          super(ability);
44      }
45  
46      @Override
47      public SupportAbility copy() {
48          return new SupportAbility(this);
49      }
50  }

1   package mage.abilities.keyword;

3   import mage.abilities.TriggeredAbilityImpl;
4   import mage.abilities.effects.Effect;
5   import mage.constants.AbilityWord;
6   import mage.constants.Zone;
7   import mage.game.Game;
8   import mage.game.events.GameEvent;
9   import mage.game.permanent.Permanent;
10  
11  import java.util.UUID;
12  
13  /**
14   * @author TheElk801
15   */
16  public class PackTacticsAbility extends TriggeredAbilityImpl {
17  
18      public PackTacticsAbility(Effect effect) {
19          super(Zone.BATTLEFIELD, effect, false);
20          this.setAbilityWord(AbilityWord.PACK_TACTICS);
21          setTriggerPhrase("Whenever {this} attacks, if you attacked with creatures with total power 6 or greater this combat, ");
22      }
23  
24      private PackTacticsAbility(final PackTacticsAbility ability) {
25          super(ability);
26      }
27  
28      @Override
29      public boolean checkEventType(GameEvent event, Game game) {
30          return event.getType() == GameEvent.EventType.ATTACKER_DECLARED;
31      }
32  
33      @Override
34      public boolean checkTrigger(GameEvent event, Game game) {
35          if (getSourceId().equals(event.getSourceId())) {
36              int power = 0;
37              for (UUID attackerId : game.getCombat().getAttackers()) {
38                  Permanent attacker = game.getPermanent(attackerId);
39                  if (attacker != null) {
40                      power += attacker.getPower().getValue();
41                  }
42              }
43              return power >= 6;
44          }
45          return false;
46      }
47  
48      @Override
49      public PackTacticsAbility copy() {
50          return new PackTacticsAbility(this);
51      }
52  }

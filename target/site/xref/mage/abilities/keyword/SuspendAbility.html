31  /**
32   * 502.59. Suspend
33   * <p>
34   * 502.59a Suspend is a keyword that represents three abilities. The first is a
35   * static ability that functions while the card with suspend is in a player's
36   * hand. The second and third are triggered abilities that function in the
37   * removed-from-the-game zone. "Suspend N--[cost]" means "If you could play this
38   * card from your hand, you may pay [cost] and remove it from the game with N
39   * time counters on it. This is a special action that doesn't use the stack,"
40   * and "At the beginning of your upkeep, if this card is suspended, remove a
41   * time counter from it," and "When the last time counter is removed from this
42   * card, if it's removed from the game, play it without paying its mana cost if
43   * able. If you can't, it remains removed from the game. If you play it this way
44   * and it's a creature, it gains haste until you lose control of it."
45   * <p>
46   * 502.59b A card is "suspended" if it's in the removed-from-the-game zone, has
47   * suspend, and has a time counter on it.
48   * <p>
49   * 502.59c Playing a spell as an effect of its suspend ability follows the rules
50   * for paying alternative costs in rules 409.1b and 409.1f-h.
51   * <p>
52   * The phrase "if you could play this card from your hand" checks only for
53   * timing restrictions and permissions. This includes both what's inherent in
54   * the card's type (for example, if the card with suspend is a creature, it must
55   * be your main phase and the stack must be empty) and what's imposed by other
56   * abilities, such as flash or Meddling Mage's ability. Whether you could
57   * actually follow all steps in playing the card is irrelevant. If the card is
58   * impossible to play due to a lack of legal targets or an unpayable mana cost,
59   * for example, it may still be removed from the game with suspend.
60   * <p>
61   * Removing a card from the game with its suspend ability is not playing that
62   * card. This action doesn't use the stack and can't be responded to.
63   * <p>
64   * If a spell with suspend has targets, the targets are chosen when the spell is
65   * played, not when it's removed from the game.
66   * <p>
67   * If the first triggered ability of suspend is countered, no time counter is
68   * removed. The ability will trigger again during its owner's next upkeep.
69   * <p>
70   * When the last time counter is removed from a suspended card, the second
71   * triggered ability of suspend will trigger. It doesn't matter why the time
72   * counter was removed or whose effect removed it. (The _Time Spiral_ reminder
73   * text is misleading on this point.)
74   * <p>
75   * If the second triggered ability of suspend is countered, the card can't be
76   * played. It remains in the removed-from-the-game zone without any time
77   * counters on it for the rest of the game, and it's no longer considered
78   * suspended.
79   * <p>
80   * If the second triggered ability of suspend resolves, the card's owner must
81   * play the spell if possible, even if that player doesn't want to. Normal
82   * timing considerations for the spell are ignored (for example, if the
83   * suspended card is a creature and this ability resolves during your upkeep,
84   * you're able to play the card), but other play restrictions are not ignored.
85   * <p>
86   * If the second triggered ability of suspend resolves and the suspended card
87   * can't be played due to a lack of legal targets or a play restriction, for
88   * example, it remains in the removed-from-the-game zone without any time
89   * counters on it for the rest of the game, and it's no longer considered
90   * suspended.
91   * <p>
92   * As the second triggered ability of suspend resolves, if playing the suspended
93   * card involves an additional cost, the card's owner must pay that cost if
94   * able. If they can't, the card remains removed from the game. If the
95   * additional cost includes mana, the situation is more complex. If the player
96   * has enough mana in their mana pool to pay the cost, that player must do so.
97   * If the player can't possibly pay the cost, the card remains removed from the
98   * game. However, if the player has the means to produce enough mana to pay the
99   * cost, then they have a choice: The player may play the spell, produce mana,
100  * and pay the cost. Or the player may choose to play no mana abilities, thus
101  * making the card impossible to play because the additional mana can't be paid.
102  * <p>
103  * A creature played via suspend comes into play with haste. It still has haste
104  * after the first turn it's in play as long as the same player controls it. As
105  * soon as another player takes control of it, it loses haste.
106  *
107  * @author LevelX2
108  */
109 public class SuspendAbility extends SpecialAction {
110 
111     private final String ruleText;
112     private boolean gainedTemporary;
113 
114     /**
115      * Gives the card the SuspendAbility
116      *
117      * @param suspend - amount of time counters, if Integer.MAX_VALUE is set
118      *                there will be {X} costs and X counters added
119      * @param cost    - null is used for temporary gained suspend ability
120      * @param card    - card that has the suspend ability
121      */
122     public SuspendAbility(int suspend, ManaCost cost, Card card) {
123         this(suspend, cost, card, false);
124     }
125 
126     public SuspendAbility(int suspend, ManaCost cost, Card card, boolean shortRule) {
127         super(Zone.HAND);
128         this.addCost(cost);
129         this.addEffect(new SuspendExileEffect(suspend));
130         this.usesStack = false;
131         if (suspend == Integer.MAX_VALUE) {
132             VariableManaCost xCosts = new VariableManaCost(VariableCostType.ALTERNATIVE);
133             xCosts.setMinX(1);
134             this.addManaCost(xCosts);
135             cost = new ManaCostsImpl<>("{X}" + cost.getText());
136         }
137         StringBuilder sb = new StringBuilder("Suspend ");
138         if (cost != null) {
139             sb.append(suspend == Integer.MAX_VALUE ? "X" : suspend).append("&mdash;")
140                     .append(cost.getText()).append(suspend
141                             == Integer.MAX_VALUE ? ". X can't be 0." : "");
142             if (!shortRule) {
143                 sb.append(" <i>(Rather than cast this card from your hand, pay ")
144                         .append(cost.getText())
145                         .append(" and exile it with ")
146                         .append((suspend == 1 ? "a time counter" : (suspend == Integer.MAX_VALUE
147                                 ? "X time counters" : suspend + " time counters")))
148                         .append(" on it.")
149                         .append(" At the beginning of your upkeep, remove a time counter. "
150                                 + "When the last is removed, cast it without paying its mana cost.")
151                         .append(card.isCreature() ? " If you play it this way and it's a creature, "
152                                 + "it gains haste until you lose control of it." : "")
153                         .append(")</i>");
154             }
155             if (card.getManaCost().isEmpty()) {
156                 setRuleAtTheTop(true);
157             }
158             addSubAbility(new SuspendBeginningOfUpkeepInterveningIfTriggeredAbility());
159             addSubAbility(new SuspendPlayCardAbility());
160         }
161         ruleText = sb.toString();
162     }
163 
164     /**
165      * Adds suspend to a card that does not have it regularly e.g. Epochrasite
166      * or added by Jhoira of the Ghitu
167      *
168      * @param card
169      * @param source
170      * @param game
171      */
172     public static void addSuspendTemporaryToCard(Card card, Ability source, Game game) {
173         SuspendAbility ability = new SuspendAbility(0, null, card, false);
174         ability.setSourceId(card.getId());
175         ability.setControllerId(card.getOwnerId());
176         game.getState().addOtherAbility(card, ability);
177 
178         SuspendBeginningOfUpkeepInterveningIfTriggeredAbility ability1
179                 = new SuspendBeginningOfUpkeepInterveningIfTriggeredAbility();
180         ability1.setSourceId(card.getId());
181         ability1.setControllerId(card.getOwnerId());
182         game.getState().addOtherAbility(card, ability1);
183 
184         SuspendPlayCardAbility ability2 = new SuspendPlayCardAbility();
185         ability2.setSourceId(card.getId());
186         ability2.setControllerId(card.getOwnerId());
187         game.getState().addOtherAbility(card, ability2);
188     }
189 
190     public static UUID getSuspendExileId(UUID controllerId, Game game) {
191         UUID exileId = (UUID) game.getState().getValue("SuspendExileId" + controllerId.toString());
192         if (exileId == null) {
193             exileId = UUID.randomUUID();
194             game.getState().setValue("SuspendExileId" + controllerId, exileId);
195         }
196         return exileId;
197     }
198 
199     public SuspendAbility(SuspendAbility ability) {
200         super(ability);
201         this.ruleText = ability.getRule();
202         this.gainedTemporary = ability.gainedTemporary;
203     }
204 
205     @Override
206     public ActivationStatus canActivate(UUID playerId, Game game) {
207         // suspend can only be activated from a hand
208         if (game.getState().getZone(getSourceId()) != Zone.HAND) {
209             return ActivationStatus.getFalse();
210         }
211         // suspend uses card's timing restriction
212         Card card = game.getCard(getSourceId());
213         if (card == null) {
214             return ActivationStatus.getFalse();
215         }
216         if (!card.getSpellAbility().spellCanBeActivatedRegularlyNow(playerId, game)) {
217             return ActivationStatus.getFalse();
218         }
219         return super.canActivate(playerId, game);
220     }
221 
222     @Override
223     public String getRule() {
224         return ruleText;
225     }
226 
227     public boolean isGainedTemporary() {
228         return gainedTemporary;
229     }
230 
231     @Override
232     public SuspendAbility copy() {
233         return new SuspendAbility(this);
234     }

238 class SuspendExileEffect extends OneShotEffect {
239 
240     private int suspend;
241 
242     public SuspendExileEffect(int suspend) {
243         super(Outcome.PutCardInPlay);
244         this.staticText = new StringBuilder("Suspend ").append(suspend
245                 == Integer.MAX_VALUE ? "X" : suspend).toString();
246         this.suspend = suspend;
247     }
248 
249     public SuspendExileEffect(final SuspendExileEffect effect) {
250         super(effect);
251         this.suspend = effect.suspend;
252     }
253 
254     @Override
255     public SuspendExileEffect copy() {
256         return new SuspendExileEffect(this);
257     }
258 
259     @Override
260     public boolean apply(Game game, Ability source) {
261         Card card = game.getCard(source.getSourceId());
262         Player controller = game.getPlayer(source.getControllerId());
263         if (card != null && controller != null) {
264             UUID exileId = SuspendAbility.getSuspendExileId(controller.getId(), game);
265             if (controller.moveCardToExileWithInfo(card, exileId, "Suspended cards of "
266                     + controller.getName(), source, game, Zone.HAND, true)) {
267                 if (suspend == Integer.MAX_VALUE) {
268                     suspend = source.getManaCostsToPay().getX();
269                 }
270                 card.addCounters(CounterType.TIME.createInstance(suspend), source.getControllerId(), source, game);
271                 if (!game.isSimulation()) {
272                     game.informPlayers(controller.getLogName()
273                             + " suspends (" + suspend + ") " + card.getLogName());
274                 }
275                 return true;
276             }
277         }
278         return false;
279     }

298     @Override
299     public boolean checkTrigger(GameEvent event, Game game) {
300         if (event.getTargetId().equals(this.getSourceId())) {
301             Card card = game.getCard(this.getSourceId());
302             return card != null
303                     && game.getState().getZone(card.getId()) == Zone.EXILED
304                     && card.getCounters(game).getCount(CounterType.TIME) == 0;
305         }
306         return false;
307     }

311         return "When the last time counter is removed from this card ({this}), "
312                 + "if it's removed from the game, ";

321 class SuspendPlayCardEffect extends OneShotEffect {
322 
323     public SuspendPlayCardEffect() {
324         super(Outcome.PlayForFree);
325         this.staticText = "play it without paying its mana cost if able. "
326                 + "If you can't, it remains removed from the game";
327     }
328 
329     public SuspendPlayCardEffect(final SuspendPlayCardEffect effect) {
330         super(effect);
331     }
332 
333     @Override
334     public SuspendPlayCardEffect copy() {
335         return new SuspendPlayCardEffect(this);
336     }
337 
338     @Override
339     public boolean apply(Game game, Ability source) {
340         Player player = game.getPlayer(source.getControllerId());
341         Card card = game.getCard(source.getSourceId());
342         if (player != null && card != null) {
343             // remove temporary suspend ability (used e.g. for Epochrasite)
344             // TODO: isGainedTemporary is not set or use in other places, so it can be deleted?!
345             List<Ability> abilitiesToRemove = new ArrayList<>();
346             for (Ability ability : card.getAbilities(game)) {
347                 if (ability instanceof SuspendAbility) {
348                     if (((SuspendAbility) ability).isGainedTemporary()) {
349                         abilitiesToRemove.add(ability);
350                     }
351                 }
352             }
353             if (!abilitiesToRemove.isEmpty()) {
354                 for (Ability ability : card.getAbilities(game)) {
355                     if (ability instanceof SuspendBeginningOfUpkeepInterveningIfTriggeredAbility
356                             || ability instanceof SuspendPlayCardAbility) {
357                         abilitiesToRemove.add(ability);
358                     }
359                 }
360                 // remove the abilities from the card
361                 // TODO: will not work with Adventure Cards and another auto-generated abilities list
362                 // TODO: is it work after blink or return to hand?
363                 /*
364                  bug example:
365                  Epochrasite bug: It comes out of suspend, is cast and enters the battlefield. THEN if it's returned to
366                  its owner's hand from battlefield, the bounced Epochrasite can't be cast for the rest of the game.
367                  */
368                 card.getAbilities().removeAll(abilitiesToRemove);
369             }
370             // cast the card for free
371             game.getState().setValue("PlayFromNotOwnHandZone" + card.getId(), Boolean.TRUE);
372             Boolean cardWasCast = player.cast(player.chooseAbilityForCast(card, game, true),
373                     game, true, new ApprovingObject(source, game));
374             game.getState().setValue("PlayFromNotOwnHandZone" + card.getId(), null);
375             if (cardWasCast) {
376                 if (card.isCreature(game)) {
377                     ContinuousEffect effect = new GainHasteEffect();
378                     effect.setTargetPointer(new FixedTarget(card.getId(), card.getZoneChangeCounter(game) + 1));
379                     game.addEffect(effect, source);
380                 }
381                 return true;
382             }
383         }
384         return false;
385     }

388 class GainHasteEffect extends ContinuousEffectImpl {
389 
390     private UUID suspendController;
391 
392     public GainHasteEffect() {
393         super(Duration.Custom, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
394         staticText = "If you play it this way and it's a creature, it gains haste until you lose control of it";
395     }
396 
397     public GainHasteEffect(final GainHasteEffect effect) {
398         super(effect);
399         this.suspendController = effect.suspendController;
400     }
401 
402     @Override
403     public GainHasteEffect copy() {
404         return new GainHasteEffect(this);
405     }
406 
407     @Override
408     public boolean apply(Game game, Ability source) {
409         if (suspendController == null) {
410             suspendController = source.getControllerId();
411         }
412         Permanent permanent = game.getPermanent(getTargetPointer().getFirst(game, source));
413         if (permanent != null) {
414             if (suspendController.equals(source.getControllerId())) {
415                 permanent.addAbility(HasteAbility.getInstance(), source.getSourceId(), game);
416             } else {
417                 this.discard();
418             }
419             return true;
420         }
421         if (game.getState().getZoneChangeCounter(((FixedTarget) getTargetPointer()).getTarget())
422                 >= ((FixedTarget) getTargetPointer()).getZoneChangeCounter()) {
423             this.discard();
424         }
425         return false;
426     }

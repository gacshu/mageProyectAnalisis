33  /**
34   * 702.65. Delve
35   * <p>
36   * 702.65a Delve is a static ability that functions while the spell with delve is on the stack. “Delve” means “For
37   * each generic mana in this spell’s total cost, you may exile a card from your graveyard rather than pay that mana.”
38   * <p>
39   * 702.65b The delve ability isn’t an additional or alternative cost and applies only after the total cost of the spell
40   * with delve is determined.
41   * <p>
42   * 702.65c Multiple instances of delve on the same spell are redundant.
43   * <p>
44   * The rules for delve have changed slightly since it was last in an expansion. Previously, delve reduced the cost
45   * to cast a spell. Under the current rules, you exile cards from your graveyard at the same time you pay the spell’s
46   * cost. Exiling a card this way is simply another way to pay that cost. (This is similar to the change made to
47   * convoke for the Magic 2015 Core Set.)
48   * <p>
49   * You can’t exile cards to pay for the colored mana requirements of a spell with delve.
50   * <p>
51   * You can’t exile more cards than the generic mana requirement of a spell with delve. For example, you can’t exile more
52   * than nine cards from your graveyard to cast Dead Drop.
53   * <p>
54   * Because delve isn’t an alternative cost, it can be used in conjunction with alternative costs.
55   *
56   * @author LevelX2, JayDi85
57   * <p>
58   * TODO: Change card exiling to a way to pay mana costs, now it's maybe not
59   * possible to pay costs from effects that increase the mana costs.
60   * If it real bug then possible fix: choose cards on apply like convoke and improvise, not as cost
61   */
62  public class DelveAbility extends SimpleStaticAbility implements AlternateManaPaymentAbility {
63  
64      private static final DynamicValue cardsInGraveyard = new CardsInControllerGraveyardCount();
65  
66      public DelveAbility() {
67          super(Zone.ALL, null);
68          this.setRuleAtTheTop(true);
69          this.addHint(new ValueHint("Cards in your graveyard", cardsInGraveyard));
70      }
71  
72      public DelveAbility(final DelveAbility ability) {
73          super(ability);
74      }
75  
76      @Override
77      public DelveAbility copy() {
78          return new DelveAbility(this);
79      }
80  
81      @Override
82      public String getRule() {
83          return "Delve <i>(Each card you exile from your graveyard while casting this spell pays for {1}.)</i>";
84      }
85  
86      @Override
87      public ActivationManaAbilityStep useOnActivationManaAbilityStep() {
88          return ActivationManaAbilityStep.AFTER;
89      }
90  
91      @Override
92      public void addSpecialAction(Ability source, Game game, ManaCost unpaid) {
93          Player controller = game.getPlayer(source.getControllerId());
94          if (controller != null && !controller.getGraveyard().isEmpty()) {
95              if (source.getAbilityType() == AbilityType.SPELL && unpaid.getMana().getGeneric() > 0) {
96                  SpecialAction specialAction = new DelveSpecialAction(this);
97                  specialAction.setControllerId(source.getControllerId());
98                  specialAction.setSourceId(source.getSourceId());
99                  int unpaidAmount = unpaid.getMana().getGeneric();
100                 if (!controller.getManaPool().isAutoPayment() && unpaidAmount > 1) {
101                     unpaidAmount = 1;
102                 }
103                 specialAction.addCost(new ExileFromGraveCost(new TargetCardInYourGraveyard(
104                         0, Math.min(controller.getGraveyard().size(), unpaidAmount),
105                         new FilterCard("cards from your graveyard"), true)));
106                 if (specialAction.canActivate(source.getControllerId(), game).canActivate()) {
107                     game.getState().getSpecialActions().add(specialAction);
108                 }
109             }
110         }
111     }
112 
113     @Override
114     public ManaOptions getManaOptions(Ability source, Game game, ManaCost unpaid) {
115         ManaOptions options = new ManaOptions();
116         Player controller = game.getPlayer(source.getControllerId());
117         int graveCount = cardsInGraveyard.calculate(game, source, null);
118         if (controller != null && graveCount > 0) {
119             options.addMana(Mana.GenericMana(Math.min(unpaid.getMana().getGeneric(), graveCount)));
120         }
121         return options;
122     }
123 }

128         super(Zone.ALL, manaAbility);

143 class DelveEffect extends OneShotEffect {
144 
145     public DelveEffect() {
146         super(Outcome.Benefit);
147         this.staticText = "Delve (Each card you exile from your graveyard while casting this spell pays for {1}.)";
148     }
149 
150     public DelveEffect(final DelveEffect effect) {
151         super(effect);
152     }
153 
154     @Override
155     public DelveEffect copy() {
156         return new DelveEffect(this);
157     }
158 
159     @Override
160     public boolean apply(Game game, Ability source) {
161         Player controller = game.getPlayer(source.getControllerId());
162         Spell spell = game.getStack().getSpell(source.getSourceId());
163         if (controller != null && spell != null) {
164             ExileFromGraveCost exileFromGraveCost = (ExileFromGraveCost) source.getCosts().get(0);
165             List<Card> exiledCards = exileFromGraveCost.getExiledCards();
166             if (!exiledCards.isEmpty()) {
167                 Cards toDelve = new CardsImpl();
168                 for (Card card : exiledCards) {
169                     toDelve.add(card);
170                 }
171                 ManaPool manaPool = controller.getManaPool();
172                 manaPool.addMana(Mana.ColorlessMana(toDelve.size()), game, source);
173                 manaPool.unlockManaType(ManaType.COLORLESS);
174                 String keyString = CardUtil.getCardZoneString("delvedCards", source.getSourceId(), game);
175                 @SuppressWarnings("unchecked")
176                 Cards delvedCards = (Cards) game.getState().getValue(keyString);
177                 if (delvedCards == null) {
178                     game.getState().setValue(keyString, toDelve);
179                 } else {
180                     delvedCards.addAll(toDelve);
181                 }
182 
183                 // can't use mana abilities after that (delve cost must be payed after mana abilities only)
184                 spell.setCurrentActivatingManaAbilitiesStep(ActivationManaAbilityStep.AFTER);
185             }
186             return true;
187         }
188         return false;
189     }

3   package mage.abilities.keyword;
4   
5   import mage.abilities.ActivatedAbilityImpl;
6   import mage.abilities.costs.mana.ManaCosts;
7   import mage.abilities.effects.common.counter.AddCountersSourceEffect;
8   import mage.constants.TimingRule;
9   import mage.constants.Zone;
10  import mage.counters.CounterType;
11  
12  /**
13   *
14   * @author BetaSteward_at_googlemail.com
15   */
16  public class LevelUpAbility extends ActivatedAbilityImpl {
17  
18      public LevelUpAbility(ManaCosts costs) {
19          super(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.LEVEL.createInstance()), costs);
20          this.timing = TimingRule.SORCERY;
21      }
22  
23      public LevelUpAbility(final LevelUpAbility ability) {
24          super(ability);
25      }
26  
27      @Override
28      public LevelUpAbility copy() {
29          return new LevelUpAbility(this);
30      }
31  
32      @Override
33      public String getRule() {
34          return new StringBuilder("Level up ").append(manaCostsToPay.getText())
35                  .append(" <i>(").append(manaCostsToPay.getText()).append(": Put a level counter on this. Level up only as a sorcery.)</i>").toString();
36      }
37  }

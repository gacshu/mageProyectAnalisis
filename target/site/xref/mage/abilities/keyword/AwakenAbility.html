30  public class AwakenAbility extends SpellAbility {
31  
32      private static final Logger logger = Logger.getLogger(AwakenAbility.class);
33  
34      static private String filterMessage = "a land you control to awake";
35  
36      private String rule;
37      private int awakenValue;
38  
39      public AwakenAbility(Card card, int awakenValue, String awakenCosts) {
40          super(card.getSpellAbility());
41          this.newId();
42          this.setCardName(card.getName() + " with awaken");
43          zone = Zone.HAND;
44          spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
45  
46          this.getManaCosts().clear();
47          this.getManaCostsToPay().clear();
48          this.addManaCost(new ManaCostsImpl<>(awakenCosts));
49  
50          this.addTarget(new TargetControlledPermanent(new FilterControlledLandPermanent(filterMessage)));
51          this.addEffect(new AwakenEffect());
52          this.awakenValue = awakenValue;
53          rule = "Awaken " + awakenValue + "&mdash;" + awakenCosts
54                  + " <i>(If you cast this spell for " + awakenCosts + ", also put "
55                  + CardUtil.numberToText(awakenValue, "a")
56                  + " +1/+1 counters on target land you control and it becomes a 0/0 Elemental creature with haste. It's still a land.)</i>";
57      }
58  
59      public AwakenAbility(final AwakenAbility ability) {
60          super(ability);
61          this.awakenValue = ability.awakenValue;
62          this.rule = ability.rule;
63      }
64  
65      @Override
66      public AwakenAbility copy() {
67          return new AwakenAbility(this);
68      }
69  
70      @Override
71      public String getRule(boolean all) {
72          return getRule();
73      }
74  
75      @Override
76      public String getRule() {
77          return rule;
78      }
79  
80      class AwakenEffect extends OneShotEffect {
81  
82          private AwakenEffect() {
83              super(Outcome.BoostCreature);
84              this.staticText = "put " + CardUtil.numberToText(awakenValue, "a") + " +1/+1 counters on target land you control";
85          }
86  
87          public AwakenEffect(final AwakenEffect effect) {
88              super(effect);
89          }
90  
91          @Override
92          public AwakenEffect copy() {
93              return new AwakenEffect(this);
94          }
95  
96          @Override
97          public boolean apply(Game game, Ability source) {
98              UUID targetId = null;
99              if (source != null && source.getTargets() != null) {
100                 for (Target target : source.getTargets()) {
101                     if (target.getFilter() != null && target.getFilter().getMessage().equals(filterMessage)) {
102                         targetId = target.getFirstTarget();
103                     }
104                 }
105                 if (targetId != null) {
106                     FixedTarget fixedTarget = new FixedTarget(targetId, game);
107                     ContinuousEffect continuousEffect = new BecomesCreatureTargetEffect(new AwakenElementalToken(), false, true, Duration.Custom);
108                     continuousEffect.setTargetPointer(fixedTarget);
109                     game.addEffect(continuousEffect, source);
110                     Effect effect = new AddCountersTargetEffect(CounterType.P1P1.createInstance(awakenValue));
111                     effect.setTargetPointer(fixedTarget);
112                     return effect.apply(game, source);
113                 }
114             } else // source should never be null, but we are seeing a lot of NPEs from this section
115                 if (source == null) {
116                     logger.fatal("Source was null in AwakenAbility: Create a bug report or fix the source code");
117                 } else if (source.getTargets() == null) {
118                     MageObject sourceObj = source.getSourceObject(game);
119                     if (sourceObj != null) {
120                         Class<? extends MageObject> sourceClass = sourceObj.getClass();
121                         if (sourceClass != null) {
122                             logger.fatal("getTargets was null in AwakenAbility for " + sourceClass.toString() + " : Create a bug report or fix the source code");
123                         }
124                     }
125                 }
126             return true;
127         }
128     }
129 
130 }

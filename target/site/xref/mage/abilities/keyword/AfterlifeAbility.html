1   package mage.abilities.keyword;
2   
3   import mage.abilities.common.DiesSourceTriggeredAbility;
4   import mage.abilities.effects.common.CreateTokenEffect;
5   import mage.game.permanent.token.WhiteBlackSpiritToken;
6   import mage.util.CardUtil;
7   
8   public class AfterlifeAbility extends DiesSourceTriggeredAbility {
9   
10      private final int tokenCount;
11  
12      public AfterlifeAbility(int tokenCount) {
13          super(new CreateTokenEffect(new WhiteBlackSpiritToken(), tokenCount), false);
14          this.tokenCount = tokenCount;
15      }
16  
17      public AfterlifeAbility(final AfterlifeAbility ability) {
18          super(ability);
19          this.tokenCount = ability.tokenCount;
20      }
21  
22      @Override
23      public String getRule() {
24          return "Afterlife " + tokenCount + " <i>(When this creature dies, create "
25                  + CardUtil.numberToText(tokenCount, "a")
26                  + " 1/1 white and black Spirit creature token"
27                  + (tokenCount > 1 ? "s" : "")
28                  + " with flying.)</i>";
29      }
30  
31      @Override
32      public AfterlifeAbility copy() {
33          return new AfterlifeAbility(this);
34      }
35  }

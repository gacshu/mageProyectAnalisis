28  public class BlitzAbility extends SpellAbility {
29  
30      public static final String BLITZ_ACTIVATION_VALUE_KEY = "blitzActivation";
31  
32      public BlitzAbility(Card card, String manaString) {
33          super(new ManaCostsImpl<>(manaString), card.getName() + " with Blitz");
34          this.spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
35          Ability ability = new EntersBattlefieldAbility(
36                  new GainAbilitySourceEffect(HasteAbility.getInstance(), Duration.Custom, false),
37                  BlitzedCondition.instance, "", ""
38          );
39          ability.addEffect(new GainAbilitySourceEffect(new DiesSourceTriggeredAbility(
40                  new DrawCardSourceControllerEffect(1)
41          ).setTriggerPhrase("When this creature dies, ")));
42          ability.addEffect(new BlitzAddDelayedTriggeredAbilityEffect());
43          ability.setRuleVisible(false);
44          addSubAbility(ability);
45          this.ruleAdditionalCostsVisible = false;
46          this.timing = TimingRule.SORCERY;
47      }
48  
49      private BlitzAbility(final BlitzAbility ability) {
50          super(ability);
51      }
52  
53      @Override
54      public BlitzAbility copy() {
55          return new BlitzAbility(this);
56      }
57  
58      @Override
59      public String getRule() {
60          StringBuilder sb = new StringBuilder("Blitz");
61          if (costs.isEmpty()) {
62              sb.append(' ');
63          } else {
64              sb.append("&mdash;");
65          }
66          sb.append(manaCosts.getText());
67          if (!costs.isEmpty()) {
68              sb.append(", ");
69              sb.append(costs.getText());
70              sb.append('.');
71          }
72          sb.append(" <i>(If you cast this spell for its blitz cost, it gains haste ");
73          sb.append("and \"When this creature dies, draw a card.\" ");
74          sb.append("Sacrifice it at the beginning of the next end step.)</i>");
75          return sb.toString();
76      }
77  
78      @Override
79      public boolean activate(Game game, boolean noMana) {
80          if (!super.activate(game, noMana)) {
81              return false;
82          }
83          Object obj = game.getState().getValue(BLITZ_ACTIVATION_VALUE_KEY + getSourceId());
84          List<Integer> blitzActivations;
85          if (obj != null) {
86              blitzActivations = (List<Integer>) obj;
87          } else {
88              blitzActivations = new ArrayList<>();
89              game.getState().setValue(BLITZ_ACTIVATION_VALUE_KEY + getSourceId(), blitzActivations);
90          }
91          blitzActivations.add(game.getState().getZoneChangeCounter(getSourceId()));
92          return true;
93      }
94  }
95  
96  class BlitzAddDelayedTriggeredAbilityEffect extends OneShotEffect {
97  
98      BlitzAddDelayedTriggeredAbilityEffect() {
99          super(Outcome.Benefit);
100     }
101 
102     private BlitzAddDelayedTriggeredAbilityEffect(final BlitzAddDelayedTriggeredAbilityEffect effect) {
103         super(effect);
104     }
105 
106     @Override
107     public BlitzAddDelayedTriggeredAbilityEffect copy() {
108         return new BlitzAddDelayedTriggeredAbilityEffect(this);
109     }
110 
111     @Override
112     public boolean apply(Game game, Ability source) {
113         if (game.getPermanentEntering(source.getSourceId()) == null) {
114             return false;
115         }
116         // init target pointer now because the Blitzed creature will only be returned from battlefield zone (now in entering state so zone change counter is not raised yet)
117         game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
118                 new SacrificeTargetEffect()
119                         .setText("sacrifice the blitzed creature")
120                         .setTargetPointer(new FixedTarget(source.getSourceId(), game.getState().getZoneChangeCounter(source.getSourceId()) + 1))
121         ), source);
122         return true;
123     }
124 }

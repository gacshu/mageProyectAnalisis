4   import mage.abilities.costs.mana.ManaCostsImpl;
5   import mage.cards.Card;
6   import mage.constants.SpellAbilityType;

9   import mage.players.Player;

11  
12  import java.util.ArrayList;
13  import java.util.List;
14  import java.util.UUID;

16  /**
17   * @author LevelX2
18   */
19  public class SurgeAbility extends SpellAbility {
20  
21      public static final String SURGE_ACTIVATION_VALUE_KEY = "surgeActivation";
22  
23      private String rule;
24  
25      public SurgeAbility(Card card, String surgeCosts) {
26          super(card.getSpellAbility());
27          this.newId();
28          this.setCardName(card.getName() + " with surge");
29          zone = Zone.HAND;
30          spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
31  
32          this.getManaCosts().clear();
33          this.getManaCostsToPay().clear();
34          this.addManaCost(new ManaCostsImpl<>(surgeCosts));
35  
36          this.setRuleAtTheTop(true);
37          this.rule = "Surge " + surgeCosts
38                  + " <i>(You may cast this spell for its surge cost if you or a teammate has cast another spell this turn.)</i>";
39      }
40  
41      public SurgeAbility(final SurgeAbility ability) {
42          super(ability);
43          this.rule = ability.rule;
44      }
45  
46      @Override
47      public ActivationStatus canActivate(UUID playerId, Game game) {
48          // check if controller or teammate has already cast a spell this turn
49          CastSpellLastTurnWatcher watcher = game.getState().getWatcher(CastSpellLastTurnWatcher.class);
50          if (watcher != null) {
51              Player player = game.getPlayer(playerId);
52              if (player != null) {
53                  for (UUID playerToCheckId : game.getState().getPlayersInRange(playerId, game)) {
54                      if (!player.hasOpponent(playerToCheckId, game)) {
55                          if (watcher.getAmountOfSpellsPlayerCastOnCurrentTurn(playerToCheckId) > 0
56                                  && super.canActivate(playerId, game).canActivate()) {
57                              return ActivationStatus.getTrue(this, game);
58                          }
59                      }
60                  }
61              }
62          }
63          return ActivationStatus.getFalse();
64      }
65  
66      @Override
67      @SuppressWarnings("unchecked")
68      public boolean activate(Game game, boolean noMana) {
69          if (super.activate(game, noMana)) {
70              List<Integer> surgeActivations = (ArrayList) game.getState().getValue(SURGE_ACTIVATION_VALUE_KEY + getSourceId());
71              if (surgeActivations == null) {
72                  surgeActivations = new ArrayList<>(); // zoneChangeCounter
73                  game.getState().setValue(SURGE_ACTIVATION_VALUE_KEY + getSourceId(), surgeActivations);
74              }
75              surgeActivations.add(game.getState().getZoneChangeCounter(getSourceId()));
76              return true;
77          }
78          return false;
79      }
80  
81      @Override
82      public SurgeAbility copy() {
83          return new SurgeAbility(this);
84      }
85  
86      @Override
87      public String getRule(boolean all) {
88          return getRule();
89      }
90  
91      @Override
92      public String getRule() {
93          return rule;
94      }
95  
96  }

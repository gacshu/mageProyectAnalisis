28  /**
29   * 702.46. Offering # 702.46a Offering is a static ability of a card that
30   * functions in any zone from which the card can be cast. "[Subtype] offering"
31   * means "You may cast this card any time you could cast an instant by
32   * sacrificing a [subtype] permanent. If you do, the total cost to cast this
33   * card is reduced by the sacrificed permanent's mana cost." #
34   * <p>
35   * 702.46b The permanent is sacrificed at the same time the spell is announced
36   * (see rule 601.2a). The total cost of the spell is reduced by the sacrificed
37   * permanent's mana cost (see rule 601.2e). #
38   * <p>
39   * 702.46c Generic mana in the sacrificed permanent's mana cost reduces generic
40   * mana in the total cost to cast the card with offering. Colored mana in the
41   * sacrificed permanent's mana cost reduces mana of the same color in the total
42   * cost to cast the card with offering. Colored mana in the sacrificed
43   * permanent's mana cost that doesn't match colored mana in the colored mana
44   * cost of the card with offering, or is in excess of the card's colored mana
45   * cost, reduces that much generic mana in the total cost. #
46   *
47   * @author LevelX2
48   */
49  public class OfferingAbility extends StaticAbility implements AlternateManaPaymentAbility {
50  
51      private final FilterControlledPermanent filter;
52  
53      /**
54       * @param subtype name of the subtype that can be offered
55       */
56      public OfferingAbility(FilterControlledPermanent filter) {
57          super(Zone.ALL, null);
58          this.filter = filter;
59          this.addEffect(new OfferingAsThoughEffect());
60      }
61  
62      public OfferingAbility(OfferingAbility ability) {
63          super(ability);
64          this.filter = ability.filter;
65      }
66  
67      @Override
68      public OfferingAbility copy() {
69          return new OfferingAbility(this);
70      }
71  
72      public FilterControlledPermanent getFilter() {
73          return filter;
74      }
75  
76      @Override
77      public String getRule(boolean all) {
78          String message = filter.getMessage();
79          return message + " offering <i>(You may cast this card any time you could cast an instant by sacrificing " +
80                  CardUtil.addArticle(message) + " and paying the difference in mana costs between this and the sacrificed " +
81                  message + ". Mana cost includes color.)</i>";
82      }
83  
84      @Override
85      public ActivationManaAbilityStep useOnActivationManaAbilityStep() {
86          return ActivationManaAbilityStep.NORMAL;
87      }
88  
89      @Override
90      public void addSpecialAction(Ability source, Game game, ManaCost unpaid) {
91          // No special Action
92      }
93  
94      @Override
95      public ManaOptions getManaOptions(Ability source, Game game, ManaCost unpaid) {
96          ManaOptions additionalManaOptionsForThisAbility = new ManaOptions();
97  
98          // Creatures from the offerd type
99          game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)
100                 .stream()
101                 .map(Card::getSpellAbility)
102                 .filter(Objects::nonNull)
103                 .forEach(spellAbility -> {
104                             ManaOptions manaOptionsForThisPermanent = new ManaOptions();
105                             for (ManaCost manaCost : spellAbility.getManaCosts()) {
106                                 if (manaCost instanceof HybridManaCost) {
107                                     ManaOptions manaOptionsForHybrid = new ManaOptions();
108                                     manaOptionsForHybrid.addAll(manaCost.getManaOptions());
109                                     manaOptionsForThisPermanent.addMana(manaOptionsForHybrid);
110                                 } else {
111                                     manaOptionsForThisPermanent.addMana(manaCost.getMana());
112                                 }
113                             }
114 
115                             additionalManaOptionsForThisAbility.addAll(manaOptionsForThisPermanent);
116                         }
117                 );
118 
119         additionalManaOptionsForThisAbility.removeFullyIncludedVariations();
120         return additionalManaOptionsForThisAbility;
121     }
122 }

124 class OfferingAsThoughEffect extends AsThoughEffectImpl {
125 
126     public OfferingAsThoughEffect() {
127         super(AsThoughEffectType.CAST_AS_INSTANT, Duration.EndOfGame, Outcome.Benefit);
128     }
129 
130     public OfferingAsThoughEffect(final OfferingAsThoughEffect effect) {
131         super(effect);
132     }
133 
134     @Override
135     public boolean apply(Game game, Ability source) {
136         return true;
137     }
138 
139     @Override
140     public OfferingAsThoughEffect copy() {
141         return new OfferingAsThoughEffect(this);
142     }
143 
144     @Override
145     public boolean applies(UUID sourceId, Ability source, UUID affectedControllerId, Game game) {
146         return false;
147     }
148 
149     @Override
150     public boolean applies(UUID sourceId, Ability affectedAbility, Ability source, Game game, UUID playerId) {
151         if (sourceId.equals(source.getSourceId())) {
152             Card card = game.getCard(sourceId);
153             if (card == null || !card.isOwnedBy(source.getControllerId())) {
154                 return false;
155             }
156             // because can activate is always called twice, result from first call will be used
157             Object object = game.getState().getValue("offering_" + card.getId());
158             if (object != null && object.equals(true)) {
159                 Object alreadyConfirmed = game.getState().getValue("offering_ok_" + card.getId());
160                 game.getState().setValue("offering_" + card.getId(), null);
161                 game.getState().setValue("offering_ok_" + card.getId(), null);
162                 return alreadyConfirmed != null;
163             } else {
164                 // first call -> remove previous Ids
165                 game.getState().setValue("offering_Id_" + card.getId(), null);
166             }
167 
168             if (game.getBattlefield().count(((OfferingAbility) source).getFilter(), source.getControllerId(), source, game) > 0) {
169 
170                 if (game.inCheckPlayableState()) {
171                     return true;
172                 }
173                 FilterControlledPermanent filter = ((OfferingAbility) source).getFilter();
174                 Card spellToCast = game.getCard(source.getSourceId());
175                 if (spellToCast == null) {
176                     return false;
177                 }
178                 Player player = game.getPlayer(source.getControllerId());
179                 if (player != null
180                         && player.chooseUse(Outcome.Benefit, "Offer a " + filter.getMessage() + " to cast " + spellToCast.getName() + '?', source, game)) {
181                     Target target = new TargetPermanent(1, 1, filter, true);
182                     player.chooseTarget(Outcome.Sacrifice, target, source, game);
183                     if (!target.isChosen()) {
184                         return false;
185                     }
186                     game.getState().setValue("offering_" + card.getId(), true);
187                     Permanent offer = game.getPermanent(target.getFirstTarget());
188                     if (offer != null) {
189                         UUID activationId = UUID.randomUUID();
190                         OfferingCostReductionEffect effect = new OfferingCostReductionEffect(activationId);
191                         effect.setTargetPointer(new FixedTarget(offer, game));
192                         game.addEffect(effect, source);
193                         game.getState().setValue("offering_ok_" + card.getId(), true);
194                         game.getState().setValue("offering_Id_" + card.getId(), activationId);
195                         game.informPlayers(player.getLogName() + " announces to offer "
196                                 + offer.getLogName() + " to cast "
197                                 + GameLog.getColoredObjectName(spellToCast));// No id name to prevent to offer hand card knowledge after cancel casting
198                         return true;
199                     }
200                 } else {
201                     game.getState().setValue("offering_" + card.getId(), true);
202                 }
203             }
204         }
205         return false;
206     }
207 }

209 class OfferingCostReductionEffect extends CostModificationEffectImpl {
210 
211     private final UUID activationId;
212 
213     OfferingCostReductionEffect(UUID activationId) {
214         super(Duration.OneUse, Outcome.Benefit, CostModificationType.REDUCE_COST);
215         this.activationId = activationId;
216         staticText = "mana costs reduction from offering";
217     }
218 
219     OfferingCostReductionEffect(OfferingCostReductionEffect effect) {
220         super(effect);
221         this.activationId = effect.activationId;
222     }
223 
224     @Override
225     public boolean apply(Game game, Ability source, Ability abilityToModify) {
226         Permanent toOffer = game.getPermanent(getTargetPointer().getFirst(game, source));
227         if (toOffer != null) {
228             toOffer.sacrifice(source, game);
229             CardUtil.reduceCost((SpellAbility) abilityToModify, toOffer.getSpellAbility().getManaCosts());
230         }
231         game.getState().setValue("offering_" + source.getSourceId(), null);
232         game.getState().setValue("offering_ok_" + source.getSourceId(), null);
233         this.discard();
234         return true;
235     }
236 
237     @Override
238     public boolean applies(Ability abilityToModify, Ability source, Game game) {
239         if (game.inCheckPlayableState()) { // Cost modifaction does not work correctly for checking available spells
240             return false;
241         }
242         if (abilityToModify.getSourceId().equals(source.getSourceId())
243                 && abilityToModify instanceof SpellAbility) {
244             Card card = game.getCard(source.getSourceId());
245             if (card != null) {
246                 Object object = game.getState().getValue("offering_Id_" + card.getId());
247                 if (object != null && object.equals(this.activationId) && getTargetPointer().getFirst(game, source) != null) {
248                     return true;
249                 }
250             }
251             // to target to offer, no correct activation ID this effect is no longer valid
252             this.discard();
253         }
254         return false;
255     }
256 
257     @Override
258     public OfferingCostReductionEffect copy() {
259         return new OfferingCostReductionEffect(this);
260     }

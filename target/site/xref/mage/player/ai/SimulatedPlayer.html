109     //add a generic mana cost for each amount possible
110     protected void simulateVariableCosts(Ability ability, Game game) {
111         int numAvailable = getAvailableManaProducers(game).size() - ability.getManaCosts().manaValue();
112         int start = 0;
113         if (!(ability instanceof SpellAbility)) {
114             //only use x=0 on spell abilities
115             if (numAvailable == 0)
116                 return;
117             else
118                 start = 1;
119         }
120         for (int i = start; i < numAvailable; i++) {
121             Ability newAbility = ability.copy();
122             newAbility.getManaCostsToPay().add(new GenericManaCost(i));
123             allActions.add(newAbility);
124         }
125     }

141     public List<Combat> addAttackers(Game game) {
142         Map<Integer, Combat> engagements = new HashMap<>();
143         //useful only for two player games - will only attack first opponent
144         UUID defenderId = game.getOpponents(playerId).iterator().next();
145         List<Permanent> attackersList = super.getAvailableAttackers(defenderId, game);
146         //use binary digits to calculate powerset of attackers
147         int powerElements = (int) Math.pow(2, attackersList.size());
148         StringBuilder binary = new StringBuilder();
149         for (int i = powerElements - 1; i >= 0; i--) {
150             Game sim = game.copy();
151             binary.setLength(0);
152             binary.append(Integer.toBinaryString(i));
153             while (binary.length() < attackersList.size()) {
154                 binary.insert(0, '0');
155             }
156             for (int j = 0; j < attackersList.size(); j++) {
157                 if (binary.charAt(j) == '1') {
158                     setStoredBookmark(sim.bookmarkState()); // makes it possible to UNDO a declared attacker with costs from e.g. Propaganda
159                     if (!sim.getCombat().declareAttacker(attackersList.get(j).getId(), defenderId, playerId, sim)) {
160                         sim.undo(playerId);
161                     }                    
162                 }
163             }
164             if (engagements.put(sim.getCombat().getValue().hashCode(), sim.getCombat()) != null) {
165                 logger.debug("simulating -- found redundant attack combination");
166             }
167             else if (logger.isDebugEnabled()) {
168                 logger.debug("simulating -- attack:" + sim.getCombat().getGroups().size());
169             }
170         }
171         return new ArrayList<>(engagements.values());
172     }

174     public List<Combat> addBlockers(Game game) {
175         Map<Integer, Combat> engagements = new HashMap<>();
176         int numGroups = game.getCombat().getGroups().size();
177         if (numGroups == 0) return new ArrayList<>();
178 
179         //add a node with no blockers
180         Game sim = game.copy();
181         engagements.put(sim.getCombat().getValue().hashCode(), sim.getCombat());
182         sim.fireEvent(GameEvent.getEvent(GameEvent.EventType.DECLARED_BLOCKERS, playerId, playerId));
183 
184         List<Permanent> blockers = getAvailableBlockers(game);
185         addBlocker(game, blockers, engagements);
186 
187         return new ArrayList<>(engagements.values());
188     }

190     protected void addBlocker(Game game, List<Permanent> blockers, Map<Integer, Combat> engagements) {
191         if (blockers.isEmpty())
192             return;
193         int numGroups = game.getCombat().getGroups().size();
194         //try to block each attacker with each potential blocker
195         Permanent blocker = blockers.get(0);
196         if (logger.isDebugEnabled())
197             logger.debug("simulating -- block:" + blocker);
198         List<Permanent> remaining = remove(blockers, blocker);
199         for (int i = 0; i < numGroups; i++) {
200             if (game.getCombat().getGroups().get(i).canBlock(blocker, game)) {
201                 Game sim = game.copy();
202                 sim.getCombat().getGroups().get(i).addBlocker(blocker.getId(), playerId, sim);
203                 if (engagements.put(sim.getCombat().getValue().hashCode(), sim.getCombat()) != null)
204                     logger.debug("simulating -- found redundant block combination");
205                 addBlocker(sim, remaining, engagements);  // and recurse minus the used blocker
206             }
207         }
208         addBlocker(game, remaining, engagements);
209     }

211     @Override
212     public boolean triggerAbility(TriggeredAbility source, Game game) {
213         Ability ability = source.copy();
214         List<Ability> options = getPlayableOptions(ability, game);
215         if (options.isEmpty()) {
216             if (logger.isDebugEnabled())
217                 logger.debug("simulating -- triggered ability:" + ability);
218             game.getStack().push(new StackAbility(ability, playerId));
219             if (ability.activate(game, false) && ability.isUsesStack()) {
220                 game.fireEvent(new GameEvent(GameEvent.EventType.TRIGGERED_ABILITY, ability.getId(), ability, ability.getControllerId()));
221             }
222             game.applyEffects();
223             game.getPlayers().resetPassed();
224         }
225         else {
226             SimulationNode parent = (SimulationNode) game.getCustomData();
227             if (parent.getDepth() == maxDepth) return true;
228             logger.debug(indent(parent.getDepth()) + "simulating -- triggered ability - adding children:" + options.size());
229             for (Ability option: options) {
230                 addAbilityNode(parent, option, game);
231             }
232         }
233         return true;
234     }

236     protected void addAbilityNode(SimulationNode parent, Ability ability, Game game) {
237         Game sim = game.copy();
238         sim.getStack().push(new StackAbility(ability, playerId));
239         ability.activate(sim, false);
240         if (ability.activate(sim, false) && ability.isUsesStack()) {
241             game.fireEvent(new GameEvent(GameEvent.EventType.TRIGGERED_ABILITY, ability.getId(), ability, ability.getControllerId()));
242         }
243         sim.applyEffects();
244         SimulationNode newNode = new SimulationNode(parent, sim, playerId);
245         logger.debug(indent(newNode.getDepth()) + "simulating -- node #:" + SimulationNode.getCount() + " triggered ability option");
246         for (Target target: ability.getTargets()) {
247             for (UUID targetId: target.getTargets()) {
248                 newNode.getTargets().add(targetId);
249             }
250         }
251         parent.children.add(newNode);
252     }

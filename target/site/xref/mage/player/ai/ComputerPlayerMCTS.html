62      @Override
63      public boolean priority(Game game) {
64          if (game.getTurnStepType() == PhaseStep.UPKEEP) {
65              if (!lastPhase.equals(game.getTurn().getValue(game.getTurnNum()))) {
66                  logList(game.getTurn().getValue(game.getTurnNum()) + name + " hand: ", new ArrayList(hand.getCards(game)));
67                  lastPhase = game.getTurn().getValue(game.getTurnNum());
68                  if (MCTSNode.USE_ACTION_CACHE) {
69                      int count = MCTSNode.cleanupCache(game.getTurnNum());
70                      if (count > 0)
71                          logger.info("Removed " + count + " cache entries");
72                  }
73              }
74          }
75          game.getState().setPriorityPlayerId(playerId);
76          game.firePriorityEvent(playerId);
77          getNextAction(game, NextAction.PRIORITY);
78          Ability ability = root.getAction();
79          if (ability == null)
80              logger.fatal("null ability");
81          activateAbility((ActivatedAbility) ability, game);
82          if (ability instanceof PassAbility)
83              return false;
84          logLife(game);
85          logger.info("choose action:" + root.getAction() + " success ratio: " + root.getWinRatio());
86          return true;
87      }

89      protected void calculateActions(Game game, NextAction action) {
90          if (root == null) {
91              Game sim = createMCTSGame(game);
92              MCTSPlayer player = (MCTSPlayer) sim.getPlayer(playerId);
93              player.setNextAction(action);
94              root = new MCTSNode(playerId, sim);
95          }
96          applyMCTS(game, action);
97          root = root.bestChild();
98          root.emancipate();
99      }

101     protected void getNextAction(Game game, NextAction nextAction) {
102         if (root != null) {
103             MCTSNode newRoot;
104             newRoot = root.getMatchingState(game.getState().getValue(game, playerId));
105             if (newRoot != null) {
106                 newRoot.emancipate();
107             } else
108                 logger.info("unable to find matching state");
109             root = newRoot;
110         }
111         calculateActions(game, nextAction);
112     }

116         StringBuilder sb = new StringBuilder();
117         sb.append(game.getTurn().getValue(game.getTurnNum())).append(" player ").append(name).append(" attacking with: ");
118         getNextAction(game, NextAction.SELECT_ATTACKERS);
119         Combat combat = root.getCombat();
120         UUID opponentId = game.getCombat().getDefenders().iterator().next();
121         for (UUID attackerId : combat.getAttackers()) {
122             this.declareAttacker(attackerId, opponentId, game, false);
123             sb.append(game.getPermanent(attackerId).getName()).append(',');
124         }

129     @Override
130     public void selectBlockers(Ability source, Game game, UUID defendingPlayerId) {
131         StringBuilder sb = new StringBuilder();
132         sb.append(game.getTurn().getValue(game.getTurnNum())).append(" player ").append(name).append(" blocking: ");
133         getNextAction(game, NextAction.SELECT_BLOCKERS);
134         Combat simulatedCombat = root.getCombat();
135         List<CombatGroup> currentGroups = game.getCombat().getGroups();
136         for (int i = 0; i < currentGroups.size(); i++) {
137             if (i < simulatedCombat.getGroups().size()) {
138                 CombatGroup currentGroup = currentGroups.get(i);
139                 CombatGroup simulatedGroup = simulatedCombat.getGroups().get(i);
140                 sb.append(game.getPermanent(currentGroup.getAttackers().get(0)).getName()).append(" with: ");
141                 for (UUID blockerId : simulatedGroup.getBlockers()) {
142                     // blockers can be added automaticly by requirement effects, so we must add only missing blockers
143                     if (!currentGroup.getBlockers().contains(blockerId)) {
144                         this.declareBlocker(this.getId(), blockerId, currentGroup.getAttackers().get(0), game);
145                         sb.append(game.getPermanent(blockerId).getName()).append(',');
146                     }
147                 }
148                 sb.append('|');
149             }
150         }
151         logger.info(sb.toString());
152         MCTSNode.logHitMiss();
153     }

156     protected long totalSimulations = 0;

158     protected void applyMCTS(final Game game, final NextAction action) {
159 
160         int thinkTime = calculateThinkTime(game, action);
161 
162         if (thinkTime > 0) {
163             if (USE_MULTIPLE_THREADS) {
164                 ExecutorService pool = Executors.newFixedThreadPool(poolSize);
165                 List<MCTSExecutor> tasks = new ArrayList<>();
166                 for (int i = 0; i < poolSize; i++) {
167                     Game sim = createMCTSGame(game);
168                     MCTSPlayer player = (MCTSPlayer) sim.getPlayer(playerId);
169                     player.setNextAction(action);
170                     MCTSExecutor exec = new MCTSExecutor(sim, playerId, thinkTime);
171                     tasks.add(exec);
172                 }
173 
174                 try {
175                     pool.invokeAll(tasks, thinkTime, TimeUnit.SECONDS);
176                     pool.awaitTermination(1, TimeUnit.SECONDS);
177                     pool.shutdownNow();
178                 } catch (InterruptedException | RejectedExecutionException ex) {
179                     logger.warn("applyMCTS interrupted");
180                 }
181 
182                 int simCount = 0;
183                 for (MCTSExecutor task : tasks) {
184                     simCount += task.getSimCount();
185                     root.merge(task.getRoot());
186                     task.clear();
187                 }
188                 tasks.clear();
189                 totalThinkTime += thinkTime;
190                 totalSimulations += simCount;
191                 logger.info("Player: " + name + " Simulated " + simCount + " games in " + thinkTime + " seconds - nodes in tree: " + root.size());
192                 logger.info("Total: Simulated " + totalSimulations + " games in " + totalThinkTime + " seconds - Average: " + totalSimulations / totalThinkTime);
193                 MCTSNode.logHitMiss();
194             } else {
195                 long startTime = System.nanoTime();
196                 long endTime = startTime + (thinkTime * 1000000000l);
197                 MCTSNode current;
198                 int simCount = 0;
199                 while (true) {
200                     long currentTime = System.nanoTime();
201                     if (currentTime > endTime)
202                         break;
203                     current = root;
204 
205                     // Selection
206                     while (!current.isLeaf()) {
207                         current = current.select(this.playerId);
208                     }
209 
210                     int result;
211                     if (!current.isTerminal()) {
212                         // Expansion
213                         current.expand();
214 
215                         // Simulation
216                         current = current.select(this.playerId);
217                         result = current.simulate(this.playerId);
218                         simCount++;
219                     } else {
220                         result = current.isWinner(this.playerId) ? 1 : -1;
221                     }
222                     // Backpropagation
223                     current.backpropagate(result);
224                 }
225                 logger.info("Simulated " + simCount + " games - nodes in tree: " + root.size());
226             }
227 //            displayMemory();
228         }
229 
230     }

232     //try to ensure that there are at least THINK_MIN_RATIO simulations per node at all times
233     private int calculateThinkTime(Game game, NextAction action) {
234         int thinkTime;
235         int nodeSizeRatio = 0;
236         if (root.getNumChildren() > 0)
237             nodeSizeRatio = root.getVisits() / root.getNumChildren();
238 //        logger.info("Ratio: " + nodeSizeRatio);
239         PhaseStep curStep = game.getTurnStepType();
240         if (action == NextAction.SELECT_ATTACKERS || action == NextAction.SELECT_BLOCKERS) {
241             if (nodeSizeRatio < THINK_MIN_RATIO) {
242                 thinkTime = maxThinkTime;
243             } else if (nodeSizeRatio >= THINK_MAX_RATIO) {
244                 thinkTime = 0;
245             } else {
246                 thinkTime = maxThinkTime / 2;
247             }
248         } else if (game.isActivePlayer(playerId) && (curStep == PhaseStep.PRECOMBAT_MAIN || curStep == PhaseStep.POSTCOMBAT_MAIN) && game.getStack().isEmpty()) {
249             if (nodeSizeRatio < THINK_MIN_RATIO) {
250                 thinkTime = maxThinkTime;
251             } else if (nodeSizeRatio >= THINK_MAX_RATIO) {
252                 thinkTime = 0;
253             } else {
254                 thinkTime = maxThinkTime / 2;
255             }
256         } else {
257             if (nodeSizeRatio < THINK_MIN_RATIO) {
258                 thinkTime = maxThinkTime / 2;
259             } else {
260                 thinkTime = 0;
261             }
262         }
263         return thinkTime;
264     }

266     /**
267      * Copies game and replaces all players in copy with mcts players
268      * Shuffles each players library so that there is no knowledge of its order
269      * Swaps all other players hands with random cards from the library so that
270      * there is no knowledge of what cards are in opponents hands
271      * The most knowledge that is known is what cards are in an opponents deck
272      *
273      * @param game
274      * @return a new game object with simulated players
275      */
276     protected Game createMCTSGame(Game game) {
277         Game mcts = game.copy();
278 
279         for (Player copyPlayer : mcts.getState().getPlayers().values()) {
280             Player origPlayer = game.getState().getPlayers().get(copyPlayer.getId());
281             MCTSPlayer newPlayer = new MCTSPlayer(copyPlayer.getId());
282             newPlayer.restore(origPlayer);
283             if (!newPlayer.getId().equals(playerId)) {
284                 int handSize = newPlayer.getHand().size();
285                 newPlayer.getLibrary().addAll(newPlayer.getHand().getCards(mcts), mcts);
286                 newPlayer.getHand().clear();
287                 newPlayer.getLibrary().shuffle();
288                 for (int i = 0; i < handSize; i++) {
289                     Card card = newPlayer.getLibrary().removeFromTop(mcts);
290                     card.setZone(Zone.HAND, mcts);
291                     newPlayer.getHand().add(card);
292                 }
293             } else {
294                 newPlayer.getLibrary().shuffle();
295             }
296             mcts.getState().getPlayers().put(copyPlayer.getId(), newPlayer);
297         }
298         mcts.setSimulation(true);
299         mcts.resume();
300         return mcts;
301     }

303     protected void displayMemory() {
304         long heapSize = Runtime.getRuntime().totalMemory();
305         long heapMaxSize = Runtime.getRuntime().maxMemory();
306         long heapFreeSize = Runtime.getRuntime().freeMemory();
307         long heapUsedSize = heapSize - heapFreeSize;
308         long mb = 1024 * 1024;
309 
310         logger.info("Max heap size: " + heapMaxSize / mb + " Heap size: " + heapSize / mb + " Used: " + heapUsedSize / mb);
311     }

157     protected void act(Game game) {
158         if (actions == null
159                 || actions.isEmpty()) {
160             pass(game);
161         } else {
162             boolean usedStack = false;
163             while (actions.peek() != null) {
164                 Ability ability = actions.poll();
165                 // example: ===> SELECTED ACTION for PlayerA: Play Swamp
166                 logger.info(String.format("===> SELECTED ACTION for %s: %s",
167                         getName(),
168                         ability.toString()
169                                 + listTargets(game, ability.getTargets(), " (targeting %s)", "")
170                 ));
171                 if (!ability.getTargets().isEmpty()) {
172                     for (Target target : ability.getTargets()) {
173                         for (UUID id : target.getTargets()) {
174                             target.updateTarget(id, game);
175                             if (!target.isNotTarget()) {
176                                 game.addSimultaneousEvent(GameEvent.getEvent(GameEvent.EventType.TARGETED, id, ability, ability.getControllerId()));
177                             }
178                         }
179                     }
180                     Player player = game.getPlayer(ability.getFirstTarget());
181                     if (player != null) {
182                         logger.info("targets = " + player.getName());
183                     }
184                 }
185                 this.activateAbility((ActivatedAbility) ability, game);
186                 if (ability.isUsesStack()) {
187                     usedStack = true;
188                 }
189                 if (!suggestedActions.isEmpty() && !(ability instanceof PassAbility)) {
190                     Iterator<String> it = suggestedActions.iterator();
191                     while (it.hasNext()) {
192                         String action = it.next();
193                         Card card = game.getCard(ability.getSourceId());
194                         if (card != null && action.equals(card.getName())) {
195                             logger.info("-> removed from suggested=" + action);
196                             it.remove();
197                         }
198                     }
199                 }
200             }
201             if (usedStack) {
202                 pass(game);
203             }
204         }
205     }

207     protected int addActions(SimulationNode2 node, int depth, int alpha, int beta) {
208         boolean stepFinished = false;
209         int val;
210         if (logger.isTraceEnabled()
211                 && node != null
212                 && node.getAbilities() != null
213                 && !node.getAbilities().toString().equals("[Pass]")) {
214             logger.trace("Add Action [" + depth + "] " + node.getAbilities().toString() + "  a: " + alpha + " b: " + beta);
215         }
216         Game game = node.getGame();
217         if (!COMPUTER_DISABLE_TIMEOUT_IN_GAME_SIMULATIONS
218                 && Thread.interrupted()) {
219             Thread.currentThread().interrupt();
220             logger.debug("interrupted");
221             return GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
222         }
223         // Condition to stop deeper simulation
224         if (depth <= 0
225                 || SimulationNode2.nodeCount > maxNodes
226                 || game.checkIfGameIsOver()) {
227             val = GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
228             if (logger.isTraceEnabled()) {
229                 StringBuilder sb = new StringBuilder("Add Actions -- reached end state  <").append(val).append('>');
230                 SimulationNode2 logNode = node;
231                 do {
232                     sb.append(new StringBuilder(" <- [" + logNode.getDepth() + ']' + (logNode.getAbilities() != null ? logNode.getAbilities().toString() : "[empty]")));
233                     logNode = logNode.getParent();
234                 } while ((logNode.getParent() != null));
235                 logger.trace(sb);
236             }
237         } else if (!node.getChildren().isEmpty()) {
238             if (logger.isDebugEnabled()) {
239                 StringBuilder sb = new StringBuilder("Add Action [").append(depth)
240                         .append("] -- something added children ")
241                         .append(node.getAbilities() != null ? node.getAbilities().toString() : "null")
242                         .append(" added children: ").append(node.getChildren().size()).append(" (");
243                 for (SimulationNode2 logNode : node.getChildren()) {
244                     sb.append(logNode.getAbilities() != null ? logNode.getAbilities().toString() : "null").append(", ");
245                 }
246                 sb.append(')');
247                 logger.debug(sb);
248             }
249             val = minimaxAB(node, depth - 1, alpha, beta);
250         } else {
251             logger.trace("Add Action -- alpha: " + alpha + " beta: " + beta + " depth:" + depth + " step:" + game.getTurnStepType() + " for player:" + game.getPlayer(game.getActivePlayerId()).getName());
252             if (allPassed(game)) {
253                 if (!game.getStack().isEmpty()) {
254                     resolve(node, depth, game);
255                 } else {
256                     stepFinished = true;
257                 }
258             }
259 
260             if (game.checkIfGameIsOver()) {
261                 val = GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
262             } else if (stepFinished) {
263                 logger.debug("Step finished");
264                 int testScore = GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
265                 if (game.isActivePlayer(playerId)) {
266                     if (testScore < currentScore) {
267                         // if score at end of step is worse than original score don't check further
268                         //logger.debug("Add Action -- abandoning check, no immediate benefit");
269                         val = testScore;
270                     } else {
271                         val = GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
272                     }
273                 } else {
274                     val = GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
275                 }
276             } else if (!node.getChildren().isEmpty()) {
277                 if (logger.isDebugEnabled()) {
278                     StringBuilder sb = new StringBuilder("Add Action [").append(depth)
279                             .append("] -- trigger ")
280                             .append(node.getAbilities() != null ? node.getAbilities().toString() : "null")
281                             .append(" added children: ").append(node.getChildren().size()).append(" (");
282                     for (SimulationNode2 logNode : node.getChildren()) {
283                         sb.append(logNode.getAbilities() != null ? logNode.getAbilities().toString() : "null").append(", ");
284                     }
285                     sb.append(')');
286                     logger.debug(sb);
287                 }
288                 val = minimaxAB(node, depth - 1, alpha, beta);
289             } else {
290                 val = simulatePriority(node, game, depth, alpha, beta);
291             }
292         }
293         node.setScore(val);
294         logger.trace("returning -- score: " + val + " depth:" + depth + " step:" + game.getTurnStepType() + " for player:" + game.getPlayer(node.getPlayerId()).getName());
295         return val;

299     protected boolean getNextAction(Game game) {
300         if (root != null
301                 && !root.children.isEmpty()) {
302             SimulationNode2 test = root;
303             root = root.children.get(0);
304             while (!root.children.isEmpty()
305                     && !root.playerId.equals(playerId)) {
306                 test = root;
307                 root = root.children.get(0);
308             }
309             logger.trace("Sim getNextAction -- game value:" + game.getState().getValue(true) + " test value:" + test.gameValue);
310             if (!suggestedActions.isEmpty()) {
311                 return false;
312             }
313             if (root.playerId.equals(playerId)
314                     && root.abilities != null
315                     && game.getState().getValue(true).hashCode() == test.gameValue) {
316                 logger.info("simulating -- continuing previous action chain");
317                 actions = new LinkedList<>(root.abilities);
318                 combat = root.combat;
319                 return true;
320             } else {
321                 return false;
322             }
323         }
324         return false;
325     }

327     protected int minimaxAB(SimulationNode2 node, int depth, int alpha, int beta) {
328         logger.trace("Sim minimaxAB [" + depth + "] -- a: " + alpha + " b: " + beta + " <" + (node != null ? node.getScore() : "null") + '>');
329         UUID currentPlayerId = node.getGame().getPlayerList().get();
330         SimulationNode2 bestChild = null;
331         for (SimulationNode2 child : node.getChildren()) {
332             Combat _combat = child.getCombat();
333             if (alpha >= beta) {
334                 break;
335             }
336             if (SimulationNode2.nodeCount > maxNodes) {
337                 break;
338             }
339             int val = addActions(child, depth - 1, alpha, beta);
340             if (!currentPlayerId.equals(playerId)) {
341                 if (val < beta) {
342                     beta = val;
343                     bestChild = child;
344                     if (node.getCombat() == null) {
345                         node.setCombat(_combat);
346                         bestChild.setCombat(_combat);
347                     }
348                 }
349                 // no need to check other actions
350                 if (val == GameStateEvaluator2.LOSE_GAME_SCORE) {
351                     logger.debug("lose - break");
352                     break;
353                 }
354             } else {
355                 if (val > alpha) {
356                     alpha = val;
357                     bestChild = child;
358                     if (node.getCombat() == null) {
359                         node.setCombat(_combat);
360                         bestChild.setCombat(_combat);
361                     }
362                 }
363                 // no need to check other actions
364                 if (val == GameStateEvaluator2.WIN_GAME_SCORE) {
365                     logger.debug("win - break");
366                     break;
367                 }
368             }
369         }
370         node.children.clear();
371         if (bestChild != null) {
372             node.children.add(bestChild);
373         }
374         if (!currentPlayerId.equals(playerId)) {
375             return beta;
376         } else {
377             return alpha;
378         }
379     }

390     protected void resolve(SimulationNode2 node, int depth, Game game) {
391         StackObject stackObject = game.getStack().getFirst();
392         if (stackObject instanceof StackAbility) {
393             // AI hint for search effects (calc all possible cards for best score)
394             SearchEffect effect = getSearchEffect((StackAbility) stackObject);
395             if (effect != null
396                     && stackObject.getControllerId().equals(playerId)) {
397                 Target target = effect.getTarget();
398                 if (!target.doneChoosing()) {
399                     for (UUID targetId : target.possibleTargets(stackObject.getControllerId(), stackObject.getStackAbility(), game)) {
400                         Game sim = game.copy();
401                         StackAbility newAbility = (StackAbility) stackObject.copy();
402                         SearchEffect newEffect = getSearchEffect(newAbility);
403                         newEffect.getTarget().addTarget(targetId, newAbility, sim);
404                         sim.getStack().push(newAbility);
405                         SimulationNode2 newNode = new SimulationNode2(node, sim, depth, stackObject.getControllerId());
406                         node.children.add(newNode);
407                         newNode.getTargets().add(targetId);
408                         logger.trace("Sim search -- node#: " + SimulationNode2.getCount() + " for player: " + sim.getPlayer(stackObject.getControllerId()).getName());
409                     }
410                     return;
411                 }
412             }
413         }
414         stackObject.resolve(game);
415         if (stackObject instanceof StackAbility) {
416             game.getStack().remove(stackObject, game);
417         }
418         game.applyEffects();
419         game.getPlayers().resetPassed();
420         game.getPlayerList().setCurrent(game.getActivePlayerId());
421     }

428     protected Integer addActionsTimed() {
429         FutureTask<Integer> task = new FutureTask<>(new Callable<Integer>() {
430             @Override
431             public Integer call() throws Exception {
432                 return addActions(root, maxDepth, Integer.MIN_VALUE, Integer.MAX_VALUE);
433             }
434         });
435         pool.execute(task);
436         try {
437             int maxSeconds = maxThink;
438             if (COMPUTER_DISABLE_TIMEOUT_IN_GAME_SIMULATIONS) {
439                 maxSeconds = 3600;
440             }
441             logger.debug("maxThink: " + maxSeconds + " seconds ");
442             if (task.get(maxSeconds, TimeUnit.SECONDS) != null) {
443                 return task.get(maxSeconds, TimeUnit.SECONDS);
444             }
445         } catch (TimeoutException e) {
446             logger.info("simulating - timed out");
447             task.cancel(true);
448         } catch (ExecutionException e) {
449             // exception error in simulated game
450             e.printStackTrace();
451             task.cancel(true);
452             // real games: must catch
453             // unit tests: must raise again for test fail
454             if (this.isTestsMode()) {
455                 throw new IllegalStateException("One of the simulated games raise the error: " + e.getCause());
456             }
457         } catch (InterruptedException e) {
458             e.printStackTrace();
459             task.cancel(true);
460         } catch (Exception e) {
461             e.printStackTrace();
462             task.cancel(true);
463         }
464         //TODO: timeout handling
465         return 0;
466     }

468     protected int simulatePriority(SimulationNode2 node, Game game, int depth, int alpha, int beta) {
469         if (!COMPUTER_DISABLE_TIMEOUT_IN_GAME_SIMULATIONS
470                 && Thread.interrupted()) {
471             Thread.currentThread().interrupt();
472             logger.info("interrupted");
473             return GameStateEvaluator2.evaluate(playerId, game).getTotalScore();
474         }
475         node.setGameValue(game.getState().getValue(true).hashCode());
476         SimulatedPlayer2 currentPlayer = (SimulatedPlayer2) game.getPlayer(game.getPlayerList().get());
477         SimulationNode2 bestNode = null;
478         List<Ability> allActions = currentPlayer.simulatePriority(game);
479         optimize(game, allActions);
480         int startedScore = GameStateEvaluator2.evaluate(this.getId(), node.getGame()).getTotalScore();
481         if (logger.isInfoEnabled()
482                 && !allActions.isEmpty()
483                 && depth == maxDepth) {
484             logger.info(String.format("POSSIBLE ACTIONS for %s (%d, started score: %d)%s",
485                     getName(),
486                     allActions.size(),
487                     startedScore,
488                     (actions.isEmpty() ? "" : ":")
489             ));
490             for (int i = 0; i < allActions.size(); i++) {
491                 logger.info(String.format("-> #%d (%s)", i + 1, allActions.get(i)));
492             }
493         }
494         int actionNumber = 0;
495         int bestValSubNodes = Integer.MIN_VALUE;
496         for (Ability action : allActions) {
497             actionNumber++;
498             if (!COMPUTER_DISABLE_TIMEOUT_IN_GAME_SIMULATIONS
499                     && Thread.interrupted()) {
500                 Thread.currentThread().interrupt();
501                 logger.info("Sim Prio [" + depth + "] -- interrupted");
502                 break;
503             }
504             Game sim = game.copy();
505             sim.setSimulation(true);
506             if (!(action instanceof StaticAbility) //for MorphAbility, etc
507                     && sim.getPlayer(currentPlayer.getId()).activateAbility((ActivatedAbility) action.copy(), sim)) {
508                 sim.applyEffects();
509                 if (checkForRepeatedAction(sim, node, action, currentPlayer.getId())) {
510                     logger.debug("Sim Prio [" + depth + "] -- repeated action: " + action);
511                     continue;
512                 }
513                 if (!sim.checkIfGameIsOver()
514                         && (action.isUsesStack() || action instanceof PassAbility)) {
515                     // skip priority for opponents before stack resolve
516                     UUID nextPlayerId = sim.getPlayerList().get();
517                     do {
518                         sim.getPlayer(nextPlayerId).pass(game);
519                         nextPlayerId = sim.getPlayerList().getNext();
520                     } while (!Objects.equals(nextPlayerId, this.getId()));
521                 }
522                 SimulationNode2 newNode = new SimulationNode2(node, sim, action, depth, currentPlayer.getId());
523                 sim.checkStateAndTriggered();
524                 int actionScore;
525                 if (action instanceof PassAbility && sim.getStack().isEmpty()) {
526                     // no more next actions, it's a final score
527                     actionScore = GameStateEvaluator2.evaluate(this.getId(), sim).getTotalScore();
528                 } else {
529                     // resolve current action and calc all next actions to find best score (return max possible score)
530                     actionScore = addActions(newNode, depth - 1, alpha, beta);
531                 }
532                 logger.debug("Sim Prio " + BLANKS.substring(0, 2 + (maxDepth - depth) * 3) + '[' + depth + "]#" + actionNumber + " <" + actionScore + "> - (" + action + ") ");
533 
534                 // Hints on data:
535                 // * node - started game with executed command (pay and put on stack)
536                 // * newNode - resolved game with resolved command (resolve stack)
537                 // * node.children - rewrites to store only best tree (e.g. contains only final data)
538                 // * node.score - rewrites to store max score (e.g. contains only final data)
539                 if (logger.isInfoEnabled()
540                         && depth >= maxDepth) {
541                     // show calculated actions and score
542                     // example: Sim Prio [6] #1 <605> (Play Swamp)
543                     int currentActionScore = GameStateEvaluator2.evaluate(this.getId(), newNode.getGame()).getTotalScore();
544                     int diffCurrentAction = currentActionScore - startedScore;
545                     int diffNextActions = actionScore - startedScore - diffCurrentAction;
546                     logger.info(String.format("Sim Prio [%d] #%d <diff %s, %s> (%s)",
547                             depth,
548                             actionNumber,
549                             printDiffScore(diffCurrentAction),
550                             printDiffScore(diffNextActions),
551                             action
552                                     + (action.isModal() ? " Mode = " + action.getModes().getMode().toString() : "")
553                                     + listTargets(game, action.getTargets(), " (targeting %s)", "")
554                                     + (logger.isTraceEnabled() ? " #" + newNode.hashCode() : "")
555                     ));
556                     // collect childs info (next actions chain)
557                     SimulationNode2 logNode = newNode;
558                     while (logNode.getChildren() != null
559                             && !logNode.getChildren().isEmpty()) {
560                         logNode = logNode.getChildren().get(0);
561                         if (logNode.getAbilities() != null
562                                 && !logNode.getAbilities().isEmpty()) {
563                             int logCurrentScore = GameStateEvaluator2.evaluate(this.getId(), logNode.getGame()).getTotalScore();
564                             int logPrevScore = GameStateEvaluator2.evaluate(this.getId(), logNode.getParent().getGame()).getTotalScore();
565                             logger.info(String.format("Sim Prio [%d] -> next action: [%d]%s <diff %s, %s>",
566                                     depth,
567                                     logNode.getDepth(),
568                                     logNode.getAbilities().toString(),
569                                     printDiffScore(logCurrentScore - logPrevScore),
570                                     printDiffScore(actionScore - logCurrentScore)
571                             ));
572                         }
573                     }
574                 }
575 
576                 if (currentPlayer.getId().equals(playerId)) {
577                     if (actionScore > bestValSubNodes) {
578                         bestValSubNodes = actionScore;
579                     }
580                     if (depth == maxDepth
581                             && action instanceof PassAbility) {
582                         actionScore = actionScore - PASSIVITY_PENALTY; // passivity penalty
583                     }
584                     if (actionScore > alpha
585                             || (depth == maxDepth
586                             && actionScore == alpha
587                             && RandomUtil.nextBoolean())) { // Adding random for equal value to get change sometimes
588                         alpha = actionScore;
589                         bestNode = newNode;
590                         bestNode.setScore(actionScore);
591                         if (!newNode.getChildren().isEmpty()) {
592                             bestNode.setCombat(newNode.getChildren().get(0).getCombat());
593                         }
594                         if (depth == maxDepth) {
595                             GameStateEvaluator2.PlayerEvaluateScore score = GameStateEvaluator2.evaluate(this.getId(), bestNode.game);
596                             String scoreInfo = " [" + score.getPlayerInfoShort() + "-" + score.getOpponentInfoShort() + "]";
597                             String abilitiesInfo = bestNode.getAbilities()
598                                     .stream()
599                                     .map(a -> a.toString() + listTargets(game, a.getTargets(), " (targeting %s)", ""))
600                                     .collect(Collectors.joining("; "));
601                             logger.info("Sim Prio [" + depth + "] >> BEST action chain found <" + bestNode.getScore() + scoreInfo + "> " + abilitiesInfo);
602                             node.children.clear();
603                             node.children.add(bestNode);
604                             node.setScore(bestNode.getScore());
605                         }
606                     }
607 
608                     // no need to check other actions
609                     if (actionScore == GameStateEvaluator2.WIN_GAME_SCORE) {
610                         logger.debug("Sim Prio -- win - break");
611                         break;
612                     }
613                 } else {
614                     if (actionScore < beta) {
615                         beta = actionScore;
616                         bestNode = newNode;
617                         bestNode.setScore(actionScore);
618                         if (!newNode.getChildren().isEmpty()) {
619                             bestNode.setCombat(newNode.getChildren().get(0).getCombat());
620                         }
621                     }
622 
623                     // no need to check other actions
624                     if (actionScore == GameStateEvaluator2.LOSE_GAME_SCORE) {
625                         logger.debug("Sim Prio -- lose - break");
626                         break;
627                     }
628                 }
629                 if (alpha >= beta) {
630                     break;
631                 }
632                 if (SimulationNode2.nodeCount > maxNodes) {
633                     logger.debug("Sim Prio -- reached end-state");
634                     break;
635                 }
636             }
637         } // end of for (allActions)
638 
639         if (depth == maxDepth) {
640             logger.info("Sim Prio [" + depth + "] -- End for Max Depth  -- Nodes calculated: " + SimulationNode2.nodeCount);
641         }
642         if (bestNode != null) {
643             node.children.clear();
644             node.children.add(bestNode);
645             node.setScore(bestNode.getScore());
646             if (logger.isTraceEnabled()
647                     && !bestNode.getAbilities().toString().equals("[Pass]")) {
648                 logger.trace(new StringBuilder("Sim Prio [").append(depth).append("] -- Set after (depth=").append(depth).append(")  <").append(bestNode.getScore()).append("> ").append(bestNode.getAbilities().toString()).toString());
649             }
650         }
651 
652         if (currentPlayer.getId().equals(playerId)) {
653             return bestValSubNodes;
654         } else {
655             return beta;
656         }
657     }

708                 if (cast1 != cast2) {
709                     if (cast1) {
710                         return -1;
711                     } else {
712                         return 1;
713                     }
714                 }
715 
716                 // default
717                 return ability1.getRule().compareTo(ability2.getRule());
718             }

782     private void declareBlockers(Game game, UUID activePlayerId) {
783         game.fireEvent(new GameEvent(GameEvent.EventType.DECLARE_BLOCKERS_STEP_PRE, null, null, activePlayerId));
784         if (!game.replaceEvent(GameEvent.getEvent(GameEvent.EventType.DECLARING_BLOCKERS, activePlayerId, activePlayerId))) {
785             List<Permanent> attackers = getAttackers(game);
786             if (attackers == null) {
787                 return;
788             }
789 
790             List<Permanent> possibleBlockers = super.getAvailableBlockers(game);
791             possibleBlockers = filterOutNonblocking(game, attackers, possibleBlockers);
792             if (possibleBlockers.isEmpty()) {
793                 return;
794             }
795 
796             attackers = filterOutUnblockable(game, attackers, possibleBlockers);
797             if (attackers.isEmpty()) {
798                 return;
799             }
800 
801             CombatUtil.sortByPower(attackers, false);
802 
803             CombatInfo combatInfo = CombatUtil.blockWithGoodTrade2(game, attackers, possibleBlockers);
804             Player player = game.getPlayer(playerId);
805 
806             boolean blocked = false;
807             for (Map.Entry<Permanent, List<Permanent>> entry : combatInfo.getCombat().entrySet()) {
808                 UUID attackerId = entry.getKey().getId();
809                 List<Permanent> blockers = entry.getValue();
810                 if (blockers != null) {
811                     for (Permanent blocker : blockers) {
812                         player.declareBlocker(player.getId(), blocker.getId(), attackerId, game);
813                         blocked = true;
814                     }
815                 }
816             }
817             if (blocked) {
818                 game.getPlayers().resetPassed();
819             }
820         }
821     }

824         List<Permanent> blockersLeft = new ArrayList<>();
825         for (Permanent blocker : blockers) {
826             for (Permanent attacker : attackers) {
827                 if (blocker.canBlock(attacker.getId(), game)) {
828                     blockersLeft.add(blocker);
829                     break;
830                 }
831             }
832         }
833         return blockersLeft;

860     /**
861      * Choose attackers based on static information. That means that AI won't
862      * look to the future as it was before, but just choose attackers based on
863      * current state of the game. This is worse, but at least it is easier to
864      * implement and won't lead to the case when AI doesn't do anything -
865      * neither attack nor block.
866      *
867      * @param game
868      * @param activePlayerId
869      */
870     private void declareAttackers(Game game, UUID activePlayerId) {
871         attackersToCheck.clear();
872         attackersList.clear();
873         game.fireEvent(new GameEvent(GameEvent.EventType.DECLARE_ATTACKERS_STEP_PRE, null, null, activePlayerId));
874         if (!game.replaceEvent(GameEvent.getEvent(GameEvent.EventType.DECLARING_ATTACKERS, activePlayerId, activePlayerId))) {
875             Player attackingPlayer = game.getPlayer(activePlayerId);
876 
877             // check alpha strike first (all in attack to kill)
878             for (UUID defenderId : game.getOpponents(playerId)) {
879                 Player defender = game.getPlayer(defenderId);
880                 if (!defender.isInGame()) {
881                     continue;
882                 }
883 
884                 attackersList = super.getAvailableAttackers(defenderId, game);
885                 if (attackersList.isEmpty()) {
886                     continue;
887                 }
888                 List<Permanent> possibleBlockers = defender.getAvailableBlockers(game);
889                 List<Permanent> killers = CombatUtil.canKillOpponent(game, attackersList, possibleBlockers, defender);
890                 if (!killers.isEmpty()) {
891                     for (Permanent attacker : killers) {
892                         attackingPlayer.declareAttacker(attacker.getId(), defenderId, game, false);
893                     }
894                     return;
895                 }
896             }
897 
898             // check all other actions
899             for (UUID defenderId : game.getOpponents(playerId)) {
900                 Player defender = game.getPlayer(defenderId);
901                 if (!defender.isInGame()) {
902                     continue;
903                 }
904                 attackersList = super.getAvailableAttackers(defenderId, game);
905                 if (attackersList.isEmpty()) {
906                     continue;
907                 }
908                 List<Permanent> possibleBlockers = defender.getAvailableBlockers(game);
909 
910                 // The AI will now attack more sanely.  Simple, but good enough for now.
911                 // The sim minmax does not work at the moment.
912                 boolean safeToAttack;
913                 CombatEvaluator eval = new CombatEvaluator();
914 
915                 for (Permanent attacker : attackersList) {
916                     safeToAttack = true;
917                     int attackerValue = eval.evaluate(attacker, game);
918                     for (Permanent blocker : possibleBlockers) {
919                         int blockerValue = eval.evaluate(blocker, game);
920 
921                         // blocker can kill attacker
922                         if (attacker.getPower().getValue() <= blocker.getToughness().getValue()
923                                 && attacker.getToughness().getValue() <= blocker.getPower().getValue()) {
924                             safeToAttack = false;
925                         }
926 
927                         // attacker and blocker have the same P/T, check their overall value
928                         if (attacker.getToughness().getValue() == blocker.getPower().getValue()
929                                 && attacker.getPower().getValue() == blocker.getToughness().getValue()) {
930                             if (attackerValue > blockerValue
931                                     || blocker.getAbilities().containsKey(FirstStrikeAbility.getInstance().getId())
932                                     || blocker.getAbilities().containsKey(DoubleStrikeAbility.getInstance().getId())
933                                     || blocker.getAbilities().contains(new ExaltedAbility())
934                                     || blocker.getAbilities().containsKey(DeathtouchAbility.getInstance().getId())
935                                     || blocker.getAbilities().containsKey(IndestructibleAbility.getInstance().getId())
936                                     || !attacker.getAbilities().containsKey(FirstStrikeAbility.getInstance().getId())
937                                     || !attacker.getAbilities().containsKey(DoubleStrikeAbility.getInstance().getId())
938                                     || !attacker.getAbilities().contains(new ExaltedAbility())) {
939                                 safeToAttack = false;
940                             }
941                         }
942 
943                         // attacker can kill by deathtouch
944                         if (attacker.getAbilities().containsKey(DeathtouchAbility.getInstance().getId())
945                                 || attacker.getAbilities().containsKey(IndestructibleAbility.getInstance().getId())) {
946                             safeToAttack = true;
947                         }
948 
949                         // attacker has flying and blocker has neither flying nor reach
950                         if (attacker.getAbilities().containsKey(FlyingAbility.getInstance().getId())
951                                 && !blocker.getAbilities().containsKey(FlyingAbility.getInstance().getId())
952                                 && !blocker.getAbilities().containsKey(ReachAbility.getInstance().getId())) {
953                             safeToAttack = true;
954                         }
955 
956                         // if any check fails, move on to the next possible attacker
957                         if (!safeToAttack) {
958                             break;
959                         }
960                     }
961 
962                     // 0 power, don't bother attacking
963                     if (attacker.getPower().getValue() == 0) {
964                         safeToAttack = false;
965                     }
966 
967                     // add attacker to the next list of all attackers that can safely attack
968                     if (safeToAttack) {
969                         attackersToCheck.add(attacker);
970                     }
971                 }
972 
973                 // now we have a list of all attackers that can safely attack:
974                 // first check to see if any Planeswalkers can be killed
975                 int totalPowerOfAttackers = 0;
976                 int loyaltyCounters = 0;
977                 for (Permanent planeswalker : game.getBattlefield().getAllActivePermanents(StaticFilters.FILTER_PERMANENT_PLANESWALKER, defender.getId(), game)) {
978                     if (planeswalker != null) {
979                         loyaltyCounters = planeswalker.getCounters(game).getCount(CounterType.LOYALTY);
980                         // verify the attackers can kill the planeswalker, otherwise attack the player
981                         for (Permanent attacker : attackersToCheck) {
982                             totalPowerOfAttackers += attacker.getPower().getValue();
983                         }
984                         if (totalPowerOfAttackers < loyaltyCounters) {
985                             break;
986                         }
987                         // kill the Planeswalker
988                         for (Permanent attacker : attackersToCheck) {
989                             loyaltyCounters -= attacker.getPower().getValue();
990                             attackingPlayer.declareAttacker(attacker.getId(), planeswalker.getId(), game, true);
991                             if (loyaltyCounters <= 0) {
992                                 break;
993                             }
994                         }
995                     }
996                 }
997 
998                 // any remaining attackers go for the player
999                 for (Permanent attackingPermanent : attackersToCheck) {
1000                     // if not already attacking a Planeswalker...
1001                     if (!attackingPermanent.isAttacking()) {
1002                         attackingPlayer.declareAttacker(attackingPermanent.getId(), defenderId, game, true);
1003                     }
1004                 }
1005             }
1006         }
1007     }

1043     private boolean checkForRepeatedAction(Game sim, SimulationNode2 node, Ability action, UUID playerId) {
1044         // pass or casting two times a spell multiple times on hand is ok
1045         if (action instanceof PassAbility || action instanceof SpellAbility || action.getAbilityType() == AbilityType.MANA) {
1046             return false;
1047         }
1048         int newVal = GameStateEvaluator2.evaluate(playerId, sim).getTotalScore();
1049         SimulationNode2 test = node.getParent();
1050         while (test != null) {
1051             if (test.getPlayerId().equals(playerId)) {
1052                 if (test.getAbilities() != null && test.getAbilities().size() == 1) {
1053                     if (action.toString().equals(test.getAbilities().get(0).toString())) {
1054                         if (test.getParent() != null) {
1055                             Game prevGame = node.getGame();
1056                             if (prevGame != null) {
1057                                 int oldVal = GameStateEvaluator2.evaluate(playerId, prevGame).getTotalScore();
1058                                 if (oldVal >= newVal) {
1059                                     return true;
1060                                 }
1061                             }
1062                         }
1063                     }
1064                 }
1065             }
1066             test = test.getParent();
1067         }
1068         return false;
1069     }

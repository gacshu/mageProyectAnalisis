17  public class CombatSimulator implements Serializable {
18  
19      public List<CombatGroupSimulator> groups = new ArrayList<>();
20      public List<UUID> defenders = new ArrayList<>();
21      public Map<UUID, Integer> playersLife = new HashMap<>();
22      public Map<UUID, Integer> planeswalkerLoyalty = new HashMap<>();
23      public UUID attackerId;
24      public int rating = 0;
25  
26      public static CombatSimulator load(Game game) {
27          CombatSimulator simCombat = new CombatSimulator();
28          for (CombatGroup group: game.getCombat().getGroups()) {
29              simCombat.groups.add(new CombatGroupSimulator(group.getDefenderId(), group.getAttackers(), group.getBlockers(), game));
30          }
31          for (UUID defenderId: game.getCombat().getDefenders()) {
32              simCombat.defenders.add(defenderId);
33              Player player = game.getPlayer(defenderId);
34              if (player != null) {
35                  simCombat.playersLife.put(defenderId, player.getLife());
36              }
37              else {
38                  Permanent permanent = game.getPermanent(defenderId);
39                  simCombat.planeswalkerLoyalty.put(defenderId, permanent.getCounters(game).getCount(CounterType.LOYALTY));
40              }
41          }
42          return simCombat;
43      }
44  
45      public CombatSimulator() {}
46  
47      public void clear() {
48          groups.clear();
49          defenders.clear();
50          attackerId = null;
51      }
52  
53      public void simulate(Game game) {
54          for (CombatGroupSimulator group: groups) {
55              group.simulateCombat(game);
56          }
57      }
58  
59      public int evaluate() {
60          Map<UUID, Integer> damage = new HashMap<>();
61          int result = 0;
62          for (CombatGroupSimulator group: groups) {
63              if (!damage.containsKey(group.defenderId)) {
64                  damage.put(group.defenderId, group.unblockedDamage);
65              }
66              else {
67                  damage.put(group.defenderId, damage.get(group.defenderId) + group.unblockedDamage);
68              }
69          }
70          //check for lethal damage to player
71          for (Entry<UUID, Integer> entry: playersLife.entrySet()) {
72              if (damage.containsKey(entry.getKey()) && entry.getValue() <= damage.get(entry.getKey())) {
73                  //TODO:  check for protection
74                  //NOTE:  not applicable for mulitplayer games
75                  return Integer.MAX_VALUE;
76              }
77          }
78  
79          for (CombatGroupSimulator group: groups) {
80              result += group.evaluateCombat();
81          }
82  
83          rating = result;
84          return result;
85      }
86  }

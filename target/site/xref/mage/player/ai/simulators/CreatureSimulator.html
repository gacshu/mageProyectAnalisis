18  public class CreatureSimulator implements Serializable {
19      public UUID id;
20      public int damage;
21      public int power;
22      public int toughness;
23      public boolean hasFirstStrike;
24      public boolean hasDoubleStrike;
25      public boolean hasTrample;
26      public Permanent permanent;
27  
28      public CreatureSimulator(Permanent permanent) {
29          this.id = permanent.getId();
30          this.damage = permanent.getDamage();
31          this.power = permanent.getPower().getValue();
32          this.toughness = permanent.getToughness().getValue();
33          this.hasDoubleStrike = permanent.getAbilities().containsKey(DoubleStrikeAbility.getInstance().getId());
34          this.hasFirstStrike = permanent.getAbilities().containsKey(FirstStrikeAbility.getInstance().getId());
35          this.hasTrample = permanent.getAbilities().containsKey(TrampleAbility.getInstance().getId());
36          this.permanent = permanent;
37      }
38  
39      public boolean isDead() {
40          return damage >= toughness;
41      }
42  
43      public int getLethalDamage(Game game) {
44          List<FilterCreaturePermanent> usePowerInsteadOfToughnessForDamageLethalityFilters = game.getState().getActivePowerInsteadOfToughnessForDamageLethalityFilters();
45          /*
46           * for handling Zilortha, Strength Incarnate:
47           * 2020-04-17
48           * Any time the game is checking whether damage is lethal or if a creature should be destroyed for having lethal damage marked on it, use the power of your creatures rather than their toughness to check the damage against. This includes being assigned trample damage, damage from Flame Spill, and so on.
49           */
50          boolean usePowerInsteadOfToughnessForDamageLethality = usePowerInsteadOfToughnessForDamageLethalityFilters.stream()
51                  .anyMatch(filter -> filter.match(permanent, game));
52          int lethalDamageThreshold = usePowerInsteadOfToughnessForDamageLethality ?
53                  // Zilortha, Strength Incarnate, 2020-04-17: A creature with 0 power isnâ€™t destroyed unless it has at least 1 damage marked on it.
54                  Math.max(power, 1) : toughness;
55          return Math.max(lethalDamageThreshold - damage, 0);
56      }
57  }

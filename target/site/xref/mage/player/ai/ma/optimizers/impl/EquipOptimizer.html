1   package mage.player.ai.ma.optimizers.impl;
2   
3   import java.util.List;
4   import mage.abilities.Ability;
5   import mage.abilities.keyword.EquipAbility;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   
9   /**
10   * Make sure that AI won't equip the same creature equip already attached to.
11   *
12   * @author ayratn
13   */
14  public class EquipOptimizer extends BaseTreeOptimizer {
15  
16      @Override
17      public void filter(Game game, List<Ability> actions) {
18          for (Ability ability : actions) {
19              if (ability instanceof EquipAbility) {
20                  Permanent permanent = game.getPermanent(ability.getFirstTarget());
21                  if (permanent != null) {
22                      // check that equipment is not already attached to {this}
23                      if (permanent.getAttachments().contains(ability.getSourceId())) {
24                          removeAbility(ability);
25                      }
26                  }
27              }
28          }
29      }
30  }

3   import mage.abilities.Ability;
4   import mage.abilities.keyword.LevelUpAbility;
5   import mage.counters.CounterType;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   import mage.game.permanent.PermanentCard;
9   
10  import java.util.List;
11  
12  /**
13   * Make sure that AI won't level up whenever there are maximum possible level up counters.
14   *
15   * @author ayratn
16   */
17  public class LevelUpOptimizer extends BaseTreeOptimizer {
18  
19      /**
20       * Check that ability is level up ability, then compare the current counters of Level type to maximum.
21       *
22       * @param game
23       * @param actions
24       */
25      @Override
26      public void filter(Game game, List<Ability> actions) {
27          for (Ability ability : actions) {
28              if (ability instanceof LevelUpAbility) {
29                  Permanent permanent = game.getPermanent(ability.getSourceId());
30                  if (permanent instanceof PermanentCard) {
31                      PermanentCard leveler = (PermanentCard) permanent;
32                      // check already existing Level counters and compare to maximum that make sense
33                      if (permanent.getCounters(game).getCount(CounterType.LEVEL) >= leveler.getMaxLevelCounters()) {
34                          removeAbility(ability);
35                      }
36                  }
37              }
38          }
39      }
40  }

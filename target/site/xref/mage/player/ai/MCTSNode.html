88      private void setPlayer() {
89          if (game.getStep().getStepPart() == StepPart.PRIORITY) {
90              playerId = game.getPriorityPlayerId();
91          } else {
92              if (game.getTurnStepType() == PhaseStep.DECLARE_BLOCKERS) {
93                  playerId = game.getCombat().getDefenders().iterator().next();
94              } else {
95                  playerId = game.getActivePlayerId();
96              }
97          }

100     public MCTSNode select(UUID targetPlayerId) {
101         double bestValue = Double.NEGATIVE_INFINITY;
102         boolean isTarget = playerId.equals(targetPlayerId);
103         MCTSNode bestChild = null;
104         if (children.size() == 1) {
105             return children.get(0);
106         }
107         for (MCTSNode node: children) {
108             double uct;
109             if (node.visits > 0)
110                 if (isTarget)
111                     uct = (node.wins / (node.visits)) + (selectionCoefficient * Math.sqrt(Math.log(visits) / (node.visits)));
112                 else
113                     uct = ((node.visits - node.wins) / (node.visits)) + (selectionCoefficient * Math.sqrt(Math.log(visits) / (node.visits)));
114             else
115                 // ensure that a random unvisited node is played first
116                 uct = 10000 + 1000 * Math.random();
117             if (uct > bestValue) {
118                 bestChild = node;
119                 bestValue = uct;
120             }
121         }
122         return bestChild;
123     }

125     public void expand() {
126         MCTSPlayer player = (MCTSPlayer) game.getPlayer(playerId);
127         if (player.getNextAction() == null) {
128             logger.fatal("next action is null");
129         }
130         switch (player.getNextAction()) {
131             case PRIORITY:
132 //                logger.info("Priority for player:" + player.getName() + " turn: " + game.getTurnNum() + " phase: " + game.getTurnPhaseType() + " step: " + game.getTurnStepType());
133                 List<Ability> abilities;
134                 if (!USE_ACTION_CACHE)
135                     abilities = player.getPlayableOptions(game);
136                 else
137                     abilities = getPlayables(player, fullStateValue, game);
138                 for (Ability ability: abilities) {
139                     Game sim = game.copy();
140 //                    logger.info("expand " + ability.toString());
141                     MCTSPlayer simPlayer = (MCTSPlayer) sim.getPlayer(player.getId());
142                     simPlayer.activateAbility((ActivatedAbility)ability, sim);
143                     sim.resume();
144                     children.add(new MCTSNode(this, sim, ability));
145                 }
146                 break;
147             case SELECT_ATTACKERS:
148 //                logger.info("Select attackers:" + player.getName());
149                 List<List<UUID>> attacks;
150                 if (!USE_ACTION_CACHE)
151                     attacks = player.getAttacks(game);
152                 else
153                     attacks = getAttacks(player, fullStateValue, game);
154                 UUID defenderId = game.getOpponents(player.getId()).iterator().next();
155                 for (List<UUID> attack: attacks) {
156                     Game sim = game.copy();
157                     MCTSPlayer simPlayer = (MCTSPlayer) sim.getPlayer(player.getId());
158                     for (UUID attackerId: attack) {
159                         simPlayer.declareAttacker(attackerId, defenderId, sim, false);
160                     }
161                     sim.resume();
162                     children.add(new MCTSNode(this, sim, sim.getCombat()));
163                 }
164                 break;
165             case SELECT_BLOCKERS:
166 //                logger.info("Select blockers:" + player.getName());
167                 List<List<List<UUID>>> blocks;
168                 if (!USE_ACTION_CACHE)
169                     blocks = player.getBlocks(game);
170                 else
171                     blocks = getBlocks(player, fullStateValue, game);
172                 for (List<List<UUID>> block: blocks) {
173                     Game sim = game.copy();
174                     MCTSPlayer simPlayer = (MCTSPlayer) sim.getPlayer(player.getId());
175                     List<CombatGroup> groups = sim.getCombat().getGroups();
176                     for (int i = 0; i < groups.size(); i++) {
177                         if (i < block.size()) {
178                             for (UUID blockerId: block.get(i)) {
179                                 simPlayer.declareBlocker(simPlayer.getId(), blockerId, groups.get(i).getAttackers().get(0), sim);
180                             }
181                         }
182                     }
183                     sim.resume();
184                     children.add(new MCTSNode(this, sim, sim.getCombat()));
185                 }
186                 break;
187         }
188         game = null;
189     }

207     public void backpropagate(int result) {
208         if (result == 0)
209             return;
210         if (result == 1)
211             wins++;
212         visits++;
213         if (parent != null)
214             parent.backpropagate(result);
215     }

221     public MCTSNode bestChild() {
222         if (children.size() == 1)
223             return children.get(0);
224         double bestCount = -1;
225         double bestRatio = 0;
226         boolean bestIsPass = false;
227         MCTSNode bestChild = null;
228         for (MCTSNode node: children) {
229             //favour passing vs any other action except for playing land if ratio is close
230             if (node.visits > bestCount) {
231                 if (bestIsPass) {
232                     double ratio = node.wins/(node.visits * 1.0);
233                     if (ratio < bestRatio + passRatioTolerance)
234                         continue;
235                 }
236                 bestChild = node;
237                 bestCount = node.visits;
238                 bestRatio = node.wins/(node.visits * 1.0);
239                 bestIsPass = false;
240             }
241             else if (node.action instanceof PassAbility && node.visits > 10 && !(bestChild.action instanceof PlayLandAbility)) {
242                 //favour passing vs any other action if ratio is close
243                 double ratio = node.wins/(node.visits * 1.0);
244                 if (ratio > bestRatio - passRatioTolerance) {
245                     logger.info("choosing pass over " + bestChild.getAction());
246                     bestChild = node;
247                     bestCount = node.visits;
248                     bestRatio = ratio;
249                     bestIsPass = true;
250                 }
251             }
252         }
253         return bestChild;
254     }

297     /**
298      * Copies game and replaces all players in copy with simulated players
299      * Shuffles each players library so that there is no knowledge of its order
300      *
301      * @param game
302      * @return a new game object with simulated players
303      */
304     protected Game createSimulation(Game game, UUID playerId) {
305         Game sim = game.copy();
306 
307         for (Player oldPlayer: sim.getState().getPlayers().values()) {
308             Player origPlayer = game.getState().getPlayers().get(oldPlayer.getId()).copy();
309             SimulatedPlayerMCTS newPlayer = new SimulatedPlayerMCTS(oldPlayer, true);
310             newPlayer.restore(origPlayer);
311             sim.getState().getPlayers().put(oldPlayer.getId(), newPlayer);
312         }
313         randomizePlayers(sim, playerId);
314         sim.setSimulation(true);
315         return sim;
316     }

318     /*
319      * Shuffles each players library so that there is no knowledge of its order
320      * Swaps all other players hands with random cards from the library so that
321      * there is no knowledge of what cards are in opponents hands
322      */
323     protected void randomizePlayers(Game game, UUID playerId) {
324         for (Player player: game.getState().getPlayers().values()) {
325             if (!player.getId().equals(playerId)) {
326                 int handSize = player.getHand().size();
327                 player.getLibrary().addAll(player.getHand().getCards(game), game);
328                 player.getHand().clear();
329                 player.getLibrary().shuffle();
330                 for (int i = 0; i < handSize; i++) {
331                     Card card = player.getLibrary().removeFromTop(game);
332                     card.setZone(Zone.HAND, game);
333                     player.getHand().add(card);
334                 }
335             }
336             else {
337                 player.getLibrary().shuffle();                
338             }
339         }
340     }

355     /**
356      * 
357      * performs a breadth first search for a matching game state
358      * 
359      * @param state - the game state that we are looking for
360      * @return the matching state or null if no match is found
361      */
362     public MCTSNode getMatchingState(String state) {
363         ArrayDeque<MCTSNode> queue = new ArrayDeque<>();
364         queue.add(this);
365 
366         while (!queue.isEmpty()) {
367             MCTSNode current = queue.remove();
368             if (current.stateValue.equals(state))
369                 return current;
370             for (MCTSNode child: current.children) {
371                 queue.add(child);
372             }
373         }
374         return null;
375     }

377     public void merge(MCTSNode merge) {
378         if (!stateValue.equals(merge.stateValue)) {
379             logger.info("mismatched merge states at root");
380             return;
381         }
382 
383         this.visits += merge.visits;
384         this.wins += merge.wins;
385         int mismatchCount = 0;
386         
387         List<MCTSNode> mergeChildren = new ArrayList<>();
388         for (MCTSNode child: merge.children) {
389             mergeChildren.add(child);
390         }
391 
392         for (MCTSNode child: children) {
393             for (MCTSNode mergeChild: mergeChildren) {
394                 if (mergeChild.action != null && child.action != null) {
395                     if (mergeChild.action.toString().equals(child.action.toString())) {
396                         if (!mergeChild.stateValue.equals(child.stateValue)) {
397                             mismatchCount++;
398 //                            logger.info("mismatched merge states");
399 //                            mergeChildren.remove(mergeChild);
400                         }
401                         else {
402                             child.merge(mergeChild);
403                             mergeChildren.remove(mergeChild);
404                         }
405                         break;
406                     }
407                 }
408                 else {
409                     if (mergeChild.combat.getValue().equals(child.combat.getValue())) {
410                         if (!mergeChild.stateValue.equals(child.stateValue)) {
411                             mismatchCount++;
412 //                            logger.info("mismatched merge states");
413 //                            mergeChildren.remove(mergeChild);
414                         }
415                         else {
416                             child.merge(mergeChild);
417                             mergeChildren.remove(mergeChild);
418                         }
419                         break;
420                     }
421                 }
422             }
423         }
424         if (!mergeChildren.isEmpty()) {
425             for (MCTSNode child: mergeChildren) {
426                 child.parent = this;
427                 children.add(child);
428             }
429         }
430 //        if (mismatchCount > 0)
431 //            logger.info("mismatched merge states: " + mismatchCount);
432     }

466     private static List<Ability> getPlayables(MCTSPlayer player, String state, Game game) {
467         if (playablesCache.containsKey(state)) {
468             playablesHit++;
469             return playablesCache.get(state);
470         }
471         else {
472             playablesMiss++;
473             List<Ability> abilities = player.getPlayableOptions(game);
474             playablesCache.put(state, abilities);
475             return abilities;
476         }
477     }

480         if (attacksCache.containsKey(state)) {
481             attacksHit++;
482             return attacksCache.get(state);
483         }
484         else {
485             attacksMiss++;
486             List<List<UUID>> attacks = player.getAttacks(game);
487             attacksCache.put(state, attacks);
488             return attacks;
489         }
490     }

505     public static int cleanupCache(int turnNum) {
506         Set<String> playablesKeys = playablesCache.keySet();
507         Iterator<String> playablesIterator = playablesKeys.iterator();
508         int count = 0;
509         while(playablesIterator.hasNext()) {
510             String next = playablesIterator.next();
511             int cacheTurn = Integer.parseInt(next.split(":", 2)[0].substring(1));
512             if (cacheTurn < turnNum) {
513                 playablesIterator.remove();
514                 count++;
515             }
516         }
517 
518         Set<String> attacksKeys = attacksCache.keySet();
519         Iterator<String> attacksIterator = attacksKeys.iterator();
520         while(attacksIterator.hasNext()) {
521             int cacheTurn = Integer.parseInt(attacksIterator.next().split(":", 2)[0].substring(1));
522             if (cacheTurn < turnNum) {
523                 attacksIterator.remove();
524                 count++;
525             }
526         }
527         
528         Set<String> blocksKeys = blocksCache.keySet();
529         Iterator<String> blocksIterator = blocksKeys.iterator();
530         while(blocksIterator.hasNext()) {
531             int cacheTurn = Integer.parseInt(blocksIterator.next().split(":", 2)[0].substring(1));
532             if (cacheTurn < turnNum) {
533                 blocksIterator.remove();
534                 count++;
535             }
536         }
537 
538         return count;
539     }

55      public List<Ability> getPlayableOptions(Game game) {
56          List<Ability> all = new ArrayList<>();
57          List<ActivatedAbility> playables = getPlayableAbilities(game);
58          for (ActivatedAbility ability: playables) {
59              List<Ability> options = game.getPlayer(playerId).getPlayableOptions(ability, game);
60              if (options.isEmpty()) {
61                  if (!ability.getManaCosts().getVariableCosts().isEmpty()) {
62                      simulateVariableCosts(ability, all, game);
63                  }
64                  else {
65                      all.add(ability);
66                  }
67              }
68              else {
69                  for (Ability option: options) {
70                      if (!ability.getManaCosts().getVariableCosts().isEmpty()) {
71                          simulateVariableCosts(option, all, game);
72                      }
73                      else {
74                          all.add(option);
75                      }
76                  }
77              }
78          }
79          return all;
80      }

82      protected void simulateVariableCosts(Ability ability, List<Ability> options, Game game) {
83          int numAvailable = getAvailableManaProducers(game).size() - ability.getManaCosts().manaValue();
84          int start = 0;
85          if (!(ability instanceof SpellAbility)) {
86              //only use x=0 on spell abilities
87              if (numAvailable == 0)
88                  return;
89              else
90                  start = 1;
91          }
92          for (int i = start; i < numAvailable; i++) {
93              Ability newAbility = ability.copy();
94              newAbility.getManaCostsToPay().add(new GenericManaCost(i));
95              options.add(newAbility);
96          }
97      }

99      public List<List<UUID>> getAttacks(Game game) {
100         List<List<UUID>> engagements = new ArrayList<>();
101         List<Permanent> attackersList = super.getAvailableAttackers(game);
102         //use binary digits to calculate powerset of attackers
103         int powerElements = (int) Math.pow(2, attackersList.size());
104         StringBuilder binary = new StringBuilder();
105         for (int i = powerElements - 1; i >= 0; i--) {
106             binary.setLength(0);
107             binary.append(Integer.toBinaryString(i));
108             while (binary.length() < attackersList.size()) {
109                 binary.insert(0, '0');
110             }
111             List<UUID> engagement = new ArrayList<>();
112             for (int j = 0; j < attackersList.size(); j++) {
113                 if (binary.charAt(j) == '1') {
114                     engagement.add(attackersList.get(j).getId());
115                 }
116             }
117             engagements.add(engagement);
118         }
119         return engagements;
120     }

122     public List<List<List<UUID>>> getBlocks(Game game) {
123         List<List<List<UUID>>> engagements = new ArrayList<>();
124         int numGroups = game.getCombat().getGroups().size();
125         if (numGroups == 0) {
126             return engagements;
127         }
128 
129         //add a node with no blockers
130         List<List<UUID>> engagement = new ArrayList<>();
131         for (int i = 0; i < numGroups; i++) {
132             engagement.add(new ArrayList<UUID>());
133         }
134         engagements.add(engagement);
135 
136         List<Permanent> blockers = getAvailableBlockers(game);
137         addBlocker(game, engagement, blockers, engagements);
138 
139         return engagements;
140     }

142     private List<List<UUID>> copyEngagement(List<List<UUID>> engagement) {
143         List<List<UUID>> newEngagement = new ArrayList<>();
144         for (List<UUID> group: engagement) {
145             newEngagement.add(new ArrayList<>(group));
146         }
147         return newEngagement;
148     }

150     protected void addBlocker(Game game, List<List<UUID>> engagement, List<Permanent> blockers, List<List<List<UUID>>> engagements) {
151         if (blockers.isEmpty())
152             return;
153         int numGroups = game.getCombat().getGroups().size();
154         //try to block each attacker with each potential blocker
155         Permanent blocker = blockers.get(0);
156 //        if (logger.isDebugEnabled())
157 //            logger.debug("simulating -- block:" + blocker);
158         List<Permanent> remaining = remove(blockers, blocker);
159         for (int i = 0; i < numGroups; i++) {
160             if (game.getCombat().getGroups().get(i).canBlock(blocker, game)) {
161                 List<List<UUID>>newEngagement = copyEngagement(engagement);
162                 newEngagement.get(i).add(blocker.getId());
163                 engagements.add(newEngagement);
164 //                    logger.debug("simulating -- found redundant block combination");
165                 addBlocker(game, newEngagement, remaining, engagements);  // and recurse minus the used blocker
166             }
167         }
168         addBlocker(game, engagement, remaining, engagements);
169     }

9   import java.util.Map;

15  public class PermanentEvaluator {
16  
17      private final Map<UUID, Integer> values = new HashMap<>();
18      private final CombatEvaluator combat = new CombatEvaluator();
19  
20      public int evaluate(Permanent permanent, Game game) {
21          // more score -- more valueable/powerfull permanent
22          if (!values.containsKey(permanent.getId())) {
23              int value = 0;
24              if (permanent.isCreature(game)) {
25                  value += combat.evaluate(permanent, game);
26              }
27              if (permanent.isPlaneswalker(game)) {
28                  value += 2 * permanent.getCounters(game).getCount(CounterType.LOYALTY); // planeswalker is more valuable
29              }
30              value += permanent.getAbilities().getActivatedManaAbilities(Zone.BATTLEFIELD).size();
31              value += permanent.getAbilities().getActivatedAbilities(Zone.BATTLEFIELD).size();
32              values.put(permanent.getId(), value);
33          }
34          return values.get(permanent.getId());
35      }
36  
37  }

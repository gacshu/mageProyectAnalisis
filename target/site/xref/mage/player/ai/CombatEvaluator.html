2   package mage.player.ai;

4   import java.util.HashMap;
5   import java.util.Map;
6   import java.util.UUID;
7   import mage.abilities.keyword.DoubleStrikeAbility;
8   import mage.abilities.keyword.FirstStrikeAbility;
9   import mage.abilities.keyword.TrampleAbility;
10  import mage.game.Game;
11  import mage.game.permanent.Permanent;
12  
13  /**
14   *
15   * @author BetaSteward_at_googlemail.com
16   */
17  public class CombatEvaluator {
18  
19      //preserve calculations for efficiency
20      private Map<UUID, Integer> values = new HashMap<>();
21  
22      public int evaluate(Permanent creature, Game game) {
23          if (!values.containsKey(creature.getId())) {
24              int value = 0;
25              if (creature.canAttack(null, game)) {
26                  value += 2;
27              }
28              value += creature.getPower().getValue();
29              value += creature.getToughness().getValue();
30              value += creature.getAbilities().getEvasionAbilities().size();
31              value += creature.getAbilities().getProtectionAbilities().size();
32              value += creature.getAbilities().containsKey(FirstStrikeAbility.getInstance().getId()) ? 1 : 0;
33              value += creature.getAbilities().containsKey(DoubleStrikeAbility.getInstance().getId()) ? 2 : 0;
34              value += creature.getAbilities().containsKey(TrampleAbility.getInstance().getId()) ? 1 : 0;
35              values.put(creature.getId(), value);
36          }
37          return values.get(creature.getId());
38      }
39  
40  }

34      public ManaPoolItem(int red, int green, int blue, int white, int black, int colorless, MageObject sourceObject, UUID originalId, boolean flag) {
35          this.red = red;
36          this.green = green;
37          this.blue = blue;
38          this.white = white;
39          this.black = black;
40          this.colorless = colorless;
41          this.sourceObject = sourceObject;
42          this.originalId = originalId;
43          this.flag = flag;
44          this.duration = Duration.EndOfStep;
45      }

47      public ManaPoolItem(ConditionalMana conditionalMana, MageObject sourceObject, UUID originalId) {
48          this.conditionalMana = conditionalMana;
49          this.sourceObject = sourceObject;
50          this.originalId = originalId;
51          this.conditionalMana.setManaProducerId(sourceObject.getId());
52          this.conditionalMana.setManaProducerOriginalId(originalId);
53          this.flag = conditionalMana.getFlag();
54          this.duration = Duration.EndOfStep;
55      }
56  
57      public ManaPoolItem(final ManaPoolItem item) {
58          this.red = item.red;
59          this.green = item.green;
60          this.blue = item.blue;
61          this.white = item.white;
62          this.black = item.black;
63          this.colorless = item.colorless;
64          if (item.conditionalMana != null) {
65              this.conditionalMana = item.conditionalMana.copy();
66          }
67          this.sourceObject = item.sourceObject;
68          this.originalId = item.originalId;
69          this.flag = item.flag;
70          this.duration = item.duration;
71          this.stock = item.stock;
72      }

169     public int count() {
170         if (conditionalMana == null) {
171             return red + green + blue + white + black + colorless;
172         }
173         return conditionalMana.count();
174     }

176     public int get(ManaType manaType) {
177         switch (manaType) {
178             case BLACK:
179                 return black;
180             case BLUE:
181                 return blue;
182             case GREEN:
183                 return green;
184             case RED:
185                 return red;
186             case WHITE:
187                 return white;
188             case COLORLESS:
189                 return colorless;
190         }
191         return 0;
192     }

194     public ManaType getFirstAvailable() {
195         if (black > 0) {
196             return ManaType.BLACK;
197         } else if (blue > 0) {
198             return ManaType.BLUE;
199         } else if (green > 0) {
200             return ManaType.GREEN;
201         } else if (red > 0) {
202             return ManaType.RED;
203         } else if (white > 0) {
204             return ManaType.WHITE;
205         } else if (colorless > 0) {
206             return ManaType.COLORLESS;
207         }
208         if (conditionalMana != null) {
209             if (conditionalMana.getBlack() > 0) {
210                 return ManaType.BLACK;
211             } else if (conditionalMana.getBlue() > 0) {
212                 return ManaType.BLUE;
213             } else if (conditionalMana.getGreen() > 0) {
214                 return ManaType.GREEN;
215             } else if (conditionalMana.getRed() > 0) {
216                 return ManaType.RED;
217             } else if (conditionalMana.getWhite() > 0) {
218                 return ManaType.WHITE;
219             } else if (conditionalMana.getColorless() > 0) {
220                 return ManaType.COLORLESS;
221             }
222         }
223         return null;
224     }

226     public void remove(ManaType manaType) {
227         int oldCount = count();
228         switch (manaType) {
229             case BLACK:
230                 if (black > 0) {
231                     black--;
232                 }
233                 break;
234             case BLUE:
235                 if (blue > 0) {
236                     blue--;
237                 }
238                 break;
239             case GREEN:
240                 if (green > 0) {
241                     green--;
242                 }
243                 break;
244             case RED:
245                 if (red > 0) {
246                     red--;
247                 }
248                 break;
249             case WHITE:
250                 if (white > 0) {
251                     white--;
252                 }
253                 break;
254             case COLORLESS:
255                 if (colorless > 0) {
256                     colorless--;
257                 }
258                 break;
259         }
260         if (stock == oldCount && oldCount > count()) {
261             stock--;
262         }
263     }

71       * Get absolute score of the card.
72       * Depends on type, manacost, rating.
73       * If allowedColors is null then the rating is retrieved from the cache
74       *
75       * @param card
76       * @param allowedColors
77       * @return

80          return rateCard(card, allowedColors, true);

83      public static int rateCard(Card card, List<ColoredManaSymbol> allowedColors, boolean useCache) {
84          if (card == null) {
85              return 0;
86          }
87  
88          if (useCache && allowedColors == null && rated.containsKey(card.getName())) {
89              int rate = rated.get(card.getName());
90              return rate;
91          }
92  
93          int type;
94          if (card.isPlaneswalker()) {
95              type = 15;
96          } else if (card.isCreature()) {
97              type = 10;
98          } else if (card.getSubtype().contains(SubType.EQUIPMENT)) {
99              type = 8;
100         } else if (card.getSubtype().contains(SubType.AURA)) {
101             type = 5;
102         } else if (card.isInstant()) {
103             type = 7;
104         } else {
105             type = 6;
106         }
107         int score = getBaseCardScore(card) + 2 * type + getManaCostScore(card, allowedColors)
108                 + 40 * isRemoval(card);
109 
110         if (useCache && allowedColors == null)
111             rated.put(card.getName(), score);
112 
113         return score;
114     }

116     private static int isRemoval(Card card) {
117         if (card.isEnchantment() || card.isInstantOrSorcery()) {
118 
119             for (Ability ability : card.getAbilities()) {
120                 for (Effect effect : ability.getEffects()) {
121                     if (isEffectRemoval(card, ability, effect) == 1) {
122                         return 1;
123                     }
124                 }
125                 for (Mode mode : ability.getModes().values()) {
126                     for (Effect effect : mode.getEffects()) {
127                         if (isEffectRemoval(card, ability, effect) == 1) {
128                             return 1;
129                         }
130                     }
131                 }
132             }
133 
134         }
135         return 0;
136     }

138     private static int isEffectRemoval(Card card, Ability ability, Effect effect) {
139         // it's effect relates score, do not use custom outcome from ability
140         if (effect.getOutcome() == Outcome.Removal) {
141             // found removal
142             return 1;
143         }
144         if (effect instanceof FightTargetsEffect
145                 || effect instanceof DamageWithPowerFromOneToAnotherTargetEffect
146                 || effect instanceof DamageWithPowerFromSourceToAnotherTargetEffect) {
147             return 1;
148         }
149         if (effect.getOutcome() == Outcome.Damage || effect instanceof DamageTargetEffect) {
150             for (Target target : ability.getTargets()) {
151                 if (!(target instanceof TargetPlayerOrPlaneswalker)) {
152                     // found damage dealer
153                     return 1;
154                 }
155             }
156         }
157         if (effect.getOutcome() == Outcome.DestroyPermanent ||
158                 effect instanceof DestroyTargetEffect ||
159                 effect instanceof ExileTargetEffect ||
160                 effect instanceof ExileUntilSourceLeavesEffect) {
161             for (Target target : ability.getTargets()) {
162                 if (target instanceof TargetPermanent) {
163                     // found destroyer/exiler
164                     return 1;
165                 }
166             }
167         }
168         return 0;
169     }

172     /**
173      * Return rating of the card.
174      *
175      * @param card Card to rate.
176      * @return Rating number from [1:100].
177      */
178     public static int getBaseCardScore(Card card) {
179         // same card name must have same rating
180 
181         // ratings from files
182         // lazy loading on first request
183         prepareAndLoadRatings();
184 
185         // ratings from card rarity
186         // some cards can have different rarity -- it's will be used from first set
187         int newRating;
188         if (card.getRarity() != null) {
189             switch (card.getRarity()) {
190                 case COMMON:
191                     newRating = DEFAULT_NOT_RATED_CARD_RATING;
192                     break;
193                 case UNCOMMON:
194                     newRating = DEFAULT_NOT_RATED_UNCOMMON_RATING;
195                     break;
196                 case RARE:
197                     newRating = DEFAULT_NOT_RATED_RARE_RATING;
198                     break;
199                 case MYTHIC:
200                     newRating = DEFAULT_NOT_RATED_MYTHIC_RATING;
201                     break;
202                 default:
203                     if (isBasicLand(card)) {
204                         newRating = DEFAULT_BASIC_LAND_RATING;
205                     } else {
206                         newRating = DEFAULT_NOT_RATED_CARD_RATING;
207                     }
208                     break;
209             }
210         } else {
211             // tokens
212             newRating = DEFAULT_NOT_RATED_CARD_RATING;
213         }
214 
215         int oldRating = baseRatings.getOrDefault(card.getName(), 0);
216         if (oldRating != 0 && oldRating != newRating) {
217             //log.info("card have different rating by sets: " + card.getName() + " (" + oldRating + " <> " + newRating + ")");
218         }
219 
220         if (oldRating != 0) {
221             return oldRating;
222         } else {
223             baseRatings.put(card.getName(), newRating);
224             return newRating;
225         }
226     }

228     /**
229      * reads the list of sets that have ratings csv files and read each file
230      */
231     public synchronized static void prepareAndLoadRatings() {
232         if (isLoaded) {
233             return;
234         }
235 
236         // load sets list
237         List<String> setsToLoad = new LinkedList<>();
238         try {
239             InputStream is = RateCard.class.getResourceAsStream(RATINGS_SET_LIST);
240             Scanner scanner = new Scanner(is);
241             while (scanner.hasNextLine()) {
242                 String line = scanner.nextLine();
243                 if (!line.substring(0, 1).equals("#")) {
244                     setsToLoad.add(line);
245                 }
246             }
247         } catch (Throwable e) {
248             log.error("Failed to read ratings sets list file: " + RATINGS_SET_LIST, e);
249         }
250 
251         // load set data
252         String rateFile = "";
253         try {
254             for (String code : setsToLoad) {
255                 //log.info("Reading ratings for the set " + code);
256                 rateFile = RATINGS_DIR + code + ".csv";
257                 readFromFile(rateFile);
258             }
259         } catch (Exception e) {
260             log.error("Failed to read ratings set file: " + rateFile, e);
261         }
262 
263         isLoaded = true;
264     }

266     /**
267      * reads ratings from the file
268      */
269     private synchronized static void readFromFile(String path) {
270         // card must get max rating from multiple cards
271         Integer min = Integer.MAX_VALUE, max = 0;
272         Map<String, Integer> thisFileRatings = new HashMap<>();
273 
274         // load
275         InputStream is = RateCard.class.getResourceAsStream(path);
276         Scanner scanner = new Scanner(is);
277         while (scanner.hasNextLine()) {
278             String line = scanner.nextLine();
279             String[] s = line.split(":");
280             if (s.length == 2) {
281                 Integer rating = Integer.parseInt(s[1].trim());
282                 String name = s[0].trim();
283                 name = name.replace("’", "'");
284                 if (rating > max) {
285                     max = rating;
286                 }
287                 if (rating < min) {
288                     min = rating;
289                 }
290                 thisFileRatings.put(name, rating);
291             }
292         }
293 
294         // normalize for the file to [1..100]
295         for (Map.Entry<String, Integer> ratingByName : thisFileRatings.entrySet()) {
296             int r = ratingByName.getValue();
297             String name = ratingByName.getKey();
298             int newRating = (int) (100.0f * (r - min) / (max - min));
299             int oldRating = baseRatings.getOrDefault(name, 0);
300             if (newRating > oldRating) {
301                 baseRatings.put(name, newRating);
302             }
303         }
304     }

306     private static final int[] SINGLE_PENALTY = {0, 1, 1, 3, 6, 9};
307     private static final int MULTICOLOR_BONUS = 15;
308 
309     /**
310      * Get manacost score.
311      * Depends on chosen colors. Returns negative score for those cards that doesn't fit allowed colors.
312      * If allowed colors are not chosen, then score based on converted cost is returned with penalty for heavy colored cards.
313      * gives bonus to multicolor cards that fit within allowed colors and if allowed colors is <5
314      *
315      * @param card
316      * @param allowedColors Can be null.
317      * @return
318      */
319     private static int getManaCostScore(Card card, List<ColoredManaSymbol> allowedColors) {
320         int converted = card.getManaValue();
321         if (allowedColors == null) {
322             int colorPenalty = 0;
323             for (String symbol : card.getManaCostSymbols()) {
324                 if (isColoredMana(symbol)) {
325                     colorPenalty++;
326                 }
327             }
328             return 2 * (converted - colorPenalty + 1);
329         }
330         
331         // Basic lands have no value so they're always treated as off-color
332         if (isBasicLand(card)) {
333             return OFF_COLOR_PENALTY;
334         }
335         
336         final Map<String, Integer> singleCount = new HashMap<>();
337         int maxSingleCount = 0;
338         for (String symbol : card.getManaCostSymbols()) {
339             int count = 0;
340             symbol = symbol.replace("{", "").replace("}", "");
341             if (isColoredMana(symbol)) {
342                 for (ColoredManaSymbol allowed : allowedColors) {
343                     if (allowed.toString().equals(symbol)) {
344                         count++;
345                     }
346                 }
347                 if (count == 0) {
348                     return OFF_COLOR_PENALTY;
349                 }
350                 Integer typeCount = singleCount.get(symbol);
351                 if (typeCount == null) {
352                     typeCount = 0;
353                 }
354                 typeCount += 1;
355                 singleCount.put(symbol, typeCount);
356                 maxSingleCount = Math.max(maxSingleCount, typeCount);
357             }
358         }
359         if (maxSingleCount > 5)
360             maxSingleCount = 5;
361 
362         int rate = 2 * converted + 3 * (10 - SINGLE_PENALTY[maxSingleCount]);
363         if (singleCount.size() > 1 && singleCount.size() < 5) {
364             rate += MULTICOLOR_BONUS;
365         }
366         return rate;
367     }

369     /**
370      * Determines whether mana symbol is color.
371      *
372      * @param symbol
373      * @return
374      */
375     public static boolean isColoredMana(String symbol) {
376         String s = symbol;
377         if (s.length() > 1) {
378             s = s.replace("{", "").replace("}", "");
379         }
380         if (s.length() > 1) {
381             return false;
382         }
383         return s.equals("W") || s.equals("G") || s.equals("U") || s.equals("B") || s.equals("R");
384     }

386     /**
387      * Return number of color mana symbols in manacost.
388      *
389      * @param card
390      * @return
391      */
392     public static int getColorManaCount(Card card) {
393         int count = 0;
394         for (String symbol : card.getManaCostSymbols()) {
395             if (isColoredMana(symbol)) {
396                 count++;
397             }
398         }
399         return count;
400     }

402     /**
403      * Return number of different color mana symbols in manacost.
404      *
405      * @param card
406      * @return
407      */
408     public static int getDifferentColorManaCount(Card card) {
409         Set<String> symbols = new HashSet<>();
410         for (String symbol : card.getManaCostSymbols()) {
411             if (isColoredMana(symbol)) {
412                 symbols.add(symbol);
413             }
414         }
415         return symbols.size();
416     }

418     /**
419      * Return true if the card is one of the basic land types that can be added to the deck for free.
420      *
421      * @param card
422      * @return
423      */
424      public static boolean isBasicLand(Card card) {
425         String name = card.getName();
426         if (name.equals("Plains")
427                 || name.equals("Island")
428                 || name.equals("Swamp")
429                 || name.equals("Mountain")
430                 || name.equals("Forest")) {
431             return true;
432         } else {
433             return false;
434         }

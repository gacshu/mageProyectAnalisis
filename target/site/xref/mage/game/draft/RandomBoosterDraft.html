3   package mage.game.draft;
4   
5   import java.util.ArrayList;
6   import java.util.List;
7   import mage.cards.ExpansionSet;
8   import java.util.Collections;
9   
10  /**
11   *
12   * @author BrodyLodmell_at_googlemail.com
13   */
14  public class RandomBoosterDraft extends BoosterDraft {
15  
16      List<ExpansionSet> allSets;
17      List<ExpansionSet> useBoosters;
18      public RandomBoosterDraft(DraftOptions options, List<ExpansionSet> sets) {
19          super(options, sets);
20          if (sets.isEmpty()){
21              throw new RuntimeException("At least one set must be selected for random booster draft");
22          }
23          allSets = new ArrayList<>(sets);
24          resetBoosters();
25      }
26  
27      @Override
28      protected void openBooster() {
29          if (boosterNum <= numberBoosters) {
30              for (DraftPlayer player: players.values()) {
31                  player.setBooster(getNextBooster().create15CardBooster());
32              }
33          }
34      }
35  
36      private ExpansionSet getNextBooster() {
37          if (useBoosters.isEmpty()){
38              resetBoosters();
39          }
40          ExpansionSet theBooster = useBoosters.get(0);
41          useBoosters.remove(theBooster);
42          return theBooster;
43      }
44  
45      private void resetBoosters(){
46          useBoosters = new ArrayList<>(allSets);
47          Collections.shuffle(useBoosters);
48      }
49  }

27      public static boolean cast(ZoneChangeInfo info, Game game, Ability source) {
28          if (maybeRemoveFromSourceZone(info, game, source)) {
29              placeInDestinationZone(info, game, 0);
30              // create a group zone change event if a card is moved to stack for casting (it's always only one card, but some effects check for group events (one or more xxx))
31              Set<Card> cards = new HashSet<>();
32              Set<PermanentToken> tokens = new HashSet<>();
33              Card targetCard = getTargetCard(game, info.event.getTargetId());
34              if (targetCard instanceof PermanentToken) {
35                  tokens.add((PermanentToken) targetCard);
36              } else {
37                  cards.add(targetCard);
38              }
39              game.fireEvent(new ZoneChangeGroupEvent(
40                      cards, 
41                      tokens, 
42                      info.event.getSourceId(), 
43                      info.event.getSource(), 
44                      info.event.getPlayerId(), 
45                      info.event.getFromZone(), 
46                      info.event.getToZone()));
47              // normal movement
48              game.fireEvent(info.event);
49              return true;
50          }
51          return false;
52      }

54      public static boolean moveCard(ZoneChangeInfo info, Game game, Ability source) {
55          List<ZoneChangeInfo> list = new ArrayList<>();
56          list.add(info);
57          return !moveCards(list, game, source).isEmpty();
58      }

60      public static List<ZoneChangeInfo> moveCards(List<ZoneChangeInfo> zoneChangeInfos, Game game, Ability source) {
61          // Handle Unmelded Meld Cards
62          for (ListIterator<ZoneChangeInfo> itr = zoneChangeInfos.listIterator(); itr.hasNext(); ) {
63              ZoneChangeInfo info = itr.next();
64              MeldCard card = game.getMeldCard(info.event.getTargetId());
65              // Copies should be handled as normal cards.
66              if (card != null && !card.isMelded(game) && !card.isCopy()) {
67                  ZoneChangeInfo.Unmelded unmelded = new ZoneChangeInfo.Unmelded(info, game);
68                  if (unmelded.subInfo.isEmpty()) {
69                      itr.remove();
70                  } else {
71                      itr.set(unmelded);
72                  }
73              }
74          }
75  
76          // process Modal Double Faces cards (e.g. put card from hand)
77          // rules:
78          // If an effect puts a double-faced card onto the battlefield, it enters with its front face up.
79          // If that front face can’t be put onto the battlefield, it doesn’t enter the battlefield.
80          // For example, if an effect exiles Sejiri Glacier and returns it to the battlefield,
81          // it remains in exile because an instant can’t be put onto the battlefield.
82          for (ListIterator<ZoneChangeInfo> itr = zoneChangeInfos.listIterator(); itr.hasNext(); ) {
83              ZoneChangeInfo info = itr.next();
84              if (info.event.getToZone().equals(Zone.BATTLEFIELD)) {
85                  Card card = game.getCard(info.event.getTargetId());
86                  if (card instanceof ModalDoubleFacesCard || card instanceof ModalDoubleFacesCardHalf) {
87                      boolean forceToMainSide = false;
88  
89                      // if effect put half mdf card to battlefield then it must be the main side only (example: return targeted half card to battle)
90                      if (card instanceof ModalDoubleFacesCardHalf && !source.getAbilityType().isPlayCardAbility()) {
91                          forceToMainSide = true;
92                      }
93  
94                      // if effect put mdf card to battlefield then it must be main side only
95                      if (card instanceof ModalDoubleFacesCard) {
96                          forceToMainSide = true;
97                      }
98  
99                      if (forceToMainSide) {
100                         info.event.setTargetId(((ModalDoubleFacesCard) card.getMainCard()).getLeftHalfCard().getId());
101                     }
102                 }
103             }
104         }
105 
106         zoneChangeInfos.removeIf(zoneChangeInfo -> !maybeRemoveFromSourceZone(zoneChangeInfo, game, source));
107         int createOrder = 0;
108         for (ZoneChangeInfo zoneChangeInfo : zoneChangeInfos) {
109             if (createOrder == 0 && Zone.BATTLEFIELD.equals(zoneChangeInfo.event.getToZone())) {
110                 // All permanents go to battlefield at the same time (=create order)
111                 createOrder = game.getState().getNextPermanentOrderNumber();
112             }
113             placeInDestinationZone(zoneChangeInfo, game, createOrder);
114             if (game.getPhase() != null) { // moving cards to zones before game started does not need events
115                 game.addSimultaneousEvent(zoneChangeInfo.event);
116             }
117         }
118         return zoneChangeInfos;
119     }

121     private static void placeInDestinationZone(ZoneChangeInfo info, Game game, int createOrder) {
122         // Handle unmelded cards
123         if (info instanceof ZoneChangeInfo.Unmelded) {
124             ZoneChangeInfo.Unmelded unmelded = (ZoneChangeInfo.Unmelded) info;
125             Zone toZone = null;
126             for (ZoneChangeInfo subInfo : unmelded.subInfo) {
127                 toZone = subInfo.event.getToZone();
128                 placeInDestinationZone(subInfo, game, createOrder);
129             }
130             // We arbitrarily prefer the bottom half card. This should never be relevant.
131             if (toZone != null) {
132                 game.setZone(unmelded.event.getTargetId(), toZone);
133             }
134             return;
135         }
136         // Handle normal cases
137         ZoneChangeEvent event = info.event;
138         Zone toZone = event.getToZone();
139         Card targetCard = getTargetCard(game, event.getTargetId());
140 
141         Cards cardsToMove = null; // moving real cards
142         Map<Zone, Cards> cardsToUpdate = new LinkedHashMap<>(); // updating all card's parts (must be ordered LinkedHashMap)
143         cardsToUpdate.put(toZone, new CardsImpl());
144         cardsToUpdate.put(Zone.OUTSIDE, new CardsImpl());
145         // if we're moving a token it shouldn't be put into any zone as an object.
146         if (!(targetCard instanceof Permanent) && targetCard != null) {
147             if (targetCard instanceof MeldCard) {
148                 // meld/group cards must be independent (use can choose order)
149                 cardsToMove = ((MeldCard) targetCard).getHalves();
150                 cardsToUpdate.get(toZone).addAll(cardsToMove);
151             } else if (targetCard instanceof ModalDoubleFacesCard
152                     || targetCard instanceof ModalDoubleFacesCardHalf) {
153                 // mdf cards must be moved as single object, but each half must be updated separately
154                 ModalDoubleFacesCard mdfCard = (ModalDoubleFacesCard) targetCard.getMainCard();
155                 cardsToMove = new CardsImpl(mdfCard);
156                 cardsToUpdate.get(toZone).add(mdfCard);
157                 // example: cast left side
158                 // result:
159                 // * main to battlefield
160                 // * left to battlefield
161                 // * right to outside (it helps to ignore all triggers and other effects from that card)
162                 switch (toZone) {
163                     case STACK:
164                     case BATTLEFIELD:
165                         if (targetCard.getId().equals(mdfCard.getLeftHalfCard().getId())) {
166                             // play left
167                             cardsToUpdate.get(toZone).add(mdfCard.getLeftHalfCard());
168                             cardsToUpdate.get(Zone.OUTSIDE).add(mdfCard.getRightHalfCard());
169                         } else if (targetCard.getId().equals(mdfCard.getRightHalfCard().getId())) {
170                             // play right
171                             cardsToUpdate.get(toZone).add(mdfCard.getRightHalfCard());
172                             cardsToUpdate.get(Zone.OUTSIDE).add(mdfCard.getLeftHalfCard());
173                         } else {
174                             // cast mdf (only on stack)
175                             if (!toZone.equals(Zone.STACK)) {
176                                 throw new IllegalStateException("Wrong mdf card move to " + toZone + " in placeInDestinationZone");
177                             }
178                             cardsToUpdate.get(toZone).add(mdfCard.getLeftHalfCard());
179                             cardsToUpdate.get(toZone).add(mdfCard.getRightHalfCard());
180                         }
181                         break;
182                     default:
183                         // move all parts
184                         cardsToUpdate.get(toZone).add(mdfCard.getLeftHalfCard());
185                         cardsToUpdate.get(toZone).add(mdfCard.getRightHalfCard());
186                         break;
187                 }
188             } else {
189                 cardsToMove = new CardsImpl(targetCard);
190                 cardsToUpdate.get(toZone).addAll(cardsToMove);
191             }
192 
193             Player owner = game.getPlayer(targetCard.getOwnerId());
194             switch (toZone) {
195                 case HAND:
196                     for (Card card : cardsToMove.getCards(game)) {
197                         game.getPlayer(card.getOwnerId()).getHand().add(card);
198                     }
199                     break;
200                 case GRAVEYARD:
201                     for (Card card : chooseOrder(
202                             "order to put in graveyard (last chosen will be on top)", cardsToMove, owner, game)) {
203                         game.getPlayer(card.getOwnerId()).getGraveyard().add(card);
204                     }
205                     break;
206                 case LIBRARY:
207                     if (info instanceof ZoneChangeInfo.Library && ((ZoneChangeInfo.Library) info).top) {
208                         // on top
209                         for (Card card : chooseOrder(
210                                 "order to put on top of library (last chosen will be topmost)", cardsToMove, owner, game)) {
211                             game.getPlayer(card.getOwnerId()).getLibrary().putOnTop(card, game);
212                         }
213                     } else {
214                         // on bottom
215                         for (Card card : chooseOrder(
216                                 "order to put on bottom of library (last chosen will be bottommost)", cardsToMove, owner, game)) {
217                             game.getPlayer(card.getOwnerId()).getLibrary().putOnBottom(card, game);
218                         }
219                     }
220                     break;
221                 case EXILED:
222                     for (Card card : cardsToMove.getCards(game)) {
223                         if (info instanceof ZoneChangeInfo.Exile && ((ZoneChangeInfo.Exile) info).id != null) {
224                             ZoneChangeInfo.Exile exileInfo = (ZoneChangeInfo.Exile) info;
225                             game.getExile().createZone(exileInfo.id, exileInfo.name).add(card);
226                         } else {
227                             game.getExile().getPermanentExile().add(card);
228                         }
229                     }
230                     break;
231                 case COMMAND:
232                     // There should never be more than one card here.
233                     for (Card card : cardsToMove.getCards(game)) {
234                         game.addCommander(new Commander(card));
235                     }
236                     break;
237                 case STACK:
238                     // There should never be more than one card here.
239                     for (Card card : cardsToMove.getCards(game)) {
240                         Spell spell;
241                         if (info instanceof ZoneChangeInfo.Stack && ((ZoneChangeInfo.Stack) info).spell != null) {
242                             spell = ((ZoneChangeInfo.Stack) info).spell;
243                         } else {
244                             spell = new Spell(card, card.getSpellAbility().copy(), card.getOwnerId(), event.getFromZone(), game);
245                         }
246                         spell.syncZoneChangeCounterOnStack(card, game);
247                         game.getStack().push(spell);
248                         game.getState().setZone(spell.getId(), Zone.STACK);
249                         game.getState().setZone(card.getId(), Zone.STACK);
250                     }
251                     break;
252                 case BATTLEFIELD:
253                     Permanent permanent = event.getTarget();
254                     game.addPermanent(permanent, createOrder);
255                     game.getPermanentsEntering().remove(permanent.getId());
256                     break;
257                 default:
258                     throw new UnsupportedOperationException("to Zone " + toZone.toString() + " not supported yet");
259             }
260         }
261 
262         // update zone in main
263         game.setZone(event.getTargetId(), event.getToZone());
264 
265         // update zone in other parts (meld cards, mdf half cards)
266         cardsToUpdate.entrySet().forEach(entry -> {
267             for (Card card : entry.getValue().getCards(game)) {
268                 if (!card.getId().equals(event.getTargetId())) {
269                     game.setZone(card.getId(), entry.getKey());
270                 }
271             }
272         });
273 
274         // reset meld status
275         if (targetCard instanceof MeldCard) {
276             if (event.getToZone() != Zone.BATTLEFIELD) {
277                 ((MeldCard) targetCard).setMelded(false, game);
278             }
279         }
280     }

282     public static Card getTargetCard(Game game, UUID targetId) {
283         Card card = game.getCard(targetId);
284         if (card != null) {
285             return card;
286         }
287 
288         card = game.getMeldCard(targetId);
289         if (card != null) {
290             return card;
291         }
292 
293         return game.getPermanent(targetId);
294     }

296     private static boolean maybeRemoveFromSourceZone(ZoneChangeInfo info, Game game, Ability source) {
297         ZoneChangeEvent event = info.event;
298 
299         // Handle Unmelded Cards
300         if (info instanceof ZoneChangeInfo.Unmelded) {
301             ZoneChangeInfo.Unmelded unmelded = (ZoneChangeInfo.Unmelded) info;
302             MeldCard meld = game.getMeldCard(event.getTargetId());
303             for (Iterator<ZoneChangeInfo> itr = unmelded.subInfo.iterator(); itr.hasNext(); ) {
304                 ZoneChangeInfo subInfo = itr.next();
305                 if (!maybeRemoveFromSourceZone(subInfo, game, source)) {
306                     itr.remove();
307                 } else if (Objects.equals(subInfo.event.getTargetId(), meld.getTopHalfCard().getId())) {
308                     meld.setTopLastZoneChangeCounter(meld.getTopHalfCard().getZoneChangeCounter(game));
309                 } else if (Objects.equals(subInfo.event.getTargetId(), meld.getBottomHalfCard().getId())) {
310                     meld.setBottomLastZoneChangeCounter(meld.getBottomHalfCard().getZoneChangeCounter(game));
311                 }
312             }
313             if (unmelded.subInfo.isEmpty()) {
314                 return false;
315             }
316             // We arbitrarily prefer the bottom half card. This should never be relevant.
317             meld.updateZoneChangeCounter(game, unmelded.subInfo.get(unmelded.subInfo.size() - 1).event);
318             return true;
319         }
320 
321         // Handle all normal cases
322         Card card = getTargetCard(game, event.getTargetId());
323         if (card == null) {
324             // If we can't find the card we can't remove it.
325             return false;
326         }
327 
328         boolean success = false;
329         if (info.faceDown) {
330             card.setFaceDown(true, game);
331         } else if (event.getToZone().equals(Zone.BATTLEFIELD)) {
332             if (!card.isPermanent(game)
333                     && (!card.isTransformable() || Boolean.FALSE.equals(game.getState().getValue(TransformAbility.VALUE_KEY_ENTER_TRANSFORMED + card.getId())))) {
334                 // Non permanents (Instants, Sorceries, ... stay in the zone they are if an abilty/effect tries to move it to the battlefield
335                 return false;
336             }
337         }
338 
339         if (!game.replaceEvent(event)) {
340             Zone fromZone = event.getFromZone();
341             if (event.getToZone() == Zone.BATTLEFIELD) {
342                 // prepare card and permanent
343                 // If needed take attributes from the spell (e.g. color of spell was changed)
344                 card = takeAttributesFromSpell(card, event, game);
345                 // controlling player can be replaced so use event player now
346                 Permanent permanent;
347                 if (card instanceof MeldCard) {
348                     permanent = new PermanentMeld(card, event.getPlayerId(), game);
349                 } else if (card instanceof ModalDoubleFacesCard) {
350                     // main mdf card must be processed before that call (e.g. only halfes can be moved to battlefield)
351                     throw new IllegalStateException("Unexpected trying of move mdf card to battlefield instead half");
352                 } else if (card instanceof Permanent) {
353                     throw new IllegalStateException("Unexpected trying of move permanent to battlefield instead card");
354                 } else {
355                     permanent = new PermanentCard(card, event.getPlayerId(), game);
356                 }
357 
358                 // put onto battlefield with possible counters
359                 game.getPermanentsEntering().put(permanent.getId(), permanent);
360                 card.checkForCountersToAdd(permanent, source, game);
361                 permanent.setTapped(info instanceof ZoneChangeInfo.Battlefield
362                         && ((ZoneChangeInfo.Battlefield) info).tapped);
363 
364                 permanent.setFaceDown(info.faceDown, game);
365                 if (info.faceDown) {
366                     card.setFaceDown(false, game);
367                 }
368 
369                 // make sure the controller of all continuous effects of this card are switched to the current controller
370                 game.setScopeRelevant(true);
371                 game.getContinuousEffects().setController(permanent.getId(), permanent.getControllerId());
372                 if (permanent.entersBattlefield(source, game, fromZone, true)
373                         && card.removeFromZone(game, fromZone, source)) {
374                     success = true;
375                     event.setTarget(permanent);
376                 } else {
377                     // revert controller to owner if permanent does not enter
378                     game.getContinuousEffects().setController(permanent.getId(), permanent.getOwnerId());
379                     game.getPermanentsEntering().remove(permanent.getId());
380                 }
381                 game.setScopeRelevant(false);
382             } else if (event.getTarget() != null) {
383                 card.setFaceDown(info.faceDown, game);
384                 Permanent target = event.getTarget();
385                 success = target.removeFromZone(game, fromZone, source)
386                         && game.getPlayer(target.getControllerId()).removeFromBattlefield(target, source, game);
387             } else {
388                 card.setFaceDown(info.faceDown, game);
389                 success = card.removeFromZone(game, fromZone, source);
390             }
391         }
392         if (success) {
393             // change ZCC on real enter
394             // warning, tokens creation code uses same zcc logic as cards (+1 zcc on enter to battlefield)
395             // so if you want to change zcc logic here (but I know you don't) then change token code
396             // too in TokenImpl.putOntoBattlefieldHelper
397             // KickerTest do many tests for token's zcc
398             if (event.getToZone() == Zone.BATTLEFIELD && event.getTarget() != null) {
399                 event.getTarget().updateZoneChangeCounter(game, event);
400             } else if (!(card instanceof Permanent)) {
401                 card.updateZoneChangeCounter(game, event);
402             }
403         }
404         return success;
405     }

407     public static List<Card> chooseOrder(String message, Cards cards, Player player, Game game) {
408         List<Card> order = new ArrayList<>();
409         if (cards.isEmpty()) {
410             return order;
411         }
412         TargetCard target = new TargetCard(Zone.ALL, new FilterCard(message));
413         target.setRequired(true);
414         while (player.canRespond() && cards.size() > 1) {
415             player.choose(Outcome.Neutral, cards, target, game);
416             UUID targetObjectId = target.getFirstTarget();
417             order.add(cards.get(targetObjectId, game));
418             cards.remove(targetObjectId);
419             target.clearChosen();
420         }
421         order.add(cards.getCards(game).iterator().next());
422         return order;
423     }

425     private static Card takeAttributesFromSpell(Card card, ZoneChangeEvent event, Game game) {
426         card = card.copy();
427         if (Zone.STACK == event.getFromZone()) {
428             Spell spell = game.getStack().getSpell(event.getTargetId());
429             if (spell != null && !spell.isFaceDown(game)) {
430                 if (!card.getColor(game).equals(spell.getColor(game))) {
431                     // the card that is referenced to in the permanent is copied and the spell attributes are set to this copied card
432                     card.getColor(game).setColor(spell.getColor(game));
433                 }
434             }
435         }
436         return card;
437     }

5   public enum MulliganType {
6   
7       GAME_DEFAULT("Game Default"),
8       VANCOUVER("Vancouver"),
9       PARIS("Paris"),

12  
13      private final String displayName;
14  
15      MulliganType(String displayName) {
16          this.displayName = displayName;
17      }
18  
19      public Mulligan getMulligan(int freeMulligans) {
20          switch (this) {
21              case PARIS:
22                  return new ParisMulligan(freeMulligans);
23              case CANADIAN_HIGHLANDER:
24                  return new CanadianHighlanderMulligan(freeMulligans);
25              case VANCOUVER:
26                  return new VancouverMulligan(freeMulligans);
27              default:
28              case LONDON:
29                  return new LondonMulligan(freeMulligans);
30          }
31      }
32  
33      @Override
34      public String toString() {
35          return displayName;
36      }
37  
38      public static MulliganType valueByName(String name) {
39          String search = (name != null ? name : "").toUpperCase(Locale.ENGLISH);
40  
41          MulliganType res = GAME_DEFAULT;
42          for (MulliganType m : values()) {
43              if (m.displayName.toUpperCase(Locale.ENGLISH).equals(search)) {
44                  res = m;
45                  break;
46              }
47          }
48          return res;
49      }
50  
51      public MulliganType orDefault(MulliganType defaultMulligan) {
52          if (this == GAME_DEFAULT) {
53              return defaultMulligan;
54          }
55          return this;
56      }

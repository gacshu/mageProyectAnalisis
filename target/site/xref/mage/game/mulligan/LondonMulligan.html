15  public class LondonMulligan extends Mulligan {
16  
17      private final Map<UUID, Integer> startingHandSizes = new HashMap<>();
18      private final Map<UUID, Integer> openingHandSizes = new HashMap<>();
19  
20      public LondonMulligan(int freeMulligans) {
21          super(freeMulligans);
22      }
23  
24      LondonMulligan(final LondonMulligan mulligan) {
25          super(mulligan);
26  
27          this.startingHandSizes.putAll(mulligan.startingHandSizes);
28          this.openingHandSizes.putAll(mulligan.openingHandSizes);
29      }
30  
31      @Override
32      public void executeMulliganPhase(Game game, int startingHandSize) {
33          /*
34           * 103.4. Each player draws a number of cards equal to their starting hand size, which is normally
35           * seven. (Some effects can modify a player’s starting hand size.) A player who is dissatisfied with
36           * their initial hand may take a mulligan. First, the starting player declares whether they will
37           * take a mulligan. Then each other player in turn order does the same. Once each player has made a
38           * declaration, all players who decided to take mulligans do so at the same time. To take a mulligan,
39           * a player shuffles the cards in their hand back into their library, draws a new hand of cards equal
40           * to their starting hand size, then puts a number of those cards onto the bottom of their library in
41           * any order equal to the number of times that player has taken a mulligan. Once a player chooses not
42           * to take a mulligan, the remaining cards become the player’s opening hand, and that player may not
43           * take any further mulligans. This process is then repeated until no player takes a mulligan. A
44           * player can’t take a number of mulligans greater their starting hand size.
45           *
46           * https://magic.wizards.com/en/articles/archive/competitive-gaming/mythic-championship-ii-format-and-london-test-2019-02-21
47           */
48  
49          for (UUID playerId : game.getState().getPlayerList(game.getStartingPlayerId())) {
50              openingHandSizes.put(playerId, startingHandSize);
51              startingHandSizes.put(playerId, startingHandSize);
52          }
53  
54          super.executeMulliganPhase(game, startingHandSize);
55      }
56  
57      @Override
58      public int mulliganDownTo(Game game, UUID playerId) {
59          Player player = game.getPlayer(playerId);
60          int deduction = 1;
61          if (freeMulligans > 0) {
62              if (usedFreeMulligans.containsKey(player.getId())) {
63                  int used = usedFreeMulligans.get(player.getId());
64                  if (used < freeMulligans) {
65                      deduction = 0;
66                  }
67              } else {
68                  deduction = 0;
69              }
70          }
71          return openingHandSizes.get(playerId) - deduction;
72      }
73  
74      @Override
75      public boolean canTakeMulligan(Game game, Player player) {
76          return super.canTakeMulligan(game, player) && openingHandSizes.get(player.getId()) > 0;
77      }
78  
79      @Override
80      public void mulligan(Game game, UUID playerId) {
81          Player player = game.getPlayer(playerId);
82          int numCards = startingHandSizes.get(player.getId());
83          player.getLibrary().addAll(player.getHand().getCards(game), game);
84          player.getHand().clear();
85          player.shuffleLibrary(null, game);
86          int deduction = 1;
87          if (freeMulligans > 0) {
88              if (usedFreeMulligans.containsKey(player.getId())) {
89                  int used = usedFreeMulligans.get(player.getId());
90                  if (used < freeMulligans) {
91                      deduction = 0;
92                      usedFreeMulligans.put(player.getId(), used + 1);
93                  }
94              } else {
95                  deduction = 0;
96                  usedFreeMulligans.put(player.getId(), 1);
97              }
98          }
99          openingHandSizes.put(playerId, openingHandSizes.get(playerId) - deduction);
100         int newHandSize = openingHandSizes.get(player.getId());
101         if (deduction == 0) {
102             game.fireInformEvent(player.getLogName() +
103                     " mulligans for free.");
104         } else {
105             game.fireInformEvent(player.getLogName() +
106                     " mulligans down to " +
107                     newHandSize +
108                     (newHandSize == 1 ? " card" : " cards"));
109         }
110         player.drawCards(numCards, null, game);
111 
112         while (player.canRespond() && player.getHand().size() > newHandSize) {
113             Target target = new TargetCardInHand(new FilterCard("card (" + (player.getHand().size() - newHandSize) + " more) to put on the bottom of your library"));
114             player.chooseTarget(Outcome.Discard, target, null, game);
115             player.putCardsOnBottomOfLibrary(new CardsImpl(target.getTargets()), game, null, true);
116         }
117     }
118 
119     @Override
120     public void endMulligan(Game game, UUID playerId) {
121     }
122 
123     @Override
124     public LondonMulligan copy() {
125         return new LondonMulligan(this);
126     }
127 }

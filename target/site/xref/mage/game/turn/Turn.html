43      public Turn(final Turn turn) {
44          if (turn.currentPhase != null) {
45              this.currentPhase = turn.currentPhase.copy();
46          }
47          this.activePlayerId = turn.activePlayerId;
48          for (Phase phase : turn.phases) {
49              this.phases.add(phase.copy());
50          }
51          this.declareAttackersStepStarted = turn.declareAttackersStepStarted;
52          this.endTurn = turn.endTurn;

88       * @param game
89       * @param activePlayer
90       * @return true if turn is skipped
91       */
92      public boolean play(Game game, Player activePlayer) {
93          // uncomment this to trace triggered abilities and/or continous effects 
94          // TraceUtil.traceTriggeredAbilities(game);
95          // game.getState().getContinuousEffects().traceContinuousEffects(game);
96          activePlayer.becomesActivePlayer();
97          this.setDeclareAttackersStepStarted(false);
98          if (game.isPaused() || game.checkIfGameIsOver()) {
99              return false;
100         }
101 
102 
103         if (game.getState().getTurnMods().skipTurn(activePlayer.getId())) {
104             game.informPlayers(activePlayer.getLogName() + " skips their turn.");
105             return true;
106         }
107         logStartOfTurn(game, activePlayer);
108 
109         checkTurnIsControlledByOtherPlayer(game, activePlayer.getId());
110 
111         this.activePlayerId = activePlayer.getId();
112         resetCounts();
113         game.getPlayer(activePlayer.getId()).beginTurn(game);
114         for (Phase phase : phases) {
115             if (game.isPaused() || game.checkIfGameIsOver()) {
116                 return false;
117             }
118             if (isEndTurnRequested() && phase.getType() != TurnPhase.END) {
119                 continue;
120             }
121             currentPhase = phase;
122             game.fireEvent(new PhaseChangedEvent(activePlayer.getId(), null));
123             if (game.getState().getTurnMods().skipPhase(
124                     activePlayer.getId(), currentPhase.getType()
125             ) || !phase.play(game, activePlayer.getId())) {
126                 continue;
127             }
128             if (game.executingRollback()) {
129                 return false;
130             }
131             //20091005 - 500.4/703.4n
132             game.emptyManaPools(null);
133             game.saveState(false);
134 
135             //20091005 - 500.8
136             while (playExtraPhases(game, phase.getType())) ;
137         }
138         return false;
139     }

141     public void resumePlay(Game game, boolean wasPaused) {
142         activePlayerId = game.getActivePlayerId();
143         UUID priorityPlayerId = game.getPriorityPlayerId();
144         TurnPhase phaseType = game.getTurnPhaseType();
145         PhaseStep stepType = game.getTurnStepType();
146 
147         Iterator<Phase> it = phases.iterator();
148         Phase phase;
149         do {
150             phase = it.next();
151             currentPhase = phase;
152         } while (phase.type != phaseType);
153         if (phase.resumePlay(game, stepType, wasPaused)) {
154             //20091005 - 500.4/703.4n
155             game.emptyManaPools(null);
156             //game.saveState();
157             //20091005 - 500.8
158             playExtraPhases(game, phase.getType());
159         }
160         while (it.hasNext()) {
161             phase = it.next();
162             if (game.isPaused() || game.checkIfGameIsOver()) {
163                 return;
164             }
165             currentPhase = phase;
166             if (!game.getState().getTurnMods().skipPhase(activePlayerId, currentPhase.getType())) {
167                 if (phase.play(game, activePlayerId)) {
168                     //20091005 - 500.4/703.4n
169                     game.emptyManaPools(null);
170                     //game.saveState();
171                     //20091005 - 500.8
172                     playExtraPhases(game, phase.getType());
173                 }
174             }
175             if (!currentPhase.equals(phase)) { // phase was changed from the card
176                 game.fireEvent(new PhaseChangedEvent(activePlayerId, null));
177                 break;
178             }
179         }
180     }

182     private void checkTurnIsControlledByOtherPlayer(Game game, UUID activePlayerId) {
183         UUID newControllerId = game.getState().getTurnMods().controlsTurn(activePlayerId);
184         if (newControllerId != null && !newControllerId.equals(activePlayerId)) {
185             game.getPlayer(newControllerId).controlPlayersTurn(game, activePlayerId);
186         }

195     private boolean playExtraPhases(Game game, TurnPhase afterPhase) {
196         while (true) {
197             TurnMod extraPhaseTurnMod = game.getState().getTurnMods().extraPhase(activePlayerId, afterPhase);
198             if (extraPhaseTurnMod == null) {
199                 return false;
200             }
201             TurnPhase extraPhase = extraPhaseTurnMod.getExtraPhase();
202             if (extraPhase == null) {
203                 return false;
204             }
205             Phase phase;
206             switch (extraPhase) {
207                 case BEGINNING:
208                     phase = new BeginningPhase();
209                     break;
210                 case PRECOMBAT_MAIN:
211                     phase = new PreCombatMainPhase();
212                     break;
213                 case COMBAT:
214                     phase = new CombatPhase();
215                     break;
216                 case POSTCOMBAT_MAIN:
217                     phase = new PostCombatMainPhase();
218                     break;
219                 default:
220                     phase = new EndPhase();
221             }
222             currentPhase = phase;
223             game.fireEvent(new PhaseChangedEvent(activePlayerId, extraPhaseTurnMod));
224             Player activePlayer = game.getPlayer(activePlayerId);
225             if (activePlayer != null && !game.isSimulation()) {
226                 game.informPlayers(activePlayer.getLogName() + " starts an additional " + phase.getType().toString() + " phase");
227             }
228             phase.play(game, activePlayerId);
229             afterPhase = extraPhase;
230         }
231     }

239     /**
240      * Used for some spells with end turn effect (e.g. Time Stop).
241      *
242      * @param game
243      * @param activePlayerId
244      * @param source
245      */
246     public void endTurn(Game game, UUID activePlayerId, Ability source) {
247         // Ending the turn this way (Time Stop) means the following things happen in order:
248 
249         setEndTurnRequested(true);
250 
251         // 1) All spells and abilities on the stack are exiled. This includes (e.g.) Time Stop, though it will continue to resolve.
252         // It also includes spells and abilities that can't be countered.
253         while (!game.hasEnded() && !game.getStack().isEmpty()) {
254             StackObject stackObject = game.getStack().peekFirst();
255             if (stackObject instanceof Spell) {
256                 ((Spell) stackObject).moveToExile(null, "", source, game);
257             } else {
258                 game.getStack().remove(stackObject, game); // stack ability
259             }
260         }
261         // 2) All attacking and blocking creatures are removed from combat.
262         for (UUID attackerId : game.getCombat().getAttackers()) {
263             Permanent permanent = game.getPermanent(attackerId);
264             if (permanent != null) {
265                 permanent.removeFromCombat(game);
266             }
267             game.getCombat().removeAttacker(attackerId, game);
268         }
269         for (UUID blockerId : game.getCombat().getBlockers()) {
270             Permanent permanent = game.getPermanent(blockerId);
271             if (permanent != null) {
272                 permanent.removeFromCombat(game);
273             }
274         }
275         // 3) State-based actions are checked. No player gets priority, and no triggered abilities are put onto the stack.
276         // seems like trigger events have to be removed: http://tabakrules.tumblr.com/post/122350751009/days-undoing-has-been-officially-spoiled-on
277         game.getState().clearTriggeredAbilities();
278         game.checkStateAndTriggered(); // triggered effects don't go to stack because check of endTurnRequested
279 
280         // 4) The current phase and/or step ends.
281         // The game skips straight to the cleanup step. The cleanup step happens in its entirety.
282         // this is caused by the endTurnRequest state
283     }

315     private void logStartOfTurn(Game game, Player player) {
316         StringBuilder sb = new StringBuilder(game.getState().isExtraTurn() ? "Extra turn" : "Turn ");
317         sb.append(game.getState().getTurnNum()).append(' ');
318         sb.append(player.getLogName());
319         sb.append(" (");
320         int delimiter = game.getPlayers().size() - 1;
321         for (Player gamePlayer : game.getPlayers().values()) {
322             sb.append(gamePlayer.getLife());
323             int poison = gamePlayer.getCounters().getCount(CounterType.POISON);
324             if (poison > 0) {
325                 sb.append("[P:").append(poison).append(']');
326             }
327             if (delimiter > 0) {
328                 sb.append(" - ");
329                 delimiter--;
330             }
331         }
332         sb.append(')');
333         game.fireStatusEvent(sb.toString(), true, false);
334     }

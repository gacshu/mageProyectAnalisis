38  public class UndercityDungeon extends Dungeon {
39  
40      public UndercityDungeon() {
41          super("Undercity", "CLB");
42  
43          // Secret Entrance — Search your library for a basic land card,
44          //                   reveal it,
45          //                   put it into your hand,
46          //                   then shuffle. (Leads to: Forge, Lost Well)
47          DungeonRoom secretEntrance = new DungeonRoom(
48                  "Secret Entrance",
49                  new SearchLibraryPutInHandEffect(
50                          new TargetCardInLibrary(StaticFilters.FILTER_CARD_BASIC_LAND), true
51                  )
52          );
53  
54          // Forge — Put two +1/+1 counters on target creature. (Leads to: Trap!, Arena)
55          DungeonRoom forge = new DungeonRoom(
56                  "Forge", new AddCountersTargetEffect(CounterType.P1P1.createInstance(2))
57          );
58          forge.addTarget(new TargetCreaturePermanent());
59  
60          // Lost Well — Scry 2. (Leads to: Arena, Stash)
61          DungeonRoom lostWell = new DungeonRoom(
62                  "Lost Well", new ScryEffect(2, false)
63          );
64  
65          // Trap! — Target player loses 5 life. (Leads to: Archives)
66          DungeonRoom trap = new DungeonRoom("Trap!", new LoseLifeTargetEffect(5));
67          trap.addTarget(new TargetPlayer());
68  
69          // Arena — Goad target creature. (Leads to: Archives, Catacombs)
70          DungeonRoom arena = new DungeonRoom("Arena", new GoadTargetEffect());
71          arena.addTarget(new TargetCreaturePermanent());
72  
73          // Stash — Create a Treasure token. (Leads to: Catacombs)
74          DungeonRoom stash = new DungeonRoom("Stash", new CreateTokenEffect(new TreasureToken()));
75  
76          // Archives — Draw a card. (Leads to: Throne of the Dead Three)
77          DungeonRoom archives = new DungeonRoom("Archives", new DrawCardSourceControllerEffect(1));
78  
79          // Catacombs — Create a 4/1 black Skeleton creature token with menace. (Leads to: Throne of the Dead Three)
80          DungeonRoom catacombs = new DungeonRoom("Catacombs", new CreateTokenEffect(new SkeletonMenaceToken()));
81  
82          // Throne of the Dead Three — Reveal the top ten cards of your library.
83          //                            Put a creature card from among them onto the battlefield with three +1/+1 counters on it.
84          //                            It gains hexproof until your next turn.
85          //                            Then shuffle.
86          DungeonRoom throneOfTheDeadThree = new DungeonRoom("Throne of the Dead Three", new ThroneOfTheDeadThreeEffect());
87  
88          secretEntrance.addNextRoom(forge);
89          secretEntrance.addNextRoom(lostWell);
90          forge.addNextRoom(trap);
91          forge.addNextRoom(arena);
92          lostWell.addNextRoom(arena);
93          lostWell.addNextRoom(stash);
94          trap.addNextRoom(archives);
95          arena.addNextRoom(archives);
96          arena.addNextRoom(catacombs);
97          stash.addNextRoom(catacombs);
98          archives.addNextRoom(throneOfTheDeadThree);
99          catacombs.addNextRoom(throneOfTheDeadThree);
100 
101         this.addRoom(secretEntrance);
102         this.addRoom(forge);
103         this.addRoom(lostWell);
104         this.addRoom(trap);
105         this.addRoom(arena);
106         this.addRoom(stash);
107         this.addRoom(archives);
108         this.addRoom(catacombs);
109         this.addRoom(throneOfTheDeadThree);
110     }
111 
112     private UndercityDungeon(final UndercityDungeon dungeon) {
113         super(dungeon);
114     }
115 
116     public UndercityDungeon copy() {
117         return new UndercityDungeon(this);
118     }
119 }

121 class ThroneOfTheDeadThreeEffect extends OneShotEffect {
122 
123     ThroneOfTheDeadThreeEffect() {
124         super(Outcome.Benefit);
125         staticText = "Reveal the top ten cards of your library. Put a creature card from among them onto the " +
126                 "battlefield with three +1/+1 counters on it. It gains hexproof until your next turn. Then shuffle.";
127     }
128 
129     private ThroneOfTheDeadThreeEffect(final ThroneOfTheDeadThreeEffect effect) {
130         super(effect);
131     }
132 
133     @Override
134     public ThroneOfTheDeadThreeEffect copy() {
135         return new ThroneOfTheDeadThreeEffect(this);
136     }
137 
138     @Override
139     public boolean apply(Game game, Ability source) {
140         Player player = game.getPlayer(source.getControllerId());
141         if (player == null) {
142             return false;
143         }
144         Cards cards = new CardsImpl(player.getLibrary().getTopCards(game, 10));
145         player.revealCards(source, cards, game);
146         Card card;
147         switch (cards.count(StaticFilters.FILTER_CARD_CREATURE, game)) {
148             case 0:
149                 card = null;
150                 break;
151             case 1:
152                 card = RandomUtil.randomFromCollection(cards.getCards(StaticFilters.FILTER_CARD_CREATURE, game));
153                 break;
154             default:
155                 TargetCardInLibrary target = new TargetCardInLibrary(StaticFilters.FILTER_CARD_CREATURE);
156                 player.choose(outcome, cards, target, game);
157                 card = cards.get(target.getFirstTarget(), game);
158         }
159         if (card != null) {
160             player.moveCards(card, Zone.BATTLEFIELD, source, game);
161             Permanent permanent = game.getPermanent(card.getId());
162             permanent.addCounters(CounterType.P1P1.createInstance(3), source, game);
163             game.addEffect(new GainAbilityTargetEffect(HexproofAbility.getInstance(), Duration.UntilYourNextTurn)
164                     .setTargetPointer(new FixedTarget(permanent, game)), source);
165         }
166         player.shuffleLibrary(source, game);
167         return true;
168     }

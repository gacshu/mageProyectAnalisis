36  public class TrailOfTheMageRingsPlane extends Plane {
37  
38      private static final FilterCard filter = new FilterCard("creature spells");
39  
40      static {
41          filter.add(CardType.CREATURE.getPredicate());
42      }
43  
44      public TrailOfTheMageRingsPlane() {
45          this.setPlaneType(Planes.PLANE_TRAIL_OF_THE_MAGE_RINGS);
46          this.setExpansionSetCodeForImage("PCA");
47  
48          // Instant and sorcery spells have rebound
49          Ability ability = new SimpleStaticAbility(Zone.COMMAND, new TrailOfTheMageRingsReboundEffect());
50          this.getAbilities().add(ability);
51  
52          // Active player can roll the planar die: Whenever you roll {CHAOS}, you may search your library for an instant or sorcery card, reveal it, put it into your hand, then shuffle your library
53          Effect chaosEffect = new SearchLibraryPutInHandEffect(new TargetCardInLibrary(0, 1, new FilterInstantOrSorceryCard()), true, true);
54          Target chaosTarget = null;
55  
56          List<Effect> chaosEffects = new ArrayList<Effect>();
57          chaosEffects.add(chaosEffect);
58          List<Target> chaosTargets = new ArrayList<Target>();
59          chaosTargets.add(chaosTarget);
60  
61          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
62          chaosAbility.addWatcher(new PlanarRollWatcher());
63          this.getAbilities().add(chaosAbility);
64          chaosAbility.setMayActivate(TargetController.ANY);
65          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
66      }
67  }

69  class TrailOfTheMageRingsReboundEffect extends ContinuousEffectImpl {
70  
71      protected static final FilterCard filter = new FilterCard("Instant and sorcery spells");
72  
73      static {
74          filter.add(Predicates.or(CardType.INSTANT.getPredicate(), CardType.SORCERY.getPredicate()));
75      }
76  
77      public TrailOfTheMageRingsReboundEffect() {
78          super(Duration.Custom, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
79          staticText = "Instant and sorcery spells have rebound";
80      }
81  
82      public TrailOfTheMageRingsReboundEffect(final TrailOfTheMageRingsReboundEffect effect) {
83          super(effect);
84      }
85  
86      @Override
87      public TrailOfTheMageRingsReboundEffect copy() {
88          return new TrailOfTheMageRingsReboundEffect(this);
89      }
90  
91      @Override
92      public boolean apply(Game game, Ability source) {
93          Plane cPlane = game.getState().getCurrentPlane();
94          if (cPlane == null) {
95              return false;
96          }
97          if (!cPlane.getPlaneType().equals(Planes.PLANE_TRAIL_OF_THE_MAGE_RINGS)) {
98              return false;
99          }
100 
101         for (UUID playerId : game.getPlayers().keySet()) {
102             Player player = game.getPlayer(playerId);
103             if (player != null) {
104                 for (Card card : player.getHand().getCards(filter, game)) {
105                     addReboundAbility(card, source, game);
106                 }
107                 for (Iterator<StackObject> iterator = game.getStack().iterator(); iterator.hasNext(); ) {
108                     StackObject stackObject = iterator.next();
109                     if (stackObject instanceof Spell && stackObject.isControlledBy(source.getControllerId())) {
110                         Spell spell = (Spell) stackObject;
111                         Card card = spell.getCard();
112                         if (card != null) {
113                             addReboundAbility(card, source, game);
114                         }
115                     }
116                 }
117             }
118         }
119         return true;
120     }
121 
122     private void addReboundAbility(Card card, Ability source, Game game) {
123         if (filter.match(card, game)) {
124             boolean found = card.getAbilities(game).containsClass(ReboundAbility.class);
125             if (!found) {
126                 Ability ability = new ReboundAbility();
127                 game.getState().addOtherAbility(card, ability);
128             }
129         }
130     }

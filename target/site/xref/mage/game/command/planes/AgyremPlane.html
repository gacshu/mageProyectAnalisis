37  public class AgyremPlane extends Plane {
38  
39      private static final FilterCreaturePermanent filterWhite = new FilterCreaturePermanent("a white creature");
40      private static final FilterCreaturePermanent filterNonWhite = new FilterCreaturePermanent("a nonwhite creature");
41  
42      static {
43          filterWhite.add(new ColorPredicate(ObjectColor.WHITE));
44          filterNonWhite.add(Predicates.not(new ColorPredicate(ObjectColor.WHITE)));
45      }
46  
47      public AgyremPlane() {
48          this.setPlaneType(Planes.PLANE_AGYREM);
49          this.setExpansionSetCodeForImage("PCA");
50  
51          // Whenever a white creature dies, return it to the battlefield under its owner's control at the beginning of the next end step
52          DiesCreatureTriggeredAbility ability = new DiesCreatureTriggeredAbility(Zone.COMMAND, new AgyremEffect(), false, filterWhite, true);
53          this.getAbilities().add(ability);
54          // Whenever a nonwhite creature dies, return it to its owner's hand at the beginning of the next end step.
55          DiesCreatureTriggeredAbility ability2 = new DiesCreatureTriggeredAbility(Zone.COMMAND, new AgyremEffect2(), false, filterNonWhite, true);
56          this.getAbilities().add(ability2);
57  
58          // Active player can roll the planar die: Whenever you roll {CHAOS}, creatures can't attack you until a player planeswalks
59          Effect chaosEffect = new AgyremRestrictionEffect();
60          Target chaosTarget = null;
61  
62          List<Effect> chaosEffects = new ArrayList<>();
63          chaosEffects.add(chaosEffect);
64          List<Target> chaosTargets = new ArrayList<>();
65          chaosTargets.add(chaosTarget);
66  
67          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
68          chaosAbility.addWatcher(new PlanarRollWatcher());
69          this.getAbilities().add(chaosAbility);
70          chaosAbility.setMayActivate(TargetController.ANY);
71          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
72      }
73  }

75  class AgyremEffect extends OneShotEffect {
76  
77      public AgyremEffect() {
78          super(Outcome.PutCardInPlay);
79          this.staticText = "return that card to the battlefield under its owner's control at the beginning of the next end step";
80      }
81  
82      public AgyremEffect(final AgyremEffect effect) {
83          super(effect);
84      }
85  
86      @Override
87      public AgyremEffect copy() {
88          return new AgyremEffect(this);
89      }
90  
91      @Override
92      public boolean apply(Game game, Ability source) {
93          Card card = game.getCard(getTargetPointer().getFirst(game, source));
94          if (card != null) {
95              Effect effect = new ReturnToBattlefieldUnderOwnerControlTargetEffect(false, false);
96              effect.setTargetPointer(new FixedTarget(card, game));
97              effect.setText("return that card to the battlefield under its owner's control at the beginning of the next end step");
98              game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(effect, TargetController.ANY), source);
99              return true;
100         }
101         return false;
102     }
103 }

135 class AgyremRestrictionEffect extends RestrictionEffect {
136 
137     AgyremRestrictionEffect() {
138         super(Duration.Custom, Outcome.Benefit);
139         staticText = "Creatures can't attack you";
140     }
141 
142     AgyremRestrictionEffect(final AgyremRestrictionEffect effect) {
143         super(effect);
144     }
145 
146     @Override
147     public boolean applies(Permanent permanent, Ability source, Game game) {
148         return permanent.isCreature(game);
149     }
150 
151     @Override
152     public boolean canAttack(Permanent attacker, UUID defenderId, Ability source, Game game, boolean canUseChooseDialogs) {
153         if (defenderId == null) {
154             return true;
155         }
156 
157         Plane cPlane = game.getState().getCurrentPlane();
158         if (cPlane != null && cPlane.getPlaneType().equals(Planes.PLANE_AGYREM)) {
159             return !defenderId.equals(source.getControllerId());
160         }
161         return true;
162     }
163 
164     @Override
165     public AgyremRestrictionEffect copy() {
166         return new AgyremRestrictionEffect(this);
167     }
168 }

31  public class FieldsOfSummerPlane extends Plane {
32  
33  
34      public FieldsOfSummerPlane() {
35          this.setPlaneType(Planes.PLANE_FIELDS_OF_SUMMER);
36          this.setExpansionSetCodeForImage("PCA");
37  
38          // Whenever a player casts a spell, that player may gain 2 life
39          SpellCastAllTriggeredAbility ability = new SpellCastAllTriggeredAbility(Zone.COMMAND, new FieldsOfSummerEffect(), StaticFilters.FILTER_SPELL_A, false, SetTargetPointer.PLAYER);
40          this.getAbilities().add(ability);
41  
42          // Active player can roll the planar die: Whenever you roll {CHAOS}, you may gain 10 life
43          Effect chaosEffect = new GainLifeEffect(10);
44          Target chaosTarget = null;
45  
46          List<Effect> chaosEffects = new ArrayList<>();
47          chaosEffects.add(chaosEffect);
48          List<Target> chaosTargets = new ArrayList<>();
49          chaosTargets.add(chaosTarget);
50  
51          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
52          chaosAbility.addWatcher(new PlanarRollWatcher());
53          this.getAbilities().add(chaosAbility);
54          chaosAbility.setMayActivate(TargetController.ANY);
55          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
56      }
57  }
58  
59  class FieldsOfSummerEffect extends OneShotEffect {
60  
61      public FieldsOfSummerEffect() {
62          super(Outcome.GainLife);
63          this.staticText = "that player may gain 2 life";
64      }
65  
66      public FieldsOfSummerEffect(final FieldsOfSummerEffect effect) {
67          super(effect);
68      }
69  
70      @Override
71      public FieldsOfSummerEffect copy() {
72          return new FieldsOfSummerEffect(this);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          Plane cPlane = game.getState().getCurrentPlane();
78          if (cPlane == null || !cPlane.getPlaneType().equals(Planes.PLANE_FIELDS_OF_SUMMER)) {
79              return false;
80          }
81          Player owner = game.getPlayer(this.getTargetPointer().getFirst(game, source));
82          if (owner != null && owner.canRespond() && owner.chooseUse(Outcome.Benefit, "Gain 2 life?", source, game)) {
83              Effect effect = new GainLifeTargetEffect(2);
84              effect.setTargetPointer(new FixedTarget(owner.getId())).apply(game, source);
85              return true;
86          }
87          return false;
88      }
89  }

30   * @author spjspj
31   */
32  public class TheDarkBaronyPlane extends Plane {
33  
34      private static final FilterCard filter = new FilterCard("a nonblack card");
35  
36      static {
37          filter.add(Predicates.not(new ColorPredicate(ObjectColor.BLACK)));
38      }
39  
40      public TheDarkBaronyPlane() {
41          this.setPlaneType(Planes.PLANE_THE_DARK_BARONY);
42          this.setExpansionSetCodeForImage("PCA");
43  
44          // Whenever a nonblack card is put into a player's graveyard from anywhere, that player loses 1 life
45          Ability ability = new PutCardIntoGraveFromAnywhereAllTriggeredAbility(Zone.COMMAND,
46                  new LoseLifeTargetEffect(1), false, filter, TargetController.ANY, SetTargetPointer.PLAYER);
47          this.getAbilities().add(ability);
48  
49          // Active player can roll the planar die: Whenever you roll {CHAOS}, each player dicards a card
50          Effect chaosEffect = new DiscardEachPlayerEffect(TargetController.OPPONENT);
51          Target chaosTarget = null;
52  
53          List<Effect> chaosEffects = new ArrayList<Effect>();
54          chaosEffects.add(chaosEffect);
55          List<Target> chaosTargets = new ArrayList<Target>();
56          chaosTargets.add(chaosTarget);
57  
58          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
59          chaosAbility.addWatcher(new PlanarRollWatcher());
60          this.getAbilities().add(chaosAbility);
61          chaosAbility.setMayActivate(TargetController.ANY);
62          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
63      }
64  }

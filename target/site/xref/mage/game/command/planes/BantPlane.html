39  public class BantPlane extends Plane {
40  
41      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("Creatures");
42      private static final FilterCreaturePermanent filter2 = new FilterCreaturePermanent("Green, White or Blue creatures");
43  
44      static {
45          filter2.add(Predicates.or(new ColorPredicate(ObjectColor.GREEN), new ColorPredicate(ObjectColor.WHITE), new ColorPredicate(ObjectColor.BLUE)));
46      }
47  
48      private static final String rule = "{this} has indestructible as long as it has a divinity counter on it";
49      private static final String exaltedRule = "All creatures have exalted";
50  
51      public BantPlane() {
52          this.setPlaneType(Planes.PLANE_BANT);
53          this.setExpansionSetCodeForImage("PCA");
54  
55          // All creatures have exalted
56          SimpleStaticAbility ability
57                  = new SimpleStaticAbility(Zone.COMMAND, new ConditionalContinuousEffect(
58                  new GainAbilityAllEffect(new ExaltedAbility(), Duration.Custom, StaticFilters.FILTER_PERMANENT_CREATURE),
59                  new IsStillOnPlaneCondition(this.getName()),
60                  exaltedRule));
61  
62          this.getAbilities().add(ability);
63  
64          // Active player can roll the planar die: Whenever you roll {CHAOS}, put a divinity counter on target green, white, or blue creature.  That creature gains indestructible for as long as it has a divinity counter on it.
65          Effect chaosEffect = new ConditionalContinuousEffect(new GainAbilityTargetEffect(IndestructibleAbility.getInstance(), Duration.Custom), new TargetHasCounterCondition(CounterType.DIVINITY), rule);
66          Target chaosTarget = new TargetCreaturePermanent(1, 1, filter2, false);
67          Effect chaosEffect2 = new AddCountersTargetEffect(CounterType.DIVINITY.createInstance());
68  
69          List<Effect> chaosEffects = new ArrayList<Effect>();
70          chaosEffects.add(chaosEffect2);
71          chaosEffects.add(chaosEffect);
72  
73          List<Target> chaosTargets = new ArrayList<Target>();
74          chaosTargets.add(chaosTarget);
75          chaosTargets.add(chaosTarget);
76  
77          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
78          chaosAbility.addWatcher(new PlanarRollWatcher());
79          this.getAbilities().add(chaosAbility);
80          chaosAbility.setMayActivate(TargetController.ANY);
81          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
82      }
83  }

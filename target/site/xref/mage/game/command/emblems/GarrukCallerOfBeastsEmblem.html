2   package mage.game.command.emblems;

4   import mage.abilities.Ability;
5   import mage.abilities.common.SpellCastControllerTriggeredAbility;
6   import mage.abilities.effects.Effect;
7   import mage.abilities.effects.common.search.SearchLibraryPutInPlayEffect;
8   import mage.constants.Outcome;
9   import mage.constants.Zone;
10  import mage.filter.StaticFilters;
11  import mage.filter.common.FilterCreatureCard;
12  import mage.game.command.Emblem;
13  import mage.target.common.TargetCardInLibrary;
14  
15  /**
16   *
17   * @author spjspj
18   */
19  public final class GarrukCallerOfBeastsEmblem extends Emblem {
20  
21      /**
22       * Emblem: "Whenever you cast a creature spell, you may search your library
23       * for a creature card, put it onto the battlefield, then shuffle your
24       * library."
25       */
26      public GarrukCallerOfBeastsEmblem() {
27          this.setName("Emblem Garruk");
28          Effect effect = new SearchLibraryPutInPlayEffect(new TargetCardInLibrary(new FilterCreatureCard("creature card")), false, true, Outcome.PutCreatureInPlay);
29          Ability ability = new SpellCastControllerTriggeredAbility(Zone.COMMAND, effect, StaticFilters.FILTER_SPELL_A_CREATURE, true, false);
30          this.getAbilities().add(ability);
31  
32          this.setExpansionSetCodeForImage("M14");
33      }
34  }

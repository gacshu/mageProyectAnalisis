16  public final class LukeSkywalkerEmblem extends Emblem {
17  
18      // -6: You get an emblem with "Prevent all damage that would be dealt to you during combat." Exile Luke Skywalker, the Last Jedi.
19      public LukeSkywalkerEmblem() {
20          this.setName("Emblem Luke Skywalker");
21          this.setExpansionSetCodeForImage("SWS");
22          this.getAbilities().add(new SimpleStaticAbility(Zone.BATTLEFIELD, new LukeSkywalkerEmblemEffect()));
23      }
24  }
25  
26  class LukeSkywalkerEmblemEffect extends PreventionEffectImpl {
27  
28      public LukeSkywalkerEmblemEffect() {
29          super(Duration.Custom, Integer.MAX_VALUE, false);
30          staticText = "Prevent all damage that would be dealt to you during combat";
31      }
32  
33      public LukeSkywalkerEmblemEffect(final LukeSkywalkerEmblemEffect effect) {
34          super(effect);
35      }
36  
37      @Override
38      public LukeSkywalkerEmblemEffect copy() {
39          return new LukeSkywalkerEmblemEffect(this);
40      }
41  
42      @Override
43      public boolean applies(GameEvent event, Ability source, Game game) {
44          if (game.getTurnPhaseType() == TurnPhase.COMBAT
45                  && super.applies(event, source, game)
46                  && event.getType() == GameEvent.EventType.DAMAGE_PLAYER) {
47              Player controller = game.getPlayer(source.getControllerId());
48              if (controller != null && game.getState().getPlayersInRange(controller.getId(), game).contains(event.getTargetId())) {
49                  return true;
50              }
51          }
52          return false;
53      }
54  }

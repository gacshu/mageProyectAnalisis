19  /**
20   *
21   * @author spjspj
22   */
23  public final class LilianaDefiantNecromancerEmblem extends Emblem {
24      // You get an emblem with "Whenever a creature you control dies, return it to the battlefield under your control at the beginning of the next end step."
25  
26      public LilianaDefiantNecromancerEmblem() {
27          this.setName("Emblem Liliana");
28          Ability ability = new DiesCreatureTriggeredAbility(Zone.COMMAND, new LilianaDefiantNecromancerEmblemEffect(), false, StaticFilters.FILTER_PERMANENT_A_CREATURE, true);
29          this.getAbilities().add(ability);
30  
31          this.setExpansionSetCodeForImage("ORI");
32      }
33  }
34  
35  class LilianaDefiantNecromancerEmblemEffect extends OneShotEffect {
36  
37      LilianaDefiantNecromancerEmblemEffect() {
38          super(Outcome.PutCardInPlay);
39          this.staticText = "return it to the battlefield under your control at the beginning of the next end step";
40      }
41  
42      LilianaDefiantNecromancerEmblemEffect(final LilianaDefiantNecromancerEmblemEffect effect) {
43          super(effect);
44      }
45  
46      @Override
47      public LilianaDefiantNecromancerEmblemEffect copy() {
48          return new LilianaDefiantNecromancerEmblemEffect(this);
49      }
50  
51      @Override
52      public boolean apply(Game game, Ability source) {
53          Card card = game.getCard(getTargetPointer().getFirst(game, source));
54          if (card != null) {
55              Effect effect = new ReturnFromGraveyardToBattlefieldTargetEffect();
56              effect.setTargetPointer(new FixedTarget(card, game));
57              effect.setText("return that card to the battlefield at the beginning of the next end step");
58              game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(effect, TargetController.ANY), source);
59              return true;
60          }
61          return false;
62      }
63  }

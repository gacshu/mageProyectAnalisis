13  
14  public final class BasriKetEmblem extends Emblem {
15      /**
16       * Emblem with "At the beginning of combat on your turn, create a 1/1 white Soldier creature token, then put a +1/+1 counter on each creature you control."
17       */
18  
19      public BasriKetEmblem() {
20          setName("Emblem Basri");
21          Ability ability = new BeginningOfCombatTriggeredAbility(
22                  Zone.COMMAND,
23                  new CreateTokenEffect(new SoldierToken()),
24                  TargetController.YOU, false, false);
25          ability.addEffect(
26                  new AddCountersAllEffect(CounterType.P1P1.createInstance(), StaticFilters.FILTER_PERMANENT_CREATURE_CONTROLLED)
27                  .setText(", then put a +1/+1 counter on each creature you control")
28          );
29          this.getAbilities().add(ability);
30  
31          this.setExpansionSetCodeForImage("M21");
32      }
33  }

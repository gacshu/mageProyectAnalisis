24  public final class WrennAndSixEmblem extends Emblem {
25  
26      public WrennAndSixEmblem() {
27          this.setName("Emblem Wrenn");
28          this.getAbilities().add(new SimpleStaticAbility(Zone.COMMAND, new WrennAndSixEmblemEffect()));
29  
30          availableImageSetCodes = Arrays.asList("MH1", "2X2");
31      }
32  }
33  
34  class WrennAndSixEmblemEffect extends ContinuousEffectImpl {
35  
36      WrennAndSixEmblemEffect() {
37          super(Duration.WhileOnBattlefield, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
38          staticText = "Instant and sorcery cards in your graveyard have retrace.";
39      }
40  
41      private WrennAndSixEmblemEffect(final WrennAndSixEmblemEffect effect) {
42          super(effect);
43      }
44  
45      @Override
46      public boolean apply(Game game, Ability source) {
47          Player controller = game.getPlayer(source.getControllerId());
48          if (controller == null) {
49              return false;
50          }
51          for (UUID cardId : controller.getGraveyard()) {
52              Card card = game.getCard(cardId);
53              if (card == null) {
54                  continue;
55              }
56              if (card instanceof SplitCard) {
57                  SplitCardHalf leftHalfCard = ((SplitCard) card).getLeftHalfCard();
58                  SplitCardHalf rightHalfCard = ((SplitCard) card).getRightHalfCard();
59                  if (leftHalfCard.isInstantOrSorcery(game)) {
60                      Ability ability = new RetraceAbility(leftHalfCard);
61                      ability.setSourceId(cardId);
62                      ability.setControllerId(leftHalfCard.getOwnerId());
63                      game.getState().addOtherAbility(leftHalfCard, ability);
64                  }
65                  if (rightHalfCard.isInstantOrSorcery(game)) {
66                      Ability ability = new RetraceAbility(rightHalfCard);
67                      ability.setSourceId(cardId);
68                      ability.setControllerId(rightHalfCard.getOwnerId());
69                      game.getState().addOtherAbility(rightHalfCard, ability);
70                  }
71              }
72              if (card instanceof ModalDoubleFacesCard) {
73                  ModalDoubleFacesCardHalf leftHalfCard = ((ModalDoubleFacesCard) card).getLeftHalfCard();
74                  ModalDoubleFacesCardHalf rightHalfCard = ((ModalDoubleFacesCard) card).getRightHalfCard();
75                  if (leftHalfCard.isInstantOrSorcery(game)) {
76                      Ability ability = new RetraceAbility(leftHalfCard);
77                      ability.setSourceId(cardId);
78                      ability.setControllerId(leftHalfCard.getOwnerId());
79                      game.getState().addOtherAbility(leftHalfCard, ability);
80                  }
81                  if (rightHalfCard.isInstantOrSorcery(game)) {
82                      Ability ability = new RetraceAbility(rightHalfCard);
83                      ability.setSourceId(cardId);
84                      ability.setControllerId(rightHalfCard.getOwnerId());
85                      game.getState().addOtherAbility(rightHalfCard, ability);
86                  }
87              }
88              if (card instanceof AdventureCard) {
89                  // Adventure cards are castable per https://twitter.com/elishffrn/status/1179047911729946624
90                  card = ((AdventureCard) card).getSpellCard();
91              }
92              if (!card.isInstantOrSorcery(game)) {
93                  continue;
94              }
95              Ability ability = new RetraceAbility(card);
96              ability.setSourceId(cardId);
97              ability.setControllerId(card.getOwnerId());
98              game.getState().addOtherAbility(card, ability);
99          }
100         return true;
101     }
102 
103     @Override
104     public WrennAndSixEmblemEffect copy() {
105         return new WrennAndSixEmblemEffect(this);
106     }
107 }

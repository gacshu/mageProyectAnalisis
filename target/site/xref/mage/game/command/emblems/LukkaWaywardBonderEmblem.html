18  public final class LukkaWaywardBonderEmblem extends Emblem {
19  
20      // âˆ’7: You get an emblem with "Whenever a creature enters the battlefield under your control, it deals damage equal to its power to any target."
21      public LukkaWaywardBonderEmblem() {
22          this.setName("Emblem Lukka");
23          this.setExpansionSetCodeForImage("STX");
24          Ability ability = new EntersBattlefieldControlledTriggeredAbility(
25                  Zone.COMMAND, new LukkaWaywardBonderEmblemEffect(),
26                  StaticFilters.FILTER_PERMANENT_A_CREATURE, false
27          );
28          ability.addTarget(new TargetAnyTarget());
29          this.getAbilities().add(ability);
30      }
31  }

33  class LukkaWaywardBonderEmblemEffect extends OneShotEffect {
34  
35      LukkaWaywardBonderEmblemEffect() {
36          super(Outcome.Benefit);
37          staticText = "it deals damage equal to its power to any target";
38      }
39  
40      private LukkaWaywardBonderEmblemEffect(final LukkaWaywardBonderEmblemEffect effect) {
41          super(effect);
42      }
43  
44      @Override
45      public LukkaWaywardBonderEmblemEffect copy() {
46          return new LukkaWaywardBonderEmblemEffect(this);
47      }
48  
49      @Override
50      public boolean apply(Game game, Ability source) {
51          Permanent permanent = (Permanent) getValue("permanentEnteringBattlefield");
52          if (permanent == null || permanent.getPower().getValue() < 1) {
53              return false;
54          }
55          Permanent targetPermanent = game.getPermanent(source.getFirstTarget());
56          if (targetPermanent != null) {
57              return targetPermanent.damage(permanent.getPower().getValue(), permanent.getId(), source, game) > 0;
58          }
59          Player targetPlayer = game.getPlayer(source.getFirstTarget());
60          if (targetPlayer != null) {
61              return targetPlayer.damage(permanent.getPower().getValue(), permanent.getId(), source, game) > 0;
62          }
63          return false;
64      }
65  }

21  public final class LolthSpiderQueenEmblem extends Emblem {
22  
23      // âˆ’8: You get an emblem with "Whenever an opponent is dealt combat damage by one or more creatures you control, if that player lost less than 8 life this turn, they lose life equal to the difference."
24      public LolthSpiderQueenEmblem() {
25          this.setName("Emblem Lolth");
26          this.setExpansionSetCodeForImage("AFR");
27          this.getAbilities().add(new ConditionalInterveningIfTriggeredAbility(
28                  new DealCombatDamageControlledTriggeredAbility(
29                          Zone.COMMAND, new LolthSpiderQueenEmblemEffect(), true, true
30                  ), LolthSpiderQueenEmblemCondition.instance, "Whenever an opponent " +
31                  "is dealt combat damage by one or more creatures you control, " +
32                  "if that player lost less than 8 life this turn, they lose life equal to the difference."
33          ));
34      }
35  }
36  
37  enum LolthSpiderQueenEmblemCondition implements Condition {
38      instance;
39  
40      @Override
41      public boolean apply(Game game, Ability source) {
42          Player player = source
43                  .getEffects()
44                  .stream()
45                  .map(Effect::getTargetPointer)
46                  .map(tp -> tp.getFirst(game, source))
47                  .map(game::getPlayer)
48                  .filter(Objects::nonNull)
49                  .findFirst()
50                  .orElse(null);
51          PlayerLostLifeWatcher watcher = game.getState().getWatcher(PlayerLostLifeWatcher.class);
52          return player != null && watcher != null && watcher.getLifeLost(player.getId()) < 8;
53      }
54  }
55  
56  class LolthSpiderQueenEmblemEffect extends OneShotEffect {
57  
58      LolthSpiderQueenEmblemEffect() {
59          super(Outcome.Benefit);
60      }
61  
62      private LolthSpiderQueenEmblemEffect(final LolthSpiderQueenEmblemEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public LolthSpiderQueenEmblemEffect copy() {
68          return new LolthSpiderQueenEmblemEffect(this);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          Player player = game.getPlayer(getTargetPointer().getFirst(game, source));
74          PlayerLostLifeWatcher watcher = game.getState().getWatcher(PlayerLostLifeWatcher.class);
75          return player != null && watcher != null && player.loseLife(
76                  Math.max(8 - watcher.getLifeLost(player.getId()), 0), game, source, false
77          ) > 0;
78      }
79  }

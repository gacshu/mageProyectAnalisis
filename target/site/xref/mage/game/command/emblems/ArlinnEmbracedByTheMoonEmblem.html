24  public final class ArlinnEmbracedByTheMoonEmblem extends Emblem {
25      // "Creatures you control have haste and '{T}: This creature deals damage equal to its power to any target.'"
26  
27      public ArlinnEmbracedByTheMoonEmblem() {
28          this.setName("Emblem Arlinn");
29          FilterPermanent filter = new FilterControlledCreaturePermanent("Creatures");
30          GainAbilityControlledEffect effect = new GainAbilityControlledEffect(HasteAbility.getInstance(), Duration.EndOfGame, filter);
31          effect.setText("Creatures you control have haste");
32          Ability ability = new SimpleStaticAbility(Zone.COMMAND, effect);
33          Effect effect2 = new DamageTargetEffect(new SourcePermanentPowerCount());
34          effect2.setText("This creature deals damage equal to its power to any target");
35          Ability ability2 = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect2, new TapSourceCost());
36          ability2.addTarget(new TargetAnyTarget());
37          effect = new GainAbilityControlledEffect(ability2, Duration.EndOfGame, filter);
38          effect.setText("and '{T}: This creature deals damage equal to its power to any target");
39          ability.addEffect(effect);
40          this.getAbilities().add(ability);
41  
42          this.setExpansionSetCodeForImage("SOI");
43      }
44  }

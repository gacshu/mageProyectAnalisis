20  public final class TamiyoTheMoonSageEmblem extends Emblem {
21  
22      /**
23       * Emblem with "You have no maximum hand size" and "Whenever a card is put
24       * into your graveyard from anywhere, you may return it to your hand."
25       */
26  
27      public TamiyoTheMoonSageEmblem() {
28          this.setName("Emblem Tamiyo");
29          this.getAbilities().add(new SimpleStaticAbility(Zone.COMMAND, new MaximumHandSizeControllerEffect(
30                  Integer.MAX_VALUE, Duration.Custom, HandSizeModification.SET
31          )));
32          this.getAbilities().add(new PutCardIntoGraveFromAnywhereAllTriggeredAbility(
33                  Zone.COMMAND, new ReturnToHandTargetEffect().setText("return it to your hand"),
34                  true, StaticFilters.FILTER_CARD_A, TargetController.YOU, SetTargetPointer.CARD
35          ));
36  
37          availableImageSetCodes = Arrays.asList("AVR", "MED");
38      }
39  }

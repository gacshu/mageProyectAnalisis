168     protected Round createRoundRandom() {
169         Round round = new Round(rounds.size() + 1, this);
170         rounds.add(round);
171         List<TournamentPlayer> roundPlayers = getActivePlayers();
172 
173         // search the player with a bye last round
174         List<TournamentPlayer> playerWithByes = getTournamentPlayersWithBye(roundPlayers);
175 
176         while (roundPlayers.size() > 1) {
177             TournamentPlayer player1 = getNextAvailablePlayer(roundPlayers, playerWithByes);
178             TournamentPlayer player2 = getNextAvailablePlayer(roundPlayers, playerWithByes);
179             round.addPairing(new TournamentPairing(player1, player2));
180         }
181 
182         if (!roundPlayers.isEmpty()) {
183             // player free round - add to bye players of this round
184             TournamentPlayer player1 = roundPlayers.get(0);
185             round.getPlayerByes().add(player1);
186             player1.setState(TournamentPlayerState.WAITING);
187             player1.setStateInfo("Round Bye");
188             updateResults();
189         }
190         return round;
191     }

193     private TournamentPlayer getNextAvailablePlayer(List<TournamentPlayer> roundPlayers, List<TournamentPlayer> playerWithByes) {
194         TournamentPlayer nextPlayer;
195         if (playerWithByes.isEmpty()) {
196             int i = RandomUtil.nextInt(roundPlayers.size());
197             nextPlayer = roundPlayers.get(i);
198             roundPlayers.remove(i);
199         } else { // prefer players with byes to pair
200             Iterator<TournamentPlayer> iterator = playerWithByes.iterator();
201             nextPlayer = iterator.next();
202             iterator.remove();
203             roundPlayers.remove(nextPlayer);
204         }
205         return nextPlayer;
206     }

208     private List<TournamentPlayer> getTournamentPlayersWithBye(List<TournamentPlayer> roundPlayers) {
209         List<TournamentPlayer> playersWithBye = new ArrayList<>();
210         if (rounds.size() > 1) {
211             for (int i = rounds.size() - 2; i >= 0; i--) {
212                 Round oldRound = rounds.get(i);
213                 if (oldRound != null && !oldRound.getPlayerByes().isEmpty()) {
214                     TournamentPlayer tournamentPlayerWithBye = oldRound.getPlayerByes().iterator().next();
215                     if (roundPlayers.contains(tournamentPlayerWithBye)) {
216                         playersWithBye.add(tournamentPlayerWithBye);
217                     }
218                 }
219             }
220         }
221         return playersWithBye;
222     }

224     protected void playRound(Round round) {
225         for (TournamentPairing pair : round.getPairs()) {
226             playMatch(pair);

228         updateResults(); // show points from byes

230             try {
231                 //TODO: improve this

234                 Logger.getLogger(TournamentImpl.class).warn("TournamentImpl playRound error ", ex);
235                 break;
236             }
237         }
238         updateResults();
239     }

260     @Override
261     public void updateResults() {
262         for (TournamentPlayer player : players.values()) {
263             player.setResults("");
264             player.setPoints(0);
265             player.setStateInfo("");
266         }
267         for (Round round : rounds) {
268             for (TournamentPairing pair : round.getPairs()) {
269                 Match match = pair.getMatch();
270                 if (match != null && match.hasEnded()) {
271                     TournamentPlayer tp1 = pair.getPlayer1();
272                     TournamentPlayer tp2 = pair.getPlayer2();
273                     MatchPlayer mp1 = match.getPlayer(pair.getPlayer1().getPlayer().getId());
274                     MatchPlayer mp2 = match.getPlayer(pair.getPlayer2().getPlayer().getId());
275                     // set player state if they finished the round
276                     if (round.getRoundNumber() == rounds.size()) { // for elimination getRoundNumber = 0 so never true here
277                         match.setTournamentRound(round.getRoundNumber());
278                         if (tp1.getState() == TournamentPlayerState.DUELING) {
279                             if (round.getRoundNumber() == getNumberRounds()) {
280                                 tp1.setState(TournamentPlayerState.FINISHED);
281                             } else {
282                                 tp1.setState(TournamentPlayerState.WAITING);
283                             }
284                         }
285                         if (tp2.getState() == TournamentPlayerState.DUELING) {
286                             if (round.getRoundNumber() == getNumberRounds()) {
287                                 tp2.setState(TournamentPlayerState.FINISHED);
288                             } else {
289                                 tp2.setState(TournamentPlayerState.WAITING);
290                             }
291                         }
292                     }
293                     // Add round result
294                     tp1.setResults(addRoundResult(round.getRoundNumber(), pair, tp1, tp2));
295                     tp2.setResults(addRoundResult(round.getRoundNumber(), pair, tp2, tp1));
296 
297                     // Add points
298                     if ((!mp1.hasQuit() && mp1.getWins() > mp2.getWins()) || mp2.hasQuit()) {
299                         tp1.setPoints(tp1.getPoints() + 3);
300                     } else if ((!mp2.hasQuit() && mp1.getWins() < mp2.getWins()) || mp1.hasQuit()) {
301                         tp2.setPoints(tp2.getPoints() + 3);
302                     } else {
303                         tp1.setPoints(tp1.getPoints() + 1);
304                         tp2.setPoints(tp2.getPoints() + 1);
305                     }
306                 }
307             }
308             for (TournamentPlayer tp : round.getPlayerByes()) {
309                 tp.setResults(new StringBuilder(tp.getResults()).append('R').append(round.getRoundNumber()).append(' ').append("Bye ").toString());
310                 tp.setPoints(tp.getPoints() + 3);
311             }
312         }
313     }

315     private static String addRoundResult(int round, TournamentPairing pair, TournamentPlayer tournamentPlayer, TournamentPlayer opponentPlayer) {
316         StringBuilder playerResult = new StringBuilder(tournamentPlayer.getResults());
317         playerResult.append('R').append(round).append(' ');
318         playerResult.append(getMatchResultString(tournamentPlayer, opponentPlayer, pair.getMatch()));
319         return playerResult.toString();
320     }

322     private static String getMatchResultString(TournamentPlayer p1, TournamentPlayer p2, Match match) {
323         MatchPlayer mp1 = match.getPlayer(p1.getPlayer().getId());
324         MatchPlayer mp2 = match.getPlayer(p2.getPlayer().getId());
325         StringBuilder matchResult = new StringBuilder();
326         matchResult.append(p2.getPlayer().getName());
327         matchResult.append(" [").append(mp1.getWins());
328         if (mp1.hasQuit()) {
329             matchResult.append(mp1.getPlayer().hasIdleTimeout() ? "I" : (mp1.getPlayer().hasTimerTimeout() ? "T" : "Q"));
330         }
331         if (match.getDraws() > 0) {
332             matchResult.append('-').append(match.getDraws());
333         }
334         matchResult.append('-').append(mp2.getWins());
335         if (mp2.hasQuit()) {
336             matchResult.append(mp2.getPlayer().hasIdleTimeout() ? "I" : (mp2.getPlayer().hasTimerTimeout() ? "T" : "Q"));
337         }
338         matchResult.append("] ");
339         return matchResult.toString();
340     }

377     public void construct() {
378         tableEventSource.fireTableEvent(EventType.CONSTRUCT);
379         if (!isAbort()) {
380             for (final TournamentPlayer player : players.values()) {
381 
382                 player.setConstructing();
383                 new Thread(
384                         () -> player.getPlayer().construct(TournamentImpl.this, player.getDeck())
385                 ).start();
386             }
387             // add autosubmit trigger
388 
389             synchronized (this) {
390                 while (!isDoneConstructing()) {
391                     try {
392                         this.wait();
393                     } catch (InterruptedException ex) {
394 
395                     }
396                 }
397             }
398         }
399         nextStep();
400     }

402     protected void openBoosters() {
403         for (TournamentPlayer player : this.players.values()) {
404             player.setDeck(new Deck());
405             if (options.getLimitedOptions().getDraftCube() != null) {
406                 DraftCube cube = options.getLimitedOptions().getDraftCube();
407                 for (int i = 0; i < options.getLimitedOptions().getNumberBoosters(); i++) {
408                     player.getDeck().getSideboard().addAll(cube.createBooster());
409                 }
410             } else if (options.getLimitedOptions().getIsJumpstart()) {
411                 if (options.getLimitedOptions().jumpstartPacks == null) {
412                     player.getDeck().getCards().addAll(JumpstartPoolGenerator.generatePool());
413                 } else {
414                     player.getDeck().getCards().addAll(JumpstartPoolGenerator.generatePool(options.getLimitedOptions().jumpstartPacks));
415                 }
416             } else {
417                 for (ExpansionSet set : sets) {
418                     player.getDeck().getSideboard().addAll(set.createBooster());
419                 }
420             }
421         }
422         nextStep();
423     }

466     protected void winners() {
467         List<TournamentPlayer> winners = new ArrayList<>();
468         int pointsWinner = 1; // with less than 1 point you can't win
469         for (TournamentPlayer tournamentPlayer : this.getPlayers()) {
470             if (pointsWinner < tournamentPlayer.getPoints()) {
471                 winners.clear();
472                 winners.add(tournamentPlayer);
473                 pointsWinner = tournamentPlayer.getPoints();
474             } else if (pointsWinner == tournamentPlayer.getPoints()) {
475                 winners.add(tournamentPlayer);
476             }
477         }
478         // set winner state for the players with the most points > 0
479         for (TournamentPlayer tournamentPlayer : winners) {
480             tournamentPlayer.setStateInfo("Winner");
481         }
482     }

539     @Override
540     public TourneyProto toProto() {
541         TourneyProto.Builder tourneyBuilder = TourneyProto.newBuilder()
542                 .setBoosterInfo(this.getBoosterInfo());
543         for (TournamentPlayer player : players.values()) {
544             TournamentPlayer replacedPlayer = player.getReplacedTournamentPlayer();
545             if (replacedPlayer != null) {
546                 player = replacedPlayer;
547             }
548             tourneyBuilder.addPlayersBuilder().mergeFrom(player.toProto());
549         }
550         for (Round round : rounds) {
551             TourneyRoundProto.Builder roundBuilder = tourneyBuilder.addRoundsBuilder()
552                     .setRound(round.getRoundNumber());
553             for (TournamentPairing pair : round.getPairs()) {
554                 Match match = pair.getMatch();
555                 if (match != null && match.hasEnded()) {
556                     MatchProto.Builder matchBuilder = roundBuilder.addMatchesBuilder()
557                             .setName(match.getName())
558                             .setGameType(match.getOptions().getGameType())
559                             .setDeckType(match.getOptions().getDeckType())
560                             .setGames(match.getNumGames())
561                             .setDraws(match.getDraws())
562                             .addPlayers(matchToProto(match, pair.getPlayer1()))
563                             .addPlayers(matchToProto(match, pair.getPlayer2()))
564                             .setMatchOptions(match.getOptions().toProto())
565                             .setEndTimeMs((match.getEndTime() != null ? match.getEndTime() : new Date()).getTime());
566                 }
567             }
568             for (TournamentPlayer tp : round.getPlayerByes()) {
569                 roundBuilder.addByes(tp.getPlayer().getName());
570             }
571         }
572         return tourneyBuilder.build();
573     }

575     private MatchPlayerProto matchToProto(Match match, TournamentPlayer player) {
576         MatchPlayer matchPlayer = match.getPlayer(player.getPlayer().getId());
577         MatchQuitStatus quit = !matchPlayer.hasQuit() ? MatchQuitStatus.NO_MATCH_QUIT
578                 : matchPlayer.getPlayer().hasIdleTimeout() ? MatchQuitStatus.IDLE_TIMEOUT
579                 : matchPlayer.getPlayer().hasTimerTimeout() ? MatchQuitStatus.TIMER_TIMEOUT
580                 : MatchQuitStatus.QUIT;
581         return MatchPlayerProto.newBuilder()
582                 .setName(player.getPlayer().getName())
583                 .setHuman(player.getPlayer().isHuman())
584                 .setWins(matchPlayer.getWins())
585                 .setQuit(quit)
586                 .build();
587     }

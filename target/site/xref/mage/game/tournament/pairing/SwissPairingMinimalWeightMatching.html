15  // SwissPairingMinimalWeightMatching creates round pairings for swiss tournament.
16  // It assigns weight to each possible pair and searches perfect matching with minimal weight
17  // for more details see https://www.leaguevine.com/blog/18/swiss-tournament-scheduling-leaguevines-new-algorithm/
18  // This implementation don't use fast minimum weight maximum matching algorithm,
19  // it uses brute-force search, so it works reasonably fast only up to 16 players.
20  
21  public class SwissPairingMinimalWeightMatching {
22  
23      private final int playersCount;
24  
25      private List<PlayerInfo> swissPlayers;
26  
27      // number of vertexes in graph
28      private final int n;
29  
30      // weight of pairings
31      private final int[][] w;
32  
33      public SwissPairingMinimalWeightMatching(List<TournamentPlayer> players, List<Round> rounds, boolean isLastRound) {
34          playersCount = players.size();
35  
36          swissPlayers = new ArrayList<>();
37          for (TournamentPlayer tournamentPlayer : players) {
38              PlayerInfo swissPlayer = new PlayerInfo();
39              swissPlayer.setTournamentPlayer(tournamentPlayer);
40              swissPlayer.setPoints(tournamentPlayer.getPoints());
41              swissPlayers.add(swissPlayer);
42          }
43  
44          // shuffle players first to add some randomness
45          Collections.shuffle(swissPlayers);
46          Map<TournamentPlayer, Integer> map = new HashMap<>();
47          for (int i = 0; i < playersCount; i++) {
48              swissPlayers.get(i).setId(i);
49              map.put(swissPlayers.get(i).getTournamentPlayer(), i);
50          }
51  
52          // calculate Tie Breaker points -- Sum of Opponents' Scores (SOS)
53          // see http://senseis.xmp.net/?SOS
54          for (Round round : rounds) {
55              for (TournamentPairing pairing : round.getPairs()) {
56                  TournamentPlayer player1 = pairing.getPlayer1();
57                  TournamentPlayer player2 = pairing.getPlayer2();
58  
59                  Integer id1 = map.get(player1);
60                  Integer id2 = map.get(player2);
61  
62                  // a player could have left the tournament, so we should check if id is not null
63                  if (id1 != null) {
64                      swissPlayers.get(id1).setSosPoints(swissPlayers.get(id1).getSosPoints() + player2.getPoints());
65                  }
66                  if (id2 != null) {
67                      swissPlayers.get(id2).setSosPoints(swissPlayers.get(id2).getSosPoints() + player1.getPoints());
68                  }
69                  // todo: sos points for byes? maybe add player points?
70              }
71          }
72  
73          // sort by points and then by sos points
74          swissPlayers.sort((p1, p2) -> {
75              int result = p2.getPoints() - p1.getPoints();
76              if (result != 0) {
77                  return result;
78              }
79              return p2.getSosPoints() - p1.getSosPoints();
80          });
81  
82          // order could be changed, update ids and mapping
83          map.clear();
84          for (int i = 0; i < playersCount; i++) {
85              swissPlayers.get(i).setId(i);
86              map.put(swissPlayers.get(i).getTournamentPlayer(), i);
87          }
88  
89          // count ties and matches between players
90          int[][] duels = new int[playersCount][playersCount];
91          int[] byes = new int[playersCount];
92          for (Round round : rounds) {
93              for (TournamentPairing pairing : round.getPairs()) {
94                  TournamentPlayer player1 = pairing.getPlayer1();
95                  TournamentPlayer player2 = pairing.getPlayer2();
96  
97                  Integer id1 = map.get(player1);
98                  Integer id2 = map.get(player2);
99  
100                 if (id1 != null && id2 != null) {
101                     duels[id1][id2]++;
102                     duels[id2][id1]++;
103                 }
104             }
105             for (TournamentPlayer playerBye : round.getPlayerByes()) {
106                 Integer id = map.get(playerBye);
107                 if (id != null) {
108                     byes[id]++;
109                 }
110             }
111         }
112 
113         // set vertex count
114         // add vertex for bye if we have odd number of players
115         n = (playersCount % 2 == 1 ? playersCount + 1 : playersCount);
116 
117         // calculate weight
118         // try to pair players with equal scores
119         w = new int[n][n];
120         int pointsDiffMultiplier = 10;
121         if (isLastRound) {
122             // for the last round, for each unpaired player starting with the first place, pair
123             // against the highest ranked player they haven't played against
124             for (int i = 0; i < playersCount; i++) {
125                 for (int j = 0; j < i; j++) {
126                     w[i][j] = Math.abs(i - j) +
127                             pointsDiffMultiplier * Math.abs(swissPlayers.get(i).getPoints() - swissPlayers.get(j).getPoints());
128                     w[j][i] = w[i][j];
129                 }
130             }
131         } else {
132             for (int i = 0; i < playersCount; i++) {
133                 PlayerInfo player = swissPlayers.get(i);
134                 for (int p = player.getPoints(); p >= 0; p--) {
135                     int first = -1;
136                     int last = -1;
137                     for (int j = 0; j < playersCount; j++) {
138                         if (swissPlayers.get(j).getPoints() == p) {
139                             if (first < 0) {
140                                 first = j;
141                             }
142                             last = j;
143                         }
144                     }
145                     if (first < 0) {
146                         continue;
147                     }
148                     int self = (p == player.getPoints() ? i : first - 1);
149                     int diff = pointsDiffMultiplier * (player.getPoints() - p);
150                     for (int j = Math.max(first, i); j <= last; j++) {
151                         w[i][j] = Math.abs(j - (last + first - self)) + diff;
152                         w[j][i] = w[i][j];
153                     }
154                 }
155             }
156         }
157 
158         // avoid pairing players that have played each other already
159         for (int i = 0; i < playersCount; i++) {
160             for (int j = 0; j < i; j++) {
161                 w[i][j] += duels[i][j] * 500;
162                 w[j][i] = w[i][j];
163             }
164         }
165 
166         // try to give bye to a player with a low score
167         // try to avoid giving the same person multiple byes
168         if (n > playersCount) {
169             for (int i = 0; i < playersCount; i++) {
170                 w[i][n - 1] = 10 * (swissPlayers.get(i).getPoints() - swissPlayers.get(playersCount - 1).getPoints()) + (playersCount - i - 1);
171                 w[i][n - 1] += byes[i] * 2000;
172                 w[n - 1][i] = w[i][n - 1];
173             }
174         }
175 
176         for (int i = 0; i < n; i++) {
177             for (int j = 0; j < n; j++) {
178                 w[i][j] *= w[i][j];
179             }
180         }
181 
182         // initialize variables for backtrack
183         used = new boolean[n];
184         pairs = new int[n];
185         Arrays.fill(pairs, -1);
186         result = new int[n];
187         weight = 0;
188         minCost = -1;
189         makePairings(0);
190     }

192     public RoundPairings getRoundPairings() {
193         // return round pairings with minimal weight
194         List<TournamentPairing> pairings = new ArrayList<>();
195         List<TournamentPlayer> playerByes = new ArrayList<>();
196 
197         Map<Integer, TournamentPlayer> map = new HashMap<>();
198         for (PlayerInfo player : swissPlayers) {
199             map.put(player.getId(), player.getTournamentPlayer());
200         }
201 
202         if (n > playersCount) {
203             // last vertex -- bye
204             playerByes.add(map.get(result[n - 1]));
205             result[result[n - 1]] = -1;
206             result[n - 1] = -1;
207         }
208 
209         for (int i = 0; i < playersCount; i++) {
210             if (result[i] >= 0) {
211                 pairings.add(new TournamentPairing(map.get(i), map.get(result[i])));
212                 result[result[i]] = -1;
213                 result[i] = -1;
214             }
215         }
216 
217         return new RoundPairings(pairings, playerByes);
218     }

220     private boolean[] used;
221 
222     // current pairs
223     private int[] pairs;
224     // current weight
225     private int weight;
226 
227     private int[] result;
228     private int minCost;
229 
230     // backtrack all possible pairings and choose one with minimal weight
231     private void makePairings(int t) {
232         if (t >= n) {
233             if (minCost < 0 || minCost > weight) {
234                 minCost = weight;
235                 System.arraycopy(pairs, 0, result, 0, n);
236             }
237             return;
238         }
239 
240         if (!used[t]) {
241             for (int i = t + 1; i < n; i++) {
242                 if (!used[i]) {
243                     pairs[t] = i;
244                     pairs[i] = t;
245                     used[t] = true;
246                     used[i] = true;
247                     weight += w[t][i];
248 
249                     makePairings(t + 1);
250 
251                     pairs[t] = -1;
252                     pairs[i] = -1;
253                     used[t] = false;
254                     used[i] = false;
255                     weight -= w[t][i];
256                 }
257             }
258         } else {
259             makePairings(t + 1);
260         }
261     }

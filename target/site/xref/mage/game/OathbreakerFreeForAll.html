25  public class OathbreakerFreeForAll extends GameCommanderImpl {
26  
27      private int numPlayers;
28      private final Map<UUID, Set<UUID>> playerSignatureSpells = new HashMap<>();
29      private final Map<UUID, Set<UUID>> playerOathbreakers = new HashMap<>();
30  
31      private static final String COMMANDER_NAME_OATHBREAKER = "Oathbreaker";
32      private static final String COMMANDER_NAME_SIGNATURE_SPELL = "Signature Spell";
33  
34      public OathbreakerFreeForAll(MultiplayerAttackOption attackOption, RangeOfInfluence range, Mulligan mulligan, int startLife) {
35          super(attackOption, range, mulligan, startLife, 100);
36          this.startingPlayerSkipsDraw = false;
37      }
38  
39      public OathbreakerFreeForAll(final OathbreakerFreeForAll game) {
40          super(game);
41          this.numPlayers = game.numPlayers;
42          game.playerSignatureSpells.forEach((key, value) -> this.playerSignatureSpells.put(key, new HashSet<>(value)));
43          game.playerOathbreakers.forEach((key, value) -> this.playerOathbreakers.put(key, new HashSet<>(value)));
44      }
45  
46      private String getCommanderTypeName(Card commander) {
47          return commander.isInstantOrSorcery() ? COMMANDER_NAME_SIGNATURE_SPELL : COMMANDER_NAME_OATHBREAKER;
48      }
49  
50      @Override
51      public CommanderInfoWatcher initCommanderWatcher(Card commander, boolean checkCommanderDamage) {
52          return new CommanderInfoWatcher(getCommanderTypeName(commander), commander.getId(), checkCommanderDamage);
53      }
54  
55      @Override
56      public void initCommanderEffects(Card commander, Player player, Ability commanderAbility) {
57          // all commander effects must be independent from sourceId or controllerId (it's limitation of current commander effects)
58  
59          boolean isSignatureSpell = this.playerSignatureSpells.getOrDefault(player.getId(), new HashSet<>()).contains(commander.getId());
60  
61          // basic commmander restrict (oathbreaker may ask to move, signature force to move)
62          commanderAbility.addEffect(new CommanderReplacementEffect(commander.getId(), alsoHand, alsoLibrary, isSignatureSpell, getCommanderTypeName(commander)));
63          commanderAbility.addEffect(new CommanderCostModification(commander));
64  
65          // signature spell restrict (spell can be casted on player's commander on battlefield)
66          if (isSignatureSpell) {
67              OathbreakerOnBattlefieldCondition condition = new OathbreakerOnBattlefieldCondition(this, player.getId(), commander.getId(),
68                      this.playerOathbreakers.getOrDefault(player.getId(), new HashSet<>()));
69              commanderAbility.addEffect(new SignatureSpellCastOnlyWithOathbreakerEffect(condition, commander.getId()));
70  
71              // hint must be added to card, not global ability
72              Ability ability = new SimpleStaticAbility(new InfoEffect("Signature spell hint"));
73              ability.addHint(new ConditionHint(condition, "Oathbreaker on battlefield (" + condition.getCompatibleNames() + ")"));
74              ability.setRuleVisible(false);
75              commander.addAbility(ability);
76          }
77      }
78  
79      private void addInnerCommander(Map<UUID, Set<UUID>> destList, UUID playerId, UUID cardId) {
80          Set<UUID> list = destList.getOrDefault(playerId, null);
81          if (list == null) {
82              list = new HashSet<>();
83              destList.put(playerId, list);
84          }
85          list.add(cardId);
86      }
87  
88      @Override
89      public void addCommander(Card card, Player player) {
90          super.addCommander(card, player);
91  
92          // prepare signature and commanders info
93          if (card.isInstantOrSorcery()) {
94              addInnerCommander(this.playerSignatureSpells, player.getId(), card.getId());
95          } else {
96              addInnerCommander(this.playerOathbreakers, player.getId(), card.getId());
97          }
98      }

116         return new OathbreakerFreeForAll(this);

119     @Override
120     public Set<UUID> getCommandersIds(Player player, CommanderCardType commanderCardType, boolean returnAllCardParts) {
121         Set<UUID> res = new HashSet<>();
122         if (player != null) {
123             Set<UUID> commanders = this.playerOathbreakers.getOrDefault(player.getId(), new HashSet<>());
124             Set<UUID> spells = this.playerSignatureSpells.getOrDefault(player.getId(), new HashSet<>());
125             for (UUID commanderId : super.getCommandersIds(player, commanderCardType, returnAllCardParts)) {
126                 switch (commanderCardType) {
127                     case ANY:
128                         res.add(commanderId);
129                         break;
130                     case COMMANDER_OR_OATHBREAKER:
131                         if (commanders.contains(commanderId)) {
132                             res.add(commanderId);
133                         }
134                         break;
135                     case SIGNATURE_SPELL:
136                         // TODO: doesn't filter mdf cards with different sides (creature + spell)
137                         if (spells.contains(commanderId)) {
138                             res.add(commanderId);
139                         }
140                         break;
141                     default:
142                         throw new IllegalStateException("Unknown commander type " + commanderCardType);
143                 }
144             }
145         }
146         return super.filterCommandersBySearchZone(res, returnAllCardParts);
147     }

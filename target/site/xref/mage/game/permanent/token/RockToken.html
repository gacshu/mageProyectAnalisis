16  public final class RockToken extends TokenImpl {
17  
18      public RockToken() {
19          super("Rock", "artifact equipment token named Rock with \"Equipped creature has '{1}, {T}, Sacrifice Rock: This creature deals 2 damage to any target'\" and equip {1}");
20          cardType.add(CardType.ARTIFACT);
21          subtype.add(SubType.EQUIPMENT);
22  
23          this.addAbility(new SimpleStaticAbility(new GainAbilityWithAttachmentEffect(
24                  "equipped creature has \"{1}, {T}, Sacrifice {this}: This creature deals 2 damage to any target.\"",
25                  new DamageTargetEffect(2), new TargetAnyTarget(), new SacrificeAttachmentCost(), new GenericManaCost(1), new TapSourceCost()
26          )));
27          this.addAbility(new EquipAbility(1));
28  
29          availableImageSetCodes = Arrays.asList("CMR");
30      }
31  
32      public RockToken(final RockToken token) {
33          super(token);
34      }
35  
36      public RockToken copy() {
37          return new RockToken(this);
38      }

43      }

12  import mage.abilities.effects.common.continuous.SetBasePowerToughnessSourceEffect;
13  import mage.constants.Duration;
14  import mage.constants.Zone;
15  import mage.counters.CounterType;
16  import mage.game.Game;
17  import mage.game.permanent.Permanent;

20   *
21   * @author spjspj
22   */
23  public final class GutterGrimeToken extends TokenImpl {
24  
25      public GutterGrimeToken() {
26          this ((UUID)null);
27          power = new MageInt(3);
28          toughness = new MageInt(3);
29      }
30  
31      public GutterGrimeToken(UUID sourceId) {
32          super("Ooze Token", "green Ooze creature token with \"This creature's power and toughness are each equal to the number of slime counters on Gutter Grime.\"");
33          cardType.add(CardType.CREATURE);
34          subtype.add(SubType.OOZE);
35          color.setGreen(true);
36          power = new MageInt(0);
37          toughness = new MageInt(0);
38          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new SetBasePowerToughnessSourceEffect(new GutterGrimeCounters(sourceId), Duration.WhileOnBattlefield)));
39      }
40  
41      public GutterGrimeToken(final GutterGrimeToken token) {
42          super(token);
43      }
44  
45      public GutterGrimeToken copy() {
46          return new GutterGrimeToken(this);
47      }
48  
49      class GutterGrimeCounters implements DynamicValue {
50  
51          private final UUID sourceId;
52  
53          public GutterGrimeCounters(UUID sourceId) {
54              this.sourceId = sourceId;
55          }
56  
57          @Override
58          public int calculate(Game game, Ability sourceAbility, Effect effect) {
59              Permanent p = game.getPermanent(sourceId);
60              if (p != null) {
61                  return p.getCounters(game).getCount(CounterType.SLIME);
62              }
63              return 0;
64          }
65  
66          @Override
67          public GutterGrimeCounters copy() {
68              return this;
69          }
70  
71          @Override
72          public String getMessage() {
73              return "slime counters on Gutter Grime";
74          }
75  
76          @Override
77          public String toString() {
78              return "1";
79          }
80      }
81  }

15  public final class TamiyosNotebookToken extends TokenImpl {
16  
17      private static final FilterCard filter = new FilterCard("spells");
18  
19      public TamiyosNotebookToken() {
20          super("Tamiyo's Notebook", "Tamiyo's Notebook, a legendary colorless artifact token with \"Spells you cast cost {2} less to cast\" and \"{T}: Draw a card.\"");
21          addSuperType(SuperType.LEGENDARY);
22          this.cardType.add(CardType.ARTIFACT);
23          this.addAbility(new SimpleStaticAbility(new SpellsCostReductionControllerEffect(filter, 2)));
24          this.addAbility(new SimpleActivatedAbility(new DrawCardSourceControllerEffect(1), new TapSourceCost()));
25      }
26  
27      public TamiyosNotebookToken(final TamiyosNotebookToken token) {
28          super(token);
29      }
30  
31      public TamiyosNotebookToken copy() {
32          return new TamiyosNotebookToken(this);
33      }

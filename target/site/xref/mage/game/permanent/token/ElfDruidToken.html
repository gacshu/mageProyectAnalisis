4   import java.util.Arrays;

6   import mage.constants.CardType;
7   import mage.constants.SubType;
8   import mage.MageInt;
9   import mage.ObjectColor;
10  import mage.abilities.mana.GreenManaAbility;
11  
12  /**
13   *
14   * @author spjspj
15   */
16  public final class ElfDruidToken extends TokenImpl {
17  
18      public ElfDruidToken() {
19          this(null, 0);
20      }
21  
22      public ElfDruidToken(String setCode) {
23          this(setCode, 0);
24      }
25  
26      public ElfDruidToken(String setCode, int tokenType) {
27          super("Elf Druid Token", "1/1 green Elf Druid creature token with \"{T}: Add {G}.\"");
28          this.cardType.add(CardType.CREATURE);
29          this.color.setGreen(true);
30          this.subtype.add(SubType.ELF);
31          this.subtype.add(SubType.DRUID);
32  
33          this.power = new MageInt(1);
34          this.toughness = new MageInt(1);
35  
36          // {T}: Add {G}.
37          this.addAbility(new GreenManaAbility());
38  
39          availableImageSetCodes = Arrays.asList("C14", "CMA");
40      }
41  
42      public ElfDruidToken(final ElfDruidToken token) {
43          super(token);
44      }
45  
46      public ElfDruidToken copy() {
47          return new ElfDruidToken(this);
48      }
49  }

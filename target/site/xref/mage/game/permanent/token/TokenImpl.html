88      private String tokenDescriptor() {
89          String strName = this.name.replaceAll("[^a-zA-Z0-9]", "");
90          String strColor = this.color.toString().replaceAll("[^a-zA-Z0-9]", "");
91          String strSubtype = this.subtype.toString().replaceAll("[^a-zA-Z0-9]", "");
92          String strCardType = this.cardType.toString().replaceAll("[^a-zA-Z0-9]", "");
93          String descriptor = strName + '.' + strColor + '.' + strSubtype + '.' + strCardType + '.' + this.power + '.' + this.toughness;
94          descriptor = descriptor.toUpperCase(Locale.ENGLISH);
95          return descriptor;
96      }

116     @Override
117     public void removeAbility(Ability abilityToRemove) {
118         if (abilityToRemove == null) {
119             return;
120         }
121 
122         // 112.10b  Effects that remove an ability remove all instances of it.
123         List<Ability> toRemove = new ArrayList<>();
124         abilities.forEach(a -> {
125             if (a.isSameInstance(abilityToRemove)) {
126                 toRemove.add(a);
127             }
128         });
129 
130         // TODO: what about triggered abilities? See addAbility above -- triggers adds to GameState
131         toRemove.forEach(r -> abilities.remove(r));
132     }

159     private String getSetCode(Game game, UUID sourceId) {
160         // moved here from CreateTokenEffect because not all cards that create tokens use CreateTokenEffect
161         // they use putOntoBattlefield directly
162         // TODO: Check this setCode handling because it makes no sense if token put into play with e.g. "Feldon of the third Path"
163         String setCode = null;
164         if (this.getOriginalExpansionSetCode() != null && !this.getOriginalExpansionSetCode().isEmpty()) {
165             setCode = this.getOriginalExpansionSetCode();
166         } else {
167             Card source = game.getCard(sourceId);
168             if (source != null) {
169                 setCode = source.getExpansionSetCode();
170             } else {
171                 MageObject object = game.getObject(sourceId);
172                 if (object instanceof PermanentToken) {
173                     setCode = ((PermanentToken) object).getExpansionSetCode();
174                 }
175             }
176         }
177 
178         if (!expansionSetCodeChecked) {
179             expansionSetCodeChecked = this.updateExpansionSetCode(setCode);
180         }
181         return setCode;
182     }

189     @Override
190     public boolean putOntoBattlefield(int amount, Game game, Ability source, UUID controllerId, boolean tapped, boolean attacking, UUID attackedPlayer, boolean created) {
191         Player controller = game.getPlayer(controllerId);
192         if (controller == null) {
193             return false;
194         }
195         if (amount == 0) {
196             return false;
197         }
198         lastAddedTokenIds.clear();
199 
200         CreateTokenEvent event = new CreateTokenEvent(source, controllerId, amount, this);
201         if (!created || !game.replaceEvent(event)) {
202             int currentTokens = game.getBattlefield().countTokens(event.getPlayerId());
203             int tokenSlots = Math.max(MAX_TOKENS_PER_GAME - currentTokens, 0);
204             int amountToRemove = event.getAmount() - tokenSlots;
205             if (amountToRemove > 0) {
206                 game.informPlayers(
207                         "The token limit per player is " + MAX_TOKENS_PER_GAME + ", " + controller.getName()
208                                 + " will only create " + tokenSlots + " tokens."
209                 );
210                 Iterator<Map.Entry<Token, Integer>> it = event.getTokens().entrySet().iterator();
211                 while (it.hasNext() && amountToRemove > 0) {
212                     Map.Entry<Token, Integer> entry = it.next();
213                     int newValue = entry.getValue() - amountToRemove;
214                     if (newValue > 0) {
215                         entry.setValue(newValue);
216                         break;
217                     }
218                     amountToRemove -= entry.getValue();
219                     it.remove();
220                 }
221             }
222             putOntoBattlefieldHelper(event, game, source, tapped, attacking, attackedPlayer, created);
223             event.getTokens()
224                     .keySet()
225                     .stream()
226                     .map(Token::getLastAddedTokenIds)
227                     .flatMap(Collection::stream)
228                     .distinct()
229                     .filter(uuid -> !this.lastAddedTokenIds.contains(uuid))
230                     .forEach(this.lastAddedTokenIds::add);
231             return true;
232         }
233         return false;
234     }

236     private static void putOntoBattlefieldHelper(CreateTokenEvent event, Game game, Ability source, boolean tapped, boolean attacking, UUID attackedPlayer, boolean created) {
237         Player controller = game.getPlayer(event.getPlayerId());
238         if (controller == null) {
239             return;
240         }
241 
242         for (Map.Entry<Token, Integer> entry : event.getTokens().entrySet()) {
243             Token token = entry.getKey();
244             int amount = entry.getValue();
245             String setCode = token instanceof TokenImpl ? ((TokenImpl) token).getSetCode(game, event.getSourceId()) : null;
246 
247             List<Permanent> needTokens = new ArrayList<>();
248             List<Permanent> allowedTokens = new ArrayList<>();
249 
250             // prepare tokens to enter
251             for (int i = 0; i < amount; i++) {
252                 // use event.getPlayerId() as controller cause it can be replaced by replacement effect
253                 PermanentToken newPermanent = new PermanentToken(token, event.getPlayerId(), setCode, game);
254                 game.getState().addCard(newPermanent);
255                 needTokens.add(newPermanent);
256                 game.getPermanentsEntering().put(newPermanent.getId(), newPermanent);
257                 newPermanent.setTapped(tapped);
258 
259                 ZoneChangeEvent emptyEvent = new ZoneChangeEvent(newPermanent, newPermanent.getControllerId(), Zone.OUTSIDE, Zone.BATTLEFIELD);
260                 // tokens zcc must simulate card's zcc too keep copied card/spell settings
261                 // (example: etb's kicker ability of copied creature spell, see tests with Deathforge Shaman)
262                 newPermanent.updateZoneChangeCounter(game, emptyEvent);
263             }
264 
265             // check ETB effects
266             game.setScopeRelevant(true);
267             for (Permanent permanent : needTokens) {
268                 if (permanent.entersBattlefield(source, game, Zone.OUTSIDE, true)) {
269                     allowedTokens.add(permanent);
270                 } else {
271                     game.getPermanentsEntering().remove(permanent.getId());
272                 }
273             }
274             game.setScopeRelevant(false);
275 
276             // put allowed tokens to play
277             int createOrder = game.getState().getNextPermanentOrderNumber();
278             for (Permanent permanent : allowedTokens) {
279                 game.addPermanent(permanent, createOrder);
280                 permanent.setZone(Zone.BATTLEFIELD, game);
281                 game.getPermanentsEntering().remove(permanent.getId());
282 
283                 // keep tokens ids
284                 if (token instanceof TokenImpl) {
285                     ((TokenImpl) token).lastAddedTokenIds.add(permanent.getId());
286                 }
287 
288                 // created token events
289                 ZoneChangeEvent zccEvent = new ZoneChangeEvent(permanent, permanent.getControllerId(), Zone.OUTSIDE, Zone.BATTLEFIELD);
290                 game.addSimultaneousEvent(zccEvent);
291                 if (permanent instanceof PermanentToken && created) {
292                     game.addSimultaneousEvent(new CreatedTokenEvent(source, (PermanentToken) permanent));
293                 }
294 
295                 // if token was created (not a spell copy) handle auras coming into the battlefield
296                 // code blindly copied from CopyPermanentEffect
297                 // TODO: clean this up -- half the comments make no sense in the context of creating a token
298                 if (created && permanent.getSubtype().contains(SubType.AURA)) {
299                     Outcome auraOutcome = Outcome.BoostCreature;
300                     Target auraTarget = null;
301 
302                     // attach - search effect in spell ability (example: cast Utopia Sprawl, cast Estrid's Invocation on it)
303                     for (Ability ability : permanent.getAbilities()) {
304                         if (!(ability instanceof SpellAbility)) {
305                             continue;
306                         }
307                         auraOutcome = ability.getEffects().getOutcome(ability);
308                         for (Effect effect : ability.getEffects()) {
309                             if (!(effect instanceof AttachEffect)) {
310                                 continue;
311                             }
312                             if (permanent.getSpellAbility().getTargets().size() > 0) {
313                                 auraTarget = permanent.getSpellAbility().getTargets().get(0);
314                             }
315                         }
316                     }
317 
318                     // enchant - search in all abilities (example: cast Estrid's Invocation on enchanted creature by Estrid, the Masked second ability, cast Estrid's Invocation on it)
319                     if (auraTarget == null) {
320                         for (Ability ability : permanent.getAbilities()) {
321                             if (!(ability instanceof EnchantAbility)) {
322                                 continue;
323                             }
324                             auraOutcome = ability.getEffects().getOutcome(ability);
325                             if (ability.getTargets().size() > 0) { // Animate Dead don't have targets
326                                 auraTarget = ability.getTargets().get(0);
327                             }
328                         }
329                     }
330 
331                     // if this is a copy of a copy, the copy's target has been copied and needs to be cleared
332                     if (auraTarget == null) {
333                         break;
334                     }
335                     // clear selected target
336                     if (auraTarget.getFirstTarget() != null) {
337                         auraTarget.remove(auraTarget.getFirstTarget());
338                     }
339 
340                     // select new target
341                     auraTarget.setNotTarget(true);
342                     if (!controller.choose(auraOutcome, auraTarget, source, game)) {
343                         break;
344                     }
345                     UUID targetId = auraTarget.getFirstTarget();
346                     Permanent targetPermanent = game.getPermanent(targetId);
347                     Player targetPlayer = game.getPlayer(targetId);
348                     if (targetPermanent != null) {
349                         targetPermanent.addAttachment(permanent.getId(), source, game);
350                     } else if (targetPlayer != null) {
351                         targetPlayer.addAttachment(permanent.getId(), source, game);
352                     }
353                 }
354                 // end of aura code : just remove this line if everything works out well
355 
356                 // must attack
357                 if (attacking && game.getCombat() != null && game.getActivePlayerId().equals(permanent.getControllerId())) {
358                     game.getCombat().addAttackingCreature(permanent.getId(), game, attackedPlayer);
359                 }
360 
361                 // game logs
362                 if (created) {
363                     game.informPlayers(controller.getLogName() + " creates a " + permanent.getLogName() + " token");
364                 } else {
365                     game.informPlayers(permanent.getLogName() + " enters the battlefield as a token under " + controller.getLogName() + "'s control'");
366                 }
367             }
368         }
369         game.getState().applyEffects(game); // Needed to do it here without LKIReset i.e. do get SwordOfTheMeekTest running correctly.
370     }

431     @Override
432     public void setExpansionSetCodeForImage(String code) {
433         if (!availableImageSetCodes.isEmpty()) {
434             if (availableImageSetCodes.contains(code)) {
435                 setOriginalExpansionSetCode(code);
436             } else // we should not set random set if appropriate set is already used
437             {
438                 if (getOriginalExpansionSetCode() == null || getOriginalExpansionSetCode().isEmpty()
439                         || !availableImageSetCodes.contains(getOriginalExpansionSetCode())) {
440                     setOriginalExpansionSetCode(availableImageSetCodes.get(RandomUtil.nextInt(availableImageSetCodes.size())));
441                 }
442             }
443         } else if (getOriginalExpansionSetCode() == null || getOriginalExpansionSetCode().isEmpty()) {
444             setOriginalExpansionSetCode(code);
445         }
446         setTokenDescriptor();
447     }

449     @Override
450     public boolean updateExpansionSetCode(String setCode) {
451         if (setCode == null || setCode.isEmpty()) {
452             return false;
453         }
454         this.setExpansionSetCodeForImage(setCode);
455         return true;

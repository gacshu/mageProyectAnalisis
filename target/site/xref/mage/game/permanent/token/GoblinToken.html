1   package mage.game.permanent.token;
2   
3   import mage.MageInt;
4   import mage.abilities.keyword.HasteAbility;
5   import mage.constants.CardType;
6   import mage.constants.SubType;
7   
8   import java.util.Arrays;
9   
10  /**
11   * @author North
12   */
13  public final class GoblinToken extends TokenImpl {
14  
15      public GoblinToken(boolean withHaste) {
16          this();
17  
18          // token image don't have haste info so it's ok to use same class for different versions
19          if (withHaste) {
20              addAbility(HasteAbility.getInstance());
21              this.description = "1/1 red Goblin creature token with haste";
22          }
23      }
24  
25      public GoblinToken() {
26          super("Goblin Token", "1/1 red Goblin creature token");
27          cardType.add(CardType.CREATURE);
28          subtype.add(SubType.GOBLIN);
29          color.setRed(true);
30          power = new MageInt(1);
31          toughness = new MageInt(1);
32  
33          availableImageSetCodes = Arrays.asList("10E", "ALA", "SOM", "M10", "M13", "RTR",
34                  "MMA", "M15", "C14", "KTK", "EVG", "DTK", "ORI", "DDG", "DDN",
35                  "MM3", "EMA", "DOM", "RNA", "WAR", "MH1", "TSR", "MH2", "AFR",
36                  "NEC", "M19", "CM2", "PCA", "DD1", "DDS", "DDT", "A25", "GRN",
37                  "GK1", "DMU", "DMR", "ONC");
38      }
39  
40      public GoblinToken(final GoblinToken token) {
41          super(token);
42      }
43  
44      public GoblinToken copy() {
45          return new GoblinToken(this);
46      }
47  }

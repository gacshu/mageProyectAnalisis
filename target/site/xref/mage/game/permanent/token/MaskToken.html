1   package mage.game.permanent.token;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.common.AttachEffect;
5   import mage.abilities.keyword.EnchantAbility;
6   import mage.abilities.keyword.TotemArmorAbility;
7   import mage.constants.CardType;
8   import mage.constants.Outcome;
9   import mage.constants.SubType;
10  import mage.target.TargetPermanent;
11  
12  /**
13   * @author TheElk801
14   */
15  public final class MaskToken extends TokenImpl {
16  
17      public MaskToken() {
18          super(
19                  "Mask", "white Aura enchantment token named Mask "
20                          + "attached to another target permanent. "
21                          + "The token has enchant permanent and totem armor."
22          );
23          cardType.add(CardType.ENCHANTMENT);
24          color.setWhite(true);
25          subtype.add(SubType.AURA);
26  
27          TargetPermanent auraTarget = new TargetPermanent();
28          Ability ability = new EnchantAbility(auraTarget);
29          ability.addTarget(auraTarget);
30          ability.addEffect(new AttachEffect(Outcome.BoostCreature));
31          this.addAbility(ability);
32  
33          // Totem armor
34          this.addAbility(new TotemArmorAbility());
35      }
36  
37      public MaskToken(final MaskToken token) {
38          super(token);
39      }
40  
41      public MaskToken copy() {
42          return new MaskToken(this);
43      }
44  }

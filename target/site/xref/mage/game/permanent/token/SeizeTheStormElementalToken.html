20  public final class SeizeTheStormElementalToken extends TokenImpl {
21  
22      public SeizeTheStormElementalToken() {
23          this(StaticValue.get(0), new StaticHint(""));
24      }
25  
26      public SeizeTheStormElementalToken(DynamicValue xValue, Hint hint) {
27          super("Elemental Token", "red Elemental creature token with trample and " +
28                  "\"This creature's power and toughness are each equal to the number of instant " +
29                  "and sorcery cards in your graveyard plus the number of cards with flashback you own in exile.\"");
30          cardType.add(CardType.CREATURE);
31          color.setRed(true);
32          subtype.add(SubType.ELEMENTAL);
33          power = new MageInt(0);
34          toughness = new MageInt(0);
35          this.addAbility(TrampleAbility.getInstance());
36          this.addAbility(new SimpleStaticAbility(new SetBasePowerToughnessSourceEffect(
37                  xValue, Duration.WhileOnBattlefield
38          ).setText("this creature's power and toughness are each equal to the number of " +
39                  "instant and sorcery cards in your graveyard, plus the number of cards with flashback you own in exile")
40          ).addHint(hint));
41  
42          availableImageSetCodes = Arrays.asList("MID");
43      }
44  
45      private SeizeTheStormElementalToken(final SeizeTheStormElementalToken token) {
46          super(token);
47      }
48  
49      @Override
50      public SeizeTheStormElementalToken copy() {
51          return new SeizeTheStormElementalToken(this);
52      }
53  }

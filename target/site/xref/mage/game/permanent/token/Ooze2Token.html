3   package mage.game.permanent.token;
4   import mage.constants.CardType;
5   import mage.constants.SubType;
6   import mage.MageInt;
7   import mage.abilities.common.DiesSourceTriggeredAbility;
8   import mage.abilities.effects.common.CreateTokenEffect;
9   
10  import java.util.Arrays;
11  
12  /**
13   *
14   * @author spjspj
15   */
16  public final class Ooze2Token extends TokenImpl {
17  
18      public Ooze2Token() {
19          super("Ooze Token", "2/2 green Ooze creature token. It has \"When this creature dies, create two 1/1 green Ooze creature tokens.\"");
20          cardType.add(CardType.CREATURE);
21          subtype.add(SubType.OOZE);
22          color.setGreen(true);
23          power = new MageInt(2);
24          toughness = new MageInt(2);
25          this.addAbility(new DiesSourceTriggeredAbility(new CreateTokenEffect(new OozeToken(1, 1), 2), false));
26  
27          availableImageSetCodes = Arrays.asList("M11", "PCA", "NCC");
28      }
29  
30      @Override
31      public void setExpansionSetCodeForImage(String code) {
32          super.setExpansionSetCodeForImage(code);
33  
34          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("M11")) {
35              setTokenType(2);
36          }
37  
38          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("PCA")) {
39              setTokenType(2);
40          }
41  
42          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("NCC")) {
43              setTokenType(2);
44          }
45      }
46  
47      public Ooze2Token(final Ooze2Token token) {
48          super(token);
49      }
50  
51      public Ooze2Token copy() {
52          return new Ooze2Token(this);
53      }
54  }

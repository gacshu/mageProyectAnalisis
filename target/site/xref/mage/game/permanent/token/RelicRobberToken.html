15  public final class RelicRobberToken extends TokenImpl {
16  
17      public RelicRobberToken() {
18          super("Goblin Construct Token", "0/1 colorless Goblin Construct artifact creature token with \"This creature can't block\" and \"At the beginning of your upkeep, this creature deals 1 damage to you.\"");
19          cardType.add(CardType.ARTIFACT);
20          cardType.add(CardType.CREATURE);
21          subtype.add(SubType.GOBLIN);
22          subtype.add(SubType.CONSTRUCT);
23          power = new MageInt(0);
24          toughness = new MageInt(1);
25  
26          this.addAbility(new SimpleStaticAbility(
27                  new CantBlockSourceEffect(Duration.WhileOnBattlefield).setText("this creature can't block")
28          ));
29          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new DamageControllerEffect(
30                  1, "this creature"
31          ), TargetController.YOU, false));
32  
33          availableImageSetCodes = Arrays.asList("ZNR");
34      }
35  
36      public RelicRobberToken(final RelicRobberToken token) {
37          super(token);
38      }
39  
40      public RelicRobberToken copy() {
41          return new RelicRobberToken(this);
42      }

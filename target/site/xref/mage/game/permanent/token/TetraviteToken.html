9   import mage.game.Game;

12  /**
13   * @author spjspj
14   */
15  public final class TetraviteToken extends TokenImpl {
16  
17      public TetraviteToken() {
18          super("Tetravite Token", "1/1 colorless Tetravite artifact creature token with flying and \"This creature can't be enchanted.\"");
19          cardType.add(CardType.CREATURE);
20          cardType.add(CardType.ARTIFACT);
21          subtype.add(SubType.TETRAVITE);
22          power = new MageInt(1);
23          toughness = new MageInt(1);
24  
25          this.addAbility(FlyingAbility.getInstance());
26          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new TetraviteTokenEffect()));
27      }
28  
29      public TetraviteToken(final TetraviteToken token) {
30          super(token);
31      }
32  
33      public TetraviteToken copy() {
34          return new TetraviteToken(this);
35      }
36  }
37  
38  class TetraviteTokenEffect extends ContinuousRuleModifyingEffectImpl {
39  
40      public TetraviteTokenEffect() {
41          super(Duration.WhileOnBattlefield, Outcome.Detriment);
42          staticText = "this creature can't be enchanted";
43      }
44  
45      public TetraviteTokenEffect(final TetraviteTokenEffect effect) {
46          super(effect);
47      }
48  
49      @Override
50      public TetraviteTokenEffect copy() {
51          return new TetraviteTokenEffect(this);
52      }
53  
54      @Override
55      public boolean checksEventType(GameEvent event, Game game) {
56          return event.getType() == GameEvent.EventType.ATTACH
57                  || event.getType() == GameEvent.EventType.STAY_ATTACHED;
58      }
59  
60      @Override
61      public boolean applies(GameEvent event, Ability source, Game game) {
62          return event.getTargetId().equals(source.getSourceId());
63      }
64  }

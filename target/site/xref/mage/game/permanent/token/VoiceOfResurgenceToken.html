17  public final class VoiceOfResurgenceToken extends TokenImpl {
18  
19      public VoiceOfResurgenceToken() {
20          super("Elemental Token", "X/X green and white Elemental creature with with \"This creature's power and toughness are each equal to the number of creatures you control.");
21          setOriginalExpansionSetCode("DGM");
22          cardType.add(CardType.CREATURE);
23          color.setGreen(true);
24          color.setWhite(true);
25          subtype.add(SubType.ELEMENTAL);
26  
27          power = new MageInt(0);
28          toughness = new MageInt(0);
29  
30          // This creature's power and toughness are each equal to the number of creatures you control.
31          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new SetBasePowerToughnessSourceEffect(
32                  CreaturesYouControlCount.instance, Duration.EndOfGame)));
33              
34          availableImageSetCodes = Arrays.asList("DGM", "MM3", "2XM");
35      }
36  
37      public VoiceOfResurgenceToken(final VoiceOfResurgenceToken token) {
38          super(token);
39      }
40  
41      public VoiceOfResurgenceToken copy() {
42          return new VoiceOfResurgenceToken(this);
43      }
44  }

23  /**
24   * @author TheElk801
25   */
26  public class IcingdeathFrostTongueToken extends TokenImpl {
27  
28      private static final FilterPermanent filter
29              = new FilterCreaturePermanent("creature defending player controls");
30  
31      static {
32          filter.add(IcingdeathFrostTonguePredicate.instance);
33      }
34  
35      public IcingdeathFrostTongueToken() {
36          super("Icingdeath, Frost Tongue", "Icingdeath, Frost Tongue, a legendary white " +
37                  "Equipment artifact token with \"Equipped creature gets +2/+0,\" " +
38                  "\"Whenever equipped creature attacks, tap target creature " +
39                  "defending player controls,\" and equip {2}");
40          supertype.add(SuperType.LEGENDARY);
41          cardType.add(CardType.ARTIFACT);
42          color.setWhite(true);
43          subtype.add(SubType.EQUIPMENT);
44  
45          // Equipped creature gets +2/+0
46          this.addAbility(new SimpleStaticAbility(new BoostEquippedEffect(2, 0)));
47  
48          // Whenever equipped creature attacks, tap target creature defending player controls
49          Ability ability = new AttacksAttachedTriggeredAbility(new TapTargetEffect(), false);
50          ability.addTarget(new TargetPermanent(filter));
51          this.addAbility(ability);
52  
53          // Equip {2}
54          this.addAbility(new EquipAbility(2));
55  
56          availableImageSetCodes = Arrays.asList("AFR");
57      }
58  
59      private IcingdeathFrostTongueToken(final IcingdeathFrostTongueToken token) {
60          super(token);
61      }
62  
63      @Override
64      public IcingdeathFrostTongueToken copy() {
65          return new IcingdeathFrostTongueToken(this);
66      }
67  }
68  
69  enum IcingdeathFrostTonguePredicate implements ObjectSourcePlayerPredicate<Permanent> {
70      instance;
71  
72      @Override
73      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
74          return Optional.ofNullable(input.getSource().getSourcePermanentOrLKI(game))
75                  .map(Permanent::getAttachedTo)
76                  .map(uuid -> game.getCombat().getDefendingPlayerId(uuid, game))
77                  .map(input.getObject()::isControlledBy)
78                  .orElse(false);
79      }
80  
81      @Override
82      public String toString() {
83          return "";
84      }
85  }

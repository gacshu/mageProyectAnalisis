23  /**
24   *
25   * @author LevelX2
26   */
27  public final class RekindlingPhoenixToken extends TokenImpl {
28  
29      public RekindlingPhoenixToken() {
30          super("Elemental Token", "0/1 red Elemental creature token with \"At the beginning of your upkeep, sacrifice this creature and return target card named Rekindling Phoenix from your graveyard to the battlefield. It gains haste until end of turn.\"");
31          setTokenType(1);
32          cardType.add(CardType.CREATURE);
33          subtype.add(SubType.ELEMENTAL);
34          color.setRed(true);
35          power = new MageInt(0);
36          toughness = new MageInt(1);
37  
38          Ability ability = new BeginningOfUpkeepTriggeredAbility(Zone.BATTLEFIELD, new SacrificeSourceEffect(), TargetController.YOU, false);
39          ability.addEffect(new ReturnFromGraveyardToBattlefieldTargetEffect().setText("and return target card named Rekindling Phoenix from your graveyard to the battlefield"));
40          ability.addEffect(new GainAbilityTargetEffect(HasteAbility.getInstance(), Duration.EndOfTurn).setText("It gains haste until end of turn"));
41          FilterCard filter = new FilterCard("card named Rekindling Phoenix from your graveyard");
42          filter.add(new NamePredicate("Rekindling Phoenix"));
43          ability.addTarget(new TargetCardInYourGraveyard(filter));
44          this.addAbility(ability);
45  
46          availableImageSetCodes = Arrays.asList("RIX", "NCC");
47      }
48  
49      @Override
50      public void setExpansionSetCodeForImage(String code) {
51          super.setExpansionSetCodeForImage(code);
52  
53          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("RIX")) {
54              this.setTokenType(1);
55          }
56  
57          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("NCC")) {
58              this.setTokenType(1);
59          }
60      }
61  
62      public RekindlingPhoenixToken(final RekindlingPhoenixToken token) {
63          super(token);
64      }
65  
66      public RekindlingPhoenixToken copy() {
67          return new RekindlingPhoenixToken(this);
68      }
69  }

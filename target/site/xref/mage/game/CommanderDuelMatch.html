10  public class CommanderDuelMatch extends MatchImpl {
11  
12      public CommanderDuelMatch(MatchOptions options) {
13          super(options);
14      }
15  
16      @Override
17      public void startGame() throws GameException {
18          int startLife = 40;
19          // Don't like it to compare but seems like it's complicated to do it in another way
20          boolean checkCommanderDamage = true;
21          if (options.getDeckType().equals("Variant Magic - Duel Commander")) {
22              startLife = 20;   // Starting with the Commander 2016 update (on November 11th, 2016), Duel Commander will be played with 20 life points instead of 30.
23              checkCommanderDamage = false; // since nov 16 duel commander uses no longer commander damage rule
24          }
25          if (options.getDeckType().equals("Variant Magic - MTGO 1v1 Commander")) {
26              startLife = 30;
27          }
28          if (options.getDeckType().equals("Variant Magic - Centurion Commander")) {
29              startLife = 25;
30          }
31          Mulligan mulligan = options.getMulliganType().getMulligan(options.getFreeMulligans());
32          CommanderDuel game = new CommanderDuel(options.getAttackOption(), options.getRange(), mulligan, startLife);
33          game.setCheckCommanderDamage(checkCommanderDamage);
34          game.setStartMessage(this.createGameStartMessage());
35          initGame(game);
36          games.add(game);
37      }
38  
39  }

143     /**
144      * Returns a Set of opponents in range for the given playerId
145      *
146      * @param playerId
147      * @param excludeDeadPlayers Determines if players who have lost are excluded from the list
148      * @return
149      */
150     default Set<UUID> getOpponents(UUID playerId, boolean excludeDeadPlayers) {
151         Player player = getPlayer(playerId);
152         if (player == null) {
153             return new HashSet<>();
154         }
155 
156         return player.getInRange().stream()
157                 .filter(opponentId -> !opponentId.equals(playerId))
158                 .filter(opponentId -> !excludeDeadPlayers || !getPlayer(opponentId).hasLost())
159                 .collect(Collectors.toSet());
160     }

568     /**
569      * Return not played commander cards from command zone
570      * Read comments for CommanderCardType for more info on commanderCardType usage
571      *
572      * @param player
573      * @return
574      */
575     default Set<Card> getCommanderCardsFromCommandZone(Player player, CommanderCardType commanderCardType) {
576         // commanders in command zone aren't cards so you must call getCard instead getObject
577         return getCommandersIds(player, commanderCardType, false).stream()
578                 .map(this::getCard)
579                 .filter(Objects::nonNull)
580                 .filter(card -> Zone.COMMAND.equals(this.getState().getZone(card.getId())))
581                 .collect(Collectors.toSet());
582     }
583 
584     /**
585      * Return commander cards from any zones (main card from command and permanent card from battlefield)
586      * Read comments for CommanderCardType for more info on commanderCardType usage
587      *
588      * @param player
589      * @param commanderCardType commander or signature spell
590      * @return
591      */
592     default Set<Card> getCommanderCardsFromAnyZones(Player player, CommanderCardType commanderCardType, Zone... searchZones) {
593         Set<Zone> needZones = Arrays.stream(searchZones).collect(Collectors.toSet());
594         if (needZones.isEmpty()) {
595             throw new IllegalArgumentException("Empty zones list in searching commanders");
596         }
597         Set<UUID> needCommandersIds = this.getCommandersIds(player, commanderCardType, true);
598         Set<Card> needCommandersCards = needCommandersIds.stream()
599                 .map(this::getCard)
600                 .filter(Objects::nonNull)
601                 .collect(Collectors.toSet());
602         Set<Card> res = new HashSet<>();
603 
604         // hand
605         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.HAND)) {
606             needCommandersCards.stream()
607                     .filter(card -> Zone.HAND.equals(this.getState().getZone(card.getId())))
608                     .forEach(res::add);
609         }
610 
611         // graveyard
612         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.GRAVEYARD)) {
613             needCommandersCards.stream()
614                     .filter(card -> Zone.GRAVEYARD.equals(this.getState().getZone(card.getId())))
615                     .forEach(res::add);
616         }
617 
618         // library
619         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.LIBRARY)) {
620             needCommandersCards.stream()
621                     .filter(card -> Zone.LIBRARY.equals(this.getState().getZone(card.getId())))
622                     .forEach(res::add);
623         }
624 
625         // battlefield (need permanent card)
626         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.BATTLEFIELD)) {
627             needCommandersIds.stream()
628                     .map(this::getPermanent)
629                     .filter(Objects::nonNull)
630                     .forEach(res::add);
631         }
632 
633         // stack
634         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.STACK)) {
635             needCommandersCards.stream()
636                     .filter(card -> Zone.STACK.equals(this.getState().getZone(card.getId())))
637                     .forEach(res::add);
638         }
639 
640         // exiled
641         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.EXILED)) {
642             needCommandersCards.stream()
643                     .filter(card -> Zone.EXILED.equals(this.getState().getZone(card.getId())))
644                     .forEach(res::add);
645         }
646 
647         // command
648         if (needZones.contains(Zone.ALL) || needZones.contains(Zone.COMMAND)) {
649             res.addAll(getCommanderCardsFromCommandZone(player, commanderCardType));
650         }
651 
652         // outside must be ignored (example: second side of MDFC commander after cast)
653         if (needZones.contains(Zone.OUTSIDE)) {
654             throw new IllegalArgumentException("Outside zone doesn't supported in searching commanders");
655         }
656 
657         return res;
658     }

660     /**
661      * Finds is it a commander card/object (use it in conditional and other things)
662      *
663      * @param player
664      * @param object
665      * @return
666      */
667     default boolean isCommanderObject(Player player, MageObject object) {
668         UUID idToCheck = null;
669         if (object instanceof Spell) {
670             idToCheck = ((Spell) object).getCard().getId();
671         }
672         if (object instanceof CommandObject) {
673             idToCheck = object.getId();
674         }
675         if (object instanceof Card) {
676             idToCheck = ((Card) object).getMainCard().getId();
677         }
678         return idToCheck != null && this.getCommandersIds(player, CommanderCardType.COMMANDER_OR_OATHBREAKER, false).contains(idToCheck);
679     }

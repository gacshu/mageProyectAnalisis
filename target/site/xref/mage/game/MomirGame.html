20  /**
21   *
22   * @author nigelzor
23   */
24  public class MomirGame extends GameImpl {
25  
26      private int numPlayers;
27  
28      public MomirGame(MultiplayerAttackOption attackOption, RangeOfInfluence range, Mulligan mulligan, int startLife) {
29          super(attackOption, range, mulligan, startLife, 60);
30      }
31  
32      public MomirGame(final MomirGame game) {
33          super(game);
34      }
35  
36      @Override
37      public MatchType getGameType() {
38          return new MomirFreeForAllType();
39      }
40  
41      @Override
42      public int getNumPlayers() {
43          return numPlayers;
44      }
45  
46      @Override
47      protected void init(UUID choosingPlayerId) {
48          Ability ability = new SimpleStaticAbility(Zone.COMMAND, new InfoEffect("Vanguard effects"));
49          for (UUID playerId : state.getPlayerList(startingPlayerId)) {
50              Player player = getPlayer(playerId);
51              if (player != null) {
52                  CardInfo cardInfo = CardRepository.instance.findCard("Momir Vig, Simic Visionary");
53                  addEmblem(new MomirEmblem(), cardInfo.getCard(), playerId);
54              }
55          }
56          getState().addAbility(ability, null);
57          super.init(choosingPlayerId);
58          state.getTurnMods().add(new TurnMod(startingPlayerId, PhaseStep.DRAW));
59      }
60  
61      @Override
62      public MomirGame copy() {
63          return new MomirGame(this);
64      }
65  
66  }

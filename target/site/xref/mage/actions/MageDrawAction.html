20  /**
21   * Action for drawing cards.
22   *
23   * @author ayrat
24   */
25  public class MageDrawAction extends MageAction {
26  
27      private static final int NEGATIVE_VALUE = -1000000;
28  
29      private final Player player;
30      private final List<Card> drawnCards;
31      private final GameEvent originalDrawEvent; // for replace effects
32  
33      private int amount;
34  
35      public MageDrawAction(Player player, int amount, GameEvent originalDrawEvent) {
36          this.player = player;
37          this.amount = amount;
38          this.drawnCards = new ArrayList<>();
39          this.originalDrawEvent = originalDrawEvent;
40      }
41  
42      /**
43       * Draw and set action score.
44       *
45       * @param source
46       * @param game     Game context.
47       * @return
48       */
49      @Override
50      public int doAction(Ability source, Game game) {
51          int numDrawn = 0;
52          int score = 0;
53          GameEvent event = new DrawCardsEvent(this.player.getId(), source, this.originalDrawEvent, this.amount);
54          // TODO: This needs a better description of how it works. Why "amount < 2"?
55          if (amount < 2 || !game.replaceEvent(event)) {
56              amount = event.getAmount();
57              for (int i = 0; i < amount; i++) {
58                  int value = drawCard(source, this.originalDrawEvent, game);
59                  if (value == NEGATIVE_VALUE) {
60                      continue;
61                  }
62                  numDrawn++;
63                  score += value;
64              }
65              if (!player.isTopCardRevealed() && numDrawn > 0) {
66                  game.fireInformEvent(player.getLogName() + " draws " + CardUtil.numberToText(numDrawn, "a") + " card" + (numDrawn > 1 ? "s" : ""));
67              }
68              setScore(player, score);
69          }
70          return numDrawn;
71      }
72  
73      /**
74       * Draw a card if possible (there is no replacement effect that prevent us
75       * from drawing). Fire event about card drawn.
76       *
77       * @param source
78       * @param originalDrawEvent original draw event for replacement effects, can be null for normal calls
79       * @param game
80       * @return
81       */
82      protected int drawCard(Ability source, GameEvent originalDrawEvent, Game game) {
83          GameEvent event = new DrawCardEvent(this.player.getId(), source, originalDrawEvent);
84          if (!game.replaceEvent(event)) {
85              Card card = player.getLibrary().removeFromTop(game);
86              if (card != null) {
87                  drawnCards.add(card);
88                  card.moveToZone(Zone.HAND, source, game, false); // if you want to use event.getSourceId() here then thinks x10 times
89                  if (player.isTopCardRevealed()) {
90                      game.fireInformEvent(player.getLogName() + " draws a revealed card  (" + card.getLogName() + ')');
91                  }
92  
93                  game.fireEvent(new DrewCardEvent(card.getId(), player.getId(), source, originalDrawEvent));
94                  return ArtificialScoringSystem.inst.getCardScore(card);
95              }
96          }
97          return NEGATIVE_VALUE;
98      }
99  
100     /**
101      * Return a card back to top.
102      *
103      * @param game Game context
104      */
105     @Override
106     public void undoAction(Game game) {
107         for (int index = drawnCards.size() - 1; index >= 0; index--) {
108             Card card = drawnCards.get(index);
109             player.getHand().remove(card);
110             player.getLibrary().putOnTop(card, game);
111         }
112     }
113 }

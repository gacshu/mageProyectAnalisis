443     /**
444      * Subtracts the passed in mana values from this instance. The difference
445      * between this and {@code subtract()} is that if we do not have the
446      * available generic mana to pay, we take mana from our colored mana pools.
447      *
448      * @param mana mana values to subtract
449      * @throws ArithmeticException thrown if there is not enough available
450      *                             mana to pay the generic cost
451      */
452     public void subtractCost(final Mana mana) throws ArithmeticException {
453         this.subtract(mana);
454 
455         // Handle any unpaid generic mana costs
456         while (generic < 0) {
457             int oldColorless = generic;
458             if (white > 0) {
459                 white = CardUtil.overflowDec(white, 1);
460                 generic = CardUtil.overflowInc(generic, 1);
461                 continue;
462             }
463             if (blue > 0) {
464                 blue = CardUtil.overflowDec(blue, 1);
465                 generic = CardUtil.overflowInc(generic, 1);
466                 continue;
467             }
468             if (black > 0) {
469                 black = CardUtil.overflowDec(black, 1);
470                 generic = CardUtil.overflowInc(generic, 1);
471                 continue;
472             }
473             if (red > 0) {
474                 red = CardUtil.overflowDec(red, 1);
475                 generic = CardUtil.overflowInc(generic, 1);
476                 continue;
477             }
478             if (green > 0) {
479                 green = CardUtil.overflowDec(green, 1);
480                 generic = CardUtil.overflowInc(generic, 1);
481                 continue;
482             }
483             if (colorless > 0) {
484                 colorless = CardUtil.overflowDec(colorless, 1);
485                 generic = CardUtil.overflowInc(generic, 1);
486                 continue;
487             }
488             if (any > 0) {
489                 any = CardUtil.overflowDec(any, 1);
490                 generic = CardUtil.overflowInc(generic, 1);
491                 continue;
492             }
493             if (oldColorless == generic) {
494                 throw new ArithmeticException("Not enough mana to pay colorless");
495             }
496         }
497     }

527     /**
528      * Returns the count of filtered mana provided by the passed in
529      * {@link FilterMana}. If {@link FilterMana} is null, the total mana count
530      * is returned via {@link #count() count}.
531      *
532      * @param filter the colors of mana to return the count for.
533      * @return the count of filtered mana provided by the passed in
534      * {@link FilterMana}.
535      */
536     public int count(final FilterMana filter) {
537         if (filter == null) {
538             return count();
539         }
540         int count = 0;
541         if (filter.isWhite()) {
542             count = CardUtil.overflowInc(count, white);
543         }
544         if (filter.isBlue()) {
545             count = CardUtil.overflowInc(count, blue);
546         }
547         if (filter.isBlack()) {
548             count = CardUtil.overflowInc(count, black);
549         }
550         if (filter.isRed()) {
551             count = CardUtil.overflowInc(count, red);
552         }
553         if (filter.isGreen()) {
554             count = CardUtil.overflowInc(count, green);
555         }
556         if (filter.isGeneric()) {
557             count = CardUtil.overflowInc(count, generic);
558         }
559         if (filter.isColorless()) {
560             count = CardUtil.overflowInc(count, colorless);
561         }
562         return count;

641     /**
642      * Returns this objects values as a {@link String}.
643      *
644      * @return this objects values as a {@link String}.
645      */
646     @Override
647     public String toString() {
648         StringBuilder sbMana = new StringBuilder();
649 
650         if (generic > 0) {
651             sbMana.append('{').append(generic).append('}');
652         }
653 
654         // normal mana
655         if (colorless < 20) {
656             for (int i = 0; i < colorless; i++) {
657                 sbMana.append("{C}");
658             }
659         } else {
660             sbMana.append(colorless).append("{C}");
661         }
662 
663         String colorsInOrder = getColorsInOrder();
664         for (char c : colorsInOrder.toCharArray()) {
665             int amount = colorCharToAmount(c);
666             if (amount < 20) {
667                 for (int i = 0; i < amount; i++) {
668                     sbMana.append('{').append(c).append('}');
669                 }
670             } else {
671                 sbMana.append(amount).append('{').append(c).append('}');
672             }
673         }
674 
675         if (any < 20) {
676             for (int i = 0; i < any; i++) {
677                 sbMana.append("{Any}");
678             }
679         } else {
680             sbMana.append(any).append("{Any}");
681         }
682 
683         return sbMana.toString();
684     }

696     /**
697      * Returns if the cost (this) can be paid by the mana provided by the passed in {@link Mana} object.
698      *
699      * @param avail The mana to compare too.
700      * @return      boolean indicating if there is enough available mana to pay.
701      */
702     public boolean enough(final Mana avail) {
703         Mana compare = avail.copy();
704 
705         // Subtract the mana cost (this) from the mana available (compare).
706         // This will only subtract like mana types from one another (e.g. green from green, coloreless from colorless).
707         compare.subtract(this);
708 
709         // A negative value for compare.X means that mana of type X from the cost could not be paid by mana
710         // of the same kind from the available mana.
711         // Check each of the types, and see if there is enough mana of any color left to pay for the colors.
712 
713         if (compare.colorless < 0) { // Put first to shortcut the calculations
714             // Colorless mana can only be paid by colorless mana.
715             // If there's a negative value, then there's nothing else that can be used to pay for it.
716             return false;
717         }
718         if (compare.white < 0) {
719             compare.any = CardUtil.overflowInc(compare.any, compare.white);
720             // A negatice value means that there was more mana of the given type required than there was mana of any
721             // color to pay for it.
722             // So, there is not enough mana to pay the avail.
723             if (compare.any < 0) {
724                 return false;
725             }
726             compare.white = 0;
727         }
728         if (compare.blue < 0) {
729             compare.any = CardUtil.overflowInc(compare.any, compare.blue);
730             if (compare.any < 0) {
731                 return false;
732             }
733             compare.blue = 0;
734         }
735         if (compare.black < 0) {
736             compare.any = CardUtil.overflowInc(compare.any, compare.black);
737             if (compare.any < 0) {
738                 return false;
739             }
740             compare.black = 0;
741         }
742         if (compare.red < 0) {
743             compare.any = CardUtil.overflowInc(compare.any, compare.red);
744             if (compare.any < 0) {
745                 return false;
746             }
747             compare.red = 0;
748         }
749         if (compare.green < 0) {
750             compare.any = CardUtil.overflowInc(compare.any, compare.green);
751             if (compare.any < 0) {
752                 return false;
753             }
754             compare.green = 0;
755         }
756         if (compare.generic < 0) {
757             compare.generic = CardUtil.overflowInc(compare.generic, compare.white);
758             compare.generic = CardUtil.overflowInc(compare.generic, compare.blue);
759             compare.generic = CardUtil.overflowInc(compare.generic, compare.black);
760             compare.generic = CardUtil.overflowInc(compare.generic, compare.red);
761             compare.generic = CardUtil.overflowInc(compare.generic, compare.green);
762             compare.generic = CardUtil.overflowInc(compare.generic, compare.colorless);
763             compare.generic = CardUtil.overflowInc(compare.generic, compare.any);
764             return compare.generic >= 0;
765         }
766         return true;
767     }

769     /**
770      * Returns the total mana needed to meet the cost of this given the available mana passed in
771      * as a {@link Mana} object.
772      *
773      * Used by the AI to calculate what mana it needs to obtain for a spell to become playable.
774      *
775      * @param avail the mana available to pay the cost
776      * @return      the total mana needed to pay this given the available mana passed in as a {@link Mana} object.
777      */
778     public Mana needed(final Mana avail) {
779         Mana compare = avail.copy();
780 
781         // Subtract the mana cost (this) from the mana available (compare).
782         // This will only subtract like mana types from one another (e.g. green from green, coloreless from colorless).
783         compare.subtract(this);
784 
785         // A negative value for compare.X means that mana of type X from the cost could not be paid by mana
786         // of the same kind from the available mana (this).
787         if (compare.white < 0 && compare.any > 0) {
788             // Calculate how much of the unpaid colored mana can be covered by mana of any color
789             int diff = Math.min(compare.any, Math.abs(compare.white));
790             // Make the payment
791             compare.any = CardUtil.overflowDec(compare.any, diff);
792             compare.white = CardUtil.overflowInc(compare.white, diff);
793         }
794         if (compare.blue < 0 && compare.any > 0) {
795             int diff = Math.min(compare.any, Math.abs(compare.blue));
796             compare.any = CardUtil.overflowDec(compare.any, diff);
797             compare.blue = CardUtil.overflowInc(compare.blue, diff);
798         }
799         if (compare.black < 0 && compare.any > 0) {
800             int diff = Math.min(compare.any, Math.abs(compare.black));
801             compare.any = CardUtil.overflowDec(compare.any, diff);
802             compare.black = CardUtil.overflowInc(compare.black, diff);
803         }
804         if (compare.red < 0 && compare.any > 0) {
805             int diff = Math.min(compare.any, Math.abs(compare.red));
806             compare.any = CardUtil.overflowDec(compare.any, diff);
807             compare.red = CardUtil.overflowInc(compare.red, diff);
808         }
809         if (compare.green < 0 && compare.any > 0) {
810             int diff = Math.min(compare.any, Math.abs(compare.green));
811             compare.any = CardUtil.overflowDec(compare.any, diff);
812             compare.green = CardUtil.overflowInc(compare.green, diff);
813         }
814 
815         // Colorless mana can only be paid by colorless sources, so a check for it is not performed.
816 
817         if (compare.generic < 0) {
818             // Calculate total leftover mana available to pay for generic costs
819             int remaining = 0;
820             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.white));
821             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.blue));
822             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.black));
823             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.red));
824             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.green));
825             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.colorless));
826             remaining = CardUtil.overflowInc(remaining, Math.max(0, compare.any));
827 
828             if (remaining > 0) {
829                 // Calculate how much of the unpaid generic cost can be paid by the leftover mana
830                 int diff = Math.min(remaining, Math.abs(compare.generic));
831                 // Make the payment
832                 compare.generic = CardUtil.overflowInc(compare.generic, diff);
833             }
834         }
835 
836         // Create the mana object holding the mana needed to pay the cost
837         // If the value in compare is positive it means that there's excess mana of that type, and no more is needed.
838         Mana needed = new Mana();
839         if (compare.white < 0) {
840             needed.white = CardUtil.overflowDec(needed.white, compare.white);
841         }
842         if (compare.blue < 0) {
843             needed.blue = CardUtil.overflowDec(needed.blue, compare.blue);
844         }
845         if (compare.black < 0) {
846             needed.black = CardUtil.overflowDec(needed.black, compare.black);
847         }
848         if (compare.red < 0) {
849             needed.red = CardUtil.overflowDec(needed.red, compare.red);
850         }
851         if (compare.green < 0) {
852             needed.green = CardUtil.overflowDec(needed.green, compare.green);
853         }
854         if (compare.colorless < 0) {
855             needed.colorless = CardUtil.overflowDec(needed.colorless, compare.colorless);
856         }
857         if (compare.generic < 0) {
858             needed.generic = CardUtil.overflowDec(needed.generic, compare.generic);
859         }
860         return needed;
861     }

1237     /**
1238      * Returns the mana that is more colored or has a greater amount but does
1239      * not contain one less mana in any type but generic.
1240      * <p>
1241      * See tests ManaTest.moreValuableManaTest for several examples
1242      *
1243      * Examples:
1244      *      {1}       and {R}       -> {R}
1245      *      {2}       and {1}{W}    -> {1}{W}
1246      *      {3}       and {1}{W}    -> {1}{W}
1247      *      {1}{W}{R} and {G}{W}{R} -> {G}{W}{R}
1248      *      {G}{W}{R} and {G}{W}{R} -> null
1249      *      {G}{W}{B} and {G}{W}{R} -> null
1250      *      {C}       and {ANY}     -> null
1251      *
1252      * @param mana1     The 1st mana to compare.
1253      * @param mana2     The 2nd mana to compare.
1254      * @return          The greater of the two manas, or null if they're the same OR they cannot be compared
1255      */
1256     public static Mana getMoreValuableMana(final Mana mana1, final Mana mana2) {
1257         if (mana1.equals(mana2)) {
1258             return null;
1259         }
1260 
1261         boolean mana1IsConditional = mana1 instanceof ConditionalMana;
1262         boolean mana2IsConditional = mana2 instanceof ConditionalMana;
1263         if (mana1IsConditional != mana2IsConditional) {
1264             return null;
1265         }
1266 
1267         if (mana1IsConditional) {
1268             List<Condition> conditions1 = ((ConditionalMana) mana1).getConditions();
1269             List<Condition> conditions2 = ((ConditionalMana) mana2).getConditions();
1270             if (!Objects.equals(conditions1, conditions2)) {
1271                 return null;
1272             }
1273         }
1274 
1275         // Set one mana as moreMana and one as lessMana.
1276         Mana moreMana;
1277         Mana lessMana;
1278         if (mana2.any > mana1.any
1279                 || mana2.colorless > mana1.colorless
1280                 || mana2.countColored() > mana1.countColored()
1281                 || (mana2.countColored() == mana1.countColored()
1282                         && mana2.colorless == mana1.colorless
1283                         && mana2.count() > mana1.count())) {
1284             moreMana = mana2;
1285             lessMana = mana1;
1286         } else {
1287             moreMana = mana1;
1288             lessMana = mana2;
1289         }
1290 
1291         if (lessMana.any > moreMana.any) {
1292             return null;
1293         }
1294         if (lessMana.colorless > moreMana.colorless) {
1295             return null; // Any (color) can't produce colorless mana
1296         }
1297 
1298         int anyDiff = CardUtil.overflowDec(moreMana.any, lessMana.any);
1299 
1300         int whiteDiff = CardUtil.overflowDec(lessMana.white, moreMana.white);
1301         if (whiteDiff > 0) {
1302             anyDiff = CardUtil.overflowDec(anyDiff, whiteDiff);
1303             if (anyDiff < 0) {
1304                 return null;
1305             }
1306         }
1307 
1308         int redDiff = CardUtil.overflowDec(lessMana.red, moreMana.red);
1309         if (redDiff > 0) {
1310             anyDiff = CardUtil.overflowDec(anyDiff, redDiff);
1311             if (anyDiff < 0) {
1312                 return null;
1313             }
1314         }
1315 
1316         int greenDiff = CardUtil.overflowDec(lessMana.green, moreMana.green);
1317         if (greenDiff > 0) {
1318             anyDiff = CardUtil.overflowDec(anyDiff, greenDiff);
1319             if (anyDiff < 0) {
1320                 return null;
1321             }
1322         }
1323 
1324         int blueDiff = CardUtil.overflowDec(lessMana.blue, moreMana.blue);
1325         if (blueDiff > 0) {
1326             anyDiff = CardUtil.overflowDec(anyDiff, blueDiff);
1327             if (anyDiff < 0) {
1328                 return null;
1329             }
1330         }
1331 
1332         int blackDiff = CardUtil.overflowDec(lessMana.black, moreMana.black);
1333         if (blackDiff > 0) {
1334             anyDiff = CardUtil.overflowDec(anyDiff, blackDiff);
1335             if (anyDiff < 0) {
1336                 return null;
1337             }
1338         }
1339 
1340         return moreMana;
1341     }

1386     /**

1388      * called thousands of times by {@link mage.abilities.mana.ManaOptions#addManaWithCost(List, Game)}

1401         result = 31 * result + generic;
1402         result = 31 * result + colorless;
1403         result = 31 * result + any;
1404         result = 31 * result + (flag ? 1 : 0);
1405 
1406         return Long.hashCode(result);
1407     }

1409     /**
1410      * Checks that the {@code value} passed in is not less than 0. If the value
1411      * is negative, it is logged and 0 is returned.
1412      *
1413      * @param value the value to check.
1414      * @param name  the name of the value to check. Used to make logging of the
1415      *              {@code value} easier
1416      * @return the {@code value} passed in, unless it is minus, in which case 0
1417      * is returned.
1418      */
1419     private static int notNegative(int value, final String name) {
1420         if (value < 0) {
1421             logger.info(name + " can not be less than 0. Passed in: " + value + " Defaulting to 0.");
1422             value = 0;
1423         }

5   public final class MtgJsonCard {
6       // v5 support
7       // https://mtgjson.com/data-models/card-atomic/
8       // contains only used fields, if you need more for tests then just add it here
9   
10      public String name;
11      public String asciiName; // mtgjson uses it for some cards like El-Hajjaj
12      public String number; // from sets source only, see https://mtgjson.com/data-models/card/
13  
14      public String faceName;
15      public String side;
16  
17      public String manaCost;
18      public List<String> colorIdentity;
19      public List<String> colors;
20  
21      public List<String> supertypes;
22      public List<String> types;
23      public List<String> subtypes;
24  
25      public String text; // rules splits by \n
26  
27      public String loyalty;
28      public String power;
29      public String toughness;
30  
31      public Integer edhrecRank;
32      public String layout;
33      public boolean isFullArt;
34      public List<String> printings; // set codes with that card
35  
36      @Override
37      public String toString() {
38          return number + " - " + this.getNameAsFull()
39                  + (this.getNameAsFull().equals(this.getNameAsFace()) ? "" : String.format(" (face: %s)", this.getNameAsFace()));
40      }
41  
42      /**
43       *
44       * @return single side name like Ice from Fire // Ice
45       */
46      public String getNameAsFace() {
47          // return single side name
48          return faceName != null ? faceName : (asciiName != null ? asciiName : name);
49      }
50  
51      /**
52       *
53       * @return full card name like Fire // Ice
54       */
55      public String getNameAsFull() {
56          // xmage split a double faced card to two different cards, but mtgjson/scryfall uses full name,
57          // so use faceName property for full name searching
58          if ("transform".equals(layout)
59                  || "flip".equals(layout)
60                  || "adventure".equals(layout)
61                  || "modal_dfc".equals(layout)
62                  || "reversible_card".equals(layout) // reversible_card - example: Zndrsplt, Eye of Wisdom
63                  || "meld".equals(layout)) { // meld - mtgjson uses composite names for meld cards, but scryfall uses simple face names
64              return getNameAsFace();
65          }
66  
67          return asciiName != null ? asciiName : name;
68      }
69  }

3   import mage.MageObject;
4   import mage.constants.SubType;

6   import java.util.LinkedHashSet;
7   import java.util.stream.Collectors;
8   
9   public class ChoiceCreatureType extends ChoiceImpl {
10  
11      private static final String DEFAULT_MESSAGE = "Choose a creature type";
12  
13      public ChoiceCreatureType() {
14          this(true, DEFAULT_MESSAGE, null);
15      }
16  
17      public ChoiceCreatureType(MageObject source) {
18          this(true, DEFAULT_MESSAGE, source);
19      }
20  
21      public ChoiceCreatureType(String chooseMessage, MageObject source) {
22          this(true, chooseMessage, source);
23      }
24  
25      public ChoiceCreatureType(boolean required, String chooseMessage, MageObject source) {
26          super(required);
27          this.setChoices(SubType.getCreatureTypes().stream().map(SubType::toString).collect(Collectors.toCollection(LinkedHashSet::new)));
28          this.setMessage(chooseMessage);
29          if (source != null) {
30              this.setSubMessage(source.getIdName());
31          }
32          this.setSearchEnabled(true);
33      }
34  
35      public ChoiceCreatureType(final ChoiceCreatureType choice) {
36          super(choice);
37      }
38  
39      @Override
40      public ChoiceCreatureType copy() {
41          return new ChoiceCreatureType(this);
42      }
43  }

5   import java.net.URL;
6   import java.time.ZoneOffset;
7   import java.time.format.DateTimeFormatter;
8   import java.time.temporal.TemporalAccessor;
9   import java.util.jar.Attributes;
10  import java.util.jar.Manifest;
11  
12  /**
13   * @author JayDi85
14   */
15  public class JarVersion {
16  
17      private static final Logger logger = Logger.getLogger(JarVersion.class);
18      private static final String JAR_BUILD_TIME_FROM_CLASSES = "runtime";
19      private static final String JAR_BUILD_TIME_ERROR = "n/a";
20  
21      public static String getBuildTime(Class clazz) {
22          // build time info inserted by maven on jar build phase (see root pom.xml)
23          String resultFormat = "uuuu-MM-dd HH:mm";
24          String className = clazz.getSimpleName() + ".class";
25          String classPath = clazz.getResource(className).toString();
26  
27          // https://stackoverflow.com/a/1273432/1276632
28          String manifestPath;
29          if (classPath.startsWith("jar")) {
30              // jar source
31              manifestPath = classPath.substring(0, classPath.lastIndexOf('!') + 1) + "/META-INF/MANIFEST.MF";
32          } else {
33              // dir source (e.g. IDE's debug)
34              // it's can be generated by runtime, but need extra code and performance: https://stackoverflow.com/questions/34674073/how-to-generate-manifest-mf-file-during-compile-phase
35              // manifestPath = classPath.substring(0, classPath.lastIndexOf("/" + className)) + "/META-INF/MANIFEST.MF";
36              return JAR_BUILD_TIME_FROM_CLASSES;
37          }
38  
39          try {
40              Manifest manifest = new Manifest(new URL(manifestPath).openStream());
41              Attributes attr = manifest.getMainAttributes();
42              String buildTime = attr.getValue("Build-Time");
43              // default maven format: yyyy-MM-dd'T'HH:mm:ss'Z' or see maven.build.timestamp.format in pom file
44              DateTimeFormatter sourceFormatter = DateTimeFormatter.ofPattern("uuuu-MM-dd'T'HH:mm:ss'Z'").withZone(ZoneOffset.UTC);
45              TemporalAccessor ta = sourceFormatter.parse(buildTime);
46              DateTimeFormatter resultFormatter = DateTimeFormatter.ofPattern(resultFormat).withZone(ZoneOffset.UTC);
47              return resultFormatter.format(ta);
48          } catch (Throwable e) {
49              logger.error("Can't read build time in jar manifest for class " + clazz.getName() + " and path " + manifestPath, e);
50              return JAR_BUILD_TIME_ERROR;
51          }
52      }
53  
54      public static boolean isBuildTimeOk(String buildTime) {
55          return buildTime != null
56                  && !buildTime.isEmpty()
57                  && !buildTime.equals(JAR_BUILD_TIME_ERROR)
58                  && !buildTime.equals(JAR_BUILD_TIME_FROM_CLASSES);
59      }
60  }

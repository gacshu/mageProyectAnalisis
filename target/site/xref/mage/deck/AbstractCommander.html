50      protected boolean checkCommander(Card commander, Set<Card> commanders) {
51          return commander.hasCardTypeForDeckbuilding(CardType.CREATURE) && commander.isLegendary()
52                  || commander.getAbilities().contains(CanBeYourCommanderAbility.getInstance())
53                  || commander.hasSubTypeForDeckbuilding(SubType.BACKGROUND) && commanders.size() == 2;
54      }

56      protected boolean checkPartners(Set<Card> commanders) {
57          switch (commanders.size()) {
58              case 1:
59                  return true;
60              case 2:
61                  break;
62              default:
63                  return false;
64          }
65          Iterator<Card> iter = commanders.iterator();
66          Card commander1 = iter.next();
67          Card commander2 = iter.next();
68          if (commander1.getAbilities().containsClass(PartnerAbility.class)
69                  && commander2.getAbilities().containsClass(PartnerAbility.class)
70                  || commander1.getAbilities().containsClass(FriendsForeverAbility.class)
71                  && commander2.getAbilities().containsClass(FriendsForeverAbility.class)) {
72              return true;
73          }
74          if (commander1.getAbilities().containsClass(PartnerWithAbility.class)
75                  && commander2.getAbilities().containsClass(PartnerWithAbility.class)) {
76              String name1 = CardUtil.castStream(commander2.getAbilities().stream(), PartnerWithAbility.class).map(PartnerWithAbility::getPartnerName).findFirst().orElse(null);
77              String name2 = CardUtil.castStream(commander1.getAbilities().stream(), PartnerWithAbility.class).map(PartnerWithAbility::getPartnerName).findFirst().orElse(null);
78              if (commander1.getName().equals(name1) && commander2.getName().equals(name2)) {
79                  return true;
80              }
81              addError(DeckValidatorErrorType.PRIMARY, commander1.getName(), "Commander with invalid Partner (" + commander1.getName() + ')', true);
82              addError(DeckValidatorErrorType.PRIMARY, commander2.getName(), "Commander with invalid Partner (" + commander2.getName() + ')', true);
83              return false;
84          }
85          if (commander1.getAbilities().containsClass(ChooseABackgroundAbility.class) == commander2.hasSubTypeForDeckbuilding(SubType.BACKGROUND)
86                  && commander2.getAbilities().containsClass(ChooseABackgroundAbility.class) == commander1.hasSubTypeForDeckbuilding(SubType.BACKGROUND)) {
87              return true;
88          }
89          if (commander1.hasSubTypeForDeckbuilding(SubType.BACKGROUND)) {
90              addError(DeckValidatorErrorType.PRIMARY, commander1.getName(), "Background without valid Commander (" + commander1.getName() + ')', true);
91          }
92          if (commander2.hasSubTypeForDeckbuilding(SubType.BACKGROUND)) {
93              addError(DeckValidatorErrorType.PRIMARY, commander1.getName(), "Background without valid Commander (" + commander2.getName() + ')', true);
94          }
95          return false;
96      }

98      private boolean checkColorIdentity(Deck deck, FilterMana colorIdentity, Set<Card> commanders) {
99          int piperCount = commanders
100                 .stream()
101                 .filter(CommanderChooseColorAbility::checkCard)
102                 .mapToInt(x -> 1)
103                 .sum();
104         if (piperCount == 0) {
105             boolean valid = true;
106             for (Card card : deck.getCards()) {
107                 if (!ManaUtil.isColorIdentityCompatible(colorIdentity, card.getColorIdentity())) {
108                     addError(DeckValidatorErrorType.OTHER, card.getName(), "Invalid color (" + colorIdentity.toString() + ')', true);
109                     valid = false;
110                 }
111             }
112             for (Card card : deck.getSideboard()) {
113                 if (!ManaUtil.isColorIdentityCompatible(colorIdentity, card.getColorIdentity())) {
114                     addError(DeckValidatorErrorType.OTHER, card.getName(), "Invalid color (" + colorIdentity.toString() + ')', true);
115                     valid = false;
116                 }
117             }
118             return valid;
119         }
120         FilterMana filterMana = new FilterMana();
121         Stream.concat(
122                 deck.getCards().stream(),
123                 deck.getSideboard().stream()
124         ).map(Card::getColorIdentity).forEach(filterMana::addAll);
125         if (colorIdentity.getColorCount() + piperCount >= filterMana.getColorCount()) {
126             return true;
127         }
128         StringBuilder sb = new StringBuilder()
129                 .append("Invalid color, commander color identity has ")
130                 .append(colorIdentity.getColorCount())
131                 .append(" color")
132                 .append(colorIdentity.getColorCount() > 1 ? "s" : "")
133                 .append(", plus ")
134                 .append(piperCount)
135                 .append(" cop")
136                 .append(piperCount > 1 ? "ies" : "y")
137                 .append(" of The Prismatic Piper, but the total amount of colors in the deck is ")
138                 .append(filterMana.getColorCount());
139         addError(DeckValidatorErrorType.OTHER, "The Prismatic Piper", sb.toString());
140         return false;
141     }

144     public boolean validate(Deck deck) {

157             case 1:
158                 companion = null;
159                 commanders.add(deck.getSideboard().iterator().next());
160                 break;
161             case 2:
162                 iter = deck.getSideboard().iterator();
163                 card1 = iter.next();
164                 card2 = iter.next();
165                 if (card1.getAbilities().stream().anyMatch(CompanionAbility.class::isInstance)) {
166                     companion = card1;
167                     commanders.add(card2);
168                 } else if (card2.getAbilities().stream().anyMatch(CompanionAbility.class::isInstance)) {
169                     companion = card2;
170                     commanders.add(card1);
171                 } else {
172                     companion = null;
173                     commanders.add(card1);
174                     commanders.add(card2);
175                 }
176                 break;
177             case 3:
178                 iter = deck.getSideboard().iterator();
179                 card1 = iter.next();
180                 card2 = iter.next();
181                 card3 = iter.next();
182                 if (card1.getAbilities().stream().anyMatch(CompanionAbility.class::isInstance)) {
183                     companion = card1;
184                     commanders.add(card2);
185                     commanders.add(card3);
186                 } else if (card2.getAbilities().stream().anyMatch(CompanionAbility.class::isInstance)) {
187                     companion = card2;
188                     commanders.add(card1);
189                     commanders.add(card3);
190                 } else if (card3.getAbilities().stream().anyMatch(CompanionAbility.class::isInstance)) {
191                     companion = card3;
192                     commanders.add(card1);
193                     commanders.add(card2);
194                 } else {
195                     companion = null;
196                     addError(DeckValidatorErrorType.PRIMARY, "Commander", "Sideboard must contain only the commander(s) and up to 1 companion");
197                     valid = false;
198                 }
199                 break;
200             default:
201                 companion = null;
202                 addError(DeckValidatorErrorType.PRIMARY, "Commander", "Sideboard must contain only the commander(s) and up to 1 companion");
203                 valid = false;
204         }
205 
206         if (companion != null && deck.getCards().size() + deck.getSideboard().size() != 101) {
207             addError(DeckValidatorErrorType.DECK_SIZE, "Deck", "Must contain " + 101 + " cards (companion doesn't count for deck size): has " + (deck.getCards().size() + deck.getSideboard().size()) + " cards");
208             valid = false;
209         } else if (companion == null && deck.getCards().size() + deck.getSideboard().size() != 100) {
210             addError(DeckValidatorErrorType.DECK_SIZE, "Deck", "Must contain " + 100 + " cards: has " + (deck.getCards().size() + deck.getSideboard().size()) + " cards");
211             valid = false;
212         }
213 
214         Map<String, Integer> counts = new HashMap<>();
215         countCards(counts, deck.getCards());
216         countCards(counts, deck.getSideboard());
217         valid = checkCounts(1, counts) && valid;
218         valid = checkBanned(counts) && valid;
219         valid = checkPartners(commanders) && valid;
220 
221         for (Card commander : commanders) {
222             if (bannedCommander.contains(commander.getName())) {
223                 addError(DeckValidatorErrorType.PRIMARY, commander.getName(), "Commander banned (" + commander.getName() + ')', true);
224                 valid = false;
225             }
226             if (!checkCommander(commander, commanders)) {
227                 addError(DeckValidatorErrorType.PRIMARY, commander.getName(), "Commander invalid (" + commander.getName() + ')', true);
228                 valid = false;
229             }
230             if (commanders.size() == 2 && bannedPartner.contains(commander.getName())) {
231                 addError(DeckValidatorErrorType.PRIMARY, commander.getName(), "Commander Partner banned (" + commander.getName() + ')', true);
232                 valid = false;
233             }
234             ManaUtil.collectColorIdentity(colorIdentity, commander.getColorIdentity());
235         }
236 
237         // no needs in cards check on wrong commanders
238         if (!valid) {
239             return false;
240         }
241 
242         valid = checkColorIdentity(deck, colorIdentity, commanders);
243 
244         for (Card card : deck.getCards()) {
245             if (!isSetAllowed(card.getExpansionSetCode())) {
246                 if (!legalSets(card)) {
247                     addError(DeckValidatorErrorType.WRONG_SET, card.getName(), "Not allowed Set: " + card.getExpansionSetCode(), true);
248                     valid = false;
249                 }
250             }
251         }
252         for (Card card : deck.getSideboard()) {
253             if (!isSetAllowed(card.getExpansionSetCode())) {
254                 if (!legalSets(card)) {
255                     addError(DeckValidatorErrorType.WRONG_SET, card.getName(), "Not allowed Set: " + card.getExpansionSetCode(), true);
256                     valid = false;
257                 }
258             }
259         }
260         // Check for companion legality
261         if (companion != null) {
262             Set<Card> cards = new HashSet<>(deck.getCards());
263             cards.addAll(commanders);
264             CompanionAbility companionAbility = CardUtil.castStream(
265                     companion.getAbilities().stream(),
266                     CompanionAbility.class
267             ).findFirst().orElse(null);
268             if (companionAbility == null || !companionAbility.isLegal(cards, getDeckMinSize())) {
269                 addError(DeckValidatorErrorType.PRIMARY, companion.getName(),
270                         String.format("Commander companion illegal: %s", companionAbility.getLegalRule()), true);
271                 valid = false;
272             }
273         }
274         return valid;
275     }

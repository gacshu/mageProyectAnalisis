17  public class Standard extends Constructed {
18  
19      public Standard() {
20          super("Constructed - Standard");
21  
22          setCodes.addAll(makeLegalSets());
23  
24          banned.add("The Meathook Massacre");
25      }
26  
27      private static boolean isFallSet(ExpansionSet set) {
28          Calendar cal = Calendar.getInstance();
29          cal.setTime(set.getReleaseDate());
30          // Fall sets are normally released during or after September and before November
31          return set.getSetType() == SetType.EXPANSION
32                  && Calendar.SEPTEMBER <= cal.get(Calendar.MONTH)
33                  && cal.get(Calendar.MONTH) < Calendar.NOVEMBER;
34      }
35  
36      static List<String> makeLegalSets() {
37          GregorianCalendar current = new GregorianCalendar();
38          // Get the second most recent fall set that's been released.
39          Date earliestDate = Sets
40                  .getInstance()
41                  .values()
42                  .stream()
43                  .filter(set -> !set.getReleaseDate().after(current.getTime()))
44                  .filter(Standard::isFallSet)
45                  .sorted(ExpansionSet.getComparator())
46                  .skip(1)
47                  .findFirst()
48                  .get()
49                  .getReleaseDate();
50          return Sets.getInstance()
51                  .values()
52                  .stream()
53                  .filter(set -> set.getSetType().isStandardLegal())
54                  .filter(set -> !set.getReleaseDate().before(earliestDate))
55  //                .filter(set -> !set.getReleaseDate().after(current.getTime())) // no after date restrict for early tests and beta
56                  .map(ExpansionSet::getCode)
57                  .collect(Collectors.toList());
58      }
59  }

61      protected Commander(String name) {
62          super(name);
63          for (ExpansionSet set : Sets.getInstance().values()) {
64              if (set.getSetType().isEternalLegal()) {
65                  setCodes.add(set.getCode());
66              }
67          }
68      }
69  
70      @Override
71      protected boolean checkBanned(Map<String, Integer> counts) {
72          boolean valid = true;
73          for (String bannedCard : banned) {
74              if (counts.containsKey(bannedCard)) {
75                  addError(DeckValidatorErrorType.BANNED, bannedCard, "Banned", true);
76                  valid = false;
77              }
78          }
79          return valid;
80      }

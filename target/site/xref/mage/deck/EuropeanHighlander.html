12  /**
13   * @author ddzn
14   * @brief European Highlander is a classic competitive singleton format formalized around 2003.
15   * It is also known as 'German Highlander'.
16   * In short: 20 pts starting life, 100 card singleton deck without sideboard.
17   * No command zone, so companions can only be played in the main deck.
18   * There is a ban list but no points system.
19   * More info at http://highlandermagic.info
20   * Banned list here http://highlandermagic.info/banned-list
21   * Watch list here http://highlandermagic.info/watch-list
22   */
23  public class EuropeanHighlander extends Constructed {

25      public EuropeanHighlander() {
26          super("European Highlander", "EuroLander");
27          for (ExpansionSet set : Sets.getInstance().values()) {
28              if (set.getSetType().isEternalLegal()) {

69      @Override
70      public int getDeckMinSize() {
71          return 100;
72      }
73  
74      @Override
75      public boolean validate(Deck deck) {
76          errorsList.clear();
77  
78          // Parent class checks the banned list
79          boolean valid = super.validate(deck);
80  
81          if (deck.getCards().size() < 100) {
82              addError(DeckValidatorErrorType.DECK_SIZE, "Deck", "Must contain 100 or more singleton cards: has " + (deck.getCards().size()) + " cards");
83              valid = false;
84          }
85  
86          if (!deck.getSideboard().isEmpty()) {
87              addError(DeckValidatorErrorType.DECK_SIZE, "Sideboard", "Sideboard can't contain any cards: has " + (deck.getSideboard().size()) + " cards");
88              valid = false;
89          }
90  
91          // Count main deck card occurrences
92          // Note: Although a singleton format, the "any-number-of" cards are allowed
93          Map<String, Integer> counts = new HashMap<>();
94          countCards(counts, deck.getCards());
95          valid = checkCounts(1, counts) && valid;
96  
97          return valid;
98      }

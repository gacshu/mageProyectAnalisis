3   import mage.cards.Card;
4   import mage.cards.ExpansionSet;
5   import mage.cards.Sets;
6   import mage.cards.decks.DeckValidatorErrorType;
7   import mage.constants.CardType;
8   
9   import java.util.Map;
10  import java.util.Set;
11  
12  /**
13   * @author spjspj
14   */
15  public class FreeformCommander extends AbstractCommander {
16  
17      public FreeformCommander() {
18          super("Freeform Commander");
19          for (ExpansionSet set : Sets.getInstance().values()) {
20              setCodes.add(set.getCode());
21          }
22  
23          // no banned cards
24          this.banned.clear();
25      }
26  
27      public FreeformCommander(String name) {
28          super(name);
29      }
30  
31      @Override
32      protected boolean checkBanned(Map<String, Integer> counts) {
33          return true;
34      }
35  
36      @Override
37      protected boolean checkCommander(Card commander, Set<Card> commanders) {
38          if (!commander.hasCardTypeForDeckbuilding(CardType.CREATURE) && !commander.isLegendary()) {
39              addError(DeckValidatorErrorType.PRIMARY, commander.getName(), "For Freeform Commander, the commander must be a creature or be legendary. Yours was: " + commander.getName(), true);
40              return false;
41          }
42          return true;
43      }
44  }

3   import mage.cards.Card;
4   import mage.cards.decks.Deck;
5   import mage.cards.decks.DeckValidator;
6   import mage.cards.decks.DeckValidatorErrorType;
7   
8   import java.util.ArrayList;
9   import java.util.Arrays;
10  import java.util.List;
11  
12  /**
13   * @author nigelzor
14   */
15  public class Momir extends DeckValidator {
16  
17      public Momir() {
18          super("Momir Basic", "Momir");
19      }
20  
21      @Override
22      public int getDeckMinSize() {
23          return 60;
24      }
25  
26      @Override
27      public int getSideboardMinSize() {
28          return 0;
29      }
30  
31      @Override
32      public boolean validate(Deck deck) {
33          boolean valid = true;
34          errorsList.clear();
35  
36          if (deck.getCards().size() != getDeckMinSize()) {
37              addError(DeckValidatorErrorType.DECK_SIZE, "Deck", "Must contain " + getDeckMinSize() + " cards: has " + deck.getCards().size() + " cards");
38              valid = false;
39          }
40  
41          List<String> basicLandNames = new ArrayList<>(Arrays.asList("Forest", "Island", "Mountain", "Swamp", "Plains", "Wastes"));
42          for (Card card : deck.getCards()) {
43              if (!basicLandNames.contains(card.getName())) {
44                  addError(DeckValidatorErrorType.OTHER, card.getName(), "Only basic lands are allowed", true);
45                  valid = false;
46              }
47          }
48  
49          return valid;
50      }
51  
52  }

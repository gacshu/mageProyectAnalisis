55      /**
56       * Constructor. Don't need to mess with any of the sets yet; that will be
57       * done in the overridden validate function.
58       */
59      public SuperType2() {
60          super("Constructed - Super Type 2");

62          banned.add("Ancient Den");
63          banned.add("Disciple of the Vault");
64          banned.add("Great Furnace");
65          banned.add("Seat of the Synod");

71      /**
72       * Overridden validate function. Changes the standard sets, then uses the
73       * regular validation function to test validity.
74       *
75       * @param deck - the deck to validate.
76       * @return boolean if valid deck
77       */
78      @Override
79      public boolean validate(Deck deck) {
80  
81          List<DeckValidatorError> leastInvalid = null;
82  
83          boolean valid = false;
84          errorsList.clear();
85  
86          // first, check whether misty and batterskull are in the same deck.
87          Map<String, Integer> counts = new HashMap<>();
88          countCards(counts, deck.getCards());
89          countCards(counts, deck.getSideboard());
90          if (counts.containsKey("Stoneforge Mystic")
91                  && counts.containsKey("Batterskull")) {
92  
93              // if both, then skip all following tests by returning
94              return false;
95          }
96  
97          // up to Lorwyn/Alara, standards will have to be hard-coded.
98          // iterate through the array of standards.
99          for (String[] sets : standards) {
100 
101             // clear the invalid list
102             errorsList.clear();
103 
104             // add the sets to the setCodes.
105             setCodes = new ArrayList<>(Arrays.asList(sets));
106 
107             // if either of the mirrodin blocks are in the time period, ban
108             // misty and darksteel citadel
109             if (setCodes.contains("MRD") || setCodes.contains("SOM")) {
110                 banned.add("Darksteel Citadel");
111             } else {
112                 banned.remove("Darksteel Citadel");
113             }
114 
115             // validate it. If it validates, clear the invalid cards and break.
116             if (super.validate(deck)) {
117                 valid = true;
118                 break;
119             }
120 
121             // if the map holding the invalid cards is empty, set it to a
122             // copy of the current invalid list.
123             if (leastInvalid == null) {
124                 leastInvalid = new ArrayList<>(this.getErrorsList());
125                 continue;
126             }
127 
128             // see how many invalid cards there are. if there are less invalid
129             // cards than the stored invalid list, assign the current invalid
130             // to leastInvalid.
131             if (leastInvalid.size() > this.getErrorsList().size()) {
132                 leastInvalid = new ArrayList<>(this.getErrorsList());
133             }
134         }
135 
136         // After testing the first few standards, do the regular ones.
137         // set the initial starting and ending date, as well as the current.
138         GregorianCalendar start = new GregorianCalendar(2006,
139                 Calendar.SEPTEMBER, 1);
140         GregorianCalendar end = new GregorianCalendar(2008,
141                 Calendar.SEPTEMBER, 1);
142         GregorianCalendar current = new GregorianCalendar();
143 
144         // use the method for determining regular standard legality, but change
145         // the date for each standard.
146         while (end.before(current) && !valid) {
147 
148             // clear the invalid list and set codes.
149             setCodes.clear();
150             errorsList.clear();
151 
152             // increment the start and end dates.
153             start.set(Calendar.YEAR, start.get(Calendar.YEAR) + 1);
154             end.set(Calendar.YEAR, start.get(Calendar.YEAR) + 2);
155 
156             // Get the sets in that time period.
157             // (code taken from standard.java)
158             for (ExpansionSet set : Sets.getInstance().values()) {
159                 if (set.getSetType().isStandardLegal()
160                         && set.getReleaseDate().after(start.getTime())
161                         && set.getReleaseDate().before(end.getTime())) {
162                     setCodes.add(set.getCode());
163                 }
164             }
165 
166             // if either of the mirrodin blocks are in the time period, ban
167             // misty and darksteel citadel
168             if (setCodes.contains("MRD") || setCodes.contains("SOM")) {
169                 banned.add("Darksteel Citadel");
170             } else {
171                 banned.remove("Darksteel Citadel");
172             }
173 
174             // validate it. If it validates, clear the invalid cards and break.
175             if (super.validate(deck)) {
176                 errorsList.clear();
177                 valid = true;
178                 break;
179             }
180 
181             // see how many invalid cards there are. if there are less invalid
182             // cards than the stored invalid list, assign the current invalid
183             // to leastInvalid.
184             if (leastInvalid == null) {
185                 leastInvalid = new ArrayList<>(this.getErrorsList());
186 
187             } else if (leastInvalid.size() > this.getErrorsList().size()) {
188                 leastInvalid = new ArrayList<>(this.getErrorsList());
189             }
190         }
191 
192         // if no standard environment is valid, set the invalid to the
193         // invalid that had the least errors.
194         if (!valid) {
195             this.errorsList = new ArrayList<>(leastInvalid);
196         }
197 
198         // return the validity.
199         return valid;
200     }

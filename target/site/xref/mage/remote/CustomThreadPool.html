3   import java.lang.reflect.Field;
4   import java.util.concurrent.ThreadPoolExecutor;
5   
6   import org.apache.log4j.Logger;
7   import org.jboss.util.threadpool.BasicThreadPool;
8   
9   public class CustomThreadPool extends BasicThreadPool {
10      private static final Logger logger = Logger.getLogger(SessionImpl.class);
11  
12      @Override
13      public void setMaximumPoolSize(int size) {
14          /*
15           * I really don't want to implement a whole new threadpool
16           * just to fix this and the executor is private
17           */
18          try {
19              Field executorField = BasicThreadPool.class.getField("executor");
20              executorField.setAccessible(true);
21              ThreadPoolExecutor executor = (ThreadPoolExecutor) executorField.get(this);
22              synchronized (executor) {
23                  executor.setMaximumPoolSize(size);
24                  executor.setCorePoolSize(size);
25              }
26          } catch (NoSuchFieldException | SecurityException e) {
27              logger.error("Failed to get field executor from BasicThreadPool", e);
28          } catch (IllegalArgumentException | IllegalAccessException e) {
29              logger.error("Failed to get executor object from BasicThreadPool", e);
30          }
31      }
32  }

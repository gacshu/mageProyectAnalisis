82      // RemotingTask - do server side works in background and return result, can be canceled at any time
83      public abstract class RemotingTask {
84  
85          SwingWorker<Boolean, Object> worker = null;
86          Throwable lastError = null;
87  
88          abstract public boolean work() throws Throwable;
89  
90          boolean doWork() throws Throwable {
91              worker = new SwingWorker<Boolean, Object>() {
92                  @Override
93                  protected Boolean doInBackground() {
94                      try {
95                          return work();
96                      } catch (Throwable t) {
97                          lastError = t;
98                          return false;
99                      }
100                 }
101             };
102             worker.execute();
103 
104             boolean res = worker.get();
105             if (lastError != null) {
106                 throw lastError;
107             }
108             return res;
109         }
110 
111         public void cancel() {
112             if (worker != null) {
113                 worker.cancel(true);
114             }
115         }
116     }

125     private boolean doRemoteWorkAndHandleErrors(boolean closeConnectionOnFinish, boolean mustWaitServerMessageOnFail,
126                                                 RemotingTask remoting) {
127         // execute remote task and wait result, can be canceled
128         lastRemotingTask = remoting;
129         try {
130             boolean res = remoting.doWork();
131             if (!res && mustWaitServerMessageOnFail) {
132                 // server send detail error as separate message by existing connection,
133                 // so you need wait some time before disconnect
134                 try {
135                     Thread.sleep(3000);
136                 } catch (InterruptedException e) {
137                     logger.fatal("waiting of error message had failed", e);
138                     Thread.currentThread().interrupt();
139                 }
140             }
141             return res;
142         } catch (InterruptedException | CancellationException t) {
143             // was canceled by user, nothing to show
144         } catch (MalformedURLException ex) {
145             logger.fatal("Connect: wrong server address", ex);
146             showMessageToUser(ex.getMessage());
147         } catch (UndeclaredThrowableException ex) {
148             String addMessage = "";
149             Throwable cause = ex.getCause();
150             if (cause instanceof InvocationFailureException) {
151                 InvocationFailureException exep = (InvocationFailureException) cause;
152                 if (exep.getCause() instanceof IOException) {
153                     if ((exep.getCause().getMessage() != null) && (exep.getCause().getMessage().startsWith("Field hash null is not available on current")
154                             || exep.getCause().getMessage().endsWith("end of file"))) {
155                         addMessage = "Probably the server version is not compatible with the client. ";
156                     }
157                 } else {
158                     logger.error("Connect: unknown server error", exep.getCause());
159                 }
160             } else if (cause instanceof NoSuchMethodException) {
161                 // NoSuchMethodException is thrown on an invocation of an unknown JBoss remoting
162                 // method, so it's likely to be because of a version incompatibility.
163                 addMessage = "The following method is not available in the server, probably the "
164                         + "server version is not compatible with the client: " + cause.getMessage();
165             }
166             if (addMessage.isEmpty()) {
167                 logger.fatal("Connect: unknown error", ex);
168             }
169             showMessageToUser(addMessage + (ex.getMessage() != null ? ex.getMessage() : ""));
170         } catch (IOException ex) {
171             logger.fatal("Connect: unknown IO error", ex);
172             String addMessage = "";
173             if (ex.getMessage() != null && ex.getMessage().startsWith("Unable to perform invocation")) {
174                 addMessage = "Maybe the server version is not compatible. ";
175             }
176             showMessageToUser(addMessage + (ex.getMessage() != null ? ex.getMessage() : ""));
177         } catch (MageVersionException ex) {
178             logger.warn("Connect: wrong versions");
179             disconnect(false);
180             if (!canceled) {
181                 showMessageToUser(ex.getMessage());
182             }
183         } catch (CannotConnectException ex) {
184             if (!canceled) {
185                 handleCannotConnectException(ex);
186             }
187         } catch (Throwable t) {
188             logger.fatal("Connect: FAIL", t);
189             disconnect(false);
190             if (!canceled) {
191                 showMessageToUser(t.getMessage());
192             }
193         } finally {
194             lastRemotingTask = null;
195             if (closeConnectionOnFinish) {
196                 disconnect(false); // it's ok on mutiple calls
197             }
198         }
199         return false;
200     }

241     @Override
242     public synchronized boolean connect(final Connection connection) {
243         return doRemoteConnection(connection) && doRemoteWorkAndHandleErrors(false, true, new RemotingTask() {
244             @Override
245             public boolean work() throws Throwable {
246                 setLastError("");
247                 logger.info("Logging: as username " + getUserName() + " to server " + connection.getHost() + ':' + connection.getPort());
248                 boolean result;
249 
250                 if (connection.getAdminPassword() == null) {
251                     // for backward compatibility. don't remove twice call - first one does nothing but for version checking
252                     result = server.connectUser(connection.getUsername(), connection.getPassword(), sessionId, client.getVersion(), connection.getUserIdStr());
253                 } else {
254                     result = server.connectAdmin(connection.getAdminPassword(), sessionId, client.getVersion());
255                 }
256 
257                 if (result) {
258                     serverState = server.getServerState();
259 
260                     // client side check for incompatible versions
261                     if (client.getVersion().compareTo(serverState.getVersion()) != 0) {
262                         throw new MageVersionException(client.getVersion(), serverState.getVersion());
263                     }
264 
265                     if (!connection.getUsername().equals("Admin")) {
266                         server.setUserData(connection.getUsername(), sessionId, connection.getUserData(), client.getVersion().toString(), connection.getUserIdStr());
267                         updateDatabase(connection.isForceDBComparison(), serverState);
268                     }
269 
270                     logger.info("Logging: DONE");
271                     client.connected(getUserName() + '@' + connection.getHost() + ':' + connection.getPort() + ' ');
272                     return true;
273                 }
274 
275                 logger.info("Logging: FAIL");
276                 return false;
277             }
278         });

295     private boolean doRemoteConnection(final Connection connection) {
296         // connect to server and setup all data, can be canceled
297         if (isConnected()) {
298             disconnect(true);
299         }
300         this.connection = connection;
301         this.canceled = false;
302         sessionState = SessionState.CONNECTING;
303         lastRemotingTask = new RemotingTask() {
304             @Override
305             public boolean work() throws Throwable {
306                 logger.info("Connect: connecting to server " + connection.getHost() + ':' + connection.getPort());
307 
308                 System.setProperty("http.nonProxyHosts", "code.google.com");
309                 System.setProperty("socksNonProxyHosts", "code.google.com");
310 
311                 // clear previous values
312                 System.clearProperty("socksProxyHost");
313                 System.clearProperty("socksProxyPort");
314                 System.clearProperty("http.proxyHost");
315                 System.clearProperty("http.proxyPort");
316 
317                 if (connection.getProxyType() != Connection.ProxyType.NONE) {
318                     logger.info("Connect: using proxy " + connection.getProxyHost() + ":" + connection.getProxyPort());
319                 }
320                 switch (connection.getProxyType()) {
321                     case SOCKS:
322                         System.setProperty("socksProxyHost", connection.getProxyHost());
323                         System.setProperty("socksProxyPort", Integer.toString(connection.getProxyPort()));
324                         break;
325                     case HTTP:
326                         System.setProperty("http.proxyHost", connection.getProxyHost());
327                         System.setProperty("http.proxyPort", Integer.toString(connection.getProxyPort()));
328                         Authenticator.setDefault(new MageAuthenticator(connection.getProxyUsername(), connection.getProxyPassword()));
329                         break;
330                 }
331                 InvokerLocator clientLocator = new InvokerLocator(connection.getURI());
332 
333                 Map<String, String> metadata = new HashMap<>();
334                 /*
335                  5.8.3.1.1. Write timeouts
336                  The socket timeout facility offered by the JDK applies only to read operations on the socket. As of release 2.5.2,
337                  the socket and bisocket (and also sslsocket and sslbisocket) transports offer a write timeout facility. When a client
338                  or server is configured, in any of the usual ways, with the parameter org.jboss.remoting.transport.socket.SocketWrapper.WRITE_TIMEOUT
339                  (actual value "writeTimeout") set to a positive value (in milliseconds), all write operations will time out if they do
340                  not complete within the configured period. When a write operation times out, the socket upon which the write was invoked
341                  will be closed, which is likely to result in a java.net.SocketException.
342                  Note. A SocketException is considered to be a "retriable" exception, so, if the parameter "numberOfCallRetries" is set
343                  to a value greater than 1, an invocation interrupted by a write timeout can be retried.
344                  Note. The write timeout facility applies to writing of both invocations and responses. It applies to push callbacks as well.
345                  */
346                 metadata.put(SocketWrapper.WRITE_TIMEOUT, String.valueOf(connection.getSocketWriteTimeout()));
347                 metadata.put("generalizeSocketException", "true");
348                 server = (MageServer) TransporterClient.createTransporterClient(clientLocator.getLocatorURI(), MageServer.class, metadata);
349 
350                 // http://docs.jboss.org/jbossremoting/docs/guide/2.5/html_single/#d0e1057
351                 Map<String, String> clientMetadata = new HashMap<>();
352 
353                 clientMetadata.put(SocketWrapper.WRITE_TIMEOUT, String.valueOf(connection.getSocketWriteTimeout()));
354                 /*  generalizeSocketException
355                  *  If set to false, a failed invocation will be retried in the case of
356                  *  SocketExceptions. If set to true, a failed invocation will be retried in the case of
357                  *  <classname>SocketException</classname>s and also any <classname>IOException</classname>
358                  *  whose message matches the regular expression
359                  *  <code>^.*(?:connection.*reset|connection.*closed|broken.*pipe).*$</code>.
360                  *  See also the "numberOfCallRetries" parameter, above. The default value is false.*/
361                 clientMetadata.put("generalizeSocketException", "true");
362 
363                 /* A remoting server also has the capability to detect when a client is no longer available.
364                  * This is done by estabilishing a lease with the remoting clients that connect to a server.
365                  * On the client side, an org.jboss.remoting.LeasePinger periodically sends PING messages to
366                  * the server, and on the server side an org.jboss.remoting.Lease informs registered listeners
367                  * if the PING doesn't arrive withing the specified timeout period. */
368                 clientMetadata.put(Client.ENABLE_LEASE, "true");
369                 /*
370                  When the socket client invoker makes its first invocation, it will check to see if there is an available
371                  socket connection in its pool. Since is the first invocation, there will not be and will create a new socket
372                  connection and use it for making the invocation. Then when finished making invocation, will return the still
373                  active socket connection to the pool. As more client invocations are made, is possible for the number of
374                  socket connections to reach the maximum allowed (which is controlled by 'clientMaxPoolSize' property). At this
375                  point, when the next client invocation is made, it will wait up to some configured number of milliseconds, at
376                  which point it will throw an org.jboss.remoting.CannotConnectException. The number of milliseconds is given by
377                  the parameter MicroSocketClientInvoker.CONNECTION_WAIT (actual value "connectionWait"), with a default of
378                  30000 milliseconds. Note that if more than one call retry is configured (see next paragraph),
379                  the CannotConnectException will be swallowed.
380                  Once the socket client invoker get an available socket connection from the pool, are not out of the woods yet.
381                  For example, a network problem could cause a java.net.SocketException. There is also a possibility that the socket
382                  connection, while still appearing to be valid, has "gone stale" while sitting in the pool. For example, a ServerThread
383                  on the other side of the connection could time out and close its socket. If the attempt to complete an invocation
384                  fails, then MicroSocketClientInvoker will make a number of attempts, according to the parameter "numberOfCallRetries",
385                  with a default value of 3. Once the configured number of retries has been exhausted,
386                  an org.jboss.remoting.InvocationFailureException will be thrown.
387                  */
388                 clientMetadata.put("numberOfCallRetries", "1");
389 
390                 /**
391                  * I'll explain the meaning of "secondaryBindPort" and
392                  * "secondaryConnectPort", and maybe that will help. The
393                  * Remoting bisocket transport creates two ServerSockets on the
394                  * server. The "primary" ServerSocket is used to create
395                  * connections used for ordinary invocations, e.g., a request to
396                  * create a JMS consumer, and the "secondary" ServerSocket is
397                  * used to create "control" connections for internal Remoting
398                  * messages. The port for the primary ServerSocket is configured
399                  * by the "serverBindPort" parameter, and the port for the
400                  * secondary ServerSocket is, by default, chosen randomly. The
401                  * "secondaryBindPort" parameter can be used to assign a
402                  * specific port to the secondary ServerSocket. Now, if there is
403                  * a translating firewall between the client and server, the
404                  * client should be given the value of the port that is
405                  * translated to the actual binding port of the secondary
406                  * ServerSocket. For example, your configuration will tell the
407                  * secondary ServerSocket to bind to port 14000, and it will
408                  * tell the client to connect to port 14001. It assumes that
409                  * there is a firewall which will translate 14001 to 14000.
410                  * Apparently, that's not happening.
411                  */
412                 // secondaryBindPort - the port to which the secondary server socket is to be bound. By default, an arbitrary port is selected.
413                 // secondaryConnectPort - the port clients are to use to connect to the secondary server socket.
414                 // By default, the value of secondaryBindPort is used. secondaryConnectPort is useful if the server is behind a translating firewall.
415                 // Indicated the max number of threads used within oneway thread pool.
416                 clientMetadata.put(Client.MAX_NUM_ONEWAY_THREADS, "10");
417                 clientMetadata.put(Remoting.USE_CLIENT_CONNECTION_IDENTITY, "true");
418                 callbackClient = new Client(clientLocator, "callback", clientMetadata);
419 
420                 Map<String, String> listenerMetadata = new HashMap<>();
421                 if (debugMode) {
422                     // prevent client from disconnecting while debugging
423                     listenerMetadata.put(ConnectionValidator.VALIDATOR_PING_PERIOD, "1000000");
424                     listenerMetadata.put(ConnectionValidator.VALIDATOR_PING_TIMEOUT, "900000");
425                 } else {
426                     listenerMetadata.put(ConnectionValidator.VALIDATOR_PING_PERIOD, "15000");
427                     listenerMetadata.put(ConnectionValidator.VALIDATOR_PING_TIMEOUT, "13000");
428                 }
429                 callbackClient.connect(new ClientConnectionListener(), listenerMetadata);
430 
431                 Map<String, String> callbackMetadata = new HashMap<>();
432                 callbackMetadata.put(Bisocket.IS_CALLBACK_SERVER, "true");
433                 if (callbackHandler == null) {
434                     callbackHandler = new CallbackHandler();
435                 }
436                 callbackClient.addListener(callbackHandler, callbackMetadata);
437 
438                 Set callbackConnectors = callbackClient.getCallbackConnectors(callbackHandler);
439                 if (callbackConnectors.size() != 1) {
440                     logger.warn("There should be one callback Connector (number existing = " + callbackConnectors.size() + ')');
441                 }
442 
443                 callbackClient.invoke(null);
444 
445                 sessionId = callbackClient.getSessionId();
446                 sessionState = SessionState.CONNECTED;
447                 logger.info("Connect: DONE");
448                 return true;
449             }
450         };
451 
452         boolean result;
453         try {
454             result = doRemoteWorkAndHandleErrors(false, false, lastRemotingTask);
455         } finally {
456             lastRemotingTask = null;
457         }
458 
459         if (result) {
460             return true;
461         } else {
462             disconnect(false);
463             return false;
464         }
465     }

467     private void updateDatabase(boolean forceDBComparison, ServerState serverState) {
468         // download NEW cards/sets, but do not download data fixes (it's an old and rare feature from old clients, e.g. one client for different servers with different cards)
469         // use case: server gets new minor version with new cards, old client can get that cards too without donwload new version
470 
471         // sets
472         long expansionDBVersion = ExpansionRepository.instance.getContentVersionFromDB();
473         if (forceDBComparison || serverState.getExpansionsContentVersion() > expansionDBVersion) {
474             List<String> setCodes = ExpansionRepository.instance.getSetCodes();
475             List<ExpansionInfo> expansions = server.getMissingExpansionData(setCodes);
476             logger.info("DB: updating sets... Found new: " + expansions.size());
477             ExpansionRepository.instance.saveSets(expansions, null, serverState.getExpansionsContentVersion());
478         }
479 
480         // cards
481         long cardDBVersion = CardRepository.instance.getContentVersionFromDB();
482         if (forceDBComparison || serverState.getCardsContentVersion() > cardDBVersion) {
483             List<String> classNames = CardRepository.instance.getClassNames();
484             List<CardInfo> cards = server.getMissingCardsData(classNames);
485             logger.info("DB: updating cards... Found new: " + cards.size());
486             CardRepository.instance.saveCards(cards, serverState.getCardsContentVersion());
487         }
488     }

490     private void handleCannotConnectException(CannotConnectException ex) {
491         logger.warn("Cannot connect", ex);
492         Throwable t = ex.getCause();
493         String message = "";
494         while (t != null) {
495             if (t instanceof ConnectException) {
496                 message = "Server is likely offline." + message;
497                 break;
498             }
499             if (t instanceof SocketException) {
500                 message = "Check your internet connection." + message;
501                 break;
502             }
503             if (t instanceof SocketTimeoutException) {
504                 message = "Server is not responding." + message;
505                 break;
506             }
507             if (t.getCause() != null && logger.isDebugEnabled()) {
508                 message = '\n' + t.getCause().getMessage() + message;
509                 logger.debug(t.getCause().getMessage());
510             }
511 
512             t = t.getCause();
513         }
514         client.showMessage("Unable connect to server. " + message);
515         if (logger.isTraceEnabled()) {
516             logger.trace("StackTrace", t);
517         }
518     }

520     /**
521      * @param askForReconnect - true = connection was lost because of error and
522      *                        ask the user if they want to try to reconnect
523      */
524     @Override
525     public synchronized void disconnect(boolean askForReconnect) {
526         if (isConnected()) {
527             logger.info("Disconnecting...");
528             sessionState = SessionState.DISCONNECTING;
529         }
530         if (connection == null || sessionState == SessionState.DISCONNECTED) {
531             return;
532         }
533 
534         try {
535             if (callbackClient != null && callbackClient.isConnected()) {
536                 callbackClient.removeListener(callbackHandler);
537                 callbackClient.disconnect();
538             }
539             TransporterClient.destroyTransporterClient(server);
540         } catch (Throwable ex) {
541             logger.fatal("Disconnecting FAIL", ex);
542         }
543 
544         if (sessionState == SessionState.DISCONNECTING || sessionState == SessionState.CONNECTING) {
545             sessionState = SessionState.DISCONNECTED;
546             serverState = null;
547             logger.info("Disconnecting DONE");
548             if (askForReconnect) {
549                 client.showError("Network error. You have been disconnected from " + connection.getHost());
550             }
551             client.disconnected(askForReconnect); // MageFrame with check to reconnect
552             pingTime.clear();
553         }
554     }

952     @Override
953     public void appendJsonLog(ActionData actionData) {
954         if (isJsonLogActive()) {
955             String dir = "gamelogsJson";
956             File saveDir = new File(dir);
957             //Existence check
958             if (!saveDir.exists()) {
959                 saveDir.mkdirs();
960             }
961             actionData.sessionId = getSessionId();
962             String logFileName = dir + File.separator + "game-" + actionData.gameId + ".json";
963             try (PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(logFileName, true)))) {
964                 out.println(actionData.toJson());
965             } catch (IOException e) {
966                 logger.error("Cant write JSON game log file - " + logFileName, e);
967             }
968         }
969     }

1629     private void handleThrowable(Throwable t) {
1630 
1631         // ignore interrupted exceptions -- it's connection problem or user's close
1632         if (t instanceof InterruptedException) {
1633             //logger.error("Connection error: was interrupted", t);
1634             Thread.currentThread().interrupt();
1635             return;
1636         }
1637 
1638         if (t instanceof RuntimeException) {
1639             RuntimeException re = (RuntimeException) t;
1640             if (t.getCause() instanceof InterruptedException) {
1641                 //logger.error("Connection error: was interrupted by runtime exception", t.getCause());
1642                 Thread.currentThread().interrupt();
1643                 return;
1644             }
1645         }
1646 
1647         logger.fatal("Connection error: other", t);
1648     }

1691     @Override
1692     public boolean ping() {
1693         try {
1694             // ping must work after login only, all other actions are single call (example: register new user)
1695             // sessionId fills on connection
1696             // serverState fills on good login
1697             if (isConnected() && sessionId != null && serverState != null) {
1698                 long startTime = System.nanoTime();
1699                 if (!server.ping(sessionId, pingInfo)) {
1700                     logger.error("Ping failed: " + this.getUserName() + " Session: " + sessionId + " to MAGE server at " + connection.getHost() + ':' + connection.getPort());
1701                     throw new MageException("Ping failed");
1702                 }
1703                 pingTime.add(System.nanoTime() - startTime);
1704                 long milliSeconds = TimeUnit.MILLISECONDS.convert(pingTime.getLast(), TimeUnit.NANOSECONDS);
1705                 String lastPing = milliSeconds > 0 ? milliSeconds + "ms" : "<1ms";
1706                 if (pingTime.size() > PING_CYCLES) {
1707                     pingTime.poll();
1708                 }
1709                 long sum = 0;
1710                 for (Long time : pingTime) {
1711                     sum += time;
1712                 }
1713                 milliSeconds = TimeUnit.MILLISECONDS.convert(sum / pingTime.size(), TimeUnit.NANOSECONDS);
1714                 pingInfo = lastPing + " (avg: " + (milliSeconds > 0 ? milliSeconds + "ms" : "<1ms") + ')';
1715             }
1716             return true;
1717         } catch (MageException ex) {
1718             handleMageException(ex);
1719             disconnect(true);
1720         } catch (Throwable t) {
1721             handleThrowable(t);
1722         }
1723         return false;
1724     }

341     @Override
342     protected List<CardInfo> findCardsByRarity(Rarity rarity) {
343         List<CardInfo> cardInfos = super.findCardsByRarity(rarity);
344         if (rarity == Rarity.LAND) {
345             // only the full-art basic lands are found in boosters
346             cardInfos.removeIf(cardInfo -> cardInfo.getCardNumber().contains("a"));
347         } else if (rarity == Rarity.SPECIAL) {
348             cardInfos.addAll(CardRepository.instance.findCards(new CardCriteria().setCodes("EXP").maxCardNumber(25)));
349         }
350         return cardInfos;
351     }

355         super.generateBoosterMap();
356         CardRepository
357                 .instance
358                 .findCards(new CardCriteria().setCodes("EXP").maxCardNumber(25))
359                 .stream()
360                 .forEach(cardInfo -> inBoosterMap.put("EXP_" + cardInfo.getCardNumber(), cardInfo));
361     }

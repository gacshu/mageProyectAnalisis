216     @Override
217     protected void addDoubleFace(List<Card> booster) {
218         Rarity rarity;
219         for (int i = 0; i < numBoosterDoubleFaced; i++) {
220             int rarityKey = RandomUtil.nextInt(121);
221             if (rarityKey < 72) {
222                 rarity = Rarity.COMMON;
223             } else if (rarityKey < 108) {
224                 rarity = Rarity.UNCOMMON;
225             } else if (rarityKey < 117) {
226                 rarity = Rarity.RARE;
227             } else {
228                 rarity = Rarity.MYTHIC;
229             }
230             List<CardInfo> doubleFacedCards = getSpecialCardsByRarity(rarity);
231             addToBooster(booster, doubleFacedCards);
232         }
233     }

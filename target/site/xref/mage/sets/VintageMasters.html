360     @Override
361     protected void addSpecialCards(List<Card> booster, int number) {
362         // number is here always 1
363         Rarity rarity;
364         // You will open a Power Nine card about once in 53 packs
365         if (RandomUtil.nextInt(53) == 0) {
366             rarity = Rarity.BONUS;
367         } else {
368             // assuming same distribution as foils in paper Masters sets, 10:3:1 C:U:R
369             int rarityKey = RandomUtil.nextInt(112);
370             if (rarityKey < 80) {
371                 rarity = Rarity.COMMON;
372             } else if (rarityKey < 104) {
373                 rarity = Rarity.UNCOMMON;
374             } else if (rarityKey < 111) {
375                 rarity = Rarity.RARE;
376             } else {
377                 rarity = Rarity.MYTHIC;
378             }
379         }
380         addToBooster(booster, getCardsByRarity(rarity));
381     }

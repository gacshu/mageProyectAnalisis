269     // add common or uncommon double faced card to booster
270     // 60/120 packs contain one of 4 common DFCs and 60/120 packs contain one of 10 uncommon DFCs
271     @Override
272     protected void addDoubleFace(List<Card> booster) {
273         Rarity rarity;
274         for (int i = 0; i < numBoosterDoubleFaced; i++) {
275             if (RandomUtil.nextInt(120) < 60) {
276                 rarity = Rarity.COMMON;
277             } else {
278                 rarity = Rarity.UNCOMMON;
279             }
280             addToBooster(booster, getSpecialCardsByRarity(rarity));
281         }
282     }
283 
284     // Then about an eighth of the packs will have a second double-faced card, which will be a rare or mythic rare
285     // 10/12 of such packs contain one of 5 rare DFCs and 2/12 packs contain one of 2 mythic DFCs
286     @Override
287     protected void addSpecialCards(List<Card> booster, int number) {
288         // number is here always 1
289         Rarity rarity;
290         if (RandomUtil.nextInt(12) < 10) {
291             rarity = Rarity.RARE;
292         } else {
293             rarity = Rarity.MYTHIC;
294         }
295         addToBooster(booster, getSpecialCardsByRarity(rarity));
296     }

300     private static final String[] rareMeldCards = {"Bruna, the Fading Light", "Hanweir Battlements", "Hanweir Garrison"};
301     private static final String[] mythicMeldCards = {"Gisela, the Broken Blade"};
302 
303     @Override
304     protected List<CardInfo> findSpecialCardsByRarity(Rarity rarity) {
305         List<CardInfo> cardInfos = super.findSpecialCardsByRarity(rarity);
306         String[] meldCardNames = {};
307         if (rarity == Rarity.COMMON) {
308             meldCardNames = commonMeldCards;
309         } else if (rarity == Rarity.RARE) {
310             meldCardNames = rareMeldCards;
311         } else if (rarity == Rarity.MYTHIC) {
312             meldCardNames = mythicMeldCards;
313         }
314         for (String name : meldCardNames) {
315             cardInfos.addAll(CardRepository.instance.findCards(new CardCriteria()
316                     .setCodes(this.code)
317                     .nameExact(name)));
318         }
319         return cardInfos;
320     }

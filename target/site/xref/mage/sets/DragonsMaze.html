200     @Override
201     protected void addSpecialCards(List<Card> booster, int number) {
202         // number is here always 1
203         // the land print sheets are believed to contain 23 copies of each Guildgate,
204         // one copy of each RTR and GTC shockland, and two copies of Maze's End
205         Rarity rarity;
206         int rarityKey = RandomUtil.nextInt(242);
207         if (rarityKey < 230) {
208             rarity = Rarity.COMMON;
209         } else if (rarityKey < 240) {
210             rarity = Rarity.RARE;
211         } else {
212             rarity = Rarity.MYTHIC;
213         }
214         addToBooster(booster, getSpecialCardsByRarity(rarity));
215     }
216 
217     @Override
218     protected List<CardInfo> findSpecialCardsByRarity(Rarity rarity) {
219         CardCriteria criteria = new CardCriteria();
220         criteria.rarities(rarity).types(CardType.LAND);
221         if (rarity == Rarity.RARE) {
222             // shocklands
223             criteria.setCodes("RTR", "GTC");
224         } else {
225             // Guildgates and Maze's End
226             criteria.setCodes(this.code);
227         }
228         List<CardInfo> cardInfos = CardRepository.instance.findCards(criteria);
229         cardInfos.removeIf(cardInfo -> (cardInfo.getName().equals("Grove of the Guardian")
230                                     || cardInfo.getName().equals("Thespian's Stage")));
231         return cardInfos;
232     }
233 
234     @Override
235     protected void generateBoosterMap() {
236         super.generateBoosterMap();
237         CardRepository
238                 .instance
239                 .findCards(new CardCriteria().setCodes("RTR", "GTC").rarities(Rarity.RARE).types(CardType.LAND))
240                 .stream()
241                 .forEach(cardInfo -> inBoosterMap.put(cardInfo.getSetCode() + "_" + cardInfo.getCardNumber(), cardInfo));
242     }

333     @Override
334     protected void addSpecialCards(List<Card> booster, int number) {
335         // number is here always 1
336         // Boosters have one card from BRR, odds are 2/3 for uncommon, 4/15 for rare, 1/15 for mythic
337         Rarity rarity;
338         int rarityKey = RandomUtil.nextInt(15);
339         if (rarityKey == 14) {
340             rarity = Rarity.MYTHIC;
341         } else if (rarityKey >= 10) {
342             rarity = Rarity.RARE;
343         } else {
344             rarity = Rarity.UNCOMMON;
345         }
346         addToBooster(booster, TheBrothersWarRetroArtifacts.getInstance().getCardsByRarity(rarity));
347     }

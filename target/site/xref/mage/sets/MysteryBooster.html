3478      *
3479      * @param slotNumber booster slot number. 1-indexed, valid range is 1-14, as 15 is the special slot
3480      * @param cardNames  List of English card names found on the given slot
3481      */
3482     private void populateSlot(int slotNumber, List<String> cardNames) {
3483         final List<CardInfo> cardInfoList = this.possibleCardsPerBoosterSlot.get(slotNumber);
3484         for (String name : cardNames) {
3485             final CardInfo cardWithGivenName = CardRepository.instance.findCardWithPreferredSetAndNumber(name, this.code, null);
3486             cardInfoList.add(cardWithGivenName);
3487         }
3488     }

3502         if (this.possibleCardsPerBoosterSlot.isEmpty()) {
3503             // Generate the map only once
3504             this.populateBoosterSlotMap();

3506         final List<Card> booster = new ArrayList<>(15);
3507         for (int slot = 1; slot < 16; ++slot) {
3508             final List<CardInfo> availableCards = this.possibleCardsPerBoosterSlot.get(slot);
3509             final int printSheetCardNumber = RandomUtil.nextInt(availableCards.size());
3510             final Card chosenCard = availableCards.get(printSheetCardNumber).getCard();
3511             booster.add(chosenCard);
3512         }
3513         return booster;
3514     }

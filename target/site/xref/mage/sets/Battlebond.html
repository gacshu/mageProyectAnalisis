297     @Override
298     public List<Card> tryBooster() {
299         List<Card> booster = new ArrayList<>();
300         boolean partnerAllowed = true;
301         List<CardInfo> uncommons = getCardsByRarity(Rarity.UNCOMMON);
302         for (int i = 0; i < numBoosterUncommon; i++) {
303             while (true) {
304                 addToBooster(booster, uncommons);
305                 int check = addMissingPartner(booster, partnerAllowed, numBoosterUncommon - 1, i);
306                 if (check == 1) {
307                     break;
308                 }
309                 if (check == 2) {
310                     partnerAllowed = false;
311                     //Be sure to account for the added card
312                     if (i != numBoosterUncommon - 1) {
313                         i += 1;
314                     }
315                     break;
316                 }
317             }
318         }
319 
320         List<CardInfo> commons = getCardsByRarity(Rarity.COMMON);
321         for (int i = 0; i < numBoosterCommon; i++) {
322             addToBooster(booster, commons);
323         }
324 
325         List<CardInfo> rares = getCardsByRarity(Rarity.RARE);
326         List<CardInfo> mythics = getCardsByRarity(Rarity.MYTHIC);
327         for (int i = 0; i < numBoosterRare; i++) {
328             List<CardInfo> cards = (checkMythic() ? mythics : rares);
329             while (true) {
330                 addToBooster(booster, cards);
331                 int check = addMissingPartner(booster, partnerAllowed, -1, 1);
332                 if (check == 1) {
333                     break;
334                 }
335                 if (check == 2) {
336                     partnerAllowed = false;
337                     break;
338                 }
339             }
340         }
341 
342         List<CardInfo> basicLands = getCardsByRarity(Rarity.LAND);
343         for (int i = 0; i < numBoosterLands; i++) {
344             addToBooster(booster, basicLands);
345         }
346 
347         return booster;
348     }

350     private int addMissingPartner(List<Card> booster, boolean partnerAllowed, int max, int i) {
351 
352         Card sourceCard = booster.get(booster.size() - 1);
353         for (Ability ability : sourceCard.getAbilities()) {
354 
355             //Check if fetched card has the PartnerWithAbility
356             if (ability instanceof PartnerWithAbility) {
357                 String partnerName = ((PartnerWithAbility) ability).getPartnerName();
358                 //Check if the pack already contains a partner pair
359                 if (partnerAllowed) {
360                     //Added card always replaces an uncommon card
361                     Card card = CardRepository.instance.findCardWithPreferredSetAndNumber(partnerName, sourceCard.getExpansionSetCode(), null).getCard();
362                     if (i < max) {
363                         booster.add(card);
364                     } else {
365                         booster.set(0, card);
366                     }
367                     //2 return value indicates found partner
368                     return 2;
369                 } else {
370                     //If partner already exists, remove card and loop again
371                     booster.remove(booster.size() - 1);
372                     return 0;
373                 }
374             }
375         }
376         return 1;
377     }

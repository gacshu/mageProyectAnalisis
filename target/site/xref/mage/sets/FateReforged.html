233         // number is here always 1
234         Rarity rarity;
235         if (RandomUtil.nextInt(24) < 23) {
236             rarity = Rarity.COMMON;
237         } else {

239         }
240         addToBooster(booster, getSpecialCardsByRarity(rarity));

243     @Override
244     protected List<CardInfo> findSpecialCardsByRarity(Rarity rarity) {
245         CardCriteria criteria = new CardCriteria();
246         criteria.rarities(rarity).types(CardType.LAND);
247         if (rarity == Rarity.RARE) {
248             // fetchlands
249             criteria.setCodes("KTK");
250         } else {
251             // gainlands
252             criteria.setCodes(this.code);
253         }
254         return CardRepository.instance.findCards(criteria);
255     }

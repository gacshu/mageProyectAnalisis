192     @Override
193     protected List<CardInfo> findCardsByRarity(Rarity rarity) {
194         List<CardInfo> cardInfos = super.findCardsByRarity(rarity);
195         if (rarity == Rarity.COMMON) {
196             // Snow basic lands are included in boosters as regular commons
197             cardInfos.addAll(CardRepository.instance.findCards(new CardCriteria().setCodes(this.code).rarities(Rarity.LAND)));
198         }
199         return cardInfos;
200     }

333         List<CardInfo> cardInfos = super.findSpecialCardsByRarity(rarity);
334         cardInfos.addAll(CardRepository.instance.findCards(new CardCriteria()
335                 .setCodes(this.code)
336                 .rarities(rarity)
337                 .supertypes(SuperType.LEGENDARY)
338                 .types(CardType.CREATURE)
339                 .maxCardNumber(maxCardNumberInBooster)));

426     // In order for equal numbers of each uncommon to exist, the average booster must contain:
427     // 1.125 A uncommons     (9 / 8)
428     // 1.125 B uncommons     (9 / 8)
429     // 0.75  uncommon legend (6 / 8)
430     private final RarityConfiguration uncommonRuns = new RarityConfiguration(
431             ABL, ABL, ABL, ABL, ABL, ABL,
432             AAB, ABB
433     );
434     private final RarityConfiguration landRuns = new RarityConfiguration(L1);
435 
436     @Override
437     public List<String> makeBooster() {
438         List<String> booster = new ArrayList<>();
439         booster.addAll(commonRuns.getNext().makeRun());
440         booster.addAll(uncommonRuns.getNext().makeRun());
441         booster.addAll(landRuns.getNext().makeRun());
442         return booster;
443     }

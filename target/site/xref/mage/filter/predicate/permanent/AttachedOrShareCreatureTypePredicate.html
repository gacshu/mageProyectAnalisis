3   import mage.filter.predicate.ObjectSourcePlayer;
4   import mage.filter.predicate.ObjectSourcePlayerPredicate;
5   import mage.game.Game;
6   import mage.game.permanent.Permanent;

8   import java.util.UUID;
9   
10  /**
11   *
12   * @author awjackson
13   */
14  public enum AttachedOrShareCreatureTypePredicate implements ObjectSourcePlayerPredicate<Permanent> {
15      instance;
16  
17      @Override
18      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
19          Permanent attachment = input.getSource().getSourcePermanentOrLKI(game);
20          if (attachment == null) {
21              return false;
22          }
23          UUID attachedId = attachment.getAttachedTo();
24          if (attachedId == null) {
25              return false;
26          }
27          Permanent otherPermanent = input.getObject();
28          if (attachedId.equals(otherPermanent.getId())) {
29              return true;
30          }
31          Permanent attachedPermanent = game.getPermanentOrLKIBattlefield(attachedId);
32          if (attachedPermanent == null) {
33              return false;
34          }
35          return attachedPermanent.shareCreatureTypes(game, otherPermanent);
36      }
37  }

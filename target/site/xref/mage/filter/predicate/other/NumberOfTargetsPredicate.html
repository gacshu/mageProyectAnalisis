1   package mage.filter.predicate.other;
2   
3   import java.util.UUID;
4   import mage.abilities.Mode;
5   import mage.filter.predicate.Predicate;
6   import mage.game.Controllable;
7   import mage.game.Game;
8   import mage.game.stack.StackObject;
9   import mage.target.Target;
10  
11  /**
12   *
13   * @author jeffwadsworth
14   */
15  public class NumberOfTargetsPredicate implements Predicate<Controllable> {
16  
17      private final int targets;
18  
19      public NumberOfTargetsPredicate(int targets) {
20          this.targets = targets;
21      }
22  
23      @Override
24      public boolean apply(Controllable input, Game game) {
25          StackObject stackObject = game.getState().getStack().getStackObject(input.getId());
26          if (stackObject != null) {
27              int numberOfTargets = 0;
28              for (UUID modeId : stackObject.getStackAbility().getModes().getSelectedModes()) {
29                  Mode mode = stackObject.getStackAbility().getModes().get(modeId);
30                  for (Target target : mode.getTargets()) {
31                      numberOfTargets += target.getTargets().size();
32                  }
33              }
34              if (numberOfTargets == targets) {
35                  return true;
36              }
37          }
38          return false;
39      }
40  
41      @Override
42      public String toString() {
43          return "Number of targets(" + targets + ')';
44      }
45  }

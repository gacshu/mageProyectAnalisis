21      /**
22       * Returns a predicate that evaluates to {@code true} if the given predicate evaluates to {@code false}.
23       *
24       * @param <T>
25       * @param predicate
26       * @return
27       */
28      public static <T> Predicate<T> not(Predicate<T> predicate) {
29          return new NotPredicate<T>(predicate);
30      }

32      /**
33       * Returns a predicate that evaluates to {@code true} if each of its components evaluates to {@code true}. The
34       * components are evaluated in order, and evaluation will be "short-circuited" as soon as a false predicate is
35       * found. It defensively copies the iterable passed in, so future changes to it won't alter the behavior of this
36       * predicate. If {@code components} is empty, the returned predicate will always evaluate to {@code true}.
37       *
38       * @param <T>
39       * @param components
40       * @return
41       */
42      public static <T> Predicate<T> and(Iterable<? extends Predicate<? super T>> components) {
43          return new AndPredicate<T>(defensiveCopy(components));
44      }

47       * Returns a predicate that evaluates to {@code true} if each of its components evaluates to {@code true}. The
48       * components are evaluated in order, and evaluation will be "short-circuited" as soon as a false predicate is
49       * found. It defensively copies the array passed in, so future changes to it won't alter the behavior of this
50       * predicate. If {@code components} is empty, the returned predicate will always evaluate to {@code true}.
51       *
52       * @param <T>
53       * @param components
54       * @return
55       */
56      public static <T> Predicate<T> and(Predicate<? super T>... components) {
57          return new AndPredicate<T>(defensiveCopy(components));

60      /**
61       * Returns a predicate that evaluates to {@code true} if both of its components evaluate to {@code true}. The
62       * components are evaluated in order, and evaluation will be "short-circuited" as soon as a false predicate is
63       * found.
64       *
65       * @param <T>
66       * @param first
67       * @param second
68       * @return

70      public static <T> Predicate<T> and(Predicate<? super T> first, Predicate<? super T> second) {

95       * @param components
96       * @return

115     /**
116      * @see Predicates#not(Predicate)
117      */
118     private static class NotPredicate<T> implements Predicate<T> {
119 
120         final Predicate<T> predicate;
121 
122         NotPredicate(Predicate<T> predicate) {
123             this.predicate = checkNotNull(predicate);
124         }
125 
126         @Override
127         public boolean apply(T t, Game game) {
128             return !predicate.apply(t, game);
129         }
130 
131         @Override
132         public String toString() {
133             return "Not(" + predicate.toString() + ')';
134         }
135 
136         private static final long serialVersionUID = 0;
137     }
138 
139     /**
140      * @see Predicates#and(Iterable)
141      */
142     private static class AndPredicate<T> implements Predicate<T> {
143 
144         private final List<? extends Predicate<? super T>> components;
145 
146         private AndPredicate(List<? extends Predicate<? super T>> components) {
147             this.components = components;
148         }
149 
150         @Override
151         public boolean apply(T t, Game game) {
152             return components.stream().allMatch(predicate -> predicate.apply(t, game));
153 
154         }
155 
156         @Override
157         public String toString() {
158             return "And(" + commaJoin(components) + ')';
159         }
160 
161         private static final long serialVersionUID = 0;
162     }
163 
164     /**
165      * @see Predicates#or(Iterable)
166      */
167     private static class OrPredicate<T> implements Predicate<T> {
168 
169         private final List<? extends Predicate<? super T>> components;
170 
171         private OrPredicate(List<? extends Predicate<? super T>> components) {
172             this.components = components;
173         }
174 
175         @Override
176         public boolean apply(T t, Game game) {
177             return components.stream().anyMatch(predicate -> predicate.apply(t, game));
178         }
179 
180         @Override
181         public String toString() {
182             return "Or(" + commaJoin(components) + ')';
183         }
184 
185         private static final long serialVersionUID = 0;
186     }

188     @SuppressWarnings("unchecked")
189     private static <T> List<Predicate<? super T>> asList(Predicate<? super T> first, Predicate<? super T> second) {
190         return Arrays.asList(first, second);
191     }

193     private static <T> List<T> defensiveCopy(T... array) {
194         return defensiveCopy(Arrays.asList(array));
195     }
196 
197     static <T> List<T> defensiveCopy(Iterable<T> iterable) {
198         List<T> list = new ArrayList<>();
199         for (T element : iterable) {
200             list.add(checkNotNull(element));
201         }
202         return list;
203     }

205     /**
206      * Ensures that an object reference passed as a parameter to the calling method is not null.
207      *
208      * @param reference an object reference
209      * @return the non-null reference that was validated
210      * @throws NullPointerException if {@code reference} is null
211      */
212     private static <T> T checkNotNull(T reference) {
213         if (reference == null) {
214             throw new NullPointerException();
215         }
216         return reference;
217     }

219     private static String commaJoin(List components) {
220         StringBuilder sb = new StringBuilder();
221         for (Object component : components) {
222             sb.append(component.toString());
223         }
224         sb.deleteCharAt(sb.length() - 1);
225         return sb.toString();
226     }

228     /**
229      * Collect real predicates for searching some data (see dependency effect code)
230      */
231     public static void collectAllComponents(Predicate predicate, List<Predicate> res) {
232         if (predicate instanceof NotPredicate) {
233             collectAllComponents(((NotPredicate) predicate).predicate, res);
234         } else if (predicate instanceof AndPredicate) {
235             collectAllComponents(((AndPredicate) predicate).components, res);
236         } else if (predicate instanceof OrPredicate) {
237             collectAllComponents(((OrPredicate) predicate).components, res);
238         } else {
239             res.add(predicate);
240         }
241     }
242 
243     public static void collectAllComponents(List<Predicate> predicates, List<Predicate> res) {
244         predicates.forEach(p -> {
245             collectAllComponents(p, res);
246         });
247     }

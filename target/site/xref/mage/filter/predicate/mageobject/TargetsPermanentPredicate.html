4   import mage.abilities.Mode;

6   import mage.filter.predicate.ObjectSourcePlayer;

8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  import mage.game.stack.StackObject;
11  import mage.target.Target;
12  
13  import java.util.UUID;
14  
15  /**
16   * @author LoneFox
17   */
18  public class TargetsPermanentPredicate implements ObjectSourcePlayerPredicate<MageObject> {
19  
20      private final FilterPermanent targetFilter;
21  
22      public TargetsPermanentPredicate(FilterPermanent targetFilter) {
23          this.targetFilter = targetFilter;
24      }
25  
26      @Override
27      public boolean apply(ObjectSourcePlayer<MageObject> input, Game game) {
28          StackObject object = game.getStack().getStackObject(input.getObject().getId());
29          if (object != null) {
30              for (UUID modeId : object.getStackAbility().getModes().getSelectedModes()) {
31                  Mode mode = object.getStackAbility().getModes().get(modeId);
32                  for (Target target : mode.getTargets()) {
33                      if (target.isNotTarget()) {
34                          continue;
35                      }
36                      for (UUID targetId : target.getTargets()) {
37                          Permanent permanent = game.getPermanentOrLKIBattlefield(targetId);
38                          if (targetFilter.match(permanent, input.getPlayerId(), input.getSource(), game)) {
39                              return true;
40                          }
41                      }
42                  }
43              }
44          }
45          return false;
46      }
47  
48      @Override
49      public String toString() {
50          return "that targets " + targetFilter.getMessage();
51      }
52  }

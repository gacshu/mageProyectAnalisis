2   package mage.filter.common;
3   
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Set;
7   import java.util.UUID;
8   import mage.filter.FilterImpl;
9   import mage.filter.FilterPlayer;
10  import mage.filter.predicate.Predicate;
11  import mage.filter.predicate.Predicates;
12  import mage.filter.predicate.other.PlayerIdPredicate;
13  import mage.filter.predicate.permanent.ControllerIdPredicate;
14  import mage.game.Game;
15  import mage.game.permanent.Permanent;
16  import mage.players.Player;
17  
18  /**
19   *
20   * @author BetaSteward_at_googlemail.com
21   */
22  public class FilterPlaneswalkerOrPlayer extends FilterImpl<Object> {
23  
24      protected final FilterPlaneswalkerPermanent planeswalkerFilter;
25      protected final FilterPlayer playerFilter;
26  
27      public FilterPlaneswalkerOrPlayer(Set<UUID> defenders) {
28          super("planeswalker or player");
29  
30          List<Predicate<Permanent>> permanentPredicates = new ArrayList<>();
31          for (UUID defenderId : defenders) {
32              permanentPredicates.add(new ControllerIdPredicate(defenderId));
33          }
34          planeswalkerFilter = new FilterPlaneswalkerPermanent();
35          planeswalkerFilter.add(Predicates.or(permanentPredicates));
36  
37          List<Predicate<Player>> playerPredicates = new ArrayList<>();
38          for (UUID defenderId : defenders) {
39              playerPredicates.add(new PlayerIdPredicate(defenderId));
40          }
41          playerFilter = new FilterPlayer();
42          playerFilter.add(Predicates.or(playerPredicates));
43      }
44  
45      public FilterPlaneswalkerOrPlayer(final FilterPlaneswalkerOrPlayer filter) {
46          super(filter);
47          this.planeswalkerFilter = filter.planeswalkerFilter.copy();
48          this.playerFilter = filter.playerFilter.copy();
49      }
50  
51      public FilterPlaneswalkerPermanent getFilterPermanent() {
52          return this.planeswalkerFilter;
53      }
54  
55      public FilterPlayer getFilterPlayer() {
56          return this.playerFilter;
57      }
58  
59      @Override
60      public boolean checkObjectClass(Object object) {
61          return true;
62      }
63  
64      @Override
65      public boolean match(Object o, Game game) {
66          if (o instanceof Player) {
67              return playerFilter.match((Player) o, game);
68          } else if (o instanceof Permanent) {
69              return planeswalkerFilter.match((Permanent) o, game);
70          }
71          return false;
72      }
73  
74      @Override
75      public FilterPlaneswalkerOrPlayer copy() {
76          return new FilterPlaneswalkerOrPlayer(this);
77      }
78  }

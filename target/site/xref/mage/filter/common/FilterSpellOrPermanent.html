1   package mage.filter.common;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.filter.FilterImpl;
6   import mage.filter.FilterInPlay;
7   import mage.filter.FilterPermanent;
8   import mage.filter.FilterSpell;
9   import mage.game.Game;
10  import mage.game.permanent.Permanent;
11  import mage.game.stack.Spell;
12  
13  import java.util.UUID;
14  
15  /**
16   * @author LevelX
17   */
18  public class FilterSpellOrPermanent extends FilterImpl<MageObject> implements FilterInPlay<MageObject> {
19  
20      protected FilterPermanent permanentFilter;
21      protected FilterSpell spellFilter;
22  
23      public FilterSpellOrPermanent() {
24          this("spell or permanent");
25      }
26  
27      public FilterSpellOrPermanent(String name) {
28          super(name);
29          permanentFilter = new FilterPermanent();
30          spellFilter = new FilterSpell();
31      }
32  
33      public FilterSpellOrPermanent(final FilterSpellOrPermanent filter) {
34          super(filter);
35          this.permanentFilter = filter.permanentFilter.copy();
36          this.spellFilter = filter.spellFilter.copy();
37      }
38  
39      @Override
40      public boolean checkObjectClass(Object object) {
41          return object instanceof MageObject;
42      }
43  
44      @Override
45      public boolean match(MageObject o, Game game) {
46          if (o instanceof Spell) {
47              return spellFilter.match((Spell) o, game);
48          } else if (o instanceof Permanent) {
49              return permanentFilter.match((Permanent) o, game);
50          }
51          return false;
52      }
53  
54      @Override
55      public boolean match(MageObject o, UUID playerId, Ability source, Game game) {
56          if (o instanceof Spell) {
57              return spellFilter.match((Spell) o, playerId, source, game);
58          } else if (o instanceof Permanent) {
59              return permanentFilter.match((Permanent) o, playerId, source, game);
60          }
61          return false;
62      }
63  
64      @Override
65      public void setLockedFilter(boolean lockedFilter) {
66          super.setLockedFilter(lockedFilter);
67          spellFilter.setLockedFilter(lockedFilter);
68          permanentFilter.setLockedFilter(lockedFilter);
69      }
70  
71      public FilterPermanent getPermanentFilter() {
72          return this.permanentFilter;
73      }
74  
75      public FilterSpell getSpellFilter() {
76          return this.spellFilter;
77      }
78  
79      public void setPermanentFilter(FilterPermanent permanentFilter) {
80          this.permanentFilter = permanentFilter;
81      }
82  
83      public void setSpellFilter(FilterSpell spellFilter) {
84          this.spellFilter = spellFilter;
85      }
86  
87      @Override
88      public FilterSpellOrPermanent copy() {
89          return new FilterSpellOrPermanent(this);
90      }
91  
92  }

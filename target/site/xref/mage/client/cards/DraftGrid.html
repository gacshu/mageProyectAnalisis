39      public DraftGrid() {
40          initComponents();
41          markedCard = null;
42          emptyGrid = true;
43  
44          // ENABLE picks and other actions
45          cardEventSource.addListener(new Listener<Event>() {
46              @Override
47              public void event(Event event) {
48                  if (event.getEventType() == ClientEventType.CARD_DOUBLE_CLICK) {
49                      CardView card = (CardView) event.getSource();
50                      cardEventSource.fireEvent(card, ClientEventType.DRAFT_PICK_CARD);
51                      hidePopup();
52                      AudioManager.playOnDraftSelect();
53                  } else if (event.getEventType() == ClientEventType.CARD_CLICK) {
54                      CardView card = (CardView) event.getSource();
55                      MageCard cardPanel = (MageCard) event.getComponent();
56                      if (markedCard != null) {
57                          markedCard.setSelected(false);
58                      }
59                      cardEventSource.fireEvent(card, ClientEventType.DRAFT_MARK_CARD);
60                      markedCard = cardPanel;
61                      markedCard.setSelected(true);
62                      repaint();
63                  }
64              }
65          });
66      }

77      public void loadBooster(CardsView booster, BigCard bigCard) {
78          if (booster != null && booster.isEmpty()) {
79              emptyGrid = true;
80          } else {
81              if (!emptyGrid) {
82                  AudioManager.playOnDraftSelect();
83              }
84              emptyGrid = false;
85          }
86          this.bigCard = bigCard;
87          this.removeAll();
88  
89          int maxRows = 4;
90  
91          int numColumns = 5;
92          int curColumn = 0;
93          int curRow = 0;
94          int offsetX = 5;
95          int offsetY = 3;
96  
97          CardDimensions cardDimension = null;
98          int maxCards;
99          double scale;
100 
101         for (int i = 1; i < maxRows; i++) {
102             scale = (double) (this.getHeight() / i) / Constants.FRAME_MAX_HEIGHT;
103             cardDimension = new CardDimensions(scale);
104             maxCards = this.getWidth() / (cardDimension.getFrameWidth() + offsetX);
105             if ((maxCards * i) >= booster.size()) {
106                 numColumns = booster.size() / i;
107                 if (booster.size() % i > 0) {
108                     numColumns++;
109                 }
110                 break;
111             }
112         }
113 
114         if (cardDimension != null) {
115             Rectangle rectangle = new Rectangle(cardDimension.getFrameWidth(), cardDimension.getFrameHeight());
116             Dimension dimension = new Dimension(cardDimension.getFrameWidth(), cardDimension.getFrameHeight());
117 
118             List<CardView> sortedCards = new ArrayList<>(booster.values());
119             sortedCards.sort(new CardViewRarityComparator());
120             for (CardView card : sortedCards) {
121                 MageCard cardImg = Plugins.instance.getMageCard(card, bigCard, new CardIconRenderSettings(), dimension, null, true, true, PreferencesDialog.getRenderMode(), true);
122                 cardImg.setCardContainerRef(this);
123                 cardImg.update(card);
124                 this.add(cardImg);
125 
126                 rectangle.setLocation(curColumn * (cardDimension.getFrameWidth() + offsetX) + offsetX, curRow * (rectangle.height + offsetY) + offsetY);
127                 cardImg.setCardBounds(rectangle.x, rectangle.y, rectangle.width, rectangle.height);
128                 curColumn++;
129                 if (curColumn == numColumns) {
130                     curColumn = 0;
131                     curRow++;
132                 }
133             }
134             repaint();
135         } else {
136             logger.warn("Draft Grid - no possible fit of cards");
137         }
138     }

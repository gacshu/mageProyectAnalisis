87       public void changeGUISize() {
88           setGUISize();
89           for (MageCard mageCard : cards.values()) {
90               mageCard.setCardBounds(0, 0, getCardDimension().width, getCardDimension().height);
91               mageCard.updateArtImage();
92               mageCard.doLayout();
93           }
94           layoutCards();
95           sizeCards(getCardDimension());
96       }

134      public boolean loadCards(CardsView cardsView, BigCard bigCard, UUID gameId, boolean revertOrder) {
135          boolean changed = false;
136 
137          // auto-move scrollbars to the end of the list
138          boolean moveScrollbar;
139          if (zone == Zone.HAND) {
140              // hand moves on new cards only
141              moveScrollbar = lastLoadedCardsCount != 0 && cardsView.size() > lastLoadedCardsCount;
142          } else {
143              // stack moves on any changes (e.g. show current stack object)
144              moveScrollbar = cardsView.size() != lastLoadedCardsCount;
145          }
146          this.lastLoadedCardsCount = cardsView.size();
147 
148          // remove objects no longer to display
149          changed = removeOutdatedCards(cardsView);
150 
151          // Workaround for bug leaving display of objects on the stack (issue #213 https://github.com/magefree/mage/issues/213)
152          if (cardsView.isEmpty() && countCards() > 0) {
153              // problem happens with transformable cards
154              logger.fatal("Card object on the cards panel was not removed");
155              for (Component comp : cardArea.getComponents()) {
156                  if (comp instanceof MageCard) {
157                      MageCard mageCard = (MageCard) comp;
158                      logger.fatal("MageCard name:" + mageCard.getName() + " toolTiptext:" + mageCard.getToolTipText());
159                  } else {
160                      logger.fatal("Unknown object:" + comp.getName() + " className:" + comp.getClass().getName());
161                  }
162                  cardArea.remove(comp);
163              }
164          }
165 
166          java.util.List<CardView> orderedList = new ArrayList<>();
167          if (revertOrder) {
168              // order objects for display
169              for (CardView card : cardsView.values()) {
170                  orderedList.add(0, card);
171              }
172          } else {
173              orderedList.addAll(cardsView.values());
174          }
175 
176          // add objects to the panel
177          for (CardView card : orderedList) {
178              if (dontDisplayTapped) {
179                  if (card instanceof PermanentView) {
180                      ((PermanentView) card).overrideTapped(false);
181                  }
182              }
183              if (card instanceof StackAbilityView) {
184                  // replace ability by original card
185                  CardView tmp = ((StackAbilityView) card).getSourceCard();
186                  // sync settings
187                  tmp.overrideRules(card.getRules());
188                  tmp.setChoosable(card.isChoosable());
189                  tmp.setPlayableStats(card.getPlayableStats().copy());
190                  tmp.setSelected(card.isSelected());
191                  tmp.setIsAbility(true);
192                  tmp.overrideTargets(card.getTargets());
193                  tmp.overrideId(card.getId());
194                  tmp.setAbilityType(card.getAbilityType());
195                  // sync card icons
196                  tmp.getCardIcons().clear();
197                  tmp.getCardIcons().addAll(card.getCardIcons());
198                  card = tmp;
199              } else {
200                  card.setAbilityType(null);
201              }
202              if (!cards.containsKey(card.getId())) {
203                  addCard(card, bigCard, gameId);
204                  changed = true;
205              }
206              cards.get(card.getId()).update(card);
207          }
208 
209          if (changed) {
210              layoutCards();
211          }
212 
213          if (!isVisibleIfEmpty) {
214              cardArea.setVisible(!cards.isEmpty());
215          }
216 
217          sizeCards(getCardDimension());
218          this.revalidate();
219          this.repaint();
220 
221          // auto-scroll (must use it at the end)
222          if (changed && moveScrollbar) {
223              SwingUtilities.invokeLater(() -> {
224                  if (jScrollPane1 != null) {
225                      jScrollPane1.getHorizontalScrollBar().setValue(jScrollPane1.getHorizontalScrollBar().getMaximum());
226                  }
227                  if (parentScrollPane != null) {
228                      parentScrollPane.getHorizontalScrollBar().setValue(parentScrollPane.getHorizontalScrollBar().getMaximum());
229                  }
230              });
231          }
232 
233          return changed;
234      }

236      public void sizeCards(Dimension cardDimension) {
237          cardArea.setPreferredSize(new Dimension(
238                  (int) ((cards.size()) * (cardDimension.getWidth() + MageActionCallback.getHandOrStackBetweenGapX(zone)))
239                          + MageActionCallback.getHandOrStackMargins(zone).getWidth(),
240                  (int) (cardDimension.getHeight())
241                          + MageActionCallback.getHandOrStackMargins(zone).getHeight()
242                  ));
243          cardArea.revalidate();
244          cardArea.repaint();
245      }

268      private void addCard(CardView card, BigCard bigCard, UUID gameId) {
269          MageCard mageCard = Plugins.instance.getMageCard(card, bigCard, new CardIconRenderSettings(), getCardDimension(), gameId, true, true, PreferencesDialog.getRenderMode(), true);
270          mageCard.setCardContainerRef(cardArea);
271          mageCard.update(card);
272          if (zone != null) {
273              mageCard.setZone(zone);
274          }
275          cards.put(card.getId(), mageCard);
276          cardArea.add(mageCard);
277          definePosition(mageCard);
278      }

280      private void definePosition(MageCard newCard) {
281          int dx = MageActionCallback.getHandOrStackMargins(zone).getLeft(); // starting position
282 
283          // search last card's position
284          for (Component currentComp : cardArea.getComponents()) {
285              if (!currentComp.equals(newCard) && currentComp instanceof MageCard) {
286                  MageCard currentCard = (MageCard) currentComp;
287                  dx = Math.max(dx, currentCard.getCardLocation().getCardX());
288              }
289          }
290 
291          // Y position sets here one time only (all sorting and drag manipulations works with X)
292          // add card to the end
293          dx += newCard.getCardLocation().getCardWidth() + MageActionCallback.getHandOrStackBetweenGapX(newCard.getZone());
294          newCard.setCardLocation(dx, MageActionCallback.getHandOrStackMargins(newCard.getZone()).getTop());
295      }

297      private boolean removeOutdatedCards(CardsView cardsView) {
298          boolean changed = false;
299 
300          // links to components
301          cards.keySet().removeIf(id -> !cardsView.containsKey(id));
302 
303          // components
304          for (Component comp : cardArea.getComponents()) {
305              if (comp instanceof MageCard) {
306                  if (!cards.containsValue(comp)) {
307                      cardArea.remove(comp);
308                      changed = true;
309                  }
310              } else {
311                  logger.error("Unknown card conponent in cards panel to remove: " + comp);
312              }
313          }
314 
315          return changed;
316      }

363      private void layoutCards() {
364          // get all the card panels
365          java.util.List<MageCard> cardsToLayout = new ArrayList<>();
366          for (Component component : cardArea.getComponents()) {
367              if (component instanceof MageCard) {
368                  cardsToLayout.add((MageCard) component);
369              }
370          }
371 
372          // WARNING, must be same sort code as MageActionCallback->sortLayout (if not then hand cards will be messed after drag)
373 
374          // sort the cards
375          cardsToLayout.sort(Comparator.comparingInt(cp -> cp.getCardLocation().getCardX()));
376 
377          // relocate the cards (support only horizontal style: hand and stack panels)
378          // TODO: add shrinking of cards list for too big amount (cards will be overlapped, use MageActionCallback.HAND_CARDS_BETWEEN_GAP_X to control it)
379          int dx = MageActionCallback.getHandOrStackBetweenGapX(zone); // starting position
380          for (MageCard component : cardsToLayout) {
381              component.setCardLocation(dx, component.getCardLocation().getCardY());
382              dx += component.getCardLocation().getCardWidth() + MageActionCallback.getHandOrStackBetweenGapX(zone);
383          }
384      }

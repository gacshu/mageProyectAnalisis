1   package mage.client.components;
2   
3   import java.awt.Color;
4   import java.awt.Font;
5   import java.awt.Graphics;
6   import javax.swing.JButton;
7   import mage.client.dialog.PreferencesDialog;
8   
9   /**
10   *
11   * @author Campbell Suter <znix@znix.xyz>
12   */
13  public class KeyboundButton extends JButton {
14  
15      private final String text;
16      private static final Font keyFont = new Font(Font.SANS_SERIF, Font.BOLD, 13);
17  
18      private boolean tinting = false;
19  
20      public KeyboundButton(String key, boolean drawText) {
21          if (drawText) {
22              text = PreferencesDialog.getCachedKeyText(key);
23          } else {
24              text = "";
25          }
26      }
27  
28      @Override
29      protected void paintComponent(Graphics g) {
30          if (ui != null && g != null) {
31              Graphics sg = g.create();
32              try {
33                  ui.update(sg, this);
34  
35                  if (tinting) {
36                      sg.setColor(new Color(0, 0, 0, 32));
37                      sg.fillRoundRect(2, 2, getWidth() - 4 , getHeight() - 4, 6, 6);
38                  }
39                  sg.setColor(tinting ? Color.lightGray : Color.white);
40  
41                  if (!text.isEmpty()) {
42                      sg.setFont(keyFont);
43  
44                      int textWidth = sg.getFontMetrics(keyFont).stringWidth(text);
45                      int centerX = (getWidth() - textWidth) / 2;
46  
47                      sg.drawString(text, centerX, 28);
48                  }
49              } finally {
50                  sg.dispose();
51              }
52          }
53      }
54  
55      public void setTint(boolean tinting) {
56          this.tinting = tinting;
57          repaint();
58      }
59  }

92      public void show(AbilityPickerView choices, Point p) {
93          this.choices = new ArrayList<>();
94          this.selected = true; // to stop previous modal
95          setMessageAndPrepare(choices.getMessage());
96  
97          // if not cancel from server then add own
98          boolean wasCancelButton = false;
99          for (Map.Entry<UUID, String> choice : choices.getChoices().entrySet()) {
100             wasCancelButton = wasCancelButton || choice.getKey().equals(Modes.CHOOSE_OPTION_CANCEL_ID);
101             this.choices.add(new AbilityPickerAction(choice.getKey(), choice.getValue()));
102         }
103         if (!wasCancelButton) {
104             this.choices.add(new AbilityPickerAction(null, "Cancel"));
105         }
106 
107         show(this.choices);
108     }

206     @Override
207     public void mouseWheelMoved(MouseWheelEvent e) {
208         int notches = e.getWheelRotation();
209         int index = rows.getSelectedIndex();
210 
211         if (notches < 0) {
212             if (index > 0) {
213                 rows.setSelectedIndex(index - 1);
214                 rows.repaint();
215             }
216         } else if (index < choices.size() - 1) {
217             rows.setSelectedIndex(index + 1);
218             rows.repaint();
219         }
220     }

230         public final Map<String, String> cache = new HashMap<>();
231 
232         @Override
233         public Component getListCellRendererComponent(
234                 javax.swing.JList list,
235                 Object value,
236                 int index,
237                 boolean isSelected,
238                 boolean cellHasFocus
239         ) {
240 
241             setBorder(BorderFactory.createEmptyBorder(3, 3, 3, 6));
242 
243             UI.setHTMLEditorKit(this);
244 
245             setOpaque(false);
246             setBackground(new Color(0, 0, 0, 0));
247 
248             String text = value.toString();
249 
250             if (cache.containsKey(text)) {
251                 text = cache.get(text);
252             } else {
253                 StringBuilder buffer = getHtmlForText(isSelected, text);
254                 String rendered = buffer.toString();
255                 cache.put(text, rendered);
256                 text = rendered;
257             }
258 
259             final String finalText = text;
260             // System.out.println(finalText);
261 
262             ImageRenderer2.super.setText(finalText);
263             setCaretPosition(0);
264 
265             return this;
266         }

268         private StringBuilder getHtmlForText(boolean isSelected, String text) {
269             int fontSize = 16;
270 
271             String fontFamily = "arial";
272 
273             final StringBuilder buffer = new StringBuilder(512);
274             buffer.append("<html><body style='font-family:");
275             buffer.append(fontFamily);
276             buffer.append(";font-size:");
277             buffer.append(fontSize);
278             buffer.append("pt;margin:3px 3px 3px 3px;");
279             if (isSelected) {
280                 buffer.append("color: #4093D0'>");
281             } else {
282                 buffer.append("color: #FFFFFF'>");
283             }
284             buffer.append("<b>");
285 
286             text = text.replaceAll("#([^#]+)#", "<i>$1</i>");
287             text = text.replaceAll("\\s*//\\s*", "<hr width='50%'>");
288             text = text.replace("\r\n", "<div style='font-size:5pt'></div>");
289             //text += "<br>";
290 
291             if (!text.isEmpty()) {
292                 buffer.append(ManaSymbols.replaceSymbolsWithHTML(text, ManaSymbols.Type.DIALOG));
293             }
294 
295             buffer.append("</b></body></html>");
296             return buffer;
297         }

417     public class AbilityPickerAction extends AbstractAction {
418 
419         private final UUID id;
420 
421         public AbilityPickerAction(UUID id, String choice) {
422             this.id = id;
423             putValue(Action.NAME, capitalizeFirstLetter(choice));
424         }
425 
426         private String capitalizeFirstLetter(String choice) {
427             if (choice == null || choice.isEmpty()) {
428                 return choice;
429             }
430             choice = Jsoup.parse(choice).text(); // decode HTML entities and strip tags
431             return choice.substring(0, 1).toUpperCase(Locale.ENGLISH) + choice.substring(1);
432         }
433 
434         @Override
435         public void actionPerformed(ActionEvent e) {
436             // cancel
437             if (id == null) {
438                 cancel();
439             } else {
440                 SessionHandler.sendPlayerUUID(gameId, id);
441             }
442             setVisible(false);
443             AbilityPicker.this.selected = true;
444         }
445 
446         @Override
447         public String toString() {
448             return (String) getValue(Action.NAME);
449         }

470     private void tryChoiceDone() {
471         // done by keyboard
472         if (!isVisible() || choices == null) {
473             return;
474         }
475         for (Object obj : choices) {
476             AbilityPickerAction action = (AbilityPickerAction) obj;
477             if (Modes.CHOOSE_OPTION_DONE_ID.equals(action.id)) {
478                 action.actionPerformed(null);
479                 break;
480             }
481         }
482     }
483 
484     private void tryChoiceCancel() {
485         // cancel by keyboard
486         if (!isVisible() || choices == null) {
487             return;
488         }
489         for (Object obj : choices) {
490             AbilityPickerAction action = (AbilityPickerAction) obj;
491             if (Modes.CHOOSE_OPTION_CANCEL_ID.equals(action.id)) {
492                 action.actionPerformed(null);
493                 break;
494             }
495         }
496 
497         // old cancel style (just close dialog)
498         for (Object obj : choices) {
499             AbilityPickerAction action = (AbilityPickerAction) obj;
500             if (action.id == null) {
501                 cancel();
502                 break;
503             }
504         }
505     }
506 
507     private void tryChoiceOption(int choiceNumber) {
508         // choice by keyboard
509         if (!isVisible() || choices == null) {
510             return;
511         }
512         String need = choiceNumber + ".";
513         for (Object obj : choices) {
514             AbilityPickerAction action = (AbilityPickerAction) obj;
515             if (action.toString().startsWith(need)) {
516                 action.actionPerformed(null);
517                 break;
518             }
519         }
520     }

522     public void injectHotkeys(GamePanel panel, String commandsPrefix) {
523         // TODO: fix that GamePanel recive imput from any place, not only active (e.g. F9 works from lobby)
524         int c = JComponent.WHEN_IN_FOCUSED_WINDOW;
525 
526         // choice keys
527         Map<Integer, Integer> numbers = new HashMap<>();
528         numbers.put(KeyEvent.VK_1, 1);
529         numbers.put(KeyEvent.VK_2, 2);
530         numbers.put(KeyEvent.VK_3, 3);
531         numbers.put(KeyEvent.VK_4, 4);
532         numbers.put(KeyEvent.VK_5, 5);
533         numbers.put(KeyEvent.VK_6, 6);
534         numbers.put(KeyEvent.VK_7, 7);
535         numbers.put(KeyEvent.VK_8, 8);
536         numbers.put(KeyEvent.VK_9, 9);
537         numbers.forEach((vk, num) -> {
538             KeyStroke ks = KeyStroke.getKeyStroke(vk, 0);
539             panel.getInputMap(c).put(ks, commandsPrefix + "_CHOOSE_" + num);
540             panel.getActionMap().put(commandsPrefix + "_CHOOSE_" + num, new AbstractAction() {
541                 @Override
542                 public void actionPerformed(ActionEvent actionEvent) {
543                     tryChoiceOption(num);
544                 }
545             });
546         });
547 
548         // done key (space, enter)
549         panel.getInputMap(c).put(KeyStroke.getKeyStroke(KeyEvent.VK_SPACE, 0), commandsPrefix + "_CHOOSE_DONE");
550         panel.getInputMap(c).put(KeyStroke.getKeyStroke(KeyEvent.VK_ENTER, 0), commandsPrefix + "_CHOOSE_DONE");
551         panel.getActionMap().put(commandsPrefix + "_CHOOSE_DONE", new AbstractAction() {
552             @Override
553             public void actionPerformed(ActionEvent actionEvent) {
554                 tryChoiceDone();
555             }
556         });
557 
558         // cancel key (esc can be lost by focus on exile window popup)
559         panel.getInputMap(c).put(KeyStroke.getKeyStroke(KeyEvent.VK_BACK_SPACE, 0), commandsPrefix + "_CHOOSE_CANCEL");
560         panel.getActionMap().put(commandsPrefix + "_CHOOSE_CANCEL", new AbstractAction() {
561             @Override
562             public void actionPerformed(ActionEvent actionEvent) {
563                 tryChoiceCancel();
564             }
565         });

119     public void mouseClicked(MouseEvent e, TransferData data, boolean doubleClick) {
120         // send mouse clicked event to the card's area and other cards list components for processing
121         if (e.isConsumed()) {
122             return;
123         }
124         if (data.getComponent().getCardContainer() instanceof CardEventProducer) {
125             ClientEventType clickType = doubleClick ? ClientEventType.CARD_DOUBLE_CLICK : ClientEventType.CARD_CLICK;
126             CardEventProducer cardContainer = (CardEventProducer) data.getComponent().getCardContainer();
127             mage.client.util.Event clientEvent = new mage.client.util.Event(
128                     data.getComponent().getOriginal(), clickType,
129                     0, e.getX(), e.getY(), data.getComponent(), e, false
130             );
131             cardContainer.getCardEventSource().fireEvent(clientEvent);
132         }
133     }

142     private void startCardHintPopup(final TransferData data, final Component parentComponent, final Point parentPoint) {
143         MageCard cardPanel = data.getComponent().getTopPanelRef();
144 
145         if (data.getTooltipDelay() > 0) {

165             tooltipPopup.show();
166             // hack to get popup to resize to fit text
167             tooltipPopup.hide();
168             tooltipPopup = factory.getPopup(cardPanel, data.getPopupText(), (int) data.getLocationOnScreen().getX() + data.getPopupOffsetX(), (int) data.getLocationOnScreen().getY() + data.getPopupOffsetY() + 40);

170         } else {
171             showCardHintPopup(data, parentComponent, parentPoint);

175     private void showCardHintPopup(final TransferData data, final Component parentComponent, final Point parentPoint) {
176         MageCard cardPanel = data.getComponent().getTopPanelRef();
177 
178         ThreadUtils.threadPool2.submit(new Runnable() {
179             @Override
180             public void run() {
181                 ThreadUtils.sleep(tooltipDelay);
182 
183                 if (tooltipCard == null
184                         || !tooltipCard.equals(data.getCard())
185                         || SessionHandler.getSession() == null
186                         || !popupTextWindowOpen
187                         || enlargedWindowState != EnlargedWindowState.CLOSED) {
188                     return;
189                 }
190 
191                 try {
192                     final Component popupContainer = MageFrame.getUI().getComponent(MageComponents.POPUP_CONTAINER);
193                     Component popupInfo = MageFrame.getUI().getComponent(MageComponents.CARD_INFO_PANE);
194                     ((CardInfoPane) popupInfo).setCard(data.getCard(), popupContainer);
195                     showPopup(popupContainer, popupInfo);
196                 } catch (InterruptedException e) {
197                     logger.error("Can't show card tooltip", e);
198                     Thread.currentThread().interrupt();
199                 }
200             }

203                 final Component c = MageFrame.getUI().getComponent(MageComponents.DESKTOP_PANE);
204                 SwingUtilities.invokeLater(() -> {

216                             popupContainer.setLocation(location);
217                             popupContainer.setVisible(true);
218                             c.repaint();
219                         }
220                 );
221             }

225     @Override
226     public void mousePressed(MouseEvent e, TransferData data) {
227         MageCard cardPanel = data.getComponent().getTopPanelRef();
228         cardPanel.requestFocusInWindow();
229 
230         // for some reason sometime mouseRelease happens before numerous Mouse_Dragged events
231         // that results in not finished dragging
232         clearDragging(this.prevCardPanel);
233 
234         isDragging = false;
235         startedDragging = false;
236         prevCardPanel = null;
237         draggingCards.clear();
238         Point mouse = new Point(e.getX(), e.getY());
239         SwingUtilities.convertPointToScreen(mouse, data.getComponent());
240         initialMousePos = new Point((int) mouse.getX(), (int) mouse.getY());
241         initialCardPos = cardPanel.getCardLocation().getCardPoint();
242         // Closes popup & enlarged view if a card/Permanent is selected
243         hideTooltipPopup();
244     }

246     @Override
247     public void mouseReleased(MouseEvent e, TransferData data) {
248         MageCard cardPanel = data.getComponent().getTopPanelRef();
249         if (e.isPopupTrigger()) {
250             hideTooltipPopup();
251         } else if (cardPanel.getZone() == Zone.HAND) {
252             // drag end
253             int maxXOffset = 0;
254             if (isDragging) {
255                 Point mouse = new Point(e.getX(), e.getY());
256                 SwingUtilities.convertPointToScreen(mouse, data.getComponent());
257                 maxXOffset = Math.abs((int) (mouse.getX() - initialMousePos.x));
258             }
259 
260             clearDragging(cardPanel);
261 
262             this.startedDragging = false;
263             if (maxXOffset < MIN_X_OFFSET_REQUIRED) { // we need this for protection from small card movements
264                 // default click simulation // TODO: replace to normal clicked events (change dialogs)
265                 cardPanel.requestFocusInWindow();
266                 DefaultActionCallback.instance.mouseClicked(data.getGameId(), data.getCard());
267                 // Closes popup & enlarged view if a card/Permanent is selected
268                 hideTooltipPopup();
269             }
270             e.consume();
271         } else {
272             // default click simulation
273             cardPanel.requestFocusInWindow();
274             DefaultActionCallback.instance.mouseClicked(data.getGameId(), data.getCard());
275             // Closes popup & enlarged view if a card/Permanent is selected
276             hideTooltipPopup();
277             e.consume();
278         }
279     }

281     private void clearDragging(MageCard clearCard) {
282         if (this.startedDragging && prevCardPanel != null && clearCard != null) {
283             // distribute cards between cards container and a drag container
284             for (Component comp : clearCard.getCardContainer().getComponents()) {
285                 if (comp instanceof MageCard) {
286                     MageCard realCard = (MageCard) comp;
287                     if (draggingCards.contains(realCard)) {
288                         realCard.setCardLocation(realCard.getCardLocation().getCardX(), realCard.getCardLocation().getCardY() - GO_DOWN_ON_DRAG_Y_OFFSET);
289                     }
290                 }
291             }
292             clearCard.setCardLocation(clearCard.getCardLocation().getCardX(), clearCard.getCardLocation().getCardY() + GO_UP_ON_DRAG_Y_OFFSET);
293             sortHandCards(clearCard, clearCard.getCardContainer(), true);
294             draggingCards.clear();
295         }
296         prevCardPanel = null;
297     }

317     @Override
318     public void mouseDragged(MouseEvent e, TransferData data) {
319         // start the dragging
320         MageCard cardPanel = data.getComponent().getTopPanelRef();
321         if (cardPanel.getZone() != Zone.HAND) {
322             // drag'n'drop is allowed for HAND zone only
323             return;
324         }
325         if (!SwingUtilities.isLeftMouseButton(e)) {
326             // only allow draging with the left mouse button
327             return;
328         }
329         isDragging = true;
330         prevCardPanel = cardPanel;
331 
332         Point cardPanelLocationOld = cardPanel.getCardLocation().getCardPoint();
333         Point mouse = new Point(e.getX(), e.getY());
334         SwingUtilities.convertPointToScreen(mouse, data.getComponent());
335         int xOffset = 0; // starting position
336         int newX = Math.max(initialCardPos.x + (int) (mouse.getX() - initialMousePos.x) - xOffset, 0); // TODO: fix
337         cardPanel.setCardBounds(
338                 newX,
339                 cardPanelLocationOld.y,
340                 cardPanel.getCardLocation().getCardWidth(),
341                 cardPanel.getCardLocation().getCardHeight());
342         cardPanel.getCardContainer().setComponentZOrder(cardPanel, 0);
343         sortHandCards(cardPanel, cardPanel.getCardContainer(), false);
344 
345         if (!this.startedDragging) {
346             this.startedDragging = true;
347         }
348     }

396     private void sortHandCards(MageCard card, Container container, boolean sortSource) {
397         java.util.List<MageCard> cards = new ArrayList<>();
398 
399         // distribute cards between cards container and a drag container
400         for (Component comp : container.getComponents()) {
401             if (comp instanceof MageCard) {
402                 MageCard realCard = (MageCard) comp;
403                 if (!realCard.equals(card)) {
404                     if (!draggingCards.contains(realCard)) {
405                         realCard.setCardLocation(realCard.getCardLocation().getCardX(), realCard.getCardLocation().getCardY() + GO_DOWN_ON_DRAG_Y_OFFSET);
406                     }
407                     draggingCards.add(realCard);
408                 } else if (!startedDragging) {
409                     realCard.setCardLocation(realCard.getCardLocation().getCardX(), realCard.getCardLocation().getCardY() - GO_DOWN_ON_DRAG_Y_OFFSET);
410                 }
411                 cards.add(realCard);
412             }
413         }
414 
415         sortAndAnimateDraggingHandCards(cards, card, sortSource);
416     }

418     private void sortAndAnimateDraggingHandCards(List<MageCard> cards, MageCard source, boolean includeSource) {
419         // special offset, allows to switch with first card
420         source.setCardLocation(source.getCardLocation().getCardX() - HAND_CARDS_COMPARE_GAP_X, source.getCardLocation().getCardY());
421 
422         // sorting card components, so the effect above will be applied too
423         cards.sort(Comparator.comparingInt(cp -> cp.getCardLocation().getCardX()));
424 
425         // WARNING, must be same sort code as Cards->layoutCards (if not then hand cards will be messed after drag)
426         int dx = MageActionCallback.getHandOrStackMargins(source.getZone()).getLeft(); // starting position
427         boolean createdGapForSource = false;
428         for (MageCard component : cards) {
429             // use real component locations, not a card's
430             if (!includeSource) {
431                 // create special hole between cards to put dragging card into it
432                 if (!component.equals(source)) {
433                     component.setCardLocation(dx, component.getCardLocation().getCardY());
434                     dx += component.getCardLocation().getCardWidth() + MageActionCallback.getHandOrStackBetweenGapX(source.getZone());
435                     // once dx is bigger than source's x position
436                     // we need to create a gap for the source card
437                     // but only once
438                     if (!createdGapForSource && (dx + HAND_CARDS_COMPARE_GAP_X) > source.getCardLocation().getCardX()) {
439                         createdGapForSource = true;
440                         int gapOffset = component.getCardLocation().getCardWidth() + MageActionCallback.getHandOrStackBetweenGapX(source.getZone());
441                         dx += gapOffset;
442                         // workaround to apply gap on the first card (if you drag over first card)
443                         if (cards.get(0).equals(source) && cards.size() > 1 && cards.get(1).equals(component)) {
444                             component.setCardLocation(component.getCardLocation().getCardX() + gapOffset, component.getCardLocation().getCardY());
445                         }
446                     }
447                 }
448             } else {
449                 component.setCardLocation(dx, component.getCardLocation().getCardY());
450                 dx += component.getCardLocation().getCardWidth() + MageActionCallback.getHandOrStackBetweenGapX(source.getZone());
451             }
452         }
453     }

455     private void handleMouseMoveOverNewCard(TransferData data) {
456         MageCard cardPanel = data.getComponent().getTopPanelRef();
457 
458         // Prevent to show tooltips from panes not in front
459         MagePane topPane = MageFrame.getTopMost(null);
460         if (topPane instanceof GamePane) {
461             if (data.getGameId() != null && !((GamePane) topPane).getGameId().equals(data.getGameId())) {
462                 return;
463             }
464         }
465 
466         hideTooltipPopup();
467         cancelTimeout();
468         Component parentComponent = SwingUtilities.getRoot(cardPanel);
469         if (parentComponent == null) {
470             // virtual card (example: show card popup in non cards panel like PickChoiceDialog)
471             parentComponent = MageFrame.getDesktop();
472         }
473         Point parentPoint = parentComponent.getLocationOnScreen();
474 
475         if (data.getLocationOnScreen() == null) {
476             data.setLocationOnScreen(cardPanel.getCardLocationOnScreen().getCardPoint());
477         }
478 
479         ArrowUtil.drawArrowsForTargets(data, parentPoint);
480         ArrowUtil.drawArrowsForSource(data, parentPoint);
481         ArrowUtil.drawArrowsForPairedCards(data, parentPoint);
482         ArrowUtil.drawArrowsForBandedCards(data, parentPoint);
483         ArrowUtil.drawArrowsForEnchantPlayers(data, parentPoint);
484 
485         tooltipCard = data.getCard();
486         startCardHintPopup(data, parentComponent, parentPoint);
487     }

489     private void updateCardHints(TransferData data) {
490         MageCard cardPanel = data.getComponent().getTopPanelRef();
491 
492         if (!popupTextWindowOpen
493                 || !Objects.equals(cardPanel.getOriginal().getId(), bigCard.getCardId())) {
494             if (bigCard.getWidth() > 0) {
495                 synchronized (MageActionCallback.class) {
496                     if (!popupTextWindowOpen || !Objects.equals(cardPanel.getOriginal().getId(), bigCard.getCardId())) {
497                         if (!popupTextWindowOpen) {
498                             bigCard.resetCardId();
499                         }
500                         popupTextWindowOpen = true;
501                         Image image = cardPanel.getImage();
502                         displayCardInfo(cardPanel.getOriginal(), image, bigCard);
503                     }
504                 }
505             } else {
506                 popupTextWindowOpen = true;
507             }
508             if (enlargedWindowState != EnlargedWindowState.CLOSED) {
509                 cancelTimeout();
510                 displayEnlargedCard(cardPanel.getOriginal(), data);
511             }
512         }
513     }

558     @Override
559     public void mouseWheelMoved(MouseWheelEvent e, TransferData data) {
560         int notches = e.getWheelRotation();
561         if (enlargedWindowState != EnlargedWindowState.CLOSED) {
562             // same move direction will be ignored, opposite direction closes the enlarged window
563             if (enlargeredViewOpened != null && new Date().getTime() - enlargeredViewOpened.getTime() > 1000) {
564                 // if the opening is back more than 1 seconds close anyway
565                 hideEnlargedCard();
566                 handleMouseMoveOverNewCard(data);
567             } else if (enlargeMode == EnlargeMode.NORMAL) {
568                 if (notches > 0) {
569                     hideEnlargedCard();
570                     handleMouseMoveOverNewCard(data);
571                 }
572             } else if (notches < 0) {
573                 hideEnlargedCard();
574                 handleMouseMoveOverNewCard(data);
575             }
576             return;
577         }
578         if (notches < 0) {
579             // move up - show normal image
580             enlargeCard(EnlargeMode.NORMAL);
581         } else {
582             // move down - show alternate image
583             enlargeCard(EnlargeMode.ALTERNATE);
584         }
585     }

587     /**
588      * Show the big card image on mouse position while hovering over a card
589      *
590      * @param showAlternative defines if the original image (if it's a copied
591      *                        card) or the opposite side of a transformable card will be shown
592      */
593     public void enlargeCard(EnlargeMode showAlternative) {
594         if (enlargedWindowState == EnlargedWindowState.CLOSED) {
595             this.enlargeMode = showAlternative;
596             CardView cardView = null;
597             if (popupData != null) {
598                 cardView = popupData.getCard();
599             }
600             if (this.popupTextWindowOpen) {
601                 hideTooltipPopup();
602             }
603             if (cardView != null) {
604                 if (cardView.isToRotate()) {
605                     enlargedWindowState = EnlargedWindowState.ROTATED;
606                 } else {
607                     enlargedWindowState = EnlargedWindowState.NORMAL;
608                 }
609                 displayEnlargedCard(cardView, popupData);
610             }
611         }
612     }

614     public void hideEnlargedCard() {
615         enlargedWindowState = EnlargedWindowState.CLOSED;
616         try {
617             Component cardPreviewContainer = MageFrame.getUI().getComponent(MageComponents.CARD_PREVIEW_CONTAINER);
618             if (cardPreviewContainer.isVisible()) {
619                 cardPreviewContainer.setVisible(false);
620                 cardPreviewContainer.repaint();
621             }
622             cardPreviewContainer = MageFrame.getUI().getComponent(MageComponents.CARD_PREVIEW_CONTAINER_ROTATED);
623             if (cardPreviewContainer.isVisible()) {
624                 cardPreviewContainer.setVisible(false);
625                 cardPreviewContainer.repaint();
626             }
627 
628         } catch (InterruptedException e) {
629             logger.warn("Can't hide enlarged card", e);
630         }
631     }

633     private void displayEnlargedCard(final CardView cardView, final TransferData data) {
634         MageCard cardPanel = data.getComponent().getTopPanelRef();
635 
636         ThreadUtils.threadPool3.submit(() -> {
637             if (cardView == null) {
638                 return;
639             }
640             try {
641                 if (enlargedWindowState == EnlargedWindowState.CLOSED) {
642                     return;
643                 }
644 
645                 MageComponents mageComponentCardPreviewContainer;
646                 MageComponents mageComponentCardPreviewPane;
647                 if (cardView.isToRotate()) {
648                     if (enlargedWindowState == EnlargedWindowState.NORMAL) {
649                         hideEnlargedCard();
650                         enlargedWindowState = EnlargedWindowState.ROTATED;
651                     }
652                     mageComponentCardPreviewContainer = MageComponents.CARD_PREVIEW_CONTAINER_ROTATED;
653                     mageComponentCardPreviewPane = MageComponents.CARD_PREVIEW_PANE_ROTATED;
654                 } else {
655                     if (enlargedWindowState == EnlargedWindowState.ROTATED) {
656                         hideEnlargedCard();
657                         enlargedWindowState = EnlargedWindowState.NORMAL;
658                     }
659                     mageComponentCardPreviewContainer = MageComponents.CARD_PREVIEW_CONTAINER;
660                     mageComponentCardPreviewPane = MageComponents.CARD_PREVIEW_PANE;
661                 }
662                 final Component popupContainer = MageFrame.getUI().getComponent(mageComponentCardPreviewContainer);
663                 Component cardPreviewPane = MageFrame.getUI().getComponent(mageComponentCardPreviewPane);
664                 Component parentComponent = SwingUtilities.getRoot(cardPanel);
665                 if (cardPreviewPane != null && parentComponent != null) {
666                     Point parentPoint = parentComponent.getLocationOnScreen();
667                     data.setLocationOnScreen(cardPanel.getCardLocationOnScreen().getCardPoint());
668                     Point location = new Point((int) data.getLocationOnScreen().getX() + data.getPopupOffsetX() - 40, (int) data.getLocationOnScreen().getY() + data.getPopupOffsetY() - 40);
669                     location = GuiDisplayUtil.keepComponentInsideParent(location, parentPoint, cardPreviewPane, parentComponent);
670                     location.translate(-parentPoint.x, -parentPoint.y);
671                     popupContainer.setLocation(location);
672                     popupContainer.setVisible(true);
673 
674                     // popup hint mode
675                     Image image = cardPanel.getImage();
676                     CardView displayCard = cardPanel.getOriginal();
677                     switch (enlargeMode) {
678                         case COPY:
679                             if (cardView instanceof PermanentView) {
680                                 image = ImageCache.getImageOriginal(((PermanentView) cardView).getOriginal()).getImage();
681                             }
682                             break;
683                         case ALTERNATE:
684                             if (cardView.getAlternateName() != null) {
685                                 if (cardView instanceof PermanentView
686                                         && !cardView.isFlipCard()
687                                         && !cardView.canTransform()
688                                         && ((PermanentView) cardView).isCopy()) {
689                                     image = ImageCache.getImageOriginal(((PermanentView) cardView).getOriginal()).getImage();
690                                 } else {
691                                     image = ImageCache.getImageOriginalAlternateName(cardView).getImage();
692                                     displayCard = displayCard.getSecondCardFace();
693                                 }
694                             }
695                             break;
696                         default:
697                             break;
698                     }
699 
700                     // shows the card in the popup Container
701                     displayCardInfo(displayCard, image, (BigCard) cardPreviewPane);
702                 } else {
703                     logger.warn("No Card preview Pane in Mage Frame defined. Card: " + cardView.getName());
704                 }
705 
706             } catch (Exception e) {
707                 logger.warn("Problem dring display of enlarged card", e);
708             }
709         });
710     }

712     private void displayCardInfo(CardView card, Image image, BigCard bigCard) {
713         if (image instanceof BufferedImage) {
714             // XXX: scaled to fit width
715             bigCard.setCard(card.getId(), enlargeMode, image, card.getRules(), card.isToRotate());
716             // if it's an ability, show only the ability text as overlay
717             if (card.isAbility() && enlargeMode == EnlargeMode.NORMAL && isAbilityTextOverlayEnabled()) {
718                 bigCard.showTextComponent();
719             } else {
720                 bigCard.hideTextComponent();
721             }
722         } else {
723             JXPanel panel = GuiDisplayUtil.getDescription(card, bigCard.getWidth(), bigCard.getHeight());
724             panel.setVisible(true);
725             bigCard.hideTextComponent();
726             bigCard.addJXPanel(card.getId(), panel);
727         }
728         enlargeredViewOpened = new Date();
729     }

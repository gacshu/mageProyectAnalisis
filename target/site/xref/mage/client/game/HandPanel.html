20      public void initComponents() {
21          jPanel = new JPanel();
22          jScrollPane1 = new JScrollPane(jPanel);
23          jScrollPane1.getViewport().setBackground(new Color(0, 0, 0, 0));
24  
25          hand = new mage.client.cards.Cards(true, jScrollPane1);
26          hand.setCardDimension(GUISizeHelper.handCardDimension);
27  
28          jPanel.setLayout(new GridBagLayout()); // centers hand
29          jPanel.setBackground(new Color(0, 0, 0, 0));
30          jPanel.add(hand);
31  
32          setOpaque(false);
33          jPanel.setOpaque(false);
34          jScrollPane1.setOpaque(false);
35  
36          jPanel.setBorder(EMPTY_BORDER);
37          jScrollPane1.setBorder(EMPTY_BORDER);
38          jScrollPane1.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_NEVER);
39          jScrollPane1.getHorizontalScrollBar().setUnitIncrement(8);
40          jScrollPane1.setViewportBorder(EMPTY_BORDER);
41  
42          setLayout(new BorderLayout());
43          add(jScrollPane1, BorderLayout.CENTER);
44  
45          hand.setHScrollSpeed(8);
46          hand.setBackgroundColor(new Color(0, 0, 0, 0));
47          hand.setVisibleIfEmpty(false);

51  
52      public void cleanUp() {
53          hand.cleanUp();
54      }
55  
56      public void changeGUISize() {
57          setGUISize();
58      }
59  
60      private void setGUISize() {
61          jScrollPane1.getVerticalScrollBar().setPreferredSize(new Dimension(GUISizeHelper.scrollBarSize, 0));
62          jScrollPane1.getHorizontalScrollBar().setPreferredSize(new Dimension(0, GUISizeHelper.scrollBarSize));
63          jScrollPane1.getHorizontalScrollBar().setUnitIncrement(GUISizeHelper.getCardsScrollbarUnitInc(GUISizeHelper.handCardDimension.width));
64          hand.setCardDimension(GUISizeHelper.handCardDimension);
65          hand.changeGUISize();
66      }
67

70      }

68      public void prepareFeedback(FeedbackMode mode, String message, boolean special, Map<String, Serializable> options,
69                                  int messageId, boolean gameNeedUserFeedback, TurnPhase gameTurnPhase) {
70          synchronized (this) {
71              if (messageId < this.lastMessageId) {
72                  // if too many warning messages here then look at GAME_REDRAW_GUI event logic
73                  LOGGER.warn("catch un-synced message from later source (possible reason: connection or performance problems): " + messageId + ", text=" + message);
74                  return;
75              }
76              this.lastMessageId = messageId;
77              this.lastOptions = options;
78              this.mode = mode;
79          }
80  
81          this.helper.setBasicMessage(message);
82          this.helper.setOriginalId(null); // reference to the feedback causing ability
83          String lblText = addAdditionalText(message, options);
84          this.helper.setTextArea(lblText);
85  
86  
87          switch (this.mode) {
88              case INFORM:
89                  setButtonState("", "", mode);
90                  break;
91              case QUESTION:
92                  setButtonState("Yes", "No", mode);
93                  if (options != null && options.containsKey(ORIGINAL_ID)) {
94                      // allows yes/no auto-answers for ability related
95                      this.helper.setOriginalId((UUID) options.get(ORIGINAL_ID));
96                  }
97                  break;
98              case CONFIRM:
99                  setButtonState("OK", "Cancel", mode);
100                 break;
101             case CANCEL:
102                 setButtonState("", "Cancel", mode);
103                 this.helper.setUndoEnabled(false);
104                 break;
105             case SELECT:
106                 setButtonState("", "Done", mode);
107                 break;
108             case END:
109                 setButtonState("", "Close game", mode);
110                 ArrowBuilder.getBuilder().removeAllArrows(gameId);
111                 endWithTimeout();
112                 break;
113         }
114         if (options != null && options.containsKey(SPECIAL_BUTTON)) {
115             this.setSpecial((String) options.get(SPECIAL_BUTTON), true);
116         } else {
117             this.setSpecial("Special", special);
118         }
119 
120         requestFocusIfPossible();
121         updateOptions(options);
122 
123         this.revalidate();
124         this.repaint();
125         this.helper.setLinks(btnLeft, btnRight, btnSpecial, btnUndo);
126 
127         this.helper.setVisible(true);
128         this.helper.setGameNeedFeedback(gameNeedUserFeedback, gameTurnPhase);
129         this.helper.autoSizeButtonsAndFeedbackState();
130     }

132     private void setButtonState(String leftText, String rightText, FeedbackMode mode) {
133         btnLeft.setVisible(!leftText.isEmpty());
134         btnLeft.setText(leftText);
135         btnRight.setVisible(!rightText.isEmpty());
136         btnRight.setText(rightText);
137         this.helper.setState(leftText, !leftText.isEmpty(), rightText, !rightText.isEmpty(), mode);
138     }

140     private String addAdditionalText(String message, Map<String, Serializable> options) {
141         if (options != null && options.containsKey(SECOND_MESSAGE)) {
142             return message + getSmallText((String) options.get(SECOND_MESSAGE));
143         } else {
144             return message;
145         }
146     }

158     /**
159      * Close game window by pressing OK button after 8 seconds
160      */
161     private void endWithTimeout() {
162         Runnable task = () -> {
163             LOGGER.info("Ending game...");
164             Component c = MageFrame.getGame(gameId);
165             while (c != null && !(c instanceof GamePane)) {
166                 c = c.getParent();
167             }
168             if (c != null && c.isVisible()) { // check if GamePanel still visible
169                 FeedbackPanel.this.btnRight.doClick();
170             }
171         };
172         WORKER.schedule(task, 8, TimeUnit.SECONDS);
173     }

175     public void updateOptions(Map<String, Serializable> options) {
176         this.lastOptions = options;
177 
178         if (this.lastOptions != null) {
179             if (this.lastOptions.containsKey("UI.left.btn.text")) {
180                 String text = (String) this.lastOptions.get("UI.left.btn.text");
181                 this.btnLeft.setText(text);
182                 this.helper.setLeft(text, !text.isEmpty());
183             }
184             if (this.lastOptions.containsKey("UI.right.btn.text")) {
185                 String text = (String) this.lastOptions.get("UI.right.btn.text");
186                 this.btnRight.setText(text);
187                 this.helper.setRight(text, !text.isEmpty());
188             }
189             updateConnectedDialog((MageDialog) this.lastOptions.getOrDefault("dialog", null));
190             this.helper.autoSizeButtonsAndFeedbackState();
191         } else {
192             updateConnectedDialog(null);
193         }
194     }

196     private void updateConnectedDialog(MageDialog newDialog) {
197         if (this.connectedDialog != null && this.connectedDialog != newDialog) {
198             // remove old
199             this.connectedDialog.removeDialog();
200         }
201         this.connectedDialog = newDialog;
202         if (this.connectedDialog != null) {
203             this.connectedDialog.setVisible(true);
204         }
205     }

207     // Issue 256: Chat+Feedback panel: request focus prevents players from chatting
208     // Issue #1054: XMage steals window focus whenever the screen updates
209     private void requestFocusIfPossible() {
210         boolean requestFocusAllowed = true;
211         if (MageFrame.getInstance().getFocusOwner() == null) {
212             requestFocusAllowed = false;
213         } else if (connectedChatPanel != null && connectedChatPanel.getTxtMessageInputComponent() != null) {
214             if (connectedChatPanel.getTxtMessageInputComponent().hasFocus()) {
215                 requestFocusAllowed = false;
216             }
217         }
218         if (requestFocusAllowed) {
219             this.btnRight.requestFocus();
220             this.helper.requestFocus();
221         }
222     }

257     private void btnRightActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnRightActionPerformed
258         updateConnectedDialog(null);
259         if (mode == FeedbackMode.SELECT && (evt.getModifiers() & ActionEvent.CTRL_MASK) == ActionEvent.CTRL_MASK) {
260             SessionHandler.sendPlayerInteger(gameId, 0);
261         } else if (mode == FeedbackMode.END) {
262             GamePanel gamePanel = MageFrame.getGame(gameId);
263             if (gamePanel != null) {
264                 gamePanel.removeGame();
265             }
266         } else {
267             SessionHandler.sendPlayerBoolean(gameId, false);
268         }
269         //AudioManager.playButtonOk();
270     }//GEN-LAST:event_btnRightActionPerformed

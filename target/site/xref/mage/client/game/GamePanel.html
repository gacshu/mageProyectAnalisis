586     private void addPlayers(GameView game) {
587         this.players.clear();
588         this.playersWhoLeft.clear();
589         this.pnlBattlefield.removeAll();
590         //arrange players in a circle with the session player at the bottom left
591         int numSeats = game.getPlayers().size();
592         int numColumns = (numSeats + 1) / 2;
593         boolean oddNumber = (numColumns > 1 && numSeats % 2 == 1);
594         int col = 0;
595         int row = 1;
596         int playerSeat = 0;
597         if (playerId != null) {
598             for (PlayerView player : game.getPlayers()) {
599                 if (playerId.equals(player.getPlayerId())) {
600                     break;
601                 }
602                 playerSeat++;
603             }
604         }
605         PlayerView player = game.getPlayers().get(playerSeat);
606         PlayAreaPanel playAreaPanel = new PlayAreaPanel(player, bigCard, gameId, game.getPriorityTime(), this,
607                 new PlayAreaPanelOptions(game.isPlayer(), player.isHuman(), game.isPlayer(), game.isRollbackTurnsAllowed(), row == 0));
608         players.put(player.getPlayerId(), playAreaPanel);
609         playersWhoLeft.put(player.getPlayerId(), false);
610         GridBagConstraints c = new GridBagConstraints();
611         c.fill = GridBagConstraints.BOTH;
612         c.weightx = 0.5;
613         c.weighty = 0.5;
614         if (oddNumber) {
615             c.gridwidth = 2;
616         }
617         c.gridx = col;
618         c.gridy = 0;
619 
620         // Top panel (row=0)
621         JPanel topPanel = new JPanel();
622         topPanel.setOpaque(false);
623 
624         // Bottom panel (row=1)
625         JPanel bottomPanel = new JPanel();
626         bottomPanel.setOpaque(false);
627         topPanel.setLayout(new GridBagLayout());
628         bottomPanel.setLayout(new GridBagLayout());
629 
630         bottomPanel.add(playAreaPanel, c);
631         playAreaPanel.setVisible(true);
632         if (oddNumber) {
633             col++;
634         }
635         int playerNum = playerSeat + 1;
636         if (playerNum >= numSeats) {
637             playerNum = 0;
638         }
639         while (true) {
640             if (row == 1) {
641                 col++;
642             } else {

2899     public void handleEvent(AWTEvent event) {
2900         if (event instanceof InputEvent) {
2901             int id = event.getID();
2902             boolean isActionEvent = false;
2903             if (id == MouseEvent.MOUSE_PRESSED) {
2904                 isActionEvent = true;
2905                 // clear chat focus on click
2906                 if (event instanceof MouseEvent) {
2907                     MouseEvent me = (MouseEvent) event;
2908                     if (isChatInputActive() && !isChatInputUnderCursor(me.getPoint())) {
2909                         KeyboardFocusManager.getCurrentKeyboardFocusManager().clearFocusOwner();
2910                     }
2911                 }
2912             } else if (id == KeyEvent.KEY_PRESSED) {
2913                 KeyEvent key = (KeyEvent) event;
2914                 int keyCode = key.getKeyCode();
2915                 if (keyCode == KeyEvent.VK_ENTER || keyCode == KeyEvent.VK_SPACE) {
2916                     isActionEvent = true;
2917                 }
2918             }
2919             if (isActionEvent) {
2920                 InputEvent input = (InputEvent) event;
2921                 if ((input.getModifiersEx() & holdPriorityMask) != 0) {
2922                     setMenuStates(
2923                             PreferencesDialog.getCachedValue(KEY_GAME_MANA_AUTOPAYMENT, "true").equals("true"),
2924                             PreferencesDialog.getCachedValue(KEY_GAME_MANA_AUTOPAYMENT_ONLY_ONE, "true").equals("true"),
2925                             PreferencesDialog.getCachedValue(KEY_USE_FIRST_MANA_ABILITY, "false").equals("true"),
2926                             true);
2927                     holdPriority(true);
2928                 }
2929             }
2930         }
2931     }

109     public void cleanUp() {
110         gameUpdateTimer.stop();
111 
112         for (Component c : this.jPanel.getComponents()) {
113             if (c instanceof MageCard) {
114                 if (((MageCard) c).getMainPanel() instanceof MagePermanent) {
115                     this.jPanel.remove(c);
116                 }
117             }
118         }
119         permanents.clear();
120         this.bigCard = null;

143     public void update(Map<UUID, PermanentView> battlefield) {
144         gameUpdateTimer.stop();
145         boolean changed = false;
146 
147         List<PermanentView> permanentsToAdd = new ArrayList<>();
148         for (PermanentView permanent : battlefield.values()) {
149             if (!permanent.isPhasedIn()) {
150                 continue;
151             }
152             MageCard oldFound = permanents.get(permanent.getId());
153             MagePermanent oldMagePermanent = oldFound == null ? null : (MagePermanent) oldFound.getMainPanel();
154             if (oldMagePermanent == null) {
155                 permanentsToAdd.add(permanent);
156                 changed = true;
157             } else {
158                 if (!changed) {
159                     changed = oldMagePermanent.getOriginalPermanent().isCreature() != permanent.isCreature();
160                     // Check if there was a change in the permanets that are the permanent attached to
161                     if (!changed) {
162                         int attachments = permanent.getAttachments() == null ? 0 : permanent.getAttachments().size();
163                         int attachmentsBefore = oldMagePermanent.getLinks().size();
164                         if (attachments != attachmentsBefore) {
165                             changed = true;
166                         } else if (attachments > 0) {
167                             Set<UUID> attachmentIds = new HashSet<>(permanent.getAttachments());
168                             for (MageCard mageCard : oldMagePermanent.getLinks()) {
169                                 MagePermanent magePermanent = (MagePermanent) mageCard.getMainPanel();
170                                 if (!attachmentIds.contains(magePermanent.getOriginalPermanent().getId())) {
171                                     // that means that the amount of attachments is the same
172                                     // but they are different:
173                                     // we've just found an attachment on previous view
174                                     // that doesn't exist anymore on current view
175                                     changed = true;
176                                     break;
177                                 }
178                             }
179                         }
180                     }
181                     // Check if permanents it now attached to another or no permanent
182                     if (!changed) {
183                         UUID attachedToIdBefore = oldMagePermanent.getOriginalPermanent().getAttachedTo();
184                         UUID attachedToId = permanent.getAttachedTo();
185                         if (attachedToIdBefore == null && attachedToId != null || attachedToId == null && attachedToIdBefore != null
186                                 || (attachedToIdBefore != null && !attachedToIdBefore.equals(attachedToId))) {
187                             changed = true;
188                         }
189                     }
190                     // Check for changes in the counters of the permanent
191                     if (!changed) {
192                         List<CounterView> counters1 = oldMagePermanent.getOriginalPermanent().getCounters();
193                         List<CounterView> counters2 = permanent.getCounters();
194                         if (counters1 == null && counters2 != null || counters1 != null && counters2 == null) {
195                             changed = true;
196                         } else if (counters1 != null && counters2 != null && counters1.size() != counters2.size()) {
197                             changed = true;
198                         }
199                     }
200                 }
201                 oldMagePermanent.update(permanent);
202             }
203         }
204 
205         addedArtifact = addedCreature = addedPermanent = false;
206 
207         int count = permanentsToAdd.size();
208         for (PermanentView permanent : permanentsToAdd) {
209             addPermanent(permanent, count);
210         }
211 
212         if (addedArtifact) {
213             AudioManager.playAddArtifact();
214         } else if (addedCreature) {
215             AudioManager.playSummon();
216         } else if (addedPermanent) {
217             AudioManager.playAddPermanent();
218         }
219 
220         removedCreature = false;
221 
222         for (Iterator<Entry<UUID, MageCard>> iterator = permanents.entrySet().iterator(); iterator.hasNext(); ) {
223             Entry<UUID, MageCard> entry = iterator.next();
224             if (!battlefield.containsKey(entry.getKey()) || !battlefield.get(entry.getKey()).isPhasedIn()) {
225                 removePermanent(entry.getKey(), 1);
226                 iterator.remove();
227                 changed = true;
228             }
229         }
230 
231         if (removedCreature) {
232             AudioManager.playDiedCreature();
233         }
234 
235         if (changed) {
236             this.battlefield = battlefield;
237             sortLayout();
238         }
239     }

241     public void sortLayout() {
242         if (battlefield == null || this.getWidth() < 1) { // Can't do layout when panel is not sized yet
243             return;
244         }
245 
246         layoutStrategy.doLayout(this, this.getWidth());
247 
248         this.jScrollPane.repaint();
249         this.jScrollPane.revalidate();
250 
251         invalidate();
252         repaint();
253     }

255     private void addPermanent(PermanentView permanent, final int count) {
256         if (cardDimension == null) {
257             cardDimension = new Dimension(ClientDefaultSettings.dimensions.getFrameWidth(), ClientDefaultSettings.dimensions.getFrameHeight());
258         }
259         final MageCard perm = Plugins.instance.getMagePermanent(permanent, bigCard, new CardIconRenderSettings(), cardDimension, gameId, true, PreferencesDialog.getRenderMode(), true);
260         perm.setCardContainerRef(jPanel);
261         perm.update(permanent);
262         // cards sizes changes in parent call by sortLayout
263         //perm.setCardBounds
264 
265         permanents.put(permanent.getId(), perm);
266 
267         this.jPanel.add(perm, (Integer) 10);
268         moveToFront(jPanel);
269         Plugins.instance.onAddCard(perm, 1);
270 
271         if (permanent.isArtifact()) {
272             addedArtifact = true;
273         } else if (permanent.isCreature()) {
274             addedCreature = true;
275         } else {
276             addedPermanent = true;
277         }
278     }

280     private void removePermanent(UUID permanentId, final int count) {
281         for (Component comp : this.jPanel.getComponents()) {
282             if (comp instanceof MageCard) {
283                 MageCard mageCard = (MageCard) comp;
284                 if (mageCard.getMainPanel() instanceof MagePermanent) {
285                     MagePermanent magePermanent = (MagePermanent) mageCard.getMainPanel();
286                     if (magePermanent.getOriginal().getId().equals(permanentId)) {
287                         Thread t = new Thread(() -> {
288                             Plugins.instance.onRemoveCard(mageCard, count);
289                             mageCard.setVisible(false);
290                             this.jPanel.remove(mageCard);
291                         });
292                         t.start();
293                     }
294                     if (magePermanent.getOriginal().isCreature()) {
295                         removedCreature = true;
296                     }
297                 }
298             }
299         }
300     }

311     private void initComponents() {
312         setOpaque(false);
313 
314         jPanel = new JLayeredPane();
315         jPanel.setLayout(null);
316         jPanel.setOpaque(false);
317         jScrollPane = new JScrollPane(jPanel);
318         if (DebugUtil.GUI_GAME_DRAW_BATTLEFIELD_BORDER) {
319             jPanel.setBorder(BorderFactory.createLineBorder(Color.MAGENTA));
320             jScrollPane.setBorder(BorderFactory.createLineBorder(Color.green));
321         }
322 
323         Border empty = new EmptyBorder(0, 0, 0, 0);
324         jScrollPane.setBorder(empty);
325         jScrollPane.setViewportBorder(empty);
326         jScrollPane.setOpaque(false);
327         jScrollPane.getViewport().setOpaque(false);
328         jScrollPane.getVerticalScrollBar().addAdjustmentListener(new AdjustmentListener() {
329             @Override
330             public void adjustmentValueChanged(AdjustmentEvent e) {
331                 if (gameUpdateTimer.isRunning()) {
332                     gameUpdateTimer.restart();
333                 } else {
334                     gameUpdateTimer.start();
335                 }
336             }
337         });
338 
339         this.add(jScrollPane);
340     }

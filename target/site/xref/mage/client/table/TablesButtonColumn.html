5   import javax.swing.*;
6   import javax.swing.table.TableCellEditor;

9   import java.awt.*;
10  import java.awt.event.ActionEvent;
11  import java.awt.event.ActionListener;
12  import java.awt.event.MouseEvent;
13  import java.awt.event.MouseListener;
14  
15  /**
16   * @author BetaSteward_at_googlemail.com
17   */
18  public class TablesButtonColumn extends AbstractCellEditor implements TableCellRenderer, TableCellEditor, ActionListener, MouseListener {
19  
20      private final JTable table;
21      private final Action action;
22      private final JButton renderButton;
23      private final JButton editButton;
24      private String text;
25      private boolean isButtonColumnEditor;
26  
27      public TablesButtonColumn(JTable table, Action action, int column) {
28          super();
29          this.table = table;
30          this.action = action;
31          renderButton = new JButton();
32  
33          editButton = new JButton();
34          editButton.setFocusPainted(false);
35          editButton.addActionListener(this);
36          setGUISize();
37  
38          TableColumnModel columnModel = table.getColumnModel();
39          columnModel.getColumn(column).setCellRenderer(this);
40          columnModel.getColumn(column).setCellEditor(this);
41          table.addMouseListener(this);
42      }
43  
44      public void changeGUISize() {
45          setGUISize();
46      }
47  
48      private void setGUISize() {
49          renderButton.setFont(GUISizeHelper.tableFont);
50          editButton.setFont(GUISizeHelper.tableFont);
51      }
52  
53      @Override
54      public Object getCellEditorValue() {
55          return text;
56      }
57  
58      @Override
59      public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
60          if (hasFocus) {
61              renderButton.setForeground(table.getForeground());
62              renderButton.setBackground(UIManager.getColor("Button.background"));
63          } else if (isSelected) {
64              renderButton.setForeground(table.getSelectionForeground());
65              renderButton.setBackground(table.getSelectionBackground());
66          } else {
67              renderButton.setForeground(table.getForeground());
68              renderButton.setBackground(UIManager.getColor("Button.background"));
69          }
70  
71          renderButton.setText((value == null) ? "" : value.toString());
72          return renderButton;
73      }
74  
75      @Override
76      public Component getTableCellEditorComponent(JTable table, Object value, boolean isSelected, int row, int column) {
77          text = (value == null) ? "" : value.toString();
78          editButton.setText(text);
79          return editButton;
80      }
81  
82      @Override
83      public void actionPerformed(ActionEvent e) {
84          if (table.getRowCount() > 0 && table.getRowCount() >= table.getEditingRow() && table.getEditingRow() >= 0) {
85              int row = table.convertRowIndexToModel(table.getEditingRow());
86              fireEditingStopped();
87              ActionEvent event = new ActionEvent(table, ActionEvent.ACTION_PERFORMED, TablesUtil.getSearchIdFromTable(table, row));
88              action.actionPerformed(event);
89          }
90      }
91  
92      @Override
93      public void mousePressed(MouseEvent arg0) {
94          if (table.isEditing() && table.getCellEditor() == this) {
95              isButtonColumnEditor = true;
96          }
97      }
98  
99      @Override
100     public void mouseReleased(MouseEvent arg0) {
101         if (isButtonColumnEditor && table.isEditing()) {
102             table.getCellEditor().stopCellEditing();
103         }
104 
105         isButtonColumnEditor = false;
106     }
107 
108     @Override
109     public void mouseClicked(MouseEvent arg0) {
110     }
111 
112     @Override
113     public void mouseEntered(MouseEvent arg0) {
114     }
115 
116     @Override
117     public void mouseExited(MouseEvent arg0) {
118     }
119 
120 }

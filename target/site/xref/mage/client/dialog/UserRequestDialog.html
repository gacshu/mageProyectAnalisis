54      public void showDialog(UserRequestMessage userRequestMessage) {
55          this.userRequestMessage = userRequestMessage;
56          this.setTitle(userRequestMessage.getTitle());
57          String text = "<html><p style=\"text-align:center; margin-left:10px; margin-right:10px\">" + userRequestMessage.getMessage() + "</p></html>";
58          this.lblText.setText(text);
59          if (userRequestMessage.getButton1Text() != null) {
60              this.btn1.setText(userRequestMessage.getButton1Text());
61              this.btn1.setFocusable(false);
62          } else {
63              this.btn1.setVisible(false);
64          }
65          if (userRequestMessage.getButton2Text() != null) {
66              this.btn2.setText(userRequestMessage.getButton2Text());
67              this.btn2.setFocusable(false);
68          } else {
69              this.btn2.setVisible(false);
70          }
71          if (userRequestMessage.getButton3Text() != null) {
72              this.btn3.setText(userRequestMessage.getButton3Text());
73              this.btn3.setFocusable(false);
74          } else {
75              this.btn3.setVisible(false);
76          }
77          this.pack();
78          this.revalidate();
79          this.repaint();
80          this.setModal(true);
81          this.setVisible(true);
82      }

36      public void showDialog(UUID roomId, UUID tableId, boolean isTournament, boolean isLimited) {

120     private void btnOKActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnOKActionPerformed
121         Session session = SessionHandler.getSession();
122         try {
123             PreferencesDialog.saveValue(PreferencesDialog.KEY_NEW_TABLE_PASSWORD_JOIN, txtPassword.getText());
124             if (isTournament) {
125                 joined = session.joinTournamentTable(roomId, tableId, this.newPlayerPanel.getPlayerName(), PlayerType.HUMAN, 1, DeckImporter.importDeckFromFile(this.newPlayerPanel.getDeckFile(), true), this.txtPassword.getText());
126             } else {
127                 joined = session.joinTable(roomId, tableId, this.newPlayerPanel.getPlayerName(), PlayerType.HUMAN, 1, DeckImporter.importDeckFromFile(this.newPlayerPanel.getDeckFile(), true), this.txtPassword.getText());
128             }
129 
130         } catch (Exception ex) {
131             handleError(ex);
132         }
133         this.hideDialog();
134     }//GEN-LAST:event_btnOKActionPerformed

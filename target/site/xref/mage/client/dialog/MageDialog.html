35      public static boolean isModalDialogActivated() {
36          for (JInternalFrame frame : MageFrame.getDesktop().getAllFrames()) {
37              if (frame instanceof MageDialog) {
38                  MageDialog md = (MageDialog) frame;
39                  if (md.isVisible() && md.isModal()) {
40                      return true;
41                  }
42              }
43          }
44          return false;
45      }

63      @Override
64      public void show() {
65          super.show();
66  
67          // frames desktop ordering
68          // more info https://docs.oracle.com/javase/7/docs/api/javax/swing/JLayeredPane.html
69          // WARNING, use
70          // - JLayeredPane.DEFAULT_LAYER: tables and games (tabs)
71          // - JLayeredPane.PALETTE_LAYER: toolbars and info windows like cards list, not modal dialogs (not required user actions)
72          // - JLayeredPane.MODAL_LAYER: all modal dialogs (user required actions - select cards in game, new game window, error windows)
73          // - JLayeredPane.POPUP_LAYER: hints and other top level graphics
74          // - JLayeredPane.DRAG_LAYER: top most layer for critical actions and user controls
75  
76          if (modal) {
77              this.setClosable(false);
78          }
79  
80          this.toFront();
81  
82          if (modal) {
83              startModal();
84          }
85      }

87      @Override
88      public void setVisible(boolean value) {
89          super.setVisible(value);
90  
91          if (value) {
92              this.toFront();
93              try {
94                  this.setSelected(true);
95              } catch (PropertyVetoException e) {
96                  //
97              }
98          }
99  
100         if (modal) {
101             this.setClosable(false);
102             if (value) {
103                 startModal();
104             } else if (SwingUtilities.isEventDispatchThread()) {
105                 stopModal();
106             } else {
107                 try {
108                     SwingUtilities.invokeAndWait(() -> stopModal());
109                 } catch (InterruptedException ex) {
110                     LOGGER.fatal("MageDialog error", ex);
111                     Thread.currentThread().interrupt();
112                 } catch (InvocationTargetException ex) {
113                     LOGGER.fatal("MageDialog error", ex);
114                 }
115             }
116         }
117     }

119     private synchronized void startModal() {
120         // modal loop -- all mouse events must be ignored by other windows
121         try {
122             if (SwingUtilities.isEventDispatchThread()) {
123                 EventQueue theQueue = getToolkit().getSystemEventQueue();
124                 while (isVisible()) {
125                     AWTEvent event = theQueue.getNextEvent();
126                     Object source = event.getSource();
127                     boolean dispatch = true;
128 
129                     // https://github.com/magefree/mage/issues/584 - Let's hope this will fix the Linux window problem
130                     if (event.getSource() != null && event.getSource() instanceof TrayIcon && !(event instanceof InvocationEvent)) {
131                         dispatch = false;
132                         //return; // JayDi85: users can move mouse over try icon to disable modal mode (it's a bug but can be used in the future)
133                     }
134 
135                     // ignore mouse events outside from panel, only drag and move allowed -- as example:
136                     // combobox's popup will be selectable outside
137                     // cards and button hints will be works
138                     Component popupComponent = null;
139                     MouseEvent popupEvent = null;
140                     if (event instanceof MouseEvent && event.getSource() instanceof Component) {
141                         MouseEvent e = (MouseEvent) event;
142                         MouseEvent m = SwingUtilities.convertMouseEvent((Component) e.getSource(), e, this);
143 
144                         // disable all outer events (except some actions)
145                         if (!this.contains(m.getPoint())) {
146                             boolean allowedEvent = false;
147 
148                             // need any mouse move (for hints)
149                             if (e.getID() == MouseEvent.MOUSE_DRAGGED || e.getID() == MouseEvent.MOUSE_MOVED) {
150                                 allowedEvent = true;
151                             }
152 
153                             // need popup clicks and mouse wheel (for out of bound actions)
154                             if (!allowedEvent) {
155                                 popupComponent = SwingUtilities.getDeepestComponentAt(e.getComponent(), e.getX(), e.getY()); // show root component (popups creates at root)
156                                 if (popupComponent != null && (popupComponent.getClass().getName().contains("BasicComboPopup")
157                                         || popupComponent.getClass().getName().contains("JMenuItem"))) {
158                                     popupEvent = SwingUtilities.convertMouseEvent((Component) e.getSource(), e, popupComponent);
159                                     allowedEvent = true;
160                                 }
161                             }
162 
163                             dispatch = allowedEvent;
164                         }
165                     }
166 
167                     if (dispatch) {
168                         if (popupEvent != null) {
169                             // process outer popup events, it's must be FIRST check
170                             popupComponent.dispatchEvent(popupEvent);
171                         } else if (event instanceof ActiveEvent) {
172                             ((ActiveEvent) event).dispatch();
173                         } else if (source instanceof Component) {
174                             ((Component) source).dispatchEvent(event);
175                         } else if (source instanceof MenuComponent) {
176                             ((MenuComponent) source).dispatchEvent(event);
177                         } else {
178                             LOGGER.warn("Unable to dispatch: " + event);
179                         }
180                     }
181                 }
182             } else {
183                 while (isVisible()) {
184                     wait();
185                 }
186             }
187         } catch (InterruptedException e) {
188             LOGGER.fatal("MageDialog error", e);
189             Thread.currentThread().interrupt();
190         }
191     }

209     public void removeDialog() {
210         // avoid memory leak of javax.swing.plaf.nimbus.NimbusStyle$CacheKey
211         KeyboardFocusManager.getCurrentKeyboardFocusManager().clearGlobalFocusOwner();
212         //this.setVisible(false);
213         // important to set close before removing the JInternalFrame to avoid memory leaks (http://bugs.java.com/view_bug.do?bug_id=7163808)
214         try {
215             this.setClosed(true);
216         } catch (PropertyVetoException ex) {
217             LOGGER.error("setClosed(false) failed", ex);
218         }
219         MageFrame.getDesktop().remove(this);
220     }

229         GuiDisplayUtil.keepComponentInsideScreen(centered.x, centered.y, component);

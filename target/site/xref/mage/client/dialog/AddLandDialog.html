44      public void showDialog(Deck deck, DeckEditorMode mode) {
45          this.deck = deck;
46          this.mode = mode;
47          SortedSet<String> landSetNames = new TreeSet<>();
48          String defaultSetName = null;
49          if (mode != DeckEditorMode.FREE_BUILDING) {
50              // decide from which sets basic lands are taken from
51              for (String setCode : deck.getExpansionSetCodes()) {
52                  ExpansionInfo expansionInfo = ExpansionRepository.instance.getSetByCode(setCode);
53                  if (expansionInfo != null && expansionInfo.hasBasicLands() && !CardRepository.haveSnowLands(expansionInfo.getCode())) {
54                      defaultSetName = expansionInfo.getName();
55                      break;
56                  }
57              }
58  
59              // if sets have no basic land, take land from block
60              if (defaultSetName == null) {
61                  for (String setCode : deck.getExpansionSetCodes()) {
62                      ExpansionInfo expansionInfo = ExpansionRepository.instance.getSetByCode(setCode);
63                      if (expansionInfo != null) {
64                          List<ExpansionInfo> blockSets = ExpansionRepository.instance.getSetsFromBlock(expansionInfo.getBlockName());
65                          for (ExpansionInfo blockSet : blockSets) {
66                              if (blockSet.hasBasicLands() && !CardRepository.haveSnowLands(expansionInfo.getCode())) {
67                                  defaultSetName = expansionInfo.getName();
68                                  break;
69                              }
70                          }
71                      }
72                  }
73              }
74          }
75  
76          // if still no set with lands found, add list of all available
77          List<ExpansionInfo> basicLandSets = ExpansionRepository.instance.getSetsWithBasicLandsByReleaseDate();
78          for (ExpansionInfo expansionInfo : basicLandSets) {
79              // snow lands only in free mode
80              if (mode != DeckEditorMode.FREE_BUILDING && CardRepository.haveSnowLands(expansionInfo.getCode())) {
81                  continue;
82              }
83              landSetNames.add(expansionInfo.getName());
84          }
85          if (landSetNames.isEmpty()) {
86              throw new IllegalArgumentException("No set with basic land was found");
87          }
88          if (landSetNames.size() > 1) {
89              landSetNames.add("<Random lands>");
90          }
91          cbLandSet.setModel(new DefaultComboBoxModel(landSetNames.toArray()));
92          if (defaultSetName != null) {
93              String item;
94              for (int i = 0; i < cbLandSet.getItemCount(); i++) {
95                  item = (String) cbLandSet.getItemAt(i);
96                  if (item.equalsIgnoreCase(defaultSetName)) {
97                      cbLandSet.setSelectedIndex(i);
98                      break;
99                  }
100             }
101 
102         }
103 
104         spnDeckSize.setValue(DEFAULT_SEALED_DECK_CARD_NUMBER);
105         BufferedImage image = ManaSymbols.getSizedManaSymbol("G", 15);
106         if (image != null) {
107             lblForestIcon.setIcon(new ImageIcon(image));
108         }
109         image = ManaSymbols.getSizedManaSymbol("U", 15);
110         if (image != null) {
111             lblIslandIcon.setIcon(new ImageIcon(image));
112         }
113         image = ManaSymbols.getSizedManaSymbol("W", 15);
114         if (image != null) {
115             lblPlainsIcon.setIcon(new ImageIcon(image));
116         }
117         image = ManaSymbols.getSizedManaSymbol("R", 15);
118         if (image != null) {
119             lblMountainIcon.setIcon(new ImageIcon(image));
120         }
121         image = ManaSymbols.getSizedManaSymbol("B", 15);
122         if (image != null) {
123             lblSwampIcon.setIcon(new ImageIcon(image));
124         }
125 
126         getRootPane().setDefaultButton(btnOK);
127 
128         // windows settings
129         MageFrame.getDesktop().remove(this);
130         if (this.isModal()) {
131             MageFrame.getDesktop().add(this, JLayeredPane.MODAL_LAYER);
132         } else {
133             MageFrame.getDesktop().add(this, JLayeredPane.PALETTE_LAYER);
134         }
135         this.makeWindowCentered();
136 
137         // Close on "ESC"
138         registerKeyboardAction(e -> onCancel(), KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
139 
140         this.setVisible(true);
141     }

143     private void addLands(String landName, int number, boolean useFullArt) {
144         String landSetName = (String) cbLandSet.getSelectedItem();
145         ExpansionInfo expansionInfo = null;
146 
147         CardCriteria criteria = new CardCriteria();
148         if (!landSetName.equals("<Random lands>")) {
149             expansionInfo = ExpansionRepository.instance.getSetByName(landSetName);
150             if (expansionInfo == null) {
151                 throw new IllegalArgumentException("Code of Set " + landSetName + " not found");
152             }
153             criteria.setCodes(expansionInfo.getCode());
154         } else {
155             criteria.ignoreSetsWithSnowLands();
156         }
157         if (mode == DeckEditorMode.FREE_BUILDING && expansionInfo != null && CardRepository.haveSnowLands(expansionInfo.getCode())) {
158             criteria.name(landName); // snow basics added only if in free mode and the chosen set has exclusively snow basics
159         } else {
160             criteria.nameExact(landName);
161         }
162         criteria.rarities(Rarity.LAND);        
163         List<CardInfo> cards = CardRepository.instance.findCards(criteria);
164         if (cards.isEmpty()) {
165             logger.error("No basic lands found in Set: " + landSetName);
166             criteria = new CardCriteria();
167             criteria.rarities(Rarity.LAND).name(landName);
168             criteria.setCodes("M15");
169             cards = CardRepository.instance.findCards(criteria);
170         }
171 
172         int foundLands = 0;
173         int foundNoneAfter = 0;
174         for (int i = 0; foundLands != number && foundNoneAfter < 1000; i++) {
175             Card land = cards.get(RandomUtil.nextInt(cards.size())).getMockCard();
176             boolean useLand = !useFullArt;
177             if (useFullArt && (land.getFrameStyle() == FrameStyle.BFZ_FULL_ART_BASIC
178                     || land.getFrameStyle() == FrameStyle.UGL_FULL_ART_BASIC
179                     || land.getFrameStyle() == FrameStyle.UNH_FULL_ART_BASIC
180                     || land.getFrameStyle() == FrameStyle.UST_FULL_ART_BASIC
181                     || land.getFrameStyle() == FrameStyle.ZEN_FULL_ART_BASIC)) {
182                 useLand = true;
183             }
184             if (useLand) {
185                 deck.getCards().add(land);
186                 foundLands++;
187                 foundNoneAfter = 0;
188             } else {
189                 foundNoneAfter++;
190             }
191         }
192 
193         if (foundNoneAfter >= 1000 && useFullArt) {
194             MageFrame.getInstance().showMessage("Unable to add enough " + landName + "s.  You encountered an error in adding chosen lands.  Unable to find enough full art lands in the set " + landSetName + ".");
195         }
196     }

202     private void onOK() {
203         int nForest = ((Number) spnForest.getValue()).intValue();
204         int nIsland = ((Number) spnIsland.getValue()).intValue();
205         int nMountain = ((Number) spnMountain.getValue()).intValue();
206         int nPlains = ((Number) spnPlains.getValue()).intValue();
207         int nSwamp = ((Number) spnSwamp.getValue()).intValue();
208         boolean useFullArt = ckbFullArtLands.isSelected();
209 
210         addLands("Forest", nForest, useFullArt);
211         addLands("Island", nIsland, useFullArt);
212         addLands("Mountain", nMountain, useFullArt);
213         addLands("Plains", nPlains, useFullArt);
214         addLands("Swamp", nSwamp, useFullArt);
215 
216         this.removeDialog();
217     }

473     private void autoAddLands() {
474         int red = 0;
475         int green = 0;
476         int black = 0;
477         int blue = 0;
478         int white = 0;
479         Set<Card> cards = deck.getCards();
480         int land_number = ((Number) spnDeckSize.getValue()).intValue() - cards.size();
481         if (land_number < 0) {
482             land_number = 0;
483         }
484         for (Card cd : cards) {
485             for (String s : cd.getManaCostSymbols()) {
486                 if (s.contains("W")) white++;
487                 if (s.contains("U")) blue++;
488                 if (s.contains("B")) black++;
489                 if (s.contains("R")) red++;
490                 if (s.contains("G")) green++;
491             }
492         }
493         int total = red + green + black + blue + white;
494 
495         int redcards = 0;
496         int greencards = 0;
497         int blackcards = 0;
498         int bluecards = 0;
499         int whitecards = 0;
500         if (total > 0) {
501             redcards = Math.round(land_number * ((float) red / (float) total));
502             total -= red;
503             land_number -= redcards;
504 
505             greencards = Math.round(land_number * ((float) green / (float) total));
506             total -= green;
507             land_number -= greencards;
508 
509             blackcards = Math.round(land_number * ((float) black / (float) total));
510             total -= black;
511             land_number -= blackcards;
512 
513             bluecards = Math.round(land_number * ((float) blue / (float) total));
514             total -= blue;
515             land_number -= bluecards;
516 
517             whitecards = land_number;
518         }
519 
520         spnMountain.setValue(redcards);
521         spnForest.setValue(greencards);
522         spnSwamp.setValue(blackcards);
523         spnIsland.setValue(bluecards);
524         spnPlains.setValue(whitecards);
525     }

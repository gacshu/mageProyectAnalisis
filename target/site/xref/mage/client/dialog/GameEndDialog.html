34       public GameEndDialog(GameEndView gameEndView) {
35           initComponents();
36           this.modal = true;
37  
38           pnlText.setOpaque(true);
39           pnlText.setBackground(new Color(240, 240, 240, 140));
40  
41           Rectangle r = new Rectangle(610, 250);
42  
43           Image image;
44           if (gameEndView.hasWon()) {
45               image = ImageHelper.getImageFromResources(PreferencesDialog.getCurrentTheme().getWinlossPath("game_won.jpg"));
46           } else {
47               image = ImageHelper.getImageFromResources(PreferencesDialog.getCurrentTheme().getWinlossPath("game_lost.jpg"));
48           }
49  
50           BufferedImage imageResult = ImageHelper.getResizedImage(BufferedImageBuilder.bufferImage(image, BufferedImage.TYPE_INT_ARGB), r);
51           ImageIcon icon = new ImageIcon(imageResult);
52           lblResultImage.setIcon(icon);
53  
54           this.lblGameInfo.setText(gameEndView.getGameInfo());
55           this.lblMatchInfo.setText(gameEndView.getMatchInfo());
56           this.lblAdditionalInfo.setText(gameEndView.getAdditionalInfo());
57  
58           String autoSave = PreferencesDialog.getCachedValue(PreferencesDialog.KEY_GAME_LOG_AUTO_SAVE, "true");
59           if (autoSave.equals("true")) {
60               this.saveGameLog(gameEndView);
61           }
62  
63           DateFormat df = DateFormat.getDateTimeInstance();
64  
65           // game duration
66           txtDurationGame.setText(" " + Format.getDuration(gameEndView.getStartTime(), gameEndView.getEndTime()));
67           txtDurationGame.setToolTipText(new StringBuilder(df.format(gameEndView.getStartTime())).append(" - ").append(df.format(gameEndView.getEndTime())).toString());
68  
69           // match duration
70           Calendar cal = Calendar.getInstance();
71           txtDurationMatch.setText(" " + Format.getDuration(gameEndView.getMatchView().getStartTime(), cal.getTime()));
72           txtDurationMatch.setToolTipText(new StringBuilder(df.format(gameEndView.getMatchView().getStartTime())).append(" - ").append(df.format(cal.getTime())).toString());
73  
74           StringBuilder sb = new StringBuilder(" ");
75           for (PlayerView player : gameEndView.getPlayers()) {
76               sb.append(player.getName()).append(" Life: ").append(player.getLife()).append(' ');
77           }
78           this.txtLife.setText(sb.toString());
79  
80           if (gameEndView.hasWon()) {
81               AudioManager.playPlayerWon();
82           } else {
83               AudioManager.playPlayerLost();
84           }
85  
86           txtMatchScore.setText(gameEndView.getMatchView().getResult());
87       }

89       private void saveGameLog(GameEndView gameEndView) {
90           String dir = "gamelogs";
91           File saveDir = new File(dir);
92           //Here comes the existence check
93           if (!saveDir.exists()) {
94               saveDir.mkdirs();
95           }
96           // get game log
97           try {
98               if (gameEndView.getMatchView().getGames().size() > 0) {
99                   GamePanel gamePanel = MageFrame.getGame(gameEndView.getMatchView().getGames().get(gameEndView.getMatchView().getGames().size() - 1));
100                  if (gamePanel != null) {
101                      SimpleDateFormat sdf = new SimpleDateFormat();
102                      sdf.applyPattern("yyyyMMdd_HHmmss");
103                      String fileName = new StringBuilder(dir).append(File.separator)
104                              .append(sdf.format(gameEndView.getStartTime()))
105                              .append('_').append(gameEndView.getMatchView().getGameType())
106                              .append('_').append(gameEndView.getMatchView().getGames().size())
107                              .append(".html").toString();
108                      PrintWriter out = new PrintWriter(fileName);
109                      String log = gamePanel.getGameLog();
110                      log = log.replace("<body>", "<body style=\"background-color:black\">");
111                      log = log.replace("<font color=\"#CCCC33\">", "<br><font color=\"#CCCC33\">"); //The color is TIMESTAMP_COLOR and we can utilize it to add line breaks to new lines
112                      out.print(log);
113                      out.close();
114                  }
115              }
116          } catch (FileNotFoundException ex) {
117              JOptionPane.showMessageDialog(this, "Error while writing game log to file\n\n" + ex, "Error writing gamelog", JOptionPane.ERROR_MESSAGE);
118          }
119 
120      }

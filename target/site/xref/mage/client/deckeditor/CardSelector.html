230     private FilterCard buildFilter() {
231         FilterCard filter = new FilterCard();
232 
233         String name = jTextFieldSearch.getText().trim();
234         filter.add(new CardTextPredicate(name, chkNames.isSelected(), chkTypes.isSelected(), chkRules.isSelected(), chkUnique.isSelected()));
235 
236         if (limited) {
237             List<Predicate<MageObject>> predicates = new ArrayList<>();
238 
239             if (this.tbGreen.isSelected()) {
240                 predicates.add(new ColorPredicate(ObjectColor.GREEN));
241             }
242             if (this.tbRed.isSelected()) {
243                 predicates.add(new ColorPredicate(ObjectColor.RED));
244             }
245             if (this.tbBlack.isSelected()) {
246                 predicates.add(new ColorPredicate(ObjectColor.BLACK));
247             }
248             if (this.tbBlue.isSelected()) {
249                 predicates.add(new ColorPredicate(ObjectColor.BLUE));
250             }
251             if (this.tbWhite.isSelected()) {
252                 predicates.add(new ColorPredicate(ObjectColor.WHITE));
253             }
254             if (this.tbColorless.isSelected()) {
255                 predicates.add(ColorlessPredicate.instance);
256             }
257             filter.add(Predicates.or(predicates));
258 
259             predicates.clear();
260             if (this.tbLand.isSelected()) {
261                 predicates.add(CardType.LAND.getPredicate());
262             }
263             if (this.tbArifiacts.isSelected()) {
264                 predicates.add(CardType.ARTIFACT.getPredicate());
265             }
266             if (this.tbCreatures.isSelected()) {
267                 predicates.add(CardType.CREATURE.getPredicate());
268             }
269             if (this.tbEnchantments.isSelected()) {
270                 predicates.add(CardType.ENCHANTMENT.getPredicate());
271             }
272             if (this.tbInstants.isSelected()) {
273                 predicates.add(CardType.INSTANT.getPredicate());
274             }
275             if (this.tbSorceries.isSelected()) {
276                 predicates.add(CardType.SORCERY.getPredicate());
277             }
278             if (this.tbPlaneswalkers.isSelected()) {
279                 predicates.add(CardType.PLANESWALKER.getPredicate());
280             }
281             filter.add(Predicates.or(predicates));
282 
283             List<String> filteredSets = getFilteredSets();
284             if (!filteredSets.isEmpty()) {
285                 List<Predicate<Card>> expansionPredicates = new ArrayList<>();
286                 for (String setCode : filteredSets) {
287                     expansionPredicates.add(new ExpansionSetPredicate(setCode));
288                 }
289                 filter.add(Predicates.or(expansionPredicates));
290             }
291         }
292 
293         return filter;
294     }

308     private List<String> getFilteredSets() {
309         // empty list - show all sets
310 
311         List<String> res = new ArrayList<>();
312         if (!this.cbExpansionSet.isVisible()) {
313             return res;
314         }
315 
316         if (listCodeSelected.getCheckedIndices().length <= 1) {
317             // single set selected
318             String expansionSelection = this.cbExpansionSet.getSelectedItem().toString();
319             if (!expansionSelection.equals(ConstructedFormats.ALL_SETS)
320                     && !expansionSelection.startsWith(MULTI_SETS_SELECTION_TEXT)) {
321                 res.addAll(ConstructedFormats.getSetsByFormat(expansionSelection));
322             }
323         } else {
324             // multiple sets selected
325             int[] choiseValue = listCodeSelected.getCheckedIndices();
326             ListModel x = listCodeSelected.getModel();
327 
328             for (int itemIndex : choiseValue) {
329                 java.util.List<String> listReceived = ConstructedFormats.getSetsByFormat(x.getElementAt(itemIndex).toString());
330                 listReceived.stream()
331                         .filter(item -> !res.contains(item))
332                         .forEachOrdered(res::add);
333             }
334         }
335 
336         return res;
337     }

339     private CardCriteria buildCriteria() {
340         CardCriteria criteria = new CardCriteria();

342         criteria.blue(this.tbBlue.isSelected());

350             criteria.types(CardType.LAND);
351         }
352         if (this.tbArifiacts.isSelected()) {
353             criteria.types(CardType.ARTIFACT);
354         }
355         if (this.tbCreatures.isSelected()) {
356             criteria.types(CardType.CREATURE);
357         }
358         if (this.tbEnchantments.isSelected()) {
359             criteria.types(CardType.ENCHANTMENT);
360         }
361         if (this.tbInstants.isSelected()) {
362             criteria.types(CardType.INSTANT);
363         }
364         if (this.tbSorceries.isSelected()) {
365             criteria.types(CardType.SORCERY);
366         }
367         if (this.tbPlaneswalkers.isSelected()) {
368             criteria.types(CardType.PLANESWALKER);
369         }
370 
371         if (this.tbCommon.isSelected()) {
372             criteria.rarities(Rarity.COMMON);
373             criteria.rarities(Rarity.LAND);
374         }
375         if (this.tbUncommon.isSelected()) {
376             criteria.rarities(Rarity.UNCOMMON);
377         }
378         if (this.tbRare.isSelected()) {
379             criteria.rarities(Rarity.RARE);
380         }
381         if (this.tbMythic.isSelected()) {
382             criteria.rarities(Rarity.MYTHIC);
383         }
384         if (this.tbSpecial.isSelected()) {
385             criteria.rarities(Rarity.SPECIAL);
386             criteria.rarities(Rarity.BONUS);
387         }
388 
389         List<String> filteredSets = getFilteredSets();
390         if (!filteredSets.isEmpty()) {
391             criteria.setCodes(filteredSets.toArray(new String[0]));
392         }
393 
394         return criteria;
395     }

447     private void filterCards() {
448         FilterCard filter = buildFilter();
449         MageFrame.getDesktop().setCursor(new Cursor(Cursor.WAIT_CURSOR));
450         try {
451 
452             // debug
453             //debugObjectMemorySize("Old cards size", this.currentView.getCardsStore());
454             this.currentView.clearCardsStoreBeforeUpdate();
455 
456             java.util.List<Card> filteredCards = new ArrayList<>();
457 
458             if (chkPennyDreadful.isSelected() && pdAllowed.isEmpty()) {
459                 pdAllowed.putAll(PennyDreadfulLegalityUtil.getLegalCardList());
460             }
461 
462             if (limited) {
463                 for (Card card : cards) {
464                     if (filter.match(card, null)) {
465                         filteredCards.add(card);
466                     }
467                 }
468             } else {
469                 java.util.List<CardInfo> foundCards = CardRepository.instance.findCards(buildCriteria());
470 
471                 for (CardInfo cardInfo : foundCards) {
472                     // filter by penny
473                     if (chkPennyDreadful.isSelected()) {
474                         if (!pdAllowed.containsKey(cardInfo.getName())) {
475                             continue;
476                         }
477                     }
478                     // filter by settings
479                     Card card = cardInfo.getMockCard();
480                     if (!filter.match(card, null)) {
481                         continue;
482                     }
483                     // found
484                     filteredCards.add(card);
485                 }
486             }
487 
488             // force to list mode on too much cards
489             if (currentView instanceof CardGrid && filteredCards.size() > CardGrid.MAX_IMAGES) {
490                 this.toggleViewMode();
491             }
492 
493             // debug
494             //debugObjectMemorySize("New cards size", filteredCards);
495 
496             this.currentView.loadCards(new CardsView(filteredCards), sortSetting, bigCard, null, false);
497             this.cardCount.setText(String.valueOf(filteredCards.size()));
498         } finally {
499             MageFrame.getDesktop().setCursor(new Cursor(Cursor.DEFAULT_CURSOR));
500         }
501     }

1252     private void cbExpansionSetActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_cbExpansionSetActionPerformed
1253         // one item selected by user
1254 
1255         // ignore combobox modifying
1256         if (isSetsFilterLoading) {
1257             return;
1258         }
1259 
1260         // auto-remove unused multi-select item and sync checkboxes
1261         if (!cbExpansionSet.getSelectedItem().toString().startsWith(MULTI_SETS_SELECTION_TEXT)) {
1262             int currentSelection = cbExpansionSet.getSelectedIndex();
1263             isSetsFilterLoading = true;
1264             try {
1265                 // remove multi item and fix selection
1266                 if (cbExpansionSet.getItemAt(0).startsWith(MULTI_SETS_SELECTION_TEXT)) {
1267                     cbExpansionSet.removeItemAt(0);
1268                     currentSelection = Math.max(0, currentSelection - 1);
1269                 }
1270             } finally {
1271                 isSetsFilterLoading = false;
1272             }
1273 
1274             // sync checkboxes
1275             listCodeSelected.uncheckAll();
1276             if (currentSelection > 0) {
1277                 // if not "all" option selected
1278                 // checkbox haven't "all", so use another indexes
1279                 listCodeSelected.setChecked(currentSelection - 1, true);
1280             }
1281         }
1282 
1283         // update data
1284         filterCards();
1285     }//GEN-LAST:event_cbExpansionSetActionPerformed

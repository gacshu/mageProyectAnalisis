214     public void removeCollectionViewer() {
215         this.cleanUp();
216         Component c = this.getParent();
217         while (c != null && !(c instanceof CollectionViewerPane)) {
218             c = c.getParent();
219         }
220         if (c != null) {
221             ((CollectionViewerPane) c).removeFrame();
222         }

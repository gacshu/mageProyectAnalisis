2   package mage.client.deckeditor;
3   
4   import mage.client.constants.Constants.SortBy;
5   import mage.client.dialog.PreferencesDialog;
6   
7   /**
8    * @author LevelX2
9    */
10  public abstract class SortSetting {
11  
12      SortBy sortBy;
13      int sortIndex;
14      boolean ascending;
15      boolean pilesToggle;
16  
17      final String prefSortBy;
18      final String prefSortIndex;
19      final String prefSortAscending;
20      final String prefPilesToggle;
21  
22      public SortSetting(String prefSortBy, String prefSortIndex, String prefSortAscending, String prefPilesToggle) {
23          this.prefSortBy = prefSortBy;
24          this.prefSortIndex = prefSortIndex;
25          this.prefSortAscending = prefSortAscending;
26          this.prefPilesToggle = prefPilesToggle;
27          this.sortBy = SortBy.getByString(PreferencesDialog.getCachedValue(this.prefSortBy, "Color"));
28          try {
29              this.sortIndex = Integer.parseInt(PreferencesDialog.getCachedValue(this.prefSortIndex, "1"));
30          } catch (NumberFormatException e) {
31              this.sortIndex = 2;
32          }
33          this.ascending = PreferencesDialog.getCachedValue(this.prefSortAscending, "1").equals("1");
34          this.pilesToggle = PreferencesDialog.getCachedValue(this.prefPilesToggle, "true").equals("true");
35      }
36  
37      public void setSortBy(SortBy sortBy) {
38          this.sortBy = sortBy;
39          PreferencesDialog.saveValue(prefSortBy, sortBy.toString());
40      }
41  
42      public void setSortIndex(int sortIndex) {
43          this.sortIndex = sortIndex;
44          PreferencesDialog.saveValue(this.prefSortIndex, Integer.toString(sortIndex));
45      }
46  
47      public void setAscending(boolean ascending) {
48          this.ascending = ascending;
49          PreferencesDialog.saveValue(this.prefSortAscending, this.ascending ? "1" : "0");
50      }
51  
52      public void setPilesToggle(boolean pileToggle) {
53          this.pilesToggle = pileToggle;
54          PreferencesDialog.saveValue(this.prefSortAscending, this.pilesToggle ? "true" : "false");
55      }
56  
57      public SortBy getSortBy() {
58          return sortBy;
59      }
60  
61      public int getSortIndex() {
62          return sortIndex;
63      }
64  
65      public boolean isAscending() {
66          return ascending;
67      }
68  
69      public boolean isPilesToggle() {
70          return pilesToggle;
71      }
72  }

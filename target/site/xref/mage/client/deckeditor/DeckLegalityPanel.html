126     public void validateDeck(Deck deck) {
127         // Non game GUI like Deck Editor works with Mock cards (fake cards with texts only), but validate must use
128         // real cards all the time. So convert it here before check.
129 
130         final Deck deckToValidate;
131         if (deck.getCards().stream().noneMatch(this::isMockCard)
132                 && deck.getSideboard().stream().noneMatch(this::isMockCard)) {
133             // contains real cards, e.g. it's a game
134             deckToValidate = deck;
135         } else {
136             // contains mock cards, e.g. it's a Deck Editor
137             try {
138                 deckToValidate = Deck.load(deck.getDeckCardLists(), true, false);
139             } catch (Exception ex) {
140                 logger.error("Can't load real deck cards for validate: " + ex.getMessage(), ex);
141                 return;
142             }
143         }
144 
145         Arrays.stream(getComponents())
146                 .filter(LegalityLabel.class::isInstance)
147                 .map(LegalityLabel.class::cast)
148                 .forEach(label -> label.validateDeck(deckToValidate));
149     }

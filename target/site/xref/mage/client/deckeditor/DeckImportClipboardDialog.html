38      public void showDialog() {
39          this.tmpPath = "";
40          onRefreshClipboard();
41  
42          this.setModal(true);
43          this.setResizable(true);
44          getRootPane().setDefaultButton(buttonOK);
45  
46          // windows settings
47          MageFrame.getDesktop().remove(this);
48          if (this.isModal()) {
49              MageFrame.getDesktop().add(this, JLayeredPane.MODAL_LAYER);
50          } else {
51              MageFrame.getDesktop().add(this, JLayeredPane.PALETTE_LAYER);
52          }
53          this.makeWindowCentered();
54  
55          // Close on "ESC"
56          registerKeyboardAction(e -> onCancel(), KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
57  
58          this.setVisible(true);
59      }

61      private Optional<String> getClipboardStringData() {
62          try {
63              return Optional.of((String) Toolkit.getDefaultToolkit().getSystemClipboard().getData(DataFlavor.stringFlavor));
64          } catch (HeadlessException | UnsupportedFlavorException | IOException e) {
65              //e.printStackTrace();
66          }
67          return Optional.empty();
68      }
69  
70      private void onOK() {
71          String decklist = editData.getText();
72          decklist = decklist.replace(FORMAT_TEXT, "");
73          // This dialog also accepts a paste in .mtga format
74          if (decklist.startsWith("Deck\n")) {  // An .mtga list always starts with the first line being "Deck". This kind of paste is processed as .mtga
75              tmpPath = DeckUtil.writeTextToTempFile("cbimportdeck", ".mtga", decklist);
76          } else {  // If the paste is not .mtga format, it's processed as plaintext
77              tmpPath = DeckUtil.writeTextToTempFile(decklist);
78          }
79          this.removeDialog();
80      }
81  
82      private void onCancel() {
83          this.removeDialog();
84      }
85  
86      private void onRefreshClipboard() {
87          editData.setText(FORMAT_TEXT + getClipboardStringData().orElse(""));
88          editData.setCaretPosition(FORMAT_TEXT.length());
89      }
90  
91      public String getTmpPath() {
92          return tmpPath;
93      }

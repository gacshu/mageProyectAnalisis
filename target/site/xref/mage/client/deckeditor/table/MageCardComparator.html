3   import mage.cards.MageCard;
4   import mage.client.util.comparators.CardViewComparator;
5   import mage.game.draft.RateCard;
6   import mage.view.CardView;
7   import org.apache.log4j.Logger;
8   
9   /**
10   * {@link MageCard} comparator. Used to sort cards in Deck Editor Table View
11   * pane.
12   *
13   * @author nantuko
14   */
15  public class MageCardComparator implements CardViewComparator {
16  
17      private static final Logger logger = Logger.getLogger(MageCardComparator.class);
18  
19      private final int column;
20      private final boolean ascending;
21  
22      public MageCardComparator(int column, boolean ascending) {
23          this.column = column;
24          this.ascending = ascending;
25      }
26  
27      @Override
28      public int compare(CardView a, CardView b) {
29          Comparable aCom = 1;
30          Comparable bCom = 1;
31  
32          switch (column) {
33              // #skip
34              case 0:
35                  break;
36              // Name
37              case 1:
38                  aCom = a.getName();
39                  bCom = b.getName();
40                  if (aCom.equals(bCom) && a.getExpansionSetCode().equals(b.getExpansionSetCode())) {
41                      aCom = a.getCardNumber();
42                      bCom = b.getCardNumber();
43                  }
44                  break;
45              // Cost
46              case 2:
47                  aCom = a.getManaValue();
48                  bCom = b.getManaValue();
49                  break;
50              // Color
51              case 3:
52                  aCom = a.getColorText();
53                  bCom = b.getColorText();
54                  break;
55              // Type
56              case 4:
57                  aCom = a.getTypeText();
58                  bCom = b.getTypeText();
59                  break;
60              // Stats, attack and defense
61              case 5:
62                  aCom = (float) -1;
63                  bCom = (float) -1;
64                  if (a.isCreature()) {
65                      aCom = new Float(a.getPower() + '.' + (a.getToughness().startsWith("-") ? "0" : a.getToughness()));
66                  }
67                  if (b.isCreature()) {
68                      bCom = new Float(b.getPower() + '.' + (b.getToughness().startsWith("-") ? "0" : b.getToughness()));
69                  }
70                  break;
71              // Rarity
72              case 6:
73                  aCom = a.getRarity() == null ? 0 : a.getRarity().getSorting();
74                  bCom = b.getRarity() == null ? 0 : b.getRarity().getSorting();
75                  break;
76              // Set name
77              case 7:
78                  aCom = a.getExpansionSetCode();
79                  bCom = b.getExpansionSetCode();
80                  break;
81              case 8:
82                  aCom = Integer.parseInt(a.getCardNumber().replaceAll("[\\D]", ""));
83                  bCom = Integer.parseInt(b.getCardNumber().replaceAll("[\\D]", ""));
84                  break;
85              case 9:
86                  aCom = RateCard.rateCard(a.getOriginalCard(), null);
87                  bCom = RateCard.rateCard(b.getOriginalCard(), null);
88                  break;
89              case 10:
90                  aCom = a.getColorIdentityStr();
91                  bCom = b.getColorIdentityStr();
92                  break;
93              default:
94                  break;
95          }
96  
97          if (ascending) {
98              return aCom.compareTo(bCom);
99          } else {
100             return bCom.compareTo(aCom);
101         }
102     }
103 
104     @Override
105     public String getCategoryName(CardView sample) {
106         return "";
107     }
108 }

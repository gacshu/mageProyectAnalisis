44      /**
45       * Creates a card pool with specified criterea used when generating a deck.
46       *
47       * @param deckSize the size of the complete deck
48       * @param creaturePercentage what percentage of creatures to use when generating the deck.
49       * @param nonCreaturePercentage percentage of non-creatures to use when generating the deck.
50       * @param landPercentage percentage of lands to use when generating the deck.
51       * @param allowedColors which card colors are allowed in the generated deck.
52       * @param isSingleton if the deck only has 1 copy of each non-land card.
53       * @param colorlessAllowed if colourless mana symbols are allowed in costs in the deck.
54       * @param isAdvanced if the user has provided advanced options to generate the deck.
55       * @param deckGeneratorCMC the CMC curve to use for this deck
56       */
57      public DeckGeneratorPool(final int deckSize, final int creaturePercentage, final int nonCreaturePercentage, final int landPercentage,
58                               final List<ColoredManaSymbol> allowedColors, boolean isSingleton, boolean colorlessAllowed, boolean isAdvanced, DeckGeneratorCMC deckGeneratorCMC)
59      {
60          this.deckSize = deckSize;
61          this.allowedColors = allowedColors;
62          this.isSingleton = isSingleton;
63          this.colorlessAllowed = colorlessAllowed;
64  
65          this.deck = new Deck();
66  
67          // Advanced (CMC Slider panel and curve drop-down in the dialog)
68          if(isAdvanced) {
69              this.creatureCount = (int)Math.ceil((deckSize / 100.0) * creaturePercentage);
70              this.nonCreatureCount = (int)Math.ceil((deckSize / 100.0)* nonCreaturePercentage);
71              this.landCount = (int)Math.ceil((deckSize / 100.0)* landPercentage);
72              if(this.deckSize == 60) {
73                  this.poolCMCs = deckGeneratorCMC.get60CardPoolCMC();
74              } else {
75                  this.poolCMCs = deckGeneratorCMC.get40CardPoolCMC();
76              }
77          } else {
78              // Ignore the advanced group, just use defaults
79              this.creatureCount = (int)Math.ceil((deckSize / 100.0) * DEFAULT_CREATURE_PERCENTAGE);
80              this.nonCreatureCount = (int)Math.ceil((deckSize / 100.0) * DEFAULT_NON_CREATURE_PERCENTAGE);
81              this.landCount = (int)Math.ceil((deckSize / 100.0) * DEFAULT_LAND_PERCENTAGE);
82              if(this.deckSize == 60) {
83                  this.poolCMCs = DeckGeneratorCMC.Default.get60CardPoolCMC();
84              } else {
85                  this.poolCMCs = DeckGeneratorCMC.Default.get40CardPoolCMC();
86              }
87          }
88  
89          if(allowedColors.size() == 1) {
90              monoColored = true;
91          }
92  
93      }

96      /**
97       * Adjusts the number of spell cards that should be in a converted mana cost (CMC) range, given the amount of cards total.
98       * @param cardsCount the number of total cards.
99       * @return a list of CMC ranges, with the amount of cards for each CMC range
100      */
101     public List<DeckGeneratorCMC.CMC> getCMCsForSpellCount(int cardsCount) {
102         List<DeckGeneratorCMC.CMC> adjustedCMCs = new ArrayList<>(this.poolCMCs);
103         // For each CMC calculate how many spell cards are needed, given the total amount of cards
104         for(DeckGeneratorCMC.CMC deckCMC : adjustedCMCs) {
105             deckCMC.setAmount((int)Math.ceil(deckCMC.percentage * cardsCount));
106         }
107         return adjustedCMCs;
108     }

110     /**
111      * Verifies if the spell card supplied is valid for this pool of cards.
112      * Checks that there isn't too many copies of this card in the deck.
113      * Checks that the card fits the chosen colors for this pool.
114      * @param card the spell card
115      * @return if the spell card is valid for this pool.
116      */
117     public boolean isValidSpellCard(Card card)
118     {
119         int cardCount = getCardCount((card.getName()));
120         // Check it hasn't already got the maximum number of copies in a deck
121         if(cardCount < (isSingleton ? 1 : 4)) {
122             if(cardFitsChosenColors(card)) {
123                 return true;
124             }
125         }
126         return false;
127     }

129     /**
130      * Verifies if the non-basic land card supplied is valid for this pool of cards.
131      * @param card the non-basic land card
132      * @return if the land card generates the allowed colors for this pool.
133      */
134     public boolean isValidLandCard(Card card)
135     {
136         int cardCount = getCardCount((card.getName()));
137         // No need to check if the land is valid for the colors chosen
138         // They are all filtered before searching for lands to include in the deck.
139         return (cardCount < (isSingleton ? 1 : 4));
140     }

143     /**
144      * Adds a card to the pool and updates the count of this card.
145      * @param card the card to add.
146      */
147     public void addCard(Card card)
148     {
149         Object cnt = cardCounts.get((card.getName()));
150         if(cnt == null)
151             cardCounts.put(card.getName(), 0);
152         int existingCount = cardCounts.get((card.getName()));
153         cardCounts.put(card.getName(), existingCount+1);
154         deckCards.add(card);
155     }

157     /**
158      * Adds a card to the reserve pool.
159      * Reserve pool is used when the deck generation fails to build a complete deck, or
160      * a partially complete deck (e.g. if there are no cards found that match a CMC)
161      * @param card the card to add
162      * @param cardCMC the converted mana cost of the card
163      */
164     public boolean tryAddReserve(Card card, int cardCMC) {
165         // Only cards with CMC < 7 and don't already exist in the deck
166         // can be added to our reserve pool as not to overwhelm the curve
167         // with high CMC cards and duplicates.
168         if(cardCMC < 7 && getCardCount(card.getName()) == 0) {
169             this.reserveSpells.add(card);
170             return true;
171         }
172         return false;
173     }

175     /**
176      * Checks if the mana symbols in the card all match the allowed colors for this pool.
177      * @param card the spell card to check.
178      * @return if all the mana symbols fit the chosen colors.
179      */
180     private boolean cardFitsChosenColors(Card card) {
181         for (String symbol : card.getManaCostSymbols()) {
182             boolean found = false;
183             symbol = symbol.replace("{", "").replace("}", "");
184             if (isColoredManaSymbol(symbol)) {
185                 for (ColoredManaSymbol allowed : allowedColors) {
186                     if (symbol.contains(allowed.toString())) {
187                         found = true;
188                         break;
189                     }
190                 }
191                 if (!found) {
192                     return false;
193                 }
194             }
195             if (symbol.equals("C") && !colorlessAllowed) {
196                 return false;
197             }
198         }
199         return true;
200     }

203     /**
204      * Calculates the percentage of colored mana symbols over all spell cards in the deck.
205      * Used to balance the generation of basic lands so the amount of lands matches the
206      * cards mana costs.
207      * @return a list of colored mana symbols and the percentage of symbols seen in cards mana costs.
208      */
209     public Map<String, Double> calculateSpellColorPercentages() {
210 
211         final Map<String, Integer> colorCount = new HashMap<>();
212         for (final ColoredManaSymbol color : ColoredManaSymbol.values()) {
213             colorCount.put(color.toString(), 0);
214         }
215 
216         // Counts how many colored mana symbols we've seen in total so we can get the percentage of each color
217         int totalCount = 0;
218 
219         List<Card> fixedSpells = getFixedSpells();
220         for(Card spell: fixedSpells) {
221             for (String symbol : spell.getManaCostSymbols()) {
222                 symbol = symbol.replace("{", "").replace("}", "");
223                 if (isColoredManaSymbol(symbol)) {
224                     for (ColoredManaSymbol allowed : allowedColors) {
225                         if (symbol.contains(allowed.toString())) {
226                             int cnt = colorCount.get(allowed.toString());
227                             colorCount.put(allowed.toString(), cnt+1);
228                             totalCount++;
229                         }
230                     }
231                 }
232             }
233         }
234         final Map<String, Double> percentages = new HashMap<>();
235         for(Map.Entry<String, Integer> singleCount: colorCount.entrySet()) {
236             String color = singleCount.getKey();
237             int count = singleCount.getValue();
238             // Calculate the percentage this color has out of the total color counts
239             double percentage = (count / (double) totalCount) * 100;
240             percentages.put(color, percentage);
241         }
242         return percentages;
243     }

245     /**
246      * Calculates how many of each mana the non-basic lands produce.
247      * @param deckLands the non-basic lands which will be used in the deck.
248      * @return a mapping of colored mana symbol to the amount that can be produced.
249      */
250     public Map<String,Integer> countManaProduced(List<Card> deckLands)
251     {
252         Map<String, Integer> manaCounts = new HashMap<>();
253         for (final ColoredManaSymbol color : ColoredManaSymbol.values()) {
254             manaCounts.put(color.toString(), 0);
255         }
256         for(Card land: deckLands)  {
257             for(Ability landAbility: land.getAbilities()) {
258                 for (ColoredManaSymbol symbol : allowedColors) {
259                     String abilityString = landAbility.getRule();
260                     if(landTapsForAllowedColor(abilityString, symbol.toString())) {
261                         Integer count = manaCounts.get(symbol.toString());
262                         manaCounts.put(symbol.toString(), count + 1);
263                     }
264                 }
265             }
266         }
267         return manaCounts;
268     }

270     /** Filter all the non-basic lands retrieved from the database.
271      * @param landCardsInfo information about all the cards.
272      * @return a list of cards that produce the allowed colors for this pool.
273      */
274     public List<Card> filterLands(List<CardInfo> landCardsInfo) {
275         List<Card> matchingLandList = new ArrayList<>();
276         for(CardInfo landCardInfo: landCardsInfo) {
277             Card landCard = landCardInfo.getMockCard();
278             if(landProducesChosenColors(landCard)) {
279                 matchingLandList.add(landCard);
280             }
281         }
282         return matchingLandList;
283     }

359     /**
360      * Fixes undersized or oversized decks that have been generated.
361      * Removes random cards from an oversized deck until it is the correct size.
362      * Uses the reserve pool to fill up and undersized deck with cards.
363      * @return a fixed list of cards for this deck.
364      */
365     private List<Card> getFixedSpells()
366     {
367         int spellSize = deckCards.size();
368         int nonLandSize = (deckSize - landCount);
369 
370         // Less spells than needed
371         if(spellSize < nonLandSize) {
372 
373             int spellsNeeded = nonLandSize-spellSize;
374 
375             // If we haven't got enough spells in reserve to fulfil the amount we need, skip adding any.
376             if(reserveSpells.size() >= spellsNeeded) {
377 
378                 List<Card> spellsToAdd = new ArrayList<>(spellsNeeded);
379 
380                 // Initial reservoir
381                 for (int i = 0; i < spellsNeeded; i++)
382                     spellsToAdd.add(reserveSpells.get(i));
383 
384                 for (int i = spellsNeeded + 1; i < reserveSpells.size() - 1; i++) {
385                     int j = RandomUtil.nextInt(i);
386                     Card randomCard = reserveSpells.get(j);
387                     if (isValidSpellCard(randomCard) && j < spellsToAdd.size()) {
388                         spellsToAdd.set(j, randomCard);
389                     }
390                 }
391                 // Add randomly selected spells needed
392                 deckCards.addAll(spellsToAdd);
393             }
394         }
395 
396         // More spells than needed
397         else if(spellSize > (deckSize - landCount)) {
398             int spellsRemoved = (spellSize)-(deckSize-landCount);
399             for(int i = 0; i < spellsRemoved; ++i) {
400                 deckCards.remove(RandomUtil.nextInt(deckCards.size()));
401             }
402         }
403 
404         // Check we have exactly the right amount of cards for a deck.
405         if(deckCards.size() != nonLandSize) {
406             throw new IllegalStateException("Not enough cards found to generate deck.");
407         }
408         // Return the fixed amount
409         return deckCards;
410     }

412     /**
413      * Returns if this land taps for the given color.
414      * Basic string matching to check the ability adds one of the chosen mana when tapped.
415      * @param ability MockAbility of the land card
416      * @param symbol colored mana symbol.
417      * @return if the ability is tapping to produce the mana the symbol represents.
418      */
419     private boolean landTapsForAllowedColor(String ability, String symbol)  {
420         return ability.matches(".*Add \\{" + symbol + "\\}.");
421     }

423     /**
424      * Returns if this land will produce the chosen colors for this pool.
425      * @param card a non-basic land card.
426      * @return if this land card taps to produces the colors chosen.
427      */
428     private boolean landProducesChosenColors(Card card) {
429         // All mock card abilities will be MockAbilities so we can't differentiate between ManaAbilities
430         // and other Abilities so we have to do some basic string matching on land cards for now.
431         List<Ability> landAbilities = card.getAbilities();
432         int count = 0;
433         for(Ability ability : landAbilities) {
434             String abilityString = ability.getRule();
435             // Lands that tap to produce mana of the chosen colors
436             for(ColoredManaSymbol symbol : allowedColors) {
437                 if(landTapsForAllowedColor(abilityString, symbol.toString())) {
438                     count++;
439                 }
440             }
441             if(count > 1) {
442                 return true;
443             }
444         }
445         return false;
446     }

448     /**
449      * Returns if the symbol is a colored mana symbol.
450      * @param symbol the symbol to check.
451      * @return If it is a basic mana symbol or a hybrid mana symbol.
452      */
453     private static boolean isColoredManaSymbol(String symbol) {
454         // Hybrid mana
455         if(symbol.contains("/")) {
456             return true;
457         }
458         for(ColoredManaSymbol c: ColoredManaSymbol.values()) {
459             if (symbol.charAt(0) == (c.toString().charAt(0))) {
460                 return true;
461             }
462         }
463         return false;
464     }

467      * Returns how many of this card is in the pool.
468      * If there are none in the pool it will initalise the card count.
469      * @param cardName the name of the card to check.
470      * @return the number of cards in the pool of this name.
471      */
472     private int getCardCount(String cardName) {
473         Object cC = cardCounts.get((cardName));
474         if(cC == null)
475             cardCounts.put(cardName, 0);
476         return  cardCounts.get((cardName));

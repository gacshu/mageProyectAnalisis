40      /**
41       * Builds a deck out of the selected block/set/format.
42       *
43       * @return a path to the generated deck.
44       */
45      public static String generateDeck() {
46  
47          genDialog = new DeckGeneratorDialog();
48          if (genDialog.getSelectedColors() != null) {
49              Deck deck = buildDeck();
50              return genDialog.saveDeck(deck);
51          }
52          // If the deck couldn't be generated or the user cancelled, repopulate the deck selection with its cached value
53          return PreferencesDialog.getCachedValue(PreferencesDialog.KEY_NEW_TABLE_DECK_FILE, null);
54      }

56      protected static Deck buildDeck() {
57  
58          String selectedColors = genDialog.getSelectedColors();
59          List<ColoredManaSymbol> allowedColors = new ArrayList<>();
60          selectedColors = selectedColors != null ? selectedColors.toUpperCase(Locale.ENGLISH) : getRandomColors("X");
61          String format = genDialog.getSelectedFormat();
62  
63          List<String> setsToUse = ConstructedFormats.getSetsByFormat(format);
64          if (setsToUse == null) {
65              throw new DeckGeneratorException("Deck sets aren't initialized; please connect to a server to update the database.");
66          }
67          if (setsToUse.isEmpty()) {
68              // Default to using all sets
69              setsToUse = ExpansionRepository.instance.getSetCodes();
70          }
71  
72          int deckSize = genDialog.getDeckSize();
73  
74          if (selectedColors.contains("X")) {
75              selectedColors = getRandomColors(selectedColors);
76          }
77  
78          for (int i = 0; i < selectedColors.length(); i++) {
79              char c = selectedColors.charAt(i);
80              allowedColors.add(ColoredManaSymbol.lookup(c));
81          }
82  
83          return generateDeck(deckSize, allowedColors, setsToUse);
84      }

86      /**
87       * If the user has selected random colors, pick them randomly for the user.
88       *
89       * @param selectedColors a string of the colors selected.
90       * @return a String representation of the new colors chosen.
91       */
92      private static String getRandomColors(String selectedColors) {
93  
94          List<Character> availableColors = new ArrayList<>();
95          for (ColoredManaSymbol cms : ColoredManaSymbol.values()) {
96              availableColors.add(cms.toString().charAt(0));
97          }
98  
99          StringBuilder generatedColors = new StringBuilder();
100         int randomColors = 0;
101         for (int i = 0; i < selectedColors.length(); i++) {
102             char currentColor = selectedColors.charAt(i);
103             if (currentColor != 'X') {
104                 generatedColors.append(currentColor);
105                 availableColors.remove(new Character(currentColor));
106             } else {
107                 randomColors++;
108             }
109         }
110         for (int i = 0; i < randomColors && !availableColors.isEmpty(); i++) {
111             int index = RandomUtil.nextInt(availableColors.size());
112             generatedColors.append(availableColors.remove(index));
113         }
114         return generatedColors.toString();
115     }

117     /**
118      * Generates all the cards to use in the deck. Adds creatures,
119      * non-creatures, lands (including non-basic). Fixes the deck, adjusting for
120      * size and color of the cards retrieved.
121      *
122      * @param deckSize      how big the deck is to generate.
123      * @param allowedColors which colors are allowed in the deck.
124      * @param setsToUse     which sets to use to retrieve cards for this deck.
125      * @return the final deck to use.
126      */
127     private static Deck generateDeck(int deckSize, List<ColoredManaSymbol> allowedColors, List<String> setsToUse) {
128 
129         genPool = new DeckGeneratorPool(deckSize, genDialog.getCreaturePercentage(), genDialog.getNonCreaturePercentage(),
130                 genDialog.getLandPercentage(), allowedColors, genDialog.isSingleton(), genDialog.isColorless(),
131                 genDialog.isAdvanced(), genDialog.getDeckGeneratorCMC());
132 
133         final String[] sets = setsToUse.toArray(new String[setsToUse.size()]);
134 
135         // Creatures
136         final CardCriteria creatureCriteria = new CardCriteria();
137         creatureCriteria.setCodes(sets);
138         creatureCriteria.notTypes(CardType.LAND);
139         creatureCriteria.types(CardType.CREATURE);
140         if (!(genDialog.useArtifacts())) {
141             creatureCriteria.notTypes(CardType.ARTIFACT);
142         }
143 
144         // Non-creatures (sorcery, instant, enchantment, artifact etc.)
145         final CardCriteria nonCreatureCriteria = new CardCriteria();
146         nonCreatureCriteria.setCodes(sets);
147         nonCreatureCriteria.notTypes(CardType.LAND);
148         nonCreatureCriteria.notTypes(CardType.CREATURE);
149         if (!(genDialog.useArtifacts())) {
150             nonCreatureCriteria.notTypes(CardType.ARTIFACT);
151         }
152 
153         // Non-basic land
154         final CardCriteria nonBasicLandCriteria = new CardCriteria();
155         nonBasicLandCriteria.setCodes(sets);
156         nonBasicLandCriteria.types(CardType.LAND);
157         nonBasicLandCriteria.notSupertypes(SuperType.BASIC);
158 
159         // Generate basic land cards
160         Map<String, List<CardInfo>> basicLands = generateBasicLands(setsToUse);
161 
162         generateSpells(creatureCriteria, genPool.getCreatureCount());
163         generateSpells(nonCreatureCriteria, genPool.getNonCreatureCount());
164         generateLands(nonBasicLandCriteria, genPool.getLandCount(), basicLands);
165 
166         // Reconstructs the final deck and adjusts for Math rounding and/or missing cards
167         return genPool.getDeck();
168     }

170     /**
171      * Generates all spells for the deck. Each card is retrieved from the
172      * database and checked against the converted mana cost (CMC) needed for the
173      * current card pool. If a card's CMC matches the CMC range required by the
174      * pool, it is added to the deck. This ensures that the majority of cards
175      * fit a fixed mana curve for the deck, and it is playable. Creatures and
176      * non-creatures are retrieved separately to ensure the deck contains a
177      * reasonable mix of both.
178      *
179      * @param criteria   the criteria to search for in the database.
180      * @param spellCount the number of spells that match the criteria needed in
181      *                   the deck.
182      */
183     private static void generateSpells(CardCriteria criteria, int spellCount) {
184         List<CardInfo> cardPool = CardRepository.instance.findCards(criteria);
185         int retrievedCount = cardPool.size();
186         List<DeckGeneratorCMC.CMC> deckCMCs = genPool.getCMCsForSpellCount(spellCount);
187         int count = 0;
188         int reservesAdded = 0;
189         boolean added;
190         if (retrievedCount > 0 && retrievedCount >= spellCount) {
191             int tries = 0;
192             while (count < spellCount) {
193                 Card card = cardPool.get(RandomUtil.nextInt(retrievedCount)).getMockCard();
194                 if (genPool.isValidSpellCard(card)) {
195                     int cardCMC = card.getManaValue();
196                     for (DeckGeneratorCMC.CMC deckCMC : deckCMCs) {
197                         if (cardCMC >= deckCMC.min && cardCMC <= deckCMC.max) {
198                             int currentAmount = deckCMC.getAmount();
199                             if (currentAmount > 0) {
200                                 deckCMC.setAmount(currentAmount - 1);
201                                 genPool.addCard(card.copy());
202                                 count++;
203                             }
204                         } else if (reservesAdded < (genPool.getDeckSize() / 2)) {
205                             added = genPool.tryAddReserve(card, cardCMC);
206                             if (added) {
207                                 reservesAdded++;
208                             }
209                         }
210                     }
211                 }
212                 tries++;
213                 if (tries > MAX_TRIES) {
214                     // Break here, we'll fill in random missing ones later
215                     break;
216                 }
217             }
218         } else {
219             throw new IllegalStateException("Not enough cards to generate deck.");
220         }
221 
222     }

224     /**
225      * Generates all the lands for the deck. Generates non-basic if selected by
226      * the user and if the deck isn't monocolored. Will fetch non-basic lands if
227      * required and then fill up the remaining space with basic lands. Basic
228      * lands are adjusted according to the mana symbols seen in the cards used
229      * in this deck. Usually the lands will be well balanced relative to the
230      * color of cards.
231      *
232      * @param criteria   the criteria of the lands to search for in the database.
233      * @param landsCount the amount of lands required for this deck.
234      * @param basicLands information about the basic lands from the sets used.
235      */
236     private static void generateLands(CardCriteria criteria, int landsCount, Map<String, List<CardInfo>> basicLands) {
237 
238         int tries = 0;
239         int countNonBasic = 0;
240         // Store the nonbasic lands (if any) we'll add
241         List<Card> deckLands = new ArrayList<>();
242 
243         // Calculates the percentage of colored mana symbols over all spells in the deck
244         Map<String, Double> percentage = genPool.calculateSpellColorPercentages();
245 
246         // Only dual/tri color lands are generated for now, and not non-basic lands that only produce colorless mana.
247         if (!genPool.isMonoColoredDeck() && genDialog.useNonBasicLand()) {
248             List<Card> landCards = genPool.filterLands(CardRepository.instance.findCards(criteria));
249             int allCount = landCards.size();
250             if (allCount > 0) {
251                 while (countNonBasic < landsCount / 2) {
252                     Card card = landCards.get(RandomUtil.nextInt(allCount));
253                     if (genPool.isValidLandCard(card)) {
254                         Card addedCard = card.copy();
255                         deckLands.add(addedCard);
256                         genPool.addCard(addedCard);
257                         countNonBasic++;
258                     }
259                     tries++;
260                     // to avoid infinite loop
261                     if (tries > MAX_TRIES) {
262                         // Not a problem, just use what we have
263                         break;
264                     }
265                 }
266             }
267         }
268         // Calculate the amount of colored mana already can be produced by the non-basic lands
269         Map<String, Integer> count = genPool.countManaProduced(deckLands);
270         // Fill up the rest of the land quota with basic lands adjusted to fit the deck's mana costs
271         addBasicLands(landsCount - countNonBasic, percentage, count, basicLands);
272     }

274     /**
275      * Returns a map of colored mana symbol to basic land cards of that color.
276      *
277      * @param setsToUse which sets to retrieve basic lands from.
278      * @return a map of color to basic lands.
279      */
280     private static Map<String, List<CardInfo>> generateBasicLands(List<String> setsToUse) {
281 
282         Set<String> landSets = TournamentUtil.getLandSetCodeForDeckSets(setsToUse);
283 
284         CardCriteria criteria = new CardCriteria();
285         if (!landSets.isEmpty()) {
286             criteria.setCodes(landSets.toArray(new String[landSets.size()]));
287         }
288         criteria.ignoreSetsWithSnowLands();
289 
290         Map<String, List<CardInfo>> basicLandMap = new HashMap<>();
291 
292         for (ColoredManaSymbol c : ColoredManaSymbol.values()) {
293             String landName = DeckGeneratorPool.getBasicLandName(c.toString());
294             criteria.rarities(Rarity.LAND).nameExact(landName);
295             List<CardInfo> cards = CardRepository.instance.findCards(criteria);
296             if (cards.isEmpty()) { // Workaround to get basic lands if lands are not available for the given sets
297                 criteria.setCodes("M15");
298                 cards = CardRepository.instance.findCards(criteria);
299             }
300             basicLandMap.put(landName, cards);
301         }
302         return basicLandMap;
303     }

305     /**
306      * Once any non-basic lands are added, add basic lands until the deck is
307      * filled.
308      *
309      * @param landsNeeded how many remaining lands are needed.
310      * @param percentage  the percentage needed for each color in the final deck.
311      * @param count       how many of each color can be produced by non-basic lands.
312      * @param basicLands  list of information about basic lands from the
313      *                    database.
314      */
315     private static void addBasicLands(int landsNeeded, Map<String, Double> percentage, Map<String, Integer> count, Map<String, List<CardInfo>> basicLands) {
316 
317         int colorTotal = 0;
318         ColoredManaSymbol colorToAdd = ColoredManaSymbol.U;
319 
320         // Add up the totals for all colors, to keep track of the percentage a color is.
321         for (Map.Entry<String, Integer> c : count.entrySet()) {
322             colorTotal += c.getValue();
323         }
324 
325         // Keep adding basic lands until we fill the deck
326         while (landsNeeded > 0) {
327 
328             double minPercentage = Integer.MIN_VALUE;
329 
330             for (ColoredManaSymbol color : ColoredManaSymbol.values()) {
331                 // What percentage of this color is requested
332                 double neededPercentage = percentage.get(color.toString());
333                 // If there is a 0% need for basic lands of this color, skip it
334                 if (neededPercentage <= 0) {
335                     continue;
336                 }
337                 int currentCount = count.get(color.toString());
338                 double thisPercentage = 0.0;
339                 // Calculate the percentage of lands so far that produce this color
340                 if (currentCount > 0) {
341                     thisPercentage = (currentCount / (double) colorTotal) * 100.0;
342                 }
343                 // Check if the color is the most "needed" (highest percentage) we have seen so far
344                 if (neededPercentage - thisPercentage > minPercentage) {
345                     // Put this color land forward to be added
346                     colorToAdd = color;
347                     minPercentage = (neededPercentage - thisPercentage);
348                 }
349             }
350             genPool.addCard(getBasicLand(colorToAdd, basicLands));
351             count.put(colorToAdd.toString(), count.get(colorToAdd.toString()) + 1);
352             colorTotal++;
353             landsNeeded--;
354         }
355     }

357     /**
358      * Return a random basic land of the chosen color.
359      *
360      * @param color      the color the basic land should produce.
361      * @param basicLands list of information about basic lands from the
362      *                   database.
363      * @return a single basic land that produces the color needed.
364      */
365     private static Card getBasicLand(ColoredManaSymbol color, Map<String, List<CardInfo>> basicLands) {
366         String landName = DeckGeneratorPool.getBasicLandName(color.toString());
367         List<CardInfo> basicLandsInfo = basicLands.get(landName);
368         return basicLandsInfo.get(RandomUtil.nextInt(basicLandsInfo.size())).getMockCard().copy();
369     }

1   package mage.client.util.stats;
2   
3   import java.awt.*;
4   import java.util.List;
5   import java.util.concurrent.CancellationException;
6   import java.util.concurrent.ExecutionException;
7   import java.util.concurrent.TimeUnit;
8   import javax.swing.*;
9   import org.apache.log4j.Logger;
10  
11  /**
12   * This updates the mem usage info in the Mage client every
13   * MEM_USAGE_UPDATE_TIME ms.
14   *
15   * @author noxx, JayDi85
16   */
17  
18  public class UpdateMemUsageTask extends SwingWorker<Void, MemoryStats> {
19  
20      private static final int MEM_USAGE_UPDATE_TIME = 2000;
21      private static final int MEM_USAGE_WARNING_PERCENT = 80; // red color for mem used more than xx%
22  
23      private final JLabel jLabelToDisplayInfo;
24  
25      private static final Logger logger = Logger.getLogger(UpdateMemUsageTask.class);
26  
27      public UpdateMemUsageTask(JLabel jLabelToDisplayInfo) {
28          this.jLabelToDisplayInfo = jLabelToDisplayInfo;
29          this.jLabelToDisplayInfo.setToolTipText("<html>Memory usage statistics");
30      }
31  
32      @Override
33      protected Void doInBackground() throws Exception {
34          while (!isCancelled()) {
35              MemoryStats memoryStats = new MemoryStats(0, 0, 0, 0);
36  
37              Runtime runtime = Runtime.getRuntime();
38              if (runtime.maxMemory() != 0) {
39                  memoryStats.setMaxAvailable(runtime.maxMemory());
40                  memoryStats.setAvailable(runtime.totalMemory());
41                  memoryStats.setFree(runtime.freeMemory());
42                  memoryStats.setUsed(runtime.totalMemory() - runtime.freeMemory());
43              }
44  
45              this.publish(memoryStats);
46              TimeUnit.MILLISECONDS.sleep(MEM_USAGE_UPDATE_TIME);
47          }
48          return null;
49      }
50  
51      @Override
52      protected void process(List<MemoryStats> chunks) {
53          if (chunks != null && !chunks.isEmpty()) {
54              MemoryStats memoryStats = chunks.get(chunks.size() - 1);
55              if (memoryStats != null) {
56                  int max = Math.round(memoryStats.getMaxAvailable() / (1000 * 1000));
57                  int used = Math.round(memoryStats.getUsed() / (1000 * 1000));
58                  int total = Math.round(memoryStats.getAvailable() / (1000 * 1000));
59                  int percent = 0;
60                  if(max != 0){
61                      percent = Math.round((used * 100) / max);
62                  }
63  
64                  jLabelToDisplayInfo.setText("Memory used: " + percent + "% (" + used + " of " + max + " MB)");
65                  String warning = "";
66                  String optimizeHint = "<br><br>If you see low memory warning and have free system memory then try to increase max limit in launcher settings:<br>"
67                          + " - Go to <i>launcher -> settings -> java tab</i>;<br>"
68                          + " - Find <i>client java options</i> (it's may contain many commands);<br>"
69                          + " - Find max available memory setting: <i>-Xmx256m</i> (it's must start with <b>-Xmx</b>);<br>"
70                          + " - Increase number in that value from 256 to 512, or 512 to 1024;<br>"
71                          + " - Save new settings and restart application.";
72                  if(percent >= MEM_USAGE_WARNING_PERCENT){
73                      jLabelToDisplayInfo.setForeground(Color.red);
74                      warning = "<br><br><b>WARNING</b><br>"
75                              + "Application memory limit almost reached. Errors and freezes are very possible.";
76  
77                  }else{
78                      jLabelToDisplayInfo.setForeground(Color.black);
79                  }
80  
81                  this.jLabelToDisplayInfo.setToolTipText("<html>Memory usage statistics" + warning + optimizeHint);
82  
83                  return;
84              }
85          }
86          jLabelToDisplayInfo.setText("Mem usage unknown");
87      }
88  
89      @Override
90      protected void done() {
91          try {
92              get();
93          } catch (InterruptedException | ExecutionException ex) {
94              logger.fatal("Update Memory Usage error", ex);
95          } catch (CancellationException ex) {
96          }
97      }
98  
99  }

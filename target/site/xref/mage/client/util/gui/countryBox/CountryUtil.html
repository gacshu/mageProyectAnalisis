16  public final class CountryUtil {

18      private static final Logger LOGGER = Logger.getLogger(CountryUtil.class);
19      private static final Map<String, ImageIcon> FLAG_ICON_CACHE = new HashMap<>();
20      private static final Map<String, String> COUNTRY_MAP = new HashMap<>();
21  
22      public static ImageIcon getCountryFlagIcon(String countryCode) {
23          ImageIcon flagIcon = FLAG_ICON_CACHE.get(countryCode);
24          if (flagIcon == null) {
25              Image flagImage = ImageHelper.getImageFromResources("/flags/" + countryCode + (countryCode.endsWith(".png") ? "" : ".png"));
26              if (flagImage != null) {
27                  if (GUISizeHelper.flagHeight > 11) {
28                      int width = Math.round(GUISizeHelper.flagHeight * flagImage.getWidth(null) / flagImage.getHeight(null));
29                      BufferedImage resized = ImageHelper.scale((BufferedImage) flagImage, BufferedImage.TYPE_4BYTE_ABGR, width, GUISizeHelper.flagHeight);
30                      flagIcon = new ImageIcon(resized);
31                  } else {
32                      flagIcon = new ImageIcon(flagImage);
33                  }
34              }
35              if (flagIcon == null || flagIcon.getImage() == null) {
36                  LOGGER.warn("Country flag resource not found: " + countryCode);
37                  FLAG_ICON_CACHE.put(countryCode, flagIcon);
38              } else {
39                  FLAG_ICON_CACHE.put(countryCode, flagIcon);
40              }
41          }
42          return flagIcon;
43      }
44  
45      public static ImageIcon getCountryFlagIconSize(String countryCode, int height) {
46          ImageIcon flagIcon = null;
47          Image flagImage = ImageHelper.getImageFromResources("/flags/" + countryCode + (countryCode.endsWith(".png") ? "" : ".png"));
48          if (flagImage != null) {
49  
50              if (height > 11) {
51                  int width = Math.round(height * flagImage.getWidth(null) / flagImage.getHeight(null));
52                  BufferedImage resized = ImageHelper.scale((BufferedImage) flagImage, BufferedImage.TYPE_4BYTE_ABGR, width, height);
53                  flagIcon = new ImageIcon(resized);
54              } else {
55                  flagIcon = new ImageIcon(flagImage);
56              }
57          }
58          return flagIcon;
59      }
60  
61      public static void changeGUISize() {
62          FLAG_ICON_CACHE.clear();
63      }
64  
65      public static String getCountryName(String countryCode) {
66          if (COUNTRY_MAP.isEmpty()) {
67              for (int i = 0; i < CountryComboBox.countryList.length; i++) {
68                  COUNTRY_MAP.put(CountryComboBox.countryList[i][1], CountryComboBox.countryList[i][0]);
69              }
70          }
71          return COUNTRY_MAP.get(countryCode);
72      }

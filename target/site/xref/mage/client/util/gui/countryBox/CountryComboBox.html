266     @SuppressWarnings("unchecked")
267     public CountryComboBox() {
268         model = new DefaultComboBoxModel();
269         setModel(model);
270         setRenderer(new CountryItemRenderer());
271         setEditor(new CountryItemEditor());
272         addItems(countryList);
273     }
274 
275     /**
276      * Add an array items to this combo box. Each item is an array of two String
277      * elements: - first element is country name. - second element is path of an
278      * image file for country flag.
279      *
280      * @param items
281      */
282     @SuppressWarnings("unchecked")
283     public void addItems(String[][] items) {
284         for (String[] anItem : items) {
285             model.addElement(anItem);
286         }
287     }

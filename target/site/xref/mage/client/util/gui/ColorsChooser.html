66      @Override
67      public Component getListCellRendererComponent(final JList list, final Object value, final int index, final boolean isSelected,
68              final boolean cellHasFocus) {
69  
70          final JPanel panel = new JPanel(new FlowLayout());
71          drawOn(panel, (String) value);
72          panel.setBorder(Constants.EMPTY_BORDER);
73          if (isSelected) {
74              panel.setBackground(Color.LIGHT_GRAY);
75          }
76          return panel;
77      }
78  
79      private void drawOn(JPanel panel, String value) {
80          List<Image> images = new ArrayList<>();
81          value = value.toUpperCase(Locale.ENGLISH);
82          for (int i = 0; i < value.length(); i++) {
83              char symbol = value.charAt(i);
84              Image image = ManaSymbols.getSizedManaSymbol(String.valueOf(symbol));
85              if (image != null) {
86                  images.add(image);
87              }
88          }
89  
90          if (images.size() == value.length()) {
91              int dx = 0;
92              for (Image image : images) {
93                  ImageIcon icon = new ImageIcon(image);
94                  JLabel imageLabel = new JLabel();
95                  imageLabel.setSize(11, 11);
96                  imageLabel.setLocation(dx, 0);
97                  imageLabel.setIcon(icon);
98                  panel.add(imageLabel);
99                  dx += 13;
100             }
101         } else {
102             String s = value.replace("B", "{B}").replace("R", "{R}").replace("G", "{G}").replace("W", "{W}").replace("U", "{U}").replace("X", "{X}");
103             panel.add(new JLabel(s));
104         }
105     }

50      public static void restoreDividerLocations(Rectangle bounds, String lastDividerLocation, JComponent component) {
51          String currentBounds = Double.toString(bounds.getWidth()) + 'x' + bounds.getHeight();
52          String savedBounds = PreferencesDialog.getCachedValue(KEY_MAGE_PANEL_LAST_SIZE, null);
53          // use divider positions only if screen size is the same as it was the time the settings were saved
54          if (savedBounds != null && savedBounds.equals(currentBounds)) {
55              if (lastDividerLocation != null && component != null) {
56                  if (component instanceof JSplitPane) {
57                      JSplitPane jSplitPane = (JSplitPane) component;
58                      jSplitPane.setDividerLocation(Integer.parseInt(lastDividerLocation));
59                  }
60  
61                  if (component instanceof PlayersChatPanel) {
62                      PlayersChatPanel playerChatPanel = (PlayersChatPanel) component;
63                      playerChatPanel.setSplitDividerLocation(Integer.parseInt(lastDividerLocation));
64                  }
65              }
66          }
67      }

107     public static String cleanString(String in) {
108         StringBuilder out = new StringBuilder();
109         char c;
110         for (int i = 0; i < in.length(); i++) {
111             c = in.charAt(i);
112             if (c == ' ' || c == '-') {
113                 out.append('_');
114             } else if (Character.isLetterOrDigit(c)) {
115                 out.append(c);
116             }
117         }
118         return out.toString().toLowerCase(Locale.ENGLISH);
119     }

121     public static void keepComponentInsideScreen(int centerX, int centerY, Component component) {
122         Dimension screenDim = component.getToolkit().getScreenSize();
123         GraphicsConfiguration g = component.getGraphicsConfiguration();
124 
125         if (g != null) {
126             Insets insets = component.getToolkit().getScreenInsets(g); // no usable space like toolbar
127             boolean setLocation = false;
128             if (centerX + component.getWidth() > screenDim.width - insets.right) {
129                 centerX = (screenDim.width - insets.right) - component.getWidth();
130                 setLocation = true;
131             } else if (centerX < insets.left) {
132                 centerX = insets.left;
133                 setLocation = true;
134             }
135 
136             if (centerY + component.getHeight() > screenDim.height - insets.bottom) {
137                 centerY = (screenDim.height - insets.bottom) - component.getHeight();
138                 setLocation = true;
139             } else if (centerY < insets.top) {
140                 centerY = insets.top;
141                 setLocation = true;
142             }
143             if (setLocation) {
144                 component.setLocation(centerX, centerY);
145             }
146         } else {
147             System.out.println("GuiDisplayUtil::keepComponentInsideScreen -> no GraphicsConfiguration");
148         }
149     }

153     public static void keepComponentInsideFrame(int centerX, int centerY, Component component) {
154         Rectangle frameRec = MageFrame.getInstance().getBounds();
155         boolean setLocation = false;
156         if (component.getX() > (frameRec.width - OVERLAP_LIMIT)) {
157             setLocation = true;
158         }
159         if (component.getY() > (frameRec.height - OVERLAP_LIMIT)) {
160             setLocation = true;
161         }
162         if (setLocation) {
163             component.setLocation(centerX, centerY);
164         }

167     public static Point keepComponentInsideParent(Point l, Point parentPoint, Component c, Component parent) {
168         int dx = parentPoint.x + parent.getWidth() - DEFAULT_INSETS.right - COMPONENT_INSETS.right;
169         if (l.x + c.getWidth() > dx) {
170             l.x = dx - c.getWidth();
171         }
172 
173         int dy = parentPoint.y + parent.getHeight() - DEFAULT_INSETS.bottom - COMPONENT_INSETS.bottom;
174         if (l.y + c.getHeight() > dy) {
175             l.y = Math.max(10, dy - c.getHeight());
176         }
177 
178         return l;
179     }

181     public static TextLines getTextLinesfromCardView(CardView card) {
182         TextLines textLines = new TextLines();
183 
184         // rules
185         textLines.setLines(new ArrayList<>(card.getRules()));
186         for (String rule : card.getRules()) {
187             textLines.setBasicTextLength(textLines.getBasicTextLength() + rule.length());
188         }
189 
190         // counters
191         if (card.getMageObjectType().canHaveCounters()) {
192             java.util.List<CounterView> counters = new ArrayList<>();
193             if (card.getCounters() != null) {
194                 counters.addAll(card.getCounters());
195             }
196             if (!counters.isEmpty()) {
197                 StringBuilder sb = new StringBuilder();
198                 int index = 0;
199                 for (CounterView counter : counters) {
200                     if (counter.getCount() > 0) {
201                         if (index == 0) {
202                             sb.append("<b>Counters:</b> ");
203                         } else {
204                             sb.append(", ");
205                         }
206                         sb.append(counter.getCount()).append(" x <i>").append(counter.getName()).append("</i>");
207                         index++;
208                     }
209                 }
210                 textLines.getLines().add(sb.toString());
211                 textLines.setBasicTextLength(textLines.getBasicTextLength() + 50);
212             }
213         }
214 
215         // damage
216         if (card.getMageObjectType().isPermanent() && card instanceof PermanentView) {
217             int damage = ((PermanentView) card).getDamage();
218             if (damage > 0) {
219                 textLines.getLines().add("<span color='red'><b>Damage dealt:</b> " + damage + "</span>");
220                 textLines.setBasicTextLength(textLines.getBasicTextLength() + 50);
221             }
222         }
223         return textLines;
224     }

230     public static StringBuilder getRulesFromCardView(CardView card, TextLines textLines) {
231         String manaCost = card.getManaCostStr();
232         String castingCost = UI.getDisplayManaCost(manaCost);
233         castingCost = ManaSymbols.replaceSymbolsWithHTML(castingCost, ManaSymbols.Type.TOOLTIP);
234 
235         int symbolCount = 0;
236         int offset = 0;
237         while ((offset = castingCost.indexOf("<img", offset) + 1) != 0) {
238             symbolCount++;
239         }
240 
241         String fontFamily = "tahoma";
242         int fontSize = GUISizeHelper.cardTooltipFontSize;
243 
244         /*if (prefs.fontFamily == CardFontFamily.arial)
245          fontFamily = "arial";
246          else if (prefs.fontFamily == CardFontFamily.verdana) {
247          fontFamily = "verdana";
248          }*/
249         final StringBuilder buffer = new StringBuilder(512);
250         buffer.append("<html><body style='font-family:");
251         buffer.append(fontFamily);
252         buffer.append(";font-size:");
253         buffer.append(fontSize);
254         buffer.append("pt;margin:0px 1px 0px 1px'>");
255         buffer.append("<table cellspacing=0 cellpadding=0 border=0 width='100%'>");
256         buffer.append("<tr><td valign='top'><b>");
257         buffer.append(card.getDisplayName());
258         if (card.isGameObject()) {
259             buffer.append(" [").append(card.getId().toString(), 0, 3).append(']');
260         }
261         buffer.append("</b></td><td align='right' valign='top' style='width:");
262         buffer.append(symbolCount * GUISizeHelper.cardTooltipFontSize);
263         buffer.append("px'>");
264         if (!card.isSplitCard()) {
265             buffer.append(castingCost);
266         }
267         buffer.append("</td></tr></table>");
268         buffer.append("<table cellspacing=0 cellpadding=0 border=0 width='100%'><tr><td style='margin-left: 1px'>");
269         String imageSize = " width=" + GUISizeHelper.cardTooltipFontSize + " height=" + GUISizeHelper.cardTooltipFontSize + '>';
270         if (card.getColor().isWhite()) {
271             buffer.append("<img src='").append(getResourcePath("card/color_ind_white.png")).append("' alt='W' ").append(imageSize);
272         }
273         if (card.getColor().isBlue()) {
274             buffer.append("<img src='").append(getResourcePath("card/color_ind_blue.png")).append("' alt='U' ").append(imageSize);
275         }
276         if (card.getColor().isBlack()) {
277             buffer.append("<img src='").append(getResourcePath("card/color_ind_black.png")).append("' alt='B' ").append(imageSize);
278         }
279         if (card.getColor().isRed()) {
280             buffer.append("<img src='").append(getResourcePath("card/color_ind_red.png")).append("' alt='R' ").append(imageSize);
281         }
282         if (card.getColor().isGreen()) {
283             buffer.append("<img src='").append(getResourcePath("card/color_ind_green.png")).append("' alt='G' ").append(imageSize);
284         }
285         if (!card.getColor().isColorless()) {
286             buffer.append("&nbsp;&nbsp;");
287         }
288         buffer.append(getTypes(card));
289         buffer.append("</td><td align='right'>");
290         String rarity;
291         if (card.getRarity() == null) {
292             rarity = Rarity.COMMON.getCode();
293             buffer.append("<b color='black'>");
294         } else {
295             switch (card.getRarity()) {
296                 case RARE:
297                     buffer.append("<b color='#FFBF00'>");
298                     break;
299                 case UNCOMMON:
300                     buffer.append("<b color='silver'>");
301                     break;
302                 case COMMON:
303                     buffer.append("<b color='black'>");
304                     break;
305                 case MYTHIC:
306                     buffer.append("<b color='#D5330B'>");
307                     break;
308             }
309             rarity = card.getRarity().getCode();
310         }
311         if (card.getExpansionSetCode() != null) {
312             buffer.append(ManaSymbols.replaceSetCodeWithHTML(card.getExpansionSetCode().toUpperCase(Locale.ENGLISH), rarity, GUISizeHelper.symbolTooltipSize));
313         }
314         buffer.append("</td></tr></table>");
315 
316         String pt = "";
317         if (card.isCreature()) {
318             pt = card.getPower() + '/' + card.getToughness();
319         } else if (card.isPlanesWalker()) {
320             pt = card.getLoyalty();
321         }
322 
323         buffer.append("<table cellspacing=0 cellpadding=0 border=0 width='100%' valign='bottom'><tr><td><b>");
324         buffer.append(pt).append("</b></td>");
325         buffer.append("<td align='right'>");
326         if (!card.isControlledByOwner()) {
327             if (card instanceof PermanentView) {
328                 buffer.append('[').append(((PermanentView) card).getNameOwner()).append("] ");
329             } else {
330                 buffer.append("[only controlled] ");
331             }
332         }
333         if (card.getMageObjectType() != MageObjectType.NULL) {
334             buffer.append(card.getMageObjectType().toString());
335         }
336         buffer.append("</td></tr></table>");
337 
338         // split card rules shows up by parts, so no needs to duplicate it later (only dynamic abilities must be shown)
339         Set<String> duplicatedRules = new HashSet<>();
340 
341         StringBuilder rule = new StringBuilder("<br/>");
342         if (card.isSplitCard()) {
343             rule.append("<table cellspacing=0 cellpadding=0 border=0 width='100%'>");
344             rule.append("<tr><td valign='top'><b>");
345             rule.append(card.getLeftSplitName());
346             rule.append("</b></td><td align='right' valign='top' style='width:");
347             rule.append(ManaSymbols.getClearManaSymbolsCount(card.getLeftSplitCostsStr()) * GUISizeHelper.symbolTooltipSize + 1);
348             rule.append("px'>");
349             rule.append(card.getLeftSplitCostsStr());
350             rule.append("</td></tr></table>");
351             for (String ruling : card.getLeftSplitRules()) {
352                 if (ruling != null && !ruling.replace(".", "").trim().isEmpty()) {
353                     // split names must be replaced
354                     duplicatedRules.add(ruling);
355                     rule.append("<p style='margin: 2px'>").append(replaceNamesInRule(ruling, card.getLeftSplitName())).append("</p>");
356                 }
357             }
358             rule.append("<table cellspacing=0 cellpadding=0 border=0 width='100%'>");
359             rule.append("<tr><td valign='top'><b>");
360             rule.append(card.getRightSplitName());
361             rule.append("</b></td><td align='right' valign='top' style='width:");
362             rule.append(ManaSymbols.getClearManaSymbolsCount(card.getRightSplitCostsStr()) * GUISizeHelper.symbolTooltipSize + 1);
363             rule.append("px'>");
364             rule.append(card.getRightSplitCostsStr());
365             rule.append("</td></tr></table>");
366             for (String ruling : card.getRightSplitRules()) {
367                 if (ruling != null && !ruling.replace(".", "").trim().isEmpty()) {
368                     // split names must be replaced
369                     duplicatedRules.add(ruling);
370                     rule.append("<p style='margin: 2px'>").append(replaceNamesInRule(ruling, card.getRightSplitName())).append("</p>");
371                 }
372             }
373         }
374         if (!textLines.getLines().isEmpty()) {
375             for (String textLine : textLines.getLines()) {
376                 if (textLine != null && !textLine.replace(".", "").trim().isEmpty()) {
377                     if (duplicatedRules.contains(textLine)) {
378                         continue;
379                     }
380                     rule.append("<p style='margin: 2px'>").append(textLine).append("</p>");
381                 }
382             }
383         }
384 
385         String legal = rule.toString();
386         if (!legal.isEmpty()) {
387             legal = replaceNamesInRule(legal, card.getDisplayName()); // must show real display name (e.g. split part, not original card)
388             buffer.append(ManaSymbols.replaceSymbolsWithHTML(legal, ManaSymbols.Type.TOOLTIP));
389         }
390 
391         Zone zone = card.getZone();
392         if (zone != null) {
393             buffer.append("<p style='margin: 2px'><b>Card Zone:</b> ").append(zone).append("</p>");
394         }
395 
396         buffer.append("<br></body></html>");
397         return buffer;
398     }

400     private static String replaceNamesInRule(String rule, String cardName) {
401         return rule.replaceAll("\\{this\\}", cardName.isEmpty() ? "this" : cardName);
402     }

408     private static String getTypes(CardView card) {
409         String types = "";
410         for (SuperType superType : card.getSuperTypes()) {
411             types += superType.toString() + ' ';
412         }
413         for (CardType cardType : card.getCardTypes()) {
414             types += cardType.toString() + ' ';
415         }
416         if (!card.getSubTypes().isEmpty()) {
417             types += "- ";
418         }
419         for (SubType subType : card.getSubTypes()) {
420             types += subType + " ";
421         }
422         return types.trim();
423     }

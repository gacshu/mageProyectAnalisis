10  public class Arrow extends JPanel {
11  
12      private static final long serialVersionUID = -4631054277822828303L;
13  
14      private int startX;
15      private int startY;
16      private int endX;
17      private int endY;
18      private int bodyWidth = 10;
19      private float headSize = 17;
20      private Composite composite;
21      private Color color = Color.red;
22      private Color strokeColor = Color.black;
23  
24      public Arrow() {
25          setOpaque(false);
26          setOpacity(0.6f);
27      }
28  
29  
30      @Override
31      protected void paintComponent(Graphics g) {
32          super.paintComponent(g);
33          float ex = endX - startX;
34          float ey = endY - startY;
35          if (ex == 0 && ey == 0) {
36              return;
37          }
38          float length = (float) Math.sqrt(ex * ex + ey * ey);
39          float bendPercent = (float) Math.asin(ey / length);
40          if (endX > startX) {
41              bendPercent = -bendPercent;
42          }
43          Area arrow = getArrow(length, bendPercent);
44          Graphics2D g2d = (Graphics2D) g;
45          g2d.translate(startX, startY);
46          g2d.rotate(Math.atan2(ey, ex));
47          g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
48          g2d.setComposite(composite);
49          g2d.setColor(this.color);
50          //debug stroke draw
51          //float[] dash2 = {10, 5};
52          //g2d.setStroke(new BasicStroke(3.5f, BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND, 10.0f, dash2, 0));
53          g2d.fill(arrow);
54          g2d.setColor(this.strokeColor);
55          g2d.draw(arrow);
56      }
57  
58      private Area getArrow(float length, float bendPercent) {
59          float p1x = 0, p1y = 0;
60          float p2x = length, p2y = 0;
61          float cx = length / 2, cy = length / 8f * bendPercent;
62  
63          float adjSize, ex, ey, abs_e;
64          adjSize = (float) (bodyWidth / 2 / Math.sqrt(2));
65          ex = p2x - cx;
66          ey = p2y - cy;
67          abs_e = (float) Math.sqrt(ex * ex + ey * ey);
68          ex /= abs_e;
69          ey /= abs_e;
70          GeneralPath bodyPath = new GeneralPath();
71          bodyPath.moveTo(p2x + (ey - ex) * adjSize, p2y - (ex + ey) * adjSize);
72          bodyPath.quadTo(cx, cy, p1x, p1y - bodyWidth / 2);
73          bodyPath.lineTo(p1x, p1y + bodyWidth / 2);
74          bodyPath.quadTo(cx, cy, p2x - (ey + ex) * adjSize, p2y + (ex - ey) * adjSize);
75          bodyPath.closePath();
76  
77          adjSize = (float) (headSize / Math.sqrt(2));
78          ex = p2x - cx;
79          ey = p2y - cy;
80          abs_e = (float) Math.sqrt(ex * ex + ey * ey);
81          ex /= abs_e;
82          ey /= abs_e;
83          GeneralPath headPath = new GeneralPath();
84          headPath.moveTo(p2x - (ey + ex) * adjSize, p2y + (ex - ey) * adjSize);
85          headPath.lineTo(p2x, p2y);
86          headPath.lineTo(p2x + (ey - ex) * adjSize, p2y - (ex + ey) * adjSize);
87          headPath.closePath();
88  
89          Area area = new Area(headPath);
90          area.add(new Area(bodyPath));
91          return area;
92      }

1   package mage.client.util.comparators;
2   
3   import mage.view.CardView;
4   
5   import java.util.stream.Collectors;
6   
7   /**
8    * @author BetaSteward_at_googlemail.com, JayDi85
9    */
10  public class CardViewColorComparator implements CardViewComparator {
11  
12      @Override
13      public int compare(CardView o1, CardView o2) {
14          return o1.getColor().compareTo(o2.getColor());
15      }
16  
17      @Override
18      public String getCategoryName(CardView sample) {
19          String res = sample.getColor().getColors()
20                  .stream()
21                  .map(c -> "{" + c.getOneColoredManaSymbol() + "}")
22                  .collect(Collectors.joining(""));
23          if (res.isEmpty()) {
24              res = "{C}"; // colorless
25          }
26          return res;
27      }
28  }

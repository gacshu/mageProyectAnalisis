12      private static final class Key {
13  
14          final int width;
15          final int height;
16          final double angle;
17  
18          public Key(int width, int height, double angle) {
19              this.width = width;
20              this.height = height;
21              this.angle = angle;
22          }
23  
24          @Override
25          public int hashCode() {
26              int hash = 3;
27              hash = 53 * hash + this.width;
28              hash = 53 * hash + this.height;
29              hash = 53 * hash + (int) (Double.doubleToLongBits(this.angle) ^ (Double.doubleToLongBits(this.angle) >>> 32));
30              return hash;
31          }
32  
33          @Override
34          public boolean equals(Object obj) {
35              if (this == obj) {
36                  return true;
37              }
38              if (obj == null) {
39                  return false;
40              }
41              if (getClass() != obj.getClass()) {
42                  return false;
43              }
44              final Key other = (Key) obj;
45              if (this.width != other.width) {
46                  return false;
47              }
48              if (this.height != other.height) {
49                  return false;
50              }
51              if (Double.doubleToLongBits(this.angle) != Double.doubleToLongBits(other.angle)) {
52                  return false;
53              }
54              return true;
55          }
56      }

58      private static final SoftValuesLoadingCache<Key, SoftValuesLoadingCache<BufferedImage, BufferedImage>> IMAGE_CACHE;
59  
60      static {
61          // TODO: can we use a single map?
62          IMAGE_CACHE = ImageCaches.register(SoftValuesLoadingCache.from(TransformedImageCache::createTransformedImageCache));
63      }
64  
65      private static SoftValuesLoadingCache<BufferedImage, BufferedImage> createTransformedImageCache(Key key) {
66          return SoftValuesLoadingCache.from(image -> {
67              if (key.width != image.getWidth() || key.height != image.getHeight()) {
68                  image = resizeImage(image, key.width, key.height);
69              }
70              if (key.angle != 0.0) {
71                  image = rotateImage(image, key.angle);
72              }
73              return image;           
74          });
75      }
76  
77      private static BufferedImage rotateImage(BufferedImage image, double angle) {
78          double sin = Math.abs(Math.sin(angle)), cos = Math.abs(Math.cos(angle));
79          int width = image.getWidth(), height = image.getHeight();
80          int newWidth = (int) Math.floor(width * cos + height * sin), newHeight = (int) Math.floor(height * cos + width * sin);
81  
82          GraphicsEnvironment ge = GraphicsEnvironment.getLocalGraphicsEnvironment();
83          GraphicsDevice gs = ge.getDefaultScreenDevice();
84          GraphicsConfiguration gc = gs.getDefaultConfiguration();
85  
86          BufferedImage result = gc.createCompatibleImage(newWidth, newHeight, Transparency.TRANSLUCENT);
87          Graphics2D g = result.createGraphics();
88          g.translate((newWidth - width) / 2, (newHeight - height) / 2);
89          g.rotate(angle, width / 2, height / 2);
90          g.drawRenderedImage(image, null);
91          g.dispose();
92          return result;
93      }
94  
95      private static BufferedImage resizeImage(BufferedImage original, int width, int height) {
96          Image scaled = original.getScaledInstance(width, height, Image.SCALE_SMOOTH);
97          BufferedImage output = new BufferedImage(width, height, original.getType());
98          Graphics2D graphics = output.createGraphics();
99          graphics.drawImage(scaled, 0, 0, null);
100         graphics.dispose();
101         return output;
102     }

104     public static BufferedImage getResizedImage(BufferedImage image, int width, int height) {
105         return getRotatedResizedImage(image, width, height, 0.0);
106     }
107 
108     public static BufferedImage getRotatedImage(BufferedImage image, double angle) {
109         return getRotatedResizedImage(image, -1, -1, angle);
110     }

112     public static BufferedImage getRotatedResizedImage(BufferedImage image, int width, int height, double angle) {
113         int imageWidth = image.getWidth();
114         int imageHeight = image.getHeight();
115 
116         if (angle == 0.0 && (width < 0 || imageWidth == width) && (height < 0 || imageHeight == height)) {
117             return image;
118         }
119 
120         int resWidth;
121         int resHeight;
122         if (width < 0 && height < 0) {
123             resWidth = imageWidth;
124             resHeight = imageHeight;
125         } else if ((height < 0) || (width >= 0 && imageHeight * width <= imageWidth * height)) {
126             resWidth = width;
127             resHeight = imageHeight * width / imageWidth;
128         } else {
129             resWidth = imageWidth * height / imageHeight;
130             resHeight = height;
131         }
132 
133         if (angle == 0.0 && imageWidth == resWidth && imageHeight == resHeight) {
134             return image;
135         }
136         if (resWidth < 3) {
137             resWidth = 3;
138         }
139         if (resHeight < 3) {
140             resHeight = 3;
141         }
142         return IMAGE_CACHE.getOrThrow(new Key(resWidth, resHeight, angle)).getOrThrow(image);
143     }

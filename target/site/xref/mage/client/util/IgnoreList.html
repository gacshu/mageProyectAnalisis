1   package mage.client.util;

3   import com.google.common.collect.ImmutableSet;
4   import mage.client.MageFrame;
5   import mage.client.preference.MagePreferences;
6   import mage.view.ChatMessage;
7   
8   import java.util.ArrayList;
9   import java.util.Collections;
10  import java.util.List;
11  import java.util.Set;
12  
13  public final class IgnoreList {
14  
15      private static final String USAGE = ""
16              + "<br><font color=yellow>\\ignore - shows your ignore list on this server."
17              + "<br>\\ignore username - add username to ignore list (they won't be able to chat or join to your new game)."
18              + "<br>\\unignore username - remove a username from your ignore list on this server.</font>";
19  
20      public static final int MAX_IGNORE_LIST_SIZE = 500;
21      public static final Set<ChatMessage.MessageType> IGNORED_MESSAGE_TYPES
22              = ImmutableSet.of(ChatMessage.MessageType.TALK,
23              ChatMessage.MessageType.WHISPER_FROM);
24  
25      public static String usage(String serverAddress) {
26          return "<br>Your ignored list on server " + serverAddress + ": " + getIgnoredUsers(serverAddress).size()
27                  + USAGE;
28      }
29  
30      public static Set<String> getIgnoredUsers(String serverAddress) {
31          return MagePreferences.ignoreList(serverAddress);
32      }
33  
34      public static String getIgnoreListInfo(String serverAddress) {
35          List<String> list = new ArrayList<>(getIgnoredUsers(serverAddress));
36          Collections.sort(list);
37          return "<font color=yellow>Current ignore list on " + serverAddress + " (total: " + list.size() + "): "
38                  + "[" + String.join(", ", list) + "]"
39                  + "</font>";
40      }
41  
42      public static String ignore(String serverAddress, String user) {
43          if (user == null || user.isEmpty()) {
44              return getIgnoreListInfo(serverAddress);
45          }
46  
47          if (MagePreferences.ignoreList(serverAddress).size() >= MAX_IGNORE_LIST_SIZE) {
48              return "Your ignore list is too big (max " + MAX_IGNORE_LIST_SIZE + "), remove a user to be able to add a new one.";
49          }
50  
51          if (userIsIgnored(serverAddress, user)) {
52              return user + " is already on your ignore list on " + serverAddress;
53          }
54  
55          MagePreferences.addIgnoredUser(serverAddress, user);
56          updateTablesTable();
57  
58          return "Added " + user + " to your ignore list on " + serverAddress + " (total: " + getIgnoredUsers(serverAddress).size() + ")";
59      }
60  
61      private static void updateTablesTable() {
62          MageFrame mageFrame = MageFrame.getInstance();
63          if (mageFrame != null) {
64              mageFrame.setTableFilter();
65          }
66      }
67  
68      public static String unignore(String serverAddress, String user) {
69          if (user == null || user.isEmpty()) {
70              return usage(serverAddress);
71          }
72          if (MagePreferences.removeIgnoredUser(serverAddress, user)) {
73              updateTablesTable();
74              return "Removed " + user + " from your ignore list on " + serverAddress + " (total: " + getIgnoredUsers(serverAddress).size() + ")";
75          } else {
76              return "No such user \"" + user + "\" on your ignore list on " + serverAddress + " (total: " + getIgnoredUsers(serverAddress).size() + ")";
77          }
78      }
79  
80      public static boolean userIsIgnored(String serverAddress, String username) {
81          return MagePreferences.ignoreList(serverAddress).contains(username);
82      }
83  }

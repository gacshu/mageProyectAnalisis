3   import mage.abilities.Ability;
4   import mage.constants.ComparisonType;
5   import mage.filter.FilterCard;
6   import mage.filter.predicate.mageobject.ManaValuePredicate;
7   import mage.game.Game;
8   import mage.target.TargetCard;
9   import mage.target.common.TargetCardInGraveyard;
10  
11  /**
12   *
13   * @author TheElk801
14   */
15  public enum XCMCGraveyardAdjuster implements TargetAdjuster {
16      instance;
17  
18      @Override
19      public void adjustTargets(Ability ability, Game game) {
20          int xValue = ability.getManaCostsToPay().getX();
21          FilterCard filterCard = ((TargetCard) ability.getTargets().get(0)).getFilter().copy();
22          filterCard.add(new ManaValuePredicate(ComparisonType.EQUAL_TO, xValue));
23          filterCard.setMessage(filterCard.getMessage().replace('X', (char) xValue));
24          ability.getTargets().clear();
25          ability.getTargets().add(new TargetCardInGraveyard(filterCard));
26      }
27  }

3   import mage.server.managers.ConfigSettings;
4   import mage.server.managers.MailClient;
5   import org.apache.log4j.Logger;
6   
7   import javax.mail.Message;
8   import javax.mail.MessagingException;
9   import javax.mail.Session;
10  import javax.mail.Transport;
11  import javax.mail.internet.InternetAddress;
12  import javax.mail.internet.MimeMessage;
13  import java.util.Properties;
14  
15  public class MailClientImpl implements MailClient {
16  
17      private static final Logger logger = Logger.getLogger(Main.class);
18  
19      private final ConfigSettings config;
20  
21      public MailClientImpl(ConfigSettings config) {
22          this.config = config;
23      }
24  
25      public boolean sendMessage(String email, String subject, String text) {
26          if (email.isEmpty()) {
27              logger.info("Email is not sent because the address is empty");
28              return false;
29          }
30  
31          Properties properties = System.getProperties();
32          properties.setProperty("mail.smtps.host", config.getMailSmtpHost());
33          properties.setProperty("mail.smtps.port", config.getMailSmtpPort());
34          properties.setProperty("mail.smtps.auth", "true");
35          properties.setProperty("mail.user", config.getMailUser());
36          properties.setProperty("mail.password", config.getMailPassword());
37  
38          Session session = Session.getDefaultInstance(properties);
39  
40          try {
41              MimeMessage message = new MimeMessage(session);
42              message.setFrom(new InternetAddress(config.getMailFromAddress()));
43              message.addRecipient(Message.RecipientType.TO, new InternetAddress(email));
44              message.setSubject(subject);
45              message.setText(text);
46  
47              Transport trnsport = session.getTransport("smtps");
48              trnsport.connect(null, properties.getProperty("mail.password"));
49              message.saveChanges();
50              trnsport.sendMessage(message, message.getAllRecipients());
51              trnsport.close();
52  
53              return true;
54          } catch (MessagingException ex) {
55              logger.error("Error sending message to " + email, ex);
56          }
57          return false;
58      }
59  }

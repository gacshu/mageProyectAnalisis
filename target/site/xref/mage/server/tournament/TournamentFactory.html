9   import mage.game.tournament.TournamentType;

13  
14  import java.lang.reflect.Constructor;
15  import java.util.*;

17  /**
18   *
19   * @author BetaSteward_at_googlemail.com
20   */
21  public enum TournamentFactory {
22      instance;
23      private static final Logger logger = Logger.getLogger(TournamentFactory.class);
24  
25      private final Map<String, Class<Tournament>> tournaments = new HashMap<>();
26      private final Map<String, TournamentType> tournamentTypes = new HashMap<>();
27      private final List<TournamentTypeView> tournamentTypeViews = new ArrayList<>();
28  
29  
30  
31      public Tournament createTournament(String tournamentType, TournamentOptions options) {
32  
33          Tournament tournament;
34          try {
35              Constructor<Tournament> con = tournaments.get(tournamentType).getConstructor(TournamentOptions.class);
36              tournament = con.newInstance(options);
37              // transfer set information, create short info string for included sets
38              tournament.setTournamentType(tournamentTypes.get(tournamentType));
39              if (tournament.getTournamentType().isLimited()) {
40                  Map<String,Integer> setInfo = new LinkedHashMap<>();
41                  for (String setCode: options.getLimitedOptions().getSetCodes()) {
42                      tournament.getSets().add(Sets.findSet(setCode));
43                      int count = setInfo.getOrDefault(setCode, 0);
44                      setInfo.put(setCode, count + 1);
45                  }
46                  tournament.getOptions().getLimitedOptions().setNumberBoosters(tournament.getTournamentType().getNumBoosters());
47                  if (tournament.getTournamentType().isCubeBooster()) {
48                      DraftCube draftCube;
49  
50                      if (tournament.getOptions().getLimitedOptions().getCubeFromDeck() != null) {
51                          draftCube = CubeFactory.instance.createDeckDraftCube(tournament.getOptions().getLimitedOptions().getDraftCubeName(), tournament.getOptions().getLimitedOptions().getCubeFromDeck());
52                      } else {
53                          draftCube = CubeFactory.instance.createDraftCube(tournament.getOptions().getLimitedOptions().getDraftCubeName());
54                      }
55                      tournament.getOptions().getLimitedOptions().setDraftCube(draftCube);
56                      tournament.setBoosterInfo(tournament.getOptions().getLimitedOptions().getDraftCubeName());
57                  } else if (tournament.getTournamentType().isRandom()) {
58                      StringBuilder rv = new StringBuilder( "Random Draft using sets: ");
59                      for (Map.Entry<String, Integer> entry: setInfo.entrySet()){
60                          rv.append(entry.getKey());
61                          rv.append(';');
62                      }
63                      tournament.setBoosterInfo(rv.toString());
64                  } else {
65                      StringBuilder sb = new StringBuilder();
66                      for (Map.Entry<String,Integer> entry:setInfo.entrySet()) {
67                          sb.append(entry.getValue().toString()).append('x').append(entry.getKey()).append(' ');
68                      }
69                      tournament.setBoosterInfo(sb.toString());
70                  }
71              } else {
72                  tournament.setBoosterInfo("");
73              }
74  
75          } catch (Exception ex) {
76              logger.fatal("TournamentFactory error ", ex);
77              return null;
78          }
79          logger.debug("Tournament created: " + tournamentType + ' ' + tournament.getId());
80  
81          return tournament;
82      }
83  
84      public List<TournamentTypeView> getTournamentTypes() {
85          return tournamentTypeViews;
86      }
87  
88  
89      public void addTournamentType(String name, TournamentType tournamentType, Class tournament) {
90          if (tournament != null) {
91              this.tournaments.put(name, tournament);
92              this.tournamentTypes.put(name, tournamentType);
93              this.tournamentTypeViews.add(new TournamentTypeView(tournamentType));
94          }
95      }
96  
97  }

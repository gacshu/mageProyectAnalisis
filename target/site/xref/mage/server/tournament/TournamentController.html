60      private void init() {
61          tournament.addTableEventListener(
62                  (Listener<TableEvent>) event -> {
63                      switch (event.getEventType()) {
64                          case CHECK_STATE_PLAYERS:
65                              checkPlayersState();
66                              break;
67                          case INFO:
68                              managerFactory.chatManager().broadcast(chatId, "", event.getMessage(), MessageColor.BLACK, true, null, MessageType.STATUS, null);
69                              logger.debug(tournament.getId() + " " + event.getMessage());
70                              break;
71                          case START_DRAFT:
72                              startDraft(event.getDraft());
73                              break;
74                          case CONSTRUCT:
75                              if (!isAbort()) {
76                                  construct();
77                              } else {
78                                  endTournament();
79                              }
80                              break;
81                          case START_MATCH:
82                              if (!isAbort()) {
83                                  initTournament(); // set state
84                                  startMatch(event.getPair(), event.getMatchOptions());
85                              }
86                              break;
87                          case START_MULTIPLAYER_MATCH:
88                              if (!isAbort()) {
89                                  initTournament(); // set state
90                                  MatchOptions matchOptions = event.getMatchOptions();
91                                  if (matchOptions != null && event.getMultiplayerRound() != null) {
92                                      for (TournamentPlayer player : event.getMultiplayerRound().getAllPlayers()) {
93                                          matchOptions.getPlayerTypes().add(player.getPlayerType());
94                                      }
95                                  }
96  
97                                  startMultiplayerMatch(event.getMultiplayerRound(), event.getMatchOptions());
98                              }
99                              break;
100                         case END:
101                             endTournament();
102                             break;
103                     }
104                 }
105         );
106         tournament.addPlayerQueryEventListener(
107                 (Listener<PlayerQueryEvent>) event -> {
108                     try {
109                         switch (event.getQueryType()) {
110                             case CONSTRUCT:
111                                 construct(event.getPlayerId(), event.getMax());
112                                 break;
113                         }
114                     } catch (MageException ex) {
115                         logger.fatal("Player event listener error", ex);
116                     }
117                 }
118         );
119         for (TournamentPlayer player : tournament.getPlayers()) {
120             if (!player.getPlayer().isHuman()) {
121                 player.setJoined();
122                 logger.debug("player " + player.getPlayer().getId() + " has joined tournament " + tournament.getId());
123                 managerFactory.chatManager().broadcast(chatId, "", player.getPlayer().getLogName() + " has joined the tournament", MessageColor.BLACK, true, null, MessageType.STATUS, null);
124             }
125         }
126         checkStart();
127     }

129     public synchronized void join(UUID userId) {
130         UUID playerId = userPlayerMap.get(userId);
131         if (playerId == null) {
132             if (logger.isDebugEnabled()) {
133                 managerFactory.userManager().getUser(userId).ifPresent(user
134                         -> logger.debug(user.getName() + " shows tournament panel  tournamentId: " + tournament.getId()));
135 
136             }
137             return;
138         }
139         if (tournamentSessions.containsKey(playerId)) {
140             logger.debug("player reopened tournament panel userId: " + userId + " tournamentId: " + tournament.getId());
141             return;
142         }
143         // first join of player
144         TournamentSession tournamentSession = new TournamentSession(managerFactory, tournament, userId, tableId, playerId);
145         tournamentSessions.put(playerId, tournamentSession);
146         Optional<User> _user = managerFactory.userManager().getUser(userId);
147         if (_user.isPresent()) {
148             User user = _user.get();
149             user.addTournament(playerId, tournament.getId());
150             TournamentPlayer player = tournament.getPlayer(playerId);
151             player.setJoined();
152             logger.debug("player " + player.getPlayer().getName() + " - client has joined tournament " + tournament.getId());
153             managerFactory.chatManager().broadcast(chatId, "", player.getPlayer().getLogName() + " has joined the tournament", MessageColor.BLACK, true, null, MessageType.STATUS, null);
154             checkStart();
155         } else {
156             logger.error("User not found  userId: " + userId + "   tournamentId: " + tournament.getId());
157         }
158     }

160     public void rejoin(UUID playerId) {
161         TournamentSession tournamentSession = tournamentSessions.get(playerId);
162         if (tournamentSession == null) {
163             logger.fatal("Tournament session not found - playerId:" + playerId + "  tournamentId " + tournament.getId());
164             return;
165         }
166         if (!tournamentSession.init()) {
167             logger.fatal("Unable to initialize client userId: " + tournamentSession.userId + "  tournamentId " + tournament.getId());
168             return;
169         }
170         tournamentSession.update();
171     }

173     private void checkStart() {
174         if (!started && allJoined()) {
175             managerFactory.threadExecutor().getCallExecutor().execute(this::startTournament);
176         }
177     }

179     private boolean allJoined() {
180         if (!tournament.allJoined()) {
181             return false;
182         }
183         for (TournamentPlayer player : tournament.getPlayers()) {
184             if (player.getPlayer().isHuman() && tournamentSessions.get(player.getPlayer().getId()) == null) {
185                 return false;
186             }
187         }
188         return true;
189     }

191     private synchronized void startTournament() {
192         for (final TournamentSession tournamentSession : tournamentSessions.values()) {
193             if (!tournamentSession.init()) {
194                 logger.fatal("Unable to initialize client userId: " + tournamentSession.userId + "  tournamentId " + tournament.getId());
195                 //TODO: generate client error message
196                 return;
197             }
198         }
199         started = true;
200         logger.debug("Tournament starts (all players joined): " + tournament.getId() + " - " + tournament.getTournamentType().toString());
201         tournament.nextStep();
202     }

204     private void endTournament() {
205         for (TournamentPlayer player : tournament.getPlayers()) {
206             player.setStateAtTournamentEnd();
207         }
208         for (final TournamentSession tournamentSession : tournamentSessions.values()) {
209             tournamentSession.tournamentOver();
210         }
211         this.tournamentSessions.clear();
212         managerFactory.tableManager().endTournament(tableId, tournament);
213         tournament.cleanUpOnTournamentEnd();
214 
215     }

217     private void startMatch(TournamentPairing pair, MatchOptions matchOptions) {
218         try {
219             TableManager tableManager = managerFactory.tableManager();
220             Table table = tableManager.createTable(managerFactory.gamesRoomManager().getMainRoomId(), matchOptions);
221             table.setTournamentSubTable(true);
222             table.setTournament(tournament);
223             table.setState(TableState.WAITING);
224             TournamentPlayer player1 = pair.getPlayer1();
225             TournamentPlayer player2 = pair.getPlayer2();
226             UUID user1Uuid = null;
227             UUID user2Uuid = null;
228             if (player1.getPlayerType() == PlayerType.HUMAN) {
229                 Optional<UUID> user1Id = getPlayerUserId(player1.getPlayer().getId());
230                 if (!user1Id.isPresent()) {
231                     logger.fatal("Player 1 not found");
232                 } else {
233                     user1Uuid = user1Id.get();
234                 }
235             }
236             if (player2.getPlayerType() == PlayerType.HUMAN) {
237                 Optional<UUID> user2Id = getPlayerUserId(player2.getPlayer().getId());
238                 if (!user2Id.isPresent()) {
239                     logger.fatal("Player 2 not found");
240                 } else {
241                     user2Uuid = user2Id.get();
242                 }
243             }
244             tableManager.addPlayer(user1Uuid, table.getId(), player1);
245             tableManager.addPlayer(user2Uuid, table.getId(), player2);
246             table.setState(TableState.STARTING);
247             tableManager.startTournamentSubMatch(null, table.getId());
248             tableManager.getMatch(table.getId()).ifPresent(match -> {
249                 match.setTableId(tableId);
250                 pair.setMatch(match);
251                 pair.setTableId(table.getId());
252                 player1.setState(TournamentPlayerState.DUELING);
253                 player2.setState(TournamentPlayerState.DUELING);
254             });
255         } catch (GameException ex) {
256             logger.fatal("TournamentController startMatch error", ex);
257         }
258     }

260     private void startMultiplayerMatch(MultiplayerRound round, MatchOptions matchOptions) {
261         try {
262             TableManager tableManager = managerFactory.tableManager();
263             Table table = tableManager.createTable(managerFactory.gamesRoomManager().getMainRoomId(), matchOptions);
264             table.setTournamentSubTable(true);
265             table.setTournament(tournament);
266             table.setState(TableState.WAITING);
267             if (round.getAllPlayers().stream().allMatch(tournamentPlayer -> getPlayerUserId(tournamentPlayer.getPlayer().getId()).isPresent())) {
268                 for (TournamentPlayer player : round.getAllPlayers()) {
269                     tableManager.addPlayer(getPlayerUserId(player.getPlayer().getId()).get(), table.getId(), player);
270                 }
271                 table.setState(TableState.STARTING);
272                 tableManager.startTournamentSubMatch(null, table.getId());
273                 tableManager.getMatch(table.getId()).ifPresent(match -> {
274                     match.setTableId(tableId);
275                     round.setMatch(match);
276                     round.setTableId(table.getId());
277                     for (TournamentPlayer player : round.getAllPlayers()) {
278                         player.setState(TournamentPlayerState.DUELING);
279                     }
280                 });
281             }
282         } catch (GameException ex) {
283             logger.fatal("TournamentController startMatch error", ex);
284         }
285     }

301     private void construct(UUID playerId, int timeout) throws MageException {
302         if (tournamentSessions.containsKey(playerId)) {
303             TournamentSession tournamentSession = tournamentSessions.get(playerId);
304             tournamentSession.construct(timeout);
305             getPlayerUserId(playerId).ifPresent(userId -> {
306                 managerFactory.userManager().getUser(userId).ifPresent(user -> {
307                     user.addConstructing(playerId, tournamentSession);
308                     TournamentPlayer player = tournament.getPlayer(playerId);
309                     player.setState(TournamentPlayerState.CONSTRUCTING);
310                 });
311             });
312         }

332     public void timeout(UUID userId) {
333         if (userPlayerMap.containsKey(userId)) {
334             TournamentPlayer tournamentPlayer = tournament.getPlayer(userPlayerMap.get(userId));
335             if (tournamentPlayer.getDeck() != null) {
336                 tournament.autoSubmit(userPlayerMap.get(userId), tournamentPlayer.generateDeck());
337             } else {
338                 StringBuilder sb = new StringBuilder();
339                 managerFactory.userManager().getUser(userId).ifPresent(user
340                         -> sb.append(user.getName()));
341 
342                 sb.append(" - no deck found for auto submit");
343                 logger.fatal(sb);
344                 tournamentPlayer.setEliminated();
345                 tournamentPlayer.setStateInfo("No deck for auto submit");
346             }
347         }

354     public void quit(UUID userId) {
355         UUID playerId = userPlayerMap.get(userId);
356         if (playerId == null) {
357             logger.debug("Player not found userId:" + userId + " tournId: " + tournament.getId());
358             return;
359         }
360         TournamentPlayer tournamentPlayer = tournament.getPlayer(playerId);
361         if (tournamentPlayer == null) {
362             logger.debug("TournamentPlayer not found userId: " + userId + " tournId: " + tournament.getId());
363             return;
364         }
365         if (!started) {
366             tournament.leave(playerId);
367             return;
368         }
369         TournamentSession tournamentSession = tournamentSessions.get(playerId);
370         if (tournamentSession == null) {
371             logger.debug("TournamentSession not found userId: " + userId + " tournId: " + tournament.getId());
372             return;
373         }
374         tournamentSession.setKilled();
375         if (tournamentPlayer.isInTournament()) {
376             String info;
377             TourneyQuitStatus status;
378             if (tournament.isDoneConstructing()) {
379                 info = new StringBuilder("during round ").append(tournament.getRounds().size()).toString();
380                 // quit active matches of that tournament
381                 managerFactory.tableManager().userQuitTournamentSubTables(tournament.getId(), userId);
382                 status = TourneyQuitStatus.DURING_ROUND;
383             } else if (tournamentPlayer.getState() == TournamentPlayerState.DRAFTING) {
384                 info = "during Draft phase";
385                 if (!checkToReplaceDraftPlayerByAi(userId, tournamentPlayer)) {
386                     this.abortDraftTournament();
387                 } else {
388                     managerFactory.draftManager().getController(tableId).ifPresent(draftController -> {
389                         draftController.getDraftSession(playerId).ifPresent(draftSession
390                                 -> managerFactory.draftManager().kill(draftSession.getDraftId(), userId));
391 
392                     });
393                 }
394                 status = TourneyQuitStatus.DURING_DRAFTING;
395             } else if (tournamentPlayer.getState() == TournamentPlayerState.CONSTRUCTING) {
396                 info = "during Construction phase";
397                 status = TourneyQuitStatus.DURING_CONSTRUCTION;
398             } else {
399                 info = "";
400                 status = TourneyQuitStatus.NO_TOURNEY_QUIT;
401             }
402             tournamentPlayer.setQuit(info, status);
403             tournament.quit(playerId);
404             tournamentSession.quit();
405             managerFactory.chatManager().broadcast(chatId, "", tournamentPlayer.getPlayer().getLogName() + " has quit the tournament", MessageColor.BLACK, true, null, MessageType.STATUS, SoundToPlay.PlayerQuitTournament);
406         }
407     }

409     private boolean checkToReplaceDraftPlayerByAi(UUID userId, TournamentPlayer leavingPlayer) {
410         int humans = 0;
411         for (TournamentPlayer tPlayer : tournament.getPlayers()) {
412             if (tPlayer.getPlayer().isHuman()) {
413                 humans++;
414             }
415         }
416         // replace player that quits with draft bot
417         if (humans > 1) {
418             Optional<User> user = managerFactory.userManager().getUser(userId);
419             managerFactory.tableManager().getController(tableId).ifPresent(tableController -> {
420 
421                 String replacePlayerName = "Draftbot";
422                 if (user.isPresent()) {
423                     replacePlayerName = "Draftbot (" + user.get().getName() + ')';
424                 }
425                 tableController.replaceDraftPlayer(leavingPlayer.getPlayer(), replacePlayerName, PlayerType.COMPUTER_DRAFT_BOT, 5);
426                 if (user.isPresent()) {
427                     user.get().removeDraft(leavingPlayer.getPlayer().getId());
428                     user.get().removeTable(leavingPlayer.getPlayer().getId());
429                     user.get().removeTournament(leavingPlayer.getPlayer().getId());
430                 }
431                 managerFactory.chatManager().broadcast(chatId, "", leavingPlayer.getPlayer().getLogName() + " was replaced by draftbot", MessageColor.BLACK, true, null, MessageType.STATUS, null);
432             });
433             return true;
434         }
435         return false;
436     }

455     private void checkPlayersState() {
456         for (TournamentPlayer tournamentPlayer : tournament.getPlayers()) {
457             if (!tournamentPlayer.isEliminated() && tournamentPlayer.getPlayer().isHuman()) {
458                 if (tournamentSessions.containsKey(tournamentPlayer.getPlayer().getId())) {
459                     if (tournamentSessions.get(tournamentPlayer.getPlayer().getId()).isKilled()) {
460                         tournamentPlayer.setEliminated();
461                         tournamentPlayer.setStateInfo("disconnected");
462                     }
463                 } else {
464                     tournamentPlayer.setEliminated();
465                     tournamentPlayer.setStateInfo("no tournament session");
466                 }
467             }
468         }

476     /**
477      * Check tournaments that are not already finished, if they are in a still
478      * valid state
479      *
480      * @param tableState state of the tournament table
481      * @return true - if tournament is valid false - if tournament is not valid
482      * and should be removed
483      */
484     public boolean isTournamentStillValid(TableState tableState) {
485         int activePlayers = 0;
486         for (Entry<UUID, UUID> entry : userPlayerMap.entrySet()) {
487             TournamentPlayer tournamentPlayer = tournament.getPlayer(entry.getValue());
488             if (tournamentPlayer != null) {
489                 if (!tournamentPlayer.hasQuit()) {
490                     if (tournamentPlayer.getPlayer().isHuman()) {
491                         Optional<User> user = managerFactory.userManager().getUser(entry.getKey());
492                         if (!user.isPresent()) {
493                             logger.debug("Tournament user is missing but player active -> start quit - tournamentId: " + tournament.getId() + " state: " + tableState.toString());
494                             // active tournament player but the user is no longer online
495                             quit(entry.getKey());
496                         } else {
497                             activePlayers++;
498                         }
499                     }
500                 }
501             } else {
502                 // tournament player is missing
503                 logger.debug("Tournament player is missing - tournamentId: " + tournament.getId() + " state: " + tableState.toString());
504             }
505         }
506         for (TournamentPlayer tournamentPlayer : tournament.getPlayers()) {
507             if (!tournamentPlayer.getPlayer().isHuman()) {
508                 if (!tournamentPlayer.hasQuit()) {
509                     activePlayers++;
510                 }
511             }
512         }
513         if (activePlayers < 2 && tableState != TableState.WAITING) {
514             logger.debug("Tournament has less than 2 active players - tournamentId: " + tournament.getId() + " state: " + tableState.toString());
515             return false;
516         }
517         return true;
518     }

2   package mage.server.game;
3   
4   import java.util.UUID;
5   import java.util.concurrent.Callable;
6   
7   import mage.MageException;
8   import mage.game.Game;
9   import org.apache.log4j.Logger;
10  
11  /**
12   * @param <T>
13   * @author BetaSteward_at_googlemail.com
14   */
15  public class GameWorker<T> implements Callable {
16  
17      private static final Logger LOGGER = Logger.getLogger(GameWorker.class);
18  
19      private final GameCallback gameController;
20      private final Game game;
21      private final UUID choosingPlayerId;
22  
23      public GameWorker(Game game, UUID choosingPlayerId, GameCallback gameController) {
24          this.game = game;
25          this.choosingPlayerId = choosingPlayerId;
26          this.gameController = gameController;
27      }
28  
29      @Override
30      public Object call() {
31          try {
32              LOGGER.debug("GAME WORKER started gameId " + game.getId());
33              Thread.currentThread().setName("GAME " + game.getId());
34              game.start(choosingPlayerId);
35              game.fireUpdatePlayersEvent();
36              gameController.gameResult(game.getWinner());
37              game.cleanUp();
38          } catch (MageException ex) {
39              LOGGER.fatal("GameWorker mage error [" + game.getId() + "] " + ex, ex);
40          } catch (Exception e) {
41              LOGGER.fatal("GameWorker general exception [" + game.getId() + "] " + e.getMessage(), e);
42              if (e instanceof NullPointerException) {
43                  LOGGER.info(e.getStackTrace());
44              }
45          } catch (Error err) {
46              LOGGER.fatal("GameWorker general error [" + game.getId() + "] " + err, err);
47          }
48          return null;
49      }
50  
51  }

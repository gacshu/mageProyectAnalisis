32      private static final Logger LOGGER = Logger.getLogger(GamesRoomImpl.class);
33  
34      private static final ScheduledExecutorService UPDATE_EXECUTOR = Executors.newSingleThreadScheduledExecutor();
35      private static List<TableView> tableView = new ArrayList<>();
36      private static List<MatchView> matchView = new ArrayList<>();
37      private static List<RoomUsersView> roomUsersView = new ArrayList<>();
38  
39      private final ManagerFactory managerFactory;
40      private final ConcurrentHashMap<UUID, Table> tables = new ConcurrentHashMap<>();
41  
42      public GamesRoomImpl(ManagerFactory managerFactory) {
43          super(managerFactory.chatManager());
44          this.managerFactory = managerFactory;
45          UPDATE_EXECUTOR.scheduleAtFixedRate(() -> {
46              try {
47                  update();
48              } catch (Exception ex) {
49                  LOGGER.fatal("Games room update exception! " + ex.toString(), ex);
50              }
51  
52          }, 2, 2, TimeUnit.SECONDS);
53      }
54  
55      @Override
56      public List<TableView> getTables() {
57          return tableView;
58      }
59  
60      private void update() {
61          List<Table> allTables = new ArrayList<>(tables.values());
62          allTables.sort(new TableListSorter());
63          List<MatchView> matchList = new ArrayList<>();
64          List<TableView> tableList = new ArrayList<>();
65          for (Table table : allTables) {
66              if (table.getState() != TableState.FINISHED) {
67                  tableList.add(new TableView(table));
68              } else if (matchList.size() < 50) {
69                  matchList.add(new MatchView(table));
70              } else {
71                  // more since 50 matches finished since this match so removeUserFromAllTablesAndChat it
72                  if (table.isTournament()) {
73                      managerFactory.tournamentManager().removeTournament(table.getTournament().getId());
74                  }
75                  this.removeTable(table.getId());
76              }
77          }
78          tableView = tableList;
79          matchView = matchList;
80          List<UsersView> users = new ArrayList<>();
81          for (User user : managerFactory.userManager().getUsers()) {
82              if (user.getUserState() != User.UserState.Offline && !user.getName().equals("Admin")) {
83                  try {
84                      users.add(new UsersView(user.getUserData().getFlagName(), user.getName(),
85                              user.getMatchHistory(), user.getMatchQuitRatio(), user.getTourneyHistory(),
86                              user.getTourneyQuitRatio(), user.getGameInfo(), user.getPingInfo(),
87                              user.getUserData().getGeneralRating(), user.getUserData().getConstructedRating(),
88                              user.getUserData().getLimitedRating()));
89                  } catch (Exception ex) {
90                      LOGGER.fatal("User update exception: " + user.getName() + " - " + ex.toString(), ex);
91                      users.add(new UsersView(
92                              (user.getUserData() != null && user.getUserData().getFlagName() != null) ? user.getUserData().getFlagName() : "world",
93                              user.getName() != null ? user.getName() : "<no name>",
94                              user.getMatchHistory() != null ? user.getMatchHistory() : "<no match history>",
95                              user.getMatchQuitRatio(),
96                              user.getTourneyHistory() != null ? user.getTourneyHistory() : "<no tourney history>",
97                              user.getTourneyQuitRatio(),
98                              "[exception]",
99                              user.getPingInfo() != null ? user.getPingInfo() : "<no ping>",
100                             user.getUserData() != null ? user.getUserData().getGeneralRating() : 0,
101                             user.getUserData() != null ? user.getUserData().getConstructedRating() : 0,
102                             user.getUserData() != null ? user.getUserData().getLimitedRating() : 0));
103                 }
104             }
105         }
106 
107         users.sort((one, two) -> one.getUserName().compareToIgnoreCase(two.getUserName()));
108         List<RoomUsersView> roomUserInfo = new ArrayList<>();
109         roomUserInfo.add(new RoomUsersView(users,
110                 managerFactory.gameManager().getNumberActiveGames(),
111                 managerFactory.threadExecutor().getActiveThreads(managerFactory.threadExecutor().getGameExecutor()),
112                 managerFactory.configSettings().getMaxGameThreads()
113         ));
114         roomUsersView = roomUserInfo;
115     }

154     @Override
155     public Optional<TableView> getTable(UUID tableId) {
156         if (tables.containsKey(tableId)) {

160     }

163     public void removeTable(UUID userId, UUID tableId) {
164         tables.remove(tableId);
165     }

167     @Override
168     public void removeTable(UUID tableId) {
169         Table table = tables.get(tableId);
170         if (table != null) {
171             table.cleanUp();
172             tables.remove(tableId);
173             if (LOGGER.isDebugEnabled()) {
174                 LOGGER.debug("Table removed: " + tableId);
175             }
176         }
177     }

182     }

196 /**
197  * Sorts the tables for table and match view of the client room
198  *
199  * @author LevelX2
200  */
201 class TableListSorter implements Comparator<Table> {
202 
203     @Override
204     public int compare(Table one, Table two) {
205         if (one.getState() != null && two.getState() != null) {
206             if (TableState.SIDEBOARDING != one.getState() && TableState.DUELING != one.getState()) {
207                 if (one.getState().compareTo(two.getState()) != 0) {
208                     return one.getState().compareTo(two.getState());
209                 }
210             } else if (TableState.SIDEBOARDING != two.getState() && TableState.DUELING != two.getState()) {
211                 if (one.getState().compareTo(two.getState()) != 0) {
212                     return one.getState().compareTo(two.getState());
213                 }
214             }
215         }
216         if (two.getEndTime() != null) {
217             if (one.getEndTime() == null) {
218                 return 1;
219             } else {
220                 return two.getEndTime().compareTo(one.getEndTime());
221             }
222         } else if (one.getEndTime() != null) {
223             return -1;
224         }
225 
226         if (two.getStartTime() != null) {
227             if (one.getStartTime() == null) {
228                 return 1;
229             } else {
230                 return two.getStartTime().compareTo(one.getStartTime());
231             }
232         } else if (one.getStartTime() != null) {
233             return -1;
234         }
235 
236         if (two.getCreateTime() != null) {
237             if (one.getCreateTime() == null) {
238                 return 1;
239             } else {
240                 return two.getCreateTime().compareTo(one.getCreateTime());
241             }
242         } else if (one.getCreateTime() != null) {
243             return -1;
244         }
245         return 0;
246     }
247 }

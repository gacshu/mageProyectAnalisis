3   package mage.server.game;
4   
5   import mage.constants.RangeOfInfluence;
6   import mage.players.Player;
7   import mage.players.PlayerType;
8   import org.apache.log4j.Logger;
9   
10  import java.lang.reflect.Constructor;
11  import java.util.EnumMap;
12  import java.util.Optional;
13  import java.util.Set;
14  
15  /**
16   * @author BetaSteward_at_googlemail.com
17   */
18  public enum PlayerFactory {
19  
20      instance;
21      private static final Logger logger = Logger.getLogger(PlayerFactory.class);
22  
23      private final EnumMap<PlayerType, Class> playerTypes = new EnumMap<>(PlayerType.class);
24  
25  
26      public Optional<Player> createPlayer(PlayerType playerType, String name, RangeOfInfluence range, int skill) {
27          try {
28              Class playerTypeClass = playerTypes.get(playerType);
29              if (playerTypeClass != null) {
30                  Constructor<?> con = playerTypeClass.getConstructor(String.class, RangeOfInfluence.class, int.class);
31                  Player player = (Player) con.newInstance(name, range, skill);
32                  logger.trace("Player created: " + name + " - " + player.getId());
33                  return Optional.of(player);
34              } else {
35                  logger.fatal("Unknown player type: " + playerType);
36              }
37          } catch (Exception ex) {
38              logger.fatal("PlayerFactory error ", ex);
39          }
40          return Optional.empty();
41      }
42  
43      public Set<PlayerType> getPlayerTypes() {
44          return playerTypes.keySet();
45      }
46  
47      public void addPlayerType(String name, Class playerType) {
48          PlayerType type = PlayerType.getByDescription(name);
49          if (type != null) {
50              if (playerType != null) {
51                  this.playerTypes.put(type, playerType);
52              }
53          }
54      }
55  
56  }

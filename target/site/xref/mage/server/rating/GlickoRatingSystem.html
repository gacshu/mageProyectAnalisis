2   package mage.server.rating;
3   
4   import java.util.Date;
5   
6   /**
7    *
8    * @author Quercitron
9    */
10  public class GlickoRatingSystem {
11  
12      // rating deviation will grow back from 50 to max 350 in 2 years
13      public static final double C = 0.00137934314767061324980397708525;
14  
15      public static final double BaseRating = 1500;
16      public static final double BaseRD = 350;
17      public static final double MinRD = 30;
18  
19      private static final double Q = Math.log(10) / 400;
20  
21      public static GlickoRating getInitialRating() {
22          return new GlickoRating(GlickoRatingSystem.BaseRating, GlickoRatingSystem.BaseRD, 0);
23      }
24  
25      public static int getDisplayedRating(GlickoRating rating) {
26          long currentTime = new Date().getTime();
27          double updatedRatingDeviation = getUpdatedRD(rating, currentTime);
28          return (int) Math.max(rating.getRating() - 2 * updatedRatingDeviation, 0);
29      }
30  
31      public static int getDefaultDisplayedRating() {
32          return getDisplayedRating(getInitialRating());
33      }
34  
35      public void updateRating(GlickoRating playerRating, GlickoRating opponentRating, double outcome, long gameTimeMs) {
36          playerRating.setRatingDeviation(getUpdatedRD(playerRating, gameTimeMs));
37          opponentRating.setRatingDeviation(getUpdatedRD(opponentRating, gameTimeMs));
38  
39          GlickoRating newPlayerRating = getNewRating(playerRating, opponentRating, outcome);
40          GlickoRating newOpponentRating = getNewRating(opponentRating, playerRating, 1 - outcome);
41  
42          playerRating.setRating(newPlayerRating.getRating());
43          playerRating.setRatingDeviation(newPlayerRating.getRatingDeviation());
44          playerRating.setLastGameTimeMs(gameTimeMs);
45  
46          opponentRating.setRating(newOpponentRating.getRating());
47          opponentRating.setRatingDeviation(newOpponentRating.getRatingDeviation());
48          opponentRating.setLastGameTimeMs(gameTimeMs);
49      }
50  
51      private static double getUpdatedRD(GlickoRating rating, long gameTimeMs) {
52          double newRatingDeviation;
53          if (rating.getLastGameTimeMs() != 0)
54          {
55              double newRD = Math.sqrt(
56                      rating.getRatingDeviation() * rating.getRatingDeviation()
57                              + C * C * Math.max(gameTimeMs - rating.getLastGameTimeMs(), 0));
58              newRatingDeviation = Math.max(Math.min(BaseRD, newRD), MinRD);
59          }
60          else
61          {
62              newRatingDeviation = BaseRD;
63          }
64          return newRatingDeviation;
65      }
66  
67      private GlickoRating getNewRating(GlickoRating playerRating, GlickoRating opponentRating, double outcome) {
68          double RD = playerRating.getRatingDeviation();
69  
70          double g = gFunc(opponentRating.getRatingDeviation());
71          double p = -g * (playerRating.getRating() - opponentRating.getRating()) / 400;
72          double e = 1 / (1 + Math.pow(10, p));
73          double d2 = 1 / (Q * Q * g * g * e * (1 - e));
74  
75          // todo: set minimum K?
76          double newRating = playerRating.getRating() + Q / (1 / RD / RD + 1 / d2) * g * (outcome - e);
77          double newRD = Math.sqrt(1 / (1 / RD / RD + 1 / d2));
78  
79          return new GlickoRating(newRating, newRD);
80      }
81  
82      private double gFunc(double rd) {
83          return 1 / Math.sqrt(1 + 3 * Q * Q * rd * rd / Math.PI / Math.PI);
84      }
85  
86  }

63       public void start() {
64           updateUsersTask = new UpdateUsersTask(ConsoleFrame.getSession(), this);
65           updateTablesTask = new UpdateTablesTask(ConsoleFrame.getSession(), ConsoleFrame.getSession().getMainRoomId(), this);
66           updateUsersTask.execute();
67           updateTablesTask.execute();
68       }

70       public void stop() {
71           if (updateUsersTask != null && !updateUsersTask.isDone()) {
72               updateUsersTask.cancel(true);
73           }
74           if (updateTablesTask != null && !updateTablesTask.isDone()) {
75               updateTablesTask.cancel(true);
76           }
77       }

338      private void btnDeActivateActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnDeActivateActionPerformed
339          String userName;
340          if (!getjUserName().getText().isEmpty()) {
341              userName = getjUserName().getText();
342          } else {
343              int row = this.tblUsers.convertRowIndexToModel(tblUsers.getSelectedRow());
344              userName = (String) tableUserModel.getValueAt(row, TableUserModel.POS_USER_NAME);
345          }
346 
347          if (JOptionPane.showConfirmDialog(null, "Did you want to set user: " + userName + " to active?", "WARNING",
348                  JOptionPane.YES_NO_OPTION) == JOptionPane.YES_OPTION) {
349              ConsoleFrame.getSession().setActivation(userName, true);
350              return;
351          }
352          if (JOptionPane.showConfirmDialog(null, "Did you want to set user: " + userName + " to inactive?", "WARNING",
353                  JOptionPane.YES_NO_OPTION) == JOptionPane.YES_OPTION) {
354              ConsoleFrame.getSession().setActivation(userName, false);
355              return;
356          }
357          if (JOptionPane.showConfirmDialog(null, "Are you sure you mean to toggle activation for user: " + userName + '?', "WARNING",
358                  JOptionPane.YES_NO_OPTION) == JOptionPane.YES_OPTION) {
359              ConsoleFrame.getSession().toggleActivation(userName);
360              return;
361          }
362      }//GEN-LAST:event_btnDeActivateActionPerformed

404  class TableUserModel extends AbstractTableModel {
405 
406      public static final int POS_USER_NAME = 0;
407      public static final int POS_HOST = 1;
408      public static final int POS_TIME_CONNECTED = 2;
409      public static final int POS_LAST_ACTIVITY = 3;
410      public static final int POS_SESSION_ID = 4;
411      public static final int POS_GAME_INFO = 5;
412      public static final int POS_USER_STATE = 6;
413      public static final int POS_CHAT_MUTE = 7;
414      public static final int POS_CLIENT_VERSION = 8;
415 
416      private final String[] columnNames = new String[]{"User Name", "Host", "Time Connected", "Last activity", "SessionId", "Gameinfo", "User state", "Chat mute", "Client Version"};
417      private UserView[] users = new UserView[0];
418      private static final DateFormat formatterTime = new SimpleDateFormat("HH:mm:ss");
419      private static final DateFormat formatterTimeStamp = new SimpleDateFormat("yy-M-dd HH:mm:ss");
420 
421      public void loadData(List<UserView> users) {
422          this.users = users.toArray(new UserView[0]);
423          this.fireTableDataChanged();
424      }
425 
426      @Override
427      public int getRowCount() {
428          return users.length;
429      }
430 
431      @Override
432      public int getColumnCount() {
433          return columnNames.length;
434      }
435 
436      @Override
437      public Object getValueAt(int arg0, int arg1) {
438          switch (arg1) {
439              case POS_USER_NAME:
440                  return users[arg0].getUserName();
441              case POS_HOST:
442                  return users[arg0].getHost();
443              case POS_TIME_CONNECTED:
444                  return formatterTime.format(users[arg0].getTimeConnected());
445              case POS_LAST_ACTIVITY:
446                  return formatterTime.format(users[arg0].getLastActivity());
447              case POS_SESSION_ID:
448                  return users[arg0].getSessionId();
449              case POS_GAME_INFO:
450                  return users[arg0].getGameInfo();
451              case POS_USER_STATE:
452                  return users[arg0].getUserState();
453              case POS_CHAT_MUTE:
454                  if (users[arg0].getMuteChatUntil() == null) {
455                      return "";
456                  }
457                  return formatterTimeStamp.format(users[arg0].getMuteChatUntil());
458              case POS_CLIENT_VERSION:
459                  return users[arg0].getClientVersion();
460          }
461          return "";
462      }
463 
464      @Override
465      public String getColumnName(int columnIndex) {
466          String colName = "";
467 
468          if (columnIndex <= getColumnCount()) {
469              colName = columnNames[columnIndex];
470          }
471 
472          return colName;
473      }
474 
475      @Override
476      public Class getColumnClass(int columnIndex) {
477          return String.class;
478      }
479 
480      @Override
481      public boolean isCellEditable(int rowIndex, int columnIndex) {
482          return false;
483      }

487  class TableTableModel extends AbstractTableModel {
488 
489      private final String[] columnNames = new String[]{"Table Name", "Owner", "Game Type", "Deck Type", "Status"};
490      private TableView[] tables = new TableView[0];
491 
492      public void loadData(Collection<TableView> tables) {
493          this.tables = tables.toArray(new TableView[0]);
494          this.fireTableDataChanged();
495      }
496 
497      @Override
498      public int getRowCount() {
499          return tables.length;
500      }
501 
502      @Override
503      public int getColumnCount() {
504          return columnNames.length;
505      }
506 
507      @Override
508      public Object getValueAt(int arg0, int arg1) {
509          switch (arg1) {
510              case 0:
511                  return tables[arg0].getTableName();
512              case 1:
513                  return tables[arg0].getControllerName();
514              case 2:
515                  return tables[arg0].getGameType();
516              case 3:
517                  return tables[arg0].getDeckType();
518              case 4:
519                  return tables[arg0].getTableState().toString();
520              case 5:
521                  return tables[arg0].isTournament();
522              case 6:
523                  if (!tables[arg0].getGames().isEmpty()) {
524                      return tables[arg0].getGames().get(0);
525                  }
526                  return null;
527              case 7:
528                  return tables[arg0].getTableId();
529          }
530          return "";
531      }
532 
533      @Override
534      public String getColumnName(int columnIndex) {

603          }
604 
605          for (UserView u1 : usersToCheck) {
606              String s = u1.getUserName() + ',' + u1.getHost();

611          }

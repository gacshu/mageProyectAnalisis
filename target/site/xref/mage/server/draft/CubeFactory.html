2   package mage.server.draft;
3   
4   import mage.cards.decks.Deck;
5   import mage.game.draft.DraftCube;
6   import org.apache.log4j.Logger;
7   
8   import java.lang.reflect.Constructor;
9   import java.util.LinkedHashMap;
10  import java.util.Map;
11  import java.util.Set;
12  
13  /**
14   *
15   * @author LevelX2
16   */
17  public enum CubeFactory {
18  
19      instance;
20      private static final Logger logger = Logger.getLogger(CubeFactory.class);
21  
22      private final Map<String, Class> draftCubes = new LinkedHashMap<>();
23  
24  
25  
26      public DraftCube createDraftCube(String draftCubeName) {
27  
28          DraftCube draftCube;
29          try {
30              Constructor<?> con = draftCubes.get(draftCubeName).getConstructor();
31              draftCube = (DraftCube)con.newInstance();
32          } catch (Exception ex) {
33              logger.fatal("CubeFactory error", ex);
34              return null;
35          }
36          logger.debug("Draft cube created: " + draftCube.getName());
37  
38          return draftCube;
39      }
40  
41      public DraftCube createDeckDraftCube(String draftCubeName, Deck cubeFromDeck) {
42  
43          DraftCube draftCube;
44          try {
45              Constructor<?> con = draftCubes.get(draftCubeName).getConstructor(Deck.class);
46              draftCube = (DraftCube)con.newInstance(cubeFromDeck);
47          } catch (Exception ex) {
48              logger.fatal("CubeFactory error", ex);
49              return null;
50          }
51          logger.debug("Draft cube created: " + draftCube.getName());
52  
53          return draftCube;
54      }
55  
56      public Set<String> getDraftCubes() {
57          return draftCubes.keySet();
58      }
59  
60      public void addDraftCube(String name, Class draftCube) {
61          if (draftCube != null) {
62              this.draftCubes.put(name, draftCube);
63          }
64      }
65  
66  }

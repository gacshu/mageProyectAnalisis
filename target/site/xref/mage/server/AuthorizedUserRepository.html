27      private static final String JDBC_URL = "jdbc:h2:file:./db/authorized_user.h2;AUTO_SERVER=TRUE";
28      private static final String VERSION_ENTITY_NAME = "authorized_user";
29      // raise this if db structure was changed
30      private static final long DB_VERSION = 2;
31      private static final RandomNumberGenerator rng = new SecureRandomNumberGenerator();
32  
33      private static final AuthorizedUserRepository instance;
34      static {
35          instance = new AuthorizedUserRepository(JDBC_URL);
36      }
37  
38      private Dao<AuthorizedUser, Object> dao;
39  
40      public AuthorizedUserRepository(String connectionString) {
41          File file = new File("db");
42          if (!file.exists()) {
43              file.mkdirs();
44          }
45          try {
46              ConnectionSource connectionSource = new JdbcConnectionSource(connectionString);
47              TableUtils.createTableIfNotExists(connectionSource, AuthorizedUser.class);
48              dao = DaoManager.createDao(connectionSource, AuthorizedUser.class);
49          } catch (SQLException ex) {
50              Logger.getLogger(AuthorizedUserRepository.class).error("Error creating / assigning authorized_user repository - ", ex);
51          }
52      }
53  
54      public static AuthorizedUserRepository getInstance() {
55          return instance;
56      }
57  
58      public void add(final String userName, final String password, final String email) {
59          try {
60              Hash hash = new SimpleHash(Sha256Hash.ALGORITHM_NAME, password, rng.nextBytes(), 1024);
61              AuthorizedUser user = new AuthorizedUser(userName, hash, email);
62              dao.create(user);
63          } catch (SQLException ex) {
64              Logger.getLogger(AuthorizedUserRepository.class).error("Error adding a user to DB - ", ex);
65          }
66      }
67  
68      public void remove(final String userName) {
69          try {
70              DeleteBuilder<AuthorizedUser, Object> db = dao.deleteBuilder();
71              db.where().eq("name", new SelectArg(userName));
72              db.delete();
73          } catch (SQLException ex) {
74              Logger.getLogger(AuthorizedUserRepository.class).error("Error removing a user from DB - ", ex);
75          }
76      }
77  
78      public AuthorizedUser getByName(String userName) {
79          try {
80              QueryBuilder<AuthorizedUser, Object> qb = dao.queryBuilder();
81              qb.where().eq("name", new SelectArg(userName));
82              List<AuthorizedUser> results = dao.query(qb.prepare());
83              if (results.size() == 1) {
84                  return results.get(0);
85              }
86              return null;
87          } catch (SQLException ex) {
88              Logger.getLogger(AuthorizedUserRepository.class).error("Error getting a authorized_user - ", ex);
89          }
90          return null;
91      }
92  
93      public void update(AuthorizedUser authorizedUser) {
94          try {
95              dao.update(authorizedUser);
96          } catch (SQLException ex) {
97              Logger.getLogger(AuthorizedUserRepository.class).error("Error updating authorized_user", ex);
98          }
99      }

116     public void closeDB() {
117         try {
118             if (dao != null && dao.getConnectionSource() != null) {
119                 DatabaseConnection conn = dao.getConnectionSource().getReadWriteConnection(dao.getTableName());
120                 conn.executeStatement("shutdown compact", 0);
121             }
122         } catch (SQLException ex) {
123             Logger.getLogger(AuthorizedUserRepository.class).error("Error closing authorized_user repository - ", ex);
124         }
125     }

127     public long getDBVersionFromDB() {
128         try {
129             ConnectionSource connectionSource = new JdbcConnectionSource(JDBC_URL);
130             return RepositoryUtil.getDatabaseVersion(connectionSource, VERSION_ENTITY_NAME);
131         } catch (SQLException ex) {
132             Logger.getLogger(CardRepository.class).error("Error getting DB version from DB - ", ex);

134         return 0;
135     }

137     public boolean checkAlterAndMigrateAuthorizedUser() {
138         long currentDBVersion = getDBVersionFromDB();
139         if (currentDBVersion == 1 && DB_VERSION == 2) {
140             return migrateFrom1To2();
141         }
142         return true;
143     }
144 
145     private boolean migrateFrom1To2() {
146         try {
147             Logger.getLogger(AuthorizedUserRepository.class).info("Starting " + VERSION_ENTITY_NAME + " DB migration from version 1 to version 2");
148             dao.executeRaw("ALTER TABLE authorized_user ADD COLUMN active BOOLEAN DEFAULT true;");
149             dao.executeRaw("ALTER TABLE authorized_user ADD COLUMN lockedUntil DATETIME;");
150             dao.executeRaw("ALTER TABLE authorized_user ADD COLUMN chatLockedUntil DATETIME;");
151             dao.executeRaw("ALTER TABLE authorized_user ADD COLUMN lastConnection DATETIME;");
152             RepositoryUtil.updateVersion(dao.getConnectionSource(), VERSION_ENTITY_NAME, DB_VERSION);
153             Logger.getLogger(AuthorizedUserRepository.class).info("Migration finished.");
154             return true;
155         } catch (SQLException ex) {
156             Logger.getLogger(AuthorizedUserRepository.class).error("Error while migrating from version 1 to version 2 - ", ex);
157             return false;
158         }
159     }

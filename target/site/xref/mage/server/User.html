133     public void setSessionId(String sessionId) {
134         this.sessionId = sessionId;
135         if (sessionId.isEmpty()) {
136             setUserState(UserState.Disconnected);
137             lostConnection();
138             logger.trace("USER - lost connection: " + userName + " id: " + userId);
139 
140         } else if (userState == UserState.Created) {
141             setUserState(UserState.Connected);
142             logger.trace("USER - created: " + userName + " id: " + userId);
143         } else {
144             setUserState(UserState.Connected);
145             reconnect();
146             logger.trace("USER - reconnected: " + userName + " id: " + userId);
147         }
148     }

184             UUID gameId = iterator.next();

215     public String getConnectionDuration() {
216         int minutes = (int) SystemUtil.getDateDiff(connectionTime, new Date(), TimeUnit.SECONDS) / 60;
217         int hours = 0;
218         if (minutes > 59) {
219             hours = minutes / 60;
220             minutes = minutes - (hours * 60);
221         }
222         String connTime = hours + ":" + (minutes > 9 ? Integer.toString(minutes) : '0' + Integer.toString(minutes));
223 
224         int lastSecs = (int) SystemUtil.getDateDiff(lastActivity, new Date(), TimeUnit.SECONDS);
225 
226         return " (online: " + connTime + "; seen: " + lastSecs + " sec ago)";
227     }

339     private void reconnect() {
340         logger.trace(userName + " started reconnect");
341         for (Entry<UUID, Table> entry : tables.entrySet()) {
342             ccJoinedTable(entry.getValue().getRoomId(), entry.getValue().getId(), entry.getValue().isTournament());
343         }
344         for (Iterator<Entry<UUID, UUID>> iterator = userTournaments.entrySet().iterator(); iterator.hasNext(); ) {
345             Entry<UUID, UUID> next = iterator.next();
346             Optional<TournamentController> tournamentController = managerFactory.tournamentManager().getTournamentController(next.getValue());
347             if (tournamentController.isPresent()) {
348                 ccTournamentStarted(next.getValue(), next.getKey());
349                 tournamentController.get().rejoin(next.getKey());
350             } else {
351                 iterator.remove(); // tournament has ended meanwhile
352             }
353         }
354         for (Entry<UUID, GameSessionPlayer> entry : gameSessions.entrySet()) {
355             ccGameStarted(entry.getValue().getGameId(), entry.getKey());
356             entry.getValue().init();
357             managerFactory.gameManager().sendPlayerString(entry.getValue().getGameId(), userId, "");
358         }
359 
360         for (Entry<UUID, DraftSession> entry : draftSessions.entrySet()) {
361             ccDraftStarted(entry.getValue().getDraftId(), entry.getKey());
362             entry.getValue().init();
363             entry.getValue().update();
364         }
365 
366         for (Entry<UUID, TournamentSession> entry : constructing.entrySet()) {
367             entry.getValue().construct(0); // TODO: Check if this is correct
368         }
369         for (Entry<UUID, Deck> entry : sideboarding.entrySet()) {
370             Optional<TableController> controller = managerFactory.tableManager().getController(entry.getKey());
371             if (controller.isPresent()) {
372                 ccSideboard(entry.getValue(), entry.getKey(), controller.get().getRemainingTime(), controller.get().getOptions().isLimited());
373             } else {
374                 // Table is missing after connection was lost during sideboard.
375                 // Means other players were removed or conceded the game?
376                 logger.debug(getName() + " reconnects during sideboarding but tableId not found: " + entry.getKey());
377             }
378         }
379         ServerMessagesUtil.instance.incReconnects();
380         logger.trace(userName + " ended reconnect");
381     }

427     public void removeUserFromAllTables(DisconnectReason reason) {
428         logger.trace("REMOVE " + userName + " Draft sessions " + draftSessions.size());
429         for (DraftSession draftSession : draftSessions.values()) {
430             draftSession.setKilled();
431         }
432         draftSessions.clear();
433         logger.trace("REMOVE " + userName + " Tournament sessions " + userTournaments.size());
434         for (UUID tournamentId : userTournaments.values()) {
435             managerFactory.tournamentManager().quit(tournamentId, userId);
436         }
437         userTournaments.clear();
438         constructing.clear();
439         logger.trace("REMOVE " + userName + " Tables " + tables.size());
440         for (Entry<UUID, Table> entry : tables.entrySet()) {
441             logger.debug("-- leave tableId: " + entry.getValue().getId());
442             managerFactory.tableManager().leaveTable(userId, entry.getValue().getId());
443         }
444         tables.clear();
445         sideboarding.clear();
446         logger.trace("REMOVE " + userName + " Game sessions: " + gameSessions.size());
447         for (GameSessionPlayer gameSessionPlayer : gameSessions.values()) {
448             logger.debug("-- kill game session of gameId: " + gameSessionPlayer.getGameId());
449             managerFactory.gameManager().quitMatch(gameSessionPlayer.getGameId(), userId);
450             gameSessionPlayer.quitGame();
451         }
452         gameSessions.clear();
453         logger.trace("REMOVE " + userName + " watched Games " + watchedGames.size());
454         for (Iterator<UUID> it = watchedGames.iterator(); it.hasNext(); ) { // Iterator to prevent ConcurrentModificationException
455             UUID gameId = it.next();
456             managerFactory.gameManager().stopWatching(gameId, userId);
457         }
458         watchedGames.clear();
459         logger.trace("REMOVE " + userName + " Chats ");
460         managerFactory.chatManager().removeUser(userId, reason);
461     }

479     public String getGameInfo() {
480         StringBuilder sb = new StringBuilder();
481 
482         int draft = 0, match = 0, sideboard = 0, tournament = 0, construct = 0, waiting = 0;
483 
484         for (Map.Entry<UUID, Table> tableEntry : tables.entrySet()) {
485             if (tableEntry != null) {
486                 Table table = tableEntry.getValue();
487                 if (table != null) {
488                     if (table.isTournament()) {
489                         if (tableEntry.getKey() != null) {
490                             TournamentPlayer tournamentPlayer = table.getTournament().getPlayer(tableEntry.getKey());
491                             if (tournamentPlayer != null) {
492                                 if (!tournamentPlayer.isEliminated()) {
493                                     switch (table.getState()) {
494                                         case WAITING:
495                                         case STARTING:
496                                         case READY_TO_START:
497                                             waiting++;
498                                             break;
499                                         case CONSTRUCTING:
500                                             construct++;
501                                             break;
502                                         case DRAFTING:
503                                             draft++;
504                                             break;
505                                         case DUELING:
506                                             tournament++;
507                                             break;
508                                     }
509                                     switch (getUserState()) {
510                                         case Disconnected:
511                                             tournamentPlayer.setDisconnectInfo(" (discon. " + getDisconnectDuration() + ')');
512                                             break;
513                                         case Offline:
514                                             tournamentPlayer.setDisconnectInfo(" Offline");
515                                             break;
516                                         default:
517                                             tournamentPlayer.setDisconnectInfo("");
518                                     }
519                                 }
520                             } else {
521                                 // can happen if tournamet has just ended
522                                 logger.debug(userName + " tournament player missing - tableId:" + table.getId(), null);
523                                 tablesToDelete.add(tableEntry.getKey());
524                             }
525                         } else {
526                             logger.error(userName + " tournament key missing - tableId: " + table.getId(), null);
527                         }
528                     } else {
529                         switch (table.getState()) {
530                             case WAITING:
531                             case STARTING:
532                             case READY_TO_START:
533                                 waiting++;
534                                 break;
535                             case SIDEBOARDING:
536                                 sideboard++;
537                                 break;
538                             case DUELING:
539                                 match++;
540                                 break;
541                         }
542                     }
543                 }
544             }
545         }
546         if (!tablesToDelete.isEmpty()) {
547             for (UUID keyId : tablesToDelete) {
548                 removeTable(keyId);
549             }
550             tablesToDelete.clear();
551         }
552         if (waiting > 0) {
553             sb.append("Wait: ").append(waiting).append(' ');
554         }
555         if (match > 0) {
556             sb.append("Match: ").append(match).append(' ');
557         }
558         if (sideboard > 0) {
559             sb.append("Sideb: ").append(sideboard).append(' ');
560         }
561         if (draft > 0) {
562             sb.append("Draft: ").append(draft).append(' ');
563         }
564         if (construct > 0) {
565             sb.append("Const: ").append(construct).append(' ');
566         }
567         if (tournament > 0) {
568             sb.append("Tourn: ").append(tournament).append(' ');
569         }
570         if (!watchedGames.isEmpty()) {
571             sb.append("Watch: ").append(watchedGames.size()).append(' ');
572         }
573         return sb.toString();
574     }

603     public void resetUserStats() {
604         if (userData == null) {
605             return;
606         }
607         userStats = UserStatsRepository.instance.getUser(this.userName);
608         if (userStats != null) {
609             ResultProtos.UserStatsProto userStatsProto = userStats.getProto();
610 
611             userData.setMatchHistory(userStatsToMatchHistory(userStatsProto));
612             userData.setMatchQuitRatio(userStatsToMatchQuitRatio(userStatsProto));
613             userData.setTourneyHistory(userStatsToTourneyHistory(userStatsProto));
614             userData.setTourneyQuitRatio(userStatsToTourneyQuitRatio(userStatsProto));
615             userData.setGeneralRating(userStatsToGeneralRating(userStatsProto));
616             userData.setConstructedRating(userStatsToConstructedRating(userStatsProto));
617             userData.setLimitedRating(userStatsToLimitedRating(userStatsProto));
618         } else {
619             userData.setMatchHistory("0");
620             userData.setMatchQuitRatio(0);
621             userData.setTourneyHistory("0");
622             userData.setTourneyQuitRatio(0);
623             userData.setGeneralRating(GlickoRatingSystem.getDefaultDisplayedRating());
624             userData.setConstructedRating(GlickoRatingSystem.getDefaultDisplayedRating());
625             userData.setLimitedRating(GlickoRatingSystem.getDefaultDisplayedRating());
626         }
627     }

665     public static String userStatsToMatchHistory(ResultProtos.UserStatsProto proto) {
666         StringBuilder builder = new StringBuilder();
667         builder.append(proto.getMatches());
668         List<String> quit = new ArrayList<>();
669         if (proto.getMatchesIdleTimeout() > 0) {
670             quit.add("I:" + proto.getMatchesIdleTimeout());
671         }
672         if (proto.getMatchesTimerTimeout() > 0) {
673             quit.add("T:" + proto.getMatchesTimerTimeout());
674         }
675         if (proto.getMatchesQuit() > 0) {

771     private static void joinStrings(StringBuilder joined, List<String> strings, String separator) {
772         for (int i = 0; i < strings.size(); ++i) {
773             if (i > 0) {
774                 joined.append(separator);
775             }
776             joined.append(strings.get(i));
777         }
778     }

790     public int getNumberOfNotFinishedTables() {
791         int number = 0;
792         for (Table table : tables.values()) {
793             if (table.getState() == TableState.FINISHED) {
794                 number++;
795             } else {
796                 Optional<TableController> tableController = managerFactory.tableManager().getController(table.getId());
797                 if (!tableController.isPresent()) {
798                     logger.error("table not found : " + table.getId());
799                 } else if (tableController.get().isUserStillActive(userId)) {
800                     number++;
801                 }
802             }
803         }
804         return number;
805     }

116     public synchronized boolean joinTournament(UUID userId, String name, PlayerType playerType, int skill, DeckCardLists deckList, String password) throws GameException {
117         if (table.getState() != TableState.WAITING) {
118             return false;
119         }
120 
121         Seat seat = table.getNextAvailableSeat(playerType);
122         if (seat == null) {
123             throw new GameException("No available seats.");
124         }
125         Optional<User> _user = managerFactory.userManager().getUser(userId);
126         if (!_user.isPresent()) {
127             logger.fatal("couldn't get user " + name + " for join tournament userId = " + userId);
128             return false;
129         }
130         User user = _user.get();
131         // check password
132         if (!table.getTournament().getOptions().getPassword().isEmpty() && playerType == PlayerType.HUMAN) {
133             if (!table.getTournament().getOptions().getPassword().equals(password)) {
134                 user.showUserMessage("Join Table", "Wrong password.");
135                 return false;
136             }
137         }
138         if (userPlayerMap.containsKey(userId) && playerType == PlayerType.HUMAN) {
139             user.showUserMessage("Join Table", "You can join a table only one time.");
140             return false;
141         }
142         Deck deck = null;
143         if (!table.getTournament().getTournamentType().isLimited()) {
144             if (deckList != null) {
145                 deck = Deck.load(deckList, false, false);
146             } else {
147                 user.showUserMessage("Join Table", "No valid deck selected!");
148                 return false;
149             }
150             if (!Main.isTestMode() && !table.getValidator().validate(deck)) {
151                 StringBuilder sb = new StringBuilder("You (").append(name).append(") have an invalid deck for the selected ").append(table.getValidator().getName()).append(" Format. \n\n");
152                 List<DeckValidatorError> errorsList = table.getValidator().getErrorsListSorted();
153                 errorsList.stream().forEach(error -> {
154                     sb.append(error.getGroup()).append(": ").append(error.getMessage()).append("\n");
155                 });
156                 sb.append("\n\nSelect a deck that is appropriate for the selected format and try again!");
157                 user.showUserMessage("Join Table", sb.toString());
158                 if (isOwner(userId)) {
159                     logger.debug("New table removed because owner submitted invalid deck tableId " + table.getId());
160                     managerFactory.tableManager().removeTable(table.getId());
161                 }
162                 return false;
163             }
164         }
165         // Check quit ratio.
166         int quitRatio = table.getTournament().getOptions().getQuitRatio();
167         if (quitRatio < user.getTourneyQuitRatio()) {
168             String message = new StringBuilder("Your quit ratio ").append(user.getTourneyQuitRatio())
169                     .append("% is higher than the table requirement ").append(quitRatio).append('%').toString();
170             user.showUserMessage("Join Table", message);
171             return false;
172         }
173 
174         // Check minimum rating.
175         int minimumRating = table.getTournament().getOptions().getMinimumRating();
176         int userRating;
177         if (table.getTournament().getOptions().getMatchOptions().isLimited()) {
178             userRating = user.getUserData().getLimitedRating();
179         } else {
180             userRating = user.getUserData().getConstructedRating();
181         }
182         if (userRating < minimumRating) {
183             String message = new StringBuilder("Your rating ").append(userRating)
184                     .append(" is lower than the table requirement ").append(minimumRating).toString();
185             user.showUserMessage("Join Table", message);
186             return false;
187         }
188 
189         Optional<Player> playerOptional = createPlayer(name, seat.getPlayerType(), skill);
190         if (playerOptional.isPresent()) {
191             Player player = playerOptional.get();
192             if (!player.canJoinTable(table)) {
193                 user.showUserMessage("Join Table", new StringBuilder("A ").append(seat.getPlayerType()).append(" player can't join this table.").toString());
194                 return false;
195             }
196             tournament.addPlayer(player, seat.getPlayerType());
197             TournamentPlayer tournamentPlayer = tournament.getPlayer(player.getId());
198             if (deck != null && tournamentPlayer != null) {
199                 tournamentPlayer.submitDeck(deck);
200             }
201             table.joinTable(player, seat);
202             logger.debug("Player " + player.getName() + " id: " + player.getId() + " joined tableId: " + table.getId());
203             //only inform human players and add them to sessionPlayerMap
204             if (seat.getPlayer().isHuman()) {
205                 seat.getPlayer().setUserData(user.getUserData());
206                 user.addTable(player.getId(), table);
207                 user.ccJoinedTable(table.getRoomId(), table.getId(), true);
208                 userPlayerMap.put(userId, player.getId());
209             }
210 
211             return true;
212         } else {
213             throw new GameException("Playertype " + seat.getPlayerType() + " could not be created.");
214         }
215     }

221     public synchronized boolean replaceDraftPlayer(Player oldPlayer, String name, PlayerType playerType, int skill) {
222         Optional<Player> newPlayerOpt = createPlayer(name, playerType, skill);

224             return false;
225         }

227         TournamentPlayer oldTournamentPlayer = tournament.getPlayer(oldPlayer.getId());

229         tournament.addPlayer(newPlayer, playerType);

231         TournamentPlayer newTournamentPlayer = tournament.getPlayer(newPlayer.getId());

233         newTournamentPlayer.setReplacedTournamentPlayer(oldTournamentPlayer);
234 
235         managerFactory.draftManager().getController(table.getId()).ifPresent(controller -> controller.replacePlayer(oldPlayer, newPlayer));
236         return true;
237     }

239     public synchronized boolean joinTable(UUID userId, String name, PlayerType playerType, int skill, DeckCardLists deckList, String password) throws MageException {
240         Optional<User> _user = managerFactory.userManager().getUser(userId);
241         if (!_user.isPresent()) {
242             logger.error("Join Table: can't find user to join " + name + " Id = " + userId);
243             return false;
244         }
245         User user = _user.get();
246         if (userPlayerMap.containsKey(userId) && playerType == PlayerType.HUMAN) {
247             user.showUserMessage("Join Table", new StringBuilder("You can join a table only one time.").toString());
248             return false;
249         }
250         if (table.getState() != TableState.WAITING) {
251             user.showUserMessage("Join Table", "No available seats.");
252             return false;
253         }
254         // check password
255         if (!table.getMatch().getOptions().getPassword().isEmpty() && playerType == PlayerType.HUMAN) {
256             if (!table.getMatch().getOptions().getPassword().equals(password)) {
257                 user.showUserMessage("Join Table", "Wrong password.");
258                 return false;
259             }
260         }
261         Seat seat = table.getNextAvailableSeat(playerType);
262         if (seat == null) {
263             user.showUserMessage("Join Table", "No available seats.");
264             return false;
265         }
266         Deck deck = Deck.load(deckList, false, false);
267 
268         if (!Main.isTestMode() && !table.getValidator().validate(deck)) {
269             StringBuilder sb = new StringBuilder("You (").append(name).append(") have an invalid deck for the selected ").append(table.getValidator().getName()).append(" Format. \n\n");
270             List<DeckValidatorError> errorsList = table.getValidator().getErrorsListSorted();
271             errorsList.stream().forEach(error -> {
272                 sb.append(error.getGroup()).append(": ").append(error.getMessage()).append("\n");
273             });
274             sb.append("\n\nSelect a deck that is appropriate for the selected format and try again!");
275             user.showUserMessage("Join Table", sb.toString());
276             if (isOwner(userId)) {
277                 logger.debug("New table removed because owner submitted invalid deck tableId " + table.getId());
278                 managerFactory.tableManager().removeTable(table.getId());
279             }
280             return false;
281         }
282         // Check quit ratio.
283         int quitRatio = table.getMatch().getOptions().getQuitRatio();
284         if (quitRatio < user.getMatchQuitRatio()) {
285             String message = new StringBuilder("Your quit ratio ").append(user.getMatchQuitRatio())
286                     .append("% is higher than the table requirement ").append(quitRatio).append('%').toString();
287             user.showUserMessage("Join Table", message);
288             return false;
289         }
290 
291         // Check minimum rating.
292         int minimumRating = table.getMatch().getOptions().getMinimumRating();
293         int userRating;
294         if (table.getMatch().getOptions().isLimited()) {
295             userRating = user.getUserData().getLimitedRating();
296         } else {
297             userRating = user.getUserData().getConstructedRating();
298         }
299         if (userRating < minimumRating) {
300             String message = new StringBuilder("Your rating ").append(userRating)
301                     .append(" is lower than the table requirement ").append(minimumRating).toString();
302             user.showUserMessage("Join Table", message);
303             return false;
304         }
305 
306         // Check power level for table (currently only used for EDH/Commander table)
307         int edhPowerLevel = table.getMatch().getOptions().getEdhPowerLevel();
308         if (edhPowerLevel > 0 && table.getValidator().getName().toLowerCase(Locale.ENGLISH).equals("commander")) {
309             int deckEdhPowerLevel = table.getValidator().getEdhPowerLevel(deck);
310             if (deckEdhPowerLevel % 100 > edhPowerLevel) {
311                 String message = new StringBuilder("Your deck appears to be too powerful for this table.\n\nReduce the number of extra turn cards, infect, counters, fogs, reconsider your commander. ")
312                         .append("\nThe table requirement has a maximum power level of ").append(edhPowerLevel).append(" whilst your deck has a calculated power level of ")
313                         .append(deckEdhPowerLevel % 100).toString();
314                 user.showUserMessage("Join Table", message);
315                 return false;
316             }
317 
318             boolean restrictedColor = false;
319             String badColor = "";
320             int colorVal = edhPowerLevel % 10;
321             if (colorVal == 6 && deckEdhPowerLevel >= 10000000) {
322                 restrictedColor = true;
323                 badColor = "white";
324             }
325             if (colorVal == 4 && deckEdhPowerLevel % 10000000 >= 1000000) {
326                 restrictedColor = true;
327                 badColor = "blue";
328             }
329             if (colorVal == 3 && deckEdhPowerLevel % 1000000 >= 100000) {
330                 restrictedColor = true;
331                 badColor = "black";
332             }
333             if (colorVal == 2 && deckEdhPowerLevel % 100000 >= 10000) {
334                 restrictedColor = true;
335                 badColor = "red";
336             }
337             if (colorVal == 1 && deckEdhPowerLevel % 10000 >= 1000) {
338                 restrictedColor = true;
339                 badColor = "green";
340             }
341             if (restrictedColor) {
342                 String message = new StringBuilder("Your deck contains ")
343                         .append(badColor)
344                         .append(".  The creator of the table has requested no ")
345                         .append(badColor)
346                         .append(" cards to be on the table!").toString();
347                 user.showUserMessage("Join Table", message);
348                 return false;
349             }
350         }
351 
352         Optional<Player> playerOpt = createPlayer(name, seat.getPlayerType(), skill);
353         if (!playerOpt.isPresent()) {
354             String message = "Could not create player " + name + " of type " + seat.getPlayerType();
355             logger.warn("User: " + user.getName() + " => " + message);
356             user.showUserMessage("Join Table", message);
357             return false;
358         }
359         Player player = playerOpt.get();
360         if (!player.canJoinTable(table)) {
361             user.showUserMessage("Join Table", "A " + seat.getPlayerType() + " player can't join this table.");
362             return false;
363         }
364         match.addPlayer(player, deck);
365         table.joinTable(player, seat);
366         logger.trace(player.getName() + " joined tableId: " + table.getId());
367         //only inform human players and add them to sessionPlayerMap
368         if (seat.getPlayer().isHuman()) {
369             seat.getPlayer().setUserData(user.getUserData());
370             if (!table.isTournamentSubTable()) {
371                 user.addTable(player.getId(), table);
372             }
373             user.ccJoinedTable(table.getRoomId(), table.getId(), false);
374             userPlayerMap.put(userId, player.getId());
375         }
376         return true;
377     }

394     public synchronized boolean submitDeck(UUID userId, DeckCardLists deckList) throws MageException {
395         UUID playerId = userPlayerMap.get(userId);
396         if (table.isTournament()) {
397             TournamentPlayer player = tournament.getPlayer(playerId);
398             if (player == null || player.hasQuit()) {
399                 return true; // so the construct panel closes after submit
400             }
401         } else if (table.getMatch() != null) {
402             MatchPlayer mPlayer = table.getMatch().getPlayer(playerId);
403             if (mPlayer == null || mPlayer.hasQuit()) {
404                 return true; // so the construct panel closes after submit
405             }
406             if (table.isTournamentSubTable()) {
407                 TournamentPlayer tournamentPlayer = table.getTournament().getPlayer(mPlayer.getPlayer().getId());
408                 if (tournamentPlayer != null) {
409                     tournamentPlayer.setStateInfo(""); // reset sideboarding state
410                 }
411             }
412         }
413         if (table.getState() != TableState.SIDEBOARDING && table.getState() != TableState.CONSTRUCTING) {
414             return false;
415         }
416         Deck deck = Deck.load(deckList, false, false);
417         if (table.getState() == TableState.SIDEBOARDING && table.getMatch() != null) {
418             MatchPlayer mPlayer = table.getMatch().getPlayer(playerId);
419             if (mPlayer != null) {
420                 deck.setName(mPlayer.getDeck().getName());
421             }
422         }
423         if (!Main.isTestMode() && !table.getValidator().validate(deck)) {
424             Optional<User> _user = managerFactory.userManager().getUser(userId);
425             if (!_user.isPresent()) {
426                 return false;
427             }
428             StringBuilder sb = new StringBuilder("Invalid deck for the selected ").append(table.getValidator().getName()).append(" format. \n\n");
429             List<DeckValidatorError> errorsList = table.getValidator().getErrorsListSorted();
430             errorsList.stream().forEach(error -> {
431                 sb.append(error.getGroup()).append(": ").append(error.getMessage()).append("\n");
432             });
433             sb.append("\n\nAdd enough cards and try again!");
434             _user.get().showUserMessage("Submit deck", sb.toString());
435             return false;
436         }
437         submitDeck(userId, playerId, deck);
438         return true;
439     }

441     public void updateDeck(UUID userId, DeckCardLists deckList) throws MageException {
442         boolean validDeck;
443         UUID playerId = userPlayerMap.get(userId);
444         if (table.getState() != TableState.SIDEBOARDING && table.getState() != TableState.CONSTRUCTING) {
445             return;
446         }
447         Deck deck = Deck.load(deckList, false, false);
448         validDeck = updateDeck(userId, playerId, deck);
449         if (!validDeck && getTableState() == TableState.SIDEBOARDING) {
450             logger.warn(" userId: " + userId + " - Modified deck card list!");
451         }
452     }

456             match.submitDeck(playerId, deck);
457             managerFactory.userManager().getUser(userId).ifPresent(user -> user.removeSideboarding(table.getId()));
458         } else {
459             managerFactory.tournamentManager().submitDeck(tournament.getId(), playerId, deck);
460             managerFactory.userManager().getUser(userId).ifPresent(user -> user.removeConstructing(playerId));
461         }
462     }

464     private boolean updateDeck(UUID userId, UUID playerId, Deck deck) {
465         boolean validDeck = true;
466         if (table.isTournament()) {
467             if (tournament != null) {
468                 validDeck = managerFactory.tournamentManager().updateDeck(tournament.getId(), playerId, deck);
469             } else {
470                 logger.fatal("Tournament == null  table: " + table.getId() + " userId: " + userId);
471             }
472         } else if (TableState.SIDEBOARDING == table.getState()) {
473             validDeck = match.updateDeck(playerId, deck);
474         } else {
475             // deck was meanwhile submitted so the autoupdate can be ignored
476         }
477         return validDeck;
478     }

492             if (userPlayerMap.get(userId) != null) {
493                 return false;

496             if (!_user.isPresent()) {
497                 return false;
498             }

524     public synchronized void leaveTable(UUID userId) {
525         if (table == null) {
526             logger.error("No table object - userId: " + userId);
527             return;
528         }
529         if (table.isTournament() && tournament == null) {
530             logger.error("No tournament object - userId: " + userId + "  table: " + table.getId());
531             return;
532         }
533         if (this.userId != null && this.userId.equals(userId) // tourn. sub tables have no creator user
534                 && (table.getState() == TableState.WAITING
535                 || table.getState() == TableState.READY_TO_START)) {
536             // table not started yet and user is the owner, removeUserFromAllTablesAndChat the table
537             managerFactory.tableManager().removeTable(table.getId());
538         } else {
539             UUID playerId = userPlayerMap.get(userId);
540             if (playerId != null) {
541                 if (table.getState() == TableState.WAITING || table.getState() == TableState.READY_TO_START) {
542                     table.leaveNotStartedTable(playerId);
543                     if (table.isTournament()) {
544                         tournament.removePlayer(playerId);
545                     } else {
546                         match.quitMatch(playerId);
547                     }
548                     Optional<User> user = managerFactory.userManager().getUser(userId);
549                     if (user.isPresent()) {
550                         managerFactory.chatManager().broadcast(chatId, user.get().getName(), "has left the table", ChatMessage.MessageColor.BLUE, true, null, ChatMessage.MessageType.STATUS, ChatMessage.SoundToPlay.PlayerLeft);
551                         if (!table.isTournamentSubTable()) {
552                             user.get().removeTable(playerId);
553                         }
554                     } else {
555                         logger.debug("User not found - userId: " + userId + " tableId:" + table.getId());
556                     }
557                     userPlayerMap.remove(userId);
558                 } else if (table.getState() != TableState.FINISHED) {
559                     if (table.isTournament()) {
560                         logger.debug("Quit tournament sub tables for userId: " + userId);
561                         managerFactory.tableManager().userQuitTournamentSubTables(tournament.getId(), userId);
562                         logger.debug("Quit tournament  Id: " + table.getTournament().getId() + '(' + table.getTournament().getTournamentState() + ')');
563                         managerFactory.tournamentManager().quit(tournament.getId(), userId);
564                     } else {
565                         MatchPlayer matchPlayer = match.getPlayer(playerId);
566                         if (matchPlayer != null && !match.hasEnded() && !matchPlayer.hasQuit()) {
567                             Game game = match.getGame();
568                             if (game != null && !game.hasEnded()) {
569                                 Player player = match.getPlayer(playerId).getPlayer();
570                                 if (player != null && player.isInGame()) {
571                                     managerFactory.gameManager().quitMatch(game.getId(), userId);
572                                 }
573                                 match.quitMatch(playerId);
574                             } else {
575                                 if (table.getState() == TableState.SIDEBOARDING) {
576                                     if (!matchPlayer.isDoneSideboarding()) {
577                                         // submit deck to finish sideboarding and trigger match start / end
578                                         matchPlayer.submitDeck(matchPlayer.getDeck());
579                                     }
580                                 }
581                                 match.quitMatch(playerId);
582                             }
583                         }
584                     }
585                 }
586             } else {
587                 logger.error("No playerId found for userId: " + userId);
588             }
589         }

623     private void startGame(UUID choosingPlayerId) throws GameException {
624         try {
625             match.startGame();
626             table.initGame();
627             GameOptions gameOptions = new GameOptions();
628             gameOptions.rollbackTurnsAllowed = match.getOptions().isRollbackTurnsAllowed();
629             gameOptions.bannedUsers = match.getOptions().getBannedUsers();
630             gameOptions.planeChase = match.getOptions().isPlaneChase();
631             match.getGame().setGameOptions(gameOptions);
632             managerFactory.gameManager().createGameSession(match.getGame(), userPlayerMap, table.getId(), choosingPlayerId, gameOptions);
633             String creator = null;
634             StringBuilder opponent = new StringBuilder();
635             for (Entry<UUID, UUID> entry : userPlayerMap.entrySet()) { // do only for no AI players
636                 if (match.getPlayer(entry.getValue()) != null && !match.getPlayer(entry.getValue()).hasQuit()) {
637                     Optional<User> _user = managerFactory.userManager().getUser(entry.getKey());
638                     if (_user.isPresent()) {
639                         User user = _user.get();
640                         user.ccGameStarted(match.getGame().getId(), entry.getValue());
641 
642                         if (creator == null) {
643                             creator = user.getName();
644                         } else {
645                             if (opponent.length() > 0) {
646                                 opponent.append(" - ");
647                             }
648                             opponent.append(user.getName());
649                         }
650                     } else {
651                         logger.error("Unable to find user: " + entry.getKey() + "  playerId: " + entry.getValue());
652                         MatchPlayer matchPlayer = match.getPlayer(entry.getValue());
653                         if (matchPlayer != null && !matchPlayer.hasQuit()) {
654                             matchPlayer.setQuit(true);
655                         }
656                     }
657                 }
658             }
659             // Append AI opponents to the log file
660             for (MatchPlayer mPlayer : match.getPlayers()) {
661                 if (!mPlayer.getPlayer().isHuman()) {
662                     if (opponent.length() > 0) {
663                         opponent.append(" - ");
664                     }
665                     opponent.append(mPlayer.getName());
666                 }
667             }
668             ServerMessagesUtil.instance.incGamesStarted();
669 
670             // log about game started
671             logger.info("GAME started " + (match.getGame() != null ? match.getGame().getId() : "no Game") + " [" + match.getName() + "] " + creator + " - " + opponent);
672             logger.debug("- matchId: " + match.getId() + " [" + match.getName() + ']');
673             if (match.getGame() != null) {
674                 logger.debug("- chatId:  " + managerFactory.gameManager().getChatId(match.getGame().getId()));
675             }
676         } catch (Exception ex) {
677             logger.fatal("Error starting game table: " + table.getId(), ex);
678             if (table != null) {
679                 managerFactory.tableManager().removeTable(table.getId());
680             }
681             if (match != null) {
682                 Game game = match.getGame();
683                 if (game != null) {
684                     managerFactory.gameManager().removeGame(game.getId());
685                     // game ended by error, so don't add it to ended stats
686                 }
687             }
688         }
689     }

758     /**
759      * Ends the current game and starts if neccessary the next game
760      *
761      * @return true if table can be closed
762      */
763     public boolean endGameAndStartNextGame() {
764         // get player that chooses who goes first
765         Game game = match.getGame();
766         if (game == null) {
767             return true;
768         }
769         UUID choosingPlayerId = match.getChooser();
770         match.endGame();
771         if (managerFactory.configSettings().isSaveGameActivated() && !game.isSimulation()) {
772             if (managerFactory.gameManager().saveGame(game.getId())) {
773                 match.setReplayAvailable(true);
774             }
775         }
776         managerFactory.gameManager().removeGame(game.getId());
777         ServerMessagesUtil.instance.incGamesEnded();
778 
779         try {
780             if (!match.hasEnded()) {
781                 if (match.getGame() != null && match.getGame().getGameType().isSideboardingAllowed()) {
782                     sideboard();
783                 }
784                 if (!match.hasEnded()) {
785                     startGame(choosingPlayerId);
786                 } else {
787                     closeTable();
788                 }
789             } else {
790                 closeTable();
791             }
792         } catch (GameException ex) {
793             logger.fatal(null, ex);
794         }
795         return match.hasEnded();
796     }

798     private void sideboard() {
799         table.sideboard();
800         setupTimeout(Match.SIDEBOARD_TIME);
801         if (table.isTournamentSubTable()) {
802             for (MatchPlayer matchPlayer : match.getPlayers()) {
803                 if (!matchPlayer.hasQuit()) {
804                     TournamentPlayer tournamentPlayer = table.getTournament().getPlayer(matchPlayer.getPlayer().getId());
805                     if (tournamentPlayer != null) {
806                         tournamentPlayer.setStateInfo("sideboarding");
807                     }
808                 }
809             }
810         }
811         match.sideboard();
812         cancelTimeout();
813         if (table.isTournamentSubTable()) {
814             for (MatchPlayer matchPlayer : match.getPlayers()) {
815                 TournamentPlayer tournamentPlayer = table.getTournament().getPlayer(matchPlayer.getPlayer().getId());
816                 if (tournamentPlayer != null && tournamentPlayer.getStateInfo().equals("sideboarding")) {
817                     tournamentPlayer.setStateInfo("");
818                 }
819             }
820         }
821     }

832     private void matchEnd() {
833         if (match != null) {
834             for (Entry<UUID, UUID> entry : userPlayerMap.entrySet()) {
835                 MatchPlayer matchPlayer = match.getPlayer(entry.getValue());
836                 // opponent(s) left during sideboarding
837                 if (matchPlayer != null) {
838                     if (!matchPlayer.hasQuit()) {
839                         managerFactory.userManager().getUser(entry.getKey()).ifPresent(user -> {
840                             if (table.getState() == TableState.SIDEBOARDING) {
841                                 StringBuilder sb = new StringBuilder();
842                                 if (table.isTournamentSubTable()) {
843                                     sb.append("Your tournament match of round ");
844                                     sb.append(table.getTournament().getRounds().size());
845                                     sb.append(" is over. ");
846                                 } else {
847                                     sb.append("Match [").append(match.getName()).append("] is over. ");
848                                 }
849                                 if (match.getPlayers().size() > 2) {
850                                     sb.append("All your opponents have lost or quit the match.");
851                                 } else {
852                                     sb.append("Your opponent has quit the match.");
853                                 }
854                                 user.showUserMessage("Match info", sb.toString());
855                             }
856                             // removeUserFromAllTablesAndChat table from user - table manager holds table for display of finished matches
857                             if (!table.isTournamentSubTable()) {
858                                 user.removeTable(entry.getValue());
859                             }
860                         });
861                     }
862                 }
863                 // free resources no longer needed
864                 match.cleanUpOnMatchEnd(managerFactory.configSettings().isSaveGameActivated(), table.isTournament());
865             }
866         }
867     }

973     public boolean isMatchTableStillValid() {
974         // removes active match only, not tourney
975         if (table.isTournament()) {
976             return true;
977         }
978 
979         // only started games need to check
980         if (Arrays.asList(
981                 TableState.WAITING,
982                 TableState.READY_TO_START,
983                 TableState.STARTING
984         ).contains(table.getState())) {
985             // waiting in start dialog
986             return true;
987         }
988         if (match != null && !match.isDoneSideboarding()) {
989             // waiting sideboard complete
990             return true;
991         }
992 
993         // no games in started match (error in match init code?)
994         if (match.getGame() == null) {
995             logger.error("- Match without started games:");
996             logger.error("-- matchId:" + match.getId());
997             return false; // critical error
998         }
999 
1000         // find player stats
1001         int validHumanPlayers = 0;
1002         int validAIPlayers = 0;
1003 
1004         // check humans
1005         for (Map.Entry<UUID, UUID> userPlayerEntry : userPlayerMap.entrySet()) {
1006             MatchPlayer matchPlayer = match.getPlayer(userPlayerEntry.getValue());
1007 
1008             // de-synced users and players listst?
1009             if (matchPlayer == null) {
1010                 logger.error("- Match player not found in started game:");
1011                 logger.error("-- matchId:" + match.getId());
1012                 logger.error("-- userId:" + userPlayerEntry.getKey());
1013                 logger.error("-- playerId:" + userPlayerEntry.getValue());
1014                 continue;
1015             }
1016 
1017             if (matchPlayer.getPlayer().isHuman()) {
1018                 if (matchPlayer.getPlayer().isInGame()) {
1019                     Optional<User> user = managerFactory.userManager().getUser(userPlayerEntry.getKey());
1020 
1021                     // user was logout or disconnected from server, but still in the game somehow
1022                     if (!user.isPresent() || !user.get().isActive()) {
1023                         logger.error("- Active user of match is missing: " + matchPlayer.getName());
1024                         logger.error("-- matchId:" + match.getId());
1025                         logger.error("-- userId:" + userPlayerEntry.getKey());
1026                         logger.error("-- playerId:" + userPlayerEntry.getValue());
1027                         return false; // critical error
1028                     }
1029 
1030                     // user exits on the server and match player has not quit -> player is valid
1031                     validHumanPlayers++;
1032                 }
1033             }
1034         }
1035 
1036         // check AI
1037         for (MatchPlayer matchPlayer : match.getPlayers()) {
1038             if (!matchPlayer.getPlayer().isHuman()) {
1039                 if (matchPlayer.getPlayer().isInGame()) {
1040                     validAIPlayers++;
1041                 }
1042             }
1043         }
1044 
1045         // if someone can play 1 vs 1 (e.g. 2+ players) then keep table
1046         return validAIPlayers + validHumanPlayers >= 2;
1047     }

29      private final ManagerFactory managerFactory;
30      private final ReadWriteLock lock = new ReentrantReadWriteLock();
31  
32      private final ConcurrentMap<UUID, String> clients = new ConcurrentHashMap<>();
33      private final UUID chatId;

37      public ChatSession(ManagerFactory managerFactory, String info) {
38          this.managerFactory = managerFactory;
39          chatId = UUID.randomUUID();
40          this.createTime = new Date();
41          this.info = info;
42      }

44      public void join(UUID userId) {
45          managerFactory.userManager().getUser(userId).ifPresent(user -> {
46              if (!clients.containsKey(userId)) {
47                  String userName = user.getName();
48                  final Lock w = lock.writeLock();
49                  w.lock();
50                  try {
51                      clients.put(userId, userName);
52                  } finally {
53                      w.unlock();
54                  }
55                  broadcast(null, userName + " has joined (" + user.getClientVersion() + ')', MessageColor.BLUE, true, null, MessageType.STATUS, null);
56                  logger.trace(userName + " joined chat " + chatId);
57              }
58          });
59      }
60  
61      public void kill(UUID userId, DisconnectReason reason) {
62  
63          try {
64              if (reason == null) {
65                  logger.fatal("User kill without disconnect reason  userId: " + userId);
66                  reason = DisconnectReason.Undefined;
67              }
68              if (userId != null && clients.containsKey(userId)) {
69                  String userName = clients.get(userId);
70                  if (reason != DisconnectReason.LostConnection) { // for lost connection the user will be reconnected or session expire so no removeUserFromAllTablesAndChat of chat yet
71                      final Lock w = lock.writeLock();
72                      w.lock();
73                      try {
74                          clients.remove(userId);
75                      } finally {
76                          w.unlock();
77                      }
78                      logger.debug(userName + '(' + reason.toString() + ')' + " removed from chatId " + chatId);
79                  }
80                  String message = reason.getMessage();
81  
82                  if (!message.isEmpty()) {
83                      broadcast(null, userName + message, MessageColor.BLUE, true, null, MessageType.STATUS, null);
84                  }
85              }
86          } catch (Exception ex) {
87              logger.fatal("exception: " + ex.toString());
88          }
89      }

91      public boolean broadcastInfoToUser(User toUser, String message) {
92          if (clients.containsKey(toUser.getId())) {
93              toUser.fireCallback(new ClientCallback(ClientCallbackMethod.CHATMESSAGE, chatId,
94                      new ChatMessage(null, message, new Date(), null, MessageColor.BLUE, MessageType.USER_INFO, null)));
95              return true;
96          }
97          return false;
98      }

100     public boolean broadcastWhisperToUser(User fromUser, User toUser, String message) {
101         if (clients.containsKey(toUser.getId())) {
102             toUser.fireCallback(new ClientCallback(ClientCallbackMethod.CHATMESSAGE, chatId,
103                     new ChatMessage(fromUser.getName(), message, new Date(), null, MessageColor.YELLOW, MessageType.WHISPER_FROM, SoundToPlay.PlayerWhispered)));
104             if (clients.containsKey(fromUser.getId())) {
105                 fromUser.fireCallback(new ClientCallback(ClientCallbackMethod.CHATMESSAGE, chatId,
106                         new ChatMessage(toUser.getName(), message, new Date(), null, MessageColor.YELLOW, MessageType.WHISPER_TO, null)));
107                 return true;
108             }
109         }
110         return false;
111     }

113     public void broadcast(String userName, String message, MessageColor color, boolean withTime, Game game, MessageType messageType, SoundToPlay soundToPlay) {
114         if (!message.isEmpty()) {
115             Set<UUID> clientsToRemove = new HashSet<>();
116             ClientCallback clientCallback = new ClientCallback(ClientCallbackMethod.CHATMESSAGE, chatId,
117                     new ChatMessage(userName, message, (withTime ? new Date() : null), game, color, messageType, soundToPlay));
118             List<UUID> chatUserIds = new ArrayList<>();
119             final Lock r = lock.readLock();
120             r.lock();
121             try {
122                 chatUserIds.addAll(clients.keySet());
123             } finally {
124                 r.unlock();
125             }
126             for (UUID userId : chatUserIds) {
127                 Optional<User> user = managerFactory.userManager().getUser(userId);
128                 if (user.isPresent()) {
129                     user.get().fireCallback(clientCallback);
130                 } else {
131                     clientsToRemove.add(userId);
132                 }
133             }
134             if (!clientsToRemove.isEmpty()) {
135                 final Lock w = lock.readLock();
136                 w.lock();
137                 try {
138                     clients.keySet().removeAll(clientsToRemove);
139                 } finally {
140                     w.unlock();
141                 }
142             }
143 
144         }
145     }

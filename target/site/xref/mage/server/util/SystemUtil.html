139     private static String getCardsListForSpecialInform(Game game, List<UUID> cardsList, List<String> commandParams) {
140         // cards list with ext info
141 
142         List<String> res = new ArrayList<>();
143 
144         for (UUID cardID : cardsList) {
145             Card card = game.getCard(cardID);
146             if (card == null) {
147                 continue;
148             }
149 
150             // basic info (card + set)
151             String cardInfo = card.getName() + " - " + card.getExpansionSetCode();
152 
153             // optional info
154             List<String> resInfo = new ArrayList<>();
155             for (String param : commandParams) {
156                 switch (param) {
157                     case PARAM_COLOR_COST:
158                         resInfo.add(card.getColor(game).toString());
159                         break;
160                     case PARAM_COLOR_COMMANDER:
161                         resInfo.add(card.getColorIdentity().toString());
162                         break;
163                     case PARAM_PT:
164                         resInfo.add(card.getPower() + " / " + card.getToughness());
165                         break;
166                     case PARAM_ABILITIES_COUNT:
167                         resInfo.add(String.valueOf(card.getAbilities(game).size()));
168                         break;
169                     case PARAM_ABILITIES_LIST:
170                         resInfo.add(card.getAbilities(game).stream()
171                                 .map(Ability::getClass)
172                                 .map(Class::getSimpleName)
173                                 .collect(Collectors.joining(", "))
174                         );
175                         break;
176                     default:
177                         logger.warn("Unknown param for cards list: " + param);
178                 }
179             }
180 
181             if (!resInfo.isEmpty()) {
182                 cardInfo += ": " + resInfo.stream().collect(Collectors.joining("; "));
183             }
184 
185             res.add(cardInfo);
186         }
187 
188         return res.stream().sorted().collect(Collectors.joining("\n"));
189     }

204     public static CardCommandData parseCardCommand(String commandLine) {
205         CardCommandData com = new CardCommandData();
206         com.source = commandLine.trim();
207         com.OK = false;
208         com.Error = "unknown error";
209 
210         Matcher matchCommand = patternCommand.matcher(com.source);
211         if (!matchCommand.matches()) {
212             com.Error = "Unknown command format";
213             return com;
214         }
215 
216         com.zone = matchCommand.group(1);
217         com.player = matchCommand.group(2);
218         try {
219             com.Amount = Integer.parseInt(matchCommand.group(4));
220         } catch (Throwable e) {
221             com.Error = "Can't parse amount value [" + matchCommand.group(4) + "]";
222             return com;
223         }
224 
225         // card name can be with set
226         // example: XLN-Island
227         List<String> cardInfo = parseSetAndCardNameCommand(matchCommand.group(3));
228         com.cardSet = cardInfo.get(0);
229         com.cardName = cardInfo.get(1);
230 
231         if (com.cardName.isEmpty()) {
232             com.Error = "Card name is empty";
233             return com;
234         }
235 
236         if (com.Amount <= 0) {
237             com.Error = "Amount [" + com.Amount + "] must be greater than 0";
238             return com;
239         }
240 
241         // all ok
242         com.Error = "";
243         com.OK = true;
244         return com;
245     }

247     /**
248      * Replaces cards in player's hands by specified in config/init.txt.<br/>
249      * <br/>
250      * <b>Implementation note:</b><br/>
251      * 1. Read init.txt line by line<br/>
252      * 2. Parse line using for searching groups like: [group 1] 3. Parse line
253      * using the following format: line ::=
254      * <zone>:<nickname>:<card name>:<amount><br/>
255      * 4. If zone equals to 'hand', add card to player's library<br/>
256      * 5a. Then swap added card with any card in player's hand<br/>
257      * 5b. Parse next line (go to 2.), If EOF go to 4.<br/>
258      * 6. Log message to all players that cards were added (to prevent unfair
259      * play).<br/>
260      * 7. Exit<br/>
261      *
262      * @param game
263      */
264     public static void addCardsForTesting(Game game, String fileSource, Player feedbackPlayer) {
265 
266         // fake test ability for triggers and events
267         Ability fakeSourceAbilityTemplate = new SimpleStaticAbility(Zone.OUTSIDE, new InfoEffect("adding testing cards"));
268         fakeSourceAbilityTemplate.setControllerId(feedbackPlayer.getId());
269 
270         try {
271             String fileName = fileSource;
272             if (fileName == null) {
273                 fileName = INIT_FILE_PATH;
274             }
275 
276             File f = new File(fileName);
277             if (!f.exists()) {
278                 logger.warn("Couldn't find init file: " + fileName);
279                 return;
280             }
281 
282             logger.info("Parsing init file... ");
283 
284             // steps:
285             // 1. parse groups and commands
286             // 2. ask user if many groups
287             // 3. process system commands
288             // 4. run commands from selected group
289             // 1. parse
290             List<CommandGroup> groups = new ArrayList<>();
291 
292             try (Scanner scanner = new Scanner(f)) {
293 
294                 CommandGroup currentGroup = null;
295 
296                 while (scanner.hasNextLine()) {
297                     String line = scanner.nextLine().trim();
298 
299                     // skip comments
300                     if (line.isEmpty() || line.startsWith("#") || line.startsWith("//")) {
301                         continue;
302                     }
303 
304                     // group
305                     Matcher matchGroup = patternGroup.matcher(line);
306                     if (matchGroup.matches()) {
307                         String groupName = matchGroup.group(1);
308                         if (groupName.startsWith("@")) {
309                             // special command group
310                             if (supportedCommands.containsKey(groupName)) {
311                                 currentGroup = new CommandGroup(groupName, true);
312                                 groups.add(currentGroup);
313                             } else {
314                                 logger.warn("Special group [" + groupName + "] is not supported.");
315                             }
316                             continue;
317                         } else {
318                             // basic group
319                             currentGroup = new CommandGroup(groupName);
320                             groups.add(currentGroup);
321                             continue;
322                         }
323                     }
324 
325                     // command
326                     if (currentGroup == null) {
327                         currentGroup = new CommandGroup("default group");
328                         groups.add(currentGroup);
329                     }
330                     currentGroup.commands.add(line);
331                 }
332             }
333 
334             // 2. ask user
335             CommandGroup runGroup = null;
336             if (groups.size() == 1) {
337                 // not need to ask
338                 runGroup = groups.get(0);
339             } else if (groups.size() > 1) {
340                 // need to ask
341                 logger.info("Found " + groups.size() + " groups. Need to select.");
342 
343                 // choice dialog
344                 Map<String, String> list = new LinkedHashMap<>();
345                 Map<String, Integer> sort = new LinkedHashMap<>();
346                 for (int i = 0; i < groups.size(); i++) {
347                     list.put(Integer.toString(i + 1), groups.get(i).getPrintNameWithStats());
348                     sort.put(Integer.toString(i + 1), i);
349                 }
350 
351                 Choice groupChoice = new ChoiceImpl(false);
352                 groupChoice.setMessage("Choose commands group to run");
353                 groupChoice.setKeyChoices(list);
354                 groupChoice.setSortData(sort);
355 
356                 if (feedbackPlayer.choose(Outcome.Benefit, groupChoice, game)) {
357                     String need = groupChoice.getChoiceKey();
358                     if ((need != null) && list.containsKey(need)) {
359                         runGroup = groups.get(Integer.parseInt(need) - 1);
360                     }
361                 }
362             }
363 
364             if (runGroup == null) {
365                 // was canceled
366                 logger.info("Command file was empty or canceled");
367                 return;
368             }
369 
370             logger.info("Selected group [" + runGroup.name + "] with " + runGroup.commands.size() + " commands");
371 
372             // 3. system commands
373             if (runGroup.isSpecialCommand) {
374 
375                 Player opponent = game.getPlayer(game.getOpponents(feedbackPlayer.getId()).stream().findFirst().orElse(null));
376 
377                 String info;
378                 switch (runGroup.name) {
379 
380                     case COMMAND_SHOW_OPPONENT_HAND:
381                         if (opponent != null) {
382                             info = getCardsListForSpecialInform(game, opponent.getHand(), runGroup.commands);
383                             game.informPlayer(feedbackPlayer, info);
384                         }
385                         break;
386 
387                     case COMMAND_SHOW_OPPONENT_LIBRARY:
388                         if (opponent != null) {
389                             info = getCardsListForSpecialInform(game, opponent.getLibrary().getCardList(), runGroup.commands);
390                             game.informPlayer(feedbackPlayer, info);
391                         }
392                         break;
393 
394                     case COMMAND_SHOW_MY_HAND:
395                         info = getCardsListForSpecialInform(game, feedbackPlayer.getHand(), runGroup.commands);
396                         game.informPlayer(feedbackPlayer, info);
397                         break;
398 
399                     case COMMAND_SHOW_MY_LIBRARY:
400                         info = getCardsListForSpecialInform(game, feedbackPlayer.getLibrary().getCardList(), runGroup.commands);
401                         game.informPlayer(feedbackPlayer, info);
402                         break;
403 
404                     case COMMAND_ACTIVATE_OPPONENT_ABILITY:
405                         // WARNING, maybe very bugged if called in wrong priority
406                         // uses choose triggered ability dialog to select it
407                         UUID savedPriorityPlayer = null;
408                         if (game.getActivePlayerId() != opponent.getId()) {
409                             savedPriorityPlayer = game.getActivePlayerId();
410                         }
411 
412                         // change active player to find and play selected abilities (it's danger and buggy code)
413                         if (savedPriorityPlayer != null) {
414                             game.getState().setPriorityPlayerId(opponent.getId());
415                             game.firePriorityEvent(opponent.getId());
416                         }
417 
418                         List<ActivatedAbility> abilities = opponent.getPlayable(game, true);
419                         Map<String, String> choices = new HashMap<>();
420                         abilities.forEach(ability -> {
421                             MageObject object = ability.getSourceObject(game);
422                             choices.put(ability.getId().toString(), object.getName() + ": " + ability.toString());
423                         });
424                         // TODO: set priority for us?
425                         Choice choice = new ChoiceImpl();
426                         choice.setMessage("Choose playable ability to activate by opponent " + opponent.getName());
427                         choice.setKeyChoices(choices);
428                         if (feedbackPlayer.choose(Outcome.Detriment, choice, game) && choice.getChoiceKey() != null) {
429                             String needId = choice.getChoiceKey();
430                             Optional<ActivatedAbility> ability = abilities.stream().filter(a -> a.getId().toString().equals(needId)).findFirst();
431                             if (ability.isPresent()) {
432                                 // TODO: set priority for player?
433                                 ActivatedAbility activatedAbility = ability.get();
434                                 game.informPlayers(feedbackPlayer.getLogName() + " as another player " + opponent.getLogName()
435                                         + " trying to force an activate ability: " + activatedAbility.getGameLogMessage(game));
436                                 if (opponent.activateAbility(activatedAbility, game)) {
437                                     game.informPlayers("Force to activate ability: DONE");
438                                 } else {
439                                     game.informPlayers("Force to activate ability: FAIL");
440                                 }
441                             }
442                         }
443                         // restore original priority player
444                         if (savedPriorityPlayer != null) {
445                             game.getState().setPriorityPlayerId(savedPriorityPlayer);
446                             game.firePriorityEvent(savedPriorityPlayer);
447                         }
448                         break;
449                 }
450                 return;
451             }
452 
453             // insert group refs as extra commands (only user defined groups allowed, no special)
454             Map<String, CommandGroup> otherGroupRefs = new HashMap<>();
455             for (CommandGroup group : groups) {
456                 if (group != runGroup) {
457                     otherGroupRefs.putIfAbsent(COMMAND_REF_PREFIX + group.name, group);
458                 }
459             }
460             for (int i = runGroup.commands.size() - 1; i >= 0; i--) {
461                 String line = runGroup.commands.get(i);
462                 // replace ref by real commands from another group
463                 if (line.startsWith(COMMAND_REF_PREFIX)) {
464                     CommandGroup other = otherGroupRefs.getOrDefault(line, null);
465                     if (other != null && !other.isSpecialCommand) {
466                         logger.info("Replace ref group " + line + " by " + other.commands.size() + " commands");
467                         runGroup.commands.remove(i);
468                         runGroup.commands.addAll(i, other.commands);
469                     } else {
470                         logger.error("Can't find ref group: " + line);
471                     }
472                 }
473             }
474 
475             // 4. run commands
476             for (String line : runGroup.commands) {
477 
478                 CardCommandData command = parseCardCommand(line);
479                 if (!command.OK) {
480                     logger.warn(command.Error + ": " + line);
481                     continue;
482                 }
483 
484                 Optional<Player> playerOptional = findPlayer(game, command.player);
485                 if (!playerOptional.isPresent()) {
486                     logger.warn("Unknown player: " + line);
487                     continue;
488                 }
489                 Player player = playerOptional.get();
490 
491                 // SPECIAL token/emblem call (without SET name)
492                 if ("token".equalsIgnoreCase(command.zone)) {
493                     // eg: token:Human:HippoToken:1
494                     Class<?> c = Class.forName("mage.game.permanent.token." + command.cardName);
495                     Constructor<?> cons = c.getConstructor();
496                     Object obj = cons.newInstance();
497                     if (obj instanceof Token) {
498                         Token token = (Token) obj;
499                         Ability fakeSourceAbility = fakeSourceAbilityTemplate.copy();
500                         fakeSourceAbility.setSourceId(token.getId());
501                         token.putOntoBattlefield(command.Amount, game, fakeSourceAbility, player.getId(), false, false);
502                         continue;
503                     }
504                 } else if ("emblem".equalsIgnoreCase(command.zone)) {
505                     // eg: emblem:Human:ElspethSunsChampionEmblem:1
506                     Class<?> c = Class.forName("mage.game.command.emblems." + command.cardName);
507                     Constructor<?> cons = c.getConstructor();
508                     Object emblem = cons.newInstance();
509                     if (emblem instanceof mage.game.command.Emblem) {
510                         ((mage.game.command.Emblem) emblem).setControllerId(player.getId());
511                         game.addEmblem((mage.game.command.Emblem) emblem, null, player.getId());
512                         continue;
513                     }
514                 } else if ("plane".equalsIgnoreCase(command.zone)) {
515                     if (putPlaneToGame(game, player, command.cardName)) {
516                         continue;
517                     }
518                 } else if ("loyalty".equalsIgnoreCase(command.zone)) {
519                     for (Permanent perm : game.getBattlefield().getAllActivePermanents(player.getId())) {
520                         if (perm.getName().equals(command.cardName) && perm.getCardType(game).contains(CardType.PLANESWALKER)) {
521                             Ability fakeSourceAbility = fakeSourceAbilityTemplate.copy();
522                             fakeSourceAbility.setSourceId(perm.getId());
523                             perm.addCounters(CounterType.LOYALTY.createInstance(command.Amount), fakeSourceAbility.getControllerId(), fakeSourceAbility, game);
524                         }
525                     }
526                     continue;
527                 } else if ("stack".equalsIgnoreCase(command.zone)) {
528                     // simple cast (without targets or modes)
529 
530                     // find card info
531                     CardInfo cardInfo = CardRepository.instance.findCard(command.cardName);
532                     if (cardInfo == null) {
533                         logger.warn("Unknown card for stack command [" + command.cardName + "]: " + line);
534                         continue;
535                     }
536 
537                     // put card to game
538                     Set<Card> cardsToLoad = new HashSet<>();
539                     for (int i = 0; i < command.Amount; i++) {
540                         cardsToLoad.add(cardInfo.getCard());
541                     }
542                     game.loadCards(cardsToLoad, player.getId());
543 
544                     // move card from exile to stack
545                     for (Card card : cardsToLoad) {
546                         Ability fakeSourceAbility = fakeSourceAbilityTemplate.copy();
547                         fakeSourceAbility.setSourceId(card.getId());
548                         putCardToZone(fakeSourceAbility, game, player, card, Zone.STACK);
549                     }
550 
551                     continue;
552                 }
553 
554                 Zone gameZone;
555                 if ("hand".equalsIgnoreCase(command.zone)) {
556                     gameZone = Zone.HAND;
557                 } else if ("battlefield".equalsIgnoreCase(command.zone)) {
558                     gameZone = Zone.BATTLEFIELD;
559                 } else if ("graveyard".equalsIgnoreCase(command.zone)) {
560                     gameZone = Zone.GRAVEYARD;
561                 } else if ("library".equalsIgnoreCase(command.zone)) {
562                     gameZone = Zone.LIBRARY;
563                 } else if ("token".equalsIgnoreCase(command.zone)) {
564                     gameZone = Zone.BATTLEFIELD;
565                 } else if ("exiled".equalsIgnoreCase(command.zone)) {
566                     gameZone = Zone.EXILED;
567                 } else if ("outside".equalsIgnoreCase(command.zone)) {
568                     gameZone = Zone.OUTSIDE;
569                 } else if ("emblem".equalsIgnoreCase(command.zone)) {
570                     gameZone = Zone.COMMAND;
571                 } else if ("plane".equalsIgnoreCase(command.zone)) {
572                     gameZone = Zone.COMMAND;
573                 } else if ("commander".equalsIgnoreCase(command.zone)) {
574                     gameZone = Zone.COMMAND;
575                 } else if ("sideboard".equalsIgnoreCase(command.zone)) {
576                     gameZone = Zone.OUTSIDE;
577                 } else {
578                     logger.warn("Unknown zone [" + command.zone + "]: " + line);
579                     continue;
580                 }
581 
582                 List<CardInfo> cards = null;
583                 if (command.cardSet.isEmpty()) {
584                     // by name
585                     cards = CardRepository.instance.findCards(command.cardName);
586                 } else {
587                     // by name and set
588                     cards = CardRepository.instance.findCards(new CardCriteria().setCodes(command.cardSet).name(command.cardName));
589                 }
590 
591                 if (cards.isEmpty()) {
592                     logger.warn("Unknown card [" + command.cardName + "]: " + line);
593                     continue;
594                 }
595 
596                 Set<Card> cardsToLoad = new HashSet<>();
597                 for (int i = 0; i < command.Amount; i++) {
598                     CardInfo cardInfo = cards.get(RandomUtil.nextInt(cards.size()));
599                     Card card = cardInfo != null ? cardInfo.getCard() : null;
600                     if (card != null) {
601                         cardsToLoad.add(card);
602                     }
603                 }
604                 game.loadCards(cardsToLoad, player.getId());
605 
606                 if ("commander".equalsIgnoreCase(command.zone) && cardsToLoad.size() > 0) {
607                     // as commander (only commander games, look at init code in GameCommanderImpl)
608                     if (game instanceof GameCommanderImpl) {
609                         GameCommanderImpl gameCommander = (GameCommanderImpl) game;
610                         cardsToLoad.forEach(card -> gameCommander.addCommander(card, player));
611                         cardsToLoad.forEach(card -> gameCommander.initCommander(card, player));
612                     } else {
613                         logger.fatal("Commander card can be used in commander game only: " + command.cardName);
614                     }
615                 } else if ("sideboard".equalsIgnoreCase(command.zone) && cardsToLoad.size() > 0) {
616                     // put to sideboard
617                     for (Card card : cardsToLoad) {
618                         player.getSideboard().add(card);
619                     }
620                 } else {
621                     // as other card
622                     for (Card card : cardsToLoad) {
623                         Ability fakeSourceAbility = fakeSourceAbilityTemplate.copy();
624                         fakeSourceAbility.setSourceId(card.getId());
625                         putCardToZone(fakeSourceAbility, game, player, card, gameZone);
626                     }
627                 }
628             }
629         } catch (Exception e) {
630             logger.fatal("", e);
631         }
632     }

638     private static void putCardToZone(Ability source, Game game, Player player, Card card, Zone zone) {
639         // TODO: replace by player.move?
640         switch (zone) {
641             case BATTLEFIELD:
642                 CardUtil.putCardOntoBattlefieldWithEffects(source, game, card, player);
643                 break;
644             case LIBRARY:
645                 card.setZone(Zone.LIBRARY, game);
646                 player.getLibrary().putOnTop(card, game);
647                 break;
648             case STACK:
649                 card.cast(game, game.getState().getZone(card.getId()), card.getSpellAbility(), player.getId());
650                 break;
651             case EXILED:
652                 card.setZone(Zone.EXILED, game);
653                 game.getExile().getPermanentExile().add(card);
654                 break;
655             case OUTSIDE:
656                 card.setZone(Zone.OUTSIDE, game);
657                 break;
658             default:
659                 card.moveToZone(zone, null, game, false);
660                 break;
661         }
662         game.applyEffects();
663         logger.info("Added card to player's " + zone.toString() + ": " + card.getName() + ", player = " + player.getName());
664     }

666     public static boolean putPlaneToGame(Game game, Player player, String planeClassName) {
667         // remove the last plane and set its effects to discarded
668         for (CommandObject comObject : game.getState().getCommand()) {
669             if (comObject instanceof Plane) {
670                 if (comObject.getAbilities() != null) {
671                     for (Ability ability : comObject.getAbilities()) {
672                         for (Effect effect : ability.getEffects()) {
673                             if (effect instanceof ContinuousEffect) {
674                                 ((ContinuousEffect) effect).discard();
675                             }
676                         }
677                     }
678                 }
679                 game.getState().removeTriggersOfSourceId(comObject.getId());
680                 game.getState().getCommand().remove(comObject);
681                 break;
682             }
683         }
684 
685         // put new plane to game
686         Planes planeType = Planes.fromClassName(planeClassName);
687         Plane plane = Plane.createPlane(planeType);
688         if (plane != null) {
689             plane.setControllerId(player.getId());
690             game.addPlane(plane, null, player.getId());
691             return true;
692         }
693         return false;
694     }

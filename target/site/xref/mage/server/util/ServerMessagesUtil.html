20  /**
21   * Handles server messages (Messages of the Day). Reloads messages every 5
22   * minutes.
23   *
24   * @author nantuko
25   */
26  public enum ServerMessagesUtil {
27      instance;
28  
29      private static final Logger LOGGER = Logger.getLogger(ServerMessagesUtil.class);
30      private static final String SERVER_MSG_TXT_FILE = "server.msg.txt";
31  
32      private final List<String> messages = new ArrayList<>();
33      private final ReadWriteLock lock = new ReentrantReadWriteLock();
34      private static boolean ignore = false;
35  
36      private static long startDate;
37      private static final AtomicInteger gamesStarted = new AtomicInteger(0);
38      private static final AtomicInteger gamesEnded = new AtomicInteger(0);
39      private static final AtomicInteger tournamentsStarted = new AtomicInteger(0);
40      private static final AtomicInteger tournamentsEnded = new AtomicInteger(0);
41      private static final AtomicInteger lostConnection = new AtomicInteger(0);
42      private static final AtomicInteger reconnects = new AtomicInteger(0);
43  
44      ServerMessagesUtil() {
45          ScheduledExecutorService updateExecutor = Executors.newSingleThreadScheduledExecutor();
46          updateExecutor.scheduleAtFixedRate(this::reloadMessages, 5, 5 * 60, TimeUnit.SECONDS);
47      }
48  
49      public List<String> getMessages() {
50          lock.readLock().lock();
51          try {
52              return messages;
53          } finally {
54              lock.readLock().unlock();
55          }
56      }
57  
58      private void reloadMessages() {
59          LOGGER.debug("Reading server messages...");
60          List<String> motdMessages = readFromFile();
61          List<String> newMessages = new ArrayList<>(motdMessages);
62          newMessages.add(getServerStatistics());
63          newMessages.add(getServerStatistics2());
64  
65          lock.writeLock().lock();
66          try {
67              messages.clear();
68              messages.addAll(newMessages);
69          } finally {
70              lock.writeLock().unlock();
71          }
72      }
73  
74      private List<String> readFromFile() {
75          if (ignore) {
76              return Collections.emptyList();
77          }
78  
79          InputStream is = null;
80          File file = new File(SERVER_MSG_TXT_FILE);
81          if (!file.exists() || !file.canRead()) {
82              // warn user about miss messages file, except dev environment
83              if (!file.getAbsolutePath().contains("Mage.Server")) {
84                  LOGGER.warn("Couldn't find server messages file using path: " + file.getAbsolutePath());
85              }
86          } else {
87              try {
88                  is = new FileInputStream(file);
89                  ignore = false;
90              } catch (Exception f) {
91                  LOGGER.error(f, f);
92                  ignore = true;
93              }
94          }
95  
96          if (is == null) {
97              return Collections.emptyList();
98          }
99  
100         List<String> newMessages = new ArrayList<>();
101         try (Scanner scanner = new Scanner(is)) {
102             while (scanner.hasNextLine()) {
103                 String message = scanner.nextLine().trim();
104                 if (message.startsWith("//") || message.isEmpty()) {
105                     continue;
106                 }
107                 newMessages.add(message.trim());
108             }
109         } catch (Exception e) {
110             LOGGER.error(e.getMessage(), e);
111         } finally {
112             StreamUtils.closeQuietly(is);
113         }
114         return newMessages;
115     }
116 
117     private String getServerStatistics() {
118         long current = System.currentTimeMillis();
119         long hours = ((current - startDate) / (1000 * 60 * 60));
120         String statistics = "Server uptime: " + hours + " hour(s)"
121                 + "; Games started: " + gamesStarted.get() + ", ended: " + gamesEnded.get()
122                 + "; Tourneys started: " + tournamentsStarted.get() + ", ended: " + tournamentsEnded.get();
123         return statistics;
124     }
125 
126     private String getServerStatistics2() {
127         long current = System.currentTimeMillis();
128         long minutes = ((current - startDate) / (1000 * 60));
129         if (minutes == 0) {
130             minutes = 1;
131         }
132         String statistics = "Disconnects: " + lostConnection.get() + ", avg/hour: " + lostConnection.get() * 60 / minutes
133                 + "; Reconnects: " + reconnects.get() + ", avg/hour: " + reconnects.get() * 60 / minutes;
134         return statistics;
135     }
136 
137     public void setStartDate(long milliseconds) {
138         startDate = milliseconds;
139     }
140 
141     public void incGamesStarted() {
142         int value;
143         do {
144             value = gamesStarted.get();
145         } while (!gamesStarted.compareAndSet(value, value + 1));
146     }
147

153     }

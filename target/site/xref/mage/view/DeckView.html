13      private final SimpleCardsView cards;
14      private final SimpleCardsView sideboard;
15  
16      public DeckView(Deck deck) {

18          cards = new SimpleCardsView(deck.getCards(), false);
19          sideboard = new SimpleCardsView(deck.getSideboard(), false);
20      }
21  
22      public String getName() {
23          return name;
24      }
25  
26      public SimpleCardsView getCards() {
27          return cards;
28      }
29

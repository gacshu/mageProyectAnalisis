3   package mage.view;
4   
5   import java.io.Serializable;
6   import java.util.ArrayList;
7   import java.util.Collections;
8   import java.util.Date;
9   import java.util.List;
10  import mage.game.tournament.Round;
11  import mage.game.tournament.Tournament;
12  import mage.game.tournament.TournamentPlayer;
13  
14  /**
15   *
16   * @author BetaSteward_at_googlemail.com
17   */
18  public class TournamentView implements Serializable {
19      private static final long serialVersionUID = 1L;
20  
21      private final String tournamentName;
22      private final String tournamentType;
23      private final String tournamentState;
24      
25      private final Date startTime;
26      private final Date endTime;
27      private final Date stepStartTime;
28      private final Date serverTime;
29      private final int constructionTime;
30      private final boolean watchingAllowed;
31  
32      private final List<RoundView> rounds = new ArrayList<>();
33      private final List<TournamentPlayerView> players = new ArrayList<>();
34      private final String runningInfo;
35      
36      public TournamentView(Tournament tournament) {
37  
38          tournamentName = tournament.getOptions().getName();
39          StringBuilder typeText = new StringBuilder(tournament.getOptions().getTournamentType());
40          if (!tournament.getTournamentType().isLimited()) {
41              typeText.append(" / ").append(tournament.getOptions().getMatchOptions().getDeckType());
42          }
43          if (tournament.getNumberRounds() > 0) {
44              typeText.append(' ').append(tournament.getNumberRounds()).append(" rounds");
45          } 
46          tournamentType = typeText.toString();
47          startTime = tournament.getStartTime();
48          endTime = tournament.getEndTime();
49          stepStartTime = tournament.getStepStartTime();
50          constructionTime = tournament.getOptions().getLimitedOptions().getConstructionTime();
51          watchingAllowed = tournament.getOptions().isWatchingAllowed();
52          serverTime = new Date();
53          tournamentState = tournament.getTournamentState();
54  
55          if (tournament.getTournamentState().equals("Drafting") && tournament.getDraft() != null) {
56              runningInfo = "booster/card: " + tournament.getDraft().getBoosterNum() + '/' + (tournament.getDraft().getCardNum());
57          } else {
58              runningInfo = "";
59          }
60          for (TournamentPlayer player: tournament.getPlayers()) {
61              players.add(new TournamentPlayerView(player));
62          }
63          Collections.sort(players);
64          for (Round round: tournament.getRounds()) {
65              rounds.add(new RoundView(round));
66          }
67      }
68  
69      public String getTournamentName() {
70          return tournamentName;
71      }
72  
73      public String getTournamentType() {
74          return tournamentType;
75      }
76  
77      public Date getStartTime() {
78          return new Date(startTime.getTime());
79      }
80  
81      public Date getEndTime() {
82          if (endTime == null) {
83              return null;
84          }
85          return new Date(endTime.getTime());
86      }
87  
88      public boolean isWatchingAllowed() {
89          return watchingAllowed;
90      }
91  
92      public List<TournamentPlayerView> getPlayers() {
93          return players;
94      }
95  
96      public List<RoundView> getRounds() {
97          return rounds;
98      }
99  
100     public String getTournamentState() {
101         return tournamentState;
102     }
103 
104     public Date getStepStartTime() {
105         return stepStartTime;
106     }
107 
108     public int getConstructionTime() {
109         return constructionTime;
110     }
111 
112     public Date getServerTime() {
113         return serverTime;
114     }
115 
116     public String getRunningInfo() {
117         return runningInfo;
118     }
119     
120 }

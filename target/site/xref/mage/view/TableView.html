47      public TableView(Table table) {
48          this.tableId = table.getId();
49          this.gameType = table.getGameType();
50          this.tableName = table.getName();
51          this.controllerName = table.getControllerName();
52          this.tableState = table.getState();
53          if (table.getState() == TableState.WAITING
54                  || table.getState() == TableState.READY_TO_START
55                  || table.getState() == TableState.STARTING) {
56              this.createTime = table.getCreateTime();
57          } else {
58              if (table.isTournament()) {
59                  this.createTime = table.getTournament().getStartTime();
60              } else {
61                  this.createTime = table.getMatch().getStartTime();
62              }
63          }
64          this.isTournament = table.isTournament();
65          for (Seat seat : table.getSeats()) {
66              seats.add(new SeatView(seat));
67          }
68          if (!table.isTournament()) {
69              // MATCH
70              seatsInfo = "" + table.getMatch().getPlayers().size() + '/' + table.getSeats().length;
71              if (table.getState() == TableState.WAITING || table.getState() == TableState.READY_TO_START) {
72                  tableStateText = table.getState().toString() + " (" + seatsInfo + ')';
73              } else {
74                  tableStateText = table.getState().toString();
75              }
76              for (Game game : table.getMatch().getGames()) {
77                  games.add(game.getId());
78              }
79              StringBuilder sb = new StringBuilder();
80              StringBuilder sbScore = new StringBuilder();
81              for (MatchPlayer matchPlayer : table.getMatch().getPlayers()) {
82                  if (matchPlayer.getPlayer() == null) {
83                      sb.append(", ").append("[unknown]");
84                      sbScore.append('-').append(matchPlayer.getWins());
85                  } else if (!matchPlayer.getName().equals(table.getControllerName())) {
86                      sb.append(", ").append(matchPlayer.getName());
87                      sbScore.append('-').append(matchPlayer.getWins());
88                  } else {
89                      sbScore.insert(0, matchPlayer.getWins()).insert(0, " Score: ");
90                  }
91              }
92              if (table.getMatch().getDraws() > 0) {
93                  sbScore.append(" Draws: ").append(table.getMatch().getDraws());
94              }
95              this.controllerName += sb.toString();
96              this.deckType = table.getDeckType();
97              StringBuilder addInfo = new StringBuilder();
98              if (table.getMatch().getGames().isEmpty()) {
99                  addInfo.append("Wins:").append(table.getMatch().getWinsNeeded());
100                 addInfo.append(" Time: ").append(table.getMatch().getOptions().getMatchTimeLimit().toString());
101                 if (table.getMatch().getFreeMulligans() > 0) {
102                     addInfo.append(" FM: ").append(table.getMatch().getFreeMulligans());
103                 }
104             } else {
105                 addInfo.append("Wins:").append(table.getMatch().getWinsNeeded());
106                 addInfo.append(sbScore.toString());
107             }
108             if (table.getMatch().getOptions().isRollbackTurnsAllowed()) {
109                 addInfo.append(" RB");
110             }
111             if (table.getMatch().getOptions().isPlaneChase()) {
112                 addInfo.append(" PC");
113             }
114             if (table.getMatch().getOptions().isSpectatorsAllowed()) {
115                 addInfo.append(" SP");
116             }
117             if (table.getNumberOfSeats() > 3) {
118                 addInfo.append(" Rng: ").append(table.getMatch().getOptions().getRange().toString());
119             }
120             this.additionalInfo = addInfo.toString();
121             this.skillLevel = table.getMatch().getOptions().getSkillLevel();
122             this.quitRatio = Integer.toString(table.getMatch().getOptions().getQuitRatio());
123             this.minimumRating = Integer.toString(table.getMatch().getOptions().getMinimumRating());
124             this.limited = table.getMatch().getOptions().isLimited();
125             this.rated = table.getMatch().getOptions().isRated();
126             this.passworded = !table.getMatch().getOptions().getPassword().isEmpty();
127             this.spectatorsAllowed = table.getMatch().getOptions().isSpectatorsAllowed();
128         } else {
129             // TOURNAMENT
130             if (table.getTournament().getOptions().getNumberRounds() > 0) {
131                 this.gameType = new StringBuilder(this.gameType).append(' ').append(table.getTournament().getOptions().getNumberRounds()).append(" Rounds").toString();
132             }
133             StringBuilder sb1 = new StringBuilder();
134             for (TournamentPlayer tp : table.getTournament().getPlayers()) {
135                 if (!tp.getPlayer().getName().equals(table.getControllerName())) {
136                     sb1.append(", ").append(tp.getPlayer().getName());
137                 }
138             }
139             this.controllerName += sb1.toString();
140             this.seatsInfo = "" + table.getTournament().getPlayers().size() + "/" + table.getNumberOfSeats();
141             StringBuilder infoText = new StringBuilder();
142             StringBuilder stateText = new StringBuilder(table.getState().toString());
143             infoText.append("Wins:").append(table.getTournament().getOptions().getMatchOptions().getWinsNeeded());
144             infoText.append(" Seats: ").append(this.seatsInfo);
145             switch (table.getState()) {
146                 case WAITING:
147                 case READY_TO_START:
148                 case STARTING:
149                     if (TableState.WAITING.equals(table.getState())) {
150                         stateText.append(" (").append(table.getTournament().getPlayers().size()).append('/').append(table.getNumberOfSeats()).append(')');
151                     }
152                     infoText.append(" Time: ").append(table.getTournament().getOptions().getMatchOptions().getMatchTimeLimit().toString());
153                     if (table.getTournament().getOptions().getMatchOptions().getFreeMulligans() > 0) {
154                         infoText.append(" FM: ").append(table.getTournament().getOptions().getMatchOptions().getFreeMulligans());
155                     }
156                     if (table.getTournament().getTournamentType().isLimited()) {
157                         infoText.append(" Constr.: ").append(table.getTournament().getOptions().getLimitedOptions().getConstructionTime() / 60).append(" Min.");
158                     }
159                     if (table.getTournament().getOptions().getLimitedOptions() instanceof DraftOptions) {
160                         DraftOptions draftOptions = (DraftOptions) table.getTournament().getOptions().getLimitedOptions();
161                         infoText.append(" Pick time: ").append(draftOptions.getTiming().getShortName());
162                     }
163                     if (table.getTournament().getOptions().getMatchOptions().isRollbackTurnsAllowed()) {
164                         infoText.append(" RB");
165                     }
166                     if (table.getTournament().getOptions().getMatchOptions().isPlaneChase()) {
167                         infoText.append(" PC");
168                     }
169                     if (table.getTournament().getOptions().isWatchingAllowed()) {
170                         infoText.append(" SP");
171                     }
172                     
173                     break;
174                 case DUELING:
175                     stateText.append(" Round: ").append(table.getTournament().getRounds().size());
176                     break;
177                 case DRAFTING:
178                     Draft draft = table.getTournament().getDraft();
179                     if (draft != null) {
180                         stateText.append(' ').append(draft.getBoosterNum()).append('/').append(draft.getCardNum());
181                     }
182                     break;
183                 default:
184             }
185             this.additionalInfo = infoText.toString();
186             this.tableStateText = stateText.toString();
187             this.deckType = table.getDeckType() + ' ' + table.getTournament().getBoosterInfo();
188             this.skillLevel = table.getTournament().getOptions().getMatchOptions().getSkillLevel();
189             this.quitRatio = Integer.toString(table.getTournament().getOptions().getQuitRatio());
190             this.minimumRating = Integer.toString(table.getTournament().getOptions().getMinimumRating());
191             this.limited = table.getTournament().getOptions().getMatchOptions().isLimited();
192             this.rated = table.getTournament().getOptions().getMatchOptions().isRated();
193             this.passworded = !table.getTournament().getOptions().getPassword().isEmpty();
194             this.spectatorsAllowed = table.getTournament().getOptions().isWatchingAllowed();
195         }
196     }

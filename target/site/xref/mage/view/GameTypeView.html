3   package mage.view;
4   
5   import java.io.Serializable;
6   import mage.game.match.MatchType;
7   
8   /**
9    *
10   * @author BetaSteward_at_googlemail.com
11   */
12  public class GameTypeView extends Object implements Serializable {
13      private static final long serialVersionUID = 1L;
14  
15      private final String name;
16      private final int minPlayers;
17      private final int maxPlayers;
18      private final int numTeams;
19      private final int playersPerTeam;
20      private final boolean useRange;
21      private final boolean useAttackOption;
22  
23      public GameTypeView(MatchType gameType) {
24          this.name = gameType.getName();
25          this.minPlayers = gameType.getMinPlayers();
26          this.maxPlayers = gameType.getMaxPlayers();
27          this.numTeams = gameType.getNumTeams();
28          this.playersPerTeam = gameType.getPlayersPerTeam();
29          this.useAttackOption = gameType.isUseAttackOption();
30          this.useRange = gameType.isUseRange();
31      }
32  
33      @Override
34      public String toString() {
35          return name;
36      }
37  
38      public String getName() {
39          return name;
40      }
41  
42      public int getMinPlayers() {
43          return minPlayers;
44      }
45  
46      public int getMaxPlayers() {
47          return maxPlayers;
48      }
49  
50      public int getNumTeams() {
51          return numTeams;
52      }
53  
54      public int getPlayersPerTeam() {
55          return playersPerTeam;
56      }
57  
58      public boolean isUseRange() {
59          return useRange;
60      }
61  
62      public boolean isUseAttackOption() {
63          return useAttackOption;
64      }
65  
66  
67  }

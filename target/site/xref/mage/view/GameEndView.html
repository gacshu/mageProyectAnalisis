19  public class GameEndView implements Serializable {
20  
21      private PlayerView clientPlayer = null;
22      private final List<PlayerView> players = new ArrayList<>();
23      private final Date startTime;
24      private final Date endTime;
25      private String gameInfo;
26      private final String matchInfo;
27      private final String additionalInfo;
28      private boolean won;
29      private final MatchView matchView;
30      private int wins;
31      private int loses;
32      private final int winsNeeded;
33  
34      public GameEndView(GameState state, Game game, UUID playerId, Table table) {
35          startTime = game.getStartTime();
36          endTime = game.getEndTime();
37  
38          // set result message
39          int winner = 0;
40          Player you = null;
41          for (Player player : state.getPlayers().values()) {
42              PlayerView playerView = new PlayerView(player, state, game, playerId, null);
43              if (playerView.getPlayerId().equals(playerId)) {
44                  clientPlayer = playerView;
45                  you = player;
46                  won = you.hasWon(); // needed to control image
47              }
48              players.add(playerView);
49              if (player.hasWon()) {
50                  winner++;
51              }
52          }
53          if (you != null) {
54              if (you.hasWon()) {
55                  gameInfo = "You won the game on turn " + game.getTurnNum() + ".";
56              } else if (winner > 0) {
57                  gameInfo = "You lost the game on turn " + game.getTurnNum() + ".";
58              } else {
59                  gameInfo = "Game is a draw on Turn " + game.getTurnNum() + ".";
60              }
61          }
62          matchView = new MatchView(table);
63  
64          Match match = table.getMatch();
65          MatchPlayer matchWinner = null;
66          winsNeeded = match.getOptions().getWinsNeeded();
67          StringBuilder additonalText = new StringBuilder();
68          for (MatchPlayer matchPlayer : match.getPlayers()) {
69              if (matchPlayer.getPlayer().equals(you)) {
70                  wins = matchPlayer.getWins();
71              }
72              if (matchPlayer.isMatchWinner()) {
73                  matchWinner = matchPlayer;
74              }
75              if (matchPlayer.getPlayer().hasTimerTimeout()) {
76                  if (matchPlayer.getPlayer().equals(you)) {
77                      additonalText.append("You run out of time. ");
78                  } else {
79                      additonalText.append(matchPlayer.getName()).append(" runs out of time. ");
80                  }
81              } else if (matchPlayer.getPlayer().hasIdleTimeout()) {
82                  if (matchPlayer.getPlayer().equals(you)) {
83                      additonalText.append("You lost the match for being idle. ");
84                  } else {
85                      additonalText.append(matchPlayer.getName()).append(" lost for being idle. ");
86                  }
87              } else if (matchPlayer.hasQuit()) {
88                  if (matchPlayer.getPlayer().equals(you)) {
89                      additonalText.append("You have quit the match. ");
90                  } else {
91                      additonalText.append(matchPlayer.getName()).append(" has quit the match. ");
92                  }
93              }
94          }
95  
96          if (matchWinner != null) {
97              if (matchWinner.getPlayer().equals(you)) {
98                  matchInfo = "You won the match!";
99              } else {
100                 matchInfo = new StringBuilder(matchWinner.getName()).append(" won the match!").toString();
101             }
102         } else {
103             matchInfo = new StringBuilder("You need ").append(winsNeeded - wins == 1 ? "one more win " : winsNeeded - wins + " more wins ").append("to win the match.").toString();
104         }
105         additionalInfo = additonalText.toString();
106 
107     }

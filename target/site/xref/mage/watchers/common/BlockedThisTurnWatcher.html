2   package mage.watchers.common;
3   
4   import mage.MageObjectReference;
5   import mage.constants.WatcherScope;
6   import mage.game.Game;
7   import mage.game.events.GameEvent;
8   import mage.game.permanent.Permanent;
9   import mage.watchers.Watcher;
10  
11  import java.util.HashSet;
12  import java.util.Set;
13  
14  /**
15   * @author Quercitron
16   */
17  public class BlockedThisTurnWatcher extends Watcher {
18  
19      private final Set<MageObjectReference> blockedThisTurnCreatures = new HashSet<>();
20  
21      public BlockedThisTurnWatcher() {
22          super(WatcherScope.GAME);
23      }
24  
25      @Override
26      public void watch(GameEvent event, Game game) {
27          if (event.getType() == GameEvent.EventType.BLOCKER_DECLARED) {
28              this.blockedThisTurnCreatures.add(new MageObjectReference(event.getSourceId(), game));
29          }
30      }
31  
32      public Set<MageObjectReference> getBlockedThisTurnCreatures() {
33          return this.blockedThisTurnCreatures;
34      }
35  
36      public boolean checkIfBlocked(Permanent permanent, Game game) {
37          for (MageObjectReference mor : blockedThisTurnCreatures) {
38              if (mor.refersTo(permanent, game)) {
39                  return true;
40              }
41          }
42          return false;
43      }
44  
45      @Override
46      public void reset() {
47          super.reset();
48          blockedThisTurnCreatures.clear();
49      }
50  
51  }

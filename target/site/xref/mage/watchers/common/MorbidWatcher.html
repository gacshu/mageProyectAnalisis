2   package mage.watchers.common;
3   
4   import mage.constants.WatcherScope;
5   import mage.game.Game;
6   import mage.game.events.GameEvent;
7   import mage.game.events.ZoneChangeEvent;
8   import mage.watchers.Watcher;
9   
10  /**
11   *
12   * @author BetaSteward_at_googlemail.com
13   */
14  public class MorbidWatcher extends Watcher {
15  
16      public MorbidWatcher() {
17          super(WatcherScope.GAME);
18      }
19  
20      @Override
21      public void watch(GameEvent event, Game game) {
22          if (condition) {
23              return;
24          }
25          if (event.getType() == GameEvent.EventType.ZONE_CHANGE
26                  && ((ZoneChangeEvent) event).isDiesEvent()
27                  && ((ZoneChangeEvent) event).getTarget().isCreature(game)) {
28              condition = true;
29          }
30      }
31  }

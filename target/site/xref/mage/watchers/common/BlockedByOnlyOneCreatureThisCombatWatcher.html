2   package mage.watchers.common;
3   
4   import java.util.*;
5   
6   import mage.constants.WatcherScope;
7   import mage.game.Game;
8   import mage.game.combat.CombatGroup;
9   import mage.game.events.GameEvent;
10  import mage.watchers.Watcher;
11  
12  /**
13   *
14   * @author L_J
15   */
16  public class BlockedByOnlyOneCreatureThisCombatWatcher extends Watcher {
17  
18      private final Map<CombatGroup, UUID> blockedByOneCreature = new HashMap<>();
19  
20      public BlockedByOnlyOneCreatureThisCombatWatcher() {
21          super(WatcherScope.GAME);
22      }
23  
24      @Override
25      public void watch(GameEvent event, Game game) {
26          if (event.getType() == GameEvent.EventType.BEGIN_COMBAT_STEP_PRE) {
27              this.blockedByOneCreature.clear();
28          }
29          else if (event.getType() == GameEvent.EventType.BLOCKER_DECLARED) {
30              CombatGroup combatGroup = game.getCombat().findGroup(event.getTargetId());
31              if (combatGroup != null) {
32                  if (combatGroup.getBlockers().size() == 1) {
33                      if (!blockedByOneCreature.containsKey(combatGroup)) {
34                          blockedByOneCreature.put(combatGroup, event.getSourceId());
35                      }
36                      else if (!Objects.equals(blockedByOneCreature.get(combatGroup), event.getSourceId())) {
37                          blockedByOneCreature.put(combatGroup, null);
38                      }
39                  }
40                  else if (combatGroup.getBlockers().size() > 1) {
41                      blockedByOneCreature.put(combatGroup, null);
42                  }
43              }
44          }
45      }
46  
47      public Set<CombatGroup> getBlockedOnlyByCreature(UUID creature) {
48          Set<CombatGroup> combatGroups = new HashSet<>();
49          for (Map.Entry<CombatGroup, UUID> entry : blockedByOneCreature.entrySet()) {
50              if (entry.getValue() != null) {
51                  if (entry.getValue().equals(creature)) {
52                      combatGroups.add(entry.getKey());
53                  }
54              }
55          }
56          if (combatGroups.size() > 0) {
57              return combatGroups;
58          }
59          return null;
60      }
61  
62  }

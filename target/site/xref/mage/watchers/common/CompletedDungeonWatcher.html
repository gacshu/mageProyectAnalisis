1   package mage.watchers.common;
2   
3   import mage.constants.WatcherScope;
4   import mage.game.Game;
5   import mage.game.events.GameEvent;
6   import mage.watchers.Watcher;
7   
8   import java.util.*;
9   
10  /**
11   * @author TheElk801
12   */
13  public class CompletedDungeonWatcher extends Watcher {
14  
15      private final Map<UUID, Set<String>> playerMap = new HashMap<>();
16      private static final Set<String> emptySet = new HashSet<>();
17  
18      public CompletedDungeonWatcher() {
19          super(WatcherScope.GAME);
20      }
21  
22      @Override
23      public void watch(GameEvent event, Game game) {
24          switch (event.getType()) {
25              case DUNGEON_COMPLETED:
26                  playerMap.computeIfAbsent(event.getPlayerId(), u -> new HashSet<>()).add(event.getData());
27                  return;
28              case BEGINNING_PHASE_PRE:
29                  if (game.getTurnNum() == 1) {
30                      playerMap.clear();
31                  }
32          }
33      }
34  
35      public static boolean checkPlayer(UUID playerId, Game game) {
36          CompletedDungeonWatcher watcher = game.getState().getWatcher(CompletedDungeonWatcher.class);
37          return watcher != null && !watcher.playerMap.getOrDefault(playerId, emptySet).isEmpty();
38      }
39  
40      public static Set<String> getCompletedNames(UUID playerId, Game game) {
41          CompletedDungeonWatcher watcher = game.getState().getWatcher(CompletedDungeonWatcher.class);
42          return watcher != null ? watcher.playerMap.getOrDefault(playerId, emptySet) : emptySet;
43      }
44  }

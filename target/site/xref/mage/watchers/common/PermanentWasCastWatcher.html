2   package mage.watchers.common;
3   
4   import java.util.HashSet;
5   import java.util.Set;
6   import java.util.UUID;
7   import mage.cards.Card;
8   import mage.constants.WatcherScope;
9   import mage.constants.Zone;
10  import mage.game.Game;
11  import mage.game.events.GameEvent;
12  import mage.game.events.ZoneChangeEvent;
13  import mage.game.stack.Spell;
14  import mage.watchers.Watcher;
15  
16  /**
17   *
18   * @author LevelX2
19   */
20  public class PermanentWasCastWatcher extends Watcher {
21  
22      private final Set<UUID> permanentsCasted = new HashSet<>();
23  
24      public PermanentWasCastWatcher() {
25          super(WatcherScope.GAME);
26      }
27  
28      @Override
29      public void watch(GameEvent event, Game game) {
30          if (event.getType() == GameEvent.EventType.SPELL_CAST) {
31              Spell spell = (Spell) game.getObject(event.getTargetId());
32              if (spell != null) {
33                  Card card = game.getCard(spell.getSourceId());
34                  if (card != null && card.isPermanent(game)) {
35                      permanentsCasted.add(card.getId());
36                  }
37              }
38          }
39          if (event.getType() == GameEvent.EventType.ZONE_CHANGE
40                  && ((ZoneChangeEvent) event).getFromZone() == Zone.BATTLEFIELD) {
41              Card card = game.getCard(event.getTargetId());
42              if (card != null && card.isPermanent(game)) {
43                  permanentsCasted.remove(card.getId());
44              }
45          }
46      }
47  
48      public boolean wasPermanentCastThisTurn(UUID permanentSourceId) {
49          return permanentsCasted.contains(permanentSourceId);
50      }
51  
52      @Override
53      public void reset() {
54          super.reset();
55          permanentsCasted.clear();
56      }
57  }

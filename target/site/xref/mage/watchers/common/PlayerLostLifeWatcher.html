9   import java.util.HashMap;
10  import java.util.Map;
11  import java.util.UUID;
12  
13  /*
14   * Counts amount of life lost current or last turn by players.
15   * This watcher is automatically started in gameImpl.init for each game
16   *
17   * @author LevelX2
18   */
19  public class PlayerLostLifeWatcher extends Watcher {
20  
21      private final Map<UUID, Integer> amountOfLifeLostThisTurn = new HashMap<>();
22      private final Map<UUID, Integer> amountOfLifeLostLastTurn = new HashMap<>();
23  
24      public PlayerLostLifeWatcher() {
25          super(WatcherScope.GAME);
26      }
27  
28      @Override
29      public void watch(GameEvent event, Game game) {
30          if (event.getType() == GameEvent.EventType.LOST_LIFE) {
31              UUID playerId = event.getPlayerId();
32              if (playerId != null) {
33                  Integer amount = amountOfLifeLostThisTurn.get(playerId);
34                  if (amount == null) {
35                      amount = event.getAmount();
36                  } else {
37                      amount = amount + event.getAmount();
38                  }
39                  amountOfLifeLostThisTurn.put(playerId, amount);
40              }
41          }
42      }
43  
44      public int getLifeLost(UUID playerId) {
45          return amountOfLifeLostThisTurn.getOrDefault(playerId, 0);
46      }
47  
48      public int getAllOppLifeLost(UUID playerId, Game game) {
49          int amount = 0;
50          for (UUID opponentId : this.amountOfLifeLostThisTurn.keySet()) {
51              Player opponent = game.getPlayer(opponentId);
52              if (opponent != null && opponent.hasOpponent(playerId, game)) {
53                  amount += this.amountOfLifeLostThisTurn.getOrDefault(opponentId, 0);
54              }
55          }
56          return amount;
57      }
58  
59      public int getLifeLostLastTurn(UUID playerId) {
60          return amountOfLifeLostLastTurn.getOrDefault(playerId, 0);
61      }
62  
63      @Override
64      public void reset() {
65          super.reset();
66          amountOfLifeLostLastTurn.clear();
67          amountOfLifeLostLastTurn.putAll(amountOfLifeLostThisTurn);
68          amountOfLifeLostThisTurn.clear();
69      }
70  }

21  public final class AzoriusPloy extends CardImpl {
22  
23      public AzoriusPloy(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{W}{W}{U}");
25  
26  
27          // Prevent all combat damage target creature would deal this turn.
28          Effect effect = new PreventDamageByTargetEffect( Duration.EndOfTurn, true);
29          effect.setText("Prevent all combat damage target creature would deal this turn.");
30          this.getSpellAbility().addEffect(effect);
31          Target target = new TargetCreaturePermanent(new FilterCreaturePermanent("first creature"));
32          this.getSpellAbility().addTarget(target);
33  
34          // Prevent all combat damage that would be dealt to target creature this turn.
35          Effect effect2 = new PreventDamageToTargetEffect(Duration.EndOfTurn, true);
36          effect2.setText("<br><br>Prevent all combat damage that would be dealt to target creature this turn.");
37          effect2.setTargetPointer(new SecondTargetPointer());
38          this.getSpellAbility().addEffect(effect2);
39          target = new TargetCreaturePermanent(new FilterCreaturePermanent("second creature (can be the same as the first)"));
40          this.getSpellAbility().addTarget(target);
41  
42      }
43  
44      private AzoriusPloy(final AzoriusPloy card) {
45          super(card);
46      }
47  
48      @Override
49      public AzoriusPloy copy() {
50          return new AzoriusPloy(this);
51      }
52  }

7   import mage.abilities.costs.common.SacrificeTargetCost;
8   import mage.abilities.dynamicvalue.common.CardsInControllerGraveyardCount;
9   import mage.abilities.effects.common.TapSourceUnlessPaysEffect;
10  import mage.abilities.effects.common.continuous.SetBasePowerToughnessSourceEffect;
11  import mage.abilities.keyword.FlyingAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.*;
15  import mage.filter.StaticFilters;
16  import mage.target.common.TargetControlledPermanent;
17  
18  /**
19   *
20   * @author MajorLazar
21   */
22  public final class ApocalypseDemon extends CardImpl {
23  
24      public ApocalypseDemon(UUID ownerId, CardSetInfo cardSetInfo) {
25          super(ownerId, cardSetInfo, new CardType[]{CardType.CREATURE}, "{4}{B}{B}");
26          subtype.add(SubType.DEMON);
27  
28          // Flying
29          this.addAbility(FlyingAbility.getInstance());
30  
31          // Apocalypse Demon's power and toughness are each equal to the number of cards in your graveyard.
32          this.addAbility(new SimpleStaticAbility(Zone.ALL, new SetBasePowerToughnessSourceEffect(new CardsInControllerGraveyardCount(), Duration.EndOfGame)));
33  
34          // At the beginning of your upkeep, tap Apocalypse Demon unless you sacrifice another creature.
35          TapSourceUnlessPaysEffect tapEffect = new TapSourceUnlessPaysEffect(new SacrificeTargetCost(new TargetControlledPermanent(StaticFilters.FILTER_CONTROLLED_ANOTHER_CREATURE)));
36          tapEffect.setText("tap {this} unless you sacrifice another creature.");
37          this.addAbility(new BeginningOfUpkeepTriggeredAbility(tapEffect, TargetController.YOU, false));
38      }
39  
40      private ApocalypseDemon(final ApocalypseDemon card) {
41          super(card);
42      }
43  
44      @Override
45      public ApocalypseDemon copy() {
46          return new ApocalypseDemon(this);
47      }
48  }

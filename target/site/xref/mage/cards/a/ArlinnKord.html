23  /**
24   *
25   * @author fireshoes
26   */
27  public final class ArlinnKord extends CardImpl {
28  
29      public ArlinnKord(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId,setInfo,new CardType[]{CardType.PLANESWALKER},"{2}{R}{G}");
31          this.addSuperType(SuperType.LEGENDARY);
32          this.subtype.add(SubType.ARLINN);
33  
34          this.secondSideCardClazz = ArlinnEmbracedByTheMoon.class;
35  
36          this.setStartingLoyalty(3);
37  
38          // +1: Until end of turn, up to one target creature gets +2/+2 and gains vigilance and haste.
39          Effect effect = new BoostTargetEffect(2, 2, Duration.EndOfTurn);
40          effect.setText("Until end of turn, up to one target creature gets +2/+2");
41          LoyaltyAbility ability = new LoyaltyAbility(effect, 1);
42          effect = new GainAbilityTargetEffect(VigilanceAbility.getInstance(), Duration.EndOfTurn);
43          effect.setText("and gains vigilance");
44          ability.addEffect(effect);
45          effect = new GainAbilityTargetEffect(HasteAbility.getInstance(), Duration.EndOfTurn);
46          effect.setText("and haste");
47          ability.addEffect(effect);
48          ability.addTarget(new TargetCreaturePermanent(0, 1));
49          this.addAbility(ability);
50  
51          // 0: Create a 2/2 green Wolf creature token. Transform Arlinn Kord.
52          this.addAbility(new TransformAbility());
53          ability = new LoyaltyAbility(new CreateTokenEffect(new WolfToken()), 0);
54          ability.addEffect(new TransformSourceEffect());
55          this.addAbility(ability);
56      }
57  
58      private ArlinnKord(final ArlinnKord card) {
59          super(card);
60      }
61  
62      @Override
63      public ArlinnKord copy() {
64          return new ArlinnKord(this);
65      }
66  }

24  public final class AgonizingRemorse extends CardImpl {
25  
26      public AgonizingRemorse(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{B}");
28  
29          // Target opponent reveals their hand. You choose a nonland card from 
30          // it or a card from their graveyard. Exile that card. You lose 1 life.
31          this.getSpellAbility().addEffect(new AgonizingRemorseEffect());
32          this.getSpellAbility().addTarget(new TargetOpponent());
33      }
34  
35      private AgonizingRemorse(final AgonizingRemorse card) {
36          super(card);
37      }
38  
39      @Override
40      public AgonizingRemorse copy() {
41          return new AgonizingRemorse(this);
42      }
43  }

45  class AgonizingRemorseEffect extends OneShotEffect {
46  
47      AgonizingRemorseEffect() {
48          super(Outcome.Benefit);
49          staticText = "Target opponent reveals their hand. You choose a nonland card from it "
50                  + "or a card from their graveyard. Exile that card. You lose 1 life.";
51      }
52  
53      private AgonizingRemorseEffect(final AgonizingRemorseEffect effect) {
54          super(effect);
55      }
56  
57      @Override
58      public AgonizingRemorseEffect copy() {
59          return new AgonizingRemorseEffect(this);
60      }
61  
62      @Override
63      public boolean apply(Game game, Ability source) {
64          Player controller = game.getPlayer(source.getControllerId());
65          Player opponent = game.getPlayer(source.getFirstTarget());
66          if (controller == null || opponent == null) {
67              return false;
68          }
69          opponent.revealCards(source, opponent.getHand(), game);
70          TargetCard target;
71          Cards cards;
72          if (controller.chooseUse(outcome, "Exile a card from hand or graveyard?", 
73                  null, "Hand", "Graveyard", source, game)) {
74              target = new TargetCard(Zone.HAND, new FilterNonlandCard("nonland card in " 
75                      + opponent.getName() + "'s hand"));
76              target.setNotTarget(true);
77              cards = opponent.getHand();
78          } else {
79              target = new TargetCard(Zone.GRAVEYARD, new FilterCard("card in " 
80                      + opponent.getName() + "'s graveyard"));
81              target.setNotTarget(true);
82              cards = opponent.getGraveyard();
83          }
84          if (controller.choose(outcome, cards, target, game)) {
85              Card card = game.getCard(target.getFirstTarget());
86              if (card != null) {
87                  controller.moveCards(card, Zone.EXILED, source, game);
88              }
89          }
90          controller.loseLife(1, game, source, false);
91          return true;
92      }
93  }

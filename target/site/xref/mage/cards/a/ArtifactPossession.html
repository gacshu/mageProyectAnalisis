22  /**
23   *
24   * @author MarcoMarin
25   */
26  public final class ArtifactPossession extends CardImpl {
27  
28      public ArtifactPossession(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{2}{B}");
30          this.subtype.add(SubType.AURA);
31  
32          // Enchant artifact
33          TargetPermanent auraTarget = new TargetArtifactPermanent();
34          this.getSpellAbility().addTarget(auraTarget);
35          this.getSpellAbility().addEffect(new AttachEffect(Outcome.AddAbility));
36          Ability ability = new EnchantAbility(auraTarget);
37          this.addAbility(ability);
38          // Whenever enchanted artifact becomes tapped or a player activates an ability of enchanted artifact without {tap} in its activation cost, Artifact Possession deals 2 damage to that artifact's controller.
39          this.addAbility(new AbilityActivatedTriggeredAbility());
40  
41      }
42  
43      private ArtifactPossession(final ArtifactPossession card) {
44          super(card);
45      }
46  
47      @Override
48      public ArtifactPossession copy() {
49          return new ArtifactPossession(this);
50      }
51  }
52  
53  class AbilityActivatedTriggeredAbility extends TriggeredAbilityImpl {
54  
55      AbilityActivatedTriggeredAbility() {
56          super(Zone.BATTLEFIELD, new DamageAttachedControllerEffect(2));
57      }
58  
59      AbilityActivatedTriggeredAbility(final AbilityActivatedTriggeredAbility ability) {
60          super(ability);
61      }
62  
63      @Override
64      public AbilityActivatedTriggeredAbility copy() {
65          return new AbilityActivatedTriggeredAbility(this);
66      }
67  
68      @Override
69      public boolean checkEventType(GameEvent event, Game game) {
70          return event.getType() == GameEvent.EventType.ACTIVATED_ABILITY ||
71                 event.getType() == GameEvent.EventType.TAPPED;
72      }
73  
74      @Override
75      public boolean checkTrigger(GameEvent event, Game game) {
76          UUID tappedPermanent = null;
77          if (event.getType() == GameEvent.EventType.ACTIVATE_ABILITY) {
78              tappedPermanent = event.getSourceId();
79          }
80          if (event.getType() == GameEvent.EventType.TAPPED) {
81              tappedPermanent = event.getTargetId();
82          }
83  
84          Permanent aura = game.getPermanent(this.getSourceId());
85          return aura != null && aura.isAttachedTo(tappedPermanent);
86      }
87  
88      @Override
89      public String getRule() {
90          return "Whenever enchanted artifact becomes tapped or a player activates an ability of enchanted artifact without {tap} in its activation cost, Artifact Possession deals 2 damage to that artifact's controller.";
91      }
92  }

37  public final class AlrundGodOfTheCosmos extends ModalDoubleFacesCard {

39      public AlrundGodOfTheCosmos(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo,
41                  new CardType[]{CardType.CREATURE}, new SubType[]{SubType.GOD}, "{3}{U}{U}",
42                  "Hakka, Whispering Raven", new CardType[]{CardType.CREATURE}, new SubType[]{SubType.BIRD}, "{1}{U}"
43          );
44  
45          // 1.
46          // Alrund, God of the Cosmos
47          // Legendary Creature - God
48          this.getLeftHalfCard().addSuperType(SuperType.LEGENDARY);
49          this.getLeftHalfCard().setPT(new MageInt(1), new MageInt(1));
50  
51          // Alrund gets +1/+1 for each card in your hand and each foretold card you own in exile.
52          Effect effect = new BoostSourceEffect(AlrundGodOfTheCosmosValue.instance, AlrundGodOfTheCosmosValue.instance, Duration.EndOfGame);
53          effect.setText("{this} gets +1/+1 for each card in your hand and each foretold card you own in exile.");
54          Ability ability = new SimpleStaticAbility(effect);
55          this.getLeftHalfCard().addAbility(ability);
56  
57          //  At the beginning of your end step, choose a card type, then reveal the top two cards of your library. Put all cards of the chosen type into your hand and the rest on the bottom of your library in any order.
58          Ability ability2 = new BeginningOfYourEndStepTriggeredAbility(new ChooseCardTypeEffect(Outcome.Neutral), false);
59          ability2.addEffect(new AlrundGodOfTheCosmosEffect());
60          this.getLeftHalfCard().addAbility(ability2);
61  
62          // 2.
63          // Hakka, Whispering Raven
64          // Legendary Creature — Bird 
65          this.getRightHalfCard().addSuperType(SuperType.LEGENDARY);
66          this.getRightHalfCard().setPT(new MageInt(2), new MageInt(3));
67  
68          // Flying
69          this.getRightHalfCard().addAbility(FlyingAbility.getInstance());
70  
71          // Whenever Hakka, Whispering Raven deals combat damage to a player, return it to its owner’s hand, then scry 2.
72          Ability ability3 = new DealsCombatDamageToAPlayerTriggeredAbility(new ReturnToHandSourceEffect().setText("return it to its owner's hand"), false);
73          ability3.addEffect(new ScryEffect(2).concatBy(", then"));
74          this.getRightHalfCard().addAbility(ability3);
75  
76      }

88  class AlrundGodOfTheCosmosEffect extends OneShotEffect {
89  
90      public AlrundGodOfTheCosmosEffect() {
91          super(Outcome.Neutral);
92          staticText = ", then reveal the top two cards of your library. Put all cards of the chosen type revealed this way into your hand and the rest on the bottom of your library in any order";
93      }
94  
95      public AlrundGodOfTheCosmosEffect(final AlrundGodOfTheCosmosEffect effect) {
96          super(effect);
97      }
98  
99      @Override
100     public AlrundGodOfTheCosmosEffect copy() {
101         return new AlrundGodOfTheCosmosEffect(this);
102     }
103 
104     @Override
105     public boolean apply(Game game, Ability source) {
106         Player controller = game.getPlayer(source.getControllerId());
107         String chosenCardType = (String) game.getState().getValue(source.getSourceId() + "_type");
108         Cards cardsToHand = new CardsImpl();
109         Cards cardsToBottomOfLibrary = new CardsImpl();
110         if (controller != null) {
111             Set<Card> twoCardsFromTop = controller.getLibrary().getTopCards(game, 2);
112             Cards cards = new CardsImpl();
113             cards.addAll(twoCardsFromTop);
114             controller.revealCards(source, cards, game);
115             for (Card card : cards.getCards(game)) {
116                 if (card.getCardType(game).toString().contains(chosenCardType)) {
117                     cardsToHand.add(card);
118                 } else {
119                     cardsToBottomOfLibrary.add(card);
120                 }
121             }
122             controller.moveCards(cardsToHand, Zone.HAND, source, game);
123             controller.putCardsOnBottomOfLibrary(cardsToBottomOfLibrary, game, source, true);
124             return true;
125         }
126         return false;
127     }
128 }

130 enum AlrundGodOfTheCosmosValue implements DynamicValue {
131     instance;
132 
133     @Override
134     public int calculate(Game game, Ability sourceAbility, Effect effect) {
135         Player controller = game.getPlayer(sourceAbility.getControllerId());
136         if (controller != null) {
137             Collection<ExileZone> exileZones = game.getState().getExile().getExileZones();
138             Cards cardsForetoldInExileZones = new CardsImpl();
139             FilterCard filter = new FilterCard();
140             filter.add(new OwnerIdPredicate(controller.getId()));
141             filter.add(new AbilityPredicate(ForetellAbility.class));
142             for (ExileZone exile : exileZones) {
143                 for (Card card : exile.getCards(filter, game)) {
144                     // verify that the card is actually Foretold
145                     UUID exileId = CardUtil.getExileZoneId(card.getId().toString() + "foretellAbility", game);
146                     if (exileId != null) {
147                         if (game.getState().getExile().getExileZone(exileId) != null) {
148                             cardsForetoldInExileZones.add(card);
149                         }
150                     }
151                 }
152             }
153             return controller.getHand().size() + cardsForetoldInExileZones.size();
154         }
155         return 0;
156     }
157 
158     @Override
159     public DynamicValue copy() {
160         return instance;
161     }
162 
163     @Override
164     public String getMessage() {
165         return "";
166     }

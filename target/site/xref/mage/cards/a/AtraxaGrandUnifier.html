28      public AtraxaGrandUnifier(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{G}{W}{U}{B}");
30  
31          this.addSuperType(SuperType.LEGENDARY);
32          this.subtype.add(SubType.PHYREXIAN);
33          this.subtype.add(SubType.ANGEL);
34          this.power = new MageInt(7);
35          this.toughness = new MageInt(7);
36  
37          // Flying
38          this.addAbility(FlyingAbility.getInstance());
39  
40          // Vigilance
41          this.addAbility(VigilanceAbility.getInstance());
42  
43          // Deathtouch
44          this.addAbility(DeathtouchAbility.getInstance());
45  
46          // Lifelink
47          this.addAbility(LifelinkAbility.getInstance());
48  
49          // When Atraxa, Grand Unifier enters the battlefield, reveal the top ten cards of your library. For each card type, you may put a card of that type from among the revealed cards into your hand. Put the rest on the bottom of your library in a random order.
50          this.addAbility(new EntersBattlefieldTriggeredAbility(new AtraxaGrandUnifierEffect()));
51      }

63  class AtraxaGrandUnifierEffect extends OneShotEffect {
64  
65      AtraxaGrandUnifierEffect() {
66          super(Outcome.Benefit);
67          staticText = "reveal the top ten cards of your library. For each card type, " +
68                  "you may put a card of that type from among the revealed cards into your hand. " +
69                  "Put the rest on the bottom of your library in a random order";
70      }
71  
72      private AtraxaGrandUnifierEffect(final AtraxaGrandUnifierEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public AtraxaGrandUnifierEffect copy() {
78          return new AtraxaGrandUnifierEffect(this);
79      }
80  
81      @Override
82      public boolean apply(Game game, Ability source) {
83          Player player = game.getPlayer(source.getControllerId());
84          if (player == null) {
85              return false;
86          }
87          Cards cards = new CardsImpl(player.getLibrary().getTopCards(game, 10));
88          player.revealCards(source, cards, game);
89          TargetCard target = new AtraxaGrandUnifierTarget();
90          player.choose(outcome, cards, target, game);
91          Cards toHand = new CardsImpl(target.getTargets());
92          player.moveCardsToHandWithInfo(toHand, source, game, true);
93          cards.retainZone(Zone.LIBRARY, game);
94          player.putCardsOnBottomOfLibrary(cards, game, source, false);
95          return true;
96      }
97  }

99  class AtraxaGrandUnifierTarget extends TargetCardInLibrary {
100 
101     private static final FilterCard filter
102             = new FilterCard("a card of each card type");
103 
104     private static final CardTypeAssignment cardTypeAssigner
105             = new CardTypeAssignment(Arrays.stream(CardType.values()).toArray(CardType[]::new));
106 
107     AtraxaGrandUnifierTarget() {
108         super(0, Integer.MAX_VALUE, filter);
109     }
110 
111     private AtraxaGrandUnifierTarget(final AtraxaGrandUnifierTarget target) {
112         super(target);
113     }
114 
115     @Override
116     public AtraxaGrandUnifierTarget copy() {
117         return new AtraxaGrandUnifierTarget(this);
118     }
119 
120     @Override
121     public boolean canTarget(UUID playerId, UUID id, Ability source, Game game) {
122         if (!super.canTarget(playerId, id, source, game)) {
123             return false;
124         }
125         Card card = game.getCard(id);
126         if (card == null) {
127             return false;
128         }
129         if (this.getTargets().isEmpty()) {
130             return true;
131         }
132         Cards cards = new CardsImpl(this.getTargets());
133         cards.add(card);
134         return cardTypeAssigner.getRoleCount(cards, game) >= cards.size();
135     }

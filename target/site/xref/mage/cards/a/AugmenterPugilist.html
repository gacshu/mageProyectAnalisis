27  public final class AugmenterPugilist extends ModalDoubleFacesCard {
28  
29      public AugmenterPugilist(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo,
31                  new CardType[]{CardType.CREATURE}, new SubType[]{SubType.TROLL, SubType.DRUID}, "{1}{G}{G}",
32                  "Echoing Equation", new CardType[]{CardType.SORCERY}, new SubType[]{}, "{3}{U}{U}");
33          
34          // 1.
35          // Augmenter Pugilist
36          // Creature — Troll Druid
37          this.getLeftHalfCard().setPT(3, 3);
38  
39          // Trample
40          this.getLeftHalfCard().addAbility(TrampleAbility.getInstance());
41  
42          // As long as you control eight or more lands, Augmenter Pugilist gets +5/+5.
43          this.getLeftHalfCard().addAbility(new SimpleStaticAbility(
44              Zone.BATTLEFIELD,
45              new ConditionalContinuousEffect(
46                      new BoostSourceEffect(
47                              5, 5, Duration.WhileOnBattlefield
48                      ),
49                      new PermanentsOnTheBattlefieldCondition(
50                              StaticFilters.FILTER_CONTROLLED_PERMANENT_LAND,
51                              ComparisonType.MORE_THAN, 7
52                      ),
53                      "as long as you control eight or more lands, {this} gets +5/+5"
54              )
55          ).addHint(LandsYouControlHint.instance));
56  
57          // 2.
58          // Echoing Equation
59          // Sorcery
60  
61          // Choose target creature you control. Each other creature you control becomes a copy of it until end of turn, except those creatures aren’t legendary if the chosen creature is legendary.
62          this.getRightHalfCard().getSpellAbility().addEffect(new EchoingEquationEffect());
63          this.getRightHalfCard().getSpellAbility().addTarget(new TargetControlledCreaturePermanent());
64  
65      }
66

68          super(card);
69      }
70

72      public AugmenterPugilist copy() {
73          return new AugmenterPugilist(this);
74      }
75  }

77  class EchoingEquationEffect extends OneShotEffect {
78  
79      public EchoingEquationEffect() {
80          super(Outcome.Benefit);
81          staticText = "choose target creature you control. Each other creature you control becomes a copy of it until end of turn, except those creatures aren't legendary if the chosen creature is legendary";
82      }
83  
84      EchoingEquationEffect(EchoingEquationEffect effect) {
85          super(effect);
86      }
87  
88      @Override
89      public EchoingEquationEffect copy() {
90          return new EchoingEquationEffect(this);
91      }
92  
93      @Override
94      public boolean apply(Game game, Ability source) {
95          Permanent copyFrom = game.getPermanent(source.getFirstTarget());
96          if (copyFrom != null) {
97              game.getBattlefield().getAllActivePermanents(source.getControllerId()).stream()
98                  .filter(permanent -> permanent.isCreature(game) && !permanent.getId().equals(copyFrom.getId()))
99                  .forEach(copyTo -> game.copyPermanent(Duration.EndOfTurn, copyFrom, copyTo.getId(), source, new CopyApplier() {
100                     @Override
101                     public boolean apply(Game game, MageObject blueprint, Ability source, UUID targetObjectId) {
102                         blueprint.getSuperType().remove(SuperType.LEGENDARY);
103                         return true;
104                     }
105                 }));
106             return true;
107         }
108         return false;
109     }

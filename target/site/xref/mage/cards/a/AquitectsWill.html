24  public final class AquitectsWill extends CardImpl {
25  
26      private static final FilterControlledPermanent filter = new FilterControlledPermanent(SubType.MERFOLK);
27      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter);
28  
29      public AquitectsWill(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.TRIBAL, CardType.SORCERY}, "{U}");
31          this.subtype.add(SubType.MERFOLK);
32  
33          // Put a flood counter on target land.
34          this.getSpellAbility().addEffect(new AddCountersTargetEffect(CounterType.FLOOD.createInstance()));
35          this.getSpellAbility().addTarget(new TargetLandPermanent());
36  
37          // That land is an Island in addition to its other types for as long as it has a flood counter on it.
38          this.getSpellAbility().addEffect(new AquitectsWillEffect());
39  
40          // If you control a Merfolk, draw a card.
41          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
42                  new DrawCardSourceControllerEffect(1), condition,
43                  "If you control a Merfolk, draw a card"
44          ));
45      }
46  
47      private AquitectsWill(final AquitectsWill card) {
48          super(card);
49      }
50  
51      @Override
52      public AquitectsWill copy() {
53          return new AquitectsWill(this);
54      }
55  }
56  
57  class AquitectsWillEffect extends BecomesBasicLandTargetEffect {
58  
59      AquitectsWillEffect() {
60          super(Duration.Custom, false, false, SubType.ISLAND);
61          staticText = "That land is an Island in addition to its other types for as long as it has a flood counter on it";
62      }
63  
64      private AquitectsWillEffect(final AquitectsWillEffect effect) {
65          super(effect);
66      }
67  
68      @Override
69      public AquitectsWillEffect copy() {
70          return new AquitectsWillEffect(this);
71      }
72  
73      @Override
74      public boolean apply(Game game, Ability source) {
75          Permanent land = game.getPermanent(this.targetPointer.getFirst(game, source));
76          if (land == null || land.getCounters(game).getCount(CounterType.FLOOD) < 1) {
77              discard();
78              return false;
79          }
80          return super.apply(game, source);
81      }
82  }

5   import java.util.UUID;
6   import mage.MageInt;
7   import mage.abilities.common.EntersBattlefieldAbility;
8   import mage.abilities.common.SimpleActivatedAbility;
9   import mage.abilities.condition.common.KickedCostCondition;
10  import mage.abilities.costs.common.PayLifeCost;
11  import mage.abilities.effects.EntersBattlefieldEffect;
12  import mage.abilities.effects.common.RegenerateSourceEffect;
13  import mage.abilities.effects.common.continuous.GainAbilitySourceEffect;
14  import mage.abilities.effects.common.counter.AddCountersSourceEffect;
15  import mage.abilities.keyword.FlyingAbility;
16  import mage.abilities.keyword.KickerAbility;
17  import mage.cards.CardImpl;
18  import mage.cards.CardSetInfo;
19  import mage.constants.CardType;
20  import mage.constants.Duration;
21  import mage.constants.SubType;
22  import mage.constants.Zone;
23  import mage.counters.CounterType;
24  
25  /**
26   * @author Loki
27   */
28  public final class Anavolver extends CardImpl {
29  
30      public Anavolver(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{G}");
32          this.subtype.add(SubType.VOLVER);
33  
34          this.power = new MageInt(3);
35          this.toughness = new MageInt(3);
36          
37          // Kicker {1}{U} and/or {B} (You may pay an additional {1}{U} and/or {B} as you cast this spell.)
38          KickerAbility kickerAbility = new KickerAbility("{1}{U}");
39          kickerAbility.addKickerCost("{B}");
40          this.addAbility(kickerAbility);
41  
42          // If Anavolver was kicked with its {1}{U} kicker, it enters the battlefield with two +1/+1 counters on it and with flying.
43          EntersBattlefieldAbility ability1 = new EntersBattlefieldAbility(
44                  new AddCountersSourceEffect(CounterType.P1P1.createInstance(2),false),
45                  new KickedCostCondition("{1}{U}"), "If {this} was kicked with its {1}{U} kicker, it enters the battlefield with two +1/+1 counters on it and with flying.",
46                  "{this} enters the battlefield with two +1/+1 counters on it and with flying");
47          ((EntersBattlefieldEffect)ability1.getEffects().get(0)).addEffect(new GainAbilitySourceEffect(FlyingAbility.getInstance(), Duration.WhileOnBattlefield));
48          this.addAbility(ability1);
49  
50          // If Anavolver was kicked with its {B} kicker, it enters the battlefield with a +1/+1 counter on it and with "Pay 3 life: Regenerate Anavolver."
51          EntersBattlefieldAbility ability2 = new EntersBattlefieldAbility(
52                  new AddCountersSourceEffect(CounterType.P1P1.createInstance(1),false), new KickedCostCondition("{B}"),
53                  "If {this} was kicked with its {B} kicker, it enters the battlefield with a +1/+1 counter on it and with \"Pay 3 life: Regenerate Anavolver.\"",
54                  "{this} enters the battlefield with a +1/+1 counter on it and with \"Pay 3 life: Regenerate Anavolver.\"");
55          ((EntersBattlefieldEffect)ability2.getEffects().get(0)).addEffect(new GainAbilitySourceEffect(new SimpleActivatedAbility(Zone.BATTLEFIELD, new RegenerateSourceEffect(), new PayLifeCost(3)), Duration.WhileOnBattlefield));
56          this.addAbility(ability2);
57      }
58  
59      private Anavolver(final Anavolver card) {
60          super(card);
61      }
62  
63      @Override
64      public Anavolver copy() {
65          return new Anavolver(this);
66      }
67  }

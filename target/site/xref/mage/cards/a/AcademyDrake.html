17  public final class AcademyDrake extends CardImpl {
18  
19      public AcademyDrake(UUID ownerId, CardSetInfo cardSetInfo) {
20          super(ownerId, cardSetInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
21          subtype.add(SubType.DRAKE);
22          power = new MageInt(2);
23          toughness = new MageInt(2);
24  
25          // Kicker {4}
26          this.addAbility(new KickerAbility("{4}"));
27  
28          // Flying
29          addAbility(FlyingAbility.getInstance());
30  
31          // If Academy Drake was kicked, it enters the battlefield with two +1/+1 counters on it.
32          this.addAbility(new EntersBattlefieldAbility(new AddCountersSourceEffect(CounterType.P1P1.createInstance(2)),
33                  KickedCondition.ONCE, "If {this} was kicked, it enters the battlefield with two +1/+1 counters on it.", ""));
34  
35      }
36  
37      public AcademyDrake(final AcademyDrake academyDrake) {
38          super(academyDrake);
39      }
40  
41      public AcademyDrake copy() {
42          return new AcademyDrake(this);
43      }
44  }

28  public final class AgadeemsAwakening extends ModalDoubleFacesCard {
29  
30      public AgadeemsAwakening(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo,
32                  new CardType[]{CardType.SORCERY}, new SubType[]{}, "{X}{B}{B}{B}",
33                  "Agadeem, the Undercrypt", new CardType[]{CardType.LAND}, new SubType[]{}, ""
34          );
35  
36          // 1.
37          // Agadeem's Awakening
38          // Sorcery
39  
40          // Return from your graveyard to the battlefield any number of target creature cards that each have a different converted mana cost X or less.
41          this.getLeftHalfCard().getSpellAbility().addEffect(new ReturnFromGraveyardToBattlefieldTargetEffect().setText(
42                  "return from your graveyard to the battlefield any number of target creature cards " +
43                          "that each have a different mana value X or less"
44          ));
45          this.getLeftHalfCard().getSpellAbility().setTargetAdjuster(AgadeemsAwakeningAdjuster.instance);
46  
47          // 2.
48          // Agadeem, the Undercrypt
49          // Land
50  
51          // As Agadeem, the Undercrypt enters the battlefield, you may pay 3 life. If you don't, it enters the battlefield tapped.
52          this.getRightHalfCard().addAbility(new AsEntersBattlefieldAbility(
53                  new TapSourceUnlessPaysEffect(new PayLifeCost(3)),
54                  "you may pay 3 life. If you don't, it enters the battlefield tapped"
55          ));
56  
57          // {T}: Add {B}.
58          this.getRightHalfCard().addAbility(new BlackManaAbility());
59      }
60  
61      private AgadeemsAwakening(final AgadeemsAwakening card) {
62          super(card);
63      }
64  
65      @Override
66      public AgadeemsAwakening copy() {
67          return new AgadeemsAwakening(this);
68      }
69  }
70  
71  enum AgadeemsAwakeningAdjuster implements TargetAdjuster {
72      instance;
73  
74      @Override
75      public void adjustTargets(Ability ability, Game game) {
76          ability.getTargets().clear();
77          ability.addTarget(new AgadeemsAwakeningTarget(ability.getManaCostsToPay().getX()));
78      }
79  }
80  
81  class AgadeemsAwakeningTarget extends TargetCardInYourGraveyard {
82  
83      private static final FilterCard filter
84              = new FilterCreatureCard("creature cards that each have a different mana value X or less");
85      private final int xValue;
86  
87      AgadeemsAwakeningTarget(int xValue) {
88          super(0, Integer.MAX_VALUE, filter, true);
89          this.xValue = xValue;
90      }
91  
92      private AgadeemsAwakeningTarget(final AgadeemsAwakeningTarget target) {
93          super(target);
94          this.xValue = target.xValue;
95      }
96  
97      @Override
98      public AgadeemsAwakeningTarget copy() {
99          return new AgadeemsAwakeningTarget(this);
100     }
101 
102     @Override
103     public Set<UUID> possibleTargets(UUID sourceControllerId, Ability source, Game game) {
104         Set<UUID> possibleTargets = super.possibleTargets(sourceControllerId, source, game);
105         Set<Integer> cmcs = this.getTargets()
106                 .stream()
107                 .map(game::getCard)
108                 .map(MageObject::getManaValue)
109                 .collect(Collectors.toSet());
110         possibleTargets.removeIf(uuid -> {
111             Card card = game.getCard(uuid);
112             return card != null && (cmcs.contains(card.getManaValue()) || card.getManaValue() > xValue);
113         });
114         return possibleTargets;
115     }
116 }

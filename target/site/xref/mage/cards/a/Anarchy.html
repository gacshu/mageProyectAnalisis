2   package mage.cards.a;

4   import java.util.UUID;
5   import mage.ObjectColor;
6   import mage.abilities.effects.common.DestroyAllEffect;
7   import mage.cards.CardImpl;
8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.filter.FilterPermanent;
11  import mage.filter.predicate.mageobject.ColorPredicate;
12  
13  /**
14   *
15   * @author LevelX2
16   */
17  public final class Anarchy extends CardImpl {
18  
19      private static final FilterPermanent filter = new FilterPermanent("white permanents");
20  
21      static {
22          filter.add(new ColorPredicate(ObjectColor.WHITE));
23      }
24  
25      public Anarchy(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{2}{R}{R}");
27  
28  
29          // Destroy all white permanents.
30          this.getSpellAbility().addEffect(new DestroyAllEffect(filter));
31      }
32  
33      private Anarchy(final Anarchy card) {
34          super(card);
35      }
36  
37      @Override
38      public Anarchy copy() {
39          return new Anarchy(this);
40      }
41  }

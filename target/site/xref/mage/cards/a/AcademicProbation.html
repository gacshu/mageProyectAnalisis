25  public final class AcademicProbation extends CardImpl {
26  
27      public AcademicProbation(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{W}");
29          
30          this.subtype.add(SubType.LESSON);
31  
32          // Choose one —
33          // • Choose a nonland card name. Opponents can't cast spells with the chosen name until your next turn.
34          Effect effect = new ChooseACardNameEffect(ChooseACardNameEffect.TypeOfName.NON_LAND_NAME);
35          effect.setText("Choose a nonland card name");
36          this.getSpellAbility().addEffect(effect);
37          this.getSpellAbility().addEffect(new OpponentsCantCastChosenUntilNextTurnEffect().setText("opponents can't cast spells with the chosen name until your next turn"));
38  
39          // • Choose target nonland permanent. Until your next turn, it can't attack or block, and its activated abilities can't be activated.
40          Mode restrictMode = new Mode(new AcademicProbationRestrictionEffect());
41          restrictMode.addTarget(new TargetNonlandPermanent());
42          this.getSpellAbility().addMode(restrictMode);
43      }
44  
45      private AcademicProbation(final AcademicProbation card) {
46          super(card);
47      }
48  
49      @Override
50      public AcademicProbation copy() {
51          return new AcademicProbation(this);
52      }
53  }

55  class AcademicProbationRestrictionEffect extends RestrictionEffect {
56  
57      AcademicProbationRestrictionEffect() {
58          super(Duration.UntilYourNextTurn, Outcome.UnboostCreature);
59          staticText = "choose target nonland permanent. Until your next turn, it can't attack or block, and its activated abilities can't be activated";
60      }
61  
62      AcademicProbationRestrictionEffect(final AcademicProbationRestrictionEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public AcademicProbationRestrictionEffect copy() {
68          return new AcademicProbationRestrictionEffect(this);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          return true;
74      }
75  
76      @Override
77      public boolean applies(Permanent permanent, Ability source, Game game) {
78          return this.targetPointer.getTargets(game, source).contains(permanent.getId());
79      }
80  
81      @Override
82      public boolean canAttack(Game game, boolean canUseChooseDialogs) {
83          return false;
84      }
85  
86      @Override
87      public boolean canBlock(Permanent attacker, Permanent blocker, Ability source, Game game, boolean canUseChooseDialogs) {
88          return false;
89      }
90  
91      @Override
92      public boolean canUseActivatedAbilities(Permanent permanent, Ability source, Game game, boolean canUseChooseDialogs) {
93          return false;
94      }

5   import mage.abilities.condition.common.CardsInOpponentGraveyardCondition;
6   import mage.abilities.costs.mana.GenericManaCost;
7   import mage.abilities.decorator.ConditionalOneShotEffect;
8   import mage.abilities.effects.common.CounterTargetEffect;
9   import mage.abilities.effects.common.CounterUnlessPaysEffect;
10  import mage.abilities.effects.keyword.ScryEffect;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.filter.FilterSpell;
15  import mage.filter.predicate.Predicates;
16  import mage.target.TargetSpell;
17  
18  /**
19   * @author TheElk801
20   */
21  public final class Anticognition extends CardImpl {
22  
23      private static final FilterSpell filter = new FilterSpell("creature or planeswalker spell");
24  
25      static {
26          filter.add(Predicates.or(
27                  CardType.CREATURE.getPredicate(),
28                  CardType.PLANESWALKER.getPredicate()
29          ));
30      }
31  
32      public Anticognition(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{U}");
34  
35          // Counter target creature or planeswalker spell unless its controller pays {2}. If an opponent has eight or more cards in their graveyard, instead counter that spell, then scry 2.
36          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
37                  new CounterTargetEffect(), new CounterUnlessPaysEffect(new GenericManaCost(2)),
38                  CardsInOpponentGraveyardCondition.EIGHT, "Counter target creature or planeswalker spell " +
39                  "unless its controller pays {2}. If an opponent has eight or more cards in their graveyard, " +
40                  "instead counter that spell"
41          ));
42          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
43                  new ScryEffect(2), CardsInOpponentGraveyardCondition.EIGHT, ", then scry 2"
44          ));
45          this.getSpellAbility().addTarget(new TargetSpell(filter));
46          this.getSpellAbility().addHint(CardsInOpponentGraveyardCondition.EIGHT.getHint());
47      }
48  
49      private Anticognition(final Anticognition card) {
50          super(card);
51      }
52  
53      @Override
54      public Anticognition copy() {
55          return new Anticognition(this);
56      }
57  }

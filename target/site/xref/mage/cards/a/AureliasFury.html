32  /**
33   * GATECRASH FAQ 11.01.2013
34   * <p>
35   * You announce the value of X and how the damage will be divided as part of
36   * casting Aurelia's Fury. Each chosen target must receive at least 1 damage.
37   * <p>
38   * Aurelia's Fury can't deal damage to both a planeswalker and that
39   * planeswalker's controller. If damage dealt by Aurelia's Fury is redirected
40   * from a player to a planeswalker they control, that player will be able
41   * to cast noncreature spells that turn. If you want to stop a player from
42   * casting noncreature spells this turn, you can't choose to redirect the
43   * damage to a planeswalker they control.
44   * <p>
45   * If Aurelia's Fury has multiple targets, and some but not all of them are
46   * illegal targets when Aurelia's Fury resolves, Aurelia's Fury will still
47   * deal damage to the remaining legal targets according to the original damage
48   * division.
49   * <p>
50   * If all of the targets are illegal when Aurelia's Fury tries to resolve,
51   * it will be countered and none of its effects will happen. No creature or
52   * player will be dealt damage.
53   *
54   * @author LevelX2
55   */
56  public final class AureliasFury extends CardImpl {
57  
58      public AureliasFury(UUID ownerId, CardSetInfo setInfo) {
59          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{X}{R}{W}");
60  
61  
62          // Aurelia's Fury deals X damage divided as you choose among any number of target creatures and/or players.
63          // Tap each creature dealt damage this way. Players dealt damage this way can't cast noncreature spells this turn.
64          DynamicValue xValue = ManacostVariableValue.REGULAR;
65          this.getSpellAbility().addEffect(new DamageMultiEffect(xValue));
66          this.getSpellAbility().addEffect(new AureliasFuryEffect());
67          this.getSpellAbility().addTarget(new TargetAnyTargetAmount(xValue));
68          this.getSpellAbility().addWatcher(new AureliasFuryDamagedByWatcher());
69  
70      }
71  
72      private AureliasFury(final AureliasFury card) {
73          super(card);
74      }
75  
76      @Override
77      public AureliasFury copy() {
78          return new AureliasFury(this);
79      }
80  }

82  class AureliasFuryEffect extends OneShotEffect {
83  
84      AureliasFuryEffect() {
85          super(Outcome.Benefit);
86          this.staticText = "Tap each creature dealt damage this way. Players dealt damage this way can't cast noncreature spells this turn";
87      }
88  
89      private AureliasFuryEffect(final AureliasFuryEffect effect) {
90          super(effect);
91      }
92  
93      @Override
94      public AureliasFuryEffect copy() {
95          return new AureliasFuryEffect(this);
96      }
97  
98      @Override
99      public boolean apply(Game game, Ability source) {
100         AureliasFuryDamagedByWatcher watcher = game.getState().getWatcher(AureliasFuryDamagedByWatcher.class, source.getSourceId());
101         if (watcher != null) {
102             for (UUID creatureId : watcher.getDamagedCreatures()) {
103                 Permanent permanent = game.getPermanent(creatureId);
104                 if (permanent != null) {
105                     permanent.tap(source, game);
106                 }
107             }
108             for (UUID playerId : watcher.getDamagedPlayers()) {
109                 ContinuousEffect effect = new AureliasFuryCantCastEffect();
110                 effect.setTargetPointer(new FixedTarget(playerId));
111                 game.addEffect(effect, source);
112             }
113             watcher.reset();
114         }
115 
116         return false;
117     }
118 }

120 class AureliasFuryCantCastEffect extends ContinuousRuleModifyingEffectImpl {
121 
122     AureliasFuryCantCastEffect() {
123         super(Duration.EndOfTurn, Outcome.Benefit);
124         staticText = "Players dealt damage this way can't cast noncreature spells this turn";
125     }
126 
127     private AureliasFuryCantCastEffect(final AureliasFuryCantCastEffect effect) {
128         super(effect);
129     }
130 
131     @Override
132     public AureliasFuryCantCastEffect copy() {
133         return new AureliasFuryCantCastEffect(this);
134     }
135 
136     @Override
137     public boolean apply(Game game, Ability source) {
138         return true;
139     }
140 
141     @Override
142     public String getInfoMessage(Ability source, GameEvent event, Game game) {
143         MageObject mageObject = game.getObject(source);
144         if (mageObject != null) {
145             return "You can't cast noncreature spells this turn (you were dealt damage by " + mageObject.getLogName() + ')';
146         }
147         return null;
148     }
149 
150     @Override
151     public boolean checksEventType(GameEvent event, Game game) {
152         return event.getType() == GameEvent.EventType.CAST_SPELL;
153     }
154 
155     @Override
156     public boolean applies(GameEvent event, Ability source, Game game) {
157         Player player = game.getPlayer(getTargetPointer().getFirst(game, source));
158         if (player != null && player.getId().equals(event.getPlayerId())) {
159             Card card = game.getCard(event.getSourceId());
160             if (card != null && !card.isCreature(game)) {
161                 return true;
162             }
163         }
164         return false;
165     }
166 }

168 class AureliasFuryDamagedByWatcher extends Watcher {
169 
170     private final Set<UUID> damagedCreatures = new HashSet<>();
171     private final Set<UUID> damagedPlayers = new HashSet<>();
172 
173     AureliasFuryDamagedByWatcher() {
174         super(WatcherScope.CARD);
175     }
176 
177     @Override
178     public void watch(GameEvent event, Game game) {
179         MageObject obj = game.getObject(event.getSourceId());
180         if (!(obj instanceof Spell) || !sourceId.equals(((Spell) obj).getSourceId())) {
181             return;
182         }
183         switch (event.getType()) {
184             case DAMAGED_PERMANENT:
185                 Permanent permanent = game.getPermanent(event.getTargetId());
186                 if (permanent != null && permanent.isCreature(game)) {
187                     damagedCreatures.add(event.getTargetId());
188                 }
189                 return;
190             case DAMAGED_PLAYER:
191                 damagedPlayers.add(event.getTargetId());
192         }
193     }
194 
195     @Override
196     public void reset() {
197         super.reset();
198         damagedCreatures.clear();
199         damagedPlayers.clear();
200     }
201 
202     Set<UUID> getDamagedCreatures() {
203         return damagedCreatures;
204     }
205 
206     Set<UUID> getDamagedPlayers() {
207         return damagedPlayers;
208     }
209 }

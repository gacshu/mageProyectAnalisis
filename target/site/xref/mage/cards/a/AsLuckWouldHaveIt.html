25  public final class AsLuckWouldHaveIt extends CardImpl {
26  
27      static final String rule = "put a number of luck counters on {this} equal to the result. Then if there are 100 or more luck counters on {this}, you win the game.";
28  
29      public AsLuckWouldHaveIt(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{G}");
31  
32          // Hexproof
33          this.addAbility(HexproofAbility.getInstance());
34  
35          // Whenever you roll a die, put a number of luck counters on As Luck Would Have It equal to the result. Then if there are 100 or more luck counters on As Luck Would Have It, you win the game.
36          this.addAbility(new AsLuckWouldHaveItTriggeredAbility());
37      }
38

40          super(card);
41      }

49  class AsLuckWouldHaveItTriggeredAbility extends TriggeredAbilityImpl {
50  
51      public AsLuckWouldHaveItTriggeredAbility() {
52          super(Zone.BATTLEFIELD, new AsLuckWouldHaveItEffect(), false);
53          setTriggerPhrase("Whenever you roll a die, ");
54      }
55  
56      public AsLuckWouldHaveItTriggeredAbility(final AsLuckWouldHaveItTriggeredAbility ability) {
57          super(ability);
58      }
59  
60      @Override
61      public AsLuckWouldHaveItTriggeredAbility copy() {
62          return new AsLuckWouldHaveItTriggeredAbility(this);
63      }
64  
65      @Override
66      public boolean checkEventType(GameEvent event, Game game) {
67          return event.getType() == GameEvent.EventType.DIE_ROLLED;
68      }
69  
70      @Override
71      public boolean checkTrigger(GameEvent event, Game game) {
72          DieRolledEvent drEvent = (DieRolledEvent) event;
73          // Any die roll with a numerical result will add luck counters to As Luck Would Have It.
74          // Rolling the planar die will not cause the second ability to trigger.
75          // (2018-01-19)
76          if (this.isControlledBy(event.getPlayerId()) && drEvent.getRollDieType() == RollDieType.NUMERICAL) {
77              // silver border card must look for "result" instead "natural result"
78              this.getEffects().setValue("rolled", drEvent.getResult());
79              return true;
80          }
81          return false;
82      }
83  }

85  class AsLuckWouldHaveItEffect extends OneShotEffect {
86  
87      public AsLuckWouldHaveItEffect() {
88          super(Outcome.Benefit);
89          this.staticText = "put a number of luck counters on {this} equal to the result. Then if there are 100 or more luck counters on {this}, you win the game.";
90      }
91  
92      public AsLuckWouldHaveItEffect(final AsLuckWouldHaveItEffect effect) {
93          super(effect);
94      }
95  
96      @Override
97      public AsLuckWouldHaveItEffect copy() {
98          return new AsLuckWouldHaveItEffect(this);
99      }
100 
101     @Override
102     public boolean apply(Game game, Ability source) {
103         Player controller = game.getPlayer(source.getControllerId());
104         Permanent permanent = game.getPermanent(source.getSourceId());
105         if (controller != null && permanent != null) {
106             if (getValue("rolled") != null) {
107                 int amount = (Integer) getValue("rolled");
108                 permanent.addCounters(new Counter("luck", amount), source.getControllerId(), source, game);
109 
110                 if (permanent.getCounters(game).getCount("luck") >= 100) {
111                     Player player = game.getPlayer(permanent.getControllerId());
112                     if (player != null) {
113                         player.won(game);
114                     }
115                 }
116 
117                 return true;
118             }
119         }
120         return false;
121 
122     }

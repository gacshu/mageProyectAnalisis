28  public final class AkroanHorse extends CardImpl {

30      public AkroanHorse(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{4}");
32          this.subtype.add(SubType.HORSE);
33          this.power = new MageInt(0);
34          this.toughness = new MageInt(4);
35  
36          // Defender
37          this.addAbility(DefenderAbility.getInstance());
38  
39          // When Akroan Horse enters the battlefield, an opponent gains control of it.
40          this.addAbility(new EntersBattlefieldTriggeredAbility(
41                  new AkroanHorseChangeControlEffect(), false
42          ));
43  
44          // At the beginning of your upkeep, each opponent create a 1/1 white Soldier creature token.
45          this.addAbility(new BeginningOfUpkeepTriggeredAbility(
46                  new CreateTokenAllEffect(new SoldierToken(), TargetController.OPPONENT),
47                  TargetController.YOU, false
48          ));
49      }
50  
51      private AkroanHorse(final AkroanHorse card) {
52          super(card);
53      }
54

56      public AkroanHorse copy() {
57          return new AkroanHorse(this);
58      }
59  }

61  class AkroanHorseChangeControlEffect extends OneShotEffect {
62  
63      public AkroanHorseChangeControlEffect() {
64          super(Outcome.Benefit);
65          this.staticText = "an opponent gains control of it";
66      }
67  
68      private AkroanHorseChangeControlEffect(final AkroanHorseChangeControlEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public AkroanHorseChangeControlEffect copy() {
74          return new AkroanHorseChangeControlEffect(this);
75      }
76  
77      @Override
78      public boolean apply(Game game, Ability source) {
79          Player controller = game.getPlayer(source.getControllerId());
80          if (controller == null) {
81              return false;
82          }
83          Target target = new TargetOpponent();
84          target.setNotTarget(true);
85          controller.chooseTarget(outcome, target, source, game);
86          ContinuousEffect effect = new AkroanHorseGainControlEffect(Duration.Custom, target.getFirstTarget());
87          effect.setTargetPointer(new FixedTarget(source.getSourceId(), game));
88          game.addEffect(effect, source);
89          return true;
90      }
91  }

93  class AkroanHorseGainControlEffect extends ContinuousEffectImpl {
94  
95      private final UUID controller;
96  
97      public AkroanHorseGainControlEffect(Duration duration, UUID controller) {
98          super(duration, Layer.ControlChangingEffects_2, SubLayer.NA, Outcome.GainControl);
99          this.controller = controller;
100         this.staticText = "Gain control of Akroan Horse";
101     }
102 
103     public AkroanHorseGainControlEffect(final AkroanHorseGainControlEffect effect) {
104         super(effect);
105         this.controller = effect.controller;
106     }
107 
108     @Override
109     public AkroanHorseGainControlEffect copy() {
110         return new AkroanHorseGainControlEffect(this);
111     }
112 
113     @Override
114     public boolean apply(Game game, Ability source) {
115         Permanent permanent;
116         if (targetPointer == null) {
117             permanent = game.getPermanent(source.getFirstTarget());
118         } else {
119             permanent = game.getPermanent(targetPointer.getFirst(game, source));
120         }
121         if (permanent == null) {
122             return false;
123         }
124         return permanent.changeControllerId(controller, game, source);
125     }

5   import mage.abilities.LoyaltyAbility;

8   import mage.abilities.effects.common.DestroyAllControlledTargetEffect;
9   import mage.abilities.effects.common.DontUntapInControllersNextUntapStepTargetEffect;
10  import mage.abilities.effects.common.GainLifeEffect;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.SubType;
15  import mage.constants.SuperType;
16  import mage.filter.FilterPermanent;
17  import mage.target.TargetPermanent;
18  import mage.target.TargetPlayer;
19  import mage.target.common.TargetAnyTarget;

21  /**
22   *
23   * @author BetaSteward_at_googlemail.com
24   */
25  public final class AjaniVengeant extends CardImpl {
26  
27      private static final FilterPermanent filter = new FilterPermanent("lands");
28  
29      static {
30          filter.add(CardType.LAND.getPredicate());
31      }
32  
33      public AjaniVengeant(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.PLANESWALKER},"{2}{R}{W}");
35          this.addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.AJANI);
37  
38          this.setStartingLoyalty(3);
39  
40          // +1: Target permanent doesn't untap during its controller's next untap step.
41          LoyaltyAbility ability1 = new LoyaltyAbility(new DontUntapInControllersNextUntapStepTargetEffect(), 1);
42          ability1.addTarget(new TargetPermanent());
43          this.addAbility(ability1);
44  
45          // −2: Ajani Vengeant deals 3 damage to any target and you gain 3 life.
46          Effects effects1 = new Effects();
47          effects1.add(new DamageTargetEffect(3));
48          effects1.add(new GainLifeEffect(3).concatBy("and"));
49          LoyaltyAbility ability2 = new LoyaltyAbility(effects1, -2);
50          ability2.addTarget(new TargetAnyTarget());
51          this.addAbility(ability2);
52  
53          // −7: Destroy all lands target player controls.
54          LoyaltyAbility ability3 = new LoyaltyAbility(new DestroyAllControlledTargetEffect(filter), -7);
55          ability3.addTarget(new TargetPlayer());
56          this.addAbility(ability3);
57  
58      }
59  
60      private AjaniVengeant(final AjaniVengeant card) {
61          super(card);
62      }
63  
64      @Override
65      public AjaniVengeant copy() {
66          return new AjaniVengeant(this);
67      }
68  }

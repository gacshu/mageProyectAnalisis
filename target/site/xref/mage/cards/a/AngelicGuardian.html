27  public final class AngelicGuardian extends CardImpl {
28  
29      public AngelicGuardian(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{W}{W}");
31          this.subtype.add(SubType.ANGEL);
32  
33          this.power = new MageInt(5);
34          this.toughness = new MageInt(5);
35  
36          // Flying
37          this.addAbility(FlyingAbility.getInstance());
38  
39          // Whenever one or more creatures you control attack, they gain indestructible until end of turn
40          this.addAbility(new AttacksWithCreaturesTriggeredAbility(
41                  new AngelicGuardianGainEffect(), 1
42          ).setTriggerPhrase("Whenever one or more creatures you control attack, "));
43      }
44  
45      private AngelicGuardian(final AngelicGuardian card) {
46          super(card);
47      }
48  
49      @Override
50      public AngelicGuardian copy() {
51          return new AngelicGuardian(this);
52      }
53  }
54  
55  class AngelicGuardianGainEffect extends OneShotEffect {
56  
57      public AngelicGuardianGainEffect() {
58          super(Outcome.Benefit);
59          staticText = "they gain indestructible until end of turn";
60      }
61  
62      public AngelicGuardianGainEffect(final AngelicGuardianGainEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public AngelicGuardianGainEffect copy() {
68          return new AngelicGuardianGainEffect(this);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          Player you = game.getPlayer(source.getControllerId());
74          if (you != null) {
75              game.getCombat().getAttackers().stream()
76                      .map(game::getPermanent)
77                      .filter(Objects::nonNull)
78                      .filter(permanent -> permanent.isControlledBy(you.getId()))
79                      .filter(permanent1 -> permanent1.isCreature(game))
80                      .forEach(permanent -> {
81                          ContinuousEffect effect = new GainAbilityTargetEffect(IndestructibleAbility.getInstance(), Duration.EndOfTurn);
82                          effect.setTargetPointer(new FixedTarget(permanent, game));
83                          game.addEffect(effect, source);
84                      });
85              return true;
86          }
87          return false;
88      }
89  }

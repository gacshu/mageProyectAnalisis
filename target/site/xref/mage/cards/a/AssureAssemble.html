15  import mage.target.common.TargetCreaturePermanent;
16  
17  /**
18   *
19   * @author TheElk801
20   */
21  public final class AssureAssemble extends SplitCard {
22  
23      public AssureAssemble(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{G/W}{G/W}", "{4}{G}{W}", SpellAbilityType.SPLIT);
25  
26          // Assure
27          // Put a +1/+1 counter on target creature. It gains indestructible until end of turn.
28          this.getLeftHalfCard().getSpellAbility().addEffect(
29                  new AddCountersTargetEffect(CounterType.P1P1.createInstance())
30          );
31          this.getLeftHalfCard().getSpellAbility().addEffect(
32                  new GainAbilityTargetEffect(
33                          IndestructibleAbility.getInstance(),
34                          Duration.EndOfTurn
35                  ).setText("That creature gains indestructible until end of turn.")
36          );
37          this.getLeftHalfCard().getSpellAbility().addTarget(
38                  new TargetCreaturePermanent()
39          );
40  
41          // Assemble
42          // Create three 2/2 green and white Elf Knight creature tokens with vigilance.
43          this.getRightHalfCard().getSpellAbility().addEffect(
44                  new CreateTokenEffect(new ElfKnightToken(), 3)
45          );
46      }
47  
48      private AssureAssemble(final AssureAssemble card) {
49          super(card);
50      }
51  
52      @Override
53      public AssureAssemble copy() {
54          return new AssureAssemble(this);
55      }
56  }

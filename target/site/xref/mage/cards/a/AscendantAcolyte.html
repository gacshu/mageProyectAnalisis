24  /**
25   * @author TheElk801

27  public final class AscendantAcolyte extends CardImpl {
28  
29      public AscendantAcolyte(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{G}");
31  
32          this.subtype.add(SubType.HUMAN);
33          this.subtype.add(SubType.MONK);
34          this.power = new MageInt(1);
35          this.toughness = new MageInt(1);
36  
37          // Ascendant Acolyte enters the battlefield with a +1/+1 counter on it for each +1/+1 counter among other creatures you control.
38          this.addAbility(new EntersBattlefieldAbility(new AddCountersSourceEffect(
39                  CounterType.P1P1.createInstance(), AscendantAcolyteValue.instance, true
40          ), "with a +1/+1 counter on it for each +1/+1 counter among other creatures you control").addHint(AscendantAcolyteValue.getHint()));
41  
42          // At the beginning of your upkeep, double the number of +1/+1 counters on Ascendant Acolyte.
43          this.addAbility(new BeginningOfUpkeepTriggeredAbility(
44                  new DoubleCountersSourceEffect(CounterType.P1P1), TargetController.YOU, false
45          ));
46      }
47  
48      private AscendantAcolyte(final AscendantAcolyte card) {
49          super(card);
50      }
51  
52      @Override
53      public AscendantAcolyte copy() {
54          return new AscendantAcolyte(this);
55      }
56  }
57  
58  enum AscendantAcolyteValue implements DynamicValue {
59      instance;
60      private static final Hint hint = new ValueHint(
61              "Total +1/+1 counters on other creatures you control", instance
62      );
63  
64      public static Hint getHint() {
65          return hint;
66      }
67  
68      @Override
69      public int calculate(Game game, Ability sourceAbility, Effect effect) {
70          return game.getBattlefield()
71                  .getActivePermanents(
72                          StaticFilters.FILTER_CONTROLLED_ANOTHER_CREATURE,
73                          sourceAbility.getControllerId(),
74                          sourceAbility, game
75                  ).stream()
76                  .mapToInt(permanent -> permanent.getCounters(game).getCount(CounterType.P1P1))
77                  .sum();
78      }
79  
80      @Override
81      public AscendantAcolyteValue copy() {
82          return this;
83      }
84  
85      @Override
86      public String getMessage() {
87          return "";
88      }
89  }

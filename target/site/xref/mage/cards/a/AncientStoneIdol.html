20  import java.util.UUID;

22  /**
23   * @author TheElk801
24   */
25  public final class AncientStoneIdol extends CardImpl {
26  
27      public AncientStoneIdol(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{10}");
29  
30          this.subtype.add(SubType.GOLEM);
31          this.power = new MageInt(12);
32          this.toughness = new MageInt(12);
33  
34          // Flash
35          this.addAbility(FlashAbility.getInstance());
36  
37          // This spell costs {1} less to cast for each attacking creature.
38          DynamicValue xValue = new AttackingCreatureCount();
39          this.addAbility(new SimpleStaticAbility(Zone.ALL, new SpellCostReductionForEachSourceEffect(1, xValue))
40                  .addHint(new ValueHint("Attacking creatures", xValue))
41          );
42  
43          // Trample
44          this.addAbility(TrampleAbility.getInstance());
45  
46          // When Ancient Stone Idol dies, create a 6/12 colorless Construct artifact creature token with trample.
47          this.addAbility(new DiesSourceTriggeredAbility(new CreateTokenEffect(new StoneTrapIdolToken())));
48      }
49  
50      private AncientStoneIdol(final AncientStoneIdol card) {
51          super(card);
52      }
53  
54      @Override
55      public AncientStoneIdol copy() {
56          return new AncientStoneIdol(this);
57      }
58  }

29  public final class AshiokNightmareMuse extends CardImpl {
30  
31      public AshiokNightmareMuse(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{3}{U}{B}");
33  
34          this.addSuperType(SuperType.LEGENDARY);
35          this.subtype.add(SubType.ASHIOK);
36          this.setStartingLoyalty(5);
37  
38          // +1: Create a 2/3 blue and black Nightmare creature token with "Whenever this creature attacks or blocks, each opponent exiles the top two cards of their library."
39          this.addAbility(new LoyaltyAbility(new CreateTokenEffect(new AshiokNightmareMuseToken()), 1));
40  
41          // −3: Return target nonland permanent to its owner's hand, then that player exiles a card from their hand.
42          Ability ability = new LoyaltyAbility(new AshiokNightmareMuseBounceEffect(), -3);
43          ability.addTarget(new TargetNonlandPermanent());
44          this.addAbility(ability);
45  
46          // −7: You may cast up to three spells from among face-up cards your opponents own from exile without paying their mana costs.
47          this.addAbility(new LoyaltyAbility(new AshiokNightmareMuseCastEffect(), -7));
48      }
49  
50      private AshiokNightmareMuse(final AshiokNightmareMuse card) {
51          super(card);
52      }
53  
54      @Override
55      public AshiokNightmareMuse copy() {
56          return new AshiokNightmareMuse(this);
57      }
58  }

60  class AshiokNightmareMuseBounceEffect extends OneShotEffect {
61  
62      AshiokNightmareMuseBounceEffect() {
63          super(Outcome.Discard);
64          staticText = "return target nonland permanent to its owner's hand, "
65                  + "then that player exiles a card from their hand";
66      }
67  
68      private AshiokNightmareMuseBounceEffect(final AshiokNightmareMuseBounceEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public AshiokNightmareMuseBounceEffect copy() {
74          return new AshiokNightmareMuseBounceEffect(this);
75      }
76  
77      @Override
78      public boolean apply(Game game, Ability source) {
79          Permanent permanent = game.getPermanent(source.getFirstTarget());
80          Player player = game.getPlayer(game.getOwnerId(source.getFirstTarget()));
81          if (permanent == null || player == null) {
82              return false;
83          }
84          player.moveCards(permanent, Zone.HAND, source, game);
85          if (player.getHand().isEmpty()) {
86              return true;
87          }
88          TargetCardInHand target = new TargetCardInHand();
89          if (!player.choose(outcome, player.getHand(), target, game)) {
90              return false;
91          }
92          return player.moveCards(game.getCard(target.getFirstTarget()), Zone.EXILED, source, game);
93      }
94  }

96  class AshiokNightmareMuseCastEffect extends OneShotEffect {
97  
98      private static final FilterCard filter = new FilterCard("face-up cards your opponents own from exile");
99  
100     AshiokNightmareMuseCastEffect() {
101         super(Outcome.Benefit);
102         staticText = "You may cast up to three spells from among face-up cards your opponents own from exile without paying their mana costs.";
103     }
104 
105     private AshiokNightmareMuseCastEffect(final AshiokNightmareMuseCastEffect effect) {
106         super(effect);
107     }
108 
109     @Override
110     public AshiokNightmareMuseCastEffect copy() {
111         return new AshiokNightmareMuseCastEffect(this);
112     }
113 
114     @Override
115     public boolean apply(Game game, Ability source) {
116         Player controller = game.getPlayer(source.getControllerId());
117         if (controller == null) {
118             return false;
119         }
120         // card is owned by an opponent and is face up
121         filter.add(Predicates.not(new OwnerIdPredicate(controller.getId())));
122         filter.add(Predicates.not(FaceDownPredicate.instance));
123         CardUtil.castMultipleWithAttributeForFree(
124                 controller, source, game, new CardsImpl(game.getExile().getCards(filter, game)),
125                 StaticFilters.FILTER_CARD, 3
126         );
127         return true;
128     }

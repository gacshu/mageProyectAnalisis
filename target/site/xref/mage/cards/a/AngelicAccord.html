21  public final class AngelicAccord extends CardImpl {
22  
23      public AngelicAccord(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{W}");
25  
26          // At the beginning of each end step, if you gained 4 or more life this turn, create a 4/4 white Angel creature token with flying.
27          this.addAbility(new BeginningOfEndStepTriggeredAbility(
28                  Zone.BATTLEFIELD, new CreateTokenEffect(new AngelToken()), TargetController.ANY,
29                  new YouGainedLifeCondition(ComparisonType.MORE_THAN, 3), false
30          ).addHint(ControllerGotLifeCount.getHint()), new PlayerGainedLifeWatcher());
31      }
32  
33      private AngelicAccord(final AngelicAccord card) {
34          super(card);
35      }
36  
37      @Override
38      public AngelicAccord copy() {
39          return new AngelicAccord(this);
40      }
41  }

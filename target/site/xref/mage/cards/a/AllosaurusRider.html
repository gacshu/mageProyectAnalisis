7   import mage.abilities.common.SimpleStaticAbility;
8   import mage.abilities.costs.AlternativeCostSourceAbility;
9   import mage.abilities.costs.common.ExileFromHandCost;
10  import mage.abilities.dynamicvalue.DynamicValue;
11  import mage.abilities.dynamicvalue.IntPlusDynamicValue;
12  import mage.abilities.dynamicvalue.common.PermanentsOnBattlefieldCount;
13  import mage.abilities.effects.common.continuous.SetBasePowerToughnessSourceEffect;
14  import mage.cards.CardImpl;
15  import mage.cards.CardSetInfo;
16  import mage.constants.CardType;
17  import mage.constants.Duration;
18  import mage.constants.SubType;
19  import mage.constants.Zone;
20  import mage.filter.FilterCard;
21  import mage.filter.common.FilterControlledLandPermanent;
22  import mage.filter.predicate.mageobject.ColorPredicate;
23  import mage.target.common.TargetCardInHand;
24  
25  /**
26   *
27   * @author LevelX2
28   */
29  public final class AllosaurusRider extends CardImpl {
30  
31      private static final FilterCard filter = new FilterCard("green cards");
32  
33      static {
34          filter.add(new ColorPredicate(ObjectColor.GREEN));
35      }
36  
37      public AllosaurusRider(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{5}{G}{G}");
39          this.subtype.add(SubType.ELF);
40          this.subtype.add(SubType.WARRIOR);
41  
42          this.power = new MageInt(0);
43          this.toughness = new MageInt(0);
44  
45          // You may exile two green cards from your hand rather than pay Allosaurus Rider's mana cost.
46          this.addAbility(new AlternativeCostSourceAbility(new ExileFromHandCost(new TargetCardInHand(2, filter))));
47  
48          // Allosaurus Rider's power and toughness are each equal to 1 plus the number of lands you control.
49          DynamicValue onePlusControlledLands = new IntPlusDynamicValue(1, new PermanentsOnBattlefieldCount(new FilterControlledLandPermanent("lands you control")));
50          this.addAbility(new SimpleStaticAbility(Zone.ALL, new SetBasePowerToughnessSourceEffect(onePlusControlledLands, Duration.EndOfGame)));
51  
52      }
53  
54      private AllosaurusRider(final AllosaurusRider card) {
55          super(card);
56      }
57  
58      @Override
59      public AllosaurusRider copy() {
60          return new AllosaurusRider(this);
61      }
62  }

17  
18  import java.util.UUID;

20  /**
21   * @author TheElk801
22   */
23  public final class AnointerOfValor extends CardImpl {
24  
25      public AnointerOfValor(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{5}{W}");
27  
28          this.subtype.add(SubType.ANGEL);
29          this.power = new MageInt(3);
30          this.toughness = new MageInt(5);
31  
32          // Flying
33          this.addAbility(FlyingAbility.getInstance());
34  
35          // Whenever a creature attacks, you may pay {3}. When you do, put a +1/+1 counter on that creature.
36          this.addAbility(new AttacksAllTriggeredAbility(
37                  new DoWhenCostPaid(
38                          new ReflexiveTriggeredAbility(
39                                  new AddCountersTargetEffect(CounterType.P1P1.createInstance()),
40                                  false, "put a +1/+1 counter on that creature"
41                          ), new GenericManaCost(3), "Pay {3}?"
42                  ), false, StaticFilters.FILTER_PERMANENT_CREATURE,
43                  SetTargetPointer.PERMANENT, false
44          ));
45      }
46  
47      private AnointerOfValor(final AnointerOfValor card) {
48          super(card);
49      }
50  
51      @Override
52      public AnointerOfValor copy() {
53          return new AnointerOfValor(this);
54      }
55  }

31  public final class AssemblyHall extends CardImpl {

33      public AssemblyHall(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{5}");
35  
36          // {4}, {tap}: Reveal a creature card in your hand. Search your library for a card with the same name as that card, reveal it, and put it into your hand. Then shuffle your library.
37          Ability ability = new SimpleActivatedAbility(new AssemblyHallEffect(), new GenericManaCost(4));
38          ability.addCost(new TapSourceCost());
39          this.addAbility(ability);
40      }
41

43          super(card);
44      }

52  class AssemblyHallEffect extends OneShotEffect {
53  
54      public AssemblyHallEffect() {
55          super(Outcome.Benefit);
56          this.staticText = "reveal a creature card from your hand. "
57                  + "Search your library for a card with the same name as that card, "
58                  + "reveal it, and put it into your hand. Then shuffle";
59      }
60  
61      public AssemblyHallEffect(final AssemblyHallEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public AssemblyHallEffect copy() {
67          return new AssemblyHallEffect(this);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          Player controller = game.getPlayer(source.getControllerId());
73          MageObject sourceObject = game.getObject(source);
74          if (controller == null || controller.getHand().isEmpty() || sourceObject == null) {
75              return false;
76          }
77          Card cardToReveal = null;
78          Target target = new TargetCardInHand(StaticFilters.FILTER_CARD_CREATURE);
79          target.setNotTarget(true);
80          if (controller.chooseTarget(outcome, target, source, game)) {
81              cardToReveal = game.getCard(target.getFirstTarget());
82          }
83          if (cardToReveal == null) {
84              return false;
85          }
86          controller.revealCards("from hand :" + sourceObject.getName(), new CardsImpl(cardToReveal), game);
87          String nameToSearch = CardUtil.getCardNameForSameNameSearch(cardToReveal);
88          FilterCard filterCard = new FilterCard("card named " + nameToSearch);
89          filterCard.add(new NamePredicate(nameToSearch));
90          return new SearchLibraryPutInHandEffect(new TargetCardInLibrary(filterCard), true, true).apply(game, source);
91      }
92  }

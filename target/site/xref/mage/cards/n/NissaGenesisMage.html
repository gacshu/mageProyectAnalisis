11  import mage.cards.CardSetInfo;
12  import mage.constants.*;
13  import mage.filter.FilterCard;
14  import mage.filter.predicate.Predicates;
15  import mage.target.common.TargetCreaturePermanent;
16  import mage.target.common.TargetLandPermanent;
17  import mage.target.targetpointer.EachTargetPointer;
18  
19  /**
20   *
21   * @author LevelX2
22   */
23  public final class NissaGenesisMage extends CardImpl {
24  
25      private static final FilterCard filter = new FilterCard("creature and/or land cards");
26  
27      static {
28          filter.add(Predicates.or(
29                  CardType.CREATURE.getPredicate(),
30                  CardType.LAND.getPredicate()
31          ));
32      }
33  
34      public NissaGenesisMage(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{5}{G}{G}");
36          this.addSuperType(SuperType.LEGENDARY);
37          this.subtype.add(SubType.NISSA);
38  
39          this.setStartingLoyalty(5);
40  
41          //+2: Untap up to two target creatures and up to two target lands.
42          Effect effect = new UntapTargetEffect("untap up to two target creatures and up to two target lands");
43          effect.setTargetPointer(new EachTargetPointer());
44          Ability ability = new LoyaltyAbility(effect, +2);
45          ability.addTarget(new TargetCreaturePermanent(0, 2));
46          ability.addTarget(new TargetLandPermanent(0, 2));
47          this.addAbility(ability);
48  
49          //-3: Target creature gets +5/+5 until end of turn.
50          ability = new LoyaltyAbility(new BoostTargetEffect(5, 5, Duration.EndOfTurn), -3);
51          ability.addTarget(new TargetCreaturePermanent());
52          this.addAbility(ability);
53  
54          //-10: Look at the top ten cards of your library.
55          //You may put any number of creature and/or land cards from among them onto the battlefield.
56          //Put the rest on the bottom of your library in a random order.);
57          this.addAbility(new LoyaltyAbility(
58                  new LookLibraryAndPickControllerEffect(10, Integer.MAX_VALUE, filter, PutCards.BATTLEFIELD, PutCards.BOTTOM_RANDOM),
59                  -10));
60      }
61  
62      private NissaGenesisMage(final NissaGenesisMage card) {
63          super(card);
64      }
65  
66      @Override
67      public NissaGenesisMage copy() {
68          return new NissaGenesisMage(this);
69      }
70  }

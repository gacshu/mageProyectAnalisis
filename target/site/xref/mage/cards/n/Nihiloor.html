29  public final class Nihiloor extends CardImpl {
30  
31      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent();
32  
33      static {
34          filter.add(TargetController.OPPONENT.getOwnerPredicate());
35          filter.add(TargetController.YOU.getControllerPredicate());
36      }
37  
38      public Nihiloor(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{W}{U}{B}");
40  
41          this.addSuperType(SuperType.LEGENDARY);
42          this.subtype.add(SubType.HORROR);
43          this.power = new MageInt(3);
44          this.toughness = new MageInt(5);
45  
46          // When Nihiloor enters the battlefield, for each opponent, tap up to one untapped creature you control. When you do, gain control of target creature that player controls with power less than or equal to the tapped creature's power for as long as you control Nihiloor.
47          this.addAbility(new EntersBattlefieldTriggeredAbility(new NihiloorControlEffect()));
48  
49          // Whenever you attack with a creature an opponent owns, you gain 2 life and that player loses 2 life.
50          Ability ability = new AttacksAllTriggeredAbility(
51                  new GainLifeEffect(2), false, filter,
52                  SetTargetPointer.PERMANENT, false
53          ).setTriggerPhrase("Whenever you attack with a creature an opponent owns, ");
54          ability.addEffect(new NihiloorLoseLifeEffect());
55          this.addAbility(ability);
56      }
57  
58      private Nihiloor(final Nihiloor card) {
59          super(card);
60      }
61  
62      @Override
63      public Nihiloor copy() {
64          return new Nihiloor(this);
65      }
66  }

68  class NihiloorControlEffect extends OneShotEffect {
69  
70      private static final FilterControlledCreaturePermanent filter
71              = new FilterControlledCreaturePermanent("untapped creatured you control");
72  
73      static {
74          filter.add(TappedPredicate.UNTAPPED);
75      }
76  
77      private static final class NihiloorPredicate implements Predicate<Permanent> {
78  
79          private final Permanent permanent;
80          private final UUID playerId;
81  
82          private NihiloorPredicate(Permanent permanent, UUID playerId) {
83              this.permanent = permanent;
84              this.playerId = playerId;
85          }
86  
87          @Override
88          public boolean apply(Permanent input, Game game) {
89              return input.isControlledBy(playerId)
90                      && input.isCreature(game)
91                      && input.getPower().getValue() <= permanent.getPower().getValue();
92          }
93      }
94  
95      NihiloorControlEffect() {
96          super(Outcome.Benefit);
97          staticText = "for each opponent, tap up to one untapped creature you control. When you do, "
98                  + "gain control of target creature that player controls with power less than "
99                  + "or equal to the tapped creature's power for as long as you control {this}";
100     }
101 
102     private NihiloorControlEffect(final NihiloorControlEffect effect) {
103         super(effect);
104     }
105 
106     @Override
107     public NihiloorControlEffect copy() {
108         return new NihiloorControlEffect(this);
109     }
110 
111     @Override
112     public boolean apply(Game game, Ability source) {
113         Player controller = game.getPlayer(source.getControllerId());
114         if (controller == null) {
115             return false;
116         }
117         for (UUID playerId : game.getOpponents(source.getControllerId())) {
118             Player opponent = game.getPlayer(playerId);
119             if (opponent == null) {
120                 continue;
121             }
122             TargetPermanent target = new TargetPermanent(0, 1, filter, true);
123             target.withChooseHint("tapping a creature controlled by " + opponent.getName());
124             controller.choose(outcome, target, source, game);
125             Permanent permanent = game.getPermanent(target.getFirstTarget());
126             if (permanent == null || !permanent.tap(source, game)) {
127                 continue;
128             }
129             FilterPermanent filter2 = new FilterPermanent(
130                     "creature controlled by " + opponent.getName()
131                     + " with power " + permanent.getPower().getValue() + " or less"
132             );
133             filter2.add(new NihiloorPredicate(permanent, playerId));
134             ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
135                     new GainControlTargetEffect(Duration.WhileControlled, true),
136                     false, "gain control of target creature that player controls with "
137                     + "power less than or equal to the tapped creature's power for as long as you control {this}"
138             );
139             ability.addTarget(new TargetPermanent(filter2));
140             game.fireReflexiveTriggeredAbility(ability, source);
141         }
142         return true;
143     }
144 
145 }

147 class NihiloorLoseLifeEffect extends OneShotEffect {
148 
149     NihiloorLoseLifeEffect() {
150         super(Outcome.Benefit);
151         staticText = "and that player loses 2 life";
152     }
153 
154     private NihiloorLoseLifeEffect(final NihiloorLoseLifeEffect effect) {
155         super(effect);
156     }
157 
158     @Override
159     public NihiloorLoseLifeEffect copy() {
160         return new NihiloorLoseLifeEffect(this);
161     }
162 
163     @Override
164     public boolean apply(Game game, Ability source) {
165         Player owner = game.getPlayer(game.getOwnerId(getTargetPointer().getFirst(game, source)));
166         return owner != null
167                 && owner.loseLife(2, game, source, false) > 0;
168     }

18   *
19   * @author jeffwadsworth
20   */
21  public final class NearDeathExperience extends CardImpl {
22  
23      public NearDeathExperience(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{2}{W}{W}{W}");
25  
26  
27          // At the beginning of your upkeep, if you have exactly 1 life, you win the game.
28          TriggeredAbility ability = new BeginningOfUpkeepTriggeredAbility(new WinGameSourceControllerEffect(), TargetController.YOU, false);
29          this.addAbility(new ConditionalInterveningIfTriggeredAbility(ability, new OneLifeCondition(), "At the beginning of your upkeep, if you have exactly 1 life, you win the game."));
30      }
31  
32      private NearDeathExperience(final NearDeathExperience card) {
33          super(card);
34      }
35  
36      @Override
37      public NearDeathExperience copy() {
38          return new NearDeathExperience(this);
39      }
40  }
41  
42  class OneLifeCondition implements Condition {
43  
44      @Override
45      public boolean apply(Game game, Ability source) {
46          return game.getPlayer(source.getControllerId()).getLife() == 1;
47      }
48  }

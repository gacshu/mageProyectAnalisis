29      public NinjasKunai(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{1}");
31  
32          this.subtype.add(SubType.EQUIPMENT);
33  
34          // Equipped creature has "{1}, {T}, Sacrifice Ninja's Kunai: Ninja's Kunai deals 3 damage to any target."
35          this.addAbility(new SimpleStaticAbility(new GainAbilityWithAttachmentEffect(
36                  "equipped creature has \"{1}, {T}, Sacrifice {this}: {this} deals 3 damage to any target.\"",
37                  new NinjasKunaiEffect(), new TargetAnyTarget(), new SacrificeAttachmentCost(), new GenericManaCost(1), new TapSourceCost()
38          )));
39  
40          // Equip {1}
41          this.addAbility(new EquipAbility(1));
42      }
43  
44      private NinjasKunai(final NinjasKunai card) {
45          super(card);
46      }
47

49      public NinjasKunai copy() {
50          return new NinjasKunai(this);
51      }
52  }

54  class NinjasKunaiEffect extends OneShotEffect {
55  
56      NinjasKunaiEffect() {
57          super(Outcome.Benefit);
58      }
59  
60      private NinjasKunaiEffect(final NinjasKunaiEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public NinjasKunaiEffect copy() {
66          return new NinjasKunaiEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Object object = getValue("attachedPermanent");
72          Player player = game.getPlayer(source.getControllerId());
73          if (!(object instanceof Permanent) || player == null) {
74              return false;
75          }
76          Permanent permanent = (Permanent) object;
77          Permanent targetedPermanent = game.getPermanent(source.getFirstTarget());
78          if (targetedPermanent == null) {
79              Player targetedPlayer = game.getPlayer(source.getFirstTarget());
80              if (targetedPlayer != null) {
81                  targetedPlayer.damage(3, permanent.getId(), source, game);
82              }
83          } else {
84              targetedPermanent.damage(3, permanent.getId(), source, game);
85          }
86          return true;
87      }
88  
89      @Override
90      public String getText(Mode mode) {
91          String name = "Ninja's Kunai";
92          Object object = getValue("attachedPermanent");
93          if (object instanceof Permanent) {
94              name = ((Permanent) object).getName();
95          }
96          return name + "deals 3 damage to target any target.";
97      }
98  }

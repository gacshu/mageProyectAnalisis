16  import mage.target.common.TargetCardInLibrary;
17  
18  import java.util.UUID;
19  
20  /**
21   * @author TheElk801
22   */
23  public final class NissasTriumph extends CardImpl {
24  
25      private static final FilterCard filter = new FilterCard("basic Forest cards");
26      private static final FilterPermanent filter2 = new FilterControlledPlaneswalkerPermanent(SubType.NISSA);
27  
28      static {
29          filter.add(SuperType.BASIC.getPredicate());
30          filter.add(SubType.FOREST.getPredicate());
31      }
32  
33      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter2);
34  
35      public NissasTriumph(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{G}{G}");
37  
38          // Search your land for up to two basic Forest cards. If you control a Nissa planeswalker, instead search your library for up to three land cards. reveal those cards, put them in your hand, then shuffle your library.
39          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
40                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary(
41                          0, 3, StaticFilters.FILTER_CARD_LAND
42                  ), true, true),
43                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary(
44                          0, 2, filter
45                  ), true, true),
46                  new PermanentsOnTheBattlefieldCondition(filter2),
47                  "Search your library for up to two basic Forest cards. If you control a Nissa planeswalker, " +
48                          "instead search your library for up to three land cards. " +
49                          "Reveal those cards, put them into your hand, then shuffle."
50          ));
51      }
52  
53      private NissasTriumph(final NissasTriumph card) {
54          super(card);
55      }
56  
57      @Override
58      public NissasTriumph copy() {
59          return new NissasTriumph(this);
60      }
61  }

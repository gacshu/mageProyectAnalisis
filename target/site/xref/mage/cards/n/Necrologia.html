12  
13  import java.util.UUID;

16   * @author LevelX2

18  public final class Necrologia extends CardImpl {
19  
20      public Necrologia(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{B}{B}");
22  
23          // Cast Necrologia only during your end step.
24          this.addAbility(new CastOnlyDuringPhaseStepSourceAbility(null, PhaseStep.END_TURN, MyTurnCondition.instance,
25                  "Cast this spell only during your end step"));
26  
27          // As an additional cost to cast to Necrologia, pay X life.
28          this.getSpellAbility().addCost(new PayVariableLifeCost(true));
29  
30          // Draw X cards.
31          this.getSpellAbility().addEffect(new DrawCardSourceControllerEffect(GetXValue.instance));
32      }
33  
34      private Necrologia(final Necrologia card) {
35          super(card);
36      }
37  
38      @Override
39      public Necrologia copy() {
40          return new Necrologia(this);
41      }
42  }

28  public final class NevinyrralUrborgTyrant extends CardImpl {
29  
30      public NevinyrralUrborgTyrant(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}{U}{B}");
32  
33          this.addSuperType(SuperType.LEGENDARY);
34          this.subtype.add(SubType.ZOMBIE);
35          this.subtype.add(SubType.WIZARD);
36          this.power = new MageInt(3);
37          this.toughness = new MageInt(6);
38  
39          // Hexproof from artifacts, creatures, and enchantments
40          this.addAbility(HexproofFromArtifactsCreaturesAndEnchantments.getInstance());
41  
42          // When Nevinyrral, Urborg Tyrant enters the battlefield, create a tapped 2/2 black Zombie creature token for each creature that died this turn.
43          this.addAbility(new EntersBattlefieldTriggeredAbility(new CreateTokenEffect(
44                  new ZombieToken(), CreaturesDiedThisTurnCount.instance, true, false
45          ).setText("create a tapped 2/2 black Zombie creature token for each creature that died this turn")).addHint(CreaturesDiedThisTurnHint.instance));
46  
47          // When Nevinyrral dies, you may pay {1}. When you do, destroy all artifacts, creatures, and enchantments.
48          this.addAbility(new DiesSourceTriggeredAbility(
49                  new DoWhenCostPaid(new ReflexiveTriggeredAbility(
50                          new DestroyAllEffect(StaticFilters.FILTER_PERMANENT_ARTIFACT_CREATURE_OR_ENCHANTMENT),
51                          false, "destroy all artifacts, creatures, and enchantments"
52                  ), new GenericManaCost(1), "Pay {1}?")
53          ));
54      }
55  
56      private NevinyrralUrborgTyrant(final NevinyrralUrborgTyrant card) {
57          super(card);
58      }
59  
60      @Override
61      public NevinyrralUrborgTyrant copy() {
62          return new NevinyrralUrborgTyrant(this);
63      }
64  }

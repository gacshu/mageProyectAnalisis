18  
19  import java.util.UUID;

21  /**
22   * @author jeffwadsworth
23   */
24  public final class NayaSojourners extends CardImpl {
25  
26      public NayaSojourners(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{G}{W}");
28          this.subtype.add(SubType.ELF);
29          this.subtype.add(SubType.SHAMAN);
30          this.power = new MageInt(5);
31          this.toughness = new MageInt(3);
32  
33          // When you cycle Naya Sojourners or it dies, you may put a +1/+1 counter on target creature.
34          Ability ability = new OrTriggeredAbility(Zone.ALL,
35                  new AddCountersTargetEffect(CounterType.P1P1.createInstance()),
36                  true,
37                  "When you cycle {this} or it dies, ",
38                  new CycleTriggeredAbility(null, true),
39                  new DiesSourceTriggeredAbility(null, true)
40          );
41          ability.addTarget(new TargetCreaturePermanent());
42          this.addAbility(ability);
43  
44          // Cycling {2}{G}
45          this.addAbility(new CyclingAbility(new ManaCostsImpl<>("{2}{G}")));
46      }
47  
48      private NayaSojourners(final NayaSojourners card) {
49          super(card);
50      }
51  
52      @Override
53      public NayaSojourners copy() {
54          return new NayaSojourners(this);
55      }
56  }

16  import static mage.cards.decks.CardNameUtil.CARD_NAME_PATTERN;
17  
18  public class MtgaImporter extends PlainTextDeckImporter {
19  
20      private static final Map<String, String> SET_REMAPPING = ImmutableMap.of("DAR", "DOM");
21      private static final Pattern MTGA_PATTERN = Pattern.compile(
22              "(\\p{Digit}+)" +
23                      "\\p{javaWhitespace}+" +
24                      "(" + CARD_NAME_PATTERN.pattern() + ")" +
25                      "(?:\\p{javaWhitespace}+\\()?" +
26                      "(\\p{Alnum}+)?" +
27                      "(?:\\)\\p{javaWhitespace}+)?" +
28                      "(\\p{Graph}+)?");
29  
30      private final CardLookup lookup = getCardLookup();
31      private boolean sideboard = false;
32  
33      @Override
34      protected void readLine(String line, DeckCardLists deckList, FixedInfo fixedInfo) {
35          
36          line = line.trim();
37          
38          if (line.equals("Deck")) {
39              return;
40          }
41  
42          if (line.equals("Sideboard") || line.equals("")) {
43              sideboard = true;
44              return;
45          }
46  
47          Matcher pattern = MTGA_PATTERN.matcher(CardNameUtil.normalizeCardName(line));
48          
49          if (!pattern.matches()) {
50              sbMessage.append("Error reading '").append(line).append("'\n");
51              return;
52          }
53          
54          Optional<CardInfo> found;
55          int count = Integer.parseInt(pattern.group(1));
56          String name = pattern.group(2);    
57          if (pattern.group(3) != null && pattern.group(4) != null) {
58              String set = SET_REMAPPING.getOrDefault(pattern.group(3), pattern.group(3));
59              String cardNumber = pattern.group(4);
60              found = lookup.lookupCardInfo(name, set, cardNumber);
61          } else {
62              found = lookup.lookupCardInfo(name);
63          }
64          
65          if (!found.isPresent()) {
66              sbMessage.append("Cound not find card for '").append(line).append("'\n");
67          } else {
68              final List<DeckCardInfo> zone = sideboard ? deckList.getSideboard() : deckList.getCards();
69              found.ifPresent(card -> zone.addAll(Collections.nCopies(count,
70                      new DeckCardInfo(card.getName(), card.getCardNumber(), card.getSetCode()))));
71          }
72      }
73  
74  }

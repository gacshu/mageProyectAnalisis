12  public class DecDeckImporter extends PlainTextDeckImporter {
13  
14      @Override
15      protected void readLine(String line, DeckCardLists deckList, FixedInfo fixedInfo) {
16  
17          if (line.isEmpty() || line.startsWith("//")) {
18              return;
19          }
20  
21          boolean sideboard = false;
22          if (line.startsWith("SB:")) {
23              line = line.substring(3).trim();
24              sideboard = true;
25          }
26  
27          int delim = line.indexOf(' ');
28          String lineNum = line.substring(0, delim).trim();
29          String lineName = line.substring(delim).trim();
30          try {
31              int num = Integer.parseInt(lineNum);
32              Optional<CardInfo> cardLookup = getCardLookup().lookupCardInfo(lineName);
33              if (!cardLookup.isPresent()) {
34                  sbMessage.append("Could not find card: '").append(lineName).append("' at line ").append(lineCount).append('\n');
35              } else {
36                  CardInfo cardInfo = cardLookup.get();
37                  for (int i = 0; i < num; i++) {
38                      if (!sideboard) {
39                          deckList.getCards().add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
40                      } else {
41                          deckList.getSideboard().add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
42                      }
43                  }
44              }
45          } catch (NumberFormatException nfe) {
46              sbMessage.append("Invalid number: ").append(lineNum).append(" at line ").append(lineCount).append('\n');
47          }
48      }
49  
50  }

220     @Override
221     public boolean validate(Deck deck) {
222         boolean valid = true;
223         errorsList.clear();
224         //20091005 - 100.2a
225         if (deck.getCards().size() < getDeckMinSize()) {
226             addError(DeckValidatorErrorType.DECK_SIZE, "Deck", "Must contain at least " + getDeckMinSize() + " cards: has only " + deck.getCards().size() + " cards");
227             valid = false;
228         }
229         //20130713 - 100.4a
230         if (deck.getSideboard().size() > 15) {
231             addError(DeckValidatorErrorType.DECK_SIZE, "Sideboard", "Must contain no more than 15 cards : has " + deck.getSideboard().size() + " cards");
232             valid = false;
233         }
234 
235         Map<String, Integer> counts = new HashMap<>();
236         countCards(counts, deck.getCards());
237         countCards(counts, deck.getSideboard());
238         valid = checkCounts(4, counts) && valid;
239 
240         for (String bannedCard : banned) {
241             if (counts.containsKey(bannedCard)) {
242                 addError(DeckValidatorErrorType.BANNED, bannedCard, "Banned", true);
243                 valid = false;
244             }
245         }
246 
247         for (String restrictedCard : restricted) {
248             if (counts.containsKey(restrictedCard)) {
249                 int count = counts.get(restrictedCard);
250                 if (count > 1) {
251                     addError(DeckValidatorErrorType.OTHER, restrictedCard, "Restricted amount: " + count, true);
252                     valid = false;
253                 }
254             }
255         }
256 
257         if (!rarities.isEmpty()) {
258             for (Card card : deck.getCards()) {
259                 if (!rarities.contains(card.getRarity())) {
260                     if (!legalRarity(card)) {
261                         valid = false;
262                     }
263                 }
264             }
265             for (Card card : deck.getSideboard()) {
266                 if (!rarities.contains(card.getRarity())) {
267                     if (!legalRarity(card)) {
268                         valid = false;
269                     }
270                 }
271             }
272         }
273 
274         for (Card card : deck.getCards()) {
275             if (!isSetAllowed(card.getExpansionSetCode())) {
276                 if (!legalSets(card)) {
277                     valid = false;
278                 }
279             }
280         }
281         for (Card card : deck.getSideboard()) {
282             if (!isSetAllowed(card.getExpansionSetCode())) {
283                 if (!legalSets(card)) {
284                     valid = false;
285                 }
286             }
287         }
288         return valid;
289     }

291     /**
292      * Checks if the given card is legal in any of the given rarities
293      *
294      * @param card - the card to check
295      * @return Whether the card was printed at any of the given rarities.
296      */
297     protected boolean legalRarity(Card card) {
298         // check if card is legal if taken from other set
299         boolean legal = false;
300         List<CardInfo> cardInfos = CardRepository.instance.findCards(card.getName());
301         for (CardInfo cardInfo : cardInfos) {
302             if (rarities.contains(cardInfo.getRarity())) {
303                 legal = true;
304                 break;
305             }
306         }
307         if (!legal && !errorsListContainsGroup(card.getName())) {
308             addError(DeckValidatorErrorType.OTHER, card.getName(), "Invalid rarity: " + card.getRarity(), true);
309         }
310         return legal;
311     }

313     /**
314      * Checks if a given set is legal in this format.
315      *
316      * @param code - the set code to check
317      * @return Whether the set is legal in this format.
318      */
319     protected boolean isSetAllowed(String code) {
320         return setCodes.isEmpty() || setCodes.contains(code);
321     }
322 
323     /**
324      * Checks if the given card is legal in any of the given sets or as single card
325      *
326      * @param card - the card to check
327      * @return Whether the card was printed in any of this format's sets.
328      */
329     protected boolean legalSets(Card card) {
330         // check if card is legal if taken from other set
331         boolean legal = false;
332         List<CardInfo> cardInfos = CardRepository.instance.findCards(card.getName());
333         for (CardInfo cardInfo : cardInfos) {
334             if (isSetAllowed(cardInfo.getSetCode())) {
335                 legal = true;
336                 break;
337             }
338         }
339 
340         // check if single card allows
341         if (singleCards.contains(card.getName())) {
342             legal = true;
343         }
344 
345         if (!legal && !errorsListContainsGroup(card.getName())) {
346             addError(DeckValidatorErrorType.WRONG_SET, card.getName(), "Invalid set: " + card.getExpansionSetCode(), true);
347         }
348         return legal;
349     }
350 
351     protected boolean checkCounts(int maxCopies, Map<String, Integer> counts) {
352         boolean valid = true;
353         for (Entry<String, Integer> entry : counts.entrySet()) {
354             if (entry.getValue() > maxCopies
355                     && !basicLandNames.contains(entry.getKey())
356                     && !anyNumberCardsAllowed.contains(entry.getKey())) {
357                 addError(DeckValidatorErrorType.OTHER, entry.getKey(), "Too many: " + entry.getValue(), true);
358                 valid = false;
359             }
360             if (entry.getValue() > 7 && entry.getKey().equals("Seven Dwarves")) {
361                 addError(DeckValidatorErrorType.OTHER, entry.getKey(), "Too many: " + entry.getValue(), true);
362                 valid = false;
363             }
364         }
365         return valid;
366     }

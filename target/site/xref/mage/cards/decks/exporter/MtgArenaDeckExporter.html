14   */
15  public class MtgArenaDeckExporter extends DeckExporter {
16  
17      private static final String ext = "mtga";
18      private static final String description = "MTG Arena's deck format (*.mtga)";
19      private static final FileFilter fileFilter = new DeckFileFilter(ext, description);
20  
21      private static final Map<String, String> SET_CODE_REPLACEMENTS = ImmutableMap.of("DOM", "DAR");
22  
23      @Override
24      public void writeDeck(PrintWriter out, DeckCardLists deck) {
25          Map<String, Integer> amount = new HashMap<>();
26          List<String> deckMain = prepareCardsList(deck.getCards(), amount, "M@");
27          List<String> deckSideboard = prepareCardsList(deck.getSideboard(), amount, "S@");
28  
29          printCards(out, deckMain, amount, "M@");
30          if (deckSideboard.size() > 0) {
31              out.println();
32              printCards(out, deckSideboard, amount, "S@");
33          }
34      }
35  
36      private List<String> prepareCardsList(List<DeckCardInfo> sourceCards, Map<String, Integer> amount, String prefix) {
37          List<String> res = new ArrayList<>();
38          for (DeckCardInfo card : sourceCards) {
39              String setCode = card.getSetCode().toUpperCase(Locale.ENGLISH);
40              setCode = SET_CODE_REPLACEMENTS.getOrDefault(setCode, setCode);
41              String name = card.getCardName() + " (" + setCode + ") " + card.getCardNum();
42              String code = prefix + name;
43              int curAmount = amount.getOrDefault(code, 0);
44              if (curAmount == 0) {
45                  res.add(name);
46              }
47              amount.put(code, curAmount + card.getQuantity());
48          }
49          return res;
50      }
51  
52      private void printCards(PrintWriter out, List<String> deck, Map<String, Integer> amount, String prefix) {
53          if (deck.size() == 0) return;
54  
55          boolean firstCard = true;
56          for (String name : deck) {
57              if (!firstCard) out.println();
58              out.print(amount.get(prefix + name));
59              out.print(' ');
60              out.print(name);
61              firstCard = false;
62          }
63          out.println();
64      }
65  
66      @Override
67      public FileFilter getFileFilter() {
68          return fileFilter;
69      }
70  
71      @Override
72      public String getDescription() {
73          return description;
74      }
75  
76      @Override
77      public String getDefaultFileExt() {
78          return ext;
79      }
80  }

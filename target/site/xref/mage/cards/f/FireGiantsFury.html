34  public final class FireGiantsFury extends CardImpl {
35  
36      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent(SubType.GIANT);
37  
38      public FireGiantsFury(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{R}");
40  
41          // Target Giant you control gets +2/+2 and gains trample until end of turn.
42          Effect effect = new BoostTargetEffect(2, 2, Duration.EndOfTurn);
43          effect.setText("Target Giant you control gets +2/+2");
44          this.getSpellAbility().addEffect(effect);
45  
46          effect = new GainAbilityTargetEffect(TrampleAbility.getInstance(), Duration.EndOfTurn);
47          effect.setText("and gains trample until end of turn");
48          this.getSpellAbility().addEffect(effect);
49  
50          // Whenever it deals combat damage to a player this turn, exile that many cards from the top of your library. Until the end of your next turn, you may play those cards.
51          effect = new FireGiantsFuryEffect();
52          this.getSpellAbility().addEffect(effect);
53  
54          this.getSpellAbility().addTarget(new TargetPermanent(filter));
55      }
56  
57      private FireGiantsFury(final FireGiantsFury card) {
58          super(card);
59      }
60  
61      @Override
62      public FireGiantsFury copy() {
63          return new FireGiantsFury(this);
64      }
65  }

70          super(Outcome.PlayForFree);

72      }

74      private FireGiantsFuryEffect(final FireGiantsFuryEffect effect) {
75          super(effect);
76      }
77  
78      @Override
79      public FireGiantsFuryEffect copy() {
80          return new FireGiantsFuryEffect(this);
81      }
82  
83      @Override
84      public boolean apply(Game game, Ability source) {
85          Permanent permanent = game.getPermanent(source.getFirstTarget());
86          if (permanent == null) {
87              return false;
88          }
89          game.addDelayedTriggeredAbility(new FireGiantsFuryDelayedTriggeredAbility(new MageObjectReference(permanent, game)), source);
90          return true;
91      }

94  class FireGiantsFuryDelayedTriggeredAbility extends DelayedTriggeredAbility {
95  
96      private final MageObjectReference mor;
97  
98      public FireGiantsFuryDelayedTriggeredAbility(MageObjectReference mor) {
99          super(new FireGiantsFuryDelayedEffect(), Duration.EndOfTurn, false, false);
100         this.mor = mor;
101     }
102 
103     private FireGiantsFuryDelayedTriggeredAbility(FireGiantsFuryDelayedTriggeredAbility ability) {
104         super(ability);
105         this.mor = ability.mor;
106     }
107 
108     @Override
109     public boolean checkEventType(GameEvent event, Game game) {
110         return event.getType() == GameEvent.EventType.DAMAGED_PLAYER;
111     }
112 
113     @Override
114     public boolean checkTrigger(GameEvent event, Game game) {
115         if (mor.refersTo(event.getSourceId(), game) && ((DamagedEvent) event).isCombatDamage()) {
116             for (Effect effect : this.getAllEffects()) {
117                 effect.setTargetPointer(new FixedTarget(event.getPlayerId()));
118                 effect.setValue("damage", event.getAmount());
119             }
120             return true;
121         }
122         return false;
123     }
124 
125     @Override
126     public FireGiantsFuryDelayedTriggeredAbility copy() {
127         return new FireGiantsFuryDelayedTriggeredAbility(this);
128     }
129 
130     @Override
131     public String getRule() {
132         return "Whenever it deals combat damage to a player this turn, exile that many cards from the top of your library. Until the end of your next turn, you may play those cards";
133     }
134 }

136 class FireGiantsFuryDelayedEffect extends OneShotEffect {
137 
138     public FireGiantsFuryDelayedEffect() {
139         super(Outcome.PlayForFree);
140         this.staticText = "exile that many cards from the top of your library. Until the end of your next turn, you may play those cards";
141     }
142 
143     private FireGiantsFuryDelayedEffect(final FireGiantsFuryDelayedEffect effect) {
144         super(effect);
145     }
146 
147     @Override
148     public FireGiantsFuryDelayedEffect copy() {
149         return new FireGiantsFuryDelayedEffect(this);
150     }
151 
152     @Override
153     public boolean apply(Game game, Ability source) {
154         Player controller = game.getPlayer(source.getControllerId());
155         if (controller != null) {
156             Integer damage = (Integer) getValue("damage");
157             if (damage != null && damage > 0) {
158                 Set<Card> cards = controller.getLibrary().getTopCards(game, damage);
159                 Card sourceCard = game.getCard(source.getSourceId());
160                 controller.moveCardsToExile(cards, source, game, true, CardUtil.getCardExileZoneId(game, source), sourceCard != null ? sourceCard.getIdName() : "");
161 
162                 for (Card card : cards) {
163                     ContinuousEffect effect = new FireGiantsFuryMayPlayEffect();
164                     effect.setTargetPointer(new FixedTarget(card.getId(), game));
165                     game.addEffect(effect, source);
166                 }
167 
168             }
169             return true;
170         }
171         return false;
172     }
173 }

175 class FireGiantsFuryMayPlayEffect extends AsThoughEffectImpl {
176 
177     private int castOnTurn = 0;
178 
179     public FireGiantsFuryMayPlayEffect() {
180         super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.Custom, Outcome.Benefit);
181         this.staticText = "Until the end of your next turn, you may play that card.";
182     }
183 
184     private FireGiantsFuryMayPlayEffect(final FireGiantsFuryMayPlayEffect effect) {
185         super(effect);
186         castOnTurn = effect.castOnTurn;
187     }
188 
189     @Override
190     public FireGiantsFuryMayPlayEffect copy() {
191         return new FireGiantsFuryMayPlayEffect(this);
192     }
193 
194     @Override
195     public void init(Ability source, Game game) {
196         super.init(source, game);
197         castOnTurn = game.getTurnNum();
198     }
199 
200     @Override
201     public boolean isInactive(Ability source, Game game) {
202         if (castOnTurn != game.getTurnNum() && game.getPhase().getStep().getType() == PhaseStep.END_TURN) {
203             return game.isActivePlayer(source.getControllerId());
204         }
205         return false;
206     }
207 
208     @Override
209     public boolean apply(Game game, Ability source) {
210         return true;
211     }
212 
213     @Override
214     public boolean applies(UUID sourceId, Ability source, UUID affectedControllerId, Game game) {
215         UUID objectIdToCast = CardUtil.getMainCardId(game, sourceId);
216         return source.isControlledBy(affectedControllerId)
217                 && getTargetPointer().getTargets(game, source).contains(objectIdToCast);
218     }

16  public final class FuryStorm extends CardImpl {
17  
18      public FuryStorm(UUID ownerId, CardSetInfo setInfo) {
19          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{2}{R}{R}");
20  
21          // When you cast this spell, copy it for each time you've cast your commander from the command zone this game. You may choose new targets for the copies.
22          this.addAbility(new CommanderStormAbility());
23  
24          // Copy target instant or sorcery spell. You may choose new targets for the copy.
25          this.getSpellAbility().addEffect(new CopyTargetSpellEffect());
26          this.getSpellAbility().addTarget(new TargetSpell(StaticFilters.FILTER_SPELL_INSTANT_OR_SORCERY));
27      }
28  
29      private FuryStorm(final FuryStorm card) {
30          super(card);
31      }
32  
33      @Override
34      public FuryStorm copy() {
35          return new FuryStorm(this);
36      }

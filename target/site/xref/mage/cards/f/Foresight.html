22  public final class Foresight extends CardImpl {
23  
24      public Foresight(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{U}");
26  
27          // Search your library for three cards, exile them, then shuffle your library.
28          this.getSpellAbility().addEffect(new ForesightEffect());
29  
30          // Draw a card at the beginning of the next turn's upkeep.
31          this.getSpellAbility().addEffect(new CreateDelayedTriggeredAbilityEffect(new AtTheBeginOfNextUpkeepDelayedTriggeredAbility(new DrawCardSourceControllerEffect(1)), false).concatBy("<br>"));
32      }
33  
34      private Foresight(final Foresight card) {
35          super(card);
36      }
37

39      public Foresight copy() {
40          return new Foresight(this);
41      }
42  }

44  class ForesightEffect extends SearchEffect {
45  
46      ForesightEffect() {
47          super(new TargetCardInLibrary(3, StaticFilters.FILTER_CARD), Outcome.Benefit);
48          staticText = "Search your library for three cards, exile them, then shuffle";
49      }
50  
51      private ForesightEffect(final ForesightEffect effect) {
52          super(effect);
53      }
54  
55      @Override
56      public ForesightEffect copy() {
57          return new ForesightEffect(this);
58      }
59  
60      @Override
61      public boolean apply(Game game, Ability source) {
62          Player player = game.getPlayer(source.getControllerId());
63          if (player == null) {
64              return false;
65          }
66          player.searchLibrary(target, source, game);
67          Cards cards = new CardsImpl();
68          for (UUID targetId : target.getTargets()) {
69              Card card = player.getLibrary().getCard(targetId, game);
70              if (card != null) {
71                  cards.add(card);
72              }
73          }
74          player.moveCards(cards, Zone.EXILED, source, game);
75          player.shuffleLibrary(source, game);
76          return true;
77      }
78  }

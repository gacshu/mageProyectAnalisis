4   import mage.abilities.condition.common.ManaWasSpentCondition;
5   import mage.abilities.decorator.ConditionalOneShotEffect;
6   import mage.abilities.effects.common.DamageAllEffect;
7   import mage.abilities.keyword.FlyingAbility;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.ColoredManaSymbol;
12  import mage.filter.common.FilterCreaturePermanent;
13  import mage.filter.predicate.Predicates;
14  import mage.filter.predicate.mageobject.AbilityPredicate;
15  
16  import java.util.UUID;
17  
18  /**
19   * @author LevelX2
20   */
21  public final class Firespout extends CardImpl {
22  
23      private static final FilterCreaturePermanent filter1 = new FilterCreaturePermanent("creature without flying");
24      private static final FilterCreaturePermanent filter2 = new FilterCreaturePermanent("creature with flying");
25  
26      static {
27          filter1.add(Predicates.not(new AbilityPredicate(FlyingAbility.class)));
28          filter2.add(new AbilityPredicate(FlyingAbility.class));
29      }
30  
31      public Firespout(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{2}{R/G}");
33  
34  
35          // Firespout deals 3 damage to each creature without flying if {R} was spent to cast Firespout and 3 damage to each creature with flying if {G} was spent to cast it.
36          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
37                  new DamageAllEffect(3, filter1),
38                  ManaWasSpentCondition.RED, "{this} deals 3 damage to each creature without flying if {R} was spent to cast this spell"));
39          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
40                  new DamageAllEffect(3, filter2),
41                  ManaWasSpentCondition.GREEN, "and 3 damage to each creature with flying if {G} was spent to cast this spell. <i>(Do both if {R}{G} was spent.)</i>"));
42      }
43  
44      private Firespout(final Firespout card) {
45          super(card);
46      }
47  
48      @Override
49      public Firespout copy() {
50          return new Firespout(this);
51      }
52  }

34  public final class FalseOrders extends CardImpl {

36      private static final FilterPermanent filter = new FilterPermanent("creature defending player controls");
37  
38      static {
39          filter.add(CardType.CREATURE.getPredicate());
40          filter.add(FalseOrdersDefendingPlayerControlsPredicate.instance);
41      }
42  
43      public FalseOrders(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{R}");
45  
46          // Cast False Orders only during the declare blockers step.
47          this.addAbility(new CastOnlyDuringPhaseStepSourceAbility(null, PhaseStep.DECLARE_BLOCKERS, null, "Cast this spell only during the declare blockers step"));
48  
49          // Remove target creature defending player controls from combat. Creatures it was blocking that had become blocked by only that creature this combat become unblocked. You may have it block an attacking creature of your choice.
50          this.getSpellAbility().addTarget(new TargetPermanent(filter));
51          this.getSpellAbility().addEffect(new FalseOrdersUnblockEffect());
52          this.getSpellAbility().addWatcher(new BlockedByOnlyOneCreatureThisCombatWatcher());
53      }
54  
55      private FalseOrders(final FalseOrders card) {
56          super(card);
57      }
58  
59      @Override
60      public FalseOrders copy() {
61          return new FalseOrders(this);
62      }

66  enum FalseOrdersDefendingPlayerControlsPredicate implements ObjectSourcePlayerPredicate<Controllable> {
67      instance;
68  
69      @Override
70      public boolean apply(ObjectSourcePlayer<Controllable> input, Game game) {
71          return game.getCombat().getPlayerDefenders(game).contains(input.getObject().getControllerId());
72      }
73  }

75  class FalseOrdersUnblockEffect extends OneShotEffect {
76  
77      FalseOrdersUnblockEffect() {
78          super(Outcome.Benefit);
79          this.staticText = "Remove target creature defending player controls from combat. Creatures it was blocking that had become blocked by only that creature this combat become unblocked. You may have it block an attacking creature of your choice";
80      }
81  
82      private FalseOrdersUnblockEffect(final FalseOrdersUnblockEffect effect) {
83          super(effect);
84      }
85  
86      @Override
87      public FalseOrdersUnblockEffect copy() {
88          return new FalseOrdersUnblockEffect(this);
89      }
90  
91      @Override
92      public boolean apply(Game game, Ability source) {
93          Player controller = game.getPlayer(source.getControllerId());
94          Permanent permanent = game.getPermanent(source.getTargets().getFirstTarget());
95          if (controller == null || permanent == null) {
96              return false;
97          }
98  
99          // Remove target creature from combat
100         Effect effect = new RemoveFromCombatTargetEffect();
101         effect.apply(game, source);
102 
103         // Make blocked creatures unblocked
104         BlockedByOnlyOneCreatureThisCombatWatcher watcher = game.getState().getWatcher(BlockedByOnlyOneCreatureThisCombatWatcher.class);
105         if (watcher != null) {
106             Set<CombatGroup> combatGroups = watcher.getBlockedOnlyByCreature(permanent.getId());
107             if (combatGroups != null) {
108                 for (CombatGroup combatGroup : combatGroups) {
109                     if (combatGroup != null) {
110                         combatGroup.setBlocked(false, game);
111                     }
112                 }
113             }
114         }
115 
116         if (!permanent.isCreature(game)
117                 || !controller.chooseUse(Outcome.Benefit, "Have " + permanent.getLogName() + " block an attacking creature?", source, game)) {
118             return false;
119         }
120         // Choose new creature to block
121 
122         // according to the following mail response from MTG Rules Management about False Orders:
123         // "if Player A attacks Players B and C, Player B's creatures cannot block creatures attacking Player C"
124         // therefore we need to single out creatures attacking the target blocker's controller (disappointing, I know)
125         List<Permanent> list = new ArrayList<>();
126         for (CombatGroup combatGroup : game.getCombat().getGroups()) {
127             if (combatGroup.getDefendingPlayerId().equals(permanent.getControllerId())) {
128                 for (UUID attackingCreatureId : combatGroup.getAttackers()) {
129                     Permanent targetsControllerAttacker = game.getPermanent(attackingCreatureId);
130                     list.add(targetsControllerAttacker);
131                 }
132             }
133         }
134         Player targetsController = game.getPlayer(permanent.getControllerId());
135         if (targetsController == null) {
136             return false;
137         }
138         FilterAttackingCreature filter = new FilterAttackingCreature("creature attacking " + targetsController.getLogName());
139         filter.add(new PermanentInListPredicate(list));
140         TargetPermanent target = new TargetPermanent(filter);
141         target.setNotTarget(true);
142         if (target.canChoose(controller.getId(), source, game)) {
143             while (!target.isChosen() && target.canChoose(controller.getId(), source, game) && controller.canRespond()) {
144                 controller.chooseTarget(outcome, target, source, game);
145             }
146         } else {
147             return true;
148         }
149         Permanent chosenPermanent = game.getPermanent(target.getFirstTarget());
150         if (chosenPermanent == null || !chosenPermanent.isCreature(game)) {
151             return false;
152         }
153         CombatGroup chosenGroup = game.getCombat().findGroup(chosenPermanent.getId());
154         if (chosenGroup != null) {
155             // Relevant ruling for Balduvian Warlord:
156             // 7/15/2006 	If an attacking creature has an ability that triggers “When this creature becomes blocked,” 
157             // it triggers when a creature blocks it due to the Warlord's ability only if it was unblocked at that point.
158             boolean notYetBlocked = chosenGroup.getBlockers().isEmpty();
159             chosenGroup.addBlockerToGroup(permanent.getId(), controller.getId(), game);
160             game.getCombat().addBlockingGroup(permanent.getId(), chosenPermanent.getId(), controller.getId(), game); // 702.21h
161             if (notYetBlocked) {
162                 game.fireEvent(GameEvent.getEvent(GameEvent.EventType.CREATURE_BLOCKED, chosenPermanent.getId(), source, null));
163                 Set<MageObjectReference> morSet = new HashSet<>();
164                 morSet.add(new MageObjectReference(chosenPermanent, game));
165                 for (UUID bandedId : chosenPermanent.getBandedCards()) {
166                     CombatGroup bandedGroup = game.getCombat().findGroup(bandedId);
167                     if (bandedGroup != null && chosenGroup.getBlockers().size() == 1) {
168                         morSet.add(new MageObjectReference(bandedId, game));
169                         game.fireEvent(GameEvent.getEvent(GameEvent.EventType.CREATURE_BLOCKED, bandedId, source, null));
170                     }
171                 }
172                 String key = UUID.randomUUID().toString();
173                 game.getState().setValue("becameBlocked_" + key, morSet);
174                 game.fireEvent(GameEvent.getEvent(
175                         GameEvent.EventType.BATCH_BLOCK_NONCOMBAT,
176                         source.getSourceId(), source,
177                         source.getControllerId(), key, 0)
178                 );
179             }
180             game.fireEvent(new BlockerDeclaredEvent(chosenPermanent.getId(), permanent.getId(), permanent.getControllerId()));
181             game.fireEvent(GameEvent.getEvent(GameEvent.EventType.CREATURE_BLOCKS, permanent.getId(), source, null));
182         }
183         CombatGroup blockGroup = findBlockingGroup(permanent, game); // a new blockingGroup is formed, so it's necessary to find it again
184         if (blockGroup != null) {
185             blockGroup.pickAttackerOrder(permanent.getControllerId(), game);
186         }
187         return true;
188     }
189 
190     private CombatGroup findBlockingGroup(Permanent blocker, Game game) {
191         if (game.getCombat().blockingGroupsContains(blocker.getId())) { // if (blocker.getBlocking() > 1) {
192             for (CombatGroup group : game.getCombat().getBlockingGroups()) {
193                 if (group.getBlockers().contains(blocker.getId())) {
194                     return group;
195                 }
196             }
197         }
198         return null;
199     }

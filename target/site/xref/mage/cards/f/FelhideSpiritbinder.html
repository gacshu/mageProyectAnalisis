31  public final class FelhideSpiritbinder extends CardImpl {
32  
33      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("another creature");
34  
35      static {
36          filter.add(AnotherPredicate.instance);
37      }
38  
39      public FelhideSpiritbinder(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{R}");
41          this.subtype.add(SubType.MINOTAUR);
42          this.subtype.add(SubType.SHAMAN);
43  
44          this.power = new MageInt(3);
45          this.toughness = new MageInt(4);
46  
47          // <i>Inspired</i> &mdash; Whenever Felhide Spiritbinder becomes untapped, you may pay {1}{R}. If you do, create a token that's a copy of another target creature except it's an enchantment in addition to its other types. It gains haste. Exile it at the beginning of the next end step.
48          Ability ability = new InspiredAbility(new DoIfCostPaid(new FelhideSpiritbinderEffect(), new ManaCostsImpl<>("{1}{R}"), "Use effect of {this}?"));
49          ability.addTarget(new TargetCreaturePermanent(filter));
50          this.addAbility(ability);
51      }
52  
53      private FelhideSpiritbinder(final FelhideSpiritbinder card) {
54          super(card);
55      }
56  
57      @Override
58      public FelhideSpiritbinder copy() {
59          return new FelhideSpiritbinder(this);
60      }
61  }
62  
63  class FelhideSpiritbinderEffect extends OneShotEffect {
64  
65      public FelhideSpiritbinderEffect() {
66          super(Outcome.PutCreatureInPlay);
67          this.staticText = "create a token that's a copy of another target creature except it's an enchantment in addition to its other types. It gains haste. Exile it at the beginning of the next end step";
68      }
69  
70      public FelhideSpiritbinderEffect(final FelhideSpiritbinderEffect effect) {
71          super(effect);
72      }
73  
74      @Override
75      public FelhideSpiritbinderEffect copy() {
76          return new FelhideSpiritbinderEffect(this);
77      }
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          Permanent permanent = getTargetPointer().getFirstTargetPermanentOrLKI(game, source);
82          if (permanent != null) {
83              CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect(null, CardType.ENCHANTMENT, true);
84              effect.setTargetPointer(getTargetPointer());
85              if (effect.apply(game, source)) {
86                  for (Permanent tokenPermanent : effect.getAddedPermanents()) {
87                      ExileTargetEffect exileEffect = new ExileTargetEffect();
88                      exileEffect.setTargetPointer(new FixedTarget(tokenPermanent, game));
89                      DelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(exileEffect);
90                      game.addDelayedTriggeredAbility(delayedAbility, source);
91                  }
92                  return true;
93              }
94          }
95  
96          return false;
97      }
98  }

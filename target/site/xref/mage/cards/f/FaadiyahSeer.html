27  public final class FaadiyahSeer extends CardImpl {
28  
29      public FaadiyahSeer(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{G}");
31          this.subtype.add(SubType.HUMAN);
32          this.subtype.add(SubType.SHAMAN);
33  
34          this.power = new MageInt(1);
35          this.toughness = new MageInt(1);
36  
37          // {tap}: Draw a card and reveal it. If it isn't a land card, discard it.
38          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new FaadiyahSeerEffect(), new TapSourceCost()));
39      }
40  
41      private FaadiyahSeer(final FaadiyahSeer card) {
42          super(card);
43      }
44  
45      @Override
46      public FaadiyahSeer copy() {
47          return new FaadiyahSeer(this);
48      }
49  }
50  
51  class FaadiyahSeerEffect extends OneShotEffect {
52  
53      private static final FilterCard filter = new FilterLandCard();
54  
55      public FaadiyahSeerEffect() {
56          super(Outcome.DrawCard);
57          this.staticText = "Draw a card and reveal it. If it isn't a land card, discard it";
58      }
59  
60      public FaadiyahSeerEffect(final FaadiyahSeerEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public FaadiyahSeerEffect copy() {
66          return new FaadiyahSeerEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Player controller = game.getPlayer(source.getControllerId());
72          if (controller == null) {
73              return false;
74          }
75          Card card = controller.getLibrary().getFromTop(game);
76          // Gatherer ruling (2007-02-01)
77          // If the draw is replaced by another effect, none of the rest of Fa’adiyah Seer’s ability applies,
78          // even if the draw is replaced by another draw (such as with Enduring Renewal).
79          if (controller.drawCards(1, source, game) != 1) {
80              return true;
81          }
82          controller.revealCards(source, new CardsImpl(card), game);
83          if (!card.isLand(game)) {
84              controller.discard(card, false, source, game);
85          }
86          return true;
87      }
88  }

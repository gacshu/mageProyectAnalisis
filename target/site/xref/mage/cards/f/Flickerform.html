37  public final class Flickerform extends CardImpl {
38  
39      public Flickerform(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}");
41          this.subtype.add(SubType.AURA);
42  
43          // Enchant creature
44          TargetPermanent auraTarget = new TargetCreaturePermanent();
45          this.getSpellAbility().addTarget(auraTarget);
46          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
47          Ability ability = new EnchantAbility(auraTarget);
48          this.addAbility(ability);
49  
50          // {2}{W}{W}: Exile enchanted creature and all Auras attached to it. At the beginning of the next end step, return that card to the battlefield under its owner's control. If you do, return the other cards exiled this way to the battlefield under their owners' control attached to that creature.
51          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new FlickerformEffect(), new ManaCostsImpl<>("{2}{W}{W}")));
52      }
53  
54      private Flickerform(final Flickerform card) {
55          super(card);
56      }
57  
58      @Override
59      public Flickerform copy() {
60          return new Flickerform(this);
61      }
62  }

64  class FlickerformEffect extends OneShotEffect {
65  
66      private static final FilterEnchantmentPermanent filter = new FilterEnchantmentPermanent();
67  
68      static {
69          filter.add(SubType.AURA.getPredicate());
70      }
71  
72      public FlickerformEffect() {
73          super(Outcome.Detriment);
74          this.staticText = "Exile enchanted creature and all Auras attached to it. At the beginning of the next end step, return that card to the battlefield under its owner's control. If you do, return the other cards exiled this way to the battlefield under their owners' control attached to that creature";
75      }
76  
77      public FlickerformEffect(final FlickerformEffect effect) {
78          super(effect);
79      }
80  
81      @Override
82      public FlickerformEffect copy() {
83          return new FlickerformEffect(this);
84      }
85  
86      @Override
87      public boolean apply(Game game, Ability source) {
88          // Exile enchanted creature and all Auras attached to it.
89          Permanent enchantment = game.getPermanent(source.getSourceId());
90          if (enchantment == null) {
91              enchantment = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
92          }
93          if (enchantment != null && enchantment.getAttachedTo() != null) {
94              Permanent enchantedCreature = game.getPermanent(enchantment.getAttachedTo());
95              if (enchantedCreature != null) {
96                  UUID exileZoneId = UUID.randomUUID();
97                  enchantedCreature.moveToExile(exileZoneId, enchantment.getName(), source, game);
98                  for (UUID attachementId : enchantedCreature.getAttachments()) {
99                      Permanent attachment = game.getPermanent(attachementId);
100                     if (attachment != null && filter.match(attachment, game)) {
101                         attachment.moveToExile(exileZoneId, enchantment.getName(), source, game);
102                     }
103                 }
104                 if (!(enchantedCreature instanceof Token)) {
105                     // At the beginning of the next end step, return that card to the battlefield under its owner's control.
106                     // If you do, return the other cards exiled this way to the battlefield under their owners' control attached to that creature
107                     AtTheBeginOfNextEndStepDelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
108                             new FlickerformReturnEffect(enchantedCreature.getId(), exileZoneId));
109                     game.addDelayedTriggeredAbility(delayedAbility, source);
110                 }
111                 return true;
112             }
113         }
114 
115         return false;
116     }
117 }

119 class FlickerformReturnEffect extends OneShotEffect {
120 
121     private static final FilterCard filterAura = new FilterCard();
122 
123     static {
124         filterAura.add(CardType.ENCHANTMENT.getPredicate());
125         filterAura.add(SubType.AURA.getPredicate());
126     }
127 
128     private final UUID enchantedCardId;
129     private final UUID exileZoneId;
130 
131     public FlickerformReturnEffect(UUID enchantedCardId, UUID exileZoneId) {
132         super(Outcome.Benefit);
133         this.enchantedCardId = enchantedCardId;
134         this.exileZoneId = exileZoneId;
135         this.staticText = "return that card to the battlefield under its owner's control. If you do, return the other cards exiled this way to the battlefield under their owners' control attached to that creature";
136     }
137 
138     public FlickerformReturnEffect(final FlickerformReturnEffect effect) {
139         super(effect);
140         this.enchantedCardId = effect.enchantedCardId;
141         this.exileZoneId = effect.exileZoneId;
142     }
143 
144     @Override
145     public FlickerformReturnEffect copy() {
146         return new FlickerformReturnEffect(this);
147     }
148 
149     @Override
150     public boolean apply(Game game, Ability source) {
151         Player controller = game.getPlayer(source.getControllerId());
152         if (controller == null) {
153             return false;
154         }
155         ExileZone exileZone = game.getExile().getExileZone(exileZoneId);
156         Card enchantedCard = exileZone.get(enchantedCardId, game);
157         //skip if exiled card is missing
158         if (enchantedCard != null) {
159             Player owner = game.getPlayer(enchantedCard.getOwnerId());
160             //skip if card's owner is missing
161             if (owner != null) {
162                 owner.moveCards(enchantedCard, Zone.BATTLEFIELD, source, game);
163                 Permanent newPermanent = game.getPermanent(enchantedCardId);
164                 if (newPermanent != null) {
165                     Set<Card> toBattlefieldAttached = new HashSet<Card>();
166                     for (Card enchantment : exileZone.getCards(game)) {
167                         if (filterAura.match(enchantment, game)) {
168                             boolean canTarget = false;
169                             for (Target target : enchantment.getSpellAbility().getTargets()) {
170                                 Filter filter = target.getFilter();
171                                 if (filter.match(newPermanent, game)) {
172                                     canTarget = true;
173                                     break;
174                                 }
175                             }
176                             if (!canTarget) {
177                                 // Aura stays exiled
178                                 continue;
179                             }
180                             game.getState().setValue("attachTo:" + enchantment.getId(), newPermanent);
181                         }
182                         toBattlefieldAttached.add(enchantment);
183                     }
184                     if (!toBattlefieldAttached.isEmpty()) {
185                         controller.moveCards(toBattlefieldAttached, Zone.BATTLEFIELD, source, game);
186                         for (Card card : toBattlefieldAttached) {
187                             if (game.getState().getZone(card.getId()) == Zone.BATTLEFIELD) {
188                                 newPermanent.addAttachment(card.getId(), source, game);
189                             }
190                         }
191                     }
192                 }
193                 return true;
194             }
195         }
196         return false;
197     }

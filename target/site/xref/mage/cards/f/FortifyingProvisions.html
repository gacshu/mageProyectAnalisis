18  public final class FortifyingProvisions extends CardImpl {
19  
20      public FortifyingProvisions(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
22  
23          // Creatures you control get +0/+1.
24          this.addAbility(new SimpleStaticAbility(new BoostControlledEffect(0, 1, Duration.WhileOnBattlefield)));
25  
26          // When Fortifying Provisions enters the battlefield, create a Food token.
27          this.addAbility(new EntersBattlefieldTriggeredAbility(new CreateTokenEffect(new FoodToken())));
28      }
29  
30      private FortifyingProvisions(final FortifyingProvisions card) {
31          super(card);
32      }
33  
34      @Override
35      public FortifyingProvisions copy() {
36          return new FortifyingProvisions(this);
37      }

9   import mage.abilities.keyword.HasteAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.SubType;

15  import mage.filter.StaticFilters;
16  import mage.target.common.TargetControlledPermanent;
17  import mage.target.common.TargetOpponentOrPlaneswalker;
18  
19  import java.util.UUID;
20  
21  /**
22   * @author TheElk801
23   */
24  public final class FirebladeArtist extends CardImpl {
25  
26      public FirebladeArtist(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{B}{R}");
28  
29          this.subtype.add(SubType.HUMAN);
30          this.subtype.add(SubType.SHAMAN);
31          this.power = new MageInt(2);
32          this.toughness = new MageInt(2);
33  
34          // Haste
35          this.addAbility(HasteAbility.getInstance());
36  
37          // At the beginning of your upkeep, you may sacrifice a creature. When you do, Fireblade Artist deals 2 damage to target opponent or planeswalker.
38          ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
39                  new DamageTargetEffect(2), false,
40                  "{this} deals 2 damage to target opponent or planeswalker"
41          );
42          ability.addTarget(new TargetOpponentOrPlaneswalker());
43          this.addAbility(new BeginningOfUpkeepTriggeredAbility(
44                  new DoWhenCostPaid(ability, new SacrificeTargetCost(
45                          new TargetControlledPermanent(StaticFilters.FILTER_CONTROLLED_CREATURE_SHORT_TEXT)
46                  ), "Sacrifice a creature?"), TargetController.YOU, false
47          ));
48      }
49  
50      private FirebladeArtist(final FirebladeArtist card) {
51          super(card);
52      }
53  
54      @Override
55      public FirebladeArtist copy() {
56          return new FirebladeArtist(this);
57      }
58  }

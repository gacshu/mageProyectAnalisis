3   import mage.ObjectColor;
4   import mage.abilities.condition.common.NotMyTurnCondition;
5   import mage.abilities.costs.AlternativeCostSourceAbility;
6   import mage.abilities.costs.common.ExileFromHandCost;
7   import mage.abilities.effects.common.DestroyTargetEffect;
8   import mage.abilities.hint.common.NotMyTurnHint;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  import mage.filter.FilterPermanent;
13  import mage.filter.common.FilterArtifactOrEnchantmentPermanent;
14  import mage.filter.common.FilterOwnedCard;
15  import mage.filter.predicate.mageobject.ColorPredicate;
16  import mage.target.TargetPermanent;
17  import mage.target.common.TargetCardInHand;
18  
19  import java.util.UUID;
20  
21  /**
22   * @author TheElk801
23   */
24  public final class ForceOfVigor extends CardImpl {
25  
26      private static final FilterOwnedCard filter
27              = new FilterOwnedCard("a green card from your hand");
28      private static final FilterPermanent filter2
29              = new FilterArtifactOrEnchantmentPermanent("artifacts and/or enchantments");
30  
31      static {
32          filter.add(new ColorPredicate(ObjectColor.GREEN));
33      }
34  
35      public ForceOfVigor(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{2}{G}{G}");
37  
38          // If it's not your turn, you may exile a green card from your hand rather than pay this spell's mana cost.
39          this.addAbility(new AlternativeCostSourceAbility(
40                  new ExileFromHandCost(new TargetCardInHand(filter)), NotMyTurnCondition.instance,
41                  "If it's not your turn, you may exile a green card from " +
42                          "your hand rather than pay this spell's mana cost."
43          ).addHint(NotMyTurnHint.instance));
44  
45          // Destroy up to two target artifacts and/or enchantments.
46          this.getSpellAbility().addEffect(new DestroyTargetEffect());
47          this.getSpellAbility().addTarget(new TargetPermanent(0, 2, filter2, false));
48      }
49  
50      private ForceOfVigor(final ForceOfVigor card) {
51          super(card);
52      }
53  
54      @Override
55      public ForceOfVigor copy() {
56          return new ForceOfVigor(this);
57      }
58  }

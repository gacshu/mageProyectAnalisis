25  public final class FoulEmissary extends CardImpl {
26  
27      public FoulEmissary(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{2}{G}");
29          this.subtype.add(SubType.HUMAN, SubType.HORROR);
30          this.power = new MageInt(1);
31          this.toughness = new MageInt(1);
32  
33          // When Foul Emissary enters the battlefield, look at the top four cards of your library. You may reveal a creature card from among them and put it into your hand. Put the rest on the bottom of your library in any order.
34          this.addAbility(new EntersBattlefieldTriggeredAbility(new LookLibraryAndPickControllerEffect(
35                  4, 1, StaticFilters.FILTER_CARD_CREATURE_A, PutCards.HAND, PutCards.BOTTOM_ANY)));
36  
37          // When you sacrifice Foul Emissary while casting a spell with emerge, create a 3/2 colorless Eldrazi Horror creature token.
38          this.addAbility(new FoulEmissaryTriggeredAbility());
39      }
40  
41      private FoulEmissary(final FoulEmissary card) {
42          super(card);
43      }
44  
45      @Override
46      public FoulEmissary copy() {
47          return new FoulEmissary(this);
48      }
49  }
50  
51  class FoulEmissaryTriggeredAbility extends SacrificeSourceTriggeredAbility {
52  
53      public FoulEmissaryTriggeredAbility() {
54          super(new CreateTokenEffect(new EldraziHorrorToken()), false);
55          setTriggerPhrase("When you sacrifice {this} while casting a spell with emerge, ");
56      }
57  
58      public FoulEmissaryTriggeredAbility(final FoulEmissaryTriggeredAbility ability) {
59          super(ability);
60      }
61  
62      @Override
63      public boolean checkTrigger(GameEvent event, Game game) {
64          if (super.checkTrigger(event, game)) {
65              Spell spell = game.getStack().getSpell(event.getSourceId());
66              return spell != null && spell.getSpellAbility() instanceof EmergeAbility;
67          }
68          return false;
69      }
70  
71      @Override
72      public FoulEmissaryTriggeredAbility copy() {
73          return new FoulEmissaryTriggeredAbility(this);
74      }
75  }

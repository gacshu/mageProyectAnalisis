1   package mage.cards.f;

3   import mage.ObjectColor;
4   import mage.abilities.costs.AlternativeCostSourceAbility;
5   import mage.abilities.costs.common.ExileFromHandCost;
6   import mage.abilities.costs.common.PayLifeCost;
7   import mage.abilities.effects.common.CounterTargetEffect;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.filter.common.FilterOwnedCard;
12  import mage.filter.predicate.mageobject.ColorPredicate;
13  import mage.target.TargetSpell;
14  import mage.target.common.TargetCardInHand;
15  
16  import java.util.UUID;
17  
18  /**
19   * @author Plopman
20   */
21  public final class ForceOfWill extends CardImpl {
22  
23      private static final FilterOwnedCard filter
24              = new FilterOwnedCard("a blue card from your hand");
25  
26      static {
27          filter.add(new ColorPredicate(ObjectColor.BLUE));
28      }
29  
30      public ForceOfWill(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{U}{U}");
32  
33          // You may pay 1 life and exile a blue card from your hand rather than pay Force of Will's mana cost.
34          AlternativeCostSourceAbility ability = new AlternativeCostSourceAbility(new PayLifeCost(1));
35          ability.addCost(new ExileFromHandCost(new TargetCardInHand(filter)));
36          this.addAbility(ability);
37  
38          // Counter target spell.
39          this.getSpellAbility().addEffect(new CounterTargetEffect());
40          this.getSpellAbility().addTarget(new TargetSpell());
41      }
42  
43      private ForceOfWill(final ForceOfWill card) {
44          super(card);
45      }
46  
47      @Override
48      public ForceOfWill copy() {
49          return new ForceOfWill(this);
50      }
51  }

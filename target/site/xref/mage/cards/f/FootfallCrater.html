27  public final class FootfallCrater extends CardImpl {
28  
29      public FootfallCrater(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{R}");
31  
32          this.subtype.add(SubType.AURA);
33  
34          // Enchant land
35          TargetPermanent auraTarget = new TargetLandPermanent();
36          this.getSpellAbility().addTarget(auraTarget);
37          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
38          Ability ability = new EnchantAbility(auraTarget);
39          this.addAbility(ability);
40  
41          // Enchanted land has "{T}: Target creature gains trample and haste until end of turn."
42          ability = new SimpleActivatedAbility(new GainAbilityTargetEffect(
43                  TrampleAbility.getInstance(), Duration.EndOfTurn
44          ).setText("Target creature gains trample"), new TapSourceCost());
45          ability.addEffect(new GainAbilityTargetEffect(
46                  HasteAbility.getInstance(), Duration.EndOfTurn
47          ).setText("and haste until end of turn"));
48          ability.addTarget(new TargetCreaturePermanent());
49          this.addAbility(new SimpleStaticAbility(new GainAbilityAttachedEffect(
50                  ability, AttachmentType.AURA, Duration.WhileOnBattlefield,
51                  "enchanted land has \"{T}: Target creature gains trample and haste until end of turn.\""
52          )));
53  
54          // Cycling {1}
55          this.addAbility(new CyclingAbility(new ManaCostsImpl<>("{1}")));
56      }
57  
58      private FootfallCrater(final FootfallCrater card) {
59          super(card);
60      }
61  
62      @Override
63      public FootfallCrater copy() {
64          return new FootfallCrater(this);
65      }
66  }

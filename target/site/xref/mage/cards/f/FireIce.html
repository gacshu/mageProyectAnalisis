14  public final class FireIce extends SplitCard {
15  
16      public FireIce(UUID ownerId, CardSetInfo setInfo) {
17          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{R}", "{1}{U}", SpellAbilityType.SPLIT);
18  
19          // Fire
20          // Fire deals 2 damage divided as you choose among one or two targets.
21          getLeftHalfCard().getSpellAbility().addEffect(new DamageMultiEffect(2, "Fire"));
22          getLeftHalfCard().getSpellAbility().addTarget(new TargetAnyTargetAmount(2));
23  
24          // Ice
25          // Tap target permanent.
26          // Draw a card.
27          getRightHalfCard().getSpellAbility().addEffect(new TapTargetEffect());
28          getRightHalfCard().getSpellAbility().addTarget(new TargetPermanent());
29          getRightHalfCard().getSpellAbility().addEffect(new DrawCardSourceControllerEffect(1));
30  
31      }
32  
33      private FireIce(final FireIce card) {
34          super(card);
35      }
36  
37      @Override
38      public FireIce copy() {
39          return new FireIce(this);
40      }

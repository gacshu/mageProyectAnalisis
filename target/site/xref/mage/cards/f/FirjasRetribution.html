32  public final class FirjasRetribution extends CardImpl {
33  
34      private static final FilterCreaturePermanent filter
35              = new FilterCreaturePermanent("creature with power less than this creature's power");
36      private static final FilterControlledPermanent filter2
37              = new FilterControlledPermanent(SubType.ANGEL, "Angels you control");
38  
39      static {
40          filter.add(FirjasRetributionPredicate.instance);
41      }
42  
43      public FirjasRetribution(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}{W}{B}");
45  
46          this.subtype.add(SubType.SAGA);
47  
48          // (As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)
49          SagaAbility sagaAbility = new SagaAbility(this);
50  
51          // I — Create a 4/4 white Angel Warrior creature token with flying and vigilance.
52          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, new CreateTokenEffect(new AngelWarriorVigilanceToken()));
53  
54          // II — Until end of turn, Angels you control gain "{T}: Destroy target creature with less power than this creature."
55          Ability ability = new SimpleActivatedAbility(new DestroyTargetEffect(), new TapSourceCost());
56          ability.addTarget(new TargetCreaturePermanent(filter));
57          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_II, new GainAbilityAllEffect(
58                  ability, Duration.EndOfTurn, filter2
59          ).setText("until end of turn, Angels you control gain \"{T}: Destroy target creature with power less than this creature's power.\""));
60  
61          // III — Angels you control gain double strike until end of turn.
62          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new GainAbilityAllEffect(
63                  DoubleStrikeAbility.getInstance(), Duration.EndOfTurn, filter2
64          ));
65          this.addAbility(sagaAbility);
66      }
67  
68      private FirjasRetribution(final FirjasRetribution card) {
69          super(card);
70      }
71  
72      @Override
73      public FirjasRetribution copy() {
74          return new FirjasRetribution(this);
75      }
76  }
77  
78  enum FirjasRetributionPredicate implements ObjectSourcePlayerPredicate<Permanent> {
79      instance;
80  
81      @Override
82      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
83          Permanent permanent = input.getSource().getSourcePermanentOrLKI(game);
84          return permanent != null && input.getObject().getPower().getValue() < permanent.getPower().getValue();
85      }
86  }

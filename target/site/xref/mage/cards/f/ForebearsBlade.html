28  public final class ForebearsBlade extends CardImpl {
29  
30      public ForebearsBlade(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
32          this.subtype.add(SubType.EQUIPMENT);
33  
34          // Equipped creature gets +3/+0
35          Effect effect = new BoostEquippedEffect(3, 0);
36          effect.setText("Equipped creature gets +3/+0");
37          Ability ability = new SimpleStaticAbility(Zone.BATTLEFIELD, effect);
38          // and has vigilance
39          effect = new GainAbilityAttachedEffect(VigilanceAbility.getInstance(), AttachmentType.EQUIPMENT);
40          effect.setText("and has vigilance");
41          ability.addEffect(effect);
42          // and trample
43          effect = new GainAbilityAttachedEffect(TrampleAbility.getInstance(), AttachmentType.EQUIPMENT);
44          effect.setText("and trample");
45          ability.addEffect(effect);
46          this.addAbility(ability);
47  
48          // Whenever equipped creature dies, attach Forebear's Blade to target creature you control.
49          ability = new DiesAttachedTriggeredAbility(
50                  new AttachEffect(Outcome.Neutral, "attach {this} to target creature you control"), "equipped creature", false);
51          ability.addTarget(new TargetControlledCreaturePermanent());
52          this.addAbility(ability);
53  
54          // Equip {3}
55          this.addAbility(new EquipAbility(Outcome.AddAbility, new GenericManaCost(3), false));
56      }
57  
58      private ForebearsBlade(final ForebearsBlade card) {
59          super(card);
60      }
61  
62      @Override
63      public ForebearsBlade copy() {
64          return new ForebearsBlade(this);
65      }
66  }

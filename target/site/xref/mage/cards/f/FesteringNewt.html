5   import mage.MageInt;

7   import mage.abilities.common.DiesSourceTriggeredAbility;
8   import mage.abilities.condition.LockedInCondition;
9   import mage.abilities.condition.common.PermanentsOnTheBattlefieldCondition;
10  import mage.abilities.decorator.ConditionalContinuousEffect;
11  import mage.abilities.effects.Effect;
12  import mage.abilities.effects.common.continuous.BoostTargetEffect;
13  import mage.cards.CardImpl;
14  import mage.cards.CardSetInfo;
15  import mage.constants.CardType;
16  import mage.constants.SubType;
17  import mage.constants.Duration;
18  import mage.filter.StaticFilters;
19  import mage.filter.common.FilterCreaturePermanent;
20  import mage.filter.predicate.mageobject.NamePredicate;
21  import mage.target.common.TargetCreaturePermanent;
22  
23  /**
24   *
25   * @author LevelX2
26   */
27  public final class FesteringNewt extends CardImpl {
28  
29      private static final FilterCreaturePermanent filterBogbrewWitch = new FilterCreaturePermanent();
30  
31      static {
32          filterBogbrewWitch.add(new NamePredicate("Bogbrew Witch"));
33      }
34  
35      public FesteringNewt(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{B}");
37          this.subtype.add(SubType.SALAMANDER);
38  
39          this.power = new MageInt(1);
40          this.toughness = new MageInt(1);
41  
42          // When Festering Newt dies, target creature an opponent controls gets -1/-1 until end of turn. That creature gets -4/-4 instead if you control a creature named Bogbrew Witch.
43          Effect effect = new ConditionalContinuousEffect(
44                  new BoostTargetEffect(-4,-4, Duration.EndOfTurn), 
45                  new BoostTargetEffect(-1,-1, Duration.EndOfTurn),
46                  new LockedInCondition(new PermanentsOnTheBattlefieldCondition(filterBogbrewWitch)),
47                  "target creature an opponent controls gets -1/-1 until end of turn. That creature gets -4/-4 instead if you control a creature named Bogbrew Witch");
48          Ability ability = new DiesSourceTriggeredAbility(effect);
49          ability.addTarget(new TargetCreaturePermanent(StaticFilters.FILTER_OPPONENTS_PERMANENT_CREATURE));
50          this.addAbility(ability);
51      }
52  
53      private FesteringNewt(final FesteringNewt card) {
54          super(card);
55      }
56  
57      @Override
58      public FesteringNewt copy() {
59          return new FesteringNewt(this);
60      }
61  }

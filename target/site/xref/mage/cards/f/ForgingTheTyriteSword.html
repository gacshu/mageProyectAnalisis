10  import mage.constants.SubType;

14  import mage.game.permanent.token.TreasureToken;
15  import mage.target.common.TargetCardInLibrary;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class ForgingTheTyriteSword extends CardImpl {
23  
24      private static final FilterCard filter
25              = new FilterCard("card named Halvar, God of Battle or an Equipment card");
26  
27      static {
28          filter.add(Predicates.or(
29                  new NamePredicate("Halvar, God of Battle"),
30                  SubType.EQUIPMENT.getPredicate()
31          ));
32      }
33  
34      public ForgingTheTyriteSword(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{R}{W}");
36  
37          this.subtype.add(SubType.SAGA);
38  
39          // (As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)
40          SagaAbility sagaAbility = new SagaAbility(this);
41  
42          // I, II — Create a Treasure token.
43          sagaAbility.addChapterEffect(
44                  this, SagaChapter.CHAPTER_I, SagaChapter.CHAPTER_II,
45                  new CreateTokenEffect(new TreasureToken())
46          );
47  
48          // III — Search your library for a card named Halvar, God of Battle or an Equipment card, reveal it, put it into your hand, then shuffle your library.
49          sagaAbility.addChapterEffect(
50                  this, SagaChapter.CHAPTER_III,
51                  new SearchLibraryPutInHandEffect(
52                          new TargetCardInLibrary(filter), true
53                  )
54          );
55  
56          this.addAbility(sagaAbility);
57      }
58  
59      private ForgingTheTyriteSword(final ForgingTheTyriteSword card) {
60          super(card);
61      }
62  
63      @Override
64      public ForgingTheTyriteSword copy() {
65          return new ForgingTheTyriteSword(this);
66      }
67  }

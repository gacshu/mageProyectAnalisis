3   import mage.abilities.common.PutIntoGraveFromBattlefieldSourceTriggeredAbility;
4   import mage.abilities.effects.common.search.SearchLibraryPutInPlayEffect;
5   import mage.abilities.mana.WhiteManaAbility;
6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.constants.SubType;
10  import mage.constants.SuperType;
11  import mage.filter.common.FilterLandCard;
12  import mage.target.common.TargetCardInLibrary;
13  
14  import java.util.UUID;
15  
16  /**
17   * @author LevelX2
18   */
19  public final class FlagstonesOfTrokair extends CardImpl {
20  
21      private static final FilterLandCard FILTER = new FilterLandCard("Plains card");
22  
23      static {
24          FILTER.add(SubType.PLAINS.getPredicate());
25      }
26  
27      public FlagstonesOfTrokair(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
29          addSuperType(SuperType.LEGENDARY);
30  
31          // {tap}: Add {W}.
32          this.addAbility(new WhiteManaAbility());
33  
34          // When Flagstones of Trokair is put into a graveyard from the battlefield, you may search your library for a Plains card and put it onto the battlefield tapped. If you do, shuffle your library.
35          this.addAbility(new PutIntoGraveFromBattlefieldSourceTriggeredAbility(new SearchLibraryPutInPlayEffect(new TargetCardInLibrary(FILTER), true, true), true, false));
36      }
37  
38      private FlagstonesOfTrokair(final FlagstonesOfTrokair card) {
39          super(card);
40      }
41  
42      @Override
43      public FlagstonesOfTrokair copy() {
44          return new FlagstonesOfTrokair(this);
45      }
46  }

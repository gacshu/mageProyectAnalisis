21  /**
22   * @author TheElk801
23   */
24  public final class KeyToTheArchive extends CardImpl {
25  
26      private static final List<String> spellbook = Collections.unmodifiableList(Arrays.asList(
27              "Approach of the Second Sun",
28              "Claim the Firstborn",
29              "Counterspell",
30              "Day of Judgment",
31              "Demonic Tutor",
32              "Despark",
33              "Doom Blade",
34              "Electrolyze",
35              "Growth Spiral",
36              "Krosan Grip",
37              "Lightning Bolt",
38              "Lightning Helix",
39              "Putrefy",
40              "Regrowth",
41              "Time Warp"
42      ));
43  
44      public KeyToTheArchive(UUID ownerId, CardSetInfo setInfo) {
45          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{4}");
46  
47          // Key to the Archive enters the battlefield tapped.
48          this.addAbility(new EntersBattlefieldTappedAbility());
49  
50          // When Key to the Archive enters the battlefield, draft a card from Key to the Archive's spellbook, then discard a card.
51          Ability ability = new EntersBattlefieldTriggeredAbility(new DraftFromSpellbookEffect(spellbook));
52          ability.addEffect(new DiscardControllerEffect(1).concatBy(", then"));
53          this.addAbility(ability);
54  
55          // {T}: Add two mana in any combination of colors.
56          this.addAbility(new SimpleManaAbility(
57                  Zone.BATTLEFIELD, new AddManaInAnyCombinationEffect(2), new TapSourceCost()
58          ));
59      }
60  
61      private KeyToTheArchive(final KeyToTheArchive card) {
62          super(card);
63      }
64  
65      @Override
66      public KeyToTheArchive copy() {
67          return new KeyToTheArchive(this);
68      }
69  }

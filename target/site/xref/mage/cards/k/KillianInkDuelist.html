22  public final class KillianInkDuelist extends CardImpl {

24      public KillianInkDuelist(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{W}{B}");
26  
27          this.addSuperType(SuperType.LEGENDARY);
28          this.subtype.add(SubType.HUMAN);
29          this.subtype.add(SubType.WARLOCK);
30          this.power = new MageInt(2);
31          this.toughness = new MageInt(2);
32  
33          // Lifelink
34          this.addAbility(LifelinkAbility.getInstance());
35  
36          // Menace
37          this.addAbility(new MenaceAbility());
38  
39          // Spells you cast that target a creature cost {2} less to cast.
40          this.addAbility(new SimpleStaticAbility(new KillianInkDuelistEffect()));
41      }
42  
43      private KillianInkDuelist(final KillianInkDuelist card) {
44          super(card);
45      }
46  
47      @Override
48      public KillianInkDuelist copy() {
49          return new KillianInkDuelist(this);
50      }
51  }

53  class KillianInkDuelistEffect extends CostModificationEffectImpl {
54  
55      KillianInkDuelistEffect() {
56          super(Duration.WhileOnBattlefield, Outcome.Neutral, CostModificationType.REDUCE_COST);
57          staticText = "spells you cast that target a creature cost {2} less to cast";
58      }
59  
60      private KillianInkDuelistEffect(KillianInkDuelistEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public KillianInkDuelistEffect copy() {
66          return new KillianInkDuelistEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source, Ability abilityToModify) {
71          // Bug #7762: https://github.com/magefree/mage/issues/7762
72          // Check possible targets for getPlayable
73          if (game.inCheckPlayableState()) {
74              if (CardUtil.getAllPossibleTargets(abilityToModify, game)
75                      .stream()
76                      .map(game::getPermanent)
77                      .filter(Objects::nonNull)
78                      .anyMatch(permanent -> permanent.isCreature(game))) {
79                  CardUtil.reduceCost(abilityToModify, 2);
80              }
81          // Check selected targets on actual cast
82          } else if (CardUtil.getAllSelectedTargets(abilityToModify, game)
83                  .stream()
84                  .map(game::getPermanent)
85                  .filter(Objects::nonNull)
86                  .anyMatch(permanent -> permanent.isCreature(game))) {
87              CardUtil.reduceCost(abilityToModify, 2);
88          }
89          return true;
90      }
91  
92      @Override
93      public boolean applies(Ability abilityToModify, Ability source, Game game) {
94          return abilityToModify instanceof SpellAbility
95                  && abilityToModify.isControlledBy(source.getControllerId());
96      }
97  }

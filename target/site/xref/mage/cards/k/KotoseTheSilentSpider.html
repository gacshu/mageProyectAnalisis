31  public final class KotoseTheSilentSpider extends CardImpl {
32  
33      private static final FilterCard filter
34              = new FilterCard("card other than a basic land card in an opponent's graveyard");
35  
36      static {
37          filter.add(Predicates.not(Predicates.and(
38                  CardType.LAND.getPredicate(),
39                  SuperType.BASIC.getPredicate()
40          )));
41      }
42  
43      public KotoseTheSilentSpider(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}{B}");
45          this.addSuperType(SuperType.LEGENDARY);
46          this.subtype.add(SubType.HUMAN);
47          this.subtype.add(SubType.NINJA);
48  
49          this.power = new MageInt(4);
50          this.toughness = new MageInt(4);
51  
52          // When Kotose, Silent Spider enters the battlefield, exile target card in an opponent's graveyard other than a basic land card. Search that player's graveyard, hand, and library for any number of cards with the same name as that card and exile them. For as long as you control Kotose, you may play one of the exiled cards, and you may spend mana as though it were mana of any color to cast it.
53          Ability ability = new EntersBattlefieldTriggeredAbility(new KotoseTheSilentSpiderEffect());
54          ability.addTarget(new TargetCardInOpponentsGraveyard(filter));
55          this.addAbility(ability, new KotoseTheSilentSpiderWatcher());
56      }
57  
58      private KotoseTheSilentSpider(final KotoseTheSilentSpider card) {
59          super(card);
60      }
61  
62      @Override
63      public KotoseTheSilentSpider copy() {
64          return new KotoseTheSilentSpider(this);
65      }
66  }

68  class KotoseTheSilentSpiderEffect extends OneShotEffect {
69  
70      public KotoseTheSilentSpiderEffect() {
71          super(Outcome.Exile);
72          this.staticText = "exile target card other than a basic land card from an opponent's graveyard. " +
73                  "Search that player's graveyard, hand, and library for any number of cards with the same name " +
74                  "as that card and exile them. Then that player shuffles. For as long as you control {this}, you may " +
75                  "play one of the exiled cards, and you may spend mana as though it were mana of any color to cast it";
76      }
77  
78      public KotoseTheSilentSpiderEffect(final KotoseTheSilentSpiderEffect effect) {
79          super(effect);
80      }
81  
82      @Override
83      public KotoseTheSilentSpiderEffect copy() {
84          return new KotoseTheSilentSpiderEffect(this);
85      }
86  
87      @Override
88      public boolean apply(Game game, Ability source) {
89          Player controller = game.getPlayer(source.getControllerId());
90          Card card = game.getCard(getTargetPointer().getFirst(game, source));
91          if (controller == null || card == null) {
92              return false;
93          }
94          Player opponent = game.getPlayer(card.getOwnerId());
95          if (opponent == null) {
96              return false;
97          }
98          UUID exileId = CardUtil.getExileZoneId(game, source);
99          String exileName = CardUtil.getSourceName(game, source);
100         controller.moveCardsToExile(card, source, game, true, exileId, exileName);
101         Cards cards = new CardsImpl();
102         FilterCard filter = new FilterCard("cards named " + card.getName() + " from " + opponent.getName() + "'s graveyard");
103         filter.add(new NamePredicate(card.getName()));
104 
105         TargetCardInGraveyard targetCardInGraveyard = new TargetCardInGraveyard(0, Integer.MAX_VALUE, filter);
106         controller.choose(outcome, opponent.getGraveyard(), targetCardInGraveyard, game);
107         cards.addAll(targetCardInGraveyard.getTargets());
108 
109         filter.setMessage("cards named " + card.getName() + " from " + opponent.getName() + "'s hand");
110         TargetCardInHand targetCardInHand = new TargetCardInHand(0, Integer.MAX_VALUE, filter);
111         controller.choose(outcome, opponent.getHand(), targetCardInHand, game);
112         cards.addAll(targetCardInHand.getTargets());
113 
114         filter.setMessage("cards named " + card.getName() + " from " + opponent.getName() + "'s library");
115         TargetCardInLibrary target = new TargetCardInLibrary(0, Integer.MAX_VALUE, filter);
116         controller.searchLibrary(target, source, game, opponent.getId());
117         target.getTargets()
118                 .stream()
119                 .map(cardId -> opponent.getLibrary().getCard(cardId, game))
120                 .forEach(cards::add);
121 
122         Set<Card> cardSet = cards.getCards(game);
123         controller.moveCardsToExile(cardSet, source, game, true, exileId, exileName);
124         opponent.shuffleLibrary(source, game);
125         cardSet.add(card);
126         if (cardSet.isEmpty() || source.getSourcePermanentIfItStillExists(game) == null) {
127             return true;
128         }
129         KotoseTheSilentSpiderWatcher.addCards(source, cardSet, game);
130         for (Card exiledCard : cardSet) {
131             CardUtil.makeCardPlayable(
132                     game, source, exiledCard, Duration.WhileControlled, true,
133                     null, new KotoseTheSilentSpiderCondition(exiledCard, game)
134             );
135         }
136         return true;
137     }
138 }

140 class KotoseTheSilentSpiderCondition implements Condition {
141 
142     private final MageObjectReference mor;
143 
144     KotoseTheSilentSpiderCondition(Card card, Game game) {
145         this.mor = new MageObjectReference(card, game);
146     }
147 
148     @Override
149     public boolean apply(Game game, Ability source) {
150         return KotoseTheSilentSpiderWatcher.checkCard(game, source, mor);
151     }
152 }
153 
154 class KotoseTheSilentSpiderWatcher extends Watcher {
155 
156     private final Map<MageObjectReference, Set<Set<MageObjectReference>>> morMap = new HashMap<>();
157 
158     public KotoseTheSilentSpiderWatcher() {
159         super(WatcherScope.GAME);
160     }
161 
162     @Override
163     public void watch(GameEvent event, Game game) {
164         if (event.getType() == GameEvent.EventType.CLEANUP_STEP_POST) {
165             morMap.entrySet().removeIf(e -> !e.getKey().zoneCounterIsCurrent(game));
166             morMap.values()
167                     .stream()
168                     .flatMap(Collection::stream)
169                     .forEach(set -> set.removeIf(mor -> !mor.zoneCounterIsCurrent(game)));
170             morMap.values().removeIf(Set::isEmpty);
171             return;
172         }
173         if (event.getType() != GameEvent.EventType.SPELL_CAST || event.getAdditionalReference() == null) {
174             return;
175         }
176         Spell spell = game.getSpell(event.getTargetId());
177         if (spell == null) {
178             return;
179         }
180         morMap.getOrDefault(
181                 event.getAdditionalReference().getApprovingMageObjectReference(), Collections.emptySet()
182         ).removeIf(set -> set
183                 .stream()
184                 .anyMatch(mor -> mor.getSourceId().equals(spell.getMainCard().getId())
185                         && mor.getZoneChangeCounter() + 1 == spell.getZoneChangeCounter(game)));
186     }
187 
188     static void addCards(Ability source, Set<Card> cards, Game game) {
189         game.getState()
190                 .getWatcher(KotoseTheSilentSpiderWatcher.class)
191                 .morMap
192                 .computeIfAbsent(new MageObjectReference(source), x -> new HashSet<>())
193                 .add(cards
194                         .stream()
195                         .map(card -> new MageObjectReference(card, game))
196                         .collect(Collectors.toSet()));
197     }
198 
199     static boolean checkCard(Game game, Ability source, MageObjectReference mor) {
200         return game.getState()
201                 .getWatcher(KotoseTheSilentSpiderWatcher.class)
202                 .morMap
203                 .getOrDefault(new MageObjectReference(source), Collections.emptySet())
204                 .stream()
205                 .flatMap(Collection::stream)
206                 .anyMatch(mor::equals);
207     }
208 }

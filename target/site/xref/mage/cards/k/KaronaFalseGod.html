29  public final class KaronaFalseGod extends CardImpl {

31      public KaronaFalseGod(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{W}{U}{B}{R}{G}");
33          addSuperType(SuperType.LEGENDARY);
34          this.subtype.add(SubType.AVATAR);
35  
36          this.power = new MageInt(5);
37          this.toughness = new MageInt(5);
38  
39          // Haste
40          this.addAbility(HasteAbility.getInstance());
41  
42          // At the beginning of each player's upkeep, that player untaps Karona, False God and gains control of it.
43          this.addAbility(new BeginningOfUpkeepTriggeredAbility(Zone.BATTLEFIELD, new KaronaFalseGodUntapGetControlEffect(), TargetController.ANY, false, true));
44  
45          // Whenever Karona attacks, creatures of the creature type of your choice get +3/+3 until end of turn.
46          this.addAbility(new AttacksTriggeredAbility(new KaronaFalseGodEffect(), false));
47      }

50          super(card);

59  class KaronaFalseGodUntapGetControlEffect extends OneShotEffect {
60  
61      public KaronaFalseGodUntapGetControlEffect() {
62          super(Outcome.GainControl);
63          this.staticText = "that player untaps {this} and gains control of it";
64      }
65  
66      public KaronaFalseGodUntapGetControlEffect(final KaronaFalseGodUntapGetControlEffect effect) {
67          super(effect);
68      }
69  
70      @Override
71      public KaronaFalseGodUntapGetControlEffect copy() {
72          return new KaronaFalseGodUntapGetControlEffect(this);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          Player controller = game.getPlayer(source.getControllerId());
78          MageObject sourceObject = source.getSourceObject(game);
79          Permanent sourcePermanent = game.getPermanent(source.getSourceId());
80          Player newController = game.getPlayer(getTargetPointer().getFirst(game, source));
81          if (newController != null && controller != null && sourceObject != null && sourceObject.equals(sourcePermanent)) {
82              sourcePermanent.untap(game);
83              game.informPlayers(newController.getLogName() + " untaps " + sourceObject.getIdName());
84              // remove old control effects of the same player
85              for (ContinuousEffect effect : game.getState().getContinuousEffects().getLayeredEffects(game)) {
86                  if (effect instanceof GainControlTargetEffect) {
87                      UUID checkId = (UUID) effect.getValue("KaronaFalseGodSourceId");
88                      UUID controllerId = (UUID) effect.getValue("KaronaFalseGodControllerId");
89                      if (source.getSourceId().equals(checkId) && newController.getId().equals(controllerId)) {
90                          effect.discard();
91                      }
92                  }
93              }
94              ContinuousEffect effect = new GainControlTargetEffect(Duration.Custom, true, newController.getId());
95              effect.setValue("KaronaFalseGodSourceId", source.getSourceId());
96              effect.setValue("KaronaFalseGodControllerId", newController.getId());
97              effect.setTargetPointer(new FixedTarget(sourcePermanent.getId(), game));
98              effect.setText("and gains control of it");
99              game.addEffect(effect, source);
100             return true;
101         }
102         return false;
103     }
104 }

106 class KaronaFalseGodEffect extends OneShotEffect {
107 
108     public KaronaFalseGodEffect() {
109         super(Outcome.BoostCreature);
110         this.staticText = "creatures of the creature type of your choice get +3/+3 until end of turn";
111     }
112 
113     public KaronaFalseGodEffect(final KaronaFalseGodEffect effect) {
114         super(effect);
115     }
116 
117     @Override
118     public KaronaFalseGodEffect copy() {
119         return new KaronaFalseGodEffect(this);
120     }
121 
122     @Override
123     public boolean apply(Game game, Ability source) {
124         Player controller = game.getPlayer(source.getControllerId());
125         MageObject sourceObject = game.getObject(source);
126         if (sourceObject != null && controller != null) {
127             Choice typeChoice = new ChoiceCreatureType(sourceObject);
128             if (!controller.choose(Outcome.BoostCreature, typeChoice, game)) {
129                 return false;
130             }
131             String typeChosen = typeChoice.getChoice();
132             if (!typeChosen.isEmpty()) {
133                 game.informPlayers(controller.getLogName() + " has chosen " + typeChosen);
134                 FilterCreaturePermanent filter = new FilterCreaturePermanent();
135                 filter.add(SubType.byDescription(typeChosen).getPredicate());
136                 game.addEffect(new BoostAllEffect(3, 3, Duration.EndOfTurn, filter, false), source);
137             }
138             return true;
139         }
140         return false;
141     }

32  public final class KeldonStrikeTeam extends CardImpl {
33  
34      public KeldonStrikeTeam(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}");
36  
37          this.subtype.add(SubType.HUMAN);
38          this.subtype.add(SubType.WARRIOR);
39          this.power = new MageInt(3);
40          this.toughness = new MageInt(1);
41  
42          // Kicker {1}{W}
43          this.addAbility(new KickerAbility("{1}{W}"));
44  
45          // When Keldon Strike Team enters the battlefield, if it was kicked, create two 1/1 white Soldier creature tokens.
46          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
47                  new EntersBattlefieldTriggeredAbility(new CreateTokenEffect(new SoldierToken(), 2)),
48                  KickedCondition.ONCE, "When {this} enters the battlefield, if it was kicked, " +
49                  "create two 1/1 white Soldier creature tokens."
50          ));
51  
52          // As long as Keldon Strike Team entered the battlefield this turn, creatures you control have haste.
53          this.addAbility(new SimpleStaticAbility(new ConditionalContinuousEffect(
54                  new GainAbilityControlledEffect(
55                          HasteAbility.getInstance(), Duration.WhileOnBattlefield,
56                          StaticFilters.FILTER_CONTROLLED_CREATURE
57                  ), KeldonStrikeTeamCondition.instance, "as long as {this} " +
58                  "entered the battlefield this turn, creatures you control have haste"
59          )));
60      }
61  
62      private KeldonStrikeTeam(final KeldonStrikeTeam card) {
63          super(card);
64      }
65  
66      @Override
67      public KeldonStrikeTeam copy() {
68          return new KeldonStrikeTeam(this);
69      }
70  }
71  
72  enum KeldonStrikeTeamCondition implements Condition {
73      instance;
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          return Optional.ofNullable(source.getSourcePermanentIfItStillExists(game))
78                  .filter(Objects::nonNull)
79                  .map(Permanent::getTurnsOnBattlefield)
80                  .orElse(-1) == 0;
81      }
82  }

27  public final class KefnetTheMindful extends CardImpl {

29      public KefnetTheMindful(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
31  
32          this.addSuperType(SuperType.LEGENDARY);
33          this.subtype.add(SubType.GOD);
34          this.power = new MageInt(5);
35          this.toughness = new MageInt(5);
36  
37          // Flying
38          this.addAbility(FlyingAbility.getInstance());
39  
40          // Indestructible
41          this.addAbility(IndestructibleAbility.getInstance());
42  
43          // Kefnet the Mindful can't attack or block unless you have seven or more cards in your hand.
44          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new KefnetTheMindfulRestrictionEffect()));
45  
46          // {3}{U}: Draw a card, then you may return a land you control to its owner's hand.
47          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new KefnetTheMindfulEffect(), new ManaCostsImpl<>("{3}{U}"));
48          this.addAbility(ability);
49  
50      }

62  class KefnetTheMindfulRestrictionEffect extends RestrictionEffect {
63  
64      public KefnetTheMindfulRestrictionEffect() {
65          super(Duration.WhileOnBattlefield);
66          staticText = "{this} can't attack or block unless you have seven or more cards in your hand";
67      }
68  
69      public KefnetTheMindfulRestrictionEffect(final KefnetTheMindfulRestrictionEffect effect) {
70          super(effect);
71      }
72  
73      @Override
74      public KefnetTheMindfulRestrictionEffect copy() {
75          return new KefnetTheMindfulRestrictionEffect(this);
76      }
77  
78      @Override
79      public boolean canAttack(Game game, boolean canUseChooseDialogs) {
80          return false;
81      }
82  
83      @Override
84      public boolean canBlock(Permanent attacker, Permanent blocker, Ability source, Game game, boolean canUseChooseDialogs) {
85          return false;
86      }
87  
88      @Override
89      public boolean applies(Permanent permanent, Ability source, Game game) {
90          if (permanent.getId().equals(source.getSourceId())) {
91              Player controller = game.getPlayer(source.getControllerId());
92              if (controller != null) {
93                  return (controller.getHand().size() < 7);
94              }
95              return true;
96          }  // do not apply to other creatures.
97          return false;
98      }
99  }

101 class KefnetTheMindfulEffect extends OneShotEffect {
102 
103     KefnetTheMindfulEffect() {
104         super(Outcome.Benefit);
105         staticText = "Draw a card, then you may return a land you control to its owner's hand";
106     }
107 
108     KefnetTheMindfulEffect(final KefnetTheMindfulEffect effect) {
109         super(effect);
110     }
111 
112     @Override
113     public boolean apply(Game game, Ability source) {
114         FilterControlledPermanent filterControlledLand = new FilterControlledPermanent("land you control");
115         filterControlledLand.add(CardType.LAND.getPredicate());
116         Player controller = game.getPlayer(source.getControllerId());
117         if (controller != null) {
118             controller.drawCards(1, source, game);
119             if (controller.chooseUse(Outcome.AIDontUseIt, "Return a land you control to its owner's hand?", source, game)) {
120                 Effect effect = new ReturnToHandChosenControlledPermanentEffect(filterControlledLand);
121                 effect.apply(game, source);
122             }
123             return true;
124         }
125         return false;
126     }
127 
128     @Override
129     public KefnetTheMindfulEffect copy() {
130         return new KefnetTheMindfulEffect(this);
131     }
132 }

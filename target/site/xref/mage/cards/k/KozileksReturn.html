4   import java.util.UUID;
5   import mage.abilities.common.SpellCastControllerTriggeredAbility;
6   import mage.abilities.costs.common.ExileSourceFromGraveCost;
7   import mage.abilities.effects.common.DamageAllEffect;
8   import mage.abilities.effects.common.DoIfCostPaid;
9   import mage.abilities.keyword.DevoidAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.ComparisonType;
14  import mage.constants.SubType;
15  import mage.constants.Zone;
16  import mage.filter.FilterSpell;
17  import mage.filter.common.FilterCreaturePermanent;
18  import mage.filter.predicate.mageobject.ManaValuePredicate;
19  
20  /**
21   *
22   * @author fireshoes
23   */
24  public final class KozileksReturn extends CardImpl {
25      
26      private static final FilterSpell filter = new FilterSpell("an Eldrazi creature spell with mana value 7 or greater");
27  
28      static {
29          filter.add(SubType.ELDRAZI.getPredicate());
30          filter.add(CardType.CREATURE.getPredicate());
31          filter.add(new ManaValuePredicate(ComparisonType.MORE_THAN, 6));
32      }
33  
34      public KozileksReturn(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{2}{R}");
36  
37          // Devoid
38          this.addAbility(new DevoidAbility(this.color));
39          
40          // Kozilek's Return deals 2 damage to each creature.
41          this.getSpellAbility().addEffect(new DamageAllEffect(2, new FilterCreaturePermanent()));
42          
43          // Whenever you cast an Eldrazi creature spell with converted mana cost 7 or greater, you may exile Kozilek's Return from your graveyard.
44          // If you do, Kozilek's Return deals 5 damage to each creature.
45         this.addAbility(new SpellCastControllerTriggeredAbility(Zone.GRAVEYARD, new DoIfCostPaid(new DamageAllEffect(5, new FilterCreaturePermanent()), 
46                  new ExileSourceFromGraveCost()), filter, false, false));
47      }
48  
49      private KozileksReturn(final KozileksReturn card) {
50          super(card);
51      }
52  
53      @Override
54      public KozileksReturn copy() {
55          return new KozileksReturn(this);
56      }
57  }

33      public KarnLiberated(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{7}");
35          this.addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.KARN);
37          this.setStartingLoyalty(6);
38  
39          // +4: Target player exiles a card from their hand.
40          LoyaltyAbility ability1 = new LoyaltyAbility(new KarnPlayerExileEffect(), 4);
41          ability1.addTarget(new TargetPlayer());
42          this.addAbility(ability1);
43  
44          // -3: Exile target permanent.
45          LoyaltyAbility ability2 = new LoyaltyAbility(new ExileTargetForSourceEffect(), -3);
46          ability2.addTarget(new TargetPermanent());
47          this.addAbility(ability2);
48  
49          // -14: Restart the game, leaving in exile all non-Aura permanent cards exiled with Karn Liberated. Then put those cards onto the battlefield under your control.
50          this.addAbility(new LoyaltyAbility(new KarnLiberatedEffect(), -14));
51      }

63  class KarnLiberatedEffect extends OneShotEffect {
64  
65      private UUID exileId;
66  
67      public KarnLiberatedEffect() {
68          super(Outcome.ExtraTurn);
69          this.staticText = "Restart the game, leaving in exile all non-Aura permanent cards exiled with {this}. Then put those cards onto the battlefield under your control";
70      }
71  
72      public KarnLiberatedEffect(final KarnLiberatedEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public boolean apply(Game game, Ability source) {
78          MageObject sourceObject = source.getSourceObject(game);
79          if (sourceObject == null) {
80              return false;
81          }
82          List<Card> cards = new ArrayList<>();
83          for (ExileZone zone : game.getExile().getExileZones()) {
84              exileId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
85              if (zone.getId().equals(exileId)) {
86                  for (Card card : zone.getCards(game)) {
87                      if (!card.hasSubtype(SubType.AURA, game)
88                              && card.isPermanent(game)) {
89                          cards.add(card);
90                      }
91                  }
92              }
93          }
94  
95          // dirty hack for game restart, can cause bugs due strange clear code (some data like ZCC keeping on new game)
96          // see testCommanderRestoredToBattlefieldAfterKarnUltimate for more details
97  
98          game.getState().clearOnGameRestart();
99          // default watchers init, TODO: remove all restart/init code to game
100         ((GameImpl) game).initGameDefaultWatchers();
101 
102         for (Card card : game.getCards()) {
103             game.getState().addCard(card);
104         }
105         for (Player player : game.getPlayers().values()) {
106             if (player.canRespond()) { // only players alive are in the restarted game
107                 player.getGraveyard().clear();
108                 player.getHand().clear();
109                 player.getLibrary().clear();
110                 for (Card card : game.getCards()) {
111                     if (card.isOwnedBy(player.getId()) && !card.isCopy() // no copies
112                             && !player.getSideboard().contains(card.getId())
113                             && !cards.contains(card)) { // not the exiled cards
114                         if (game.getCommandersIds(player, CommanderCardType.ANY, false).contains(card.getId())) {
115                             game.addCommander(new Commander(card)); // TODO: check restart and init
116                             // no needs in initCommander call -- it's used on game startup (init)
117                             game.setZone(card.getId(), Zone.COMMAND);
118                         } else {
119                             player.getLibrary().putOnTop(card, game);
120                         }
121                     }
122                 }
123                 ((GameImpl) game).initPlayerDefaultWatchers(player.getId());
124                 player.init(game);
125             }
126         }
127         for (Card card : cards) {
128             game.getState().setZone(card.getId(), Zone.EXILED);
129             game.getExile().add(exileId, sourceObject.getIdName(), card);
130         }
131         game.addDelayedTriggeredAbility(new KarnLiberatedDelayedTriggeredAbility(exileId), source);
132         game.setStartingPlayerId(source.getControllerId());
133         game.start(null);
134         return true;
135     }
136 
137     @Override
138     public KarnLiberatedEffect copy() {
139         return new KarnLiberatedEffect(this);
140     }

171 class KarnLiberatedDelayedEffect extends OneShotEffect {
172 
173     private final UUID exileId;
174 
175     public KarnLiberatedDelayedEffect(UUID exileId) {
176         super(Outcome.PlayForFree);
177         this.exileId = exileId;
178         this.staticText = "Put those cards onto the battlefield under your control";
179     }
180 
181     public KarnLiberatedDelayedEffect(final KarnLiberatedDelayedEffect effect) {
182         super(effect);
183         this.exileId = effect.exileId;
184     }
185 
186     @Override
187     public boolean apply(Game game, Ability source) {
188         Player controller = game.getPlayer(source.getControllerId());
189         ExileZone exileZone = game.getExile().getExileZone(exileId);
190         if (controller == null || exileZone == null) {
191             return false;
192         }
193 
194         // Creatures put onto the battlefield due to Karn's ability will have been under their controller's control continuously
195         // since the beginning of the first turn. They can attack and their activated abilities with {T} in the cost can be activated.
196         Cards cards = new CardsImpl(exileZone); // needed because putOntoTheBattlefield removes from exile
197         if (cards.isEmpty()) {
198             return false;
199         }
200 
201         controller.moveCards(cards, Zone.BATTLEFIELD, source, game);
202         for (Card card : cards.getCards(game)) {
203             if (card != null) {
204                 Permanent permanent = game.getPermanent(card.getId());
205                 if (permanent != null) {
206                     ((PermanentImpl) permanent).removeSummoningSickness();
207                 }
208             }
209         }
210 
211         return true;
212     }
213 
214     @Override
215     public KarnLiberatedDelayedEffect copy() {
216         return new KarnLiberatedDelayedEffect(this);
217     }

221 class KarnPlayerExileEffect extends OneShotEffect {
222 
223     public KarnPlayerExileEffect() {
224         super(Outcome.Exile);
225         staticText = "target player exiles a card from their hand.";
226     }
227 
228     public KarnPlayerExileEffect(final KarnPlayerExileEffect effect) {
229         super(effect);
230     }
231 
232     @Override
233     public boolean apply(Game game, Ability source) {
234         Player player = game.getPlayer(targetPointer.getFirst(game, source));
235         MageObject sourceObject = source.getSourceObject(game);
236         if (sourceObject == null) {
237             return false;
238         }
239         if (player == null) {
240             return false;
241         }
242         TargetCardInHand target = new TargetCardInHand();
243         if (target.canChoose(player.getId(), source, game)
244                 && target.chooseTarget(Outcome.Exile, player.getId(), source, game)) {
245             UUID exileId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
246             return player.moveCardsToExile(new CardsImpl(target.getTargets()).getCards(game), source, game, true, exileId, sourceObject.getIdName());
247         }
248         return false;
249     }
250 
251     @Override
252     public KarnPlayerExileEffect copy() {
253         return new KarnPlayerExileEffect(this);
254     }

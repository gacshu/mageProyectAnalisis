29  public final class KrovikanVampire extends CardImpl {
30  
31      public KrovikanVampire(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{B}{B}");
33  
34          this.subtype.add(SubType.VAMPIRE);
35          this.power = new MageInt(3);
36          this.toughness = new MageInt(3);
37  
38          // At the beginning of each end step, if a creature dealt damage by Krovikan Vampire this turn died, put that card onto the battlefield under your control. Sacrifice it when you lose control of Krovikan Vampire.
39          Ability ability = new BeginningOfEndStepTriggeredAbility(
40                  Zone.BATTLEFIELD,
41                  new KrovikanVampireEffect(),
42                  TargetController.ANY,
43                  new KrovikanVampireInterveningIfCondition(),
44                  false);
45          ability.addWatcher(new KrovikanVampireCreaturesDamagedWatcher());
46          ability.addWatcher(new KrovikanVampireCreaturesDiedWatcher());
47          this.addAbility(ability);
48      }
49  
50      private KrovikanVampire(final KrovikanVampire card) {
51          super(card);
52      }
53  
54      @Override
55      public KrovikanVampire copy() {
56          return new KrovikanVampire(this);
57      }

60  class KrovikanVampireEffect extends OneShotEffect {
61  
62      Set<UUID> creaturesAffected = new HashSet<>();
63  
64      KrovikanVampireEffect() {
65          super(Outcome.Neutral);
66          staticText = "put that card onto the battlefield under your control. Sacrifice it when you lose control of {this}";
67      }
68  
69      KrovikanVampireEffect(KrovikanVampireEffect effect) {
70          super(effect);
71      }
72  
73      @Override
74      public boolean apply(Game game, Ability source) {
75          Player controller = game.getPlayer(source.getControllerId());
76          Permanent krovikanVampire = game.getPermanent(source.getSourceId());
77          creaturesAffected = (Set<UUID>) game.getState().getValue(source.getSourceId() + "creatureToGainControl");
78          if (creaturesAffected != null
79                  && controller != null
80                  && krovikanVampire != null) {
81              creaturesAffected.stream().map((creatureId) -> {
82                  controller.moveCards(game.getCard(creatureId), Zone.BATTLEFIELD, source, game, false, false, false, null);
83                  return creatureId;
84              }).map((creatureId) -> {
85                  OneShotEffect effect = new SacrificeTargetEffect();
86                  effect.setText("Sacrifice this if Krovikan Vampire leaves the battlefield or its current controller loses control of it.");
87                  effect.setTargetPointer(new FixedTarget(creatureId, game));
88                  return effect;
89              }).map((effect) -> new KrovikanVampireDelayedTriggeredAbility(effect, krovikanVampire.getId())).forEachOrdered((dTA) -> {
90                  game.addDelayedTriggeredAbility(dTA, source);
91              });
92              creaturesAffected.clear();
93              return true;
94          }
95          return false;
96      }
97  
98      @Override
99      public KrovikanVampireEffect copy() {
100         return new KrovikanVampireEffect(this);
101     }
102 }

104 class KrovikanVampireInterveningIfCondition implements Condition {
105 
106     Set<UUID> creaturesAffected = new HashSet<>();
107 
108     @Override
109     public boolean apply(Game game, Ability source) {
110         KrovikanVampireCreaturesDiedWatcher watcherDied = game.getState().getWatcher(KrovikanVampireCreaturesDiedWatcher.class);
111         KrovikanVampireCreaturesDamagedWatcher watcherDamaged = game.getState().getWatcher(KrovikanVampireCreaturesDamagedWatcher.class);
112         if (watcherDied != null) {
113             Set<UUID> creaturesThatDiedThisTurn = watcherDied.getDiedThisTurn();
114             creaturesThatDiedThisTurn.stream().filter((mor) -> (watcherDamaged != null)).forEachOrdered((mor) -> {
115                 watcherDamaged.getDamagedBySource().stream().filter((mor2) -> (mor2 != null
116                         && mor == mor2)).forEachOrdered((_item) -> {
117                     creaturesAffected.add(mor);
118                 });
119             });
120             if (creaturesAffected != null
121                     && creaturesAffected.size() > 0) {
122                 game.getState().setValue(source.getSourceId() + "creatureToGainControl", creaturesAffected);
123                 return true;
124             }
125         }
126         return false;
127     }
128 
129     @Override
130     public String toString() {
131         return "if a creature dealt damage by Krovikan Vampire this turn died";
132     }
133 }

135 class KrovikanVampireCreaturesDamagedWatcher extends Watcher {
136 
137     private final Set<UUID> damagedBySource = new HashSet<>();
138 
139     public KrovikanVampireCreaturesDamagedWatcher() {
140         super(WatcherScope.GAME);
141     }
142 
143     @Override
144     public void watch(GameEvent event, Game game) {
145         if (event.getType() != GameEvent.EventType.DAMAGED_PERMANENT
146                 || !sourceId.equals(event.getSourceId())) {
147             return;
148         }
149         Permanent permanent = game.getPermanent(event.getTargetId());
150         if (permanent == null || !permanent.isCreature(game)) {
151             return;
152         }
153         damagedBySource.add(event.getTargetId());
154     }
155 
156     public Set<UUID> getDamagedBySource() {
157         return this.damagedBySource;
158     }
159 
160     @Override
161     public void reset() {
162         damagedBySource.clear();
163     }
164 }

166 class KrovikanVampireCreaturesDiedWatcher extends Watcher {
167 
168     private final Set<UUID> diedThisTurn = new HashSet<>();
169 
170     public KrovikanVampireCreaturesDiedWatcher() {
171         super(WatcherScope.GAME);
172     }
173 
174     @Override
175     public void watch(GameEvent event, Game game) {
176         if (event.getType() == GameEvent.EventType.ZONE_CHANGE) {
177             ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
178             if (zEvent.isDiesEvent()
179                     && zEvent.getTarget() != null
180                     && zEvent.getTarget().isCreature(game)) {
181                 diedThisTurn.add(zEvent.getTargetId());
182             }
183         }
184     }
185 
186     @Override
187     public void reset() {
188         diedThisTurn.clear();
189     }
190 
191     public Set<UUID> getDiedThisTurn() {
192         return this.diedThisTurn;
193     }
194 }

196 class KrovikanVampireDelayedTriggeredAbility extends DelayedTriggeredAbility {
197 
198     UUID krovikanVampire;
199 
200     KrovikanVampireDelayedTriggeredAbility(Effect effect, UUID krovikanVampire) {
201         super(effect, Duration.EndOfGame, true);
202         this.krovikanVampire = krovikanVampire;
203     }
204 
205     KrovikanVampireDelayedTriggeredAbility(KrovikanVampireDelayedTriggeredAbility ability) {
206         super(ability);
207         this.krovikanVampire = ability.krovikanVampire;
208     }
209 
210     @Override
211     public boolean checkEventType(GameEvent event, Game game) {
212         return event.getType() == GameEvent.EventType.LOST_CONTROL
213                 || event.getType() == GameEvent.EventType.ZONE_CHANGE;
214     }
215 
216     @Override
217     public boolean checkTrigger(GameEvent event, Game game) {
218         if (event.getType() == GameEvent.EventType.LOST_CONTROL
219                 && event.getTargetId().equals(krovikanVampire)) {
220             return true;
221         }
222         if (event.getType() == GameEvent.EventType.ZONE_CHANGE
223                 && event.getTargetId().equals(krovikanVampire)) {
224             return true;
225         }
226         return false;
227     }
228 
229     @Override
230     public KrovikanVampireDelayedTriggeredAbility copy() {
231         return new KrovikanVampireDelayedTriggeredAbility(this);
232     }
233 
234     @Override
235     public String getRule() {
236         return "Control of Krovikan Vampire lost";
237     }

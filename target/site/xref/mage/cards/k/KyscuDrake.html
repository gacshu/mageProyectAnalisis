28  public final class KyscuDrake extends CardImpl {
29  
30      private static final FilterCard filter = new FilterCard("card named Viashivan Dragon");
31      private static final FilterControlledCreaturePermanent filterSpitting = new FilterControlledCreaturePermanent("creature named Spitting Drake");
32  
33      static {
34          filter.add(new NamePredicate("Viashivan Dragon"));
35          filterSpitting.add(new NamePredicate("Spitting Drake"));
36      }
37  
38      public KyscuDrake(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{G}");
40          this.subtype.add(SubType.DRAKE);
41  
42          this.power = new MageInt(2);
43          this.toughness = new MageInt(2);
44  
45          // Flying
46          this.addAbility(FlyingAbility.getInstance());
47  
48          // {G}: Kyscu Drake gets +0/+1 until end of turn. Activate this ability only once each turn.
49          this.addAbility(new LimitedTimesPerTurnActivatedAbility(Zone.BATTLEFIELD, new BoostSourceEffect(0, 1, Duration.EndOfTurn), new ManaCostsImpl<>("{G}")));
50  
51          // Sacrifice Kyscu Drake and a creature named Spitting Drake: Search your library for a card named Viashivan Dragon and put that card onto the battlefield. Then shuffle your library.
52          this.addAbility(new SimpleActivatedAbility(new SearchLibraryPutInPlayEffect(new TargetCardInLibrary(1, 1, filter), false, true, Outcome.PutCardInPlay), new CompositeCost(new SacrificeSourceCost(), new SacrificeTargetCost(new TargetControlledCreaturePermanent(1, 1, filterSpitting, false)), "sacrifice {this} and a creature named Spitting Drake")));
53      }
54  
55      private KyscuDrake(final KyscuDrake card) {
56          super(card);
57      }
58  
59      @Override
60      public KyscuDrake copy() {
61          return new KyscuDrake(this);
62      }
63  }

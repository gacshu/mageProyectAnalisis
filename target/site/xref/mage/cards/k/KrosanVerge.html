26  public final class KrosanVerge extends CardImpl {
27  
28      public KrosanVerge(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
30  
31          // Krosan Verge enters the battlefield tapped.
32          this.addAbility(new EntersBattlefieldTappedAbility());
33  
34          // {tap}: Add {C}.
35          this.addAbility(new ColorlessManaAbility());
36  
37          // {2}, {T}, Sacrifice Krosan Verge: Search your library for a Forest card and a Plains card and put them onto the battlefield tapped. Then shuffle your library.
38          Ability ability = new SimpleActivatedAbility(new SearchLibraryPutInPlayEffect(
39                  new KrosanVergeTarget(), true, Outcome.PutLandInPlay
40          ).setText("search your library for a Forest card and a Plains card, put them onto the battlefield tapped, then shuffle"), new GenericManaCost(2));
41          ability.addCost(new TapSourceCost());
42          ability.addCost(new SacrificeSourceCost());
43          this.addAbility(ability);
44      }
45  
46      private KrosanVerge(final KrosanVerge card) {
47          super(card);
48      }
49  
50      @Override
51      public KrosanVerge copy() {
52          return new KrosanVerge(this);
53      }
54  }

56  class KrosanVergeTarget extends TargetCardInLibrary {
57  
58      private static final FilterCard filter
59              = new FilterCard("a Forest card and a Plains card");
60  
61      static {
62          filter.add(Predicates.or(
63                  SubType.FOREST.getPredicate(),
64                  SubType.PLAINS.getPredicate()
65          ));
66      }
67  
68      private static final SubTypeAssignment subTypeAssigner
69              = new SubTypeAssignment(SubType.FOREST, SubType.PLAINS);
70  
71      KrosanVergeTarget() {
72          super(0, 2, filter);
73      }
74  
75      private KrosanVergeTarget(final KrosanVergeTarget target) {
76          super(target);
77      }
78  
79      @Override
80      public KrosanVergeTarget copy() {
81          return new KrosanVergeTarget(this);
82      }
83  
84      @Override
85      public boolean canTarget(UUID playerId, UUID id, Ability source, Game game) {
86          if (!super.canTarget(playerId, id, source, game)) {
87              return false;
88          }
89          Card card = game.getCard(id);
90          if (card == null) {
91              return false;
92          }
93          if (this.getTargets().isEmpty()) {
94              return true;
95          }
96          Cards cards = new CardsImpl(this.getTargets());
97          cards.add(card);
98          return subTypeAssigner.getRoleCount(cards, game) >= cards.size();
99      }
100 }

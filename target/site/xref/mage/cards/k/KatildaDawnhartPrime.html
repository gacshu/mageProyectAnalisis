38      private static final FilterPermanent filter = new FilterPermanent(SubType.WEREWOLF, "Werewolves");
39      private static final FilterPermanent filter2 = new FilterControlledCreaturePermanent(SubType.HUMAN, "Human creatures");
40  
41      public KatildaDawnhartPrime(UUID ownerId, CardSetInfo setInfo) {
42          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{G}{W}");
43  
44          this.addSuperType(SuperType.LEGENDARY);
45          this.subtype.add(SubType.HUMAN);
46          this.subtype.add(SubType.WARLOCK);
47          this.power = new MageInt(1);
48          this.toughness = new MageInt(1);
49  
50          // Protection from Werewolves
51          this.addAbility(new ProtectionAbility(filter));
52  
53          // Human creatures you control have "{T}: Add one mana of any of this creature's colors."
54          this.addAbility(new SimpleStaticAbility(new GainAbilityControlledEffect(
55                  new KatildaDawnhartPrimeManaAbility(), Duration.WhileOnBattlefield, filter2
56          )));
57  
58          // {4}{G}{W}, {T}: Put a +1/+1 counter on each creature you control.
59          Ability ability = new SimpleActivatedAbility(new AddCountersAllEffect(
60                  CounterType.P1P1.createInstance(), StaticFilters.FILTER_CONTROLLED_CREATURE
61          ), new ManaCostsImpl<>("{4}{G}{W}"));
62          ability.addCost(new TapSourceCost());
63          this.addAbility(ability);
64      }

114     @Override
115     public List<Mana> getNetMana(Game game, Ability source) {
116         List<Mana> netMana = new ArrayList<>();
117         if (game != null) {
118             Permanent permanent = source.getSourcePermanentIfItStillExists(game);
119             if (permanent != null) {
120                 ObjectColor color = permanent.getColor(game);
121                 if (color.isWhite()) {
122                     netMana.add(new Mana(ColoredManaSymbol.W));
123                 }
124                 if (color.isBlue()) {
125                     netMana.add(new Mana(ColoredManaSymbol.U));
126                 }
127                 if (color.isBlack()) {
128                     netMana.add(new Mana(ColoredManaSymbol.B));
129                 }
130                 if (color.isRed()) {
131                     netMana.add(new Mana(ColoredManaSymbol.R));
132                 }
133                 if (color.isGreen()) {
134                     netMana.add(new Mana(ColoredManaSymbol.G));
135                 }
136             }
137         }
138         return netMana;
139     }

141     @Override
142     public Mana produceMana(Game game, Ability source) {
143         Mana mana = new Mana();
144         if (game != null) {
145             Player controller = game.getPlayer(source.getControllerId());
146             Permanent permanent = source.getSourcePermanentIfItStillExists(game);
147             if (controller != null && permanent != null) {
148                 Choice choice = new ChoiceImpl();
149                 choice.setMessage("Pick a mana color");
150                 ObjectColor color = permanent.getColor(game);
151                 if (color.isWhite()) {
152                     choice.getChoices().add("White");
153                 }
154                 if (color.isBlue()) {
155                     choice.getChoices().add("Blue");
156                 }
157                 if (color.isBlack()) {
158                     choice.getChoices().add("Black");
159                 }
160                 if (color.isRed()) {
161                     choice.getChoices().add("Red");
162                 }
163                 if (color.isGreen()) {
164                     choice.getChoices().add("Green");
165                 }
166                 if (!choice.getChoices().isEmpty()) {
167                     if (choice.getChoices().size() == 1) {
168                         choice.setChoice(choice.getChoices().iterator().next());
169                     } else {
170                         controller.choose(outcome, choice, game);
171                     }
172 
173                     if (choice.getChoice() != null) {
174                         switch (choice.getChoice()) {
175                             case "White":
176                                 mana.setWhite(1);
177                                 break;
178                             case "Blue":
179                                 mana.setBlue(1);
180                                 break;
181                             case "Black":
182                                 mana.setBlack(1);
183                                 break;
184                             case "Red":
185                                 mana.setRed(1);
186                                 break;
187                             case "Green":
188                                 mana.setGreen(1);
189                                 break;
190                         }
191                     }
192                 }
193             }
194         }
195         return mana;
196     }

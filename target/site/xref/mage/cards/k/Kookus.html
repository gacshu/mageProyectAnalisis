5   import mage.MageInt;

15  import mage.abilities.effects.common.continuous.BoostSourceEffect;
16  import mage.constants.SubType;
17  import mage.abilities.keyword.TrampleAbility;
18  import mage.cards.CardImpl;
19  import mage.cards.CardSetInfo;
20  import mage.constants.CardType;
21  import mage.constants.ColoredManaSymbol;
22  import mage.constants.Duration;
23  import mage.constants.TargetController;
24  import mage.constants.Zone;
25  import mage.filter.common.FilterCreaturePermanent;
26  import mage.filter.predicate.mageobject.NamePredicate;
27  
28  /**
29   *
30   * @author TheElk801
31   */
32  public final class Kookus extends CardImpl {
33  
34      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent();
35  
36      static {
37          filter.add(new NamePredicate("Keeper of Kookus"));
38      }
39  
40      public Kookus(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}{R}");
42  
43          this.subtype.add(SubType.DJINN);
44          this.power = new MageInt(3);
45          this.toughness = new MageInt(5);
46  
47          // Trample
48          this.addAbility(TrampleAbility.getInstance());
49  
50          // At the beginning of your upkeep, if you don't control a creature named Keeper of Kookus, Kookus deals 3 damage to you and attacks this turn if able.
51          TriggeredAbility ability = new BeginningOfUpkeepTriggeredAbility(new DamageControllerEffect(3), TargetController.YOU, false);
52          ability.addEffect(new AttacksIfAbleSourceEffect(Duration.EndOfTurn, false));
53          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
54                  ability,
55                  new InvertCondition(new PermanentsOnTheBattlefieldCondition(filter)),
56                  "At the beginning of your upkeep, "
57                  + "if you don't control a creature named Keeper of Kookus, "
58                  + "{this} deals 3 damage to you and attacks this turn if able"
59          ));
60  
61          // {R}: Kookus gets +1/+0 until end of turn.
62          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new BoostSourceEffect(1, 0, Duration.EndOfTurn), new ColoredManaCost(ColoredManaSymbol.R)));
63      }
64  
65      private Kookus(final Kookus card) {
66          super(card);
67      }
68  
69      @Override
70      public Kookus copy() {
71          return new Kookus(this);
72      }
73  }

4   import java.util.UUID;
5   import mage.abilities.Ability;

9   import mage.abilities.mana.ColorlessManaAbility;

15   *
16   * @author jonubuu

18  public final class TalismanOfDominance extends CardImpl {
19  
20      public TalismanOfDominance(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT},"{2}");
22  
23          // {tap}: Add {C}.
24          this.addAbility(new ColorlessManaAbility());
25          // {tap}: Add {U} or {B}. Talisman of Dominance deals 1 damage to you.
26          Ability blueManaAbility = new BlueManaAbility();
27          blueManaAbility.addEffect(new DamageControllerEffect(1));
28          this.addAbility(blueManaAbility);
29          Ability blackManaAbility = new BlackManaAbility();
30          blackManaAbility.addEffect(new DamageControllerEffect(1));
31          this.addAbility(blackManaAbility);
32      }
33  
34      private TalismanOfDominance(final TalismanOfDominance card) {
35          super(card);
36      }
37  
38      @Override
39      public TalismanOfDominance copy() {
40          return new TalismanOfDominance(this);
41      }
42  }

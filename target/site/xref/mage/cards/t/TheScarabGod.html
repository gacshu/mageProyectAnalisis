42  public final class TheScarabGod extends CardImpl {
43  
44      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent(SubType.ZOMBIE, "Zombies you control");
45      private static final DynamicValue xValue = new PermanentsOnBattlefieldCount(filter, null);
46      private static final Hint hint = new ValueHint(
47              "Number of Zombies you control", xValue
48      );
49  
50      public TheScarabGod(UUID ownerId, CardSetInfo setInfo) {
51          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}{B}");
52  
53          addSuperType(SuperType.LEGENDARY);
54          this.subtype.add(SubType.GOD);
55          this.power = new MageInt(5);
56          this.toughness = new MageInt(5);
57  
58          // At the beginning of your upkeep, each opponent loses X life and you scry X, where X is the number of Zombies you control.
59          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new TheScarabGodEffect(xValue), TargetController.YOU, false).addHint(hint));
60  
61          // {2}{U}{B}: Exile target creature card from a graveyard. Create a token that's a copy of it, except it's a 4/4 black Zombie.
62          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new TheScarabGodEffect2(), new ManaCostsImpl<>("{2}{U}{B}"));
63          ability.addTarget(new TargetCardInGraveyard(1, 1, new FilterCreatureCard("creature card from a graveyard")));
64          this.addAbility(ability);
65  
66          // When The Scarab God dies, return it to its owner's hand at the beginning of the next end step.
67          this.addAbility(new DiesSourceTriggeredAbility(new TheScarabGodEffect3()));
68      }
69  
70      private TheScarabGod(final TheScarabGod card) {
71          super(card);
72      }
73  
74      @Override
75      public TheScarabGod copy() {
76          return new TheScarabGod(this);
77      }
78  }

80  class TheScarabGodEffect extends OneShotEffect {
81  
82      private final DynamicValue numOfZombies;
83  
84      public TheScarabGodEffect(DynamicValue numOfZombies) {
85          super(Outcome.Benefit);
86          this.numOfZombies = numOfZombies;
87          staticText = "each opponent loses X life and you scry X, where X is the number of Zombies you control";
88      }
89  
90      public TheScarabGodEffect(final TheScarabGodEffect effect) {
91          super(effect);
92          this.numOfZombies = effect.numOfZombies;
93      }
94  
95      @Override
96      public TheScarabGodEffect copy() {
97          return new TheScarabGodEffect(this);
98      }
99  
100     @Override
101     public boolean apply(Game game, Ability source) {
102         Player controller = game.getPlayer(source.getControllerId());
103         if (controller != null) {
104             int numZombies = numOfZombies.calculate(game, source, this);
105             if (numZombies > 0) {
106                 for (UUID playerId : game.getOpponents(source.getControllerId())) {
107                     Player opponent = game.getPlayer(playerId);
108                     if (opponent != null) {
109                         opponent.loseLife(numZombies, game, source, false);
110                     }
111                 }
112                 controller.scry(numZombies, source, game);
113             }
114 
115             return true;
116         }
117         return false;
118     }
119 }

121 class TheScarabGodEffect2 extends OneShotEffect {
122 
123     public TheScarabGodEffect2() {
124         super(Outcome.PutCreatureInPlay);
125         this.staticText = "Exile target creature card from a graveyard. Create a token that's a copy of it, except it's a 4/4 black Zombie.";
126     }
127 
128     public TheScarabGodEffect2(final TheScarabGodEffect2 effect) {
129         super(effect);
130     }
131 
132     @Override
133     public TheScarabGodEffect2 copy() {
134         return new TheScarabGodEffect2(this);
135     }
136 
137     @Override
138     public boolean apply(Game game, Ability source) {
139         Card card = game.getCard(source.getFirstTarget());
140         Player controller = game.getPlayer(source.getControllerId());
141         if (controller != null && card != null) {
142             controller.moveCards(card, Zone.EXILED, source, game); // Also if the move to exile is replaced, the copy takes place
143             CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect(source.getControllerId(), null, false, 1, false, false, null, 4, 4, false);
144             effect.setTargetPointer(new FixedTarget(card, game));
145             effect.setOnlySubType(SubType.ZOMBIE);
146             effect.setOnlyColor(ObjectColor.BLACK);
147             effect.apply(game, source);
148             return true;
149         }
150 
151         return false;
152     }
153 }

155 class TheScarabGodEffect3 extends OneShotEffect {
156 
157     private static final String effectText = "return it to its owner's hand at the beginning of the next end step";
158 
159     TheScarabGodEffect3() {
160         super(Outcome.Benefit);
161         staticText = effectText;
162     }
163 
164     TheScarabGodEffect3(TheScarabGodEffect3 effect) {
165         super(effect);
166     }
167 
168     @Override
169     public boolean apply(Game game, Ability source) {
170         // Create delayed triggered ability
171         Effect effect = new ReturnToHandTargetEffect();
172         effect.setText("return {this} to its owner's hand");
173         effect.setTargetPointer(new FixedTarget(source.getSourceId(), source.getSourceObjectZoneChangeCounter()));
174         DelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(effect);
175         game.addDelayedTriggeredAbility(delayedAbility, source);
176         return true;
177     }
178 
179     @Override
180     public TheScarabGodEffect3 copy() {
181         return new TheScarabGodEffect3(this);
182     }

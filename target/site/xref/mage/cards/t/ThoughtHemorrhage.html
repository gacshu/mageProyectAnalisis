29  public final class ThoughtHemorrhage extends CardImpl {
30  
31      public ThoughtHemorrhage(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{2}{B}{R}");
33  
34          // Name a nonland card. Target player reveals their hand. Thought 
35          // Hemorrhage deals 3 damage to that player for each card with that 
36          // name revealed this way. Search that player's graveyard, hand, and 
37          // library for all cards with that name and exile them. Then that 
38          // player shuffles their library.
39          this.getSpellAbility().addTarget(new TargetPlayer());
40          this.getSpellAbility().addEffect(new ChooseACardNameEffect(
41                  ChooseACardNameEffect.TypeOfName.NON_LAND_NAME));
42          this.getSpellAbility().addEffect(new ThoughtHemorrhageEffect());
43      }
44  
45      private ThoughtHemorrhage(final ThoughtHemorrhage card) {
46          super(card);
47      }
48  
49      @Override
50      public ThoughtHemorrhage copy() {
51          return new ThoughtHemorrhage(this);
52      }
53  }

55  class ThoughtHemorrhageEffect extends OneShotEffect {
56  
57      static final String rule = "Target player reveals their hand. "
58              + "{this} deals 3 damage to that player for each card with "
59              + "the chosen name revealed this way. "
60              + "Search that player's graveyard, hand, and library for "
61              + "all cards with that name and exile them. "
62              + "Then that player shuffles";
63  
64      public ThoughtHemorrhageEffect() {
65          super(Outcome.Exile);
66          staticText = rule;
67      }
68  
69      public ThoughtHemorrhageEffect(final ThoughtHemorrhageEffect effect) {
70          super(effect);
71      }
72  
73      @Override
74      public boolean apply(Game game, Ability source) {
75          Player controller = game.getPlayer(source.getControllerId());
76          MageObject sourceObject = game.getObject(source);
77          String cardName = (String) game.getState().getValue(
78                  source.getSourceId().toString() + ChooseACardNameEffect.INFO_KEY);
79          if (sourceObject != null
80                  && controller != null
81                  && cardName != null
82                  && !cardName.isEmpty()) {
83              Player targetPlayer = game.getPlayer(source.getFirstTarget());
84              if (targetPlayer != null) {
85                  targetPlayer.revealCards("hand of "
86                          + targetPlayer.getName(), targetPlayer.getHand(), game);
87                  int cardsFound = 0;
88                  for (Card card : targetPlayer.getHand().getCards(game)) {
89                      if (CardUtil.haveSameNames(card, cardName, game)) {
90                          cardsFound++;
91                      }
92                  }
93                  if (cardsFound > 0) {
94                      targetPlayer.damage(3 * cardsFound, source.getSourceId(), source, game);
95                  }
96                  // Exile all cards with the same name
97                  // Building a card filter with the name
98                  FilterCard filterNamedCards = new FilterCard();
99                  filterNamedCards.add(new NamePredicate(cardName));
100 
101                 Set<Card> toExile = new LinkedHashSet<>();
102                 // The cards you're searching for must be found and exiled if 
103                 // they're in the graveyard because it's a public zone.
104                 // Finding those cards in the hand and library is optional, 
105                 // because those zones are hidden (even if the hand is temporarily revealed).
106                 // search cards in graveyard
107                 for (Card checkCard : targetPlayer.getGraveyard().getCards(game)) {
108                     if (checkCard.getName().equals(cardName)) {
109                         toExile.add(checkCard);
110                     }
111                 }
112 
113                 // search cards in Hand
114                 TargetCard targetCardInHand = new TargetCard(0, Integer.MAX_VALUE, Zone.HAND, filterNamedCards);
115                 if (controller.chooseTarget(Outcome.Exile, targetPlayer.getHand(), targetCardInHand, source, game)) {
116                     List<UUID> targets = targetCardInHand.getTargets();
117                     for (UUID targetId : targets) {
118                         Card targetCard = targetPlayer.getHand().get(targetId, game);
119                         if (targetCard != null) {
120                             toExile.add(targetCard);
121                         }
122                     }
123                 }
124 
125                 // search cards in Library
126                 // If the player has no nonland cards in their hand, you can still search 
127                 // that player's library and have that player shuffle it.
128                 TargetCardInLibrary targetCardsLibrary = new TargetCardInLibrary(0, Integer.MAX_VALUE, filterNamedCards);
129                 controller.searchLibrary(targetCardsLibrary, source, game, targetPlayer.getId());
130                 for (UUID cardId : targetCardsLibrary.getTargets()) {
131                     Card card = game.getCard(cardId);
132                     if (card != null) {
133                         toExile.add(card);
134                     }
135                 }
136                 controller.moveCards(toExile, Zone.EXILED, source, game);
137                 targetPlayer.shuffleLibrary(source, game);
138                 return true;
139             }
140         }
141         return false;
142     }
143 
144     @Override
145     public ThoughtHemorrhageEffect copy() {
146         return new ThoughtHemorrhageEffect(this);
147     }
148 }

1   package mage.cards.t;
2   
3   import mage.MageInt;
4   import mage.abilities.common.DiesSourceTriggeredAbility;
5   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
6   import mage.abilities.effects.common.CreateTokenEffect;
7   import mage.abilities.effects.common.RevealLibraryPickControllerEffect;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.PutCards;
12  import mage.constants.SubType;
13  import mage.constants.SuperType;
14  import mage.filter.FilterCard;
15  import mage.filter.predicate.Predicates;
16  import mage.game.permanent.token.SoldierToken;
17  
18  import java.util.UUID;
19  
20  /**
21   * @author TheElk801
22   */
23  public final class TorstenFounderOfBenalia extends CardImpl {
24  
25      private static final FilterCard filter = new FilterCard("creature and/or land cards");
26  
27      static {
28          filter.add(Predicates.or(
29                  CardType.CREATURE.getPredicate(),
30                  CardType.LAND.getPredicate()
31          ));
32      }
33  
34      public TorstenFounderOfBenalia(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{5}{G}{W}");
36  
37          this.addSuperType(SuperType.LEGENDARY);
38          this.subtype.add(SubType.HUMAN, SubType.SOLDIER);
39          this.power = new MageInt(7);
40          this.toughness = new MageInt(7);
41  
42          // When Torsten, Founder of Benalia enters the battlefield, reveal the top seven cards of your library. Put any number of creature and/or land cards from among them into your hand and the rest on the bottom of your library in a random order.
43          this.addAbility(new EntersBattlefieldTriggeredAbility(new RevealLibraryPickControllerEffect(
44                  7, Integer.MAX_VALUE, filter, PutCards.HAND, PutCards.BOTTOM_RANDOM, false
45          )));
46  
47          // When Torsten dies, create seven 1/1 white Soldier creature tokens.
48          this.addAbility(new DiesSourceTriggeredAbility(new CreateTokenEffect(new SoldierToken(), 7)));
49      }
50  
51      private TorstenFounderOfBenalia(final TorstenFounderOfBenalia card) {
52          super(card);
53      }
54  
55      @Override
56      public TorstenFounderOfBenalia copy() {
57          return new TorstenFounderOfBenalia(this);
58      }
59  }

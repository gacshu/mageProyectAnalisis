37  public final class TerritorialHellkite extends CardImpl {

39      public TerritorialHellkite(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{R}");
41  
42          this.subtype.add(SubType.DRAGON);
43          this.power = new MageInt(6);
44          this.toughness = new MageInt(5);
45  
46          // Flying
47          this.addAbility(FlyingAbility.getInstance());
48  
49          // Haste
50          this.addAbility(HasteAbility.getInstance());
51  
52          // At the beginning of combat on your turn, choose an opponent at random that Territorial Hellkite didn't attack during your last combat. Territorial Hellkite attacks that player this combat if able. If you can't choose an opponent this way, tap Territorial Hellkite.
53          this.addAbility(new BeginningOfCombatTriggeredAbility(new AttackIfAbleTargetRandoOpponentSourceEffect(), TargetController.YOU, false), new AttackedLastCombatWatcher());
54      }

57          super(card);
58      }

66  class AttackedLastCombatWatcher extends Watcher {
67  
68      // Map<lastCombatOfPlayerId, Map<attackingCreature, attackedPlayerId>>
69      private final Map<UUID, Map<MageObjectReference, UUID>> attackedLastCombatPlayers = new HashMap<>();
70  
71      public AttackedLastCombatWatcher() {
72          super(WatcherScope.GAME);
73      }
74  
75      @Override
76      public void watch(GameEvent event, Game game) {
77          if (event.getType() == GameEvent.EventType.DECLARE_ATTACKERS_STEP_PRE) {
78              // Remove previous attacking creatures of the current combat's player if info exists
79              attackedLastCombatPlayers.remove(game.getCombat().getAttackingPlayerId());
80          }
81          if (event.getType() == GameEvent.EventType.ATTACKER_DECLARED) {
82              // remember which attacker attacked which player
83              Map<MageObjectReference, UUID> attackedPlayers = new HashMap<>();
84              for (UUID attackerId : game.getCombat().getAttackers()) {
85                  Permanent attacker = game.getPermanent(attackerId);
86                  if (attacker != null) {
87                      attackedPlayers.put(new MageObjectReference(attacker, game), game.getCombat().getDefenderId(attackerId));
88                  }
89              }
90              attackedLastCombatPlayers.put(game.getCombat().getAttackingPlayerId(), attackedPlayers);
91          }
92      }
93  
94      public Map<MageObjectReference, UUID> getAttackedLastCombatPlayers(UUID combatPlayerId) {
95          return attackedLastCombatPlayers.get(combatPlayerId);
96      }
97  }

99  class AttackIfAbleTargetRandoOpponentSourceEffect extends OneShotEffect {
100 
101     public AttackIfAbleTargetRandoOpponentSourceEffect() {
102         super(Outcome.Benefit);
103         this.staticText = "choose an opponent at random that {this} didn't attack during your last combat. {this} attacks that player this combat if able. If you can't choose an opponent this way, tap {this}";
104     }
105 
106     public AttackIfAbleTargetRandoOpponentSourceEffect(final AttackIfAbleTargetRandoOpponentSourceEffect effect) {
107         super(effect);
108     }
109 
110     @Override
111     public AttackIfAbleTargetRandoOpponentSourceEffect copy() {
112         return new AttackIfAbleTargetRandoOpponentSourceEffect(this);
113     }
114 
115     @Override
116     public boolean apply(Game game, Ability source) {
117         Player controller = game.getPlayer(source.getControllerId());
118         Permanent sourcePermanent = game.getPermanent(source.getSourceId());
119         AttackedLastCombatWatcher watcher = game.getState().getWatcher(AttackedLastCombatWatcher.class);
120         if (controller != null && sourcePermanent != null && watcher != null) {
121             List<UUID> opponents = new ArrayList<>();
122             Map<MageObjectReference, UUID> attackedPlayers = watcher.getAttackedLastCombatPlayers(source.getControllerId());
123             MageObjectReference mor = new MageObjectReference(sourcePermanent, game);
124             if (attackedPlayers == null) {
125                 opponents.addAll(game.getOpponents(controller.getId()));
126             } else {
127                 for (UUID opp : game.getOpponents(controller.getId())) {
128                     if (!opp.equals(attackedPlayers.getOrDefault(mor, null))) {
129                         opponents.add(opp);
130                     }
131                 }
132             }
133             if (!opponents.isEmpty()) {
134                 Player opponent = game.getPlayer(opponents.get(RandomUtil.nextInt(opponents.size())));
135                 if (opponent != null) {
136                     ContinuousEffect effect = new AttacksIfAbleTargetPlayerSourceEffect();
137                     effect.setTargetPointer(new FixedTarget(opponent.getId()));
138                     game.addEffect(effect, source);
139                     game.informPlayers(sourcePermanent.getLogName() + " has to attack " + opponent.getLogName() + ".");
140                 }
141             } else {
142                 game.informPlayers(sourcePermanent.getLogName() + " can't attack an opponent it didn't attack last combat.");
143                 sourcePermanent.tap(source, game);
144             }
145             return true;
146         }
147 
148         return false;
149     }

31  public final class TouchTheSpiritRealm extends CardImpl {
32  
33      public TouchTheSpiritRealm(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
35  
36          // When Touch the Spirit Realm enters the battlefield, exile up to one target artifact or creature until
37          // Touch the Spirit Realm leaves the battlefield.
38          Ability ability = new EntersBattlefieldTriggeredAbility(new ExileUntilSourceLeavesEffect());
39          ability.addTarget(new TargetPermanent(0, 1, StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_CREATURE));
40          ability.addEffect(new CreateDelayedTriggeredAbilityEffect(new OnLeaveReturnExiledToBattlefieldAbility()));
41          this.addAbility(ability);
42          
43          // Channel - {1}{W}, Discard Touch the Spirit Realm: Exile target artifact or creature.
44          // Return it to the battlefield under its owner's control at the beginning of the next end step.
45          Ability channelAbility = new ChannelAbility("{1}{W}", new TouchTheSpiritRealmEffect());
46          channelAbility.addTarget(new TargetPermanent(StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_CREATURE));
47          this.addAbility(channelAbility);
48      }
49  
50      private TouchTheSpiritRealm(final TouchTheSpiritRealm card) {
51          super(card);
52      }
53  
54      @Override
55      public TouchTheSpiritRealm copy() {
56          return new TouchTheSpiritRealm(this);
57      }
58  }
59  
60  class TouchTheSpiritRealmEffect extends OneShotEffect {
61  
62      public TouchTheSpiritRealmEffect() {
63          super(Outcome.Detriment);
64          staticText = "Exile target artifact or creature. Return it to the battlefield under its owner's control at the beginning of the next end step";
65      }
66  
67      private TouchTheSpiritRealmEffect(final TouchTheSpiritRealmEffect effect) {
68          super(effect);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          Player player = game.getPlayer(source.getControllerId());
74          Permanent permanent = game.getPermanent(source.getFirstTarget());
75          if (player == null || permanent == null) {
76              return false;
77          }
78          Card card = permanent.getMainCard();
79          player.moveCardsToExile(permanent, source, game, true, CardUtil.getExileZoneId(game, source), CardUtil.getSourceName(game, source));
80          ReturnToBattlefieldUnderOwnerControlTargetEffect returnEffect = new ReturnToBattlefieldUnderOwnerControlTargetEffect(false, true);
81          returnEffect.setTargetPointer(new FixedTarget(card,game));
82          game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(returnEffect), source);
83          return true;
84      }
85      
86      @Override
87      public TouchTheSpiritRealmEffect copy() {
88          return new TouchTheSpiritRealmEffect(this);
89      }
90  }

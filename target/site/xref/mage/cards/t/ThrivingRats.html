9   import mage.abilities.effects.common.counter.GetEnergyCountersControllerEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.SubType;
14  import mage.counters.CounterType;
15  
16  import java.util.UUID;
17  
18  /**
19   * @author fireshoes
20   */
21  public final class ThrivingRats extends CardImpl {
22  
23      public ThrivingRats(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{B}");
25          this.subtype.add(SubType.RAT);
26          this.power = new MageInt(1);
27          this.toughness = new MageInt(2);
28  
29          // When Thriving Rats enters the battlefield, you get {E}{E}.
30          this.addAbility(new EntersBattlefieldTriggeredAbility(new GetEnergyCountersControllerEffect(2)));
31  
32          // Whenever Thriving Rats attacks, you may pay {E}{E}. If you do, put a +1/+1 counter on it.
33          this.addAbility(new AttacksTriggeredAbility(new DoIfCostPaid(
34                  new AddCountersSourceEffect(CounterType.P1P1.createInstance())
35                          .setText("put a +1/+1 counter on it"),
36                  new PayEnergyCost(2)
37          ), false));
38      }
39  
40      private ThrivingRats(final ThrivingRats card) {
41          super(card);
42      }
43  
44      @Override
45      public ThrivingRats copy() {
46          return new ThrivingRats(this);
47      }
48  }

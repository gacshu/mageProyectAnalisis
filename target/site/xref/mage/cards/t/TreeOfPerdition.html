28  public final class TreeOfPerdition extends CardImpl {
29  
30      public TreeOfPerdition(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{B}");
32          this.subtype.add(SubType.PLANT);
33          this.power = new MageInt(0);
34          this.toughness = new MageInt(13);
35  
36          // Defender
37          this.addAbility(DefenderAbility.getInstance());
38  
39          // {tap}: Exchange target opponent's life total with Tree of Perdition's toughness.
40          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new TreeOfPerditionEffect(), new TapSourceCost());
41          ability.addTarget(new TargetOpponent());
42          this.addAbility(ability);
43      }
44  
45      private TreeOfPerdition(final TreeOfPerdition card) {
46          super(card);
47      }
48  
49      @Override
50      public TreeOfPerdition copy() {
51          return new TreeOfPerdition(this);
52      }
53  }

55  class TreeOfPerditionEffect extends OneShotEffect {
56  
57      public TreeOfPerditionEffect() {
58          super(Outcome.Neutral);
59          staticText = "Exchange target opponent's life total with {this}'s toughness";
60      }
61  
62      public TreeOfPerditionEffect(final TreeOfPerditionEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public boolean apply(Game game, Ability source) {
68          Player opponent = game.getPlayer(source.getFirstTarget());
69          Permanent perm = game.getPermanent(source.getSourceId());
70          if (perm == null || opponent == null || !opponent.isLifeTotalCanChange()) {
71              return false;
72          }
73  
74          int amount = perm.getToughness().getValue(); // Must get total value
75          int life = opponent.getLife();
76          if (life == amount) {
77              return false;
78          }
79          if (life < amount && !opponent.isCanGainLife()) {
80              return false;
81          }
82          if (life > amount && !opponent.isCanLoseLife()) {
83              return false;
84          }
85          opponent.setLife(amount, game, source);
86  
87          // Any toughness-modifying effects, counters, Auras, or Equipment will apply after its toughness is set to the player’s former life total.
88          // For example, say Tree of Perdition is equipped with Cultist’s Staff (which makes it 2/15) and the player’s life total is 7.
89          // After the exchange, Tree of Perdition would be a 2/9 creature (its toughness became 7, which was then modified by Cultist’s Staff) and the player’s life total would be 15.
90          // (2016-07-13)
91          ContinuousEffect powerToughnessEffect = new SetBasePowerToughnessTargetEffect(null, StaticValue.get(life), Duration.Custom);
92          powerToughnessEffect.setTargetPointer(new FixedTarget(perm.getId()));
93          game.addEffect(powerToughnessEffect, source);
94          return true;
95      }
96  
97      @Override
98      public TreeOfPerditionEffect copy() {
99          return new TreeOfPerditionEffect(this);
100     }

24  /**
25   *
26   * @author LoneFox
27   */
28  public final class Thirst extends CardImpl {
29  
30      public Thirst(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{2}{U}");
32          this.subtype.add(SubType.AURA);
33  
34          // Enchant creature
35          TargetPermanent auraTarget = new TargetCreaturePermanent();
36          this.getSpellAbility().addTarget(auraTarget);
37          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
38          this.addAbility(new EnchantAbility(auraTarget));
39          // When Thirst enters the battlefield, tap enchanted creature.
40          this.addAbility(new EntersBattlefieldTriggeredAbility(new TapEnchantedEffect()));
41          // Enchanted creature doesn't untap during its controller's untap step.
42          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new DontUntapInControllersUntapStepEnchantedEffect()));
43          // At the beginning of your upkeep, sacrifice Thirst unless you pay {U}.
44          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new SacrificeSourceUnlessPaysEffect(new ManaCostsImpl<>("{U}")),
45              TargetController.YOU, false));
46      }
47  
48      private Thirst(final Thirst card) {
49          super(card);
50      }
51  
52      @Override
53      public Thirst copy() {
54          return new Thirst(this);
55      }
56  }

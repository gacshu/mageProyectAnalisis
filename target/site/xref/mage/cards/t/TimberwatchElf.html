3   import java.util.UUID;
4   import mage.MageInt;
5   import mage.abilities.common.SimpleActivatedAbility;
6   import mage.abilities.costs.common.TapSourceCost;
7   import mage.abilities.dynamicvalue.DynamicValue;
8   import mage.abilities.dynamicvalue.common.PermanentsOnBattlefieldCount;
9   import mage.abilities.effects.common.continuous.BoostTargetEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.Duration;
14  import mage.constants.SubType;
15  import mage.filter.FilterPermanent;
16  import mage.target.common.TargetCreaturePermanent;
17  
18  /**
19   *
20   * @author LevelX2
21   */
22  public final class TimberwatchElf extends CardImpl {
23  
24      private static final DynamicValue xValue = new PermanentsOnBattlefieldCount(
25              new FilterPermanent(SubType.ELF, "Elves on the battlefield"), null
26      );
27  
28      public TimberwatchElf(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{2}{G}");
30          this.subtype.add(SubType.ELF);
31  
32          this.power = new MageInt(1);
33          this.toughness = new MageInt(2);
34  
35          // {tap}: Target creature gets +X/+X until end of turn, where X is the number of Elves on the battlefield.
36          SimpleActivatedAbility ability = new SimpleActivatedAbility(new BoostTargetEffect(xValue, xValue, Duration.EndOfTurn), new TapSourceCost());
37          ability.addTarget(new TargetCreaturePermanent());
38          this.addAbility(ability);
39      }
40  
41      private TimberwatchElf(final TimberwatchElf card) {
42          super(card);
43      }
44  
45      @Override
46      public TimberwatchElf copy() {
47          return new TimberwatchElf(this);
48      }
49  }

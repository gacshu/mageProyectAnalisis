32  public final class TheFirstTyrannicWar extends CardImpl {
33  
34      public TheFirstTyrannicWar(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{G}{U}{R}");
36  
37          this.subtype.add(SubType.SAGA);
38  
39          // (As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)
40          SagaAbility sagaAbility = new SagaAbility(this);
41  
42          // I -- You may put a creature card from your hand onto the battlefield. If its mana cost contains {X}, it enters the battlefield with a number of +1/+1 counters on it equal to the number of lands you control.
43          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, new TheFirstTyrannicWarFirstEffect());
44  
45          // II, III -- Double the number of each kind of counter on target creature you control.
46          sagaAbility.addChapterEffect(
47                  this, SagaChapter.CHAPTER_II, SagaChapter.CHAPTER_III,
48                  new TheFirstTyrannicWarSecondEffect(), new TargetControlledCreaturePermanent()
49          );
50          this.addAbility(sagaAbility);
51      }
52  
53      private TheFirstTyrannicWar(final TheFirstTyrannicWar card) {
54          super(card);
55      }
56  
57      @Override
58      public TheFirstTyrannicWar copy() {
59          return new TheFirstTyrannicWar(this);
60      }
61  }

63  class TheFirstTyrannicWarFirstEffect extends OneShotEffect {
64  
65      TheFirstTyrannicWarFirstEffect() {
66          super(Outcome.Benefit);
67          staticText = "you may put a creature card from your hand onto the battlefield. " +
68                  "If its mana cost contains {X}, it enters the battlefield " +
69                  "with a number of +1/+1 counters on it equal to the number of lands you control";
70      }
71  
72      private TheFirstTyrannicWarFirstEffect(final TheFirstTyrannicWarFirstEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public TheFirstTyrannicWarFirstEffect copy() {
78          return new TheFirstTyrannicWarFirstEffect(this);
79      }
80  
81      @Override
82      public boolean apply(Game game, Ability source) {
83          Player player = game.getPlayer(source.getControllerId());
84          if (player == null) {
85              return false;
86          }
87          TargetCard target = new TargetCardInHand(0, 1, StaticFilters.FILTER_CARD_CREATURE);
88          player.choose(outcome, player.getHand(), target, game);
89          Card card = game.getCard(target.getFirstTarget());
90          if (card == null) {
91              return false;
92          }
93          if (card.getManaCostSymbols().stream().anyMatch(s -> s.contains("{X}"))) {
94              game.addEffect(new TheFirstTyrannicWarReplacementEffect()
95                      .setTargetPointer(new FixedTarget(card, game)), source);
96          }
97          return player.moveCards(card, Zone.BATTLEFIELD, source, game);
98      }
99  }

101 class TheFirstTyrannicWarReplacementEffect extends ReplacementEffectImpl {
102 
103     TheFirstTyrannicWarReplacementEffect() {
104         super(Duration.EndOfStep, Outcome.BoostCreature);
105     }
106 
107     TheFirstTyrannicWarReplacementEffect(TheFirstTyrannicWarReplacementEffect effect) {
108         super(effect);
109     }
110 
111     @Override
112     public boolean checksEventType(GameEvent event, Game game) {
113         return event.getType() == GameEvent.EventType.ENTERS_THE_BATTLEFIELD;
114     }
115 
116     @Override
117     public boolean applies(GameEvent event, Ability source, Game game) {
118         return event.getTargetId().equals(getTargetPointer().getFirst(game, source));
119     }
120 
121     @Override
122     public boolean apply(Game game, Ability source) {
123         return false;
124     }
125 
126     @Override
127     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
128         Permanent creature = ((EntersTheBattlefieldEvent) event).getTarget();
129         if (creature != null) {
130             creature.addCounters(CounterType.P1P1.createInstance(
131                     LandsYouControlCount.instance.calculate(game, source, this)
132             ), source.getControllerId(), source, game, event.getAppliedEffects());
133             discard();
134         }
135         return false;
136     }
137 
138     @Override
139     public TheFirstTyrannicWarReplacementEffect copy() {
140         return new TheFirstTyrannicWarReplacementEffect(this);
141     }
142 }

145 class TheFirstTyrannicWarSecondEffect extends OneShotEffect {
146 
147     TheFirstTyrannicWarSecondEffect() {
148         super(Outcome.Benefit);
149         staticText = "double the number of each kind of counter on target creature you control";
150     }
151 
152     private TheFirstTyrannicWarSecondEffect(final TheFirstTyrannicWarSecondEffect effect) {
153         super(effect);
154     }
155 
156     @Override
157     public TheFirstTyrannicWarSecondEffect copy() {
158         return new TheFirstTyrannicWarSecondEffect(this);
159     }
160 
161     @Override
162     public boolean apply(Game game, Ability source) {
163         Permanent permanent = game.getPermanent(getTargetPointer().getFirst(game, source));
164         if (permanent == null) {
165             return false;
166         }
167         Set<Counter> counters = permanent
168                 .getCounters(game)
169                 .entrySet()
170                 .stream()
171                 .map(entry -> CounterType
172                         .findByName(entry.getKey())
173                         .createInstance(entry.getValue().getCount()))
174                 .collect(Collectors.toSet());
175         if (counters.isEmpty()) {
176             return false;
177         }
178         for (Counter counter : counters) {
179             permanent.addCounters(counter, source, game);
180         }
181         return true;
182     }

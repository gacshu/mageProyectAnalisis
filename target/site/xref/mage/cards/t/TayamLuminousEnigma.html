42  public final class TayamLuminousEnigma extends CardImpl {

44      public TayamLuminousEnigma(UUID ownerId, CardSetInfo setInfo) {
45          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{W}{B}{G}");
46  
47          this.addSuperType(SuperType.LEGENDARY);
48          this.subtype.add(SubType.NIGHTMARE);
49          this.subtype.add(SubType.BEAST);
50          this.power = new MageInt(3);
51          this.toughness = new MageInt(3);
52  
53          // Each other creature you control enters the battlefield with an additional vigilance counter on it.
54          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new TayamLuminousEnigmaReplacementEffect()));
55  
56          // {3}, Remove three counters from among creatures you control: Put the top three cards of your library into your graveyard, then return a permanent card with converted mana cost 3 or less from your graveyard to the battlefield.
57          MillCardsControllerEffect millEffect = new MillCardsControllerEffect(3);
58          millEffect.concatBy(".");
59          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, millEffect, new GenericManaCost(3));
60          ability.addCost(new TayamLuminousEnigmaCost());
61          TayamLuminousEnigmaEffect effect = new TayamLuminousEnigmaEffect();
62          ability.addEffect(effect);
63          this.addAbility(ability);
64      }

76  class TayamLuminousEnigmaCost extends RemoveCounterCost {
77  
78      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent("a creature with a counter among creatures you control");
79  
80      static {
81          filter.add(CounterAnyPredicate.instance);
82      }
83  
84      public TayamLuminousEnigmaCost() {
85          super(new TargetPermanent(1, 1, filter, true), null, 3);
86      }
87  
88      public TayamLuminousEnigmaCost(TayamLuminousEnigmaCost cost) {
89          super(cost);
90      }
91  
92      @Override
93      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
94          paid = false;
95          int countersRemoved = 0;
96          Player controller = game.getPlayer(controllerId);
97          for (int i = 0; i < countersToRemove; i++) {
98              if (target.choose(Outcome.UnboostCreature, controllerId, source.getSourceId(), source, game)) {
99                  Permanent permanent = game.getPermanent(target.getFirstTarget());
100                 if (permanent != null) {
101                     if (!permanent.getCounters(game).isEmpty()) {
102                         String counterName = null;
103                         if (permanent.getCounters(game).size() > 1) {
104                             Choice choice = new ChoiceImpl(true);
105                             Set<String> choices = new LinkedHashSet<>();
106                             for (Counter counter : permanent.getCounters(game).values()) {
107                                 if (permanent.getCounters(game).getCount(counter.getName()) > 0) {
108                                     choices.add(counter.getName());
109                                 }
110                             }
111                             choice.setChoices(choices);
112                             choice.setMessage("Choose a counter to remove from " + permanent.getLogName());
113                             if (!controller.choose(Outcome.UnboostCreature, choice, game)) {
114                                 return false;
115                             }
116                             counterName = choice.getChoice();
117                         } else {
118                             for (Counter counter : permanent.getCounters(game).values()) {
119                                 if (counter.getCount() > 0) {
120                                     counterName = counter.getName();
121                                 }
122                             }
123                         }
124                         if (counterName != null) {
125                             permanent.removeCounters(counterName, 1, source, game);
126                             target.clearChosen();
127                             if (!game.isSimulation()) {
128                                 game.informPlayers(new StringBuilder(controller.getLogName())
129                                         .append(" removes a ")
130                                         .append(counterName).append(" counter from ")
131                                         .append(permanent.getName()).toString());
132                             }
133                             countersRemoved++;
134                             if (countersRemoved == countersToRemove) {
135                                 paid = true;
136                                 break;
137                             }
138                         }
139                     }
140                 }
141             } else {
142                 break;
143             }
144         }
145         return paid;
146     }

149     public TayamLuminousEnigmaCost copy() {
150         return new TayamLuminousEnigmaCost(this);
151     }

155         return "Remove three counters from among creatures you control";
156     }

159 class TayamLuminousEnigmaEffect extends OneShotEffect {
160 
161     private static final FilterCard filter = new FilterPermanentCard("permanent card in your graveyard with mana value 3 or less");
162 
163     static {
164         filter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 4));
165     }
166 
167     TayamLuminousEnigmaEffect() {
168         super(Outcome.Benefit);
169         staticText = ", then return a permanent card with mana value 3 or less from your graveyard to the battlefield";
170     }
171 
172     private TayamLuminousEnigmaEffect(TayamLuminousEnigmaEffect effect) {
173         super(effect);
174     }
175 
176     @Override
177     public TayamLuminousEnigmaEffect copy() {
178         return new TayamLuminousEnigmaEffect(this);
179     }
180 
181     @Override
182     public boolean apply(Game game, Ability source) {
183         Player player = game.getPlayer(source.getControllerId());
184         if (player == null || player.getGraveyard().count(filter, game) == 0) {
185             return false;
186         }
187         TargetCard target = new TargetCardInYourGraveyard(filter);
188         target.setNotTarget(true);
189         if (!player.choose(outcome, player.getGraveyard(), target, game)) {
190             return false;
191         }
192         return player.moveCards(game.getCard(target.getFirstTarget()), Zone.BATTLEFIELD, source, game);
193     }

196 class TayamLuminousEnigmaReplacementEffect extends ReplacementEffectImpl {
197 
198     TayamLuminousEnigmaReplacementEffect() {
199         super(Duration.WhileOnBattlefield, Outcome.BoostCreature);
200         staticText = "Each other creature you control enters the battlefield with an additional vigilance counter on it.";
201     }
202 
203     private TayamLuminousEnigmaReplacementEffect(final TayamLuminousEnigmaReplacementEffect effect) {
204         super(effect);
205     }
206 
207     @Override
208     public boolean checksEventType(GameEvent event, Game game) {
209         return event.getType() == GameEvent.EventType.ENTERS_THE_BATTLEFIELD;
210     }
211 
212     @Override
213     public boolean applies(GameEvent event, Ability source, Game game) {
214         Permanent creature = ((EntersTheBattlefieldEvent) event).getTarget();
215         return creature != null
216                 && creature.isCreature(game)
217                 && !source.getSourceId().equals(creature.getId())
218                 && creature.isControlledBy(source.getControllerId());
219     }
220 
221     @Override
222     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
223         Permanent creature = ((EntersTheBattlefieldEvent) event).getTarget();
224         if (creature != null) {
225             creature.addCounters(CounterType.VIGILANCE.createInstance(), source.getControllerId(), source, game, event.getAppliedEffects());
226         }
227         return false;
228     }
229 
230     @Override
231     public TayamLuminousEnigmaReplacementEffect copy() {
232         return new TayamLuminousEnigmaReplacementEffect(this);
233     }

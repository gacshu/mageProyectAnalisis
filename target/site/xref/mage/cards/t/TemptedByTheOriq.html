25  public final class TemptedByTheOriq extends CardImpl {
26  
27      public TemptedByTheOriq(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{U}{U}{U}");
29  
30          // For each opponent, gain control of up to one target creature or planeswalker that player controls with mana value 3 or less.
31          this.getSpellAbility().addEffect(new GainControlTargetEffect(Duration.Custom, true)
32                  .setTargetPointer(new EachTargetPointer())
33                  .setText("for each opponent, gain control of up to one target creature " +
34                          "or planeswalker that player controls with mana value 3 or less"));
35          this.getSpellAbility().setTargetAdjuster(TemptedByTheOriqAdjuster.instance);
36      }
37  
38      private TemptedByTheOriq(final TemptedByTheOriq card) {
39          super(card);
40      }
41  
42      @Override
43      public TemptedByTheOriq copy() {
44          return new TemptedByTheOriq(this);
45      }
46  }
47  
48  enum TemptedByTheOriqAdjuster implements TargetAdjuster {
49      instance;
50  
51      @Override
52      public void adjustTargets(Ability ability, Game game) {
53          ability.getTargets().clear();
54          for (UUID opponentId : game.getOpponents(ability.getControllerId())) {
55              Player opponent = game.getPlayer(opponentId);
56              if (opponent == null) {
57                  continue;
58              }
59              FilterPermanent filter = new FilterCreatureOrPlaneswalkerPermanent(
60                      "creature or planeswalker " + opponent.getName() + " controls with mana value 3 or less"
61              );
62              filter.add(new ControllerIdPredicate(opponentId));
63              filter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 4));
64              ability.addTarget(new TargetPermanent(0, 1, filter, false));
65          }
66      }
67  }

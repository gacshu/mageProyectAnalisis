29  public final class ThiefOfSanity extends CardImpl {
30  
31      protected static final String VALUE_PREFIX = "ExileZones";
32  
33      public ThiefOfSanity(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{U}{B}");
35  
36          this.subtype.add(SubType.SPECTER);
37          this.power = new MageInt(2);
38          this.toughness = new MageInt(2);
39  
40          // Flying
41          this.addAbility(FlyingAbility.getInstance());
42  
43          // Whenever Thief of Sanity deals combat damage to a player, look at the top three cards of that player's library, exile one of them face down, then put the rest into their graveyard. For as long as that card remains exiled, you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
44          this.addAbility(new DealsCombatDamageToAPlayerTriggeredAbility(new ThiefOfSanityEffect(), false, true));
45      }
46  
47      private ThiefOfSanity(final ThiefOfSanity card) {
48          super(card);
49      }
50  
51      @Override
52      public ThiefOfSanity copy() {
53          return new ThiefOfSanity(this);
54      }
55  }

57  class ThiefOfSanityEffect extends OneShotEffect {
58  
59      public ThiefOfSanityEffect() {
60          super(Outcome.Benefit);
61          this.staticText = "look at the top three cards of that player's library, exile one of them face down, then put the rest into their graveyard. "
62                  + "For as long as that card remains exiled, you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it";
63      }
64  
65      private ThiefOfSanityEffect(final ThiefOfSanityEffect effect) {
66          super(effect);
67      }
68  
69      @Override
70      public ThiefOfSanityEffect copy() {
71          return new ThiefOfSanityEffect(this);
72      }
73  
74      @Override
75      public boolean apply(Game game, Ability source) {
76          Player controller = game.getPlayer(source.getControllerId());
77          Player damagedPlayer = game.getPlayer(getTargetPointer().getFirst(game, source));
78          MageObject sourceObject = source.getSourceObject(game);
79          if (controller != null && damagedPlayer != null && sourceObject != null) {
80              Cards topCards = new CardsImpl();
81              topCards.addAll(damagedPlayer.getLibrary().getTopCards(game, 3));
82              TargetCard target = new TargetCard(Zone.LIBRARY, new FilterCard("card to exile face down"));
83              if (controller.choose(outcome, topCards, target, game)) {
84                  Card card = game.getCard(target.getFirstTarget());
85                  if (card != null) {
86                      topCards.remove(card);
87                      // move card to exile
88                      UUID exileZoneId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
89                      card.setFaceDown(true, game);
90                      if (controller.moveCardsToExile(card, source, game, false, exileZoneId, sourceObject.getIdName() + " (" + controller.getName() + ")")) {
91                          card.setFaceDown(true, game);
92                          Set<UUID> exileZones = (Set<UUID>) game.getState().getValue(ThiefOfSanity.VALUE_PREFIX + source.getSourceId().toString());
93                          if (exileZones == null) {
94                              exileZones = new HashSet<>();
95                              game.getState().setValue(ThiefOfSanity.VALUE_PREFIX + source.getSourceId().toString(), exileZones);
96                          }
97                          exileZones.add(exileZoneId);
98                          // rule information: https://blogs.magicjudges.org/rulestips/2018/11/thief-of-sanity-and-control-changing/
99                          // allow to cast the card
100                         ContinuousEffect effect = new ThiefOfSanityCastFromExileEffect(controller.getId());
101                         effect.setTargetPointer(new FixedTarget(card.getId(), game));
102                         game.addEffect(effect, source);
103                         // and you may spend mana as though it were mana of any color to cast it
104                         effect = new ThiefOfSanitySpendAnyManaEffect(controller.getId());
105                         effect.setTargetPointer(new FixedTarget(card.getId(), game));
106                         game.addEffect(effect, source);
107                         // For as long as that card remains exiled, you may look at it
108                         effect = new ThiefOfSanityLookEffect(controller.getId());
109                         effect.setTargetPointer(new FixedTarget(card.getId(), game));
110                         game.addEffect(effect, source);
111                     }
112                 }
113             }
114             // put the rest into their graveyard
115             controller.moveCards(topCards, Zone.GRAVEYARD, source, game);
116             return true;
117         }
118 
119         return false;
120     }
121 }

123 class ThiefOfSanityCastFromExileEffect extends AsThoughEffectImpl {
124 
125     private final UUID authorizedPlayerId;
126 
127     public ThiefOfSanityCastFromExileEffect(UUID authorizedPlayerId) {
128         super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.Custom, Outcome.Benefit);
129         this.authorizedPlayerId = authorizedPlayerId;
130         staticText = "For as long as that card remains exiled, you may cast it";
131     }
132 
133     private ThiefOfSanityCastFromExileEffect(final ThiefOfSanityCastFromExileEffect effect) {
134         super(effect);
135         this.authorizedPlayerId = effect.authorizedPlayerId;
136     }
137 
138     @Override
139     public boolean apply(Game game, Ability source) {
140         return true;
141     }
142 
143     @Override
144     public ThiefOfSanityCastFromExileEffect copy() {
145         return new ThiefOfSanityCastFromExileEffect(this);
146     }
147 
148     @Override
149     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
150         UUID cardId = getTargetPointer().getFirst(game, source);
151         if (cardId == null) {
152             this.discard(); // card is no longer in the origin zone, effect can be discarded
153             return false;
154         }
155         Card theCard = game.getCard(objectId);
156         if (theCard == null || theCard.isLand(game)) {
157             return false;
158         }
159         objectId = theCard.getMainCard().getId(); // for split cards
160 
161         if (objectId.equals(cardId)
162                 && affectedControllerId.equals(authorizedPlayerId)) {
163             Card card = game.getCard(objectId);
164             // TODO: Allow to cast Zoetic Cavern face down
165             return card != null;
166         }
167         return false;
168     }
169 }

171 class ThiefOfSanitySpendAnyManaEffect extends AsThoughEffectImpl implements AsThoughManaEffect {
172 
173     private final UUID authorizedPlayerId;
174 
175     public ThiefOfSanitySpendAnyManaEffect(UUID authorizedPlayerId) {
176         super(AsThoughEffectType.SPEND_OTHER_MANA, Duration.Custom, Outcome.Benefit);
177         this.authorizedPlayerId = authorizedPlayerId;
178         staticText = "For as long as that card remains exiled, you may spend mana as though it were mana of any color to cast it";
179     }
180 
181     private ThiefOfSanitySpendAnyManaEffect(final ThiefOfSanitySpendAnyManaEffect effect) {
182         super(effect);
183         this.authorizedPlayerId = effect.authorizedPlayerId;
184     }
185 
186     @Override
187     public boolean apply(Game game, Ability source) {
188         return true;
189     }
190 
191     @Override
192     public ThiefOfSanitySpendAnyManaEffect copy() {
193         return new ThiefOfSanitySpendAnyManaEffect(this);
194     }
195 
196     @Override
197     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
198         objectId = CardUtil.getMainCardId(game, objectId); // for split cards
199         if (objectId.equals(((FixedTarget) getTargetPointer()).getTarget())
200                 && game.getState().getZoneChangeCounter(objectId) <= ((FixedTarget) getTargetPointer()).getZoneChangeCounter() + 1) {
201             // if the card moved from exile to spell the zone change counter is increased by 1 (effect must applies before and on stack, use isCheckPlayableMode?)
202             return affectedControllerId.equals(authorizedPlayerId);
203         } else if (((FixedTarget) getTargetPointer()).getTarget().equals(objectId)) {
204             // object has moved zone so effect can be discarded
205             this.discard();
206         }
207         return false;
208     }
209 
210     @Override
211     public ManaType getAsThoughManaType(ManaType manaType, ManaPoolItem mana, UUID affectedControllerId, Ability source, Game game) {
212         return mana.getFirstAvailable();
213     }

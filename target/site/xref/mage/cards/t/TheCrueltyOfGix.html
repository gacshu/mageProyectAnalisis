9   import mage.abilities.effects.common.search.SearchLibraryPutInHandEffect;
10  import mage.constants.SagaChapter;
11  import mage.constants.SubType;
12  import mage.cards.CardImpl;

16  import mage.filter.common.FilterCreatureCard;
17  import mage.filter.predicate.Predicates;
18  import mage.target.common.TargetCardInGraveyard;
19  import mage.target.common.TargetCardInLibrary;
20  import mage.target.common.TargetOpponent;
21  
22  /**
23   *
24   * @author weirddan455
25   */
26  public final class TheCrueltyOfGix extends CardImpl {
27  
28      private static final FilterCard filter = new FilterCard("creature or planeswalker card");
29      private static final FilterCreatureCard filter2 = new FilterCreatureCard("creature card from a graveyard");
30  
31      static {
32          filter.add(Predicates.or(
33                  CardType.CREATURE.getPredicate(),
34                  CardType.PLANESWALKER.getPredicate()
35          ));
36      }
37  
38      public TheCrueltyOfGix(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{B}{B}");
40  
41          this.subtype.add(SubType.SAGA);
42  
43          // Read ahead
44          SagaAbility sagaAbility = new SagaAbility(this, SagaChapter.CHAPTER_III, true);
45  
46          // I -- Target opponent reveals their hand. You choose a creature or planeswalker card from it. That player discards that card.
47          sagaAbility.addChapterEffect(
48                  this, SagaChapter.CHAPTER_I, SagaChapter.CHAPTER_I,
49                  new DiscardCardYouChooseTargetEffect(filter), new TargetOpponent()
50          );
51  
52          // II -- Search your library for a card, put that card into your hand, then shuffle. You lose 3 life.
53          sagaAbility.addChapterEffect(
54                  this, SagaChapter.CHAPTER_II,
55                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary()),
56                  new LoseLifeSourceControllerEffect(3)
57          );
58  
59          // III -- Put target creature card from a graveyard onto the battlefield under your control.
60          sagaAbility.addChapterEffect(
61                  this, SagaChapter.CHAPTER_III, SagaChapter.CHAPTER_III,
62                  new ReturnFromGraveyardToBattlefieldTargetEffect(),
63                  new TargetCardInGraveyard(filter2)
64          );
65          this.addAbility(sagaAbility);
66      }
67  
68      private TheCrueltyOfGix(final TheCrueltyOfGix card) {
69          super(card);
70      }
71  
72      @Override
73      public TheCrueltyOfGix copy() {
74          return new TheCrueltyOfGix(this);
75      }
76  }

26  public final class TheFallOfKroog extends CardImpl {
27  
28      public TheFallOfKroog(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{R}{R}");
30  
31          // Choose target opponent. Destroy target land that player controls. The Fall of Kroog deals 3 damage to that player and 1 damage to each creature they control.
32          this.getSpellAbility().addEffect(new DestroyTargetEffect()
33                  .setText("choose target opponent. Destroy target land that player controls")
34                  .setTargetPointer(new SecondTargetPointer()));
35          this.getSpellAbility().addEffect(new DamageTargetEffect(
36                  3, true, "that player"
37          ));
38          this.getSpellAbility().addEffect(new DamageAllControlledTargetEffect(
39                  1, StaticFilters.FILTER_PERMANENT_CREATURE
40          ).setText("and 1 damage to each creature they control"));
41          this.getSpellAbility().addTarget(new TargetOpponent());
42          this.getSpellAbility().addTarget(new TheFallOfKroogTarget());
43      }
44  
45      private TheFallOfKroog(final TheFallOfKroog card) {
46          super(card);
47      }
48  
49      @Override
50      public TheFallOfKroog copy() {
51          return new TheFallOfKroog(this);
52      }
53  }

55  class TheFallOfKroogTarget extends TargetPermanent {
56  
57      TheFallOfKroogTarget() {
58          super(1, 1, StaticFilters.FILTER_LAND, false);
59      }
60  
61      private TheFallOfKroogTarget(final TheFallOfKroogTarget target) {
62          super(target);
63      }
64  
65      @Override
66      public boolean canTarget(UUID id, Ability source, Game game) {
67          Player player = game.getPlayer(source.getFirstTarget());
68          Permanent permanent = game.getPermanent(id);
69          return player != null
70                  && permanent != null
71                  && permanent.isControlledBy(player.getId())
72                  && super.canTarget(id, source, game);
73      }
74  
75      @Override
76      public Set<UUID> possibleTargets(UUID sourceControllerId, Ability source, Game game) {
77          Player player = Optional
78                  .ofNullable(game.getObject(source))
79                  .filter(StackObject.class::isInstance)
80                  .map(StackObject.class::cast)
81                  .map(StackObject::getStackAbility)
82                  .map(Ability::getFirstTarget)
83                  .map(game::getPlayer)
84                  .orElse(null);
85          if (player == null) {
86              return new HashSet<>();
87          }
88          return super
89                  .possibleTargets(sourceControllerId, source, game)
90                  .stream()
91                  .map(game::getPermanent)
92                  .filter(Objects::nonNull)
93                  .filter(permanent -> permanent.isControlledBy(player.getId()))
94                  .map(MageItem::getId)
95                  .collect(Collectors.toSet());
96      }
97  
98      @Override
99      public TheFallOfKroogTarget copy() {
100         return new TheFallOfKroogTarget(this);
101     }
102 }

27  public final class TormentOfScarabs extends CardImpl {
28  
29      public TormentOfScarabs(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{B}");
31  
32          this.subtype.add(SubType.AURA, SubType.CURSE);
33  
34          // Enchant player
35          TargetPlayer auraTarget = new TargetPlayer();
36          this.getSpellAbility().addTarget(auraTarget);
37          this.getSpellAbility().addEffect(new AttachEffect(Outcome.LoseLife));
38          Ability ability = new EnchantAbility(auraTarget);
39          this.addAbility(ability);
40  
41          // At the beginning of enchanted player's upkeep, that player loses 3 life unless they sacrifice a nonland permanent or discards a card.
42          this.addAbility(new BeginningOfUpkeepTriggeredAbility(
43                  new TormentOfScarabsEffect(), TargetController.ENCHANTED, false
44          ));
45      }
46  
47      private TormentOfScarabs(final TormentOfScarabs card) {
48          super(card);
49      }
50  
51      @Override
52      public TormentOfScarabs copy() {
53          return new TormentOfScarabs(this);
54      }
55  }

57  class TormentOfScarabsEffect extends OneShotEffect {
58  
59      TormentOfScarabsEffect() {
60          super(Outcome.LoseLife);
61          this.staticText = "that player loses 3 life unless they sacrifice a nonland permanent or discard a card";
62      }
63  
64      private TormentOfScarabsEffect(final TormentOfScarabsEffect effect) {
65          super(effect);
66      }
67  
68      @Override
69      public TormentOfScarabsEffect copy() {
70          return new TormentOfScarabsEffect(this);
71      }
72  
73      @Override
74      public boolean apply(Game game, Ability source) {
75          Player enchantedPlayer = game.getPlayer(targetPointer.getFirst(game, source));
76          if (enchantedPlayer == null) {
77              return false;
78          }
79          int permanents = game.getBattlefield().countAll(StaticFilters.FILTER_PERMANENT_NON_LAND, enchantedPlayer.getId(), game);
80          if (permanents > 0 && enchantedPlayer.chooseUse(outcome, "Sacrifice a nonland permanent?",
81                  "Otherwise you have to discard a card or lose 3 life.", "Sacrifice", "Discard or life loss", source, game)) {
82              Target target = new TargetPermanent(StaticFilters.FILTER_CONTROLLED_PERMANENT_NON_LAND);
83              if (enchantedPlayer.choose(outcome, target, source, game)) {
84                  Permanent permanent = game.getPermanent(target.getFirstTarget());
85                  if (permanent != null) {
86                      permanent.sacrifice(source, game);
87                      return true;
88                  }
89              }
90          }
91          if (!enchantedPlayer.getHand().isEmpty() && enchantedPlayer.chooseUse(outcome, "Discard a card?",
92                  "Otherwise you lose 3 life.", "Discard", "Lose 3 life", source, game)) {
93              enchantedPlayer.discardOne(false, false, source, game);
94              return true;
95          }
96          enchantedPlayer.loseLife(3, game, source, false);
97          return true;
98      }

21  /**
22   * @author Alex-Vasile
23   */
24  public class ThreefoldSignal extends CardImpl {
25  
26      private static final FilterSpell filter = new FilterSpell("spell you cast that's exactly three colors");
27      static {
28          filter.add(ThreeColorPredicate.instance);
29      }
30  
31      public ThreefoldSignal(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
33  
34          // When Threefold Signal enters the battlefield, scry 3.
35          this.addAbility(new EntersBattlefieldTriggeredAbility(new ScryEffect(3)));
36  
37          // Each spell you cast that's exactly three colors has replicate {3}.
38          //  (When you cast it, copy it for each time you paid its replicate cost.
39          //   You may choose new targets for the copies.
40          //   A copy of a permanent spell becomes a token.)
41          this.addAbility(new SimpleStaticAbility(new EachSpellYouCastHasReplicateEffect(filter, new GenericManaCost(3))));
42      }
43  
44      private ThreefoldSignal(final ThreefoldSignal card) {
45          super(card);
46      }
47  
48      @Override
49      public ThreefoldSignal copy() {
50          return new ThreefoldSignal(this);
51      }
52  }
53  
54  /**
55   * Based on MultiColorPredicate
56   */
57  enum ThreeColorPredicate implements Predicate<MageObject> {
58      instance;
59  
60      @Override
61      public boolean apply(MageObject input, Game game) {
62          return 3 == input.getColor(game).getColorCount();
63      }
64  
65      @Override
66      public String toString() {
67          return "Multicolored";
68      }
69  }

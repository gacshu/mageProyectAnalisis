21  import java.util.UUID;

23  /**
24   * @author Loki
25   */
26  public final class TimberProtector extends CardImpl {
27  
28      private static final FilterCreaturePermanent filterTreefolk
29              = new FilterCreaturePermanent("Treefolk creatures");
30      private static final FilterControlledPermanent filterBoth
31              = new FilterControlledPermanent("other Treefolk and Forests you control");
32  
33      static {
34          filterTreefolk.add(SubType.TREEFOLK.getPredicate());
35          filterBoth.add(TimberProtectorPredicate.instance);
36      }
37  
38      public TimberProtector(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{G}");
40          this.subtype.add(SubType.TREEFOLK);
41          this.subtype.add(SubType.WARRIOR);
42  
43          this.power = new MageInt(4);
44          this.toughness = new MageInt(6);
45  
46          // Other Treefolk creatures you control get +1/+1.
47          this.addAbility(new SimpleStaticAbility(new BoostControlledEffect(
48                  1, 1, Duration.WhileOnBattlefield, filterTreefolk, true
49          )));
50  
51          // Other Treefolk and Forests you control are indestructible.
52          this.addAbility(new SimpleStaticAbility(new GainAbilityAllEffect(
53                  IndestructibleAbility.getInstance(), Duration.WhileOnBattlefield, filterBoth
54          )));
55      }
56  
57      private TimberProtector(final TimberProtector card) {
58          super(card);
59      }
60  
61      @Override
62      public TimberProtector copy() {
63          return new TimberProtector(this);
64      }
65  }
66  
67  enum TimberProtectorPredicate implements ObjectSourcePlayerPredicate<Permanent> {
68      instance;
69  
70      @Override
71      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
72          MageObject obj = input.getObject();
73          if (obj.getId().equals(input.getSourceId())) {
74              return obj.hasSubtype(SubType.FOREST, game);
75          }
76          return obj.hasSubtype(SubType.FOREST, game)
77                  || obj.hasSubtype(SubType.TREEFOLK, game);
78      }
79  }

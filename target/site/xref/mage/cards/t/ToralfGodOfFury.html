40  public final class ToralfGodOfFury extends ModalDoubleFacesCard {
41  
42      private static final Condition condition
43              = new AttachedToMatchesFilterCondition(StaticFilters.FILTER_PERMANENT_LEGENDARY);
44  
45      public ToralfGodOfFury(UUID ownerId, CardSetInfo setInfo) {
46          super(ownerId, setInfo,
47                  new CardType[]{CardType.CREATURE}, new SubType[]{SubType.GOD}, "{2}{R}{R}",
48                  "Toralf's Hammer", new CardType[]{CardType.ARTIFACT}, new SubType[]{SubType.EQUIPMENT}, "{1}{R}"
49          );
50  
51          // 1.
52          // Toralf, God of Fury
53          // Legendary Creature - God
54          this.getLeftHalfCard().addSuperType(SuperType.LEGENDARY);
55          this.getLeftHalfCard().setPT(new MageInt(5), new MageInt(4));
56  
57          // Trample
58          this.getLeftHalfCard().addAbility(TrampleAbility.getInstance());
59  
60          // Whenever a creature or planeswalker an opponent controls is dealt excess noncombat damage, Toralf, God of Fury deals damage equal to the excess to any target other than that permanent.
61          this.getLeftHalfCard().addAbility(new ToralfGodOfFuryTriggeredAbility());
62  
63          // 2.
64          // Toralf's Hammer
65          // Legendary Artifact - Equipment
66          this.getRightHalfCard().addSuperType(SuperType.LEGENDARY);
67  
68          // Equipped creature has "{1}{R}, {T}, Unattach Toralf's Hammer: It deals 3 damage to any target. Return Toralf's Hammer to its owner's hand."
69          this.getRightHalfCard().addAbility(new SimpleStaticAbility(new GainAbilityWithAttachmentEffect(
70                  "equipped creature has \"{1}{R}, {T}, Unattach {this}: It deals 3 damage to any target. Return {this} to its owner's hand.\"",
71                  new ToralfsHammerEffect(), new TargetAnyTarget(), new UnattachCost(), new ManaCostsImpl<>("{1}{R}"), new TapSourceCost()
72          )));
73  
74          // Equipped creature get +3/+0 as long as it's legendary.
75          this.getRightHalfCard().addAbility(new SimpleStaticAbility(new ConditionalContinuousEffect(
76                  new BoostEquippedEffect(3, 0), condition,
77                  "equipped creature gets +3/+0 as long as it's legendary"
78          )));
79  
80          // Equip {1}{R}
81          this.getRightHalfCard().addAbility(new EquipAbility(Outcome.BoostCreature, new ManaCostsImpl<>("{1}{R}"), false));
82      }

94  class ToralfGodOfFuryTriggeredAbility extends TriggeredAbilityImpl {
95  
96      ToralfGodOfFuryTriggeredAbility() {
97          super(Zone.BATTLEFIELD, null);
98      }
99  
100     private ToralfGodOfFuryTriggeredAbility(final ToralfGodOfFuryTriggeredAbility ability) {
101         super(ability);
102     }
103 
104     @Override
105     public boolean checkEventType(GameEvent event, Game game) {
106         return event.getType() == GameEvent.EventType.DAMAGED_PERMANENT;
107     }
108 
109     @Override
110     public boolean checkTrigger(GameEvent event, Game game) {
111         DamagedEvent dEvent = (DamagedEvent) event;
112         if (dEvent.getExcess() < 1
113                 || dEvent.isCombatDamage()
114                 || !game.getOpponents(getControllerId()).contains(game.getControllerId(event.getTargetId()))) {
115             return false;
116         }
117         this.getEffects().clear();
118         this.getTargets().clear();
119         int excessDamage = dEvent.getExcess();
120         this.addEffect(new DamageTargetEffect(excessDamage));
121         FilterCreaturePlayerOrPlaneswalker filter = new FilterCreaturePlayerOrPlaneswalker();
122         filter.getPermanentFilter().add(Predicates.not(new MageObjectReferencePredicate(new MageObjectReference(event.getTargetId(), game))));
123         this.addTarget(new TargetAnyTarget(filter).withChooseHint(Integer.toString(excessDamage) + " damage"));
124         return true;
125     }
126 
127     @Override
128     public ToralfGodOfFuryTriggeredAbility copy() {
129         return new ToralfGodOfFuryTriggeredAbility(this);
130     }
131 
132     @Override
133     public String getRule() {
134         return "Whenever a creature or planeswalker an opponent controls is dealt excess noncombat damage, " +
135                 "{this} deals damage equal to the excess to any target other than that permanent.";
136     }
137 }

139 class ToralfsHammerEffect extends OneShotEffect {
140 
141     ToralfsHammerEffect() {
142         super(Outcome.Benefit);
143     }
144 
145     private ToralfsHammerEffect(final ToralfsHammerEffect effect) {
146         super(effect);
147     }
148 
149     @Override
150     public ToralfsHammerEffect copy() {
151         return new ToralfsHammerEffect(this);
152     }
153 
154     @Override
155     public boolean apply(Game game, Ability source) {
156         Object object = getValue("attachedPermanent");
157         Player player = game.getPlayer(source.getControllerId());
158         if (!(object instanceof Permanent) || player == null) {
159             return false;
160         }
161         Permanent permanent = (Permanent) object;
162         Permanent targetedPermanent = game.getPermanent(source.getFirstTarget());
163         if (targetedPermanent == null) {
164             Player targetedPlayer = game.getPlayer(source.getFirstTarget());
165             if (targetedPlayer != null) {
166                 targetedPlayer.damage(3, permanent.getId(), source, game);
167             }
168         } else {
169             targetedPermanent.damage(3, permanent.getId(), source, game);
170         }
171         player.moveCards(permanent, Zone.HAND, source, game);
172         return true;
173     }
174 
175     @Override
176     public String getText(Mode mode) {
177         String name = "Toralf's Hammer";
178         Object object = getValue("attachedPermanent");
179         if (object instanceof Permanent) {
180             name = ((Permanent) object).getName();
181         }
182         return "It deals 3 damage to any target. Return " + name + " to its owner's hand.";
183     }

31      public TormentOfVenom(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{2}{B}{B}");
33  
34          // Put three -1/-1 counters on target creature. Its controller loses 3 life unless they sacrifice another nonland permanent or discards a card.
35          this.getSpellAbility().addEffect(new TormentOfVenomEffect());
36          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
37      }

40          super(card);
41      }

49  class TormentOfVenomEffect extends OneShotEffect {
50  
51      public TormentOfVenomEffect() {
52          super(Outcome.LoseLife);
53          this.staticText = "Put three -1/-1 counters on target creature. Its controller loses 3 life unless they sacrifice another nonland permanent or discards a card";
54      }
55  
56      public TormentOfVenomEffect(final TormentOfVenomEffect effect) {
57          super(effect);
58      }
59  
60      @Override
61      public TormentOfVenomEffect copy() {
62          return new TormentOfVenomEffect(this);
63      }
64  
65      @Override
66      public boolean apply(Game game, Ability source) {
67          Permanent targetCreature = game.getPermanent(getTargetPointer().getFirst(game, source));
68          if (targetCreature != null) {
69              new AddCountersTargetEffect(CounterType.M1M1.createInstance(3)).apply(game, source);
70              Player controllingPlayer = game.getPlayer(targetCreature.getControllerId());
71              if (controllingPlayer != null) {
72                  int permanents = game.getBattlefield().countAll(StaticFilters.FILTER_PERMANENT_NON_LAND, controllingPlayer.getId(), game);
73                  if (permanents > 0 && controllingPlayer.chooseUse(outcome, "Sacrifices a nonland permanent?",
74                          "Otherwise you have to discard a card or lose 3 life.", "Sacrifice", "Discard or life loss", source, game)) {
75                      FilterPermanent filter = new FilterControlledPermanent("another nonland permanent");
76                      filter.add(Predicates.not(CardType.LAND.getPredicate()));
77                      filter.add(Predicates.not(new PermanentIdPredicate(targetCreature.getId())));
78                      Target target = new TargetPermanent(filter);
79                      if (controllingPlayer.choose(outcome, target, source, game)) {
80                          Permanent permanent = game.getPermanent(target.getFirstTarget());
81                          if (permanent != null) {
82                              permanent.sacrifice(source, game);
83                              return true;
84                          }
85                      }
86                  }
87                  if (!controllingPlayer.getHand().isEmpty() && controllingPlayer.chooseUse(outcome, "Discard a card?",
88                          "Otherwise you lose 3 life.", "Discard", "Lose 3 life", source, game)) {
89                      controllingPlayer.discardOne(false, false, source, game);
90                      return true;
91                  }
92                  controllingPlayer.loseLife(3, game, source, false);
93                  return true;
94              }
95          }
96  
97          return false;
98      }
99  }

28  public final class TibaltTheFiendBlooded extends CardImpl {
29  
30      public TibaltTheFiendBlooded(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{R}{R}");
32          this.addSuperType(SuperType.LEGENDARY);
33          this.subtype.add(SubType.TIBALT);
34  
35          this.setStartingLoyalty(2);
36  
37          // +1: Draw a card, then discard a card at random.
38          Ability ability = new LoyaltyAbility(new DrawCardSourceControllerEffect(1), 1);
39          ability.addEffect(new DiscardControllerEffect(1, true).concatBy(", then"));
40          this.addAbility(ability);
41  
42          // -4: Tibalt, the Fiend-Blooded deals damage equal to the number of cards in target player's hand to that player.
43          Effect effect = new DamageTargetEffect(CardsInTargetHandCount.instance);
44          effect.setText("{this} deals damage equal to the number of cards in target player's hand to that player");
45          ability = new LoyaltyAbility(effect, -4);
46          ability.addTarget(new TargetPlayer());
47          this.addAbility(ability);
48  
49          // -6: Gain control of all creatures until end of turn. Untap them. They gain haste until end of turn.
50          ability = new LoyaltyAbility(new GainControlAllEffect(Duration.EndOfTurn, StaticFilters.FILTER_PERMANENT_CREATURES), -6);
51          ability.addEffect(new UntapAllEffect(StaticFilters.FILTER_PERMANENT_CREATURES).setText("untap them"));
52          ability.addEffect(new GainAbilityAllEffect(
53                  HasteAbility.getInstance(),
54                  Duration.EndOfTurn,
55                  StaticFilters.FILTER_PERMANENT_CREATURES,
56                  "they gain haste until end of turn"
57          ));
58          this.addAbility(ability);
59      }
60  
61      private TibaltTheFiendBlooded(final TibaltTheFiendBlooded card) {
62          super(card);
63      }
64  
65      @Override
66      public TibaltTheFiendBlooded copy() {
67          return new TibaltTheFiendBlooded(this);
68      }
69  }

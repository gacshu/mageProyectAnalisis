40      public TheBookOfVileDarkness(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{B}{B}{B}");
42  
43          this.addSuperType(SuperType.LEGENDARY);
44  
45          // At the beginning of your end step, if you lost 2 or more life this turn, create a 2/2 black Zombie creature token.
46          this.addAbility(new BeginningOfEndStepTriggeredAbility(
47                  Zone.BATTLEFIELD, new CreateTokenEffect(new ZombieToken()), TargetController.YOU,
48                  TheBookOfVileDarknessCondition.instance, false
49          ));
50  
51          // {T}, Exile The Book of Vile Darkness and artifacts you control named Eye of Vecna and Hand of Vecna: Create Vecna, a legendary 8/8 black Zombie God creature token with indestructible and all triggered abilities of the exiled cards.
52          Ability ability = new SimpleActivatedAbility(new TheBookOfVileDarknessEffect(), new TapSourceCost());
53          ability.addCost(new TheBookOfVileDarknessCost());
54          this.addAbility(ability);
55      }
56  
57      private TheBookOfVileDarkness(final TheBookOfVileDarkness card) {
58          super(card);
59      }

62      public TheBookOfVileDarkness copy() {
63          return new TheBookOfVileDarkness(this);
64      }

67  enum TheBookOfVileDarknessCondition implements Condition {
68      instance;
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          PlayerLostLifeWatcher watcher = game.getState().getWatcher(PlayerLostLifeWatcher.class);
73          return watcher != null && watcher.getLifeLost(source.getControllerId()) >= 2;
74      }
75  
76      @Override
77      public String toString() {
78          return "if you lost 2 or more life this turn";
79      }

82  class TheBookOfVileDarknessCost extends CostImpl {
83  
84      private static final FilterPermanent filter1
85              = new FilterControlledArtifactPermanent("artifact you control named Eye of Vecna");
86      private static final FilterPermanent filter2
87              = new FilterControlledArtifactPermanent("artifact you control named Hand of Vecna");
88  
89      static {
90          filter1.add(new NamePredicate("Eye of Vecna"));
91          filter2.add(new NamePredicate("Hand of Vecna"));
92      }
93  
94      public TheBookOfVileDarknessCost() {
95          this.text = "exile {this} and artifacts you control named Eye of Vecna and Hand of Vecna";
96      }
97  
98      public TheBookOfVileDarknessCost(final TheBookOfVileDarknessCost cost) {
99          super(cost);
100     }
101 
102     @Override
103     public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
104         Player controller = game.getPlayer(controllerId);
105         Permanent sourcePermanent = ability.getSourcePermanentIfItStillExists(game);
106         if (controller == null || sourcePermanent == null) {
107             return paid;
108         }
109         Permanent eye = getPermanent(filter1, controller, ability, game);
110         if (eye == null) {
111             return paid;
112         }
113         Permanent hand = getPermanent(filter2, controller, ability, game);
114         if (hand == null) {
115             return paid;
116         }
117         controller.moveCards(new CardsImpl(Arrays.asList(sourcePermanent, eye, hand)), Zone.EXILED, source, game);
118         Set<MageObjectReference> morSet = new HashSet<>();
119         morSet.add(new MageObjectReference(sourcePermanent, game));
120         morSet.add(new MageObjectReference(eye, game));
121         morSet.add(new MageObjectReference(hand, game));
122         ability.getEffects().setValue("BookEyeHand", morSet);
123         paid = true;
124         return paid;
125     }
126 
127     @Override
128     public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
129         return source.getSourcePermanentIfItStillExists(game) != null
130                 && game.getBattlefield().count(filter1, source.getControllerId(), source, game) > 0
131                 && game.getBattlefield().count(filter2, source.getControllerId(), source, game) > 0;
132     }
133 
134     private static Permanent getPermanent(FilterPermanent filter, Player controller, Ability source, Game game) {
135         int count = game.getBattlefield().count(filter, source.getControllerId(), source, game);
136         switch (count) {
137             case 0:
138                 return null;
139             case 1:
140                 return game.getBattlefield().getActivePermanents(
141                         filter, source.getControllerId(), source, game
142                 ).stream().findFirst().orElse(null);
143             default:
144                 break;
145         }
146         TargetPermanent target = new TargetPermanent(filter);
147         target.setNotTarget(true);
148         controller.choose(Outcome.Sacrifice, target, source, game);
149         return game.getPermanent(target.getFirstTarget());
150     }
151 
152     @Override
153     public TheBookOfVileDarknessCost copy() {
154         return new TheBookOfVileDarknessCost(this);
155     }
156 }

158 class TheBookOfVileDarknessEffect extends OneShotEffect {
159 
160     TheBookOfVileDarknessEffect() {
161         super(Outcome.Benefit);
162         staticText = "create Vecna, a legendary 8/8 black Zombie God creature token "
163                 + "with indestructible and all triggered abilities of the exiled cards";
164     }
165 
166     private TheBookOfVileDarknessEffect(final TheBookOfVileDarknessEffect effect) {
167         super(effect);
168     }
169 
170     @Override
171     public TheBookOfVileDarknessEffect copy() {
172         return new TheBookOfVileDarknessEffect(this);
173     }
174 
175     @Override
176     public boolean apply(Game game, Ability source) {
177         Set<MageObjectReference> morSet = (Set<MageObjectReference>) getValue("BookEyeHand");
178         if (morSet == null) {
179             return false;
180         }
181         Token token = new VecnaToken();
182         for (MageObjectReference mor : morSet) {
183             // the card object in the mor doesn't work, so the permanent object is used
184             Permanent card = mor.getPermanentOrLKIBattlefield(game);
185             if (card == null) {
186                 continue;
187             }
188             for (Ability ability : card.getAbilities(game)) {
189                 if (ability instanceof TriggeredAbility) {
190                     Ability copyAbility = ability.copy();
191                     copyAbility.newId();
192                     copyAbility.setControllerId(source.getControllerId());
193                     token.addAbility(copyAbility);
194                 }
195             }
196         }
197         return token.putOntoBattlefield(1, game, source, source.getControllerId());
198     }

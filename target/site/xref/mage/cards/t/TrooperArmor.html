9   import mage.abilities.effects.common.continuous.AddCardSubtypeAttachedEffect;
10  import mage.abilities.effects.common.continuous.BoostEquippedEffect;
11  import mage.abilities.keyword.EquipAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.*;
15  import mage.filter.FilterPermanent;
16  
17  /**
18   *
19   * @author Styxo
20   */
21  public final class TrooperArmor extends CardImpl {
22  
23      public TrooperArmor(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
25          this.subtype.add(SubType.EQUIPMENT);
26  
27          // Equiped creature gets +1/+1 and is a Trooper in addition to its other types.
28          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new BoostEquippedEffect(1, 1)));
29          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new AddCardSubtypeAttachedEffect(SubType.TROOPER, Duration.WhileOnBattlefield, AttachmentType.EQUIPMENT)));
30  
31          // Whenever a Trooper enters the battlefield under your control, you may attach {this} to it.
32          this.addAbility(new EntersBattlefieldControlledTriggeredAbility(
33                  Zone.BATTLEFIELD,
34                  new AttachEffect(Outcome.BoostCreature, "attach {this} to it"),
35                  new FilterPermanent(SubType.TROOPER, "Trooper"),
36                  true,
37                  SetTargetPointer.PERMANENT,
38                  null));
39          // Equip {2}
40          this.addAbility(new EquipAbility(Outcome.BoostCreature, new GenericManaCost(2)));
41      }
42  
43      private TrooperArmor(final TrooperArmor card) {
44          super(card);
45      }
46  
47      @Override
48      public TrooperArmor copy() {
49          return new TrooperArmor(this);
50      }
51  }

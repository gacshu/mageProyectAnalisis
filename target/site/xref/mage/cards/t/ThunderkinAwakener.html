35  public final class ThunderkinAwakener extends CardImpl {
36  
37      private static final FilterCard filter
38              = new FilterCreatureCard("Elemental creature card in your graveyard with lesser toughness");
39  
40      static {
41          filter.add(SubType.ELEMENTAL.getPredicate());
42          filter.add(ThunderkinAwakenerPredicate.instance);
43      }
44  
45      public ThunderkinAwakener(UUID ownerId, CardSetInfo setInfo) {
46          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{R}");
47          this.subtype.add(SubType.ELEMENTAL);
48          this.subtype.add(SubType.SHAMAN);
49          this.power = new MageInt(1);
50          this.toughness = new MageInt(2);
51  
52          // Haste
53          this.addAbility(HasteAbility.getInstance());
54  
55          // Whenever Thunderkin Awakener attacks, choose target Elemental creature card in your graveyard
56          // with toughness less than Thunderkin Awakener's toughness. Return that card to the battlefield tapped and attacking.
57          // Sacrifice it at the beginning of the next end step.
58          Ability ability = new AttacksTriggeredAbility(new ThunderkinAwakenerEffect(), false);
59          ability.addTarget(new TargetCardInYourGraveyard(filter));
60          this.addAbility(ability);
61      }
62  
63      private ThunderkinAwakener(final ThunderkinAwakener card) {
64          super(card);
65      }
66  
67      @Override
68      public ThunderkinAwakener copy() {
69          return new ThunderkinAwakener(this);
70      }
71  }
72  
73  enum ThunderkinAwakenerPredicate implements ObjectSourcePlayerPredicate<Card> {
74      instance;
75  
76      @Override
77      public boolean apply(ObjectSourcePlayer<Card> input, Game game) {
78          Permanent sourcePermanent = input.getSource().getSourcePermanentOrLKI(game);
79          return sourcePermanent != null
80                  && input.getObject().getToughness().getValue() < sourcePermanent.getToughness().getValue();
81      }
82  }
83  
84  class ThunderkinAwakenerEffect extends OneShotEffect {
85  
86      ThunderkinAwakenerEffect() {
87          super(Outcome.Benefit);
88          staticText = "choose target Elemental creature card in your graveyard with toughness less than {this}'s toughness."
89                  + " Return that card to the battlefield tapped and attacking. Sacrifice it at the beginning of the next end step";
90      }
91  
92      private ThunderkinAwakenerEffect(final ThunderkinAwakenerEffect effect) {
93          super(effect);
94      }
95  
96      @Override
97      public ThunderkinAwakenerEffect copy() {
98          return new ThunderkinAwakenerEffect(this);
99      }
100 
101     @Override
102     public boolean apply(Game game, Ability source) {
103         Player controller = game.getPlayer(source.getControllerId());
104         Card creatureCard = game.getCard(this.getTargetPointer().getFirst(game, source));
105         if (controller != null && creatureCard != null) {
106 
107             // Return that card to the battlefield tapped and attacking
108             Effect effect = new ReturnFromGraveyardToBattlefieldTargetEffect(true, true);
109             effect.setTargetPointer(new FixedTarget(creatureCard.getId()));
110             effect.apply(game, source);
111 
112             Permanent permanent = game.getPermanent(getTargetPointer().getFirst(game, source));
113             if (permanent != null) {
114                 // Sacrifice it at the beginning of the next end step
115                 SacrificeTargetEffect sacrificeEffect = new SacrificeTargetEffect("Sacrifice " + permanent.getName(), source.getControllerId());
116                 sacrificeEffect.setTargetPointer(new FixedTarget(permanent, game));
117                 DelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(sacrificeEffect);
118                 game.addDelayedTriggeredAbility(delayedAbility, source);
119 
120                 // info
121                 InfoEffect.addInfoToPermanent(game, source, permanent, "<i><b>Warning</b>: It will be sacrificed at the beginning of the next end step<i>");
122             }
123 
124             return true;
125         }
126         return false;
127     }

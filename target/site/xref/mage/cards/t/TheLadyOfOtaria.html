31  public final class TheLadyOfOtaria extends CardImpl {
32  
33      private static final FilterControlledPermanent filter
34              = new FilterControlledPermanent(SubType.DWARF, "untapped Dwarves you control");
35      private static final FilterCard filter2 = new FilterCard("Dwarf cards");
36  
37      static {
38          filter.add(TappedPredicate.UNTAPPED);
39          filter2.add(SubType.DWARF.getPredicate());
40      }
41  
42      public TheLadyOfOtaria(UUID ownerId, CardSetInfo setInfo) {
43          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}{G}");
44  
45          this.addSuperType(SuperType.LEGENDARY);
46          this.subtype.add(SubType.AVATAR);
47          this.power = new MageInt(5);
48          this.toughness = new MageInt(5);
49  
50          // You may tap three untapped Dwarves you control rather than pay this spell's mana cost.
51          this.addAbility(new AlternativeCostSourceAbility(
52                  new TapTargetCost(new TargetControlledPermanent(3, filter))
53          ));
54  
55          // At the beginning of each end step, if a land you controlled was put into your graveyard from the battlefield this turn, reveal the top four cards of your library. Put any number of Dwarf cards from among them into your hand and the rest on the bottom of your library in a random order.
56          this.addAbility(new BeginningOfEndStepTriggeredAbility(
57                  new RevealLibraryPickControllerEffect(
58                          4, Integer.MAX_VALUE, filter2, PutCards.HAND, PutCards.BOTTOM_RANDOM, false
59                  ), TargetController.ANY, TheLadyOfOtariaCondition.instance, false
60          ).addHint(TheLadyOfOtariaCondition.getHint()), new TheLadyOfOtariaWatcher());
61      }
62  
63      private TheLadyOfOtaria(final TheLadyOfOtaria card) {
64          super(card);
65      }
66  
67      @Override
68      public TheLadyOfOtaria copy() {
69          return new TheLadyOfOtaria(this);
70      }
71  }

73  enum TheLadyOfOtariaCondition implements Condition {
74      instance;
75      private static final Hint hint = new ConditionHint(
76              instance, "A land you controlled was put into your graveyard"
77      );
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          return false;
82      }
83  
84      @Override
85      public String toString() {
86          return "if a land you controlled was put into your graveyard from the battlefield this turn";
87      }

94  class TheLadyOfOtariaWatcher extends Watcher {
95  
96      private final Set<UUID> playerMap = new HashSet<>();
97  
98      TheLadyOfOtariaWatcher() {
99          super(WatcherScope.GAME);
100     }
101 
102     @Override
103     public void watch(GameEvent event, Game game) {
104         if (event.getType() != GameEvent.EventType.ZONE_CHANGE) {
105             return;
106         }
107         ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
108         if (zEvent.isDiesEvent()
109                 && zEvent.getTarget() != null
110                 && zEvent.getTarget().isLand(game)
111                 && zEvent.getTarget().isOwnedBy(zEvent.getTarget().getControllerId())) {
112             playerMap.add(zEvent.getTarget().getControllerId());
113         }
114     }
115 
116     @Override
117     public void reset() {
118         super.reset();
119         playerMap.clear();
120     }
121 
122     static boolean checkPlayer(Ability source, Game game) {
123         return game
124                 .getState()
125                 .getWatcher(TheLadyOfOtariaWatcher.class)
126                 .playerMap
127                 .contains(source.getControllerId());
128     }

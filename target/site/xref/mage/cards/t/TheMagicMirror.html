9   import mage.abilities.dynamicvalue.common.CountersSourceCount;
10  import mage.abilities.effects.common.DrawCardSourceControllerEffect;

12  import mage.abilities.effects.common.cost.SpellCostReductionForEachSourceEffect;

15  import mage.cards.CardImpl;
16  import mage.cards.CardSetInfo;
17  import mage.constants.*;
18  import mage.counters.CounterType;
19  import mage.filter.StaticFilters;

21  /**
22   * @author TheElk801
23   */
24  public final class TheMagicMirror extends CardImpl {
25  
26      public TheMagicMirror(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{6}{U}{U}{U}");
28  
29          this.addSuperType(SuperType.LEGENDARY);
30  
31          // This spell costs {1} less to cast for each instant and sorcery card in your graveyard.
32          DynamicValue xValue = new CardsInControllerGraveyardCount(StaticFilters.FILTER_CARD_INSTANT_AND_SORCERY);
33          Ability ability = new SimpleStaticAbility(Zone.ALL, new SpellCostReductionForEachSourceEffect(1, xValue));
34          ability.setRuleAtTheTop(true);
35          ability.addHint(new ValueHint("Instant and sorcery card in your graveyard", xValue));
36          this.addAbility(ability);
37  
38          // You have no maximum hand size.
39          this.addAbility(new SimpleStaticAbility(new MaximumHandSizeControllerEffect(
40                  Integer.MAX_VALUE, Duration.WhileOnBattlefield,
41                  MaximumHandSizeControllerEffect.HandSizeModification.SET
42          )));
43  
44          // At the beginning of your upkeep, put a knowledge counter on The Magic Mirror, then draw a card for each knowledge counter on The Magic Mirror.
45          ability = new BeginningOfUpkeepTriggeredAbility(
46                  new AddCountersSourceEffect(CounterType.KNOWLEDGE.createInstance())
47                          .setText("put a knowledge counter on {this},"),
48                  TargetController.YOU, false);
49          ability.addEffect(new DrawCardSourceControllerEffect(new CountersSourceCount(CounterType.KNOWLEDGE)).concatBy("then"));
50          this.addAbility(ability);
51      }
52  
53      private TheMagicMirror(final TheMagicMirror card) {
54          super(card);
55      }
56  
57      @Override
58      public TheMagicMirror copy() {
59          return new TheMagicMirror(this);
60      }
61  }

28   * @author TheElk801
29   */
30  public final class ThaumaticCompass extends CardImpl {
31  
32      public ThaumaticCompass(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
34          this.secondSideCardClazz = mage.cards.s.SpiresOfOrazca.class;
35  
36          // {3}, {T}: Search your library for a basic land card, reveal it, put it into your hand, then shuffle your library.
37          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD,
38                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary(StaticFilters.FILTER_CARD_BASIC_LAND), true),
39                  new GenericManaCost(3));
40          ability.addCost(new TapSourceCost());
41          this.addAbility(ability);
42  
43          // At the beginning of your end step, if you control seven or more lands, transform Thaumatic Compass.
44          this.addAbility(new TransformAbility());
45          TriggeredAbility ability2 = new BeginningOfEndStepTriggeredAbility(new TransformSourceEffect(), TargetController.YOU, false);
46          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
47                  ability2,
48                  new PermanentsOnTheBattlefieldCondition(new FilterLandPermanent(), ComparisonType.MORE_THAN, 6, true),
49                  "At the beginning of your end step, if you control seven or more lands, transform {this}."));
50      }
51  
52      private ThaumaticCompass(final ThaumaticCompass card) {
53          super(card);
54      }
55  
56      @Override
57      public ThaumaticCompass copy() {
58          return new ThaumaticCompass(this);
59      }
60  }

25  public final class TheRestorationOfEiganjo extends CardImpl {
26  
27      private static final FilterCard filter
28              = new FilterCard("a basic Plains card");
29      private static final FilterCard filter2
30              = new FilterPermanentCard("permanent card with mana value 2 or less from your graveyard");
31  
32      static {
33          filter.add(SubType.PLAINS.getPredicate());
34          filter.add(SuperType.BASIC.getPredicate());
35          filter2.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 3));
36      }
37  
38      public TheRestorationOfEiganjo(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
40  
41          this.subtype.add(SubType.SAGA);
42          this.secondSideCardClazz = mage.cards.a.ArchitectOfRestoration.class;
43  
44          // (As this Saga enters and after your draw step, add a lore counter.)
45          SagaAbility sagaAbility = new SagaAbility(this);
46  
47          // I - Search your library for a basic Plains card, reveal it, put it into your hand, then shuffle.
48          sagaAbility.addChapterEffect(
49                  this, SagaChapter.CHAPTER_I, new SearchLibraryPutInHandEffect(
50                          new TargetCardInLibrary(filter), true, true
51                  )
52          );
53  
54          // II â€” You may discard a card. When you do, return target permanent card with mana value 2 or less from your graveyard to the battlefield tapped
55          ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
56                  new ReturnFromGraveyardToBattlefieldTargetEffect(true), false
57          );
58          ability.addTarget(new TargetCardInYourGraveyard(filter2));
59          sagaAbility.addChapterEffect(
60                  this, SagaChapter.CHAPTER_II, new DoWhenCostPaid(
61                          ability, new DiscardCardCost(), "Discard a card?"
62                  )
63          );
64  
65          // III â€” Exile this Saga, then return it to the battlefield transformed under your control.
66          this.addAbility(new TransformAbility());
67          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new ExileSagaAndReturnTransformedEffect());
68  
69          this.addAbility(sagaAbility);
70      }
71  
72      private TheRestorationOfEiganjo(final TheRestorationOfEiganjo card) {
73          super(card);
74      }
75  
76      @Override
77      public TheRestorationOfEiganjo copy() {
78          return new TheRestorationOfEiganjo(this);
79      }
80  }

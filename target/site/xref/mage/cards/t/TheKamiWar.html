17  import mage.filter.common.FilterNonlandPermanent;
18  import mage.filter.predicate.mageobject.AnotherPredicate;
19  import mage.target.TargetPermanent;
20  
21  import java.util.UUID;
22  
23  /**
24   * @author TheElk801
25   */
26  public final class TheKamiWar extends CardImpl {
27  
28      private static final FilterPermanent filter
29              = new FilterNonlandPermanent("nonland permanent an opponent controls");
30      private static final FilterPermanent filter2
31              = new FilterNonlandPermanent("other target nonland permanent");
32  
33      static {
34          filter.add(TargetController.OPPONENT.getControllerPredicate());
35          filter2.add(AnotherPredicate.instance);
36      }
37  
38      public TheKamiWar(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}{U}{B}{R}{G}");
40  
41          this.subtype.add(SubType.SAGA);
42          this.secondSideCardClazz = mage.cards.o.OKagachiMadeManifest.class;
43  
44          // (As this Saga enters and after your draw step, add a lore counter.)
45          SagaAbility sagaAbility = new SagaAbility(this);
46  
47          // I — Exile target nonland permanent an opponent controls.
48          sagaAbility.addChapterEffect(
49                  this, SagaChapter.CHAPTER_I, SagaChapter.CHAPTER_I,
50                  new ExileTargetEffect(), new TargetPermanent(filter)
51          );
52  
53          // II — Return up to one other target nonland permanent to its owner's hand. Then each opponent discards a card.
54          sagaAbility.addChapterEffect(
55                  this, SagaChapter.CHAPTER_II, SagaChapter.CHAPTER_II,
56                  new Effects(
57                          new ReturnToHandTargetEffect(),
58                          new DiscardEachPlayerEffect(TargetController.OPPONENT)
59                                  .concatBy("Then")
60                  ), new TargetPermanent(0, 1, filter2)
61          );
62  
63          // III — Exile this Saga, then return it to the battlefield transformed under your control.
64          this.addAbility(new TransformAbility());
65          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new ExileSagaAndReturnTransformedEffect());
66  
67          this.addAbility(sagaAbility);
68      }
69  
70      private TheKamiWar(final TheKamiWar card) {
71          super(card);
72      }
73  
74      @Override
75      public TheKamiWar copy() {
76          return new TheKamiWar(this);
77      }
78  }

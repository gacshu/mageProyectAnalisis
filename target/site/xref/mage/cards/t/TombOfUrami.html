25  /**
26   *
27   * @author Plopman
28   */
29  public final class TombOfUrami extends CardImpl {
30  
31      private static final FilterControlledPermanent filter = new FilterControlledPermanent("Ogre");
32  
33      static {
34          filter.add(SubType.OGRE.getPredicate());
35      }
36  
37      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter, ComparisonType.EQUAL_TO, 0);
38  
39      public TombOfUrami(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
41          this.addSuperType(SuperType.LEGENDARY);
42  
43          // {tap}: Add {B}. Tomb of Urami deals 1 damage to you if you don't control an Ogre.
44          Ability ability = new BlackManaAbility();
45          ability.addEffect(new ConditionalOneShotEffect(
46                  new DamageControllerEffect(1),
47                  condition,
48                  "{this} deals 1 damage to you if you don't control an Ogre"
49          ));
50          this.addAbility(ability);
51  
52          // {2}{B}{B}, {tap}, Sacrifice all lands you control: Create a legendary 5/5 black Demon Spirit creature token with flying named Urami.
53          Ability ability2 = new SimpleActivatedAbility(new CreateTokenEffect(new UramiToken()), new ManaCostsImpl<>("{2}{B}{B}"));
54          ability2.addCost(new TapSourceCost());
55          ability2.addCost(new SacrificeAllCost(StaticFilters.FILTER_CONTROLLED_PERMANENT_LANDS));
56          this.addAbility(ability2);
57      }
58  
59      private TombOfUrami(final TombOfUrami card) {
60          super(card);
61      }
62  
63      @Override
64      public TombOfUrami copy() {
65          return new TombOfUrami(this);
66      }
67  }

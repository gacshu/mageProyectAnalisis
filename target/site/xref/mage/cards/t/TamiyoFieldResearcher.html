40  public final class TamiyoFieldResearcher extends CardImpl {
41  
42      private static final FilterPermanent filter = new FilterPermanent("nonland permanent");
43  
44      static {
45          filter.add(Predicates.not(CardType.LAND.getPredicate()));
46      }
47  
48      public TamiyoFieldResearcher(UUID ownerId, CardSetInfo setInfo) {
49          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{1}{G}{W}{U}");
50          this.addSuperType(SuperType.LEGENDARY);
51          this.subtype.add(SubType.TAMIYO);
52  
53          this.setStartingLoyalty(4);
54  
55          // +1: Choose up to two target creatures. Until your next turn, whenever either of those creatures deals combat damage, you draw a card.
56          Ability ability = new LoyaltyAbility(new TamiyoFieldResearcherEffect1(), 1);
57          ability.addTarget(new TargetCreaturePermanent(0, 2, StaticFilters.FILTER_PERMANENT_CREATURES, false));
58          this.addAbility(ability);
59  
60          // -2: Tap up to two target nonland permanents. They don't untap during their controller's next untap step.
61          ability = new LoyaltyAbility(new TapTargetEffect(), -2);
62          ability.addTarget(new TargetPermanent(0, 2, filter, false));
63          ability.addEffect(new DontUntapInControllersNextUntapStepTargetEffect("They"));
64          this.addAbility(ability);
65  
66          // -7: Draw three cards. You get an emblem with "You may cast nonland cards from your hand without paying their mana costs."
67          ability = new LoyaltyAbility(new DrawCardSourceControllerEffect(3), -7);
68          ability.addEffect(new GetEmblemEffect(new TamiyoFieldResearcherEmblem()));
69          this.addAbility(ability);
70      }
71  
72      private TamiyoFieldResearcher(final TamiyoFieldResearcher card) {
73          super(card);
74      }
75  
76      @Override
77      public TamiyoFieldResearcher copy() {
78          return new TamiyoFieldResearcher(this);
79      }
80  }

82  class TamiyoFieldResearcherEffect1 extends OneShotEffect {
83  
84      public TamiyoFieldResearcherEffect1() {
85          super(Outcome.PreventDamage);
86          this.staticText = "Choose up to two target creatures. Until your next turn, whenever either of those creatures deals combat damage, you draw a card";
87      }
88  
89      public TamiyoFieldResearcherEffect1(final TamiyoFieldResearcherEffect1 effect) {
90          super(effect);
91      }
92  
93      @Override
94      public TamiyoFieldResearcherEffect1 copy() {
95          return new TamiyoFieldResearcherEffect1(this);
96      }
97  
98      @Override
99      public boolean apply(Game game, Ability source) {
100         Player controller = game.getPlayer(source.getControllerId());
101         if (controller != null) {
102             List<MageObjectReference> creatures = new ArrayList<>();
103             for (UUID uuid : getTargetPointer().getTargets(game, source)) {
104                 creatures.add(new MageObjectReference(uuid, game));
105             }
106             if (!creatures.isEmpty()) {
107                 DelayedTriggeredAbility delayedAbility = new TamiyoFieldResearcherDelayedTriggeredAbility(creatures, game.getTurnNum());
108                 game.addDelayedTriggeredAbility(delayedAbility, source);
109             }
110             return true;
111         }
112         return false;
113     }
114 }

116 class TamiyoFieldResearcherDelayedTriggeredAbility extends DelayedTriggeredAbility {
117 
118     private int startingTurn;
119     private List<MageObjectReference> creatures;
120 
121     public TamiyoFieldResearcherDelayedTriggeredAbility(List<MageObjectReference> creatures, int startingTurn) {
122         super(new DrawCardSourceControllerEffect(1), Duration.Custom, false);
123         this.creatures = creatures;
124         this.startingTurn = startingTurn;
125     }
126 
127     public TamiyoFieldResearcherDelayedTriggeredAbility(final TamiyoFieldResearcherDelayedTriggeredAbility ability) {
128         super(ability);
129         this.creatures = ability.creatures;
130         this.startingTurn = ability.startingTurn;
131     }
132 
133     @Override
134     public boolean checkEventType(GameEvent event, Game game) {
135         return event instanceof DamagedEvent;
136     }
137 
138     @Override
139     public boolean checkTrigger(GameEvent event, Game game) {
140         if (((DamagedEvent) event).isCombatDamage()) {
141             Permanent damageSource = game.getPermanent(event.getSourceId());
142             if (damageSource != null) {
143                 return creatures.contains(new MageObjectReference(damageSource, game));
144             }
145         }
146         return false;
147     }
148 
149     @Override
150     public boolean isInactive(Game game) {
151         return game.isActivePlayer(getControllerId()) && game.getTurnNum() != startingTurn;
152     }
153 
154     @Override
155     public TamiyoFieldResearcherDelayedTriggeredAbility copy() {
156         return new TamiyoFieldResearcherDelayedTriggeredAbility(this);
157     }
158 
159     @Override
160     public String getRule() {
161         return "Until your next turn, whenever either of those creatures deals combat damage, you draw a card.";
162     }
163 }

5   import java.util.UUID;
6   import mage.abilities.Ability;
7   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
8   import mage.abilities.common.SimpleStaticAbility;
9   import mage.abilities.effects.OneShotEffect;
10  import mage.abilities.effects.common.AttachEffect;
11  import mage.abilities.effects.common.continuous.ControlEnchantedEffect;
12  import mage.abilities.keyword.EnchantAbility;
13  import mage.cards.CardImpl;
14  import mage.cards.CardSetInfo;
15  import mage.constants.CardType;
16  import mage.constants.SubType;
17  import mage.constants.Outcome;
18  import mage.constants.Zone;
19  import mage.game.Game;
20  import mage.game.permanent.Permanent;
21  import mage.target.TargetPermanent;
22  
23  /**
24   * @author nantuko
25   */
26  public final class VolitionReins extends CardImpl {
27  
28      public VolitionReins(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{3}{U}{U}{U}");
30          this.subtype.add(SubType.AURA);
31  
32          // Enchant permanent
33          TargetPermanent auraTarget = new TargetPermanent();
34          this.getSpellAbility().addTarget(auraTarget);
35          this.getSpellAbility().addEffect(new AttachEffect(Outcome.GainControl));
36          Ability ability = new EnchantAbility(auraTarget);
37          this.addAbility(ability);
38          // When Volition Reins enters the battlefield, if enchanted permanent is tapped, untap it.
39          this.addAbility(new EntersBattlefieldTriggeredAbility(new UntapVolitionReinsEffect()));
40          // You control enchanted permanent.
41          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new ControlEnchantedEffect("permanent")));
42      }
43  
44      private VolitionReins(final VolitionReins card) {
45          super(card);
46      }
47  
48      @Override
49      public VolitionReins copy() {
50          return new VolitionReins(this);
51      }
52  
53      public static class UntapVolitionReinsEffect extends OneShotEffect {
54  
55          public UntapVolitionReinsEffect() {
56              super(Outcome.Untap);
57              staticText = "if enchanted permanent is tapped, untap it";
58          }
59  
60          public UntapVolitionReinsEffect(final UntapVolitionReinsEffect effect) {
61              super(effect);
62          }
63  
64          @Override
65          public UntapVolitionReinsEffect copy() {
66              return new UntapVolitionReinsEffect(this);
67          }
68  
69          @Override
70          public boolean apply(Game game, Ability source) {
71              Permanent enchantment = game.getPermanent(source.getSourceId());
72              if (enchantment != null && enchantment.getAttachedTo() != null) {
73                  Permanent permanent = game.getPermanent(enchantment.getAttachedTo());
74                  if (permanent != null && permanent.isTapped()) {
75                      permanent.untap(game);
76                      return true;
77                  }
78              }
79              return false;
80          }
81  
82      }
83  
84  }

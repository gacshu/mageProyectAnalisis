10  import mage.abilities.keyword.WardAbility;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.SubType;
15  import mage.constants.SuperType;
16  import mage.filter.FilterSpell;
17  import mage.filter.predicate.Predicates;
18  import mage.game.permanent.token.WolfToken;
19  
20  import java.util.UUID;
21  
22  /**
23   * @author TheElk801
24   */
25  public final class VarisSilverymoonRanger extends CardImpl {
26  
27      private static final FilterSpell filter = new FilterSpell("a creature or planeswalker spell");
28  
29      static {
30          filter.add(Predicates.or(
31                  CardType.CREATURE.getPredicate(),
32                  CardType.PLANESWALKER.getPredicate()
33          ));
34      }
35  
36      public VarisSilverymoonRanger(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{G}{G}");
38  
39          this.addSuperType(SuperType.LEGENDARY);
40          this.subtype.add(SubType.HUMAN);
41          this.subtype.add(SubType.ELF);
42          this.subtype.add(SubType.RANGER);
43          this.power = new MageInt(3);
44          this.toughness = new MageInt(3);
45  
46          // Reach
47          this.addAbility(ReachAbility.getInstance());
48  
49          // Ward {1}
50          this.addAbility(new WardAbility(new ManaCostsImpl<>("{1}"), false));
51  
52          // Whenever you cast a creature or planeswalker spell, venture into the dungeon. This ability triggers only once each turn.
53          this.addAbility(new SpellCastControllerTriggeredAbility(
54                  new VentureIntoTheDungeonEffect(), filter, false
55          ).setTriggersOnce(true));
56  
57          // Whenever you complete a dungeon, create a 2/2 green Wolf creature token.
58          this.addAbility(new CompletedDungeonTriggeredAbility(new CreateTokenEffect(new WolfToken())));
59      }
60  
61      private VarisSilverymoonRanger(final VarisSilverymoonRanger card) {
62          super(card);
63      }
64  
65      @Override
66      public VarisSilverymoonRanger copy() {
67          return new VarisSilverymoonRanger(this);
68      }
69  }

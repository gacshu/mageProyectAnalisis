31  public final class VarinaLichQueen extends CardImpl {
32  
33      public VarinaLichQueen(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{W}{U}{B}");
35  
36          this.addSuperType(SuperType.LEGENDARY);
37          this.subtype.add(SubType.ZOMBIE);
38          this.subtype.add(SubType.WIZARD);
39          this.power = new MageInt(4);
40          this.toughness = new MageInt(4);
41  
42          // Whenever you attack with one or more Zombies, draw that many cards, then discard that many cards. You gain that much life.
43          this.addAbility(new VarinaLichQueenTriggeredAbility());
44  
45          // {2}, Exile two cards from your graveyard: Create a tapped 2/2 black Zombie creature token.
46          Ability ability = new SimpleActivatedAbility(
47                  Zone.BATTLEFIELD,
48                  new CreateTokenEffect(
49                          new ZombieToken(),
50                          1, true, false
51                  ), new GenericManaCost(2)
52          );
53          ability.addCost(new ExileFromGraveCost(new TargetCardInYourGraveyard(
54                  2, StaticFilters.FILTER_CARDS_FROM_YOUR_GRAVEYARD
55          )));
56          this.addAbility(ability);
57      }
58  
59      private VarinaLichQueen(final VarinaLichQueen card) {
60          super(card);
61      }
62

64      public VarinaLichQueen copy() {
65          return new VarinaLichQueen(this);

69  class VarinaLichQueenTriggeredAbility extends TriggeredAbilityImpl {
70  
71      public VarinaLichQueenTriggeredAbility() {
72          super(Zone.BATTLEFIELD, null, false);
73      }
74  
75      public VarinaLichQueenTriggeredAbility(final VarinaLichQueenTriggeredAbility ability) {
76          super(ability);
77      }
78  
79      @Override
80      public VarinaLichQueenTriggeredAbility copy() {
81          return new VarinaLichQueenTriggeredAbility(this);
82      }
83  
84      @Override
85      public boolean checkEventType(GameEvent event, Game game) {
86          return event.getType() == GameEvent.EventType.DECLARED_ATTACKERS;
87      }
88  
89      @Override
90      public boolean checkTrigger(GameEvent event, Game game) {
91          int attackingZombies = 0;
92          for (UUID attacker : game.getCombat().getAttackers()) {
93              Permanent creature = game.getPermanent(attacker);
94              if (creature != null
95                      && creature.getControllerId() != null
96                      && creature.isControlledBy(this.getControllerId())
97                      && creature.hasSubtype(SubType.ZOMBIE, game)) {
98                  attackingZombies++;
99              }
100         }
101         if (attackingZombies > 0) {
102             this.getEffects().clear();
103             addEffect(new DrawCardSourceControllerEffect(attackingZombies));
104             addEffect(new DiscardControllerEffect(attackingZombies, false));
105             addEffect(new GainLifeEffect(attackingZombies));
106             return true;
107         }
108         return false;
109     }
110 
111     @Override
112     public String getRule() {
113         return "Whenever you attack with one or more Zombies, "
114                 + "draw that many cards, then discard that many cards. "
115                 + "You gain that much life.";
116     }

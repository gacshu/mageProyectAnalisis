34  public final class VizierOfTheAnointed extends CardImpl {
35  
36      private static final FilterCard filter = new FilterCard("a creature card with eternalize or embalm");
37  
38      static {
39          filter.add(CardType.CREATURE.getPredicate());
40          filter.add(new VizierOfTheAnointedAbilityPredicate());
41      }
42  
43      public VizierOfTheAnointed(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
45  
46          this.subtype.add(SubType.HUMAN);
47          this.subtype.add(SubType.CLERIC);
48          this.power = new MageInt(2);
49          this.toughness = new MageInt(4);
50  
51          // When Vizier of the Anointed enters the battlefield, you may search your library for a creature card with eternalize or embalm, put that card into your graveyard, then shuffle your library.
52          this.addAbility(new EntersBattlefieldTriggeredAbility(new SearchLibraryPutInGraveyard(filter), true));
53  
54          // Whenever you activate an eternalize or embalm ability, draw a card.
55          this.addAbility(new VizierOfTheAnointedTriggeredAbility());
56  
57      }
58  
59      private VizierOfTheAnointed(final VizierOfTheAnointed card) {
60          super(card);
61      }
62  
63      @Override
64      public VizierOfTheAnointed copy() {
65          return new VizierOfTheAnointed(this);
66      }
67  }

69  class VizierOfTheAnointedAbilityPredicate implements Predicate<MageObject> {
70  
71      public VizierOfTheAnointedAbilityPredicate() {
72      }
73  
74      @Override
75      public boolean apply(MageObject input, Game game) {
76          Abilities<Ability> abilities = input.getAbilities();
77          for (int i = 0; i < abilities.size(); i++) {
78              if (abilities.get(i) instanceof EmbalmAbility) {
79                  return true;
80              }
81              if (abilities.get(i) instanceof EternalizeAbility) {
82                  return true;
83              }
84          }
85          return false;
86      }
87  
88      @Override
89      public String toString() {
90          return "Creature card with Eternalize or Embalm";
91      }
92  }

94  class VizierOfTheAnointedTriggeredAbility extends TriggeredAbilityImpl {
95  
96      VizierOfTheAnointedTriggeredAbility() {
97          super(Zone.BATTLEFIELD, new DrawCardSourceControllerEffect(1), false);
98      }
99  
100     VizierOfTheAnointedTriggeredAbility(final VizierOfTheAnointedTriggeredAbility ability) {
101         super(ability);
102     }
103 
104     @Override
105     public VizierOfTheAnointedTriggeredAbility copy() {
106         return new VizierOfTheAnointedTriggeredAbility(this);
107     }
108 
109     @Override
110     public boolean checkEventType(GameEvent event, Game game) {
111         return event.getType() == GameEvent.EventType.ACTIVATED_ABILITY;
112     }
113 
114     @Override
115     public boolean checkTrigger(GameEvent event, Game game) {
116         if (event.getPlayerId().equals(getControllerId())) {
117             StackAbility stackAbility = (StackAbility) game.getStack().getStackObject(event.getSourceId());
118             if (stackAbility.getStackAbility() instanceof EternalizeAbility
119                     || stackAbility.getStackAbility() instanceof EmbalmAbility) {
120                 return true;
121             }
122         }
123         return false;
124     }
125 
126     @Override
127     public String getRule() {
128         return "Whenever you activate an eternalize or embalm ability, draw a card.";
129     }
130 }

132 class SearchLibraryPutInGraveyard extends SearchEffect {
133 
134     public SearchLibraryPutInGraveyard(FilterCard filter) {
135         super(new TargetCardInLibrary(filter), Outcome.Neutral);
136         staticText = "search your library for a creature card with eternalize or embalm, put that card into your graveyard, then shuffle.";
137     }
138 
139     public SearchLibraryPutInGraveyard(final SearchLibraryPutInGraveyard effect) {
140         super(effect);
141     }
142 
143     @Override
144     public SearchLibraryPutInGraveyard copy() {
145         return new SearchLibraryPutInGraveyard(this);
146     }
147 
148     @Override
149     public boolean apply(Game game, Ability source) {
150         Player controller = game.getPlayer(source.getControllerId());
151         if (controller != null) {
152             if (controller.searchLibrary(target, source, game)) {
153                 if (!target.getTargets().isEmpty()) {
154                     Card card = controller.getLibrary().getCard(target.getFirstTarget(), game);
155                     if (card != null) {
156                         controller.moveCards(card, Zone.GRAVEYARD, source, game);
157                     }
158                 }
159             }
160             controller.shuffleLibrary(source, game);
161             return true;
162         }
163         return false;
164     }

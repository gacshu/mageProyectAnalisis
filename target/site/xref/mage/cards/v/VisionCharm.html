30  public final class VisionCharm extends CardImpl {
31  
32      public VisionCharm(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{U}");
34  
35          // Choose one - Target player puts the top four cards of their library into their graveyard;
36          this.getSpellAbility().addEffect(new PutLibraryIntoGraveTargetEffect(4));
37          this.getSpellAbility().addTarget(new TargetPlayer());
38  
39          // or choose a land type and a basic land type. Each land of the first chosen type becomes the second chosen type until end of turn;
40          this.getSpellAbility().addMode(new Mode(new VisionCharmEffect()));
41  
42          // or target artifact phases out.
43          Mode mode = new Mode(new PhaseOutTargetEffect());
44          mode.addTarget(new TargetArtifactPermanent());
45          this.getSpellAbility().addMode(mode);
46      }
47  
48      private VisionCharm(final VisionCharm card) {
49          super(card);
50      }
51  
52      @Override
53      public VisionCharm copy() {
54          return new VisionCharm(this);
55      }
56  }

58  class VisionCharmEffect extends ContinuousEffectImpl {
59  
60      private String targetLandType;
61      private SubType targetBasicLandType;
62  
63      VisionCharmEffect() {
64          super(Duration.EndOfTurn, Layer.TypeChangingEffects_4, SubLayer.NA, Outcome.Neutral);
65          staticText = "Choose a land type and a basic land type. Each land of the first chosen type becomes the second chosen type until end of turn.";
66      }
67  
68      private VisionCharmEffect(final VisionCharmEffect effect) {
69          super(effect);
70          targetLandType = effect.targetLandType;
71          targetBasicLandType = effect.targetBasicLandType;
72      }
73  
74      @Override
75      public VisionCharmEffect copy() {
76          return new VisionCharmEffect(this);
77      }
78  
79      @Override
80      public void init(Ability source, Game game) {
81          super.init(source, game);
82          Player controller = game.getPlayer(source.getControllerId());
83          if (controller != null) {
84              Choice choice = new ChoiceLandType();
85              if (!controller.choose(outcome, choice, game)) {
86                  discard();
87                  return;
88              }
89              targetLandType = choice.getChoice();
90              choice = new ChoiceBasicLandType();
91              if (!controller.choose(outcome, choice, game)) {
92                  discard();
93                  return;
94              }
95              targetBasicLandType = SubType.byDescription(choice.getChoice());
96              if (targetLandType == null || targetBasicLandType == null) {
97                  this.discard();
98                  return;
99              }
100         } else {
101             this.discard();
102             return;
103         }
104         FilterPermanent filter = new FilterLandPermanent();
105         filter.add(SubType.byDescription(targetLandType).getPredicate());
106         if (this.affectedObjectsSet) {
107             for (Permanent permanent : game.getBattlefield().getAllActivePermanents(filter, game)) {
108                 affectedObjectList.add(new MageObjectReference(permanent, game));
109             }
110         }
111     }
112 
113     @Override
114     public boolean apply(Game game, Ability source) {
115         for (Iterator<MageObjectReference> it = affectedObjectList.iterator(); it.hasNext(); ) {
116             Permanent land = it.next().getPermanent(game);
117             if (land == null) {
118                 it.remove();
119                 continue;
120             }
121             land.removeAllSubTypes(game, SubTypeSet.NonBasicLandType);
122             land.addSubType(game, targetBasicLandType);
123             land.removeAllAbilities(source.getSourceId(), game);
124             switch (targetBasicLandType) {
125                 case FOREST:
126                     land.addAbility(new GreenManaAbility(), source.getSourceId(), game);
127                     break;
128                 case PLAINS:
129                     land.addAbility(new WhiteManaAbility(), source.getSourceId(), game);
130                     break;
131                 case MOUNTAIN:
132                     land.addAbility(new RedManaAbility(), source.getSourceId(), game);
133                     break;
134                 case ISLAND:
135                     land.addAbility(new BlueManaAbility(), source.getSourceId(), game);
136                     break;
137                 case SWAMP:
138                     land.addAbility(new BlackManaAbility(), source.getSourceId(), game);
139                     break;
140             }
141         }
142 
143         return true;
144     }

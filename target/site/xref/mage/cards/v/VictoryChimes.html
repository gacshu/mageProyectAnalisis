24  public final class VictoryChimes extends CardImpl {
25  
26      public VictoryChimes(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
28  
29          // Untap Victory Chimes during each other player's untap step.
30          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new UntapSourceDuringEachOtherPlayersUntapStepEffect()));
31  
32          // {T}: A player of your choice adds {C}.
33          ManaEffect effect = new VictoryChimesManaEffect("chosen player");
34          effect.setText("a player of your choice adds {C}");
35          Ability ability = new SimpleManaAbility(Zone.BATTLEFIELD, effect, new TapSourceCost());
36          // choosing player as first effect, before adding mana effect
37          ability.getEffects().add(0, new ChoosePlayerEffect(Outcome.PutManaInPool).setText(""));
38          this.addAbility(ability);
39      }
40  
41      private VictoryChimes(final VictoryChimes card) {
42          super(card);
43      }
44  
45      @Override
46      public VictoryChimes copy() {
47          return new VictoryChimes(this);
48      }
49  }
50  
51  class VictoryChimesManaEffect extends ManaEffect {
52  
53      public VictoryChimesManaEffect(String textManaPoolOwner) {
54          super();
55          this.staticText = "a player of your choice adds {C}";
56      }
57  
58      public VictoryChimesManaEffect(final VictoryChimesManaEffect effect) {
59          super(effect);
60      }
61  
62      @Override
63      public Player getPlayer(Game game, Ability source) {
64          return game.getPlayer((UUID) game.getState().getValue(source.getSourceId() + "_player"));
65      }
66  
67      @Override
68      public Mana produceMana(Game game, Ability source) {
69          return Mana.ColorlessMana(1);
70      }
71  
72      @Override
73      public VictoryChimesManaEffect copy() {
74          return new VictoryChimesManaEffect(this);
75      }
76  
77  }

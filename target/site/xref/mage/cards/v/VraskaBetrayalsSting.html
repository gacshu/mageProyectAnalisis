26  public final class VraskaBetrayalsSting extends CardImpl {
27  
28      public VraskaBetrayalsSting(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{4}{B}{B/P}");
30  
31          this.addSuperType(SuperType.LEGENDARY);
32          this.subtype.add(SubType.VRASKA);
33          this.setStartingLoyalty(6);
34  
35          // Compleated
36          this.addAbility(CompleatedAbility.getInstance());
37  
38          // 0: You draw a card and you lose 1 life. Proliferate.
39          Ability ability = new LoyaltyAbility(new DrawCardSourceControllerEffect(1).setText("You draw a card"), 0);
40          ability.addEffect(new LoseLifeSourceControllerEffect(1).concatBy("and"));
41          ability.addEffect(new ProliferateEffect(false));
42          this.addAbility(ability);
43  
44          // −2: Target creature becomes a Treasure artifact with "{T}, Sacrifice this artifact:
45          // Add one mana of any color" and loses all other card types and abilities.
46          ability = new LoyaltyAbility(new BecomesCreatureTargetEffect(
47                  new TreasureToken(), true, false, Duration.WhileOnBattlefield, false, false, true)
48                  .setText("Target creature becomes a Treasure artifact with \"{T}, Sacrifice this artifact: " +
49                          "Add one mana of any color\" and loses all other card types and abilities"), -2
50          );
51          ability.addTarget(new TargetCreaturePermanent());
52          this.addAbility(ability);
53  
54          // −9: If target player has fewer than nine poison counters, they get a number of poison counters equal to the difference.
55          ability = new LoyaltyAbility(new VraskaBetrayalsStingEffect(), -9);
56          ability.addTarget(new TargetPlayer());
57          this.addAbility(ability);
58      }
59  
60      private VraskaBetrayalsSting(final VraskaBetrayalsSting card) {
61          super(card);
62      }
63  
64      @Override
65      public VraskaBetrayalsSting copy() {
66          return new VraskaBetrayalsSting(this);
67      }
68  }

70  class VraskaBetrayalsStingEffect extends OneShotEffect {
71  
72      VraskaBetrayalsStingEffect() {
73          super(Outcome.Benefit);
74          staticText = "If target player has fewer than nine poison counters, they get a number of poison counters equal to the difference";
75      }
76  
77      private VraskaBetrayalsStingEffect(final VraskaBetrayalsStingEffect effect) {
78          super(effect);
79      }
80  
81      @Override
82      public VraskaBetrayalsStingEffect copy() {
83          return new VraskaBetrayalsStingEffect(this);
84      }
85  
86      @Override
87      public boolean apply(Game game, Ability source) {
88          Player targetPlayer = game.getPlayer(source.getFirstTarget());
89          if (targetPlayer == null) {
90              return false;
91          }
92          int totalPoison = targetPlayer.getCounters().getCount(CounterType.POISON);
93          if (totalPoison < 9) {
94              targetPlayer.addCounters(CounterType.POISON.createInstance(9 - totalPoison), source.getControllerId(), source, game);
95          }
96          return true;
97      }
98  }

3   import java.util.UUID;
4   import mage.abilities.Ability;
5   import mage.abilities.common.SimpleActivatedAbility;
6   import mage.abilities.costs.common.SacrificeSourceCost;
7   import mage.abilities.costs.mana.ManaCostsImpl;
8   import mage.abilities.effects.common.RevealLibraryPickControllerEffect;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  import mage.constants.PutCards;
13  import mage.filter.FilterCard;
14  import mage.filter.predicate.Predicates;
15  
16  /**
17   *
18   * @author awjackson
19   */
20  public final class VesselOfNascency extends CardImpl {
21  
22      private static final FilterCard filter = new FilterCard("an artifact, creature, enchantment, land, or planeswalker card");
23  
24      static {
25          filter.add(Predicates.or(CardType.ARTIFACT.getPredicate(),
26                  CardType.CREATURE.getPredicate(),
27                  CardType.ENCHANTMENT.getPredicate(),
28                  CardType.LAND.getPredicate(),
29                  CardType.PLANESWALKER.getPredicate()
30          ));
31      }
32  
33      public VesselOfNascency(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{G}");
35  
36          // {1}{G}, Sacrifice Vessel of Nascency: Reveal the top four cards of your library. You may put an artifact, creature, enchantment, land, or
37          // planeswalker card from among them into your hand. Put the rest into your graveyard.
38          Ability ability = new SimpleActivatedAbility(
39                  new RevealLibraryPickControllerEffect(4, 1, filter, PutCards.HAND, PutCards.GRAVEYARD),
40                  new ManaCostsImpl<>("{1}{G}"));
41          ability.addCost(new SacrificeSourceCost());
42          this.addAbility(ability);
43      }
44  
45      private VesselOfNascency(final VesselOfNascency card) {
46          super(card);
47      }
48  
49      @Override
50      public VesselOfNascency copy() {
51          return new VesselOfNascency(this);
52      }
53  }

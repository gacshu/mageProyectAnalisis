25  public final class VolcanicEruption extends CardImpl {
26  
27      public VolcanicEruption(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{U}{U}{U}");
29  
30          // Destroy X target Mountains. Volcanic Eruption deals damage to each creature and each player equal to the number of Mountains put into a graveyard this way.
31          this.getSpellAbility().addEffect(new VolcanicEruptionEffect());
32          this.getSpellAbility().setTargetAdjuster(VolcanicEruptionAdjuster.instance);
33      }
34  
35      private VolcanicEruption(final VolcanicEruption card) {
36          super(card);
37      }
38  
39      @Override
40      public VolcanicEruption copy() {
41          return new VolcanicEruption(this);
42      }
43  }
44  
45  enum VolcanicEruptionAdjuster implements TargetAdjuster {
46      instance;
47      private static final FilterLandPermanent filter
48              = new FilterLandPermanent(SubType.MOUNTAIN, "Mountain");
49  
50      @Override
51      public void adjustTargets(Ability ability, Game game) {
52          ability.getTargets().clear();
53          int xValue = ability.getManaCostsToPay().getX();
54          ability.addTarget(new TargetLandPermanent(xValue, xValue, filter, false));
55      }
56  }
57  
58  class VolcanicEruptionEffect extends OneShotEffect {
59  
60      public VolcanicEruptionEffect() {
61          super(Outcome.DestroyPermanent);
62          this.staticText = "Destroy X target Mountains. {this} deals damage to each creature and each player equal to the number of Mountains put into a graveyard this way.";
63      }
64  
65      public VolcanicEruptionEffect(final VolcanicEruptionEffect effect) {
66          super(effect);
67      }
68  
69      @Override
70      public VolcanicEruptionEffect copy() {
71          return new VolcanicEruptionEffect(this);
72      }
73  
74      @Override
75      public boolean apply(Game game, Ability source) {
76  
77          int destroyedCount = 0;
78          for (UUID targetID : this.targetPointer.getTargets(game, source)) {
79              Permanent permanent = game.getPermanent(targetID);
80              if (permanent != null) {
81                  if (permanent.destroy(source, game, false)) {
82                      if (game.getState().getZone(permanent.getId()) == Zone.GRAVEYARD) {
83                          destroyedCount++;
84                      }
85                  }
86              }
87          }
88  
89          if (destroyedCount > 0) {
90              List<Permanent> permanents = game.getBattlefield().getActivePermanents(StaticFilters.FILTER_PERMANENT_CREATURE, source.getControllerId(), game);
91              for (Permanent permanent : permanents) {
92                  permanent.damage(destroyedCount, source.getSourceId(), source, game, false, true);
93              }
94              for (UUID playerId : game.getState().getPlayersInRange(source.getControllerId(), game)) {
95                  Player player = game.getPlayer(playerId);
96                  if (player != null) {
97                      player.damage(destroyedCount, source.getSourceId(), source, game);
98                  }
99              }
100         }
101         return true;
102     }
103 }

28  public final class VoiceOfTheBlessed extends CardImpl {
29  
30      public VoiceOfTheBlessed(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{W}{W}");
32  
33          this.subtype.add(SubType.SPIRIT);
34          this.subtype.add(SubType.CLERIC);
35          this.power = new MageInt(2);
36          this.toughness = new MageInt(2);
37  
38          // Whenever you gain life, put a +1/+1 counter on Voice of the Blessed.
39          this.addAbility(new GainLifeControllerTriggeredAbility(
40                  new AddCountersSourceEffect(CounterType.P1P1.createInstance())
41          ));
42  
43          // As long as Voice of the Blessed has four or more +1/+1 counters on it, it has flying and vigilance.
44          Ability ability = new SimpleStaticAbility(new ConditionalContinuousEffect(
45                  new GainAbilitySourceEffect(
46                          FlyingAbility.getInstance(), Duration.WhileOnBattlefield
47                  ), VoiceOfTheBlessedCondition.FOUR, "as long as {this} has " +
48                  "four or more +1/+1 counters on it, it has flying"
49          ));
50          ability.addEffect(new ConditionalContinuousEffect(new GainAbilitySourceEffect(
51                  VigilanceAbility.getInstance(), Duration.WhileOnBattlefield
52          ), VoiceOfTheBlessedCondition.FOUR, "and vigilance"));
53          this.addAbility(ability);
54  
55          // As long as Voice of the Blessed has ten or more +1/+1 counters on it, it has indestructible.
56          this.addAbility(new SimpleStaticAbility(new ConditionalContinuousEffect(
57                  new GainAbilitySourceEffect(
58                          IndestructibleAbility.getInstance(), Duration.WhileOnBattlefield
59                  ), VoiceOfTheBlessedCondition.TEN, "as long as {this} has " +
60                  "ten or more +1/+1 counters on it, it has indestructible"
61          )));
62      }
63  
64      private VoiceOfTheBlessed(final VoiceOfTheBlessed card) {
65          super(card);
66      }
67  
68      @Override
69      public VoiceOfTheBlessed copy() {
70          return new VoiceOfTheBlessed(this);
71      }
72  }
73  
74  enum VoiceOfTheBlessedCondition implements Condition {
75      FOUR(4),
76      TEN(10);
77      private final int counters;
78  
79      VoiceOfTheBlessedCondition(int counters) {
80          this.counters = counters;
81      }
82  
83      @Override
84      public boolean apply(Game game, Ability source) {
85          Permanent permanent = source.getSourcePermanentIfItStillExists(game);
86          return permanent != null && permanent.getCounters(game).getCount(CounterType.P1P1) >= counters;
87      }
88  }

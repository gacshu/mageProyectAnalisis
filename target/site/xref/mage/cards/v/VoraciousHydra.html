8   import mage.abilities.effects.common.DoubleCountersSourceEffect;
9   import mage.abilities.effects.common.EntersBattlefieldWithXCountersEffect;
10  import mage.abilities.effects.common.FightTargetSourceEffect;
11  import mage.abilities.keyword.TrampleAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.counters.CounterType;
17  import mage.filter.StaticFilters;
18  import mage.target.TargetPermanent;
19  
20  import java.util.UUID;
21  
22  /**
23   * @author TheElk801
24   */
25  public final class VoraciousHydra extends CardImpl {
26  
27      public VoraciousHydra(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{X}{G}{G}");
29  
30          this.subtype.add(SubType.HYDRA);
31          this.power = new MageInt(0);
32          this.toughness = new MageInt(1);
33  
34          // Trample
35          this.addAbility(TrampleAbility.getInstance());
36  
37          // Voracious Hydra enters the battlefield with X +1/+1 counters on it.
38          this.addAbility(new EntersBattlefieldAbility(
39                  new EntersBattlefieldWithXCountersEffect(CounterType.P1P1.createInstance())
40          ));
41  
42          // When Voracious Hydra enters the battlefield, choose one —
43          // • Double the number of +1/+1 counters on Voracious Hydra.
44          Ability ability = new EntersBattlefieldTriggeredAbility(new DoubleCountersSourceEffect(CounterType.P1P1), false);
45  
46          // • Voracious Hydra fights target creature you don't control.
47          Mode mode = new Mode(
48                  new FightTargetSourceEffect()
49                          .setText("{this} fights target creature you don't control")
50          );
51          mode.addTarget(new TargetPermanent(StaticFilters.FILTER_CREATURE_YOU_DONT_CONTROL));
52          ability.addMode(mode);
53          this.addAbility(ability);
54      }
55  
56      private VoraciousHydra(final VoraciousHydra card) {
57          super(card);
58      }
59  
60      @Override
61      public VoraciousHydra copy() {
62          return new VoraciousHydra(this);
63      }
64  }

33  public final class OtharriSunsGlory extends CardImpl {
34  
35      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent("untapped Rebel you control");
36  
37      static {
38          filter.add(TappedPredicate.UNTAPPED);
39          filter.add(SubType.REBEL.getPredicate());
40      }
41  
42      public OtharriSunsGlory(UUID ownerId, CardSetInfo setInfo) {
43          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}{W}");
44  
45          this.supertype.add(SuperType.LEGENDARY);
46          this.subtype.add(SubType.PHOENIX);
47          this.power = new MageInt(3);
48          this.toughness = new MageInt(3);
49  
50          // Flying,
51          this.addAbility(FlyingAbility.getInstance());
52  
53          // Lifelink,
54          this.addAbility(LifelinkAbility.getInstance());
55  
56          // Haste
57          this.addAbility(HasteAbility.getInstance());
58  
59          // Whenever Otharri, Suns' Glory attacks, you get an experience counter.
60          // Then create a 2/2 red Rebel creature token thatâ€™s tapped and attacking for each experience counter you have.
61          Ability ability = new AttacksTriggeredAbility(new AddCountersPlayersEffect(CounterType.EXPERIENCE.createInstance(), TargetController.YOU));
62          DynamicValue value = new OtharriSunsGloryCount();
63          ability.addEffect(new CreateTokenEffect(new RebelRedToken(), value, true, true).concatBy("Then"));
64          this.addAbility(ability);
65  
66          // {2}{R}{W}, Tap an untapped Rebel you control: Return Otharri from your graveyard to the battlefield tapped.
67          ability = new SimpleActivatedAbility(
68                  Zone.GRAVEYARD,
69                  new ReturnToBattlefieldUnderOwnerControlSourceEffect(true, -1)
70                          .setText("return {this} from your graveyard to the battlefield tapped"),
71                  new ManaCostsImpl<>("{2}{R}{W}"));
72          ability.addCost(new TapTargetCost(new TargetControlledPermanent(1, 1, filter, false)));
73          this.addAbility(ability);
74      }
75  
76      private OtharriSunsGlory(final OtharriSunsGlory card) {
77          super(card);
78      }
79  
80      @Override
81      public OtharriSunsGlory copy() {
82          return new OtharriSunsGlory(this);
83      }
84  }
85  
86  class OtharriSunsGloryCount implements DynamicValue {
87  
88      @Override
89      public int calculate(Game game, Ability sourceAbility, Effect effect) {
90          int amount = 0;
91          Player player = game.getPlayer(sourceAbility.getControllerId());
92          if (player != null) {
93              amount = player.getCounters().getCount(CounterType.EXPERIENCE);
94          }
95          return amount;
96      }
97  
98      @Override
99      public OtharriSunsGloryCount copy() {
100         return new OtharriSunsGloryCount();
101     }
102 
103     @Override
104     public String toString() {
105         return "1";
106     }
107 
108     @Override
109     public String getMessage() {
110         return "experience counter you have";
111     }
112 }

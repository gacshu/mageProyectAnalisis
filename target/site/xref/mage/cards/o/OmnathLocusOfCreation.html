22  public final class OmnathLocusOfCreation extends CardImpl {
23  
24      private static final FilterPermanent filter = new FilterPlaneswalkerPermanent();
25  
26      static {
27          filter.add(TargetController.NOT_YOU.getControllerPredicate());
28      }
29  
30      public OmnathLocusOfCreation(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{R}{G}{W}{U}");
32  
33          this.addSuperType(SuperType.LEGENDARY);
34          this.subtype.add(SubType.ELEMENTAL);
35          this.power = new MageInt(4);
36          this.toughness = new MageInt(4);
37  
38          // When Omnath, Locus of Creation enters the battlefield, draw a card.
39          this.addAbility(new EntersBattlefieldTriggeredAbility(new DrawCardSourceControllerEffect(1)));
40  
41          // Landfall â€” Whenever a land enters the battlefield under your control, you gain 4 life if this is the first time this ability has resolved this turn. If it's the second time, add {R}{G}{W}{U}. If it's the third time, Omnath deals 4 damage to each opponent and each planeswalker you don't control.
42          Ability ability = new LandfallAbility(new IfAbilityHasResolvedXTimesEffect(
43                  Outcome.GainLife, 1, new GainLifeEffect(4)
44          ).setText("you gain 4 life if this is the first time this ability has resolved this turn."), false);
45          ability.addEffect(new IfAbilityHasResolvedXTimesEffect(
46                  Outcome.PutManaInPool, 2, new BasicManaEffect(new Mana(
47                  1, 1, 0, 1, 1, 0, 0, 0
48          ))).setText("If it's the second time, add {R}{G}{W}{U}."));
49          ability.addEffect(new IfAbilityHasResolvedXTimesEffect(
50                  Outcome.Damage, 3, new DamagePlayersEffect(4, TargetController.OPPONENT)
51          ).setText("If it's the third time, {this} deals 4 damage to each opponent"));
52          ability.addEffect(new IfAbilityHasResolvedXTimesEffect(
53                  Outcome.Damage, 3, new DamageAllEffect(4, filter)
54          ).setText("and each planeswalker you don't control."));
55          this.addAbility(ability, new AbilityResolvedWatcher());
56      }
57  
58      private OmnathLocusOfCreation(final OmnathLocusOfCreation card) {
59          super(card);
60      }
61  
62      @Override
63      public OmnathLocusOfCreation copy() {
64          return new OmnathLocusOfCreation(this);
65      }
66  }

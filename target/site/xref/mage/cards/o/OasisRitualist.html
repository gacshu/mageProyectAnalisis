10  import mage.abilities.mana.AnyColorManaAbility;
11  import mage.abilities.mana.SimpleManaAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;

18  /**
19   *
20   * @author LevelX2
21   */
22  public final class OasisRitualist extends CardImpl {
23  
24      public OasisRitualist(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{G}");
26  
27          this.subtype.add(SubType.NAGA);
28          this.subtype.add(SubType.DRUID);
29          this.power = new MageInt(2);
30          this.toughness = new MageInt(4);
31  
32          // {T}: Add one mana of any color.
33          this.addAbility(new AnyColorManaAbility());
34          // {T}, Exert Oasis Ritualist: Add two mana of any one color to your manna pool.
35          SimpleManaAbility ability = new SimpleManaAbility(Zone.BATTLEFIELD,
36                  (ManaEffect) new AddManaOfAnyColorEffect(2).setText("Add two mana of any one color to your manna pool. <i> (An exerted creature won't untap during your next untap step.)</i>"),
37                   new TapSourceCost());
38          ability.addCost(new ExertSourceCost());
39          this.addAbility(ability);
40      }
41  
42      private OasisRitualist(final OasisRitualist card) {
43          super(card);
44      }
45  
46      @Override
47      public OasisRitualist copy() {
48          return new OasisRitualist(this);
49      }
50  }

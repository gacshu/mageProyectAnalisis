7   import mage.abilities.effects.common.cost.SpellCostReductionForEachSourceEffect;
8   import mage.abilities.hint.Hint;
9   import mage.abilities.hint.ValueHint;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.Zone;
14  import mage.filter.StaticFilters;
15  import mage.target.common.TargetCreatureOrPlaneswalker;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class OverwhelmingRemorse extends CardImpl {
23  
24      private static final DynamicValue xValue = new CardsInControllerGraveyardCount(StaticFilters.FILTER_CARD_CREATURE);
25      private static final Hint hint = new ValueHint("Creatures in your graveyard", xValue);
26  
27      public OverwhelmingRemorse(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{4}{B}");
29  
30          // This spell costs {1} less to cast for each creature card in your graveyard.
31          this.addAbility(new SimpleStaticAbility(
32                  Zone.ALL, new SpellCostReductionForEachSourceEffect(1, xValue)
33          ).addHint(hint));
34  
35          // Exile target creature or planeswalker.
36          this.getSpellAbility().addEffect(new ExileTargetEffect());
37          this.getSpellAbility().addTarget(new TargetCreatureOrPlaneswalker());
38      }
39  
40      private OverwhelmingRemorse(final OverwhelmingRemorse card) {
41          super(card);
42      }
43  
44      @Override
45      public OverwhelmingRemorse copy() {
46          return new OverwhelmingRemorse(this);
47      }
48  }

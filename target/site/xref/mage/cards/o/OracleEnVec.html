37  public final class OracleEnVec extends CardImpl {
38  
39      public OracleEnVec(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{W}");
41          this.subtype.add(SubType.HUMAN);
42          this.subtype.add(SubType.WIZARD);
43          this.power = new MageInt(1);
44          this.toughness = new MageInt(1);
45  
46          // {T}: Target opponent chooses any number of creatures they control. During that player’s next turn, the chosen
47          // creatures attack if able, and other creatures can’t attack. At the beginning of that turn’s end step,
48          // destroy each of the chosen creatures that didn’t attack this turn. Activate this ability only during your turn.
49          Ability ability = new ActivateIfConditionActivatedAbility(Zone.BATTLEFIELD, new OracleEnVecEffect(), new TapSourceCost(), MyTurnCondition.instance);
50          ability.addTarget(new TargetOpponent());
51          ability.addHint(MyTurnHint.instance);
52          this.addAbility(ability, new AttackedThisTurnWatcher());
53      }
54  
55      private OracleEnVec(final OracleEnVec card) {
56          super(card);
57      }
58  
59      @Override
60      public OracleEnVec copy() {
61          return new OracleEnVec(this);
62      }
63  }

65  class OracleEnVecEffect extends OneShotEffect {
66  
67      OracleEnVecEffect() {
68          super(Outcome.Benefit);
69          this.staticText = "Target opponent chooses any number of creatures they control. During that player's next turn, "
70                  + "the chosen creatures attack if able, and other creatures can't attack. At the beginning of that turn's end step, "
71                  + "destroy each of the chosen creatures that didn't attack this turn";
72      }
73  
74      OracleEnVecEffect(final OracleEnVecEffect effect) {
75          super(effect);
76      }
77  
78      @Override
79      public OracleEnVecEffect copy() {
80          return new OracleEnVecEffect(this);
81      }
82  
83      @Override
84      public boolean apply(Game game, Ability source) {
85          Player opponent = game.getPlayer(this.getTargetPointer().getFirst(game, source));
86          if (opponent != null) {
87              Target target = new TargetControlledCreaturePermanent(0, Integer.MAX_VALUE, new FilterControlledCreaturePermanent(), true);
88              if (target.choose(Outcome.Neutral, opponent.getId(), source.getSourceId(), source, game)) {
89                  for (Permanent permanent : game.getBattlefield().getActivePermanents(new FilterControlledCreaturePermanent(), opponent.getId(), source, game)) {
90                      if (target.getTargets().contains(permanent.getId())) {
91                          RequirementEffect effect = new OracleEnVecMustAttackRequirementEffect();
92                          effect.setTargetPointer(new FixedTarget(permanent, game));
93                          game.addEffect(effect, source);
94                      } else {
95                          RestrictionEffect effect = new OracleEnVecCantAttackRestrictionEffect();
96                          effect.setTargetPointer(new FixedTarget(permanent, game));
97                          game.addEffect(effect, source);
98                      }
99                  }
100                 game.addDelayedTriggeredAbility(new OracleEnVecDelayedTriggeredAbility(game.getTurnNum(), target.getTargets()), source);
101                 return true;
102             }
103         }
104         return false;
105     }
106 }

108 class OracleEnVecMustAttackRequirementEffect extends RequirementEffect {
109 
110     OracleEnVecMustAttackRequirementEffect() {
111         super(Duration.UntilEndOfYourNextTurn);
112     }
113 
114     OracleEnVecMustAttackRequirementEffect(final OracleEnVecMustAttackRequirementEffect effect) {
115         super(effect);
116     }
117 
118     @Override
119     public OracleEnVecMustAttackRequirementEffect copy() {
120         return new OracleEnVecMustAttackRequirementEffect(this);
121     }
122 
123     @Override
124     public boolean applies(Permanent permanent, Ability source, Game game) {
125         return this.getTargetPointer().getFirst(game, source) != null
126                 && this.getTargetPointer().getFirst(game, source).equals(permanent.getId())
127                 && this.isYourNextTurn(game);
128     }
129 
130     @Override
131     public boolean mustAttack(Game game) {
132         return true;
133     }
134 
135     @Override
136     public boolean mustBlock(Game game) {
137         return false;
138     }
139 
140     @Override
141     public void init(Ability source, Game game) {
142         super.init(source, game);
143         Permanent perm = game.getPermanent(this.getTargetPointer().getFirst(game, source));
144         if (perm != null) {
145             setStartingControllerAndTurnNum(game, perm.getControllerId(), game.getActivePlayerId()); // setup startingController to calc isYourTurn calls
146         } else {
147             discard();
148         }
149     }
150 
151     @Override
152     public boolean isInactive(Ability source, Game game) {
153         return game.getTurnPhaseType() == TurnPhase.END && this.isYourNextTurn(game);
154     }
155 
156     @Override
157     public String getText(Mode mode) {
158         return "{this} attack if able.";
159     }

206 class OracleEnVecDelayedTriggeredAbility extends DelayedTriggeredAbility {
207 
208     private final int startingTurn;
209 
210     OracleEnVecDelayedTriggeredAbility(int startingTurn, List<UUID> chosenCreatures) {
211         super(new OracleEnVecDestroyEffect(chosenCreatures), Duration.EndOfGame, true);
212         this.startingTurn = startingTurn;
213         setTriggerPhrase("At the beginning of that turn's end step, ");
214     }
215 
216     OracleEnVecDelayedTriggeredAbility(final OracleEnVecDelayedTriggeredAbility ability) {
217         super(ability);
218         this.startingTurn = ability.startingTurn;
219     }
220 
221     @Override
222     public boolean checkEventType(GameEvent event, Game game) {
223         return event.getType() == GameEvent.EventType.END_TURN_STEP_PRE;
224     }
225 
226     @Override
227     public boolean checkTrigger(GameEvent event, Game game) {
228         return startingTurn != game.getTurnNum() && game.isActivePlayer(event.getPlayerId());
229     }
230 
231     @Override
232     public OracleEnVecDelayedTriggeredAbility copy() {
233         return new OracleEnVecDelayedTriggeredAbility(this);
234     }
235 }

237 class OracleEnVecDestroyEffect extends OneShotEffect {
238 
239     private final List<UUID> chosenCreatures;
240 
241     OracleEnVecDestroyEffect(List<UUID> chosenCreatures) { // need to be changed to MageObjectReference
242         super(Outcome.DestroyPermanent);
243         this.chosenCreatures = chosenCreatures;
244         this.staticText = "destroy each of the chosen creatures that didn't attack";
245     }
246 
247     OracleEnVecDestroyEffect(final OracleEnVecDestroyEffect effect) {
248         super(effect);
249         this.chosenCreatures = effect.chosenCreatures;
250     }
251 
252     @Override
253     public OracleEnVecDestroyEffect copy() {
254         return new OracleEnVecDestroyEffect(this);
255     }
256 
257     @Override
258     public boolean apply(Game game, Ability source) {
259         AttackedThisTurnWatcher watcher = game.getState().getWatcher(AttackedThisTurnWatcher.class);
260         if (watcher != null) {
261             for (UUID targetId : chosenCreatures) {
262                 Permanent permanent = game.getPermanent(targetId);
263                 if (permanent != null && !watcher.getAttackedThisTurnCreatures().contains(new MageObjectReference(permanent, game))) {
264                     Effect effect = new DestroyTargetEffect();
265                     effect.setTargetPointer(new FixedTarget(targetId, game));
266                     effect.apply(game, source);
267                 }
268             }
269             return true;
270         }
271         return false;
272     }

41  public final class OldGrowthTroll extends CardImpl {
42  
43      public OldGrowthTroll(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{G}{G}{G}");
45  
46          this.subtype.add(SubType.TROLL);
47          this.subtype.add(SubType.WARRIOR);
48          this.power = new MageInt(4);
49          this.toughness = new MageInt(4);
50  
51          // Trample
52          this.addAbility(TrampleAbility.getInstance());
53  
54          // When Old-Growth Troll dies, if it was a creature, return it to the battlefield. It's an Aura enchantment with enchant Forest you control and "Enchanted Forest has '{T}: Add {G}{G}' and '{1}, {T}, Sacrifice this land: Create a 4/4 green Troll Warrior creature token with trample.'"
55          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
56                  new DiesSourceTriggeredAbility(new OldGrowthTrollReturnEffect()),
57                  OldGrowthTrollCondition.instance, "When {this} dies, if it was a creature, " +
58                  "return it to the battlefield. It's an Aura enchantment with enchant Forest you control " +
59                  "and \"Enchanted Forest has '{T}: Add {G}{G}' and '{1}, {T}, Sacrifice this land: " +
60                  "Create a tapped 4/4 green Troll Warrior creature token with trample.'\""
61          ));
62      }
63  
64      private OldGrowthTroll(final OldGrowthTroll card) {
65          super(card);
66      }
67  
68      @Override
69      public OldGrowthTroll copy() {
70          return new OldGrowthTroll(this);
71      }
72  }

77      @Override
78      public boolean apply(Game game, Ability source) {
79          Permanent permanent = (Permanent) source.getEffects().get(0).getValue("permanentLeftBattlefield");

84  class OldGrowthTrollReturnEffect extends OneShotEffect {
85  
86      private static final FilterPermanent filter = new FilterControlledPermanent(SubType.FOREST);
87  
88      OldGrowthTrollReturnEffect() {
89          super(Outcome.PutCardInPlay);
90      }
91  
92      private OldGrowthTrollReturnEffect(final OldGrowthTrollReturnEffect effect) {
93          super(effect);
94      }
95  
96      @Override
97      public OldGrowthTrollReturnEffect copy() {
98          return new OldGrowthTrollReturnEffect(this);
99      }
100 
101     @Override
102     public boolean apply(Game game, Ability source) {
103         Player controller = game.getPlayer(source.getControllerId());
104         if (controller == null || !(game.getState().getZone(source.getSourceId()) == Zone.GRAVEYARD)) {
105             return false;
106         }
107 
108         Card card = game.getCard(source.getSourceId());
109         if (card == null) {
110             return false;
111         }
112         TargetPermanent target = new TargetPermanent(filter);
113         target.setNotTarget(true);
114         if (controller.choose(outcome, target, source, game)
115                 && game.getPermanent(target.getFirstTarget()) != null) {
116             game.getState().setValue("attachTo:" + source.getSourceId(), target.getFirstTarget());
117         }
118         game.addEffect(new OldGrowthTrollContinuousEffect(game.getState().getZoneChangeCounter(source.getSourceId()) + 1), source);
119         controller.moveCards(card, Zone.BATTLEFIELD, source, game);
120         Permanent aura = game.getPermanent(card.getId());
121         Permanent creature = game.getPermanent(target.getFirstTarget());
122         if (aura == null || creature == null) {
123             return true;
124         }
125         creature.addAttachment(aura.getId(), source, game);
126         return true;
127     }
128 }

130 class OldGrowthTrollContinuousEffect extends ContinuousEffectImpl {
131 
132     private static final FilterPermanent filter = new FilterControlledPermanent(SubType.FOREST);
133     private final int zoneChangeCounter;
134 
135     OldGrowthTrollContinuousEffect(int zoneChangeCounter) {
136         super(Duration.Custom, Outcome.Neutral);
137         this.zoneChangeCounter = zoneChangeCounter;
138         dependencyTypes.add(DependencyType.AuraAddingRemoving);
139     }
140 
141     private OldGrowthTrollContinuousEffect(final OldGrowthTrollContinuousEffect ability) {
142         super(ability);
143         this.zoneChangeCounter = ability.zoneChangeCounter;
144     }
145 
146     @Override
147     public boolean apply(Game game, Ability source) {
148         return false;
149     }
150 
151     @Override
152     public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
153         if (game.getState().getZoneChangeCounter(source.getSourceId()) > zoneChangeCounter) {
154             discard();
155             return false;
156         }
157         Permanent sourceObject = game.getPermanent(source.getSourceId());
158         if (sourceObject == null) {
159             sourceObject = game.getPermanentEntering(source.getSourceId());
160         }
161         if (sourceObject == null) {
162             return false;
163         }
164         Permanent troll = sourceObject;
165         switch (layer) {
166             case TypeChangingEffects_4:
167                 troll.removeAllCardTypes(game);
168                 troll.addCardType(game, CardType.ENCHANTMENT);
169                 troll.removeAllSubTypes(game);
170                 troll.addSubType(game, SubType.AURA);
171                 break;
172             case AbilityAddingRemovingEffects_6:
173                 // Spell Ability can be null with clone effects (ex. Moritte)
174                 if (troll.getSpellAbility() == null) {
175                     troll.addAbility(new SpellAbility(null, null), source.getSourceId(), game);
176                 }
177                 troll.getSpellAbility().getTargets().clear();
178                 troll.getSpellAbility().getEffects().clear();
179                 TargetPermanent auraTarget = new TargetPermanent(filter);
180                 troll.getSpellAbility().addTarget(auraTarget);
181                 troll.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
182                 troll.addAbility(new EnchantAbility(auraTarget), source.getSourceId(), game);
183 
184                 // add the activated ability
185                 troll.addAbility(makeAbility(), source.getSourceId(), game);
186         }
187         return true;
188     }
189 
190     @Override
191     public boolean hasLayer(Layer layer) {
192         return layer == Layer.TypeChangingEffects_4
193                 || layer == Layer.AbilityAddingRemovingEffects_6;
194     }
195 
196     @Override
197     public OldGrowthTrollContinuousEffect copy() {
198         return new OldGrowthTrollContinuousEffect(this);
199     }
200 
201     private static final Ability makeAbility() {
202         Ability activatedAbility = new SimpleActivatedAbility(
203                 new CreateTokenEffect(new TrollWarriorToken(), 1, true, false), new GenericManaCost(1)
204         );
205         activatedAbility.addCost(new TapSourceCost());
206         Cost cost = new SacrificeSourceCost();
207         cost.setText("sacrifice this land");
208         activatedAbility.addCost(cost);
209         Ability ability = new SimpleStaticAbility(new GainAbilityAttachedEffect(new SimpleManaAbility(
210                 Zone.BATTLEFIELD, Mana.GreenMana(2), new TapSourceCost()), AttachmentType.AURA
211         ).setText("enchanted Forest has \"{T}: Add {G}{G}\""));
212         ability.addEffect(new GainAbilityAttachedEffect(
213                 activatedAbility, AttachmentType.AURA
214         ).setText("and \"{1}, {T}, Sacrifice this land: Create a tapped 4/4 green Troll Warrior creature token with trample.\""));
215         return ability;
216     }

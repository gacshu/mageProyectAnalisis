4   import mage.MageInt;
5   import mage.abilities.common.BeginningOfCombatTriggeredAbility;
6   import mage.abilities.common.WinsCoinFlipTriggeredAbility;
7   import mage.abilities.dynamicvalue.DynamicValue;
8   import mage.abilities.dynamicvalue.common.SourcePermanentPowerCount;
9   import mage.abilities.dynamicvalue.common.SourcePermanentToughnessValue;
10  import mage.abilities.effects.common.FlipUntilLoseEffect;
11  import mage.abilities.effects.common.continuous.BoostSourceEffect;
12  import mage.abilities.keyword.PartnerWithAbility;
13  import mage.cards.CardImpl;
14  import mage.cards.CardSetInfo;
15  import mage.constants.*;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class OkaunEyeOfChaos extends CardImpl {
23  
24      private static final DynamicValue sourcePower = new SourcePermanentPowerCount();
25  
26      public OkaunEyeOfChaos(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{R}");
28  
29          this.addSuperType(SuperType.LEGENDARY);
30          this.subtype.add(SubType.CYCLOPS);
31          this.subtype.add(SubType.BERSERKER);
32          this.power = new MageInt(3);
33          this.toughness = new MageInt(3);
34  
35          // Partner with Zndrsplt, Eye of Wisdom (When this creature enters the battlefield, target player may put Zndrsplt into their hand from their library, then shuffle.)
36          this.addAbility(new PartnerWithAbility("Zndrsplt, Eye of Wisdom", true));
37  
38          // At the beginning of combat on your turn, flip a coin until you lose a flip.
39          this.addAbility(new BeginningOfCombatTriggeredAbility(new FlipUntilLoseEffect(), TargetController.YOU, false));
40  
41          // Whenever a player wins a coin flip, double Okaun's power and toughness until end of turn.
42          this.addAbility(new WinsCoinFlipTriggeredAbility(
43                  new BoostSourceEffect(
44                          sourcePower,
45                          SourcePermanentToughnessValue.getInstance(),
46                          Duration.EndOfTurn,
47                          true
48                  ).setText("double {this}'s power and toughness until end of turn")
49          ));
50      }
51  
52      private OkaunEyeOfChaos(final OkaunEyeOfChaos card) {
53          super(card);
54      }
55  
56      @Override
57      public OkaunEyeOfChaos copy() {
58          return new OkaunEyeOfChaos(this);
59      }
60  }

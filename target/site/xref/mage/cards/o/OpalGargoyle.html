27  public final class OpalGargoyle extends CardImpl {
28  
29      public OpalGargoyle(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{W}");
31  
32          // When an opponent casts a creature spell, if Opal Gargoyle is an enchantment, Opal Gargoyle becomes a 2/2 Gargoyle creature with flying.
33          TriggeredAbility ability = new SpellCastOpponentTriggeredAbility(new BecomesCreatureSourceEffect(new OpalGargoyleToken(), "", Duration.WhileOnBattlefield, true, false),
34                  new FilterCreatureSpell(), false);
35          this.addAbility(new ConditionalInterveningIfTriggeredAbility(ability, new SourceMatchesFilterCondition(StaticFilters.FILTER_PERMANENT_ENCHANTMENT),
36                  "When an opponent casts a creature spell, if {this} is an enchantment, {this} becomes a 2/2 Gargoyle creature with flying."));
37      }
38  
39      private OpalGargoyle(final OpalGargoyle card) {
40          super(card);
41      }
42  
43      @Override
44      public OpalGargoyle copy() {
45          return new OpalGargoyle(this);
46      }
47  }
48  
49  class OpalGargoyleToken extends TokenImpl {
50  
51      public OpalGargoyleToken() {
52          super("Gargoyle", "2/2 Gargoyle creature with flying");
53          cardType.add(CardType.CREATURE);
54          subtype.add(SubType.GARGOYLE);
55          power = new MageInt(2);
56          toughness = new MageInt(2);
57          this.addAbility(FlyingAbility.getInstance());
58      }
59      public OpalGargoyleToken(final OpalGargoyleToken token) {
60          super(token);
61      }
62  
63      public OpalGargoyleToken copy() {
64          return new OpalGargoyleToken(this);
65      }
66  }

30  public final class OrbOfDragonkind extends CardImpl {
31  
32      private static final FilterCard filter = new FilterCard("a Dragon card");
33  
34      static {
35          filter.add(SubType.DRAGON.getPredicate());
36      }
37  
38      public OrbOfDragonkind(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{1}{R}");
40  
41          // {1}, {T}: Add two mana in any combination of colors. Spend this mana only to cast Dragon spells or to activate abilities of Dragons.
42          Ability ability = new ConditionalAnyColorManaAbility(new GenericManaCost(1), 2, new OrbOfDragonkindManaBuilder());
43          ability.addCost(new TapSourceCost());
44          this.addAbility(ability);
45  
46          // {R}, {T}, Sacrifice Orb of Dragonkind: Look at the top seven cards of your library.
47          // You may reveal a Dragon card from among them and put it into your hand. Put the rest on the bottom of your library in a random order.
48          ability = new SimpleActivatedAbility(
49                  new LookLibraryAndPickControllerEffect(7, 1, filter, PutCards.HAND, PutCards.BOTTOM_RANDOM),
50                  new ColoredManaCost(ColoredManaSymbol.R));
51          ability.addCost(new TapSourceCost());
52          ability.addCost(new SacrificeSourceCost());
53          this.addAbility(ability);
54      }
55  
56      private OrbOfDragonkind(final OrbOfDragonkind card) {
57          super(card);
58      }
59  
60      @Override
61      public OrbOfDragonkind copy() {
62          return new OrbOfDragonkind(this);
63      }
64  }
65  
66  class OrbOfDragonkindManaBuilder extends ConditionalManaBuilder {
67  
68      @Override
69      public ConditionalMana build(Object... options) {
70          return new OrbOfDragonkindConditionalMana(this.mana);
71      }
72  
73      @Override
74      public String getRule() {
75          return "Spend this mana only to cast Dragon spells or activate abilities of Dragons";
76      }
77  }
78  
79  class OrbOfDragonkindConditionalMana extends ConditionalMana {
80  
81      public OrbOfDragonkindConditionalMana(Mana mana) {
82          super(mana);
83          this.staticText = "Spend this mana only to cast Dragon spells or activate abilities of Dragons";
84          addCondition(OrbOfDragonkindManaCondition.instance);
85      }
86  }
87  
88  enum OrbOfDragonkindManaCondition implements Condition {
89      instance;
90  
91      @Override
92      public boolean apply(Game game, Ability source) {
93          MageObject object = game.getObject(source);
94          return object != null && object.hasSubtype(SubType.DRAGON, game);
95      }
96  }

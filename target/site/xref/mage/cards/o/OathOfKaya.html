25  public final class OathOfKaya extends CardImpl {
26  
27      public OathOfKaya(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}{B}");
29  
30          this.addSuperType(SuperType.LEGENDARY);
31  
32          // When Oath of Kaya enters the battlefield, it deals 3 damage to any target and you gain 3 life.
33          Ability ability = new EntersBattlefieldTriggeredAbility(new DamageTargetEffect(3, "it"));
34          ability.addEffect(new GainLifeEffect(3).concatBy("and"));
35          ability.addTarget(new TargetAnyTarget());
36          this.addAbility(ability);
37  
38          // Whenever an opponent attacks a planeswalker you control with one or more creatures,
39          // Oath of Kaya deals 2 damage to that player and you gain 2 life.
40          this.addAbility(new OathOfKayaTriggeredAbility());
41      }
42  
43      private OathOfKaya(final OathOfKaya card) {
44          super(card);
45      }
46  
47      @Override
48      public OathOfKaya copy() {
49          return new OathOfKaya(this);
50      }
51  }
52  
53  class OathOfKayaTriggeredAbility extends TriggeredAbilityImpl {
54  
55      public OathOfKayaTriggeredAbility() {
56          super(Zone.BATTLEFIELD, new DamageTargetEffect(2), false);
57          this.addEffect(new GainLifeEffect(2));
58      }
59  
60      public OathOfKayaTriggeredAbility(final OathOfKayaTriggeredAbility ability) {
61          super(ability);
62      }
63  
64      @Override
65      public OathOfKayaTriggeredAbility copy() {
66          return new OathOfKayaTriggeredAbility(this);
67      }
68  
69      @Override
70      public boolean checkEventType(GameEvent event, Game game) {
71          return event.getType() == GameEvent.EventType.DECLARED_ATTACKERS;
72      }
73  
74      @Override
75      public boolean checkTrigger(GameEvent event, Game game) {
76          Player you = game.getPlayer(this.getControllerId());
77          if (you == null) {
78              return false;
79          }
80  
81          if (game.getCombat().isPlaneswalkerAttacked(you.getId(), game)) {
82              for (UUID attacker : game.getCombat().getAttackers()) {
83                  Permanent attackingPermanent = game.getPermanent(attacker);
84                  if (attackingPermanent != null && attackingPermanent.isCreature(game)) {
85                      getEffects().setTargetPointer(new FixedTarget(attackingPermanent.getControllerId(), game));
86                      return true;
87                  }
88              }
89          }
90          return false;
91      }
92  
93      @Override
94      public String getRule() {
95          return "Whenever an opponent attacks a planeswalker you control with one or more creatures, "
96                  + "{this} deals 2 damage to that player and you gain 2 life.";
97      }
98  }

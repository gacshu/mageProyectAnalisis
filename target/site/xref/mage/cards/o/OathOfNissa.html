25  public final class OathOfNissa extends CardImpl {
26  
27      private static final FilterCard filter = new FilterCard("a creature, land, or planeswalker card");
28  
29      static {
30          filter.add(Predicates.or(CardType.CREATURE.getPredicate(),
31                  CardType.LAND.getPredicate(),
32                  CardType.PLANESWALKER.getPredicate()));
33      }
34  
35      public OathOfNissa(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{G}");
37          addSuperType(SuperType.LEGENDARY);
38  
39          // When Oath of Nissa enters the battlefield, look at the top three cards of your library.
40          // You may reveal a creature, land, or planeswalker card from among them and put it into your hand.
41          // Put the rest on the bottom of your library in any order.
42          this.addAbility(new EntersBattlefieldTriggeredAbility(new LookLibraryAndPickControllerEffect(
43                  3, 1, filter, PutCards.HAND, PutCards.BOTTOM_ANY)));
44  
45          // You may spend mana as though it were mana of any color to cast planeswalker spells.
46          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new OathOfNissaSpendAnyManaEffect()));
47      }
48  
49      private OathOfNissa(final OathOfNissa card) {
50          super(card);
51      }
52  
53      @Override
54      public OathOfNissa copy() {
55          return new OathOfNissa(this);
56      }
57  }
58  
59  class OathOfNissaSpendAnyManaEffect extends AsThoughEffectImpl implements AsThoughManaEffect {
60  
61      public OathOfNissaSpendAnyManaEffect() {
62          super(AsThoughEffectType.SPEND_OTHER_MANA, Duration.WhileOnBattlefield, Outcome.Benefit);
63          staticText = "you may spend mana as though it were mana of any color to cast planeswalker spells";
64      }
65  
66      public OathOfNissaSpendAnyManaEffect(final OathOfNissaSpendAnyManaEffect effect) {
67          super(effect);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          return true;
73      }
74  
75      @Override
76      public OathOfNissaSpendAnyManaEffect copy() {
77          return new OathOfNissaSpendAnyManaEffect(this);
78      }
79  
80      @Override
81      public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
82          objectId = CardUtil.getMainCardId(game, objectId); // for split cards
83          if (source.isControlledBy(affectedControllerId)) {
84              MageObject mageObject = game.getObject(objectId);
85              return mageObject != null && mageObject.isPlaneswalker(game);
86          }
87          return false;
88      }
89  
90      @Override
91      public ManaType getAsThoughManaType(ManaType manaType, ManaPoolItem mana, UUID affectedControllerId, Ability source, Game game) {
92          return mana.getFirstAvailable();
93      }
94  }

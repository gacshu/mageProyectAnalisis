24  public final class OnceUponATime extends CardImpl {
25  
26      public OnceUponATime(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{G}");
28  
29          // If this spell is the first spell you've cast this game, you may cast it without paying its mana cost.
30          this.addAbility(new AlternativeCostSourceAbility(
31                  null, OnceUponATimeCondition.instance, "If this spell is the first spell " +
32                  "you've cast this game, you may cast it without paying its mana cost."
33          ), new OnceUponATimeWatcher());
34  
35          // Look at the top five cards of your library. You may reveal a creature or land card from among them and put it into your hand. Put the rest on the bottom of your library in a random order.
36          this.getSpellAbility().addEffect(new LookLibraryAndPickControllerEffect(
37                  5, 1, StaticFilters.FILTER_CARD_CREATURE_OR_LAND, PutCards.HAND, PutCards.BOTTOM_RANDOM));
38      }
39  
40      private OnceUponATime(final OnceUponATime card) {
41          super(card);
42      }
43  
44      @Override
45      public OnceUponATime copy() {
46          return new OnceUponATime(this);
47      }
48  }
49  
50  enum OnceUponATimeCondition implements Condition {
51      instance;
52  
53      @Override
54      public boolean apply(Game game, Ability source) {
55          OnceUponATimeWatcher watcher = game.getState().getWatcher(OnceUponATimeWatcher.class);
56          return watcher != null && watcher.getSpellsCastThisGame(source.getControllerId());
57      }
58  }
59  
60  class OnceUponATimeWatcher extends Watcher {
61  
62      private final Set<UUID> castSpells = new HashSet();
63  
64      OnceUponATimeWatcher() {
65          super(WatcherScope.GAME);
66      }
67  
68      @Override
69      public void watch(GameEvent event, Game game) {
70          switch (event.getType()) {
71              case SPELL_CAST:
72                  castSpells.add(event.getPlayerId());
73                  return;
74              case BEGINNING_PHASE_PRE:
75                  if (game.getTurnNum() == 1) {
76                      castSpells.clear();
77                  }
78          }
79      }
80  
81      boolean getSpellsCastThisGame(UUID playerId) {
82          return !castSpells.contains(playerId);
83      }
84  }

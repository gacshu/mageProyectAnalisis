18  
19  import java.util.UUID;
20  
21  /**
22   * @author PurpleCrowbar
23   */
24  public final class OrcaSiegeDemon extends CardImpl {
25  
26      public OrcaSiegeDemon(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{5}{B}{R}");
28          addSuperType(SuperType.LEGENDARY);
29          this.subtype.add(SubType.DEMON);
30  
31          this.power = new MageInt(5);
32          this.toughness = new MageInt(5);
33  
34          // Trample
35          this.addAbility(TrampleAbility.getInstance());
36  
37          // Whenever another creature dies, put a +1/+1 counter on Orca, Siege Demon.
38          this.addAbility(new DiesCreatureTriggeredAbility(new AddCountersSourceEffect(CounterType.P1P1.createInstance()), false, true));
39  
40          // When Orca dies, it deals damage equal to its power divided as you choose among any number of targets.
41          Ability ability = new DiesSourceTriggeredAbility(new DamageMultiEffect(new SourcePermanentPowerCount())
42                  .setText("it deals damage equal to its power divided as you choose among any number of targets."));
43          ability.addTarget(new TargetAnyTargetAmount(new SourcePermanentPowerCount()));
44          this.addAbility(ability);
45      }
46  
47      private OrcaSiegeDemon(final OrcaSiegeDemon card) {
48          super(card);
49      }
50  
51      @Override
52      public OrcaSiegeDemon copy() {
53          return new OrcaSiegeDemon(this);
54      }
55  }

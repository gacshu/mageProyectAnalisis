30      public OlagLudevicsHubris(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "");
32  
33          this.addSuperType(SuperType.LEGENDARY);
34          this.subtype.add(SubType.ZOMBIE);
35          this.power = new MageInt(4);
36          this.toughness = new MageInt(4);
37          this.color.setBlue(true);
38          this.color.setBlack(true);
39          this.nightCard = true;
40  
41          // As this creature transforms in Olag, Ludevic's Hubris, it becomes a copy of a creature card exiled with it, except its name is Olag, Ludevic's Hubris, it's 4/4, and it's a legendary blue and black Zombie in addition to its other colors and types. Put a number of +1/+1 counters on Olag equal to the number of creature cards exiled with it.
42          this.addAbility(new SimpleStaticAbility(new OlagLudevicsHubrisEffect()));
43      }

55  class OlagLudevicsHubrisEffect extends ReplacementEffectImpl {
56  
57      OlagLudevicsHubrisEffect() {
58          super(Duration.WhileOnBattlefield, Outcome.Benefit);
59          staticText = "as this creature transforms into {this}, it becomes a copy of a creature card exiled with it, " +
60                  "except its name is Olag, Ludevic's Hubris, it's 4/4, and it's a legendary blue and black " +
61                  "Zombie in addition to its other colors and types. Put a number of +1/+1 counters on {this} " +
62                  "equal to the number of creature cards exiled with it";
63      }
64  
65      private OlagLudevicsHubrisEffect(final OlagLudevicsHubrisEffect effect) {
66          super(effect);
67      }
68  
69      @Override
70      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
71          ExileZone exileZone = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source));
72          if (exileZone == null) {
73              return false;
74          }
75  
76          Cards cards = new CardsImpl(exileZone);
77          cards.removeIf(uuid -> !game.getCard(uuid).isCreature(game));
78          if (cards.isEmpty()) {
79              return false;
80          }
81  
82          Card copyFromCard = getCard(cards, source, game);
83          if (copyFromCard == null) {
84              return false;
85          }
86          Permanent newBluePrint = new PermanentCard(copyFromCard, source.getControllerId(), game);
87          newBluePrint.assignNewId();
88          CopyApplier applier = new OlagLudevicsHubrisCopyApplier();
89          applier.apply(game, newBluePrint, source, source.getSourceId());
90          CopyEffect copyEffect = new CopyEffect(Duration.Custom, newBluePrint, source.getSourceId());
91          copyEffect.newId();
92          copyEffect.setApplier(applier);
93          Ability newAbility = source.copy();
94          copyEffect.init(newAbility, game);
95          game.addEffect(copyEffect, newAbility);
96          return false;
97      }
98  
99      private Card getCard(Cards cards, Ability source, Game game) {
100         if (cards.size() == 1) {
101             return cards.getRandom(game);
102         }
103         Player player = game.getPlayer(source.getControllerId());
104         if (player == null) {
105             return cards.getRandom(game);
106         }
107         TargetCard target = new TargetCardInExile(StaticFilters.FILTER_CARD);
108         player.choose(outcome, target, source, game);
109         return cards.get(target.getFirstTarget(), game);
110     }
111 
112     @Override
113     public boolean checksEventType(GameEvent event, Game game) {
114         return event.getType() == GameEvent.EventType.TRANSFORMING;
115     }
116 
117     @Override
118     public boolean applies(GameEvent event, Ability source, Game game) {
119         return source.getSourceId().equals(event.getTargetId())
120                 && source.getSourcePermanentIfItStillExists(game) != null;
121     }
122 
123     @Override
124     public OlagLudevicsHubrisEffect copy() {
125         return new OlagLudevicsHubrisEffect(this);
126     }

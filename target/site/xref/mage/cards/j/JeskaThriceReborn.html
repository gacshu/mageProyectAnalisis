36      public JeskaThriceReborn(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{2}{R}");
38  
39          this.addSuperType(SuperType.LEGENDARY);
40          this.subtype.add(SubType.JESKA);
41          this.setStartingLoyalty(0);
42  
43          // Jeska, Thrice Reborn enters the battlefield with a loyalty counter on it for each time you've cast a commander from the command zone this game.
44          this.addAbility(new EntersBattlefieldAbility(new AddCountersSourceEffect(
45                  CounterType.LOYALTY.createInstance(0), JeskaThriceRebornValue.instance, false
46          ), "with a loyalty counter on her for each time " +
47                  "you've cast a commander from the command zone this game"
48          ).addHint(JeskaThriceRebornValue.getHint()));
49  
50          // +0: Choose target creature. Until your next turn, if that creature would deal combat damage to one of your opponents, it deals triple that damage to that player instead.
51          Ability ability = new LoyaltyAbility(new JeskaThriceRebornEffect(), 0);
52          ability.addTarget(new TargetCreaturePermanent());
53          this.addAbility(ability);
54  
55          // âˆ’X: Jeska, Thrice Reborn deals X damage to each of up to three targets.
56          ability = new LoyaltyAbility(new DamageTargetEffect(GetXLoyaltyValue.instance)
57                  .setText("{this} deals X damage to each of up to three targets"));
58          ability.addTarget(new TargetAnyTarget(0, 3));
59          this.addAbility(ability);
60  
61          // Jeska, Thrice Reborn can be your commander.
62          this.addAbility(CanBeYourCommanderAbility.getInstance());
63  
64          // Partner
65          this.addAbility(PartnerAbility.getInstance());
66      }

70      }

78  enum JeskaThriceRebornValue implements DynamicValue {
79      instance;
80      private static final Hint hint = new ValueHint(
81              "Number of times you've cast a commander this game", instance
82      );
83  
84      @Override
85      public int calculate(Game game, Ability sourceAbility, Effect effect) {
86          Player player = game.getPlayer(sourceAbility.getControllerId());
87          if (player == null) {
88              return 0;
89          }
90          CommanderPlaysCountWatcher watcher = game.getState().getWatcher(CommanderPlaysCountWatcher.class);
91          return watcher == null ? 0 : game
92                  .getCommandersIds(player, CommanderCardType.COMMANDER_OR_OATHBREAKER, false)
93                  .stream()
94                  .mapToInt(watcher::getPlaysCount)
95                  .sum();
96      }
97  
98      @Override
99      public JeskaThriceRebornValue copy() {
100         return instance;
101     }
102 
103     @Override
104     public String getMessage() {
105         return "";
106     }
107 
108     public static Hint getHint() {
109         return hint;
110     }
111 }

113 class JeskaThriceRebornEffect extends ReplacementEffectImpl {
114 
115     JeskaThriceRebornEffect() {
116         super(Duration.UntilYourNextTurn, Outcome.Damage);
117         staticText = "choose target creature. Until your next turn, " +
118                 "if that creature would deal combat damage to one of your opponents, " +
119                 "it deals triple that damage to that player instead";
120     }
121 
122     private JeskaThriceRebornEffect(final JeskaThriceRebornEffect effect) {
123         super(effect);
124     }
125 
126     @Override
127     public JeskaThriceRebornEffect copy() {
128         return new JeskaThriceRebornEffect(this);
129     }
130 
131     @Override
132     public boolean checksEventType(GameEvent event, Game game) {
133         return event.getType() == GameEvent.EventType.DAMAGE_PLAYER;
134     }
135 
136     @Override
137     public boolean applies(GameEvent event, Ability source, Game game) {
138         if (!((DamagePlayerEvent) event).isCombatDamage()
139                 || !event.getSourceId().equals(targetPointer.getFirst(game, source))) {
140             return false;
141         }
142         Player player = game.getPlayer(source.getControllerId());
143         return player != null && player.hasOpponent(event.getTargetId(), game);
144     }
145 
146     @Override
147     public boolean apply(Game game, Ability source) {
148         return true;
149     }
150 
151     @Override
152     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
153         int amount = event.getAmount();
154         event.setAmount(CardUtil.overflowInc(amount, event.getAmount()));
155         event.setAmount(CardUtil.overflowInc(amount, event.getAmount()));
156         return false;
157     }

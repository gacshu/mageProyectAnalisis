4   import java.util.UUID;
5   import mage.abilities.LoyaltyAbility;
6   import mage.abilities.effects.common.DrawCardAllEffect;
7   import mage.abilities.effects.common.DrawCardTargetEffect;
8   import mage.abilities.effects.common.PutLibraryIntoGraveTargetEffect;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  import mage.constants.SubType;
13  import mage.constants.SuperType;
14  import mage.target.TargetPlayer;
15  
16  /**
17   *
18   * @author BetaSteward_at_googlemail.com
19   */
20  public final class JaceBeleren extends CardImpl {
21  
22      public JaceBeleren(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId,setInfo,new CardType[]{CardType.PLANESWALKER},"{1}{U}{U}");
24          this.addSuperType(SuperType.LEGENDARY);
25          this.subtype.add(SubType.JACE);
26  
27          this.setStartingLoyalty(3);
28  
29          // +2: Each player draws a card.
30          this.addAbility(new LoyaltyAbility(new DrawCardAllEffect(1), 2));
31  
32          // -1: Target player draws a card.
33          LoyaltyAbility ability1 = new LoyaltyAbility(new DrawCardTargetEffect(1), -1);
34          ability1.addTarget(new TargetPlayer());
35          this.addAbility(ability1);
36  
37          // -10: Target player puts the top twenty cards of their library into their graveyard.
38          LoyaltyAbility ability2 = new LoyaltyAbility(new PutLibraryIntoGraveTargetEffect(20), -10);
39          ability2.addTarget(new TargetPlayer());
40          this.addAbility(ability2);
41      }
42  
43      private JaceBeleren(final JaceBeleren card) {
44          super(card);
45      }
46  
47      @Override
48      public JaceBeleren copy() {
49          return new JaceBeleren(this);
50      }
51  
52  }

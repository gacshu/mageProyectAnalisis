38      public JadziOracleOfArcavios(UUID ownerId, CardSetInfo setInfo) {
39          super(
40                  ownerId, setInfo,
41                  new CardType[]{CardType.CREATURE}, new SubType[]{SubType.HUMAN, SubType.WIZARD}, "{6}{U}{U}",
42                  "Journey to the Oracle",
43                  new CardType[]{CardType.SORCERY}, new SubType[]{}, "{2}{G}{G}"
44          );
45  
46          // 1.
47          // Jadzi, Oracle of Arcavios
48          // Legendary Creature - Human Wizard
49          this.getLeftHalfCard().addSuperType(SuperType.LEGENDARY);
50          this.getLeftHalfCard().setPT(5, 5);
51  
52          // Discard a card: Return Jadzi, Oracle of Arcavios to its owner's hand.
53          this.getLeftHalfCard().addAbility(new SimpleActivatedAbility(
54                  new ReturnToHandSourceEffect(true), new DiscardCardCost()
55          ));
56  
57          // Magecraft â€” Whenever you cast or copy an instant or sorcery spell, reveal the top card of your library. If it's a nonland card, you may cast it by paying {1} rather than paying its mana cost. If it's a land card, put it onto the battlefield.
58          this.getLeftHalfCard().addAbility(new MagecraftAbility(new JadziOracleOfArcaviosEffect()));
59  
60          // 1.
61          // Journey to the Oracle
62          // Sorcery
63          // You may put any number of land cards from your hand onto the battlefield. Then if you control eight or more lands, you may discard a card. If you do, return Journey to the Oracle to it owner's hand.
64          this.getRightHalfCard().getSpellAbility().addEffect(new JourneyToTheOracleEffect());
65          this.getRightHalfCard().getSpellAbility().addEffect(new ConditionalOneShotEffect(
66                  new DoIfCostPaid(
67                          new ReturnToHandSourceEffect(), new DiscardCardCost()
68                  ), condition, "Then if you control eight or more lands, "
69                  + "you may discard a card. If you do, return {this} to its owner's hand."
70          ));
71          this.getRightHalfCard().getSpellAbility().addHint(LandsYouControlHint.instance);
72      }

84  class JadziOracleOfArcaviosEffect extends OneShotEffect {
85  
86      JadziOracleOfArcaviosEffect() {
87          super(Outcome.Benefit);
88          staticText = "reveal the top card of your library. If it's a nonland card, you may cast it "
89                  + "by paying {1} rather than paying its mana cost. If it's a land card, put it onto the battlefield";
90      }
91  
92      private JadziOracleOfArcaviosEffect(final JadziOracleOfArcaviosEffect effect) {
93          super(effect);
94      }
95  
96      @Override
97      public JadziOracleOfArcaviosEffect copy() {
98          return new JadziOracleOfArcaviosEffect(this);
99      }
100 
101     @Override
102     public boolean apply(Game game, Ability source) {
103         Player controller = game.getPlayer(source.getControllerId());
104         if (controller == null) {
105             return false;
106         }
107         Card card = controller.getLibrary().getFromTop(game);
108         if (card == null) {
109             return false;
110         }
111         controller.revealCards(source, new CardsImpl(card), game);
112         if (card.isLand(game)) {
113             // this is a bit wierd in game, though it works fine
114             // note that MDFC land cards are handled differently: they can't be moved
115             return controller.moveCards(card, Zone.BATTLEFIELD, source, game);
116         }
117 
118         // query player
119         if (!controller.chooseUse(outcome, "Cast revealed card " + card.getName() + " by paying {1}?", source, game)) {
120             return false;
121         }
122         CardUtil.castSingle(controller, source, game, card, new ManaCostsImpl<>("{1}"));
123 
124         return true;
125     }
126 }

128 class JourneyToTheOracleEffect extends OneShotEffect {
129 
130     JourneyToTheOracleEffect() {
131         super(Outcome.Benefit);
132         staticText = "You may put any number of land cards from your hand onto the battlefield";
133     }
134 
135     private JourneyToTheOracleEffect(final JourneyToTheOracleEffect effect) {
136         super(effect);
137     }
138 
139     @Override
140     public JourneyToTheOracleEffect copy() {
141         return new JourneyToTheOracleEffect(this);
142     }
143 
144     @Override
145     public boolean apply(Game game, Ability source) {
146         Player player = game.getPlayer(source.getControllerId());
147         if (player == null) {
148             return false;
149         }
150         TargetCardInHand target = new TargetCardInHand(
151                 0, Integer.MAX_VALUE, StaticFilters.FILTER_CARD_LANDS
152         );
153         player.choose(Outcome.PutCardInPlay, target, source, game);
154         return player.moveCards(new CardsImpl(target.getTargets()), Zone.BATTLEFIELD, source, game);
155     }

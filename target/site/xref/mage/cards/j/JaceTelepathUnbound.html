24  public final class JaceTelepathUnbound extends CardImpl {
25  
26      public JaceTelepathUnbound(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "");
28          this.addSuperType(SuperType.LEGENDARY);
29          this.subtype.add(SubType.JACE);
30  
31          this.color.setBlue(true);
32          this.nightCard = true;
33  
34          this.setStartingLoyalty(5);
35  
36          // +1: Up to one target creature gets -2/-0 until your next turn.
37          Effect effect = new BoostTargetEffect(-2, 0, Duration.UntilYourNextTurn);
38          effect.setText("Up to one target creature gets -2/-0 until your next turn");
39          Ability ability = new LoyaltyAbility(effect, 1);
40          ability.addTarget(new TargetCreaturePermanent(0, 1));
41          this.addAbility(ability);
42  
43          // -3: You may cast target instant or sorcery card from your graveyard this turn. If that card would be put into your graveyard this turn, exile it instead.
44          CastCardFromGraveyardThenExileItEffect minusEffect = new CastCardFromGraveyardThenExileItEffect();
45          minusEffect.setText("You may cast target instant or sorcery card from your graveyard this turn. If that card would be put into your graveyard this turn, exile it instead");
46          ability = new LoyaltyAbility(minusEffect, -3);
47          ability.addTarget(new TargetCardInYourGraveyard(new FilterInstantOrSorceryCard()));
48          this.addAbility(ability);
49  
50          // âˆ’9: You get an emblem with "Whenever you cast a spell, target opponent mills five cards."
51          this.addAbility(new LoyaltyAbility(new GetEmblemEffect(new JaceTelepathUnboundEmblem()), -9));
52      }
53  
54      private JaceTelepathUnbound(final JaceTelepathUnbound card) {
55          super(card);
56      }
57  
58      @Override
59      public JaceTelepathUnbound copy() {
60          return new JaceTelepathUnbound(this);
61      }
62  }

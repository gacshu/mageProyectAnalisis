26  public final class JediHolocron extends CardImpl {
27  
28      public JediHolocron(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
30  
31          // {T}: Put a charge counter on Jedi Holocron.
32          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.CHARGE.createInstance(1)), new TapSourceCost()));
33  
34          // {T}, Remove a charge counter from Jedi Holocron: Add {G}, {W} or {U}.
35          Cost cost = new RemoveCountersSourceCost(CounterType.CHARGE.createInstance(1));
36          Ability ability = new GreenManaAbility();
37          ability.addCost(cost);
38          this.addAbility(ability);
39  
40          ability = new WhiteManaAbility();
41          ability.addCost(cost);
42          this.addAbility(ability);
43  
44          ability = new BlueManaAbility();
45          ability.addCost(cost);
46          this.addAbility(ability);
47  
48          // {T}, Remove two charge counters from Jedi Holocron: Add GW or WU.
49          cost = new RemoveCountersSourceCost(CounterType.CHARGE.createInstance(2));
50  
51          ability = new SimpleManaAbility(Zone.BATTLEFIELD, new Mana(1, 0, 0, 0, 1, 0, 0, 0), new TapSourceCost());
52          ability.addCost(cost);
53          this.addAbility(ability);
54  
55          ability = new SimpleManaAbility(Zone.BATTLEFIELD, new Mana(1, 1, 0, 0, 0, 0, 0, 0), new TapSourceCost());
56          ability.addCost(cost);
57          this.addAbility(ability);
58      }
59  
60      private JediHolocron(final JediHolocron card) {
61          super(card);
62      }
63  
64      @Override
65      public JediHolocron copy() {
66          return new JediHolocron(this);
67      }
68  }

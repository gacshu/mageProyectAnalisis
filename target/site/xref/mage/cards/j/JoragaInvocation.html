25  public final class JoragaInvocation extends CardImpl {
26  
27      public JoragaInvocation(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{G}{G}");
29  
30          // Each creature you control gets +3/+3 until end of turn and must be blocked this turn if able.
31          this.getSpellAbility().addEffect(new BoostControlledEffect(3, 3, Duration.EndOfTurn));
32          this.getSpellAbility().addEffect(new JoragaInvocationEffect());
33  
34      }
35  
36      private JoragaInvocation(final JoragaInvocation card) {
37          super(card);
38      }
39  
40      @Override
41      public JoragaInvocation copy() {
42          return new JoragaInvocation(this);
43      }
44  }
45  
46  class JoragaInvocationEffect extends OneShotEffect {
47  
48      public JoragaInvocationEffect() {
49          super(Outcome.Detriment);
50          this.staticText = "and must be blocked this turn if able";
51      }
52  
53      public JoragaInvocationEffect(final JoragaInvocationEffect effect) {
54          super(effect);
55      }
56  
57      @Override
58      public JoragaInvocationEffect copy() {
59          return new JoragaInvocationEffect(this);
60      }
61  
62      @Override
63      public boolean apply(Game game, Ability source) {
64          Player controller = game.getPlayer(source.getControllerId());
65          if (controller != null) {
66              for (Permanent permanent : game.getBattlefield().getAllActivePermanents(StaticFilters.FILTER_PERMANENT_CREATURE, game)) {
67                  ContinuousEffect effect = new MustBeBlockedByAtLeastOneTargetEffect();
68                  effect.setTargetPointer(new FixedTarget(permanent, game));
69                  game.addEffect(effect, source);
70              }
71              return true;
72          }
73          return false;
74      }
75  }

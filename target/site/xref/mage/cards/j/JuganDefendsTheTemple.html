23  public final class JuganDefendsTheTemple extends CardImpl {
24  
25      public JuganDefendsTheTemple(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{G}");
27  
28          this.subtype.add(SubType.SAGA);
29          this.secondSideCardClazz = mage.cards.r.RemnantOfTheRisingStar.class;
30  
31          // (As this Saga enters and after your draw step, add a lore counter.)
32          SagaAbility sagaAbility = new SagaAbility(this);
33  
34          // I — Create a 1/1 green Human Monk creature token with "{T}: Add {G}."
35          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, new CreateTokenEffect(new HumanMonkToken()));
36  
37          // II — Put a +1/+1 counter on each of up to two target creatures.
38          sagaAbility.addChapterEffect(
39                  this, SagaChapter.CHAPTER_II, SagaChapter.CHAPTER_II,
40                  new AddCountersTargetEffect(CounterType.P1P1.createInstance()),
41                  new TargetPermanent(0, 2, StaticFilters.FILTER_PERMANENT_CREATURES)
42          );
43  
44          // III — Exile this Saga, then return it to the battlefield transformed under your control.
45          this.addAbility(new TransformAbility());
46          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new ExileSagaAndReturnTransformedEffect());
47  
48          this.addAbility(sagaAbility);
49      }
50  
51      private JuganDefendsTheTemple(final JuganDefendsTheTemple card) {
52          super(card);
53      }
54  
55      @Override
56      public JuganDefendsTheTemple copy() {
57          return new JuganDefendsTheTemple(this);
58      }
59  }

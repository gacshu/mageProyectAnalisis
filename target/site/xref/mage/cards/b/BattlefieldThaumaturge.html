32      public BattlefieldThaumaturge(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{U}");
34          this.subtype.add(SubType.HUMAN, SubType.WIZARD);
35  
36          this.power = new MageInt(2);
37          this.toughness = new MageInt(1);
38  
39          // Each instant and sorcery spell you cast costs 1 less to cast for each creature it targets.
40          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new BattlefieldThaumaturgeSpellsCostReductionEffect()));
41  
42          // Heroic - Whenever you cast a spell that targets Battlefield Thaumaturge, Battlefield Thaumaturge gains hexproof until end of turn.
43          this.addAbility(new HeroicAbility(new GainAbilitySourceEffect(HexproofAbility.getInstance(), Duration.EndOfTurn)));
44      }

47          super(card);
48      }

56  class BattlefieldThaumaturgeSpellsCostReductionEffect extends CostModificationEffectImpl {
57  
58      public BattlefieldThaumaturgeSpellsCostReductionEffect() {
59          super(Duration.WhileOnBattlefield, Outcome.Benefit, CostModificationType.REDUCE_COST);
60          this.staticText = "Each instant and sorcery spell you cast costs {1} less to cast for each creature it targets";
61      }
62  
63      protected BattlefieldThaumaturgeSpellsCostReductionEffect(BattlefieldThaumaturgeSpellsCostReductionEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public boolean apply(Game game, Ability source, Ability abilityToModify) {
69          int reduceAmount;
70          if (game.inCheckPlayableState()) {
71              // checking state (search max possible targets)
72              reduceAmount = getMaxPossibleTargetCreatures(abilityToModify, game);
73          } else {
74              // real cast check
75              Set<UUID> creaturesTargeted = new HashSet<>();
76              for (Target target : abilityToModify.getAllSelectedTargets()) {
77                  if (target.isNotTarget()) {
78                      continue;
79                  }
80                  for (UUID uuid : target.getTargets()) {
81                      Permanent permanent = game.getPermanent(uuid);
82                      if (permanent != null && permanent.isCreature(game)) {
83                          creaturesTargeted.add(permanent.getId());
84                      }
85                  }
86              }
87              reduceAmount = creaturesTargeted.size();
88          }
89          CardUtil.reduceCost(abilityToModify, reduceAmount);
90          return true;
91      }
92  
93  
94      private int getMaxPossibleTargetCreatures(Ability ability, Game game) {
95          // checks only one mode, so it can be wrong in rare use cases with multi-modes (example: mode one gives +2 and mode two gives another +1 -- total +3)
96          int maxAmount = 0;
97          for (Mode mode : ability.getModes().values()) {
98              for (Target target : mode.getTargets()) {
99                  if (target.isNotTarget()) {
100                     continue;
101                 }
102                 Set<UUID> possibleList = target.possibleTargets(ability.getControllerId(), ability, game);
103                 possibleList.removeIf(id -> {
104                     Permanent permanent = game.getPermanent(id);
105                     return permanent == null || !permanent.isCreature(game);
106                 });
107                 int possibleAmount = Math.min(possibleList.size(), target.getMaxNumberOfTargets());
108                 maxAmount = Math.max(maxAmount, possibleAmount);
109             }
110         }
111         return maxAmount;
112     }
113 
114     @Override
115     public boolean applies(Ability abilityToModify, Ability source, Game game) {
116         if ((abilityToModify instanceof SpellAbility)
117                 && abilityToModify.isControlledBy(source.getControllerId())) {
118             Spell spell = (Spell) game.getStack().getStackObject(abilityToModify.getId());
119             if (spell != null) {
120                 return StaticFilters.FILTER_CARD_INSTANT_OR_SORCERY.match(spell, game);
121             } else {
122                 Card sourceCard = game.getCard(abilityToModify.getSourceId());
123                 return sourceCard != null && StaticFilters.FILTER_CARD_INSTANT_OR_SORCERY.match(sourceCard, game);
124             }
125         }
126         return false;
127     }
128 
129     @Override
130     public BattlefieldThaumaturgeSpellsCostReductionEffect copy() {
131         return new BattlefieldThaumaturgeSpellsCostReductionEffect(this);
132     }
133 }

29  public final class BirthOfTheImperium extends CardImpl {
30  
31      public BirthOfTheImperium(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}{U}{B}");
33  
34          this.subtype.add(SubType.SAGA);
35  
36          // (As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)
37          SagaAbility sagaAbility = new SagaAbility(this);
38  
39          // I -- Create a 2/2 white Astartes Warrior creature token with vigilance for each opponent you have.
40          sagaAbility.addChapterEffect(
41                  this, SagaChapter.CHAPTER_I,
42                  new CreateTokenEffect(new WhiteAstartesWarriorToken(), OpponentsCount.instance)
43                          .setText("create a 2/2 white Astartes Warrior creature " +
44                                  "token with vigilance for each opponent you have")
45          );
46  
47          // II -- Each opponent sacrifices a creature.
48          sagaAbility.addChapterEffect(
49                  this, SagaChapter.CHAPTER_II,
50                  new SacrificeOpponentsEffect(StaticFilters.FILTER_PERMANENT_CREATURE)
51          );
52  
53          // III -- Draw two cards for each opponent who controls fewer creatures than you.
54          sagaAbility.addChapterEffect(
55                  this, SagaChapter.CHAPTER_III,
56                  new DrawCardSourceControllerEffect(BirthOfTheImperiumValue.instance)
57          );
58          this.addAbility(sagaAbility);
59      }
60  
61      private BirthOfTheImperium(final BirthOfTheImperium card) {
62          super(card);
63      }
64  
65      @Override
66      public BirthOfTheImperium copy() {
67          return new BirthOfTheImperium(this);
68      }
69  }

71  enum BirthOfTheImperiumValue implements DynamicValue {
72      instance;
73  
74      @Override
75      public int calculate(Game game, Ability sourceAbility, Effect effect) {
76          Map<UUID, Integer> map = game
77                  .getBattlefield()
78                  .getActivePermanents(
79                          StaticFilters.FILTER_PERMANENT_CREATURE,
80                          sourceAbility.getControllerId(), sourceAbility, game
81                  ).stream()
82                  .map(Controllable::getControllerId)
83                  .collect(Collectors.toMap(Function.identity(), x -> 1, Integer::sum));
84          int yourCreatures = map.getOrDefault(sourceAbility.getControllerId(), 0);
85          return yourCreatures > 0 ? 2 * game
86                  .getOpponents(sourceAbility.getControllerId())
87                  .stream().mapToInt(uuid -> map.getOrDefault(uuid, 0))
88                  .filter(x -> x < yourCreatures)
89                  .sum() : 0;
90      }
91  
92      @Override
93      public BirthOfTheImperiumValue copy() {
94          return this;
95      }
96  
97      @Override
98      public String getMessage() {
99          return "opponent who controls fewer creatures than you";
100     }
101 
102     @Override
103     public String toString() {
104         return "two";
105     }

20   *
21   * @author maurer.it_at_gmail.com
22   */
23  public final class BlackcleaveCliffs extends CardImpl {
24  
25      private static final FilterLandPermanent filter = new FilterLandPermanent();
26  
27      public BlackcleaveCliffs(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.LAND}, null);
29  
30          Condition controls = new InvertCondition(new PermanentsOnTheBattlefieldCondition(filter, ComparisonType.FEWER_THAN, 3));
31          String abilityText = " tapped unless you control two or fewer other lands";
32          this.addAbility(new EntersBattlefieldAbility(new ConditionalOneShotEffect(new TapSourceEffect(), controls, abilityText), abilityText));
33          this.addAbility(new BlackManaAbility());
34          this.addAbility(new RedManaAbility());
35      }
36  
37      private BlackcleaveCliffs(final BlackcleaveCliffs card) {
38          super(card);
39      }
40  
41      @Override
42      public BlackcleaveCliffs copy() {
43          return new BlackcleaveCliffs(this);
44      }
45  
46  }

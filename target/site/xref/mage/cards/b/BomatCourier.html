28  public final class BomatCourier extends CardImpl {
29  
30      public BomatCourier(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{1}");
32          this.subtype.add(SubType.CONSTRUCT);
33          this.power = new MageInt(1);
34          this.toughness = new MageInt(1);
35  
36          // Haste
37          this.addAbility(HasteAbility.getInstance());
38  
39          // Whenever Bomat Courier attacks, exile the top card of your library face down.
40          this.addAbility(new AttacksTriggeredAbility(new BomatCourierExileEffect(), false));
41  
42          // {R}, Discard your hand, Sacrifice Bomat Courier: Put all cards exiled with Bomat Courier into their owners' hands.
43          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new BomatCourierReturnEffect(), new ColoredManaCost(ColoredManaSymbol.R));
44          ability.addCost(new DiscardHandCost());
45          ability.addCost(new SacrificeSourceCost());
46          this.addAbility(ability);
47      }
48  
49      private BomatCourier(final BomatCourier card) {
50          super(card);
51      }
52  
53      @Override
54      public BomatCourier copy() {
55          return new BomatCourier(this);
56      }
57  }

59  class BomatCourierExileEffect extends OneShotEffect {
60  
61      BomatCourierExileEffect() {
62          super(Outcome.Exile);
63          this.staticText = "exile the top card of your library face down";
64      }
65  
66      BomatCourierExileEffect(final BomatCourierExileEffect effect) {
67          super(effect);
68      }
69  
70      @Override
71      public BomatCourierExileEffect copy() {
72          return new BomatCourierExileEffect(this);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          Player controller = game.getPlayer(source.getControllerId());
78          MageObject sourceObject = source.getSourceObject(game);
79          if (controller != null && sourceObject != null) {
80              Card card = controller.getLibrary().getFromTop(game);
81              if (card != null) {
82                  UUID exileZoneId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
83                  card.setFaceDown(true, game);
84                  controller.moveCardsToExile(card, source, game, false, exileZoneId, sourceObject.getIdName());
85                  card.setFaceDown(true, game);
86                  return true;
87              }
88          }
89          return false;
90      }
91  }

93  class BomatCourierReturnEffect extends OneShotEffect {
94  
95      BomatCourierReturnEffect() {
96          super(Outcome.DrawCard);
97          this.staticText = "Put all cards exiled with {this} into their owners' hands";
98      }
99  
100     BomatCourierReturnEffect(final BomatCourierReturnEffect effect) {
101         super(effect);
102     }
103 
104     @Override
105     public BomatCourierReturnEffect copy() {
106         return new BomatCourierReturnEffect(this);
107     }
108 
109     @Override
110     public boolean apply(Game game, Ability source) {
111         Player controller = game.getPlayer(source.getControllerId());
112         if (controller != null) {
113             ExileZone exileZone = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter()));
114             if (exileZone != null) {
115                 controller.moveCards(exileZone, Zone.HAND, source, game);
116             }
117             return true;
118         }
119         return false;
120     }

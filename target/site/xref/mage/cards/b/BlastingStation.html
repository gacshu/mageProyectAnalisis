4   import java.util.UUID;
5   import mage.abilities.Ability;
6   import mage.abilities.common.EntersBattlefieldAllTriggeredAbility;
7   import mage.abilities.common.SimpleActivatedAbility;
8   import mage.abilities.costs.common.SacrificeTargetCost;
9   import mage.abilities.costs.common.TapSourceCost;
10  import mage.abilities.effects.common.DamageTargetEffect;
11  import mage.abilities.effects.common.UntapSourceEffect;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.Zone;
16  import static mage.filter.StaticFilters.FILTER_CONTROLLED_CREATURE_SHORT_TEXT;
17  import mage.filter.StaticFilters;
18  import mage.target.common.TargetControlledCreaturePermanent;
19  import mage.target.common.TargetAnyTarget;
20  
21  /**
22   *
23   * @author LevelX2
24   */
25  public final class BlastingStation extends CardImpl {
26  
27      public BlastingStation(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
29  
30          // {tap}, Sacrifice a creature: Blasting Station deals 1 damage to any target.
31          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DamageTargetEffect(1), new TapSourceCost());
32          ability.addCost(new SacrificeTargetCost(new TargetControlledCreaturePermanent(FILTER_CONTROLLED_CREATURE_SHORT_TEXT)));
33          ability.addTarget(new TargetAnyTarget());
34          this.addAbility(ability);
35  
36          // Whenever a creature enters the battlefield, you may untap Blasting Station.
37          this.addAbility(new EntersBattlefieldAllTriggeredAbility(Zone.BATTLEFIELD, new UntapSourceEffect(), StaticFilters.FILTER_PERMANENT_A_CREATURE, true));
38  
39      }
40  
41      private BlastingStation(final BlastingStation card) {
42          super(card);
43      }
44  
45      @Override
46      public BlastingStation copy() {
47          return new BlastingStation(this);
48      }
49  }

29  public final class Banefire extends CardImpl {
30  
31      public Banefire(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{R}");
33  
34          // Banefire deals X damage to any target.       
35          this.getSpellAbility().addEffect(new BaneFireEffect());
36          this.getSpellAbility().addTarget(new TargetAnyTarget());
37  
38          // If X is 5 or more, Banefire can't be countered and the damage can't be prevented.
39          this.addAbility(new SimpleStaticAbility(Zone.STACK, new BanefireCantCounterEffect()));
40      }
41  
42      private Banefire(final Banefire card) {
43          super(card);
44      }
45  
46      @Override
47      public Banefire copy() {
48          return new Banefire(this);
49      }
50  }

52  class testCondition implements Condition {
53  
54      private final DynamicValue xValue;
55      private final int limit;
56  
57      public testCondition(DynamicValue xValue, int limit) {
58          this.xValue = xValue;
59          this.limit = limit;
60      }
61  
62      @Override
63      public boolean apply(Game game, Ability source) {
64          Spell spell = (Spell) game.getStack().getStackObject(source.getSourceId());
65          if (spell != null) {
66              return (xValue.calculate(game, spell.getSpellAbility(), null) >= limit);
67          }
68          return false;
69  
70      }
71  }

73  class BaneFireEffect extends OneShotEffect {
74  
75      public BaneFireEffect() {
76          super(Outcome.Damage);
77          staticText = "{this} deals X damage to any target";
78      }
79  
80      public BaneFireEffect(final BaneFireEffect effect) {
81          super(effect);
82      }
83  
84      @Override
85      public BaneFireEffect copy() {
86          return new BaneFireEffect(this);
87      }
88  
89      @Override
90      public boolean apply(Game game, Ability source) {
91          Player targetPlayer = game.getPlayer(source.getFirstTarget());
92          Permanent targetCreature = game.getPermanent(source.getFirstTarget());
93          int damage = source.getManaCostsToPay().getX();
94          boolean preventable = damage < 5;
95          if (targetPlayer != null) {
96              targetPlayer.damage(damage, source.getSourceId(), source, game, false, preventable);
97              return true;
98          }
99          if (targetCreature != null) {
100             targetCreature.damage(damage, source.getSourceId(), source, game, false, preventable);
101             return true;
102         }
103         return false;
104     }
105 }

107 class BanefireCantCounterEffect extends ContinuousRuleModifyingEffectImpl {
108 
109     private Condition condition = new testCondition(ManacostVariableValue.REGULAR, 5);
110 
111     public BanefireCantCounterEffect() {
112         super(Duration.WhileOnStack, Outcome.Benefit);
113         staticText = "If X is 5 or more, this spell can't be countered and the damage can't be prevented";
114     }
115 
116     public BanefireCantCounterEffect(final BanefireCantCounterEffect effect) {
117         super(effect);
118         this.condition = effect.condition;
119     }
120 
121     @Override
122     public BanefireCantCounterEffect copy() {
123         return new BanefireCantCounterEffect(this);
124     }
125 
126     @Override
127     public boolean apply(Game game, Ability source) {
128         return true;
129     }
130 
131     @Override
132     public boolean applies(GameEvent event, Ability source, Game game) {
133         if (event.getType() != GameEvent.EventType.COUNTER) { return false; }
134 
135         Card card = game.getCard(source.getSourceId());
136         if (card == null) { return false; }
137 
138         UUID spellId = card.getSpellAbility().getId();
139         if (!event.getTargetId().equals(spellId)) { return false; }
140 
141         return condition.apply(game, source);
142     }

20  public final class BuyYourSilence extends CardImpl {
21  
22      public BuyYourSilence(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{W}");
24  
25          // Exile target nonland permanent. Its controller creates a Treasure token.
26          this.getSpellAbility().addEffect(new ExileTargetEffect());
27          this.getSpellAbility().addEffect(new BuyYourSilenceEffect());
28          this.getSpellAbility().addTarget(new TargetNonlandPermanent());
29      }
30  
31      private BuyYourSilence(final BuyYourSilence card) {
32          super(card);
33      }
34  
35      @Override
36      public BuyYourSilence copy() {
37          return new BuyYourSilence(this);
38      }
39  }
40  
41  class BuyYourSilenceEffect extends OneShotEffect {
42  
43      BuyYourSilenceEffect() {
44          super(Outcome.Benefit);
45          staticText = "Its controller creates a Treasure token";
46      }
47  
48      private BuyYourSilenceEffect(final BuyYourSilenceEffect effect) {
49          super(effect);
50      }
51  
52      @Override
53      public BuyYourSilenceEffect copy() {
54          return new BuyYourSilenceEffect(this);
55      }
56  
57      @Override
58      public boolean apply(Game game, Ability source) {
59          Permanent permanent = getTargetPointer().getFirstTargetPermanentOrLKI(game, source);
60          if (permanent == null) {
61              return false;
62          }
63          return new TreasureToken().putOntoBattlefield(1, game, source, permanent.getControllerId());
64      }
65  }

20  /**
21   * @author LevelX2
22   */
23  public final class BorosKeyrune extends CardImpl {
24  
25      public BorosKeyrune(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
27  
28          // {T}: Add {R} or {W}.
29          this.addAbility(new RedManaAbility());
30          this.addAbility(new WhiteManaAbility());
31  
32          // {R}{W}: Boros Keyrune becomes a 1/1 red and white Soldier artifact creature with double strike until end of turn.
33          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new BecomesCreatureSourceEffect(
34                  new CreatureToken(1, 1, "1/1 red and white Soldier artifact creature with double strike")
35                          .withColor("RW")
36                          .withSubType(SubType.SOLDIER)
37                          .withType(CardType.ARTIFACT)
38                          .withAbility(DoubleStrikeAbility.getInstance()),
39                  "", Duration.EndOfTurn), new ManaCostsImpl<>("{R}{W}")));
40      }
41  
42      private BorosKeyrune(final BorosKeyrune card) {
43          super(card);
44      }
45  
46      @Override
47      public BorosKeyrune copy() {
48          return new BorosKeyrune(this);
49      }
50  }

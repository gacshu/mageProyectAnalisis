23  public final class BlatantThievery extends CardImpl {
24  
25      public BlatantThievery(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{U}{U}{U}");
27  
28          // For each opponent, gain control of target permanent that player controls.
29          this.getSpellAbility().addEffect(new GainControlTargetEffect(Duration.Custom, true)
30                  .setTargetPointer(new EachTargetPointer())
31                  .setText("for each opponent, gain control of target permanent that player controls"));
32          this.getSpellAbility().setTargetAdjuster(BlatantThieveryAdjuster.instance);
33      }
34  
35      private BlatantThievery(final BlatantThievery card) {
36          super(card);
37      }
38  
39      @Override
40      public BlatantThievery copy() {
41          return new BlatantThievery(this);
42      }
43  }
44  
45  enum BlatantThieveryAdjuster implements TargetAdjuster {
46      instance;
47  
48      @Override
49      public void adjustTargets(Ability ability, Game game) {
50          ability.getTargets().clear();
51          for (UUID opponentId : game.getOpponents(ability.getControllerId())) {
52              Player opponent = game.getPlayer(opponentId);
53              if (opponent == null || game.getBattlefield().count(
54                      StaticFilters.FILTER_CONTROLLED_PERMANENT,
55                      opponentId, ability, game
56              ) < 1) {
57                  continue;
58              }
59              FilterPermanent filter = new FilterPermanent("Permanent controlled by " + opponent.getName());
60              filter.add(new ControllerIdPredicate(opponentId));
61              TargetPermanent targetPermanent = new TargetPermanent(filter);
62              ability.addTarget(targetPermanent);
63          }
64      }
65  }

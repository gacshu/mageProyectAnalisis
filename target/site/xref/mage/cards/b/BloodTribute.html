27  public final class BloodTribute extends CardImpl {
28  
29      private static final FilterControlledPermanent filter
30              = new FilterControlledPermanent(SubType.VAMPIRE, "an untapped Vampire you control");
31  
32      static {
33          filter.add(TappedPredicate.UNTAPPED);
34      }
35  
36      public BloodTribute(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{B}{B}");
38  
39          // Kicker - Tap an untapped Vampire you control.
40          this.addAbility(new KickerAbility(new TapTargetCost(new TargetControlledPermanent(filter))));
41  
42          // Target opponent loses half their life, rounded up.
43          this.getSpellAbility().addEffect(new BloodTributeLoseLifeEffect());
44          this.getSpellAbility().addTarget(new TargetOpponent());
45  
46          // If Blood Tribute was kicked, you gain life equal to the life lost this way.
47          Effect effect = new ConditionalOneShotEffect(
48                  new BloodTributeGainLifeEffect(),
49                  KickedCondition.ONCE,
50                  "if this spell was kicked, you gain life equal to the life lost this way");
51          this.getSpellAbility().addEffect(effect);
52      }
53  
54      private BloodTribute(final BloodTribute card) {
55          super(card);
56      }
57  
58      @Override
59      public BloodTribute copy() {
60          return new BloodTribute(this);
61      }
62  }
63  
64  class BloodTributeLoseLifeEffect extends OneShotEffect {
65  
66      public BloodTributeLoseLifeEffect() {
67          super(Outcome.Damage);
68          this.staticText = "Target opponent loses half their life, rounded up";
69      }
70  
71      public BloodTributeLoseLifeEffect(final BloodTributeLoseLifeEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public BloodTributeLoseLifeEffect copy() {
77          return new BloodTributeLoseLifeEffect(this);
78      }
79  
80      @Override
81      public boolean apply(Game game, Ability source) {
82          Player player = game.getPlayer(source.getFirstTarget());
83          if (player != null) {
84              Integer amount = (int) Math.ceil(player.getLife() / 2f);
85              if (amount > 0) {
86                  player.loseLife(amount, game, source, false);
87                  game.getState().setValue(source.getSourceId().toString() + "_BloodTribute", amount);
88                  return true;
89              }
90          }
91          return false;
92      }
93  }
94  
95  class BloodTributeGainLifeEffect extends OneShotEffect {
96  
97      public BloodTributeGainLifeEffect() {
98          super(Outcome.GainLife);
99          this.staticText = "If Blood Tribute was kicked, you gain life equal to the life lost this way";
100     }
101 
102     public BloodTributeGainLifeEffect(final BloodTributeGainLifeEffect effect) {
103         super(effect);
104     }
105 
106     @Override
107     public BloodTributeGainLifeEffect copy() {
108         return new BloodTributeGainLifeEffect(this);
109     }
110 
111     @Override
112     public boolean apply(Game game, Ability source) {
113         Player player = game.getPlayer(source.getControllerId());
114         if (player != null) {
115             Integer amount = (Integer) game.getState().getValue(source.getSourceId().toString() + "_BloodTribute");
116             if (amount != null && amount > 0) {
117                 player.gainLife(amount, game, source);
118                 return true;
119             }
120         }
121         return false;
122     }
123 }

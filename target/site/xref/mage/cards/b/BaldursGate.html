20  import mage.filter.predicate.mageobject.AnotherPredicate;
21  
22  import java.util.UUID;
23  
24  /**
25   * @author TheElk801
26   */
27  public final class BaldursGate extends CardImpl {
28  
29      private static final FilterPermanent filter
30              = new FilterControlledPermanent(SubType.GATE, "other Gates you control");
31  
32      static {
33          filter.add(AnotherPredicate.instance);
34      }
35  
36      private static final DynamicValue xValue = new PermanentsOnBattlefieldCount(filter, null);
37      private static final Hint hint = new ValueHint("Other Gates you control", xValue);
38  
39      public BaldursGate(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
41  
42          this.addSuperType(SuperType.LEGENDARY);
43          this.subtype.add(SubType.GATE);
44  
45          // {T}: Add {C}.
46          this.addAbility(new ColorlessManaAbility());
47  
48          // {2}, {T}: Add X mana of any one color, where X is the number of other Gates you control.
49          Ability ability = new DynamicManaAbility(
50                  Mana.AnyMana(1), xValue, new GenericManaCost(2), "Add X mana of any one color, " +
51                  "where X is the number of other Gates you control.", true
52          );
53          ability.addCost(new TapSourceCost());
54          this.addAbility(ability.addHint(hint));
55      }
56  
57      private BaldursGate(final BaldursGate card) {
58          super(card);
59      }
60  
61      @Override
62      public BaldursGate copy() {
63          return new BaldursGate(this);
64      }
65  }

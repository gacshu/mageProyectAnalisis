20  import mage.filter.common.FilterCreaturePermanent;
21  import mage.filter.predicate.mageobject.MulticoloredPredicate;
22  import mage.game.permanent.token.KnightToken;
23  import mage.target.common.TargetControlledCreaturePermanent;
24  
25  /**
26   *
27   * @author htrajan
28   */
29  public final class BasrisLieutenant extends CardImpl {
30  
31      private static final FilterObject<MageObject> multicoloredFilter = new FilterObject<>("multicolored");
32      private static final FilterCreaturePermanent controlledCreatureWithP1P1CounterFilter = new FilterCreaturePermanent("creature you control");
33  
34      static {
35          multicoloredFilter.add(MulticoloredPredicate.instance);
36          controlledCreatureWithP1P1CounterFilter.add(TargetController.YOU.getControllerPredicate());
37          controlledCreatureWithP1P1CounterFilter.add(CounterType.P1P1.getPredicate());
38      }
39  
40      public BasrisLieutenant(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}");
42  
43          this.subtype.add(SubType.HUMAN);
44          this.subtype.add(SubType.KNIGHT);
45          this.power = new MageInt(3);
46          this.toughness = new MageInt(4);
47  
48          // Vigilance, protection from multicolored
49          this.addAbility(VigilanceAbility.getInstance());
50  
51          // protection from multicolored
52          this.addAbility(new ProtectionAbility(multicoloredFilter));
53  
54          // When Basri's Lieutenant enters the battlefield, put a +1/+1 counter on target creature you control.
55          Ability ability = new EntersBattlefieldTriggeredAbility(new AddCountersTargetEffect(CounterType.P1P1.createInstance()));
56          ability.addTarget(new TargetControlledCreaturePermanent());
57          this.addAbility(ability);
58  
59          // Whenever Basri's Lieutenant or another creature you control dies, if it had a +1/+1 counter on it, create a 2/2 white Knight creature token with vigilance.
60          this.addAbility(new DiesThisOrAnotherCreatureTriggeredAbility(
61                  new CreateTokenEffect(new KnightToken()).setText("if it had a +1/+1 counter on it, create a 2/2 white Knight creature token with vigilance"),
62                  false,
63                  controlledCreatureWithP1P1CounterFilter
64          ).setApplyFilterOnSource(true));
65      }
66  
67      private BasrisLieutenant(final BasrisLieutenant card) {
68          super(card);
69      }
70  
71      @Override
72      public BasrisLieutenant copy() {
73          return new BasrisLieutenant(this);
74      }
75  }

29

31          this.addAbility(new BecomesTargetTriggeredAbility(
32                  new DamageTargetEffect(
33                          2, true, "that spell's controller", "{this}"
34                  ), StaticFilters.FILTER_SPELL_A, SetTargetPointer.PLAYER
35          ).setTriggerPhrase("Whenever {this} becomes the target of a spell, "));
36  
37          // Stomp
38          // Damage canâ€™t be prevented this turn. Stomp deals 2 damage to any target.
39          this.getSpellCard().getSpellAbility().addEffect(new DamageCantBePreventedEffect(Duration.EndOfTurn, "Damage can't be prevented this turn", false, false));
40          this.getSpellCard().getSpellAbility().addEffect(new DamageTargetEffect(2));
41          this.getSpellCard().getSpellAbility().addTarget(new TargetAnyTarget());
42      }

3   import mage.abilities.Mode;
4   import mage.abilities.effects.common.FightTargetsEffect;
5   import mage.abilities.effects.common.search.SearchLibraryPutInHandEffect;
6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.filter.StaticFilters;
10  import mage.target.TargetPermanent;
11  import mage.target.common.TargetCardInLibrary;
12  import mage.target.common.TargetControlledCreaturePermanent;
13  
14  import java.util.UUID;
15  
16  /**
17   * @author TheElk801
18   */
19  public final class Bushwhack extends CardImpl {
20  
21      public Bushwhack(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{G}");
23  
24          // Choose one --
25          // * Search your library for a basic land card, reveal it, put it into your hand, then shuffle.
26          this.getSpellAbility().addEffect(new SearchLibraryPutInHandEffect(
27                  new TargetCardInLibrary(StaticFilters.FILTER_CARD_BASIC_LAND), true
28          ));
29  
30          // * Target creature you control fights target creature you don't control.
31          this.getSpellAbility().addMode(new Mode(new FightTargetsEffect(true))
32                  .addTarget(new TargetControlledCreaturePermanent())
33                  .addTarget(new TargetPermanent(StaticFilters.FILTER_CREATURE_YOU_DONT_CONTROL)));
34      }
35  
36      private Bushwhack(final Bushwhack card) {
37          super(card);
38      }
39  
40      @Override
41      public Bushwhack copy() {
42          return new Bushwhack(this);
43      }
44  }

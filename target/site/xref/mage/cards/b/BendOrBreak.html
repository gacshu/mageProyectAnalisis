47  class BendOrBreakEffect extends OneShotEffect {
48  
49  
50      public BendOrBreakEffect() {
51          super(Outcome.DestroyPermanent);
52          this.staticText = "Each player separates all nontoken lands they control into two piles. For each player, one of their piles is chosen by one of their opponents of their choice. Destroy all lands in the chosen piles. Tap all lands in the other piles";
53      }
54  
55      public BendOrBreakEffect(final BendOrBreakEffect effect) {
56          super(effect);
57      }
58  
59      @Override
60      public BendOrBreakEffect copy() {
61          return new BendOrBreakEffect(this);
62      }
63  
64      @Override
65      public boolean apply(Game game, Ability source) {
66          Player controller = game.getPlayer(source.getControllerId());
67          if (controller == null) { return false; }
68  
69          // Map of players and their piles
70          Map<UUID, List<List<Permanent>>> playerPermanents = new LinkedHashMap<>();
71  
72          PlayerList playerList = game.getState().getPlayerList().copy();
73          while (!playerList.get().equals(source.getControllerId()) && controller.canRespond()) {
74              playerList.getNext();
75          }
76          Player currentPlayer = game.getPlayer(playerList.get());
77          Player nextPlayer;
78          UUID firstNextPlayer = null;
79  
80          while (!getNextPlayerInDirection(true, playerList).equals(firstNextPlayer) && controller.canRespond()) {
81              nextPlayer = game.getPlayer(playerList.get());
82              if (nextPlayer == null) {
83                  return false;
84              }
85              if (firstNextPlayer == null) {
86                  firstNextPlayer = nextPlayer.getId();
87              }
88              if (!nextPlayer.canRespond()) {
89                  continue;
90              }
91              // Each player separates all nontoken lands they control into two piles
92              if (currentPlayer == null || !game.getState().getPlayersInRange(controller.getId(), game).contains(currentPlayer.getId())) {
93                  continue;
94              }
95              List<Permanent> firstPile = new ArrayList<>();
96              List<Permanent> secondPile = new ArrayList<>();
97              FilterControlledLandPermanent filter = new FilterControlledLandPermanent("lands you control to assign to the first pile (lands not chosen will be assigned to the second pile)");
98              TargetPermanent target = new TargetControlledPermanent(0, Integer.MAX_VALUE, filter, true);
99              if (!target.canChoose(currentPlayer.getId(), source, game)) { continue; }
100 
101             // TODO: add support for AI (50/50), need AI hints mechanic here
102             currentPlayer.chooseTarget(Outcome.Neutral, target, source, game);
103             for (Permanent permanent : game.getBattlefield().getAllActivePermanents(filter, currentPlayer.getId(), game)) {
104                 if (target.getTargets().contains(permanent.getId())) {
105                     firstPile.add(permanent);
106                 } else {
107                     secondPile.add(permanent);
108                 }
109             }
110 
111             StringBuilder sb = new StringBuilder("First pile of ").append(currentPlayer.getLogName()).append(": ");
112             sb.append(firstPile.stream().map(Permanent::getLogName).collect(Collectors.joining(", ")));
113 
114             game.informPlayers(sb.toString());
115 
116             sb = new StringBuilder("Second pile of ").append(currentPlayer.getLogName()).append(": ");
117             sb.append(secondPile.stream().map(Permanent::getLogName).collect(Collectors.joining(", ")));
118 
119             game.informPlayers(sb.toString());
120 
121             List<List<Permanent>> playerPiles = new ArrayList<>();
122             playerPiles.add(firstPile);
123             playerPiles.add(secondPile);
124             playerPermanents.put(currentPlayer.getId(), playerPiles);
125 
126             currentPlayer = nextPlayer;
127         }
128 
129         // For each player, one of their piles is chosen by one of their opponents of their choice
130         for (Map.Entry<UUID, List<List<Permanent>>> playerPiles : playerPermanents.entrySet()) {
131             Player player = game.getPlayer(playerPiles.getKey());
132             if (player == null) { continue; }
133 
134             FilterPlayer filter = new FilterPlayer("opponent");
135             List<PlayerIdPredicate> opponentPredicates = new ArrayList<>();
136             for (UUID opponentId : game.getOpponents(player.getId())) {
137                 opponentPredicates.add(new PlayerIdPredicate(opponentId));
138             }
139             filter.add(Predicates.or(opponentPredicates));
140             Target target = new TargetPlayer(1, 1, true, filter);
141             target.setTargetController(player.getId());
142             target.setAbilityController(source.getControllerId());
143             if (!player.chooseTarget(outcome, target, source, game)) { continue; }
144 
145             Player chosenOpponent = game.getPlayer(target.getFirstTarget());
146             if (chosenOpponent == null) { continue; }
147 
148             List<Permanent> firstPile = playerPiles.getValue().get(0);
149             List<Permanent> secondPile = playerPiles.getValue().get(1);
150             game.informPlayers(player.getLogName() + " chose " + chosenOpponent.getLogName() + " to choose their pile");
151             if (chosenOpponent.choosePile(outcome, "Piles of " + player.getName(), firstPile, secondPile, game)) {
152                 List<List<Permanent>> lists = playerPiles.getValue();
153                 lists.clear();
154                 lists.add(firstPile);
155                 lists.add(secondPile);
156                 game.informPlayers(player.getLogName() + " will have their first pile destroyed");
157             } else {
158                 List<List<Permanent>> lists = playerPiles.getValue();
159                 lists.clear();
160                 lists.add(secondPile);
161                 lists.add(firstPile);
162                 game.informPlayers(player.getLogName() + " will have their second pile destroyed");
163             }
164         }
165         // Destroy all lands in the chosen piles. Tap all lands in the other piles
166         for (Map.Entry<UUID, List<List<Permanent>>> playerPiles : playerPermanents.entrySet()) {
167             Player player = game.getPlayer(playerPiles.getKey());
168             if (player == null) { continue; }
169 
170             List<Permanent> pileToSac = playerPiles.getValue().get(0);
171             List<Permanent> pileToTap = playerPiles.getValue().get(1);
172             for (Permanent permanent : pileToSac) {
173                 if (permanent != null) {
174                     permanent.destroy(source, game, false);
175                 }
176             }
177             for (Permanent permanent : pileToTap) {
178                 if (permanent != null) {
179                     permanent.tap(source, game);
180                 }
181             }
182         }
183         return true;
184     }
185 
186     private UUID getNextPlayerInDirection(boolean left, PlayerList playerList) {
187         UUID nextPlayerId;
188         if (left) {
189             nextPlayerId = playerList.getNext();
190         } else {
191             nextPlayerId = playerList.getPrevious();
192         }
193         return nextPlayerId;
194     }

33  public final class BhaalLordOfMurder extends CardImpl {
34  
35      private static final FilterPermanent filter
36              = new FilterControlledCreaturePermanent("another nontoken creature you control");
37  
38      static {
39          filter.add(AnotherPredicate.instance);
40          filter.add(TokenPredicate.FALSE);
41      }
42  
43      public BhaalLordOfMurder(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{B}{R}{G}");
45  
46          this.addSuperType(SuperType.LEGENDARY);
47          this.subtype.add(SubType.GOD);
48          this.power = new MageInt(4);
49          this.toughness = new MageInt(4);
50  
51          // As long as your life total is less than or equal to half your starting life total, Bhaal, Lord of Murder has indestructible.
52          this.addAbility(new SimpleStaticAbility(new ConditionalContinuousEffect(
53                  new GainAbilitySourceEffect(IndestructibleAbility.getInstance()), BhaalLordOfMurderCondition.instance,
54                  "as long as your life total is less than or equal to half your starting life total, {this} has indestructible"
55          )));
56  
57          // Whenever another nontoken creature you control dies, put a +1/+1 counter on target creature and goad it.
58          Ability ability = new DiesCreatureTriggeredAbility(
59                  new AddCountersTargetEffect(CounterType.P1P1.createInstance()), false, filter
60          );
61          ability.addEffect(new GoadTargetEffect().setText("and goad it"));
62          ability.addTarget(new TargetCreaturePermanent());
63          this.addAbility(ability);
64      }
65  
66      private BhaalLordOfMurder(final BhaalLordOfMurder card) {
67          super(card);
68      }
69  
70      @Override
71      public BhaalLordOfMurder copy() {
72          return new BhaalLordOfMurder(this);
73      }
74  }
75  
76  enum BhaalLordOfMurderCondition implements Condition {
77      instance;
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          return Optional.ofNullable(game.getPlayer(source.getControllerId()))
82                  .map(Player::getLife)
83                  .map(x -> 2 * x <= game.getStartingLife())
84                  .orElse(false);
85      }
86  }

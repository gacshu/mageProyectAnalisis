24   * @author TheElk801

26  public final class BurnDownTheHouse extends CardImpl {
27  
28      private static final FilterPermanent filter
29              = new FilterCreatureOrPlaneswalkerPermanent("creature and each planeswalker");
30  
31      public BurnDownTheHouse(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{R}{R}");
33  
34          // Choose one —
35          // • Burn Down the House deals 5 damage to each creature and each planeswalker.
36          this.getSpellAbility().addEffect(new DamageAllEffect(5, filter));
37  
38          // • Create three 1/1 red Devil creature tokens with "When this creature dies, it deals 1 damage to any target." They gain haste until end of turn.
39          this.getSpellAbility().addMode(new Mode(new BurnDownTheHouseEffect()));
40      }
41  
42      private BurnDownTheHouse(final BurnDownTheHouse card) {
43          super(card);
44      }
45  
46      @Override
47      public BurnDownTheHouse copy() {
48          return new BurnDownTheHouse(this);
49      }
50  }
51  
52  class BurnDownTheHouseEffect extends OneShotEffect {
53  
54      BurnDownTheHouseEffect() {
55          super(Outcome.Benefit);
56          staticText = "create three 1/1 red Devil creature tokens with " +
57                  "\"When this creature dies, it deals 1 damage to any target.\" They gain haste until end of turn";
58      }
59  
60      private BurnDownTheHouseEffect(final BurnDownTheHouseEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public BurnDownTheHouseEffect copy() {
66          return new BurnDownTheHouseEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Token token = new DevilToken();
72          token.putOntoBattlefield(3, game, source, source.getControllerId());
73          game.addEffect(new GainAbilityTargetEffect(
74                  HasteAbility.getInstance(), Duration.EndOfTurn
75          ).setTargetPointer(new FixedTargets(token, game)), source);
76          return true;
77      }
78  }

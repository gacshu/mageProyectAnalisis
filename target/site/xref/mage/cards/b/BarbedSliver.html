20   *
21   * @author Loki

23  public final class BarbedSliver extends CardImpl {
24  
25      public BarbedSliver(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}");
27          this.subtype.add(SubType.SLIVER);
28  
29          this.power = new MageInt(2);
30          this.toughness = new MageInt(2);
31          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new GainAbilityAllEffect(new SimpleActivatedAbility(Zone.BATTLEFIELD, new BoostSourceEffect(1, 0, Duration.EndOfTurn).setText("this creature gets +1/+0 until end of turn"),
32                  new GenericManaCost(2)), Duration.WhileOnBattlefield, StaticFilters.FILTER_PERMANENT_ALL_SLIVERS, false)));
33      }
34  
35      private BarbedSliver(final BarbedSliver card) {
36          super(card);
37      }
38  
39      @Override
40      public BarbedSliver copy() {
41          return new BarbedSliver(this);
42      }
43  }

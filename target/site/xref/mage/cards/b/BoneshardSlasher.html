20  /**
21   *
22   * @author LoneFox
23   */
24  public final class BoneshardSlasher extends CardImpl {
25  
26      public BoneshardSlasher(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{B}");
28          this.subtype.add(SubType.HORROR);
29          this.power = new MageInt(1);
30          this.toughness = new MageInt(1);
31  
32          // Flying
33          this.addAbility(FlyingAbility.getInstance());
34          // Threshold - As long as seven or more cards are in your graveyard, Boneshard Slasher gets +2/+2 and has "When Boneshard Slasher becomes the target of a spell or ability, sacrifice it."
35          Ability ability = new SimpleStaticAbility(Zone.BATTLEFIELD, new ConditionalContinuousEffect(
36              new BoostSourceEffect(2, 2, Duration.WhileOnBattlefield), new CardsInControllerGraveyardCondition(7),
37              "As long as seven or more cards are in your graveyard, {this} gets +2/+2"));
38          Effect effect = new ConditionalContinuousEffect(new GainAbilitySourceEffect(new BecomesTargetTriggeredAbility(new SacrificeSourceEffect())),
39              new CardsInControllerGraveyardCondition(7), "and has \"When {this} becomes the target of a spell or ability, sacrifice it.\"");
40          ability.addEffect(effect);
41          ability.setAbilityWord(AbilityWord.THRESHOLD);
42          this.addAbility(ability);
43      }
44  
45      private BoneshardSlasher(final BoneshardSlasher card) {
46          super(card);
47      }
48  
49      @Override
50      public BoneshardSlasher copy() {
51          return new BoneshardSlasher(this);
52      }
53  }

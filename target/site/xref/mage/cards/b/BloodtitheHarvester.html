31  public final class BloodtitheHarvester extends CardImpl {
32  
33      private static final FilterPermanent filter = new FilterControlledPermanent(SubType.BLOOD);
34  
35      static {
36          filter.add(TokenPredicate.TRUE);
37      }
38  
39      private static final DynamicValue xValue = new PermanentsOnBattlefieldCount(filter, -2);
40      private static final Hint hint = new ValueHint(
41              "Blood tokens you control", new PermanentsOnBattlefieldCount(filter)
42      );
43  
44      public BloodtitheHarvester(UUID ownerId, CardSetInfo setInfo) {
45          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{B}{R}");
46  
47          this.subtype.add(SubType.VAMPIRE);
48          this.power = new MageInt(3);
49          this.toughness = new MageInt(2);
50  
51          // When Bloodtithe Harvester enters the battlefield, create a Blood token.
52          this.addAbility(new EntersBattlefieldTriggeredAbility(new CreateTokenEffect(new BloodToken())));
53  
54          // {T}, Sacrifice Bloodtithe Harvester: Target creature gets -X/-X until end of turn, where X is twice the number of Blood tokens you control. Activate only as a sorcery.
55          Ability ability = new ActivateAsSorceryActivatedAbility(new BoostTargetEffect(
56                  xValue, xValue, Duration.EndOfTurn
57          ).setText("target creature gets -X/-X until end of turn, where X is twice the number of Blood tokens you control"), new TapSourceCost());
58          ability.addCost(new SacrificeSourceCost());
59          ability.addTarget(new TargetCreaturePermanent());
60          this.addAbility(ability.addHint(hint));
61      }
62  
63      private BloodtitheHarvester(final BloodtitheHarvester card) {
64          super(card);
65      }
66  
67      @Override
68      public BloodtitheHarvester copy() {
69          return new BloodtitheHarvester(this);
70      }
71  }

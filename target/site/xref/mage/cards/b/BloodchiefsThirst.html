21  /**
22   * @author TheElk801

24  public final class BloodchiefsThirst extends CardImpl {
25  
26      public BloodchiefsThirst(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{B}");
28  
29          // Kicker {2}{B}
30          this.addAbility(new KickerAbility("{2}{B}"));
31  
32          // Destroy target creature or planeswalker with converted mana cost 2 or less. If this spell was kicked, instead destroy target creature or planeswalker.
33          this.getSpellAbility().addEffect(new DestroyTargetEffect(
34                  "Destroy target creature or planeswalker with mana value 2 or less. " +
35                          "If this spell was kicked, instead destroy target creature or planeswalker."
36          ));
37          this.getSpellAbility().setTargetAdjuster(BloodchiefsThirstAdjuster.instance);
38      }
39  
40      private BloodchiefsThirst(final BloodchiefsThirst card) {
41          super(card);
42      }
43  
44      @Override
45      public BloodchiefsThirst copy() {
46          return new BloodchiefsThirst(this);
47      }
48  }
49  
50  enum BloodchiefsThirstAdjuster implements TargetAdjuster {
51      instance;
52  
53      private static final FilterPermanent filter = new FilterCreatureOrPlaneswalkerPermanent(
54              "creature or planeswalker with mana value 2 or less"
55      );
56  
57      static {
58          filter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 3));
59      }
60  
61      @Override
62      public void adjustTargets(Ability ability, Game game) {
63          ability.getTargets().clear();
64          if (KickedCondition.ONCE.apply(game, ability)) {
65              ability.addTarget(new TargetCreatureOrPlaneswalker());
66          } else {
67              ability.addTarget(new TargetPermanent(filter));
68          }
69      }
70  }

4   import java.util.UUID;
5   import mage.abilities.costs.common.PayLifeCost;
6   import mage.abilities.dynamicvalue.DynamicValue;
7   import mage.abilities.dynamicvalue.common.ManacostVariableValue;
8   import mage.abilities.effects.common.LoseLifeOpponentsEffect;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  
13  /**
14   * @author magenoxx
15   */
16  public final class BondOfAgony extends CardImpl {
17  
18      public BondOfAgony(UUID ownerId, CardSetInfo setInfo) {
19          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{X}{B}");
20  
21          DynamicValue xValue = ManacostVariableValue.REGULAR;
22  
23          // As an additional cost to cast Bond of Agony, pay X life.
24          // magenoxx: here we don't use PayVariableLifeCost as {X} shouldn't actually be announced
25          this.getSpellAbility().addCost(new PayLifeCost(xValue, "X life"));
26  
27          // Each other player loses X life.
28          this.getSpellAbility().addEffect(new LoseLifeOpponentsEffect(xValue));
29      }
30  
31      private BondOfAgony(final BondOfAgony card) {
32          super(card);
33      }
34  
35      @Override
36      public BondOfAgony copy() {
37          return new BondOfAgony(this);
38      }
39  }

27  public final class BattleOfFrostAndFire extends CardImpl {
28  
29      private static final FilterPermanent filter
30              = new FilterPermanent("non-Giant creature and each planeswalker");
31  
32      static {
33          filter.add(Predicates.or(
34                  Predicates.and(CardType.CREATURE.getPredicate(), Predicates.not(SubType.GIANT.getPredicate())),
35                  CardType.PLANESWALKER.getPredicate()
36          ));
37      }
38  
39      public BattleOfFrostAndFire(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{U}{R}");
41  
42          this.subtype.add(SubType.SAGA);
43  
44          // (As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)
45          SagaAbility sagaAbility = new SagaAbility(this);
46  
47          // I — Battle of Frost and Fire deals 4 damage to each non-Giant creature and each planeswalker.
48          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, new DamageAllEffect(4, filter));
49  
50          // II — Scry 3.
51          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_II, new ScryEffect(3));
52  
53          // III — Whenever you cast a spell with converted mana cost 5 or greater this turn, draw two cards, then discard a card.
54          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new CreateDelayedTriggeredAbilityEffect(
55                  new BattleOfFrostAndFireTriggeredAbility(), false
56          ));
57          this.addAbility(sagaAbility);
58      }
59  
60      private BattleOfFrostAndFire(final BattleOfFrostAndFire card) {
61          super(card);
62      }
63  
64      @Override
65      public BattleOfFrostAndFire copy() {
66          return new BattleOfFrostAndFire(this);
67      }
68  }
69  
70  class BattleOfFrostAndFireTriggeredAbility extends DelayedTriggeredAbility {
71  
72      public BattleOfFrostAndFireTriggeredAbility() {
73          super(new DrawDiscardControllerEffect(2, 1), Duration.EndOfTurn, false);
74          setTriggerPhrase("Whenever you cast a spell with mana value 5 or greater this turn, ");
75      }
76  
77      private BattleOfFrostAndFireTriggeredAbility(BattleOfFrostAndFireTriggeredAbility ability) {
78          super(ability);
79      }
80  
81      @Override
82      public BattleOfFrostAndFireTriggeredAbility copy() {
83          return new BattleOfFrostAndFireTriggeredAbility(this);
84      }
85  
86      @Override
87      public boolean checkEventType(GameEvent event, Game game) {
88          return event.getType() == GameEvent.EventType.SPELL_CAST;
89      }
90  
91      @Override
92      public boolean checkTrigger(GameEvent event, Game game) {
93          if (event.getPlayerId().equals(this.getControllerId())) {
94              Spell spell = game.getStack().getSpell(event.getTargetId());
95              return spell != null && spell.getManaValue() >= 5;
96          }
97          return false;
98      }
99  }

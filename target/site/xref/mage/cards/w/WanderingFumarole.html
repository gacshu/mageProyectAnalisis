26  public final class WanderingFumarole extends CardImpl {
27  
28      public WanderingFumarole(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.LAND},"");
30  
31          // Wandering Fumarole enters the battlefield tapped.
32          this.addAbility(new EntersBattlefieldTappedAbility());
33          
34          // {T}: Add {U} or {R}.
35          this.addAbility(new BlueManaAbility());
36          this.addAbility(new RedManaAbility());
37          
38          // {2}{U}{R}: Until end of turn, Wandering Fumarole becomes a 1/4 blue and red Elemental creature with 
39          // "0: Switch this creature's power and toughness until end of turn." It's still a land.
40          Effect effect = new BecomesCreatureSourceEffect(new WanderingFumaroleToken(), "land", Duration.EndOfTurn);
41          effect.setText("{this} becomes a 1/4 blue and red Elemental creature with \"0: Switch this creature's power and toughness until end of turn.\" It's still a land");
42          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new ManaCostsImpl<>("{2}{U}{R}")));
43      }
44  
45      private WanderingFumarole(final WanderingFumarole card) {
46          super(card);
47      }
48  
49      @Override
50      public WanderingFumarole copy() {
51          return new WanderingFumarole(this);
52      }
53  }
54  
55  class WanderingFumaroleToken extends TokenImpl {
56  
57      public WanderingFumaroleToken() {
58          super("", "1/4 blue and red Elemental creature with \"0: Switch this creature's power and toughness until end of turn.\"");
59          cardType.add(CardType.CREATURE);
60          subtype.add(SubType.ELEMENTAL);
61          color.setRed(true);
62          color.setBlue(true);
63          power = new MageInt(1);
64          toughness = new MageInt(4);
65          addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new SwitchPowerToughnessSourceEffect(Duration.EndOfTurn), new ManaCostsImpl<>("{0}")));
66      }
67      public WanderingFumaroleToken(final WanderingFumaroleToken token) {
68          super(token);
69      }
70  
71      public WanderingFumaroleToken copy() {
72          return new WanderingFumaroleToken(this);
73      }
74  }

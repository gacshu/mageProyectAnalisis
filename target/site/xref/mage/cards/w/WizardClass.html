21  import java.util.UUID;

23  /**
24   * @author TheElk801
25   */
26  public final class WizardClass extends CardImpl {
27  
28      public WizardClass(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{U}");
30  
31          this.subtype.add(SubType.CLASS);
32  
33          // (Gain the next level as a sorcery to add its ability.)
34          this.addAbility(new ClassReminderAbility());
35  
36          // You have no maximum hand size.
37          this.addAbility(new SimpleStaticAbility(new MaximumHandSizeControllerEffect(
38                  Integer.MAX_VALUE, Duration.WhileOnBattlefield,
39                  MaximumHandSizeControllerEffect.HandSizeModification.SET
40          )));
41  
42          // {2}{U}: Level 2
43          this.addAbility(new ClassLevelAbility(2, "{2}{U}"));
44  
45          // When this Class becomes level 2, draw two cards.
46          this.addAbility(new BecomesClassLevelTriggeredAbility(new DrawCardSourceControllerEffect(2), 2));
47  
48          // {4}{U}: Level 3
49          this.addAbility(new ClassLevelAbility(3, "{4}{U}"));
50  
51          // Whenever you draw a card, put a +1/+1 counter on target creature you control.
52          Ability ability = new DrawCardControllerTriggeredAbility(
53                  new AddCountersTargetEffect(CounterType.P1P1.createInstance()), false
54          );
55          ability.addTarget(new TargetControlledCreaturePermanent());
56          this.addAbility(new SimpleStaticAbility(new GainClassAbilitySourceEffect(ability, 3)));
57      }
58  
59      private WizardClass(final WizardClass card) {
60          super(card);
61      }
62  
63      @Override
64      public WizardClass copy() {
65          return new WizardClass(this);
66      }
67  }

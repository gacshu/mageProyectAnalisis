32  public final class WeightOfConscience extends CardImpl {
33  
34      public WeightOfConscience(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}");
36          this.subtype.add(SubType.AURA);
37  
38          // Enchant creature
39          TargetPermanent auraTarget = new TargetCreaturePermanent();
40          this.getSpellAbility().addTarget(auraTarget);
41          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
42          Ability ability = new EnchantAbility(auraTarget);
43          this.addAbility(ability);
44  
45          // Enchanted creature can't attack.
46          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new CantAttackAttachedEffect(AttachmentType.AURA)));
47  
48          // Tap two untapped creatures you control that share a creature type: Exile enchanted creature.
49          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new ExileAttachedEffect(), new TapTargetCost(new WeightOfConscienceTarget())));
50      }
51  
52      private WeightOfConscience(final WeightOfConscience card) {
53          super(card);
54      }
55  
56      @Override
57      public WeightOfConscience copy() {
58          return new WeightOfConscience(this);
59      }
60  }

62  class WeightOfConscienceTarget extends TargetControlledCreaturePermanent {
63  
64      private static final FilterControlledCreaturePermanent filterUntapped = new FilterControlledCreaturePermanent("untapped creatures you control that share a creature type");
65  
66      static {
67          filterUntapped.add(TappedPredicate.UNTAPPED);
68          filterUntapped.add(WeightOfConsciencePredicate.instance);
69      }
70  
71      WeightOfConscienceTarget() {
72          super(2, 2, filterUntapped, true);
73      }
74  
75      private WeightOfConscienceTarget(final WeightOfConscienceTarget target) {
76          super(target);
77      }
78  
79      @Override
80      public Set<UUID> possibleTargets(UUID sourceControllerId, Ability source, Game game) {
81          Player player = game.getPlayer(sourceControllerId);
82          Set<UUID> possibleTargets = new HashSet<>(0);
83          if (player == null) {
84              return possibleTargets;
85          }
86          // Choosing first target
87          if (this.getTargets().isEmpty()) {
88              List<Permanent> permanentList = game.getBattlefield().getActivePermanents(filterUntapped, sourceControllerId, source, game);
89              if (permanentList.size() < 2) {
90                  return possibleTargets;
91              }
92              for (Permanent permanent : permanentList) {
93                  if (permanent.isAllCreatureTypes(game)) {
94                      possibleTargets.add(permanent.getId());
95                      continue;
96                  }
97                  FilterPermanent filter = filterUntapped.copy();
98                  filter.add(new SharesCreatureTypePredicate(permanent));
99                  if (game.getBattlefield().count(filter, sourceControllerId, source, game) > 1) {
100                     possibleTargets.add(permanent.getId());
101                 }
102             }
103         } // Choosing second target
104         else {
105             Permanent firstTargetCreature = game.getPermanent(this.getFirstTarget());
106             if (firstTargetCreature == null) {
107                 return possibleTargets;
108             }
109             FilterPermanent filter = filterUntapped.copy();
110             filter.add(new SharesCreatureTypePredicate(firstTargetCreature));
111             for (Permanent permanent : game.getBattlefield().getActivePermanents(filterUntapped, sourceControllerId, source, game)) {
112                 if (permanent != null) {
113                     possibleTargets.add(permanent.getId());
114                 }
115             }
116         }
117         return possibleTargets;
118     }
119 
120     @Override
121     public boolean canChoose(UUID sourceControllerId, Ability source, Game game) {
122         for (Permanent permanent1 : game.getBattlefield().getActivePermanents(filterUntapped, sourceControllerId, source, game)) {
123             for (Permanent permanent2 : game.getBattlefield().getActivePermanents(filterUntapped, sourceControllerId, source, game)) {
124                 if (!Objects.equals(permanent1, permanent2) && permanent1.shareCreatureTypes(game, permanent2)) {
125                     return true;
126                 }
127             }
128         }
129         return false;
130     }
131 
132     @Override
133     public boolean canTarget(UUID id, Ability source, Game game) {
134         if (!super.canTarget(id, game)) {
135             return false;
136         }
137         Permanent targetPermanent = game.getPermanent(id);
138         if (targetPermanent == null) {
139             return false;
140         }
141         if (this.getTargets().isEmpty()) {
142             List<Permanent> permanentList = game.getBattlefield().getActivePermanents(filterUntapped, source.getControllerId(), source, game);
143             if (permanentList.size() < 2) {
144                 return false;
145             }
146             for (Permanent permanent : permanentList) {
147                 if (permanent.isAllCreatureTypes(game)) {
148                     return true;
149                 }
150                 FilterPermanent filter = filterUntapped.copy();
151                 filter.add(new SharesCreatureTypePredicate(permanent));
152                 if (game.getBattlefield().count(filter, source.getControllerId(), source, game) > 1) {
153                     return true;
154                 }
155             }
156         } else {
157             Permanent firstTarget = game.getPermanent(this.getTargets().get(0));
158             return firstTarget != null && firstTarget.shareCreatureTypes(game, targetPermanent);
159         }
160         return false;
161     }
162 
163     @Override
164     public WeightOfConscienceTarget copy() {
165         return new WeightOfConscienceTarget(this);
166     }
167 }
168 
169 enum WeightOfConsciencePredicate implements Predicate<MageObject> {
170     instance;
171 
172     @Override
173     public boolean apply(MageObject input, Game game) {
174         return input.isAllCreatureTypes(game)
175                 || input
176                 .getSubtype(game)
177                 .stream()
178                 .map(SubType::getSubTypeSet)
179                 .anyMatch(SubTypeSet.CreatureType::equals);
180     }
181 }

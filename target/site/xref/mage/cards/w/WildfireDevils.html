25  public final class WildfireDevils extends CardImpl {
26  
27      public WildfireDevils(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}");
29  
30          this.subtype.add(SubType.DEVIL);
31          this.power = new MageInt(4);
32          this.toughness = new MageInt(2);
33  
34          // When Wildfire Devils enters the battlefield and at the beginning of your upkeep, choose a player at random. That player exiles an instant or sorcery card from their graveyard. Copy that card. You may cast the copy without paying its mana cost.
35          this.addAbility(new OrTriggeredAbility(
36                  Zone.BATTLEFIELD, new WildfireDevilsEffect(), false,
37                  "When {this} enters the battlefield and at the beginning of your upkeep, ",
38                  new EntersBattlefieldTriggeredAbility(null, false),
39                  new BeginningOfUpkeepTriggeredAbility(null, TargetController.YOU, false)
40          ));
41      }
42  
43      private WildfireDevils(final WildfireDevils card) {
44          super(card);
45      }
46  
47      @Override
48      public WildfireDevils copy() {
49          return new WildfireDevils(this);
50      }
51  }

53  class WildfireDevilsEffect extends OneShotEffect {
54  
55      WildfireDevilsEffect() {
56          super(Outcome.Neutral);
57          staticText = "choose a player at random. That player exiles an instant or sorcery card from their graveyard. "
58                  + "Copy that card. You may cast the copy without paying its mana cost.";
59      }
60  
61      private WildfireDevilsEffect(final WildfireDevilsEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public WildfireDevilsEffect copy() {
67          return new WildfireDevilsEffect(this);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          Player controller = game.getPlayer(source.getControllerId());
73          if (controller == null) {
74              return false;
75          }
76          PlayerList players = game.getState().getPlayersInRange(controller.getId(), game);
77          if (players == null) {
78              return false;
79          }
80          Player randomPlayer = game.getPlayer(players.get(RandomUtil.nextInt(players.size())));
81          if (randomPlayer == null) {
82              return false;
83          }
84          game.informPlayers("The chosen random player is " + randomPlayer.getLogName());
85          if (randomPlayer.getGraveyard().getCards(game).stream().noneMatch(card1 -> card1.isInstantOrSorcery(game))) {
86              return false;
87          }
88          TargetCardInGraveyard targetCard = new TargetCardInGraveyard(StaticFilters.FILTER_CARD_INSTANT_OR_SORCERY);
89          targetCard.setNotTarget(true);
90          if (!randomPlayer.choose(Outcome.Discard, randomPlayer.getGraveyard(), targetCard, game)) {
91              return false;
92          }
93          Card card = game.getCard(targetCard.getFirstTarget());
94          if (card == null) {
95              return false;
96          }
97          randomPlayer.moveCards(card, Zone.EXILED, source, game);
98          if (game.getState().getZone(card.getId()) != Zone.EXILED) {
99              return false;
100         }
101         Card copiedCard = game.copyCard(card, source, controller.getId());
102         if (copiedCard == null) {
103             return false;
104         }
105         randomPlayer.moveCards(copiedCard, Zone.EXILED, source, game);
106         if (!controller.chooseUse(outcome, "Cast the copy of the exiled card?", source, game)) {
107             return false;
108         }
109         game.getState().setValue("PlayFromNotOwnHandZone" + copiedCard.getId(), Boolean.TRUE);
110         Boolean cardWasCast = controller.cast(controller.chooseAbilityForCast(copiedCard, game, true), game, true,
111                 new ApprovingObject(source, game));
112         game.getState().setValue("PlayFromNotOwnHandZone" + copiedCard.getId(), null);
113         return cardWasCast;
114     }

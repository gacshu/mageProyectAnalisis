33  public final class WallOfStolenIdentity extends CardImpl {
34  
35      final static private String rule = "You may have {this} enter the battlefield as a copy of any "
36              + "creature on the battlefield, except it's a Wall in addition to its other types and it has defender. "
37              + "When you do, tap the copied creature and it doesn't untap during its "
38              + "controller's untap step for as long as you control {this}";
39  
40      public WallOfStolenIdentity(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
42  
43          this.subtype.add(SubType.SHAPESHIFTER);
44          this.subtype.add(SubType.WALL);
45          this.power = new MageInt(0);
46          this.toughness = new MageInt(0);
47  
48          // You may have Wall of Stolen Identity enter the battlefield as a copy of any creature on the battlefield, except it's a wall in addition to its other types and it has defender. When you do, tap the copied creature and it doesn't untap during its controller's untap step for as long as you control Wall of Stolen Identity.
49          Ability ability = new SimpleStaticAbility(new EntersBattlefieldEffect(
50                  new WallOfStolenIdentityCopyEffect(), rule, true
51          ));
52          this.addAbility(ability);
53      }
54  
55      private WallOfStolenIdentity(final WallOfStolenIdentity card) {
56          super(card);
57      }
58  
59      @Override
60      public WallOfStolenIdentity copy() {
61          return new WallOfStolenIdentity(this);
62      }
63  }

65  class WallOfStolenIdentityCopyEffect extends OneShotEffect {
66  
67      private static final String rule2 = "When you do, .";
68  
69      public WallOfStolenIdentityCopyEffect() {
70          super(Outcome.Copy);
71          staticText = rule2;
72      }
73  
74      public WallOfStolenIdentityCopyEffect(final WallOfStolenIdentityCopyEffect effect) {
75          super(effect);
76      }
77  
78      @Override
79      public boolean apply(Game game, Ability source) {
80          Player controller = game.getPlayer(source.getControllerId());
81          Permanent permanent = game.getPermanent(source.getSourceId());
82          if (permanent == null) {
83              permanent = game.getPermanentEntering(source.getSourceId());
84          }
85          final Permanent sourcePermanent = permanent;
86          if (controller == null
87                  || sourcePermanent == null) {
88              return false;
89          }
90          Target target = new TargetPermanent(new FilterCreaturePermanent("target creature (you copy from)"));
91          target.setRequired(true);
92          if (source instanceof SimpleStaticAbility) {
93              target = new TargetPermanent(new FilterCreaturePermanent("creature (you copy from)"));
94              target.setRequired(false);
95              target.setNotTarget(true);
96          }
97          if (!target.canChoose(source.getControllerId(), source, game)) {
98              return false;
99          }
100         controller.choose(Outcome.Copy, target, source, game);
101         Permanent copyFromPermanent = game.getPermanent(target.getFirstTarget());
102         if (copyFromPermanent == null) {
103             return false;
104         }
105         game.copyPermanent(copyFromPermanent, sourcePermanent.getId(), source, new CopyApplier() {
106             @Override
107             public boolean apply(Game game, MageObject blueprint, Ability source, UUID copyToObjectId) {
108                 blueprint.addSubType(SubType.WALL);
109                 blueprint.getAbilities().add(DefenderAbility.getInstance());
110                 return true;
111             }
112         });
113 
114         ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
115                 new TapTargetEffect(), false, "tap the copied creature " +
116                 "and it doesn't untap during its controller's untap step for as long as you control {this}"
117         );
118         ability.addEffect(new DontUntapInControllersUntapStepTargetEffect(Duration.WhileControlled));
119         ability.getEffects().setTargetPointer(new FixedTarget(copyFromPermanent, game));
120         game.fireReflexiveTriggeredAbility(ability, source);
121         return true;
122     }
123 
124     @Override
125     public WallOfStolenIdentityCopyEffect copy() {
126         return new WallOfStolenIdentityCopyEffect(this);
127     }
128 }

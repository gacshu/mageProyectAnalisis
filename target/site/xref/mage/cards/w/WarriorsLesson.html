20  public final class WarriorsLesson extends CardImpl {
21  
22      public WarriorsLesson(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{G}");
24  
25  
26          // Until end of turn, up to two target creatures you control each gain "Whenever this creature deals combat damage to a player, draw a card."
27          Ability ability = new DealsCombatDamageToAPlayerTriggeredAbility(new DrawCardSourceControllerEffect(1),false);
28          Effect effect = new GainAbilityTargetEffect(ability, Duration.EndOfTurn);
29          effect.setText("Until end of turn, up to two target creatures you control each gain \"Whenever this creature deals combat damage to a player, draw a card.\"");
30          this.getSpellAbility().addEffect(effect);
31          this.getSpellAbility().addTarget(new TargetCreaturePermanent(0,2));
32      }
33  
34      private WarriorsLesson(final WarriorsLesson card) {
35          super(card);
36      }
37

39      public WarriorsLesson copy() {

2   package mage.cards.w;

4   import java.util.UUID;
5   import mage.abilities.Mode;
6   import mage.abilities.effects.Effect;
7   import mage.abilities.effects.common.CounterTargetEffect;
8   import mage.abilities.effects.common.CreateTokenEffect;
9   import mage.abilities.effects.common.ExileTargetEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.ComparisonType;
14  import mage.filter.FilterSpell;
15  import mage.filter.common.FilterCreaturePermanent;
16  import mage.filter.predicate.Predicates;
17  import mage.filter.predicate.mageobject.PowerPredicate;
18  import mage.filter.predicate.mageobject.ToughnessPredicate;
19  import mage.game.permanent.token.EldraziScionToken;
20  import mage.target.TargetSpell;
21  import mage.target.common.TargetCreaturePermanent;
22  
23  /**
24   *
25   * @author fireshoes
26   */
27  public final class WarpingWail extends CardImpl {
28  
29      private static final FilterCreaturePermanent filterCreature = new FilterCreaturePermanent("creature with power or toughness 1 or less");
30      private static final FilterSpell filterSorcery = new FilterSpell("sorcery spell");
31  
32      static {
33          filterCreature.add(Predicates.or(
34                  new PowerPredicate(ComparisonType.FEWER_THAN, 2),
35                  new ToughnessPredicate(ComparisonType.FEWER_THAN, 2)));
36          filterSorcery.add(CardType.SORCERY.getPredicate());
37      }
38  
39      public WarpingWail(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{1}{C}");
41  
42          // Choose one &mdash; Exile target creature with power or toughness 1 or less.
43          Effect effect = new ExileTargetEffect();
44          effect.setText("Exile target creature with power or toughness 1 or less.");
45          this.getSpellAbility().addEffect(effect);
46          this.getSpellAbility().addTarget(new TargetCreaturePermanent(filterCreature));
47  
48          // Counter target sorcery spell.
49          Mode mode = new Mode(new CounterTargetEffect());
50          mode.addTarget(new TargetSpell(filterSorcery));
51          this.getSpellAbility().addMode(mode);
52  
53          // Create a 1/1 colorless Eldrazi Scion creature token. It has "Sacrifice this creature: Add {C}."
54          effect = new CreateTokenEffect(new EldraziScionToken());
55          effect.setText("Create a 1/1 colorless Eldrazi Scion creature token. It has \"Sacrifice this creature: Add {C}.\"");
56          this.getSpellAbility().addMode(new Mode(effect));
57      }
58  
59      private WarpingWail(final WarpingWail card) {
60          super(card);
61      }
62  
63      @Override
64      public WarpingWail copy() {
65          return new WarpingWail(this);
66      }
67  }

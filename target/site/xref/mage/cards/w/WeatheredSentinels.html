31  public class WeatheredSentinels extends CardImpl {
32  
33      public WeatheredSentinels(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{3}");
35  
36          addSubType(SubType.WALL);
37          this.power = new MageInt(2);
38          this.toughness = new MageInt(5);
39  
40          // Defender, vigilance, reach, trample
41          this.addAbility(DefenderAbility.getInstance());
42          this.addAbility(VigilanceAbility.getInstance());
43          this.addAbility(ReachAbility.getInstance());
44          this.addAbility(TrampleAbility.getInstance());
45  
46          // Weathered Sentinels can attack players who attacked you during their last turn as though it didn't have defender.
47          this.addAbility(new SimpleStaticAbility(
48                  new ConditionalAsThoughEffect(
49                          new CanAttackAsThoughItDidntHaveDefenderSourceEffect(Duration.WhileOnBattlefield),
50                          WeatheredSentinelsCanAttackSomeoneCondition.instance)
51                          .setText("Weathered Sentinels can attack players who attacked you during their last turn as though it didn't have defender.")),
52                  new WeatheredSentinelsLastTurnAttackersWatcher()
53          );
54  
55          // Whenever Weathered Sentinels attacks, it gets +3/+3 and gains indestructible until end of turn.
56          Ability ability = new AttacksTriggeredAbility(
57                  new BoostSourceEffect(3, 3, Duration.EndOfTurn).setText("it gets +3/+3")
58          );
59          ability.addEffect(new GainAbilitySourceEffect(
60                  IndestructibleAbility.getInstance(), Duration.EndOfTurn
61                  ).concatBy("and").setText("gains indestructible until end of turn")
62          );
63          this.addAbility(ability);
64  
65          // Ability to limit who Weathered Sentinels can attack
66          this.addAbility(new SimpleStaticAbility(
67                  new WeatheredSentinelsAttackerReplacementEffect()
68                  ).addHint(WeatheredSentinelsPlayersWhoAttackedYouLastTurn.instance)
69          );
70      }

73          super(card);
74      }

78          return new WeatheredSentinels(this);

82  class WeatheredSentinelsAttackerReplacementEffect extends ReplacementEffectImpl {
83  
84      WeatheredSentinelsAttackerReplacementEffect() {
85          super(Duration.WhileOnBattlefield, Outcome.Benefit);
86      }
87  
88      WeatheredSentinelsAttackerReplacementEffect(final WeatheredSentinelsAttackerReplacementEffect effect) {
89          super(effect);
90      }
91  
92      @Override
93      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
94          return true;
95      }
96  
97      @Override
98      public boolean checksEventType(GameEvent event, Game game) {
99          return event.getType() == GameEvent.EventType.DECLARE_ATTACKER;
100     }
101 
102     @Override
103     public boolean applies(GameEvent event, Ability source, Game game) {
104         Player controller = game.getPlayer(source.getControllerId());
105         Player attacker = game.getPlayer(event.getPlayerId());
106         if (controller == null || attacker == null) {
107             return false;
108         }
109 
110         if (!attacker.equals(controller)) {
111             return false;
112         }
113 
114         Player defender;
115         if (game.getPlayer(event.getTargetId()) != null) {
116             defender = game.getPlayer(event.getTargetId());
117         } else {
118             Permanent planeswalker = game.getPermanent(event.getTargetId());
119             defender = (planeswalker == null) ? null : game.getPlayer(planeswalker.getControllerId());
120         }
121         if (defender == null) {
122             return false;
123         }
124 
125         WeatheredSentinelsLastTurnAttackersWatcher watcher = game.getState().getWatcher(WeatheredSentinelsLastTurnAttackersWatcher.class);
126         if (watcher == null) {
127             return false;
128         }
129 
130         // Attacker and defender are supposed to be flipped here
131         return watcher.checkPlayer(defender.getId(), attacker.getId());
132     }
133 
134     @Override
135     public ContinuousEffect copy() {
136         return new WeatheredSentinelsAttackerReplacementEffect(this);
137     }

140 enum WeatheredSentinelsPlayersWhoAttackedYouLastTurn implements Hint {
141     instance;
142 
143     @Override
144     public String getText(Game game, Ability ability) {
145         Player controller = game.getPlayer(ability.getControllerId());
146         WeatheredSentinelsLastTurnAttackersWatcher watcher = game.getState().getWatcher(WeatheredSentinelsLastTurnAttackersWatcher.class);
147         if (controller == null || watcher == null) {
148             return "";
149         }
150 
151         StringBuilder stringBuilder = new StringBuilder("Attacked you on their last turn: ");
152         Iterator<UUID> opponentIdIterator = game.getOpponents(controller.getId()).iterator();
153 
154         while (opponentIdIterator.hasNext()) {
155             UUID opponentId = opponentIdIterator.next();
156             Player opponent = game.getPlayer(opponentId);
157             if (opponent != null && watcher.checkPlayer(opponentId, controller.getId())) {
158                 stringBuilder.append(opponent.getName());
159                 // Add a ", " between names, but exclude adding one at the end
160                 if (opponentIdIterator.hasNext()) {
161                     stringBuilder.append(", ");
162                 } else {
163                     stringBuilder.append('.');
164                 }
165             }
166         }
167 
168         return stringBuilder.toString();
169     }
170 
171     @Override
172     public Hint copy() {
173         return instance;
174     }
175 }

177 enum WeatheredSentinelsCanAttackSomeoneCondition implements Condition {
178     instance;
179 
180     @Override
181     public boolean apply(Game game, Ability source) {
182         Player controller = game.getPlayer(source.getControllerId());
183         WeatheredSentinelsLastTurnAttackersWatcher watcher = game.getState().getWatcher(WeatheredSentinelsLastTurnAttackersWatcher.class);
184         if (controller == null || watcher == null) {
185             return false;
186         }
187 
188         for (UUID opponentId : game.getOpponents(controller.getId())) {
189             Player opponent = game.getPlayer(opponentId);
190             if (opponent != null && watcher.checkPlayer(controller.getId(), opponentId)) {
191                 return true;
192             }
193         }
194         return false;
195     }

198 class WeatheredSentinelsLastTurnAttackersWatcher extends Watcher {
199 
200     private final Map<UUID, Set<UUID>> playerMap = new HashMap<>();
201 
202     WeatheredSentinelsLastTurnAttackersWatcher() {
203         super(WatcherScope.GAME);
204     }
205 
206     @Override
207     public void watch(GameEvent event, Game game) {
208         switch (event.getType()) {
209             case BEGINNING_PHASE_PRE:
210                 playerMap.remove(game.getActivePlayerId());
211                 return;
212             case ATTACKER_DECLARED:
213                 UUID attacker = event.getPlayerId();
214                 Set<UUID> defenders = playerMap.getOrDefault(attacker, new HashSet<>());
215                 defenders.add(event.getTargetId());
216                 playerMap.put(attacker, defenders);
217         }
218     }
219 
220     /**
221      * Checks if on attackerId's last turn they attacked defenderId.
222      *
223      * @param attackerId    The ID of the player to see if they attacked the given defender on the attacker's last turn
224      * @param defenderId    The ID of the player to see if they were attacked by the attacker on the attacker's last turn
225      * @return              Whether the attacker attacked the defender on the attacker's last turn
226      */
227     boolean checkPlayer(UUID attackerId, UUID defenderId) {
228         if (attackerId == null || defenderId == null) {
229             return false;
230         }
231         Set<UUID> defendersLastTurn = playerMap.get(defenderId);
232         return defendersLastTurn != null && defendersLastTurn.contains(attackerId);
233     }

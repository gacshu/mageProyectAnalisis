28  public final class WarlockClass extends CardImpl {
29  
30      public WarlockClass(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{B}");
32  
33          this.subtype.add(SubType.CLASS);
34  
35          // (Gain the next level as a sorcery to add its ability.)
36          this.addAbility(new ClassReminderAbility());
37  
38          // At the beginning of your end step, if a creature died this turn, each opponent loses 1 life.
39          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
40                  new BeginningOfEndStepTriggeredAbility(
41                          new LoseLifeOpponentsEffect(1), TargetController.YOU, false
42                  ), MorbidCondition.instance, "At the beginning of your end step, " +
43                  "if a creature died this turn, each opponent loses 1 life."
44          ).addHint(MorbidHint.instance));
45  
46          // {1}{B}: Level 2
47          this.addAbility(new ClassLevelAbility(2, "{1}{B}"));
48  
49          // When this Class becomes level 2, look at the top three cards of your library. Put one of them into your hand and the rest into your graveyard.
50          this.addAbility(new BecomesClassLevelTriggeredAbility(
51                  new LookLibraryAndPickControllerEffect(3, 1, PutCards.HAND, PutCards.GRAVEYARD), 2));
52  
53          // {6}{B}: Level 3
54          this.addAbility(new ClassLevelAbility(3, "{6}{B}"));
55  
56          // At the beginning of your end step, each opponent loses life equal to the life they lost this turn.
57          this.addAbility(new SimpleStaticAbility(new GainClassAbilitySourceEffect(
58                  new BeginningOfEndStepTriggeredAbility(
59                          new WarlockClassEffect(), TargetController.YOU, false
60                  ), 3
61          )));
62      }
63  
64      private WarlockClass(final WarlockClass card) {
65          super(card);
66      }
67  
68      @Override
69      public WarlockClass copy() {
70          return new WarlockClass(this);
71      }
72  }

74  class WarlockClassEffect extends OneShotEffect {
75  
76      WarlockClassEffect() {
77          super(Outcome.Benefit);
78          staticText = "each opponent loses life equal to the life they lost this turn.";
79      }
80  
81      private WarlockClassEffect(final WarlockClassEffect effect) {
82          super(effect);
83      }
84  
85      @Override
86      public WarlockClassEffect copy() {
87          return new WarlockClassEffect(this);
88      }
89  
90      @Override
91      public boolean apply(Game game, Ability source) {
92          PlayerLostLifeWatcher watcher = game.getState().getWatcher(PlayerLostLifeWatcher.class);
93          if (watcher == null) {
94              return false;
95          }
96          for (UUID playerId : game.getOpponents(source.getControllerId())) {
97              Player opponent = game.getPlayer(playerId);
98              if (opponent == null) {
99                  continue;
100             }
101             int lifeLost = watcher.getLifeLost(playerId);
102             if (lifeLost > 0) {
103                 opponent.loseLife(lifeLost, game, source, false);
104             }
105         }
106         return true;
107     }
108 }

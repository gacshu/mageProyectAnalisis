25   * @author TheElk801

27  public final class WolfwillowHaven extends CardImpl {
28  
29      public WolfwillowHaven(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{G}");
31  
32          this.subtype.add(SubType.AURA);
33  
34          // Enchant land
35          TargetPermanent auraTarget = new TargetLandPermanent();
36          this.getSpellAbility().addTarget(auraTarget);
37          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
38          Ability ability = new EnchantAbility(auraTarget);
39          this.addAbility(ability);
40  
41          // Whenever enchanted land is tapped for mana, its controller adds an additional {G}.
42          this.addAbility(new EnchantedTappedTriggeredManaAbility(new AddManaToManaPoolTargetControllerEffect(
43                  new Mana(ColoredManaSymbol.G), "their"
44          ).setText("its controller adds an additional {G}")));
45  
46          // {4}{G}, Sacrifice Wolfwillow Haven: Create a 2/2 green Wolf creature token. Activate this ability only during your turn.
47          ability = new ActivateIfConditionActivatedAbility(
48                  Zone.BATTLEFIELD, new CreateTokenEffect(new WolfToken()),
49                  new ManaCostsImpl<>("{4}{G}"), MyTurnCondition.instance
50          );
51          ability.addCost(new SacrificeSourceCost());
52          ability.addHint(MyTurnHint.instance);
53          this.addAbility(ability);
54      }
55  
56      private WolfwillowHaven(final WolfwillowHaven card) {
57          super(card);
58      }
59  
60      @Override
61      public WolfwillowHaven copy() {
62          return new WolfwillowHaven(this);
63      }
64  }

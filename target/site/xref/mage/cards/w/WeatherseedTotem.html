25   * @author TheElk801
26   */
27  public final class WeatherseedTotem extends CardImpl {
28  
29      public WeatherseedTotem(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
31  
32          // {tap}: Add {G}.
33          this.addAbility(new GreenManaAbility());
34  
35          // {2}{G}{G}{G}: Weatherseed Totem becomes a 5/3 green Treefolk artifact creature with trample until end of turn.
36          this.addAbility(new SimpleActivatedAbility(new BecomesCreatureSourceEffect(
37                  new CreatureToken(5, 3, "5/3 green Treefolk artifact creature with trample")
38                          .withColor("G")
39                          .withSubType(SubType.TREEFOLK)
40                          .withType(CardType.ARTIFACT)
41                          .withAbility(TrampleAbility.getInstance()),
42                  "", Duration.EndOfTurn
43          ), new ManaCostsImpl<>("{2}{G}{G}{G}")));
44  
45          // When Weatherseed Totem is put into a graveyard from the battlefield, if it was a creature, return this card to its owner's hand.
46          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
47                  new PutIntoGraveFromBattlefieldSourceTriggeredAbility(new ReturnSourceFromGraveyardToHandEffect()),
48                  WeatherseedTotemCondition.instance, "When {this} is put into a graveyard from the battlefield, "
49                  + "if it was a creature, return this card to its owner's hand"
50          ));
51      }
52  
53      private WeatherseedTotem(final WeatherseedTotem card) {
54          super(card);
55      }
56  
57      @Override
58      public WeatherseedTotem copy() {
59          return new WeatherseedTotem(this);
60      }
61  }
62  
63  enum WeatherseedTotemCondition implements Condition {
64      instance;
65  
66      @Override
67      public boolean apply(Game game, Ability source) {
68          return source
69                  .getEffects()
70                  .stream()
71                  .map(effect -> effect.getValue("permanentWasCreature"))
72                  .filter(Objects::nonNull)
73                  .anyMatch(Boolean.class::cast);
74      }
75  }

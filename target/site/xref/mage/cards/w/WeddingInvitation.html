26   * @author TheElk801

28  public final class WeddingInvitation extends CardImpl {
29  
30      public WeddingInvitation(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
32  
33          // When Wedding Invitation enters the battlefield, draw a card.
34          this.addAbility(new EntersBattlefieldTriggeredAbility(new DrawCardSourceControllerEffect(1)));
35  
36          // {T}, Sacrifice Wedding Invitation: Target creature can't be blocked this turn. If it's a Vampire, it also gains lifelink until end of turn.
37          Ability ability = new SimpleActivatedAbility(
38                  new CantBeBlockedTargetEffect(Duration.EndOfTurn), new TapSourceCost()
39          );
40          ability.addCost(new SacrificeSourceCost());
41          ability.addEffect(new WeddingInvitationEffect());
42          ability.addTarget(new TargetCreaturePermanent());
43          this.addAbility(ability);
44      }
45  
46      private WeddingInvitation(final WeddingInvitation card) {
47          super(card);
48      }
49  
50      @Override
51      public WeddingInvitation copy() {
52          return new WeddingInvitation(this);
53      }
54  }
55  
56  class WeddingInvitationEffect extends OneShotEffect {
57  
58      WeddingInvitationEffect() {
59          super(Outcome.Benefit);
60          staticText = "If it's a Vampire, it also gains lifelink until end of turn";
61      }
62  
63      private WeddingInvitationEffect(final WeddingInvitationEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public WeddingInvitationEffect copy() {
69          return new WeddingInvitationEffect(this);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          Permanent permanent = game.getPermanent(source.getFirstTarget());
75          if (permanent == null || !permanent.hasSubtype(SubType.VAMPIRE, game)) {
76              return false;
77          }
78          game.addEffect(new GainAbilityTargetEffect(LifelinkAbility.getInstance(), Duration.EndOfTurn), source);
79          return true;
80      }
81  }

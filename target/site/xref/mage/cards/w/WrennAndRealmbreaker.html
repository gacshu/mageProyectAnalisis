30  public final class WrennAndRealmbreaker extends CardImpl {
31  
32      public WrennAndRealmbreaker(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{1}{G}{G}");
34  
35          this.addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.WRENN);
37          this.setStartingLoyalty(4);
38  
39          // Lands you control have "{T}: Add one mana of any color."
40          this.addAbility(new SimpleStaticAbility(new GainAbilityControlledEffect(
41                  new AnyColorManaAbility(), Duration.WhileOnBattlefield,
42                  StaticFilters.FILTER_LANDS, false
43          )));
44  
45          // +1: Up to one target land you control becomes a 3/3 Elemental creature with vigilance, hexproof, and haste until your next turn. It's still a land.
46          Ability ability = new LoyaltyAbility(new BecomesCreatureTargetEffect(
47                  new CreatureToken(
48                          3, 3, "3/3 Elemental creature " +
49                          "with vigilance, hexproof, and haste", SubType.ELEMENTAL
50                  ).withAbility(VigilanceAbility.getInstance())
51                          .withAbility(HexproofAbility.getInstance())
52                          .withAbility(HasteAbility.getInstance()),
53                  false, true, Duration.UntilYourNextEndStep
54          ), 1);
55          ability.addTarget(new TargetPermanent(0, 1, StaticFilters.FILTER_CONTROLLED_PERMANENT_LAND));
56          this.addAbility(ability);
57  
58          // -2: Mill three cards. You may put a permanent card from among the milled cards into your hand.
59          this.addAbility(new LoyaltyAbility(new MillThenPutInHandEffect(5, StaticFilters.FILTER_CARD_A_PERMANENT), -2));
60  
61          // -7: You get an emblem with "You may play lands and cast permanent spells from your graveyard."
62          this.addAbility(new LoyaltyAbility(new GetEmblemEffect(new WrennAndRealmbreakerEmblem()), -7));
63      }
64  
65      private WrennAndRealmbreaker(final WrennAndRealmbreaker card) {
66          super(card);
67      }
68  
69      @Override
70      public WrennAndRealmbreaker copy() {
71          return new WrennAndRealmbreaker(this);
72      }
73  }

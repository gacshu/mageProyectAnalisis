22  /**
23   *
24   * @author LevelX2
25   */
26  public final class WoodwraithCorrupter extends CardImpl {
27  
28      private static final FilterLandPermanent filter = new FilterLandPermanent("Forest");
29  
30      static {
31          filter.add(SubType.FOREST.getPredicate());
32      }
33  
34      public WoodwraithCorrupter(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{B}{B}{G}");
36  
37          this.subtype.add(SubType.ELEMENTAL);
38          this.subtype.add(SubType.HORROR);
39          this.power = new MageInt(3);
40          this.toughness = new MageInt(6);
41  
42          // {1}{B}{G}, {tap}: Target Forest becomes a 4/4 black and green Elemental Horror creature. It's still a land.
43          Effect effect = new BecomesCreatureTargetEffect(new WoodwraithCorrupterToken(), false, true, Duration.EndOfGame);
44          effect.setText("Target Forest becomes a 4/4 black and green Elemental Horror creature. It's still a land");
45          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new ManaCostsImpl<>("{1}{B}{G}"));
46          ability.addTarget(new TargetPermanent(filter));
47          ability.addCost(new TapSourceCost());
48          this.addAbility(ability);
49      }
50  
51      private WoodwraithCorrupter(final WoodwraithCorrupter card) {
52          super(card);
53      }
54  
55      @Override
56      public WoodwraithCorrupter copy() {
57          return new WoodwraithCorrupter(this);
58      }
59  }
60  
61  class WoodwraithCorrupterToken extends TokenImpl {
62  
63      public WoodwraithCorrupterToken() {
64          super("", "4/4 black and green Elemental Horror creature");
65          cardType.add(CardType.CREATURE);
66  
67          color.setBlack(true);
68          color.setGreen(true);
69          subtype.add(SubType.ELEMENTAL);
70          subtype.add(SubType.HORROR);
71  
72          power = new MageInt(4);
73          toughness = new MageInt(4);
74      }
75      public WoodwraithCorrupterToken(final WoodwraithCorrupterToken token) {
76          super(token);
77      }
78  
79      public WoodwraithCorrupterToken copy() {
80          return new WoodwraithCorrupterToken(this);
81      }
82  }

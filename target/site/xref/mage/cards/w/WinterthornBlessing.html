23  public final class WinterthornBlessing extends CardImpl {
24  
25      public WinterthornBlessing(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{G}{U}");
27  
28          // Put a +1/+1 counter on up to one target creature you control. Tap up to one target creature you don't control, and that creature doesn't untap during its controller's next untap step.
29          this.getSpellAbility().addTarget(new TargetControlledCreaturePermanent(0, 1));
30          this.getSpellAbility().addTarget(new TargetPermanent(0, 1, StaticFilters.FILTER_CREATURE_YOU_DONT_CONTROL));
31          this.getSpellAbility().addEffect(new AddCountersTargetEffect(CounterType.P1P1.createInstance()).setTargetPointer(new FirstTargetPointer()));
32          this.getSpellAbility().addEffect(new TapTargetEffect().setTargetPointer(new SecondTargetPointer()).setText("tap up to one target creature you don't control"));
33          this.getSpellAbility().addEffect(new DontUntapInControllersNextUntapStepTargetEffect().setTargetPointer(new SecondTargetPointer()).setText(", and that creature doesn't untap during its controller's next untap step"));
34  
35          // Flashback {1}{G}{U}
36          this.addAbility(new FlashbackAbility(this, new ManaCostsImpl<>("{1}{G}{U}")));
37  
38      }
39  
40      private WinterthornBlessing(final WinterthornBlessing card) {
41          super(card);
42      }
43  
44      @Override
45      public WinterthornBlessing copy() {
46          return new WinterthornBlessing(this);
47      }
48  }

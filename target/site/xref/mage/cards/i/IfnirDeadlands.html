23  
24  /**
25   *
26   * @author jeffwadsworth
27   */
28  public final class IfnirDeadlands extends CardImpl {
29      
30      private static final FilterControlledPermanent filter = new FilterControlledPermanent("Desert");
31  
32      static {
33          filter.add(SubType.DESERT.getPredicate());
34      }
35  
36      public IfnirDeadlands(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
38          
39          this.subtype.add(SubType.DESERT);
40  
41          // {t}: Add {C}.
42          this.addAbility(new ColorlessManaAbility());
43          
44          // {t}, Pay 1 life: Add {B}.
45          Ability manaAbility = new BlackManaAbility();
46          manaAbility.addCost(new PayLifeCost(1));
47          this.addAbility(manaAbility);
48          
49          // {2}{B}{B}, {t}, Sacrifice a Desert: Put two -1/-1 counters on target creature an opponent controls. Activate this ability only any time you could cast a sorcery.
50          Ability ability = new ActivateAsSorceryActivatedAbility(Zone.BATTLEFIELD, new AddCountersTargetEffect(CounterType.M1M1.createInstance(2)), new ManaCostsImpl<>("{2}{B}{B}"));
51          ability.addCost(new TapSourceCost());
52          ability.addCost(new SacrificeTargetCost(new TargetControlledPermanent(1, 1, filter, true)));
53          ability.addTarget(new TargetOpponentsCreaturePermanent());
54          this.addAbility(ability);
55          
56      }
57  
58      private IfnirDeadlands(final IfnirDeadlands card) {
59          super(card);
60      }
61  
62      @Override
63      public IfnirDeadlands copy() {
64          return new IfnirDeadlands(this);
65      }
66  }

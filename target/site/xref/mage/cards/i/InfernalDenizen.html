40  public final class InfernalDenizen extends CardImpl {
41  
42      public InfernalDenizen(UUID ownerId, CardSetInfo setInfo) {
43          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{7}{B}");
44  
45          this.subtype.add(SubType.DEMON);
46          this.power = new MageInt(5);
47          this.toughness = new MageInt(7);
48  
49          // At the beginning of your upkeep, sacrifice two Swamps. If you can't, tap Infernal Denizen, and an opponent may gain control of a creature you control of their choice for as long as Infernal Denizen remains on the battlefield.
50          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new InfernalDenizenEffect(), TargetController.YOU, false));
51  
52          // {tap}: Gain control of target creature for as long as Infernal Denizen remains on the battlefield.
53          ConditionalContinuousEffect effect = new ConditionalContinuousEffect(
54                  new GainControlTargetEffect(Duration.Custom, true),
55                  new SourceRemainsInZoneCondition(Zone.BATTLEFIELD),
56                  "gain control of target creature for as long as {this} remains on the battlefield");
57          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new TapSourceCost());
58          ability.addTarget(new TargetCreaturePermanent());
59          this.addAbility(ability);
60      }
61  
62      private InfernalDenizen(final InfernalDenizen card) {
63          super(card);
64      }
65  
66      @Override
67      public InfernalDenizen copy() {
68          return new InfernalDenizen(this);
69      }
70  }

72  class InfernalDenizenEffect extends OneShotEffect {
73  
74      private static final FilterPermanent filter = new FilterPermanent();
75  
76      static {
77          filter.add(SubType.SWAMP.getPredicate());
78          filter.add(TargetController.YOU.getControllerPredicate());
79      }
80  
81      InfernalDenizenEffect() {
82          super(Outcome.Benefit);
83          this.staticText = "sacrifice two Swamps. If you can't, tap {this}, "
84                  + "and an opponent may gain control of a creature you control of their choice "
85                  + "for as long as {this} remains on the battlefield";
86      }
87  
88      InfernalDenizenEffect(final InfernalDenizenEffect effect) {
89          super(effect);
90      }
91  
92      @Override
93      public InfernalDenizenEffect copy() {
94          return new InfernalDenizenEffect(this);
95      }
96  
97      @Override
98      public boolean apply(Game game, Ability source) {
99          Permanent creature = game.getPermanent(source.getSourceId());
100         Player player = game.getPlayer(source.getControllerId());
101         if (player != null) {
102             DynamicValue swamps = new PermanentsOnBattlefieldCount(filter);
103             boolean canSac = swamps.calculate(game, source, this) > 1;
104             Effect effect = new SacrificeControllerEffect(filter, 2, "Sacrifice two Swamps");
105             effect.apply(game, source);
106             if (!canSac) {
107                 if (creature != null) {
108                     creature.tap(source, game);
109                 }
110                 TargetOpponent targetOpp = new TargetOpponent(true);
111                 if (targetOpp.canChoose(player.getId(), source, game)
112                         && targetOpp.choose(Outcome.Detriment, player.getId(), source.getSourceId(), source, game)) {
113                     Player opponent = game.getPlayer(targetOpp.getFirstTarget());
114                     if (opponent != null) {
115                         FilterCreaturePermanent filter2 = new FilterCreaturePermanent("creature controlled by " + player.getLogName());
116                         filter2.add(new ControllerIdPredicate(player.getId()));
117                         TargetCreaturePermanent targetCreature = new TargetCreaturePermanent(1, 1, filter2, true);
118                         targetCreature.setTargetController(opponent.getId());
119                         if (targetCreature.canChoose(id, source, game)
120                                 && opponent.chooseUse(Outcome.GainControl, "Gain control of a creature?", source, game)
121                                 && opponent.chooseTarget(Outcome.GainControl, targetCreature, source, game)) {
122                             ConditionalContinuousEffect giveEffect = new ConditionalContinuousEffect(
123                                     new GainControlTargetEffect(Duration.Custom, true, opponent.getId()),
124                                     SourceOnBattlefieldCondition.instance,
125                                     "");
126                             giveEffect.setTargetPointer(new FixedTarget(targetCreature.getFirstTarget(), game));
127                             game.addEffect(giveEffect, source);
128                             return true;
129                         }
130                     }
131                 }
132             }
133         }
134         return false;
135     }

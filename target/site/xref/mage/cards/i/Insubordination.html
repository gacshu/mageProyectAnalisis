7   import mage.abilities.common.delayed.AtTheBeginOfNextEndStepDelayedTriggeredAbility;
8   import mage.abilities.condition.common.DidNotAttackThisTurnEnchantedCondition;
9   import mage.abilities.decorator.ConditionalTriggeredAbility;
10  import mage.abilities.effects.common.AttachEffect;
11  import mage.abilities.effects.common.DamageAttachedControllerEffect;
12  import mage.constants.Outcome;
13  import mage.target.TargetPermanent;
14  import mage.abilities.keyword.EnchantAbility;
15  import mage.cards.CardImpl;
16  import mage.cards.CardSetInfo;
17  import mage.constants.CardType;
18  import mage.constants.TargetController;
19  import mage.watchers.common.AttackedThisTurnWatcher;
20  
21  /**
22   *
23   * @author jeffwadsworth
24   */
25  public final class Insubordination extends CardImpl {
26  
27      public Insubordination(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{B}{B}");
29  
30          this.subtype.add(SubType.AURA);
31  
32          // Enchant creature
33          TargetPermanent auraTarget = new TargetCreaturePermanent();
34          this.getSpellAbility().addTarget(auraTarget);
35          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
36          Ability ability = new EnchantAbility(auraTarget);
37          this.addAbility(ability);
38  
39          // At the beginning of the end step of enchanted creature's controller, Insubordination deals 2 damage to that player unless that creature attacked this turn.
40          this.addAbility(new ConditionalTriggeredAbility(
41                  new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
42                          new DamageAttachedControllerEffect(2),
43                          TargetController.CONTROLLER_ATTACHED_TO),
44                  DidNotAttackThisTurnEnchantedCondition.instance,
45                  "At the beginning of the end step of enchanted creature's controller, {this} deals 2 damage to that player unless that creature attacked this turn."),
46                  new AttackedThisTurnWatcher());
47      }
48  
49      private Insubordination(final Insubordination card) {
50          super(card);
51      }
52  
53      @Override
54      public Insubordination copy() {
55          return new Insubordination(this);
56      }
57  }

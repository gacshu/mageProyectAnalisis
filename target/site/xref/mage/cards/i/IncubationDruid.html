61  class AnyColorLandsProduceManaEffect extends ManaEffect {
62  
63      private boolean inManaTypeCalculation = false;
64  
65      AnyColorLandsProduceManaEffect() {
66          super();
67          staticText = "Add one mana of any type that a land you control could produce. " +
68                  "If {this} has a +1/+1 counter on it, add three mana of that type instead.";
69      }
70  
71      private AnyColorLandsProduceManaEffect(final AnyColorLandsProduceManaEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public List<Mana> getNetMana(Game game, Ability source) {
77          int manaAmount = getManaAmount(game, source);
78          List<Mana> netManas = new ArrayList<>();
79          Mana types = getManaTypes(game, source);
80          if (types.getRed() > 0) {
81              netManas.add(Mana.RedMana(manaAmount));
82          }
83          if (types.getGreen() > 0) {
84              netManas.add(Mana.GreenMana(manaAmount));
85          }
86          if (types.getBlue() > 0) {
87              netManas.add(Mana.BlueMana(manaAmount));
88          }
89          if (types.getWhite() > 0) {
90              netManas.add(Mana.WhiteMana(manaAmount));
91          }
92          if (types.getBlack() > 0) {
93              netManas.add(Mana.BlackMana(manaAmount));
94          }
95          if (types.getColorless() > 0) {
96              netManas.add(Mana.ColorlessMana(manaAmount));
97          }
98          if (types.getAny() > 0) {
99              netManas.add(Mana.AnyMana(manaAmount));
100         }
101         return netManas;
102     }
103 
104     @Override
105     public Mana produceMana(Game game, Ability source) {
106         Mana mana = new Mana();
107         if (game == null) {
108             return mana;
109         }
110         int manaAmount = getManaAmount(game, source);
111         Mana types = getManaTypes(game, source);
112         Choice choice = new ChoiceColor(true);
113         choice.getChoices().clear();
114         choice.setMessage("Pick a mana color");
115         if (types.getBlack() > 0) {
116             choice.getChoices().add("Black");
117         }
118         if (types.getRed() > 0) {
119             choice.getChoices().add("Red");
120         }
121         if (types.getBlue() > 0) {
122             choice.getChoices().add("Blue");
123         }
124         if (types.getGreen() > 0) {
125             choice.getChoices().add("Green");
126         }
127         if (types.getWhite() > 0) {
128             choice.getChoices().add("White");
129         }
130         if (types.getColorless() > 0) {
131             choice.getChoices().add("Colorless");
132         }
133         if (types.getAny() > 0) {
134             choice.getChoices().add("Black");
135             choice.getChoices().add("Red");
136             choice.getChoices().add("Blue");
137             choice.getChoices().add("Green");
138             choice.getChoices().add("White");
139             choice.getChoices().add("Colorless");
140         }
141         if (!choice.getChoices().isEmpty()) {
142             Player player = game.getPlayer(source.getControllerId());
143             if (choice.getChoices().size() == 1) {
144                 choice.setChoice(choice.getChoices().iterator().next());
145             } else if (player == null || !player.choose(outcome, choice, game)) {
146                 return mana;
147             }
148             if (choice.getChoice() != null) {
149                 switch (choice.getChoice()) {
150                     case "Black":
151                         mana.setBlack(manaAmount);
152                         break;
153                     case "Blue":
154                         mana.setBlue(manaAmount);
155                         break;
156                     case "Red":
157                         mana.setRed(manaAmount);
158                         break;
159                     case "Green":
160                         mana.setGreen(manaAmount);
161                         break;
162                     case "White":
163                         mana.setWhite(manaAmount);
164                         break;
165                     case "Colorless":
166                         mana.setColorless(manaAmount);
167                         break;
168                 }
169             }
170         }
171         return mana;
172     }

179     private int getManaAmount(Game game, Ability source) {
180         if (game != null) {
181             Permanent permanent = game.getPermanent(source.getSourceId());
182             if (permanent != null && permanent.getCounters(game).getCount(CounterType.P1P1) > 0) {
183                 return 3;
184             }
185         }
186         return 1;
187     }
188 
189     private Mana getManaTypes(Game game, Ability source) {
190         Mana types = new Mana();
191         if (game == null || game.getPhase() == null) {
192             return types;
193         }
194         if (inManaTypeCalculation) {
195             return types;
196         }
197         inManaTypeCalculation = true;
198         List<Permanent> lands = game.getBattlefield().getActivePermanents(StaticFilters.FILTER_CONTROLLED_PERMANENT_LAND, source.getControllerId(), source, game);
199         for (Permanent land : lands) {
200             Abilities<ActivatedManaAbilityImpl> mana = land.getAbilities().getActivatedManaAbilities(Zone.BATTLEFIELD);
201             for (ActivatedManaAbilityImpl ability : mana) {
202                 if (!ability.equals(source) && ability.definesMana(game)) {
203                     for (Mana netMana : ability.getNetMana(game)) {
204                         types.add(netMana);
205                     }
206                 }
207             }
208         }
209         inManaTypeCalculation = false;
210         return types;
211     }

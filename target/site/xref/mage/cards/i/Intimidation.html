19  public final class Intimidation extends CardImpl {
20  
21      public Intimidation(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{B}{B}{B}");
23  
24          // Creatures you control have fear.
25          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new GainAbilityControlledEffect(FearAbility.getInstance(), Duration.WhileOnBattlefield, StaticFilters.FILTER_PERMANENT_CREATURES)));
26      }
27  
28      private Intimidation(final Intimidation card) {
29          super(card);
30      }
31  
32      @Override
33      public Intimidation copy() {
34          return new Intimidation(this);
35      }
36  }

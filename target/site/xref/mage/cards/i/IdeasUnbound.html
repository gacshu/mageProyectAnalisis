2   package mage.cards.i;

4   import mage.abilities.common.delayed.AtTheBeginOfNextEndStepDelayedTriggeredAbility;
5   import mage.abilities.effects.common.CreateDelayedTriggeredAbilityEffect;
6   import mage.abilities.effects.common.DrawCardSourceControllerEffect;
7   import mage.abilities.effects.common.discard.DiscardControllerEffect;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.SubType;
12  
13  import java.util.UUID;
14  
15  /**
16   *
17   * @author Plopman
18   */
19  public final class IdeasUnbound extends CardImpl {
20  
21      public IdeasUnbound(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{U}{U}");
23          this.subtype.add(SubType.ARCANE);
24  
25  
26          // Draw three cards. Discard three cards at the beginning of the next end step.
27          this.getSpellAbility().addEffect(new DrawCardSourceControllerEffect(3));
28          this.getSpellAbility().addEffect(new CreateDelayedTriggeredAbilityEffect(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(new DiscardControllerEffect(3))));
29      }
30  
31      private IdeasUnbound(final IdeasUnbound card) {
32          super(card);
33      }
34  
35      @Override
36      public IdeasUnbound copy() {
37          return new IdeasUnbound(this);
38      }
39  }

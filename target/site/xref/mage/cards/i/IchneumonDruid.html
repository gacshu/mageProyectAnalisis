28  public final class IchneumonDruid extends CardImpl {
29  
30      public IchneumonDruid(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{G}{G}");
32          this.subtype.add(SubType.HUMAN);
33          this.subtype.add(SubType.DRUID);
34          this.power = new MageInt(1);
35          this.toughness = new MageInt(1);
36  
37          // Whenever an opponent casts an instant spell other than the first instant spell that player casts each turn, Ichneumon Druid deals 4 damage to that player.
38          this.addAbility(new IchneumonDruidAbility(), new IchneumonDruidWatcher());
39      }
40  
41      private IchneumonDruid(final IchneumonDruid card) {
42          super(card);
43      }
44

46      public IchneumonDruid copy() {
47          return new IchneumonDruid(this);
48      }
49  }

51  class IchneumonDruidAbility extends TriggeredAbilityImpl {
52  
53      public IchneumonDruidAbility() {
54          super(Zone.BATTLEFIELD, new DamageTargetEffect(StaticValue.get(4), false, "that player", true));
55      }
56  
57      public IchneumonDruidAbility(final IchneumonDruidAbility ability) {
58          super(ability);
59      }
60  
61      @Override
62      public IchneumonDruidAbility copy() {
63          return new IchneumonDruidAbility(this);
64      }
65  
66      @Override
67      public boolean checkEventType(GameEvent event, Game game) {
68          return event.getType() == GameEvent.EventType.SPELL_CAST;
69      }
70  
71      @Override
72      public boolean checkTrigger(GameEvent event, Game game) {
73          if (!event.getPlayerId().equals(controllerId)) {
74              Spell spell = game.getStack().getSpell(event.getTargetId());
75              if (spell != null && spell.isInstant(game)) {
76                  IchneumonDruidWatcher watcher = game.getState().getWatcher(IchneumonDruidWatcher.class);
77                  if (watcher != null && watcher.moreThanTwoInstantsCast(event.getPlayerId(), game)) {
78                      for (Effect effect : getEffects()) {
79                          effect.setTargetPointer(new FixedTarget(event.getPlayerId()));
80                      }
81                      return true;
82                  }
83              }
84          }
85          return false;
86      }
87  
88      @Override
89      public String getRule() {
90          return "Whenever an opponent casts an instant spell other than the first instant spell that player casts each turn, {this} deals 4 damage to that player.";
91      }
92  }

94  class IchneumonDruidWatcher extends Watcher {
95  
96      private final Map<UUID, Integer> playerInstantCount = new HashMap<>();
97  
98      public IchneumonDruidWatcher() {
99          super(WatcherScope.GAME);
100     }
101 
102     @Override
103     public void watch(GameEvent event, Game game) {
104         if (event.getType() == GameEvent.EventType.SPELL_CAST) {
105             UUID playerId = event.getPlayerId();
106             Spell spell = game.getStack().getSpell(event.getTargetId());
107             if (spell != null && spell.isInstant(game)) {
108                 playerInstantCount.putIfAbsent(event.getPlayerId(), 0);
109                 playerInstantCount.compute(playerId, (k, v) -> v + 1);
110             }
111         }
112     }
113 
114     public boolean moreThanTwoInstantsCast(UUID playerId, Game game) {
115         return playerInstantCount.get(playerId) > 1;
116     }
117 
118     @Override
119     public void reset() {
120         playerInstantCount.clear();
121     }
122 }

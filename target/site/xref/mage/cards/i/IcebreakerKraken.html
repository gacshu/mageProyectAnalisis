35  public final class IcebreakerKraken extends CardImpl {
36  
37      private static final FilterControlledPermanent filter
38              = new FilterControlledLandPermanent("snow land you control");
39  
40      static {
41          filter.add(SuperType.SNOW.getPredicate());
42      }
43  
44      private static final DynamicValue xValue = new PermanentsOnBattlefieldCount(filter);
45      private static final Hint hint = new ValueHint("Snow lands you control", xValue);
46  
47      public IcebreakerKraken(UUID ownerId, CardSetInfo setInfo) {
48          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{10}{U}{U}");
49  
50          this.addSuperType(SuperType.SNOW);
51          this.subtype.add(SubType.KRAKEN);
52          this.power = new MageInt(8);
53          this.toughness = new MageInt(8);
54  
55          // This spell costs {1} less to cast for each snow land you control.
56          this.addAbility(new SimpleStaticAbility(
57                  Zone.ALL, new SpellCostReductionForEachSourceEffect(1, xValue)
58          ).addHint(hint));
59  
60          // When Icebreaker Kraken enters the battlefield, artifacts and creatures target opponent controls don't untap during that player's next untap step.
61          Ability ability = new EntersBattlefieldTriggeredAbility(new IcebreakerKrakenEffect());
62          ability.addTarget(new TargetOpponent());
63          this.addAbility(ability);
64  
65          // Return three snow lands you control to their owner's hand: Return Icebreaker Kraken to its owner's hand.
66          Cost cost = new ReturnToHandChosenControlledPermanentCost(new TargetControlledPermanent(3, filter));
67          cost.setText("return three snow lands you control to their owner's hand");
68          this.addAbility(new SimpleActivatedAbility(new ReturnToHandSourceEffect(), cost));
69      }
70  
71      private IcebreakerKraken(final IcebreakerKraken card) {
72          super(card);
73      }
74  
75      @Override
76      public IcebreakerKraken copy() {
77          return new IcebreakerKraken(this);
78      }
79  }
80  
81  class IcebreakerKrakenEffect extends OneShotEffect {
82  
83      IcebreakerKrakenEffect() {
84          super(Outcome.Benefit);
85          staticText = "artifacts and creatures target opponent controls don't untap during that player's next untap step";
86      }
87  
88      private IcebreakerKrakenEffect(final IcebreakerKrakenEffect effect) {
89          super(effect);
90      }
91  
92      @Override
93      public IcebreakerKrakenEffect copy() {
94          return new IcebreakerKrakenEffect(this);
95      }
96  
97      @Override
98      public boolean apply(Game game, Ability source) {
99          for (Permanent permanent : game.getBattlefield().getActivePermanents(
100                 StaticFilters.FILTER_CONTROLLED_PERMANENT_ARTIFACT_OR_CREATURE,
101                 source.getFirstTarget(), source, game
102         )) {
103             if (permanent == null) {
104                 continue;
105             }
106             game.addEffect(new DontUntapInControllersNextUntapStepTargetEffect()
107                     .setTargetPointer(
108                             new FixedTarget(permanent, game)), source
109             );
110         }
111         return true;
112     }
113 }

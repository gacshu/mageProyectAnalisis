34  public final class InniazTheGaleForce extends CardImpl {
35  
36      private static final FilterCreaturePermanent filter
37              = new FilterCreaturePermanent("Attacking creatures with flying");
38      private static final FilterCreaturePermanent filter2
39              = new FilterCreaturePermanent("creatures you control with flying");
40  
41      static {
42          filter.add(AttackingPredicate.instance);
43          filter.add(new AbilityPredicate(FlyingAbility.class));
44          filter2.add(new AbilityPredicate(FlyingAbility.class));
45      }
46  
47      public InniazTheGaleForce(UUID ownerId, CardSetInfo setInfo) {
48          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}{U}");
49  
50          this.addSuperType(SuperType.LEGENDARY);
51          this.subtype.add(SubType.DJINN);
52          this.power = new MageInt(4);
53          this.toughness = new MageInt(4);
54  
55          // Flying
56          this.addAbility(FlyingAbility.getInstance());
57  
58          // {2}{W/U}: Attacking creatures with flying get +1/+1 until end of turn.
59          this.addAbility(new SimpleActivatedAbility(new BoostAllEffect(
60                  1, 1, Duration.EndOfTurn, filter, false
61          ), new ManaCostsImpl<>("{2}{W/U}")));
62  
63          // Whenever three or more creatures you control with flying attack, each player gains control 
64          // of a nonland permanent of your choice controlled by the player to their right.
65          this.addAbility(new AttacksWithCreaturesTriggeredAbility(
66                  new InniazTheGaleForceEffect(), 3, filter2
67          ));
68      }
69  
70      private InniazTheGaleForce(final InniazTheGaleForce card) {
71          super(card);
72      }
73  
74      @Override
75      public InniazTheGaleForce copy() {
76          return new InniazTheGaleForce(this);
77      }
78  }

80  class InniazTheGaleForceEffect extends OneShotEffect {
81  
82      private static final class PlayerPair {
83  
84          private final Player leftPlayer;
85          private final Player rightPlayer;
86          private final FilterPermanent filter;
87          private final TargetPermanent target;
88  
89          private PlayerPair(Player rightPlayer, Player leftPlayer) {
90              this.leftPlayer = leftPlayer;
91              this.rightPlayer = rightPlayer;
92              this.filter = this.makeFilter();
93              this.target = this.makeTarget();
94          }
95  
96          private FilterPermanent makeFilter() {
97              FilterPermanent filter = new FilterNonlandPermanent(
98                      "nonland permanent controlled by " + rightPlayer.getName()
99                              + " to give to " + leftPlayer.getName()
100             );
101             filter.add(new ControllerIdPredicate(rightPlayer.getId()));
102             return filter;
103         }
104 
105         private TargetPermanent makeTarget() {
106             return new TargetPermanent(1, 1, this.filter, true);
107         }
108 
109         private void chooseTargets(Player controller, Game game, Ability source) {
110             if (game.getBattlefield().count(this.filter, source.getControllerId(), source, game) > 0) {
111                 controller.choose(Outcome.Neutral, this.target, source, game);
112             }
113         }
114 
115         private void createEffect(Game game, Ability source) {
116             if (this.target.getFirstTarget() == null) {
117                 return;
118             }
119             game.addEffect(new GainControlTargetEffect(
120                     Duration.Custom, true, leftPlayer.getId()
121             ).setTargetPointer(new FixedTarget(target.getFirstTarget(), game)), source);
122         }
123     }
124 
125     InniazTheGaleForceEffect() {
126         super(Outcome.Benefit);
127         staticText = "each player gains control of a nonland permanent of your choice controlled by the player to their right.";
128     }
129 
130     private InniazTheGaleForceEffect(final InniazTheGaleForceEffect effect) {
131         super(effect);
132     }
133 
134     @Override
135     public InniazTheGaleForceEffect copy() {
136         return new InniazTheGaleForceEffect(this);
137     }
138 
139     @Override
140     public boolean apply(Game game, Ability source) {
141         Player controller = game.getPlayer(source.getControllerId());
142         if (controller == null) {
143             return false;
144         }
145         PlayerList playerList = game.getState().getPlayersInRange(source.getControllerId(), game);
146         List<PlayerPair> playerPairList = new ArrayList<>();
147         for (int i = 0; i < playerList.size() - 1; i++) {
148             playerPairList.add(new PlayerPair(
149                     game.getPlayer(playerList.get(i)),
150                     game.getPlayer(playerList.get(i + 1))
151             ));
152         }
153         playerPairList.add(new PlayerPair(
154                 game.getPlayer(playerList.get(playerList.size() - 1)),
155                 game.getPlayer(playerList.get(0))
156         ));
157         for (PlayerPair playerPair : playerPairList) {
158             playerPair.chooseTargets(controller, game, source);
159         }
160         for (PlayerPair playerPair : playerPairList) {
161             playerPair.createEffect(game, source);
162         }
163         return true;
164     }

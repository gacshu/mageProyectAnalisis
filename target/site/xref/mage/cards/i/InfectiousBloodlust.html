22  import java.util.UUID;

24  /**
25   * @author fireshoes
26   */
27  public final class InfectiousBloodlust extends CardImpl {
28  
29      private static final FilterCard filter = new FilterCard("card named Infectious Bloodlust");
30  
31      static {
32          filter.add(new NamePredicate("Infectious Bloodlust"));
33      }
34  
35      public InfectiousBloodlust(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{R}");
37          this.subtype.add(SubType.AURA);
38  
39          // Enchant creature
40          TargetPermanent auraTarget = new TargetCreaturePermanent();
41          this.getSpellAbility().addTarget(auraTarget);
42          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
43          Ability ability = new EnchantAbility(auraTarget);
44          this.addAbility(ability);
45  
46          // Enchanted creature gets +2/+1, has haste, and attacks each turn if able.
47          ability = new SimpleStaticAbility(new BoostEnchantedEffect(2, 1));
48          ability.addEffect(new GainAbilityAttachedEffect(
49                  HasteAbility.getInstance(), AttachmentType.AURA
50          ).setText(", has haste"));
51          ability.addEffect(new AttacksIfAbleAttachedEffect(
52                  Duration.WhileOnBattlefield, AttachmentType.AURA
53          ).setText(", and attacks each combat if able"));
54          this.addAbility(ability);
55  
56          // When enchanted creature dies, you may search your library for a card named Infectious Bloodlust, reveal it, put it into your hand, then shuffle your library.
57          this.addAbility(new DiesAttachedTriggeredAbility(new SearchLibraryPutInHandEffect(
58                  new TargetCardInLibrary(filter), true, true
59          ), "enchanted creature", true));
60      }
61  
62      private InfectiousBloodlust(final InfectiousBloodlust card) {
63          super(card);
64      }
65  
66      @Override
67      public InfectiousBloodlust copy() {
68          return new InfectiousBloodlust(this);
69      }
70  }

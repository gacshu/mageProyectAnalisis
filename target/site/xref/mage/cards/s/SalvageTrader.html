25  public final class SalvageTrader extends CardImpl {
26  
27      public SalvageTrader(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
29  
30          this.subtype.add(SubType.CROLUTE);
31          this.power = new MageInt(2);
32          this.toughness = new MageInt(3);
33  
34          // {T}: Exchange control of target artifact you control and target artifact an opponent controls with the same converted mana cost.
35          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD,
36                  new ExchangeControlTargetEffect(Duration.EndOfGame,
37                          "Exchange control of target artifact you control and target artifact an opponent controls with the same mana value", false, true),
38                  new TapSourceCost());
39          FilterArtifactPermanent filterYou = new FilterArtifactPermanent("artifact you control");
40          filterYou.add(TargetController.YOU.getControllerPredicate());
41          ability.addTarget(new TargetArtifactPermanent(filterYou));
42          FilterArtifactPermanent filterOpponent = new FilterArtifactPermanent("artifact an opponent controls with the same casting cost as your targeted artifact");
43          filterOpponent.add(TargetController.OPPONENT.getControllerPredicate());
44          filterOpponent.add(new SameCastingCostPredicate());
45          ability.addTarget(new TargetArtifactPermanent(filterOpponent));
46  
47          this.addAbility(ability);
48      }
49  
50      private SalvageTrader(final SalvageTrader card) {
51          super(card);
52      }
53  
54      @Override
55      public SalvageTrader copy() {
56          return new SalvageTrader(this);
57      }
58  }

60  class SameCastingCostPredicate implements ObjectSourcePlayerPredicate<MageItem> {
61  
62      @Override
63      public boolean apply(ObjectSourcePlayer<MageItem> input, Game game) {
64          StackObject source = game.getStack().getStackObject(input.getSourceId());
65          if (source != null) {
66              if (source.getStackAbility().getTargets().isEmpty()
67                      || source.getStackAbility().getTargets().get(0).getTargets().isEmpty()) {
68                  return true;
69              }
70              Permanent firstTarget = game.getPermanent(
71                      source.getStackAbility().getTargets().get(0).getTargets().get(0));
72              Permanent inputPermanent = game.getPermanent(input.getObject().getId());
73              if (firstTarget != null && inputPermanent != null) {
74                  return firstTarget.getManaValue() == inputPermanent.getManaValue();
75              }
76          }
77          return true;
78      }
79  
80      @Override
81      public String toString() {
82          return "Target with the same casting cost";
83      }

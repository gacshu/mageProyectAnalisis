22  public final class StenchOfEvil extends CardImpl {
23  
24      public StenchOfEvil(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{2}{B}{B}");
26  
27          // Destroy all Plains. For each land destroyed this way, Stench of Evil deals 1 damage to that land's controller unless they pay {2}.
28          this.getSpellAbility().addEffect(new StenchOfEvilEffect());
29  
30      }
31  
32      private StenchOfEvil(final StenchOfEvil card) {
33          super(card);
34      }
35  
36      @Override
37      public StenchOfEvil copy() {
38          return new StenchOfEvil(this);
39      }
40  }
41  
42  class StenchOfEvilEffect extends OneShotEffect {
43  
44      private static final FilterLandPermanent filter = new FilterLandPermanent();
45  
46      static {
47          filter.add(SubType.PLAINS.getPredicate());
48      }
49  
50      public StenchOfEvilEffect() {
51          super(Outcome.Benefit);
52          this.staticText = "Destroy all Plains. For each land destroyed this way, {this} deals 1 damage to that land's controller unless they pay {2}";
53      }
54  
55      public StenchOfEvilEffect(final StenchOfEvilEffect effect) {
56          super(effect);
57      }
58  
59      @Override
60      public StenchOfEvilEffect copy() {
61          return new StenchOfEvilEffect(this);
62      }
63  
64      @Override
65      public boolean apply(Game game, Ability source) {
66          for (Permanent land : game.getBattlefield().getAllActivePermanents(filter, game)) {
67              UUID landControllerId = land.getControllerId();
68              if (land.destroy(source, game, false)) {
69                  Cost cost = new ManaCostsImpl<>("{2}");
70                  Player landController = game.getPlayer(landControllerId);
71                  if (landController != null
72                          && cost.canPay(source, source, landControllerId, game)
73                          && !cost.pay(source, game, source, landControllerId, false)) {
74                      landController.damage(1, source.getSourceId(), source, game);
75                  }
76              }
77          }
78          return true;
79      }
80  }

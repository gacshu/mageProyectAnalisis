7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;

10  import mage.target.common.TargetCreaturePermanent;
11  
12  import java.util.UUID;
13  
14  /**
15   * @author LevelX2
16   */
17  public final class SearingBlood extends CardImpl {
18  
19      public SearingBlood(UUID ownerId, CardSetInfo setInfo) {
20          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{R}{R}");
21  
22          // Searing Blood deals 2 damage to target creature. When that creature dies this turn, Searing Blood deals 3 damage to that creature's controller.
23          this.getSpellAbility().addEffect(new DamageTargetEffect(2));
24          this.getSpellAbility().addEffect(new CreateDelayedTriggeredAbilityEffect(
25                  new WhenTargetDiesDelayedTriggeredAbility(
26                          new DamageTargetEffect(3, true, "that creature's controller"),
27                          SetTargetPointer.PLAYER
28                  )
29          ));
30          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
31      }
32  
33      private SearingBlood(final SearingBlood card) {
34          super(card);
35      }
36  
37      @Override
38      public SearingBlood copy() {
39          return new SearingBlood(this);
40      }
41  }

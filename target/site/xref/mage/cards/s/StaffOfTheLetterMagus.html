49  class StaffOfTheLetterMagusChooseLetterEffect extends OneShotEffect {
50  
51      public StaffOfTheLetterMagusChooseLetterEffect() {
52          super(Outcome.Benefit);
53          staticText = "choose a consonant other than N, R, S, or T";
54      }
55  
56      public StaffOfTheLetterMagusChooseLetterEffect(final StaffOfTheLetterMagusChooseLetterEffect effect) {
57          super(effect);
58      }
59  
60      @Override
61      public boolean apply(Game game, Ability source) {
62          Player controller = game.getPlayer(source.getControllerId());
63          MageObject mageObject = game.getPermanentEntering(source.getSourceId());
64          if (mageObject == null) {
65              mageObject = game.getObject(source);
66          }
67  
68          ChoiceImpl choice = new ChoiceImpl(true);
69          choice.setMessage("Choose letter");
70          Set<String> choices = new LinkedHashSet<>();
71          // Can I choose Y?
72          // Yes. We play by popular game show rules here. Y is a consonant.
73          // https://magic.wizards.com/en/articles/archive/news/unstable-faqawaslfaqpaftidawabiajtbt-2017-12-06
74          Character[] forbiddenChars = {'A', 'E', 'I', 'N', 'O', 'R', 'S', 'T', 'U'};
75          for (Character letter = 'A'; letter <= 'Z'; letter++) {
76              if (Arrays.binarySearch(forbiddenChars, letter) < 0) {
77                  choices.add(letter.toString());
78              }
79          }
80          choice.setChoices(choices);
81  
82          if (controller != null && mageObject != null && controller.choose(outcome, choice, game)) {
83              if (!game.isSimulation()) {
84                  game.informPlayers(mageObject.getLogName() + ": " + controller.getLogName() + " has chosen " + choice.getChoice());
85              }
86              game.getState().setValue(mageObject.getId() + "_letter", choice.getChoice());
87              if (mageObject instanceof Permanent) {
88                  ((Permanent) mageObject).addInfo("chosen letter", CardUtil.addToolTipMarkTags("Chosen letter: " + choice.getChoice()), game);
89              }
90              return true;
91          }
92          return false;
93      }
94  
95      @Override
96      public StaffOfTheLetterMagusChooseLetterEffect copy() {
97          return new StaffOfTheLetterMagusChooseLetterEffect(this);
98      }
99  }

101 class StaffOfTheLetterMagusEffect extends OneShotEffect {
102 
103     public StaffOfTheLetterMagusEffect() {
104         super(Outcome.GainLife);
105         staticText = "you gain 1 life for each time the chosen letter appears in that spell's name";
106     }
107 
108     public StaffOfTheLetterMagusEffect(final StaffOfTheLetterMagusEffect effect) {
109         super(effect);
110     }
111 
112     @Override
113     public StaffOfTheLetterMagusEffect copy() {
114         return new StaffOfTheLetterMagusEffect(this);
115     }
116 
117     @Override
118     public boolean apply(Game game, Ability source) {
119         Player controller = game.getPlayer(source.getControllerId());
120         Spell spell = game.getStack().getSpell(this.getTargetPointer().getFirst(game, source));
121         if (controller != null && spell != null) {
122             MageObject mageObject = game.getObject(source);
123             if (mageObject instanceof Permanent) {
124                 Permanent permanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
125                 if (permanent != null && game.getState().getValue(mageObject.getId() + "_letter") != null) {
126                     int lifegainValue = 0;
127                     String spellName = spell.getName();
128                     for (int i = 0; i < spellName.length(); i++) {
129                         char letter = spellName.charAt(i);
130                         String chosenLetter = (String) game.getState().getValue(mageObject.getId() + "_letter");
131                         if (chosenLetter != null && Character.isLetter(letter) && Character.toUpperCase(letter) == chosenLetter.charAt(0)) {
132                             lifegainValue++;
133                         }
134                     }
135                     controller.gainLife(lifegainValue, game, source);
136                     return true;
137                 }
138             }
139         }
140         return false;
141     }

28  public final class ShatterskullSmashing extends ModalDoubleFacesCard {
29  
30      private static final DynamicValue xValue = new MultipliedValue(ManacostVariableValue.REGULAR, 2);
31  
32      public ShatterskullSmashing(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo,
34                  new CardType[]{CardType.SORCERY}, new SubType[]{}, "{X}{R}{R}",
35                  "Shatterskull, the Hammer Pass", new CardType[]{CardType.LAND}, new SubType[]{}, ""
36          );
37  
38          // 1.
39          // Shatterskull Smashing
40          // Sorcery
41  
42          // Shatterskull Smashing deals X damage divided as you choose among up to two target creatures and/or planeswalkers. If X is 6 or more, Shatterskull Smashing deals twice X damage divided as you choose among them instead.
43          this.getLeftHalfCard().getSpellAbility().addEffect(new ConditionalOneShotEffect(
44                  new DamageMultiEffect(xValue), new DamageMultiEffect(ManacostVariableValue.REGULAR),
45                  ShatterskullSmashingCondition.instance, "{this} deals X damage divided as you choose " +
46                  "among up to two target creatures and/or planeswalkers. If X is 6 or more, " +
47                  "{this} deals twice X damage divided as you choose among them instead."
48          ));
49          this.getLeftHalfCard().getSpellAbility().setTargetAdjuster(ShatterskullSmashingAdjuster.instance);
50  
51          // 2.
52          // Shatterskull, the Hammer Pass
53          // Land
54  
55          // As Shatterskull, the Hammer Pass enters the battlefield, you may pay 3 life. If you don't, it enters the battlefield tapped.
56          this.getRightHalfCard().addAbility(new AsEntersBattlefieldAbility(
57                  new TapSourceUnlessPaysEffect(new PayLifeCost(3)),
58                  "you may pay 3 life. If you don't, it enters the battlefield tapped"
59          ));
60  
61          // {T}: Add {R}.
62          this.getRightHalfCard().addAbility(new RedManaAbility());
63      }
64  
65      private ShatterskullSmashing(final ShatterskullSmashing card) {
66          super(card);
67      }
68  
69      @Override
70      public ShatterskullSmashing copy() {
71          return new ShatterskullSmashing(this);
72      }
73  }
74  
75  enum ShatterskullSmashingCondition implements Condition {
76      instance;
77  
78      @Override
79      public boolean apply(Game game, Ability source) {
80          return source.getManaCostsToPay().getX() >= 6;
81      }
82  }
83  
84  enum ShatterskullSmashingAdjuster implements TargetAdjuster {
85      instance;
86  
87      @Override
88      public void adjustTargets(Ability ability, Game game) {
89          ability.getTargets().clear();
90          TargetAmount target;
91          if (ability.getManaCostsToPay().getX() >= 6) {
92              target = new TargetCreatureOrPlaneswalkerAmount(2 * ability.getManaCostsToPay().getX());
93          } else {
94              target = new TargetCreatureOrPlaneswalkerAmount(ability.getManaCostsToPay().getX());
95          }
96          target.setMinNumberOfTargets(0);
97          target.setMaxNumberOfTargets(2);
98          ability.addTarget(target);
99      }
100 }

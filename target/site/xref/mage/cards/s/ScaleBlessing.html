14   * @author LevelX2

16  public final class ScaleBlessing extends CardImpl {
17  
18      public ScaleBlessing(UUID ownerId, CardSetInfo setInfo) {
19          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{W}");
20  
21          // Bolster 1,
22          this.getSpellAbility().addEffect(new BolsterEffect(1).setText("Bolster 1"));
23  
24          // then put a +1/+1 counter on each creature you control with a +1/+1 counter on it. <i.(To bolster 1, choose a creature with the least toughness among creatures you control and put +1/+1 counter on it.)</i>
25          this.getSpellAbility().addEffect(new AddCountersAllEffect(
26                  CounterType.P1P1.createInstance(), StaticFilters.FILTER_EACH_CONTROLLED_CREATURE_P1P1
27          ).setText(", then put a +1/+1 counter on each creature you control with a +1/+1 counter on it. " +
28                  "<i>(To bolster 1, choose a creature with the least toughness among creatures you control " +
29                  "and put a +1/+1 counter on it.)</i>"));
30      }
31  
32      private ScaleBlessing(final ScaleBlessing card) {
33          super(card);
34      }
35  
36      @Override
37      public ScaleBlessing copy() {
38          return new ScaleBlessing(this);
39      }
40  }

16   *
17   * @author TheElk801

19  public final class SeveredStrands extends CardImpl {
20  
21      public SeveredStrands(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{B}");
23  
24          // As an additional cost to cast this spell, sacrifice a creature.
25          this.getSpellAbility().addCost(new SacrificeTargetCost(new TargetControlledCreaturePermanent(1, 1, StaticFilters.FILTER_CONTROLLED_CREATURE_SHORT_TEXT, true)));
26  
27          // You gain life equal to the sacrificed creature's toughness. Destroy target creature an opponent controls.
28          this.getSpellAbility().addEffect(new GainLifeEffect(
29                  SacrificeCostCreaturesToughness.instance,
30                  "You gain life equal to the "
31                  + "sacrificed creature's toughness."
32          ));
33          this.getSpellAbility().addEffect(new DestroyTargetEffect());
34          this.getSpellAbility().addTarget(new TargetOpponentsCreaturePermanent());
35      }
36  
37      private SeveredStrands(final SeveredStrands card) {
38          super(card);
39      }
40  
41      @Override
42      public SeveredStrands copy() {
43          return new SeveredStrands(this);
44      }
45  }

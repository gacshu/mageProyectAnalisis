18  public final class SilundiVision extends ModalDoubleFacesCard {
19  
20      public SilundiVision(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo,
22                  new CardType[]{CardType.INSTANT}, new SubType[]{}, "{2}{U}",
23                  "Silundi Isle", new CardType[]{CardType.LAND}, new SubType[]{}, ""
24          );
25  
26          // 1.
27          // Silundi Vision
28          // Instant
29  
30          // Look at the top six cards of your library. You may reveal an instant or sorcery card from among them and put it into your hand. Put the rest on the bottom of your library in a random order.
31          this.getLeftHalfCard().getSpellAbility().addEffect(new LookLibraryAndPickControllerEffect(
32                  6, 1, StaticFilters.FILTER_CARD_INSTANT_OR_SORCERY, PutCards.HAND, PutCards.BOTTOM_RANDOM));
33  
34          // 2.
35          // Silundi Isle
36          // Land
37  
38          // Silundi Isle enters the battlefield tapped.
39          this.getRightHalfCard().addAbility(new EntersBattlefieldTappedAbility());
40  
41          // {T}: Add {U}.
42          this.getRightHalfCard().addAbility(new BlueManaAbility());
43      }
44  
45      private SilundiVision(final SilundiVision card) {
46          super(card);
47      }
48  
49      @Override
50      public SilundiVision copy() {
51          return new SilundiVision(this);
52      }
53  }

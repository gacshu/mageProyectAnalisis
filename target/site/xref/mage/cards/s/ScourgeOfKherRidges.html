4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.SimpleActivatedAbility;
8   import mage.abilities.costs.mana.ManaCostsImpl;
9   import mage.abilities.effects.common.DamageAllEffect;
10  import mage.abilities.keyword.FlyingAbility;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.SubType;
15  import mage.constants.Zone;
16  import mage.filter.common.FilterCreaturePermanent;
17  import mage.filter.predicate.Predicates;
18  import mage.filter.predicate.mageobject.AbilityPredicate;
19  import mage.filter.predicate.mageobject.AnotherPredicate;
20  
21  /**
22   *
23   * @author fireshoes
24   */
25  public final class ScourgeOfKherRidges extends CardImpl {
26      
27      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("creature without flying");
28      private static final FilterCreaturePermanent filter2 = new FilterCreaturePermanent("other creature with flying");
29  
30      static {
31          filter.add(Predicates.not(new AbilityPredicate(FlyingAbility.class)));
32          filter2.add(new AbilityPredicate(FlyingAbility.class));
33          filter2.add(AnotherPredicate.instance);
34      }
35  
36      public ScourgeOfKherRidges(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{6}{R}{R}");
38          this.subtype.add(SubType.DRAGON);
39          this.power = new MageInt(6);
40          this.toughness = new MageInt(6);
41  
42          // Flying
43          this.addAbility(FlyingAbility.getInstance());
44          
45          // {1}{R}: Scourge of Kher Ridges deals 2 damage to each creature without flying.
46          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DamageAllEffect(2, filter), new ManaCostsImpl<>("{1}{R}"));
47          this.addAbility(ability);
48          
49          // {5}{R}: Scourge of Kher Ridges deals 6 damage to each other creature with flying.
50          Ability ability2 = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DamageAllEffect(6, filter2), new ManaCostsImpl<>("{5}{R}"));
51          this.addAbility(ability2);
52      }
53  
54      private ScourgeOfKherRidges(final ScourgeOfKherRidges card) {
55          super(card);
56      }
57  
58      @Override
59      public ScourgeOfKherRidges copy() {
60          return new ScourgeOfKherRidges(this);
61      }
62  }

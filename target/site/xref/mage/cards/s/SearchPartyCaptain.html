21  import java.util.UUID;

23  /**
24   * @author TheElk801
25   */
26  public final class SearchPartyCaptain extends CardImpl {
27  
28      private static final Hint hint = new ValueHint(
29              "Creatures you attacked with this turn", SearchPartyCaptainValue.instance
30      );
31  
32      public SearchPartyCaptain(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}");
34  
35          this.subtype.add(SubType.HUMAN);
36          this.subtype.add(SubType.SOLDIER);
37          this.power = new MageInt(2);
38          this.toughness = new MageInt(2);
39  
40          // This spell costs {1} less to cast for each creature you attacked with this turn.
41          this.addAbility(new SimpleStaticAbility(
42                  Zone.ALL, new SpellCostReductionSourceEffect(SearchPartyCaptainValue.instance)
43                  .setText("this spell costs {1} less to cast for each creature you attacked with this turn")
44          ).addHint(hint), new PlayerAttackedWatcher());
45  
46          // When Search Party Captain enters the battlefield, draw a card.
47          this.addAbility(new EntersBattlefieldTriggeredAbility(new DrawCardSourceControllerEffect(1)));
48      }
49  
50      private SearchPartyCaptain(final SearchPartyCaptain card) {
51          super(card);
52      }
53  
54      @Override
55      public SearchPartyCaptain copy() {
56          return new SearchPartyCaptain(this);
57      }
58  }
59  
60  enum SearchPartyCaptainValue implements DynamicValue {
61      instance;
62  
63      @Override
64      public int calculate(Game game, Ability sourceAbility, Effect effect) {
65          return game
66                  .getState()
67                  .getWatcher(PlayerAttackedWatcher.class)
68                  .getNumberOfAttackersCurrentTurn(sourceAbility.getControllerId());
69      }
70  
71      @Override
72      public SearchPartyCaptainValue copy() {
73          return this;
74      }
75  
76      @Override
77      public String getMessage() {
78          return "";
79      }
80  }

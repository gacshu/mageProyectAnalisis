2   package mage.cards.s;
3   
4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.DiesSourceTriggeredAbility;
8   import mage.abilities.effects.common.counter.AddCountersTargetEffect;
9   import mage.constants.SubType;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.counters.CounterType;
14  import mage.target.common.TargetControlledCreaturePermanent;
15  
16  /**
17   *
18   * @author TheElk801
19   */
20  public final class SparringConstruct extends CardImpl {
21  
22      public SparringConstruct(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{1}");
24          this.subtype.add(SubType.CONSTRUCT);
25  
26          this.power = new MageInt(1);
27          this.toughness = new MageInt(1);
28  
29          // When Sparring Construct dies, put a +1/+1 counter on target creature you control.
30          Ability ability = new DiesSourceTriggeredAbility(new AddCountersTargetEffect(CounterType.P1P1.createInstance()), false);
31          ability.addTarget(new TargetControlledCreaturePermanent());
32          this.addAbility(ability);
33      }
34  
35      private SparringConstruct(final SparringConstruct card) {
36          super(card);
37      }
38  
39      @Override
40      public SparringConstruct copy() {
41          return new SparringConstruct(this);
42      }
43  }

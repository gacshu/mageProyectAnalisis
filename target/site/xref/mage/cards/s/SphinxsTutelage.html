28  public final class SphinxsTutelage extends CardImpl {
29  
30      public SphinxsTutelage(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{U}");
32  
33          // Whenever you draw a card, target opponent puts the top two cards of their library into their graveyard. If they're both nonland cards that share a color, repeat this process.
34          Ability ability = new DrawCardControllerTriggeredAbility(new SphinxsTutelageEffect(), false);
35          ability.addTarget(new TargetOpponent());
36          this.addAbility(ability);
37  
38          // {5}{U}: Draw a card, then discard a card.
39          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new DrawDiscardControllerEffect(1, 1), new ManaCostsImpl<>("{5}{U}")));
40      }
41  
42      private SphinxsTutelage(final SphinxsTutelage card) {
43          super(card);
44      }
45  
46      @Override
47      public SphinxsTutelage copy() {
48          return new SphinxsTutelage(this);
49      }
50  }

52  class SphinxsTutelageEffect extends OneShotEffect {
53  
54      public SphinxsTutelageEffect() {
55          super(Outcome.Benefit);
56          this.staticText = "target opponent mills two cards. If two nonland cards that share a color were milled this way, repeat this process.";
57      }
58  
59      public SphinxsTutelageEffect(final SphinxsTutelageEffect effect) {
60          super(effect);
61      }
62  
63      @Override
64      public SphinxsTutelageEffect copy() {
65          return new SphinxsTutelageEffect(this);
66      }
67  
68      @Override
69      public boolean apply(Game game, Ability source) {
70          Player targetPlayer = game.getPlayer(this.getTargetPointer().getFirst(game, source));
71  
72          if (targetPlayer != null) {
73              int possibleIterations = targetPlayer.getLibrary().size() / 2;
74              int iteration = 0;
75              boolean colorShared;
76              do {
77                  iteration++;
78                  if (iteration > possibleIterations + 20) {
79                      // 801.16. If the game somehow enters a "loop" of mandatory actions, repeating a sequence of events
80                      // with no way to stop, the game is a draw for each player who controls an object that's involved in
81                      // that loop, as well as for each player within the range of influence of any of those players. They
82                      // leave the game. All remaining players continue to play the game.
83                      game.setDraw(source.getControllerId());
84                      return true;
85                  }
86                  colorShared = false;
87                  List<Card> cards = targetPlayer
88                          .millCards(2, source, game)
89                          .getCards(game)
90                          .stream()
91                          .filter(card -> !card.isLand(game))
92                          .collect(Collectors.toList());
93                  if (cards.size() < 2) {
94                      break;
95                  }
96                  for (int i = 0; i < cards.size(); i++) {
97                      if (colorShared) {
98                          break;
99                      }
100                     ObjectColor color1 = cards.get(i).getColor(game);
101                     if (color1.isColorless()) {
102                         continue;
103                     }
104                     for (int j = 0; j < cards.size(); j++) {
105                         if (i >= j) {
106                             continue;
107                         }
108                         ObjectColor color2 = cards.get(j).getColor(game);
109                         if (color1.shares(color2)) {
110                             colorShared = true;
111                             break;
112                         }
113                     }
114                 }
115             } while (colorShared);
116             return true;
117         }
118         return false;
119     }

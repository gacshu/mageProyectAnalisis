23  /**
24   *
25   * @author fireshoes
26   */
27  public final class SigardaHeronsGrace extends CardImpl {
28  
29      private static final FilterControlledPermanent filter = new FilterControlledPermanent("Humans");
30  
31      static {
32          filter.add(SubType.HUMAN.getPredicate());
33      }
34  
35      public SigardaHeronsGrace(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{G}{W}");
37          addSuperType(SuperType.LEGENDARY);
38          this.subtype.add(SubType.ANGEL);
39          this.power = new MageInt(4);
40          this.toughness = new MageInt(5);
41  
42          // Flying
43          this.addAbility(FlyingAbility.getInstance());
44  
45          // You and Humans you control have hexproof.
46          Effect effect = new GainAbilityControllerEffect(HexproofAbility.getInstance(), Duration.WhileOnBattlefield);
47          effect.setText("You");
48          Ability ability =new SimpleStaticAbility(Zone.BATTLEFIELD, effect);
49          effect = new GainAbilityControlledEffect(HexproofAbility.getInstance(), Duration.WhileOnBattlefield, filter);
50          effect.concatBy("and");
51          ability.addEffect(effect);
52          this.addAbility(ability);
53  
54          // {2}, Exile a card from your graveyard: Create a 1/1 white Human Soldier creature token.
55          ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new CreateTokenEffect(new HumanSoldierToken()), new GenericManaCost(2));
56          ability.addCost(new ExileFromGraveCost(new TargetCardInYourGraveyard()));
57          this.addAbility(ability);
58      }
59  
60      private SigardaHeronsGrace(final SigardaHeronsGrace card) {
61          super(card);
62      }
63  
64      @Override
65      public SigardaHeronsGrace copy() {
66          return new SigardaHeronsGrace(this);
67      }
68  }

21  public final class ShapeAnew extends CardImpl {
22  
23      public ShapeAnew(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{U}");
25  
26          // The controller of target artifact sacrifices it, then reveals cards from the top
27          // of their library until they reveal an artifact card. That player puts
28          // that card onto the battlefield, then shuffles all other cards revealed this way into their library.
29          this.getSpellAbility().addEffect(new ShapeAnewEffect());
30          this.getSpellAbility().addTarget(new TargetPermanent(StaticFilters.FILTER_PERMANENT_ARTIFACT_AN));
31      }
32  
33      private ShapeAnew(final ShapeAnew card) {
34          super(card);
35      }
36  
37      @Override
38      public ShapeAnew copy() {
39          return new ShapeAnew(this);
40      }
41  
42      private static class ShapeAnewEffect extends OneShotEffect {
43  
44          public ShapeAnewEffect() {
45              super(Outcome.PutCardInPlay);
46              staticText = "The controller of target artifact sacrifices it, then reveals cards from the top of their library until they reveal an artifact card. That player puts that card onto the battlefield, then shuffles all other cards revealed this way into their library";
47          }
48  
49          public ShapeAnewEffect(ShapeAnewEffect effect) {
50              super(effect);
51          }
52  
53          @Override
54          public boolean apply(Game game, Ability source) {
55              Permanent targetPermanent = game.getPermanent(getTargetPointer().getFirst(game, source));
56              if (targetPermanent == null) {
57                  return false;
58              }
59              targetPermanent.sacrifice(source, game);
60              Player targetController = game.getPlayer(targetPermanent.getControllerId());
61              if (targetController == null) {
62                  return false;
63              }
64              Cards revealed = new CardsImpl();
65              Card artifactCard = null;
66              for (Card card : targetController.getLibrary().getCards(game)) {
67                  revealed.add(card);
68                  if (card.isArtifact(game)) {
69                      artifactCard = card;
70                      break;
71                  }
72              }
73              targetController.revealCards(source, revealed, game);
74              if (artifactCard != null) {
75                  targetController.moveCards(artifactCard, Zone.BATTLEFIELD, source, game);
76              }
77              // 1/1/2011: If the first card the player reveals is an artifact card, they will still have to shuffle their library even though no other cards were revealed this way.
78              targetController.shuffleLibrary(source, game);
79              return true;
80          }
81  
82          @Override
83          public ShapeAnewEffect copy() {
84              return new ShapeAnewEffect(this);
85          }
86      }
87  }

22  public final class Sliversmith extends CardImpl {

24      public Sliversmith(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{2}");
26          this.subtype.add(SubType.SPELLSHAPER);
27          this.power = new MageInt(1);
28          this.toughness = new MageInt(1);
29  
30          // {1}, {T}, Discard a card: Create a 1/1 colorless Sliver artifact creature token named Metallic Sliver.
31          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new CreateTokenEffect(new MetallicSliverToken()), new ManaCostsImpl<>("{1}"));
32          ability.addCost(new TapSourceCost());
33          ability.addCost(new DiscardCardCost());
34          this.addAbility(ability);
35      }
36

38          super(card);
39      }

18  public final class ShowerOfCoals extends CardImpl {
19  
20      public ShowerOfCoals(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{R}{R}");
22  
23          // Shower of Coals deals 2 damage to each of up to three target creatures and/or players.
24          // Threshold - Shower of Coals deals 4 damage to each of those creatures and/or players instead if seven or more cards are in your graveyard.
25          Effect effect = new ConditionalOneShotEffect(new DamageTargetEffect(4),
26                  new DamageTargetEffect(2),
27                  new CardsInControllerGraveyardCondition(7),
28                  "{this} deals 2 damage to each of up to three targets.<br/><br/><i>Threshold</i> &mdash; {this} deals 4 damage to each of those permanents and/or players instead if seven or more cards are in your graveyard.");
29          this.getSpellAbility().addTarget(new TargetAnyTarget(0, 3));
30          this.getSpellAbility().addEffect(effect);
31  
32      }
33  
34      private ShowerOfCoals(final ShowerOfCoals card) {
35          super(card);
36      }
37  
38      @Override
39      public ShowerOfCoals copy() {
40          return new ShowerOfCoals(this);
41      }
42  }

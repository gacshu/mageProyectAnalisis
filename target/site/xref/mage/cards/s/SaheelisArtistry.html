21          // Choose one or both —
22          this.getSpellAbility().getModes().setMinModes(1);
23          this.getSpellAbility().getModes().setMaxModes(2);
24          // • Create a token that's a copy of target artifact.
25          CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect();
26          effect.setText("Create a token that's a copy of target artifact");
27          this.getSpellAbility().addEffect(effect);
28          this.getSpellAbility().addTarget(new TargetArtifactPermanent().withChooseHint("create copy of that"));
29          // • Create a token that's a copy of target creature, except that it's an artifact in addition to its other types.
30          effect = new CreateTokenCopyTargetEffect();
31          effect.setBecomesArtifact(true);
32          effect.setText("Create a token that's a copy of target creature, except it's an artifact in addition to its other types");
33          Mode mode1 = new Mode(effect);
34          mode1.addTarget(new TargetCreaturePermanent().withChooseHint("create copy of that, artifact type"));
35          this.getSpellAbility().addMode(mode1);
36      }

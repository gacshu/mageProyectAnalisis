25  public final class SengirTheDarkBaron extends CardImpl {
26  
27      public SengirTheDarkBaron(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{B}{B}");
29  
30          this.addSuperType(SuperType.LEGENDARY);
31          this.subtype.add(SubType.VAMPIRE);
32          this.subtype.add(SubType.NOBLE);
33          this.power = new MageInt(4);
34          this.toughness = new MageInt(4);
35  
36          // Flying
37          this.addAbility(FlyingAbility.getInstance());
38  
39          // Whenever another creature dies, put two +1/+1 counters on Sengir, the Dark Baron.
40          this.addAbility(new DiesCreatureTriggeredAbility(
41                  new AddCountersSourceEffect(CounterType.P1P1.createInstance(2)), false, true
42          ));
43  
44          // Whenever another player loses the game, you gain life equal to that player's life total as the turn began.
45          this.addAbility(new SengirTheDarkBaronTriggeredAbility());
46  
47          // Partner
48          this.addAbility(PartnerAbility.getInstance());
49      }
50  
51      private SengirTheDarkBaron(final SengirTheDarkBaron card) {
52          super(card);
53      }
54  
55      @Override
56      public SengirTheDarkBaron copy() {
57          return new SengirTheDarkBaron(this);
58      }
59  }

61  class SengirTheDarkBaronTriggeredAbility extends TriggeredAbilityImpl {
62  
63      SengirTheDarkBaronTriggeredAbility() {
64          super(Zone.BATTLEFIELD, null, false);
65          this.addWatcher(new SengirTheDarkBaronWatcher());
66      }
67  
68      private SengirTheDarkBaronTriggeredAbility(final SengirTheDarkBaronTriggeredAbility ability) {
69          super(ability);
70      }
71  
72      @Override
73      public SengirTheDarkBaronTriggeredAbility copy() {
74          return new SengirTheDarkBaronTriggeredAbility(this);
75      }
76  
77      @Override
78      public boolean checkEventType(GameEvent event, Game game) {
79          return event.getType() == GameEvent.EventType.LOST;
80      }
81  
82      @Override
83      public boolean checkTrigger(GameEvent event, Game game) {
84          SengirTheDarkBaronWatcher watcher = game.getState().getWatcher(SengirTheDarkBaronWatcher.class);
85          if (watcher == null) {
86              return false;
87          }
88          this.getEffects().clear();
89          this.addEffect(new GainLifeEffect(watcher.getLife(event.getPlayerId())));
90          return true;
91      }
92  
93      @Override
94      public String getRule() {
95          return "Whenever another player loses the game, " +
96                  "you gain life equal to that player's life total as the turn began.";
97      }
98  }

100 class SengirTheDarkBaronWatcher extends Watcher {
101 
102     private final Map<UUID, Integer> playerMap = new HashMap<>();
103 
104     SengirTheDarkBaronWatcher() {
105         super(WatcherScope.GAME);
106     }
107 
108     @Override
109     public void watch(GameEvent event, Game game) {
110         if (event.getType() == GameEvent.EventType.BEGINNING_PHASE_PRE) {
111             game.getPlayers()
112                     .values()
113                     .stream()
114                     .forEach(player -> playerMap.put(player.getId(), player.getLife()));
115         }
116     }
117 
118     @Override
119     public void reset() {
120         playerMap.clear();
121         super.reset();
122     }
123 
124     int getLife(UUID playerId) {
125         return Math.max(0, playerMap.getOrDefault(playerId, 0));
126     }
127 }

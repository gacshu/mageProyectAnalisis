23  public final class SteelbaneHydra extends CardImpl {
24  
25      public SteelbaneHydra(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{X}{G}{G}");
27  
28          this.subtype.add(SubType.TURTLE);
29          this.subtype.add(SubType.HYDRA);
30  
31          // Steelbane Hydra enters the battlefield with X +1/+1 counters on it.
32          this.addAbility(new EntersBattlefieldAbility(new EntersBattlefieldWithXCountersEffect(CounterType.P1P1.createInstance())));
33  
34          // {2}{G}, Remove a +1/+1 counter from Steelbane Hydra: Destroy target artifact or enchantment.
35          Ability ability = new SimpleActivatedAbility(new DestroyTargetEffect(), new ManaCostsImpl<>("{2}{G}"));
36          ability.addCost(new RemoveCountersSourceCost(CounterType.P1P1.createInstance(1)));
37          ability.addTarget(new TargetPermanent(StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_ENCHANTMENT));
38          this.addAbility(ability);
39      }
40  
41      private SteelbaneHydra(final SteelbaneHydra card) {
42          super(card);
43      }
44  
45      @Override
46      public SteelbaneHydra copy() {
47          return new SteelbaneHydra(this);
48      }
49  }

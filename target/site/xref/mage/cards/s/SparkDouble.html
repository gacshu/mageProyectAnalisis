27  public final class SparkDouble extends CardImpl {
28  
29      private static final FilterPermanent filter = new FilterControlledPermanent("a creature or planeswalker you control");
30  
31      static {
32          filter.add(Predicates.or(
33                  CardType.CREATURE.getPredicate(),
34                  CardType.PLANESWALKER.getPredicate()));
35      }
36  
37      public SparkDouble(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
39          this.subtype.add(SubType.ILLUSION);
40          this.power = new MageInt(0);
41          this.toughness = new MageInt(0);
42  
43          // You may have Spark Double enter the battlefield as a copy of a creature or planeswalker you control, except it enters with an additional +1/+1 counter on it if it’s a creature, it enters with an additional loyalty counter on it if it’s a planeswalker, and it isn’t legendary if that permanent is legendary.
44          Effect effect = new CopyPermanentEffect(filter, new SparkDoubleCopyApplier());
45          EntersBattlefieldAbility ability = new EntersBattlefieldAbility(effect, true);
46          this.addAbility(ability);
47      }
48  
49      private SparkDouble(final SparkDouble card) {
50          super(card);
51      }
52  
53      @Override
54      public SparkDouble copy() {
55          return new SparkDouble(this);
56      }
57  }
58  
59  class SparkDoubleCopyApplier extends CopyApplier {
60  
61      @Override
62      public String getText() {
63          return ", except it enters with an additional +1/+1 counter on it if it's a creature, it enters with "
64                  + "an additional loyalty counter on it if it's a planeswalker, and it isn't legendary if "
65                  + "that permanent is legendary";
66      }
67  
68      @Override
69      public boolean apply(Game game, MageObject blueprint, Ability source, UUID copyToObjectId) {
70          // copyToObjectId can be new token outside from game, don't use it
71  
72          // it isn’t legendary if that permanent is legendary
73          //
74          // Spark Double isn’t legendary if it copies a legendary permanent, and this exception is copiable. If something
75          // else copies Spark Double later, that copy also won’t be legendary. If you control two or more permanents
76          // with the same name but only one is legendary, the “legend rule” doesn’t apply.
77          // (2019-05-03)
78          //
79          // So, it's must make changes in blueprint (for farther copyable)
80          blueprint.getSuperType().remove(SuperType.LEGENDARY);
81  
82          // TODO: Blood Moon problem, can't apply on type changing effects (same as TeferisTimeTwist)
83          // see https://magic.wizards.com/en/articles/archive/feature/war-spark-release-notes-2019-04-19
84          // If the copied permanent is affected by a type-changing effect, Spark Double may enter the battlefield with
85          // different permanent types than the copied permanent currently has. Use the characteristics of Spark Double as
86          // it enters the battlefield, not of the copied permanent, to determine whether it enters with an additional
87          // counter on it. Notably, if Spark Double copies a Gideon planeswalker that's a creature because its loyalty
88          // ability caused it to become a planeswalker creature, Spark Double enters as a noncreature planeswalker and
89          // doesn't get a +1/+1 counter. On the other hand, if Spark Double copies Gideon Blackblade during your turn,
90          // Spark Double enters as a planeswalker creature and gets both kinds of counters.
91  
92          // counters only for original card, not copies
93          if (!isCopyOfCopy(source, blueprint, copyToObjectId)) {
94              // enters with an additional +1/+1 counter on it if it’s a creature
95              if (blueprint.isCreature(game)) {
96                  blueprint.getAbilities().add(new EntersBattlefieldAbility(
97                          new AddCountersSourceEffect(CounterType.P1P1.createInstance(), false)
98                                  .setText("with an additional +1/+1 counter on it")
99                  ));
100             }
101 
102             // enters with an additional loyalty counter on it if it’s a planeswalker
103             if (blueprint.isPlaneswalker(game)) {
104                 blueprint.getAbilities().add(new EntersBattlefieldAbility(
105                         new AddCountersSourceEffect(CounterType.LOYALTY.createInstance(), false)
106                                 .setText("with an additional loyalty counter on it")
107                 ));
108             }
109         }
110 
111         return true;
112     }
113 
114 }

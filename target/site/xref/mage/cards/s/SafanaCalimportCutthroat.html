22  /**
23   * @author TheElk801
24   */
25  public final class SafanaCalimportCutthroat extends CardImpl {
26  
27      public SafanaCalimportCutthroat(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{B}");
29  
30          this.addSuperType(SuperType.LEGENDARY);
31          this.subtype.add(SubType.HUMAN);
32          this.subtype.add(SubType.ROGUE);
33          this.power = new MageInt(3);
34          this.toughness = new MageInt(2);
35  
36          // Menace
37          this.addAbility(new MenaceAbility(false));
38  
39          // At the beginning of your end step, if you have the initiative, create a Treasure token. If you've completed a dungeon, create three of those tokens instead.
40          this.addAbility(new BeginningOfEndStepTriggeredAbility(
41                  new ConditionalOneShotEffect(
42                          new CreateTokenEffect(new TreasureToken(), 3),
43                          new CreateTokenEffect(new TreasureToken()),
44                          CompletedDungeonCondition.instance, "create a Treasure token. " +
45                          "If you've completed a dungeon, create three of those tokens instead"
46                  ),
47                  TargetController.YOU, HaveInitiativeCondition.instance, false
48          ).addHint(CompletedDungeonCondition.getHint()), new CompletedDungeonWatcher());
49  
50          // Choose a Background
51          this.addAbility(ChooseABackgroundAbility.getInstance());
52      }
53  
54      private SafanaCalimportCutthroat(final SafanaCalimportCutthroat card) {
55          super(card);
56      }
57  
58      @Override
59      public SafanaCalimportCutthroat copy() {
60          return new SafanaCalimportCutthroat(this);
61      }
62  }

29  public final class Solitude extends CardImpl {
30  
31      private static final FilterPermanent filter = new FilterCreaturePermanent("other creature");
32      private static final FilterCard filter2 = new FilterCard("a white card from your hand");
33  
34      static {
35          filter.add(AnotherPredicate.instance);
36          filter2.add(new ColorPredicate(ObjectColor.WHITE));
37      }
38  
39      public Solitude(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}{W}");
41  
42          this.subtype.add(SubType.ELEMENTAL);
43          this.subtype.add(SubType.INCARNATION);
44          this.power = new MageInt(3);
45          this.toughness = new MageInt(2);
46  
47          // Flash
48          this.addAbility(FlashAbility.getInstance());
49  
50          // Lifelink
51          this.addAbility(LifelinkAbility.getInstance());
52  
53          // When Solitude enters the battlefield, exile up to one other target creature. That creature's controller gains life equal to its power.
54          Ability ability = new EntersBattlefieldTriggeredAbility(new ExileAndGainLifeEqualPowerTargetEffect()
55                  .setText("exile up to one other target creature. That creature's controller gains life equal to its power"));
56          ability.addTarget(new TargetPermanent(0, 1, filter));
57          this.addAbility(ability);
58  
59          // Evokeâ€”Exile a white card from your hand.
60          this.addAbility(new EvokeAbility(new ExileFromHandCost(new TargetCardInHand(filter2))));
61      }
62  
63      private Solitude(final Solitude card) {
64          super(card);
65      }
66  
67      @Override
68      public Solitude copy() {
69          return new Solitude(this);
70      }
71  }

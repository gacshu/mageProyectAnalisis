22  import mage.target.TargetPermanent;
23  import mage.target.common.TargetCardInYourGraveyard;
24  
25  import java.util.UUID;
26  
27  /**
28   * @author TheElk801
29   */
30  public final class SatsukiTheLivingLore extends CardImpl {
31  
32      private static final FilterPermanent filter
33              = new FilterControlledPermanent(SubType.SAGA);
34      private static final FilterPermanent filter2
35              = new FilterControlledPermanent("Saga or enchantment creature you control");
36      private static final FilterCard filter3
37              = new FilterCard("Saga card from your graveyard");
38  
39      static {
40          filter2.add(Predicates.or(
41                  SubType.SAGA.getPredicate(),
42                  Predicates.and(
43                          CardType.ENCHANTMENT.getPredicate(),
44                          CardType.CREATURE.getPredicate()
45                  )
46          ));
47          filter3.add(SubType.SAGA.getPredicate());
48      }
49  
50      public SatsukiTheLivingLore(UUID ownerId, CardSetInfo setInfo) {
51          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{G}{W}");
52  
53          this.addSuperType(SuperType.LEGENDARY);
54          this.subtype.add(SubType.HUMAN);
55          this.subtype.add(SubType.DRUID);
56          this.power = new MageInt(1);
57          this.toughness = new MageInt(3);
58  
59          // {T}: Put a lore counter on each Saga you control. Activate only as a sorcery.
60          this.addAbility(new ActivateAsSorceryActivatedAbility(new AddCountersAllEffect(
61                  CounterType.LORE.createInstance(), filter
62          ), new TapSourceCost()));
63  
64          // When Satsuki, the Living Lore dies, choose up to one —
65          // • Return target Saga or enchantment creature you control to its owner's hand.
66          Ability ability = new DiesSourceTriggeredAbility(new ReturnToHandTargetEffect());
67          ability.getModes().setMinModes(0);
68          ability.getModes().setMaxModes(1);
69          ability.addTarget(new TargetPermanent(filter2));
70  
71          // • Return target Saga card from your graveyard to your hand.
72          Mode mode = new Mode(new ReturnFromGraveyardToHandTargetEffect());
73          mode.addTarget(new TargetCardInYourGraveyard(filter3));
74          ability.addMode(mode);
75          this.addAbility(ability);
76      }
77  
78      private SatsukiTheLivingLore(final SatsukiTheLivingLore card) {
79          super(card);
80      }
81  
82      @Override
83      public SatsukiTheLivingLore copy() {
84          return new SatsukiTheLivingLore(this);
85      }
86  }

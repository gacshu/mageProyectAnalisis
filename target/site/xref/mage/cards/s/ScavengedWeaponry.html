22  /**
23   *
24   * @author michael.napoleon@gmail.com
25   */
26  public final class ScavengedWeaponry extends CardImpl {
27  
28      public ScavengedWeaponry(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{2}{B}");
30          this.subtype.add(SubType.AURA);
31  
32           // Enchant creature
33           TargetPermanent auraTarget = new TargetCreaturePermanent();
34           this.getSpellAbility().addTarget(auraTarget);
35           this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
36           Ability ability = new EnchantAbility(auraTarget);
37           this.addAbility(ability);
38           
39           // When Scavenged Weaponry enters the battlefield, draw a card.
40           this.addAbility(new EntersBattlefieldTriggeredAbility(new DrawCardSourceControllerEffect(1), false));
41           
42           // Enchanted creature gets +1/+1.
43           this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new BoostEnchantedEffect(1, 1, Duration.WhileOnBattlefield)));
44      }
45  
46      private ScavengedWeaponry(final ScavengedWeaponry card) {
47          super(card);
48      }
49  
50      @Override
51      public ScavengedWeaponry copy() {
52          return new ScavengedWeaponry(this);
53      }
54  }

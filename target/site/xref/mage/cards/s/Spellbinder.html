30  public final class Spellbinder extends CardImpl {
31  
32      public Spellbinder(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
34          this.subtype.add(SubType.EQUIPMENT);
35  
36          // Imprint - When Spellbinder enters the battlefield, you may exile 
37          // an instant card from your hand.
38          this.addAbility(new EntersBattlefieldTriggeredAbility(
39                  new SpellbinderImprintEffect(), true)
40                  .setAbilityWord(AbilityWord.IMPRINT)
41          );
42  
43          // Whenever equipped creature deals combat damage to a player, you may 
44          // copy the exiled card. If you do, you may cast the copy without paying its mana cost.
45          this.addAbility(new SpellbinderTriggeredAbility());
46  
47          // Equip {4}
48          this.addAbility(new EquipAbility(Outcome.BoostCreature, new GenericManaCost(4), new TargetControlledCreaturePermanent(), false));
49      }
50  
51      private Spellbinder(final Spellbinder card) {
52          super(card);
53      }
54  
55      @Override
56      public Spellbinder copy() {
57          return new Spellbinder(this);
58      }
59  }

61  class SpellbinderTriggeredAbility extends TriggeredAbilityImpl {
62  
63      SpellbinderTriggeredAbility() {
64          super(Zone.BATTLEFIELD, new SpellbinderCopyEffect(), true);
65      }
66  
67      SpellbinderTriggeredAbility(final SpellbinderTriggeredAbility ability) {
68          super(ability);
69      }
70  
71      @Override
72      public SpellbinderTriggeredAbility copy() {
73          return new SpellbinderTriggeredAbility(this);
74      }
75  
76      @Override
77      public boolean checkEventType(GameEvent event, Game game) {
78          return event.getType() == GameEvent.EventType.DAMAGED_PLAYER;
79      }
80  
81      @Override
82      public boolean checkTrigger(GameEvent event, Game game) {
83          DamagedPlayerEvent damageEvent = (DamagedPlayerEvent) event;
84          Permanent p = game.getPermanent(event.getSourceId());
85          return damageEvent.isCombatDamage()
86                  && p != null
87                  && p.getAttachments().contains(this.getSourceId());
88      }
89  
90      @Override
91      public String getRule() {
92          return "Whenever equipped creature deals combat damage to a player, "
93                  + "you may copy the exiled card. If you do, you may cast "
94                  + "the copy without paying its mana cost.";
95      }
96  }

98  class SpellbinderImprintEffect extends OneShotEffect {
99  
100     private static final FilterCard filter = new FilterCard("instant card");
101 
102     static {
103         filter.add(CardType.INSTANT.getPredicate());
104     }
105 
106     public SpellbinderImprintEffect() {
107         super(Outcome.Benefit);
108         staticText = "you may exile an instant card from your hand";
109     }
110 
111     public SpellbinderImprintEffect(SpellbinderImprintEffect effect) {
112         super(effect);
113     }
114 
115     @Override
116     public boolean apply(Game game, Ability source) {
117         Player controller = game.getPlayer(source.getControllerId());
118         Permanent sourcePermanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
119         if (controller != null) {
120             if (!controller.getHand().isEmpty()) {
121                 TargetCard target = new TargetCard(Zone.HAND, filter);
122                 if (target.canChoose(source.getControllerId(), source, game)
123                         && controller.choose(Outcome.Benefit, controller.getHand(), target, game)) {
124                     Card card = controller.getHand().get(target.getFirstTarget(), game);
125                     if (card != null) {
126                         controller.moveCardToExileWithInfo(card, source.getSourceId(),
127                                 sourcePermanent.getIdName() + " (Imprint)", source, game, Zone.HAND, true);
128                         Permanent permanent = game.getPermanent(source.getSourceId());
129                         if (permanent != null) {
130                             permanent.imprint(card.getId(), game);
131                             permanent.addInfo("imprint", CardUtil.addToolTipMarkTags("[Imprinted card - "
132                                     + card.getLogName() + ']'), game);
133                         }
134                     }
135                 }
136             }
137             return true;
138         }
139         return false;
140 
141     }
142 
143     @Override
144     public SpellbinderImprintEffect copy() {
145         return new SpellbinderImprintEffect(this);
146     }

150 class SpellbinderCopyEffect extends OneShotEffect {
151 
152     public SpellbinderCopyEffect() {
153         super(Outcome.Copy);
154         this.staticText = "You may copy the exiled card. If you do, "
155                 + "you may cast the copy without paying its mana cost";
156     }
157 
158     public SpellbinderCopyEffect(final SpellbinderCopyEffect effect) {
159         super(effect);
160     }
161 
162     @Override
163     public SpellbinderCopyEffect copy() {
164         return new SpellbinderCopyEffect(this);
165     }
166 
167     @Override
168     public boolean apply(Game game, Ability source) {
169         Player controller = game.getPlayer(source.getControllerId());
170         if (controller != null) {
171             Permanent spellbinder = game.getPermanentOrLKIBattlefield(source.getSourceId());
172             if (spellbinder != null && spellbinder.getImprinted() != null && !spellbinder.getImprinted().isEmpty()) {
173                 Card imprintedInstant = game.getCard(spellbinder.getImprinted().get(0));
174                 if (imprintedInstant != null && game.getState().getZone(imprintedInstant.getId()) == Zone.EXILED) {
175                     if (controller.chooseUse(outcome, "Create a copy of " + imprintedInstant.getName() + '?', source, game)) {
176                         Card copiedCard = game.copyCard(imprintedInstant, source, source.getControllerId());
177                         if (copiedCard != null) {
178                             game.getExile().add(source.getSourceId(), "", copiedCard);
179                             game.getState().setZone(copiedCard.getId(), Zone.EXILED);
180                             if (controller.chooseUse(outcome, "Cast the copied card without paying mana cost?", source, game)) {
181                                 if (copiedCard.getSpellAbility() != null) {
182                                     game.getState().setValue("PlayFromNotOwnHandZone" + copiedCard.getId(), Boolean.TRUE);
183                                     controller.cast(controller.chooseAbilityForCast(copiedCard, game, true),
184                                             game, true, new ApprovingObject(source, game));
185                                     game.getState().setValue("PlayFromNotOwnHandZone" + copiedCard.getId(), null);
186                                 } else {
187                                     Logger.getLogger(SpellbinderCopyEffect.class).error(
188                                             "Spellbinder: spell ability == null " + copiedCard.getName());
189                                 }
190                             }
191                         }
192                     }
193                 }
194             }
195             return true;
196         }
197         return false;
198     }

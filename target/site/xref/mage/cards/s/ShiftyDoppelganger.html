29  public final class ShiftyDoppelganger extends CardImpl {
30  
31      public ShiftyDoppelganger(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
33  
34          this.subtype.add(SubType.SHAPESHIFTER);
35          this.power = new MageInt(1);
36          this.toughness = new MageInt(1);
37  
38          // {3}{U}, Exile Shifty Doppelganger: You may put a creature card from your hand onto the battlefield. If you do, that creature gains haste until end of turn. At the beginning of the next end step, sacrifice that creature. If you do, return Shifty Doppelganger to the battlefield.
39          Ability ability = new SimpleActivatedAbility(new ShiftyDoppelgangerExileEffect(), new ManaCostsImpl<>("{3}{U}"));
40          ability.addCost(new ExileSourceCost());
41          this.addAbility(ability);
42      }
43  
44      private ShiftyDoppelganger(final ShiftyDoppelganger card) {
45          super(card);
46      }
47  
48      @Override
49      public ShiftyDoppelganger copy() {
50          return new ShiftyDoppelganger(this);
51      }
52  }

54  class ShiftyDoppelgangerExileEffect extends OneShotEffect {
55  
56      public ShiftyDoppelgangerExileEffect() {
57          super(Outcome.PutCreatureInPlay);
58          this.staticText = "You may put a creature card from your hand onto the battlefield. If you do, " +
59                  "that creature gains haste until end of turn. At the beginning of the next end step, " +
60                  "sacrifice that creature. If you do, return {this} to the battlefield";
61      }
62  
63      private ShiftyDoppelgangerExileEffect(final ShiftyDoppelgangerExileEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public ShiftyDoppelgangerExileEffect copy() {
69          return new ShiftyDoppelgangerExileEffect(this);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          Player player = game.getPlayer(source.getControllerId());
75          if (player == null
76                  || player.getHand().count(StaticFilters.FILTER_CARD_CREATURE, game) < 1
77                  || !player.chooseUse(
78                  Outcome.PutCardInPlay, "Put a creature card " +
79                          "from your hand onto the battlefield?", source, game
80          )) {
81              return false;
82          }
83          TargetCardInHand target = new TargetCardInHand(StaticFilters.FILTER_CARD_CREATURE);
84          player.choose(Outcome.PutCreatureInPlay, player.getHand(), target, game);
85          Card card = game.getCard(target.getFirstTarget());
86          if (card == null) {
87              return false;
88          }
89          player.moveCards(card, Zone.BATTLEFIELD, source, game);
90          Permanent creature = game.getPermanent(card.getId());
91          if (creature == null) {
92              return false;
93          }
94          game.addEffect(new GainAbilityTargetEffect(
95                  HasteAbility.getInstance(), Duration.EndOfTurn
96          ).setTargetPointer(new FixedTarget(creature, game)), source);
97          game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
98                  new ShiftyDoppelgangerReturnEffect(creature, source, game)
99          ), source);
100         return true;
101     }
102 }

104 class ShiftyDoppelgangerReturnEffect extends OneShotEffect {
105 
106     private final MageObjectReference creatureMor;
107     private final MageObjectReference sourceMor;
108 
109     ShiftyDoppelgangerReturnEffect(Permanent creature, Ability source, Game game) {
110         super(Outcome.Benefit);
111         this.staticText = "sacrifice that creature. If you do, return {this} to the battlefield";
112         this.creatureMor = new MageObjectReference(creature, game);
113         this.sourceMor = new MageObjectReference(source, 1);
114     }
115 
116     private ShiftyDoppelgangerReturnEffect(final ShiftyDoppelgangerReturnEffect effect) {
117         super(effect);
118         this.creatureMor = effect.creatureMor;
119         this.sourceMor = effect.sourceMor;
120     }
121 
122     @Override
123     public ShiftyDoppelgangerReturnEffect copy() {
124         return new ShiftyDoppelgangerReturnEffect(this);
125     }
126 
127     @Override
128     public boolean apply(Game game, Ability source) {
129         Permanent creature = creatureMor.getPermanent(game);
130         Player player = game.getPlayer(source.getControllerId());
131         if (creature == null || player == null
132                 || !creature.sacrifice(source, game)) {
133             return false;
134         }
135         Card sourceCard = sourceMor.getCard(game);
136         if (sourceCard != null) {
137             player.moveCards(sourceCard, Zone.BATTLEFIELD, source, game);
138         }
139         return true;
140     }

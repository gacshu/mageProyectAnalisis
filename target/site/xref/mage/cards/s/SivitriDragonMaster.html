21  import java.util.UUID;

23  /**
24   * @author PurpleCrowbar
25   */
26  public final class SivitriDragonMaster extends CardImpl {
27  
28      private static final FilterCard filter = new FilterCard("a Dragon card");
29      private static final FilterCreaturePermanent filterNonDragon = new FilterCreaturePermanent("non-Dragon creatures");
30  
31      static {
32          filter.add(SubType.DRAGON.getPredicate());
33          filterNonDragon.add(Predicates.not(SubType.DRAGON.getPredicate()));
34      }
35  
36      public SivitriDragonMaster(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{2}{U}{B}");
38          addSuperType(SuperType.LEGENDARY);
39          this.subtype.add(SubType.SIVITRI);
40  
41          this.setStartingLoyalty(4);
42  
43          // +1: Until your next turn, creatures canâ€™t attack you or planeswalkers you
44          // control unless their controller pays 2 life for each of those creatures.
45          ContinuousEffect effect = new CantAttackYouUnlessPayAllEffect(
46                  new PayLifeCost(2), true
47          );
48          effect.setDuration(Duration.UntilYourNextTurn);
49          effect.setText("until your next turn, creatures can't attack you or planeswalkers you control " +
50                  "unless their controller pays 2 life for each of those creatures.");
51          this.addAbility(new LoyaltyAbility(effect, 1));
52  
53          // -3: Search your library for a Dragon card, reveal it, put it into your hand, then shuffle.
54          this.addAbility(new LoyaltyAbility(new SearchLibraryPutInHandEffect(
55                  new TargetCardInLibrary(filter), true, true
56          ), -3));
57  
58          // -7: Destroy all non-Dragon creatures.
59          this.addAbility(new LoyaltyAbility(new DestroyAllEffect(filterNonDragon), -7));
60  
61          // Sivitri, Dragon Master can be your commander.
62          this.addAbility(CanBeYourCommanderAbility.getInstance());
63      }
64  
65      private SivitriDragonMaster(final SivitriDragonMaster card) {
66          super(card);
67      }
68  
69      @Override
70      public SivitriDragonMaster copy() {
71          return new SivitriDragonMaster(this);
72      }
73  }

26   *
27   * @author fireshoes
28   */
29  public final class ShieldedByFaith extends CardImpl {
30  
31      public ShieldedByFaith(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{W}{W}");
33          this.subtype.add(SubType.AURA);
34  
35          // Enchant creature
36          TargetPermanent auraTarget = new TargetCreaturePermanent();
37          this.getSpellAbility().addTarget(auraTarget);
38          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Benefit));
39          this.addAbility(new EnchantAbility(auraTarget));
40          
41          // Enchanted creature has indestructible.
42          Effect effect = new GainAbilityAttachedEffect(IndestructibleAbility.getInstance(), 
43                  AttachmentType.AURA, Duration.WhileOnBattlefield);
44          effect.setText("Enchanted creature has indestructible");
45          effect.setOutcome(Outcome.Benefit);
46          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
47          
48          // Whenever a creature enters the battlefield, you may attach Shielded by Faith to that creature.
49          this.addAbility(new EntersBattlefieldAllTriggeredAbility(
50                  Zone.BATTLEFIELD, new AttachEffect(Outcome.Benefit, "attach {this} to that creature"),
51                  StaticFilters.FILTER_PERMANENT_A_CREATURE, true, SetTargetPointer.PERMANENT, null, false));
52      }
53  
54      private ShieldedByFaith(final ShieldedByFaith card) {
55          super(card);
56      }
57  
58      @Override
59      public ShieldedByFaith copy() {
60          return new ShieldedByFaith(this);
61      }
62  }

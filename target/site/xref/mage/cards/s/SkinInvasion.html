31  public final class SkinInvasion extends CardImpl {
32  
33      public SkinInvasion(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{R}");
35          this.subtype.add(SubType.AURA);
36  
37          this.secondSideCardClazz = SkinShedder.class;
38  
39          // Enchant creature
40          TargetPermanent auraTarget = new TargetCreaturePermanent();
41          this.getSpellAbility().addTarget(auraTarget);
42          this.getSpellAbility().addEffect(new AttachEffect(Outcome.AddAbility));
43          Ability ability = new EnchantAbility(auraTarget);
44          this.addAbility(ability);
45  
46          // Enchanted creature attacks each combat if able.
47          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD,
48                  new AttacksIfAbleAttachedEffect(Duration.WhileOnBattlefield, AttachmentType.AURA)));
49  
50          // When enchanted creature dies, return Skin Invasion to the battlefield transformed under your control.
51          this.addAbility(new TransformAbility());
52          this.addAbility(new DiesAttachedTriggeredAbility(new SkinInvasionEffect(), "enchanted creature", false));
53      }
54  
55      private SkinInvasion(final SkinInvasion card) {
56          super(card);
57      }
58  
59      @Override
60      public SkinInvasion copy() {
61          return new SkinInvasion(this);
62      }
63  }
64  
65  class SkinInvasionEffect extends OneShotEffect {
66  
67      public SkinInvasionEffect() {
68          super(Outcome.PutCardInPlay);
69          this.staticText = "return {this} to the battlefield transformed under your control";
70      }
71  
72      public SkinInvasionEffect(final SkinInvasionEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public SkinInvasionEffect copy() {
78          return new SkinInvasionEffect(this);
79      }
80  
81      @Override
82      public boolean apply(Game game, Ability source) {
83          Card card = game.getCard(source.getSourceId());
84          Player controller = game.getPlayer(source.getControllerId());
85          if (card != null && controller != null) {
86              game.getState().setValue(TransformAbility.VALUE_KEY_ENTER_TRANSFORMED + source.getSourceId(), Boolean.TRUE);
87              controller.moveCards(card, Zone.BATTLEFIELD, source, game);
88              return true;
89          }
90          return false;
91      }

93  }

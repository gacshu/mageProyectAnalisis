8   import mage.abilities.effects.common.combat.CantBeBlockedTargetEffect;
9   import mage.abilities.effects.common.combat.GoadTargetEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.Duration;

15  import mage.target.common.TargetOpponentsCreaturePermanent;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class SlyInstigator extends CardImpl {
23  
24      public SlyInstigator(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
26  
27          this.subtype.add(SubType.HUMAN);
28          this.subtype.add(SubType.WIZARD);
29          this.power = new MageInt(2);
30          this.toughness = new MageInt(4);
31  
32          // {U}, {T}: Until your next turn, target creature an opponent controls can't be blocked. Goad that creature.
33          Ability ability = new SimpleActivatedAbility(
34                  new CantBeBlockedTargetEffect(Duration.UntilYourNextTurn)
35                          .setText("until your next turn, target creature an opponent controls can't be blocked."),
36                  new ManaCostsImpl<>("{U}")
37          );
38          ability.addCost(new TapSourceCost());
39          ability.addEffect(new GoadTargetEffect().setText("Goad that creature"));
40          ability.addTarget(new TargetOpponentsCreaturePermanent());
41          this.addAbility(ability);
42      }
43  
44      private SlyInstigator(final SlyInstigator card) {
45          super(card);
46      }
47  
48      @Override
49      public SlyInstigator copy() {
50          return new SlyInstigator(this);
51      }
52  }

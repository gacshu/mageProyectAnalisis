31  public final class SummonThePack extends CardImpl {
32  
33      public SummonThePack(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{7}{B}");
35  
36          // Open a sealed Magic booster pack, reveal the cards, and put all creature cards revealed this way onto the battlefield under your control. They're Zombies in addition to their other types. (Remove those cards from your deck before beginning a new game)
37          this.getSpellAbility().addEffect(new SummonThePackEffect());
38      }
39  
40      private SummonThePack(final SummonThePack card) {
41          super(card);
42      }
43

45      public SummonThePack copy() {
46          return new SummonThePack(this);
47      }
48  }

50  class SummonThePackEffect extends OneShotEffect {
51  
52      public SummonThePackEffect() {
53          super(Outcome.DestroyPermanent);
54          this.staticText = "Open a sealed Magic booster pack, reveal the cards, and put all creature cards revealed this way onto the battlefield under your control. They're Zombies in addition to their other types";
55      }
56  
57      public SummonThePackEffect(final SummonThePackEffect effect) {
58          super(effect);
59      }
60  
61      @Override
62      public SummonThePackEffect copy() {
63          return new SummonThePackEffect(this);
64      }
65  
66      @Override
67      public boolean apply(Game game, Ability source) {
68          ChooseExpansionSetEffect effect = new ChooseExpansionSetEffect(Outcome.UnboostCreature);
69          effect.apply(game, source);
70          Player controller = game.getPlayer(source.getControllerId());
71  
72          String setChosen = null;
73          if (effect.getValue("setchosen") != null) {
74              setChosen = (String) effect.getValue("setchosen");
75          } else if (game.getState().getValue(this.getId() + "_set") != null) {
76              setChosen = (String) game.getState().getValue(this.getId() + "_set");
77          }
78  
79          if (setChosen != null && controller != null) {
80              //ExpansionInfo set = ExpansionRepository.instance.getSetByName(setChosen);
81              ExpansionSet expansionSet = Sets.findSet(setChosen);
82              if (expansionSet != null) {
83                  List<Card> boosterPack = expansionSet.create15CardBooster();
84                  List<Card> creatureCards = new ArrayList<>();
85  
86                  if (boosterPack != null) {
87                      StringBuilder message = new StringBuilder(controller.getLogName()).append(" opened: ");
88  
89                      for (Card c : boosterPack) {
90                          if (c != null && c.isCreature(game)) {
91                              message.append(c.getName()).append(" ");
92                              message.append(" (creature card) ");
93                              ContinuousEffect effect2 = new AddCreatureTypeAdditionEffect(SubType.ZOMBIE, false);
94                              effect2.setTargetPointer(new FixedTarget(c.getId()));
95                              game.addEffect(effect2, source);
96                              creatureCards.add(c);
97                              c.setZone(Zone.OUTSIDE, game);
98                          }
99                      }
100 
101                     if (creatureCards.size() > 0) {
102                         Set<Card> ccs = new HashSet<>(creatureCards);
103                         game.loadCards(ccs, controller.getId());
104                         controller.moveCards(ccs, Zone.BATTLEFIELD, source, game);
105                     }
106 
107                     game.informPlayers(message.toString());
108                 }
109             }
110         }
111 
112         return false;
113     }

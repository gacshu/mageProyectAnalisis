29  /**
30   *
31   * @author LevelX2
32   */
33  public final class ShamanOfTheGreatHunt extends CardImpl {
34      
35      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent("creature you control with power 4 or greater");
36  
37      static {
38          filter.add(new PowerPredicate(ComparisonType.MORE_THAN, 3));
39      }
40      
41      public ShamanOfTheGreatHunt(UUID ownerId, CardSetInfo setInfo) {
42          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{R}");
43          this.subtype.add(SubType.ORC);
44          this.subtype.add(SubType.SHAMAN);
45          this.power = new MageInt(4);
46          this.toughness = new MageInt(2);
47  
48          // Haste
49          this.addAbility(HasteAbility.getInstance());
50          
51          // Whenever a creature you control deals combat damage to a player, put a +1/+1 counter on that creature.
52          Effect effect = new AddCountersTargetEffect(CounterType.P1P1.createInstance());
53          effect.setText("put a +1/+1 counter on that creature");
54          this.addAbility(new DealsDamageToAPlayerAllTriggeredAbility(
55                  effect,
56                  StaticFilters.FILTER_CONTROLLED_A_CREATURE, false, SetTargetPointer.PERMANENT, true
57          ));
58          
59          // <i>Ferocious</i> &mdash; {2}{G/U}{G/U}: Draw a card for each creature you control with power 4 or greater.
60          DynamicValue amount = new PermanentsOnBattlefieldCount(filter);
61          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DrawCardSourceControllerEffect(amount), new ManaCostsImpl<>("{2}{G/U}{G/U}"));
62          ability.setAbilityWord(AbilityWord.FEROCIOUS);
63          this.addAbility(ability);
64          
65      }
66  
67      private ShamanOfTheGreatHunt(final ShamanOfTheGreatHunt card) {
68          super(card);
69      }
70  
71      @Override
72      public ShamanOfTheGreatHunt copy() {
73          return new ShamanOfTheGreatHunt(this);
74      }
75  }

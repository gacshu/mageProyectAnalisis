25  public final class SilvanussInvoker extends CardImpl {
26  
27      public SilvanussInvoker(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{G}");
29  
30          this.subtype.add(SubType.DRAGON);
31          this.subtype.add(SubType.DRUID);
32          this.power = new MageInt(3);
33          this.toughness = new MageInt(2);
34  
35          // Conjure Elemental â€” {8}: Untap target land you control. It becomes an 8/8 Elemental creature with trample and haste until end of turn. It's still a land.
36          Ability ability = new SimpleActivatedAbility(new UntapTargetEffect(), new GenericManaCost(8));
37          ability.addEffect(new BecomesCreatureTargetEffect(
38                  new CreatureToken(8, 8, "8/8 Elemental creature with trample and haste")
39                          .withSubType(SubType.ELEMENTAL)
40                          .withAbility(TrampleAbility.getInstance())
41                          .withAbility(HasteAbility.getInstance()),
42                  false, true, Duration.EndOfTurn
43          ).setText("It becomes an 8/8 Elemental creature with trample and haste until end of turn. It's still a land"));
44          ability.addTarget(new TargetPermanent(StaticFilters.FILTER_CONTROLLED_PERMANENT_LAND));
45          this.addAbility(ability.withFlavorWord("Conjure Elemental"));
46      }
47  
48      private SilvanussInvoker(final SilvanussInvoker card) {
49          super(card);
50      }
51  
52      @Override
53      public SilvanussInvoker copy() {
54          return new SilvanussInvoker(this);
55      }

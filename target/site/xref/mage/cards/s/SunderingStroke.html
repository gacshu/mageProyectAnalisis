21  public final class SunderingStroke extends CardImpl {
22  
23      public SunderingStroke(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{6}{R}");
25  
26          // Sundering Stroke deals 7 damage divided as you choose among one, two, or three targets. If at least seven red mana was spent to cast this spell, instead Sundering Stroke deals 7 damage to each of those permanents and/or players.
27          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
28                  new DamageTargetEffect(7), new DamageMultiEffect(7), SunderingStrokeCondtition.instance,
29                  "{this} deals 7 damage divided as you choose among one, two, or three targets. " +
30                          "If at least seven red mana was spent to cast this spell, " +
31                          "instead {this} deals 7 damage to each of those permanents and/or players"
32          ));
33          this.getSpellAbility().addTarget(new TargetAnyTargetAmount(7, 3));
34          this.getSpellAbility().addHint(new StaticHint(
35                  "(You have to choose how 7 damage is divided even if you spend seven red mana.)"
36          ));
37      }
38  
39      private SunderingStroke(final SunderingStroke card) {
40          super(card);
41      }
42  
43      @Override
44      public SunderingStroke copy() {
45          return new SunderingStroke(this);
46      }
47  }
48  
49  enum SunderingStrokeCondtition implements Condition {
50      instance;
51  
52      @Override
53      public boolean apply(Game game, Ability source) {
54          return source.getManaCostsToPay().getUsedManaToPay().getColor(ColoredManaSymbol.R) >= 6;
55      }
56  }

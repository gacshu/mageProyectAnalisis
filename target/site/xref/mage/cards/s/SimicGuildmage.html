38  public final class SimicGuildmage extends CardImpl {
39  
40      private static final FilterEnchantmentPermanent auraFilter = new FilterEnchantmentPermanent("Aura");
41  
42      static {
43          auraFilter.add(SubType.AURA.getPredicate());
44      }
45  
46      public SimicGuildmage(UUID ownerId, CardSetInfo setInfo) {
47          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{G/U}{G/U}");
48          this.subtype.add(SubType.ELF);
49          this.subtype.add(SubType.WIZARD);
50          this.power = new MageInt(2);
51          this.toughness = new MageInt(2);
52  
53          // {1}{G}: Move a +1/+1 counter from target creature onto another target creature with the same controller.
54          Ability countersAbility = new SimpleActivatedAbility(Zone.BATTLEFIELD, new MoveCounterFromTargetToTargetEffect(), new ManaCostsImpl<>("{1}{G}"));
55          TargetCreaturePermanent target = new TargetCreaturePermanent(
56                  new FilterCreaturePermanent("creature (you take counter from)"));
57          target.setTargetTag(1);
58          countersAbility.addTarget(target);
59  
60          FilterCreaturePermanent filter = new FilterCreaturePermanent(
61                  "another target creature with the same controller (counter goes to)");
62          filter.add(new AnotherTargetPredicate(2));
63          filter.add(new SameControllerPredicate());
64          TargetCreaturePermanent target2 = new TargetCreaturePermanent(filter);
65          target2.setTargetTag(2);
66          countersAbility.addTarget(target2);
67          this.addAbility(countersAbility);
68  
69          // {1}{U}: Attach target Aura enchanting a permanent to another permanent with the same controller.
70          Ability auraAbility = new SimpleActivatedAbility(Zone.BATTLEFIELD, new MoveAuraEffect(), new ManaCostsImpl<>("{1}{U}"));
71          auraAbility.addTarget(new TargetPermanent(auraFilter));
72          this.addAbility(auraAbility);
73  
74      }
75

77          super(card);
78      }
79

81      public SimicGuildmage copy() {
82          return new SimicGuildmage(this);
83      }

86  class MoveCounterFromTargetToTargetEffect extends OneShotEffect {
87  
88      public MoveCounterFromTargetToTargetEffect() {
89          super(Outcome.Detriment);
90          this.staticText = "Move a +1/+1 counter from target creature onto another target creature with the same controller";
91      }
92  
93      public MoveCounterFromTargetToTargetEffect(final MoveCounterFromTargetToTargetEffect effect) {
94          super(effect);
95      }
96  
97      @Override
98      public MoveCounterFromTargetToTargetEffect copy() {
99          return new MoveCounterFromTargetToTargetEffect(this);
100     }
101 
102     @Override
103     public boolean apply(Game game, Ability source) {
104         Player controller = game.getPlayer(source.getControllerId());
105         if (controller != null) {
106             Permanent fromPermanent = game.getPermanent(getTargetPointer().getFirst(game, source));
107             Permanent toPermanent = null;
108             if (source.getTargets().size() > 1) {
109                 toPermanent = game.getPermanent(source.getTargets().get(1).getFirstTarget());
110             }
111             if (fromPermanent == null || toPermanent == null || !fromPermanent.isControlledBy(toPermanent.getControllerId())) {
112                 return false;
113             }
114             fromPermanent.removeCounters(CounterType.P1P1.createInstance(1), source, game);
115             toPermanent.addCounters(CounterType.P1P1.createInstance(1), source.getControllerId(), source, game);
116             return true;
117         }
118         return false;
119 
120     }
121 }

123 class SameControllerPredicate implements ObjectSourcePlayerPredicate<MageItem> {
124 
125     @Override
126     public boolean apply(ObjectSourcePlayer<MageItem> input, Game game) {
127         StackObject source = game.getStack().getStackObject(input.getSourceId());
128         if (source != null) {
129             if (source.getStackAbility().getTargets().isEmpty()
130                     || source.getStackAbility().getTargets().get(0).getTargets().isEmpty()) {
131                 return true;
132             }
133             Permanent firstTarget = game.getPermanent(
134                     source.getStackAbility().getTargets().get(0).getTargets().get(0));
135             Permanent inputPermanent = game.getPermanent(input.getObject().getId());
136             if (firstTarget != null && inputPermanent != null) {
137                 return firstTarget.isControlledBy(inputPermanent.getControllerId());
138             }
139         }
140         return true;
141     }
142 
143     @Override
144     public String toString() {
145         return "Target with the same controller";
146     }

150 class MoveAuraEffect extends OneShotEffect {
151 
152     public MoveAuraEffect() {
153         super(Outcome.AIDontUseIt);
154         staticText = "Attach target Aura enchanting a permanent to another permanent with the same controller.";
155     }
156 
157     public MoveAuraEffect(final MoveAuraEffect effect) {
158         super(effect);
159     }
160 
161     @Override
162     public MoveAuraEffect copy() {
163         return new MoveAuraEffect(this);
164     }
165 
166     @Override
167     public boolean apply(Game game, Ability source) {
168         /*
169             5/1/2006
170             For the second ability, only the Aura is targeted.
171             When the ability resolves, you choose a permanent to move the Aura onto.
172             It can't be the permanent the Aura is already attached to, it must be controlled by the player who controls the permanent the Aura is attached to, and it must be able to be enchanted by the Aura.
173             (It doesn't matter who controls the Aura or who controls Simic Guildmage.)
174             If no such permanent exists, the Aura doesn't move.
175          */
176 
177         Player controller = game.getPlayer(source.getControllerId());
178         if (controller != null) {
179             Permanent aura = game.getPermanent(getTargetPointer().getFirst(game, source));
180             if (aura == null) {
181                 return true;
182             }
183             Permanent fromPermanent = game.getPermanent(aura.getAttachedTo());
184             if (fromPermanent == null) {
185                 return false;
186             }
187             boolean passed = true;
188             Target chosenPermanentToAttachAuras = aura.getSpellAbility().getTargets().get(0).copy();
189             chosenPermanentToAttachAuras.setNotTarget(true);
190             Filter filterChoice = chosenPermanentToAttachAuras.getFilter();
191             filterChoice.add(new ControllerIdPredicate(fromPermanent.getControllerId()));
192             filterChoice.add(Predicates.not(new PermanentIdPredicate(fromPermanent.getId())));
193             chosenPermanentToAttachAuras.setTargetName("a different " + filterChoice.getMessage() + " with the same controller as the " + filterChoice.getMessage() + " the target aura is attached to");
194             if (chosenPermanentToAttachAuras.canChoose(source.getControllerId(), source, game)
195                     && controller.choose(Outcome.Neutral, chosenPermanentToAttachAuras, source, game)) {
196                 Permanent permanentToAttachAura = game.getPermanent(chosenPermanentToAttachAuras.getFirstTarget());
197                 if (permanentToAttachAura != null) {
198                     // Check for protection
199                     if (permanentToAttachAura.cantBeAttachedBy(aura, source, game, true)) {
200                         passed = false;
201                     }
202                     if (passed) {
203                         fromPermanent.removeAttachment(aura.getId(), source, game);
204                         permanentToAttachAura.addAttachment(aura.getId(), source, game);
205                         return true;
206                     }
207                 }
208             }
209             return true;
210         }
211 
212         return false;
213     }

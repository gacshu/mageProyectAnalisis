21  public final class SplitDecision extends CardImpl {
22  
23      public SplitDecision(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{U}");
25  
26          // Will of the council - Choose target instant or sorcery spell. Starting with you, each player votes for denial or duplication. If denial gets more votes, counter the spell. If duplication gets more votes or the vote is tied, copy the spell. You may choose new targets for the copy.
27          this.getSpellAbility().addEffect(new SplitDecisionEffect());
28          this.getSpellAbility().addTarget(new TargetSpell(StaticFilters.FILTER_SPELL_INSTANT_OR_SORCERY));
29      }
30  
31      private SplitDecision(final SplitDecision card) {
32          super(card);
33      }
34  
35      @Override
36      public SplitDecision copy() {
37          return new SplitDecision(this);
38      }
39  }
40  
41  class SplitDecisionEffect extends OneShotEffect {
42  
43      SplitDecisionEffect() {
44          super(Outcome.Removal); // cause AI votes for counter all the time so it must it target opponent's spell, not own
45          this.staticText = "<i>Will of the council</i> &mdash; Choose target instant or sorcery spell. " +
46                  "Starting with you, each player votes for denial or duplication. If denial gets more votes, " +
47                  "counter the spell. If duplication gets more votes or the vote is tied, copy the spell. " +
48                  "You may choose new targets for the copy";
49      }
50  
51      private SplitDecisionEffect(final SplitDecisionEffect effect) {
52          super(effect);
53      }
54  
55      @Override
56      public SplitDecisionEffect copy() {
57          return new SplitDecisionEffect(this);
58      }
59  
60      @Override
61      public boolean apply(Game game, Ability source) {
62          Spell spell = game.getStack().getSpell(getTargetPointer().getFirst(game, source));
63          if (spell == null) {
64              return false;
65          }
66  
67          // Outcome.Benefit - AI will use counter all the time (Denial choice)
68          // TODO: add AI hint logic in the choice method, see Tyrant's Choice as example
69          TwoChoiceVote vote = new TwoChoiceVote(
70                  "Denial (counter " + spell.getIdName() + ")",
71                  "Duplication (copy " + spell.getIdName() + ")",
72                  Outcome.Benefit
73          );
74          vote.doVotes(source, game);
75  
76          int denialCount = vote.getVoteCount(true);
77          int duplicationCount = vote.getVoteCount(false);
78          if (denialCount > duplicationCount) {
79              return game.getStack().counter(spell.getId(), source, game);
80          } else {
81              return new CopyTargetSpellEffect().apply(game, source);
82          }
83      }
84  }

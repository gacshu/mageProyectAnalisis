8   import mage.cards.CardSetInfo;

14  import java.util.UUID;

17   * @author TheElk801

19  public final class SettleBeyondReality extends CardImpl {
20  
21      public SettleBeyondReality(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{W}");
23  
24          // Choose one or both —
25          this.getSpellAbility().getModes().setMinModes(1);
26          this.getSpellAbility().getModes().setMaxModes(2);
27  
28          // • Exile target creature you don't control.
29          this.getSpellAbility().addEffect(new ExileTargetEffect());
30          this.getSpellAbility().addTarget(new TargetPermanent(StaticFilters.FILTER_CREATURE_YOU_DONT_CONTROL));
31  
32          // • Exile target creature you control, then return it to the battlefield under its owner's control.
33          Mode mode = new Mode(new ExileTargetForSourceEffect());
34          mode.addEffect(new ReturnToBattlefieldUnderOwnerControlTargetEffect(false, false, "it").concatBy(", then"));
35          mode.addTarget(new TargetControlledCreaturePermanent());
36          this.getSpellAbility().addMode(mode);
37      }
38  
39      private SettleBeyondReality(final SettleBeyondReality card) {
40          super(card);
41      }
42  
43      @Override
44      public SettleBeyondReality copy() {
45          return new SettleBeyondReality(this);
46      }
47  }

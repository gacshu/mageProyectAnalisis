21   * @author jeffwadsworth

23  public final class SpringjackShepherd extends CardImpl {
24  
25      public SpringjackShepherd(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}");
27          this.subtype.add(SubType.KITHKIN);
28          this.subtype.add(SubType.WIZARD);
29  
30          this.power = new MageInt(1);
31          this.toughness = new MageInt(2);
32  
33          // Chroma - When Springjack Shepherd enters the battlefield, create a 0/1 white Goat creature token for each white mana symbol in the mana costs of permanents you control.
34          DynamicValue xValue = new ChromaCount(ManaType.WHITE);
35          Effect effect = new CreateTokenEffect(new GoatToken(), xValue);
36          effect.setText("create a 0/1 white Goat creature token for each white mana symbol in the mana costs of permanents you control.");
37          this.addAbility(new EntersBattlefieldTriggeredAbility(
38                  effect, false)
39                  .setAbilityWord(AbilityWord.CHROMA)
40                  .addHint(new ValueHint("White mana symbols in your permanents", xValue))
41          );
42      }
43  
44      private SpringjackShepherd(final SpringjackShepherd card) {
45          super(card);
46      }
47  
48      @Override
49      public SpringjackShepherd copy() {
50          return new SpringjackShepherd(this);
51      }
52  }

16  import java.util.UUID;
17  
18  public final class SinuousStriker extends CardImpl {
19  
20      private static final String rule = "Eternalize &mdash; {3}{U}{U}, Discard a card. <i>({3}{U}{U}, Discard a card, Exile this card from your graveyard: Create a token that's a copy of it, except it's a 4/4 black Zombie)</i>";
21  
22      public SinuousStriker(UUID ownerId, CardSetInfo cardSetInfo) {
23          super(ownerId, cardSetInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
24          subtype.add(SubType.NAGA);
25          subtype.add(SubType.WARRIOR);
26  
27          power = new MageInt(2);
28          toughness = new MageInt(2);
29  
30          //U : Sinious Striker gets +1/-1 until end of turn
31          this.addAbility(new SimpleActivatedAbility(new BoostSourceEffect(
32                  +1, -1, Duration.EndOfTurn
33          ), new ManaCostsImpl<>("{U}")));
34  
35          //Eternalize 3UU, Discard a card
36          Ability ability = new EternalizeAbility(new ManaCostsImpl<>("{3}{U}{U}"), this, rule);
37          ability.addCost(new DiscardCardCost());
38          this.addAbility(ability);
39      }
40  
41      private SinuousStriker(final SinuousStriker sinuousStriker) {
42          super(sinuousStriker);
43      }
44  
45      @Override
46      public SinuousStriker copy() {
47          return new SinuousStriker(this);
48      }
49  }

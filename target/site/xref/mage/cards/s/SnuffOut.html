15  import java.util.UUID;

17  /**
18   *
19   * @author LevelX2
20   */
21  public final class SnuffOut extends CardImpl {
22  
23      private static final FilterLandPermanent filterSwamp = new FilterLandPermanent("If you control a Swamp");
24  
25      static {
26          filterSwamp.add(SubType.SWAMP.getPredicate());
27      }
28  
29      public SnuffOut(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{3}{B}");
31  
32          // If you control a Swamp, you may pay 4 life rather than pay Snuff Out's mana cost.
33          this.addAbility(new AlternativeCostSourceAbility(
34                  new PayLifeCost(4),
35                  new PermanentsOnTheBattlefieldCondition(filterSwamp), null));
36          // 
37          // Destroy target nonblack creature. It can't be regenerated.
38          this.getSpellAbility().addEffect(new DestroyTargetEffect(true));
39          this.getSpellAbility().addTarget(new TargetCreaturePermanent(StaticFilters.FILTER_PERMANENT_CREATURE_NON_BLACK));
40      }
41  
42      private SnuffOut(final SnuffOut card) {
43          super(card);
44      }
45  
46      @Override
47      public SnuffOut copy() {
48          return new SnuffOut(this);
49      }
50  }

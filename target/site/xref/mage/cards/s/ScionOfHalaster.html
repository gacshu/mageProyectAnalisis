22  public final class ScionOfHalaster extends CardImpl {
23  
24      public ScionOfHalaster(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{B}");
26  
27          this.addSuperType(SuperType.LEGENDARY);
28          this.subtype.add(SubType.BACKGROUND);
29  
30          // Commander creatures you own have "The first time you would draw a card each turn, instead look at the top two cards of your library. Put one of them into your graveyard and the other back on top of your library. Then draw a card."
31          this.addAbility(new SimpleStaticAbility(new GainAbilityAllEffect(
32                  new SimpleStaticAbility(new ScionOfHalasterReplacementEffect()),
33                  Duration.WhileOnBattlefield, StaticFilters.FILTER_CREATURES_OWNED_COMMANDER
34          )));
35      }
36  
37      private ScionOfHalaster(final ScionOfHalaster card) {
38          super(card);
39      }
40  
41      @Override
42      public ScionOfHalaster copy() {
43          return new ScionOfHalaster(this);
44      }
45  }
46  
47  class ScionOfHalasterReplacementEffect extends ReplacementEffectImpl {
48  
49      ScionOfHalasterReplacementEffect() {
50          super(Duration.WhileOnBattlefield, Outcome.Neutral);
51          staticText = "The first time you would draw a card each turn, instead look at the top two cards of your library. Put one of them into your graveyard and the other back on top of your library. Then draw a card";
52      }
53  
54      ScionOfHalasterReplacementEffect(final mage.cards.s.ScionOfHalasterReplacementEffect effect) {
55          super(effect);
56      }
57  
58      @Override
59      public mage.cards.s.ScionOfHalasterReplacementEffect copy() {
60          return new mage.cards.s.ScionOfHalasterReplacementEffect(this);
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          return true;
66      }
67  
68      @Override
69      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
70          new LookLibraryAndPickControllerEffect(2, 1, PutCards.GRAVEYARD, PutCards.TOP_ANY).apply(game, source);
71          Player you = game.getPlayer(event.getPlayerId());
72          if (you != null) {
73              you.drawCards(1, source, game, event);
74          }
75          return true;
76      }
77  
78      @Override
79      public boolean checksEventType(GameEvent event, Game game) {
80          return event.getType() == GameEvent.EventType.DRAW_CARD;
81      }
82  
83      @Override
84      public boolean applies(GameEvent event, Ability source, Game game) {
85          if (!event.getPlayerId().equals(source.getControllerId())) {
86              return false;
87          }
88          CardsDrawnThisTurnWatcher watcher = game.getState().getWatcher(CardsDrawnThisTurnWatcher.class);
89          return watcher != null && watcher.getCardsDrawnThisTurn(event.getPlayerId()) == 0;
90      }
91  }

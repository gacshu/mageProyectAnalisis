23  public final class SurtlandFrostpyre extends CardImpl {
24  
25      public SurtlandFrostpyre(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
27  
28          // Surtland Frostpyre enters the battlefield tapped.
29          this.addAbility(new EntersBattlefieldTappedAbility());
30  
31          // {T}: Add {R}.
32          this.addAbility(new RedManaAbility());
33  
34          // {2}{U}{U}{R}, {T}, Sacrifice Surtland Frostpyre: Scry 2. Surtland Frostpyre deals 2 damage to each creature. Activate this ability only any time you could cast a sorcery.
35          Ability ability = new ActivateAsSorceryActivatedAbility(
36                  Zone.BATTLEFIELD,
37                  new ScryEffect(2).setText("Scry 2."),
38                  new ManaCostsImpl<>("{2}{U}{U}{R}")
39          );
40          ability.addCost(new TapSourceCost());
41          ability.addCost(new SacrificeSourceCost());
42          ability.addEffect(new DamageAllEffect(2, StaticFilters.FILTER_PERMANENT_CREATURE));
43          this.addAbility(ability);
44      }
45  
46      private SurtlandFrostpyre(final SurtlandFrostpyre card) {
47          super(card);
48      }
49  
50      @Override
51      public SurtlandFrostpyre copy() {
52          return new SurtlandFrostpyre(this);
53      }
54  }

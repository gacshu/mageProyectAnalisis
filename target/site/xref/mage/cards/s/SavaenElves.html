24  /**
25   *
26   * @author fireshoes
27   */
28  public final class SavaenElves extends CardImpl {
29      
30      private static final FilterEnchantmentPermanent filter = new FilterEnchantmentPermanent("Auras attached to a land");
31      
32      static {
33          filter.add(SubType.AURA.getPredicate());
34          filter.add(new SavaenElvesPredicate());
35      }
36  
37      public SavaenElves(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{G}");
39          this.subtype.add(SubType.ELF);
40          this.power = new MageInt(1);
41          this.toughness = new MageInt(1);
42  
43          // {G}{G}, {tap}: Destroy target Aura attached to a land.
44          Effect effect = new DestroyTargetEffect();
45          effect.setText("Destroy target Aura attached to a land");
46          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new ManaCostsImpl<>("{G}{G}"));
47          ability.addCost(new TapSourceCost());
48          ability.addTarget(new TargetPermanent(filter));
49          this.addAbility(ability);
50      }
51  
52      private SavaenElves(final SavaenElves card) {
53          super(card);
54      }
55  
56      @Override
57      public SavaenElves copy() {
58          return new SavaenElves(this);
59      }
60  }
61  
62  class SavaenElvesPredicate implements ObjectSourcePlayerPredicate<Permanent> {
63      @Override
64      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
65          Permanent attachment = input.getObject();
66          if (attachment != null) {
67              Permanent permanent = game.getPermanent(attachment.getAttachedTo());
68              if (permanent != null && permanent.isLand(game)) {
69                  return true;
70              }
71          }
72          return false;
73      }
74      @Override
75      public String toString() {
76          return "Attached to a land";
77      }
78  }

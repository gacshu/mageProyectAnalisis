32  public final class SaprolingBurst extends CardImpl {
33  
34      public SaprolingBurst(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{4}{G}");
36  
37          // Fading 7
38          this.addAbility(new FadingAbility(7, this));
39  
40          // Remove a fade counter from Saproling Burst: create a green Saproling creature token. It has "This creature's power and toughness are each equal to the number of fade counters on Saproling Burst."
41          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new SaprolingBurstCreateTokenEffect(), new RemoveCountersSourceCost(CounterType.FADE.createInstance())));
42  
43          // When Saproling Burst leaves the battlefield, destroy all tokens created with Saproling Burst. They can't be regenerated.
44          this.addAbility(new SaprolingBurstLeavesBattlefieldTriggeredAbility());
45      }
46  
47      private SaprolingBurst(final SaprolingBurst card) {
48          super(card);
49      }
50  
51      @Override
52      public SaprolingBurst copy() {
53          return new SaprolingBurst(this);
54      }
55  }

57  class SaprolingBurstCreateTokenEffect extends OneShotEffect {
58  
59      SaprolingBurstCreateTokenEffect() {
60          super(Outcome.PutCreatureInPlay);
61          this.staticText = "create a green Saproling creature token. It has \"This creature's power and toughness are each equal to the number of fade counters on {this}.\"";
62      }
63  
64      SaprolingBurstCreateTokenEffect(final SaprolingBurstCreateTokenEffect effect) {
65          super(effect);
66      }
67  
68      @Override
69      public SaprolingBurstCreateTokenEffect copy() {
70          return new SaprolingBurstCreateTokenEffect(this);
71      }
72  
73      @Override
74      @SuppressWarnings("unchecked")
75      public boolean apply(Game game, Ability source) {
76          Token token = new SaprolingBurstToken(new MageObjectReference(source.getSourceObject(game), game));
77          token.putOntoBattlefield(1, game, source, source.getControllerId());
78          Permanent permanent = game.getPermanent(source.getSourceId());
79          if (permanent != null) {
80              Object object = game.getState().getValue(CardUtil.getCardZoneString("_tokensCreated", source.getSourceId(), game));
81              Set<UUID> tokensCreated;
82              if (object != null) {
83                  tokensCreated = (Set<UUID>) object;
84              } else {
85                  tokensCreated = new HashSet<>();
86              }
87              for (UUID tokenId : token.getLastAddedTokenIds()) {
88                  tokensCreated.add(tokenId);
89              }
90              game.getState().setValue(CardUtil.getCardZoneString("_tokensCreated", source.getSourceId(), game), tokensCreated);
91          }
92          return true;
93      }
94  }

96  class SaprolingBurstLeavesBattlefieldTriggeredAbility extends ZoneChangeTriggeredAbility {
97  
98      SaprolingBurstLeavesBattlefieldTriggeredAbility() {
99          super(Zone.BATTLEFIELD, null, new SaprolingBurstDestroyEffect(), "When {this} leaves the battlefield, ", false);
100     }
101 
102     SaprolingBurstLeavesBattlefieldTriggeredAbility(SaprolingBurstLeavesBattlefieldTriggeredAbility ability) {
103         super(ability);
104     }
105 
106     @Override
107     public boolean checkTrigger(GameEvent event, Game game) {
108         if (super.checkTrigger(event, game)) {
109             for (Effect effect : this.getEffects()) {
110                 if (effect instanceof SaprolingBurstDestroyEffect) {
111                     ((SaprolingBurstDestroyEffect) effect).setCardZoneString(CardUtil.getCardZoneString("_tokensCreated", this.getSourceId(), game, true));
112                 }
113             }
114             return true;
115         }
116         return false;
117     }
118 
119     @Override
120     public SaprolingBurstLeavesBattlefieldTriggeredAbility copy() {
121         return new SaprolingBurstLeavesBattlefieldTriggeredAbility(this);
122     }
123 }

125 class SaprolingBurstDestroyEffect extends OneShotEffect {
126 
127     private String cardZoneString;
128 
129     SaprolingBurstDestroyEffect() {
130         super(Outcome.Benefit);
131         this.staticText = "destroy all tokens created with {this}. They can't be regenerated";
132     }
133 
134     SaprolingBurstDestroyEffect(final SaprolingBurstDestroyEffect effect) {
135         super(effect);
136         this.cardZoneString = effect.cardZoneString;
137     }
138 
139     @Override
140     public SaprolingBurstDestroyEffect copy() {
141         return new SaprolingBurstDestroyEffect(this);
142     }
143 
144     @Override
145     @SuppressWarnings("unchecked")
146     public boolean apply(Game game, Ability source) {
147         Object object = game.getState().getValue(cardZoneString);
148         if (object != null) {
149             Set<UUID> tokensCreated = (Set<UUID>) object;
150             for (UUID tokenId : tokensCreated) {
151                 Permanent token = game.getPermanent(tokenId);
152                 if (token != null) {
153                     token.destroy(source, game, true);
154                 }
155             }
156         }
157         return true;
158     }
159 
160     public void setCardZoneString(String cardZoneString) {
161         this.cardZoneString = cardZoneString;
162     }
163 }

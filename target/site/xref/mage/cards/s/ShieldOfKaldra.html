4   import java.util.UUID;
5   import mage.abilities.common.SimpleStaticAbility;
6   import mage.abilities.costs.mana.GenericManaCost;
7   import mage.abilities.effects.Effect;
8   import mage.abilities.effects.common.continuous.GainAbilityAllEffect;
9   import mage.abilities.effects.common.continuous.GainAbilityAttachedEffect;
10  import mage.abilities.keyword.EquipAbility;
11  import mage.abilities.keyword.IndestructibleAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.*;
15  import mage.filter.FilterPermanent;
16  import mage.filter.predicate.Predicates;
17  import mage.filter.predicate.mageobject.NamePredicate;
18  
19  /**
20   *
21   * @author North
22   */
23  public final class ShieldOfKaldra extends CardImpl {
24  
25      private static final FilterPermanent filter = new FilterPermanent("Equipment named Sword of Kaldra, Shield of Kaldra, and Helm of Kaldra");
26  
27      static {
28          filter.add(SubType.EQUIPMENT.getPredicate());
29          filter.add(Predicates.or(
30                  new NamePredicate("Sword of Kaldra"),
31                  new NamePredicate("Shield of Kaldra"),
32                  new NamePredicate("Helm of Kaldra")));
33      }
34  
35      public ShieldOfKaldra(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT},"{4}");
37          addSuperType(SuperType.LEGENDARY);
38          this.subtype.add(SubType.EQUIPMENT);
39  
40          // Equipment named Sword of Kaldra, Shield of Kaldra, and Helm of Kaldra are indestructible.
41          Effect effect = new GainAbilityAllEffect(IndestructibleAbility.getInstance(), Duration.WhileOnBattlefield, filter, false);
42          effect.setText("Equipment named Sword of Kaldra, Shield of Kaldra, and Helm of Kaldra are indestructible");
43          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
44          // Equipped creature is indestructible.
45          effect = new GainAbilityAttachedEffect(IndestructibleAbility.getInstance(), AttachmentType.EQUIPMENT, Duration.WhileOnBattlefield);
46          effect.setText("Equipped creature is indestructible");
47          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
48          // Equip {4}
49          this.addAbility(new EquipAbility(Outcome.Benefit, new GenericManaCost(4), false));
50      }
51  
52      private ShieldOfKaldra(final ShieldOfKaldra card) {
53          super(card);
54      }
55  
56      @Override
57      public ShieldOfKaldra copy() {
58          return new ShieldOfKaldra(this);
59      }
60  }

27  public final class SuddenReclamation extends CardImpl {
28  
29      public SuddenReclamation(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{G}");
31  
32          // Put the top four cards of your library into your graveyard, then return a creature card and a land card from your graveyard to your hand.
33          this.getSpellAbility().addEffect(new MillCardsControllerEffect(4));
34          this.getSpellAbility().addEffect(new SuddenReclamationEffect());
35      }
36  
37      private SuddenReclamation(final SuddenReclamation card) {
38          super(card);
39      }
40  
41      @Override
42      public SuddenReclamation copy() {
43          return new SuddenReclamation(this);
44      }
45  }

47  class SuddenReclamationEffect extends OneShotEffect {
48  
49      public SuddenReclamationEffect() {
50          super(Outcome.ReturnToHand);
51          this.staticText = ", then return a creature card and a land card from your graveyard to your hand";
52      }
53  
54      public SuddenReclamationEffect(final SuddenReclamationEffect effect) {
55          super(effect);
56      }
57  
58      @Override
59      public SuddenReclamationEffect copy() {
60          return new SuddenReclamationEffect(this);
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          Player controller = game.getPlayer(source.getControllerId());
66          if (controller != null) {
67              Cards cardsToHand = new CardsImpl();
68              Target target = new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD);
69              target.setNotTarget(true);
70              if (target.canChoose(controller.getId(), source, game)
71                      && controller.chooseTarget(outcome, target, source, game)) {
72                  Card card = game.getCard(target.getFirstTarget());
73                  if (card != null) {
74                      cardsToHand.add(card);
75                  }
76              }
77              target = new TargetCardInYourGraveyard(new FilterLandCard("land card from your graveyard"));
78              target.setNotTarget(true);
79              if (target.canChoose(controller.getId(), source, game)
80                      && controller.chooseTarget(outcome, target, source, game)) {
81                  Card card = game.getCard(target.getFirstTarget());
82                  if (card != null) {
83                      cardsToHand.add(card);
84                  }
85              }
86              controller.moveCards(cardsToHand, Zone.HAND, source, game);
87              return true;
88          }
89          return false;
90      }
91  }

27  public final class ScepterOfCelebration extends CardImpl {
28  
29      public ScepterOfCelebration(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}{G}");
31  
32          this.subtype.add(SubType.EQUIPMENT);
33  
34          // Equipped creature gets +2/+0 and has trample.
35          Ability ability = new SimpleStaticAbility(new BoostEquippedEffect(2, 0));
36          ability.addEffect(new GainAbilityAttachedEffect(
37                  TrampleAbility.getInstance(), AttachmentType.EQUIPMENT
38          ).setText("and has trample"));
39          this.addAbility(ability);
40  
41          // Whenever equipped creature deals combat damage to a player, create that many 1/1 green and white Citizen creature tokens.
42          this.addAbility(new DealsDamageToAPlayerAttachedTriggeredAbility(new CreateTokenEffect(
43                  new CitizenGreenWhiteToken(), SavedDamageValue.MANY
44          ), "equipped", false));
45  
46          // Equip {3}
47          this.addAbility(new EquipAbility(Outcome.BoostCreature, new GenericManaCost(3), new TargetControlledCreaturePermanent(), false));
48      }
49  
50      private ScepterOfCelebration(final ScepterOfCelebration card) {
51          super(card);
52      }
53  
54      @Override
55      public ScepterOfCelebration copy() {
56          return new ScepterOfCelebration(this);
57      }
58  }

30  public final class SlivdraziMonstrosity extends CardImpl {
31  
32      private static final FilterCreaturePermanent eldrazi_you_control = new FilterCreaturePermanent("Eldrazi you control");
33  
34      static {
35          eldrazi_you_control.add(SubType.ELDRAZI.getPredicate());
36      }
37  
38      public SlivdraziMonstrosity(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{C}{W}{U}{B}{R}{G}");
40  
41          this.addSuperType(SuperType.LEGENDARY);
42          this.subtype.add(SubType.SLIVER);
43          this.subtype.add(SubType.ELDRAZI);
44          this.power = new MageInt(8);
45          this.toughness = new MageInt(8);
46  
47          // Eldrazi you control are Slivers in addition to their other types.
48          this.addAbility(new SimpleStaticAbility(new BecomesSubtypeAllEffect(
49                  Duration.WhileOnBattlefield, Arrays.asList(SubType.SLIVER), eldrazi_you_control, false
50          ).setText("Eldrazi you control are Slivers in addition to their other types.")));
51  
52          // Slivers you control have devoid and annihilator 1.
53          Ability ability = new SimpleStaticAbility(new SlivdraziMonstrosityEffect());
54          ability.addEffect(
55                  new GainAbilityControlledEffect(
56                  new AnnihilatorAbility(1), Duration.WhileOnBattlefield,
57                  StaticFilters.FILTER_PERMANENT_SLIVERS
58          ).setText("and annihilator 1"));
59          this.addAbility(ability);
60  
61          // {3}: Create a 1/1 colorless Eldrazi Sliver creature token. It has “Sacrifice this creature: Add {C}.”
62          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new CreateTokenEffect(new EldraziSliverToken()), new ManaCostsImpl<>("{3}")));
63      }
64  
65      private SlivdraziMonstrosity(final SlivdraziMonstrosity card) {
66          super(card);
67      }
68  
69      @Override
70      public SlivdraziMonstrosity copy() {
71          return new SlivdraziMonstrosity(this);
72      }
73  }

75  class SlivdraziMonstrosityEffect extends ContinuousEffectImpl {
76  
77      private static final FilterPermanent filter = new FilterPermanent(SubType.SLIVER, "Slivers");
78  
79      public SlivdraziMonstrosityEffect() {
80          super(Duration.WhileOnBattlefield, Layer.ColorChangingEffects_5, SubLayer.NA, Outcome.Benefit);
81          staticText = "Slivers you control have devoid";
82      }
83  
84      @Override
85      public boolean apply(Game game, Ability source) {
86          for (Permanent perm : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
87              if (perm.isControlledBy(source.getControllerId())) {
88                  // Due to the carefully calibrated system of layers that Magic uses to determine the interaction of continuous
89                  // effects, gaining an ability that changes the color of an object has no effect on the color of that object.
90                  // Despite this, Slivdrazi Monstrosity causes Slivers to become colorless as they gain devoid. Rather than try
91                  // to figure out how this could work, this card should just be played as though it read “Slivers you control
92                  // have devoid and annihilator 1 and are colorless.”
93                  // (2019-11-12)
94                  // This ruling implemented here by adding this ability in the color-changing effect layer.
95                  perm.addAbility(new DevoidAbility(perm.getColor(game)), source.getSourceId(), game);
96              }
97          }
98          return true;
99      }
100 
101     @Override
102     public SlivdraziMonstrosityEffect copy() {
103         return new SlivdraziMonstrosityEffect(this);
104     }
105 
106     private SlivdraziMonstrosityEffect(final SlivdraziMonstrosityEffect effect) {
107         super(effect);
108     }
109 }

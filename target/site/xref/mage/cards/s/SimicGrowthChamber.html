4   import java.util.UUID;
5   import mage.Mana;
6   import mage.abilities.common.EntersBattlefieldTappedAbility;
7   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
8   import mage.abilities.costs.common.TapSourceCost;
9   import mage.abilities.effects.common.ReturnToHandChosenControlledPermanentEffect;
10  import mage.abilities.mana.SimpleManaAbility;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.Zone;
15  import mage.filter.common.FilterControlledLandPermanent;
16  import mage.filter.common.FilterControlledPermanent;
17  
18  /**
19   *
20   * @author Loki
21   */
22  public final class SimicGrowthChamber extends CardImpl {
23  
24      private static final FilterControlledPermanent filter = new FilterControlledLandPermanent();
25  
26      public SimicGrowthChamber(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId,setInfo,new CardType[]{CardType.LAND},"");
28  
29          // Simic Growth Chamber enters the battlefield tapped.
30          this.addAbility(new EntersBattlefieldTappedAbility());
31          // When Simic Growth Chamber enters the battlefield, return a land you control to its owner's hand.
32          this.addAbility(new EntersBattlefieldTriggeredAbility(new ReturnToHandChosenControlledPermanentEffect(filter), false));
33          // {tap}: Add {G}{U}.
34          this.addAbility(new SimpleManaAbility(Zone.BATTLEFIELD, new Mana(0, 1, 0, 0, 1, 0, 0, 0), new TapSourceCost()));
35      }
36  
37      private SimicGrowthChamber(final SimicGrowthChamber card) {
38          super(card);
39      }
40  
41      @Override
42      public SimicGrowthChamber copy() {
43          return new SimicGrowthChamber(this);
44      }
45  }

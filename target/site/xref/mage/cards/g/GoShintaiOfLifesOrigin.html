23  
24  import java.util.UUID;

26  /**
27   * @author TheElk801
28   */
29  public final class GoShintaiOfLifesOrigin extends CardImpl {
30  
31      private static final FilterCard filter
32              = new FilterEnchantmentCard("enchantment card from your graveyard");
33      private static final FilterPermanent filter2
34              = new FilterControlledPermanent(SubType.SHRINE, "nontoken Shrine");
35  
36      static {
37          filter2.add(TokenPredicate.FALSE);
38      }
39  
40      public GoShintaiOfLifesOrigin(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT, CardType.CREATURE}, "{3}{G}");
42  
43          this.addSuperType(SuperType.LEGENDARY);
44          this.subtype.add(SubType.SHRINE);
45          this.power = new MageInt(3);
46          this.toughness = new MageInt(4);
47  
48          // {W}{U}{B}{R}{G}, {T}: Return target enchantment card from your graveyard to the battlefield.
49          Ability ability = new SimpleActivatedAbility(
50                  new ReturnFromGraveyardToBattlefieldTargetEffect(),
51                  new ManaCostsImpl<>("{W}{U}{B}{R}{G}")
52          );
53          ability.addCost(new TapSourceCost());
54          ability.addTarget(new TargetCardInYourGraveyard(filter));
55          this.addAbility(ability);
56  
57          // Whenever Go-Shintai of Life's Origin or another nontoken Shrine enters the battlefield under your control, create a 1/1 colorless Shrine enchantment creature token.
58          this.addAbility(new EntersBattlefieldThisOrAnotherTriggeredAbility(
59                  new CreateTokenEffect(new ShrineToken()),
60                  filter2, false, true
61          ));
62      }
63  
64      private GoShintaiOfLifesOrigin(final GoShintaiOfLifesOrigin card) {
65          super(card);
66      }
67  
68      @Override
69      public GoShintaiOfLifesOrigin copy() {
70          return new GoShintaiOfLifesOrigin(this);
71      }
72  }

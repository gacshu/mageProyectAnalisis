27   * @author LevelX2
28   */
29  public final class GenjuOfTheFields extends CardImpl {
30  
31      private static final FilterLandPermanent FILTER = new FilterLandPermanent(SubType.PLAINS, "Plains");
32  
33      public GenjuOfTheFields(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{W}");
35          this.subtype.add(SubType.AURA);
36  
37          // Enchant Plains
38          TargetPermanent auraTarget = new TargetLandPermanent(FILTER);
39          this.getSpellAbility().addTarget(auraTarget);
40          this.getSpellAbility().addEffect(new AttachEffect(Outcome.PutCreatureInPlay));
41          Ability ability = new EnchantAbility(auraTarget);
42          this.addAbility(ability);
43  
44          // {2}: Until end of turn, enchanted Plains becomes a 2/5 white Spirit creature with "Whenever this creature deals damage, its controller gains that much life." It's still a land.
45          Effect effect = new BecomesCreatureAttachedWithActivatedAbilityOrSpellEffect(new SpiritToken(),
46                  "Until end of turn, enchanted Plains becomes a 2/5 white Spirit creature", Duration.EndOfTurn);
47          Ability ability2 = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new GenericManaCost(2));
48          effect = new GainAbilityAttachedEffect(new DealsDamageGainLifeSourceTriggeredAbility(), AttachmentType.AURA, Duration.EndOfTurn);
49          effect.setText("with \"Whenever this creature deals damage, its controller gains that much life.\" It's still a land");
50          ability2.addEffect(effect);
51          this.addAbility(ability2);
52  
53          // When enchanted Plains is put into a graveyard, you may return Genju of the Fields from your graveyard to your hand.
54          Ability ability3 = new DiesAttachedTriggeredAbility(new ReturnToHandSourceEffect(false, true), "enchanted Plains", true, false);
55          this.addAbility(ability3);
56      }
57  
58      private GenjuOfTheFields(final GenjuOfTheFields card) {
59          super(card);
60      }
61  
62      @Override
63      public GenjuOfTheFields copy() {
64          return new GenjuOfTheFields(this);
65      }
66  
67      private static class SpiritToken extends TokenImpl {
68  
69          SpiritToken() {
70              super("Spirit", "2/5 white Spirit creature");
71              cardType.add(CardType.CREATURE);
72              color.setWhite(true);
73              subtype.add(SubType.SPIRIT);
74              power = new MageInt(2);
75              toughness = new MageInt(5);
76          }
77          public SpiritToken(final SpiritToken token) {
78              super(token);
79          }
80  
81          public SpiritToken copy() {
82              return new SpiritToken(this);
83          }
84      }
85  }

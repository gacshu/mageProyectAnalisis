17  import mage.constants.Outcome;
18  import mage.target.common.TargetControlledCreaturePermanent;
19  
20  /**
21   * @author TheElk801
22   */
23  public final class GhostLantern extends AdventureCard {
24  
25      public GhostLantern(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, new CardType[]{CardType.INSTANT}, "{B}", "Bind Spirit", "{1}{B}");
27  
28          this.subtype.add(SubType.EQUIPMENT);
29  
30          // Whenever a creature you control dies, put a +1/+1 counter on equipped creature.
31          this.addAbility(new DiesCreatureTriggeredAbility(new AddCountersAttachedEffect(
32                  CounterType.P1P1.createInstance(), "equipped creature"
33          ), false, StaticFilters.FILTER_CONTROLLED_A_CREATURE));
34  
35          // Equip {1}
36          this.addAbility(new EquipAbility(Outcome.BoostCreature, new GenericManaCost(1), new TargetControlledCreaturePermanent(), false));
37  
38          // Bind Spirit
39          // Return target creature card from your graveyard to your hand.
40          this.getSpellCard().getSpellAbility().addEffect(new ReturnFromGraveyardToHandTargetEffect());
41          this.getSpellCard().getSpellAbility().addTarget(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD));
42      }
43  
44      private GhostLantern(final GhostLantern card) {
45          super(card);
46      }
47  
48      @Override
49      public GhostLantern copy() {
50          return new GhostLantern(this);
51      }
52  }

18  public final class GraveUpheaval extends CardImpl {

20      public GraveUpheaval(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{B}{R}");
22  
23          // Put target creature card from a graveyard onto the battlefield under your control. It gains haste.
24          this.getSpellAbility().addEffect(new ReturnCreatureFromGraveyardToBattlefieldAndGainHasteEffect());
25          this.getSpellAbility().addTarget(new TargetCardInGraveyard(StaticFilters.FILTER_CARD_CREATURE));
26  
27          // Basic landcycling {2}
28          this.addAbility(new BasicLandcyclingAbility(new ManaCostsImpl<>("{2}")));
29      }
30

32          super(card);
33      }

38      }
39  }

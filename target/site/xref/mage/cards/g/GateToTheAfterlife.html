36  public final class GateToTheAfterlife extends CardImpl {

38      public GateToTheAfterlife(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
40  
41          // Whenever a nontoken creature you control dies, you gain 1 life. Then you may draw a card. If you do, discard a card.
42          Ability ability = new DiesCreatureTriggeredAbility(new GainLifeEffect(1), false, StaticFilters.FILTER_CONTROLLED_CREATURE_NON_TOKEN, false);
43          Effect effect = new DrawDiscardControllerEffect(1, 1, true);
44          effect.setText("Then you may draw a card. If you do, discard a card");
45          ability.addEffect(effect);
46          this.addAbility(ability);
47  
48          // {2}, {T}, Sacrifice Gate to the Afterlife: Search your graveyard, hand, and/or library for a card named God-Pharaoh's Gift and put it onto the battlefield. If you seearch your library this way, shuffle it. Activate this ability only if there are six or more creature cards in your graveyard.
49          ability = new ConditionalActivatedAbility(
50                  Zone.BATTLEFIELD, new GateToTheAfterlifeEffect(), new GenericManaCost(2),
51                  new CardsInControllerGraveyardCondition(6, StaticFilters.FILTER_CARD_CREATURES)
52          );
53          ability.addCost(new TapSourceCost());
54          ability.addCost(new SacrificeSourceCost());
55          this.addAbility(ability);
56      }

60      }

68  class GateToTheAfterlifeEffect extends OneShotEffect {
69  
70      static private String cardName = "God-Pharaoh's Gift";
71  
72      public GateToTheAfterlifeEffect() {
73          super(Outcome.Benefit);
74          this.staticText = "Search your graveyard, hand, and/or library for a card named "
75                  + cardName
76                  + " and put it onto the battlefield. If you search your library this way, shuffle";
77      }
78  
79      public GateToTheAfterlifeEffect(final GateToTheAfterlifeEffect effect) {
80          super(effect);
81      }
82  
83      @Override
84      public GateToTheAfterlifeEffect copy() {
85          return new GateToTheAfterlifeEffect(this);
86      }
87  
88      @Override
89      public boolean apply(Game game, Ability source) {
90          Player controller = game.getPlayer(source.getControllerId());
91          if (controller == null) {
92              return false;
93          }
94          FilterCard filter = new FilterCard("card named " + cardName);
95          filter.add(new NamePredicate(cardName));
96          Card card = null;
97          // Graveyard check
98          if (controller.chooseUse(Outcome.Benefit, "Search your graveyard for " + cardName + "?", source, game)) {
99              TargetCardInYourGraveyard target = new TargetCardInYourGraveyard(1, 1, filter, true);
100             if (controller.choose(outcome, controller.getGraveyard(), target, game)) {
101                 card = game.getCard(target.getFirstTarget());
102             }
103         }
104         // Hand check
105         if (card == null && controller.chooseUse(Outcome.Benefit, "Search your hand for " + cardName + "?", source, game)) {
106             TargetCardInHand target = new TargetCardInHand(0, 1, filter);
107             if (controller.choose(Outcome.PutCardInPlay, controller.getHand(), target, game)) {
108                 card = game.getCard(target.getFirstTarget());
109             }
110         }
111         // Library check
112         boolean librarySearched = false;
113         if (card == null && controller.chooseUse(Outcome.Benefit, "Search your library for " + cardName + "?", source, game)) {
114             librarySearched = true;
115             TargetCardInLibrary target = new TargetCardInLibrary(filter);
116             if (controller.searchLibrary(target, source, game)) {
117                 card = game.getCard(target.getFirstTarget());
118             }
119             controller.shuffleLibrary(source, game);
120         }
121         if (card != null) {
122             controller.moveCards(card, Zone.BATTLEFIELD, source, game);
123         }
124         if (librarySearched) {
125             controller.shuffleLibrary(source, game);
126         }
127         return true;
128     }

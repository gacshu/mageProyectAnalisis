1   package mage.cards.g;

3   import java.util.UUID;
4   import mage.MageInt;
5   import mage.abilities.TriggeredAbility;
6   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
7   import mage.abilities.costs.mana.ManaCostsImpl;
8   import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
9   import mage.abilities.effects.common.ReturnToHandTargetEffect;
10  import mage.abilities.keyword.MadnessAbility;
11  import mage.cards.Card;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.filter.common.FilterCreatureCard;
17  import mage.target.common.TargetCardInGraveyard;
18  
19  public final class GraveScrabbler extends CardImpl {
20  
21      public GraveScrabbler(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{B}");
23          this.subtype.add(SubType.ZOMBIE);
24  
25          this.power = new MageInt(2);
26          this.toughness = new MageInt(2);
27  
28          //Madness {1}{B}
29          this.addAbility(new MadnessAbility(new ManaCostsImpl<>("{1}{B}")));
30  
31          //When Grave Scrabbler enters the battlefield, if its madness cost was paid,
32          //you may return target creature card from a graveyard to its owner's hand.
33          TriggeredAbility ability = new EntersBattlefieldTriggeredAbility(new ReturnToHandTargetEffect(), true);
34          ability.addTarget(new TargetCardInGraveyard(new FilterCreatureCard("creature card in a graveyard")));
35          this.addAbility(new ConditionalInterveningIfTriggeredAbility(ability, MadnessAbility.getCondition(),
36                  "When {this} enters the battlefield, if its madness cost was paid, you may return target creature card from a graveyard to its owner's hand."));
37      }
38  
39      private GraveScrabbler(final GraveScrabbler card) {
40          super(card);
41      }
42  
43      @Override
44      public Card copy() {
45          return new GraveScrabbler(this);
46      }
47  
48  }

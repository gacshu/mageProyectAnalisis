30  public final class GishathSunsAvatar extends CardImpl {
31  
32      public GishathSunsAvatar(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{5}{R}{G}{W}");
34  
35          addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.DINOSAUR);
37          this.subtype.add(SubType.AVATAR);
38          this.power = new MageInt(7);
39          this.toughness = new MageInt(6);
40  
41          // Trample
42          this.addAbility(TrampleAbility.getInstance());
43  
44          // Vigilance
45          this.addAbility(VigilanceAbility.getInstance());
46  
47          // Haste
48          this.addAbility(HasteAbility.getInstance());
49  
50          // Whenever Gishath, Sun's Avatar deals combat damage to a player, reveal that many cards from the top of your library. Put any number of Dinosaur creature cards from among them onto the battlefield and the rest on the bottom of your library in a random order.
51          this.addAbility(new DealsCombatDamageToAPlayerTriggeredAbility(new GishathSunsAvatarEffect(), false, true));
52      }
53

64  class GishathSunsAvatarEffect extends OneShotEffect {
65  
66      GishathSunsAvatarEffect() {
67          super(Outcome.Benefit);
68          this.staticText = "reveal that many cards from the top of your library. Put any number of Dinosaur creature cards from among them onto the battlefield and the rest on the bottom of your library in a random order";
69      }
70  
71      GishathSunsAvatarEffect(final GishathSunsAvatarEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public GishathSunsAvatarEffect copy() {
77          return new GishathSunsAvatarEffect(this);
78      }
79  
80      @Override
81      public boolean apply(Game game, Ability source) {
82          Player controller = game.getPlayer(source.getControllerId());
83          if (controller == null) {
84              return false;
85          }
86          int xValue = (Integer) getValue("damage");
87          Cards cards = new CardsImpl(controller.getLibrary().getTopCards(game, xValue));
88          if (!cards.isEmpty()) {
89              controller.revealCards(source, cards, game);
90              FilterCreatureCard filter = new FilterCreatureCard("Dinosaur creature cards");
91              filter.add(SubType.DINOSAUR.getPredicate());
92              TargetCard target1 = new TargetCard(0, Integer.MAX_VALUE, Zone.LIBRARY, filter);
93              target1.setNotTarget(true);
94              controller.choose(Outcome.PutCardInPlay, cards, target1, game);
95              Cards toBattlefield = new CardsImpl(target1.getTargets());
96              cards.removeAll(toBattlefield);
97              controller.moveCards(toBattlefield.getCards(game), Zone.BATTLEFIELD, source, game, false, false, false, null);
98              controller.putCardsOnBottomOfLibrary(cards, game, source, false);
99          }
100         return true;
101     }

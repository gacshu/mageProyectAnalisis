24  /**
25   * @author TheElk801
26   */
27  public final class GoblinMorningstar extends CardImpl {
28  
29      public GoblinMorningstar(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{1}{R}");
31  
32          this.subtype.add(SubType.EQUIPMENT);
33  
34          // Equipped creature gets +1/+0 and has trample.
35          Ability ability = new SimpleStaticAbility(new BoostEquippedEffect(1, 0));
36          ability.addEffect(new GainAbilityAttachedEffect(
37                  TrampleAbility.getInstance(), AttachmentType.EQUIPMENT
38          ).setText("and has trample"));
39          this.addAbility(ability);
40  
41          // Equip {1}{R}
42          this.addAbility(new EquipAbility(Outcome.BoostCreature, new ManaCostsImpl<>("{1}{R}"), false));
43  
44          // When Goblin Morningstar enters the battlefield, roll a d20.
45          RollDieWithResultTableEffect effect = new RollDieWithResultTableEffect();
46          this.addAbility(new EntersBattlefieldTriggeredAbility(effect));
47  
48          // 1-9 | Create a 1/1 red Goblin creature token.
49          effect.addTableEntry(1, 9, new CreateTokenEffect(new GoblinToken()));
50  
51          // 10-20 | Create a 1/1 red Goblin creature token, then attach Goblin Morningstar more it.
52          effect.addTableEntry(10, 20, new CreateTokenAttachSourceEffect(new GoblinToken()));
53      }
54  
55      private GoblinMorningstar(final GoblinMorningstar card) {
56          super(card);
57      }
58  
59      @Override
60      public GoblinMorningstar copy() {
61          return new GoblinMorningstar(this);
62      }
63  }

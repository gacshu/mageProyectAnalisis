19  public final class GameOfChaos extends CardImpl {
20  
21      public GameOfChaos(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{R}{R}{R}");
23  
24          // Flip a coin. 
25          // If you win the flip, you gain 1 life and target opponent loses 1 life, and you decide whether to flip again. 
26          // If you lose the flip, you lose 1 life and that opponent gains 1 life, and that player decides whether to flip again. 
27          // Double the life stakes with each flip.
28          this.getSpellAbility().addEffect(new GameOfChaosEffect());
29          this.getSpellAbility().addTarget(new TargetOpponent());
30      }
31  
32      private GameOfChaos(final GameOfChaos card) {
33          super(card);
34      }
35  
36      @Override
37      public GameOfChaos copy() {
38          return new GameOfChaos(this);
39      }
40  }

42  class GameOfChaosEffect extends OneShotEffect {
43  
44      public GameOfChaosEffect() {
45          super(Outcome.Detriment);
46          this.staticText = "Flip a coin. If you win the flip, you gain 1 life and target opponent loses 1 life, and you decide whether to flip again. If you lose the flip, you lose 1 life and that opponent gains 1 life, and that player decides whether to flip again. Double the life stakes with each flip.";
47      }
48  
49      public GameOfChaosEffect(final GameOfChaosEffect effect) {
50          super(effect);
51      }
52  
53      @Override
54      public GameOfChaosEffect copy() {
55          return new GameOfChaosEffect(this);
56      }
57  
58      @Override
59      public boolean apply(Game game, Ability source) {
60          Player you = game.getPlayer(source.getControllerId());
61          Player targetOpponent = game.getPlayer(getTargetPointer().getFirst(game, source));
62  
63          if (you != null && targetOpponent != null) {
64  
65              boolean continueFlipping = true;
66              boolean youWonFlip = you.flipCoin(source, game, true); // controller flips first
67              boolean youWonLastFlip = false; // tracks if you won the flip last, negation of it means opponent won last
68              int lifeAmount = 1; // starts stakes with 1 life
69  
70              while (continueFlipping) {
71                  if (youWonFlip) { // flipper of coin wins, flipper gain 1 and non-flipper loses 1
72                      handleLifeChangesFromFlip(game, you, targetOpponent, lifeAmount, source);
73                      if (!cannotContinueFlipping(you, targetOpponent)) {
74                          continueFlipping = you.chooseUse(outcome, "You gained " + lifeAmount + " life! Flip again for double the life stakes?", source, game);
75                          youWonLastFlip = true;
76                      }
77                  } else { // non-flipper wins, flipper lose 1 and non-flipper gains 1
78                      handleLifeChangesFromFlip(game, targetOpponent, you, lifeAmount, source);
79                      if (!cannotContinueFlipping(you, targetOpponent)) {
80                          continueFlipping = targetOpponent.chooseUse(outcome, "You gained " + lifeAmount + " life! Flip again for double the life stakes?", source, game);
81                          youWonLastFlip = false;
82                      }
83                  }
84  
85                  if (cannotContinueFlipping(you, targetOpponent)) {
86                      continueFlipping = false;
87                  }
88  
89                  if (continueFlipping) {
90                      lifeAmount *= 2; // double the life each time
91                      youWonFlip = youWonLastFlip ? you.flipCoin(source, game, true) : !targetOpponent.flipCoin(source, game, true); // negate the opponent's results for proper evaluation of if you won in next iteration
92                  }
93              }
94  
95              return true;
96          }
97          return false;
98      }
99  
100     private void handleLifeChangesFromFlip(Game game, Player playerGainingLife, Player playerLosingLife, int lifeAmount, Ability source) {
101         playerGainingLife.gainLife(lifeAmount, game, source);
102         playerLosingLife.loseLife(lifeAmount, game, source, false);
103     }
104 
105     private boolean cannotContinueFlipping(Player you, Player opponent) {
106         return (!you.canRespond() || !opponent.canRespond() || (you.canLoseByZeroOrLessLife() && you.getLife() <= 0) || (opponent.canLoseByZeroOrLessLife() && opponent.getLife() <= 0));
107     }
108 }

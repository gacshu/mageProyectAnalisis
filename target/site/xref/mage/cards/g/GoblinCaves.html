24   * @author nigelzor
25   */
26  public final class GoblinCaves extends CardImpl {
27  
28      public GoblinCaves(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{R}{R}");
30          this.subtype.add(SubType.AURA);
31  
32          // Enchant land
33          TargetPermanent auraTarget = new TargetLandPermanent();
34          this.getSpellAbility().addTarget(auraTarget);
35          this.getSpellAbility().addEffect(new AttachEffect(Outcome.AddAbility));
36          this.addAbility(new EnchantAbility(auraTarget));
37          // As long as enchanted land is a basic Mountain, Goblin creatures get +0/+2.
38          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new ConditionalContinuousEffect(
39                  new BoostAllEffect(0, 2, Duration.WhileOnBattlefield, StaticFilters.FILTER_PERMANENT_CREATURE_GOBLINS, false),
40                  new AttachedToBasicMountainCondition(),
41                  "As long as enchanted land is a basic Mountain, Goblin creatures get +0/+2"
42          )));
43      }
44  
45      private GoblinCaves(final GoblinCaves card) {
46          super(card);
47      }
48  
49      @Override
50      public GoblinCaves copy() {
51          return new GoblinCaves(this);
52      }
53  }
54  
55  class AttachedToBasicMountainCondition implements Condition {
56  
57      @Override
58      public boolean apply(Game game, Ability source) {
59          Permanent enchantment = game.getPermanent(source.getSourceId());
60          if (enchantment != null) {
61              Permanent enchanted = game.getPermanent(enchantment.getAttachedTo());
62              if (enchanted != null) {
63                  if (enchanted.hasSubtype(SubType.MOUNTAIN, game) && enchanted.isBasic()) {
64                      return true;
65                  }
66              }
67          }
68          return false;
69      }
70  }

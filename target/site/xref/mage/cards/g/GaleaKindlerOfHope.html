29  public final class GaleaKindlerOfHope extends CardImpl {
30  
31      private static final FilterCard filter = new FilterCard("cast Aura and Equipment spells");
32  
33      static {
34          filter.add(Predicates.or(
35                  SubType.AURA.getPredicate(),
36                  SubType.EQUIPMENT.getPredicate()
37          ));
38      }
39  
40      public GaleaKindlerOfHope(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{G}{W}{U}");
42  
43          this.addSuperType(SuperType.LEGENDARY);
44          this.subtype.add(SubType.ELF);
45          this.subtype.add(SubType.KNIGHT);
46          this.power = new MageInt(4);
47          this.toughness = new MageInt(4);
48  
49          // Vigilance
50          this.addAbility(VigilanceAbility.getInstance());
51  
52          // You may look at the top card of your library any time.
53          this.addAbility(new SimpleStaticAbility(new LookAtTopCardOfLibraryAnyTimeEffect()));
54  
55          // You may cast Aura and Equipment spells from the top of your library. When you cast an Equipment spell this way, it gains "When this Equipment enters the battlefield, attach it to target creature you control."
56          Ability ability = new SimpleStaticAbility(new PlayTheTopCardEffect(TargetController.YOU, filter, false));
57          ability.addEffect(new InfoEffect("When you cast an Equipment spell this way, it gains " +
58                  "\"When this Equipment enters the battlefield, attach it to target creature you control.\""));
59          this.addAbility(ability);
60          this.addAbility(new GaleaKindlerOfHopeTriggeredAbility());
61      }
62  
63      private GaleaKindlerOfHope(final GaleaKindlerOfHope card) {
64          super(card);
65      }
66  
67      @Override
68      public GaleaKindlerOfHope copy() {
69          return new GaleaKindlerOfHope(this);
70      }
71  }

73  class GaleaKindlerOfHopeTriggeredAbility extends TriggeredAbilityImpl {
74  
75      public GaleaKindlerOfHopeTriggeredAbility() {
76          super(Zone.BATTLEFIELD, null);
77          this.setRuleVisible(false);
78      }
79  
80      private GaleaKindlerOfHopeTriggeredAbility(final GaleaKindlerOfHopeTriggeredAbility ability) {
81          super(ability);
82      }
83  
84      @Override
85      public boolean checkEventType(GameEvent event, Game game) {
86          return event.getType() == GameEvent.EventType.SPELL_CAST;
87      }
88  
89      @Override
90      public boolean checkTrigger(GameEvent event, Game game) {
91          if (!isControlledBy(event.getPlayerId())
92                  || event.getZone() != Zone.LIBRARY
93                  || !event
94                  .getAdditionalReference()
95                  .getApprovingMageObjectReference()
96                  .refersTo(this.getSourceObject(game), game)) {
97              return false;
98          }
99          Spell spell = game.getSpell(event.getTargetId());
100         if (spell == null || !spell.hasSubtype(SubType.EQUIPMENT, game)) {
101             return false;
102         }
103         this.getEffects().clear();
104         this.addEffect(new GaleaKindlerOfHopeEffect(spell, game));
105         return true;
106     }
107 
108     @Override
109     public GaleaKindlerOfHopeTriggeredAbility copy() {
110         return new GaleaKindlerOfHopeTriggeredAbility(this);
111     }
112 
113     @Override
114     public String getRule() {
115         return "When you cast an Equipment spell this way, it gains " +
116                 "\"When this Equipment enters the battlefield, attach it to target creature you control.\"";
117     }
118 }

120 class GaleaKindlerOfHopeEffect extends ContinuousEffectImpl {
121 
122     private final MageObjectReference spellRef;
123     private final MageObjectReference permRef;
124     private final Ability ability = new EntersBattlefieldAttachToTarget()
125             .setTriggerPhrase("When this Equipment enters the battlefield, ");
126 
127     GaleaKindlerOfHopeEffect(Spell spell, Game game) {
128         super(Duration.Custom, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.Benefit);
129         this.spellRef = new MageObjectReference(spell, game);
130         this.permRef = new MageObjectReference(spell.getMainCard(), game, 1);
131     }
132 
133     private GaleaKindlerOfHopeEffect(final GaleaKindlerOfHopeEffect effect) {
134         super(effect);
135         this.spellRef = effect.spellRef;
136         this.permRef = effect.permRef;
137     }
138 
139     @Override
140     public GaleaKindlerOfHopeEffect copy() {
141         return new GaleaKindlerOfHopeEffect(this);
142     }
143 
144     @Override
145     public boolean apply(Game game, Ability source) {
146         Spell spell = game.getSpell(spellRef.getSourceId());
147         if (spell != null && spell.getZoneChangeCounter(game) == spellRef.getZoneChangeCounter()) {
148             game.getState().addOtherAbility(spell.getCard(), ability);
149             return true;
150         }
151         Permanent permanent = permRef.getPermanent(game);
152         if (permanent == null) {
153             discard();
154             return false;
155         }
156         permanent.addAbility(ability, source.getSourceId(), game);
157         return true;
158     }
159 }

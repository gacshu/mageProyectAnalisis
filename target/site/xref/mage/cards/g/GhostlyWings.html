29  public final class GhostlyWings extends CardImpl {
30  
31      public GhostlyWings(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{U}");
33          this.subtype.add(SubType.AURA);
34  
35          // Enchant creature
36          TargetPermanent auraTarget = new TargetCreaturePermanent();
37          this.getSpellAbility().addTarget(auraTarget);
38          this.getSpellAbility().addEffect(new AttachEffect(Outcome.AddAbility));
39          Ability ability = new EnchantAbility(auraTarget);
40          this.addAbility(ability);
41  
42          // Enchanted creature gets +1/+1 and has flying.
43          Effect effect = new BoostEnchantedEffect(1, 1, Duration.WhileOnBattlefield);
44          effect.setText("Enchanted creature gets +1/+1");
45          ability = new SimpleStaticAbility(Zone.BATTLEFIELD, effect);
46          effect = new GainAbilityAttachedEffect(FlyingAbility.getInstance(), AttachmentType.AURA);
47          effect.setText("and has flying");
48          ability.addEffect(effect);
49          this.addAbility(ability);
50          // Discard a card: Return enchanted creature to its owner's hand.
51          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new GhostlyWingsReturnEffect(), new DiscardCardCost()));
52  
53      }
54  
55      private GhostlyWings(final GhostlyWings card) {
56          super(card);
57      }
58  
59      @Override
60      public GhostlyWings copy() {
61          return new GhostlyWings(this);
62      }
63  }
64  
65  class GhostlyWingsReturnEffect extends OneShotEffect {
66  
67      public GhostlyWingsReturnEffect() {
68          super(Outcome.ReturnToHand);
69          staticText = "Return enchanted creature to its owner's hand";
70      }
71  
72      public GhostlyWingsReturnEffect(final GhostlyWingsReturnEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public GhostlyWingsReturnEffect copy() {
78          return new GhostlyWingsReturnEffect(this);
79      }
80  
81      @Override
82      public boolean apply(Game game, Ability source) {
83          Permanent permanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
84          Player controller = game.getPlayer(source.getControllerId());
85          if (controller != null && permanent != null && permanent.getAttachedTo() != null) {
86              Permanent enchantedCreature = game.getPermanent(permanent.getAttachedTo());
87              if (enchantedCreature != null) {
88                  controller.moveCards(enchantedCreature, Zone.HAND, source, game);
89              }
90              return true;
91          }
92          return false;
93      }
94  }

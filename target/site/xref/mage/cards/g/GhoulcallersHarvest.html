25  public final class GhoulcallersHarvest extends CardImpl {
26  
27      private static final Hint hint = new ValueHint(
28              "Half the number of creature cards in your graveyard", GhoulcallersHarvestValue.instance
29      );
30  
31      public GhoulcallersHarvest(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{B}{G}");
33  
34          // Create X 2/2 black Zombie creature tokens with decayed, where X is half the number of creature cards in your graveyard, rounded up.
35          this.getSpellAbility().addEffect(new CreateTokenEffect(
36                  new ZombieDecayedToken(), GhoulcallersHarvestValue.instance
37          ).setText("create X 2/2 black Zombie creature tokens with decayed, " +
38                  "where X is half the number of creature cards in your graveyard, rounded up"));
39          this.getSpellAbility().addHint(hint);
40  
41          // Flashback {3}{B}{G}
42          this.addAbility(new FlashbackAbility(this, new ManaCostsImpl<>("{3}{B}{G}")));
43      }
44  
45      private GhoulcallersHarvest(final GhoulcallersHarvest card) {
46          super(card);
47      }
48  
49      @Override
50      public GhoulcallersHarvest copy() {
51          return new GhoulcallersHarvest(this);
52      }
53  }
54  
55  enum GhoulcallersHarvestValue implements DynamicValue {
56      instance;
57  
58      @Override
59      public int calculate(Game game, Ability sourceAbility, Effect effect) {
60          Player player = game.getPlayer(sourceAbility.getControllerId());
61          if (player == null) {
62              return 0;
63          }
64          int amount = player.getGraveyard().count(StaticFilters.FILTER_CARD_CREATURE, game);
65          return Math.floorDiv(amount, 2) + amount % 2;
66      }
67  
68      @Override
69      public GhoulcallersHarvestValue copy() {
70          return this;
71      }
72  
73      @Override
74      public String getMessage() {
75          return "";
76      }
77  }

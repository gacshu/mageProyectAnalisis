28  public final class GeminiEngine extends CardImpl {
29  
30      public GeminiEngine(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{6}");
32  
33          this.subtype.add(SubType.CONSTRUCT);
34          this.power = new MageInt(3);
35          this.toughness = new MageInt(4);
36  
37          // Whenever Gemini Engine attacks, create a colorless Construct artifact creature token named Twin that's attacking. Its power is equal to Gemini Engine's power and its toughness is equal to Gemini Engine's toughness. Sacrifice the token at end of combat.
38          this.addAbility(new AttacksTriggeredAbility(new GeminiEngineCreateTokenEffect(), false));
39      }
40  
41      private GeminiEngine(final GeminiEngine card) {
42          super(card);
43      }
44  
45      @Override
46      public GeminiEngine copy() {
47          return new GeminiEngine(this);
48      }
49  }
50  
51  class GeminiEngineCreateTokenEffect extends OneShotEffect {
52  
53      GeminiEngineCreateTokenEffect() {
54          super(Outcome.Benefit);
55          this.staticText = "create a colorless Construct artifact creature token named Twin that's attacking. Its power is equal to {this}'s power and its toughness is equal to {this}'s toughness. Sacrifice the token at end of combat.";
56      }
57  
58      GeminiEngineCreateTokenEffect(final GeminiEngineCreateTokenEffect effect) {
59          super(effect);
60      }
61  
62      @Override
63      public GeminiEngineCreateTokenEffect copy() {
64          return new GeminiEngineCreateTokenEffect(this);
65      }
66  
67      @Override
68      public boolean apply(Game game, Ability source) {
69          Player player = game.getPlayer(source.getControllerId());
70          if (player == null) {
71              return false;
72          }
73          Permanent permanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
74          Token token;
75          if (permanent != null) {
76              token = new GeminiEngineTwinToken(permanent.getPower().getValue(), permanent.getToughness().getValue());
77          } else {
78              token = new GeminiEngineTwinToken(0, 0);
79          }
80          token.putOntoBattlefield(1, game, source, source.getControllerId(), false, true);
81          for (UUID tokenId : token.getLastAddedTokenIds()) {
82              Permanent tokenPerm = game.getPermanent(tokenId);
83              if (tokenPerm != null) {
84                  Effect effect = new SacrificeTargetEffect("sacrifice " + tokenPerm.getLogName(), player.getId());
85                  effect.setTargetPointer(new FixedTarget(tokenPerm, game));
86                  game.addDelayedTriggeredAbility(new AtTheEndOfCombatDelayedTriggeredAbility(effect), source);
87              }
88          }
89          return true;
90      }
91  }

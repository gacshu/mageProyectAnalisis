16  import mage.cards.CardSetInfo;
17  import mage.constants.*;
18  import mage.filter.FilterPermanent;
19  import mage.filter.common.FilterCreaturePermanent;
20  import mage.filter.predicate.mageobject.ColorPredicate;
21  import mage.target.TargetPermanent;
22  import mage.target.common.TargetCreaturePermanent;

24  /**
25   *
26   * @author Zeplar1_at_googlemail.com
27   */
28  public final class GreenScarab extends CardImpl {
29      
30      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("green creatures");
31      private static final FilterPermanent filter2 = new FilterPermanent("green permanent");
32  
33      static {
34          filter.add(new ColorPredicate(ObjectColor.GREEN));
35          filter2.add(new ColorPredicate(ObjectColor.GREEN));
36      }
37  
38      public GreenScarab(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{W}");
40          this.subtype.add(SubType.AURA);
41  
42          // Enchant creature
43          TargetPermanent auraTarget = new TargetCreaturePermanent();
44          this.getSpellAbility().addTarget(auraTarget);
45          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
46          Ability ability = new EnchantAbility(auraTarget);
47          this.addAbility(ability);
48  
49          // Enchanted creature can't be blocked by green creatures.
50          Effect effect = new CantBeBlockedByCreaturesAttachedEffect(Duration.WhileOnBattlefield, filter, AttachmentType.AURA);
51          ability = new SimpleStaticAbility(Zone.BATTLEFIELD, effect);
52          this.addAbility(ability);
53                 
54          // Enchanted creature gets +2/+2 as long as an opponent controls a white permanent.
55          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new ConditionalContinuousEffect(
56                  new BoostEnchantedEffect(2, 2, Duration.WhileOnBattlefield),
57                  new OpponentControlsPermanentCondition(filter2),
58                  "Enchanted creature gets +2/+2 as long as an opponent controls a green permanent")));
59      }
60  
61      private GreenScarab(final GreenScarab card) {
62          super(card);
63      }
64  
65      @Override
66      public GreenScarab copy() {
67          return new GreenScarab(this);
68      }
69  }

18  import java.util.UUID;

20  /**
21   * @author TheElk801
22   */
23  public final class GriffinAerie extends CardImpl {
24  
25      private static final Condition condition = new YouGainedLifeCondition(ComparisonType.MORE_THAN, 2);
26      private static final Hint hint = new ConditionHint(condition, "You gained 3 or more life this turn");
27  
28      public GriffinAerie(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}");
30  
31          // At the beginning of your end step, if you gained 3 or more life this turn, create a 2/2 white Griffin creature token with flying.
32          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
33                  new BeginningOfEndStepTriggeredAbility(
34                          new CreateTokenEffect(new GriffinToken()), TargetController.YOU, false
35                  ), condition, "At the beginning of your end step, " +
36                  "if you gained 3 or more life this turn, create a 2/2 white Griffin creature token with flying."
37          ).addHint(hint), new PlayerGainedLifeWatcher());
38      }
39  
40      private GriffinAerie(final GriffinAerie card) {
41          super(card);
42      }
43  
44      @Override
45      public GriffinAerie copy() {
46          return new GriffinAerie(this);
47      }
48  }

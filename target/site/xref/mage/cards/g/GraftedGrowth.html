23  import java.util.UUID;
24  
25  /**
26   * @author TheElk801
27   */
28  public final class GraftedGrowth extends CardImpl {
29  
30      private static final FilterPermanent filter
31              = new FilterControlledPermanent("creature or Vehicle you control");
32  
33      static {
34          filter.add(Predicates.or(
35                  CardType.CREATURE.getPredicate(),
36                  SubType.VEHICLE.getPredicate()
37          ));
38      }
39  
40      public GraftedGrowth(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{G}");
42  
43          this.subtype.add(SubType.AURA);
44  
45          // Enchant land
46          TargetPermanent auraTarget = new TargetLandPermanent();
47          this.getSpellAbility().addTarget(auraTarget);
48          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
49          this.addAbility(new EnchantAbility(auraTarget));
50  
51          // When Grafted Growth enters the battlefield, put a +1/+1 counter on target creature or Vehicle you control.
52          Ability ability = new EntersBattlefieldTriggeredAbility(
53                  new AddCountersTargetEffect(CounterType.P1P1.createInstance())
54          );
55          ability.addTarget(new TargetPermanent(filter));
56          this.addAbility(ability);
57  
58          // Enchanted land has "{T}: Add two mana of any one color."
59          this.addAbility(new SimpleStaticAbility(new GainAbilityAttachedEffect(new SimpleManaAbility(
60                  Zone.BATTLEFIELD, new AddManaOfAnyColorEffect(2), new TapSourceCost()
61          ), AttachmentType.AURA).setText("enchanted land has \"{T}: Add two mana of any one color.\"")));
62      }
63  
64      private GraftedGrowth(final GraftedGrowth card) {
65          super(card);
66      }
67  
68      @Override
69      public GraftedGrowth copy() {
70          return new GraftedGrowth(this);
71      }
72  }

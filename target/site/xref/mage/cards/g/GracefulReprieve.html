3   import java.util.UUID;
4   import mage.abilities.DelayedTriggeredAbility;
5   import mage.abilities.common.delayed.WhenTargetDiesDelayedTriggeredAbility;
6   import mage.abilities.effects.common.CreateDelayedTriggeredAbilityEffect;
7   import mage.abilities.effects.common.ReturnToBattlefieldUnderOwnerControlTargetEffect;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.SetTargetPointer;
12  import mage.target.common.TargetCreaturePermanent;
13  
14  /**
15   *
16   * @author jeffwadsworth
17   */
18  public final class GracefulReprieve extends CardImpl {
19  
20      public GracefulReprieve(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{W}");
22  
23          // When target creature dies this turn, return that card to the battlefield under its owner's control.
24          DelayedTriggeredAbility ability = new WhenTargetDiesDelayedTriggeredAbility(
25                  new ReturnToBattlefieldUnderOwnerControlTargetEffect(false, false),
26                  SetTargetPointer.CARD
27          );
28          ability.setTriggerPhrase("When target creature dies this turn, ");
29          this.getSpellAbility().addEffect(new CreateDelayedTriggeredAbilityEffect(ability));
30          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
31      }
32  
33      private GracefulReprieve(final GracefulReprieve card) {
34          super(card);
35      }
36  
37      @Override
38      public GracefulReprieve copy() {
39          return new GracefulReprieve(this);
40      }
41  }

1   package mage.cards.g;

3   import mage.MageInt;
4   import mage.abilities.Ability;
5   import mage.abilities.common.BeginningOfUpkeepTriggeredAbility;
6   import mage.abilities.common.ExploitCreatureTriggeredAbility;
7   import mage.abilities.effects.common.DamageControllerEffect;
8   import mage.abilities.effects.common.DestroyTargetEffect;
9   import mage.abilities.keyword.ExploitAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.SubType;
14  import mage.constants.TargetController;
15  import mage.target.common.TargetPlaneswalkerPermanent;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class GrafReaver extends CardImpl {
23  
24      public GrafReaver(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{B}");
26  
27          this.subtype.add(SubType.ZOMBIE);
28          this.subtype.add(SubType.WARRIOR);
29          this.power = new MageInt(3);
30          this.toughness = new MageInt(3);
31  
32          // Exploit
33          this.addAbility(new ExploitAbility());
34  
35          // When Graf Reaver exploits a creature, destroy target planeswalker.
36          Ability ability = new ExploitCreatureTriggeredAbility(new DestroyTargetEffect());
37          ability.addTarget(new TargetPlaneswalkerPermanent());
38          this.addAbility(ability);
39  
40          // At the beginning of your upkeep, Graf Reaver deals 1 damage to you.
41          this.addAbility(new BeginningOfUpkeepTriggeredAbility(
42                  new DamageControllerEffect(1), TargetController.YOU, false
43          ));
44      }
45  
46      private GrafReaver(final GrafReaver card) {
47          super(card);
48      }
49  
50      @Override
51      public GrafReaver copy() {
52          return new GrafReaver(this);
53      }
54  }

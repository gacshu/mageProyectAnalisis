4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.BeginningOfYourEndStepTriggeredAbility;
8   import mage.abilities.common.SimpleActivatedAbility;
9   import mage.abilities.costs.common.SacrificeTargetCost;
10  import mage.abilities.costs.mana.ManaCostsImpl;
11  import mage.abilities.dynamicvalue.common.CountersSourceCount;
12  import mage.abilities.effects.common.DamageTargetEffect;
13  import mage.abilities.effects.common.counter.AddCountersSourceEffect;
14  import mage.cards.CardImpl;
15  import mage.cards.CardSetInfo;
16  import mage.constants.CardType;
17  import mage.constants.SubType;
18  import mage.constants.Zone;
19  import mage.counters.CounterType;
20  import mage.filter.common.FilterControlledLandPermanent;
21  import mage.filter.common.FilterControlledPermanent;
22  import mage.target.common.TargetControlledPermanent;
23  import mage.target.common.TargetPlayerOrPlaneswalker;
24  
25  /**
26   *
27   * @author Loki
28   */
29  public final class GoblinRazerunners extends CardImpl {
30  
31      private static final FilterControlledPermanent filter = new FilterControlledLandPermanent("a land");
32  
33      public GoblinRazerunners(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{R}");
35          this.subtype.add(SubType.GOBLIN);
36          this.subtype.add(SubType.WARRIOR);
37  
38          this.power = new MageInt(3);
39          this.toughness = new MageInt(4);
40  
41          // {1}{R}, Sacrifice a land: Put a +1/+1 counter on Goblin Razerunners.
42          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.P1P1.createInstance()), new ManaCostsImpl<>("{1}{R}"));
43          ability.addCost(new SacrificeTargetCost(new TargetControlledPermanent(filter)));
44          this.addAbility(ability);
45  
46          // At the beginning of your end step, you may have Goblin Razerunners deal damage equal to the number of +1/+1 counters on it to target player.
47          ability = new BeginningOfYourEndStepTriggeredAbility(new DamageTargetEffect(new CountersSourceCount(CounterType.P1P1)).setText("you may have {this} deal damage equal to the number of +1/+1 counters on it to target player or planeswalker"), true);
48          ability.addTarget(new TargetPlayerOrPlaneswalker());
49          this.addAbility(ability);
50      }
51  
52      private GoblinRazerunners(final GoblinRazerunners card) {
53          super(card);
54      }
55  
56      @Override
57      public GoblinRazerunners copy() {
58          return new GoblinRazerunners(this);
59      }
60  }

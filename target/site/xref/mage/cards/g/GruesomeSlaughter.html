12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.Duration;
16  import mage.constants.Zone;
17  import mage.filter.common.FilterCreaturePermanent;
18  import mage.filter.predicate.mageobject.ColorlessPredicate;
19  import mage.target.common.TargetCreaturePermanent;

21  /**
22   *
23   * @author fireshoes
24   */
25  public final class GruesomeSlaughter extends CardImpl {
26  
27      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("colorless creatures you control");
28  
29      static {
30          filter.add(ColorlessPredicate.instance);
31      }
32  
33      public GruesomeSlaughter(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{6}");
35  
36          // Until end of turn, colorless creatures you control gain "{T}: This creature deals damage equal to its power to target creature."
37          Effect effect = new DamageTargetEffect(new SourcePermanentPowerCount());
38          effect.setText("{this} deals damage equal to its power to target creature.");
39          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new TapSourceCost());
40          ability.addTarget(new TargetCreaturePermanent());
41          effect = new GainAbilityControlledEffect(ability, Duration.EndOfTurn, filter);
42          effect.setText("Until end of turn, colorless creatures you control gain \"{T}: This creature deals damage equal to its power to target creature.\"");
43          this.getSpellAbility().addEffect(effect);
44      }
45  
46      private GruesomeSlaughter(final GruesomeSlaughter card) {
47          super(card);
48      }
49  
50      @Override
51      public GruesomeSlaughter copy() {
52          return new GruesomeSlaughter(this);
53      }
54  }

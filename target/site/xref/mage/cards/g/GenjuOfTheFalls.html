27  public final class GenjuOfTheFalls extends CardImpl {
28  
29      private static final FilterLandPermanent FILTER = new FilterLandPermanent(SubType.ISLAND, "Island");
30  
31      public GenjuOfTheFalls(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{U}");
33          this.subtype.add(SubType.AURA);
34  
35          // Enchant Island
36          TargetPermanent auraTarget = new TargetLandPermanent(FILTER);
37          this.getSpellAbility().addTarget(auraTarget);
38          this.getSpellAbility().addEffect(new AttachEffect(Outcome.PutCreatureInPlay));
39          Ability ability = new EnchantAbility(auraTarget);
40          this.addAbility(ability);
41  
42          // {2}: Enchanted Island becomes a 3/2 blue Spirit creature with flying until end of turn. It's still a land.
43          Ability ability2 = new SimpleActivatedAbility(Zone.BATTLEFIELD, new BecomesCreatureAttachedWithActivatedAbilityOrSpellEffect(new SpiritToken(), "Enchanted Island becomes a 3/2 blue Spirit creature with flying until end of turn. It's still a land", Duration.EndOfTurn), new GenericManaCost(2));
44          this.addAbility(ability2);
45  
46          // When enchanted Island is put into a graveyard, you may return Genju of the Falls from your graveyard to your hand.        TargetPermanent auraTarget = new TargetLandPermanent(filter);
47          Ability ability3 = new DiesAttachedTriggeredAbility(new ReturnToHandSourceEffect(false, true), "enchanted Island", true, false);
48          this.addAbility(ability3);
49      }
50  
51      private GenjuOfTheFalls(final GenjuOfTheFalls card) {
52          super(card);
53      }
54  
55      @Override
56      public GenjuOfTheFalls copy() {
57          return new GenjuOfTheFalls(this);
58      }
59  
60      private static class SpiritToken extends TokenImpl {
61  
62          SpiritToken() {
63              super("Spirit", "3/2 blue Spirit creature with flying");
64              cardType.add(CardType.CREATURE);
65              color.setBlue(true);
66              subtype.add(SubType.SPIRIT);
67              power = new MageInt(3);
68              toughness = new MageInt(2);
69              addAbility(FlyingAbility.getInstance());
70          }
71  
72          public SpiritToken(final SpiritToken token) {
73              super(token);
74          }
75  
76          public SpiritToken copy() {
77              return new SpiritToken(this);
78          }
79      }
80  }

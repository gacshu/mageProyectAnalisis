25  /**
26   * @author TheElk801
27   */
28  public final class GutterShortcut extends CardImpl {
29  
30      public GutterShortcut(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "");
32  
33          this.subtype.add(SubType.AURA);
34          this.color.setBlue(true);
35          this.nightCard = true;
36  
37          // Enchant creature
38          TargetPermanent auraTarget = new TargetCreaturePermanent();
39          this.getSpellAbility().addTarget(auraTarget);
40          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
41          Ability ability = new EnchantAbility(auraTarget);
42          this.addAbility(ability);
43  
44          // Enchanted creature can't be blocked as long as it's attacking alone.
45          this.addAbility(new SimpleStaticAbility(new ConditionalRestrictionEffect(
46                  new CantBeBlockedAttachedEffect(AttachmentType.AURA), GutterShortcutCondition.instance,
47                  "enchanted creature can't be blocked as long as it's attacking alone"
48          )));
49  
50          // If Gutter Shortcut would be put into a graveyard from anywhere, exile it instead.
51          this.addAbility(new PutIntoGraveFromAnywhereSourceAbility(new ExileSourceEffect().setText("exile it instead")));
52      }
53  
54      private GutterShortcut(final GutterShortcut card) {
55          super(card);
56      }
57  
58      @Override
59      public GutterShortcut copy() {
60          return new GutterShortcut(this);
61      }
62  }
63  
64  enum GutterShortcutCondition implements Condition {
65      instance;
66  
67      @Override
68      public boolean apply(Game game, Ability source) {
69          Permanent permanent = source.getSourcePermanentIfItStillExists(game);
70          return permanent != null && game.getCombat().attacksAlone()
71                  && game.getCombat().getAttackers().contains(permanent.getAttachedTo());
72      }
73  }

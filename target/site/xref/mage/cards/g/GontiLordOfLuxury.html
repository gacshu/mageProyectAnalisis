30  public final class GontiLordOfLuxury extends CardImpl {
31  
32      public GontiLordOfLuxury(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{B}{B}");
34          addSuperType(SuperType.LEGENDARY);
35          this.subtype.add(SubType.AETHERBORN);
36          this.subtype.add(SubType.ROGUE);
37          this.power = new MageInt(2);
38          this.toughness = new MageInt(3);
39  
40          // Deathtouch
41          this.addAbility(DeathtouchAbility.getInstance());
42  
43          // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
44          // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
45          // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
46          Ability ability = new EntersBattlefieldTriggeredAbility(new GontiLordOfLuxuryEffect());
47          ability.addTarget(new TargetOpponent());
48          this.addAbility(ability);
49      }
50  
51      private GontiLordOfLuxury(final GontiLordOfLuxury card) {
52          super(card);
53      }
54  
55      @Override
56      public GontiLordOfLuxury copy() {
57          return new GontiLordOfLuxury(this);
58      }

61  class GontiLordOfLuxuryEffect extends OneShotEffect {
62  
63      private static final String VALUE_PREFIX = "ExileZones";
64  
65      public GontiLordOfLuxuryEffect() {
66          super(Outcome.Benefit);
67          this.staticText = "look at the top four cards of target opponent's library, exile one of them face down, then put the rest on the bottom of that library in a random order. You may look at and cast that card for as long as it remains exiled, and you may spend mana as though it were mana of any type to cast that spell";
68      }
69  
70      private GontiLordOfLuxuryEffect(final GontiLordOfLuxuryEffect effect) {
71          super(effect);
72      }
73  
74      @Override
75      public GontiLordOfLuxuryEffect copy() {
76          return new GontiLordOfLuxuryEffect(this);
77      }
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          Player controller = game.getPlayer(source.getControllerId());
82          Player opponent = game.getPlayer(getTargetPointer().getFirst(game, source));
83          MageObject sourceObject = source.getSourceObject(game);
84          if (controller == null || opponent == null || sourceObject == null) {
85              return false;
86          }
87          Cards topCards = new CardsImpl();
88          topCards.addAll(opponent.getLibrary().getTopCards(game, 4));
89          TargetCard target = new TargetCard(Zone.LIBRARY, new FilterCard("card to exile"));
90          controller.choose(outcome, topCards, target, game);
91          Card card = game.getCard(target.getFirstTarget());
92          if (card == null) {
93              controller.putCardsOnBottomOfLibrary(topCards, game, source, false);
94              return true;
95          }
96          topCards.remove(card);
97          // move card to exile
98          UUID exileZoneId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
99          card.setFaceDown(true, game);
100         if (controller.moveCardsToExile(card, source, game, false, exileZoneId, sourceObject.getIdName())) {
101             card.setFaceDown(true, game);
102             Set<UUID> exileZones = (Set<UUID>) game.getState().getValue(VALUE_PREFIX + source.getSourceId().toString());
103             if (exileZones == null) {
104                 exileZones = new HashSet<>();
105                 game.getState().setValue(VALUE_PREFIX + source.getSourceId().toString(), exileZones);
106             }
107             exileZones.add(exileZoneId);
108             // allow to cast the card
109             ContinuousEffect effect = new GontiLordOfLuxuryCastFromExileEffect();
110             effect.setTargetPointer(new FixedTarget(card.getId(), game));
111             game.addEffect(effect, source);
112             // and you may spend mana as though it were mana of any color to cast it
113             effect = new GontiLordOfLuxurySpendAnyManaEffect();
114             effect.setTargetPointer(new FixedTarget(card.getId(), game));
115             game.addEffect(effect, source);
116             // For as long as that card remains exiled, you may look at it
117             effect = new GontiLordOfLuxuryLookEffect(controller.getId());
118             effect.setTargetPointer(new FixedTarget(card.getId(), game));
119             game.addEffect(effect, source);
120         }
121         // then put the rest on the bottom of that library in a random order
122         controller.putCardsOnBottomOfLibrary(topCards, game, source, false);
123         return true;
124     }
125 }

127 class GontiLordOfLuxuryCastFromExileEffect extends AsThoughEffectImpl {
128 
129     public GontiLordOfLuxuryCastFromExileEffect() {
130         super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.Custom, Outcome.Benefit);
131         staticText = "You may cast that card for as long as it remains exiled, and you may spend mana as though it were mana of any color to cast that spell";
132     }
133 
134     private GontiLordOfLuxuryCastFromExileEffect(final GontiLordOfLuxuryCastFromExileEffect effect) {
135         super(effect);
136     }
137 
138     @Override
139     public boolean apply(Game game, Ability source) {
140         return true;
141     }
142 
143     @Override
144     public GontiLordOfLuxuryCastFromExileEffect copy() {
145         return new GontiLordOfLuxuryCastFromExileEffect(this);
146     }
147 
148     @Override
149     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
150         UUID targetId = getTargetPointer().getFirst(game, source);
151         if (targetId == null) {
152             this.discard(); // card is no longer in the origin zone, effect can be discarded
153             return false;
154         }
155         Card theCard = game.getCard(objectId);
156         if (theCard == null || theCard.isLand(game)) {
157             return false;
158         }
159         objectId = theCard.getMainCard().getId(); // for split cards
160 
161         if (objectId.equals(targetId)
162                 && affectedControllerId.equals(source.getControllerId())) {
163             Card card = game.getCard(objectId);
164             // TODO: Allow to cast Zoetic Cavern face down
165             return card != null;
166         }
167         return false;
168     }

174         super(AsThoughEffectType.SPEND_OTHER_MANA, Duration.Custom, Outcome.Benefit);
175         staticText = "you may spend mana as though it were mana of any color to cast it";
176     }

179         super(effect);
180     }
181 
182     @Override
183     public boolean apply(Game game, Ability source) {
184         return true;
185     }
186 
187     @Override
188     public GontiLordOfLuxurySpendAnyManaEffect copy() {
189         return new GontiLordOfLuxurySpendAnyManaEffect(this);
190     }
191 
192     @Override
193     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
194         Card theCard = game.getCard(objectId);
195         if (theCard == null) {
196             return false;
197         }
198         objectId = theCard.getMainCard().getId(); // for split cards
199         if (objectId.equals(((FixedTarget) getTargetPointer()).getTarget())
200                 && game.getState().getZoneChangeCounter(objectId) <= ((FixedTarget) getTargetPointer()).getZoneChangeCounter() + 1) {
201             // if the card moved from exile to spell the zone change counter is increased by 1 (effect must applies before and on stack, use isCheckPlayableMode?)
202             return source.isControlledBy(affectedControllerId);
203         } else if (((FixedTarget) getTargetPointer()).getTarget().equals(objectId)) {
204             // object has moved zone so effect can be discarded
205             this.discard();
206         }
207         return false;
208     }
209 
210     @Override
211     public ManaType getAsThoughManaType(ManaType manaType, ManaPoolItem mana, UUID affectedControllerId, Ability source, Game game) {
212         return mana.getFirstAvailable();
213     }

216 class GontiLordOfLuxuryLookEffect extends AsThoughEffectImpl {
217 
218     private final UUID authorizedPlayerId;
219 
220     public GontiLordOfLuxuryLookEffect(UUID authorizedPlayerId) {
221         super(AsThoughEffectType.LOOK_AT_FACE_DOWN, Duration.EndOfGame, Outcome.Benefit);
222         this.authorizedPlayerId = authorizedPlayerId;
223         staticText = "You may look at the cards exiled with {this}";
224     }
225 
226     private GontiLordOfLuxuryLookEffect(final GontiLordOfLuxuryLookEffect effect) {
227         super(effect);
228         this.authorizedPlayerId = effect.authorizedPlayerId;
229     }
230 
231     @Override
232     public boolean apply(Game game, Ability source) {
233         return true;
234     }
235 
236     @Override
237     public GontiLordOfLuxuryLookEffect copy() {
238         return new GontiLordOfLuxuryLookEffect(this);
239     }
240 
241     @Override
242     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
243         UUID cardId = getTargetPointer().getFirst(game, source);
244         if (cardId == null) {
245             this.discard(); // card is no longer in the origin zone, effect can be discarded
246         }
247         return affectedControllerId.equals(authorizedPlayerId)
248                 && objectId.equals(cardId);

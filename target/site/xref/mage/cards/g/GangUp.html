4   import mage.abilities.Ability;
5   import mage.abilities.effects.common.DestroyTargetEffect;
6   import mage.abilities.keyword.AssistAbility;
7   import mage.cards.CardImpl;
8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.constants.ComparisonType;
11  import mage.filter.common.FilterCreaturePermanent;
12  import mage.filter.predicate.mageobject.PowerPredicate;
13  import mage.game.Game;
14  import mage.target.common.TargetCreaturePermanent;
15  import mage.target.targetadjustment.TargetAdjuster;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class GangUp extends CardImpl {
23  
24      public GangUp(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{X}{B}");
26  
27          // Assist
28          this.addAbility(new AssistAbility());
29  
30          // Destroy target creature with power X or less.
31          this.getSpellAbility().addEffect(new DestroyTargetEffect("destroy target creature with power X or less"));
32          this.getSpellAbility().setTargetAdjuster(GangUpAdjuster.instance);
33      }
34  
35      private GangUp(final GangUp card) {
36          super(card);
37      }
38  
39      @Override
40      public GangUp copy() {
41          return new GangUp(this);
42      }
43  }
44  
45  enum GangUpAdjuster implements TargetAdjuster {
46      instance;
47  
48      @Override
49      public void adjustTargets(Ability ability, Game game) {
50          int xValue = ability.getManaCostsToPay().getX();
51          ability.getTargets().clear();
52          FilterCreaturePermanent filter = new FilterCreaturePermanent("creature with power " + xValue + " or less");
53          filter.add(new PowerPredicate(ComparisonType.FEWER_THAN, xValue + 1));
54          ability.addTarget(new TargetCreaturePermanent(filter));
55      }
56  }

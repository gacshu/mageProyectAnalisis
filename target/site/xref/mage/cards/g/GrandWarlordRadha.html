35  public final class GrandWarlordRadha extends CardImpl {

37      public GrandWarlordRadha(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{G}");
39  
40          this.addSuperType(SuperType.LEGENDARY);
41          this.subtype.add(SubType.ELF);
42          this.subtype.add(SubType.WARRIOR);
43          this.power = new MageInt(3);
44          this.toughness = new MageInt(4);
45  
46          // Haste
47          this.addAbility(HasteAbility.getInstance());
48  
49          // Whenever one or more creatures you control attack, add that much mana in any combination of {R} and/or {G}. Until end of turn, you don't lose this mana as steps and phases end.
50          this.addAbility(new GrandWarlordRadhaTriggeredAbility(), new CreaturesAttackedWatcher());
51      }

63  class CreaturesAttackedWatcher extends Watcher {
64  
65      private final Set<MageObjectReference> attackedThisTurnCreatures = new HashSet<>();
66  
67      public CreaturesAttackedWatcher() {
68          super(WatcherScope.GAME);
69      }
70  
71      @Override
72      public void watch(GameEvent event, Game game) {
73          if (event.getType() == GameEvent.EventType.BEGIN_COMBAT_STEP_PRE) {
74              this.attackedThisTurnCreatures.clear();
75          }
76          if (event.getType() == GameEvent.EventType.ATTACKER_DECLARED) {
77              this.attackedThisTurnCreatures.add(new MageObjectReference(event.getSourceId(), game));
78          }
79      }
80  
81      public Set<MageObjectReference> getAttackedThisTurnCreatures() {
82          return this.attackedThisTurnCreatures;
83      }
84  }

86  class GrandWarlordRadhaTriggeredAbility extends TriggeredAbilityImpl {
87  
88      public GrandWarlordRadhaTriggeredAbility() {
89          super(Zone.BATTLEFIELD, new GrandWarlordRadhaEffect(), false);
90          setTriggerPhrase("Whenever one or more creatures you control attack, ");
91      }
92  
93      public GrandWarlordRadhaTriggeredAbility(final GrandWarlordRadhaTriggeredAbility ability) {
94          super(ability);
95      }
96  
97      @Override
98      public GrandWarlordRadhaTriggeredAbility copy() {
99          return new GrandWarlordRadhaTriggeredAbility(this);
100     }
101 
102     @Override
103     public boolean checkEventType(GameEvent event, Game game) {
104         return event.getType() == GameEvent.EventType.DECLARED_ATTACKERS;
105     }
106 
107     @Override
108     public boolean checkTrigger(GameEvent event, Game game) {
109         for (UUID attacker : game.getCombat().getAttackers()) {
110             Permanent creature = game.getPermanent(attacker);
111             if (creature != null
112                     && creature.getControllerId() != null
113                     && creature.isControlledBy(this.getControllerId())) {
114                 return true;
115             }
116         }
117         return false;
118     }
119 }

121 class GrandWarlordRadhaEffect extends OneShotEffect {
122 
123     public GrandWarlordRadhaEffect() {
124         super(Outcome.Benefit);
125         this.staticText = "add that much mana in any combination of {R} and/or {G}. Until end of turn, you don't lose this mana as steps and phases end";
126     }
127 
128     public GrandWarlordRadhaEffect(final GrandWarlordRadhaEffect effect) {
129         super(effect);
130     }
131 
132     @Override
133     public GrandWarlordRadhaEffect copy() {
134         return new GrandWarlordRadhaEffect(this);
135     }
136 
137     @Override
138     public boolean apply(Game game, Ability source) {
139         Player controller = game.getPlayer(source.getControllerId());
140         if (controller != null) {
141             CreaturesAttackedWatcher watcher = game.getState().getWatcher(CreaturesAttackedWatcher.class);
142             if (watcher != null) {
143                 int attackingCreatures = 0;
144                 for (MageObjectReference attacker : watcher.getAttackedThisTurnCreatures()) {
145                     if (attacker.getPermanentOrLKIBattlefield(game).isControlledBy(controller.getId())) {
146                         attackingCreatures++;
147                     }
148                 }
149                 if (attackingCreatures > 0) {
150                     Choice manaChoice = new ChoiceImpl();
151                     Set<String> choices = new LinkedHashSet<>();
152                     choices.add("Red");
153                     choices.add("Green");
154                     manaChoice.setChoices(choices);
155                     manaChoice.setMessage("Select color of mana to add");
156 
157                     for (int i = 0; i < attackingCreatures; i++) {
158                         Mana mana = new Mana();
159                         if (!controller.choose(Outcome.Benefit, manaChoice, game)) {
160                             return false;
161                         }
162                         if (manaChoice.getChoice() == null) {  // can happen if player leaves game
163                             return false;
164                         }
165                         switch (manaChoice.getChoice()) {
166                             case "Green":
167                                 mana.increaseGreen();
168                                 break;
169                             case "Red":
170                                 mana.increaseRed();
171                                 break;
172                         }
173                         controller.getManaPool().addMana(mana, game, source, true);
174                     }
175                     return true;
176                 }
177                 return true;
178             }
179         }
180         return false;
181     }

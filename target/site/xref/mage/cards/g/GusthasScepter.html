30  public final class GusthasScepter extends CardImpl {
31  
32      public GusthasScepter(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{0}");
34  
35          // {T}: Exile a card from your hand face down. You may look at it for as long as it remains exiled.
36          this.addAbility(new SimpleActivatedAbility(new GusthasScepterExileEffect(), new TapSourceCost()));
37  
38          // {T}: Return a card you own exiled with Gustha’s Scepter to your hand.
39          this.addAbility(new SimpleActivatedAbility(new GusthasScepterReturnEffect(), new TapSourceCost()));
40  
41          // When you lose control of Gustha’s Scepter, put all cards exiled with Gustha’s Scepter into their owner’s graveyard.
42          this.addAbility(new GusthasScepterLoseControlAbility());
43      }
44  
45      private GusthasScepter(final GusthasScepter card) {
46          super(card);
47      }
48  
49      @Override
50      public GusthasScepter copy() {
51          return new GusthasScepter(this);
52      }
53  }

55  class GusthasScepterExileEffect extends OneShotEffect {
56  
57      public GusthasScepterExileEffect() {
58          super(Outcome.DrawCard);
59          staticText = "exile a card from your hand face down. You may look at it for as long as it remains exiled";
60      }
61  
62      public GusthasScepterExileEffect(final GusthasScepterExileEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public boolean apply(Game game, Ability source) {
68          Player controller = game.getPlayer(source.getControllerId());
69          if (controller == null || controller.getHand().isEmpty()) {
70              return false;
71          }
72          TargetCard target = new TargetCardInHand();
73          controller.chooseTarget(outcome, controller.getHand(), target, source, game);
74          Card card = game.getCard(target.getFirstTarget());
75          if (card == null) {
76              return false;
77          }
78          controller.moveCardsToExile(
79                  card, source, game, false,
80                  CardUtil.getExileZoneId(game, source),
81                  CardUtil.getSourceName(game, source)
82          );
83          card.setFaceDown(true, game);
84          game.addEffect(new GusthasScepterLookAtCardEffect(card, game), source);
85          return true;
86      }
87  
88      @Override
89      public GusthasScepterExileEffect copy() {
90          return new GusthasScepterExileEffect(this);
91      }
92  }

94  class GusthasScepterReturnEffect extends OneShotEffect {
95  
96      private static final FilterCard filter = new FilterCard("card you own exiled with this permanent");
97  
98      GusthasScepterReturnEffect() {
99          super(Outcome.Benefit);
100         staticText = "return a card you own exiled with {this} to your hand";
101     }
102 
103     private GusthasScepterReturnEffect(final GusthasScepterReturnEffect effect) {
104         super(effect);
105     }
106 
107     @Override
108     public GusthasScepterReturnEffect copy() {
109         return new GusthasScepterReturnEffect(this);
110     }
111 
112     @Override
113     public boolean apply(Game game, Ability source) {
114         Player player = game.getPlayer(source.getControllerId());
115         if (player == null) {
116             return false;
117         }
118         TargetCard target = new TargetCardInExile(filter, CardUtil.getExileZoneId(game, source));
119         target.setNotTarget(true);
120         if (!target.canChoose(source.getControllerId(), source, game)) {
121             return false;
122         }
123         player.choose(outcome, target, source, game);
124         Card card = game.getCard(target.getFirstTarget());
125         return card != null && player.moveCards(card, Zone.HAND, source, game);
126     }
127 }

129 class GusthasScepterLookAtCardEffect extends AsThoughEffectImpl {
130 
131     private final MageObjectReference mor;
132 
133     public GusthasScepterLookAtCardEffect(Card card, Game game) {
134         super(AsThoughEffectType.LOOK_AT_FACE_DOWN, Duration.EndOfGame, Outcome.Benefit);
135         this.mor = new MageObjectReference(card, game);
136         staticText = "You may look at it for as long as it remains exiled";
137     }
138 
139     public GusthasScepterLookAtCardEffect(final GusthasScepterLookAtCardEffect effect) {
140         super(effect);
141         this.mor = effect.mor;
142     }
143 
144     @Override
145     public boolean apply(Game game, Ability source) {
146         return true;
147     }
148 
149     @Override
150     public GusthasScepterLookAtCardEffect copy() {
151         return new GusthasScepterLookAtCardEffect(this);
152     }
153 
154     @Override
155     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
156         if (!mor.zoneCounterIsCurrent(game)) {
157             discard();
158             return false;
159         }
160         ExileZone exileZone = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source));
161         if (exileZone == null || !exileZone.contains(mor.getSourceId())) {
162             discard();
163             return false;
164         }
165         return mor.refersTo(objectId, game) && source.isControlledBy(affectedControllerId);
166     }
167 }

169 class GusthasScepterLoseControlAbility extends DelayedTriggeredAbility {
170 
171     public GusthasScepterLoseControlAbility() {
172         super(new GusthasScepterPutExiledCardsInOwnersGraveyard(), Duration.EndOfGame, false);
173     }
174 
175     public GusthasScepterLoseControlAbility(final GusthasScepterLoseControlAbility ability) {
176         super(ability);
177     }
178 
179     @Override
180     public GusthasScepterLoseControlAbility copy() {
181         return new GusthasScepterLoseControlAbility(this);
182     }
183 
184     @Override
185     public boolean checkEventType(GameEvent event, Game game) {
186         return event.getType() == GameEvent.EventType.LOST_CONTROL
187                 || event.getType() == GameEvent.EventType.ZONE_CHANGE;
188     }
189 
190     public boolean checkTrigger(GameEvent event, Game game) {
191         switch (event.getType()) {
192             case LOST_CONTROL:
193                 return event.getPlayerId().equals(controllerId)
194                         && event.getTargetId().equals(this.getSourceId());
195             case ZONE_CHANGE:
196                 return event.getTargetId().equals(this.getSourceId()) && ((ZoneChangeEvent) event).getFromZone() == Zone.BATTLEFIELD;
197         }
198         return false;
199     }
200 
201     @Override
202     public String getRule() {
203         return "When you lose control of {this}, put all cards exiled with {this} into their owner's graveyard.";
204     }
205 }

207 class GusthasScepterPutExiledCardsInOwnersGraveyard extends OneShotEffect {
208 
209     public GusthasScepterPutExiledCardsInOwnersGraveyard() {
210         super(Outcome.Neutral);
211     }
212 
213     public GusthasScepterPutExiledCardsInOwnersGraveyard(final GusthasScepterPutExiledCardsInOwnersGraveyard effect) {
214         super(effect);
215     }
216 
217     @Override
218     public boolean apply(Game game, Ability source) {
219         Player controller = game.getPlayer(source.getControllerId());
220         if (controller == null) {
221             return false;
222         }
223         ExileZone exileZone = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source));
224         return exileZone != null && controller.moveCards(exileZone.getCards(game), Zone.GRAVEYARD, source, game);
225     }
226 
227     @Override
228     public GusthasScepterPutExiledCardsInOwnersGraveyard copy() {
229         return new GusthasScepterPutExiledCardsInOwnersGraveyard(this);
230     }

22  public final class GripOfAmnesia extends CardImpl {
23  
24      public GripOfAmnesia(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{U}");
26  
27          // Counter target spell unless its controller exiles all cards from their graveyard.
28          // Draw a card.
29          this.getSpellAbility().addEffect(new GripOfAmnesiaEffect());
30          this.getSpellAbility().addEffect(new DrawCardSourceControllerEffect(1));
31          this.getSpellAbility().addTarget(new TargetSpell());
32      }
33  
34      private GripOfAmnesia(final GripOfAmnesia card) {
35          super(card);
36      }
37  
38      @Override
39      public GripOfAmnesia copy() {
40          return new GripOfAmnesia(this);
41      }
42  }
43  
44  class GripOfAmnesiaEffect extends OneShotEffect {
45  
46      public GripOfAmnesiaEffect() {
47          super(Outcome.Detriment);
48          staticText = "Counter target spell unless its controller exiles all cards from their graveyard";
49      }
50  
51      private GripOfAmnesiaEffect(final GripOfAmnesiaEffect effect) {
52          super(effect);
53      }
54  
55      @Override
56      public GripOfAmnesiaEffect copy() {
57          return new GripOfAmnesiaEffect(this);
58      }
59  
60      @Override
61      public boolean apply(Game game, Ability source) {
62          Spell spell = game.getStack().getSpell(source.getFirstTarget());
63          if (spell != null) {
64              Player controller = game.getPlayer(spell.getControllerId());
65              if (controller != null && controller.chooseUse(Outcome.Benefit, "Exile all cards from your graveyard? (Otherwise "
66                      + spell.getLogName() + " will be countered)", source, game)) {
67                  controller.moveCards(controller.getGraveyard(), Zone.EXILED, source, game);
68                  game.informPlayers(controller.getLogName() + " exiles all cards from their graveyard to prevent counter effect");
69              } else {
70                  game.getStack().counter(spell.getId(), source, game);
71              }
72              return true;
73          }
74          return false;
75      }
76  }

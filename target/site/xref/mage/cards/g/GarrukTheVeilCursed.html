30  public final class GarrukTheVeilCursed extends CardImpl {
31  
32      private static final DynamicValue xValue = new CardsInControllerGraveyardCount(StaticFilters.FILTER_CARD_CREATURE);
33  
34      public GarrukTheVeilCursed(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "");
36          this.addSuperType(SuperType.LEGENDARY);
37          this.subtype.add(SubType.GARRUK);
38  
39          // this card is the second face of double-faced card
40          this.nightCard = true;
41  
42          this.color.setGreen(true);
43          this.color.setBlack(true);
44  
45          // +1 : Create a 1/1 black Wolf creature token with deathtouch.
46          this.addAbility(new LoyaltyAbility(new CreateTokenEffect(new WolfTokenWithDeathtouch()), 1));
47  
48          // -1 : Sacrifice a creature. If you do, search your library for a creature card, reveal it, put it into your hand, then shuffle your library.
49          this.addAbility(new LoyaltyAbility(new DoIfCostPaid(
50                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary(
51                          StaticFilters.FILTER_CARD_CREATURE_A
52                  ), true, true),
53                  null,
54                  new SacrificeTargetCost(new TargetControlledPermanent(
55                          StaticFilters.FILTER_CONTROLLED_CREATURE_SHORT_TEXT
56                  )),
57                  false
58          ), -1));
59  
60          // -3 : Creatures you control gain trample and get +X/+X until end of turn, where X is the number of creature cards in your graveyard.
61          Ability ability = new LoyaltyAbility(new GainAbilityControlledEffect(
62                  TrampleAbility.getInstance(), Duration.EndOfTurn, StaticFilters.FILTER_CONTROLLED_CREATURE
63          ).setText("creatures you control gain trample"), -3);
64          ability.addEffect(new BoostControlledEffect(
65                  xValue, xValue, Duration.EndOfTurn
66          ).setText("and get +X/+X until end of turn, where X is the number of creature cards in your graveyard"));
67          this.addAbility(ability);
68      }
69  
70      private GarrukTheVeilCursed(final GarrukTheVeilCursed card) {
71          super(card);
72      }
73  
74      @Override
75      public GarrukTheVeilCursed copy() {
76          return new GarrukTheVeilCursed(this);
77      }
78  }

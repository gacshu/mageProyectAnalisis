42  public final class GideonBlackblade extends CardImpl {
43  
44      private static final FilterPermanent filter = new FilterControlledCreaturePermanent("another other creature you control");
45  
46      static {
47          filter.add(AnotherPredicate.instance);
48      }
49  
50      public GideonBlackblade(UUID ownerId, CardSetInfo setInfo) {
51          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{1}{W}{W}");
52  
53          this.addSuperType(SuperType.LEGENDARY);
54          this.subtype.add(SubType.GIDEON);
55          this.setStartingLoyalty(4);
56  
57          // As long as it's your turn, Gideon Blackblade is a 4/4 Human Soldier creature with indestructible that's still a planeswalker.
58          this.addAbility(new SimpleStaticAbility(new ConditionalContinuousEffect(
59                  new BecomesCreatureSourceEffect(
60                          new GideonBlackbladeToken(), "planeswalker", Duration.WhileOnBattlefield
61                  ), MyTurnCondition.instance, "As long as it's your turn, " +
62                  "{this} is a 4/4 Human Soldier creature with indestructible that's still a planeswalker."
63          )).addHint(MyTurnHint.instance));
64  
65          // Prevent all damage that would be dealt to Gideon Blackblade during your turn.
66          this.addAbility(new SimpleStaticAbility(new ConditionalPreventionEffect(
67                  new PreventAllDamageToSourceEffect(Duration.WhileOnBattlefield),
68                  MyTurnCondition.instance, "Prevent all damage that would be dealt to {this} during your turn."
69          )));
70  
71          // +1: Up to one other target creature you control gains your choice of vigilance, lifelink, or indestructible until end of turn.
72          Ability ability = new LoyaltyAbility(new GideonBlackbladeEffect(), 1);
73          ability.addTarget(new TargetPermanent(0, 1, filter, false));
74          this.addAbility(ability);
75  
76          // -6: Exile target nonland permanent.
77          ability = new LoyaltyAbility(new ExileTargetEffect(), -6);
78          ability.addTarget(new TargetNonlandPermanent());
79          this.addAbility(ability);
80      }
81  
82      private GideonBlackblade(final GideonBlackblade card) {
83          super(card);
84      }
85  
86      @Override
87      public GideonBlackblade copy() {
88          return new GideonBlackblade(this);
89      }
90  }

114 class GideonBlackbladeEffect extends OneShotEffect {
115     private static final Set<String> choices = new LinkedHashSet<>();
116 
117     static {
118         choices.add("Vigilance");
119         choices.add("Lifelink");
120         choices.add("Indestructible");
121     }
122 
123     GideonBlackbladeEffect() {
124         super(Outcome.Benefit);
125         staticText = "Up to one other target creature you control gains your choice of " +
126                 "vigilance, lifelink, or indestructible until end of turn.";
127     }
128 
129     private GideonBlackbladeEffect(final GideonBlackbladeEffect effect) {
130         super(effect);
131     }
132 
133     @Override
134     public GideonBlackbladeEffect copy() {
135         return new GideonBlackbladeEffect(this);
136     }
137 
138     @Override
139     public boolean apply(Game game, Ability source) {
140         Player player = game.getPlayer(source.getControllerId());
141         Permanent permanent = game.getPermanent(source.getFirstTarget());
142         if (player == null || permanent == null) {
143             return false;
144         }
145         Choice choice = new ChoiceImpl(true);
146         choice.setMessage("Choose an ability to give to " + permanent.getLogName());
147         choice.setChoices(choices);
148         if (!player.choose(outcome, choice, game)) {
149             return false;
150         }
151         Ability ability = null;
152         switch (choice.getChoice()) {
153             case "Vigilance":
154                 ability = VigilanceAbility.getInstance();
155                 break;
156             case "Lifelink":
157                 ability = LifelinkAbility.getInstance();
158                 break;
159             case "Indestructible":
160                 ability = IndestructibleAbility.getInstance();
161                 break;
162         }
163         if (ability != null) {
164             game.addEffect(new GainAbilityTargetEffect(ability, Duration.EndOfTurn), source);
165         }
166         return true;
167     }

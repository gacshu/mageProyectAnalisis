31  public final class GrimoireThief extends CardImpl {
32  
33      protected static final String VALUE_PREFIX = "ExileZones";
34  
35      public GrimoireThief(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{U}{U}");
37          this.subtype.add(SubType.MERFOLK);
38          this.subtype.add(SubType.ROGUE);
39          this.power = new MageInt(2);
40          this.toughness = new MageInt(2);
41  
42          // Whenever Grimoire Thief becomes tapped, exile the top three 
43          // cards of target opponent's library face down.
44          Ability ability = new BecomesTappedSourceTriggeredAbility(
45                  new GrimoireThiefExileEffect(), false);
46          ability.addTarget(new TargetOpponent());
47          this.addAbility(ability);
48  
49          // You may look at cards exiled with Grimoire Thief.
50          this.addAbility(new SimpleStaticAbility(Zone.ALL, new GrimoireThiefLookEffect()));
51  
52          // {U}, Sacrifice Grimoire Thief: Turn all cards exiled with 
53          //Grimoire Thief face up. Counter all spells with those names.
54          Ability ability2 = new SimpleActivatedAbility(Zone.BATTLEFIELD,
55                  new GrimoireThiefCounterspellEffect(), new ManaCostsImpl<>("{U}"));
56          ability2.addCost(new SacrificeSourceCost());
57          this.addAbility(ability2);
58  
59      }
60  
61      private GrimoireThief(final GrimoireThief card) {
62          super(card);
63      }
64  
65      @Override
66      public GrimoireThief copy() {
67          return new GrimoireThief(this);
68      }
69  }

71  class GrimoireThiefExileEffect extends OneShotEffect {
72  
73      public GrimoireThiefExileEffect() {
74          super(Outcome.Discard);
75          staticText = "exile the top three cards of target opponent's library face down";
76      }
77  
78      public GrimoireThiefExileEffect(final GrimoireThiefExileEffect effect) {
79          super(effect);
80      }
81  
82      @Override
83      public boolean apply(Game game, Ability source) {
84          Player targetOpponent = game.getPlayer(source.getFirstTarget());
85          if (targetOpponent != null) {
86              Set<Card> cards = targetOpponent.getLibrary().getTopCards(game, 3);
87              MageObject sourceObject = source.getSourceObject(game);
88              if (!cards.isEmpty() && sourceObject != null) {
89                  for (Card card : cards) {
90                      card.setFaceDown(true, game);
91                  }
92                  UUID exileZoneId = CardUtil.getExileZoneId(game,
93                          source.getSourceId(), source.getSourceObjectZoneChangeCounter());
94                  targetOpponent.moveCardsToExile(cards, source, game, false,
95                          exileZoneId, sourceObject.getIdName());
96                  for (Card card : cards) {
97                      card.setFaceDown(true, game);
98                  }
99                  Set<UUID> exileZones = (Set<UUID>) game.getState().getValue(
100                         GrimoireThief.VALUE_PREFIX + source.getSourceId().toString());
101                 if (exileZones == null) {
102                     exileZones = new HashSet<>();
103                     game.getState().setValue(GrimoireThief.VALUE_PREFIX
104                             + source.getSourceId().toString(), exileZones);
105                 }
106                 exileZones.add(exileZoneId);
107                 return true;
108             }
109         }
110         return false;
111     }
112 
113     @Override
114     public GrimoireThiefExileEffect copy() {
115         return new GrimoireThiefExileEffect(this);
116     }
117 }

119 class GrimoireThiefLookEffect extends AsThoughEffectImpl {
120 
121     public GrimoireThiefLookEffect() {
122         super(AsThoughEffectType.LOOK_AT_FACE_DOWN, Duration.EndOfGame, Outcome.Benefit);
123         staticText = "You may look at cards exiled with {this}";
124     }
125 
126     public GrimoireThiefLookEffect(final GrimoireThiefLookEffect effect) {
127         super(effect);
128     }
129 
130     @Override
131     public boolean apply(Game game, Ability source) {
132         return true;
133     }
134 
135     @Override
136     public GrimoireThiefLookEffect copy() {
137         return new GrimoireThiefLookEffect(this);
138     }
139 
140     @Override
141     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
142         if (affectedControllerId.equals(source.getControllerId())
143                 && game.getState().getZone(objectId) == Zone.EXILED) {
144             Player controller = game.getPlayer(source.getControllerId());
145             MageObject sourceObject = source.getSourceObject(game);
146             if (controller != null
147                     && sourceObject != null) {
148                 Card card = game.getCard(objectId);
149                 if (card != null
150                         && card.isFaceDown(game)) {
151                     Set<UUID> exileZones = (Set<UUID>) game.getState().
152                             getValue(GrimoireThief.VALUE_PREFIX + source.getSourceId().toString());
153                     if (exileZones != null) {
154                         for (ExileZone exileZone : game.getExile().getExileZones()) {
155                             if (exileZone.contains(objectId)) {
156                                 if (!exileZones.contains(exileZone.getId())) {
157                                     return false;
158                                 }
159                             }
160                         }
161                         return true;
162                     }
163                 }
164             }
165         }
166         return false;
167     }

170 class GrimoireThiefCounterspellEffect extends OneShotEffect {
171 
172     public GrimoireThiefCounterspellEffect() {
173         super(Outcome.Discard);
174         staticText = "Turn all cards exiled with {this} face up. "
175                 + "Counter all spells with those names";
176     }
177 
178     public GrimoireThiefCounterspellEffect(final GrimoireThiefCounterspellEffect effect) {
179         super(effect);
180     }
181 
182     @Override
183     public boolean apply(Game game, Ability source) {
184         Cards cards = new CardsImpl();
185         MageObject sourceObject = game.getObject(source);
186         Set<UUID> exileZones = (Set<UUID>) game.getState().getValue(
187                 GrimoireThief.VALUE_PREFIX + source.getSourceId().toString());
188         if (exileZones != null && sourceObject != null) {
189             for (ExileZone exileZone : game.getExile().getExileZones()) {
190                 if (!exileZone.isEmpty()) {
191                     cards.addAll(exileZone.getCards(game));
192                 }
193             }
194             // set face up first
195             for (Card card : cards.getCards(game)) {
196                 card.setFaceDown(false, game);
197             }
198             // then counter any with the same name as the card exiled with Grimoire Thief
199             for (Card card : cards.getCards(game)) {
200                 for (Iterator<StackObject> iterator = game.getStack().iterator(); iterator.hasNext(); ) {
201                     StackObject stackObject = iterator.next();
202                     MageObject mageObject = game.getObject(card.getId());
203                     String name1;
204                     String name2;
205                     if (mageObject instanceof SplitCard) {
206                         name1 = ((SplitCard) mageObject).getLeftHalfCard().getName();
207                         name2 = ((SplitCard) mageObject).getRightHalfCard().getName();
208                     } else {
209                         // modal double faces cards, adventure cards -- all have one name in non stack/battlefield zone
210                         name1 = mageObject.getName();
211                         name2 = name1;
212                     }
213 
214                     if (CardUtil.haveSameNames(stackObject, name1, game) || CardUtil.haveSameNames(stackObject, name2, game)) {
215                         Spell spell = (Spell) stackObject;
216                         game.getStack().counter(stackObject.getId(), source, game);
217                         game.informPlayers(sourceObject.getLogName() + ": spell " + spell.getIdName() + " was countered.");
218                     }
219                 }
220             }
221             return true;
222         }
223         return false;
224     }
225 
226     @Override
227     public GrimoireThiefCounterspellEffect copy() {
228         return new GrimoireThiefCounterspellEffect(this);
229     }

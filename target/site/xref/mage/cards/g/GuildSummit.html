26  public final class GuildSummit extends CardImpl {
27  
28      private static final FilterPermanent filter = new FilterPermanent("a Gate");
29  
30      static {
31          filter.add(SubType.GATE.getPredicate());
32      }
33  
34      public GuildSummit(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{U}");
36  
37          // When Guild Summit enters the battlefield, you may tap any number of untapped Gates you control. Draw a card for each Gate tapped this way.
38          this.addAbility(new EntersBattlefieldTriggeredAbility(
39                  new GuildSummitEffect(), true
40          ));
41  
42          // Whenever a Gate enters the battlefield under your control, draw a card.
43          this.addAbility(new EntersBattlefieldControlledTriggeredAbility(
44                  new DrawCardSourceControllerEffect(1), filter
45          ));
46      }
47  
48      private GuildSummit(final GuildSummit card) {
49          super(card);
50      }
51  
52      @Override
53      public GuildSummit copy() {
54          return new GuildSummit(this);
55      }
56  }

58  class GuildSummitEffect extends OneShotEffect {
59  
60      private static final FilterPermanent filter
61              = new FilterPermanent("untapped Gates you control");
62  
63      static {
64          filter.add(TargetController.YOU.getControllerPredicate());
65          filter.add(TappedPredicate.UNTAPPED);
66          filter.add(SubType.GATE.getPredicate());
67      }
68  
69      public GuildSummitEffect() {
70          super(Outcome.GainLife);
71          staticText = "you may tap any number of untapped Gates you control. "
72                  + "Draw a card for each Gate tapped this way";
73      }
74  
75      public GuildSummitEffect(GuildSummitEffect effect) {
76          super(effect);
77      }
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          int tappedAmount = 0;
82          Player you = game.getPlayer(source.getControllerId());
83          TargetPermanent target = new TargetPermanent(0, Integer.MAX_VALUE, filter, true);
84          if (target.canChoose(source.getControllerId(), source, game) && target.choose(Outcome.Tap, source.getControllerId(), source.getSourceId(), source, game)) {
85              for (UUID creatureId : target.getTargets()) {
86                  Permanent creature = game.getPermanent(creatureId);
87                  if (creature != null) {
88                      creature.tap(source, game);
89                      tappedAmount++;
90                  }
91              }
92          }
93          if (tappedAmount > 0) {
94              you.drawCards(tappedAmount, source, game);
95              return true;
96          }
97          return false;
98      }
99  
100     @Override
101     public GuildSummitEffect copy() {
102         return new GuildSummitEffect(this);
103     }
104 
105 }

24  import java.util.UUID;

26  /**
27   * @author TheElk801
28   */
29  public final class GoroGoroDiscipleOfRyusei extends CardImpl {
30  
31      private static final FilterPermanent filter
32              = new FilterControlledCreaturePermanent("if you control an attacking modified creature");
33  
34      static {
35          filter.add(AttackingPredicate.instance);
36          filter.add(ModifiedPredicate.instance);
37      }
38  
39      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter);
40      private static final Hint hint = new ConditionHint(
41              condition, "You control and attacking modified creature"
42      );
43  
44      public GoroGoroDiscipleOfRyusei(UUID ownerId, CardSetInfo setInfo) {
45          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{R}");
46  
47          this.addSuperType(SuperType.LEGENDARY);
48          this.subtype.add(SubType.GOBLIN);
49          this.subtype.add(SubType.SAMURAI);
50          this.power = new MageInt(2);
51          this.toughness = new MageInt(2);
52  
53          // {R}: Creatures you control gain haste until end of turn.
54          this.addAbility(new SimpleActivatedAbility(new GainAbilityAllEffect(
55                  HasteAbility.getInstance(), Duration.EndOfTurn,
56                  StaticFilters.FILTER_CONTROLLED_CREATURES
57          ), new ManaCostsImpl<>("{R}")));
58  
59          // {3}{R}{R}: Create a 5/5 red Dragon Spirit creature token with flying. Activate only if you control an attacking modified creature.
60          this.addAbility(new ConditionalActivatedAbility(
61                  Zone.BATTLEFIELD, new CreateTokenEffect(new DragonSpiritToken()),
62                  new ManaCostsImpl<>("{3}{R}{R}"), condition
63          ).addHint(hint));
64      }
65  
66      private GoroGoroDiscipleOfRyusei(final GoroGoroDiscipleOfRyusei card) {
67          super(card);
68      }
69  
70      @Override
71      public GoroGoroDiscipleOfRyusei copy() {
72          return new GoroGoroDiscipleOfRyusei(this);
73      }
74  }

5   import mage.MageInt;

8   import mage.abilities.costs.common.RemoveCountersSourceCost;
9   import mage.abilities.effects.common.ReturnToHandSourceEffect;
10  import mage.abilities.effects.common.counter.AddCountersSourceEffect;
11  import mage.abilities.keyword.DoubleStrikeAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.constants.SuperType;
17  import mage.constants.Zone;
18  import mage.counters.CounterType;
19  
20  /**
21   *
22   * @author Styxo
23   */
24  public final class GeneralGrievous extends CardImpl {
25  
26      public GeneralGrievous(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT,CardType.CREATURE},"{W}{U}{B}");
28          addSuperType(SuperType.LEGENDARY);
29          this.subtype.add(SubType.KALEESH);
30          this.power = new MageInt(2);
31          this.toughness = new MageInt(2);
32  
33          // Double Strike
34          this.addAbility(DoubleStrikeAbility.getInstance());
35  
36          // Whenever another creature dies, you may put a +1/+1 counter on General Grievous.
37          this.addAbility(new DiesCreatureTriggeredAbility(new AddCountersSourceEffect(CounterType.P1P1.createInstance()), true, true));
38  
39          // Remove two +1/+1 counters from General Grievous: Return General Grievous to its owner's hand.
40          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new ReturnToHandSourceEffect(), new RemoveCountersSourceCost(CounterType.P1P1.createInstance(2))));
41      }
42  
43      private GeneralGrievous(final GeneralGrievous card) {
44          super(card);
45      }
46  
47      @Override
48      public GeneralGrievous copy() {
49          return new GeneralGrievous(this);
50      }
51  }

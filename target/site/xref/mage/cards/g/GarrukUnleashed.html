31  public final class GarrukUnleashed extends CardImpl {
32  
33      public GarrukUnleashed(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{2}{G}{G}");
35          
36          this.addSuperType(SuperType.LEGENDARY);
37          this.subtype.add(SubType.GARRUK);
38          this.setStartingLoyalty(4);
39  
40          // +1: Up to one target creature gets +3/+3 and gains trample until end of turn.
41          Effect effect = new BoostTargetEffect(3, 3, Duration.EndOfTurn)
42              .setText("up to one target creature gets +3/+3");
43          LoyaltyAbility ability = new LoyaltyAbility(effect, 1);
44          effect = new GainAbilityTargetEffect(TrampleAbility.getInstance(), Duration.EndOfTurn)
45              .setText("and gains trample until end of turn");
46          ability.addEffect(effect);
47          ability.addTarget(new TargetCreaturePermanent(0, 1));
48          this.addAbility(ability);
49  
50          // âˆ’2: Create a 3/3 green Beast creature token. Then if an opponent controls more creatures than you, put a loyalty counter on Garruk, Unleashed.
51          ability = new LoyaltyAbility(new CreateTokenEffect(new BeastToken()), -2);
52          ability.addEffect(new ConditionalOneShotEffect(
53                  new AddCountersSourceEffect(CounterType.LOYALTY.createInstance()),
54                  new OpponentControlsMoreCondition(new FilterCreaturePermanent()))
55                  .setText("Then if an opponent controls more creatures than you, put a loyalty counter on {this}"));
56          this.addAbility(ability);
57  
58          // âˆ’7: You get an emblem with "At the beginning of your end step, you may search your library for a creature card, put it onto the battlefield, then shuffle your library."
59          this.addAbility(new LoyaltyAbility(new GetEmblemEffect(new GarrukUnleashedEmblem()), -7));
60      }
61  
62      private GarrukUnleashed(final GarrukUnleashed card) {
63          super(card);
64      }
65  
66      @Override
67      public GarrukUnleashed copy() {
68          return new GarrukUnleashed(this);
69      }
70  }

5   /**
6    * Inner our outer/draw spaces in cards (example: if you want add free space for animation for icons drawing)
7    *
8    * @author JayDi85
9    */
10  public class MageCardSpace {
11  
12      public static final MageCardSpace empty = new MageCardSpace(0, 0, 0, 0);
13  
14      int left;
15      int right;
16      int top;
17      int bottom;
18  
19      Color debugColor = null; // add colored border for draw debug
20  
21      public MageCardSpace(int left, int right, int top, int bottom) {
22          this.left = left;
23          this.right = right;
24          this.top = top;
25          this.bottom = bottom;
26      }
27  
28      public MageCardSpace withDebugColor(Color color) {
29          this.debugColor = color;
30          return this;
31      }
32  
33      public int getLeft() {
34          return left;
35      }
36  
37      public int getRight() {
38          return right;
39      }
40  
41      public int getTop() {
42          return top;
43      }
44  
45      public int getBottom() {
46          return bottom;
47      }
48  
49      public int getWidth() {
50          return this.left + this.right;
51      }
52  
53      public int getHeight() {
54          return this.top + this.bottom;
55      }
56  
57      public Color getDebugColor() {
58          return debugColor;
59      }
60  
61      /**
62       * Creates combined space (sums all values from each space and clear a debug color)
63       *
64       * @param spaces list of spaces to combine
65       * @return
66       */
67      public static MageCardSpace combine(MageCardSpace... spaces) {
68          MageCardSpace res = new MageCardSpace(0, 0, 0, 0);
69          for (MageCardSpace space : spaces) {
70              res.left += space.left;
71              res.right += space.right;
72              res.top += space.top;
73              res.bottom += space.bottom;
74          }
75          return res;
76      }
77  }

10  import mage.abilities.keyword.IslandwalkAbility;

20   * @author North, noxx
21   */
22  public final class HalimarWavewatch extends LevelerCard {
23  
24      public HalimarWavewatch(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{U}");
26          this.subtype.add(SubType.MERFOLK);
27          this.subtype.add(SubType.SOLDIER);
28  
29          this.color.setBlue(true);
30          this.power = new MageInt(0);
31          this.toughness = new MageInt(3);
32  
33          this.addAbility(new LevelUpAbility(new ManaCostsImpl<>("{2}")));
34  
35          Abilities<Ability> levelAbilities = new AbilitiesImpl<>();
36          levelAbilities.add(new IslandwalkAbility());
37  
38          this.addAbilities(LevelerCardBuilder.construct(
39                  new LevelerCardBuilder.LevelAbility(1, 4, new AbilitiesImpl<>(), 0, 6),
40                  new LevelerCardBuilder.LevelAbility(5, -1, levelAbilities, 6, 6)
41          ));
42          setMaxLevelCounters(5);
43      }
44  
45      private HalimarWavewatch(final HalimarWavewatch card) {
46          super(card);
47      }
48  
49      @Override
50      public HalimarWavewatch copy() {
51          return new HalimarWavewatch(this);
52      }
53  }

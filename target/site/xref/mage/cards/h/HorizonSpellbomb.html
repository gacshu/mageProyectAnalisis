12  import mage.cards.CardImpl;

15  import mage.constants.Zone;
16  import mage.filter.StaticFilters;

19  import java.util.UUID;

21  /**
22   *
23   * @author North
24   */
25  public final class HorizonSpellbomb extends CardImpl {
26  
27      public HorizonSpellbomb(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT},"{1}");
29  
30          // {2}, {tap}, Sacrifice Horizon Spellbomb: Search your library for a basic land card, reveal it, and put it into your hand. Then shuffle your library.
31          SimpleActivatedAbility ability = new SimpleActivatedAbility(Zone.BATTLEFIELD,
32                  new SearchLibraryPutInHandEffect(new TargetCardInLibrary(StaticFilters.FILTER_CARD_BASIC_LAND), true),
33                  new ManaCostsImpl<>("{2}"));
34          ability.addCost(new TapSourceCost());
35          ability.addCost(new SacrificeSourceCost());
36          this.addAbility(ability);
37          // When Horizon Spellbomb is put into a graveyard from the battlefield, you may pay {G}. If you do, draw a card.
38          this.addAbility(new PutIntoGraveFromBattlefieldSourceTriggeredAbility(new DoIfCostPaid(new DrawCardSourceControllerEffect(1), new ManaCostsImpl<>("{G}"))));
39      }
40  
41      private HorizonSpellbomb(final HorizonSpellbomb card) {
42          super(card);
43      }
44  
45      @Override
46      public HorizonSpellbomb copy() {
47          return new HorizonSpellbomb(this);
48      }
49  }

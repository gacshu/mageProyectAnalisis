4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
8   import mage.abilities.condition.common.PermanentsOnTheBattlefieldCondition;
9   import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
10  import mage.abilities.effects.common.PutOnLibraryTargetEffect;
11  import mage.abilities.effects.common.discard.DiscardTargetEffect;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.ComparisonType;
16  import mage.constants.SubType;
17  import mage.filter.common.FilterLandPermanent;
18  import mage.target.TargetPlayer;
19  import mage.target.common.TargetCardInYourGraveyard;
20  
21  /**
22   *
23   * @author jeffwadsworth
24   *
25   */
26  public final class HagHedgeMage extends CardImpl {
27  
28      private static final FilterLandPermanent filter = new FilterLandPermanent();
29      private static final FilterLandPermanent filter2 = new FilterLandPermanent();
30  
31      static {
32          filter.add(SubType.SWAMP.getPredicate());
33          filter2.add(SubType.FOREST.getPredicate());
34      }
35  
36      private static final String rule = "When {this} enters the battlefield, if you control two or more Swamps, you may have target player discard a card.";
37      private static final String rule2 = "When {this} enters the battlefield, if you control two or more Forests, you may put target card from your graveyard on top of your library.";
38  
39      public HagHedgeMage(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{2}{B/G}");
41          this.subtype.add(SubType.HAG);
42          this.subtype.add(SubType.SHAMAN);
43  
44          this.power = new MageInt(2);
45          this.toughness = new MageInt(2);
46  
47          // When Hag Hedge-Mage enters the battlefield, if you control two or more Swamps, you may have target player discard a card.
48          Ability ability = new ConditionalInterveningIfTriggeredAbility(new EntersBattlefieldTriggeredAbility(new DiscardTargetEffect(1), true), new PermanentsOnTheBattlefieldCondition(filter, ComparisonType.MORE_THAN, 1), rule);
49          ability.addTarget(new TargetPlayer());
50          this.addAbility(ability);
51  
52          // When Hag Hedge-Mage enters the battlefield, if you control two or more Forests, you may put target card from your graveyard on top of your library.
53          Ability ability2 = new ConditionalInterveningIfTriggeredAbility(new EntersBattlefieldTriggeredAbility(new PutOnLibraryTargetEffect(true), true), new PermanentsOnTheBattlefieldCondition(filter2, ComparisonType.MORE_THAN, 1), rule2);
54          ability2.addTarget(new TargetCardInYourGraveyard());
55          this.addAbility(ability2);
56      }
57  
58      private HagHedgeMage(final HagHedgeMage card) {
59          super(card);
60      }
61  
62      @Override
63      public HagHedgeMage copy() {
64          return new HagHedgeMage(this);
65      }
66  }

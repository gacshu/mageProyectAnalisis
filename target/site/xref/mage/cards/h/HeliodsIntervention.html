21  /**
22   * @author TheElk801
23   */
24  public final class HeliodsIntervention extends CardImpl {
25  
26      private static final DynamicValue xValue = new MultipliedValue(ManacostVariableValue.REGULAR, 2);
27  
28      public HeliodsIntervention(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{X}{W}{W}");
30  
31          // Choose one —
32          // • Destroy X target artifacts and/or enchantments.
33          this.getSpellAbility().addEffect(new DestroyTargetEffect()
34                  .setText("Destroy X target artifacts and/or enchantments."));
35          this.getSpellAbility().setTargetAdjuster(HeliodsInterventionAdjuster.instance);
36  
37          // • Target player gains twice X life.
38          Mode mode = new Mode(new GainLifeTargetEffect(xValue).setText("Target player gains twice X life"));
39          mode.addTarget(new TargetPlayer());
40          this.getSpellAbility().addMode(mode);
41      }
42  
43      private HeliodsIntervention(final HeliodsIntervention card) {
44          super(card);
45      }
46  
47      @Override
48      public HeliodsIntervention copy() {
49          return new HeliodsIntervention(this);
50      }
51  }
52  
53  enum HeliodsInterventionAdjuster implements TargetAdjuster {
54      instance;
55  
56      @Override
57      public void adjustTargets(Ability ability, Game game) {
58          Mode mode = ability.getModes().getMode();
59          if (mode.getEffects().stream().noneMatch(DestroyTargetEffect.class::isInstance)) {
60              return;
61          }
62          mode.getTargets().clear();
63          mode.addTarget(new TargetPermanent(
64                  ability.getManaCostsToPay().getX(), StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_ENCHANTMENT
65          ));
66      }
67  }

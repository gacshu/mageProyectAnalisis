15  
16  import java.util.UUID;
17  
18  /**
19   * @author TheElk801
20   */
21  public final class HexgoldHoverwings extends CardImpl {
22  
23      private static final FilterCreaturePermanent filter
24              = new FilterCreaturePermanent("creatures you control that are equipped");
25  
26      static {
27          filter.add(TargetController.YOU.getControllerPredicate());
28          filter.add(EquippedPredicate.instance);
29      }
30  
31      public HexgoldHoverwings(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}{W}");
33  
34          this.subtype.add(SubType.EQUIPMENT);
35  
36          // For Mirrodin!
37          this.addAbility(new ForMirrodinAbility());
38  
39          // Equipped creature has flying.
40          this.addAbility(new SimpleStaticAbility(new GainAbilityAttachedEffect(
41                  FlyingAbility.getInstance(), AttachmentType.EQUIPMENT
42          )));
43  
44          // Creatures you control that are equipped get +1/+0.
45          this.addAbility(new SimpleStaticAbility(new BoostAllEffect(
46                  1, 0, Duration.WhileOnBattlefield, filter, false
47          )));
48  
49          // Equip {2}{W}
50          this.addAbility(new EquipAbility(Outcome.AddAbility, new ManaCostsImpl<>("{2}{W}"), false));
51      }
52  
53      private HexgoldHoverwings(final HexgoldHoverwings card) {
54          super(card);
55      }
56  
57      @Override
58      public HexgoldHoverwings copy() {
59          return new HexgoldHoverwings(this);
60      }
61  }

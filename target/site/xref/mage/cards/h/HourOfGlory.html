29      public HourOfGlory(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{B}");
31  
32          // Exile target creature. If that creature was a God, its controller reveals their hand and exiles all cards with the same name as that creature.
33          this.getSpellAbility().addEffect(new HourOfGloryEffect());
34          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
35      }

38          super(card);
39      }

47  class HourOfGloryEffect extends OneShotEffect {
48  
49      public HourOfGloryEffect() {
50          super(Outcome.Exile);
51          this.staticText = "Exile target creature. If that creature was a God, its controller reveals their hand and exiles all cards with the same name as that creature";
52      }
53  
54      public HourOfGloryEffect(final HourOfGloryEffect effect) {
55          super(effect);
56      }
57  
58      @Override
59      public HourOfGloryEffect copy() {
60          return new HourOfGloryEffect(this);
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          Player controller = game.getPlayer(source.getControllerId());
66          MageObject sourceObject = source.getSourceObject(game);
67          if (controller != null && sourceObject != null) {
68              Permanent targetCreature = game.getPermanent(getTargetPointer().getFirst(game, source));
69              if (targetCreature != null) {
70                  controller.moveCards(targetCreature, Zone.EXILED, source, game);
71                  if (targetCreature.hasSubtype(SubType.GOD, game)) {
72                      game.getState().processAction(game);
73                      Player targetController = game.getPlayer(targetCreature.getControllerId());
74                      if (targetController != null) {
75                          targetController.revealCards(sourceObject.getIdName(), targetController.getHand(), game);
76                          Set<Card> toExile = new HashSet<>();
77                          for (Card card : targetController.getHand().getCards(game)) {
78                              if (card.getName().equals(targetCreature.getName())) {
79                                  toExile.add(card);
80                              }
81                          }
82                          targetController.moveCards(toExile, Zone.EXILED, source, game);
83                      }
84                  }
85              }
86              return true;
87          }
88          return false;
89      }
90  }

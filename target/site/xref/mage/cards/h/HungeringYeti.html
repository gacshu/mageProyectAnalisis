10  import mage.cards.CardImpl;

12  import mage.constants.CardType;
13  import mage.constants.Duration;
14  import mage.constants.SubType;

16  import mage.filter.common.FilterControlledPermanent;
17  import mage.filter.predicate.Predicates;
18  import mage.filter.predicate.mageobject.ColorPredicate;
19  
20  import java.util.UUID;
21  
22  /**
23   * @author LevelX2
24   */
25  public final class HungeringYeti extends CardImpl {
26  
27      private static final FilterControlledPermanent filter = new FilterControlledPermanent();
28  
29      static {
30          filter.add(Predicates.or(
31                  new ColorPredicate(ObjectColor.GREEN),
32                  new ColorPredicate(ObjectColor.BLUE)
33          ));
34      }
35  
36      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter);
37  
38      public HungeringYeti(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{R}");
40          this.subtype.add(SubType.YETI);
41          this.power = new MageInt(4);
42          this.toughness = new MageInt(4);
43  
44          // As long as you control a green or blue permanent, you may cast Hungering Yeti as though it had flash.
45          this.addAbility(new SimpleStaticAbility(Zone.ALL, new ConditionalAsThoughEffect(
46                  new CastAsThoughItHadFlashSourceEffect(Duration.EndOfGame), condition
47          ).setText("as long as you control a green or blue permanent, you may cast {this} as though it had flash")));
48      }
49  
50      private HungeringYeti(final HungeringYeti card) {
51          super(card);
52      }
53  
54      @Override
55      public HungeringYeti copy() {
56          return new HungeringYeti(this);
57      }
58  }

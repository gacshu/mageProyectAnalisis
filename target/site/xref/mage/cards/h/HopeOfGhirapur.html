34  public final class HopeOfGhirapur extends CardImpl {
35  
36      private static final FilterPlayer filter = new FilterPlayer("player who was dealt combat damage by {this} this turn");
37  
38      static {
39          filter.add(new HopeOfGhirapurPlayerLostLifePredicate());
40      }
41  
42      public HopeOfGhirapur(UUID ownerId, CardSetInfo setInfo) {
43          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{1}");
44  
45          this.addSuperType(SuperType.LEGENDARY);
46          this.subtype.add(SubType.THOPTER);
47          this.power = new MageInt(1);
48          this.toughness = new MageInt(1);
49  
50          // Flying
51          this.addAbility(FlyingAbility.getInstance());
52  
53          // Sacrifice Hope of Ghirapur: Until your next turn, target player who was dealt combat damage by Hope of Ghirapur this turn can't cast noncreature spells.
54          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new HopeOfGhirapurCantCastEffect(), new SacrificeSourceCost());
55          ability.addTarget(new TargetPlayer(1, 1, false, filter));
56          this.addAbility(ability, new HopeOfGhirapurCombatDamageWatcher());
57      }
58  
59      private HopeOfGhirapur(final HopeOfGhirapur card) {
60          super(card);
61      }
62  
63      @Override
64      public HopeOfGhirapur copy() {
65          return new HopeOfGhirapur(this);
66      }
67  }

69  class HopeOfGhirapurCantCastEffect extends ContinuousRuleModifyingEffectImpl {
70  
71      public HopeOfGhirapurCantCastEffect() {
72          super(Duration.UntilYourNextTurn, Outcome.Benefit);
73          staticText = "Until your next turn, target player who was dealt combat damage by {this} this turn can't cast noncreature spells";
74      }
75  
76      public HopeOfGhirapurCantCastEffect(final HopeOfGhirapurCantCastEffect effect) {
77          super(effect);
78      }
79  
80      @Override
81      public HopeOfGhirapurCantCastEffect copy() {
82          return new HopeOfGhirapurCantCastEffect(this);
83      }
84  
85      @Override
86      public boolean apply(Game game, Ability source) {
87          return true;
88      }
89  
90      @Override
91      public String getInfoMessage(Ability source, GameEvent event, Game game) {
92          MageObject mageObject = source.getSourceObject(game);
93          if (mageObject != null) {
94              return "You can't cast noncreature spells this turn (you were dealt damage by " + mageObject.getLogName() + ')';
95          }
96          return null;
97      }
98  
99      @Override
100     public boolean checksEventType(GameEvent event, Game game) {
101         return event.getType() == GameEvent.EventType.CAST_SPELL;
102     }
103 
104     @Override
105     public boolean applies(GameEvent event, Ability source, Game game) {
106         Player player = game.getPlayer(getTargetPointer().getFirst(game, source));
107         if (player != null && player.getId().equals(event.getPlayerId())) {
108             Card card = game.getCard(event.getSourceId());
109             if (card != null && !card.isCreature(game)) {
110                 return true;
111             }
112         }
113         return false;
114     }
115 }

117 class HopeOfGhirapurPlayerLostLifePredicate implements ObjectSourcePlayerPredicate<Player> {
118 
119     public HopeOfGhirapurPlayerLostLifePredicate() {
120     }
121 
122     @Override
123     public boolean apply(ObjectSourcePlayer<Player> input, Game game) {
124         Player targetPlayer = input.getObject();
125         if (targetPlayer == null) {
126             return false;
127         }
128         HopeOfGhirapurCombatDamageWatcher watcher = game.getState().getWatcher(HopeOfGhirapurCombatDamageWatcher.class);
129         if (watcher != null) {
130             return watcher.playerGotCombatDamage(input.getSourceId(), input.getObject().getId(), game);
131         }
132         return false;
133     }
134 }

136 class HopeOfGhirapurCombatDamageWatcher extends Watcher {
137 
138     private final Map<MageObjectReference, Set<UUID>> combatDamagedPlayers = new HashMap<>();
139 
140     public HopeOfGhirapurCombatDamageWatcher() {
141         super(WatcherScope.GAME);
142     }
143 
144     @Override
145     public void watch(GameEvent event, Game game) {
146         if (event.getType() == GameEvent.EventType.DAMAGED_PLAYER && ((DamagedPlayerEvent) event).isCombatDamage()) {
147             MageObjectReference damager = new MageObjectReference(event.getSourceId(), game);
148             Set<UUID> players;
149             if (combatDamagedPlayers.containsKey(damager)) {
150                 players = combatDamagedPlayers.get(damager);
151             } else {
152                 players = new HashSet<>();
153                 combatDamagedPlayers.put(damager, players);
154             }
155             players.add(event.getTargetId());
156         }
157     }
158 
159     /**
160      * Checks if the current object has damaged the player during the current
161      * turn.
162      *
163      * @param objectId
164      * @param playerId
165      * @return
166      */
167     public boolean playerGotCombatDamage(UUID objectId, UUID playerId, Game game) {
168         StackObject stackObject = game.getState().getStack().getStackObject(objectId);
169         MageObjectReference mor;
170         if (stackObject instanceof StackAbility) {
171             // This is neccessary because the source object was sacrificed as cost and the correct zone change counter for target calid check can only be get from stack
172             mor = new MageObjectReference(objectId, ((StackAbility) stackObject).getSourceObjectZoneChangeCounter(), game);
173         } else {
174             mor = new MageObjectReference(objectId, game);
175         }
176         if (combatDamagedPlayers.containsKey(mor)) {
177             return combatDamagedPlayers.get(mor).contains(playerId);
178         }
179         return false;
180     }
181 
182     @Override
183     public void reset() {
184         super.reset();
185         combatDamagedPlayers.clear();
186     }
187 }

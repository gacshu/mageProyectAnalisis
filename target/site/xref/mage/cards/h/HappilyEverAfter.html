29  public final class HappilyEverAfter extends CardImpl {
30  
31      public HappilyEverAfter(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
33  
34          // When Happily Ever After enters the battlefield, each player gains 5 life and draws a card.
35          this.addAbility(new EntersBattlefieldTriggeredAbility(new HappilyEverAfterEffect()));
36  
37          // At the beginning of your upkeep, if there are five colors among permanents you control, there are six or more card types among permanents you control and/or cards in your graveyard, and your life total is greater than or equal to your starting life total, you win the game.
38          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
39                  new BeginningOfUpkeepTriggeredAbility(
40                          new WinGameSourceControllerEffect(), TargetController.YOU, false
41                  ), HappilyEverAfterCondition.instance, "At the beginning of your upkeep, " +
42                  "if there are five colors among permanents you control, there are six or more card types " +
43                  "among permanents you control and/or cards in your graveyard, and your life total is " +
44                  "greater than or equal to your starting life total, you win the game.")
45                  .addHint(HappilyEverAfterColorHint.instance)
46                  .addHint(HappilyEverAfterCardTypeHint.instance)
47                  .addHint(HappilyEverAfterLifeHint.instance)
48          );
49      }
50  
51      private HappilyEverAfter(final HappilyEverAfter card) {
52          super(card);
53      }
54  
55      @Override
56      public HappilyEverAfter copy() {
57          return new HappilyEverAfter(this);
58      }
59  }

61  class HappilyEverAfterEffect extends OneShotEffect {
62  
63      HappilyEverAfterEffect() {
64          super(Outcome.GainLife);
65          staticText = "each player gains 5 life and draws a card";
66      }
67  
68      private HappilyEverAfterEffect(final HappilyEverAfterEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public HappilyEverAfterEffect copy() {
74          return new HappilyEverAfterEffect(this);
75      }
76  
77      @Override
78      public boolean apply(Game game, Ability source) {
79          game.getState()
80                  .getPlayersInRange(source.getControllerId(), game)
81                  .stream()
82                  .map(game::getPlayer)
83                  .filter(Objects::nonNull)
84                  .forEachOrdered(player -> {
85                      player.gainLife(5, game, source);
86                      player.drawCards(1, source, game);
87                  });
88          return true;
89      }
90  }

92  enum HappilyEverAfterCondition implements Condition {
93      instance;
94  
95      @Override
96      public boolean apply(Game game, Ability source) {
97          Player player = game.getPlayer(source.getControllerId());
98          if (player == null || player.getLife() < game.getStartingLife()) {
99              return false;
100         }
101         ObjectColor color = new ObjectColor("");
102         game.getBattlefield()
103                 .getAllActivePermanents(source.getControllerId())
104                 .stream()
105                 .map(permanent -> permanent.getColor(game))
106                 .forEach(color::addColor);
107         if (color.getColorCount() < 5) {
108             return false;
109         }
110         EnumSet<CardType> cardTypeEnumSet = EnumSet.noneOf(CardType.class);
111         game.getBattlefield()
112                 .getAllActivePermanents(source.getControllerId())
113                 .stream()
114                 .map(permanent -> permanent.getCardType(game))
115                 .flatMap(Collection::stream)
116                 .forEach(cardTypeEnumSet::add);
117         if (cardTypeEnumSet.size() >= 6) {
118             return true;
119         }
120         player.getGraveyard()
121                 .getCards(game)
122                 .stream()
123                 .map(card -> card.getCardType(game))
124                 .flatMap(Collection::stream)
125                 .forEach(cardTypeEnumSet::add);
126         return cardTypeEnumSet.size() >= 6;
127     }

130 enum HappilyEverAfterColorHint implements Hint {
131     instance;
132 
133     @Override
134     public String getText(Game game, Ability ability) {
135         Player player = game.getPlayer(ability.getControllerId());
136         if (player == null) {
137             return null;
138         }
139         ObjectColor color = new ObjectColor("");
140         game.getBattlefield()
141                 .getAllActivePermanents(ability.getControllerId())
142                 .stream()
143                 .map(permanent -> permanent.getColor(game))
144                 .forEach(color::addColor);
145         if (color.isColorless()) {
146             return HintUtils.prepareText("Colors of permanents you control: 0 (None)", null, HintUtils.HINT_ICON_BAD);
147         }
148         List<String> colors = new ArrayList<>();
149         if (color.isWhite()) {
150             colors.add("white");
151         }
152         if (color.isBlue()) {
153             colors.add("blue");
154         }
155         if (color.isBlack()) {
156             colors.add("black");
157         }
158         if (color.isRed()) {
159             colors.add("red");
160         }
161         if (color.isGreen()) {
162             colors.add("green");
163         }
164 
165         String hintText = "Colors of permanents you control: " + color.getColorCount() + " (" + colors.stream().reduce((a, b) -> a + ", " + b).get() + ")";
166 
167         if (color.getColors().size() == 5) {
168             return HintUtils.prepareText(hintText, null, HintUtils.HINT_ICON_GOOD);
169         } else {
170             return HintUtils.prepareText(hintText, null, HintUtils.HINT_ICON_BAD);
171         }
172     }
173 
174     @Override
175     public HappilyEverAfterColorHint copy() {
176         return instance;
177     }

180 enum HappilyEverAfterCardTypeHint implements Hint {
181     instance;
182 
183     @Override
184     public String getText(Game game, Ability ability) {
185         Player player = game.getPlayer(ability.getControllerId());
186         if (player == null) {
187             return null;
188         }
189         Set<CardType> cardTypeSet = EnumSet.noneOf(CardType.class);
190         game.getBattlefield()
191                 .getAllActivePermanents(ability.getControllerId())
192                 .stream()
193                 .map(permanent -> permanent.getCardType(game))
194                 .flatMap(Collection::stream)
195                 .forEach(cardTypeSet::add);
196         player.getGraveyard()
197                 .getCards(game)
198                 .stream()
199                 .map(card -> card.getCardType(game))
200                 .flatMap(Collection::stream)
201                 .forEach(cardTypeSet::add);
202         if (cardTypeSet.isEmpty()) {
203             return HintUtils.prepareText("Card types on battlefield and in graveyard: None", null, HintUtils.HINT_ICON_BAD);
204         }
205         String message = cardTypeSet
206                 .stream()
207                 .sorted()
208                 .map(CardType::toString)
209                 .reduce((a, b) -> a + ", " + b)
210                 .get();
211         return HintUtils.prepareText(
212                 "Card types on battlefield and in graveyard: " + cardTypeSet.size() + " (" + message + ")",
213                 null,
214                 cardTypeSet.size() >= 6 ? HintUtils.HINT_ICON_GOOD : HintUtils.HINT_ICON_BAD);
215     }
216 
217     @Override
218     public HappilyEverAfterCardTypeHint copy() {
219         return instance;
220     }
221 }

223 enum HappilyEverAfterLifeHint implements Hint {
224     instance;
225 
226     @Override
227     public String getText(Game game, Ability ability) {
228         Player player = game.getPlayer(ability.getControllerId());
229         if (player == null) {
230             return null;
231         }
232 
233         if (player.getLife() >= game.getStartingLife()) {
234             return HintUtils.prepareText("Life total is greater than or equal to your starting life total", null, HintUtils.HINT_ICON_GOOD);
235         } else {
236             return HintUtils.prepareText("Life total less than starting amount", null, HintUtils.HINT_ICON_BAD);
237         }
238     }

241     public Hint copy() {
242         return instance;
243     }

29  /**
30   *
31   * @author Saga
32   */
33  public final class HungryLynx extends CardImpl {
34      
35      private static final FilterControlledCreaturePermanent filterCat = new FilterControlledCreaturePermanent("Cats");
36      static {
37          filterCat.add(SubType.CAT.getPredicate());
38      }
39      
40      private static final FilterCard filterProRat = new FilterCard("Rats");
41      static {
42          filterProRat.add(SubType.RAT.getPredicate());
43      }
44      
45      private static final FilterCreaturePermanent filterRat = new FilterCreaturePermanent("a Rat");
46      static {
47          filterRat.add(SubType.RAT.getPredicate());
48      }
49      
50      public HungryLynx(UUID ownerId, CardSetInfo setInfo) {
51          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{G}");
52          this.subtype.add(SubType.CAT);
53          this.power = new MageInt(2);
54          this.toughness = new MageInt(2);
55  
56          // Cats you control have protection from Rats.
57          Effect effect = new GainAbilityAllEffect(new ProtectionAbility(filterProRat), Duration.WhileOnBattlefield, filterCat);
58          effect.setText("Cats you control have protection from Rats. <i>(They can't be blocked, targeted, or dealt damage by Rats.)</i>");
59          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
60          
61          // At the beginning of your end step, target opponent creates a 1/1 black Rat creature token with deathtouch. 
62          Ability ability = new BeginningOfEndStepTriggeredAbility(Zone.BATTLEFIELD, new CreateTokenTargetEffect(new DeathtouchRatToken()), TargetController.YOU, null, false);
63          Target target = new TargetOpponent();
64          ability.addTarget(target);
65          this.addAbility(ability);
66          
67          // Whenever a Rat dies, put a +1/+1 counter on each Cat you control. 
68          Effect effect2 = new AddCountersAllEffect(CounterType.P1P1.createInstance(), filterCat);
69          effect2.setText("put a +1/+1 counter on each Cat you control");
70          this.addAbility(new DiesCreatureTriggeredAbility(effect2, false, filterRat));
71      }
72  
73      private HungryLynx(final HungryLynx card) {
74          super(card);
75      }
76  
77      @Override
78      public HungryLynx copy() {
79          return new HungryLynx(this);
80      }
81  }

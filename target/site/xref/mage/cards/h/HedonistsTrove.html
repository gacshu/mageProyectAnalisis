27  public final class HedonistsTrove extends CardImpl {
28  
29      public HedonistsTrove(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{5}{B}{B}");
31  
32          // When Hedonist's Trove enters the battlefield, exile all cards from target opponent's graveyard.
33          Ability ability = new EntersBattlefieldTriggeredAbility(new HedonistsTroveExileEffect());
34          ability.addTarget(new TargetOpponent());
35          this.addAbility(ability);
36  
37          // You may play land cards exiled by Hedonist's Trove.
38          this.addAbility(new SimpleStaticAbility(new HedonistsTrovePlayLandEffect()));
39  
40          // You may cast nonland cards exiled with Hedonist's Trove. You can't cast more than one spell this way each turn.
41          this.addAbility(new SimpleStaticAbility(new HedonistsTroveCastNonlandCardsEffect()), new HedonistsTroveWatcher());
42      }
43  
44      private HedonistsTrove(final HedonistsTrove card) {
45          super(card);
46      }
47  
48      @Override
49      public HedonistsTrove copy() {
50          return new HedonistsTrove(this);
51      }
52  }

54  class HedonistsTroveExileEffect extends OneShotEffect {
55  
56      HedonistsTroveExileEffect() {
57          super(Outcome.Exile);
58          staticText = "exile target opponent's graveyard";
59      }
60  
61      private HedonistsTroveExileEffect(final HedonistsTroveExileEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public HedonistsTroveExileEffect copy() {
67          return new HedonistsTroveExileEffect(this);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          Player controller = game.getPlayer(source.getControllerId());
73          Player targetPlayer = game.getPlayer(source.getFirstTarget());
74          UUID exileId = CardUtil.getExileZoneId(game, source);
75          // save the exileId associated with this specific source
76          game.getState().setValue(source.getSourceId().toString(), exileId);
77          return controller != null
78                  && targetPlayer != null
79                  && controller.moveCardsToExile(
80                  targetPlayer.getGraveyard().getCards(game), source, game, true,
81                  exileId, CardUtil.getSourceName(game, source)
82          );
83      }
84  }

86  class HedonistsTrovePlayLandEffect extends AsThoughEffectImpl {
87  
88      HedonistsTrovePlayLandEffect() {
89          super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.WhileOnBattlefield, Outcome.Benefit);
90          staticText = "You may play lands from among cards exiled with {this}";
91      }
92  
93      private HedonistsTrovePlayLandEffect(final HedonistsTrovePlayLandEffect effect) {
94          super(effect);
95      }
96  
97      @Override
98      public boolean apply(Game game, Ability source) {
99          return true;
100     }
101 
102     @Override
103     public HedonistsTrovePlayLandEffect copy() {
104         return new HedonistsTrovePlayLandEffect(this);
105     }
106 
107     @Override
108     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
109         Card cardToCheck = game.getCard(objectId);
110         if (cardToCheck == null
111                 || !cardToCheck.isLand(game)
112                 || !source.isControlledBy(affectedControllerId)) {
113             return false;
114         }
115         // use the correct exileId
116         UUID exileId = (UUID) game.getState().getValue(source.getSourceId().toString());
117         ExileZone exileZone = game.getExile().getExileZone(exileId);
118         return exileZone != null
119                 && exileZone.contains(cardToCheck.getMainCard().getId());
120     }
121 }

123 class HedonistsTroveCastNonlandCardsEffect extends AsThoughEffectImpl {

125     HedonistsTroveCastNonlandCardsEffect() {
126         super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.WhileOnBattlefield, Outcome.Benefit);
127         staticText = "You may cast spells from among cards exiled with {this}. "
128                 + "You can't cast more than one spell this way each turn.";

132         super(effect);
133     }
134 
135     @Override
136     public boolean apply(Game game, Ability source) {
137         return true;
138     }
139 
140     @Override
141     public HedonistsTroveCastNonlandCardsEffect copy() {
142         return new HedonistsTroveCastNonlandCardsEffect(this);
143     }
144 
145     @Override
146     public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
147         HedonistsTroveWatcher watcher = game.getState().getWatcher(HedonistsTroveWatcher.class);
148         if (watcher == null
149                 || !watcher.checkPlayer(affectedControllerId, source, game)) {
150             return false;
151         }
152         Card cardToCheck = game.getCard(objectId);
153         // use the correct exileId
154         UUID exileId = (UUID) game.getState().getValue(source.getSourceId().toString());
155         ExileZone exileZone = game.getExile().getExileZone(exileId);
156         return cardToCheck != null
157                 && !cardToCheck.isLand(game)
158                 && source.isControlledBy(affectedControllerId)
159                 && exileZone != null
160                 && exileZone.contains(cardToCheck.getMainCard().getId());
161     }
162 }

164 class HedonistsTroveWatcher extends Watcher {
165 
166     private static final Set<MageObjectReference> emptySet = new HashSet<>();
167     private final Map<UUID, Set<MageObjectReference>> playerMap = new HashMap<>();
168 
169     HedonistsTroveWatcher() {
170         super(WatcherScope.GAME);
171     }
172 
173     @Override
174     public void watch(GameEvent event, Game game) {
175         if (event.getType() != GameEvent.EventType.SPELL_CAST
176                 || event.getAdditionalReference() == null) {
177             return;
178         }
179         playerMap
180                 .computeIfAbsent(event.getPlayerId(), x -> new HashSet<>())
181                 .add(event.getAdditionalReference().getApprovingMageObjectReference());
182         playerMap.get(event.getPlayerId()).removeIf(Objects::isNull);
183     }
184 
185     boolean checkPlayer(UUID playerId, Ability source, Game game) {
186         Permanent permanent = source.getSourcePermanentOrLKI(game);
187         return permanent != null
188                 && playerMap
189                 .getOrDefault(playerId, emptySet)
190                 .stream()
191                 .noneMatch(mor -> mor.refersTo(permanent, game));
192     }
193 
194     @Override
195     public void reset() {
196         super.reset();
197         playerMap.clear();
198     }
199 }

39      public EverythingamajigE(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{5}");
41  
42          // Zuran Orb
43          // Sacrifice a land: You gain 2 life.
44          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new GainLifeEffect(2), new SacrificeTargetCost(new TargetControlledPermanent(StaticFilters.FILTER_CONTROLLED_LAND_SHORT_TEXT))));
45  
46          // Ashnod's Altar
47          // Sacrifice a creature: Add {C}{C} to your mana pool.
48          SacrificeTargetCost cost = new SacrificeTargetCost(new TargetControlledCreaturePermanent(StaticFilters.FILTER_CONTROLLED_CREATURE_SHORT_TEXT));
49          this.addAbility(new SimpleManaAbility(Zone.BATTLEFIELD,
50                  new BasicManaEffect(Mana.ColorlessMana(2), CreaturesYouControlCount.instance),
51                  cost));
52  
53          // Urza's Hot Tub
54          // 2, Discard a card: Search your library for a card that shares a complete word in its name with the name of the discarded card, reveal it, put it into your hand, then shuffle your library.
55          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new UrzasHotTubEffect(), new GenericManaCost(2));
56          ability.addCost(new DiscardTargetCost(new TargetCardInHand()));
57          this.addAbility(ability);
58      }

70  class UrzasHotTubEffect extends OneShotEffect {
71  
72      public UrzasHotTubEffect() {
73          super(Outcome.ReturnToHand);
74          this.staticText = "Search your library for a card that shares a complete word in its name with the discarded card, reveal it, put it into your hand, then shuffle";
75      }
76  
77      public UrzasHotTubEffect(final UrzasHotTubEffect effect) {
78          super(effect);
79      }
80  
81      @Override
82      public UrzasHotTubEffect copy() {
83          return new UrzasHotTubEffect(this);
84      }
85  
86      @Override
87      public boolean apply(Game game, Ability source) {
88          for (Cost cost : source.getCosts()) {
89              if (cost instanceof DiscardTargetCost) {
90                  DiscardTargetCost discardCost = (DiscardTargetCost) cost;
91                  Card discardedCard = discardCost.getCards().get(0);
92                  if (discardedCard != null) {
93                      FilterCard filter = new FilterCard();
94                      filter.add(new UrzasHotTubPredicate(discardedCard.getName()));
95                      return new SearchLibraryPutInHandEffect(new TargetCardInLibrary(filter), true, true).apply(game, source);
96                  }
97              }
98          }
99          return false;
100     }
101 }

103 class UrzasHotTubPredicate implements Predicate<MageObject> {
104 
105     private final String referenceName;
106 
107     public UrzasHotTubPredicate(String referenceName) {
108         this.referenceName = referenceName;
109     }
110 
111     @Override
112     public boolean apply(MageObject input, Game game) {
113         String name = input.getName();
114         if (input instanceof SplitCard) {
115             return sharesWordWithName(((SplitCard) input).getLeftHalfCard().getName()) || sharesWordWithName(((SplitCard) input).getRightHalfCard().getName());
116         } else if (input instanceof ModalDoubleFacesCard) {
117             return sharesWordWithName(((ModalDoubleFacesCard) input).getLeftHalfCard().getName()) || sharesWordWithName(((ModalDoubleFacesCard) input).getRightHalfCard().getName());
118         } else if (input instanceof Spell && ((Spell) input).getSpellAbility().getSpellAbilityType() == SpellAbilityType.SPLIT_FUSED) {
119             SplitCard card = (SplitCard) ((Spell) input).getCard();
120             return sharesWordWithName(card.getLeftHalfCard().getName()) || sharesWordWithName(card.getRightHalfCard().getName());
121         } else {
122             if (name.contains(" // ")) {
123                 String leftName = name.substring(0, name.indexOf(" // "));
124                 String rightName = name.substring(name.indexOf(" // ") + 4);
125                 return sharesWordWithName(leftName) || sharesWordWithName(rightName);
126             } else {
127                 return sharesWordWithName(name);
128             }
129         }
130     }
131 
132     private boolean sharesWordWithName(String str) {
133         if (referenceName == null || referenceName.equals("")) {
134             return false;
135         }
136         String[] arr = referenceName.split("\\s+");
137         for (int i = 0; i < arr.length; i++) {
138             if (str.contains(arr[i].replaceAll(",", ""))) {
139                 return true;
140             }
141         }
142         return false;
143     }
144 
145     @Override
146     public String toString() {
147         return "";
148     }
149 }

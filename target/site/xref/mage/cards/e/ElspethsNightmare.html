8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.constants.ComparisonType;
11  import mage.constants.SagaChapter;
12  import mage.constants.SubType;
13  import mage.filter.FilterCard;
14  import mage.filter.FilterPermanent;
15  import mage.filter.common.FilterNonlandCard;
16  import mage.filter.common.FilterOpponentsCreaturePermanent;
17  import mage.filter.predicate.Predicates;
18  import mage.filter.predicate.mageobject.PowerPredicate;
19  import mage.target.TargetPermanent;
20  import mage.target.common.TargetOpponent;
21  
22  import java.util.UUID;
23  
24  /**
25   * @author TheElk801
26   */
27  public final class ElspethsNightmare extends CardImpl {
28  
29      private static final FilterPermanent filter
30              = new FilterOpponentsCreaturePermanent("creature an opponent controls with power 2 or less");
31      private static final FilterCard filter2 = new FilterNonlandCard("noncreature, nonland card");
32  
33      static {
34          filter.add(new PowerPredicate(ComparisonType.FEWER_THAN, 3));
35          filter2.add(Predicates.not(CardType.CREATURE.getPredicate()));
36      }
37  
38      public ElspethsNightmare(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{B}");
40  
41          this.subtype.add(SubType.SAGA);
42  
43          // <i>(As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)</i>
44          SagaAbility sagaAbility = new SagaAbility(this);
45  
46          // I - Destroy target creature an opponent controls with power 2 or less.
47          sagaAbility.addChapterEffect(
48                  this, SagaChapter.CHAPTER_I, SagaChapter.CHAPTER_I,
49                  new DestroyTargetEffect(), new TargetPermanent(filter)
50          );
51  
52          // II - Target opponent reveals their hand. You choose a noncreature, nonland card from it. That player discards that card.
53          sagaAbility.addChapterEffect(
54                  this, SagaChapter.CHAPTER_II, SagaChapter.CHAPTER_II,
55                  new DiscardCardYouChooseTargetEffect(filter2), new TargetOpponent()
56          );
57  
58          // III - Exile target opponent's graveyard.
59          sagaAbility.addChapterEffect(
60                  this, SagaChapter.CHAPTER_III, SagaChapter.CHAPTER_III,
61                  new ExileGraveyardAllTargetPlayerEffect()
62                          .setText("exile target opponent's graveyard"),
63                  new TargetOpponent()
64          );
65          this.addAbility(sagaAbility);
66      }
67  
68      private ElspethsNightmare(final ElspethsNightmare card) {
69          super(card);
70      }
71  
72      @Override
73      public ElspethsNightmare copy() {
74          return new ElspethsNightmare(this);
75      }
76  }

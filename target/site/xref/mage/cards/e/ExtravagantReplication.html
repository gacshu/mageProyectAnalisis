1   package mage.cards.e;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.common.BeginningOfUpkeepTriggeredAbility;
5   import mage.abilities.effects.common.CreateTokenCopyTargetEffect;
6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.constants.TargetController;
10  import mage.filter.FilterPermanent;
11  import mage.filter.common.FilterNonlandPermanent;
12  import mage.filter.predicate.mageobject.AnotherPredicate;
13  import mage.target.TargetPermanent;
14  
15  import java.util.UUID;
16  
17  /**
18   * @author TheElk801
19   */
20  public final class ExtravagantReplication extends CardImpl {
21  
22      private static final FilterPermanent filter
23              = new FilterNonlandPermanent("another target nonland permanent you control");
24  
25      static {
26          filter.add(AnotherPredicate.instance);
27          filter.add(TargetController.YOU.getControllerPredicate());
28      }
29  
30      public ExtravagantReplication(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{4}{U}{U}");
32  
33          // At the beginning of your upkeep, create a token that's a copy of another target nonland permanent you control.
34          Ability ability = new BeginningOfUpkeepTriggeredAbility(
35                  new CreateTokenCopyTargetEffect(), TargetController.YOU, false
36          );
37          ability.addTarget(new TargetPermanent(filter));
38          this.addAbility(ability);
39      }
40  
41      private ExtravagantReplication(final ExtravagantReplication card) {
42          super(card);
43      }
44  
45      @Override
46      public ExtravagantReplication copy() {
47          return new ExtravagantReplication(this);
48      }
49  }

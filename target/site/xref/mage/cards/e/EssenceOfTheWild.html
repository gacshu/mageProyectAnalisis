19  public final class EssenceOfTheWild extends CardImpl {
20  
21      public EssenceOfTheWild(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{G}{G}{G}");
23          this.subtype.add(SubType.AVATAR);
24  
25          this.power = new MageInt(6);
26          this.toughness = new MageInt(6);
27  
28          // Creatures you control enter the battlefield as a copy of Essence of the Wild.
29          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new EssenceOfTheWildEffect()));
30      }
31  
32      private EssenceOfTheWild(final EssenceOfTheWild card) {
33          super(card);
34      }
35  
36      @Override
37      public EssenceOfTheWild copy() {
38          return new EssenceOfTheWild(this);
39      }
40  }
41  
42  class EssenceOfTheWildEffect extends ReplacementEffectImpl {
43  
44      public EssenceOfTheWildEffect() {
45          super(Duration.WhileOnBattlefield, Outcome.Copy, false);
46          staticText = "Creatures you control enter the battlefield as a copy of {this}";
47      }
48  
49      public EssenceOfTheWildEffect(EssenceOfTheWildEffect effect) {
50          super(effect);
51      }
52  
53      @Override
54      public boolean checksEventType(GameEvent event, Game game) {
55          return event.getType() == GameEvent.EventType.ENTERS_THE_BATTLEFIELD;
56      }
57  
58      @Override
59      public boolean applies(GameEvent event, Ability source, Game game) {
60          Permanent perm = ((EntersTheBattlefieldEvent) event).getTarget();
61          return perm != null && perm.isCreature(game) && perm.isControlledBy(source.getControllerId());
62      }
63  
64      @Override
65      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
66          Permanent sourceObject = game.getPermanentOrLKIBattlefield(source.getSourceId());
67          if (sourceObject != null) {
68              game.copyPermanent(sourceObject, event.getTargetId(), source, null);
69          }
70          return false;
71      }
72  
73      @Override
74      public EssenceOfTheWildEffect copy() {
75          return new EssenceOfTheWildEffect(this);
76      }
77  
78  }

25  public final class EngulfTheShore extends CardImpl {
26  
27      public EngulfTheShore(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{3}{U}");
29  
30          // Return to their owners' hands all creatures with toughness less than or equal to the number of Islands you control.
31          getSpellAbility().addEffect(new EngulfTheShoreEffect());
32      }
33  
34      private EngulfTheShore(final EngulfTheShore card) {
35          super(card);
36      }
37  
38      @Override
39      public EngulfTheShore copy() {
40          return new EngulfTheShore(this);
41      }
42  }
43  
44  class EngulfTheShoreEffect extends OneShotEffect {
45  
46      private static final FilterControlledPermanent filter = new FilterControlledPermanent("number of Islands you control");
47  
48      static {
49          filter.add(SubType.ISLAND.getPredicate());
50      }
51  
52      public EngulfTheShoreEffect() {
53          super(Outcome.Benefit);
54          this.staticText = "Return to their owners' hands all creatures with toughness less than or equal to the number of Islands you control";
55      }
56  
57      public EngulfTheShoreEffect(final EngulfTheShoreEffect effect) {
58          super(effect);
59      }
60  
61      @Override
62      public EngulfTheShoreEffect copy() {
63          return new EngulfTheShoreEffect(this);
64      }
65  
66      @Override
67      public boolean apply(Game game, Ability source) {
68          Player controller = game.getPlayer(source.getControllerId());
69          if (controller != null) {
70              int islands = game.getBattlefield().count(filter, source.getControllerId(), source, game);
71              FilterPermanent creatureFilter = new FilterCreaturePermanent();
72              creatureFilter.add(new ToughnessPredicate(ComparisonType.FEWER_THAN, islands + 1));
73              Set<Card> cardsToHand = new HashSet<>();
74              for (Permanent permanent : game.getBattlefield().getActivePermanents(creatureFilter, source.getControllerId(), source, game)) {
75                  cardsToHand.add(permanent);
76              }
77              controller.moveCards(cardsToHand, Zone.HAND, source, game);
78              return true;
79          }
80          return false;
81      }
82  }

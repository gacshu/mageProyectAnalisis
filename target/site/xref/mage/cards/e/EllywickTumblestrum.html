22  public final class EllywickTumblestrum extends CardImpl {

24      public EllywickTumblestrum(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{2}{G}{G}");
26  
27          this.addSuperType(SuperType.LEGENDARY);
28          this.subtype.add(SubType.ELLYWICK);
29          this.setStartingLoyalty(4);
30  
31          // +1: Venture into the dungeon.
32          this.addAbility(new LoyaltyAbility(new VentureIntoTheDungeonEffect(), 1));
33  
34          // −2: Look at the top six cards of your library. You may reveal a creature card from among them and put it into your hand. If it's legendary, you gain 3 life. Put the rest on the bottom of your library in a random order.
35          this.addAbility(new LoyaltyAbility(new EllywickTumblestrumEffect(), -2));
36  
37          // −7: You get an emblem with "Creatures you control have trample and haste and get +2/+2 for each differently named dungeon you've completed."
38          this.addAbility(new LoyaltyAbility(
39                  new GetEmblemEffect(new EllywickTumblestrumEmblem()), -7
40          ), new CompletedDungeonWatcher());
41      }

53  class EllywickTumblestrumEffect extends OneShotEffect {
54  
55      EllywickTumblestrumEffect() {
56          super(Outcome.Benefit);
57          staticText = "look at the top six cards of your library. You may reveal a creature card " +
58                  "from among them and put it into your hand. If it's legendary, you gain 3 life. " +
59                  "Put the rest on the bottom of your library in a random order";
60      }
61  
62      private EllywickTumblestrumEffect(final EllywickTumblestrumEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public EllywickTumblestrumEffect copy() {
68          return new EllywickTumblestrumEffect(this);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          Player player = game.getPlayer(source.getControllerId());
74          if (player == null) {
75              return false;
76          }
77          Cards cards = new CardsImpl(player.getLibrary().getTopCards(game, 6));
78          TargetCardInLibrary target = new TargetCardInLibrary(0, 1, StaticFilters.FILTER_CARD_CREATURE);
79          player.choose(outcome, cards, target, game);
80          Card card = cards.get(target.getFirstTarget(), game);
81          if (card != null) {
82              player.revealCards(source, new CardsImpl(card), game);
83              player.moveCards(card, Zone.HAND, source, game);
84              cards.remove(card);
85              if (card.isLegendary()) {
86                  player.gainLife(3, game, source);
87              }
88          }
89          player.putCardsOnBottomOfLibrary(cards, game, source, false);
90          return true;
91      }

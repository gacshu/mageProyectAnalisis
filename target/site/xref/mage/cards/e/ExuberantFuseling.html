14  import mage.cards.CardImpl;
15  import mage.cards.CardSetInfo;
16  import mage.constants.CardType;
17  import mage.constants.Duration;
18  import mage.constants.SubType;

20  import mage.counters.CounterType;
21  import mage.filter.StaticFilters;
22  
23  import java.util.UUID;
24  
25  public class ExuberantFuseling extends CardImpl {
26  
27      private static final DynamicValue oilCounters = new CountersSourceCount(CounterType.OIL);
28  
29      public ExuberantFuseling(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{R}");
31          this.addSubType(SubType.PHYREXIAN);
32          this.addSubType(SubType.GOBLIN);
33          this.addSubType(SubType.WARRIOR);
34          this.power = new MageInt(0);
35          this.toughness = new MageInt(1);
36  
37          //Trample
38          this.addAbility(TrampleAbility.getInstance());
39  
40          //Exuberant Fuseling gets +1/+0 for each oil counter on it.
41          this.addAbility(new SimpleStaticAbility(new BoostSourceEffect(oilCounters, StaticValue.get(0),
42                  Duration.WhileOnBattlefield).setText("{this} gets +1/+0 for each oil counter on it")));
43  
44          //When Exuberant Fuseling enters the battlefield and whenever another creature or artifact you control is put
45          //into a graveyard from the battlefield, put an oil counter on Exuberant Fuseling.
46          this.addAbility(new OrTriggeredAbility(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.OIL.createInstance()),
47                  new EntersBattlefieldTriggeredAbility(null, false),
48                  new PutIntoGraveFromBattlefieldAllTriggeredAbility(null, false,
49                          StaticFilters.FILTER_CONTROLLED_ANOTHER_ARTIFACT_OR_CREATURE, false))
50                  .setTriggerPhrase("When {this} enters the battlefield and whenever another creature or artifact you " +
51                          "control is put into a graveyard from the battlefield, "));
52      }
53  
54      private ExuberantFuseling(final ExuberantFuseling card) {
55          super(card);
56      }
57  
58      @Override
59      public ExuberantFuseling copy() {
60          return new ExuberantFuseling(this);
61      }
62  }

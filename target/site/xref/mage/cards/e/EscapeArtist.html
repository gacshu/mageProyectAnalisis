5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.SimpleActivatedAbility;
8   import mage.abilities.costs.common.DiscardTargetCost;
9   import mage.abilities.costs.mana.ManaCostsImpl;
10  import mage.abilities.effects.common.ReturnToHandSourceEffect;
11  import mage.abilities.keyword.CantBeBlockedSourceAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.constants.Zone;
17  import mage.target.common.TargetCardInHand;
18  
19  /**
20   *
21   * @author cbt33
22   */
23  public final class EscapeArtist extends CardImpl {
24  
25      public EscapeArtist(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{U}");
27          this.subtype.add(SubType.HUMAN);
28          this.subtype.add(SubType.WIZARD);
29  
30          this.power = new MageInt(1);
31          this.toughness = new MageInt(1);
32  
33          // Escape Artist can't be blocked.
34          this.addAbility(new CantBeBlockedSourceAbility());
35          // {U}, Discard a card: Return Escape Artist to its owner's hand.
36          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new ReturnToHandSourceEffect(true), new ManaCostsImpl<>("{U}"));
37          ability.addCost(new DiscardTargetCost(new TargetCardInHand()));
38          this.addAbility(ability);
39      }
40  
41      private EscapeArtist(final EscapeArtist card) {
42          super(card);
43      }
44  
45      @Override
46      public EscapeArtist copy() {
47          return new EscapeArtist(this);
48      }
49  }

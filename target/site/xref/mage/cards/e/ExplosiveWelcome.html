3   import mage.Mana;
4   import mage.abilities.dynamicvalue.common.StaticValue;
5   import mage.abilities.effects.common.DamageTargetEffect;
6   import mage.abilities.effects.mana.BasicManaEffect;
7   import mage.cards.CardImpl;
8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.constants.ManaType;
11  import mage.filter.common.FilterCreaturePlayerOrPlaneswalker;
12  import mage.filter.predicate.other.AnotherTargetPredicate;
13  import mage.target.Target;
14  import mage.target.common.TargetAnyTarget;
15  import mage.target.targetpointer.SecondTargetPointer;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class ExplosiveWelcome extends CardImpl {
23  
24      private static final FilterCreaturePlayerOrPlaneswalker filter = new FilterCreaturePlayerOrPlaneswalker();
25  
26      static {
27          filter.getPermanentFilter().add(new AnotherTargetPredicate(2));
28          filter.getPlayerFilter().add(new AnotherTargetPredicate(2));
29      }
30  
31      public ExplosiveWelcome(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{7}{R}");
33  
34          // Explosive Welcome deals 5 damage to any target and 3 damage to any other target. Add {R}{R}{R}.
35          this.getSpellAbility().addEffect(new DamageTargetEffect(StaticValue.get(5), true, "", true));
36          this.getSpellAbility().addEffect(
37                  new DamageTargetEffect(StaticValue.get(3), true, "", true)
38                          .setTargetPointer(new SecondTargetPointer())
39                          .setText("and 3 damage to any other target.")
40          );
41          this.getSpellAbility().addEffect(new BasicManaEffect(new Mana(ManaType.RED, 3)));
42          Target target = new TargetAnyTarget();
43          target.setTargetTag(1);
44          this.getSpellAbility().addTarget(target);
45          target = new TargetAnyTarget(filter);
46          target.setTargetTag(2);
47          this.getSpellAbility().addTarget(target);
48      }
49  
50      private ExplosiveWelcome(final ExplosiveWelcome card) {
51          super(card);
52      }
53  
54      @Override
55      public ExplosiveWelcome copy() {
56          return new ExplosiveWelcome(this);
57      }
58  }

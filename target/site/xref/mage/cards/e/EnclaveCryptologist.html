10  import mage.abilities.common.SimpleActivatedAbility;
11  import mage.abilities.costs.common.TapSourceCost;
12  import mage.abilities.costs.mana.ManaCostsImpl;
13  import mage.abilities.effects.common.DrawCardSourceControllerEffect;
14  import mage.abilities.effects.common.DrawDiscardControllerEffect;
15  import mage.abilities.keyword.LevelUpAbility;
16  import mage.abilities.keyword.LevelerCardBuilder;
17  import mage.cards.CardSetInfo;
18  import mage.cards.LevelerCard;
19  import mage.constants.CardType;
20  import mage.constants.SubType;
21  import mage.constants.Zone;

23  /**
24   *
25   * @author BetaSteward_at_googlemail.com, noxx
26   */
27  public final class EnclaveCryptologist extends LevelerCard {
28  
29      public EnclaveCryptologist (UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{U}");
31          this.subtype.add(SubType.MERFOLK);
32          this.subtype.add(SubType.WIZARD);
33          this.color.setBlue(true);
34          this.power = new MageInt(0);
35          this.toughness = new MageInt(1);
36  
37          this.addAbility(new LevelUpAbility(new ManaCostsImpl<>("{1}{U}")));
38  
39          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DrawDiscardControllerEffect(), new TapSourceCost());
40          Abilities<Ability> abilities1 = new AbilitiesImpl<>(ability);
41  
42          ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new DrawCardSourceControllerEffect(1), new TapSourceCost());
43          Abilities<Ability> abilities2 = new AbilitiesImpl<>(ability);
44  
45          this.addAbilities(LevelerCardBuilder.construct(
46                  new LevelerCardBuilder.LevelAbility(1, 2, abilities1, 0, 1),
47                  new LevelerCardBuilder.LevelAbility(3, -1, abilities2, 0, 1)
48          ));
49          setMaxLevelCounters(3);
50      }
51  
52      public EnclaveCryptologist (final EnclaveCryptologist card) {
53          super(card);
54      }
55  
56      @Override
57      public EnclaveCryptologist copy() {
58          return new EnclaveCryptologist(this);
59      }
60  
61  }

21   * @author TheElk801

23  public final class EdgarsAwakening extends CardImpl {
24  
25      public EdgarsAwakening(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{B}{B}");
27  
28          // Return target creature card from your graveyard to the battlefield.
29          this.getSpellAbility().addEffect(new ReturnFromGraveyardToBattlefieldTargetEffect());
30          this.getSpellAbility().addTarget(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD));
31  
32          // When you discard Edgar's Awakening, you may pay {B}. When you do, return target creature card from your graveyard to your hand.
33          this.addAbility(new EdgarsAwakeningTriggeredAbility());
34      }
35  
36      private EdgarsAwakening(final EdgarsAwakening card) {
37          super(card);
38      }
39  
40      @Override
41      public EdgarsAwakening copy() {
42          return new EdgarsAwakening(this);
43      }
44  }
45  
46  class EdgarsAwakeningTriggeredAbility extends TriggeredAbilityImpl {
47  
48      private static final ReflexiveTriggeredAbility makeAbility() {
49          ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
50                  new ReturnFromGraveyardToHandTargetEffect(), false,
51                  "return target creature card from your graveyard to your hand"
52          );
53          ability.addTarget(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD));
54          return ability;
55      }
56  
57      EdgarsAwakeningTriggeredAbility() {
58          super(Zone.ALL, new DoWhenCostPaid(makeAbility(), new ManaCostsImpl<>("{B}"), "Pay {B}?"));
59      }
60  
61      private EdgarsAwakeningTriggeredAbility(final EdgarsAwakeningTriggeredAbility ability) {
62          super(ability);
63      }
64  
65      @Override
66      public EdgarsAwakeningTriggeredAbility copy() {
67          return new EdgarsAwakeningTriggeredAbility(this);
68      }
69  
70      @Override
71      public boolean checkEventType(GameEvent event, Game game) {
72          return event.getType() == GameEvent.EventType.DISCARDED_CARD;
73      }
74  
75      @Override
76      public boolean checkTrigger(GameEvent event, Game game) {
77          return this.getSourceId().equals(event.getTargetId());
78      }
79  
80      @Override
81      public String getRule() {
82          return "When you discard {this}, you may pay {B}. " +
83                  "When you do, return target creature card from your graveyard to your hand.";
84      }
85  }

4   import java.util.UUID;
5   import mage.abilities.Ability;
6   import mage.abilities.common.SimpleStaticAbility;
7   import mage.abilities.condition.common.SpellMasteryCondition;
8   import mage.abilities.decorator.ConditionalContinuousRuleModifyingEffect;
9   import mage.abilities.effects.ContinuousRuleModifyingEffect;
10  import mage.abilities.effects.common.CantBeCounteredSourceEffect;
11  import mage.abilities.effects.common.DamageTargetEffect;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.Zone;
16  import mage.target.common.TargetAnyTarget;
17  
18  /**
19   *
20   * @author fireshoes
21   */
22  public final class ExquisiteFirecraft extends CardImpl {
23  
24      public ExquisiteFirecraft(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{1}{R}{R}");
26  
27          // Exquisite Firecraft deals 4 damage to any target.        
28          this.getSpellAbility().addTarget(new TargetAnyTarget());
29          this.getSpellAbility().addEffect(new DamageTargetEffect(4));
30          
31          // <i>Spell mastery</i> &mdash; If there are two or more instant and/or sorcery cards in your graveyard, Exquisite Firecraft can't be countered.
32          ContinuousRuleModifyingEffect cantBeCountered = new CantBeCounteredSourceEffect();
33          ConditionalContinuousRuleModifyingEffect conditionalCantBeCountered = new ConditionalContinuousRuleModifyingEffect(cantBeCountered, SpellMasteryCondition.instance);
34          conditionalCantBeCountered.setText("<br/><i>Spell mastery</i> &mdash; If there are two or more instant and/or sorcery cards in your graveyard, this spell can't be countered");
35          Ability ability = new SimpleStaticAbility(Zone.STACK, conditionalCantBeCountered);
36          this.addAbility(ability);
37      }
38  
39      private ExquisiteFirecraft(final ExquisiteFirecraft card) {
40          super(card);
41      }
42  
43      @Override
44      public ExquisiteFirecraft copy() {
45          return new ExquisiteFirecraft(this);
46      }
47  }

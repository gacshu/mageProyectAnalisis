4   import mage.abilities.effects.common.counter.AddCountersTargetEffect;

13  
14  import java.util.UUID;
15  
16  /**
17   * @author TheElk801
18   */
19  public final class ExplosiveEntry extends CardImpl {
20  
21      public ExplosiveEntry(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{R}");
23  
24          // Destroy up to one target artifact. Put a +1/+1 counter on up to one target creature.
25          this.getSpellAbility().addEffect(new DestroyTargetEffect("destroy up to one target artifact")
26                  .setTargetPointer(new FirstTargetPointer()));
27          this.getSpellAbility().addTarget(new TargetArtifactPermanent(0, 1));
28          this.getSpellAbility().addEffect(new AddCountersTargetEffect(CounterType.P1P1.createInstance())
29                  .setText("Put a +1/+1 counter on up to one target creature")
30                  .setTargetPointer(new SecondTargetPointer()));
31          this.getSpellAbility().addTarget(new TargetCreaturePermanent(0, 1));
32      }
33  
34      private ExplosiveEntry(final ExplosiveEntry card) {
35          super(card);
36      }
37  
38      @Override
39      public ExplosiveEntry copy() {
40          return new ExplosiveEntry(this);
41      }
42  }

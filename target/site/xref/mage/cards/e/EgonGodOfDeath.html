24   * @author weirddan455
25   */
26  public final class EgonGodOfDeath extends ModalDoubleFacesCard {
27  
28      public EgonGodOfDeath(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo,
30                  new CardType[]{CardType.CREATURE}, new SubType[]{SubType.GOD}, "{2}{B}",
31                  "Throne of Death", new CardType[]{CardType.ARTIFACT}, new SubType[]{}, "{B}"
32          );
33  
34          // 1.
35          // Egon, God of Death
36          // Legendary Creature - God
37          this.getLeftHalfCard().addSuperType(SuperType.LEGENDARY);
38          this.getLeftHalfCard().setPT(new MageInt(6), new MageInt(6));
39  
40          // Deathtouch
41          this.getLeftHalfCard().addAbility(DeathtouchAbility.getInstance());
42  
43          // At the beginning of your upkeep, exile two cards from your graveyard. If you can't, sacrifice Egon and draw a card.
44          DoIfCostPaid effect = new DoIfCostPaid(null, new SacrificeSourceEffect(), new ExileFromGraveCost(
45                  new TargetCardInYourGraveyard(2, 2)), false
46          );
47          effect.addOtherwiseEffect(new DrawCardSourceControllerEffect(1));
48          effect.setText("exile two cards from your graveyard. If you can't, sacrifice {this} and draw a card.");
49          this.getLeftHalfCard().addAbility(new BeginningOfUpkeepTriggeredAbility(
50                  Zone.BATTLEFIELD, effect, TargetController.YOU, false, false
51          ));
52  
53          // 2.
54          // Throne of Death
55          // Legendary Artifact
56          this.getRightHalfCard().addSuperType(SuperType.LEGENDARY);
57  
58          // At the beginning of your upkeep, mill a card.
59          this.getRightHalfCard().addAbility(new BeginningOfUpkeepTriggeredAbility(
60                  Zone.BATTLEFIELD, new MillCardsControllerEffect(1), TargetController.YOU, false, false
61          ));
62  
63          // {2}{B}, {T}, Exile a creature card from your graveyard: Draw a card
64          Ability ability = new SimpleActivatedAbility(new DrawCardSourceControllerEffect(1), new ManaCostsImpl<>("{2}{B}"));
65          ability.addCost(new TapSourceCost());
66          ability.addCost(new ExileFromGraveCost(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_A)));
67          this.getRightHalfCard().addAbility(ability);
68      }
69  
70      private EgonGodOfDeath(final EgonGodOfDeath card) {
71          super(card);
72      }
73  
74      @Override
75      public EgonGodOfDeath copy() {
76          return new EgonGodOfDeath(this);
77      }
78  }

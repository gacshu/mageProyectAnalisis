33  public final class EstridTheMasked extends CardImpl {
34  
35      private static final FilterPermanent filter = new FilterPermanent();
36      private static final FilterPermanent filter2 = new FilterPermanent("another permanent");
37  
38      static {
39          filter.add(EnchantedPredicate.instance);
40          filter2.add(AnotherPredicate.instance);
41      }
42  
43      public EstridTheMasked(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{1}{G}{W}{U}");
45  
46          this.addSuperType(SuperType.LEGENDARY);
47          this.subtype.add(SubType.ESTRID);
48          this.setStartingLoyalty(3);
49  
50          // +2: Untap each enchanted permanent you control.
51          this.addAbility(new LoyaltyAbility(new UntapAllControllerEffect(
52                  filter, "untap each enchanted permanent you control"
53          ), 2));
54  
55          // -1: Create a white Aura enchantment token named Mask attached to another target permanent. The token has enchant permanent and totem armor.
56          Ability ability = new LoyaltyAbility(
57                  new EstridTheMaskedTokenEffect(), -1
58          );
59          ability.addTarget(new TargetPermanent(filter2));
60          this.addAbility(ability);
61  
62          // -7: Put the top seven cards of your library into your graveyard. Return all non-Aura enchantment cards from your graveyard to the battlefield, then do the same for Aura cards.
63          this.addAbility(new LoyaltyAbility(
64                  new EstridTheMaskedGraveyardEffect(), -7
65          ));
66  
67          // Estrid, the Masked can be your commander.
68          this.addAbility(CanBeYourCommanderAbility.getInstance());
69      }
70  
71      private EstridTheMasked(final EstridTheMasked card) {
72          super(card);
73      }
74  
75      @Override
76      public EstridTheMasked copy() {
77          return new EstridTheMasked(this);
78      }
79  }

81  class EstridTheMaskedTokenEffect extends OneShotEffect {
82  
83      public EstridTheMaskedTokenEffect() {
84          super(Outcome.Benefit);
85          this.staticText = "create a white Aura enchantment token named Mask "
86                  + "attached to another target permanent. "
87                  + "The token has enchant permanent and totem armor";
88      }
89  
90      public EstridTheMaskedTokenEffect(final EstridTheMaskedTokenEffect effect) {
91          super(effect);
92      }
93  
94      @Override
95      public EstridTheMaskedTokenEffect copy() {
96          return new EstridTheMaskedTokenEffect(this);
97      }
98  
99      @Override
100     public boolean apply(Game game, Ability source) {
101         CreateTokenEffect effect = new CreateTokenEffect(new MaskToken());
102         effect.apply(game, source);
103         for (UUID tokenId : effect.getLastAddedTokenIds()) {
104             Permanent token = game.getPermanent(tokenId);
105             if (token == null) {
106                 continue;
107             }
108             token.getAbilities().get(0).getTargets().get(0).add(source.getFirstTarget(), game);
109             token.getAbilities().get(0).getEffects().get(0).apply(game, token.getAbilities().get(0));
110         }
111         return true;
112     }
113 }
114 
115 class EstridTheMaskedGraveyardEffect extends OneShotEffect {
116 
117     private static final FilterEnchantmentCard filter
118             = new FilterEnchantmentCard();
119     private static final FilterEnchantmentCard filter2
120             = new FilterEnchantmentCard();
121 
122     static {
123         filter.add(Predicates.not(SubType.AURA.getPredicate()));
124         filter2.add(SubType.AURA.getPredicate());
125     }
126 
127     public EstridTheMaskedGraveyardEffect() {
128         super(Outcome.PutCardInPlay);
129         this.staticText = "put the top seven cards of your library "
130                 + "into your graveyard. Return all non-Aura enchantment cards "
131                 + "from your graveyard to the battlefield, "
132                 + "then do the same for Aura cards";
133     }
134 
135     public EstridTheMaskedGraveyardEffect(final EstridTheMaskedGraveyardEffect effect) {
136         super(effect);
137     }
138 
139     @Override
140     public EstridTheMaskedGraveyardEffect copy() {
141         return new EstridTheMaskedGraveyardEffect(this);
142     }
143 
144     @Override
145     public boolean apply(Game game, Ability source) {
146         new MillCardsControllerEffect(7).apply(game, source);
147         Player controller = game.getPlayer(source.getControllerId());
148         if (controller == null) {
149             return false;
150         }
151         controller.moveCards(controller.getGraveyard().getCards(
152                 filter, source.getControllerId(), source, game
153         ), Zone.BATTLEFIELD, source, game);
154         controller.moveCards(controller.getGraveyard().getCards(
155                 filter2, source.getControllerId(), source, game
156         ), Zone.BATTLEFIELD, source, game);
157         return true;
158     }
159 }

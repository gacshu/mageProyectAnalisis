35  public final class ProfessorOnyx extends CardImpl {
36  
37      private static final FilterPermanent filter = new FilterControlledCreaturePermanent(
38              "a creature with the greatest power among creatures that player controls"
39      );
40  
41      static {
42          filter.add(ProfessorOnyxPredicate.instance);
43      }
44  
45      public ProfessorOnyx(UUID ownerId, CardSetInfo setInfo) {
46          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{4}{B}{B}");
47  
48          this.addSuperType(SuperType.LEGENDARY);
49          this.subtype.add(SubType.LILIANA);
50          this.setStartingLoyalty(5);
51  
52          // Magecraft — Whenever you cast or copy an instant or sorcery spell, each opponent loses 2 life and you gain 2 life.
53          Ability ability = new MagecraftAbility(new LoseLifeOpponentsEffect(2), false);
54          ability.addEffect(new GainLifeEffect(2).concatBy("and"));
55          this.addAbility(ability);
56  
57          // +1: You lose 1 life. Look at the top three cards of your library. Put one of them into your hand and the rest into your graveyard.
58          ability = new LoyaltyAbility(new LoseLifeSourceControllerEffect(1), 1);
59          ability.addEffect(new LookLibraryAndPickControllerEffect(3, 1, PutCards.HAND, PutCards.GRAVEYARD));
60          this.addAbility(ability);
61  
62          // −3: Each opponent sacrifices a creature with the greatest power among creatures that player controls.
63          this.addAbility(new LoyaltyAbility(new SacrificeOpponentsEffect(filter), -3));
64  
65          // −8: Each opponent may discard a card. If they don't, they lose 3 life. Repeat this process six more times.
66          this.addAbility(new LoyaltyAbility(new ProfessorOnyxEffect(), -8));
67      }
68  
69      private ProfessorOnyx(final ProfessorOnyx card) {
70          super(card);
71      }
72  
73      @Override
74      public ProfessorOnyx copy() {
75          return new ProfessorOnyx(this);
76      }
77  }
78  
79  enum ProfessorOnyxPredicate implements Predicate<Permanent> {
80      instance;
81  
82      @Override
83      public boolean apply(Permanent input, Game game) {
84          return input.getPower().getValue()
85                  >= game
86                  .getBattlefield()
87                  .getActivePermanents(
88                          StaticFilters.FILTER_CONTROLLED_CREATURE,
89                          input.getControllerId(), game
90                  )
91                  .stream()
92                  .filter(Objects::nonNull)
93                  .map(MageObject::getPower)
94                  .mapToInt(MageInt::getValue)
95                  .max()
96                  .orElse(Integer.MIN_VALUE);
97      }
98  }

100 class ProfessorOnyxEffect extends OneShotEffect {
101 
102     ProfessorOnyxEffect() {
103         super(Outcome.Benefit);
104         staticText = "each opponent may discard a card. If they don't, " +
105                 "they lose 3 life. Repeat this process six more times";
106     }
107 
108     private ProfessorOnyxEffect(final ProfessorOnyxEffect effect) {
109         super(effect);
110     }
111 
112     @Override
113     public ProfessorOnyxEffect copy() {
114         return new ProfessorOnyxEffect(this);
115     }
116 
117     @Override
118     public boolean apply(Game game, Ability source) {
119         for (int i = 0; i < 6; i++) {
120             Map<UUID, Card> playerMap = new HashMap<>();
121             for (UUID playerId : game.getOpponents(source.getControllerId())) {
122                 Player player = game.getPlayer(playerId);
123                 if (player == null) {
124                     continue;
125                 }
126                 TargetDiscard target = new TargetDiscard(
127                         0, 1, StaticFilters.FILTER_CARD, playerId
128                 );
129                 player.choose(Outcome.Discard, target, source, game);
130                 playerMap.put(playerId, game.getCard(target.getFirstTarget()));
131             }
132             for (UUID playerId : game.getOpponents(source.getControllerId())) {
133                 Player player = game.getPlayer(playerId);
134                 if (player == null) {
135                     continue;
136                 }
137                 if (playerMap.get(playerId) == null
138                         || !player.discard(playerMap.get(playerId), false, source, game)) {
139                     player.loseLife(3, game, source, false);
140                 }
141             }
142         }
143         return true;
144     }

32  public final class PestilentCauldron extends ModalDoubleFacesCard {
33  
34      private static final FilterCard filter
35              = new FilterCard("creature, land, and/or planeswalker cards from your graveyard");
36  
37      static {
38          filter.add(Predicates.or(
39                  CardType.CREATURE.getPredicate(),
40                  CardType.LAND.getPredicate(),
41                  CardType.PLANESWALKER.getPredicate()
42          ));
43      }
44  
45      public PestilentCauldron(UUID ownerId, CardSetInfo setInfo) {
46          super(
47                  ownerId, setInfo,
48                  new CardType[]{CardType.ARTIFACT}, new SubType[]{}, "{2}{B}",
49                  "Restorative Burst",
50                  new CardType[]{CardType.SORCERY}, new SubType[]{}, "{3}{G}{G}"
51          );
52  
53          // 1.
54          // Pestilent Cauldron
55          // Artifact
56          // {T}, Discard a card: Create a 1/1 black and green Pest creature token with "When this creature dies, you gain 1 life."
57          Ability ability = new SimpleActivatedAbility(
58                  new CreateTokenEffect(new Pest11GainLifeToken()), new TapSourceCost()
59          );
60          ability.addCost(new DiscardCardCost());
61          this.getLeftHalfCard().addAbility(ability);
62  
63          // {1}, {T}: Each opponent mills cards equal to the amount of life you gained this turn.
64          ability = new SimpleActivatedAbility(new MillCardsEachPlayerEffect(
65                  ControllerGotLifeCount.instance, TargetController.OPPONENT
66          ).setText("each opponent mills cards equal to the amount of life you gained this turn"), new GenericManaCost(1));
67          ability.addCost(new TapSourceCost());
68          ability.addWatcher(new PlayerGainedLifeWatcher());
69          this.getLeftHalfCard().addAbility(ability.addHint(ControllerGotLifeCount.getHint()));
70  
71          // {4}, {T}: Exile four target cards from a single graveyard. Draw a card.
72          ability = new SimpleActivatedAbility(new ExileTargetEffect(), new GenericManaCost(4));
73          ability.addCost(new TapSourceCost());
74          ability.addTarget(new TargetCardInASingleGraveyard(
75                  4, 4, StaticFilters.FILTER_CARD_CARDS
76          ));
77          ability.addEffect(new DrawCardSourceControllerEffect(1));
78          this.getLeftHalfCard().addAbility(ability);
79  
80          // 2.
81          // Restorative Burst
82          // Sorcery
83          // Return up to two target creature, land, and/or planeswalker cards from your graveyard to your hand. Each player gains 4 life. Exile Restorative Burst.
84          this.getRightHalfCard().getSpellAbility().addEffect(new ReturnFromGraveyardToHandTargetEffect());
85          this.getRightHalfCard().getSpellAbility().addEffect(new RestorativeBurstEffect());
86          this.getRightHalfCard().getSpellAbility().addEffect(new ExileSpellEffect());
87          this.getRightHalfCard().getSpellAbility().addTarget(new TargetCardInYourGraveyard(0, 2, filter));
88      }
89  
90      private PestilentCauldron(final PestilentCauldron card) {
91          super(card);
92      }
93  
94      @Override
95      public PestilentCauldron copy() {
96          return new PestilentCauldron(this);
97      }
98  }
99  
100 class RestorativeBurstEffect extends OneShotEffect {
101 
102     RestorativeBurstEffect() {
103         super(Outcome.GainLife);
104         staticText = "Each player gains 4 life.";
105     }
106 
107     private RestorativeBurstEffect(final RestorativeBurstEffect effect) {
108         super(effect);
109     }
110 
111     @Override
112     public RestorativeBurstEffect copy() {
113         return new RestorativeBurstEffect(this);
114     }
115 
116     @Override
117     public boolean apply(Game game, Ability source) {
118         for (UUID playerId : game.getState().getPlayersInRange(source.getControllerId(), game)) {
119             Player player = game.getPlayer(playerId);
120             if (player != null) {
121                 player.gainLife(4, game, source);
122             }
123         }
124         return true;
125     }
126 }

32  public final class PlagueReaver extends CardImpl {
33  
34      public PlagueReaver(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{B}");
36  
37          this.subtype.add(SubType.BEAST);
38          this.power = new MageInt(6);
39          this.toughness = new MageInt(5);
40  
41          // At the beginning of your end step, sacrifice each other creature you control.
42          this.addAbility(new BeginningOfEndStepTriggeredAbility(
43                  new PlagueReaverSacrificeEffect(), TargetController.YOU, false
44          ));
45  
46          // Discard two cards, Sacrifice Plague Reaver: Choose target opponent. Return Plague Reaver to the battlefield under that player's control at the beginning of their next upkeep.
47          Ability ability = new SimpleActivatedAbility(
48                  new PlagueReaverTriggerEffect(), new DiscardTargetCost(new TargetCardInHand(2, StaticFilters.FILTER_CARD_CARDS))
49          );
50          ability.addCost(new SacrificeSourceCost());
51          this.addAbility(ability);
52      }
53  
54      private PlagueReaver(final PlagueReaver card) {
55          super(card);
56      }
57  
58      @Override
59      public PlagueReaver copy() {
60          return new PlagueReaver(this);
61      }
62  }

64  class PlagueReaverSacrificeEffect extends OneShotEffect {
65  
66      private static final FilterPermanent filter = new FilterControlledCreaturePermanent();
67  
68      static {
69          filter.add(AnotherPredicate.instance);
70      }
71  
72      PlagueReaverSacrificeEffect() {
73          super(Outcome.Benefit);
74          staticText = "sacrifice each other creature you control";
75      }
76  
77      private PlagueReaverSacrificeEffect(final PlagueReaverSacrificeEffect effect) {
78          super(effect);
79      }
80  
81      @Override
82      public PlagueReaverSacrificeEffect copy() {
83          return new PlagueReaverSacrificeEffect(this);
84      }
85  
86      @Override
87      public boolean apply(Game game, Ability source) {
88          for (Permanent permanent : game.getBattlefield().getActivePermanents(
89                  filter, source.getControllerId(), source, game
90          )) {
91              if (permanent == null) {
92                  continue;
93              }
94              permanent.sacrifice(source, game);
95          }
96          return true;
97      }
98  }

109         super(effect);
110     }

113     public PlagueReaverTriggerEffect copy() {
114         return new PlagueReaverTriggerEffect(this);
115     }
116 
117     @Override
118     public boolean apply(Game game, Ability source) {
119         game.addDelayedTriggeredAbility(new PlagueReaverDelayedTriggeredAbility(source.getFirstTarget(), source), source);
120         return true;

124 class PlagueReaverDelayedTriggeredAbility extends DelayedTriggeredAbility {
125 
126     private final UUID playerId;
127 
128     PlagueReaverDelayedTriggeredAbility(UUID playerId, Ability source) {
129         super(new PlagueReaverReturnEffect(playerId).setTargetPointer(
130                 new FixedTarget(source.getSourceId(), source.getSourceObjectZoneChangeCounter())
131         ), Duration.Custom, true, false);
132         this.playerId = playerId;
133     }
134 
135     private PlagueReaverDelayedTriggeredAbility(final PlagueReaverDelayedTriggeredAbility ability) {
136         super(ability);
137         this.playerId = ability.playerId;
138     }
139 
140     @Override
141     public boolean checkEventType(GameEvent event, Game game) {
142         return event.getType() == GameEvent.EventType.UPKEEP_STEP_PRE;
143     }
144 
145     @Override
146     public boolean checkTrigger(GameEvent event, Game game) {
147         return game.isActivePlayer(playerId);
148     }
149 
150     @Override
151     public PlagueReaverDelayedTriggeredAbility copy() {
152         return new PlagueReaverDelayedTriggeredAbility(this);
153     }
154 
155     @Override
156     public String getRule() {
157         return "Return {this} to the battlefield under that player's control " +
158                 "at the beginning of their next upkeep.";
159     }
160 }

162 class PlagueReaverReturnEffect extends OneShotEffect {
163 
164     private final UUID playerId;
165 
166     PlagueReaverReturnEffect(UUID playerId) {
167         super(Outcome.Benefit);
168         this.playerId = playerId;
169     }
170 
171     private PlagueReaverReturnEffect(final PlagueReaverReturnEffect effect) {
172         super(effect);
173         this.playerId = effect.playerId;
174     }
175 
176     @Override
177     public PlagueReaverReturnEffect copy() {
178         return new PlagueReaverReturnEffect(this);
179     }
180 
181     @Override
182     public boolean apply(Game game, Ability source) {
183         Player player = game.getPlayer(playerId);
184         Card card = game.getCard(targetPointer.getFirst(game, source));
185         return player != null
186                 && card != null
187                 && player.moveCards(card, Zone.BATTLEFIELD, source, game);
188     }

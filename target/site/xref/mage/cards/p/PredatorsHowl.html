4   import java.util.UUID;
5   import mage.abilities.condition.common.MorbidCondition;
6   import mage.abilities.decorator.ConditionalOneShotEffect;
7   import mage.abilities.effects.Effect;
8   import mage.abilities.effects.common.CreateTokenEffect;
9   import mage.abilities.hint.common.MorbidHint;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.game.permanent.token.WolfToken;
14  
15  /**
16   *
17   * @author fireshoes
18   */
19  public final class PredatorsHowl extends CardImpl {
20  
21      public PredatorsHowl(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{3}{G}");
23  
24          // Create a 2/2 green Wolf creature token.
25          // <i>Morbid</i> &mdash; Create three 2/2 green Wolf creature tokens instead if a creature died this turn.
26          Effect effect = new ConditionalOneShotEffect(
27                  new CreateTokenEffect(new WolfToken(), 3),
28                  new CreateTokenEffect(new WolfToken(), 1),
29                  MorbidCondition.instance,
30                  "Create a 2/2 green Wolf creature token. <br/><br/><i>Morbid</i> &mdash; Create three 2/2 green Wolf creature tokens instead if a creature died this turn.");
31          this.getSpellAbility().addEffect(effect);
32          this.getSpellAbility().addHint(MorbidHint.instance);
33      }
34  
35      private PredatorsHowl(final PredatorsHowl card) {
36          super(card);
37      }
38  
39      @Override
40      public PredatorsHowl copy() {
41          return new PredatorsHowl(this);
42      }
43  }

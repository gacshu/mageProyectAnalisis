27  public final class PurphorossIntervention extends CardImpl {
28  
29      private static final DynamicValue xValue = new MultipliedValue(ManacostVariableValue.REGULAR, 2);
30  
31      public PurphorossIntervention(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{R}");
33  
34          // Choose one —
35          // • Create an X/1 red Elemental creature token with trample and haste. Sacrifice it at the beginning of the next end step.
36          this.getSpellAbility().addEffect(new PurphorossInterventionEffect());
37  
38          // • Purphoros's Intervention deals twice X damage to target creature or planeswalker.
39          Mode mode = new Mode(new DamageTargetEffect(xValue)
40                  .setText("{this} deals twice X damage to target creature or planeswalker"));
41          mode.addTarget(new TargetCreatureOrPlaneswalker());
42          this.getSpellAbility().addMode(mode);
43      }
44  
45      private PurphorossIntervention(final PurphorossIntervention card) {
46          super(card);
47      }
48  
49      @Override
50      public PurphorossIntervention copy() {
51          return new PurphorossIntervention(this);
52      }
53  }
54  
55  class PurphorossInterventionEffect extends OneShotEffect {
56  
57      PurphorossInterventionEffect() {
58          super(Outcome.Benefit);
59          staticText = "Create an X/1 red Elemental creature token with trample and haste. " +
60                  "Sacrifice it at the beginning of the next end step.";
61      }
62  
63      private PurphorossInterventionEffect(final PurphorossInterventionEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public PurphorossInterventionEffect copy() {
69          return new PurphorossInterventionEffect(this);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          Token token = new PurphorossInterventionToken(source.getManaCostsToPay().getX());
75          token.putOntoBattlefield(1, game, source, source.getControllerId());
76          token.getLastAddedTokenIds()
77                  .stream()
78                  .forEach(uuid -> game.addDelayedTriggeredAbility(
79                          new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
80                                  new SacrificeTargetEffect()
81                                          .setText("sacrifice this creature")
82                                          .setTargetPointer(new FixedTarget(uuid, game))
83                          ), source
84                  ));
85          return true;
86      }
87  }

14  import mage.cards.CardImpl;
15  import mage.cards.CardSetInfo;
16  import mage.constants.CardType;
17  import mage.constants.TargetController;
18  import mage.filter.StaticFilters;
19  import mage.target.common.TargetCardInYourGraveyard;

21  /**
22   *
23   * @author LevelX2
24   */
25  public final class PalaceSiege extends CardImpl {
26  
27      private static final String ruleTrigger1 = "&bull  Khans &mdash; At the beginning of your upkeep, return target creature card from your graveyard to your hand.";
28      private static final String ruleTrigger2 = "&bull  Dragons &mdash; At the beginning of your upkeep, each opponent loses 2 life and you gain 2 life.";
29  
30      public PalaceSiege(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{B}{B}");
32  
33          // As Palace Siege enters the battlefield, choose Khans or Dragons.
34          this.addAbility(new EntersBattlefieldAbility(new ChooseModeEffect("Khans or Dragons?", "Khans", "Dragons"), null,
35                  "As {this} enters the battlefield, choose Khans or Dragons.", ""));
36  
37          // * Khans - At the beginning of your upkeep, return target creature card from your graveyard to your hand.
38          Ability ability1 = new ConditionalTriggeredAbility(
39                  new BeginningOfUpkeepTriggeredAbility(new ReturnFromGraveyardToHandTargetEffect(), TargetController.YOU, false),
40                  new ModeChoiceSourceCondition("Khans"),
41                  ruleTrigger1);
42          ability1.addTarget(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE));
43          this.addAbility(ability1);
44  
45          // * Dragons - At the beginning of your upkeep, each opponent loses 2 life and you gain 2 life.
46          Ability ability2 = new ConditionalTriggeredAbility(
47                  new BeginningOfUpkeepTriggeredAbility(new LoseLifeOpponentsEffect(2), TargetController.YOU, false),
48                  new ModeChoiceSourceCondition("Dragons"),
49                  ruleTrigger2);
50          Effect effect = new GainLifeEffect(2);
51          effect.setText("and you gain 2 life");
52          ability2.addEffect(effect);
53          this.addAbility(ability2);
54  
55      }
56  
57      private PalaceSiege(final PalaceSiege card) {
58          super(card);
59      }
60  
61      @Override
62      public PalaceSiege copy() {
63          return new PalaceSiege(this);
64      }
65  }

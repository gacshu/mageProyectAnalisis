4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.common.DealsCombatDamageToACreatureTriggeredAbility;
7   import mage.abilities.common.DealsCombatDamageToAPlayerTriggeredAbility;
8   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
9   import mage.abilities.condition.InvertCondition;
10  import mage.abilities.condition.common.CastFromHandSourcePermanentCondition;
11  import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
12  import mage.abilities.effects.common.DestroyTargetEffect;
13  import mage.abilities.effects.common.LoseGameSourceControllerEffect;
14  import mage.abilities.effects.common.LoseGameTargetPlayerEffect;
15  import mage.cards.CardImpl;
16  import mage.cards.CardSetInfo;
17  import mage.constants.CardType;
18  import mage.constants.SubType;
19  import mage.constants.SuperType;
20  import mage.watchers.common.CastFromHandWatcher;
21  
22  /**
23   *
24   * @author LevelX2
25   */
26  public final class PhageTheUntouchable extends CardImpl {
27  
28      public PhageTheUntouchable(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{B}{B}{B}{B}");
30          addSuperType(SuperType.LEGENDARY);
31          this.subtype.add(SubType.AVATAR);
32          this.subtype.add(SubType.MINION);
33  
34          this.power = new MageInt(4);
35          this.toughness = new MageInt(4);
36  
37          // When Phage the Untouchable enters the battlefield, if you didn't cast it from your hand, you lose the game.
38          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
39                  new EntersBattlefieldTriggeredAbility(new LoseGameSourceControllerEffect(), false),
40                  new InvertCondition(CastFromHandSourcePermanentCondition.instance),
41                  "When {this} enters the battlefield, if you didn't cast it from your hand, you lose the game"
42          ), new CastFromHandWatcher());
43  
44          // Whenever Phage deals combat damage to a creature, destroy that creature. It can't be regenerated.
45          this.addAbility(new DealsCombatDamageToACreatureTriggeredAbility(new DestroyTargetEffect(true), false, true));
46  
47          // Whenever Phage deals combat damage to a player, that player loses the game.
48          this.addAbility(new DealsCombatDamageToAPlayerTriggeredAbility(new LoseGameTargetPlayerEffect(), false, true));
49      }
50  
51      private PhageTheUntouchable(final PhageTheUntouchable card) {
52          super(card);
53      }
54  
55      @Override
56      public PhageTheUntouchable copy() {
57          return new PhageTheUntouchable(this);
58      }
59  }

26  /**
27   * @author TheElk801
28   */
29  public final class PoppetStitcher extends CardImpl {
30  
31      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(
32              StaticFilters.FILTER_CREATURE_TOKEN, ComparisonType.MORE_THAN, 2
33      );
34      private static final Hint hint = new ValueHint(
35              "Creature tokens you control", new PermanentsOnBattlefieldCount(StaticFilters.FILTER_CREATURE_TOKEN)
36      );
37  
38      public PoppetStitcher(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
40  
41          this.subtype.add(SubType.HUMAN);
42          this.subtype.add(SubType.WIZARD);
43          this.power = new MageInt(2);
44          this.toughness = new MageInt(3);
45  
46          this.secondSideCardClazz = mage.cards.p.PoppetFactory.class;
47  
48          // Whenever you cast an instant or sorcery spell, create a 2/2 black Zombie creature token with decayed.
49          this.addAbility(new SpellCastControllerTriggeredAbility(
50                  new CreateTokenEffect(new ZombieDecayedToken()),
51                  StaticFilters.FILTER_SPELL_AN_INSTANT_OR_SORCERY, false
52          ));
53  
54          // At the beginning of your upkeep, if you control three or more creature tokens, you may transform Poppet Sticher.
55          this.addAbility(new TransformAbility());
56          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
57                  new BeginningOfUpkeepTriggeredAbility(
58                          new TransformSourceEffect(),
59                          TargetController.YOU, true
60                  ), condition, "At the beginning of your upkeep, " +
61                  "if you control three or more creature tokens, you may transform {this}."
62          ).addHint(hint));
63      }
64  
65      private PoppetStitcher(final PoppetStitcher card) {
66          super(card);
67      }
68  
69      @Override
70      public PoppetStitcher copy() {
71          return new PoppetStitcher(this);
72      }
73  }

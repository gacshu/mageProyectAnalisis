29  public final class PossibilityStorm extends CardImpl {
30  
31      public PossibilityStorm(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{R}{R}");
33  
34          // Whenever a player casts a spell from their hand, that player exiles it, then exiles cards from
35          // the top of their library until they exile a card that shares a card type with it. That
36          // player may cast that card without paying its mana cost. Then they put all cards exiled with
37          // Possibility Storm on the bottom of their library in a random order.
38          this.addAbility(new PossibilityStormTriggeredAbility());
39      }
40  
41      private PossibilityStorm(final PossibilityStorm card) {
42          super(card);
43      }
44  
45      @Override
46      public PossibilityStorm copy() {
47          return new PossibilityStorm(this);
48      }
49  }

51  class PossibilityStormTriggeredAbility extends TriggeredAbilityImpl {
52  
53      public PossibilityStormTriggeredAbility() {
54          super(Zone.BATTLEFIELD, new PossibilityStormEffect(), false);
55          setTriggerPhrase("Whenever a player casts a spell from their hand, ");
56      }
57  
58      public PossibilityStormTriggeredAbility(final PossibilityStormTriggeredAbility ability) {
59          super(ability);
60      }
61  
62      @Override
63      public PossibilityStormTriggeredAbility copy() {
64          return new PossibilityStormTriggeredAbility(this);
65      }
66  
67      @Override
68      public boolean checkEventType(GameEvent event, Game game) {
69          return event.getType() == GameEvent.EventType.SPELL_CAST;
70      }
71  
72      @Override
73      public boolean checkTrigger(GameEvent event, Game game) {
74          if (event.getZone() != Zone.HAND) { return false; }
75  
76          Spell spell = game.getStack().getSpell(event.getTargetId());
77          if (spell == null) { return false; }
78  
79          for (Effect effect : this.getEffects()) {
80              effect.setTargetPointer(new FixedTarget(event.getTargetId()));
81          }
82          return true;
83      }
84  }

86  class PossibilityStormEffect extends OneShotEffect {
87  
88      public PossibilityStormEffect() {
89          super(Outcome.Neutral);
90          staticText = "that player exiles it, then exiles cards from the top of their library until they exile a card that shares a card type with it. That player may cast that card without paying its mana cost. Then they put all cards exiled with {this} on the bottom of their library in a random order";
91      }
92  
93      public PossibilityStormEffect(final PossibilityStormEffect effect) {
94          super(effect);
95      }
96  
97      @Override
98      public boolean apply(Game game, Ability source) {
99          Spell spell = game.getStack().getSpell(targetPointer.getFirst(game, source));
100         boolean noLongerOnStack = false; // spell was exiled already by another effect, for example NivMagus Elemental
101         if (spell == null) {
102             spell = ((Spell) game.getLastKnownInformation(targetPointer.getFirst(game, source), Zone.STACK));
103             noLongerOnStack = true;
104         }
105         if (spell == null) { return false; }
106 
107         Player spellController = game.getPlayer(spell.getControllerId());
108         if (spellController == null) { return false; }
109 
110         MageObject sourceObject = source.getSourceObject(game);
111         if (sourceObject == null) { return false; }
112 
113         if (!noLongerOnStack) {
114             spellController.moveCardsToExile(spell, source, game, true, source.getSourceId(), sourceObject.getIdName());
115         }
116 
117         if (!spellController.getLibrary().hasCards()) { return true; }
118         Library library = spellController.getLibrary();
119         Card card;
120         do {
121             card = library.getFromTop(game);
122             if (card != null) {
123                 spellController.moveCardsToExile(card, source, game, true, source.getSourceId(), sourceObject.getIdName());
124             }
125         } while (library.hasCards() && card != null && !sharesType(card, spell.getCardType(game), game));
126 
127         if (card != null && sharesType(card, spell.getCardType(game), game)
128                 && !card.isLand(game)
129                 && card.getSpellAbility().canChooseTarget(game, spellController.getId())) {
130             if (spellController.chooseUse(Outcome.PlayForFree, "Cast " + card.getLogName() + " without paying cost?", source, game)) {
131                 game.getState().setValue("PlayFromNotOwnHandZone" + card.getId(), Boolean.TRUE);
132                 spellController.cast(spellController.chooseAbilityForCast(card, game, true), game, true, new ApprovingObject(source, game));
133                 game.getState().setValue("PlayFromNotOwnHandZone" + card.getId(), null);
134             }
135         }
136 
137         ExileZone exile = game.getExile().getExileZone(source.getSourceId());
138         if (exile != null) {
139             spellController.putCardsOnBottomOfLibrary(exile, game, source, false);
140         }
141 
142         return true;
143     }
144 
145     private boolean sharesType(Card card, List<CardType> cardTypes, Game game) {
146         for (CardType type : card.getCardType(game)) {
147             if (cardTypes.contains(type)) {
148                 return true;
149             }
150         }
151         return false;
152     }
153 
154     @Override
155     public PossibilityStormEffect copy() {
156         return new PossibilityStormEffect(this);
157     }
158 
159 }

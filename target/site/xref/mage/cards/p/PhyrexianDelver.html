26  public final class PhyrexianDelver extends CardImpl {
27  
28      public PhyrexianDelver(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{B}{B}");
30          this.subtype.add(SubType.PHYREXIAN);
31          this.subtype.add(SubType.ZOMBIE);
32  
33          this.power = new MageInt(3);
34          this.toughness = new MageInt(2);
35  
36          // When Phyrexian Delver enters the battlefield, return target creature card from your graveyard to the battlefield. You lose life equal to that card's converted mana cost.
37          Ability ability = new EntersBattlefieldTriggeredAbility(new PhyrexianDelverEffect(), false);
38          Target target = new TargetCardInYourGraveyard(1, StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD);
39          ability.addTarget(target);
40          this.addAbility(ability);
41      }
42  
43      private PhyrexianDelver(final PhyrexianDelver card) {
44          super(card);
45      }
46  
47      @Override
48      public PhyrexianDelver copy() {
49          return new PhyrexianDelver(this);
50      }
51  }
52  
53  class PhyrexianDelverEffect extends OneShotEffect {
54  
55      public PhyrexianDelverEffect() {
56          super(Outcome.PutCreatureInPlay);
57          this.staticText = "return target creature card from your graveyard to the battlefield. You lose life equal to that card's mana value";
58      }
59  
60      public PhyrexianDelverEffect(final PhyrexianDelverEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public PhyrexianDelverEffect copy() {
66          return new PhyrexianDelverEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Card creatureCard = game.getCard(this.getTargetPointer().getFirst(game, source));
72          Player controller = game.getPlayer(source.getControllerId());
73          if (creatureCard != null && controller != null) {
74              boolean result = false;
75              if (game.getState().getZone(creatureCard.getId()) == Zone.GRAVEYARD) {
76                  result = controller.moveCards(creatureCard, Zone.BATTLEFIELD, source, game);
77              }
78              controller.loseLife(creatureCard.getManaValue(), game, source, false);
79              return result;
80          }
81          return false;
82      }
83  }

25  public final class PublicExecution extends CardImpl {
26  
27      public PublicExecution(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{5}{B}");
29  
30          // Destroy target creature an opponent controls. Each other creature that player controls gets -2/-0 until end of turn.
31          this.getSpellAbility().addEffect(new DestroyTargetEffect());
32          this.getSpellAbility().addTarget(new TargetCreaturePermanent(StaticFilters.FILTER_OPPONENTS_PERMANENT_CREATURE));
33          this.getSpellAbility().addEffect(new PublicExecutionEffect());
34      }
35  
36      private PublicExecution(final PublicExecution card) {
37          super(card);
38      }
39  
40      @Override
41      public PublicExecution copy() {
42          return new PublicExecution(this);
43      }
44  }
45  
46  class PublicExecutionEffect extends OneShotEffect {
47  
48      public PublicExecutionEffect() {
49          super(Outcome.Benefit);
50          staticText = "Each other creature that player controls gets -2/-0 until end of turn";
51      }
52  
53      public PublicExecutionEffect(final PublicExecutionEffect effect) {
54          super(effect);
55      }
56  
57      @Override
58      public boolean apply(Game game, Ability source) {
59          Permanent target = (Permanent) game.getLastKnownInformation(source.getFirstTarget(), Zone.BATTLEFIELD);
60          if (target != null) {
61              UUID opponent = target.getControllerId();
62              if (opponent != null) {
63                  FilterCreaturePermanent filter = new FilterCreaturePermanent("each other creature that player controls");
64                  filter.add(new ControllerIdPredicate(opponent));
65                  filter.add(Predicates.not(new PermanentIdPredicate(target.getId())));
66                  ContinuousEffect effect = new BoostAllEffect(-2, 0, Duration.EndOfTurn, filter, false);
67                  game.addEffect(effect, source);
68                  return true;
69              }
70          }
71          return false;
72      }
73  
74      @Override
75      public PublicExecutionEffect copy() {
76          return new PublicExecutionEffect(this);
77      }
78  }

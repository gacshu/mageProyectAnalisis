25  public final class PaintersServant extends CardImpl {
26  
27      public PaintersServant(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{2}");
29          this.subtype.add(SubType.SCARECROW);
30  
31          this.power = new MageInt(1);
32          this.toughness = new MageInt(3);
33  
34          // As Painter's Servant enters the battlefield, choose a color.
35          this.addAbility(new AsEntersBattlefieldAbility(new ChooseColorEffect(Outcome.Detriment)));
36  
37          // All cards that aren't on the battlefield, spells, and permanents are the chosen color in addition to their other colors.
38          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new PaintersServantEffect()));
39      }
40  
41      private PaintersServant(final PaintersServant card) {
42          super(card);
43      }
44

46      public PaintersServant copy() {
47          return new PaintersServant(this);
48      }
49  }

51  class PaintersServantEffect extends ContinuousEffectImpl {
52  
53      public PaintersServantEffect() {
54          super(Duration.WhileOnBattlefield, Layer.ColorChangingEffects_5, SubLayer.NA, Outcome.Benefit);
55          staticText = "All cards that aren't on the battlefield, spells, and permanents are the chosen color in addition to their other colors";
56      }
57  
58      @Override
59      public boolean apply(Game game, Ability source) {
60          Player controller = game.getPlayer(source.getControllerId());
61          if (controller != null) {
62              ObjectColor color = (ObjectColor) game.getState().getValue(source.getSourceId() + "_color");
63              if (color == null) {
64                  return false;
65              }
66  
67              // permanents
68              for (Permanent perm : game.getBattlefield().getActivePermanents(source.getControllerId(), game)) {
69                  perm.getColor(game).addColor(color);
70              }
71  
72              List<Card> affectedCards = new ArrayList<>();
73  
74              // stack
75              for (MageObject object : game.getStack()) {
76                  if (object instanceof Spell) {
77                      game.getState().getCreateMageObjectAttribute(object, game).getColor().addColor(color);
78  
79                      Card card = ((Spell) object).getCard();
80                      affectedCards.add(card);
81                  }
82              }
83  
84              // exile
85              affectedCards.addAll(game.getExile().getAllCardsByRange(game, controller.getId()));
86  
87              for (UUID playerId : game.getState().getPlayersInRange(controller.getId(), game)) {
88                  Player player = game.getPlayer(playerId);
89                  if (player == null) {
90                      continue;
91                  }
92  
93                  // command
94                  affectedCards.addAll(game.getCommanderCardsFromCommandZone(player, CommanderCardType.ANY));
95  
96                  // hand
97                  affectedCards.addAll(player.getHand().getCards(game));
98  
99                  // library
100                 affectedCards.addAll(player.getLibrary().getCards(game));
101 
102                 // graveyard
103                 affectedCards.addAll(player.getGraveyard().getCards(game));
104             }
105 
106             // apply colors to all cards
107             affectedCards.forEach(card -> {
108                 game.getState().getCreateMageObjectAttribute(card, game).getColor().addColor(color);
109 
110                 // mdf cards
111                 if (card instanceof ModalDoubleFacesCard) {
112                     ModalDoubleFacesCardHalf leftHalfCard = ((ModalDoubleFacesCard) card).getLeftHalfCard();
113                     ModalDoubleFacesCardHalf rightHalfCard = ((ModalDoubleFacesCard) card).getRightHalfCard();
114                     game.getState().getCreateMageObjectAttribute(leftHalfCard, game).getColor().addColor(color);
115                     game.getState().getCreateMageObjectAttribute(rightHalfCard, game).getColor().addColor(color);
116                 }
117 
118                 // split cards
119                 if (card instanceof SplitCard) {
120                     SplitCardHalf leftHalfCard = ((SplitCard) card).getLeftHalfCard();
121                     SplitCardHalf rightHalfCard = ((SplitCard) card).getRightHalfCard();
122                     game.getState().getCreateMageObjectAttribute(leftHalfCard, game).getColor().addColor(color);
123                     game.getState().getCreateMageObjectAttribute(rightHalfCard, game).getColor().addColor(color);
124                 }
125 
126                 // double faces cards
127                 if (card.getSecondCardFace() != null) {
128                     game.getState().getCreateMageObjectAttribute(card.getSecondCardFace(), game).getColor().addColor(color);
129                 }
130             });
131             return true;
132         }
133         return false;
134     }
135 
136     @Override
137     public PaintersServantEffect copy() {
138         return new PaintersServantEffect(this);
139     }
140 
141     private PaintersServantEffect(PaintersServantEffect effect) {
142         super(effect);
143     }

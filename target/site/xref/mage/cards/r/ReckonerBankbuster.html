30  public final class ReckonerBankbuster extends CardImpl {
31  
32      private static final Condition condition = new SourceHasCounterCondition(CounterType.CHARGE, 0, 0);
33  
34      public ReckonerBankbuster(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
36  
37          this.subtype.add(SubType.VEHICLE);
38          this.power = new MageInt(4);
39          this.toughness = new MageInt(4);
40  
41          // Reckoner Bankbuster enters the battlefield with three charge counters on it.
42          this.addAbility(new EntersBattlefieldAbility(new AddCountersSourceEffect(
43                  CounterType.CHARGE.createInstance(3)
44          ), "with three charge counters on it"));
45  
46          // {2}, {T}, Remove a charge counter from Reckoner Bankbuster: Draw a card. Then if there are no charge counters on Reckoner Bankbuster, create a Treasure token and a 1/1 colorless Pilot creature token with "This creature crews Vehicles as though its power were 2 greater."
47          Ability ability = new SimpleActivatedAbility(
48                  new DrawCardSourceControllerEffect(1), new GenericManaCost(2)
49          );
50          ability.addEffect(new ConditionalOneShotEffect(
51                  new CreateTokenEffect(new TreasureToken()), condition,
52                  "Then if there are no charge counters on {this}, create a Treasure token"
53          ));
54          ability.addEffect(new ConditionalOneShotEffect(
55                  new CreateTokenEffect(new PilotToken()), condition, "and a 1/1 colorless Pilot creature token " +
56                  "with \"This creature crews Vehicles as though its power were 2 greater.\""
57          ));
58          ability.addCost(new TapSourceCost());
59          ability.addCost(new RemoveCountersSourceCost(CounterType.CHARGE.createInstance()));
60          this.addAbility(ability);
61  
62          // Crew 3
63          this.addAbility(new CrewAbility(3));
64      }
65  
66      private ReckonerBankbuster(final ReckonerBankbuster card) {
67          super(card);
68      }
69  
70      @Override
71      public ReckonerBankbuster copy() {
72          return new ReckonerBankbuster(this);
73      }
74  }

25  public final class RuinGrinder extends CardImpl {
26  
27      public RuinGrinder(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{5}{R}");
29  
30          this.subtype.add(SubType.CONSTRUCT);
31          this.power = new MageInt(7);
32          this.toughness = new MageInt(4);
33  
34          // Menace
35          this.addAbility(new MenaceAbility(false));
36  
37          // When Ruin Grinder dies, each player may discard their hand and draw seven cards.
38          this.addAbility(new DiesSourceTriggeredAbility(new RuinGrinderEffect()));
39  
40          // Mountaincycling {2}
41          this.addAbility(new MountaincyclingAbility(new ManaCostsImpl<>("{2}")));
42      }
43  
44      private RuinGrinder(final RuinGrinder card) {
45          super(card);
46      }
47  
48      @Override
49      public RuinGrinder copy() {
50          return new RuinGrinder(this);
51      }
52  }
53  
54  class RuinGrinderEffect extends OneShotEffect {
55  
56      RuinGrinderEffect() {
57          super(Outcome.Benefit);
58          staticText = "each player may discard their hand and draw seven cards";
59      }
60  
61      private RuinGrinderEffect(final RuinGrinderEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public RuinGrinderEffect copy() {
67          return new RuinGrinderEffect(this);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          List<Player> wheelers = new ArrayList<>();
73          for (UUID playerId : game.getState().getPlayersInRange(source.getControllerId(), game)) {
74              Player player = game.getPlayer(playerId);
75              if (player != null && player.chooseUse(
76                      Outcome.DrawCard, "Discard your hand and draw seven?", source, game
77              )) {
78                  game.informPlayers(player.getName() + " chooses to discard their hand and draw seven");
79                  wheelers.add(player);
80              }
81          }
82          for (Player player : wheelers) {
83              player.discard(player.getHand(), false, source, game);
84              player.drawCards(7, source, game);
85          }
86          return true;
87      }
88  }

2   package mage.cards.r;
3   
4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
8   import mage.abilities.effects.common.ExileTargetForSourceEffect;
9   import mage.abilities.effects.common.ReturnToBattlefieldUnderYourControlTargetEffect;
10  import mage.abilities.keyword.FlashAbility;
11  import mage.abilities.keyword.FlyingAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.filter.common.FilterControlledCreaturePermanent;
17  import mage.filter.predicate.Predicates;
18  import mage.target.TargetPermanent;
19  
20  /**
21   *
22   * @author noxx
23   *
24   */
25  public final class RestorationAngel extends CardImpl {
26  
27      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent("non-Angel creature you control");
28  
29      static {
30          filter.add(Predicates.not(SubType.ANGEL.getPredicate()));
31      }
32  
33      public RestorationAngel(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{W}");
35          this.subtype.add(SubType.ANGEL);
36  
37          this.power = new MageInt(3);
38          this.toughness = new MageInt(4);
39  
40          this.addAbility(FlashAbility.getInstance());
41          this.addAbility(FlyingAbility.getInstance());
42  
43          // When Restoration Angel enters the battlefield, you may exile target non-Angel creature you control, then return that card to the battlefield under your control
44          Ability ability = new EntersBattlefieldTriggeredAbility(new ExileTargetForSourceEffect(), true);
45          ability.addEffect(new ReturnToBattlefieldUnderYourControlTargetEffect().concatBy(", then"));
46          ability.addTarget(new TargetPermanent(filter));
47          this.addAbility(ability);
48      }
49  
50      private RestorationAngel(final RestorationAngel card) {
51          super(card);
52      }
53  
54      @Override
55      public RestorationAngel copy() {
56          return new RestorationAngel(this);
57      }
58  }

18  import mage.target.common.TargetCardInYourGraveyard;
19  
20  import java.util.Objects;
21  import java.util.UUID;
22  
23  /**
24   * @author TheElk801
25   */
26  public final class RiveteersAscendancy extends CardImpl {
27  
28      private static final FilterCard filter
29              = new FilterCreatureCard("creature card with lesser mana value from your graveyard");
30  
31      static {
32          filter.add(RiveteersAscendancyPredicate.instance);
33      }
34  
35      public RiveteersAscendancy(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{B}{R}{G}");
37  
38          // Whenever you sacrifice a creature, you may return target creature card with lesser mana value from your graveyard to the battlefield tapped. Do this only once each turn.
39          Ability ability = new SacrificePermanentTriggeredAbility(
40                  new ReturnFromGraveyardToBattlefieldTargetEffect(true),
41                  StaticFilters.FILTER_PERMANENT_A_CREATURE
42          ).setDoOnlyOnce(true);
43          ability.addTarget(new TargetCardInYourGraveyard(filter));
44          this.addAbility(ability);
45      }
46  
47      private RiveteersAscendancy(final RiveteersAscendancy card) {
48          super(card);
49      }
50  
51      @Override
52      public RiveteersAscendancy copy() {
53          return new RiveteersAscendancy(this);
54      }
55  }
56  
57  enum RiveteersAscendancyPredicate implements ObjectSourcePlayerPredicate<Card> {
58      instance;
59  
60      @Override
61      public boolean apply(ObjectSourcePlayer<Card> input, Game game) {
62          return input
63                  .getObject()
64                  .getManaValue()
65                  < input
66                  .getSource()
67                  .getEffects()
68                  .stream()
69                  .map(effect -> effect.getValue("sacrificedPermanent"))
70                  .filter(Objects::nonNull)
71                  .map(Permanent.class::cast)
72                  .mapToInt(MageObject::getManaValue)
73                  .max()
74                  .orElse(0);
75      }
76  }

4   import java.util.UUID;
5   import mage.abilities.effects.Effect;
6   import mage.abilities.effects.common.UntapAllControllerEffect;
7   import mage.abilities.effects.common.continuous.GainAbilityAllEffect;
8   import mage.abilities.effects.common.continuous.GainAbilityControlledEffect;
9   import mage.abilities.keyword.DeathtouchAbility;
10  import mage.abilities.keyword.IndestructibleAbility;
11  import mage.abilities.keyword.LifelinkAbility;
12  import mage.cards.CardSetInfo;
13  import mage.cards.SplitCard;
14  import mage.constants.CardType;
15  import mage.constants.Duration;
16  import mage.constants.SpellAbilityType;
17  import mage.filter.StaticFilters;
18  import mage.filter.common.FilterControlledCreaturePermanent;
19  
20  /**
21   *
22   * @author LevelX2
23   */
24  public final class ReadyWilling extends SplitCard {
25  
26      public ReadyWilling(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{G}{W}", "{1}{W}{B}", SpellAbilityType.SPLIT_FUSED);
28  
29          // Ready
30          // Creatures you control are indestructible this turn. Untap each creature you control.
31          Effect effect = new GainAbilityAllEffect(IndestructibleAbility.getInstance(), Duration.EndOfTurn, new FilterControlledCreaturePermanent("Creatures you controll"), false);
32          effect.setText("Creatures you control are indestructible this turn");
33          getLeftHalfCard().getSpellAbility().addEffect(effect);
34          getLeftHalfCard().getSpellAbility().addEffect(new UntapAllControllerEffect(new FilterControlledCreaturePermanent(), "Untap each creature you control"));
35  
36          // Willing
37          // Creatures you control gain deathtouch and lifelink until end of turn.
38          getRightHalfCard().getSpellAbility().addEffect(new GainAbilityControlledEffect(DeathtouchAbility.getInstance(), Duration.EndOfTurn, StaticFilters.FILTER_PERMANENT_CREATURES));
39          effect = new GainAbilityControlledEffect(LifelinkAbility.getInstance(), Duration.EndOfTurn, StaticFilters.FILTER_PERMANENT_CREATURES);
40          effect.setText("Creatures you control gain lifelink until end of turn");
41          getRightHalfCard().getSpellAbility().addEffect(effect);
42  
43      }
44  
45      private ReadyWilling(final ReadyWilling card) {
46          super(card);
47      }
48  
49      @Override
50      public ReadyWilling copy() {
51          return new ReadyWilling(this);
52      }
53  
54  }

17  public final class RhythmicWaterVortex extends CardImpl {
18  
19      private static final FilterCard filter = new FilterCard("Mu Yanling");
20  
21      static {
22          filter.add(new NamePredicate("Mu Yanling"));
23      }
24  
25      public RhythmicWaterVortex(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{U}{U}");
27  
28          // Return up to two target creatures to their owner's hand.
29          this.getSpellAbility().addEffect(new ReturnToHandTargetEffect());
30          this.getSpellAbility().addTarget(new TargetCreaturePermanent(0, 2));
31  
32          // Search your library and/or graveyard for a card named Mu Yanling, reveal it, and put it into your hand. If you searched your library this way, shuffle it.
33          this.getSpellAbility().addEffect(new SearchLibraryGraveyardPutInHandEffect(filter).concatBy("<br>"));
34      }
35  
36      private RhythmicWaterVortex(final RhythmicWaterVortex card) {
37          super(card);
38      }
39  
40      @Override
41      public RhythmicWaterVortex copy() {
42          return new RhythmicWaterVortex(this);
43      }
44  }

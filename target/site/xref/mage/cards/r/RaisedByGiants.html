8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.constants.Duration;

14  
15  import java.util.UUID;
16  
17  /**
18   * @author TheElk801
19   */
20  public final class RaisedByGiants extends CardImpl {
21  
22      public RaisedByGiants(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{5}{G}");
24  
25          this.addSuperType(SuperType.LEGENDARY);
26          this.subtype.add(SubType.BACKGROUND);
27  
28          // Commander creatures you own have base power and toughness 10/10 and are Giants in addition to their other types.
29          Ability ability = new SimpleStaticAbility(new SetBasePowerToughnessAllEffect(
30                  10, 10, Duration.WhileOnBattlefield,
31                  StaticFilters.FILTER_CREATURES_OWNED_COMMANDER, true
32          ));
33          ability.addEffect(new AddCardSubtypeAllEffect(
34                  StaticFilters.FILTER_CREATURES_OWNED_COMMANDER,
35                  SubType.GIANT, null
36          ).setText("and are Giants in addition to their other types"));
37          this.addAbility(ability);
38      }
39  
40      private RaisedByGiants(final RaisedByGiants card) {
41          super(card);
42      }
43  
44      @Override
45      public RaisedByGiants copy() {
46          return new RaisedByGiants(this);
47      }
48  }

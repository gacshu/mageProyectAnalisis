43      private static final FilterCard filter = new FilterInstantOrSorceryCard("instant and sorcery spells");
44  
45      public RowanScholarOfSparks(UUID ownerId, CardSetInfo setInfo) {
46          super(
47                  ownerId, setInfo,
48                  new CardType[]{CardType.PLANESWALKER}, new SubType[]{SubType.ROWAN}, "{2}{R}",
49                  "Will, Scholar of Frost",
50                  new CardType[]{CardType.PLANESWALKER}, new SubType[]{SubType.WILL}, "{4}{U}"
51          );
52  
53          // 1.
54          // Rowan, Scholar of Sparks
55          // Legendary Planeswalker - Rowan
56          this.getLeftHalfCard().addSuperType(SuperType.LEGENDARY);
57          this.getLeftHalfCard().setStartingLoyalty(2);
58  
59          // Instant and sorcery spells you cast cost {1} less to cast.
60          this.getLeftHalfCard().addAbility(new SimpleStaticAbility(new SpellsCostReductionControllerEffect(filter, 1)));
61  
62          // +1: Rowan, Scholar of Sparks deals 1 damage to each opponent. If you've drawn three or more cards this turn, she deals 3 damage to each opponent instead.
63          Ability ability = new LoyaltyAbility(new ConditionalOneShotEffect(
64                  new DamagePlayersEffect(3, TargetController.OPPONENT),
65                  new DamagePlayersEffect(1, TargetController.OPPONENT),
66                  RowanScholarOfSparksCondition.instance, "{this} deals 1 damage to each opponent. " +
67                  "If you've drawn three or more cards this turn, she deals 3 damage to each opponent instead"
68          ), 1);
69          ability.addWatcher(new CardsAmountDrawnThisTurnWatcher());
70          this.getLeftHalfCard().addAbility(ability.addHint(RowanScholarOfSparksHint.instance));
71  
72          // −4: You get an emblem with "Whenever you cast an instant or sorcery spell, you may pay {2}. If you do, copy that spell. You may choose new targets for the copy."
73          this.getLeftHalfCard().addAbility(new LoyaltyAbility(
74                  new GetEmblemEffect(new RowanScholarOfSparksEmblem()), -4
75          ));
76  
77          // 2.
78          // Will, Scholar of Frost
79          // Legendary Planeswalker - Will
80          this.getRightHalfCard().addSuperType(SuperType.LEGENDARY);
81          this.getRightHalfCard().setStartingLoyalty(4);
82  
83          // Instant and sorcery spells you cast cost {1} less to cast.
84          this.getRightHalfCard().addAbility(new SimpleStaticAbility(
85                  new SpellsCostReductionControllerEffect(filter, 1)
86          ));
87  
88          // +1: Up to one target creature has base power and toughness 0/2 until your next turn.
89          ability = new LoyaltyAbility(new SetBasePowerToughnessTargetEffect(
90                  0, 2, Duration.UntilYourNextTurn
91          ), 1);
92          ability.addTarget(new TargetCreaturePermanent(0, 1));
93          this.getRightHalfCard().addAbility(ability);
94  
95          // −3: Draw two cards.
96          this.getRightHalfCard().addAbility(new LoyaltyAbility(new DrawCardSourceControllerEffect(2), -3));
97  
98          // −7: Exile up to five target permanents. For each permanent exiled this way, its controller creates a 4/4 blue and red Elemental Creature token.
99          ability = new LoyaltyAbility(new WillScholarOfFrostEffect(), -7);
100         ability.addTarget(new TargetPermanent(0, 5, StaticFilters.FILTER_PERMANENTS));
101         this.getRightHalfCard().addAbility(ability);
102     }

114 enum RowanScholarOfSparksCondition implements Condition {
115     instance;
116 
117     @Override
118     public boolean apply(Game game, Ability source) {
119         CardsAmountDrawnThisTurnWatcher watcher = game.getState().getWatcher(CardsAmountDrawnThisTurnWatcher.class);
120         return watcher != null && watcher.getAmountCardsDrawn(source.getControllerId()) >= 3;
121     }
122 }

124 enum RowanScholarOfSparksHint implements Hint {
125     instance;

128     public String getText(Game game, Ability ability) {

140 class WillScholarOfFrostEffect extends OneShotEffect {
141 
142     WillScholarOfFrostEffect() {
143         super(Outcome.Benefit);
144         staticText = "exile up to five target permanents. For each permanent exiled this way, " +
145                 "its controller creates a 4/4 blue and red Elemental creature token";
146     }
147 
148     private WillScholarOfFrostEffect(final WillScholarOfFrostEffect effect) {
149         super(effect);
150     }
151 
152     @Override
153     public WillScholarOfFrostEffect copy() {
154         return new WillScholarOfFrostEffect(this);
155     }
156 
157     @Override
158     public boolean apply(Game game, Ability source) {
159         Player player = game.getPlayer(source.getControllerId());
160         if (player == null) {
161             return false;
162         }
163         Cards cards = new CardsImpl(targetPointer.getTargets(game, source));
164         Map<UUID, Integer> playerMap = cards
165                 .getCards(game)
166                 .stream()
167                 .filter(Objects::nonNull)
168                 .map(MageItem::getId)
169                 .map(game::getControllerId)
170                 .filter(Objects::nonNull)
171                 .collect(Collectors.toMap(
172                         Function.identity(),
173                         x -> 1, Integer::sum
174                 ));
175         player.moveCards(cards, Zone.EXILED, source, game);
176         for (Map.Entry<UUID, Integer> entry : playerMap.entrySet()) {
177             new Elemental44Token().putOntoBattlefield(entry.getValue(), game, source, entry.getKey());
178         }
179         return true;
180     }

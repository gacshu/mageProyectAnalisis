2   package mage.cards.r;

4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.costs.common.ExileOpponentsCardFromExileToGraveyardCost;
7   import mage.abilities.effects.common.CastSourceTriggeredAbility;
8   import mage.abilities.effects.common.DoIfCostPaid;
9   import mage.abilities.effects.common.GainLifeEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.SubType;
14  
15  /**
16   *
17   * @author LevelX2
18   */
19  public final class RuinProcessor extends CardImpl {
20  
21      public RuinProcessor(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{7}");
23          this.subtype.add(SubType.ELDRAZI);
24          this.subtype.add(SubType.PROCESSOR);
25          this.power = new MageInt(7);
26          this.toughness = new MageInt(8);
27  
28          // When you cast Ruin Processor, you may put a card an opponent owns from exile into that player's graveyard. If you do, you gain 5 life.
29          this.addAbility(new CastSourceTriggeredAbility(
30                  new DoIfCostPaid(new GainLifeEffect(5), new ExileOpponentsCardFromExileToGraveyardCost(true)), false));
31  
32      }
33  
34      private RuinProcessor(final RuinProcessor card) {
35          super(card);
36      }
37  
38      @Override
39      public RuinProcessor copy() {
40          return new RuinProcessor(this);
41      }
42  }

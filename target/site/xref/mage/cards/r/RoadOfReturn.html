28  public final class RoadOfReturn extends CardImpl {
29  
30      private static final FilterCard filter = new FilterPermanentCard("permanent card from your graveyard");
31  
32      public RoadOfReturn(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{G}{G}");
34  
35          // Choose one —
36          // • Return target permanent card from your graveyard to your hand.
37          this.getSpellAbility().addEffect(new ReturnFromGraveyardToHandTargetEffect());
38          this.getSpellAbility().addTarget(new TargetCardInYourGraveyard(filter));
39  
40          // • Put your commander into your hand from the command zone.
41          this.getSpellAbility().addMode(new Mode(new RoadOfReturnEffect()));
42  
43          // Entwine {2}
44          this.addAbility(new EntwineAbility("{2}"));
45      }
46  
47      private RoadOfReturn(final RoadOfReturn card) {
48          super(card);
49      }
50  
51      @Override
52      public RoadOfReturn copy() {
53          return new RoadOfReturn(this);
54      }
55  }

57  class RoadOfReturnEffect extends OneShotEffect {
58  
59      RoadOfReturnEffect() {
60          super(Outcome.Benefit);
61          staticText = "Put your commander into your hand from the command zone.";
62      }
63  
64      private RoadOfReturnEffect(final RoadOfReturnEffect effect) {
65          super(effect);
66      }
67  
68      @Override
69      public RoadOfReturnEffect copy() {
70          return new RoadOfReturnEffect(this);
71      }
72  
73      @Override
74      public boolean apply(Game game, Ability source) {
75          Player controller = game.getPlayer(source.getControllerId());
76          if (controller == null) {
77              return false;
78          }
79          List<Card> commandersInCommandZone = new ArrayList<>(game.getCommanderCardsFromCommandZone(controller, CommanderCardType.COMMANDER_OR_OATHBREAKER));
80          if (commandersInCommandZone.size() == 1) {
81              controller.moveCards(commandersInCommandZone.get(0), Zone.HAND, source, game);
82          } else if (commandersInCommandZone.size() == 2) {
83              Card firstCommander = commandersInCommandZone.get(0);
84              Card secondCommander = commandersInCommandZone.get(1);
85              if (controller.chooseUse(Outcome.ReturnToHand, "Return which commander to hand?", null, firstCommander.getName(), secondCommander.getName(), source, game)) {
86                  controller.moveCards(firstCommander, Zone.HAND, source, game);
87              } else {
88                  controller.moveCards(secondCommander, Zone.HAND, source, game);
89              }
90          }
91          return true;
92      }
93  }

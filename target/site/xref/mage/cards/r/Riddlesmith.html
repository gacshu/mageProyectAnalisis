1   package mage.cards.r;

3   import mage.MageInt;
4   import mage.abilities.common.SpellCastControllerTriggeredAbility;
5   import mage.abilities.effects.common.DrawDiscardControllerEffect;
6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.constants.SubType;
10  import mage.filter.FilterSpell;
11  import mage.filter.common.FilterArtifactSpell;
12  
13  import java.util.UUID;
14  
15  /**
16   * @author Loki, North
17   */
18  public final class Riddlesmith extends CardImpl {
19  
20      private static final FilterSpell filter = new FilterArtifactSpell("an artifact spell");
21  
22      public Riddlesmith(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{U}");
24          this.subtype.add(SubType.HUMAN);
25          this.subtype.add(SubType.ARTIFICER);
26  
27          this.power = new MageInt(2);
28          this.toughness = new MageInt(1);
29  
30          // Whenever you cast an artifact spell, you may draw a card. If you do, discard a card.
31          this.addAbility(new SpellCastControllerTriggeredAbility(
32                  new DrawDiscardControllerEffect(true), filter, false
33          ));
34      }
35  
36      public Riddlesmith(final Riddlesmith card) {
37          super(card);
38      }
39  
40      @Override
41      public Riddlesmith copy() {
42          return new Riddlesmith(this);
43      }
44  }

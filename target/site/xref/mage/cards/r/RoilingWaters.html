18  public final class RoilingWaters extends CardImpl {
19  
20      public RoilingWaters(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{5}{U}{U}");
22  
23          // Return up to two target creatures your opponents control to their owners' hands.
24          this.getSpellAbility().addEffect(new ReturnToHandTargetEffect());
25          this.getSpellAbility().addTarget(new TargetOpponentsCreaturePermanent(0, 2));
26          // Target player draws two cards.
27          Effect effect = new DrawCardTargetEffect(2);
28          effect.setTargetPointer(new SecondTargetPointer());
29          getSpellAbility().addEffect(effect);
30          getSpellAbility().addTarget(new TargetPlayer());
31      }
32  
33      private RoilingWaters(final RoilingWaters card) {
34          super(card);
35      }
36  
37      @Override
38      public RoilingWaters copy() {
39          return new RoilingWaters(this);
40      }
41  }

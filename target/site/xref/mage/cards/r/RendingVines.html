24  public final class RendingVines extends CardImpl {
25  
26      public RendingVines(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId,setInfo,new CardType[]{CardType.INSTANT},"{1}{G}{G}");
28          this.subtype.add(SubType.ARCANE);
29  
30          // Destroy target artifact or enchantment if its converted mana cost is less than or equal to the number of cards in your hand.
31          this.getSpellAbility().addEffect(new RendingVinesEffect());
32          this.getSpellAbility().addTarget(new TargetPermanent(StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_ENCHANTMENT));
33  
34          // Draw a card.
35          this.getSpellAbility().addEffect(new DrawCardSourceControllerEffect(1).concatBy("<br>"));
36      }
37  
38      private RendingVines(final RendingVines card) {
39          super(card);
40      }
41  
42      @Override
43      public RendingVines copy() {
44          return new RendingVines(this);
45      }
46  }
47  
48  class RendingVinesEffect extends OneShotEffect {
49  
50      public RendingVinesEffect() {
51          super(Outcome.DestroyPermanent);
52          this.staticText = "Destroy target artifact or enchantment if its mana value is less than or equal to the number of cards in your hand";
53      }
54  
55      public RendingVinesEffect(final RendingVinesEffect effect) {
56          super(effect);
57      }
58  
59      @Override
60      public RendingVinesEffect copy() {
61          return new RendingVinesEffect(this);
62      }
63  
64      @Override
65      public boolean apply(Game game, Ability source) {
66          Player controller = game.getPlayer(source.getControllerId());
67          Permanent permanent = game.getPermanent(getTargetPointer().getFirst(game, source));
68          if (controller != null) {
69              if (permanent.getManaValue() <= controller.getHand().size()) {
70                  return permanent.destroy(source, game, false);
71              }
72          }
73          return false;
74      }
75  }

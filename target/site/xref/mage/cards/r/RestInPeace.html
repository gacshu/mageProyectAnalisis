28  //    http://www.wizards.com/magic/magazine/article.aspx?x=mtg/faq/rtr
29  //
30  //    While Rest in Peace is on the battlefield, abilities that trigger whenever a
31  //    creature dies won't trigger because cards and tokens never reach a player's graveyard.
32  //
33  //    If Rest in Peace is destroyed by a spell, Rest in Peace will be exiled and
34  //    then the spell will be put into its owner's graveyard.
35  //
36  //    If a card is discarded while Rest in Peace is on the battlefield, abilities
37  //    that function when a card is discarded (such as madness) still work, even
38  //    though that card never reaches a graveyard. In addition, spells or abilities
39  //    that check the characteristics of a discarded card (such as Chandra Ablaze's
40  //    first ability) can find that card in exile.
41  //
42  
43  public final class RestInPeace extends CardImpl {
44  
45      public RestInPeace(UUID ownerId, CardSetInfo setInfo) {
46          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{W}");
47  
48          // When Rest in Peace enters the battlefield, exile all cards from all graveyards.
49          this.addAbility(new EntersBattlefieldTriggeredAbility(new ExileGraveyardAllPlayersEffect()));
50  
51          // If a card or token would be put into a graveyard from anywhere, exile it instead.
52          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new RestInPeaceReplacementEffect()));
53      }
54  
55      private RestInPeace(final RestInPeace card) {
56          super(card);
57      }
58  
59      @Override
60      public RestInPeace copy() {
61          return new RestInPeace(this);
62      }
63  }
64  
65  class RestInPeaceReplacementEffect extends ReplacementEffectImpl {
66  
67      public RestInPeaceReplacementEffect() {
68          super(Duration.WhileOnBattlefield, Outcome.Exile);
69          staticText = "If a card or token would be put into a graveyard from anywhere, exile it instead";
70      }
71  
72      public RestInPeaceReplacementEffect(final RestInPeaceReplacementEffect effect) {
73          super(effect);
74      }
75  
76      @Override
77      public RestInPeaceReplacementEffect copy() {
78          return new RestInPeaceReplacementEffect(this);
79      }
80  
81      @Override
82      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
83          ((ZoneChangeEvent) event).setToZone(Zone.EXILED);
84          return false;
85      }
86      
87      @Override
88      public boolean checksEventType(GameEvent event, Game game) {
89          return event.getType() == GameEvent.EventType.ZONE_CHANGE;
90      }    
91  
92      @Override
93      public boolean applies(GameEvent event, Ability source, Game game) {
94          return ((ZoneChangeEvent)event).getToZone() == Zone.GRAVEYARD;
95      }
96  }

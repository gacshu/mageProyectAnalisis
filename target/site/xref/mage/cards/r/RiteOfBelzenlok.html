5   import mage.abilities.common.SagaAbility;
6   import mage.abilities.effects.common.CreateTokenEffect;
7   import mage.constants.SubType;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.SagaChapter;
12  import mage.game.permanent.token.BelzenlokClericToken;
13  import mage.game.permanent.token.BelzenlokDemonToken;
14  
15  /**
16   *
17   * @author TheElk801

19  public final class RiteOfBelzenlok extends CardImpl {
20  
21      public RiteOfBelzenlok(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{B}{B}");
23  
24          this.subtype.add(SubType.SAGA);
25  
26          // <i>(As this Saga enters and after your draw step, add a lore counter. Sacrifice after III.)</i>
27          SagaAbility sagaAbility = new SagaAbility(this);
28          // I, II — Create two 0/1 black Cleric creature tokens.
29          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, SagaChapter.CHAPTER_II, new CreateTokenEffect(new BelzenlokClericToken(), 2));
30          // III — Create a 6/6 black Demon creature token with flying, trample, and "At the beginning of your upkeep, sacrifice another creature.  If you can't, this creature deals 6 damage to you."
31          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_III, new CreateTokenEffect(new BelzenlokDemonToken()));
32          this.addAbility(sagaAbility);
33      }
34  
35      private RiteOfBelzenlok(final RiteOfBelzenlok card) {
36          super(card);
37      }
38  
39      @Override
40      public RiteOfBelzenlok copy() {
41          return new RiteOfBelzenlok(this);
42      }
43  }

31  public final class RodOfAbsorption extends CardImpl {
32  
33      public RodOfAbsorption(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}{U}");
35  
36          // Whenever a player casts an instant or sorcery spell, exile it instead of putting it into a graveyard as it resolves.
37          this.addAbility(new RodOfAbsorptionTriggeredAbility());
38  
39          // {X}, {T}, Sacrifice Rod of Absorption: You may cast any number of spells from among cards exiled with Rod of Absorption with total mana value X or less without paying their mana costs.
40          Ability ability = new SimpleActivatedAbility(new RodOfAbsorptionCastEffect(), new ManaCostsImpl<>("{X}"));
41          ability.addCost(new TapSourceCost());
42          ability.addCost(new SacrificeSourceCost());
43          this.addAbility(ability);
44      }
45  
46      private RodOfAbsorption(final RodOfAbsorption card) {
47          super(card);
48      }
49  
50      @Override
51      public RodOfAbsorption copy() {
52          return new RodOfAbsorption(this);
53      }
54  }

56  class RodOfAbsorptionTriggeredAbility extends TriggeredAbilityImpl {
57  
58      RodOfAbsorptionTriggeredAbility() {
59          super(Zone.BATTLEFIELD, null, false);
60      }
61  
62      private RodOfAbsorptionTriggeredAbility(final RodOfAbsorptionTriggeredAbility ability) {
63          super(ability);
64      }
65  
66      @Override
67      public RodOfAbsorptionTriggeredAbility copy() {
68          return new RodOfAbsorptionTriggeredAbility(this);
69      }
70  
71      @Override
72      public boolean checkEventType(GameEvent event, Game game) {
73          return event.getType() == GameEvent.EventType.SPELL_CAST;
74      }
75  
76      @Override
77      public boolean checkTrigger(GameEvent event, Game game) {
78          Spell spell = game.getStack().getSpell(event.getTargetId());
79          if (spell == null || !spell.isInstantOrSorcery(game)) {
80              return false;
81          }
82          this.getEffects().clear();
83          this.addEffect(new RodOfAbsorptionExileEffect(spell, game));
84          return true;
85      }
86  
87      @Override
88      public String getRule() {
89          return "Whenever a player casts an instant or sorcery spell, " +
90                  "exile it instead of putting it into a graveyard as it resolves.";
91      }
92  }

94  class RodOfAbsorptionExileEffect extends ReplacementEffectImpl {
95  
96      private final MageObjectReference mor;
97  
98      RodOfAbsorptionExileEffect(Spell spell, Game game) {
99          super(Duration.WhileOnStack, Outcome.Benefit);
100         this.mor = new MageObjectReference(spell, game);
101     }
102 
103     private RodOfAbsorptionExileEffect(final RodOfAbsorptionExileEffect effect) {
104         super(effect);
105         this.mor = effect.mor;
106     }
107 
108     @Override
109     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
110         Spell sourceSpell = game.getStack().getSpell(event.getTargetId());
111         if (sourceSpell == null || sourceSpell.isCopy()) {
112             return false;
113         }
114         Player player = game.getPlayer(sourceSpell.getOwnerId());
115         if (player == null) {
116             return false;
117         }
118         player.moveCardsToExile(
119                 sourceSpell, source, game, false,
120                 CardUtil.getExileZoneId(game, source),
121                 CardUtil.getSourceName(game, source)
122         );
123         return true;
124     }
125 
126     @Override
127     public boolean checksEventType(GameEvent event, Game game) {
128         return event.getType() == GameEvent.EventType.ZONE_CHANGE;
129     }
130 
131     @Override
132     public boolean applies(GameEvent event, Ability source, Game game) {
133         ZoneChangeEvent zEvent = ((ZoneChangeEvent) event);
134         if (zEvent.getFromZone() != Zone.STACK
135                 || zEvent.getToZone() != Zone.GRAVEYARD
136                 || event.getSourceId() == null
137                 || !event.getSourceId().equals(event.getTargetId())
138                 || mor.getZoneChangeCounter() != game.getState().getZoneChangeCounter(event.getSourceId())) {
139             return false;
140         }
141         Spell spell = game.getStack().getSpell(mor.getSourceId());
142         return spell != null && spell.isInstantOrSorcery(game);
143     }
144 
145     @Override
146     public RodOfAbsorptionExileEffect copy() {
147         return new RodOfAbsorptionExileEffect(this);
148     }
149 }

151 class RodOfAbsorptionCastEffect extends OneShotEffect {
152 
153     RodOfAbsorptionCastEffect() {
154         super(Outcome.Benefit);
155         staticText = "you may cast any number of spells from among cards exiled with " +
156                 "{this} with total mana value X or less without paying their mana costs";
157     }
158 
159     private RodOfAbsorptionCastEffect(final RodOfAbsorptionCastEffect effect) {
160         super(effect);
161     }
162 
163     @Override
164     public RodOfAbsorptionCastEffect copy() {
165         return new RodOfAbsorptionCastEffect(this);
166     }
167 
168     @Override
169     public boolean apply(Game game, Ability source) {
170         Player player = game.getPlayer(source.getControllerId());
171         ExileZone exileZone = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source));
172         if (exileZone == null || exileZone.isEmpty()) {
173             return false;
174         }
175 
176         Cards cards = new CardsImpl(exileZone);
177         if (player == null || cards.isEmpty()) {
178             return false;
179         }
180 
181         CardUtil.castMultipleWithAttributeForFree(
182                 player, source, game, cards, StaticFilters.FILTER_CARD, Integer.MAX_VALUE,
183                 new RodOfAbsorptionTracker(source.getManaCostsToPay().getX())
184         );
185         return true;
186     }
187 }

189 class RodOfAbsorptionTracker implements CardUtil.SpellCastTracker {
190 
191     private final int xValue;
192     private int totalManaValue = 0;
193 
194     RodOfAbsorptionTracker(int xValue) {
195         this.xValue = xValue;
196     }
197 
198     @Override
199     public boolean checkCard(Card card, Game game) {
200         return card.getManaValue() + totalManaValue <= xValue;
201     }
202 
203     @Override
204     public void addCard(Card card, Ability source, Game game) {
205         totalManaValue += card.getManaValue();
206     }

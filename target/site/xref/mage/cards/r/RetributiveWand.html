4   import mage.abilities.common.PutIntoGraveFromBattlefieldSourceTriggeredAbility;

8   import mage.abilities.effects.common.DamageTargetEffect;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  import mage.target.common.TargetAnyTarget;
13  
14  import java.util.UUID;
15  
16  /**
17   * @author TheElk801
18   */
19  public final class RetributiveWand extends CardImpl {
20  
21      public RetributiveWand(UUID ownerId, CardSetInfo setInfo) {
22          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
23  
24          // {3}, {T}: Retributive Wand deals 1 damage to any target.
25          Ability ability = new SimpleActivatedAbility(new DamageTargetEffect(1), new GenericManaCost(3));
26          ability.addCost(new TapSourceCost());
27          ability.addTarget(new TargetAnyTarget());
28          this.addAbility(ability);
29  
30          // When Retributive Wand is put into a graveyard from the battlefield, it deals 5 damage to any target.
31          ability = new PutIntoGraveFromBattlefieldSourceTriggeredAbility(
32                  new DamageTargetEffect(5, "it")
33          );
34          ability.addTarget(new TargetAnyTarget());
35          this.addAbility(ability);
36      }
37  
38      private RetributiveWand(final RetributiveWand card) {
39          super(card);
40      }
41  
42      @Override
43      public RetributiveWand copy() {
44          return new RetributiveWand(this);
45      }
46  }

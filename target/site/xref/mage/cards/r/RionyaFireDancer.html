29  public final class RionyaFireDancer extends CardImpl {
30  
31      private static final FilterPermanent filter
32              = new FilterControlledCreaturePermanent("another creature you control");
33  
34      static {
35          filter.add(AnotherPredicate.instance);
36      }
37  
38      public RionyaFireDancer(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}{R}");
40  
41          this.addSuperType(SuperType.LEGENDARY);
42          this.subtype.add(SubType.HUMAN);
43          this.subtype.add(SubType.WIZARD);
44          this.power = new MageInt(3);
45          this.toughness = new MageInt(4);
46  
47          // At the beginning of combat on your turn, create X tokens that are copies of another target creature you control, where X is one plus the number of instant and sorcery spells you've cast this turn. They gain haste. Exile them at the beginning of the next end step.
48          Ability ability = new BeginningOfCombatTriggeredAbility(
49                  new RionyaFireDancerEffect(), TargetController.YOU, false
50          );
51          ability.addTarget(new TargetPermanent(filter));
52          this.addAbility(ability.addHint(RionyaFireDancerHint.instance), new RionyaFireDancerWatcher());
53      }
54  
55      private RionyaFireDancer(final RionyaFireDancer card) {
56          super(card);
57      }
58  
59      @Override
60      public RionyaFireDancer copy() {
61          return new RionyaFireDancer(this);
62      }
63  }
64  
65  enum RionyaFireDancerHint implements Hint {
66      instance;
67  
68      @Override
69      public String getText(Game game, Ability ability) {
70          return "Instants and sorceries you've cast this turn: "
71                  + RionyaFireDancerWatcher.getValue(ability.getControllerId(), game);
72      }
73  
74      @Override
75      public RionyaFireDancerHint copy() {
76          return instance;
77      }
78  }

80  class RionyaFireDancerEffect extends OneShotEffect {
81  
82      RionyaFireDancerEffect() {
83          super(Outcome.Benefit);
84          staticText = "create X tokens that are copies of another target creature you control, " +
85                  "where X is one plus the number of instant and sorcery spells you've cast this turn. " +
86                  "They gain haste. Exile them at the beginning of the next end step";
87      }
88  
89      private RionyaFireDancerEffect(final RionyaFireDancerEffect effect) {
90          super(effect);
91      }
92  
93      @Override
94      public RionyaFireDancerEffect copy() {
95          return new RionyaFireDancerEffect(this);
96      }
97  
98      @Override
99      public boolean apply(Game game, Ability source) {
100         CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect(
101                 source.getControllerId(), null, true,
102                 RionyaFireDancerWatcher.getValue(source.getControllerId(), game) + 1
103         );
104         effect.apply(game, source);
105         effect.exileTokensCreatedAtNextEndStep(game, source);
106         return true;
107     }
108 }

110 class RionyaFireDancerWatcher extends Watcher {
111 
112     private final Map<UUID, Integer> playerMap = new HashMap<>();
113 
114     RionyaFireDancerWatcher() {
115         super(WatcherScope.GAME);
116     }
117 
118     @Override
119     public void watch(GameEvent event, Game game) {
120         if (event.getType() != GameEvent.EventType.SPELL_CAST) {
121             return;
122         }
123         Spell spell = game.getSpell(event.getTargetId());
124         if (spell != null && spell.isInstantOrSorcery(game)) {
125             playerMap.compute(spell.getControllerId(), CardUtil::setOrIncrementValue);
126         }
127     }
128 
129     @Override
130     public void reset() {
131         super.reset();
132         playerMap.clear();
133     }
134 
135     static int getValue(UUID playerId, Game game) {
136         RionyaFireDancerWatcher watcher = game.getState().getWatcher(RionyaFireDancerWatcher.class);
137         return watcher == null ? 0 : watcher.playerMap.getOrDefault(playerId, 0);
138     }

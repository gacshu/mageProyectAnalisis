31  public final class RaggadraggaGoregutsBoss extends CardImpl {
32  
33      private static final FilterCreaturePermanent filter
34              = new FilterCreaturePermanent("creature you control with a mana ability");
35      private static final FilterSpell filter2 = new FilterSpell();
36  
37      static {
38          filter.add(TargetController.YOU.getOwnerPredicate());
39          filter.add(RaggadraggaGoregutsBossCreaturePredicate.instance);
40          filter2.add(RaggadraggaGoregutsBossSpellPredicate.instance);
41      }
42  
43      public RaggadraggaGoregutsBoss(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{G}");
45  
46          this.addSuperType(SuperType.LEGENDARY);
47          this.subtype.add(SubType.HUMAN);
48          this.subtype.add(SubType.BOAR);
49          this.power = new MageInt(4);
50          this.toughness = new MageInt(4);
51  
52          // Each creature you control with a mana ability gets +2/+2.
53          this.addAbility(new SimpleStaticAbility(new BoostAllEffect(
54                  2, 2, Duration.WhileOnBattlefield, filter, false
55          ).setText("Each " + filter.getMessage() + " gets +2/+2")));
56  
57          // Whenever a creature you control with a mana ability attacks, untap it.
58          this.addAbility(new AttacksAllTriggeredAbility(
59                  new UntapTargetEffect("untap it"), false, filter,
60                  SetTargetPointer.PERMANENT, false
61          ));
62  
63          // Whenever you cast a spell, if at least seven mana was spent to cast it, untap target creature. It gets +7/+7 and gains trample until end of turn.
64          Ability ability = new SpellCastControllerTriggeredAbility(
65                  new UntapTargetEffect(), filter2, false
66          ).setTriggerPhrase("Whenever you cast a spell, if at least seven mana was spent to cast it, ");
67          ability.addEffect(new BoostTargetEffect(7, 7).setText("It gets +7/+7"));
68          ability.addEffect(new GainAbilityTargetEffect(
69                  TrampleAbility.getInstance(), Duration.EndOfTurn
70          ).setText("and gains trample until end of turn"));
71          ability.addTarget(new TargetCreaturePermanent());
72          this.addAbility(ability);
73      }
74  
75      private RaggadraggaGoregutsBoss(final RaggadraggaGoregutsBoss card) {
76          super(card);
77      }
78  
79      @Override
80      public RaggadraggaGoregutsBoss copy() {
81          return new RaggadraggaGoregutsBoss(this);
82      }
83  }

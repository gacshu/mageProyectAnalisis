24  /**
25   * @author TheElk801
26   */
27  public final class RavagerWurm extends CardImpl {
28  
29      private static final FilterPermanent filter
30              = new FilterPermanent("land with an activated ability that isn't a mana ability");
31  
32      static {
33          filter.add(RavagerWurmPredicate.instance);
34      }
35  
36      public RavagerWurm(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}{G}{G}");
38  
39          this.subtype.add(SubType.WURM);
40          this.power = new MageInt(4);
41          this.toughness = new MageInt(5);
42  
43          // Riot
44          this.addAbility(new RiotAbility());
45  
46          // When Ravager Wurm enters the battlefield, choose up to one —
47          // • Ravager Wurm fights target creature you don't control.
48          Ability ability = new EntersBattlefieldTriggeredAbility(
49                  new FightTargetSourceEffect().setText("{this} fights target creature you don't control"), false
50          );
51          ability.addTarget(new TargetPermanent(StaticFilters.FILTER_CREATURE_YOU_DONT_CONTROL));
52          ability.getModes().setMinModes(0);
53          ability.getModes().setMaxModes(1);
54  
55          // • Destroy target land with an activated ability that isn't a mana ability.
56          Mode mode = new Mode(new DestroyTargetEffect());
57          mode.addTarget(new TargetPermanent(filter));
58          ability.addMode(mode);
59          this.addAbility(ability);
60      }
61  
62      private RavagerWurm(final RavagerWurm card) {
63          super(card);
64      }
65  
66      @Override
67      public RavagerWurm copy() {
68          return new RavagerWurm(this);
69      }
70  }
71  
72  enum RavagerWurmPredicate implements Predicate<Permanent> {
73      instance;
74  
75      @Override
76      public boolean apply(Permanent input, Game game) {
77          return input != null && input.isLand(game)
78                  && input
79                  .getAbilities(game)
80                  .stream()
81                  .map(Ability::getAbilityType)
82                  .anyMatch(AbilityType.ACTIVATED::equals);
83      }
84  }

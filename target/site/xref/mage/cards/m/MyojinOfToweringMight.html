29  public final class MyojinOfToweringMight extends CardImpl {
30  
31      public MyojinOfToweringMight(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{5}{G}{G}{G}");
33  
34          this.addSuperType(SuperType.LEGENDARY);
35          this.subtype.add(SubType.SPIRIT);
36          this.power = new MageInt(8);
37          this.toughness = new MageInt(8);
38  
39          // Myojin of Towering Might enters the battlefield with an indestructible counter on it if you cast it from your hand.
40          this.addAbility(new EntersBattlefieldAbility(
41                  new AddCountersSourceEffect(CounterType.INDESTRUCTIBLE.createInstance()),
42                  CastFromHandSourcePermanentCondition.instance, null,
43                  "with an indestructible counter on it if you cast it from your hand"
44          ), new CastFromHandWatcher());
45  
46          // Remove an indestructible counter from Myojin of Towering Might: Distribute eight +1/+1 counters among any number of target creatures you control. They gain trample until end of turn.
47          Ability ability = new SimpleActivatedAbility(new DistributeCountersEffect(
48                  CounterType.P1P1, 8, false,
49                  "any number of target creatures you control"
50          ), new RemoveCountersSourceCost(CounterType.INDESTRUCTIBLE.createInstance()));
51          ability.addEffect(new GainAbilityTargetEffect(
52                  TrampleAbility.getInstance(), Duration.EndOfTurn
53          ).setText("They gain trample until end of turn"));
54          ability.addTarget(new TargetCreaturePermanentAmount(8, StaticFilters.FILTER_CONTROLLED_CREATURES));
55          this.addAbility(ability);
56      }
57  
58      private MyojinOfToweringMight(final MyojinOfToweringMight card) {
59          super(card);
60      }
61  
62      @Override
63      public MyojinOfToweringMight copy() {
64          return new MyojinOfToweringMight(this);
65      }
66  }

12  import mage.abilities.effects.common.PutOnLibraryTargetEffect;

14  import mage.cards.CardSetInfo;
15  import mage.constants.CardType;
16  import mage.constants.Duration;
17  import mage.constants.SubType;
18  import mage.constants.Zone;
19  import mage.filter.FilterObject;
20  import mage.filter.FilterPermanent;
21  import mage.filter.FilterStackObject;
22  import mage.filter.predicate.mageobject.ColorPredicate;
23  import mage.filter.predicate.permanent.TokenPredicate;
24  import mage.target.TargetPermanent;
25  
26  /**
27   *
28   * @author fireshoes
29   */
30  public final class MercenaryInformer extends CardImpl {
31      
32      private static final FilterObject filterBlack = new FilterStackObject("black spells or abilities from black sources");
33      private static final FilterPermanent filterMercenary = new FilterPermanent("nontoken Mercenary");
34  
35      static {
36          filterBlack.add(new ColorPredicate(ObjectColor.BLACK));
37          filterMercenary.add(TokenPredicate.FALSE);
38          filterMercenary.add(SubType.MERCENARY.getPredicate());
39      }
40  
41      public MercenaryInformer(UUID ownerId, CardSetInfo setInfo) {
42          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{2}{W}");
43          this.subtype.add(SubType.HUMAN);
44          this.subtype.add(SubType.REBEL);
45          this.subtype.add(SubType.MERCENARY);
46          this.power = new MageInt(2);
47          this.toughness = new MageInt(1);
48  
49          // Mercenary Informer can't be the target of black spells or abilities from black sources.
50          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new CantBeTargetedSourceEffect(filterBlack, Duration.WhileOnBattlefield)));
51          
52          // {2}{W}: Put target nontoken Mercenary on the bottom of its owner's library.
53          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new PutOnLibraryTargetEffect(false), new ManaCostsImpl<>("{2}{W}"));
54          ability.addTarget(new TargetPermanent(filterMercenary));
55          this.addAbility(ability);
56      }
57  
58      private MercenaryInformer(final MercenaryInformer card) {
59          super(card);
60      }
61  
62      @Override
63      public MercenaryInformer copy() {
64          return new MercenaryInformer(this);
65      }
66  }

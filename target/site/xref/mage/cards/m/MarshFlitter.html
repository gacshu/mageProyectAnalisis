9   import mage.abilities.effects.common.CreateTokenEffect;

15  import mage.filter.common.FilterControlledPermanent;
16  import mage.game.permanent.token.GoblinRogueToken;
17  import mage.target.common.TargetControlledPermanent;
18  
19  import java.util.UUID;
20  
21  /**
22   * @author LevelX2
23   */
24  public final class MarshFlitter extends CardImpl {
25  
26      private static final FilterControlledPermanent filter = new FilterControlledPermanent("a Goblin");
27  
28      static {
29          filter.add(SubType.GOBLIN.getPredicate());
30      }
31  
32      public MarshFlitter(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{B}");
34          this.subtype.add(SubType.FAERIE);
35          this.subtype.add(SubType.ROGUE);
36  
37          this.power = new MageInt(1);
38          this.toughness = new MageInt(1);
39  
40          // Flying
41          this.addAbility(FlyingAbility.getInstance());
42  
43          // When Marsh Flitter enters the battlefield, create two 1/1 black Goblin Rogue creature tokens.
44          this.addAbility(new EntersBattlefieldTriggeredAbility(new CreateTokenEffect(new GoblinRogueToken(), 2), false));
45  
46          // Sacrifice a Goblin: Marsh Flitter has base power and toughness 3/3 until end of turn.
47          Effect effect = new SetBasePowerToughnessSourceEffect(3, 3, Duration.EndOfTurn, SubLayer.SetPT_7b, true);
48          effect.setText("{this} has base power and toughness 3/3 until end of turn");
49          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new SacrificeTargetCost(new TargetControlledPermanent(filter)));
50          this.addAbility(ability);
51      }
52  
53      private MarshFlitter(final MarshFlitter card) {
54          super(card);
55      }
56  
57      @Override
58      public MarshFlitter copy() {
59          return new MarshFlitter(this);
60      }
61  }

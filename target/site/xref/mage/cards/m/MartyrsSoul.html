1   package mage.cards.m;

3   import mage.MageInt;
4   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
5   import mage.abilities.condition.Condition;
6   import mage.abilities.condition.common.PermanentsOnTheBattlefieldCondition;
7   import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
8   import mage.abilities.effects.common.counter.AddCountersSourceEffect;
9   import mage.abilities.keyword.ConvokeAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.ComparisonType;
14  import mage.constants.SubType;
15  import mage.counters.CounterType;
16  import mage.filter.FilterPermanent;
17  import mage.filter.common.FilterControlledLandPermanent;
18  import mage.filter.predicate.permanent.TappedPredicate;
19  
20  import java.util.UUID;
21  
22  /**
23   * @author TheElk801
24   */
25  public final class MartyrsSoul extends CardImpl {
26  
27      private static final FilterPermanent filter = new FilterControlledLandPermanent();
28  
29      static {
30          filter.add(TappedPredicate.TAPPED);
31      }
32  
33      private static final Condition condition
34              = new PermanentsOnTheBattlefieldCondition(filter, ComparisonType.EQUAL_TO, 0);
35  
36      public MartyrsSoul(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{W}");
38  
39          this.subtype.add(SubType.SPIRIT);
40          this.subtype.add(SubType.SOLDIER);
41          this.power = new MageInt(3);
42          this.toughness = new MageInt(2);
43  
44          // Convoke
45          this.addAbility(new ConvokeAbility());
46  
47          // When Martyr's Soul enters the battlefield, if you control no tapped lands, put two +1/+1 counters on it.
48          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
49                  new EntersBattlefieldTriggeredAbility(new AddCountersSourceEffect(
50                          CounterType.P1P1.createInstance(2)
51                  )), condition, "When {this} enters the battlefield, " +
52                  "if you control no tapped lands, put two +1/+1 counters on it."
53          ));
54      }
55  
56      private MartyrsSoul(final MartyrsSoul card) {
57          super(card);
58      }
59  
60      @Override
61      public MartyrsSoul copy() {
62          return new MartyrsSoul(this);
63      }
64  }

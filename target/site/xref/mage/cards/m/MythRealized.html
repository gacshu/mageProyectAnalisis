28  public final class MythRealized extends CardImpl {
29  
30      private static final DynamicValue loreCounterCount = new CountersSourceCount(CounterType.LORE);
31      private static final FilterSpell filterNonCreature = new FilterSpell("a noncreature spell");
32  
33      static {
34          filterNonCreature.add(Predicates.not(CardType.CREATURE.getPredicate()));
35      }
36      
37      public MythRealized(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{W}");
39  
40          // Whenever you cast a noncreature spell, put a lore counter on Myth Realized.
41          this.addAbility(new SpellCastControllerTriggeredAbility(new AddCountersSourceEffect(CounterType.LORE.createInstance()), filterNonCreature, false));
42  
43          // {2}{W}: Put a lore counter on Myth Realized.
44          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.LORE.createInstance()), new ManaCostsImpl<>("{2}{W}")));
45  
46          // {W}: Until end of turn, Myth Realized becomes a Monk Avatar creature in addition to its other types and gains "This creature's power and toughness are each equal to the number of lore counters on it."
47          Effect effect = new BecomesCreatureSourceEffect(new MythRealizedToken(), null, Duration.EndOfTurn);
48          effect.setText("Until end of turn, {this} becomes a Monk Avatar creature in addition to its other types ");
49          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect, new ManaCostsImpl<>("{W}"));
50          ability.addEffect(new SetBasePowerToughnessSourceEffect(loreCounterCount, loreCounterCount, Duration.EndOfTurn, SubLayer.SetPT_7b, false).setText("and gains \"This creature's power and toughness are each equal to the number of lore counters on it.\""));
51  
52          this.addAbility(ability);
53      }
54  
55      private MythRealized(final MythRealized card) {
56          super(card);
57      }
58  
59      @Override
60      public MythRealized copy() {
61          return new MythRealized(this);
62      }
63  }
64  
65  class MythRealizedToken extends TokenImpl {
66  
67      public MythRealizedToken() {
68          super("", "Monk Avatar creature in addition to its other types and gains \"This creature's power and toughness are each equal to the number of lore counters on it.\"");
69          cardType.add(CardType.CREATURE);
70          subtype.add(SubType.MONK);
71          subtype.add(SubType.AVATAR);
72          power = new MageInt(0);
73          toughness = new MageInt(0);
74      }
75      public MythRealizedToken(final MythRealizedToken token) {
76          super(token);
77      }
78  
79      @Override
80      public MythRealizedToken copy() {
81          return new MythRealizedToken(this);
82      }
83  }

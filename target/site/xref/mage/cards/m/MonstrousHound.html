24  public final class MonstrousHound extends CardImpl {

26      public MonstrousHound(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{R}");
28  
29          this.subtype.add(SubType.DOG);
30          this.power = new MageInt(4);
31          this.toughness = new MageInt(4);
32  
33          // Monstrous Hound can't attack unless you control more lands than defending player.
34          Effect effect = new CantAttackUnlessControllerControlsMoreLandsEffect();
35          this.addAbility(new SimpleStaticAbility(
36                  Zone.BATTLEFIELD,
37                  effect.setText("{this} can't attack unless you control more lands than defending player")));
38  
39          // Monstrous Hound can't block unless you control more lands than attacking player.
40          Effect effect2 = new CantBlockUnlessControllerControlsMoreLandsEffect();
41          this.addAbility(new SimpleStaticAbility(
42                  Zone.BATTLEFIELD,
43                  effect2.setText("{this} can't block unless you control more lands than attacking player")));
44  
45      }
46

48          super(card);
49      }

57  class CantAttackUnlessControllerControlsMoreLandsEffect extends RestrictionEffect {
58  
59      CantAttackUnlessControllerControlsMoreLandsEffect() {
60          super(Duration.WhileOnBattlefield);
61      }
62  
63      CantAttackUnlessControllerControlsMoreLandsEffect(final CantAttackUnlessControllerControlsMoreLandsEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public boolean applies(Permanent permanent, Ability source, Game game) {
69          return permanent.getId().equals(source.getSourceId());
70      }
71  
72      @Override
73      public boolean canAttack(Permanent attacker, UUID defenderId, Ability source, Game game, boolean canUseChooseDialogs) {
74          if (defenderId == null) {
75              return true;
76          }
77  
78          UUID defendingPlayerId;
79          Player defender = game.getPlayer(defenderId);
80          if (defender == null) {
81              Permanent permanent = game.getPermanent(defenderId);
82              if (permanent != null) {
83                  defendingPlayerId = permanent.getControllerId();
84              } else {
85                  return false;
86              }
87          } else {
88              defendingPlayerId = defenderId;
89          }
90          if (defendingPlayerId != null) {
91              return game.getBattlefield().countAll(new FilterControlledLandPermanent(),
92                      source.getControllerId(), game) > game.getBattlefield().countAll(new FilterControlledLandPermanent(),
93                      defendingPlayerId, game);
94          } else {
95              return true;
96          }
97      }
98  
99      @Override
100     public CantAttackUnlessControllerControlsMoreLandsEffect copy() {
101         return new CantAttackUnlessControllerControlsMoreLandsEffect(this);
102     }
103 }

105 class CantBlockUnlessControllerControlsMoreLandsEffect extends RestrictionEffect {
106 
107     CantBlockUnlessControllerControlsMoreLandsEffect() {
108         super(Duration.WhileOnBattlefield);
109     }
110 
111     CantBlockUnlessControllerControlsMoreLandsEffect(final CantBlockUnlessControllerControlsMoreLandsEffect effect) {
112         super(effect);
113     }
114 
115     @Override
116     public CantBlockUnlessControllerControlsMoreLandsEffect copy() {
117         return new CantBlockUnlessControllerControlsMoreLandsEffect(this);
118     }
119 
120     @Override
121     public boolean canBlock(Permanent attacker, Permanent blocker, Ability source, Game game, boolean canUseChooseDialogs) {
122         if (attacker == null) {
123             return true;
124         }
125         UUID attackingPlayerId = attacker.getControllerId();
126         if (attackingPlayerId != null) {
127             return game.getBattlefield().countAll(new FilterControlledLandPermanent(),
128                     source.getControllerId(), game) > game.getBattlefield().countAll(new FilterControlledLandPermanent(),
129                     attackingPlayerId, game);
130         }
131         return true;
132     }
133 
134     @Override
135     public boolean applies(Permanent permanent, Ability source, Game game) {
136         return permanent.getId().equals(source.getSourceId());
137     }

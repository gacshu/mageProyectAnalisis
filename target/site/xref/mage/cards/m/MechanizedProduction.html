32  public final class MechanizedProduction extends CardImpl {
33  
34      public MechanizedProduction(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{U}{U}");
36  
37          this.subtype.add(SubType.AURA);
38  
39          // Enchant artifact you control
40          TargetPermanent auraTarget = new TargetControlledPermanent(new FilterControlledArtifactPermanent());
41          this.getSpellAbility().addTarget(auraTarget);
42          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Copy));
43          Ability ability = new EnchantAbility(auraTarget);
44          this.addAbility(ability);
45  
46          // At the beginning of your upkeep, create a token that's a copy of enchanted artifact. Then if you control eight or more artifacts with the same name as one another, you win the game.
47          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new MechanizedProductionEffect(), TargetController.YOU, false));
48      }
49  
50      private MechanizedProduction(final MechanizedProduction card) {
51          super(card);
52      }
53  
54      @Override
55      public MechanizedProduction copy() {
56          return new MechanizedProduction(this);
57      }
58  }

60  class MechanizedProductionEffect extends OneShotEffect {
61  
62      public MechanizedProductionEffect() {
63          super(Outcome.Benefit);
64          this.staticText = "create a token that's a copy of enchanted artifact. Then if you control eight or more artifacts with the same name as one another, you win the game";
65      }
66  
67      public MechanizedProductionEffect(final MechanizedProductionEffect effect) {
68          super(effect);
69      }
70  
71      @Override
72      public MechanizedProductionEffect copy() {
73          return new MechanizedProductionEffect(this);
74      }
75  
76      @Override
77      public boolean apply(Game game, Ability source) {
78          Permanent sourceObject = game.getPermanentOrLKIBattlefield(source.getSourceId());
79          if (sourceObject != null && sourceObject.getAttachedTo() != null) {
80              Permanent enchantedArtifact = game.getPermanentOrLKIBattlefield(sourceObject.getAttachedTo());
81              if (enchantedArtifact != null) {
82                  EmptyToken token = new EmptyToken();
83                  CardUtil.copyTo(token).from(enchantedArtifact, game);
84                  token.putOntoBattlefield(1, game, source, source.getControllerId());
85              }
86              Map<String, Integer> countNames = new HashMap<>();
87              for (Permanent permanent : game.getBattlefield().getAllActivePermanents(new FilterArtifactPermanent(), source.getControllerId(), game)) {
88                  int counter = countNames.getOrDefault(permanent.getName(), 0);
89                  countNames.put(permanent.getName(), counter + 1);
90              }
91              for (Entry<String, Integer> entry : countNames.entrySet()) {
92                  if (entry.getValue() > 7) {
93                      Player controller = game.getPlayer(source.getControllerId());
94                      if (controller != null) {
95                          game.informPlayers(controller.getLogName() + " controls eight or more artifacts with the same name as one another (" + entry.getKey() + ").");
96                          controller.won(game);
97                          return true;
98                      }
99                  }
100             }
101             return true;
102 
103         }
104         return false;
105     }

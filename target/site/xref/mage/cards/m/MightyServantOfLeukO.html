95  class MightyServantOfLeukOWatcher extends Watcher {
96  
97      private final Map<MageObjectReference, Integer> crewCount = new HashMap<>();
98      private final Map<MageObjectReference, Set<MageObjectReference>> crewMap = new HashMap<>();
99  
100     MightyServantOfLeukOWatcher() {
101         super(WatcherScope.GAME);
102     }
103 
104     @Override
105     public void watch(GameEvent event, Game game) {
106         Permanent vehicle;
107         Permanent crewer;
108         switch (event.getType()) {
109             case VEHICLE_CREWED:
110                 vehicle = game.getPermanent(event.getTargetId());
111                 crewer = null;
112                 break;
113             case CREWED_VEHICLE:
114                 vehicle = game.getPermanent(event.getSourceId());
115                 crewer = game.getPermanent(event.getTargetId());
116                 break;
117             default:
118                 return;
119         }
120         if (vehicle == null) {
121             return;
122         }
123         if (crewer == null) {
124             crewCount.compute(new MageObjectReference(vehicle, game), CardUtil::setOrIncrementValue);
125             return;
126         }
127         crewMap.computeIfAbsent(
128                 new MageObjectReference(vehicle, game), x -> new HashSet<>()
129         ).add(new MageObjectReference(crewer, game));
130     }
131 
132     @Override
133     public void reset() {
134         super.reset();
135         crewCount.clear();
136         crewMap.clear();
137     }
138 
139     public static boolean checkVehicle(Ability source, Game game) {
140         MightyServantOfLeukOWatcher watcher = game
141                 .getState()
142                 .getWatcher(MightyServantOfLeukOWatcher.class);
143         MageObjectReference mor = new MageObjectReference(source);
144         return watcher
145                 .crewCount
146                 .getOrDefault(mor, 0) < 2
147                 && watcher
148                 .crewMap
149                 .getOrDefault(mor, Collections.emptySet())
150                 .size() == 2;
151     }

28  public final class Mindblaze extends CardImpl {
29  
30      public Mindblaze(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{5}{R}");
32  
33          // Name a nonland card and choose a number greater than 0. Target player reveals their library.
34          // If that library contains exactly the chosen number of the named card,
35          // Mindblaze deals 8 damage to that player.
36          // Then that player shuffles their library.
37          this.getSpellAbility().addEffect(new MindblazeEffect());
38          this.getSpellAbility().addTarget(new TargetPlayer());
39      }
40  
41      private Mindblaze(final Mindblaze card) {
42          super(card);
43      }
44  
45      @Override
46      public Mindblaze copy() {
47          return new Mindblaze(this);
48      }
49  
50  }

52  class MindblazeEffect extends OneShotEffect {
53  
54      MindblazeEffect() {
55          super(Outcome.Damage);
56          staticText = "Choose a nonland card name and a number greater than 0. Target player reveals their library. " +
57                  "If that library contains exactly the chosen number of cards with the chosen name, " +
58                  "{this} deals 8 damage to that player. Then that player shuffles";
59      }
60  
61      MindblazeEffect(final MindblazeEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public boolean apply(Game game, Ability source) {
67          Player player = game.getPlayer(targetPointer.getFirst(game, source));
68          Player playerControls = game.getPlayer(source.getControllerId());
69          MageObject sourceObject = source.getSourceObject(game);
70          if (player == null || playerControls == null || sourceObject == null) {
71              return false;
72          }
73          Choice numberChoice = new ChoiceImpl();
74          numberChoice.setMessage("Choose a number greater than 0");
75          Set<String> numbers = new HashSet<>();
76          for (int i = 1; i <= 4; i++) {
77              numbers.add(Integer.toString(i));
78          }
79          numberChoice.setChoices(numbers);
80  
81          String cardName = ChooseACardNameEffect.TypeOfName.NON_LAND_NAME.getChoice(playerControls, game, source, false);
82          playerControls.choose(Outcome.Neutral, numberChoice, game);
83          game.informPlayers(sourceObject.getIdName() + " - Chosen number: [" + numberChoice.getChoice() + ']');
84  
85          Cards cards = new CardsImpl();
86          cards.addAll(player.getLibrary().getCards(game));
87          playerControls.revealCards("Library", cards, game);
88          FilterCard filter = new FilterCard();
89          filter.add(new NamePredicate(cardName));
90          int count = Integer.parseInt(numberChoice.getChoice());
91          if (player.getLibrary().count(filter, game) == count) {
92              player.damage(8, source.getSourceId(), source, game);
93          }
94          player.shuffleLibrary(source, game);
95          return true;
96      }
97  
98      @Override
99      public MindblazeEffect copy() {
100         return new MindblazeEffect(this);
101     }
102 }

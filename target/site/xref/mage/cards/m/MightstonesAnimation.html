19  import java.util.UUID;

21  /**
22   * @author TheElk801
23   */
24  public final class MightstonesAnimation extends CardImpl {
25  
26      public MightstonesAnimation(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{U}");
28  
29          this.subtype.add(SubType.AURA);
30  
31          // Enchant artifact
32          TargetPermanent auraTarget = new TargetArtifactPermanent();
33          this.getSpellAbility().addTarget(auraTarget);
34          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
35          this.addAbility(new EnchantAbility(auraTarget));
36  
37          // When Mightstone's Animation enters the battlefield, draw a card.
38          this.addAbility(new EntersBattlefieldTriggeredAbility(new DrawCardSourceControllerEffect(1)));
39  
40          // Enchanted artifact is a creature with base power and toughness 4/4 in addition to its other types.
41          this.addAbility(new SimpleStaticAbility(new BecomesCreatureAttachedEffect(
42                  new CreatureToken(4, 4), "Enchanted artifact is a creature " +
43                  "with base power and toughness 4/4 in addition to its other types", Duration.WhileOnBattlefield
44          )));
45      }
46  
47      private MightstonesAnimation(final MightstonesAnimation card) {
48          super(card);
49      }
50  
51      @Override
52      public MightstonesAnimation copy() {
53          return new MightstonesAnimation(this);
54      }
55  }

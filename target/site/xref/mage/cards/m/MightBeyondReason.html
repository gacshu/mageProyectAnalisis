18  public final class MightBeyondReason extends CardImpl {
19  
20      public MightBeyondReason(UUID ownerId, CardSetInfo setInfo) {
21          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{3}{G}");
22  
23          // Put two +1/+1 counter on target creature.
24          // <i>Delirium</i> &mdash; Put three +1/+1 counter on that creature instead if there are four or more card types among cards in your graveyard.
25          getSpellAbility().addEffect(new ConditionalOneShotEffect(
26                  new AddCountersTargetEffect(CounterType.P1P1.createInstance(3)),
27                  new AddCountersTargetEffect(CounterType.P1P1.createInstance(2)),
28                  DeliriumCondition.instance,
29                  "Put two +1/+1 counter on target creature.<br>"
30                          + "<i>Delirium</i> &mdash; Put three +1/+1 counters on that creature instead if there are four or more card types among cards in your graveyard"
31          ));
32          getSpellAbility().addTarget(new TargetCreaturePermanent());
33          getSpellAbility().addHint(CardTypesInGraveyardHint.YOU);
34      }
35  
36      private MightBeyondReason(final MightBeyondReason card) {
37          super(card);
38      }
39  
40      @Override
41      public MightBeyondReason copy() {
42          return new MightBeyondReason(this);
43      }
44  }

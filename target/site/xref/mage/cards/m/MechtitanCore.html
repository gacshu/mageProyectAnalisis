39  public final class MechtitanCore extends CardImpl {
40  
41      private static final FilterControlledPermanent filter
42              = new FilterControlledArtifactPermanent("other artifact creatures and/or Vehicles you control");
43  
44      static {
45          filter.add(AnotherPredicate.instance);
46          filter.add(Predicates.or(
47                  CardType.CREATURE.getPredicate(),
48                  SubType.VEHICLE.getPredicate()
49          ));
50      }
51  
52      public MechtitanCore(UUID ownerId, CardSetInfo setInfo) {
53          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
54  
55          this.subtype.add(SubType.VEHICLE);
56          this.power = new MageInt(2);
57          this.toughness = new MageInt(4);
58  
59          // {5}, Exile Mechtitan Core and four other artifact creatures and/or Vehicles you control: Create Mechtitan, a legendary 10/10 Construct artifact creature token with flying, vigilance, trample, lifelink, and haste that's all colors. When that token leaves the battlefield, return all cards exiled with Mechtitan Core except Mechtitan Core to the battlefield tapped under their owners' control.
60          Ability ability = new SimpleActivatedAbility(new MechtitanCoreTokenEffect(), new GenericManaCost(5));
61          ability.addCost(new CompositeCost(
62                  new ExileSourceCost(), new ExileTargetCost(new TargetControlledPermanent(4, 4, filter, true)),
63                  "exile {this} and four other artifact creatures and/or Vehicles you control"));
64          this.addAbility(ability);
65  
66          // Crew 2
67          this.addAbility(new CrewAbility(2));
68      }
69  
70      private MechtitanCore(final MechtitanCore card) {
71          super(card);
72      }
73  
74      @Override
75      public MechtitanCore copy() {
76          return new MechtitanCore(this);
77      }
78  }

108 class MechtitanCoreTriggeredAbility extends DelayedTriggeredAbility {
109 
110     private final Set<UUID> tokenIds = new HashSet<>();
111 
112     MechtitanCoreTriggeredAbility(Token token, Ability source, Game game) {
113         super(new MechtitanCoreReturnEffect(), Duration.Custom, false, false);
114         this.getEffects().setTargetPointer(new FixedTargets(game.getExile().getExileZone(CardUtil.getExileZoneId(game, source)), game));
115         tokenIds.addAll(token.getLastAddedTokenIds());
116     }
117 
118     private MechtitanCoreTriggeredAbility(final MechtitanCoreTriggeredAbility ability) {
119         super(ability);
120         this.tokenIds.addAll(ability.tokenIds);
121     }
122 
123     @Override
124     public MechtitanCoreTriggeredAbility copy() {
125         return new MechtitanCoreTriggeredAbility(this);
126     }
127 
128     @Override
129     public boolean checkEventType(GameEvent event, Game game) {
130         return event.getType() == GameEvent.EventType.ZONE_CHANGE;
131     }
132 
133     @Override
134     public boolean checkTrigger(GameEvent event, Game game) {
135         if (tokenIds.contains(event.getTargetId()) && ((ZoneChangeEvent) event).getFromZone() == Zone.BATTLEFIELD) {
136             tokenIds.remove(event.getTargetId());
137             return true;
138         }
139         return false;
140     }
141 
142     @Override
143     public boolean isInactive(Game game) {
144         return tokenIds.isEmpty();
145     }
146 
147     @Override
148     public String getRule() {
149         return "When that token leaves the battlefield, return all cards exiled with {this} except "
150                 + "{this} to the battlefield tapped under their owners' control";
151     }
152 }

154 class MechtitanCoreReturnEffect extends OneShotEffect {
155 
156     MechtitanCoreReturnEffect() {
157         super(Outcome.Benefit);
158     }
159 
160     private MechtitanCoreReturnEffect(final MechtitanCoreReturnEffect effect) {
161         super(effect);
162     }
163 
164     @Override
165     public MechtitanCoreReturnEffect copy() {
166         return new MechtitanCoreReturnEffect(this);
167     }
168 
169     @Override
170     public boolean apply(Game game, Ability source) {
171         Player player = game.getPlayer(source.getControllerId());
172         if (player == null) {
173             return false;
174         }
175         Cards cards = new CardsImpl(getTargetPointer().getTargets(game, source));
176         return player.moveCards(
177                 cards.getCards(game), Zone.BATTLEFIELD, source, game,
178                 true, false, true, null
179         );
180     }

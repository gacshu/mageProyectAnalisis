4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.Ability;
7   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
8   import mage.abilities.dynamicvalue.common.StaticValue;
9   import mage.abilities.effects.Effect;
10  import mage.abilities.effects.common.DamageTargetEffect;
11  import mage.abilities.effects.common.counter.AddCountersTargetEffect;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.counters.CounterType;
17  import mage.target.common.TargetControlledCreaturePermanent;
18  import mage.target.common.TargetOpponentOrPlaneswalker;
19  import mage.target.targetpointer.FirstTargetPointer;
20  import mage.target.targetpointer.SecondTargetPointer;
21  
22  /**
23   *
24   * @author stravant
25   */
26  public final class ManticoreOfTheGauntlet extends CardImpl {
27  
28      public ManticoreOfTheGauntlet(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{R}");
30  
31          this.subtype.add(SubType.MANTICORE);
32          this.power = new MageInt(5);
33          this.toughness = new MageInt(4);
34  
35          // When Manticore of the Gauntlet enters the battlefield, put a -1/-1 counter on target creature you control. Manticore of the Gauntlet deals 3 damage to target opponent.
36          Effect counters = new AddCountersTargetEffect(CounterType.M1M1.createInstance());
37          counters.setText("put a -1/-1 counter on target creature you control");
38          counters.setTargetPointer(new FirstTargetPointer());
39  
40          Effect damage = new DamageTargetEffect(StaticValue.get(3), true, "", true);
41          damage.setText("{this} deals 3 damage to target opponent or planeswalker.");
42          damage.setTargetPointer(new SecondTargetPointer());
43  
44          Ability ability = new EntersBattlefieldTriggeredAbility(counters);
45          ability.addEffect(damage);
46  
47          ability.addTarget(new TargetControlledCreaturePermanent());
48          ability.addTarget(new TargetOpponentOrPlaneswalker());
49  
50          this.addAbility(ability);
51      }
52  
53      private ManticoreOfTheGauntlet(final ManticoreOfTheGauntlet card) {
54          super(card);
55      }
56  
57      @Override
58      public ManticoreOfTheGauntlet copy() {
59          return new ManticoreOfTheGauntlet(this);
60      }
61  }

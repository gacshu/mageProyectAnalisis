41  public class MariTheKillingQuill extends CardImpl {
42  
43      private static final FilterControlledCreaturePermanent filter = new FilterControlledCreaturePermanent("Assassins, Mercenaries, and Rogues");
44      static {
45          filter.add(Predicates.or(
46                  SubType.ASSASSIN.getPredicate(),
47                  SubType.MERCENARY.getPredicate(),
48                  SubType.ROGUE.getPredicate()
49          ));
50      }
51  
52      public MariTheKillingQuill(UUID ownerId, CardSetInfo setInfo) {
53          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{B}{B}");
54  
55          this.addSuperType(SuperType.LEGENDARY);
56          this.addSubType(SubType.VAMPIRE, SubType.ASSASSIN);
57          this.power = new MageInt(3);
58          this.toughness = new MageInt(2);
59  
60          // Whenever a creature an opponent controls dies, exile it with a hit counter on it.
61          this.addAbility(new MariTheKillingQuillCreatureDiesAbility());
62  
63          // Assassins, Mercenaries, and Rogues you control have deathtouch and
64          //  "Whenever this creature deals combat damage to a player, you may remove a hit counter from a card that player owns in exile.
65          //   If you do, draw a card and create two Treasure tokens."
66          GainAbilityControlledEffect gainDeathTouchEffect = new GainAbilityControlledEffect(DeathtouchAbility.getInstance(), Duration.WhileOnBattlefield, filter);
67          Ability mainAbility = new SimpleStaticAbility(Zone.BATTLEFIELD, gainDeathTouchEffect);
68  
69          // NOTE: Optional part is handled inside the effect
70          Ability dealsDamageAbility = new DealsCombatDamageToAPlayerTriggeredAbility(new MariTheKillingQuillDealsDamageEffect(), false, true);
71          Effect drawAndTreasureEffect = new GainAbilityControlledEffect(dealsDamageAbility, Duration.WhileOnBattlefield, filter);
72          drawAndTreasureEffect.setText(
73                  "\"Whenever this creature deals combat damage to a player, you may remove a hit counter from a card that player owns in exile. " +
74                  "If you do, draw a card and create two Treasure tokens.\"");
75          drawAndTreasureEffect.concatBy("and");
76  
77          mainAbility.addEffect(drawAndTreasureEffect);
78  
79          this.addAbility(mainAbility);
80      }
81  
82      private MariTheKillingQuill(final MariTheKillingQuill card) {
83          super(card);
84      }
85  
86      @Override
87      public MariTheKillingQuill copy() {
88          return new MariTheKillingQuill(this);
89      }
90  }

92  class MariTheKillingQuillDealsDamageEffect extends OneShotEffect {
93  
94      MariTheKillingQuillDealsDamageEffect() {
95          super(Outcome.Benefit);
96      }
97  
98      private MariTheKillingQuillDealsDamageEffect(final MariTheKillingQuillDealsDamageEffect effect) {
99          super(effect);
100     }
101 
102     @Override
103     public boolean apply(Game game, Ability source) {
104         Player controller = game.getPlayer(source.getControllerId());
105         Player opponent = game.getPlayer(getTargetPointer().getFirst(game, source));
106         if (controller == null || opponent == null) {
107             return false;
108         }
109 
110         FilterCard filterCard = new FilterCard("a card that player owns in exile");
111         filterCard.add(new OwnerIdPredicate(opponent.getId()));
112         filterCard.add(Predicates.not(FaceDownPredicate.instance));
113         filterCard.add(new CastFromZonePredicate(Zone.EXILED));
114 
115         Effect doIfCostPaidEffect =  new DoIfCostPaid(
116                 new MariTheKillingQuillDrawAndTokenEffect(),
117                 new RemoveCounterCost(new TargetCard(Zone.EXILED, filterCard))
118         );
119 
120         return doIfCostPaidEffect.apply(game, source);
121     }
122 
123     @Override
124     public MariTheKillingQuillDealsDamageEffect copy() {
125         return new MariTheKillingQuillDealsDamageEffect(this);
126     }
127 }

132         super(Outcome.Benefit);
133     }
134 
135     private MariTheKillingQuillDrawAndTokenEffect(final MariTheKillingQuillDrawAndTokenEffect effect) {
136         super(effect);
137     }
138 
139     @Override
140     public boolean apply(Game game, Ability source) {
141         Player controller = game.getPlayer(source.getControllerId());
142         if (controller == null) {
143             return false;
144         }
145 
146         Effect drawCardEffect = new DrawCardSourceControllerEffect(1);
147         Effect createTreasureEffect = new CreateTokenEffect(new TreasureToken(), 2);
148 
149         boolean success = drawCardEffect.apply(game, source);
150         success |= createTreasureEffect.apply(game, source);
151 
152         return success;
153     }
154

156     public MariTheKillingQuillDrawAndTokenEffect copy() {
157         return new MariTheKillingQuillDrawAndTokenEffect(this);
158     }

161 // Based on Patron of the Vein
162 class MariTheKillingQuillCreatureDiesAbility extends TriggeredAbilityImpl {
163 
164     public MariTheKillingQuillCreatureDiesAbility() {
165         super(Zone.BATTLEFIELD, new MariTheKillingQuillExileCreatureEffect(), false);
166     }
167 
168     public MariTheKillingQuillCreatureDiesAbility(final MariTheKillingQuillCreatureDiesAbility ability) {
169         super(ability);
170     }
171 
172     @Override
173     public MariTheKillingQuillCreatureDiesAbility copy() {
174         return new MariTheKillingQuillCreatureDiesAbility(this);
175     }
176 
177     @Override
178     public boolean checkEventType(GameEvent event, Game game) {
179         return event.getType() == GameEvent.EventType.ZONE_CHANGE;
180     }
181 
182     @Override
183     public boolean checkTrigger(GameEvent event, Game game) {
184         if (((ZoneChangeEvent) event).isDiesEvent()) {
185             if (game.getOpponents(this.controllerId).contains(event.getPlayerId())) {
186                 Card creature = game.getPermanentOrLKIBattlefield(event.getTargetId());
187                 if (creature != null
188                         && creature.isCreature(game)) {
189                     for (Effect effect : this.getEffects()) {
190                         effect.setTargetPointer(new FixedTarget(creature.getId(), game));
191                     }
192                     return true;
193                 }
194             }
195         }
196         return false;
197     }
198 
199     @Override
200     public String getRule() {
201         return "Whenever a creature an opponent controls dies, exile it with a hit counter on it.";
202     }
203 }

205 class MariTheKillingQuillExileCreatureEffect extends OneShotEffect {
206 
207     public MariTheKillingQuillExileCreatureEffect() {
208         super(Outcome.Benefit);
209         this.staticText = "exile it with a hit counter on it";
210     }
211 
212     public MariTheKillingQuillExileCreatureEffect(final MariTheKillingQuillExileCreatureEffect effect) {
213         super(effect);
214     }
215 
216     @Override
217     public MariTheKillingQuillExileCreatureEffect copy() {
218         return new MariTheKillingQuillExileCreatureEffect(this);
219     }
220 
221     @Override
222     public boolean apply(Game game, Ability source) {
223         Player controller = game.getPlayer(source.getControllerId());
224         if (controller == null) {
225             return false;
226         }
227         Card card = game.getCard(this.getTargetPointer().getFirst(game, source));
228 
229         if (card != null) {
230             return CardUtil.moveCardWithCounter(game, source, controller, card, Zone.EXILED, CounterType.HIT.createInstance());
231         }
232 
233         return false;
234 
235     }

25   *
26   * @author LevelX2
27   */
28  public final class MinamoSightbender extends CardImpl {
29  
30      private static final FilterPermanent filter = new FilterPermanent("creature with power X or less");
31  
32      static {
33          filter.add(CardType.CREATURE.getPredicate());
34      }
35  
36      public MinamoSightbender(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{U}");
38          this.subtype.add(SubType.HUMAN);
39          this.subtype.add(SubType.WIZARD);
40  
41          this.power = new MageInt(1);
42          this.toughness = new MageInt(2);
43  
44          // {X}, {T}: Target creature with power X or less can't be blocked this turn.
45          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new CantBeBlockedTargetEffect(), new ManaCostsImpl<>("{X}"));
46          Target target = new TargetPermanent(filter);
47          ability.setTargetAdjuster(MinamoSightbenderAdjuster.instance);
48          ability.addTarget(target);
49          ability.addCost(new TapSourceCost());
50          this.addAbility(ability);
51  
52      }
53  
54      private MinamoSightbender(final MinamoSightbender card) {
55          super(card);
56      }
57  
58      @Override
59      public MinamoSightbender copy() {
60          return new MinamoSightbender(this);
61      }
62  }
63  
64  enum MinamoSightbenderAdjuster implements TargetAdjuster {
65      instance;
66  
67      @Override
68      public void adjustTargets(Ability ability, Game game) {
69          int xValue = ability.getManaCostsToPay().getX();
70          FilterPermanent permanentFilter = new FilterCreaturePermanent("creature with power " + xValue + " or less");
71          permanentFilter.add(new PowerPredicate(ComparisonType.FEWER_THAN, xValue + 1));
72          ability.getTargets().clear();
73          ability.getTargets().add(new TargetPermanent(permanentFilter));
74      }
75  }

27  public final class MartialImpetus extends CardImpl {
28  
29      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent();
30  
31      static {
32          filter.add(MartialImpetusPredicate.instance);
33      }
34  
35      public MartialImpetus(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
37  
38          this.subtype.add(SubType.AURA);
39  
40          // Enchant creature
41          TargetPermanent auraTarget = new TargetCreaturePermanent();
42          this.getSpellAbility().addTarget(auraTarget);
43          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
44          Ability ability = new EnchantAbility(auraTarget);
45          this.addAbility(ability);
46  
47          // Enchanted creature gets +1/+1 and is goaded.
48          ability = new SimpleStaticAbility(new BoostEnchantedEffect(1, 1));
49          ability.addEffect(new GoadAttachedEffect());
50          this.addAbility(ability);
51  
52          // Whenever enchanted creature attacks, each other creature that's attacking one of your opponents gets +1/+1 until end of turn.
53          this.addAbility(new AttacksAttachedTriggeredAbility(
54                  new BoostAllEffect(1, 1, Duration.EndOfTurn, filter, true).setText(
55                          "each other creature that's attacking one of your opponents gets +1/+1 until end of turn"
56                  ), AttachmentType.AURA, false)
57          );
58      }
59  
60      private MartialImpetus(final MartialImpetus card) {
61          super(card);
62      }
63  
64      @Override
65      public MartialImpetus copy() {
66          return new MartialImpetus(this);
67      }
68  }
69  
70  enum MartialImpetusPredicate implements ObjectSourcePlayerPredicate<Permanent> {
71      instance;
72  
73      @Override
74      public boolean apply(ObjectSourcePlayer<Permanent> input, Game game) {
75          Permanent martialImpetus = input.getSource().getSourcePermanentOrLKI(game);
76          if (martialImpetus != null) {
77              Permanent attachedTo = game.getPermanentOrLKIBattlefield(martialImpetus.getAttachedTo());
78              UUID auraControllerId = martialImpetus.getControllerId();
79              if (attachedTo != null // check the creature that the aura is attached to, not the aura itself
80                      && input.getObject() != null // creature being checked for predicate
81                      && input.getObject() != attachedTo // must be other creature
82                      && input.getObject().isAttacking() // attacking
83                      && game.getOpponents(auraControllerId) // check for opponents of aura's controller
84                      .contains(game.getCombat().getDefendingPlayerId(input.getObject().getId(), game))) {
85                  return true;
86              }
87          }
88          return false;
89      }

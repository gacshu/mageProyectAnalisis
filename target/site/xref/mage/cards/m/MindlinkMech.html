34      public MindlinkMech(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}{U}");
36  
37          this.subtype.add(SubType.VEHICLE);
38          this.power = new MageInt(4);
39          this.toughness = new MageInt(3);

45          this.addAbility(new MindlinkMechTriggeredAbility());

47          // Crew 1
48          this.addAbility(new CrewAbility(1));
49      }

52          super(card);
53      }

61  class MindlinkMechTriggeredAbility extends TriggeredAbilityImpl {
62  
63      MindlinkMechTriggeredAbility() {
64          super(Zone.BATTLEFIELD, new MindlinkMechEffect());
65          this.addWatcher(new MindlinkMechWatcher());
66      }
67  
68      private MindlinkMechTriggeredAbility(final MindlinkMechTriggeredAbility ability) {
69          super(ability);
70      }
71  
72      @Override
73      public MindlinkMechTriggeredAbility copy() {
74          return new MindlinkMechTriggeredAbility(this);
75      }
76  
77      @Override
78      public boolean checkEventType(GameEvent event, Game game) {
79          return event.getType() == GameEvent.EventType.VEHICLE_CREWED;
80      }
81  
82      @Override
83      public boolean checkTrigger(GameEvent event, Game game) {
84          if (!event.getSourceId().equals(getSourceId()) || !MindlinkMechWatcher.checkVehicle(this, game)) {
85              return false;
86          }
87          this.getTargets().clear();
88          this.addTarget(new TargetPermanent(MindlinkMechWatcher.makeFilter(this, game)));
89          return true;
90      }
91  
92      @Override
93      public String getRule() {
94          return "Whenever {this} becomes crewed for the first time each turn, until end of turn, " +
95                  "{this} becomes a copy of target nonlegendary creature that crewed it this turn, " +
96                  "except it's 4/3, it's a Vehicle artifact in addition to its other types, and it has flying.";
97      }
98  }

100 class MindlinkMechWatcher extends Watcher {
101 
102     private final Map<MageObjectReference, Integer> crewCount = new HashMap<>();
103     private final Map<MageObjectReference, Set<MageObjectReference>> crewMap = new HashMap<>();
104     private static final FilterPermanent invalidFilter = new FilterPermanent();
105 
106     static {
107         invalidFilter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, -2));
108     }
109 
110     MindlinkMechWatcher() {
111         super(WatcherScope.GAME);
112     }
113 
114     @Override
115     public void watch(GameEvent event, Game game) {
116         Permanent vehicle;
117         Permanent crewer;
118         switch (event.getType()) {
119             case VEHICLE_CREWED:
120                 vehicle = game.getPermanent(event.getTargetId());
121                 crewer = null;
122                 break;
123             case CREWED_VEHICLE:
124                 vehicle = game.getPermanent(event.getSourceId());
125                 crewer = game.getPermanent(event.getTargetId());
126                 break;
127             default:
128                 return;
129         }
130         if (vehicle == null) {
131             return;
132         }
133         if (crewer == null) {
134             crewCount.compute(new MageObjectReference(vehicle, game), (m, i) -> i == null ? 1 : Integer.sum(i, 1));
135             return;
136         }
137         crewMap.computeIfAbsent(new MageObjectReference(vehicle, game), x -> new HashSet<>()).add(new MageObjectReference(crewer, game));
138     }
139 
140     @Override
141     public void reset() {
142         super.reset();
143         crewCount.clear();
144         crewMap.clear();
145     }
146 
147     public static boolean checkVehicle(Ability source, Game game) {
148         return game
149                 .getState()
150                 .getWatcher(MindlinkMechWatcher.class)
151                 .crewCount
152                 .getOrDefault(new MageObjectReference(source), 0) < 2;
153     }
154 
155     public static FilterPermanent makeFilter(Ability source, Game game) {
156         Set<MageObjectReferencePredicate> predicates = game
157                 .getState()
158                 .getWatcher(MindlinkMechWatcher.class)
159                 .crewMap
160                 .computeIfAbsent(new MageObjectReference(source), x -> new HashSet<>())
161                 .stream()
162                 .filter(mor -> {
163                     Permanent permanent = mor.getPermanent(game);
164                     return permanent != null && !permanent.isLegendary() && permanent.isCreature(game);
165                 }).map(MageObjectReferencePredicate::new)
166                 .collect(Collectors.toSet());
167         if (predicates.isEmpty()) {
168             return invalidFilter;
169         }
170         FilterPermanent filterPermanent = new FilterPermanent(
171                 "nonlegendary creature that crewed " + CardUtil.getSourceName(game, source) + " this turn"
172         );
173         filterPermanent.add(Predicates.or(predicates));
174         return filterPermanent;
175     }
176 }

178 class MindlinkMechEffect extends OneShotEffect {
179 
180     MindlinkMechEffect() {
181         super(Outcome.Benefit);
182     }
183 
184     private MindlinkMechEffect(final MindlinkMechEffect effect) {
185         super(effect);
186     }
187 
188     @Override
189     public MindlinkMechEffect copy() {
190         return new MindlinkMechEffect(this);
191     }
192 
193     @Override
194     public boolean apply(Game game, Ability source) {
195         Permanent permanent = source.getSourcePermanentIfItStillExists(game);
196         Permanent creature = game.getPermanent(getTargetPointer().getFirst(game, source));
197         if (permanent == null || creature == null) {
198             return false;
199         }
200         game.copyPermanent(Duration.EndOfTurn, creature, permanent.getId(), source, new MindlinkMechApplier());
201         return true;
202     }
203 }

205 class MindlinkMechApplier extends CopyApplier {
206     @Override
207     public boolean apply(Game game, MageObject blueprint, Ability source, UUID targetObjectId) {
208         blueprint.getPower().setModifiedBaseValue(4);
209         blueprint.getToughness().setModifiedBaseValue(3);
210         blueprint.addCardType(game, CardType.ARTIFACT);
211         blueprint.addSubType(game, SubType.VEHICLE);
212         blueprint.getAbilities().add(FlyingAbility.getInstance());
213         return true;
214     }

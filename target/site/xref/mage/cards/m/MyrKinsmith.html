6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.constants.SubType;
10  import mage.filter.FilterCard;
11  import mage.target.common.TargetCardInLibrary;
12  
13  import java.util.UUID;
14  
15  /**
16   * @author TheElk801
17   */
18  public final class MyrKinsmith extends CardImpl {
19  
20      private static final FilterCard filter = new FilterCard("Myr card");
21  
22      static {
23          filter.add(SubType.MYR.getPredicate());
24      }
25  
26      public MyrKinsmith(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{4}");
28  
29          this.subtype.add(SubType.MYR);
30          this.power = new MageInt(3);
31          this.toughness = new MageInt(1);
32  
33          // When Myr Kinsmith enters the battlefield, you may search your library for a Myr card, reveal it, put it into your hand, then shuffle.
34          this.addAbility(new EntersBattlefieldTriggeredAbility(new SearchLibraryPutInHandEffect(
35                  new TargetCardInLibrary(filter), true, true
36          ), true));
37      }
38  
39      private MyrKinsmith(final MyrKinsmith card) {
40          super(card);
41      }
42  
43      @Override
44      public MyrKinsmith copy() {
45          return new MyrKinsmith(this);
46      }
47  }

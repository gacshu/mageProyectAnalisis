33  public final class ManorGargoyle extends CardImpl {
34  
35      private static final String rule = "{this} is indestructible as long as it has defender.";
36  
37      public ManorGargoyle(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT,CardType.CREATURE},"{5}");
39          this.subtype.add(SubType.GARGOYLE);
40  
41          this.power = new MageInt(4);
42          this.toughness = new MageInt(4);
43  
44          this.addAbility(DefenderAbility.getInstance());
45  
46          /*
47         613.7. Within a layer or sublayer, determining which order effects are applied in is sometimes done using a dependency system.
48                If a dependency exists, it will override the timestamp system.
49          613.7a An effect is said to “depend on” another if
50                  (a) it's applied in the same layer (and, if applicable, sublayer) as the other effect (see rules 613.1 and 613.3);
51                  (b) applying the other would change the text or the existence of the first effect, what it applies to, or what
52                      it does to any of the things it applies to; and
53                  (c) neither effect is from a characteristic-defining ability or both effects are from characteristic-defining
54                      abilities. Otherwise, the effect is considered to be independent of the other effect.
55          613.7b  An effect dependent on one or more other effects waits to apply until just after all of those effects have been applied.
56                  If multiple dependent effects would apply simultaneously in this way, they're applied in timestamp order relative to each
57                  other. If several dependent effects form a dependency loop, then this rule is ignored and the effects in the dependency
58                  loop are applied in timestamp order.
59          613.7c  After each effect is applied, the order of remaining effects is reevaluated and may change if an effect that has not yet
60                  been applied becomes dependent on or independent of one or more other effects that have not yet been applied.
61           */
62          // Manor Gargoyle has indestructible as long as it has defender.
63          GainAbilitySourceEffect gainEffect = new GainAbilitySourceEffect(IndestructibleAbility.getInstance());
64          gainEffect.setDependedToType(DependencyType.LooseDefenderEffect);
65          ConditionalContinuousEffect effect = new ConditionalContinuousEffect(gainEffect, HasDefenderCondition.instance, rule);
66  
67          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
68  
69          // {1}: Until end of turn, Manor Gargoyle loses defender and gains flying.
70          ContinuousEffect effect2 = new LoseAbilitySourceEffect(DefenderAbility.getInstance(), Duration.EndOfTurn);
71          effect2.addDependencyType(DependencyType.LooseDefenderEffect);
72          effect2.setText("Until end of turn, {this} loses defender");
73          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, effect2, new ManaCostsImpl<>("{1}"));
74          effect2 = new GainAbilitySourceEffect(FlyingAbility.getInstance(), Duration.EndOfTurn);
75          effect2.setText("and gains flying");
76          ability.addEffect(effect2);
77          this.addAbility(ability);
78      }
79  
80      private ManorGargoyle(final ManorGargoyle card) {
81          super(card);
82      }
83  
84      @Override
85      public ManorGargoyle copy() {
86          return new ManorGargoyle(this);
87      }
88  }
89  
90  enum HasDefenderCondition implements Condition {
91  
92      instance;
93  
94      @Override
95      public boolean apply(Game game, Ability source) {
96          Permanent permanent = game.getPermanent(source.getSourceId());
97          if (permanent != null) {
98              Abilities<Ability> abilities = permanent.getAbilities();
99              for (Ability ability : abilities) {
100                 if (ability.getClass().equals(DefenderAbility.getInstance().getClass())) {
101                     return true;
102                 }
103             }
104         }
105         return false;
106     }
107 }

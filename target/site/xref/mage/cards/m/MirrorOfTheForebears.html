26  /**
27   * @author TheElk801
28   */
29  public final class MirrorOfTheForebears extends CardImpl {
30  
31      private static final FilterPermanent filter = new FilterControlledCreaturePermanent();
32  
33      static {
34          filter.add(ChosenSubtypePredicate.TRUE);
35      }
36  
37      public MirrorOfTheForebears(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
39  
40          // As Mirror of the Forebears enters the battlefield, choose a creature type.
41          this.addAbility(new AsEntersBattlefieldAbility(new ChooseCreatureTypeEffect(Outcome.Copy)));
42  
43          // 1: Until end of turn, Mirror of the Forebears becomes a copy of target creature you control of the chosen type, except it's an artifact in addition to its other types.
44          Ability ability = new SimpleActivatedAbility(new MirrorOfTheForebearsCopyEffect(), new GenericManaCost(1));
45          ability.addTarget(new TargetPermanent(filter));
46          this.addAbility(ability);
47      }
48  
49      private MirrorOfTheForebears(final MirrorOfTheForebears card) {
50          super(card);
51      }
52  
53      @Override
54      public MirrorOfTheForebears copy() {
55          return new MirrorOfTheForebears(this);
56      }
57  }
58  
59  class MirrorOfTheForebearsCopyEffect extends OneShotEffect {
60  
61      MirrorOfTheForebearsCopyEffect() {
62          super(Outcome.Copy);
63          this.staticText = "until end of turn, {this} becomes a copy of target creature you control of the chosen type, except it's an artifact in addition to its other types";
64      }
65  
66      private MirrorOfTheForebearsCopyEffect(final MirrorOfTheForebearsCopyEffect effect) {
67          super(effect);
68      }
69  
70      @Override
71      public MirrorOfTheForebearsCopyEffect copy() {
72          return new MirrorOfTheForebearsCopyEffect(this);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          Permanent sourcePermanent = game.getPermanent(source.getSourceId());
78          Permanent copyFromPermanent = game.getPermanent(getTargetPointer().getFirst(game, source));
79          if (sourcePermanent != null && copyFromPermanent != null) {
80              game.copyPermanent(Duration.EndOfTurn, copyFromPermanent, sourcePermanent.getId(), source, new EmptyCopyApplier());
81              game.addEffect(new AddCardTypeSourceEffect(Duration.EndOfTurn, CardType.ARTIFACT), source);
82              return true;
83          }
84          return false;
85      }
86  }

22  import java.util.UUID;

24  /**
25   * @author jeffwadsworth
26   */
27  public final class MarshalsAnthem extends CardImpl {
28  
29      private static final String rule = "return up to X target creature cards from your graveyard to the battlefield, " +
30              "where X is the number of times {this} was kicked";
31  
32      public MarshalsAnthem(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}{W}");
34  
35          // Multikicker {1}{W}
36          this.addAbility(new MultikickerAbility("{1}{W}"));
37  
38          // Creatures you control get +1/+1.
39          this.addAbility(new SimpleStaticAbility(
40                  Zone.BATTLEFIELD, new BoostControlledEffect(1, 1, Duration.WhileOnBattlefield)
41          ));
42  
43          // When Marshal's Anthem enters the battlefield, return up to X target creature cards from your graveyard to the battlefield, where X is the number of times Marshal's Anthem was kicked.
44          Ability ability = new EntersBattlefieldTriggeredAbility(
45                  new ReturnFromGraveyardToBattlefieldTargetEffect().setText(rule), false
46          );
47          ability.setTargetAdjuster(MarshalsAnthemAdjuster.instance);
48          this.addAbility(ability);
49      }
50  
51      private MarshalsAnthem(final MarshalsAnthem card) {
52          super(card);
53      }
54  
55      @Override
56      public MarshalsAnthem copy() {
57          return new MarshalsAnthem(this);
58      }
59  }
60  
61  enum MarshalsAnthemAdjuster implements TargetAdjuster {
62      instance;
63      private static final FilterCard filter = new FilterCreatureCard("creature card in your graveyard");
64  
65      @Override
66      public void adjustTargets(Ability ability, Game game) {
67          ability.getTargets().clear();
68          int numbTargets = MultikickerCount.instance.calculate(game, ability, null);
69          if (numbTargets > 0) {
70              ability.addTarget(new TargetCardInYourGraveyard(0, numbTargets, filter));
71          }
72      }
73  }

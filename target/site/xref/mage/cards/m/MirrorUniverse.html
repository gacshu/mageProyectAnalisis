4   import java.util.UUID;
5   import mage.abilities.Ability;
6   import mage.abilities.condition.common.IsStepCondition;
7   import mage.abilities.costs.common.SacrificeSourceCost;
8   import mage.abilities.costs.common.TapSourceCost;
9   import mage.abilities.decorator.ConditionalActivatedAbility;
10  import mage.abilities.effects.common.ExchangeLifeControllerTargetEffect;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.PhaseStep;
15  import mage.constants.Zone;
16  import mage.target.common.TargetOpponent;
17  
18  /**
19   *
20   * @author Styxo
21   */
22  public final class MirrorUniverse extends CardImpl {
23  
24      public MirrorUniverse(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{6}");
26  
27          // {tap}, Sacrifice Mirror Universe: Exchange life totals with target opponent. Activate this ability only during your upkeep.
28          Ability ability = new ConditionalActivatedAbility(
29                  Zone.BATTLEFIELD,
30                  new ExchangeLifeControllerTargetEffect(),
31                  new TapSourceCost(),
32                  new IsStepCondition(PhaseStep.UPKEEP),
33                  null);
34          ability.addCost(new SacrificeSourceCost());
35          ability.addTarget(new TargetOpponent());
36          this.addAbility(ability);
37      }
38  
39      private MirrorUniverse(final MirrorUniverse card) {
40          super(card);
41      }
42  
43      @Override
44      public MirrorUniverse copy() {
45          return new MirrorUniverse(this);
46      }
47  }

2   package mage.cards.m;
3   
4   import java.util.UUID;
5   import mage.abilities.costs.AlternativeCostSourceAbility;
6   import mage.abilities.costs.common.SacrificeTargetCost;
7   import mage.abilities.effects.common.CreateTokenEffect;
8   import mage.cards.CardImpl;
9   import mage.cards.CardSetInfo;
10  import mage.constants.CardType;
11  import mage.constants.SubType;
12  import mage.filter.common.FilterControlledPermanent;
13  import mage.game.permanent.token.GoblinToken;
14  import mage.target.common.TargetControlledPermanent;
15  
16  /**
17   *
18   * @author markedagain
19   */
20  public final class MoggAlarm extends CardImpl {
21  
22       private static final FilterControlledPermanent filter = new FilterControlledPermanent("Mountains");
23      static {
24          filter.add(SubType.MOUNTAIN.getPredicate());
25      }
26      
27      public MoggAlarm(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{1}{R}{R}");
29  
30          // You may sacrifice two Mountains rather than pay Mogg Alarm's mana cost.
31          this.addAbility(new AlternativeCostSourceAbility(new SacrificeTargetCost(new TargetControlledPermanent(2, 2, filter, true))));
32          // Create two 1/1 red Goblin creature tokens.
33          this.getSpellAbility().addEffect(new CreateTokenEffect(new GoblinToken(),2));
34          
35      }
36  
37      private MoggAlarm(final MoggAlarm card) {
38          super(card);
39      }
40  
41      @Override
42      public MoggAlarm copy() {
43          return new MoggAlarm(this);
44      }
45  }

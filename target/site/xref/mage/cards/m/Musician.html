26   * @author jeffwadsworth
27   */
28  public final class Musician extends CardImpl {
29  
30      public Musician(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{U}");
32  
33          this.subtype.add(SubType.HUMAN);
34          this.subtype.add(SubType.WIZARD);
35          this.power = new MageInt(1);
36          this.toughness = new MageInt(3);
37  
38          // Cumulative upkeep {1}
39          this.addAbility(new CumulativeUpkeepAbility(new ManaCostsImpl<>("{1}")));
40  
41          // {tap}: Put a music counter on target creature. If it doesn't have "At the beginning of your upkeep, destroy this creature unless you pay {1} for each music counter on it," it gains that ability.
42          Effect effect = new DoUnlessControllerPaysEffect(
43                  new DestroySourceEffect(),
44                  new DynamicValueGenericManaCost(
45                          new CountersSourceCount(CounterType.MUSIC),
46                          "{1} for each music counter on {this}"));
47          effect.setText("destroy this creature unless you pay {1} for each music counter on it");
48          Ability ability = new BeginningOfUpkeepTriggeredAbility(
49                  Zone.BATTLEFIELD,
50                  effect,
51                  TargetController.YOU,
52                  false,
53                  false,
54                  "At the beginning of your upkeep, ");
55          Effect effect2 = new AddCountersTargetEffect(CounterType.MUSIC.createInstance());
56          effect2.setText("Put a music counter on target creature");
57          Effect effect3 = new GainAbilityTargetEffect(
58                  ability,
59                  Duration.WhileOnBattlefield);
60          effect3.setText("If it doesn't have \"At the beginning of your upkeep, destroy this creature unless you pay {1} for each music counter on it,\" it gains that ability");
61          Ability ability2 = new SimpleActivatedAbility(
62                  Zone.BATTLEFIELD,
63                  effect2,
64                  new TapSourceCost());
65          ability2.addTarget(new TargetCreaturePermanent());
66          ability2.addEffect(effect3);
67          this.addAbility(ability2);
68  
69      }
70  
71      private Musician(final Musician card) {
72          super(card);
73      }
74  
75      @Override
76      public Musician copy() {
77          return new Musician(this);
78      }
79  }

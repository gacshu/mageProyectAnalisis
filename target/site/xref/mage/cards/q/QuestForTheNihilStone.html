26  public final class QuestForTheNihilStone extends CardImpl {
27  
28      public QuestForTheNihilStone(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{B}");
30  
31          // Whenever an opponent discards a card, you may put a quest counter on Quest for the Nihil Stone.
32          this.addAbility(new DiscardsACardOpponentTriggeredAbility(new AddCountersSourceEffect(CounterType.QUEST.createInstance(), true), true));
33  
34          // At the beginning of each opponent's upkeep, if that player has no cards in hand and Quest for the Nihil Stone has two or more quest counters on it, you may have that player lose 5 life.
35          this.addAbility(new QuestForTheNihilStoneTriggeredAbility());
36      }
37  
38      private QuestForTheNihilStone(final QuestForTheNihilStone card) {
39          super(card);
40      }
41  
42      @Override
43      public QuestForTheNihilStone copy() {
44          return new QuestForTheNihilStone(this);
45      }
46  }

48  class QuestForTheNihilStoneTriggeredAbility extends TriggeredAbilityImpl {
49  
50      public QuestForTheNihilStoneTriggeredAbility() {
51          super(Zone.BATTLEFIELD, new LoseLifeTargetEffect(5), true);
52      }
53  
54      public QuestForTheNihilStoneTriggeredAbility(final QuestForTheNihilStoneTriggeredAbility ability) {
55          super(ability);
56      }
57  
58      @Override
59      public QuestForTheNihilStoneTriggeredAbility copy() {
60          return new QuestForTheNihilStoneTriggeredAbility(this);
61      }
62  
63      @Override
64      public boolean checkEventType(GameEvent event, Game game) {
65          return event.getType() == GameEvent.EventType.UPKEEP_STEP_PRE;
66      }
67  
68      @Override
69      public boolean checkTrigger(GameEvent event, Game game) {
70          Permanent quest = game.getPermanent(super.getSourceId());
71          if (quest == null) {
72              Permanent questLKI = (Permanent) game.getLastKnownInformation(super.getSourceId(), Zone.BATTLEFIELD);
73              quest = questLKI;
74          }
75          if (game.getOpponents(controllerId).contains(event.getPlayerId())) {
76              Player opponent = game.getPlayer(event.getPlayerId());
77              if (opponent != null
78                      && opponent.getHand().isEmpty()
79                      && quest.getCounters(game).getCount(CounterType.QUEST) >= 2) {
80                  for (Effect effect : this.getEffects()) {
81                      effect.setTargetPointer(new FixedTarget(opponent.getId()));
82                      return true;
83                  }
84              }
85          }
86          return false;
87      }
88  
89      @Override
90      public String getRule() {
91          return "At the beginning of each opponent's upkeep, if that player has no cards in hand and {this} has two or more quest counters on it, you may have that player lose 5 life.";
92      }

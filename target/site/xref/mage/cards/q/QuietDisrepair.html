20  import java.util.UUID;

22  /**
23   *
24   * @author LevelX2
25   */
26  public final class QuietDisrepair extends CardImpl {
27  
28      public QuietDisrepair(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{G}");
30          this.subtype.add(SubType.AURA);
31  
32  
33          // Enchant artifact or enchantment
34          TargetPermanent auraTarget = new TargetPermanent(StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_ENCHANTMENT);
35          this.getSpellAbility().addTarget(auraTarget);
36          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
37          Ability ability = new EnchantAbility(auraTarget);
38          this.addAbility(ability);
39  
40          // At the beginning of your upkeep, choose one - Destroy enchanted permanent; or you gain 2 life.
41          ability = new BeginningOfUpkeepTriggeredAbility(new DestroyAttachedToEffect("enchanted permanent"), TargetController.YOU, false);
42          Mode mode = new Mode(new GainLifeEffect(2));
43          ability.addMode(mode);
44          this.addAbility(ability);
45      }
46  
47      private QuietDisrepair(final QuietDisrepair card) {
48          super(card);
49      }
50  
51      @Override
52      public QuietDisrepair copy() {
53          return new QuietDisrepair(this);
54      }
55  }

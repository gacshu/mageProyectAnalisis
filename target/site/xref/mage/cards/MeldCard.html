17  public abstract class MeldCard extends CardImpl {

25      public MeldCard(UUID ownerId, CardSetInfo setInfo, CardType[] cardTypes, String costs) {
26          super(ownerId, setInfo, cardTypes, costs);
27          halves = new CardsImpl();
28          this.nightCard = true;
29      }

31      public MeldCard(final MeldCard card) {
32          super(card);
33          this.topHalfCard = card.topHalfCard;
34          this.bottomHalfCard = card.bottomHalfCard;
35          this.topLastZoneChangeCounter = card.topLastZoneChangeCounter;
36          this.bottomLastZoneChangeCounter = card.bottomLastZoneChangeCounter;
37          this.halves = new CardsImpl(card.halves);
38      }

100     @Override
101     public int getManaValue() {
102         if (this.isCopy()) {
103             return 0;
104         } else {
105             return (this.topHalfCard != null ? this.topHalfCard.getManaValue() : 0)
106                     + (this.bottomHalfCard != null ? this.bottomHalfCard.getManaValue() : 0);
107         }
108     }

110     @Override
111     public boolean isTransformable() {
112         // there are multiple day cards for one meld card, so can't show it as second side
113         // TODO: can be fixed after mutiple sides implement, e.g. with Mutate support
114         return false;
115     }
116 
117     @Override
118     public boolean addCounters(Counter counter, UUID playerAddingCounters, Ability source, Game game, List<UUID> appliedEffects) {
119         if (this.isMelded(game)) {
120             return super.addCounters(counter, playerAddingCounters, source, game, appliedEffects);
121         } else {
122             // can this really happen?
123             boolean returnState = true;
124             if (hasTopHalf(game)) {
125                 returnState |= topHalfCard.addCounters(counter, playerAddingCounters, source, game, appliedEffects);
126             }
127             if (hasBottomHalf(game)) {
128                 returnState |= bottomHalfCard.addCounters(counter, playerAddingCounters, source, game, appliedEffects);
129             }
130             return returnState;
131         }
132     }
133 
134     public boolean hasTopHalf(Game game) {
135         boolean value = topLastZoneChangeCounter == topHalfCard.getZoneChangeCounter(game)
136                 && halves.contains(topHalfCard.getId());
137         if (!value) {
138             halves.remove(topHalfCard);
139         }
140         return value;
141     }
142 
143     public boolean hasBottomHalf(Game game) {
144         boolean value = bottomLastZoneChangeCounter == bottomHalfCard.getZoneChangeCounter(game)
145                 && halves.contains(bottomHalfCard.getId());
146         if (!value) {
147             halves.remove(bottomHalfCard);
148         }
149         return value;
150     }

170     @Override
171     public boolean removeFromZone(Game game, Zone fromZone, Ability source) {
172         if (isCopy()) {
173             return super.removeFromZone(game, fromZone, source);
174         }
175         if (isMelded(game) && fromZone == Zone.BATTLEFIELD) {
176             Permanent permanent = game.getPermanent(objectId);
177             return permanent != null && permanent.removeFromZone(game, fromZone, source);
178         }
179         boolean topRemoved = hasTopHalf(game) && topHalfCard.removeFromZone(game, fromZone, source);
180         if (!topRemoved) {
181             // The top half isn't being moved with the pair anymore.
182             halves.remove(topHalfCard);
183         }
184         boolean bottomRemoved = hasBottomHalf(game) && bottomHalfCard.removeFromZone(game, fromZone, source);
185         if (!bottomRemoved) {
186             // The bottom half isn't being moved with the pair anymore.
187             halves.remove(bottomHalfCard);
188         }
189         return topRemoved || bottomRemoved;
190     }
191 
192     @Override
193     public void updateZoneChangeCounter(Game game, ZoneChangeEvent event) {
194         if (isCopy() || !isMelded(game)) {
195             super.updateZoneChangeCounter(game, event);
196             return;
197         }
198         super.updateZoneChangeCounter(game, event);
199         if (topLastZoneChangeCounter == topHalfCard.getZoneChangeCounter(game)
200                 && halves.contains(topHalfCard.getId())) {
201             topHalfCard.updateZoneChangeCounter(game, event);
202             topLastZoneChangeCounter = topHalfCard.getZoneChangeCounter(game);
203         }
204         if (bottomLastZoneChangeCounter == bottomHalfCard.getZoneChangeCounter(game)
205                 && halves.contains(bottomHalfCard.getId())) {
206             bottomHalfCard.updateZoneChangeCounter(game, event);
207             bottomLastZoneChangeCounter = bottomHalfCard.getZoneChangeCounter(game);
208         }
209     }
210 
211     public Cards getHalves() {
212         return halves;
213     }

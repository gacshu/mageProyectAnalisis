30      public CodieVociferousCodex(UUID ownerId, CardSetInfo setInfo) {

35          this.power = new MageInt(1);
36          this.toughness = new MageInt(4);

38          // You can't cast permanent spells.

Until end of turn, you may cast that card without paying its mana cost. Put each other card exiled this way on the bottom of your library in a random order.
42          Ability ability = new SimpleManaAbility(Zone.BATTLEFIELD, new Mana(
43                  1, 1, 1, 1, 1, 0, 0, 0
44          ), new GenericManaCost(4));
45          ability.addEffect(new CreateDelayedTriggeredAbilityEffect(new CodieVociferousCodexDelayedTriggeredAbility()));
46          ability.addCost(new TapSourceCost());
47          this.addAbility(ability);
48      }

63          super(Duration.WhileOnBattlefield, Outcome.Detriment);
64          staticText = "you can't cast permanent spells";
65      }
66  
67      private CodieVociferousCodexCantCastEffect(final CodieVociferousCodexCantCastEffect effect) {
68          super(effect);
69      }
70

73          return new CodieVociferousCodexCantCastEffect(this);
74      }
75  
76      @Override
77      public boolean checksEventType(GameEvent event, Game game) {
78          return event.getType() == GameEvent.EventType.CAST_SPELL;
79      }
80  
81      @Override
82      public boolean applies(GameEvent event, Ability source, Game game) {
83          if (!source.isControlledBy(event.getPlayerId())) {
84              return false;
85          }
86          Card card = game.getCard(event.getSourceId());
87          return card != null && card.isPermanent(game);
88      }

92  
93      CodieVociferousCodexDelayedTriggeredAbility() {
94          super(new CodieVociferousCodexEffect(), Duration.EndOfTurn, true, false);
95      }
96  
97      private CodieVociferousCodexDelayedTriggeredAbility(final CodieVociferousCodexDelayedTriggeredAbility ability) {
98          super(ability);
99      }
100 
101     @Override
102     public boolean checkEventType(GameEvent event, Game game) {
103         return event.getType() == GameEvent.EventType.SPELL_CAST;
104     }
105 
106     @Override
107     public boolean checkTrigger(GameEvent event, Game game) {
108         if (!isControlledBy(event.getPlayerId())) {
109             return false;
110         }
111         Spell spell = game.getStack().getSpell(event.getTargetId());
112         if (spell == null) {
113             return false;
114         }
115         this.getEffects().setValue("spellCast", spell);
116         return true;
117     }
118 
119     @Override
120     public CodieVociferousCodexDelayedTriggeredAbility copy() {
121         return new CodieVociferousCodexDelayedTriggeredAbility(this);
122     }
123 
124     @Override
125     public String getRule() {
126         return "When you cast your next spell this turn, exile cards from the top of your library "
127                 + "until you exile an instant or sorcery card with lesser mana value. Until end of turn, "
128                 + "you may cast that card without paying its mana cost. Put each other card exiled this way "
129                 + "on the bottom of your library in a random order.";
130     }

133 class CodieVociferousCodexEffect extends OneShotEffect {
134 
135     CodieVociferousCodexEffect() {
136         super(Outcome.Benefit);
137     }
138 
139     private CodieVociferousCodexEffect(final CodieVociferousCodexEffect effect) {
140         super(effect);
141     }
142 
143     @Override
144     public CodieVociferousCodexEffect copy() {
145         return new CodieVociferousCodexEffect(this);
146     }
147 
148     @Override
149     public boolean apply(Game game, Ability source) {
150         Player controller = game.getPlayer(source.getControllerId());
151         Spell spell = (Spell) getValue("spellCast");
152         if (controller == null
153                 || spell == null) {
154             return false;
155         }
156         Cards toExile = new CardsImpl();
157         Card toCast = null;
158         for (Card card : controller.getLibrary().getCards(game)) {
159             toExile.add(card);
160             if (card.isInstantOrSorcery(game)
161                     && card.getManaValue() < spell.getManaValue()) {
162                 toCast = card;
163                 break;
164             }
165         }
166         if (toCast == null) {
167             controller.moveCards(toExile, Zone.EXILED, source, game);
168             controller.putCardsOnBottomOfLibrary(toExile, game, source, false);
169             return true;
170         }
171         controller.moveCards(toCast, Zone.EXILED, source, game);
172         PlayFromNotOwnHandZoneTargetEffect effect = new PlayFromNotOwnHandZoneTargetEffect(Zone.EXILED, TargetController.YOU, Duration.EndOfTurn, true);
173         effect.setTargetPointer(new FixedTarget(toCast.getId(), game));
174         game.addEffect(effect, source);
175         toExile.remove(toCast);
176         controller.putCardsOnBottomOfLibrary(toExile, game, source, false);
177         return true;
178     }

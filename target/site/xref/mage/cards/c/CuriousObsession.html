34  public final class CuriousObsession extends CardImpl {
35  
36      public CuriousObsession(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{U}");
38  
39          this.subtype.add(SubType.AURA);
40  
41          // Enchant creature
42          TargetPermanent auraTarget = new TargetCreaturePermanent();
43          this.getSpellAbility().addTarget(auraTarget);
44          this.getSpellAbility().addEffect(new AttachEffect(Outcome.BoostCreature));
45          Ability ability = new EnchantAbility(auraTarget);
46          this.addAbility(ability);
47  
48          // Enchanted creature gets +1/+1 and has "Whenever this creature deals combat damage to a player, you may draw a card.
49          ability = new SimpleStaticAbility(new BoostEnchantedEffect(1, 1, Duration.WhileOnBattlefield));
50          Ability gainedAbility = new DealsCombatDamageToAPlayerTriggeredAbility(new DrawCardSourceControllerEffect(1), true);
51          Effect effect = new GainAbilityAttachedEffect(gainedAbility, AttachmentType.AURA);
52          effect.setText("and has \"Whenever this creature deals combat damage to a player, you may draw a card.\"");
53          ability.addEffect(effect);
54          this.addAbility(ability);
55  
56          // At the beginning of your end step, if you didn't attack with a creature this turn sacrifice Curious Obsession.
57          ability = new ConditionalInterveningIfTriggeredAbility(
58                  new BeginningOfYourEndStepTriggeredAbility(new SacrificeSourceEffect(), false),
59                  new InvertCondition(RaidCondition.instance),
60                  "At the beginning of your end step, if you didn't attack with a creature this turn, sacrifice {this}."
61          );
62          ability.addHint(RaidHint.instance);
63          this.addAbility(ability, new PlayerAttackedWatcher());
64      }
65  
66      private CuriousObsession(final CuriousObsession card) {
67          super(card);
68      }
69  
70      @Override
71      public CuriousObsession copy() {
72          return new CuriousObsession(this);
73      }
74  }

4   import java.util.UUID;
5   import mage.Mana;
6   import mage.abilities.Ability;
7   import mage.abilities.common.EntersBattlefieldAllTriggeredAbility;
8   import mage.abilities.effects.Effect;
9   import mage.abilities.effects.mana.AddManaToManaPoolSourceControllerEffect;
10  import mage.abilities.effects.common.LoseLifeSourceControllerEffect;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.SetTargetPointer;
15  import mage.constants.Zone;
16  import mage.filter.StaticFilters;
17  
18  /**
19   *
20   * @author Galatolol
21   */
22  public final class CarnivalOfSouls extends CardImpl {
23  
24      public CarnivalOfSouls(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{B}");
26  
27          // Whenever a creature enters the battlefield, you lose 1 life and add {B}.
28          Ability ability = new EntersBattlefieldAllTriggeredAbility(Zone.BATTLEFIELD, new LoseLifeSourceControllerEffect(1),
29                  StaticFilters.FILTER_PERMANENT_A_CREATURE, false, SetTargetPointer.PERMANENT, null, false);
30          Effect effect = new AddManaToManaPoolSourceControllerEffect(Mana.BlackMana(1));
31          effect.setText("and add {B}.");
32          ability.addEffect(effect);
33          this.addAbility(ability);
34      }
35  
36      private CarnivalOfSouls(final CarnivalOfSouls card) {
37          super(card);
38      }
39  
40      @Override
41      public CarnivalOfSouls copy() {
42          return new CarnivalOfSouls(this);
43      }
44  }

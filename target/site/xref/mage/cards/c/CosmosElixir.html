18  /**
19   * @author TheElk801

21  public final class CosmosElixir extends CardImpl {
22  
23      public CosmosElixir(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{4}");
25  
26          // At the beginning of your end step, draw a card if your life total is greater than your starting life total. Otherwise, you gain 2 life.
27          this.addAbility(new BeginningOfEndStepTriggeredAbility(new ConditionalOneShotEffect(
28                  new DrawCardSourceControllerEffect(1), new GainLifeEffect(2),
29                  CosmosElixirCondition.instance, "draw a card if your life total " +
30                  "is greater than your starting life total. Otherwise, you gain 2 life"
31          ), TargetController.YOU, false));
32      }
33  
34      private CosmosElixir(final CosmosElixir card) {
35          super(card);
36      }
37  
38      @Override
39      public CosmosElixir copy() {
40          return new CosmosElixir(this);
41      }
42  }
43  
44  enum CosmosElixirCondition implements Condition {
45      instance;
46  
47      @Override
48      public boolean apply(Game game, Ability source) {
49          Player player = game.getPlayer(source.getControllerId());
50          return player != null && player.getLife() > game.getStartingLife();
51      }
52  }

19  import mage.target.TargetPlayer;
20  import mage.target.targetpointer.SecondTargetPointer;

22  import java.util.UUID;

24  /**
25   * @author TheElk801
26   */
27  public final class CyberneticaDatasmith extends CardImpl {
28  
29      private static final FilterCard filter = new FilterCard("Robots");
30      private static final FilterPlayer filter2 = new FilterPlayer();
31  
32      static {
33          filter.add(SubType.ROBOT.getPredicate());
34          filter2.add(new AnotherTargetPredicate(2));
35      }
36  
37      public CyberneticaDatasmith(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{1}{U}{B}");
39  
40          this.subtype.add(SubType.HUMAN);
41          this.subtype.add(SubType.ARTIFICER);
42          this.power = new MageInt(0);
43          this.toughness = new MageInt(1);
44  
45          // Protection from Robots
46          this.addAbility(new ProtectionAbility(filter));
47  
48          // Field Reprogramming -- {U}, {T}: Target player draws a card. Another target player creates a 4/4 colorless Robot artifact creature token with "This creature can't block."
49          Ability ability = new SimpleActivatedAbility(new DrawCardTargetEffect(1), new ManaCostsImpl<>("{U}"));
50          ability.addCost(new TapSourceCost());
51          ability.addEffect(new CreateTokenTargetEffect(new RobotToken())
52                  .setTargetPointer(new SecondTargetPointer())
53                  .concatBy("another"));
54          ability.addTarget(new TargetPlayer()
55                  .setTargetTag(1)
56                  .withChooseHint("draws a card"));
57          ability.addTarget(new TargetPlayer(filter2)
58                  .setTargetTag(2)
59                  .withChooseHint("creates a token"));
60          this.addAbility(ability.withFlavorWord("Field Reprogramming"));
61      }
62  
63      private CyberneticaDatasmith(final CyberneticaDatasmith card) {
64          super(card);
65      }
66  
67      @Override
68      public CyberneticaDatasmith copy() {
69          return new CyberneticaDatasmith(this);
70      }
71  }

24  import java.util.UUID;

26  /**
27   * @author LevelX2
28   */
29  public final class CitadelSiege extends CardImpl {
30  
31      private static final String ruleTrigger1 = "&bull  Khans &mdash; At the beginning of combat on your turn, put two +1/+1 counters on target creature you control.";
32      private static final String ruleTrigger2 = "&bull  Dragons &mdash; At the beginning of combat on each opponent's turn, tap target creature that player controls.";
33      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("creature controlled by the active player");
34  
35      static {
36          filter.add(CitadelSiegePredicate.instance);
37      }
38  
39      public CitadelSiege(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}{W}");
41  
42          // As Citadel Siege enters the battlefield, choose Khans or Dragons.
43          this.addAbility(new EntersBattlefieldAbility(new ChooseModeEffect("Khans or Dragons?", "Khans", "Dragons"), null,
44                  "As {this} enters the battlefield, choose Khans or Dragons.", ""));
45  
46          // * Khans - At the beginning of combat on your turn, put two +1/+1 counters on target creature you control.
47          Ability ability = new ConditionalTriggeredAbility(
48                  new BeginningOfCombatTriggeredAbility(new AddCountersTargetEffect(CounterType.P1P1.createInstance(2)), TargetController.YOU, false),
49                  new ModeChoiceSourceCondition("Khans"),
50                  ruleTrigger1);
51          ability.addTarget(new TargetControlledCreaturePermanent());
52          this.addAbility(ability);
53  
54          // * Dragons - At the beginning of combat on each opponent's turn, tap target creature that player controls.
55          ability = new ConditionalTriggeredAbility(
56                  new BeginningOfCombatTriggeredAbility(new TapTargetEffect(), TargetController.OPPONENT, false),
57                  new ModeChoiceSourceCondition("Dragons"),
58                  ruleTrigger2);
59          ability.addTarget(new TargetCreaturePermanent(filter));
60          this.addAbility(ability);
61      }
62  
63      private CitadelSiege(final CitadelSiege card) {
64          super(card);
65      }
66  
67      @Override
68      public CitadelSiege copy() {
69          return new CitadelSiege(this);
70      }
71  }
72  
73  enum CitadelSiegePredicate implements Predicate<Permanent> {
74      instance;
75  
76      @Override
77      public boolean apply(Permanent input, Game game) {
78          return input.getControllerId().equals(game.getActivePlayerId());
79      }
80  }

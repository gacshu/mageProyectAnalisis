22  public final class CalamitysWake extends CardImpl {
23  
24      public CalamitysWake(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{1}{W}");
26  
27          // Exile all graveyards. Players can't cast noncreature spells this turn. Exile Calamity's Wake.
28          this.getSpellAbility().addEffect(new ExileGraveyardAllPlayersEffect());
29          this.getSpellAbility().addEffect(new CalamitysWakeEffect());
30          this.getSpellAbility().addEffect(new ExileSpellEffect());
31      }
32  
33      private CalamitysWake(final CalamitysWake card) {
34          super(card);
35      }
36  
37      @Override
38      public CalamitysWake copy() {
39          return new CalamitysWake(this);
40      }
41  }
42  
43  class CalamitysWakeEffect extends ContinuousRuleModifyingEffectImpl {
44  
45      public CalamitysWakeEffect() {
46          super(Duration.EndOfTurn, Outcome.Benefit);
47          this.staticText = "Players can't cast noncreature spells this turn.";
48      }
49  
50      private CalamitysWakeEffect(final CalamitysWakeEffect effect) {
51          super(effect);
52      }
53  
54      @Override
55      public CalamitysWakeEffect copy() {
56          return new CalamitysWakeEffect(this);
57      }
58  
59      @Override
60      public boolean checksEventType(GameEvent event, Game game) {
61          return event.getType() == GameEvent.EventType.CAST_SPELL;
62      }
63  
64      @Override
65      public boolean applies(GameEvent event, Ability source, Game game) {
66          Card card = game.getCard(event.getSourceId());
67          return card != null && !card.isCreature(game);
68      }
69  }

22  public final class CloakedCadet extends CardImpl {
23  
24      public CloakedCadet(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{G}");
26  
27          this.subtype.add(SubType.HUMAN);
28          this.subtype.add(SubType.RANGER);
29          this.power = new MageInt(2);
30          this.toughness = new MageInt(4);
31  
32          // Training
33          this.addAbility(new TrainingAbility());
34  
35          // Whenever one or more +1/+1 counters are put on one or more Humans you control, draw a card. This ability triggers only once each turn.
36          this.addAbility(new CloakedCadetTriggeredAbility());
37      }
38  
39      private CloakedCadet(final CloakedCadet card) {
40          super(card);
41      }
42  
43      @Override
44      public CloakedCadet copy() {
45          return new CloakedCadet(this);
46      }
47  }
48  
49  class CloakedCadetTriggeredAbility extends TriggeredAbilityImpl {
50  
51      CloakedCadetTriggeredAbility() {
52          super(Zone.BATTLEFIELD, new DrawCardSourceControllerEffect(1));
53          this.setTriggersOnce(true);
54          setTriggerPhrase("Whenever one or more +1/+1 counters are put on one or more Humans you control, ");
55      }
56  
57      private CloakedCadetTriggeredAbility(final CloakedCadetTriggeredAbility ability) {
58          super(ability);
59      }
60  
61      @Override
62      public CloakedCadetTriggeredAbility copy() {
63          return new CloakedCadetTriggeredAbility(this);
64      }
65  
66      @Override
67      public boolean checkEventType(GameEvent event, Game game) {
68          return event.getType() == GameEvent.EventType.COUNTER_ADDED;
69      }
70  
71      @Override
72      public boolean checkTrigger(GameEvent event, Game game) {
73          Permanent permanent = game.getPermanent(event.getTargetId());
74          return permanent != null
75                  && isControlledBy(permanent.getControllerId())
76                  && permanent.hasSubtype(SubType.HUMAN, game)
77                  && event.getData().equals(CounterType.P1P1.getName());
78      }
79  }

41  class ConfluxTarget extends TargetCardInLibrary {
42  
43      private static final FilterCard filter
44              = new FilterCard("a white card, a blue card, a black card, a red card, and a green card");
45  
46      static {
47          filter.add(Predicates.not(ColorlessPredicate.instance));
48      }
49  
50      private static final ColorAssignment colorAssigner = new ColorAssignment("W", "U", "B", "R", "G");
51  
52      ConfluxTarget() {
53          super(0, 5, filter);
54      }
55  
56      private ConfluxTarget(final ConfluxTarget target) {
57          super(target);
58      }
59  
60      @Override
61      public ConfluxTarget copy() {
62          return new ConfluxTarget(this);
63      }
64  
65      @Override
66      public boolean canTarget(UUID playerId, UUID id, Ability source, Game game) {
67          if (!super.canTarget(playerId, id, source, game)) {
68              return false;
69          }
70          Card card = game.getCard(id);
71          if (card == null) {
72              return false;
73          }
74          if (this.getTargets().isEmpty()) {
75              return true;
76          }
77          Cards cards = new CardsImpl(this.getTargets());
78          cards.add(card);
79          return colorAssigner.getRoleCount(cards, game) >= cards.size();
80      }

29  public final class CogworkAssembler extends CardImpl {
30  
31      public CogworkAssembler(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{3}");
33          
34          this.subtype.add(SubType.ASSEMBLY_WORKER);
35          this.power = new MageInt(2);
36          this.toughness = new MageInt(3);
37  
38          // {7}: Create a token that's a copy of target artifact. That token gains haste. Exile it at the beginning of the next end step.
39          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new CogworkAssemblerCreateTokenEffect(), new GenericManaCost(7));
40          ability.addTarget(new TargetArtifactPermanent());
41          this.addAbility(ability);
42      }
43  
44      private CogworkAssembler(final CogworkAssembler card) {
45          super(card);
46      }
47  
48      @Override
49      public CogworkAssembler copy() {
50          return new CogworkAssembler(this);
51      }
52  }
53  
54  class CogworkAssemblerCreateTokenEffect extends OneShotEffect {
55  
56      CogworkAssemblerCreateTokenEffect() {
57          super(Outcome.Copy);
58          this.staticText = "Create a token that's a copy of target artifact. That token gains haste. Exile it at the beginning of the next end step";
59      }
60  
61      CogworkAssemblerCreateTokenEffect(final CogworkAssemblerCreateTokenEffect effect) {
62          super(effect);
63      }
64  
65      @Override
66      public CogworkAssemblerCreateTokenEffect copy() {
67          return new CogworkAssemblerCreateTokenEffect(this);
68      }
69  
70      @Override
71      public boolean apply(Game game, Ability source) {
72          Permanent copiedPermanent = game.getPermanent(this.getTargetPointer().getFirst(game, source));
73          if (copiedPermanent != null) {
74              CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect(null, CardType.ARTIFACT, true);
75              if (effect.apply(game, source)) {
76                  for (Permanent copyPermanent : effect.getAddedPermanents()) {
77                      ExileTargetEffect exileEffect = new ExileTargetEffect();
78                      exileEffect.setTargetPointer(new FixedTarget(copyPermanent, game));
79                      DelayedTriggeredAbility delayedAbility = new AtTheBeginOfNextEndStepDelayedTriggeredAbility(exileEffect);
80                      game.addDelayedTriggeredAbility(delayedAbility, source);
81                  }
82                  return true;
83              }
84          }
85          return false;
86      }
87  }

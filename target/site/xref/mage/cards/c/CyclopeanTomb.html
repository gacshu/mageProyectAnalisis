37  public final class CyclopeanTomb extends CardImpl {
38  
39      private static final FilterLandPermanent filter = new FilterLandPermanent();
40  
41      static {
42          filter.add(Predicates.not(SubType.SWAMP.getPredicate()));
43      }
44  
45      public CyclopeanTomb(UUID ownerId, CardSetInfo setInfo) {
46          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{4}");
47  
48          // {2}, {tap}: Put a mire counter on target non-Swamp land. That land is a Swamp for as long as it has a mire counter on it. Activate this ability only during your upkeep.
49          Ability ability = new ConditionalActivatedAbility(Zone.BATTLEFIELD, new AddCountersTargetEffect(CounterType.MIRE.createInstance()), new GenericManaCost(2), new IsStepCondition(PhaseStep.UPKEEP), "{2}, {T}: Put a mire counter on target non-Swamp land. That land is a Swamp for as long as it has a mire counter on it. Activate only during your upkeep.");
50          ability.addCost(new TapSourceCost());
51          ability.addTarget(new TargetLandPermanent(filter));
52          ability.addEffect(new BecomeSwampEffect());
53          this.addAbility(ability, new CyclopeanTombCounterWatcher());
54  
55          // When Cyclopean Tomb is put into a graveyard from the battlefield, at the beginning of each of your upkeeps for the rest of the game, remove all mire counters from a land that a mire counter was put onto with Cyclopean Tomb but that a mire counter has not been removed from with Cyclopean Tomb.
56          this.addAbility(new PutIntoGraveFromBattlefieldSourceTriggeredAbility(new CyclopeanTombCreateTriggeredEffect()));
57      }
58  
59      private CyclopeanTomb(final CyclopeanTomb card) {
60          super(card);
61      }
62  
63      @Override
64      public CyclopeanTomb copy() {
65          return new CyclopeanTomb(this);
66      }
67  }

69  class BecomeSwampEffect extends BecomesBasicLandTargetEffect {
70  
71      BecomeSwampEffect() {
72          super(Duration.Custom, false, true, SubType.SWAMP);
73          staticText = "That land is a Swamp for as long as it has a mire counter on it";
74      }
75  
76      private BecomeSwampEffect(final BecomeSwampEffect effect) {
77          super(effect);
78      }
79  
80      @Override
81      public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
82          Permanent land = game.getPermanent(this.targetPointer.getFirst(game, source));
83          if (land == null || land.getCounters(game).getCount(CounterType.MIRE) < 1) {
84              this.discard();
85              return false;
86          }
87          return super.apply(layer, sublayer, source, game);
88      }
89  
90      @Override
91      public BecomeSwampEffect copy() {
92          return new BecomeSwampEffect(this);
93      }
94  }

99          super(Outcome.Benefit);
100         this.staticText = "at the beginning of each of your upkeeps for the rest of the game, remove all mire counters from a land that a mire counter was put onto with {this} but that a mire counter has not been removed from with {this}";

104         super(effect);
105     }
106 
107     @Override
108     public CyclopeanTombCreateTriggeredEffect copy() {
109         return new CyclopeanTombCreateTriggeredEffect(this);
110     }
111 
112     @Override
113     public boolean apply(Game game, Ability source) {
114         Player controller = game.getPlayer(source.getControllerId());
115         if (controller != null) {
116             Permanent tomb = game.getPermanentOrLKIBattlefield(source.getSourceId()); // we need to set the correct source object
117             DelayedTriggeredAbility ability = new AtTheBeginOfYourNextUpkeepDelayedTriggeredAbility(new CyclopeanTombEffect(), Duration.EndOfGame, false);
118             ability.setControllerId(source.getControllerId());
119             ability.setSourceId(source.getSourceId());
120             game.addDelayedTriggeredAbility(ability, source);
121             return true;
122         }
123         return false;
124     }

127 class CyclopeanTombEffect extends OneShotEffect {
128 
129     CyclopeanTombEffect() {
130         super(Outcome.Benefit);
131         this.staticText = "At the beginning of each of your upkeeps for the rest of the game, remove all mire counters from a land that a mire counter was put onto with {this} but that a mire counter has not been removed from with {this}";
132     }
133 
134     private CyclopeanTombEffect(final CyclopeanTombEffect effect) {
135         super(effect);
136     }
137 
138     @Override
139     public CyclopeanTombEffect copy() {
140         return new CyclopeanTombEffect(this);
141     }
142 
143     @Override
144     public boolean apply(Game game, Ability source) {
145         Player controller = game.getPlayer(source.getControllerId());
146         MageObjectReference mor = new MageObjectReference(source.getSourceId(), source.getSourceObjectZoneChangeCounter(), game);
147         CyclopeanTombCounterWatcher watcher = game.getState().getWatcher(CyclopeanTombCounterWatcher.class);
148         if (controller != null && watcher != null) {
149 
150             Set<MageObjectReference> landRef = watcher.landMiredByCyclopeanTombInstance(mor, game);
151             if (landRef == null || landRef.isEmpty()) { // no lands got mire counter from that instance
152                 return true;
153             }
154             FilterLandPermanent filter = new FilterLandPermanent("a land with a mire counter added from the Cyclopean Tomb instance (" + landRef.size() + " left)");
155             Set<PermanentIdPredicate> idPref = new HashSet<>();
156             for (MageObjectReference ref : landRef) {
157                 Permanent land = ref.getPermanent(game);
158                 if (land != null) {
159                     idPref.add(new PermanentIdPredicate(land.getId()));
160                 }
161             }
162             filter.add(Predicates.or(idPref));
163             TargetLandPermanent target = new TargetLandPermanent(1, 1, filter, true);
164             /*Player must choose a land each upkeep. Using the message are above the player hand where frequent interactions
165              * take place is the most logical way to prompt for this scenario. A new constructor added to provide a not optional
166              * option for any cards like this where the player must choose a target in such the way this card requires.
167              */
168             if (controller.chooseTarget(Outcome.Neutral, target, source, game)) {
169                 Permanent chosenLand = game.getPermanent(target.getFirstTarget());
170                 if (chosenLand != null) {
171                     Effect effect = new RemoveAllCountersTargetEffect(CounterType.MIRE);
172                     effect.setTargetPointer(new FixedTarget(chosenLand, game));
173                     effect.apply(game, source);
174                     landRef.remove(new MageObjectReference(chosenLand, game));
175                 }
176             }
177 
178             return true;
179         }
180         return false;
181     }
182 }

184 class CyclopeanTombCounterWatcher extends Watcher {
185 
186     private final Map<MageObjectReference, Set<MageObjectReference>> counterData = new HashMap<>();
187 
188     CyclopeanTombCounterWatcher() {
189         super(WatcherScope.GAME);
190     }
191 
192     @Override
193     public void watch(GameEvent event, Game game) {
194         if (event.getType() == GameEvent.EventType.COUNTERS_ADDED && event.getData().equals(CounterType.MIRE.getName()) && event.getAmount() > 0) {
195             Permanent tomb = game.getPermanentOrLKIBattlefield(event.getTargetId());
196             if (tomb != null) {
197                 MageObjectReference cylopeanTombInstance = new MageObjectReference(tomb, game);
198                 Set<MageObjectReference> miredLands;
199                 if (counterData.containsKey(cylopeanTombInstance)) {
200                     miredLands = counterData.get(cylopeanTombInstance);
201                 } else {
202                     miredLands = new HashSet<>();
203                     counterData.put(cylopeanTombInstance, miredLands);
204                 }
205                 miredLands.add(new MageObjectReference(event.getTargetId(), game));
206             }
207 
208         }
209     }
210 
211     public Set<MageObjectReference> landMiredByCyclopeanTombInstance(MageObjectReference mor, Game game) {
212         if (counterData.containsKey(mor)) {
213             return counterData.get(mor);
214         }
215         return null;
216     }

26  /**
27   *
28   * @author LoneFox
29   */
30  public final class CentaurChieftain extends CardImpl {
31  
32      public CentaurChieftain(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{G}");
34          this.subtype.add(SubType.CENTAUR);
35          this.power = new MageInt(3);
36          this.toughness = new MageInt(3);
37  
38          // Haste
39          this.addAbility(HasteAbility.getInstance());
40          // Threshold - As long as seven or more cards are in your graveyard, Centaur Chieftain has "When Centaur Chieftain enters the battlefield, creatures you control get +1/+1 and gain trample until end of turn."
41          Effect effect = new BoostControlledEffect(1, 1, Duration.EndOfTurn);
42          effect.setText("creatures you control get +1/+1");
43          Ability gainedAbility = new EntersBattlefieldTriggeredAbility(effect);
44          effect = new GainAbilityControlledEffect(TrampleAbility.getInstance(), Duration.EndOfTurn,
45              new FilterControlledCreaturePermanent());
46          effect.setText("and gain trample until end of turn");
47          gainedAbility.addEffect(effect);
48  
49          Ability ability = new SimpleStaticAbility(Zone.BATTLEFIELD, new ConditionalContinuousEffect(
50              new GainAbilitySourceEffect(gainedAbility), new CardsInControllerGraveyardCondition(7),
51              "As long as seven or more cards are in your graveyard, {this} has \"When {this} enters the battlefield, creatures you control get +1/+1 and gain trample until end of turn.\""));
52          ability.setAbilityWord(AbilityWord.THRESHOLD);
53          this.addAbility(ability);
54      }
55  
56      private CentaurChieftain(final CentaurChieftain card) {
57          super(card);
58      }
59  
60      @Override
61      public CentaurChieftain copy() {
62          return new CentaurChieftain(this);
63      }
64  }

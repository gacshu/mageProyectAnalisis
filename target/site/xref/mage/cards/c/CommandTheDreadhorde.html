20  public final class CommandTheDreadhorde extends CardImpl {
21  
22      private static final FilterCard filter = new FilterCard("creature and/or planeswalker cards in graveyards");
23  
24      static {
25          filter.add(Predicates.or(
26                  CardType.PLANESWALKER.getPredicate(),
27                  CardType.CREATURE.getPredicate()
28          ));
29      }
30  
31      public CommandTheDreadhorde(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{B}{B}");
33  
34          // Choose any number of target creature and/or planeswalker cards in graveyards. Command the Dreadhorde deals damage to you equal to the total converted mana cost of those cards. Put them onto the battlefield under your control.
35          this.getSpellAbility().addEffect(new CommandTheDreadhordeEffect());
36          this.getSpellAbility().addTarget(new TargetCardInGraveyard(0, Integer.MAX_VALUE, filter));
37      }
38  
39      private CommandTheDreadhorde(final CommandTheDreadhorde card) {
40          super(card);
41      }
42  
43      @Override
44      public CommandTheDreadhorde copy() {
45          return new CommandTheDreadhorde(this);
46      }
47  }
48  
49  class CommandTheDreadhordeEffect extends OneShotEffect {
50  
51      CommandTheDreadhordeEffect() {
52          super(Outcome.Benefit);
53          staticText = "Choose any number of target creature and/or planeswalker cards in graveyards. " +
54                  "{this} deals damage to you equal to the total mana value of those cards. " +
55                  "Put them onto the battlefield under your control.";
56      }
57  
58      private CommandTheDreadhordeEffect(final CommandTheDreadhordeEffect effect) {
59          super(effect);
60      }
61  
62      @Override
63      public CommandTheDreadhordeEffect copy() {
64          return new CommandTheDreadhordeEffect(this);
65      }
66  
67      @Override
68      public boolean apply(Game game, Ability source) {
69          Player player = game.getPlayer(source.getControllerId());
70          if (player == null) {
71              return false;
72          }
73          Cards cards = new CardsImpl(source.getTargets().get(0).getTargets());
74          int damage = cards.getCards(game).stream().mapToInt(Card::getManaValue).sum();
75          player.damage(damage, source.getSourceId(), source, game);
76          return player.moveCards(cards, Zone.BATTLEFIELD, source, game);
77      }
78  }

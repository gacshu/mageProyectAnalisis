32  public final class ChaosMoon extends CardImpl {
33  
34      public ChaosMoon(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{R}");
36  
37          // At the beginning of each upkeep, count the number of permanents. If the number is odd, until end of turn, red creatures get +1/+1 and whenever a player taps a Mountain for mana, that player adds {R} (in addition to the mana the land produces). If the number is even, until end of turn, red creatures get -1/-1 and if a player taps a Mountain for mana, that Mountain produces colorless mana instead of any other type.
38          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new ChaosMoonEffect(), TargetController.ANY, false));
39      }

42          super(card);
43      }
44

47          return new ChaosMoon(this);
48      }

51  class ChaosMoonEffect extends OneShotEffect {
52  
53      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("red creatures");
54  
55      static {
56          filter.add(new ColorPredicate(ObjectColor.RED));
57      }
58  
59      ChaosMoonEffect() {
60          super(Outcome.Neutral);
61          this.staticText = "count the number of permanents. If the number is odd, " +
62                  "until end of turn, red creatures get +1/+1 and whenever a player taps a Mountain for mana, " +
63                  "that player adds {R} (in addition to the mana the land produces). If the number is even, " +
64                  "until end of turn, red creatures get -1/-1 and if a player taps a Mountain for mana, " +
65                  "that Mountain produces colorless mana instead of any other type";
66      }
67  
68      private ChaosMoonEffect(final ChaosMoonEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public ChaosMoonEffect copy() {
74          return new ChaosMoonEffect(this);
75      }
76  
77      @Override
78      public boolean apply(Game game, Ability source) {
79          Player player = game.getPlayer(source.getControllerId());
80          if (player == null) {
81              return false;
82          }
83          int permanentsInPlay = game.getBattlefield().count(
84                  StaticFilters.FILTER_PERMANENT, source.getControllerId(), source, game
85          );
86          // Odd
87          if (permanentsInPlay % 2 == 1) {
88              game.addEffect(new BoostAllEffect(1, 1, Duration.EndOfTurn, filter, false), source);
89              new CreateDelayedTriggeredAbilityEffect(new ChaosMoonOddTriggeredAbility()).apply(game, source);
90          } // Even
91          else {
92              game.addEffect(new BoostAllEffect(-1, -1, Duration.EndOfTurn, filter, false), source);
93              game.addEffect(new ChaosMoonEvenReplacementEffect(), source);
94          }
95          return true;
96      }
97  }

99  class ChaosMoonOddTriggeredAbility extends DelayedTriggeredManaAbility {
100 
101     ChaosMoonOddTriggeredAbility() {
102         super(new AddManaToManaPoolTargetControllerEffect(new Mana(ColoredManaSymbol.R), "their"), Duration.EndOfTurn, false);
103         this.usesStack = false;
104     }
105 
106     private ChaosMoonOddTriggeredAbility(ChaosMoonOddTriggeredAbility ability) {
107         super(ability);
108     }
109 
110     @Override
111     public boolean checkEventType(GameEvent event, Game game) {
112         return event.getType() == GameEvent.EventType.TAPPED_FOR_MANA;
113     }
114 
115     @Override
116     public boolean checkTrigger(GameEvent event, Game game) {
117         Permanent land = ((TappedForManaEvent) event).getPermanent();
118         if (land == null || !land.hasSubtype(SubType.MOUNTAIN, game)) {
119             return false;
120         }
121         this.getEffects().setTargetPointer(new FixedTarget(land.getControllerId()));
122         return true;
123     }
124 
125     @Override
126     public ChaosMoonOddTriggeredAbility copy() {
127         return new ChaosMoonOddTriggeredAbility(this);
128     }
129 
130     @Override
131     public String getRule() {
132         return "Until end of turn, whenever a player taps a Mountain for mana, that player adds {R}";
133     }
134 }

136 class ChaosMoonEvenReplacementEffect extends ReplacementEffectImpl {
137 
138     ChaosMoonEvenReplacementEffect() {
139         super(Duration.EndOfTurn, Outcome.Neutral);
140         staticText = "Until end of turn, if a Mountain is tapped for mana, it produces colorless mana instead of any other type.";
141     }
142 
143     private ChaosMoonEvenReplacementEffect(final ChaosMoonEvenReplacementEffect effect) {
144         super(effect);
145     }
146 
147     @Override
148     public ChaosMoonEvenReplacementEffect copy() {
149         return new ChaosMoonEvenReplacementEffect(this);
150     }
151 
152     @Override
153     public boolean apply(Game game, Ability source) {
154         return true;
155     }
156 
157     @Override
158     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
159         ManaEvent manaEvent = (ManaEvent) event;
160         Mana mana = manaEvent.getMana();
161         mana.setToMana(Mana.ColorlessMana(mana.count()));
162         return false;
163     }
164 
165     @Override
166     public boolean checksEventType(GameEvent event, Game game) {
167         return event.getType() == GameEvent.EventType.TAPPED_FOR_MANA;
168     }
169 
170     @Override
171     public boolean applies(GameEvent event, Ability source, Game game) {
172         Permanent land = ((TappedForManaEvent) event).getPermanent();
173         return land != null && land.hasSubtype(SubType.MOUNTAIN, game);
174     }

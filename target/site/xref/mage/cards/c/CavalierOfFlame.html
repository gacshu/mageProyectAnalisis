24  import java.util.UUID;

26  /**
27   * @author TheElk801
28   */
29  public final class CavalierOfFlame extends CardImpl {
30  
31      private static final DynamicValue xValue = new CardsInControllerGraveyardCount(StaticFilters.FILTER_CARD_LAND);
32      private static final FilterPermanent filter = new FilterPlaneswalkerPermanent();
33  
34      static {
35          filter.add(TargetController.OPPONENT.getControllerPredicate());
36      }
37  
38      public CavalierOfFlame(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{R}{R}{R}");
40  
41          this.subtype.add(SubType.ELEMENTAL);
42          this.subtype.add(SubType.KNIGHT);
43          this.power = new MageInt(6);
44          this.toughness = new MageInt(5);
45  
46          // {1}{R}: Creatures you control get +1/+0 and gain haste until end of turn.
47          Ability ability = new SimpleActivatedAbility(new BoostControlledEffect(
48                  1, 0, Duration.EndOfTurn
49          ).setText("Creatures you control get +1/+0"), new ManaCostsImpl<>("{1}{R}"));
50          ability.addEffect(new GainAbilityControlledEffect(
51                  HasteAbility.getInstance(), Duration.EndOfTurn,
52                  StaticFilters.FILTER_PERMANENT_CREATURE
53          ).setText("and gain haste until end of turn"));
54          this.addAbility(ability);
55  
56          // When Cavalier of Flame enters the battlefield, discard any number of cards, then draw that many cards.
57          this.addAbility(new EntersBattlefieldTriggeredAbility(new DiscardAndDrawThatManyEffect(Integer.MAX_VALUE)));
58  
59          // When Cavalier of Flame dies, it deals X damage to each opponent and each planeswalker they control, where X is the number of land cards in your graveyard.
60          ability = new DiesSourceTriggeredAbility(new DamagePlayersEffect(
61                  Outcome.Damage, xValue, TargetController.OPPONENT
62          ).setText("it deals X damage to each opponent"));
63          ability.addEffect(new DamageAllEffect(
64                  xValue, filter
65          ).setText("and each planeswalker they control, where X is the number of land cards in your graveyard"));
66          this.addAbility(ability);
67      }
68  
69      private CavalierOfFlame(final CavalierOfFlame card) {
70          super(card);
71      }
72  
73      @Override
74      public CavalierOfFlame copy() {
75          return new CavalierOfFlame(this);
76      }
77  }

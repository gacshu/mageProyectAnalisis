33      public ChandraPyromaster(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{2}{R}{R}");
35          this.addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.CHANDRA);
37  
38          this.setStartingLoyalty(4);
39  
40          // +1: Chandra, Pyromaster deals 1 damage to target player and 1 damage to 
41          // up to one target creature that player controls. That creature can't block this turn.
42          LoyaltyAbility ability1 = new LoyaltyAbility(new ChandraPyromasterEffect1(), 1);
43          Target target1 = new TargetPlayerOrPlaneswalker();
44          ability1.addTarget(target1);
45          ability1.addTarget(new ChandraPyromasterTarget());
46          this.addAbility(ability1);
47  
48          // 0: Exile the top card of your library. You may play it this turn.
49          LoyaltyAbility ability2 = new LoyaltyAbility(new ChandraPyromasterEffect2(), 0);
50          this.addAbility(ability2);
51  
52          // -7: Exile the top ten cards of your library. Choose an instant or sorcery 
53          // card exiled this way and copy it three times. You may cast the copies 
54          // without paying their mana costs.
55          LoyaltyAbility ability3 = new LoyaltyAbility(new ChandraPyromasterEffect3(), -7);
56          this.addAbility(ability3);
57  
58      }

70  class ChandraPyromasterEffect1 extends OneShotEffect {
71  
72      public ChandraPyromasterEffect1() {
73          super(Outcome.Damage);
74          staticText = "{this} deals 1 damage to target player or planeswalker "
75                  + "and 1 damage to up to one target creature that player or that "
76                  + "planeswalker's controller controls. That creature can't block this turn.";
77      }
78  
79      public ChandraPyromasterEffect1(final ChandraPyromasterEffect1 effect) {
80          super(effect);
81      }
82  
83      @Override
84      public ChandraPyromasterEffect1 copy() {
85          return new ChandraPyromasterEffect1(this);
86      }
87  
88      @Override
89      public boolean apply(Game game, Ability source) {
90          game.damagePlayerOrPlaneswalker(source.getTargets().get(0).getFirstTarget(),
91                  1, source.getSourceId(), source, game, false, true);
92          Permanent creature = game.getPermanent(source.getTargets().get(1).getFirstTarget());
93          if (creature != null) {
94              creature.damage(1, source.getSourceId(), source, game, false, true);
95              ContinuousEffect effect = new CantBlockTargetEffect(Duration.EndOfTurn);
96              effect.setTargetPointer(new FixedTarget(creature.getId(), game));
97              game.addEffect(effect, source);
98          }
99          return true;
100     }

103 class ChandraPyromasterTarget extends TargetPermanent {
104 
105     public ChandraPyromasterTarget() {
106         super(0, 1, new FilterCreaturePermanent("creature that the targeted player "
107                 + "or planeswalker's controller controls"), false);
108     }
109 
110     public ChandraPyromasterTarget(final ChandraPyromasterTarget target) {
111         super(target);
112     }
113 
114     @Override
115     public boolean canTarget(UUID id, Ability source, Game game) {
116         Player player = game.getPlayerOrPlaneswalkerController(source.getFirstTarget());
117         if (player == null) {
118             return false;
119         }
120         UUID firstTarget = player.getId();
121         Permanent permanent = game.getPermanent(id);
122         if (firstTarget != null && permanent != null && permanent.isControlledBy(firstTarget)) {
123             return super.canTarget(id, source, game);
124         }
125         return false;
126     }
127 
128     @Override
129     public Set<UUID> possibleTargets(UUID sourceControllerId, Ability source, Game game) {
130         Set<UUID> availablePossibleTargets = super.possibleTargets(sourceControllerId, source, game);
131         Set<UUID> possibleTargets = new HashSet<>();
132         MageObject object = game.getObject(source);
133 
134         for (StackObject item : game.getState().getStack()) {
135             if (item.getId().equals(source.getSourceId())) {
136                 object = item;
137             }
138             if (item.getSourceId().equals(source.getSourceId())) {
139                 object = item;
140             }
141         }
142 
143         if (object instanceof StackObject) {
144             UUID playerId = ((StackObject) object).getStackAbility().getFirstTarget();
145             Player player = game.getPlayerOrPlaneswalkerController(playerId);
146             if (player != null) {
147                 for (UUID targetId : availablePossibleTargets) {
148                     Permanent permanent = game.getPermanent(targetId);
149                     if (permanent != null && permanent.isControlledBy(player.getId())) {
150                         possibleTargets.add(targetId);
151                     }
152                 }
153             }
154         }
155         return possibleTargets;
156     }
157 
158     @Override
159     public ChandraPyromasterTarget copy() {
160         return new ChandraPyromasterTarget(this);
161     }
162 }

164 class ChandraPyromasterEffect2 extends OneShotEffect {
165 
166     public ChandraPyromasterEffect2() {
167         super(Outcome.Detriment);
168         this.staticText = "Exile the top card of your library. You may play it this turn";
169     }
170 
171     public ChandraPyromasterEffect2(final ChandraPyromasterEffect2 effect) {
172         super(effect);
173     }
174 
175     @Override
176     public ChandraPyromasterEffect2 copy() {
177         return new ChandraPyromasterEffect2(this);
178     }
179 
180     @Override
181     public boolean apply(Game game, Ability source) {
182         Player controller = game.getPlayer(source.getControllerId());
183         MageObject sourceObject = source.getSourceObject(game);
184         if (controller != null && sourceObject != null) {
185             Card card = controller.getLibrary().getFromTop(game);
186             if (card != null) {
187                 controller.moveCards(card, Zone.EXILED, source, game);
188                 ContinuousEffect effect = new PlayFromNotOwnHandZoneTargetEffect(
189                         Zone.EXILED, Duration.EndOfTurn);
190                 effect.setTargetPointer(new FixedTarget(card, game));
191                 game.addEffect(effect, source);
192             }
193             return true;
194         }
195         return false;
196     }

199 class ChandraPyromasterEffect3 extends OneShotEffect {
200 
201     public ChandraPyromasterEffect3() {
202         super(Outcome.PlayForFree);
203         this.staticText = "Exile the top ten cards of your library. Choose an instant "
204                 + "or sorcery card exiled this way and copy it three times. "
205                 + "You may cast the copies without paying their mana costs";
206     }
207 
208     public ChandraPyromasterEffect3(final ChandraPyromasterEffect3 effect) {
209         super(effect);
210     }
211 
212     @Override
213     public ChandraPyromasterEffect3 copy() {
214         return new ChandraPyromasterEffect3(this);
215     }
216 
217     @Override
218     public boolean apply(Game game, Ability source) {
219         Player controller = game.getPlayer(source.getControllerId());
220         MageObject sourceObject = source.getSourceObject(game);
221         if (controller == null || sourceObject == null) {
222             return false;
223         }
224         Cards cards = new CardsImpl(controller.getLibrary().getTopCards(game, 10));
225         controller.moveCardsToExile(cards.getCards(game), source, game, true, source.getSourceId(), sourceObject.getIdName());
226 
227         if (!cards.getCards(new FilterInstantOrSorceryCard(), game).isEmpty()) {
228             TargetCard target = new TargetCard(Zone.EXILED, new FilterInstantOrSorceryCard());
229             if (controller.chooseTarget(Outcome.PlayForFree, cards, target, source, game)) {
230                 Card card = cards.get(target.getFirstTarget(), game);
231                 if (card != null) {
232                     ApprovingObject approvingObject = new ApprovingObject(source, game);
233                     if (controller.chooseUse(outcome, "Cast copy 1 of " + card.getName(), source, game)) {
234                         Card copy1 = game.copyCard(card, source, source.getControllerId());
235                         game.getState().setValue("PlayFromNotOwnHandZone" + copy1.getId(), Boolean.TRUE);
236                         controller.cast(controller.chooseAbilityForCast(copy1, game, true),
237                                 game, true, approvingObject);
238                         game.getState().setValue("PlayFromNotOwnHandZone" + copy1.getId(), null);
239                     }
240                     if (controller.chooseUse(outcome, "Cast copy 2 of " + card.getName(), source, game)) {
241                         Card copy2 = game.copyCard(card, source, source.getControllerId());
242                         game.getState().setValue("PlayFromNotOwnHandZone" + copy2.getId(), Boolean.TRUE);
243                         controller.cast(controller.chooseAbilityForCast(copy2, game, true),
244                                 game, true, approvingObject);
245                         game.getState().setValue("PlayFromNotOwnHandZone" + copy2.getId(), null);
246                     }
247                     if (controller.chooseUse(outcome, "Cast copy 3 of " + card.getName(), source, game)) {
248                         Card copy3 = game.copyCard(card, source, source.getControllerId());
249                         game.getState().setValue("PlayFromNotOwnHandZone" + copy3.getId(), Boolean.TRUE);
250                         controller.cast(controller.chooseAbilityForCast(copy3, game, true),
251                                 game, true, approvingObject);
252                         game.getState().setValue("PlayFromNotOwnHandZone" + copy3.getId(), null);
253                     }
254                     return true;
255                 }
256             }
257         }
258         return false;
259     }

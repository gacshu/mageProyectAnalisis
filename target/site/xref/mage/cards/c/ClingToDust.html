23      public ClingToDust(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{B}");
25  
26          // Exile target card from a graveyard. If it was a creature card, you gain 3 life. Otherwise, you draw a card.
27          this.getSpellAbility().addEffect(new ClingToDustEffect());
28          this.getSpellAbility().addTarget(new TargetCardInGraveyard());
29  
30          // Escapeâ€”{3}{B}, Exile five other cards from your graveyard.
31          this.addAbility(new EscapeAbility(this, "{3}{B}", 5));
32      }

44  class ClingToDustEffect extends OneShotEffect {
45  
46      ClingToDustEffect() {
47          super(Outcome.Benefit);
48          staticText = "exile target card from a graveyard. If it was a creature card, " +
49                  "you gain 3 life. Otherwise, you draw a card";
50      }
51  
52      private ClingToDustEffect(final ClingToDustEffect effect) {
53          super(effect);
54      }
55  
56      @Override
57      public ClingToDustEffect copy() {
58          return new ClingToDustEffect(this);
59      }
60  
61      @Override
62      public boolean apply(Game game, Ability source) {
63          Player player = game.getPlayer(source.getControllerId());
64          Card card = game.getCard(source.getFirstTarget());
65          if (player == null || card == null) {
66              return false;
67          }
68          boolean isCreature = card.isCreature(game);
69          if (!player.moveCards(card, Zone.EXILED, source, game)) {
70              return false;
71          }
72          if (isCreature) {
73              player.gainLife(3, game, source);
74          } else {
75              player.drawCards(1, source, game);
76          }
77          return true;
78      }

2   package mage.cards.c;

4   import java.util.UUID;
5   import mage.abilities.Ability;
6   import mage.abilities.common.AttacksOrBlocksAttachedTriggeredAbility;
7   import mage.abilities.effects.common.AttachEffect;
8   import mage.abilities.effects.common.LoseLifeControllerAttachedEffect;
9   import mage.abilities.keyword.EnchantAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.*;
13  import mage.target.TargetPermanent;
14  import mage.target.common.TargetCreaturePermanent;
15  
16  /**
17   *
18   * @author Loki
19   */
20  public final class ContaminatedBond extends CardImpl {
21  
22      public ContaminatedBond(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{1}{B}");
24          this.subtype.add(SubType.AURA);
25  
26          // Enchant creature
27          TargetPermanent auraTarget = new TargetCreaturePermanent();
28          this.getSpellAbility().addTarget(auraTarget);
29          this.getSpellAbility().addEffect(new AttachEffect(Outcome.UnboostCreature));
30          Ability ability = new EnchantAbility(auraTarget);
31          this.addAbility(ability);
32  
33          // Whenever enchanted creature attacks or blocks, its controller loses 3 life.
34          this.addAbility(new AttacksOrBlocksAttachedTriggeredAbility(new LoseLifeControllerAttachedEffect(3), AttachmentType.AURA));
35      }
36  
37      private ContaminatedBond(final ContaminatedBond card) {
38          super(card);
39      }
40  
41      @Override
42      public ContaminatedBond copy() {
43          return new ContaminatedBond(this);
44      }
45  }

20   * @author TheElk801

22  public final class ChanneledForce extends CardImpl {
23  
24      public ChanneledForce(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{2}{U}{R}");
26  
27          // As an additional cost to cast this spell, discard X cards.
28          this.getSpellAbility().addCost(new DiscardXTargetCost(StaticFilters.FILTER_CARD_CARDS));
29  
30          // Target player draws X cards. Channeled Force deals X damage to up to one target creature or planeswalker.
31          this.getSpellAbility().addEffect(new ChanneledForceEffect());
32          this.getSpellAbility().addTarget(new TargetPlayer());
33          this.getSpellAbility().addTarget(new TargetCreatureOrPlaneswalker());
34      }
35  
36      private ChanneledForce(final ChanneledForce card) {
37          super(card);
38      }
39  
40      @Override
41      public ChanneledForce copy() {
42          return new ChanneledForce(this);
43      }
44  }
45  
46  class ChanneledForceEffect extends OneShotEffect {
47  
48      ChanneledForceEffect() {
49          super(Outcome.Benefit);
50          staticText = "Target player draws X cards. {this} deals X damage to up to one target creature or planeswalker.";
51      }
52  
53      private ChanneledForceEffect(final ChanneledForceEffect effect) {
54          super(effect);
55      }
56  
57      @Override
58      public ChanneledForceEffect copy() {
59          return new ChanneledForceEffect(this);
60      }
61  
62      @Override
63      public boolean apply(Game game, Ability source) {
64          int xValue = GetXValue.instance.calculate(game, source, this);
65          if (xValue == 0) {
66              return false;
67          }
68          Player player = game.getPlayer(source.getTargets().get(0).getFirstTarget());
69          if (player != null) {
70              player.drawCards(xValue, source, game);
71          }
72          game.damagePlayerOrPlaneswalker(
73                  source.getTargets().get(1).getFirstTarget(), xValue,
74                  source.getSourceId(), source, game, false, true
75          );
76          return true;
77      }
78  }

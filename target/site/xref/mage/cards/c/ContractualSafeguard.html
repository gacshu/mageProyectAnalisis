56  class ContractualSafeguardFirstEffect extends OneShotEffect {
57  
58      ContractualSafeguardFirstEffect() {
59          super(Outcome.Benefit);
60          staticText = "if you cast this spell during your main phase, put a shield counter on a creature you control";
61      }
62  
63      private ContractualSafeguardFirstEffect(final ContractualSafeguardFirstEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public ContractualSafeguardFirstEffect copy() {
69          return new ContractualSafeguardFirstEffect(this);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          if (!AddendumCondition.instance.apply(game, source)) {
75              return false;
76          }
77          Player player = game.getPlayer(source.getControllerId());
78          if (player == null || !game.getBattlefield().contains(
79                  StaticFilters.FILTER_CONTROLLED_CREATURE, source, game, 1
80          )) {
81              return false;
82          }
83          TargetPermanent target = new TargetControlledCreaturePermanent();
84          target.withChooseHint("to give a shield counter").setNotTarget(true);
85          player.choose(outcome, target, source, game);
86          Permanent permanent = game.getPermanent(target.getFirstTarget());
87          return permanent != null && permanent.addCounters(CounterType.SHIELD.createInstance(), source, game);
88      }
89  }

91  class ContractualSafeguardSecondEffect extends OneShotEffect {
92  
93      private static final FilterPermanent filter
94              = new FilterControlledCreaturePermanent("creature you control with a counter on it");
95  
96      static {
97          filter.add(CounterAnyPredicate.instance);
98      }
99  
100     ContractualSafeguardSecondEffect() {
101         super(Outcome.Benefit);
102         staticText = "choose a kind of counter on a creature you control. " +
103                 "Put a counter of that kind on each other creature you control";
104         concatPrefix = "<br>";
105     }
106 
107     private ContractualSafeguardSecondEffect(final ContractualSafeguardSecondEffect effect) {
108         super(effect);
109     }
110 
111     @Override
112     public ContractualSafeguardSecondEffect copy() {
113         return new ContractualSafeguardSecondEffect(this);
114     }
115 
116     @Override
117     public boolean apply(Game game, Ability source) {
118         Player player = game.getPlayer(source.getControllerId());
119         if (player == null || !game.getBattlefield().contains(filter, source, game, 1)) {
120             return false;
121         }
122         TargetPermanent target = new TargetPermanent(filter);
123         target.setNotTarget(true);
124         player.choose(outcome, target, source, game);
125         Permanent permanent = game.getPermanent(target.getFirstTarget());
126         if (permanent == null) {
127             return false;
128         }
129         Set<String> counterTypes = new LinkedHashSet<>(permanent.getCounters(game).keySet());
130         String chosenType;
131         switch (counterTypes.size()) {
132             case 0:
133                 return false;
134             case 1:
135                 chosenType = RandomUtil.randomFromCollection(counterTypes);
136                 break;
137             case 2:
138                 Iterator<String> iterator = counterTypes.iterator();
139                 String type1 = iterator.next();
140                 String type2 = iterator.next();
141                 chosenType = player.chooseUse(
142                         outcome, "Choose which kind of counter to add to each other creature you control",
143                         null, type1, type2, source, game
144                 ) ? type1 : type2;
145                 break;
146             default:
147                 Choice choice = new ChoiceImpl(true);
148                 choice.setMessage("Choose a kind of counter to add to each other creature you control");
149                 choice.setChoices(counterTypes);
150                 player.choose(outcome, choice, game);
151                 chosenType = choice.getChoice();
152         }
153         CounterType counterType = CounterType.findByName(chosenType);
154         for (Permanent creature : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
155             if (!creature.getId().equals(permanent.getId())) {
156                 creature.addCounters(counterType.createInstance(), source, game);
157             }
158         }
159         return true;
160     }

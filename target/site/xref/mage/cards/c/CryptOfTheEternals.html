18  import java.util.UUID;

20  public final class CryptOfTheEternals extends CardImpl {
21  
22      public CryptOfTheEternals(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
24  
25          // When Crypt of the Eternals enters the battlefield, you gain 1 life.
26          this.addAbility(new EntersBattlefieldTriggeredAbility(new GainLifeEffect(1)));
27  
28          // {T}: Add {C}.
29          this.addAbility(new ColorlessManaAbility());
30  
31          // {1}, {T}: Add {U}, {B}, or {R}.
32          List<Mana> list = new ArrayList<>(Arrays.asList(Mana.BlueMana(1), Mana.BlackMana(1), Mana.RedMana(1)));
33  
34  
35          for (Mana m : list) {
36              SimpleManaAbility uAbility = new SimpleManaAbility(Zone.BATTLEFIELD, m, new ManaCostsImpl<>("{1}"));
37              uAbility.addCost(new TapSourceCost());
38              this.addAbility(uAbility);
39          }
40      }
41  
42      private CryptOfTheEternals(final CryptOfTheEternals card) {
43          super(card);
44      }
45  
46      @Override
47      public CryptOfTheEternals copy() {
48          return new CryptOfTheEternals(this);
49      }
50  }

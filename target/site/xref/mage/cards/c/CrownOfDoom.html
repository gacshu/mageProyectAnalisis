33  public final class CrownOfDoom extends CardImpl {
34  
35      private static final FilterPlayer filter = new FilterPlayer("player other than {this}'s owner");
36  
37      static {
38          filter.add(CrownOfDoomPredicate.instance);
39      }
40  
41      public CrownOfDoom(UUID ownerId, CardSetInfo setInfo) {
42          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{3}");
43  
44          // Whenever a creature attacks you or a planeswalker you control, it gets +2/+0 until end of turn.
45          Effect effect = new BoostTargetEffect(2, 0, Duration.EndOfTurn);
46          effect.setText("it gets +2/+0 until end of turn");
47          this.addAbility(new AttacksAllTriggeredAbility(
48                  effect,
49                  false,
50                  StaticFilters.FILTER_PERMANENT_CREATURE,
51                  SetTargetPointer.PERMANENT,
52                  true));
53  
54          // {2}: Target player other than Crown of Doom's owner gains control of it. Activate this ability only during your turn.
55          Ability ability = new ActivateIfConditionActivatedAbility(
56                  Zone.BATTLEFIELD,
57                  new CrownOfDoomEffect(),
58                  new ManaCostsImpl<>("{2}"),
59                  MyTurnCondition.instance);
60          ability.addTarget(new TargetPlayer(1, 1, false, filter));
61          ability.addHint(MyTurnHint.instance);
62          this.addAbility(ability);
63      }
64  
65      private CrownOfDoom(final CrownOfDoom card) {
66          super(card);
67      }
68  
69      @Override
70      public CrownOfDoom copy() {
71          return new CrownOfDoom(this);
72      }
73  }
74  
75  enum CrownOfDoomPredicate implements ObjectSourcePlayerPredicate<Player> {
76      instance;
77  
78      @Override
79      public boolean apply(ObjectSourcePlayer<Player> input, Game game) {
80          Player targetPlayer = input.getObject();
81          Permanent sourceObject = input.getSource().getSourcePermanentOrLKI(game);
82          if (targetPlayer == null || sourceObject == null) {
83              return false;
84          }
85          return !targetPlayer.getId().equals(sourceObject.getOwnerId());
86      }
87  
88      @Override
89      public String toString() {
90          return "Owner(" + ')';
91      }
92  }
93  
94  class CrownOfDoomEffect extends OneShotEffect {
95  
96      public CrownOfDoomEffect() {
97          super(Outcome.Detriment);
98          this.staticText = "Target player other than {this}'s owner gains control of it";
99      }
100 
101     public CrownOfDoomEffect(final CrownOfDoomEffect effect) {
102         super(effect);
103     }
104 
105     @Override
106     public CrownOfDoomEffect copy() {
107         return new CrownOfDoomEffect(this);
108     }
109 
110     @Override
111     public boolean apply(Game game, Ability source) {
112         Player controller = game.getPlayer(source.getControllerId());
113         Player newController = game.getPlayer(getTargetPointer().getFirst(game, source));
114         if (controller != null
115                 && newController != null
116                 && !Objects.equals(controller.getId(), newController.getId())) {
117             ContinuousEffect effect = new GainControlTargetEffect(Duration.Custom, newController.getId());
118             effect.setTargetPointer(new FixedTarget(source.getSourceId(), game));
119             game.addEffect(effect, source);
120             return true;
121         }
122         return false;
123     }
124 }

36  public final class CurseOfTheCabal extends CardImpl {
37  
38      public CurseOfTheCabal(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{9}{B}");
40  
41          // Target player sacrifices half the permanents they control, rounded down.
42          this.getSpellAbility().addTarget(new TargetPlayer());
43          this.getSpellAbility().addEffect(new CurseOfTheCabalSacrificeEffect());
44  
45          // Suspend 2-{2}{B}{B}
46          this.addAbility(new SuspendAbility(2, new ManaCostsImpl<>("{2}{B}{B}"), this));
47  
48          // At the beginning of each player's upkeep, if Curse of the Cabal is suspended, that player may sacrifice a permanent. If they do, put two time counters on Curse of the Cabal.
49          this.addAbility(new CurseOfTheCabalInterveningIfTriggeredAbility());
50  
51      }

54          super(card);
55      }

63  class CurseOfTheCabalSacrificeEffect extends OneShotEffect {
64  
65      public CurseOfTheCabalSacrificeEffect() {
66          super(Outcome.Sacrifice);
67          this.staticText = "Target player sacrifices half the permanents they control, rounded down.";
68      }
69  
70      public CurseOfTheCabalSacrificeEffect(final CurseOfTheCabalSacrificeEffect effect) {
71          super(effect);
72      }
73  
74      @Override
75      public CurseOfTheCabalSacrificeEffect copy() {
76          return new CurseOfTheCabalSacrificeEffect(this);
77      }
78  
79      @Override
80      public boolean apply(Game game, Ability source) {
81          Player targetPlayer = game.getPlayer(source.getFirstTarget());
82          if (targetPlayer != null) {
83              int amount = game.getBattlefield().countAll(StaticFilters.FILTER_CONTROLLED_PERMANENT, targetPlayer.getId(), game) / 2;
84              if (amount < 1) {
85                  return true;
86              }
87              Target target = new TargetControlledPermanent(amount, amount, StaticFilters.FILTER_CONTROLLED_PERMANENT, true);
88              if (target.canChoose(targetPlayer.getId(), source, game)) {
89                  while (!target.isChosen() 
90                          && target.canChoose(targetPlayer.getId(), source, game) && targetPlayer.canRespond()) {
91                      targetPlayer.choose(Outcome.Sacrifice, target, source, game);
92                  }
93                  //sacrifice all chosen (non null) permanents
94                  target.getTargets().stream()
95                          .map(game::getPermanent)
96                          .filter(Objects::nonNull)
97                          .forEach(permanent -> permanent.sacrifice(source, game));
98              }
99              return true;
100         }
101         return false;
102     }
103 }

131 class CurseOfTheCabalTriggeredAbilityConditionalDelay extends AddCountersSourceEffect {
132 
133     public CurseOfTheCabalTriggeredAbilityConditionalDelay() {
134         super(CounterType.TIME.createInstance(), StaticValue.get(2), false, true);
135     }
136 
137     @Override
138     public boolean apply(Game game, Ability source) {
139         UUID activePlayerId = game.getActivePlayerId();
140         Player target = game.getPlayer(activePlayerId);
141         Cost cost = new SacrificeTargetCost(new TargetControlledPermanent(new FilterControlledPermanent()));
142         if (target == null) {
143             return false;
144         }
145         if (cost.canPay(source, source, activePlayerId, game)
146                 && target.chooseUse(Outcome.Sacrifice, "Sacrifice a permanent to delay Curse of the Cabal?", source, game)
147                 && cost.pay(source, game, source, activePlayerId, true, null)) {
148             return super.apply(game, source);
149         }
150         return true;
151     }
152 
153     public CurseOfTheCabalTriggeredAbilityConditionalDelay(final CurseOfTheCabalTriggeredAbilityConditionalDelay effect) {
154         super(effect);
155     }
156 
157     @Override
158     public CurseOfTheCabalTriggeredAbilityConditionalDelay copy() {
159         return new CurseOfTheCabalTriggeredAbilityConditionalDelay(this);
160     }

29  public final class CurseOfVitality extends CardImpl {
30  
31      public CurseOfVitality(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{2}{W}");
33          this.subtype.add(SubType.AURA, SubType.CURSE);
34  
35          // Enchant player
36          TargetPlayer auraTarget = new TargetPlayer();
37          this.getSpellAbility().addTarget(auraTarget);
38          this.getSpellAbility().addEffect(new AttachEffect(Outcome.Detriment));
39          this.addAbility(new EnchantAbility(auraTarget));
40  
41          // Whenever enchanted player is attacked, you gain 2 life. Each opponent attacking that player does the same.
42          this.addAbility(new EnchantedPlayerAttackedTriggeredAbility(new CurseOfVitalityEffect()));
43      }
44  
45      private CurseOfVitality(final CurseOfVitality card) {
46          super(card);
47      }
48  
49      @Override
50      public CurseOfVitality copy() {
51          return new CurseOfVitality(this);
52      }
53  }

55  class CurseOfVitalityEffect extends OneShotEffect {
56  
57      CurseOfVitalityEffect() {
58          super(Outcome.Benefit);
59          this.staticText = "gain 2 life. Each opponent attacking that player does the same.";
60      }
61  
62      CurseOfVitalityEffect(final CurseOfVitalityEffect effect) {
63          super(effect);
64      }
65  
66      @Override
67      public CurseOfVitalityEffect copy() {
68          return new CurseOfVitalityEffect(this);
69      }
70  
71      @Override
72      public boolean apply(Game game, Ability source) {
73          // In the case that the enchantment is blinked
74          Permanent enchantment = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
75          if (enchantment == null) {
76              // It was not blinked, use the standard method
77              enchantment = game.getPermanentOrLKIBattlefield(source.getSourceId());
78          }
79          if (enchantment != null) {
80              Player enchantedPlayer = game.getPlayer(enchantment.getAttachedTo());
81              if (enchantedPlayer != null) {
82                  Set<UUID> players = new HashSet<>();
83                  for (UUID attacker : game.getCombat().getAttackers()) {
84                      UUID defender = game.getCombat().getDefenderId(attacker);
85                      if (defender.equals(enchantedPlayer.getId())
86                              && game.getPlayer(source.getControllerId()).hasOpponent(game.getPermanent(attacker).getControllerId(), game)) {
87                          players.add(game.getPermanent(attacker).getControllerId());
88                      }
89                  }
90                  players.add(source.getControllerId());
91                  for (UUID player : players) {
92                      game.getPlayer(player);
93                      Effect effect = new GainLifeTargetEffect(2);
94                      effect.setTargetPointer(new FixedTarget(player));
95                      effect.apply(game, source);
96                  }
97              }
98              return true;
99  
100         }
101         return false;
102     }

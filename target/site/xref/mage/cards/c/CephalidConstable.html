26  public final class CephalidConstable extends CardImpl {
27  
28      public CephalidConstable(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{1}{U}{U}");
30          this.subtype.add(SubType.CEPHALID);
31          this.subtype.add(SubType.WIZARD);
32          this.power = new MageInt(1);
33          this.toughness = new MageInt(1);
34  
35          // Whenever Cephalid Constable deals combat damage to a player, return up to that many target permanents that player controls to their owners' hands.
36          Ability ability = new CephalidConstableTriggeredAbility();
37          ability.addTarget(new TargetPermanent(0, 1, new FilterPermanent(), false)); // Ajusted when it triggers.
38          this.addAbility(ability);
39      }
40  
41      private CephalidConstable(final CephalidConstable card) {
42          super(card);
43      }
44  
45      @Override
46      public CephalidConstable copy() {
47          return new CephalidConstable(this);
48      }
49  }
50  
51  class CephalidConstableTriggeredAbility extends TriggeredAbilityImpl {
52  
53      CephalidConstableTriggeredAbility() {
54          super(Zone.BATTLEFIELD, new ReturnToHandTargetEffect(), false);
55      }
56  
57      CephalidConstableTriggeredAbility(final CephalidConstableTriggeredAbility ability) {
58          super(ability);
59      }
60  
61      @Override
62      public CephalidConstableTriggeredAbility copy() {
63          return new CephalidConstableTriggeredAbility(this);
64      }
65  
66      @Override
67      public boolean checkEventType(GameEvent event, Game game) {
68          return event.getType() == GameEvent.EventType.DAMAGED_PLAYER;
69      }
70  
71      @Override
72      public boolean checkTrigger(GameEvent event, Game game) {
73          if (event.getSourceId().equals(this.sourceId) && ((DamagedPlayerEvent) event).isCombatDamage()) {
74              FilterPermanent filter = new FilterPermanent("permanent" + (event.getAmount() > 1 ? "s" : "") + " damaged player control");
75              filter.add(new ControllerIdPredicate(event.getPlayerId()));
76              Target target = new TargetPermanent(0, event.getAmount(), filter, false);
77              this.getTargets().clear();
78              this.getTargets().add(target);
79              return true;
80          }
81          return false;
82      }
83  
84      @Override
85      public String getRule() {
86          return "Whenever {this} deals combat damage to a player, return up to that many target permanents that player controls to their owners' hands";
87      }
88  
89  }

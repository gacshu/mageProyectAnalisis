36          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");

43      }

136 class CityInABottleStateTriggeredAbility extends StateTriggeredAbility {
137 
138     private static final FilterPermanent filter = new FilterPermanent("a nontoken permanent originally printed in the Arabian Nights expansion other than City in a Bottle");
139 
140     static {
141         filter.add(TokenPredicate.FALSE);
142         filter.add(Predicates.or(getArabianNightsNamePredicates()));
143 
144     }
145 
146     CityInABottleStateTriggeredAbility() {
147         super(Zone.BATTLEFIELD, new CityInABottleSacrificeEffect());
148     }
149 
150     CityInABottleStateTriggeredAbility(final CityInABottleStateTriggeredAbility ability) {
151         super(ability);
152     }
153 
154     @Override
155     public CityInABottleStateTriggeredAbility copy() {
156         return new CityInABottleStateTriggeredAbility(this);
157     }
158 
159     @Override
160     public boolean checkTrigger(GameEvent event, Game game) {
161         return game.getBattlefield().contains(filter, this.getControllerId(), this, game, 1);
162     }
163 
164     @Override
165     public String getRule() {
166         return "Whenever one or more other nontoken permanents with a name originally printed in the <i>Arabian Nights</i> expansion are on the battlefield, their controllers sacrifice them";
167     }
168 }

170 class CityInABottleSacrificeEffect extends OneShotEffect {
171 
172     private static final FilterPermanent filter = new FilterPermanent("a nontoken permanent originally printed in the Arabian Nights expansion other than City in a Bottle");
173 
174     static {
175         filter.add(TokenPredicate.FALSE);
176         filter.add(Predicates.or(getArabianNightsNamePredicates()));
177     }
178 
179     CityInABottleSacrificeEffect() {
180         super(Outcome.Sacrifice);
181         this.staticText = "its controller sacrifices it";
182     }
183 
184     CityInABottleSacrificeEffect(final CityInABottleSacrificeEffect effect) {
185         super(effect);
186     }
187 
188     @Override
189     public CityInABottleSacrificeEffect copy() {
190         return new CityInABottleSacrificeEffect(this);
191     }
192 
193     @Override
194     public boolean apply(Game game, Ability source) {
195         for (Permanent permanent : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
196             permanent.sacrifice(source, game);
197         }
198         return true;
199     }
200 }

202 class CityInABottleCantPlayEffect extends ContinuousRuleModifyingEffectImpl {
203 
204     private static final FilterCard filter = new FilterCard("cards originally printed in the Arabian Nights expansion");
205 
206     static {
207         filter.add(Predicates.or(getArabianNightsNamePredicates()));
208     }
209 
210     CityInABottleCantPlayEffect() {
211         super(Duration.WhileOnBattlefield, Outcome.Detriment);
212         staticText = "Players can't play cards originally printed in the <i>Arabian Nights</i> expansion";
213     }
214 
215     CityInABottleCantPlayEffect(final CityInABottleCantPlayEffect effect) {
216         super(effect);
217     }
218 
219     @Override
220     public boolean apply(Game game, Ability source) {
221         return true;
222     }
223 
224     @Override
225     public CityInABottleCantPlayEffect copy() {
226         return new CityInABottleCantPlayEffect(this);
227     }
228 
229     @Override
230     public String getInfoMessage(Ability source, GameEvent event, Game game) {
231         return "You can't play cards originally printed in the Arabian Nights expansion";
232     }
233 
234     @Override
235     public boolean checksEventType(GameEvent event, Game game) {
236         return event.getType() == GameEvent.EventType.PLAY_LAND || event.getType() == GameEvent.EventType.CAST_SPELL;
237     }
238 
239     @Override
240     public boolean applies(GameEvent event, Ability source, Game game) {
241         Card card = game.getCard(event.getSourceId());
242         return filter.match(card, source.getControllerId(), source, game);
243     }

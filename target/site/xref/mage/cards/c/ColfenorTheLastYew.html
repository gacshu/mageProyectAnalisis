31      public ColfenorTheLastYew(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}{B}{G}");
33  
34          this.addSuperType(SuperType.LEGENDARY);
35          this.subtype.add(SubType.TREEFOLK);
36          this.subtype.add(SubType.SHAMAN);
37          this.power = new MageInt(3);
38          this.toughness = new MageInt(7);
39  
40          // Vigilance
41          this.addAbility(VigilanceAbility.getInstance());
42  
43          // Reach
44          this.addAbility(ReachAbility.getInstance());
45  
46          // Whenever Colfenor, the Last Yew or another creature you control dies, return up to one other target creature card with lesser toughness from your graveyard to your hand.
47          this.addAbility(new ColfenorTheLastYewTriggeredAbility());
48      }

52      }

60  class ColfenorTheLastYewTriggeredAbility extends DiesThisOrAnotherCreatureTriggeredAbility {
61  
62      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent();
63  
64      static {
65          filter.add(TargetController.YOU.getControllerPredicate());
66      }
67  
68      ColfenorTheLastYewTriggeredAbility() {
69          super(new ReturnFromGraveyardToHandTargetEffect(), false, filter);
70      }
71  
72      private ColfenorTheLastYewTriggeredAbility(final ColfenorTheLastYewTriggeredAbility ability) {
73          super(ability);
74      }
75  
76      @Override
77      public DiesThisOrAnotherCreatureTriggeredAbility copy() {
78          return new ColfenorTheLastYewTriggeredAbility(this);
79      }
80  
81      @Override
82      public boolean checkTrigger(GameEvent event, Game game) {
83          if (!super.checkTrigger(event, game)) {
84              return false;
85          }
86          ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
87          Permanent permanent = zEvent.getTarget();
88          if (permanent == null) {
89              return false;
90          }
91          FilterCard filterCard = new FilterCreatureCard("creature card with toughness less than " + permanent.getToughness().getValue());
92          filterCard.add(new ToughnessPredicate(ComparisonType.FEWER_THAN, permanent.getToughness().getValue()));
93          filterCard.add(Predicates.not(new MageObjectReferencePredicate(new MageObjectReference(permanent, game))));
94          this.getTargets().clear();
95          this.addTarget(new TargetCardInYourGraveyard(0, 1, filterCard));
96          return true;
97      }
98  
99      @Override
100     public String getRule() {
101         return "Whenever {this} or another creature you control dies, return up to one other target creature card " +
102                 "with lesser toughness from your graveyard to your hand.";
103     }
104 }

2   package mage.cards.c;
3   
4   import java.util.UUID;
5   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
6   import mage.abilities.common.SimpleStaticAbility;
7   import mage.abilities.effects.common.DontUntapInControllersUntapStepAllEffect;
8   import mage.abilities.effects.common.TapAllEffect;
9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.*;
12  import mage.filter.FilterPermanent;
13  
14  /**
15   *
16   * @author fireshoes
17   */
18  public final class CurseOfMaritLage extends CardImpl {
19      
20      private static final FilterPermanent filter = new FilterPermanent("Islands");
21      
22      static {
23          filter.add(SubType.ISLAND.getPredicate());
24      }    
25  
26      public CurseOfMaritLage(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{3}{R}{R}");
28  
29          // When Curse of Marit Lage enters the battlefield, tap all Islands.
30          this.addAbility(new EntersBattlefieldTriggeredAbility(new TapAllEffect(filter)));
31          
32          // Islands don't untap during their controllers' untap steps.
33          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, 
34                  new DontUntapInControllersUntapStepAllEffect(Duration.WhileOnBattlefield, TargetController.ANY, filter)));
35      }
36  
37      private CurseOfMaritLage(final CurseOfMaritLage card) {
38          super(card);
39      }
40  
41      @Override
42      public CurseOfMaritLage copy() {
43          return new CurseOfMaritLage(this);
44      }
45  }

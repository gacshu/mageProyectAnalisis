33  public final class CunningBandit extends CardImpl {
34  
35      public CunningBandit(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{R}{R}");
37          this.subtype.add(SubType.HUMAN);
38          this.subtype.add(SubType.WARRIOR);
39  
40          this.power = new MageInt(2);
41          this.toughness = new MageInt(2);
42          this.flipCard = true;
43          this.flipCardName = "Azamuki, Treachery Incarnate";
44  
45          // Whenever you cast a Spirit or Arcane spell, you may put a ki counter on Cunning Bandit.
46          this.addAbility(new SpellCastControllerTriggeredAbility(new AddCountersSourceEffect(CounterType.KI.createInstance()), StaticFilters.FILTER_SPIRIT_OR_ARCANE_CARD, true));
47  
48          // At the beginning of the end step, if there are two or more ki counters on Cunning Bandit, you may flip it.
49          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
50                  new OnEventTriggeredAbility(GameEvent.EventType.END_TURN_STEP_PRE, "beginning of the end step", true, new FlipSourceEffect(new AzamukiTreacheryIncarnate()), true),
51                  new SourceHasCounterCondition(CounterType.KI, 2, Integer.MAX_VALUE),
52                  "At the beginning of the end step, if there are two or more ki counters on {this}, you may flip it."));
53      }
54  
55      private CunningBandit(final CunningBandit card) {
56          super(card);
57      }
58  
59      @Override
60      public CunningBandit copy() {
61          return new CunningBandit(this);
62      }
63  }

65  class AzamukiTreacheryIncarnate extends TokenImpl {
66  
67      AzamukiTreacheryIncarnate() {
68          super("Azamuki, Treachery Incarnate", "");
69          addSuperType(SuperType.LEGENDARY);
70          cardType.add(CardType.CREATURE);
71          color.setRed(true);
72          subtype.add(SubType.SPIRIT);
73          power = new MageInt(5);
74          toughness = new MageInt(2);
75  
76          // Remove a ki counter from Azamuki, Treachery Incarnate: Gain control of target creature until end of turn.
77          Ability ability = new SimpleActivatedAbility(
78                  Zone.BATTLEFIELD,
79                  new GainControlTargetEffect(Duration.EndOfTurn),
80                  new RemoveCountersSourceCost(CounterType.KI.createInstance()));
81          ability.addTarget(new TargetCreaturePermanent());
82          this.addAbility(ability);
83      }
84      public AzamukiTreacheryIncarnate(final AzamukiTreacheryIncarnate token) {
85          super(token);
86      }
87  
88      public AzamukiTreacheryIncarnate copy() {
89          return new AzamukiTreacheryIncarnate(this);
90      }

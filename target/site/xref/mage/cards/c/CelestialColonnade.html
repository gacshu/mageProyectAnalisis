22  /**
23   * @author BetaSteward_at_googlemail.com
24   */
25  public final class CelestialColonnade extends CardImpl {
26  
27      public CelestialColonnade(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.LAND}, null);
29  
30          // Celestial Colonnade enters the battlefield tapped.
31          this.addAbility(new EntersBattlefieldTappedAbility());
32  
33          // {T}: Add {W} or {U]
34          this.addAbility(new BlueManaAbility());
35          this.addAbility(new WhiteManaAbility());
36  
37          // {3}{W}{U}: Until end of turn, Celestial Colonnade becomes a 4/4 white and blue Elemental creature with flying and vigilance. It's still a land.
38          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new BecomesCreatureSourceEffect(
39                  new CreatureToken(4, 4, "4/4 white and blue Elemental creature with flying and vigilance")
40                          .withColor("WU")
41                          .withSubType(SubType.ELEMENTAL)
42                          .withAbility(FlyingAbility.getInstance())
43                          .withAbility(VigilanceAbility.getInstance()),
44                  "land", Duration.EndOfTurn), new ManaCostsImpl<>("{3}{W}{U}")));
45      }
46  
47      private CelestialColonnade(final CelestialColonnade card) {
48          super(card);
49      }
50  
51      @Override
52      public CelestialColonnade copy() {
53          return new CelestialColonnade(this);
54      }
55  
56  }

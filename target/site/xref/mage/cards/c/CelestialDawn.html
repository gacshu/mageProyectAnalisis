30  public final class CelestialDawn extends CardImpl {
31  
32      public CelestialDawn(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}{W}");
34  
35          // Lands you control are Plains.
36          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new CelestialDawnToPlainsEffect()));
37  
38          // Nonland cards you own that aren't on the battlefield, spells you control, and nonland permanents you control are white.
39          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new CelestialDawnToWhiteEffect()));
40  
41          // You may spend white mana as though it were mana of any color.
42          // You may spend other mana only as though it were colorless mana.
43          Ability ability = new SimpleStaticAbility(Zone.BATTLEFIELD, new CelestialDawnSpendAnyManaEffect());
44          ability.addEffect(new CelestialDawnSpendColorlessManaEffect());
45          this.addAbility(ability);
46  
47      }
48  
49      private CelestialDawn(final CelestialDawn card) {
50          super(card);
51      }
52  
53      @Override
54      public CelestialDawn copy() {
55          return new CelestialDawn(this);
56      }
57  }

59  class CelestialDawnToPlainsEffect extends ContinuousEffectImpl {
60  
61      private static final FilterLandPermanent filter = new FilterLandPermanent();
62  
63      CelestialDawnToPlainsEffect() {
64          super(Duration.WhileOnBattlefield, Outcome.Detriment);
65          this.staticText = "Lands you control are Plains";
66      }
67  
68      CelestialDawnToPlainsEffect(final CelestialDawnToPlainsEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          return false;
75      }
76  
77      @Override
78      public CelestialDawnToPlainsEffect copy() {
79          return new CelestialDawnToPlainsEffect(this);
80      }
81  
82      @Override
83      public boolean apply(Layer layer, SubLayer sublayer, Ability source, Game game) {
84          for (Permanent land : game.getBattlefield().getAllActivePermanents(filter, source.getControllerId(), game)) {
85              switch (layer) {
86                  case AbilityAddingRemovingEffects_6:
87                      land.removeAllAbilities(source.getSourceId(), game);
88                      land.addAbility(new WhiteManaAbility(), source.getSourceId(), game);
89                      break;
90                  case TypeChangingEffects_4:
91                      land.removeAllSubTypes(game, SubTypeSet.NonBasicLandType);
92                      land.addSubType(game, SubType.PLAINS);
93                      break;
94              }
95          }
96          return true;
97      }
98  
99      @Override
100     public boolean hasLayer(Layer layer) {
101         return layer == Layer.AbilityAddingRemovingEffects_6 || layer == Layer.TypeChangingEffects_4;
102     }
103 
104 }

106 class CelestialDawnToWhiteEffect extends ContinuousEffectImpl {
107 
108     private static final FilterNonlandPermanent filter = new FilterNonlandPermanent();
109 
110     public CelestialDawnToWhiteEffect() {
111         super(Duration.WhileOnBattlefield, Layer.ColorChangingEffects_5, SubLayer.NA, Outcome.Benefit);
112         staticText = "Nonland permanents you control are white. The same is true for spells you control and nonland cards you own that aren't on the battlefield.";
113     }
114 
115     @Override
116     public boolean apply(Game game, Ability source) {
117         Player controller = game.getPlayer(source.getControllerId());
118         if (controller != null) {
119             for (Permanent perm : game.getBattlefield().getAllActivePermanents(filter, source.getControllerId(), game)) {
120                 setColor(perm.getColor(game), game);
121             }
122             // Stack
123             for (MageObject object : game.getStack()) {
124                 if (object instanceof Spell && ((Spell) object).isControlledBy(controller.getId())) {
125                     setColor(object.getColor(game), game);
126                 }
127             }
128             // Exile
129             for (Card card : game.getExile().getAllCards(game)) {
130                 if (card.isOwnedBy(controller.getId())) {
131                     setColor(card.getColor(game), game);
132                 }
133             }
134             // Command
135             for (CommandObject commandObject : game.getState().getCommand()) {
136                 if (commandObject instanceof Commander) {
137                     if (commandObject.isControlledBy(controller.getId())) {
138                         setColor(commandObject.getColor(game), game);
139                     }
140                 }
141             }
142 
143             // Hand
144             for (Card card : controller.getHand().getCards(game)) {
145                 setColor(card.getColor(game), game);
146             }
147             // Library
148             for (Card card : controller.getLibrary().getCards(game)) {
149                 setColor(card.getColor(game), game);
150             }
151             // Graveyard
152             for (Card card : controller.getGraveyard().getCards(game)) {
153                 setColor(card.getColor(game), game);
154             }
155             return true;
156         }
157         return false;
158     }
159 
160     protected static void setColor(ObjectColor color, Game game) {
161         color.setWhite(true);
162         color.setGreen(false);
163         color.setBlue(false);
164         color.setBlack(false);
165         color.setRed(false);
166     }
167 
168     @Override
169     public CelestialDawnToWhiteEffect copy() {
170         return new CelestialDawnToWhiteEffect(this);
171     }
172 
173     private CelestialDawnToWhiteEffect(CelestialDawnToWhiteEffect effect) {
174         super(effect);
175     }
176 
177 }

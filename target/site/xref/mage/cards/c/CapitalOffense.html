24  public final class CapitalOffense extends CardImpl {
25  
26      public CapitalOffense(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{2}{B}{B}");
28  
29          // target creature gets -x/-x until end of turn, where x is the number of times a capital letter appears in its rules text. (ignore reminder text and flavor text.)
30          this.getSpellAbility().addEffect(new BoostTargetEffect(
31                  capitaloffensecount.instance, capitaloffensecount.instance, Duration.EndOfTurn
32          ).setText("target creature gets -x/-x until end of turn, where x is the number of times " +
33                  "a capital letter appears in its rules text. <i>(ignore reminder text and flavor text.)</i>"
34          ));
35          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
36      }
37  
38      private CapitalOffense(final CapitalOffense card) {
39          super(card);
40      }
41  
42      @Override
43      public CapitalOffense copy() {
44          return new CapitalOffense(this);
45      }
46  }
47  
48  enum capitaloffensecount implements DynamicValue {
49      instance;
50  
51      @Override
52      public capitaloffensecount copy() {
53          return instance;
54      }
55  
56      @Override
57      public int calculate(Game game, Ability sourceAbility, Effect effect) {
58          Permanent permanent = game.getPermanent(sourceAbility.getTargets().get(0).getFirstTarget());
59          if (permanent == null) {
60              return 0;
61          }
62          int capitals = 0;
63          List<CardInfo> cards = CardRepository.instance.findCards(permanent.getName());
64  
65          if (cards == null) {
66              return 0;
67          }
68          for (CardInfo cardInfo : cards) {
69              Card dummy = cardInfo != null ? cardInfo.getCard() : null;
70              if (dummy == null) {
71                  return -1 * capitals;
72              }
73              for (String line : dummy.getRules()) {
74                  line = line.replaceAll("(?i)<i.*?</i>", ""); // Ignoring reminder text in italic
75                  line = line.replaceAll("\\{this\\}", permanent.getName());
76                  capitals += line.length() - line.replaceAll("[A-Z]", "").length();
77              }
78              return -1 * capitals;
79          }
80          return 0;
81      }
82  
83      @Override
84      public String getMessage() {
85          return "";
86      }
87  }

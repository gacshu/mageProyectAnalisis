31  public final class CrownOfConvergence extends CardImpl {
32  
33      private static final String rule1 = "As long as the top card of your library is a creature card, creatures you control that share a color with that card get +1/+1";
34  
35      public CrownOfConvergence(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
37  
38          // Play with the top card of your library revealed.
39          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new PlayWithTheTopCardRevealedEffect()));
40  
41          // As long as the top card of your library is a creature card, creatures you control that share a color with that card get +1/+1.
42          ConditionalContinuousEffect effect = new ConditionalContinuousEffect(new CrownOfConvergenceColorBoostEffect(), new TopLibraryCardTypeCondition(CardType.CREATURE), rule1);
43          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
44  
45          // {G}{W}: Put the top card of your library on the bottom of your library.
46          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new CrownOfConvergenceEffect(), new ManaCostsImpl<>("{G}{W}")));
47      }
48  
49      private CrownOfConvergence(final CrownOfConvergence card) {
50          super(card);
51      }
52  
53      @Override
54      public CrownOfConvergence copy() {
55          return new CrownOfConvergence(this);
56      }
57  }

59  class CrownOfConvergenceColorBoostEffect extends BoostAllEffect {
60  
61      private static final String effectText = "creatures you control that share a color with that card get +1/+1";
62  
63      CrownOfConvergenceColorBoostEffect() {
64          super(1, 1, Duration.WhileOnBattlefield, StaticFilters.FILTER_PERMANENT_CREATURE, false);
65          staticText = effectText;
66      }
67  
68      private CrownOfConvergenceColorBoostEffect(CrownOfConvergenceColorBoostEffect effect) {
69          super(effect);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          Player you = game.getPlayer(source.getControllerId());
75          if (you != null) {
76              Card topCard = you.getLibrary().getFromTop(game);
77              if (topCard != null) {
78                  for (Permanent permanent : game.getBattlefield().getActivePermanents(StaticFilters.FILTER_CONTROLLED_CREATURE, source.getControllerId(), source, game)) {
79                      if (permanent.getColor(game).shares(topCard.getColor(game)) && !permanent.getColor(game).isColorless()) {
80                          permanent.addPower(power.calculate(game, source, this));
81                          permanent.addToughness(toughness.calculate(game, source, this));
82                      }
83                  }
84              }
85              return true;
86          }
87          return false;
88      }
89  
90      @Override
91      public CrownOfConvergenceColorBoostEffect copy() {
92          return new CrownOfConvergenceColorBoostEffect(this);
93      }
94  }

96  class CrownOfConvergenceEffect extends OneShotEffect {
97  
98      CrownOfConvergenceEffect() {
99          super(Outcome.Neutral);
100         staticText = "Put the top card of your library on the bottom of your library";
101     }
102 
103     private CrownOfConvergenceEffect(final CrownOfConvergenceEffect effect) {
104         super(effect);
105     }
106 
107     @Override
108     public CrownOfConvergenceEffect copy() {
109         return new CrownOfConvergenceEffect(this);
110     }
111 
112     @Override
113     public boolean apply(Game game, Ability source) {
114         Player controller = game.getPlayer(source.getControllerId());
115         if (controller != null) {
116             Card card = controller.getLibrary().getFromTop(game);
117             if (card != null) {
118                 controller.putCardsOnBottomOfLibrary(new CardsImpl(card), game, source, true);
119             }
120             return true;
121         }
122         return false;
123     }

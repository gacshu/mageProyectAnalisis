25  /**
26   *
27   * @author LevelX2
28   */
29  public final class CentaurVinecrasher extends CardImpl {
30  
31      public CentaurVinecrasher(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{G}");
33          this.subtype.add(SubType.PLANT);
34          this.subtype.add(SubType.CENTAUR);
35          this.power = new MageInt(1);
36          this.toughness = new MageInt(1);
37  
38          // Trample
39          this.addAbility(TrampleAbility.getInstance());
40          // Centaur Vinecrasher enters the battlefield with a number of +1/+1 counters on it equal to the number of land cards in all graveyards.
41          Effect effect = new AddCountersSourceEffect(CounterType.P1P1.createInstance(0), new CardsInAllGraveyardsCount(new FilterLandCard()), true);
42          effect.setText("with a number of +1/+1 counters on it equal to the number of land cards in all graveyards");
43          this.addAbility(new EntersBattlefieldAbility(effect));
44          // Whenever a land card is put into a graveyard from anywhere, you may pay {G}{G}. If you do, return Centaur Vinecrasher from your graveyard to your hand.
45          this.addAbility(new PutCardIntoGraveFromAnywhereAllTriggeredAbility(Zone.GRAVEYARD,
46                  new DoIfCostPaid(new ReturnSourceFromGraveyardToHandEffect(), new ManaCostsImpl<>("{G}{G}")),
47                  false, new FilterLandCard("a land card"), TargetController.ANY, SetTargetPointer.NONE
48          ));
49      }
50  
51      private CentaurVinecrasher(final CentaurVinecrasher card) {
52          super(card);
53      }
54  
55      @Override
56      public CentaurVinecrasher copy() {
57          return new CentaurVinecrasher(this);
58      }
59  }

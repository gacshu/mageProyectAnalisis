38  public final class ChandraHeartOfFire extends CardImpl {
39  
40      public ChandraHeartOfFire(UUID ownerId, CardSetInfo setInfo) {
41          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{3}{R}{R}");
42  
43          this.addSuperType(SuperType.LEGENDARY);
44          this.subtype.add(SubType.CHANDRA);
45          this.setStartingLoyalty(5);
46  
47          // +1: Discard your hand, then exile the top three cards of your library. Until end of turn, you may play cards exiled this way.
48          Ability ability = new LoyaltyAbility(new DiscardHandControllerEffect(), 1);
49          ability.addEffect(new ExileTopXMayPlayUntilEndOfTurnEffect(3).concatBy(", then"));
50          this.addAbility(ability);
51  
52          // +1: Chandra, Heart of Fire deals 2 damage to any target.
53          Ability damageAbility = new LoyaltyAbility(new DamageTargetEffect(2), 1);
54          damageAbility.addTarget(new TargetAnyTarget());
55          this.addAbility(damageAbility);
56  
57          // âˆ’9: Search your graveyard and library for any number of red instant and/or sorcery cards, exile them, then shuffle your library. You may cast them this turn. Add six {R}.
58          Ability ultimateAbility = new LoyaltyAbility(new ChandraHeartOfFireUltimateEffect(), -9);
59          ultimateAbility.addEffect(new BasicManaEffect(Mana.RedMana(6)).setText("Add six {R}"));
60          this.addAbility(ultimateAbility);
61      }

73  class ChandraHeartOfFireUltimateEffect extends OneShotEffect {
74  
75      private static final FilterCard filter = new FilterCard("red instant or sorcery");
76  
77      static {
78          filter.add(new ColorPredicate(ObjectColor.RED));
79          filter.add(Predicates.or(CardType.INSTANT.getPredicate(), CardType.SORCERY.getPredicate()));
80      }
81  
82      ChandraHeartOfFireUltimateEffect() {
83          super(Outcome.Benefit);
84          staticText = "Search your graveyard and library for any number of red instant and/or sorcery cards, exile them, then shuffle. You may cast them this turn";
85      }
86  
87      private ChandraHeartOfFireUltimateEffect(ChandraHeartOfFireUltimateEffect effect) {
88          super(effect);
89      }
90  
91      @Override
92      public ChandraHeartOfFireUltimateEffect copy() {
93          return new ChandraHeartOfFireUltimateEffect(this);
94      }
95  
96      @Override
97      public boolean apply(Game game, Ability source) {
98          Player controller = game.getPlayer(source.getControllerId());
99          if (controller != null) {
100             Set<Card> exiledCards = new HashSet<>();
101 
102             // from graveyard
103             Target target = new TargetCardInYourGraveyard(0, Integer.MAX_VALUE, filter, true).withChooseHint("from graveyard");
104             if (target.canChoose(controller.getId(), source, game)
105                     && target.choose(Outcome.AIDontUseIt, controller.getId(), source.getSourceId(), source, game)) {
106                 Set<Card> cards = new CardsImpl(target.getTargets()).getCards(game);
107                 exiledCards.addAll(cards);
108             }
109 
110             // from library
111             target = new TargetCardInLibrary(0, Integer.MAX_VALUE, filter).withChooseHint("from library");
112             if (target.canChoose(controller.getId(), source, game)
113                     && target.choose(Outcome.AIDontUseIt, controller.getId(), source.getSourceId(), source, game)) {
114                 Set<Card> cards = new CardsImpl(target.getTargets()).getCards(game);
115                 exiledCards.addAll(cards);
116             }
117 
118             // exile cards all at once and set the exile name to the source card
119             controller.moveCardsToExile(exiledCards, source, game, true, CardUtil.getExileZoneId(game, source), CardUtil.getSourceName(game, source));
120             controller.shuffleLibrary(source, game);
121 
122             exiledCards.removeIf(card -> !Zone.EXILED.equals(game.getState().getZone(card.getId())));
123 
124             if (!exiledCards.isEmpty()) {
125                 ContinuousEffect effect = new PlayFromNotOwnHandZoneTargetEffect(Zone.EXILED, Duration.EndOfTurn);
126                 effect.setTargetPointer(new FixedTargets(exiledCards, game));
127                 game.addEffect(effect, source);
128             }
129 
130             return true;
131         }
132         return false;
133     }

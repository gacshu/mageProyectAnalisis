23  public final class CloneShell extends CardImpl {
24  
25      public CloneShell(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{5}");
27          this.subtype.add(SubType.SHAPESHIFTER);
28  
29          this.power = new MageInt(2);
30          this.toughness = new MageInt(2);
31  
32          // Imprint - When Clone Shell enters the battlefield, look at the top four cards of your library, exile one face down, then put the rest on the bottom of your library in any order.
33          this.addAbility(new EntersBattlefieldTriggeredAbility(new CloneShellEffect(), false).setAbilityWord(AbilityWord.IMPRINT));
34  
35          // When Clone Shell dies, turn the exiled card face up. If it's a creature card, put it onto the battlefield under your control.
36          this.addAbility(new DiesSourceTriggeredAbility(new CloneShellDiesEffect()));
37      }
38  
39      private CloneShell(final CloneShell card) {
40          super(card);
41      }
42  
43      @Override
44      public CloneShell copy() {
45          return new CloneShell(this);
46      }
47  }

49  class CloneShellEffect extends OneShotEffect {
50  
51      protected static FilterCard filter1 = new FilterCard("card to exile face down");
52      protected static FilterCard filter2 = new FilterCard("card to put on the bottom of your library");
53  
54      public CloneShellEffect() {
55          super(Outcome.Benefit);
56          staticText = "look at the top four cards of your library, exile one face down, then put the rest on the bottom of your library in any order";
57      }
58  
59      public CloneShellEffect(CloneShellEffect effect) {
60          super(effect);
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          Player controller = game.getPlayer(source.getControllerId());
66          if (controller == null) {
67              return false;
68          }
69          Cards cards = new CardsImpl(controller.getLibrary().getTopCards(game, 4));
70          if (!cards.isEmpty()) {
71              TargetCard target1 = new TargetCard(Zone.LIBRARY, filter1);
72              if (controller.choose(Outcome.Benefit, cards, target1, game)) {
73                  Card card = cards.get(target1.getFirstTarget(), game);
74                  if (card != null) {
75                      cards.remove(card);
76                      controller.moveCardsToExile(card, source, game, false, CardUtil.getCardExileZoneId(game, source), CardUtil.createObjectRealtedWindowTitle(source, game, "(Imprint)"));
77                      card.setFaceDown(true, game);
78                      Permanent permanent = game.getPermanentOrLKIBattlefield(source.getSourceId());
79                      if (permanent != null) {
80                          permanent.imprint(card.getId(), game);
81                      }
82                  }
83                  target1.clearChosen();
84              }
85              controller.putCardsOnBottomOfLibrary(cards, game, source, true);
86          }
87          return true;
88      }
89  
90      @Override
91      public CloneShellEffect copy() {
92          return new CloneShellEffect(this);
93      }
94  
95  }

97  class CloneShellDiesEffect extends OneShotEffect {
98  
99      public CloneShellDiesEffect() {
100         super(Outcome.Benefit);
101         staticText = "turn the exiled card face up. If it's a creature card, put it onto the battlefield under your control";
102     }
103 
104     public CloneShellDiesEffect(CloneShellDiesEffect effect) {
105         super(effect);
106     }
107 
108     @Override
109     public boolean apply(Game game, Ability source) {
110         Player controller = game.getPlayer(source.getControllerId());
111         if (controller != null) {
112             Permanent permanent = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
113             if (permanent != null) {
114                 List<UUID> imprinted = permanent.getImprinted();
115                 if (imprinted != null
116                         && !imprinted.isEmpty()) {
117                     for (UUID imprintedId : imprinted) {
118                         Card imprintedCard = game.getCard(imprintedId);
119                         if (imprintedCard != null) {
120                             imprintedCard.setFaceDown(false, game);
121                             if (imprintedCard.isCreature(game)) {
122                                 controller.moveCards(imprintedCard, Zone.BATTLEFIELD, source, game);
123                             }
124                         }
125                     }
126                 }
127             }
128             return true;
129         }
130         return false;
131     }
132 
133     @Override
134     public CloneShellDiesEffect copy() {
135         return new CloneShellDiesEffect(this);
136     }

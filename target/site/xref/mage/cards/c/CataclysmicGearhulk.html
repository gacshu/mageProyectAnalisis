33  public final class CataclysmicGearhulk extends CardImpl {
34  
35      public CataclysmicGearhulk(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{3}{W}{W}");
37          this.subtype.add(SubType.CONSTRUCT);
38          this.power = new MageInt(4);
39          this.toughness = new MageInt(5);
40  
41          // Vigilance
42          this.addAbility(VigilanceAbility.getInstance());
43  
44          // When Cataclysmic Gearhulk enters the battlefield, each player chooses from among the non-land permanents they control an artifact, a creature,
45          // an enchantment, and a planeswalker, then sacrifices the rest.
46          this.addAbility(new EntersBattlefieldTriggeredAbility(new CataclysmicGearhulkEffect(), false));
47      }
48  
49      private CataclysmicGearhulk(final CataclysmicGearhulk card) {
50          super(card);
51      }
52  
53      @Override
54      public CataclysmicGearhulk copy() {
55          return new CataclysmicGearhulk(this);
56      }
57  }

59  class CataclysmicGearhulkEffect extends OneShotEffect {
60  
61      private static final FilterControlledArtifactPermanent filterArtifact = new FilterControlledArtifactPermanent("artifact");
62      private static final FilterControlledCreaturePermanent filterCreature = new FilterControlledCreaturePermanent("creature");
63      private static final FilterControlledEnchantmentPermanent filterEnchantment = new FilterControlledEnchantmentPermanent("enchantment");
64      private static final FilterControlledPlaneswalkerPermanent filterPlaneswalker = new FilterControlledPlaneswalkerPermanent("planeswalker");
65  
66      static {
67          filterArtifact.add(Predicates.and(
68                  CardType.ARTIFACT.getPredicate(),
69                  Predicates.not(CardType.LAND.getPredicate())));
70          filterCreature.add(Predicates.and(
71                  CardType.CREATURE.getPredicate(),
72                  Predicates.not(CardType.LAND.getPredicate())));
73          filterEnchantment.add(Predicates.and(
74                  CardType.ENCHANTMENT.getPredicate(),
75                  Predicates.not(CardType.LAND.getPredicate())));
76          filterPlaneswalker.add(Predicates.and(
77                  CardType.PLANESWALKER.getPredicate(),
78                  Predicates.not(CardType.LAND.getPredicate())));
79      }
80  
81      public CataclysmicGearhulkEffect() {
82          super(Outcome.DestroyPermanent);
83          staticText = "each player chooses an artifact, a creature, an enchantment, and a planeswalker " +
84                  "from among the nonland permanents they control, then sacrifices the rest";
85      }
86  
87      public CataclysmicGearhulkEffect(CataclysmicGearhulkEffect effect) {
88          super(effect);
89      }
90  
91      @Override
92      public boolean apply(Game game, Ability source) {
93          List<Card> chosen = new ArrayList<>();
94  
95          for (UUID playerId : game.getState().getPlayersInRange(source.getControllerId(), game)) {
96              Player player = game.getPlayer(playerId);
97  
98              Target target1 = new TargetControlledPermanent(1, 1, filterArtifact, true);
99              Target target2 = new TargetControlledPermanent(1, 1, filterCreature, true);
100             Target target3 = new TargetControlledPermanent(1, 1, filterEnchantment, true);
101             Target target4 = new TargetControlledPermanent(1, 1, filterPlaneswalker, true);
102 
103             if (target1.canChoose(player.getId(), source, game)) {
104                 while (player.canRespond() && !target1.isChosen() && target1.canChoose(player.getId(), source, game)) {
105                     player.chooseTarget(Outcome.Benefit, target1, source, game);
106                 }
107                 Permanent artifact = game.getPermanent(target1.getFirstTarget());
108                 if (artifact != null) {
109                     chosen.add(artifact);
110                 }
111                 target1.clearChosen();
112             }
113 
114             if (target2.canChoose(player.getId(), source, game)) {
115                 while (player.canRespond() && !target2.isChosen() && target2.canChoose(player.getId(), source, game)) {
116                     player.chooseTarget(Outcome.Benefit, target2, source, game);
117                 }
118                 Permanent creature = game.getPermanent(target2.getFirstTarget());
119                 if (creature != null) {
120                     chosen.add(creature);
121                 }
122                 target2.clearChosen();
123             }
124 
125             if (target3.canChoose(player.getId(), source, game)) {
126                 while (player.canRespond() && !target3.isChosen() && target3.canChoose(player.getId(), source, game)) {
127                     player.chooseTarget(Outcome.Benefit, target3, source, game);
128                 }
129                 Permanent enchantment = game.getPermanent(target3.getFirstTarget());
130                 if (enchantment != null) {
131                     chosen.add(enchantment);
132                 }
133                 target3.clearChosen();
134             }
135 
136             if (target4.canChoose(player.getId(), source, game)) {
137                 while (player.canRespond() && !target4.isChosen() && target4.canChoose(player.getId(), source, game)) {
138                     player.chooseTarget(Outcome.Benefit, target4, source, game);
139                 }
140                 Permanent planeswalker = game.getPermanent(target4.getFirstTarget());
141                 if (planeswalker != null) {
142                     chosen.add(planeswalker);
143                 }
144                 target4.clearChosen();
145             }
146 
147         }
148 
149         for (Permanent permanent : game.getBattlefield().getAllActivePermanents()) {
150             if (!chosen.contains(permanent) && !permanent.isLand(game)) {
151                 permanent.sacrifice(source, game);
152             }
153         }
154         return true;
155     }
156 
157     @Override
158     public CataclysmicGearhulkEffect copy() {
159         return new CataclysmicGearhulkEffect(this);
160     }
161 }

17      public ChordOfCalling(UUID ownerId, CardSetInfo setInfo) {
18          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{X}{G}{G}{G}");
19  
20          // Convoke (Your creatures can help cast this spell. Each creature you tap while casting this spell pays for {1} or one mana of that creature's color.)
21          this.addAbility(new ConvokeAbility());
22  
23          // Search your library for a creature card with converted mana cost X or less and put it onto the battlefield. Then shuffle your library.
24          this.getSpellAbility().addEffect(new SearchLibraryWithLessCMCPutInPlayEffect(StaticFilters.FILTER_CARD_CREATURE));
25      }

29      }

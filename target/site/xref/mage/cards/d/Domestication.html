27   *
28   * @author jeffwadsworth
29   */
30  public final class Domestication extends CardImpl {
31  
32      public Domestication(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{2}{U}{U}");
34          this.subtype.add(SubType.AURA);
35  
36  
37          // Enchant creature
38          TargetPermanent auraTarget = new TargetCreaturePermanent();
39          this.getSpellAbility().addTarget(auraTarget);
40          this.getSpellAbility().addEffect(new AttachEffect(Outcome.GainControl));
41          Ability ability = new EnchantAbility(auraTarget);
42          this.addAbility(ability);
43          
44          // You control enchanted creature.
45          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new ControlEnchantedEffect()));
46          
47          // At the beginning of your end step, if enchanted creature's power is 4 or greater, sacrifice Domestication.
48          TriggeredAbility ability2 = new BeginningOfYourEndStepTriggeredAbility(new SacrificeSourceEffect(), false);
49          this.addAbility(new ConditionalInterveningIfTriggeredAbility(ability2, new DomesticationCondition(), "At the beginning of your end step, if enchanted creature's power is 4 or greater, sacrifice {this}"));
50      }
51  
52      private Domestication(final Domestication card) {
53          super(card);
54      }
55  
56      @Override
57      public Domestication copy() {
58          return new Domestication(this);
59      }
60  }
61  
62  class DomesticationCondition implements Condition {
63      
64      @Override
65      public boolean apply(Game game, Ability source) {
66          Permanent enchantment = game.getPermanent(source.getSourceId());
67          if (enchantment != null) {
68              Permanent enchanted = game.getPermanent(enchantment.getAttachedTo());
69              if (enchanted != null) {
70                  if (enchanted.getPower().getValue() >= 4) {
71                      return true;
72                  }
73              }
74          }
75          return false;
76      }
77  }

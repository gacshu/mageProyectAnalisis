23  public final class DeadeyeRigHauler extends CardImpl {

25      public DeadeyeRigHauler(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{U}");
27  
28          this.subtype.add(SubType.HUMAN);
29          this.subtype.add(SubType.PIRATE);
30          this.power = new MageInt(3);
31          this.toughness = new MageInt(2);
32  
33          // <i>Raid</i>â€” When Deadeye Rig-Hauler enters the battlefield, if you attacked this turn, you may return target creature to its owner's hand.
34          Ability ability = new ConditionalInterveningIfTriggeredAbility(new EntersBattlefieldTriggeredAbility(new ReturnToHandTargetEffect(), true), RaidCondition.instance,
35                  "When {this} enters the battlefield, if you attacked this turn, you may return target creature to its owner's hand.");
36          ability.addTarget(new TargetCreaturePermanent());
37          ability.setAbilityWord(AbilityWord.RAID);
38          ability.addHint(RaidHint.instance);
39          this.addAbility(ability, new PlayerAttackedWatcher());
40      }
41

43          super(card);
44      }

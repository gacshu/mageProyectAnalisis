31  public final class DihadaBinderOfWills extends CardImpl {
32  
33      private static final FilterCreaturePermanent legendarycreaturefilter = new FilterCreaturePermanent("legendary creature");
34  
35      static {
36          legendarycreaturefilter.add(SuperType.LEGENDARY.getPredicate());
37      }
38  
39  public DihadaBinderOfWills(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{1}{R}{W}{B}");
41          this.addSuperType(SuperType.LEGENDARY);
42          this.subtype.add(SubType.DIHADA);
43  
44          this.setStartingLoyalty(5);
45  
46          // +2: Up to one target legendary creature gains vigilance, lifelink, and indestructible until your next turn.
47          Ability ability = new LoyaltyAbility(new GainAbilityTargetEffect(
48                  VigilanceAbility.getInstance(), Duration.UntilYourNextTurn
49          ).setText("Up to one target legendary creature gains vigilance"), 2);
50          ability.addEffect(new GainAbilityTargetEffect(
51                  LifelinkAbility.getInstance(), Duration.UntilYourNextTurn
52          ).setText(", lifelink"));
53          ability.addEffect(new GainAbilityTargetEffect(
54                  IndestructibleAbility.getInstance(), Duration.UntilYourNextTurn
55          ).setText(", and indestructible until your next turn."));
56          ability.addTarget(new TargetCreaturePermanent(0, 1, legendarycreaturefilter, false));
57          this.addAbility(ability);
58  
59          // -3: Reveal the top four cards of your library.
60          // Put any number of legendary cards from among them into your hand and the rest into your graveyard.
61          // Create a Treasure token for each card put into your graveyard this way.
62          this.addAbility(new LoyaltyAbility(new DihadaFilterEffect(), -3));
63  
64          // -11: Gain control of all nonland permanents until end of turn. Untap them. They gain haste until end of turn.
65          ability = new LoyaltyAbility(new GainControlAllEffect(Duration.EndOfTurn, StaticFilters.FILTER_PERMANENTS_NON_LAND), -11);
66          ability.addEffect(new UntapAllEffect(StaticFilters.FILTER_PERMANENTS_NON_LAND).setText("untap them"));
67          ability.addEffect(new GainAbilityAllEffect(
68                  HasteAbility.getInstance(),
69                  Duration.EndOfTurn,
70                  StaticFilters.FILTER_PERMANENTS_NON_LAND,
71                  "they gain haste until end of turn"
72          ));
73          this.addAbility(ability);
74          
75          // Dihada, Binder of Wills can be your commander.
76          this.addAbility(CanBeYourCommanderAbility.getInstance());
77      }
78  
79      private DihadaBinderOfWills(final DihadaBinderOfWills card) {
80          super(card);
81      }
82  
83      @Override
84      public DihadaBinderOfWills copy() {
85          return new DihadaBinderOfWills(this);
86      }
87  }

89  class DihadaFilterEffect extends RevealLibraryPickControllerEffect {
90  
91      private static final FilterCard legendaryfilter = new FilterCard("legendary cards");
92  
93      static {
94          legendaryfilter.add(SuperType.LEGENDARY.getPredicate());
95      }
96  
97      public DihadaFilterEffect() {
98          super(4, Integer.MAX_VALUE, legendaryfilter, PutCards.HAND, PutCards.GRAVEYARD, false);
99          staticText = "Reveal the top four cards of your library. " +
100                 "Put any number of legendary cards from among them into your hand and the rest into your graveyard. " +
101                 "Create a Treasure token for each card put into your graveyard this way";
102     }
103 
104     private DihadaFilterEffect(final DihadaFilterEffect effect) {
105         super(effect);
106     }
107 
108     @Override
109     public DihadaFilterEffect copy() {
110         return new DihadaFilterEffect(this);
111     }
112 
113     @Override
114     protected boolean actionWithPickedCards(Game game, Ability source, Player player, Cards pickedCards, Cards otherCards) {
115         super.actionWithPickedCards(game, source, player, pickedCards, otherCards);
116         otherCards.retainZone(Zone.GRAVEYARD, game);
117         if (!otherCards.isEmpty()) {
118             new TreasureToken().putOntoBattlefield(otherCards.size(), game, source, player.getId());
119         }
120         return true;
121     }

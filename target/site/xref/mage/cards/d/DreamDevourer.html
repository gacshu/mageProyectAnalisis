30  public final class DreamDevourer extends CardImpl {
31  
32      public DreamDevourer(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{B}");
34  
35          this.subtype.add(SubType.DEMON);
36          this.subtype.add(SubType.CLERIC);
37          this.power = new MageInt(0);
38          this.toughness = new MageInt(3);
39  
40          // Each nonland card in your hand without foretell has foretell. Its foretell cost is equal to its mana cost reduced by 2.
41          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, new DreamDevourerAddAbilityEffect()));
42  
43          // Whenever you foretell a card, Dream Devourer gets +2/+0 until end of turn.
44          this.addAbility(new ForetellSourceControllerTriggeredAbility(new BoostSourceEffect(2, 0, Duration.EndOfTurn)));
45  
46      }
47  
48      private DreamDevourer(final DreamDevourer card) {
49          super(card);
50      }
51  
52      @Override
53      public DreamDevourer copy() {
54          return new DreamDevourer(this);
55      }
56  }
57  
58  class DreamDevourerAddAbilityEffect extends ContinuousEffectImpl {
59  
60      private static final FilterNonlandCard filter = new FilterNonlandCard();
61  
62      static {
63          filter.add(Predicates.not(new AbilityPredicate(ForetellAbility.class)));
64      }
65  
66      DreamDevourerAddAbilityEffect() {
67          super(Duration.WhileOnBattlefield, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
68          this.staticText = "Each nonland card in your hand without foretell has foretell. Its foretell cost is equal to its mana cost reduced by {2}";
69      }
70  
71      private DreamDevourerAddAbilityEffect(final DreamDevourerAddAbilityEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public DreamDevourerAddAbilityEffect copy() {
77          return new DreamDevourerAddAbilityEffect(this);
78      }
79  
80      @Override
81      public boolean apply(Game game, Ability source) {
82          Player controller = game.getPlayer(source.getControllerId());
83          if (controller == null) {
84              return false;
85          }
86          for (Card card : controller.getHand().getCards(filter, game)) {
87              ForetellAbility foretellAbility = null;
88              if (card instanceof SplitCard) {
89                  String leftHalfCost = CardUtil.reduceCost(((SplitCard) card).getLeftHalfCard().getManaCost(), 2).getText();
90                  String rightHalfCost = CardUtil.reduceCost(((SplitCard) card).getRightHalfCard().getManaCost(), 2).getText();
91                  foretellAbility = new ForetellAbility(card, leftHalfCost, rightHalfCost);
92              } else if (card instanceof ModalDoubleFacesCard) {
93                  ModalDoubleFacesCardHalf leftHalfCard = ((ModalDoubleFacesCard) card).getLeftHalfCard();
94                  // If front side of MDFC is land, do nothing as Dream Devourer does not apply to lands
95                  // MDFC cards in hand are considered lands if front side is land
96                  if (!leftHalfCard.isLand(game)) {
97                      String leftHalfCost = CardUtil.reduceCost(leftHalfCard.getManaCost(), 2).getText();
98                      ModalDoubleFacesCardHalf rightHalfCard = ((ModalDoubleFacesCard) card).getRightHalfCard();
99                      if (rightHalfCard.isLand(game)) {
100                         foretellAbility = new ForetellAbility(card, leftHalfCost);
101                     } else {
102                         String rightHalfCost = CardUtil.reduceCost(rightHalfCard.getManaCost(), 2).getText();
103                         foretellAbility = new ForetellAbility(card, leftHalfCost, rightHalfCost);
104                     }
105                 }
106             } else if (card instanceof AdventureCard) {
107                 String creatureCost = CardUtil.reduceCost(card.getMainCard().getManaCost(), 2).getText();
108                 String spellCost = CardUtil.reduceCost(((AdventureCard) card).getSpellCard().getManaCost(), 2).getText();
109                 foretellAbility = new ForetellAbility(card, creatureCost, spellCost);
110             } else {
111                 String costText = CardUtil.reduceCost(card.getManaCost(), 2).getText();
112                 foretellAbility = new ForetellAbility(card, costText);
113             }
114             if (foretellAbility != null) {
115                 foretellAbility.setSourceId(card.getId());
116                 foretellAbility.setControllerId(card.getOwnerId());
117                 game.getState().addOtherAbility(card, foretellAbility);
118             }
119         }
120         return true;
121     }
122 }

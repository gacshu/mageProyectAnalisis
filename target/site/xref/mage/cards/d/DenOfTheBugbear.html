24  /**
25   *
26   * @author weirddan455
27   */
28  public final class DenOfTheBugbear extends CardImpl {
29  
30      private static final FilterLandPermanent filter = new FilterLandPermanent();
31  
32      static {
33          filter.add(AnotherPredicate.instance);
34      }
35  
36      private static final Condition condition = new PermanentsOnTheBattlefieldCondition(filter, ComparisonType.MORE_THAN, 1, true);
37  
38      public DenOfTheBugbear(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.LAND}, "");
40  
41          // If you control two or more other lands, Den of the Bugbear enters the battlefield tapped.
42          this.addAbility(new EntersBattlefieldAbility(
43                  new TapSourceEffect(), condition, "If you control two or more other lands, {this} enters the battlefield tapped.", null
44          ));
45  
46          // {T}: Add {R}.
47          this.addAbility(new RedManaAbility());
48  
49          // {3}{R}: Until end of turn, Den of the Bugbear becomes a 3/2 red Goblin creature with
50          // "Whenever this creature attacks, create a 1/1 red Goblin creature token that's tapped and attacking." It's still a land.
51          Ability ability = new AttacksTriggeredAbility(
52                  new CreateTokenEffect(new GoblinToken(), 1, true, true), false,
53                  "Whenever this creature attacks, create a 1/1 red Goblin creature token that's tapped and attacking."
54          );
55          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new BecomesCreatureSourceEffect(
56                  new CreatureToken(3, 2, "3/2 red Goblin creature with \"Whenever this creature attacks, create a 1/1 red Goblin creature token that's tapped and attacking\"")
57                          .withColor("R")
58                          .withSubType(SubType.GOBLIN)
59                          .withAbility(ability),
60                  "land", Duration.EndOfTurn), new ManaCostsImpl<>("{3}{R}")));
61      }
62  
63      private DenOfTheBugbear(final DenOfTheBugbear card) {
64          super(card);
65      }
66  
67      @Override
68      public DenOfTheBugbear copy() {
69          return new DenOfTheBugbear(this);
70      }
71  }

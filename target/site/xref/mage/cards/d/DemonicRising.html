2   package mage.cards.d;

4   import java.util.UUID;
5   import mage.abilities.TriggeredAbility;
6   import mage.abilities.common.BeginningOfYourEndStepTriggeredAbility;
7   import mage.abilities.condition.common.CreatureCountCondition;
8   import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
9   import mage.abilities.effects.common.CreateTokenEffect;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.TargetController;
14  import mage.game.permanent.token.DemonToken;
15  
16  /**
17   * @author noxx
18   */
19  public final class DemonicRising extends CardImpl {
20  
21      private static final String ruleText = "At the beginning of your end step, if you control exactly one creature, create a 5/5 black Demon creature token with flying";
22  
23      public DemonicRising(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId,setInfo,new CardType[]{CardType.ENCHANTMENT},"{3}{B}{B}");
25  
26          // At the beginning of your end step, if you control exactly one creature, create a 5/5 black Demon creature token with flying.
27          TriggeredAbility ability = new BeginningOfYourEndStepTriggeredAbility(new CreateTokenEffect(new DemonToken()), false);
28          this.addAbility(new ConditionalInterveningIfTriggeredAbility(ability, new CreatureCountCondition(1, TargetController.YOU), ruleText));
29      }
30  
31      private DemonicRising(final DemonicRising card) {
32          super(card);
33      }
34  
35      @Override
36      public DemonicRising copy() {
37          return new DemonicRising(this);
38      }
39  }

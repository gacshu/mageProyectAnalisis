10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;

16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class DireMimic extends CardImpl {
23  
24      public DireMimic(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{2}");
26  
27          this.subtype.add(SubType.TREASURE);
28  
29          // Flash
30          this.addAbility(FlashAbility.getInstance());
31  
32          // {T}, Sacrifice Dire Mimic: Add one mana of any color.
33          Ability ability = new AnyColorManaAbility();
34          ability.addCost(new SacrificeSourceCost());
35          this.addAbility(ability);
36  
37          // {3}: Dire Mimic becomes a Shapeshifter artifact creature with base power and toughness 5/5 until end of turn.
38          this.addAbility(new SimpleActivatedAbility(new BecomesCreatureSourceEffect(
39                  new CreatureToken(
40                          5, 5, "Shapeshifter artifact creature " +
41                          "with base power and toughness 5/5", SubType.SHAPESHIFTER
42                  ).withType(CardType.ARTIFACT), "", Duration.EndOfTurn
43          ), new GenericManaCost(3)));
44      }
45  
46      private DireMimic(final DireMimic card) {
47          super(card);
48      }
49  
50      @Override
51      public DireMimic copy() {
52          return new DireMimic(this);
53      }
54  }

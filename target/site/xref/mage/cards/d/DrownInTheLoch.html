8   import mage.cards.CardSetInfo;
9   import mage.constants.CardType;
10  import mage.filter.FilterPermanent;
11  import mage.filter.FilterSpell;
12  import mage.filter.common.FilterCreaturePermanent;
13  import mage.filter.predicate.Predicate;
14  import mage.game.Game;
15  import mage.players.Player;
16  import mage.target.TargetPermanent;
17  import mage.target.TargetSpell;
18  
19  import java.util.UUID;
20  
21  /**
22   * @author TheElk801
23   */
24  public final class DrownInTheLoch extends CardImpl {
25  
26      private static final FilterSpell filter
27              = new FilterSpell("spell with mana value less than or equal to " +
28              "the number of cards in its controller's graveyard");
29      private static final FilterPermanent filter2
30              = new FilterCreaturePermanent("creature with mana value less than or equal to " +
31              "the number of cards in its controller's graveyard");
32  
33      static {
34          filter.add(DrownInTheLochPredicate.instance);
35          filter2.add(DrownInTheLochPredicate.instance);
36      }
37  
38      public DrownInTheLoch(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{U}{B}");
40  
41          // Choose one —
42          // • Counter target spell with converted mana cost less than or equal to the number of cards in its controller's graveyard.
43          this.getSpellAbility().addEffect(new CounterTargetEffect());
44          this.getSpellAbility().addTarget(new TargetSpell(filter));
45  
46          // • Destroy target creature with converted mana cost less than or equal to the number of cards in its controller's graveyard.
47          Mode mode = new Mode(new DestroyTargetEffect());
48          mode.addTarget(new TargetPermanent(filter2));
49          this.getSpellAbility().addMode(mode);
50      }
51  
52      private DrownInTheLoch(final DrownInTheLoch card) {
53          super(card);
54      }
55  
56      @Override
57      public DrownInTheLoch copy() {
58          return new DrownInTheLoch(this);
59      }
60  }
61  
62  enum DrownInTheLochPredicate implements Predicate<MageObject> {
63      instance;
64  
65      @Override
66      public boolean apply(MageObject input, Game game) {
67          Player player = game.getPlayer(game.getControllerId(input.getId()));
68          return player != null && input.getManaValue() <= player.getGraveyard().size();
69      }
70  }

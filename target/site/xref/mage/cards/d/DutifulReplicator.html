10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;
12  import mage.constants.SubType;

14  import mage.filter.common.FilterControlledPermanent;
15  import mage.filter.predicate.Predicates;
16  import mage.filter.predicate.mageobject.NamePredicate;
17  import mage.filter.predicate.permanent.TokenPredicate;
18  import mage.target.TargetPermanent;
19  
20  import java.util.UUID;
21  
22  /**
23   * @author TheElk801
24   */
25  public final class DutifulReplicator extends CardImpl {
26  
27      private static final FilterPermanent filter
28              = new FilterControlledPermanent("token you control not named Dutiful Replicator");
29  
30      static {
31          filter.add(TokenPredicate.TRUE);
32          filter.add(Predicates.not(new NamePredicate("Dutiful Replicator")));
33      }
34  
35      public DutifulReplicator(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{3}");
37  
38          this.subtype.add(SubType.ASSEMBLY_WORKER);
39          this.power = new MageInt(3);
40          this.toughness = new MageInt(2);
41  
42          // When Dutiful Replicator enters the battlefield, you may pay {1}. When you do, create a token that's a copy of target token you control not named Dutiful Replicator.
43          ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
44                  new CreateTokenCopyTargetEffect(), false
45          );
46          ability.addTarget(new TargetPermanent(filter));
47          this.addAbility(new EntersBattlefieldTriggeredAbility(
48                  new DoWhenCostPaid(ability, new GenericManaCost(1), "Pay {1}?")
49          ));
50      }
51  
52      private DutifulReplicator(final DutifulReplicator card) {
53          super(card);
54      }
55  
56      @Override
57      public DutifulReplicator copy() {
58          return new DutifulReplicator(this);
59      }
60  }

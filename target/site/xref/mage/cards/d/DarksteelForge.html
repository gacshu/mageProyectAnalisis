3   package mage.cards.d;
4   
5   import java.util.UUID;
6   import mage.abilities.common.SimpleStaticAbility;
7   import mage.abilities.effects.Effect;
8   import mage.abilities.effects.common.continuous.GainAbilityAllEffect;
9   import mage.abilities.keyword.IndestructibleAbility;
10  import mage.cards.CardImpl;
11  import mage.cards.CardSetInfo;
12  import mage.constants.CardType;
13  import mage.constants.Duration;
14  import mage.constants.Zone;
15  import mage.filter.common.FilterControlledPermanent;
16  
17  /**
18   * @author Loki
19   */
20  public final class DarksteelForge extends CardImpl {
21  
22      private static final FilterControlledPermanent filter = new FilterControlledPermanent("Artifacts you control");
23  
24      static {
25          filter.add(CardType.ARTIFACT.getPredicate());
26      }
27  
28      public DarksteelForge(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId,setInfo,new CardType[]{CardType.ARTIFACT},"{9}");
30  
31          // Artifacts you control are indestructible.
32          Effect effect = new GainAbilityAllEffect(IndestructibleAbility.getInstance(), Duration.WhileOnBattlefield, filter, false);
33          effect.setText("Artifacts you control are indestructible");
34          this.addAbility(new SimpleStaticAbility(Zone.BATTLEFIELD, effect));
35  
36      }
37  
38      private DarksteelForge(final DarksteelForge card) {
39          super(card);
40      }
41  
42      @Override
43      public DarksteelForge copy() {
44          return new DarksteelForge(this);
45      }
46  
47  }

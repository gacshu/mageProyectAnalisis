27  public final class DarkSalvation extends CardImpl {
28  
29      public DarkSalvation(UUID ownerId, CardSetInfo setInfo) {
30          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{X}{B}");
31  
32          // Target player creates X 2/2 black Zombie creature tokens, then up to one target creature gets -1/-1 until end of turn for each Zombie that player controls.
33          this.getSpellAbility().addTarget(new TargetPlayer());
34          this.getSpellAbility().addEffect(new CreateTokenTargetEffect(new ZombieToken(), ManacostVariableValue.REGULAR));
35  
36          this.getSpellAbility().addTarget(new TargetCreaturePermanent(0, 1));
37          Effect effect = new BoostTargetEffect(
38                  ZombiesControlledByTargetPlayerCount.instance,
39                  ZombiesControlledByTargetPlayerCount.instance,
40                  Duration.EndOfTurn
41          );
42          effect.setTargetPointer(new SecondTargetPointer());
43          effect.setText(", then up to one target creature gets -1/-1 until end of turn for each Zombie that player controls");
44          this.getSpellAbility().addEffect(effect);
45      }
46  
47      private DarkSalvation(final DarkSalvation card) {
48          super(card);
49      }
50  
51      @Override
52      public DarkSalvation copy() {
53          return new DarkSalvation(this);
54      }
55  }
56  
57  enum ZombiesControlledByTargetPlayerCount implements DynamicValue {
58      instance;
59  
60      private static final FilterPermanent filter = new FilterPermanent(SubType.ZOMBIE, "Zombie");
61  
62      @Override
63      public int calculate(Game game, Ability sourceAbility, Effect effect) {
64          Player player = game.getPlayer(sourceAbility.getTargets().get(0).getFirstTarget());
65          if (player != null) {
66              int value = game.getBattlefield().countAll(filter, player.getId(), game);
67              return -1 * value;
68          } else {
69              return 0;
70          }
71      }
72  
73      @Override
74      public ZombiesControlledByTargetPlayerCount copy() {
75          return instance;
76      }
77  
78      @Override
79      public String toString() {
80          return "-1";
81      }
82  
83      @Override
84      public String getMessage() {
85          return "Zombie that player controls";
86      }
87  
88      @Override
89      public int getSign() {
90          return -1;
91      }
92  }

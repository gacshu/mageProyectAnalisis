28  public final class DelifsCube extends CardImpl {
29  
30      public DelifsCube(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT}, "{1}");
32  
33          // {2}, {tap}: This turn, when target creature you control attacks and isn't blocked, it assigns no combat damage this turn and you put a cube counter on Delif's Cube.
34          Ability ability = new SimpleActivatedAbility(
35                  new CreateDelayedTriggeredAbilityEffect(new DelifsCubeTriggeredAbility()), new GenericManaCost(2)
36          );
37          ability.addCost(new TapSourceCost());
38          ability.addTarget(new TargetControlledCreaturePermanent());
39          this.addAbility(ability);
40  
41          // {2}, Remove a cube counter from Delif's Cube: Regenerate target creature.
42          ability = new SimpleActivatedAbility(new RegenerateTargetEffect(), new GenericManaCost(2));
43          ability.addCost(new RemoveCountersSourceCost(CounterType.CUBE.createInstance()));
44          ability.addTarget(new TargetControlledCreaturePermanent());
45          this.addAbility(ability);
46      }
47  
48      private DelifsCube(final DelifsCube card) {
49          super(card);
50      }
51  
52      @Override
53      public DelifsCube copy() {
54          return new DelifsCube(this);
55      }
56  }
57  
58  class DelifsCubeTriggeredAbility extends DelayedTriggeredAbility {
59  
60      DelifsCubeTriggeredAbility() {
61          super(new AssignNoCombatDamageTargetEffect(Duration.EndOfTurn, "it assigns no combat damage this turn"), Duration.EndOfTurn, false, false);
62          this.addEffect(new AddCountersSourceEffect(CounterType.CUBE.createInstance()));
63      }
64  
65      private DelifsCubeTriggeredAbility(final DelifsCubeTriggeredAbility ability) {
66          super(ability);
67      }
68  
69      @Override
70      public boolean checkEventType(GameEvent event, Game game) {
71          return event.getType() == GameEvent.EventType.UNBLOCKED_ATTACKER;
72      }
73  
74      @Override
75      public boolean checkTrigger(GameEvent event, Game game) {
76          return event.getTargetId().equals(getFirstTarget());
77      }
78  
79      @Override
80      public DelifsCubeTriggeredAbility copy() {
81          return new DelifsCubeTriggeredAbility(this);
82      }
83  
84      @Override
85      public String getRule() {
86          return "This turn, when target creature you control attacks and isn't blocked, " +
87                  "it assigns no combat damage this turn and you put a cube counter on {this}.";
88      }
89  }

35  public final class DanceOfMany extends CardImpl {
36  
37      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent("nontoken creature");
38  
39      static {
40          filter.add(TokenPredicate.FALSE);
41      }
42  
43      public DanceOfMany(UUID ownerId, CardSetInfo setInfo) {
44          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{U}{U}");
45  
46          // When Dance of Many enters the battlefield, create a token that's a copy of target nontoken creature.
47          Ability ability = new EntersBattlefieldTriggeredAbility(new DanceOfManyCreateTokenCopyEffect(), false);
48          ability.addTarget(new TargetCreaturePermanent(filter));
49          this.addAbility(ability);
50  
51          // When Dance of Many leaves the battlefield, exile the token.
52          // When the token leaves the battlefield, sacrifice Dance of Many.
53          Ability ability2 = new LeavesBattlefieldTriggeredAbility(new DanceOfManyExileTokenEffect(), false);
54          ability2.addEffect(new InfoEffect("When the token leaves the battlefield, sacrifice {this}"));
55          this.addAbility(ability2);
56  
57          // At the beginning of your upkeep, sacrifice Dance of Many unless you pay {U}{U}.
58          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new SacrificeSourceUnlessPaysEffect(new ManaCostsImpl<>("{U}{U}")), TargetController.YOU, false));
59  
60      }
61  
62      private DanceOfMany(final DanceOfMany card) {
63          super(card);
64      }
65  
66      @Override
67      public DanceOfMany copy() {
68          return new DanceOfMany(this);
69      }
70  }

72  class DanceOfManyCreateTokenCopyEffect extends OneShotEffect {
73  
74      public DanceOfManyCreateTokenCopyEffect() {
75          super(Outcome.PutCreatureInPlay);
76          staticText = "create a token that's a copy of target nontoken creature";
77      }
78  
79      public DanceOfManyCreateTokenCopyEffect(final DanceOfManyCreateTokenCopyEffect effect) {
80          super(effect);
81      }
82  
83      @Override
84      public boolean apply(Game game, Ability source) {
85          // The ability is targeted and checks the validity of the target when put on the stack and when resolving.
86          // If the creature is not still there when the copy ability resolves, the ability doesnâ€™t resolve and no token
87          // is put onto the battlefield. This card remains on the battlefield as an enchantment with no token.
88          // (2004-10-04)
89          Permanent permanent = game.getPermanent(getTargetPointer().getFirst(game, source));
90          Permanent sourceObject = game.getPermanent(source.getSourceId());
91          if (permanent != null && sourceObject != null) {
92  
93              CreateTokenCopyTargetEffect effect = new CreateTokenCopyTargetEffect();
94              effect.setTargetPointer(new FixedTarget(permanent, game));
95              effect.apply(game, source);
96              game.getState().setValue(source.getSourceId() + "_token", effect.getAddedPermanents());
97              for (Permanent addedToken : effect.getAddedPermanents()) {
98                  Effect sacrificeEffect = new SacrificeTargetEffect("sacrifice Dance of Many");
99                  sacrificeEffect.setTargetPointer(new FixedTarget(sourceObject, game));
100                 LeavesBattlefieldTriggeredAbility triggerAbility = new LeavesBattlefieldTriggeredAbility(sacrificeEffect, false);
101                 ContinuousEffect continuousEffect = new GainAbilityTargetEffect(triggerAbility, Duration.WhileOnBattlefield);
102                 continuousEffect.setTargetPointer(new FixedTarget(addedToken, game));
103                 game.addEffect(continuousEffect, source);
104             }
105             return true;
106         }
107         return false;
108     }
109 
110     @Override
111     public DanceOfManyCreateTokenCopyEffect copy() {
112         return new DanceOfManyCreateTokenCopyEffect(this);
113     }
114 }

116 class DanceOfManyExileTokenEffect extends OneShotEffect {
117 
118     public DanceOfManyExileTokenEffect() {
119         super(Outcome.Removal);
120         staticText = "exile the token";
121     }
122 
123     public DanceOfManyExileTokenEffect(final DanceOfManyExileTokenEffect effect) {
124         super(effect);
125     }
126 
127     @Override
128     public boolean apply(Game game, Ability source) {
129         Player controller = game.getPlayer(source.getControllerId());
130         if (controller != null) {
131             List<Permanent> tokenPermanents = (List<Permanent>) game.getState().getValue(source.getSourceId() + "_token");
132             if (tokenPermanents != null) {
133                 Cards cards = new CardsImpl();
134                 for (Permanent permanent : tokenPermanents) {
135                     cards.add(permanent);
136                 }
137                 controller.moveCards(cards, Zone.EXILED, source, game);
138                 return true;
139             }
140         }
141         return false;
142     }
143 
144     @Override
145     public DanceOfManyExileTokenEffect copy() {
146         return new DanceOfManyExileTokenEffect(this);
147     }

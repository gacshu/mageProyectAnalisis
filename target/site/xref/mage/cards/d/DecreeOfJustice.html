25  public final class DecreeOfJustice extends CardImpl {
26  
27      public DecreeOfJustice(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{X}{2}{W}{W}");
29  
30          // Create X 4/4 white Angel creature tokens with flying.
31          this.getSpellAbility().addEffect(new CreateTokenEffect(new AngelToken(), ManacostVariableValue.REGULAR));
32  
33          // Cycling {2}{W}
34          this.addAbility(new CyclingAbility(new ManaCostsImpl<>("{2}{W}")));
35  
36          // When you cycle Decree of Justice, you may pay {X}. If you do, create X 1/1 white Soldier creature tokens.
37          Ability ability = new CycleTriggeredAbility(new DecreeOfJusticeCycleEffect());
38          this.addAbility(ability);
39      }
40  
41      private DecreeOfJustice(final DecreeOfJustice card) {
42          super(card);
43      }
44  
45      @Override
46      public DecreeOfJustice copy() {
47          return new DecreeOfJustice(this);
48      }
49  }
50  
51  class DecreeOfJusticeCycleEffect extends OneShotEffect {
52  
53      DecreeOfJusticeCycleEffect() {
54          super(Outcome.Benefit);
55          this.staticText = "you may pay {X}. If you do, create X 1/1 white Soldier creature tokens";
56      }
57  
58      private DecreeOfJusticeCycleEffect(final DecreeOfJusticeCycleEffect effect) {
59          super(effect);
60      }
61  
62      @Override
63      public DecreeOfJusticeCycleEffect copy() {
64          return new DecreeOfJusticeCycleEffect(this);
65      }
66  
67      @Override
68      public boolean apply(Game game, Ability source) {
69          Player player = game.getPlayer(source.getControllerId());
70          if (player == null || !player.chooseUse(Outcome.Benefit, "Pay {X} to create X tokens?", source, game)) {
71              return false;
72          }
73          int payCount = ManaUtil.playerPaysXGenericMana(true, "Decree of Justice", player, source, game);
74          if (payCount > 0) {
75              return new SoldierToken().putOntoBattlefield(payCount, game, source, source.getControllerId());
76          }
77          return false;
78      }
79  }

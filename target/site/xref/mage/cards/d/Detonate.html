9   import mage.cards.CardImpl;
10  import mage.cards.CardSetInfo;
11  import mage.constants.CardType;

13  import mage.filter.common.FilterArtifactPermanent;

18  
19  import java.util.UUID;
20  
21  /**
22   * @author LoneFox
23   */
24  public final class Detonate extends CardImpl {
25  
26      public Detonate(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{X}{R}");
28  
29          // Destroy target artifact with converted mana cost X. It can't be regenerated. Detonate deals X damage to that artifact's controller.
30          this.getSpellAbility().addEffect(new DestroyTargetEffect(true));
31          this.getSpellAbility().addTarget(new TargetArtifactPermanent(new FilterArtifactPermanent("artifact with mana value X")));
32          Effect effect = new DamageTargetControllerEffect(ManacostVariableValue.REGULAR);
33          effect.setText("{this} deals X damage to that artifact's controller");
34          this.getSpellAbility().addEffect(effect);
35          this.getSpellAbility().setTargetAdjuster(DetonateAdjuster.instance);
36      }
37  
38      private Detonate(final Detonate card) {
39          super(card);
40      }
41  
42      @Override
43      public Detonate copy() {
44          return new Detonate(this);
45      }
46  }
47  
48  enum DetonateAdjuster implements TargetAdjuster {
49      instance;
50  
51      @Override
52      public void adjustTargets(Ability ability, Game game) {
53          ability.getTargets().clear();
54          int xValue = ability.getManaCostsToPay().getX();
55          FilterArtifactPermanent filter = new FilterArtifactPermanent("artifact with mana value X");
56          filter.add(new ManaValuePredicate(ComparisonType.EQUAL_TO, xValue));
57          ability.addTarget(new TargetArtifactPermanent(filter));
58      }
59  }

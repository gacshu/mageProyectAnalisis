32  public final class DuskmantleGuildmage extends CardImpl {
33  
34      public DuskmantleGuildmage(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{U}{B}");
36          this.subtype.add(SubType.HUMAN);
37          this.subtype.add(SubType.WIZARD);
38  
39          this.power = new MageInt(2);
40          this.toughness = new MageInt(2);
41  
42          // {1}{U}{B}: Whenever a card is put into an opponent's graveyard from anywhere this turn, that player loses 1 life.
43          this.addAbility(new SimpleActivatedAbility(Zone.BATTLEFIELD, new CreateDelayedTriggeredAbilityEffect(new CardPutIntoOpponentGraveThisTurn()), new ManaCostsImpl<>("{1}{U}{B}")));
44  
45          // {2}{U}{B}: Target player puts the top two cards of their library into their graveyard.
46          Ability ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new PutLibraryIntoGraveTargetEffect(2), new ManaCostsImpl<>("{2}{U}{B}"));
47          ability.addTarget(new TargetPlayer());
48          this.addAbility(ability);
49      }
50  
51      private DuskmantleGuildmage(final DuskmantleGuildmage card) {
52          super(card);
53      }
54  
55      @Override
56      public DuskmantleGuildmage copy() {
57          return new DuskmantleGuildmage(this);
58      }
59  }

61  class CardPutIntoOpponentGraveThisTurn extends DelayedTriggeredAbility {
62  
63      public CardPutIntoOpponentGraveThisTurn() {
64          super(new LoseLifeTargetEffect(1), Duration.EndOfTurn, false);
65      }
66  
67      public CardPutIntoOpponentGraveThisTurn(final CardPutIntoOpponentGraveThisTurn ability) {
68          super(ability);
69      }
70  
71      @Override
72      public CardPutIntoOpponentGraveThisTurn copy() {
73          return new CardPutIntoOpponentGraveThisTurn(this);
74      }
75  
76      @Override
77      public boolean checkEventType(GameEvent event, Game game) {
78          return event.getType() == GameEvent.EventType.ZONE_CHANGE;
79      }
80  
81      @Override
82      public boolean checkTrigger(GameEvent event, Game game) {
83          ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
84          UUID cardId = event.getTargetId();
85          Card card = game.getCard(cardId);
86          if (card != null && zEvent.getToZone() == Zone.GRAVEYARD && !card.isCopy()
87                  && game.getOpponents(controllerId).contains(card.getOwnerId())) {
88              for (Effect effect : this.getEffects()) {
89                  effect.setTargetPointer(new FixedTarget(card.getOwnerId()));
90              }
91              return true;
92          }
93          return false;
94      }
95  
96      @Override
97      public String getRule() {
98          return "Whenever a card is put into an opponent's graveyard from anywhere this turn, that player loses 1 life";
99      }

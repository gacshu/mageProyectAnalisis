12  import mage.cards.CardImpl;

14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  
17  import java.util.UUID;
18  
19  /**
20   * @author TheElk801
21   */
22  public final class DoomWeaver extends CardImpl {
23  
24      private static final DynamicValue xValue = new SourcePermanentPowerCount();
25  
26      public DoomWeaver(UUID ownerId, CardSetInfo setInfo) {
27          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{B}{B}");
28  
29          this.subtype.add(SubType.SPIDER);
30          this.subtype.add(SubType.HORROR);
31          this.power = new MageInt(1);
32          this.toughness = new MageInt(8);
33  
34          // Reach
35          this.addAbility(ReachAbility.getInstance());
36  
37          // Soulbond
38          this.addAbility(new SoulbondAbility());
39  
40          // As long as Doom Weaver is paired with another creature, each of those creatures has "When this creature dies, draw cards equal to its power."
41          this.addAbility(new SimpleStaticAbility(new GainAbilityPairedEffect(new DiesSourceTriggeredAbility(
42                  new DrawCardSourceControllerEffect(xValue).setText("draw cards equal to its power")
43          ).setTriggerPhrase("When this creature dies, "), "As long as {this} is paired with another creature, " +
44                  "each of those creatures has \"When this creature dies, draw cards equal to its power.\"")));
45      }
46  
47      private DoomWeaver(final DoomWeaver card) {
48          super(card);
49      }
50  
51      @Override
52      public DoomWeaver copy() {
53          return new DoomWeaver(this);
54      }
55  }

35  public final class Duplicity extends CardImpl {
36  
37      public Duplicity(UUID ownerId, CardSetInfo setInfo) {
38          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{U}{U}");
39  
40          // When Duplicity enters the battlefield, exile the top five cards of your library face down.
41          this.addAbility(new EntersBattlefieldTriggeredAbility(new DuplicityEffect(), false));
42  
43          // At the beginning of your upkeep, you may exile all cards from your hand face down. If you do, put all other cards you own exiled with Duplicity into your hand.
44          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new DuplicityExileHandEffect(), TargetController.YOU, true));
45  
46          // At the beginning of your end step, discard a card.
47          this.addAbility(new BeginningOfYourEndStepTriggeredAbility(new DiscardControllerEffect(1), false));
48  
49          // When you lose control of Duplicity, put all cards exiled with Duplicity into their owner's graveyard.
50          this.addAbility(new DuplicityEntersBattlefieldAbility(new CreateDelayedTriggeredAbilityEffect(new LoseControlDuplicity())));
51  
52      }
53  
54      private Duplicity(final Duplicity card) {
55          super(card);
56      }
57  
58      @Override
59      public Duplicity copy() {
60          return new Duplicity(this);
61      }
62  }

64  class DuplicityEffect extends OneShotEffect {
65  
66      public DuplicityEffect() {
67          super(Outcome.Exile);
68          staticText = "exile the top five cards of your library face down";
69      }
70  
71      public DuplicityEffect(final DuplicityEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          Player controller = game.getPlayer(source.getControllerId());
78          MageObject sourceObject = game.getObject(source);
79          if (controller != null
80                  && sourceObject != null) {
81              if (controller.getLibrary().hasCards()) {
82                  UUID exileId = source.getSourceId();
83                  Set<Card> cardsToExile = controller.getLibrary().getTopCards(game, 5);
84                  for (Card card : cardsToExile) {
85                      controller.moveCardsToExile(card, source, game, true, exileId, sourceObject.getName());
86                      card.setFaceDown(true, game);
87                  }
88              }
89              return true;
90          }
91          return false;
92      }
93  
94      @Override
95      public DuplicityEffect copy() {
96          return new DuplicityEffect(this);
97      }
98  }

100 class DuplicityExileHandEffect extends OneShotEffect {
101 
102     public DuplicityExileHandEffect() {
103         super(Outcome.Exile);
104         staticText = "you may exile all cards from your hand face down. If you do, put all other cards you own exiled with {this} into your hand";
105     }
106 
107     public DuplicityExileHandEffect(final DuplicityExileHandEffect effect) {
108         super(effect);
109     }
110 
111     @Override
112     public boolean apply(Game game, Ability source) {
113         Player controller = game.getPlayer(source.getControllerId());
114         MageObject sourceObject = game.getObject(source);
115         if (controller != null
116                 && sourceObject != null) {
117             if (!controller.getHand().isEmpty()) {
118                 UUID exileId = source.getSourceId();
119                 Set<Card> cardsFromHandToExile = controller.getHand().getCards(game);
120                 for (Card card : cardsFromHandToExile) {
121                     controller.moveCardsToExile(card, source, game, true, exileId, sourceObject.getName());
122                     card.setFaceDown(true, game);
123                 }
124                 Set<Card> cardsInExile = game.getExile().getExileZone(exileId).getCards(game);
125                 Set<Card> cardsToReturnToHandFromExile = new HashSet<>();
126                 for (Card card : cardsInExile) {
127                     if (!cardsFromHandToExile.contains(card)) {
128                         cardsToReturnToHandFromExile.add(card);
129                     }
130                 }
131                 controller.moveCards(cardsToReturnToHandFromExile, Zone.HAND, source, game);
132             }
133             return true;
134         }
135         return false;
136     }
137 
138     @Override
139     public DuplicityExileHandEffect copy() {
140         return new DuplicityExileHandEffect(this);
141     }
142 }

144 class LoseControlDuplicity extends DelayedTriggeredAbility {
145 
146     public LoseControlDuplicity() {
147         super(new PutExiledCardsInOwnersGraveyard(), Duration.EndOfGame, false);
148     }
149 
150     public LoseControlDuplicity(final LoseControlDuplicity ability) {
151         super(ability);
152     }
153 
154     @Override
155     public LoseControlDuplicity copy() {
156         return new LoseControlDuplicity(this);
157     }
158 
159     @Override
160     public boolean checkEventType(GameEvent event, Game game) {
161         return event.getType() == GameEvent.EventType.LOST_CONTROL
162                 || event.getType() == GameEvent.EventType.ZONE_CHANGE;
163     }
164 
165     @Override
166     public boolean checkTrigger(GameEvent event, Game game) {
167         if (event.getType() == GameEvent.EventType.LOST_CONTROL
168                 && event.getPlayerId().equals(controllerId)
169                 && event.getTargetId().equals(this.getSourceId())) {
170             return true;
171         }
172         if (event.getType() == GameEvent.EventType.ZONE_CHANGE
173                 && event.getTargetId().equals(this.getSourceId())
174                 && ((ZoneChangeEvent) event).getToZone() != Zone.BATTLEFIELD) {
175             return true;
176         }
177         return false;
178     }
179 
180     @Override
181     public String getRule() {
182         return "When you lose control of {this}, put all cards exiled with {this} into their owner's graveyard.";
183     }
184 }

186 class PutExiledCardsInOwnersGraveyard extends OneShotEffect {
187 
188     public PutExiledCardsInOwnersGraveyard() {
189         super(Outcome.Neutral);
190         staticText = " put all cards exiled with {this} into their owner's graveyard.";
191     }
192 
193     public PutExiledCardsInOwnersGraveyard(final PutExiledCardsInOwnersGraveyard effect) {
194         super(effect);
195     }
196 
197     @Override
198     public boolean apply(Game game, Ability source) {
199         Player controller = game.getPlayer(source.getControllerId());
200         if (controller != null) {
201             UUID exileId = source.getSourceId();
202             Set<Card> cardsInExile = game.getExile().getExileZone(exileId).getCards(game);
203             if (cardsInExile != null) {
204                 controller.moveCards(cardsInExile, Zone.GRAVEYARD, source, game);
205                 return true;
206             }
207         }
208         return false;
209     }
210 
211     @Override
212     public PutExiledCardsInOwnersGraveyard copy() {
213         return new PutExiledCardsInOwnersGraveyard(this);
214     }

217 class DuplicityEntersBattlefieldAbility extends StaticAbility {
218     
219     public DuplicityEntersBattlefieldAbility(Effect effect) {
220         super(Zone.ALL, new EntersBattlefieldEffect(effect, null, null, true, false));
221     }
222 
223     public DuplicityEntersBattlefieldAbility(final DuplicityEntersBattlefieldAbility ability) {
224         super(ability);
225     }
226 
227     @Override
228     public void addEffect(Effect effect) {
229         if (!getEffects().isEmpty()) {
230             Effect entersBattlefieldEffect = this.getEffects().get(0);
231             if (entersBattlefieldEffect instanceof EntersBattlefieldEffect) {
232                 ((EntersBattlefieldEffect) entersBattlefieldEffect).addEffect(effect);
233                 return;
234             }
235         }
236         super.addEffect(effect);
237     }
238 
239     @Override
240     public DuplicityEntersBattlefieldAbility copy() {
241         return new DuplicityEntersBattlefieldAbility(this);
242     }

31  public final class DetentionSphere extends CardImpl {
32  
33      static final protected FilterPermanent filter = new FilterNonlandPermanent("nonland permanent not named Detention Sphere");
34  
35      static {
36          filter.add(Predicates.not(new NamePredicate("Detention Sphere")));
37      }
38  
39      public DetentionSphere(UUID ownerId, CardSetInfo setInfo) {
40          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{W}{U}");
41  
42          // When Detention Sphere enters the battlefield, you may exile
43          // target nonland permanent not named Detention Sphere and all
44          // other permanents with the same name as that permanent.
45          Ability ability = new EntersBattlefieldTriggeredAbility(new DetentionSphereEntersEffect(), true);
46          ability.addTarget(new TargetPermanent(filter));
47          this.addAbility(ability);
48  
49          // When Detention Sphere leaves the battlefield, return the exiled
50          // cards to the battlefield under their owner's control.
51          this.addAbility(new LeavesBattlefieldTriggeredAbility(new DetentionSphereLeavesEffect(), false));
52      }
53  
54      private DetentionSphere(final DetentionSphere card) {
55          super(card);
56      }
57  
58      @Override
59      public DetentionSphere copy() {
60          return new DetentionSphere(this);
61      }
62  }

64  class DetentionSphereEntersEffect extends OneShotEffect {
65  
66      public DetentionSphereEntersEffect() {
67          super(Outcome.Exile);
68          staticText = "you may exile target nonland permanent not named Detention Sphere and all other permanents with the same name as that permanent";
69      }
70  
71      public DetentionSphereEntersEffect(final DetentionSphereEntersEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public boolean apply(Game game, Ability source) {
77          UUID exileId = CardUtil.getExileZoneId(game, source.getSourceId(), source.getSourceObjectZoneChangeCounter());
78          Permanent targetPermanent = game.getPermanent(getTargetPointer().getFirst(game, source));
79          Player controller = game.getPlayer(source.getControllerId());
80          MageObject sourceObject = game.getObject(source);
81          if (sourceObject != null && exileId != null && targetPermanent != null && controller != null) {
82  
83              if (CardUtil.haveEmptyName(targetPermanent)) { // face down creature
84                  controller.moveCardToExileWithInfo(targetPermanent, exileId, sourceObject.getIdName(), source, game, Zone.BATTLEFIELD, true);
85              } else {
86                  String name = targetPermanent.getName();
87                  for (Permanent permanent : game.getBattlefield().getActivePermanents(source.getControllerId(), game)) {
88                      if (permanent != null && CardUtil.haveSameNames(permanent, name, game)) {
89                          controller.moveCardToExileWithInfo(permanent, exileId, sourceObject.getIdName(), source, game, Zone.BATTLEFIELD, true);
90                      }
91                  }
92              }
93              return true;
94          }
95          return false;
96      }
97  
98      @Override
99      public DetentionSphereEntersEffect copy() {
100         return new DetentionSphereEntersEffect(this);
101     }
102 }

104 class DetentionSphereLeavesEffect extends OneShotEffect {
105 
106     public DetentionSphereLeavesEffect() {
107         super(Outcome.Neutral);
108         staticText = "return the exiled cards to the battlefield under their owner's control";
109     }
110 
111     public DetentionSphereLeavesEffect(final DetentionSphereLeavesEffect effect) {
112         super(effect);
113     }
114 
115     @Override
116     public boolean apply(Game game, Ability source) {
117         Player controller = game.getPlayer(source.getControllerId());
118         MageObject sourceObject = source.getSourceObject(game);
119         if (sourceObject != null && controller != null) {
120             Permanent permanentLeftBattlefield = (Permanent) getValue("permanentLeftBattlefield");
121             if (permanentLeftBattlefield == null) {
122                 Logger.getLogger(ReturnFromExileForSourceEffect.class).error("Permanent not found: " + sourceObject.getName());
123                 return false;
124             }
125             ExileZone exile = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source.getSourceId(), permanentLeftBattlefield.getZoneChangeCounter(game)));
126             if (exile != null) {
127                 controller.moveCards(exile.getCards(game), Zone.BATTLEFIELD, source, game, false, false, true, null);
128             }
129             return true;
130         }
131         return false;
132     }
133 
134     @Override
135     public DetentionSphereLeavesEffect copy() {
136         return new DetentionSphereLeavesEffect(this);
137     }
138 }

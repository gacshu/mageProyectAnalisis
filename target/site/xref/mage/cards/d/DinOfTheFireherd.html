28  public final class DinOfTheFireherd extends CardImpl {
29  
30      public DinOfTheFireherd(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{5}{B/R}{B/R}{B/R}");
32  
33          // Create a 5/5 black and red Elemental creature token. Target opponent sacrifices a creature for each black creature you control, then sacrifices a land for each red creature you control.
34          this.getSpellAbility().addEffect(new DinOfTheFireherdEffect());
35          this.getSpellAbility().addTarget(new TargetOpponent());
36      }
37  
38      private DinOfTheFireherd(final DinOfTheFireherd card) {
39          super(card);
40      }
41  
42      @Override
43      public DinOfTheFireherd copy() {
44          return new DinOfTheFireherd(this);
45      }
46  }
47  
48  class DinOfTheFireherdEffect extends OneShotEffect {
49  
50      private static final FilterControlledCreaturePermanent blackCreatureFilter = new FilterControlledCreaturePermanent("black creatures you control");
51      private static final FilterControlledCreaturePermanent redCreatureFilter = new FilterControlledCreaturePermanent("red creatures you control");
52  
53      static {
54          blackCreatureFilter.add(new ColorPredicate(ObjectColor.BLACK));
55          redCreatureFilter.add(new ColorPredicate(ObjectColor.RED));
56      }
57  
58      public DinOfTheFireherdEffect() {
59          super(Outcome.Neutral);
60          this.staticText = "create a 5/5 black and red Elemental creature token. Target opponent sacrifices a creature for each black creature you control, then sacrifices a land for each red creature you control";
61      }
62  
63      public DinOfTheFireherdEffect(final DinOfTheFireherdEffect effect) {
64          super(effect);
65      }
66  
67      @Override
68      public DinOfTheFireherdEffect copy() {
69          return new DinOfTheFireherdEffect(this);
70      }
71  
72      @Override
73      public boolean apply(Game game, Ability source) {
74          boolean applied;
75          Token token = new DinOfTheFireherdToken();
76          applied = token.putOntoBattlefield(1, game, source, source.getControllerId());
77  
78          int blackCreaturesControllerControls = game.getBattlefield().countAll(blackCreatureFilter, source.getControllerId(), game);
79          int redCreaturesControllerControls = game.getBattlefield().countAll(redCreatureFilter, source.getControllerId(), game);
80  
81          Player targetOpponent = game.getPlayer(targetPointer.getFirst(game, source));
82          if (targetOpponent != null) {
83              Effect effect = new SacrificeEffect(new FilterControlledCreaturePermanent(), blackCreaturesControllerControls, "Target Opponent");
84              effect.setTargetPointer(new FixedTarget(targetOpponent.getId()));
85              effect.apply(game, source);
86  
87              Effect effect2 = new SacrificeEffect(new FilterControlledLandPermanent(), redCreaturesControllerControls, "Target Opponent");
88              effect2.setTargetPointer(new FixedTarget(targetOpponent.getId()));
89              effect2.apply(game, source);
90              applied = true;
91          }
92          return applied;
93      }
94  }

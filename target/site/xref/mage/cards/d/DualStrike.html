1   package mage.cards.d;
2   
3   import mage.abilities.common.delayed.CopyNextSpellDelayedTriggeredAbility;
4   import mage.abilities.effects.common.CreateDelayedTriggeredAbilityEffect;
5   import mage.abilities.keyword.ForetellAbility;
6   import mage.cards.CardImpl;
7   import mage.cards.CardSetInfo;
8   import mage.constants.CardType;
9   import mage.constants.ComparisonType;
10  import mage.filter.FilterSpell;
11  import mage.filter.common.FilterInstantOrSorcerySpell;
12  import mage.filter.predicate.mageobject.ManaValuePredicate;
13  
14  import java.util.UUID;
15  
16  /**
17   * @author TheElk801
18   */
19  public final class DualStrike extends CardImpl {
20  
21      private static final FilterSpell filter
22              = new FilterInstantOrSorcerySpell("instant or sorcery spell with mana value 4 or less");
23  
24      static {
25          filter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 5));
26      }
27  
28      public DualStrike(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{R}{R}");
30  
31          // When you cast your next instant or sorcery spell with converted mana cost 4 or less this turn, copy that spell. You may choose new targets for the copy.
32          this.getSpellAbility().addEffect(new CreateDelayedTriggeredAbilityEffect(
33                  new CopyNextSpellDelayedTriggeredAbility(filter)
34          ));
35  
36          // Foretell {R}
37          this.addAbility(new ForetellAbility(this, "{R}"));
38      }
39  
40      private DualStrike(final DualStrike card) {
41          super(card);
42      }
43  
44      @Override
45      public DualStrike copy() {
46          return new DualStrike(this);
47      }
48  }

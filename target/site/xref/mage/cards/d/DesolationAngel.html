2   package mage.cards.d;

4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
7   import mage.abilities.condition.common.KickedCondition;
8   import mage.abilities.decorator.ConditionalOneShotEffect;
9   import mage.abilities.effects.common.DestroyAllEffect;
10  import mage.abilities.keyword.FlyingAbility;
11  import mage.abilities.keyword.KickerAbility;
12  import mage.cards.CardImpl;
13  import mage.cards.CardSetInfo;
14  import mage.constants.CardType;
15  import mage.constants.SubType;
16  import mage.constants.TargetController;
17  import mage.filter.StaticFilters;
18  import mage.filter.common.FilterLandPermanent;
19  
20  /**
21   * @author Loki
22   */
23  public final class DesolationAngel extends CardImpl {
24  
25      private static final FilterLandPermanent filter2 = new FilterLandPermanent("lands you control");
26  
27      static {
28          filter2.add(TargetController.YOU.getControllerPredicate());
29      }
30  
31      public DesolationAngel(UUID ownerId, CardSetInfo setInfo) {
32          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{B}{B}");
33          this.subtype.add(SubType.ANGEL);
34  
35          this.power = new MageInt(5);
36          this.toughness = new MageInt(4);
37  
38          // Kicker (You may pay an additional as you cast this spell.)
39          this.addAbility(new KickerAbility("{W}{W}"));
40  
41          // Flying
42          this.addAbility(FlyingAbility.getInstance());
43  
44          // When Desolation Angel enters the battlefield, destroy all lands you control. If it was kicked, destroy all lands instead.
45          this.addAbility(new EntersBattlefieldTriggeredAbility(new ConditionalOneShotEffect(new DestroyAllEffect(StaticFilters.FILTER_LANDS),
46                  new DestroyAllEffect(filter2), KickedCondition.ONCE, "destroy all lands you control. If it was kicked, destroy all lands instead.")));
47      }
48  
49      private DesolationAngel(final DesolationAngel card) {
50          super(card);
51      }
52  
53      @Override
54      public DesolationAngel copy() {
55          return new DesolationAngel(this);
56      }
57  }

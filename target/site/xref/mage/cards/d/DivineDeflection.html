21  public final class DivineDeflection extends CardImpl {

23      public DivineDeflection(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{X}{W}");
25  
26  
27          // Prevent the next X damage that would be dealt to you and/or permanents you control this turn. If damage is prevented this way, Divine Deflection deals that much damage to any target.
28          this.getSpellAbility().addEffect(new DivineDeflectionPreventDamageTargetEffect(Duration.EndOfTurn));
29          this.getSpellAbility().addTarget(new TargetAnyTarget());
30      }
31

33          super(card);
34      }
35

37      public DivineDeflection copy() {
38          return new DivineDeflection(this);
39      }
40  }

42  class DivineDeflectionPreventDamageTargetEffect extends PreventionEffectImpl {
43  
44      public DivineDeflectionPreventDamageTargetEffect(Duration duration) {
45          super(duration, Integer.MIN_VALUE, false, true);
46          staticText = "Prevent the next X damage that would be dealt to you and/or permanents you control this turn. If damage is prevented this way, {this} deals that much damage to any target";
47      }
48  
49      public DivineDeflectionPreventDamageTargetEffect(final DivineDeflectionPreventDamageTargetEffect effect) {
50          super(effect);
51      }
52  
53      @Override
54      public DivineDeflectionPreventDamageTargetEffect copy() {
55          return new DivineDeflectionPreventDamageTargetEffect(this);
56      }
57  
58      @Override
59      public boolean apply(Game game, Ability source) {
60          return true;
61      }
62  
63      @Override
64      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
65          /*
66          If damage is dealt to multiple permanents you control, or is dealt to you and at least
67          one permanent you control, you choose which of that damage to prevent if the chosen value
68          for X won't prevent all the damage. For example, if 3 damage would be dealt to you and to
69          each of two creatures you control, and Divine Deflection will prevent the next 3 damage,
70          you might choose to prevent the next 2 damage it would deal to you and the next 1 damage
71          it would deal to one of the creatures, among other choices. You don't decide until the
72          point at which the damage would be dealt.
73          TODO: Support to select which damage to prevent
74          */
75  
76          PreventionEffectData preventionData = preventDamageAction(event, source, game);
77          /*
78          Divine Deflection's effect is not a redirection effect. If it prevents damage,
79          Divine Deflection (not the original source) deals damage to the targeted creature
80          or player as part of that prevention effect. Divine Deflection is the source of
81          the new damage, so the characteristics of the original source (such as its color,
82          or whether it had lifelink or deathtouch) don't affect this damage. The new damage
83          is not combat damage, even if the prevented damage was. Since you control the source
84          of the new damage, if you targeted an opponent with Divine Deflection, you may
85          have Divine Deflection deal its damage to a planeswalker that opponent controls.
86          */
87          // deal damage now
88          int prevented = preventionData.getPreventedDamage();
89          if (prevented > 0) {
90              UUID dealDamageTo = source.getFirstTarget();
91              /*
92            	Whether the targeted creature or player is still a legal target is not checked after
93              Divine Deflection resolves. For example, if a creature targeted by Divine Deflection
94              gains shroud after Divine Deflection resolves, Divine Deflection can still deal damage
95              to that creature.
96              */
97  
98              Permanent permanent = game.getPermanent(dealDamageTo);
99              if (permanent != null) {
100                 game.informPlayers("Dealing " + prevented + " to " + permanent.getName() + " instead");
101                 permanent.damage(prevented, source.getSourceId(), source, game, false, true);
102             }
103             Player player = game.getPlayer(dealDamageTo);
104             if (player != null) {
105                 game.informPlayers("Dealing " + prevented + " to " + player.getLogName() + " instead");
106                 player.damage(prevented, source.getSourceId(), source, game);
107             }
108         }
109         return false;
110     }
111 
112     @Override
113     public boolean applies(GameEvent event, Ability source, Game game) {
114         if (!this.used && super.applies(event, source, game)) {
115             if (amountToPrevent == Integer.MIN_VALUE) {
116                 amountToPrevent = source.getManaCostsToPay().getX();
117             }
118             //   check permanent first
119             Permanent permanent = game.getPermanent(event.getTargetId());
120             if (permanent != null) {
121                 if (permanent.isControlledBy(source.getControllerId())) {
122                     return true;
123                 }
124             }
125             //   check player
126             return source.isControlledBy(event.getTargetId());
127         }
128         return false;
129     }

16  
17  /**
18   *
19   * @author LevelX2
20   */
21  public final class DiplomacyOfTheWastes extends CardImpl {
22  
23      public DiplomacyOfTheWastes(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId,setInfo,new CardType[]{CardType.SORCERY},"{2}{B}");
25  
26          // Target opponent reveals their hand. You choose a nonland card from it. That player discards that card. If you control a Warrior, that player loses 2 life.
27          this.getSpellAbility().addTarget(new TargetOpponent());
28          this.getSpellAbility().addEffect(new DiscardCardYouChooseTargetEffect(new FilterNonlandCard()));
29          this.getSpellAbility().addEffect(new ConditionalOneShotEffect(
30                  new LoseLifeTargetEffect(2), 
31                  new PermanentsOnTheBattlefieldCondition(new FilterControlledCreaturePermanent(SubType.WARRIOR, "Warrior")),
32                  "If you control a Warrior, that player loses 2 life"));
33  
34      }
35  
36      private DiplomacyOfTheWastes(final DiplomacyOfTheWastes card) {
37          super(card);
38      }
39  
40      @Override
41      public DiplomacyOfTheWastes copy() {
42          return new DiplomacyOfTheWastes(this);
43      }
44  }

31  /**
32   * 5/1/2012 	For each despair counter on Descent into Madness, you'll exile a permanent
33   * you control or exile a card from your hand, not both.
34   * 5/1/2012 	First you choose the permanents and/or cards from your hand that will be
35   * exiled. Then each other player in turn order does the same. Then all the chosen permanents
36   * and cards are exiled simultaneously. Players who choose after you will know what permanents
37   * you'll be exiling when they choose. They'll know how many cards you'll be exiling from
38   * your hand, but they won't see those cards.
39   * 5/1/2012 	If there are more counters on Descent into Madness than the total number of
40   * permanents you control plus the number of cards in your hand, you'll exile all permanents
41   * you control (including Descent into Madness) and all cards from your hand.
42   * 5/1/2012 	If Descent into Madness isn't on the battlefield when its ability resolves,
43   * use the number of counters on it when it left the battlefield to determine how many permanents
44   * and/or cards from hands to exile.
45   *
46   * @author noxx
47   */
48  public final class DescentIntoMadness extends CardImpl {
49  
50      public DescentIntoMadness(UUID ownerId, CardSetInfo setInfo) {
51          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{B}{B}");
52  
53  
54          // At the beginning of your upkeep, put a despair counter on Descent into Madness, then each player exiles X permanents they control and/or cards from their hand, where X is the number of despair counters on Descent into Madness.
55          this.addAbility(new BeginningOfUpkeepTriggeredAbility(new DescentIntoMadnessEffect(), TargetController.YOU, false));
56      }

68  class DescentIntoMadnessEffect extends OneShotEffect {
69  
70      public DescentIntoMadnessEffect() {
71          super(Outcome.Sacrifice);
72          this.staticText = "put a despair counter on {this}, then each player exiles X permanents they control and/or cards from their hand, where X is the number of despair counters on {this}";
73      }
74  
75      public DescentIntoMadnessEffect(final DescentIntoMadnessEffect effect) {
76          super(effect);
77      }
78  
79      @Override
80      public DescentIntoMadnessEffect copy() {
81          return new DescentIntoMadnessEffect(this);
82      }
83  
84      @Override
85      public boolean apply(Game game, Ability source) {
86          Player controller = game.getPlayer(source.getControllerId());
87          Permanent sourcePermanent = game.getPermanent(source.getSourceId());
88          if (sourcePermanent != null && controller != null) {
89              sourcePermanent.addCounters(CounterType.DESPAIR.createInstance(), source.getControllerId(), source, game);
90          }
91          if (sourcePermanent == null) {
92              sourcePermanent = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
93          }
94          if (sourcePermanent != null && controller != null) {
95              int count = sourcePermanent.getCounters(game).getCount(CounterType.DESPAIR);
96              if (count > 0) {
97                  // select the permanents and hand cards in turn order
98                  LinkedList<UUID> selectedObjects = new LinkedList<>();
99  
100                 // ask all players
101                 for (UUID playerId : game.getState().getPlayersInRange(controller.getId(), game)) {
102                     Player currentPlayer = game.getPlayer(playerId);
103                     if (currentPlayer != null && currentPlayer.canRespond()) {
104                         selectCards(currentPlayer, selectedObjects, count, source, game);
105                     }
106                 }
107 
108                 // move permanents and hand cards to exile
109                 for (UUID objectId : selectedObjects) {
110                     if (game.getState().getZone(objectId) == Zone.BATTLEFIELD) {
111                         Permanent permanent = game.getPermanent(objectId);
112                         if (permanent != null) {
113                             Player player = game.getPlayer(permanent.getControllerId());
114                             if (player != null) {
115                                 player.moveCardToExileWithInfo(permanent, null, "", source, game, Zone.BATTLEFIELD, true);
116                             }
117                         }
118                     } else if (game.getState().getZone(objectId) == Zone.HAND) {
119                         Card card = game.getCard(objectId);
120                         if (card != null) {
121                             Player player = game.getPlayer(card.getOwnerId());
122                             if (player != null) {
123                                 player.moveCardToExileWithInfo(card, null, "", source, game, Zone.HAND, true);
124                             }
125                         }
126                     }
127                 }
128 
129             }
130             return true;
131         }
132         return false;
133     }

135     private void selectCards(Player player, List<UUID> selectedObjects, int count, Ability source, Game game) {
136         int amount = Math.min(count, player.getHand().size() + game.getBattlefield().getAllActivePermanents(player.getId()).size());
137         int cardsFromHand = 0;
138 
139         while (player.canRespond() && amount > 0) {
140 
141             Target target;
142             do {
143                 FilterControlledPermanent filter = new FilterControlledPermanent();
144                 filter.setMessage("permanent you control (" + amount + " left in total)");
145                 List<PermanentIdPredicate> uuidPredicates = new ArrayList<>();
146                 for (UUID uuid : selectedObjects) {
147                     uuidPredicates.add(new PermanentIdPredicate(uuid));
148                 }
149                 filter.add(Predicates.not(Predicates.or(uuidPredicates)));
150 
151                 target = new TargetControlledPermanent(0, 1, filter, true);
152                 if (target.canChoose(player.getId(), source, game)
153                         && player.choose(Outcome.Exile, target, source, game)) {
154                     for (UUID targetId : target.getTargets()) {
155                         if (!selectedObjects.contains(targetId)) {
156                             Permanent chosen = game.getPermanent(targetId);
157                             if (chosen != null) {
158                                 amount--;
159                                 game.informPlayers(player.getLogName() + " selects " + chosen.getLogName() + " from battlefield");
160                                 selectedObjects.add(targetId);
161                             }
162                         }
163                     }
164                 }
165             } while (amount > 0 && !target.getTargets().isEmpty() && player.canRespond());
166             if (amount > 0) {
167                 TargetCard targetInHand;
168                 do {
169                     FilterCard filterInHand = new FilterCard();
170                     filterInHand.setMessage("card from your hand (" + amount + " left in total)");
171                     targetInHand = new TargetCard(0, 1, Zone.HAND, filterInHand);
172                     List<CardIdPredicate> uuidPredicates = new ArrayList<>();
173                     for (UUID uuid : selectedObjects) {
174                         uuidPredicates.add(new CardIdPredicate(uuid));
175                     }
176                     filterInHand.add(Predicates.not(Predicates.or(uuidPredicates)));
177                     if (targetInHand.canChoose(player.getId(), source, game) &&
178                             player.choose(Outcome.Exile, player.getHand(), targetInHand, game)) {
179 
180                         Card card = player.getHand().get(targetInHand.getFirstTarget(), game);
181                         if (card != null) {
182                             selectedObjects.add(targetInHand.getFirstTarget());
183                             amount--;
184                             cardsFromHand++;
185                         }
186                     }
187                 } while (amount > 0 && !targetInHand.getTargets().isEmpty() && player.canRespond());
188 
189             }
190         }
191         if (cardsFromHand > 0) {
192             game.informPlayers(player.getLogName() + " selects " + cardsFromHand + (cardsFromHand == 1 ? " card" : " cards") + " from their hand");
193         }
194     }

19  import java.util.UUID;

21  /**
22   * @author jeffwadsworth
23   */
24  public final class DeclarationOfNaught extends CardImpl {
25  
26      static final private FilterSpell filter = new FilterSpell("spell with the chosen name");
27  
28      public DeclarationOfNaught(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{U}{U}");
30  
31          // As Declaration of Naught enters the battlefield, name a card.
32          this.addAbility(new AsEntersBattlefieldAbility(new ChooseACardNameEffect(ChooseACardNameEffect.TypeOfName.ALL)));
33  
34          // {U}: Counter target spell with the chosen name.
35          SimpleActivatedAbility ability = new SimpleActivatedAbility(Zone.BATTLEFIELD, new CounterTargetEffect(), new ManaCostsImpl<>("{U}"));
36          ability.setTargetAdjuster(DeclarationOfNaughtAdjuster.instance);
37          ability.addTarget(new TargetSpell(filter));
38          this.addAbility(ability);
39      }
40  
41      private DeclarationOfNaught(final DeclarationOfNaught card) {
42          super(card);
43      }
44  
45      @Override
46      public DeclarationOfNaught copy() {
47          return new DeclarationOfNaught(this);
48      }
49  }
50  
51  enum DeclarationOfNaughtAdjuster implements TargetAdjuster {
52      instance;
53  
54      @Override
55      public void adjustTargets(Ability ability, Game game) {
56          ability.getTargets().clear();
57          String chosenName = (String) game.getState().getValue(ability.getSourceId().toString() + ChooseACardNameEffect.INFO_KEY);
58          if (chosenName == null) {
59              return;
60          }
61          FilterSpell filterSpell = new FilterSpell("spell named " + chosenName);
62          filterSpell.add(new NamePredicate(chosenName));
63          TargetSpell target = new TargetSpell(1, filterSpell);
64          ability.addTarget(target);
65      }
66  }

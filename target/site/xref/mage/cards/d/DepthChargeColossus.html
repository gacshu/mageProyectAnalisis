20  public final class DepthChargeColossus extends CardImpl {
21  
22      public DepthChargeColossus(UUID ownerId, CardSetInfo setInfo) {
23          super(ownerId, setInfo, new CardType[]{CardType.ARTIFACT, CardType.CREATURE}, "{9}");
24  
25          this.subtype.add(SubType.DREADNOUGHT);
26          this.power = new MageInt(9);
27          this.toughness = new MageInt(9);
28  
29          // Prototype {4}{U}{U} -- 6/6
30          this.addAbility(new PrototypeAbility(this, "{4}{U}{U}", 6, 6));
31  
32          // Depth Charge Colossus doesn't untap during your untap step.
33          this.addAbility(new SimpleStaticAbility(new DontUntapInControllersUntapStepSourceEffect()));
34  
35          // {3}: Untap Depth Charge Colossus.
36          this.addAbility(new SimpleActivatedAbility(new UntapSourceEffect(), new GenericManaCost(3)));
37      }
38  
39      private DepthChargeColossus(final DepthChargeColossus card) {
40          super(card);
41      }
42  
43      @Override
44      public DepthChargeColossus copy() {
45          return new DepthChargeColossus(this);
46      }
47  }

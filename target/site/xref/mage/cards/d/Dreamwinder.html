28  public final class Dreamwinder extends CardImpl {
29                 
30      public Dreamwinder(UUID ownerId, CardSetInfo setInfo) {
31          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{U}");
32          this.subtype.add(SubType.SERPENT);
33  
34          this.power = new MageInt(4);
35          this.toughness = new MageInt(3);
36  
37          // Dreamwinder can't attack unless defending player controls an Island.
38          this.addAbility(new SimpleStaticAbility(new CantAttackUnlessDefenderControllsPermanent(new FilterLandPermanent(SubType.ISLAND,"an Island"))));
39          // {U}, Sacrifice an Island: Target land becomes an Island until end of turn.
40          Ability ability = new SimpleActivatedAbility(new BecomesBasicLandTargetEffect(Duration.EndOfTurn, SubType.ISLAND), new ColoredManaCost(ColoredManaSymbol.U));
41          ability.addTarget(new TargetLandPermanent().withChooseHint("becomes an Island"));
42          FilterControlledLandPermanent filter = new FilterControlledLandPermanent("an Island");
43          filter.add(SubType.ISLAND.getPredicate());
44          ability.addCost(new SacrificeTargetCost(new TargetControlledPermanent(1, 1, filter, true)));
45          this.addAbility(ability);
46      }   
47      
48  
49      private Dreamwinder(final Dreamwinder card) {
50          super(card);
51      }
52  
53      @Override
54      public Dreamwinder copy() {
55          return new Dreamwinder(this);
56      }
57  }

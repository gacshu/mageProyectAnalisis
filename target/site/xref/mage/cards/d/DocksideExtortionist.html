19  /**
20   * @author TheElk801
21   */
22  public final class DocksideExtortionist extends CardImpl {
23  
24      public DocksideExtortionist(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{R}");
26  
27          this.subtype.add(SubType.GOBLIN);
28          this.subtype.add(SubType.PIRATE);
29          this.power = new MageInt(1);
30          this.toughness = new MageInt(2);
31  
32          // When Dockside Extortionist enters the battlefield, create X Treasure tokens, where X is the number of artifacts and enchantments your opponents control.
33          this.addAbility(new EntersBattlefieldTriggeredAbility(
34                  new CreateTokenEffect(new TreasureToken(), DocksideExtortionistValue.instance)
35          ));
36      }
37  
38      private DocksideExtortionist(final DocksideExtortionist card) {
39          super(card);
40      }
41  
42      @Override
43      public DocksideExtortionist copy() {
44          return new DocksideExtortionist(this);
45      }
46  }
47  
48  enum DocksideExtortionistValue implements DynamicValue {
49      instance;
50  
51      @Override
52      public int calculate(Game game, Ability sourceAbility, Effect effect) {
53          return game
54                  .getOpponents(sourceAbility.getControllerId())
55                  .stream()
56                  .mapToInt(uuid -> game.getBattlefield().getAllActivePermanents(
57                          StaticFilters.FILTER_PERMANENT_ARTIFACT_OR_ENCHANTMENT, uuid, game
58                  ).size())
59                  .sum();
60      }
61  
62      @Override
63      public DocksideExtortionistValue copy() {
64          return instance;
65      }
66  
67      @Override
68      public String getMessage() {
69          return "the number of artifacts and enchantments your opponents control";
70      }
71  
72      @Override
73      public String toString() {
74          return "X";
75      }
76  }

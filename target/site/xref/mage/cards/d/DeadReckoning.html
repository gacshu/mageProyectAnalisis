23  public final class DeadReckoning extends CardImpl {

25      public DeadReckoning(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{1}{B}{B}");
27  
28          // You may put target creature card from your graveyard on top of your library. If you do, Dead Reckoning deals damage equal to that card's power to target creature.
29          this.getSpellAbility().addEffect(new DeadReckoningEffect());
30  
31      }
32

34          super(card);
35      }
36

38      public DeadReckoning copy() {
39          return new DeadReckoning(this);
40      }
41  }

43  class DeadReckoningEffect extends OneShotEffect {
44  
45      public DeadReckoningEffect() {
46          super(Outcome.Damage);
47          this.staticText = "You may put target creature card from your graveyard on top of your library. If you do, {this} deals damage equal to that card's power to target creature";
48      }
49  
50      public DeadReckoningEffect(final DeadReckoningEffect effect) {
51          super(effect);
52      }
53  
54      @Override
55      public DeadReckoningEffect copy() {
56          return new DeadReckoningEffect(this);
57      }
58  
59      @Override
60      public boolean apply(Game game, Ability source) {
61          Player controller = game.getPlayer(source.getControllerId());
62          TargetCardInYourGraveyard target1 = new TargetCardInYourGraveyard(new FilterCreatureCard("creature card in your graveyard"));
63          TargetCreaturePermanent target2 = new TargetCreaturePermanent();
64  
65          if (controller != null) {
66              if (target1.canChoose(source.getControllerId(), source, game)
67                      && controller.choose(Outcome.Benefit, target1, source, game)
68                      && target2.canChoose(source.getControllerId(), source, game)
69                      && controller.choose(Outcome.Damage, target2, source, game)) {
70                  Card creatureInGraveyard = game.getCard(target1.getFirstTarget());
71                  if (creatureInGraveyard != null) {
72                      if (controller.putCardsOnTopOfLibrary(creatureInGraveyard, game, source, true)) {
73                          int power = creatureInGraveyard.getPower().getValue();
74                          Permanent creature = game.getPermanent(target2.getFirstTarget());
75                          if (creature != null) {
76                              creature.damage(power, source.getSourceId(), source, game, false, true);
77                              return true;
78                          }
79                      }
80                  }
81              }
82          }
83          return false;
84      }
85  }

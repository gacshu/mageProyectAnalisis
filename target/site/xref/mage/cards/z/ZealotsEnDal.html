4   import java.util.UUID;
5   import mage.MageInt;
6   import mage.ObjectColor;
7   import mage.abilities.common.BeginningOfUpkeepTriggeredAbility;
8   import mage.abilities.condition.InvertCondition;
9   import mage.abilities.condition.common.PermanentsOnTheBattlefieldCondition;
10  import mage.abilities.decorator.ConditionalInterveningIfTriggeredAbility;
11  import mage.abilities.effects.common.GainLifeEffect;
12  import mage.constants.SubType;
13  import mage.cards.CardImpl;
14  import mage.cards.CardSetInfo;
15  import mage.constants.CardType;
16  import mage.constants.TargetController;
17  import mage.filter.common.FilterControlledPermanent;
18  import mage.filter.predicate.Predicates;
19  import mage.filter.predicate.mageobject.ColorPredicate;
20  
21  /**
22   *
23   * @author TheElk801
24   */
25  public final class ZealotsEnDal extends CardImpl {
26  
27      private static final FilterControlledPermanent filter = new FilterControlledPermanent();
28  
29      static {
30          filter.add(Predicates.not(new ColorPredicate(ObjectColor.WHITE)));
31          filter.add(Predicates.not(CardType.LAND.getPredicate()));
32      }
33  
34      public ZealotsEnDal(UUID ownerId, CardSetInfo setInfo) {
35          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{3}{W}");
36  
37          this.subtype.add(SubType.HUMAN);
38          this.subtype.add(SubType.SOLDIER);
39          this.power = new MageInt(2);
40          this.toughness = new MageInt(4);
41  
42          // At the beginning of your upkeep, if all nonland permanents you control are white, you gain 1 life.
43          this.addAbility(new ConditionalInterveningIfTriggeredAbility(
44                  new BeginningOfUpkeepTriggeredAbility(new GainLifeEffect(1), TargetController.YOU, false),
45                  new InvertCondition(new PermanentsOnTheBattlefieldCondition(filter)),
46                  "At the beginning of your upkeep, if all nonland permanents you control are white, you gain 1 life."
47          ));
48      }
49  
50      private ZealotsEnDal(final ZealotsEnDal card) {
51          super(card);
52      }
53  
54      @Override
55      public ZealotsEnDal copy() {
56          return new ZealotsEnDal(this);
57      }
58  }

19  /**
20   *
21   * @author LevelX2
22   */
23  public final class ZahidDjinnOfTheLamp extends CardImpl {
24  
25      public ZahidDjinnOfTheLamp(UUID ownerId, CardSetInfo setInfo) {
26          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{4}{U}{U}");
27  
28          this.addSuperType(SuperType.LEGENDARY);
29          this.subtype.add(SubType.DJINN);
30          this.power = new MageInt(5);
31          this.toughness = new MageInt(6);
32  
33          // You may pay {3}{U} and tap an untapped artifact you control rather than pay this spell's mana cost.
34          AlternativeCostSourceAbility alternativeCostSourceAbility = new AlternativeCostSourceAbility(new ManaCostsImpl<>("{3}{U}"), null,
35                  "You may pay {3}{U} and tap an untapped artifact you control rather than pay this spell's mana cost.");
36          FilterControlledArtifactPermanent filter = new FilterControlledArtifactPermanent("untapped artifact you control");
37          filter.add(TappedPredicate.UNTAPPED);
38          alternativeCostSourceAbility.addCost(new TapTargetCost(new TargetControlledPermanent(filter)));
39          this.addAbility(alternativeCostSourceAbility);
40  
41          // Flying
42          this.addAbility(FlyingAbility.getInstance());
43  
44      }
45  
46      private ZahidDjinnOfTheLamp(final ZahidDjinnOfTheLamp card) {
47          super(card);
48      }
49  
50      @Override
51      public ZahidDjinnOfTheLamp copy() {
52          return new ZahidDjinnOfTheLamp(this);
53      }
54  }

28   * @author LevelX2

30  public final class LilianaDefiantNecromancer extends CardImpl {
31  
32      protected static final FilterCreatureCard filter = new FilterCreatureCard("nonlegendary creature card with mana value X from your graveyard");
33  
34      static {
35          filter.add(Predicates.not(SuperType.LEGENDARY.getPredicate()));
36      }
37  
38      public LilianaDefiantNecromancer(UUID ownerId, CardSetInfo setInfo) {
39          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "");
40          this.addSuperType(SuperType.LEGENDARY);
41          this.subtype.add(SubType.LILIANA);
42          this.color.setBlack(true);
43  
44          this.nightCard = true;
45  
46          this.setStartingLoyalty(3);
47  
48          // +2: Each player discards a card.
49          this.addAbility(new LoyaltyAbility(new DiscardEachPlayerEffect(1, false), 2));
50  
51          // -X: Return target nonlegendary creature with converted mana cost X from your graveyard to the battlefield.
52          Ability ability = new LoyaltyAbility(new ReturnFromGraveyardToBattlefieldTargetEffect());
53          ability.addTarget(new TargetCardInYourGraveyard(filter));
54          ability.setTargetAdjuster(LilianaDefiantNecromancerAdjuster.instance);
55          this.addAbility(ability);
56  
57          //-8: You get an emblem with "Whenever a creature dies, return it to the battlefield under your control at the beginning of the next end step.";
58          this.addAbility(new LoyaltyAbility(new GetEmblemEffect(new LilianaDefiantNecromancerEmblem()), -8));
59      }
60  
61      private LilianaDefiantNecromancer(final LilianaDefiantNecromancer card) {
62          super(card);
63      }
64  
65      @Override
66      public LilianaDefiantNecromancer copy() {
67          return new LilianaDefiantNecromancer(this);
68      }
69  }
70  
71  enum LilianaDefiantNecromancerAdjuster implements TargetAdjuster {
72      instance;
73  
74      @Override
75      public void adjustTargets(Ability ability, Game game) {
76          int cmc = 0;
77          for (Cost cost : ability.getCosts()) {
78              if (cost instanceof PayVariableLoyaltyCost) {
79                  cmc = ((PayVariableLoyaltyCost) cost).getAmount();
80              }
81          }
82          FilterCard newFilter = LilianaDefiantNecromancer.filter.copy();
83          newFilter.add(new ManaValuePredicate(ComparisonType.EQUAL_TO, cmc));
84          ability.getTargets().clear();
85          ability.addTarget(new TargetCardInYourGraveyard(newFilter));
86      }
87  }

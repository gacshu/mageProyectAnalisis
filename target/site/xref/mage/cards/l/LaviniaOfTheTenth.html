3   package mage.cards.l;
4   
5   import java.util.UUID;
6   import mage.MageInt;
7   import mage.ObjectColor;
8   import mage.abilities.common.EntersBattlefieldTriggeredAbility;
9   import mage.abilities.effects.common.DetainAllEffect;
10  import mage.abilities.keyword.ProtectionAbility;
11  import mage.cards.CardImpl;
12  import mage.cards.CardSetInfo;
13  import mage.constants.CardType;
14  import mage.constants.SubType;
15  import mage.constants.ComparisonType;
16  import mage.constants.SuperType;
17  import mage.constants.TargetController;
18  import mage.filter.FilterPermanent;
19  import mage.filter.predicate.Predicates;
20  import mage.filter.predicate.mageobject.ManaValuePredicate;
21  
22  /**
23   *
24   * @author LevelX2
25   */
26  public final class LaviniaOfTheTenth  extends CardImpl {
27  
28      private static final FilterPermanent filterDetain = new FilterPermanent("each nonland permanent your opponents control with mana value 4 or less");
29      static {
30          filterDetain.add(TargetController.OPPONENT.getControllerPredicate());
31          filterDetain.add(Predicates.not(CardType.LAND.getPredicate()));
32          filterDetain.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 5));
33      }
34  
35      public LaviniaOfTheTenth (UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId,setInfo,new CardType[]{CardType.CREATURE},"{3}{W}{U}");
37          this.subtype.add(SubType.HUMAN);
38          this.subtype.add(SubType.SOLDIER);
39          this.addSuperType(SuperType.LEGENDARY);
40  
41  
42          this.power = new MageInt(4);
43          this.toughness = new MageInt(4);
44  
45          // Protection from red
46          this.addAbility(ProtectionAbility.from(ObjectColor.RED));
47          // When Lavinia of the Tenth enters the battlefield, detain each nonland permanent your opponents control with converted mana cost 4 or less.
48          this.addAbility(new EntersBattlefieldTriggeredAbility(new DetainAllEffect(filterDetain)));
49  
50      }
51  
52      public LaviniaOfTheTenth (final LaviniaOfTheTenth card) {
53          super(card);
54      }
55  
56      @Override
57      public LaviniaOfTheTenth copy() {
58          return new LaviniaOfTheTenth(this);
59      }
60  
61  }

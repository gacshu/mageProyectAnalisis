22  public final class LifeInsurance extends CardImpl {
23  
24      public LifeInsurance(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{3}{W}{B}");
26  
27          // Extort
28          this.addAbility(new ExtortAbility());
29  
30          // Whenever a nontoken creature dies, you lose 1 life and create a Treasure token.
31          Ability ability = new DiesCreatureTriggeredAbility(
32                  new LoseLifeSourceControllerEffect(1), false, StaticFilters.FILTER_CREATURE_NON_TOKEN
33          );
34          ability.addEffect(new CreateTokenEffect(new TreasureToken()).concatBy("and"));
35          this.addAbility(ability);
36      }
37  
38      private LifeInsurance(final LifeInsurance card) {
39          super(card);
40      }
41  
42      @Override
43      public LifeInsurance copy() {
44          return new LifeInsurance(this);
45      }
46  }

26  public final class LilianaTheNecromancer extends CardImpl {
27  
28      public LilianaTheNecromancer(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "{3}{B}{B}");
30  
31          this.addSuperType(SuperType.LEGENDARY);
32          this.subtype.add(SubType.LILIANA);
33          this.setStartingLoyalty(4);
34  
35          // +1: Target player loses 2 life.
36          Ability ability = new LoyaltyAbility(new LoseLifeTargetEffect(2), 1);
37          ability.addTarget(new TargetPlayer());
38          this.addAbility(ability);
39  
40          // −1: Return target creature card from your graveyard to your hand.
41          ability = new LoyaltyAbility(new ReturnFromGraveyardToHandTargetEffect(), -1);
42          ability.addTarget(new TargetCardInYourGraveyard(StaticFilters.FILTER_CARD_CREATURE_YOUR_GRAVEYARD));
43          this.addAbility(ability);
44  
45          // −7: Destroy up to two target creatures. Put up to two creature cards from graveyards onto the battlefield under your control.
46          ability = new LoyaltyAbility(new DestroyTargetEffect(), -7);
47          ability.addTarget(new TargetCreaturePermanent(0, 2, StaticFilters.FILTER_PERMANENT_CREATURES, false));
48          ability.addEffect(new LilianaTheNecromancerEffect());
49          this.addAbility(ability);
50      }
51  
52      private LilianaTheNecromancer(final LilianaTheNecromancer card) {
53          super(card);
54      }
55  
56      @Override
57      public LilianaTheNecromancer copy() {
58          return new LilianaTheNecromancer(this);
59      }
60  }

62  class LilianaTheNecromancerEffect extends OneShotEffect {
63  
64      private static final FilterCreatureCard filter = new FilterCreatureCard("creature cards from graveyards");
65  
66      public LilianaTheNecromancerEffect() {
67          super(Outcome.Benefit);
68          this.staticText = "Put up to two creature cards from graveyards onto the battlefield under your control";
69      }
70  
71      public LilianaTheNecromancerEffect(final LilianaTheNecromancerEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public LilianaTheNecromancerEffect copy() {
77          return new LilianaTheNecromancerEffect(this);
78      }
79  
80      @Override
81      public boolean apply(Game game, Ability source) {
82          Player player = game.getPlayer(source.getControllerId());
83          if (player == null) {
84              return false;
85          }
86          Target target = new TargetCardInGraveyard(0, 2, filter);
87          target.setNotTarget(true);
88          if (!player.choose(outcome, target, source, game)) {
89              return false;
90          }
91          Cards cardsToMove = new CardsImpl();
92          for (UUID targetId : target.getTargets()) {
93              Card card = game.getCard(targetId);
94              if (card != null) {
95                  cardsToMove.add(card);
96              }
97          }
98          return player.moveCards(cardsToMove, Zone.BATTLEFIELD, source, game);
99      }
100 }

4   import java.util.UUID;
5   import mage.Mana;
6   import mage.abilities.Ability;
7   import mage.abilities.common.SimpleStaticAbility;
8   import mage.abilities.costs.common.SacrificeTargetCost;
9   import mage.abilities.costs.common.TapSourceCost;
10  import mage.abilities.effects.common.EnterBattlefieldPayCostOrPutGraveyardEffect;
11  import mage.abilities.mana.BlackManaAbility;
12  import mage.abilities.mana.SimpleManaAbility;
13  import mage.cards.CardImpl;
14  import mage.cards.CardSetInfo;
15  import mage.constants.CardType;
16  import mage.constants.SubType;
17  import mage.constants.Zone;
18  import mage.filter.common.FilterControlledPermanent;
19  import mage.target.common.TargetControlledPermanent;
20  
21  /**
22   *
23   * @author LevelX2
24   */
25  public final class LakeOfTheDead extends CardImpl {
26  
27      private static final FilterControlledPermanent filter = new FilterControlledPermanent("a Swamp");
28  
29      static {
30          filter.add(SubType.SWAMP.getPredicate());
31      }
32  
33      public LakeOfTheDead(UUID ownerId, CardSetInfo setInfo) {
34          super(ownerId,setInfo,new CardType[]{CardType.LAND},"");
35  
36          // If Lake of the Dead would enter the battlefield, sacrifice a Swamp instead. If you do, put Lake of the Dead onto the battlefield. If you don't, put it into its owner's graveyard.
37          this.addAbility(new SimpleStaticAbility(Zone.ALL, new EnterBattlefieldPayCostOrPutGraveyardEffect(new SacrificeTargetCost(new TargetControlledPermanent(filter)))));
38          // {tap}: Add {B}.
39          this.addAbility(new BlackManaAbility());
40  
41          // {tap}, Sacrifice a Swamp: Add {B}{B}{B}{B}.
42          Ability ability = new SimpleManaAbility(Zone.BATTLEFIELD, Mana.BlackMana(4), new TapSourceCost());
43          ability.addCost(new SacrificeTargetCost(new TargetControlledPermanent(filter)));
44          this.addAbility(ability);
45  
46      }
47  
48      private LakeOfTheDead(final LakeOfTheDead card) {
49          super(card);
50      }
51  
52      @Override
53      public LakeOfTheDead copy() {
54          return new LakeOfTheDead(this);
55      }
56  }

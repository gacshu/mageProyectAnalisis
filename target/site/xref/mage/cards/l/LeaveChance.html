19  public final class LeaveChance extends SplitCard {
20  
21      private static final FilterPermanent filter = new FilterPermanent("permanents you own");
22  
23      static {
24          filter.add(TargetController.YOU.getOwnerPredicate());
25      }
26  
27      public LeaveChance(UUID ownerId, CardSetInfo setInfo) {
28          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, new CardType[]{CardType.SORCERY}, "{1}{W}", "{3}{R}", SpellAbilityType.SPLIT_AFTERMATH);
29  
30          // Return any number of target permanents you own to your hand.
31          getLeftHalfCard().getSpellAbility().addEffect(new ReturnToHandTargetEffect());
32          getLeftHalfCard().getSpellAbility().addTarget(new TargetPermanent(0, Integer.MAX_VALUE, filter, false));
33  
34          // Chance
35          // Sorcery
36          // Aftermath
37          getRightHalfCard().addAbility(new AftermathAbility().setRuleAtTheTop(true));
38  
39          // Discard any number of cards, then draw that many cards.
40          getRightHalfCard().getSpellAbility().addEffect(new DiscardAndDrawThatManyEffect(Integer.MAX_VALUE));
41      }
42  
43      private LeaveChance(final LeaveChance card) {
44          super(card);
45      }
46  
47      @Override
48      public LeaveChance copy() {
49          return new LeaveChance(this);
50      }
51  }

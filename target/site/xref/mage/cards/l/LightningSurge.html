21  public final class LightningSurge extends CardImpl {
22  
23      public LightningSurge(UUID ownerId, CardSetInfo setInfo) {
24          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{3}{R}{R}");
25  
26          // Lightning Surge deals 4 damage to any target.
27          // Threshold - If seven or more cards are in your graveyard, instead Lightning Surge deals 6 damage to that creature or player and the damage can't be prevented.
28          Effect effect = new ConditionalOneShotEffect(new DamageTargetEffect(6, false),
29                  new DamageTargetEffect(4),
30                  new CardsInControllerGraveyardCondition(7),
31                  "{this} deals 4 damage to any target.<br/><br/><i>Threshold</i> &mdash; {this} deals 6 damage to that permanent or player and the damage can't be prevented instead if seven or more cards are in your graveyard.");
32          this.getSpellAbility().addTarget(new TargetAnyTarget());
33          this.getSpellAbility().addEffect(effect);
34  
35          // Flashback {5}{R}{R}
36          this.addAbility(new FlashbackAbility(this, new ManaCostsImpl<>("{5}{R}{R}")));
37      }
38  
39      private LightningSurge(final LightningSurge card) {
40          super(card);
41      }
42  
43      @Override
44      public LightningSurge copy() {
45          return new LightningSurge(this);
46      }
47  }

21  /**
22   * @author TheElk801
23   */
24  public final class LurrusOfTheDreamDen extends CardImpl {
25  
26      private static final FilterPermanentCard filter = new FilterPermanentCard();
27  
28      static {
29          filter.add(new ManaValuePredicate(ComparisonType.FEWER_THAN, 3));
30      }
31  
32      public LurrusOfTheDreamDen(UUID ownerId, CardSetInfo setInfo) {
33          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{1}{W/B}{W/B}");
34  
35          this.addSuperType(SuperType.LEGENDARY);
36          this.subtype.add(SubType.CAT);
37          this.subtype.add(SubType.NIGHTMARE);
38          this.power = new MageInt(3);
39          this.toughness = new MageInt(2);
40  
41          // Companion â€” Each permanent card in your starting deck has converted mana cost 2 or less.
42          this.addAbility(new CompanionAbility(LurrusOfTheDreamDenCompanionCondition.instance));
43  
44          // Lifelink
45          this.addAbility(LifelinkAbility.getInstance());
46  
47          // During each of your turns, you may cast one permanent spell with converted mana cost 2 or less from your graveyard.
48          this.addAbility(new CastFromGraveyardOnceStaticAbility(filter, "During each of your turns, you may cast one permanent spell with mana value 2 or less from your graveyard"));
49      }
50  
51      private LurrusOfTheDreamDen(final LurrusOfTheDreamDen card) {
52          super(card);
53      }
54  
55      @Override
56      public LurrusOfTheDreamDen copy() {
57          return new LurrusOfTheDreamDen(this);
58      }
59  }
60  
61  enum LurrusOfTheDreamDenCompanionCondition implements CompanionCondition {
62      instance;
63  
64      @Override
65      public String getRule() {
66          return "Each permanent card in your starting deck has mana value 2 or less.";
67      }
68  
69      @Override
70      public boolean isLegal(Set<Card> deck, int startingHandSize) {
71          return deck.stream()
72                  .filter(MageObject::isPermanent)
73                  .mapToInt(MageObject::getManaValue)
74                  .max()
75                  .orElse(0) <= 2;
76      }
77  }

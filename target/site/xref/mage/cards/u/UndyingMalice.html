26  public final class UndyingMalice extends CardImpl {
27  
28      public UndyingMalice(UUID ownerId, CardSetInfo setInfo) {
29          super(ownerId, setInfo, new CardType[]{CardType.INSTANT}, "{B}");
30  
31          // Until end of turn, target creature gains "When this creature dies, return it to the battlefield tapped under its owner's control with a +1/+1 counter on it."
32          this.getSpellAbility().addTarget(new TargetCreaturePermanent());
33          this.getSpellAbility().addEffect(new GainAbilityTargetEffect(new DiesSourceTriggeredAbility(new UndyingMaliceReturnToBattlefieldEffect()))
34                  .setText("Until end of turn, target creature gains \"When this creature dies, return it to the battlefield tapped under its owner's control with a +1/+1 counter on it.\""));
35      }
36  
37      private UndyingMalice(final UndyingMalice card) {
38          super(card);
39      }
40  
41      @Override
42      public UndyingMalice copy() {
43          return new UndyingMalice(this);
44      }
45  }
46  
47  class UndyingMaliceReturnToBattlefieldEffect extends ReturnSourceFromGraveyardToBattlefieldEffect {
48  
49      public UndyingMaliceReturnToBattlefieldEffect() {
50          super(true);
51          staticText = "return it to the battlefield tapped under its owner's control with a +1/+1 counter on it";
52      }
53  
54      private UndyingMaliceReturnToBattlefieldEffect(final UndyingMaliceReturnToBattlefieldEffect effect) {
55          super(effect);
56      }
57  
58      @Override
59      public UndyingMaliceReturnToBattlefieldEffect copy() {
60          return new UndyingMaliceReturnToBattlefieldEffect(this);
61      }
62  
63      @Override
64      public boolean apply(Game game, Ability source) {
65          UndyingMaliceCounterEffect counterEffect = new UndyingMaliceCounterEffect();
66          game.addEffect(counterEffect, source);
67          return super.apply(game, source);
68      }
69  }
70  
71  class UndyingMaliceCounterEffect extends ReplacementEffectImpl {
72  
73      public UndyingMaliceCounterEffect() {
74          super(Duration.EndOfStep, Outcome.BoostCreature);
75      }
76  
77      private UndyingMaliceCounterEffect(final UndyingMaliceCounterEffect effect) {
78          super(effect);
79      }
80  
81      @Override
82      public UndyingMaliceCounterEffect copy() {
83          return new UndyingMaliceCounterEffect(this);
84      }
85  
86      @Override
87      public boolean checksEventType(GameEvent event, Game game) {
88          return event.getType() == GameEvent.EventType.ENTERS_THE_BATTLEFIELD;
89      }
90  
91      @Override
92      public boolean applies(GameEvent event, Ability source, Game game) {
93          return event.getTargetId().equals(source.getSourceId());
94      }
95  
96      @Override
97      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
98          Permanent creature = ((EntersTheBattlefieldEvent) event).getTarget();
99          if (creature == null) {
100             return false;
101         }
102         creature.addCounters(CounterType.P1P1.createInstance(), source.getControllerId(), source, game, event.getAppliedEffects());
103         discard();
104         return false;
105     }
106 }

22  public final class UndyingFlames extends CardImpl {
23  
24      public UndyingFlames(UUID ownerId, CardSetInfo setInfo) {
25          super(ownerId, setInfo, new CardType[]{CardType.SORCERY}, "{4}{R}{R}");
26  
27          // Exile cards from the top of your library until you exile a nonland card. Undying Flames deals damage to any target equal to that card's converted mana cost.
28          this.getSpellAbility().addEffect(new UndyingFlamesEffect());
29          this.getSpellAbility().addTarget(new TargetAnyTarget());
30  
31          // Epic
32          this.getSpellAbility().addEffect(new EpicEffect());
33  
34      }
35  
36      private UndyingFlames(final UndyingFlames card) {
37          super(card);
38      }
39  
40      @Override
41      public UndyingFlames copy() {
42          return new UndyingFlames(this);
43      }
44  }
45  
46  class UndyingFlamesEffect extends OneShotEffect {
47  
48      public UndyingFlamesEffect() {
49          super(Outcome.Benefit);
50          staticText = "Exile cards from the top of your library until you exile a nonland card. {this} deals damage to any target equal to that card's mana value";
51      }
52  
53      public UndyingFlamesEffect(final UndyingFlamesEffect effect) {
54          super(effect);
55      }
56  
57      @Override
58      public boolean apply(Game game, Ability source) {
59  
60          Player controller = game.getPlayer(source.getControllerId());
61          if (controller != null) {
62              while (controller.canRespond() && controller.getLibrary().hasCards()) {
63                  Card card = controller.getLibrary().getFromTop(game);
64                  if (card != null) {
65                      controller.moveCards(card, Zone.EXILED, source, game);
66                      if (!card.isLand(game)) {
67                          new DamageTargetEffect(card.getManaValue()).apply(game, source);
68                          break;
69                      }
70                  }
71              }
72              return true;
73          }
74          return false;
75      }
76  
77      @Override
78      public UndyingFlamesEffect copy() {
79          return new UndyingFlamesEffect(this);
80      }
81  }

29  public final class UrzaPlaneswalker extends MeldCard {
30  
31      private static final FilterCard filter = new FilterCard("artifact, instant, and sorcery spells");
32      private static final FilterPermanent filter2 = new FilterPermanent("artifacts and planeswalkers");
33  
34      static {
35          filter.add(Predicates.or(
36                  CardType.ARTIFACT.getPredicate(),
37                  CardType.INSTANT.getPredicate(),
38                  CardType.SORCERY.getPredicate()
39          ));
40          filter2.add(Predicates.or(
41                  CardType.ARTIFACT.getPredicate(),
42                  CardType.PLANESWALKER.getPredicate()
43          ));
44      }
45  
46      public UrzaPlaneswalker(UUID ownerId, CardSetInfo setInfo) {
47          super(ownerId, setInfo, new CardType[]{CardType.PLANESWALKER}, "");
48  
49          this.addSuperType(SuperType.LEGENDARY);
50          this.subtype.add(SubType.URZA);
51  
52          this.setStartingLoyalty(7);
53  
54          this.color.setWhite(true);
55          this.color.setBlue(true);
56  
57          // You may activate the loyalty abilities of Urza, Planeswalker twice each turn rather than only once.
58          this.addAbility(new SimpleStaticAbility(new UrzaPlaneswalkerEffect()));
59  
60          // +2: Artifact, instant, and sorcery spells you cast this turn cost {2} less to cast. You gain 2 life.
61          Ability ability = new LoyaltyAbility(new SpellsCostReductionControllerEffect(filter, 2)
62                  .setDuration(Duration.EndOfTurn)
63                  .setText("artifact, instant, and sorcery spells you cast this turn cost {2} less to cast"), 2);
64          ability.addEffect(new GainLifeEffect(2));
65          this.addAbility(ability);
66  
67          // +1: Draw two cards, then discard a card.
68          this.addAbility(new LoyaltyAbility(new DrawDiscardControllerEffect(2, 1), 1));
69  
70          // 0: Create two 1/1 colorless Soldier artifact creature tokens.
71          this.addAbility(new LoyaltyAbility(new CreateTokenEffect(new SoldierArtifactToken(), 2), 0));
72  
73          // -3: Exile target nonland permanent.
74          ability = new LoyaltyAbility(new ExileTargetEffect(), -3);
75          ability.addTarget(new TargetNonlandPermanent());
76          this.addAbility(ability);
77  
78          // -10: Artifacts and planeswalkers you control gain indestructible until end of turn. Destroy all nonland permanents.
79          ability = new LoyaltyAbility(new GainAbilityControlledEffect(
80                  IndestructibleAbility.getInstance(), Duration.EndOfTurn, filter2
81          ), -10);
82          ability.addEffect(new DestroyAllEffect(StaticFilters.FILTER_PERMANENTS_NON_LAND));
83          this.addAbility(ability);
84      }
85  
86      private UrzaPlaneswalker(final UrzaPlaneswalker card) {
87          super(card);
88      }
89  
90      @Override
91      public UrzaPlaneswalker copy() {
92          return new UrzaPlaneswalker(this);
93      }
94  }
95  
96  class UrzaPlaneswalkerEffect extends ContinuousEffectImpl {
97  
98      public UrzaPlaneswalkerEffect() {
99          super(Duration.WhileOnBattlefield, Layer.RulesEffects, SubLayer.NA, Outcome.Benefit);
100         staticText = "once during each of your turns, you may activate an additional loyalty ability of {this}";
101     }
102 
103     public UrzaPlaneswalkerEffect(final UrzaPlaneswalkerEffect effect) {
104         super(effect);
105     }
106 
107     @Override
108     public UrzaPlaneswalkerEffect copy() {
109         return new UrzaPlaneswalkerEffect(this);
110     }
111 
112     @Override
113     public boolean apply(Game game, Ability source) {
114         Permanent permanent = source.getSourcePermanentIfItStillExists(game);
115         if (permanent == null) {
116             return false;
117         }
118 
119         permanent.setLoyaltyActivationsAvailable(2);
120         return true;
121     }
122 }

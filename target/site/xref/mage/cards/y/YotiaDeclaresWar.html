36      public YotiaDeclaresWar(UUID ownerId, CardSetInfo setInfo) {
37          super(ownerId, setInfo, new CardType[]{CardType.ENCHANTMENT}, "{1}{R}");
38  
39          this.subtype.add(SubType.SAGA);
40  
41          // Read ahead
42          SagaAbility sagaAbility = new SagaAbility(this, SagaChapter.CHAPTER_III, true);
43  
44          // I -- Create a 0/2 colorless Thopter artifact creature token with flying named Ornithopter.
45          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_I, new CreateTokenEffect(new OrnithopterToken()));
46  
47          // II -- Tap any number of untapped artifacts you control. When you do, Yotia Declares War deals that much damage to target creature or planeswalker.
48          sagaAbility.addChapterEffect(this, SagaChapter.CHAPTER_II, new YotiaDeclaresWarEffect());
49  
50          // III -- Up to one target artifact you control becomes an artifact creature with base power and toughness 4/4 until end of turn.
51          sagaAbility.addChapterEffect(
52                  this, SagaChapter.CHAPTER_III, SagaChapter.CHAPTER_III,
53                  new Effects(
54                          new AddCardTypeTargetEffect(Duration.EndOfTurn, CardType.ARTIFACT, CardType.CREATURE)
55                                  .setText("up to one target artifact you control becomes an artifact creature"),
56                          new SetBasePowerToughnessTargetEffect(4, 4, Duration.EndOfTurn)
57                                  .setText("with base power and toughness 4/4 until end of turn")
58                  ), new TargetPermanent(0, 1, StaticFilters.FILTER_CONTROLLED_PERMANENT_ARTIFACT)
59          );
60          this.addAbility(sagaAbility);
61      }

73  class YotiaDeclaresWarEffect extends OneShotEffect {
74  
75      private static final FilterPermanent filter
76              = new FilterControlledArtifactPermanent("untapped artifacts you control");
77  
78      static {
79          filter.add(TappedPredicate.UNTAPPED);
80      }
81  
82      YotiaDeclaresWarEffect() {
83          super(Outcome.Benefit);
84          staticText = "tap any number of untapped artifacts you control. " +
85                  "When you do, {this} deals that much damage to target creature or planeswalker";
86      }
87  
88      private YotiaDeclaresWarEffect(final YotiaDeclaresWarEffect effect) {
89          super(effect);
90      }
91  
92      @Override
93      public YotiaDeclaresWarEffect copy() {
94          return new YotiaDeclaresWarEffect(this);
95      }
96  
97      @Override
98      public boolean apply(Game game, Ability source) {
99          Player player = game.getPlayer(source.getControllerId());
100         if (player == null) {
101             return false;
102         }
103         TargetPermanent target = new TargetPermanent(0, Integer.MAX_VALUE, filter, true);
104         player.choose(outcome, target, source, game);
105         List<Permanent> permanents = target
106                 .getTargets()
107                 .stream()
108                 .map(game::getPermanent)
109                 .filter(Objects::nonNull)
110                 .collect(Collectors.toList());
111         if (permanents.isEmpty()) {
112             return false;
113         }
114         for (Permanent permanent:permanents){
115             permanent.tap(source, game);
116         }
117         ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
118                 new DamageTargetEffect(permanents.size()), false
119         );
120         ability.addTarget(new TargetCreatureOrPlaneswalker());
121         game.fireReflexiveTriggeredAbility(ability, source);
122         return true;
123     }
124 }

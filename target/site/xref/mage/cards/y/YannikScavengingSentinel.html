33  public final class YannikScavengingSentinel extends CardImpl {

35      public YannikScavengingSentinel(UUID ownerId, CardSetInfo setInfo) {
36          super(ownerId, setInfo, new CardType[]{CardType.CREATURE}, "{2}{G}{W}");
37  
38          this.addSuperType(SuperType.LEGENDARY);
39          this.subtype.add(SubType.HYENA);
40          this.subtype.add(SubType.BEAST);
41          this.power = new MageInt(3);
42          this.toughness = new MageInt(3);
43  
44          // Partner with Nikara, Lair Scavenger
45          this.addAbility(new PartnerWithAbility("Nikara, Lair Scavenger"));
46  
47          // Vigilance
48          this.addAbility(VigilanceAbility.getInstance());
49  
50          // When Yannik, Scavenging Sentinel enters the battlefield, exile another creature you control until Yannik leaves the battlefield. When you do, distribute X +1/+1 counters among any number of target creatures, where X is the exiled creature's power.
51          this.addAbility(new EntersBattlefieldTriggeredAbility(new YannikScavengingSentinelEffect()));
52      }
53  
54      private YannikScavengingSentinel(final YannikScavengingSentinel card) {
55          super(card);
56      }
57

59      public YannikScavengingSentinel copy() {

61      }

64  class YannikScavengingSentinelEffect extends OneShotEffect {
65  
66      private static final FilterPermanent filter
67              = new FilterControlledCreaturePermanent("another creature you control");
68  
69      static {
70          filter.add(AnotherPredicate.instance);
71      }
72  
73      YannikScavengingSentinelEffect() {
74          super(Outcome.Benefit);
75          staticText = "exile another creature you control until {this} leaves the battlefield. " +
76                  "When you do, distribute X +1/+1 counters among any number of target creatures, " +
77                  "where X is the exiled creature's power.";
78      }
79  
80      private YannikScavengingSentinelEffect(final YannikScavengingSentinelEffect effect) {
81          super(effect);
82      }
83  
84      @Override
85      public YannikScavengingSentinelEffect copy() {
86          return new YannikScavengingSentinelEffect(this);
87      }
88  
89      @Override
90      public boolean apply(Game game, Ability source) {
91          Player player = game.getPlayer(source.getControllerId());
92          Permanent sourcePermanent = source.getSourcePermanentIfItStillExists(game);
93          if (player == null || sourcePermanent == null
94                  || game.getBattlefield().count(filter, source.getControllerId(), source, game) < 1) {
95              return false;
96          }
97          TargetPermanent target = new TargetPermanent(filter);
98          target.setNotTarget(true);
99          player.choose(outcome, target, source, game);
100         Permanent permanent = game.getPermanent(target.getFirstTarget());
101         if (permanent == null) {
102             return false;
103         }
104         int power = permanent.getPower().getValue();
105         new ExileTargetEffect(CardUtil.getExileZoneId(
106                 game, source.getSourceId(), source.getSourceObjectZoneChangeCounter()
107         ), permanent.getIdName()).setTargetPointer(new FixedTarget(permanent, game)).apply(game, source);
108         game.addDelayedTriggeredAbility(new OnLeaveReturnExiledToBattlefieldAbility(), source);
109         if (game.getState().getZone(permanent.getId()) != Zone.BATTLEFIELD) {
110             ReflexiveTriggeredAbility ability = new ReflexiveTriggeredAbility(
111                     new DistributeCountersEffect(
112                             CounterType.P1P1, power, false, ""
113                     ), false, "distribute X +1/+1 counters among any number of target creatures, " +
114                     "where X is the exiled creature's power"
115             );
116             ability.addTarget(new TargetCreaturePermanentAmount(power));
117             game.fireReflexiveTriggeredAbility(ability, source);
118         }
119         return true;
120     }

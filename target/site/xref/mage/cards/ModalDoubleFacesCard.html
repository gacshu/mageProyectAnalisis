30      public ModalDoubleFacesCard(UUID ownerId, CardSetInfo setInfo,
31                                  CardType[] typesLeft, SubType[] subTypesLeft, String costsLeft,
32                                  String secondSideName, CardType[] typesRight, SubType[] subTypesRight, String costsRight) {
33          super(ownerId, setInfo, typesLeft, costsLeft + costsRight, SpellAbilityType.MODAL);
34          // main card name must be same as left side
35          leftHalfCard = new ModalDoubleFacesCardHalfImpl(this.getOwnerId(), new CardSetInfo(setInfo.getName(), setInfo.getExpansionSetCode(), setInfo.getCardNumber(), setInfo.getRarity(), setInfo.getGraphicInfo()),
36                  typesLeft, subTypesLeft, costsLeft, this, SpellAbilityType.MODAL_LEFT);
37          rightHalfCard = new ModalDoubleFacesCardHalfImpl(this.getOwnerId(), new CardSetInfo(secondSideName, setInfo.getExpansionSetCode(), setInfo.getCardNumber(), setInfo.getRarity(), setInfo.getGraphicInfo()),
38                  typesRight, subTypesRight, costsRight, this, SpellAbilityType.MODAL_RIGHT);
39      }

41      public ModalDoubleFacesCard(ModalDoubleFacesCard card) {
42          super(card);

45          this.rightHalfCard = card.rightHalfCard.copy();

144     @Override
145     public boolean cast(Game game, Zone fromZone, SpellAbility ability, UUID controllerId) {
146         switch (ability.getSpellAbilityType()) {
147             case MODAL_LEFT:
148                 return this.leftHalfCard.cast(game, fromZone, ability, controllerId);
149             case MODAL_RIGHT:
150                 return this.rightHalfCard.cast(game, fromZone, ability, controllerId);
151             default:
152                 if (this.leftHalfCard.getSpellAbility() != null)
153                     this.leftHalfCard.getSpellAbility().setControllerId(controllerId);
154                 if (this.rightHalfCard.getSpellAbility() != null)
155                     this.rightHalfCard.getSpellAbility().setControllerId(controllerId);
156                 return super.cast(game, fromZone, ability, controllerId);
157         }
158     }

214     private boolean isIgnoreDefaultAbility(Ability ability) {
215         // ignore default play/spell ability from main card (only halfes are actual)
216         // default abilities added on card creation from card type and can't be skipped
217 
218         // skip cast spell
219         if (ability instanceof SpellAbility && ((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.MODAL) {
220             return true;
221         }
222 
223         // skip play land
224         return ability instanceof PlayLandAbility;
225     }

227     private Abilities<Ability> getInnerAbilities(Game game, boolean showLeftSide, boolean showRightSide) {
228         Abilities<Ability> allAbilites = new AbilitiesImpl<>();
229 
230         for (Ability ability : super.getAbilities(game)) {
231             if (isIgnoreDefaultAbility(ability)) {
232                 continue;
233             }
234             allAbilites.add(ability);
235         }
236 
237         if (showLeftSide) {
238             allAbilites.addAll(leftHalfCard.getAbilities(game));
239         }
240         if (showRightSide) {
241             allAbilites.addAll(rightHalfCard.getAbilities(game));
242         }
243 
244         return allAbilites;
245     }
246 
247     private Abilities<Ability> getInnerAbilities(boolean showLeftSide, boolean showRightSide) {
248         Abilities<Ability> allAbilites = new AbilitiesImpl<>();
249 
250         for (Ability ability : super.getAbilities()) {
251             if (isIgnoreDefaultAbility(ability)) {
252                 continue;
253             }
254             allAbilites.add(ability);
255         }
256 
257         if (showLeftSide) {
258             allAbilites.addAll(leftHalfCard.getAbilities());
259         }
260 
261         if (showRightSide) {
262             allAbilites.addAll(rightHalfCard.getAbilities());
263         }
264 
265         return allAbilites;
266     }

328     @Override
329     public int getManaValue() {
330         // Rules:
331         // The converted mana cost of a modal double-faced card is based on the characteristics of the
332         // face that’s being considered. On the stack and battlefield, consider whichever face is up.
333         // In all other zones, consider only the front face. This is different than how the converted
334         // mana cost of a transforming double-faced card is determined.
335 
336         // on stack or battlefield it must be half card with own cost
337         return leftHalfCard.getManaValue();
338     }

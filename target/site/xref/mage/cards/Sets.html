31      private Sets() {
32          List<String> packages = new ArrayList<>();
33          packages.add("mage.sets");
34          for (Class c : ClassScanner.findClasses(null, packages, ExpansionSet.class)) {
35              try {
36                  addSet((ExpansionSet) c.getMethod("getInstance").invoke(null));
37              } catch (Exception ex) {
38                  logger.error(ex);
39              }
40          }
41      }

43      public void addSet(ExpansionSet set) {
44          if (containsKey(set.getCode())) {
45              throw new IllegalArgumentException("Set code " + set.getCode() + " already exists.");
46          }
47          this.put(set.getCode(), set);
48          if (set.getSetType().isCustomSet()) {
49              customSets.add(set.getCode());
50          }
51      }

69      public static List<Card> generateRandomCardPool(int cardsCount, List<ColoredManaSymbol> allowedColors, boolean onlyBasicLands, List<String> allowedSets) {
70          CardCriteria criteria = new CardCriteria();
71  
72          if (onlyBasicLands) {
73              // only lands
74              criteria.rarities(Rarity.LAND);
75              criteria.colorless(true); // basic lands is colorless
76          } else {
77              // any card, but not basic lands
78              criteria.notTypes(CardType.LAND);
79  
80              // clear colors
81              criteria.white(false);
82              criteria.blue(false);
83              criteria.black(false);
84              criteria.red(false);
85              criteria.green(false);
86              criteria.colorless(false); // colorless is not allowed for gen
87          }
88  
89          if (allowedSets != null && allowedSets.size() > 0) {
90              for (String code : allowedSets) {
91                  criteria.setCodes(code);
92              }
93          }
94  
95          FilterMana manaNeed = new FilterMana();
96          for (ColoredManaSymbol color : allowedColors) {
97              switch (color) {
98                  case W:
99                      manaNeed.setWhite(true);
100                     criteria.white(true);
101                     break;
102                 case U:
103                     manaNeed.setBlue(true);
104                     criteria.blue(true);
105                     break;
106                 case B:
107                     manaNeed.setBlack(true);
108                     criteria.black(true);
109                     break;
110                 case R:
111                     manaNeed.setRed(true);
112                     criteria.red(true);
113                     break;
114                 case G:
115                     manaNeed.setGreen(true);
116                     criteria.green(true);
117                     break;
118             }
119         }
120         List<CardInfo> cards = CardRepository.instance.findCards(criteria);
121         if (cards.isEmpty()) {
122             throw new IllegalStateException("Can't find cards for chosen colors to generate deck: " + allowedColors);
123         }
124 
125         int count = 0;
126         int tries = 0;
127         List<Card> cardPool = new ArrayList<>();
128         while (count < cardsCount) {
129             CardInfo cardInfo = cards.get(RandomUtil.nextInt(cards.size()));
130             Card card = cardInfo != null ? cardInfo.getCard() : null;
131             if (card != null) {
132 
133                 FilterMana manaCard = card.getColorIdentity();
134                 boolean cardManaOK = true;
135 
136                 if (onlyBasicLands) {
137                     // lands is colorless
138                     // discard not needed color by mana produce
139                     //Assert.assertEquals("only basic lands allow, but found " + card.getName(), 1, card.getMana().size());
140                     for (Mana manaLand : card.getMana()) {
141                         if (manaLand.getWhite() > 0 && !manaNeed.isWhite()) {
142                             cardManaOK = false;
143                         }
144                         if (manaLand.getBlue() > 0 && !manaNeed.isBlue()) {
145                             cardManaOK = false;
146                         }
147                         if (manaLand.getBlack() > 0 && !manaNeed.isBlack()) {
148                             cardManaOK = false;
149                         }
150                         if (manaLand.getRed() > 0 && !manaNeed.isRed()) {
151                             cardManaOK = false;
152                         }
153                         if (manaLand.getGreen() > 0 && !manaNeed.isGreen()) {
154                             cardManaOK = false;
155                         }
156                         if (manaLand.getColorless() > 0) {
157                             cardManaOK = false;
158                         } // ignore colorless land (wastes)
159                     }
160                 } else {
161                     // cards
162                     // discard any card that have not needed color
163                     if (manaCard.isWhite() && !manaNeed.isWhite()) {
164                         cardManaOK = false;
165                     }
166                     if (manaCard.isBlue() && !manaNeed.isBlue()) {
167                         cardManaOK = false;
168                     }
169                     if (manaCard.isBlack() && !manaNeed.isBlack()) {
170                         cardManaOK = false;
171                     }
172                     if (manaCard.isRed() && !manaNeed.isRed()) {
173                         cardManaOK = false;
174                     }
175                     if (manaCard.isGreen() && !manaNeed.isGreen()) {
176                         cardManaOK = false;
177                     }
178                 }
179 
180                 if (cardManaOK) {
181                     cardPool.add(card);
182                     count++;
183                 }
184             }
185 
186             tries++;
187             if (tries > 4096) { // to avoid infinite loop
188                 throw new IllegalStateException("Not enough cards for chosen colors to generate deck: " + allowedColors);
189             }
190         }
191 
192         return cardPool;
193     }

195     public static ExpansionSet findSet(String code) {
196         if (instance.containsKey(code)) {
197             return instance.get(code);
198         }
199         return null;
200     }

202     public static ExpansionSet findSetByName(String setName) {
203         return instance.values().stream()
204                 .filter(expansionSet -> expansionSet.getName().equals(setName))
205                 .findFirst()
206                 .orElse(null);
207     }

209     public static ExpansionSet.SetCardInfo findCardByClass(Class<?> clazz, String preferredSetCode, String preferredCardNumber) {
210         ExpansionSet.SetCardInfo info = null;
211         if (instance.containsKey(preferredSetCode)) {
212             info = instance.get(preferredSetCode).findCardInfoByClass(clazz)
213                     .stream()
214                     .filter(card -> preferredCardNumber == null || card.getCardNumber().equals(preferredCardNumber))
215                     .findFirst()
216                     .orElse(null);
217         }
218 
219         if (info == null) {
220             for (Map.Entry<String, ExpansionSet> entry : instance.entrySet()) {
221                 info = entry.getValue().findCardInfoByClass(clazz)
222                         .stream()
223                         .filter(card -> preferredCardNumber == null || card.getCardNumber().equals(preferredCardNumber))
224                         .findFirst()
225                         .orElse(null);
226                 if (info != null) {
227                     break;
228                 }
229             }
230         }
231 
232         return info;
233     }

1   package mage.utils;
2   
3   import mage.util.JarVersion;
4   
5   import java.io.Serializable;
6   
7   /**
8    * @author BetaSteward_at_googlemail.com
9    */
10  public class MageVersion implements Serializable, Comparable<MageVersion> {
11  
12      // version must be compatible with maven version numbers
13      // launcher can update only to the newest version
14      // example: 1.4.48-V1-beta3
15      // * 1.4.48 compares as numbers
16      // * V1-beta3 is qualifier and compares as string
17      // * launcher gives priority to 1.4.48 instead 1.4.48-any-text, so don't use empty release info
18      public static final int MAGE_VERSION_MAJOR = 1;
19      public static final int MAGE_VERSION_MINOR = 4;
20      public static final int MAGE_VERSION_RELEASE = 50;
21      public static final String MAGE_VERSION_RELEASE_INFO = "V2"; // V1 for releases, V1-beta3 for betas
22  
23      // strict mode
24      // Each update requires a strict version
25      // If you disable it then server can accept multiple versions with same release number
26      // Since incompatible changes are no longer monitored - it must always be true
27      private static final boolean MAGE_VERSION_RELEASE_INFO_MUST_BE_SAME = true;
28      // build info
29      public static final boolean MAGE_VERSION_SHOW_BUILD_TIME = true;
30  
31      private final int major;
32      private final int minor;
33      private final int release;
34      private final String releaseInfo;
35      private final String buildTime;
36  
37      public MageVersion(Class sourceClass) {
38          this(MAGE_VERSION_MAJOR, MAGE_VERSION_MINOR, MAGE_VERSION_RELEASE, MAGE_VERSION_RELEASE_INFO, sourceClass);
39      }
40  
41      public MageVersion(int major, int minor, int release, String releaseInfo, Class sourceClass) {
42          this.major = major;
43          this.minor = minor;
44          this.release = release;
45          this.releaseInfo = releaseInfo;
46  
47          if (!releaseInfo.startsWith("V")) {
48              // release: V1
49              // beta: V1-beta3
50              throw new IllegalArgumentException("ERROR, release info must be started from V.");
51          }
52  
53          // build time
54          this.buildTime = JarVersion.getBuildTime(sourceClass);
55      }
56  
57      public String toString(boolean showBuildTime) {
58          // 1.4.32-V1-beta2 (build: time)
59          String res = major + "." + minor + '.' + release + "-" + releaseInfo;
60          if (showBuildTime && !this.buildTime.isEmpty()) {
61              res += " (build: " + this.buildTime + ")";
62          }
63          return res;
64      }
65  
66      @Override
67      public String toString() {
68          return toString(MAGE_VERSION_SHOW_BUILD_TIME);
69      }
70  
71      @Override
72      public int compareTo(MageVersion o) {
73          if (major != o.major) {
74              return major - o.major;
75          }
76          if (minor != o.minor) {
77              return minor - o.minor;
78          }
79          if (release != o.release) {
80              return release - o.release;
81          }
82  
83          if (MAGE_VERSION_RELEASE_INFO_MUST_BE_SAME && !releaseInfo.equals(o.releaseInfo)) {
84              return releaseInfo.compareTo(o.releaseInfo);
85          }
86  
87          // all fine
88          return 0;
89      }
90  }

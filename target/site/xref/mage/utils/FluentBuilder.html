1   package mage.utils;

3   import java.util.ArrayList;
4   import java.util.function.Consumer;
5   import java.util.function.Supplier;
6   
7   /**
8    * A base class for fluent, immutable, composable builders.
9    *
10   * @see <a href="https://github.com/fburato/functionalutils/blob/master/utils/src/main/java/com/github/fburato/functionalutils/utils/Builder.java">Builder</a>
11   */
12  public abstract class FluentBuilder<ToBuild, RealBuilder extends FluentBuilder<ToBuild, RealBuilder>> {
13  
14      final ArrayList<Consumer<RealBuilder>> buildSequence;
15      private final Supplier<RealBuilder> newReference;
16  
17      protected FluentBuilder(Supplier<RealBuilder> newReference) {
18          this.buildSequence = new ArrayList<>();
19          this.newReference = newReference;
20      }
21  
22      private RealBuilder copy() {
23          final RealBuilder realBuilder = newReference.get();
24          realBuilder.buildSequence.addAll(buildSequence);
25          return realBuilder;
26      }
27  
28      protected abstract ToBuild makeValue();
29  
30      public RealBuilder with(Consumer<RealBuilder> consumer) {
31          final RealBuilder nextBuilder = this.copy();
32          nextBuilder.buildSequence.add(consumer);
33          return nextBuilder;
34      }
35  
36      public ToBuild build() {
37          final RealBuilder instance = this.copy();
38          instance.buildSequence.forEach(c -> c.accept(instance));
39          return instance.makeValue();
40      }
41  }

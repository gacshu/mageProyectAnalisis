38      public synchronized static Deck buildDeck(List<Card> spellCardPool, List<ColoredManaSymbol> allowedColors, List<String> setsToUse, List<Card> landCardPool, int deckCardSize, RateCallback callback) {
39          deckSize = deckCardSize;
40          deck = new Deck();
41  
42          final Collection<MageScoredCard> remainingCards = new ArrayList<>();
43          Set<String> names = new HashSet<>();
44          for (final Card card : spellCardPool) {
45              if (names.contains(card.getName())) {
46                  continue;
47              }
48              remainingCards.add(new MageScoredCard(card, allowedColors, callback));
49              names.add(card.getName());
50          }
51  // prints score and manaScore to log
52  //        for(MageScoredCard scoreCard :remainingCards) {
53  //            Logger.getLogger(DeckBuilder.class).info(
54  //                    new StringBuilder("Score: ")
55  //                    .append(scoreCard.getScore())
56  //                    .append(" ManaScore: ")
57  //                    .append(scoreCard.getManaCostScore(scoreCard.getCard(), allowedColors))
58  //                    .append(" ")
59  //                    .append(scoreCard.getCard().getName())
60  //                    .append(" ")
61  //                    .append(scoreCard.getCard().getManaCost().getText()).toString()
62  //                    );
63  //        }
64          int min = 0;
65          if (deckSize == 40) {
66              deckCount = DECK_COUNT40;
67              deckSpells = 23;
68              deckLands = 17;
69          } else {
70              deckCount = DECK_COUNT60;
71              deckSpells = 35;
72              deckLands = 25;
73          }
74  
75          for (int index = 0; index < deckCount.length; index++) {
76              final int max = DECK_COST[index];
77              addCardsToDeck(remainingCards, min, max, deckCount[index]);
78              min = max + 1;
79          }
80          addCardsToDeck(remainingCards, 0, 4, deckSpells - deck.getCards().size());
81          addCardsToDeck(remainingCards, 5, 10, deckSpells - deck.getCards().size());
82          addLandsToDeck(allowedColors, setsToUse, landCardPool, callback);
83  
84          Deck returnedDeck = deck;
85          deck = null;
86          return returnedDeck;
87      }

89      /**
90       * Chosed best scored card and adds it to the deck.
91       *
92       * @param remainingCards
93       * @param minCost
94       * @param maxCost
95       * @param count
96       */
97      private static void addCardsToDeck(final Collection<MageScoredCard> remainingCards, final int minCost, final int maxCost,
98                                         final int count) {
99  
100         for (int c = count; c > 0; c--) {
101 
102             MageScoredCard bestCard = null;
103             int bestScore = -1;
104 
105             for (final MageScoredCard draftedCard : remainingCards) {
106 
107                 final int score = draftedCard.getScore();
108                 final int cost = draftedCard.getConvertedCost();
109                 if (score > bestScore && cost >= minCost && cost <= maxCost) {
110                     bestScore = score;
111                     bestCard = draftedCard;
112                 }
113             }
114 
115             if (bestCard == null || bestScore < MIN_CARD_SCORE) {
116                 break;
117             }
118             deck.getCards().add(bestCard.card);
119             remainingCards.remove(bestCard);
120         }
121     }

123     /**
124      * Adds lands from non basic land (if provided), adds basic lands getting
125      * them from provided {@link RateCallback}}.
126      *
127      * @param allowedColors
128      * @param landCardPool
129      * @param callback
130      */
131     private static void addLandsToDeck(List<ColoredManaSymbol> allowedColors, List<String> setsToUse, List<Card> landCardPool, RateCallback callback) {
132 
133         // Calculate statistics per color.
134         final Map<String, Integer> colorCount = new HashMap<>();
135         for (final Card card : deck.getCards()) {
136 
137             for (String symbol : card.getManaCostSymbols()) {
138                 int count = 0;
139                 symbol = symbol.replace("{", "").replace("}", "");
140                 if (isColoredMana(symbol)) {
141                     for (ColoredManaSymbol allowed : allowedColors) {
142                         if (symbol.contains(allowed.toString())) {
143                             count++;
144                         }
145                     }
146                     if (count > 0) {
147                         Integer typeCount = colorCount.getOrDefault(symbol, 0);
148                         typeCount += 1;
149                         colorCount.put(symbol, typeCount);
150                     }
151                 }
152             }
153         }
154 
155         // Add suitable non basic lands to deck in order of pack.
156         final Map<String, Integer> colorSource = new HashMap<>();
157         for (final ColoredManaSymbol color : ColoredManaSymbol.values()) {
158             colorSource.put(color.toString(), 0);
159         }
160         if (landCardPool != null) {
161             for (final Card landCard : landCardPool) {
162                 deck.getCards().add(landCard);
163                 for (Mana mana : landCard.getMana()) {
164                     for (ColoredManaSymbol color : allowedColors) {
165                         int amount = mana.getColor(color);
166                         if (amount > 0) {
167                             Integer count = colorSource.get(color.toString());
168                             count += amount;
169                             colorSource.put(color.toString(), count);
170                         }
171                     }
172                 }
173             }
174         }
175 
176         // Add optimal basic lands to deck.
177         while (deck.getCards().size() < deckSize) {
178             ColoredManaSymbol bestColor = null;
179             //Default to a color in the allowed colors
180             if (allowedColors != null && !allowedColors.isEmpty()) {
181                 bestColor = allowedColors.get(RandomUtil.nextInt(allowedColors.size()));
182             }
183             int lowestRatio = Integer.MAX_VALUE;
184             for (final ColoredManaSymbol color : ColoredManaSymbol.values()) {
185 
186                 final Integer count = colorCount.get(color.toString());
187                 if (count != null && count > 0) {
188                     final int source = colorSource.get(color.toString());
189                     final int ratio;
190                     if (source < MIN_SOURCE) {
191                         ratio = source - count;
192                     } else {
193                         ratio = source * 100 / count;
194                     }
195                     if (ratio < lowestRatio) {
196                         lowestRatio = ratio;
197                         bestColor = color;
198                     }
199                 }
200             }
201             final Card landCard = callback.getBestBasicLand(bestColor, setsToUse);
202             Integer count = colorSource.get(bestColor.toString());
203             count++;
204             colorSource.put(bestColor.toString(), count);
205             deck.getCards().add(landCard);
206         }
207     }

209     private static class MageScoredCard {
210 
211         private Card card;
212         private final int score;
213 
214         private static final int[] SINGLE_PENALTY = {0, 1, 1, 3, 6, 9};
215         //private static final int DOUBLE_PENALTY[] = { 0, 0, 1, 2, 4, 6 };
216 
217         public MageScoredCard(Card card, List<ColoredManaSymbol> allowedColors, RateCallback cardRater) {
218             this.card = card;
219 
220             int type;
221             if (card.isCreature()) {
222                 type = 10;
223             } else if (card.hasSubtype(SubType.EQUIPMENT, null)) {
224                 type = 8;
225             } else if (card.hasSubtype(SubType.AURA, null)) {
226                 type = 5;
227             } else if (card.isInstant()) {
228                 type = 7;
229             } else {
230                 type = 6;
231             }
232 
233             this.score
234                     = // 5*card.getValue() + // not possible now
235                     3 * cardRater.rateCard(card)
236                             + // 3*card.getRemoval() + // not possible now
237                             type + getManaCostScore(card, allowedColors);
238         }
239 
240         private int getManaCostScore(Card card, List<ColoredManaSymbol> allowedColors) {
241             int converted = card.getManaValue();
242             final Map<String, Integer> singleCount = new HashMap<>();
243             int maxSingleCount = 0;
244             int multicolor = 0;
245             Set<String> colors = new HashSet<>();
246             for (String symbol : card.getManaCostSymbols()) {
247                 int count = 0;
248                 symbol = symbol.replace("{", "").replace("}", "");
249                 if (isColoredMana(symbol)) {
250                     for (ColoredManaSymbol allowed : allowedColors) {
251                         if (symbol.contains(allowed.toString())) {
252                             count++;
253                         }
254                     }
255                     // colored but no selected colors, go back with negative value
256                     if (count == 0) {
257                         return -30;
258                     }
259                     if (!colors.contains(symbol)) {
260                         multicolor += 1;
261                         colors.add(symbol);
262                     }
263                     Integer typeCount = singleCount.getOrDefault(symbol, 0);
264                     typeCount += 1;
265                     singleCount.put(symbol, typeCount);
266                     maxSingleCount = Math.max(maxSingleCount, typeCount);
267                 }
268             }
269             int multicolorBonus = multicolor > 1 ? 30 : 0;
270             maxSingleCount = Math.min(maxSingleCount, SINGLE_PENALTY.length - 1);
271             return 2 * converted + 3 * (10 - SINGLE_PENALTY[maxSingleCount]/*-DOUBLE_PENALTY[doubleCount]*/) + multicolorBonus;
272         }

287     protected static boolean isColoredMana(String symbol) {
288         return symbol.equals("W") || symbol.equals("G") || symbol.equals("U") || symbol.equals("B") || symbol.equals("R") || symbol.contains("/");
289     }

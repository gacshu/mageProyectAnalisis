2   package mage.tournament;
3   
4   import mage.constants.TournamentPlayerState;
5   import mage.game.draft.DraftOptions;
6   import mage.game.draft.RichManCubeBoosterDraft;
7   import mage.game.events.TableEvent.EventType;
8   import mage.game.tournament.TournamentOptions;
9   import mage.game.tournament.TournamentPlayer;
10  import mage.game.tournament.TournamentSingleElimination;
11  
12  /**
13   *
14   * @author spjspj
15   */
16  public class RichManCubeDraftEliminationTournament extends TournamentSingleElimination {
17  
18      protected enum TournamentStep {
19          START, DRAFT, CONSTRUCT, COMPETE, WINNERS
20      }
21  
22      protected TournamentStep currentStep;
23  
24      public RichManCubeDraftEliminationTournament(TournamentOptions options) {
25          super(options);
26          currentStep = TournamentStep.START;
27      }
28  
29      protected void draft() {
30          draft = new RichManCubeBoosterDraft((DraftOptions) options.getLimitedOptions(), getSets());
31          for (TournamentPlayer player : players.values()) {
32              draft.addPlayer(player.getPlayer());
33              player.setState(TournamentPlayerState.DRAFTING);
34          }
35          tableEventSource.fireTableEvent(EventType.START_DRAFT, null, draft);
36      }
37  
38      @Override
39      public void nextStep() {
40          if (isAbort()) {
41              currentStep = TournamentStep.COMPETE;
42          }
43          switch (currentStep) {
44              case START:
45                  currentStep = TournamentStep.DRAFT;
46                  draft();
47                  break;
48              case DRAFT:
49                  currentStep = TournamentStep.CONSTRUCT;
50                  construct();
51                  break;
52              case CONSTRUCT:
53                  currentStep = TournamentStep.COMPETE;
54                  runTournament();
55                  break;
56              case COMPETE:
57                  currentStep = TournamentStep.WINNERS;
58                  winners();
59                  end();
60                  break;
61          }
62      }
63  }

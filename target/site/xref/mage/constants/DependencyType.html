3   /**
4    * Dependency types are a part of a workaround solution to handle dependencies
5    * of continuous effects.
6    * <p>
7    * All continuous effects can: addDependencyType -- make dependency (effect
8    * makes some changes) addDependedToType -- wait another dependency (effect must
9    * wait until all other effects finished)
10   * <p>
11   * http://magiccards.info/rule/613-interaction-of-continuous-effects.html
12   * https://github.com/magefree/mage/issues/1259
13   *
14   * @author LevelX2
15   */
16  public enum DependencyType {
17      AuraAddingRemoving,
18      ArtifactAddingRemoving,
19      AddingAbility,
20      AddingCreatureType,
21      BecomeNonbasicLand,
22      BecomeForest,
23      BecomeIsland,
24      BecomeMountain,
25      BecomePlains,
26      BecomeSwamp,
27      BecomeCreature,
28      EnchantmentAddingRemoving,
29      LooseDefenderEffect
30  }

11  public enum SpellAbilityCastMode {

23

27      }
28  
29      public Card getTypeModifiedCardObjectCopy(Card card, Game game) {
30          Card cardCopy = card.copy();
31          if (this.equals(BESTOW)) {
32              BestowAbility.becomeAura(cardCopy);
33          }
34          return cardCopy;
35      }

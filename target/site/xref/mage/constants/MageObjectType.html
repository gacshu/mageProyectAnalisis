7   /**
8    * 109.1. An object is an ability on the stack, a card, a copy of a card, a
9    * token, a spell, a permanent, or an emblem. 109.2. If a spell or ability uses
10   * a description of an object that includes a card type or subtype, but doesn't
11   * include the word "card," "spell," "source," or "scheme," it means a permanent
12   * of that card type or subtype on the battlefield. 109.2a If a spell or ability
13   * uses a description of an object that includes the word "card" and the name of
14   * a zone, it means a card matching that description in the stated zone. #
15   * 109.2b If a spell or ability uses a description of an object that includes
16   * the word "spell," it means a spell matching that description on the stack. #
17   * 109.2c If a spell or ability uses a description of an object that includes
18   * the word "source," it means a source matching that description -- either a
19   * source of an ability or a source of damage -- in any zone. See rule 609.7. #
20   * 109.2d If an ability of a scheme card includes the text "this scheme," it
21   * means the scheme card in the command zone on which that ability is printed. #
22   * 109.3. An object's characteristics are name, mana cost, color, color
23   * indicator, card type, subtype, supertype, expansion symbol, rules text,
24   * abilities, power, toughness, loyalty, hand modifier, and life modifier.
25   * Objects can have some or all of these characteristics. Any other information
26   * about an object isn't a characteristic. For example, characteristics don't
27   * include whether a permanent is tapped, a spell's target, an object's owner or
28   * controller, what an Aura enchants, and so on. # 109.4. Only objects on the
29   * stack or on the battlefield have a controller. Objects that are neither on
30   * the stack nor on the battlefield aren't controlled by any player. See rule
31   * 108.4. There are four exceptions to this rule: # 109.4a An emblem is
32   * controlled by the player that puts it into the command zone. See rule 113,
33   * "Emblems." # 109.4b In a Planechase game, a face-up plane or phenomenon card
34   * is controlled by the player designated as the planar controller. This is
35   * usually the active player. See rule 901.6. # 109.4c In a Vanguard game, each
36   * vanguard card is controlled by its owner. See rule 902.6. # 109.4d In an
37   * Archenemy game, each scheme card is controlled by its owner. See rule 904.7.
38   * # 109.5. The words "you" and "your" on an object refer to the object's
39   * controller, its would-be controller (if a player is attempting to play, cast,
40   * or activate it), or its owner (if it has no controller). For a static
41   * ability, this is the current controller of the object it's on. For an
42   * activated ability, this is the player who activated the ability. For a
43   * triggered ability, this is the controller of the object when the ability
44   * triggered, unless it's a delayed triggered ability. To determine the
45   * controller of a delayed triggered ability, see rules 603.7d-f
46   *
47   * @author LevelX2
48   */
49  public enum MageObjectType {
50      ABILITY_STACK("Ability on the Stack", false, false),
51      CARD("Card", false, true),
52      COPY_CARD("Copy of a Card", false, true),
53      TOKEN("Token", true, true),
54      SPELL("Spell", false, true),
55      PERMANENT("Permanent", true, true),
56      DUNGEON("Dungeon", false, false),
57      EMBLEM("Emblem", false, false),
58      COMMANDER("Commander", false, false),
59      DESIGNATION("Designation", false, false),
60      PLANE("Plane", false, false),
61      NULL("NullObject", false, false);
62  
63      private final String text;
64      private final boolean permanent;
65      private final boolean canHaveCounters;
66  
67      MageObjectType(String text, boolean permanent, boolean canHaveCounters) {
68          this.text = text;
69          this.permanent = permanent;
70          this.canHaveCounters = canHaveCounters;
71      }
72  
73      @Override
74      public String toString() {
75          return text;
76      }
77  
78      public boolean isPermanent() {
79          return permanent;
80      }
81  
82      public boolean canHaveCounters() {
83          return canHaveCounters;
84      }
85  
86  }

22  @Ignore
23  public class MultiConnectTest {
24  
25      private static final Logger logger = Logger.getLogger(MultiConnectTest.class);
26  
27      /**
28       * Amount of games to be started from this test.
29       */
30      private static final Integer USER_CONNECT_COUNT = 200;
31  
32      private static final CountDownLatch latch = new CountDownLatch(USER_CONNECT_COUNT);
33  
34      private static final MageVersion version = new MageVersion(MultiConnectTest.class);
35  
36      private static volatile int connected;
37  
38      private final Object sync = new Object();
39      private MageUI ui;
40  
41      private static class ClientMock implements MageClient {
42  
43          private Session session;
44          private final String username;
45  
46          public ClientMock(String username) {
47              this.username = username;
48          }
49  
50          public void connect() {
51              session = new SessionImpl(this);
52              Connection connection = new Connection();
53              connection.setUsername(username);
54              connection.setHost("localhost");
55              connection.setPort(17171);
56              connection.setProxyType(Connection.ProxyType.NONE);
57  
58              session.connect(connection);
59          }
60  
61          @Override
62          public MageVersion getVersion() {
63              logger.info("getVersion");
64              return version;
65          }
66  
67          @Override
68          public void connected(String message) {
69              logger.info("connected: " + message);
70              connected++;
71          }
72  
73          @Override
74          public void disconnected(boolean askToReconnect) {
75              logger.info("disconnected");
76          }
77  
78          @Override
79          public void showMessage(String message) {
80              logger.info("showMessage: " + message);
81          }
82  
83          @Override
84          public void showError(String message) {
85              logger.info("showError: " + message);
86          }
87  
88          @Override
89          public void processCallback(ClientCallback callback) {
90              logger.info("processCallback");
91          }
92      }
93  
94      public static void main(String[] argv) throws Exception {
95          new MultiConnectTest().startMultiGames();
96      }
97  
98      public void startMultiGames() throws Exception {
99          for (int i = 0; i < USER_CONNECT_COUNT; i++) {
100             logger.info("Starting game");
101             connect(i);
102         }
103         latch.await();
104         logger.info("Finished");
105         logger.info("Connected: " + connected + " of " + USER_CONNECT_COUNT);
106     }
107 
108     private void connect(final int index) throws Exception {
109         Thread.setDefaultUncaughtExceptionHandler((t, e) -> logger.fatal(null, e));
110         SwingUtilities.invokeLater(() -> {
111             String username = "player" + index;
112             ClientMock client = new ClientMock(username);
113             client.connect();
114             latch.countDown();
115         });
116     }
117 
118     private void sleep(int ms) {
119         try {
120             TimeUnit.MILLISECONDS.sleep(ms);
121         } catch (Exception e) {
122             e.printStackTrace();
123         }
124     }

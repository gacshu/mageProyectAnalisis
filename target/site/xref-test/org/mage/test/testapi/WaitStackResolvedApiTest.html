13      @Test
14      public void test_Spells() {
15          addCard(Zone.HAND, playerA, "Firebolt", 1); // sorcery
16          addCard(Zone.HAND, playerA, "Lightning Bolt", 2); // instant
17          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
18  
19          checkStackSize("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
20  
21          // prepare - cast 3 spells
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Firebolt", playerB);
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
25          checkStackSize("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 3);
26          checkStackObject("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Firebolt", 1);
27          checkStackObject("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", 2);
28  
29          // skip 1 (bolt)
30          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
31          checkStackSize("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
32          checkStackObject("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Firebolt", 1);
33          checkStackObject("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", 1);
34  
35          // skip 2 (bolt)
36          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
37          checkStackSize("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1);
38          checkStackObject("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Firebolt", 1);
39          checkStackObject("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", 0);
40  
41          // skip 3 (firebolt)
42          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
43          checkStackSize("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
44          checkStackObject("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Firebolt", 0);
45          checkStackObject("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", 0);
46  
47          setStrictChooseMode(true);
48          setStopAt(1, PhaseStep.END_TURN);
49          execute();
50      }

52      @Test
53      public void test_Abilities() {
54          // {1}{R}, Sacrifice Pyromania: Pyromania deals 1 damage to any target.
55          addCard(Zone.BATTLEFIELD, playerA, "Pyromania", 1);
56          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
57          //
58          // {2}: Goblin Cannon deals 1 damage to any target. Sacrifice Goblin Cannon.
59          addCard(Zone.BATTLEFIELD, playerA, "Goblin Cannon", 2);
60          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2 * 2);
61  
62          String goblinAbility = "{2}:";
63          String pyroAbility = "{1}{R}, Sacrifice";
64  
65          checkStackSize("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
66  
67          // prepare - activate 3 abilities
68          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, playerB);
69          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, pyroAbility, playerB);
70          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, playerB);
71          checkStackSize("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 3);
72          checkStackObject("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, pyroAbility, 1);
73          checkStackObject("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, 2);
74          showStack("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
75  
76          // skip 1 (goblin)
77          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
78          checkStackSize("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
79          checkStackObject("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, pyroAbility, 1);
80          checkStackObject("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, 1);
81  
82          // skip 2 (pyro)
83          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
84          checkStackSize("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1);
85          checkStackObject("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, pyroAbility, 0);
86          checkStackObject("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, 1);
87  
88          // skip 3 (goblin)
89          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
90          checkStackSize("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
91          checkStackObject("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, pyroAbility, 0);
92          checkStackObject("after 3", 1, PhaseStep.PRECOMBAT_MAIN, playerA, goblinAbility, 0);
93  
94          setStrictChooseMode(true);
95          setStopAt(1, PhaseStep.END_TURN);
96          execute();
97      }

20      @Test
21      public void blockerNotFoundTest() {
22          addCard(Zone.BATTLEFIELD, playerA, "Ember Swallower");
23  
24          attack(3, playerA, "Ember Swallower");
25          // Blocking but playerB doesn't control a Sedge Scorpion
26          block(3, playerB, "Sedge Scorpion", "Ember Swallower");
27  
28          setStopAt(3, PhaseStep.END_TURN);
29  
30          try {
31              execute();
32              fail("Expected exception not thrown");
33          } catch (AssertionError e) {
34              // Error - blocking with permanents you don't control
35              TestCase.assertEquals("No permanents found called Sedge Scorpion that match the filter criteria \"permanent you control\"", e.getMessage());
36          }
37      }

217     @Test
218     public void minimumBlockNotReachedTest() {
219         /* Underworld Cerberus {3}{B}{3} 6/6
220          *  Underworld Cerberus can't be blocked except by three or more creatures.
221          *  Cards in graveyards can't be the targets of spells or abilities.
222          *  When Underworld Cerberus dies, exile it and each player returns all creature cards from their graveyard to their hand.
223          */
224         addCard(Zone.BATTLEFIELD, playerA, "Underworld Cerberus");
225         addCard(Zone.BATTLEFIELD, playerB, "Memnite", 2); // 1/1
226 
227         attack(3, playerA, "Underworld Cerberus");
228         block(3, playerB, "Memnite:0", "Underworld Cerberus");
229         block(3, playerB, "Memnite:1", "Underworld Cerberus");
230 
231         setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
232 
233         try {
234             execute();
235             fail("Expected exception not thrown");
236         } catch (UnsupportedOperationException e) {
237             assertEquals("Underworld Cerberus is blocked by 2 creature(s). It has to be blocked by 3 or more.", e.getMessage());
238         }

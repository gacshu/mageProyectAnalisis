19      @Test
20      public void test_NamesEquals() {
21          // empty names for face down cards
22          Assert.assertTrue(CardUtil.haveEmptyName(""));
23          Assert.assertTrue(CardUtil.haveEmptyName(EmptyNames.FACE_DOWN_CREATURE.toString()));
24          Assert.assertFalse(CardUtil.haveEmptyName(" "));
25          Assert.assertFalse(CardUtil.haveEmptyName("123"));
26          Assert.assertFalse(CardUtil.haveEmptyName("Sample Name"));
27  
28          // same names (empty names can't be same)
29          Assert.assertFalse(CardUtil.haveSameNames("", ""));
30          Assert.assertFalse(CardUtil.haveSameNames(EmptyNames.FACE_DOWN_CREATURE.toString(), ""));
31          Assert.assertFalse(CardUtil.haveSameNames(EmptyNames.FACE_DOWN_CREATURE.toString(), EmptyNames.FACE_DOWN_CREATURE.toString()));
32          Assert.assertFalse(CardUtil.haveSameNames(EmptyNames.FACE_DOWN_TOKEN.toString(), ""));
33          Assert.assertFalse(CardUtil.haveSameNames(EmptyNames.FACE_DOWN_TOKEN.toString(), EmptyNames.FACE_DOWN_CREATURE.toString()));
34          Assert.assertTrue(CardUtil.haveSameNames("Name", "Name"));
35          Assert.assertFalse(CardUtil.haveSameNames("Name", ""));
36          Assert.assertFalse(CardUtil.haveSameNames("Name", " "));
37          Assert.assertFalse(CardUtil.haveSameNames("Name", "123"));
38          Assert.assertFalse(CardUtil.haveSameNames("Name", EmptyNames.FACE_DOWN_CREATURE.toString()));
39          Assert.assertFalse(CardUtil.haveSameNames("Name1", "Name2"));
40  
41          // ignore mtg rules (empty names must be same)
42          Assert.assertTrue(CardUtil.haveSameNames("", "", true));
43          Assert.assertTrue(CardUtil.haveSameNames(EmptyNames.FACE_DOWN_CREATURE.toString(), EmptyNames.FACE_DOWN_CREATURE.toString(), true));
44          Assert.assertTrue(CardUtil.haveSameNames("Name", "Name", true));
45          Assert.assertFalse(CardUtil.haveSameNames("Name", "", true));
46          Assert.assertFalse(CardUtil.haveSameNames("Name", " ", true));
47          Assert.assertFalse(CardUtil.haveSameNames("Name", "123", true));
48          Assert.assertFalse(CardUtil.haveSameNames("Name", EmptyNames.FACE_DOWN_CREATURE.toString(), true));
49          Assert.assertFalse(CardUtil.haveSameNames("Name1", "Name2", true));
50  
51          // name with split card
52          Card splitCard1 = CardRepository.instance.findCard("Armed // Dangerous").getCard();
53          Card splitCard2 = CardRepository.instance.findCard("Alive // Well").getCard();
54          Assert.assertTrue(CardUtil.haveSameNames(splitCard1, "Armed", currentGame));
55          Assert.assertTrue(CardUtil.haveSameNames(splitCard1, "Dangerous", currentGame));
56          Assert.assertTrue(CardUtil.haveSameNames(splitCard1, "Armed // Dangerous", currentGame));
57          Assert.assertTrue(CardUtil.haveSameNames(splitCard1, splitCard1));
58          Assert.assertFalse(CardUtil.haveSameNames(splitCard1, "Other", currentGame));
59          Assert.assertFalse(CardUtil.haveSameNames(splitCard1, "Other // Dangerous", currentGame));
60          Assert.assertFalse(CardUtil.haveSameNames(splitCard1, "Armed // Other", currentGame));
61          Assert.assertFalse(CardUtil.haveSameNames(splitCard1, splitCard2));
62  
63          // name with face down spells: face down spells don't have names, see https://github.com/magefree/mage/issues/6569
64          Card bearCard = CardRepository.instance.findCard("Balduvian Bears").getCard();
65          Spell normalSpell = new Spell(bearCard, bearCard.getSpellAbility(), playerA.getId(), Zone.HAND, currentGame);
66          Spell faceDownSpell = new Spell(bearCard, bearCard.getSpellAbility(), playerA.getId(), Zone.HAND, currentGame);
67          faceDownSpell.setFaceDown(true, currentGame);
68          // normal spell
69          Assert.assertFalse(CardUtil.haveSameNames(normalSpell, "", currentGame));
70          Assert.assertFalse(CardUtil.haveSameNames(normalSpell, "Other", currentGame));
71          Assert.assertFalse(CardUtil.haveSameNames(normalSpell, EmptyNames.FACE_DOWN_CREATURE.toString(), currentGame));
72          Assert.assertTrue(CardUtil.haveSameNames(normalSpell, "Balduvian Bears", currentGame));
73          // face down spell
74          Assert.assertFalse(CardUtil.haveSameNames(faceDownSpell, "", currentGame));
75          Assert.assertFalse(CardUtil.haveSameNames(faceDownSpell, "Other", currentGame));
76          Assert.assertFalse(CardUtil.haveSameNames(faceDownSpell, EmptyNames.FACE_DOWN_CREATURE.toString(), currentGame));
77          Assert.assertFalse(CardUtil.haveSameNames(faceDownSpell, "Balduvian Bears", currentGame));
78      }

199     @Test
200     public void test_ActivateAbility_Alias() {
201         // {T}: Embermage Goblin deals 1 damage to any target.
202         addCard(Zone.BATTLEFIELD, playerA, "Embermage Goblin@goblin", 3);
203         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion@lion", 1);
204 
205         // use 2 of 3 goblins
206         showAvailableAbilities("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
207         showAliases("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
208         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "@goblin.1 {T}: {this} deals", "@lion");
209         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "@goblin.3 {T}: {this} deals", "@lion");
210 
211         setStopAt(1, PhaseStep.END_TURN);
212         execute();
213 
214         assertPermanentCount(playerA, "@goblin.1", 1);
215         assertPermanentCount(playerA, "@goblin.2", 1);
216         assertPermanentCount(playerA, "@goblin.3", 1);
217         assertGraveyardCount(playerA, "@lion", 1);
218         assertTapped("@goblin.1", true);
219         assertTapped("@goblin.2", false);
220         assertTapped("@goblin.3", true);
221     }

4   import mage.constants.PhaseStep;

11   *
12   * @author LevelX2
13   */
14  public class SkipTurnTest extends CardTestPlayerBase {
15  
16      @Test
17      public void testEaterOfDays() {
18          // At the beginning of your upkeep or whenever you cast a green spell, put a charge counter on Shrine of Boundless Growth.
19          // {T}, Sacrifice Shrine of Boundless Growth: Add {C} for each charge counter on Shrine of Boundless Growth.
20          addCard(Zone.HAND, playerA, "Shrine of Boundless Growth", 1);
21          addCard(Zone.BATTLEFIELD, playerA, "Island", 7);
22          // Flying
23          // Trample
24          // When Eater of Days enters the battlefield, you skip your next two turns.
25          addCard(Zone.HAND, playerA, "Eater of Days", 1);
26  
27          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Eater of Days", true);
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Shrine of Boundless Growth");
29  
30          setStopAt(5, PhaseStep.PRECOMBAT_MAIN);
31          execute();
32  
33          assertPermanentCount(playerA, "Eater of Days", 1);
34          assertPermanentCount(playerA, "Shrine of Boundless Growth", 1);
35          assertCounterCount("Shrine of Boundless Growth", CounterType.CHARGE, 1);
36      }
37  }

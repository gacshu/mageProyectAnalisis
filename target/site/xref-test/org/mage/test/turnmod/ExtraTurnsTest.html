15      /**
16       * Emrakul, the Promised End not giving an extra turn when cast in the
17       * opponent's turn
18       */
19      @Test
20      public void testEmrakulCastOnOpponentsTurnCheckTurn3() {
21          addCard(Zone.BATTLEFIELD, playerA, "Island", 12);
22          addCard(Zone.GRAVEYARD, playerA, "Island", 1);
23          // Emrakul, the Promised End costs {1} less to cast for each card type among cards in your graveyard.
24          // When you cast Emrakul, you gain control of target opponent during that player's next turn. After that turn, that player takes an extra turn.
25          // Flying
26          // Trample
27          // Protection from instants
28          addCard(Zone.HAND, playerA, "Emrakul, the Promised End", 1); // {13}
29          // Flash (You may cast this spell any time you could cast an instant.)
30          // Creature cards you own that aren't on the battlefield have flash.
31          // Each opponent can cast spells only any time they could cast a sorcery.
32          addCard(Zone.BATTLEFIELD, playerA, "Teferi, Mage of Zhalfir", 1);
33  
34          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Emrakul, the Promised End");
35          // Turn 4 is the next turn of opponent (player B)  that player A controls
36          // So Turn 5 is the extra turn for player B after Turn 4
37          setStopAt(3, PhaseStep.DRAW);
38          execute();
39  
40          assertPermanentCount(playerA, "Emrakul, the Promised End", 1);
41          Assert.assertTrue("Turn 3 is no extra turn ", !currentGame.getState().isExtraTurn());
42          Assert.assertEquals("For turn " + currentGame.getTurnNum() + ", playerA has to be the active player but active player is: "
43                  + currentGame.getPlayer(currentGame.getActivePlayerId()).getName(), currentGame.getActivePlayerId(), playerA.getId());
44      }

100     /**
101      * https://github.com/magefree/mage/issues/6824
102      *
103      * When you cast miracled Temporal Mastery with God-Eternal Kefnet on the
104      * battlefield and copy it with it's ability you get only 1 extra turn. It
105      * should be 2, since you cast Temporal Mastery with it's miracle ability +
106      * you get a copy from Kefnet's ability. Still after first extra turn game
107      * proceeds to next player.
108      */
109     @Test
110     public void testCopyMiracledTemporalMastery4TwoExtraTurns() {
111         setStrictChooseMode(true);
112 
113         addCard(Zone.BATTLEFIELD, playerB, "Island", 7);
114         // Flying
115         // You may reveal the first card you draw each turn as you draw it. Whenever you reveal an instant or sorcery card this way,
116         // copy that card and you may cast the copy. That copy costs {2} less to cast.
117         // When God-Eternal Kefnet dies or is put into exile from the battlefield, you may put it into its owner’s library third from the top.
118         addCard(Zone.BATTLEFIELD, playerB, "God-Eternal Kefnet", 1);
119         // Take an extra turn after this one. Exile Temporal Mastery.
120         // Miracle {1}{U} (You may cast this card for its miracle cost when you draw it if it's the first card you drew this turn.)
121         addCard(Zone.LIBRARY, playerB, "Temporal Mastery", 1); // Sorcery {5}{U}{U}
122         skipInitShuffling();
123 
124         setChoice(playerB, true); // Would you like to reveal first drawn card (Temporal Mastery, you can copy it and cast {2} less)?
125         setChoice(playerB, true); // Would you like to copy Temporal Mastery and cast it {2} less?
126         setChoice(playerB, true); // Reveal Temporal Mastery to be able to use Miracle?
127         setChoice(playerB, true); // Miracle {1}{U} (You may cast this card for its miracle cost when you draw it if it's the first card you drew this turn.)
128 
129         setChoice(playerB, false); // Would you like to reveal first drawn card? (Turn 3)
130         setChoice(playerB, false); // Would you like to reveal first drawn card? (Turn 4)
131 
132         // Turn 3 + 4 are extra turns
133         setStopAt(4, PhaseStep.PRECOMBAT_MAIN);
134         execute();
135 
136         assertExileCount(playerB, "Temporal Mastery", 1);
137 
138         Assert.assertTrue("Turn 4 is an extra turn ", currentGame.getState().isExtraTurn());
139         Assert.assertEquals("For turn " + currentGame.getTurnNum() + ", playerB has to be the active player but active player is: "
140                 + currentGame.getPlayer(currentGame.getActivePlayerId()).getName(), currentGame.getActivePlayerId(), playerB.getId());
141     }

34      private static String str(List<Card> cards) {
35          StringBuilder sb = new StringBuilder("[");
36          Iterator<Card> iterator = cards.iterator();
37          while (iterator.hasNext()) {
38              Card next = iterator.next();
39              sb.append(next.getName());
40              if (iterator.hasNext()) {
41                  sb.append(", ");
42              }
43          }
44          sb.append("] (").append(cards.size()).append(')');
45          return sb.toString();
46      }

48      private static boolean contains(List<Card> cards, List<String> names, String code) {
49          return names.stream().anyMatch((name) -> (contains(cards, name, code)));
50      }

52      private static boolean contains(List<Card> cards, String name, String code) {
53          return cards.stream().anyMatch((card)
54                  -> (card.getName().equals(name)
55                  && (code == null || card.getExpansionSetCode().equals(code)))
56          );
57      }

322     @Test
323     public void testKaldheim_SnowLandAndMDFC() {
324         boolean foundVale = false;
325         boolean foundMDFC = false;
326         boolean foundNoMDFC = false;
327 
328         for (int i = 1; i <= 100; i++) {
329             List<Card> booster = Kaldheim.getInstance().createBooster();
330 
331             assertEquals("Booster does not have 15 cards", 15, booster.size());
332             assertTrue(
333                     "Booster contains cards from another set",
334                     booster.stream().map(Card::getExpansionSetCode).allMatch("KHM"::equals)
335             );
336             assertFalse(
337                     "Booster cannot contain non-snow basic lands",
338                     booster.stream().anyMatch(card -> card.isBasic() && !card.isSnow())
339             );
340             assertEquals(
341                     "Booster must contain exactly 1 rare or mythic", 1,
342                     booster.stream().map(Card::getRarity).filter(rarity -> rarity == Rarity.RARE || rarity == Rarity.MYTHIC).count()
343             );
344             assertEquals(
345                     "Booster must contain exactly 3 uncommons", 3,
346                     booster.stream().map(Card::getRarity).filter(Rarity.UNCOMMON::equals).count()
347             );
348 
349             List<Card> snowLands = booster.stream().filter(card -> card.isSnow() && card.isLand(currentGame)).collect(Collectors.toList());
350             switch (snowLands.size()) {
351                 case 0:
352                     fail("Booster must have snow lands");
353                 case 1:
354                     Card snowLand = snowLands.get(0);
355                     assertTrue(
356                             "Only one snow land, must be basic or common",
357                             snowLand.isBasic() || snowLand.getRarity() == Rarity.COMMON
358                     );
359                     assertNotEquals(
360                             "Only one snow land, can't be Shimmerdrift Vale",
361                             "Shimmerdrift Vale", snowLand.getName()
362                     );
363                     assertNotEquals(
364                             "Only one snow land, can't be Faceless Haven",
365                             "Faceless Haven", snowLand.getName()
366                     );
367                     break;
368                 case 2:
369                     assertEquals(
370                             "Booster can't have two snow lands unless one is Shimmerdrift Vale or Faceless Haven", 1,
371                             snowLands.stream().filter(card -> card.getName().equals("Shimmerdrift Vale") || card.getName().equals("Faceless Haven")).count()
372                     );
373                     assertEquals(
374                             "Booster can't have two snow lands unless one is not Shimmerdrift Vale or Faceless Haven", 1,
375                             snowLands.stream().filter(card -> !card.getName().equals("Shimmerdrift Vale") && !card.getName().equals("Faceless Haven")).count()
376                     );
377                     break;
378                 case 3:
379                     assertEquals("Booster can't have three snow lands unless one is Shimmerdrift Vale", 1,
380                             snowLands.stream().filter(card -> card.getName().equals("Shimmerdrift Vale")).count()
381                     );
382                     assertEquals("Booster can't have three snow lands unless one is Faceless Haven", 1,
383                             snowLands.stream().filter(card -> card.getName().equals("Faceless Haven")).count()
384                     );
385                     assertEquals("Booster can't have three snow lands unless one is not Shimmerdrift Vale or Faceless Haven", 1,
386                             snowLands.stream().filter(card -> !card.getName().equals("Shimmerdrift Vale") && !card.getName().equals("Faceless Haven")).count()
387                     );
388                     break;
389                 default:
390                     fail("Booster can't have more than three snow lands");
391             }
392 
393             long mdfcCount = booster.stream().filter(ModalDoubleFacesCard.class::isInstance).count();
394             assertTrue("Booster can't have more than one MDFC", mdfcCount < 2);
395 
396             foundMDFC |= mdfcCount > 0;
397             foundNoMDFC |= mdfcCount == 0;
398             foundVale |= booster.stream().map(MageObject::getName).anyMatch("Shimmerdrift Vale"::equals);
399             if (foundVale && foundMDFC && foundNoMDFC && i > 20) {
400                 break;
401             }
402         }
403         assertTrue("No booster contained Shimmerdrift Vale", foundVale);
404         assertTrue("No booster contained an MDFC", foundMDFC);
405         assertTrue("Every booster contained an MDFC", foundNoMDFC);
406     }

566             });

577                         e.getKey(),

581                 .collect(Collectors.toList());

583                 + info.stream().collect(Collectors.joining("\n")));

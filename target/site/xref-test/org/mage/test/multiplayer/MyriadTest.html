69      @Test
70      public void CallerOfThePackTestExilePlaneswalker() {
71          // Trample
72          // Myriad (Whenever this creature attacks, for each opponent other than the defending player, put a token that's a copy of this creature onto the battlefield tapped and attacking that player or a planeswalker they control. Exile those tokens at the end of combat.)
73          addCard(Zone.BATTLEFIELD, playerD, "Caller of the Pack"); // 8/6
74  
75          // turns: A, D, C, B
76  
77          // +1: You gain 2 life.
78          // -1: Put a +1/+1 counter on each creature you control. Those creatures gain vigilance until end of turn.
79          // -6: Create a white Avatar creature token. It has "This creature's power and toughness are each equal to your life total."
80          addCard(Zone.BATTLEFIELD, playerA, "Ajani Goldmane");
81  
82          // +2 life
83          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1:");
84          checkLife("must have +2 life", 2, PhaseStep.PRECOMBAT_MAIN, playerA, 40 + 2);
85  
86          // D attack C, create 2 copy of caller (for each opponent exclude defender = 2) and attack to ajani
87          attack(2, playerD, "Caller of the Pack", playerC);
88          addTarget(playerD, "Ajani Goldmane"); // select ajani instead playerA for pack attack
89          checkPermanentCount("must have 3 packs", 2, PhaseStep.END_COMBAT, playerD, "Caller of the Pack", 3);
90          checkPermanentCount("ajani must die", 2, PhaseStep.END_COMBAT, playerA, "Ajani Goldmane", 0);
91          checkLife("pack must not damage playerA", 2, PhaseStep.END_COMBAT, playerA, 40 + 2);
92          checkLife("pack must damage playerB by 8", 2, PhaseStep.END_COMBAT, playerB, 40 - 8);
93          checkLife("pack must damage playerC", 2, PhaseStep.END_COMBAT, playerC, 40 - 8);
94          checkLife("pack must not damage playerD", 2, PhaseStep.END_COMBAT, playerD, 40);
95  
96          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
97          execute();
98  
99          assertPermanentCount(playerD, "Caller of the Pack", 1);
100         assertGraveyardCount(playerA, "Ajani Goldmane", 1);
101     }

103     /**
104      * 4 player commander game, Man-o-War equipped with Blade of Selves attacks.
105      * The myriad trigger creates copies for each opponent, including the
106      * defending player, rather than the just two not being attacked.
107      */
108     @Test
109     public void BladeOfSelves() {
110         addCard(Zone.BATTLEFIELD, playerC, "Silvercoat Lion", 2);
111         addCard(Zone.BATTLEFIELD, playerC, "Pillarfield Ox", 1);
112         addCard(Zone.BATTLEFIELD, playerC, "Grizzly Bears", 1);
113 
114         // Equipped creature has myriad. (Whenever this creature attacks, for each opponent other than the defending
115         //                                player, put a token that's a copy of this creature onto the battlefield tapped
116         //                                and attacking that player or a planeswalker they control.
117         //                                Exile those tokens at the end of combat.)
118         // Equip {4}
119         addCard(Zone.BATTLEFIELD, playerD, "Blade of Selves");
120         addCard(Zone.BATTLEFIELD, playerD, "Island", 4);
121 
122         // When Man-o'-War enters the battlefield, return target creature to its owner's hand.
123         addCard(Zone.HAND, playerD, "Man-o'-War"); // {2}{U}  2/2
124 
125         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Man-o'-War");
126         addTarget(playerD, "Silvercoat Lion");
127 
128         activateAbility(6, PhaseStep.PRECOMBAT_MAIN, playerD, "Equip", "Man-o'-War");
129 
130         attack(6, playerD, "Man-o'-War", playerC);
131         setChoice(playerD, "Yes"); // for playerB
132         setChoice(playerD, "Yes"); // for playerA
133         addTarget(playerD, "Silvercoat Lion");
134         addTarget(playerD, "Pillarfield Ox");
135 
136         setStopAt(6, PhaseStep.POSTCOMBAT_MAIN);
137         execute();
138 
139         // Tokens exiled, only original left
140         assertPermanentCount(playerD, "Man-o'-War", 1);
141 
142         // Returned to hand by Man-o'-War
143         assertPermanentCount(playerC, "Silvercoat Lion", 0);  // 1 by original and 1 by token
144         assertPermanentCount(playerC, "Pillarfield Ox", 0);   // by token
145 
146         // Still on the battlefield since only 3 Man-o'-War triggers
147         assertPermanentCount(playerC, "Grizzly Bears", 1);
148 
149         assertHandCount(playerC, "Silvercoat Lion", 2);
150         assertHandCount(playerC, "Pillarfield Ox", 1);
151 
152         assertLife(playerA, 38);
153         assertLife(playerB, 38);
154         assertLife(playerC, 38);
155         assertLife(playerD, 40);
156 
157     }

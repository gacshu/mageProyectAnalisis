18      @Override
19      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
20          Game game = new FreeForAll(MultiplayerAttackOption.MULTIPLE, RangeOfInfluence.ALL, MulliganType.GAME_DEFAULT.getMulligan(0), 40);
21          // Player order: A -> D -> C -> B
22          playerA = createPlayer(game, "PlayerA");
23          playerB = createPlayer(game, "PlayerB");
24          playerC = createPlayer(game, "PlayerC");
25          playerD = createPlayer(game, "PlayerD");
26          return game;
27      }

29      @Test
30      public void testMultiplayerAttackStinkdrinkerBanditTrigger() {
31          // Flash
32          // Flying
33          // When Pestermite enters the battlefield, you may tap or untap target permanent.
34          String pestermite = "Pestermite"; // 2/1
35  
36          // Prowl {1}, {B} (You may cast this for its prowl cost if you dealt combat damage to a player this turn with a Goblin or Rogue.)
37          // Whenever a Rogue you control attacks and isn't blocked, it gets +2/+1 until end of turn.
38          String stinkdrinker = "Stinkdrinker Bandit"; // 2/1
39          addCard(Zone.BATTLEFIELD, playerA, stinkdrinker);
40          addCard(Zone.BATTLEFIELD, playerA, pestermite, 1);
41  
42          attack(1, playerA, pestermite, playerB);
43          attack(1, playerA, stinkdrinker, playerC);
44  
45          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
46          execute();
47  
48          assertPowerToughness(playerA, stinkdrinker, 4, 2);
49          assertPowerToughness(playerA, pestermite, 4, 2);
50      }

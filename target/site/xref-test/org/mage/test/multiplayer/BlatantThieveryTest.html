21      @Override
22      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
23          Game game = new FreeForAll(MultiplayerAttackOption.MULTIPLE, RangeOfInfluence.ALL, MulliganType.GAME_DEFAULT.getMulligan(0), 20);
24          // Player order: A -> D -> C -> B
25          playerA = createPlayer(game, "PlayerA");
26          playerB = createPlayer(game, "PlayerB");
27          playerC = createPlayer(game, "PlayerC");
28          playerD = createPlayer(game, "PlayerD");
29          return game;
30      }
31  
32      @Test
33      public void NormalTest() {
34          // For each opponent, gain control of target permanent that player controls.
35          addCard(Zone.HAND, playerA, "Blatant Thievery"); // Sorcery {4}{U}{U}{U}
36          addCard(Zone.BATTLEFIELD, playerA, "Island", 7);
37  
38          // Player order: A -> D -> C -> B
39          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2);
40          addCard(Zone.BATTLEFIELD, playerC, "Walking Corpse", 2);
41          addCard(Zone.BATTLEFIELD, playerD, "Pillarfield Ox", 2);
42  
43          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Blatant Thievery");
44          addTarget(playerA, "Silvercoat Lion");
45          addTarget(playerA, "Walking Corpse");
46          addTarget(playerA, "Pillarfield Ox");
47  
48          setStrictChooseMode(true);
49          setStopAt(1, PhaseStep.BEGIN_COMBAT);
50          execute();
51  
52          assertGraveyardCount(playerA, "Blatant Thievery", 1);
53          assertPermanentCount(playerA, "Silvercoat Lion", 1);
54          assertPermanentCount(playerA, "Walking Corpse", 1);
55          assertPermanentCount(playerA, "Pillarfield Ox", 1);
56          assertPermanentCount(playerB, "Silvercoat Lion", 1);
57          assertPermanentCount(playerC, "Walking Corpse", 1);
58          assertPermanentCount(playerD, "Pillarfield Ox", 1);
59      }

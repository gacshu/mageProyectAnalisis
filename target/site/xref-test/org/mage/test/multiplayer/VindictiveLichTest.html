19  public class VindictiveLichTest extends CardTestMultiPlayerBase {
20  
21      @Override
22      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
23          Game game = new FreeForAll(MultiplayerAttackOption.MULTIPLE, RangeOfInfluence.ALL, MulliganType.GAME_DEFAULT.getMulligan(0), 40);
24          // Player order: A -> D -> C -> B
25          playerA = createPlayer(game, "PlayerA");
26          playerB = createPlayer(game, "PlayerB");
27          playerC = createPlayer(game, "PlayerC");
28          playerD = createPlayer(game, "PlayerD");
29          return game;
30      }
31  
32      /**
33       * Tests multiplayer effects Player order: A -> D -> C -> B
34       */
35      @Test
36      public void CallerOfThePackTest() {
37  
38          // When Vindictive Lich dies, choose one or more. Each mode must target a different player.
39          // *Target opponent sacrifices a creature.
40          // *Target opponent discards two cards.
41          // *Target opponent loses 5 life.
42          addCard(Zone.BATTLEFIELD, playerA, "Vindictive Lich"); // Creature {3}{B} 4/1
43  
44          // Sacrifice a creature: Put a +1/+1 counter on Bloodflow Connoisseur.
45          addCard(Zone.BATTLEFIELD, playerA, "Bloodflow Connoisseur"); // Creature {2}{B} 1/1
46  
47          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2);
48          addCard(Zone.HAND, playerC, "Lightning Bolt", 2);
49  
50          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice");
51          setChoice(playerA, "Vindictive Lich");
52  
53          setModeChoice(playerA, "1");
54          addTarget(playerA, playerB);
55          setModeChoice(playerA, "2");
56          addTarget(playerA, playerC);
57          setModeChoice(playerA, "3");
58          // Player D is auto-chosen since each mode needs a different opponent
59  
60          setStopAt(1, PhaseStep.BEGIN_COMBAT);
61          execute();
62  
63          assertGraveyardCount(playerA, "Vindictive Lich", 1);
64          assertPowerToughness(playerA, "Bloodflow Connoisseur", 2, 2);
65  
66          assertPermanentCount(playerB, "Silvercoat Lion", 1);
67          assertHandCount(playerC, 0);
68          assertLife(playerD, 35);
69  
70      }

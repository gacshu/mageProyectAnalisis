20  public class PlayerDiedStackTargetHandlingTest extends CardTestMultiPlayerBase {

22      @Override
23      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
24          // Start Life = 2
25          Game game = new FreeForAll(MultiplayerAttackOption.MULTIPLE, RangeOfInfluence.ONE, MulliganType.GAME_DEFAULT.getMulligan(0), 3);
26          // Player order: A -> D -> C -> B
27          playerA = createPlayer(game, "PlayerA");
28          playerB = createPlayer(game, "PlayerB");
29          playerC = createPlayer(game, "PlayerC");
30          playerD = createPlayer(game, "PlayerD");
31          return game;
32      }
33  
34      /**
35       * If a spell or ability on the stack is targeting a player (or an object
36       * controlled by that player), and that player leaves the game, then the
37       * target becomes illegal and the spell or ability should be countered upon
38       * resolution (assuming that player or object is the only target of the
39       * spell). However, Xmage regularly continues resolving spells and abilities
40       * targeting players after that player concedes, which is a problem in
41       * multiplayer.
42       */
43      @Test
44      public void TestDeadPlayerIsNoLongerValidTarget() {
45          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
46          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
47          // Lightning Helix deals 3 damage to any target and you gain 3 life.
48          addCard(Zone.HAND, playerA, "Lightning Helix", 2); // Instant {R}{W}
49  
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Helix", playerD);
51          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Helix", playerD);
52  
53          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
54          execute();
55  
56          assertGraveyardCount(playerA, "Lightning Helix", 2);
57          Assert.assertTrue("Active player has to be player C", currentGame.getActivePlayerId().equals(playerC.getId()));
58  
59          assertLife(playerA, 6);
60  
61      }
62  
63      /**
64       * Player A casts Tendrils of Agony with a storm count of 10 targeting
65       * Player B, who is at 10 life. After the first five copies of Tendrils
66       * resolve, player A will have gained 10 life and Player B will have lost 10
67       * life, so player B should die as a state-based action before the remaining
68       * six copies resolve. In similar circumstances, Xmage has continued
69       * resolving the additional six copies, putting player B at -12 life and
70       * letting player A gain an additional 12 life inappropriately
71       */
72      @Test
73      public void TestDeadPlayerIsNoLongerValidTarget2() {
74          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
75          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
76          addCard(Zone.HAND, playerA, "Silvercoat Lion", 2);
77          // Target player loses 2 life and you gain 2 life.
78          // Storm (When you cast this spell, copy it for each spell cast before it this turn. You may choose new targets for the copies.)
79          addCard(Zone.HAND, playerA, "Tendrils of Agony", 1); // Sorcery {2}{B}{B}
80  
81          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Silvercoat Lion", true);
82          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Silvercoat Lion", true);
83          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tendrils of Agony", playerD);
84          addTarget(playerA, playerD);
85          addTarget(playerA, playerD);
86  
87          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
88          execute();
89  
90          assertPermanentCount(playerA, "Silvercoat Lion", 2);
91          assertGraveyardCount(playerA, "Tendrils of Agony", 1);
92          Assert.assertTrue("Active player has to be player C", currentGame.getActivePlayerId().equals(playerC.getId()));
93  
94          assertLife(playerA, 7);
95      }
96  }

123     /**
124      * Xmage throws an error involving an emblem unable to find the initial
125      * source if it has a proc. To reproduce, a Planeswalker was taken from an
126      * original player's control, such as using Scrambleverse to shuffle Jace,
127      * Unraveler of Secrets, to a second player and then the second player uses
128      * Jace's ability to create an emblem ("Whenever an opponent casts their
129      * first spell each turn, counter that spell."). Then the original player
130      * concedes the game and removes the Planeswalker. Once it becomes an
131      * opponent of the original player's turn and that opponent plays a spell,
132      * Xmage throws an error and rollsback the turn.
133      * <p>
134      * I don't have the actual error report on my due to negligence, but what I
135      * can recollect is that the error message was along the lines of "The
136      * emblem cannot find the original source. This turn will be rolled back".
137      * This error message will always appear when an opponent tries to play a
138      * spell. Player order: A -> D -> C -> B
139      */
140     @Test
141     public void TestOtherPlayerPlaneswalkerCreatedEmblem() {
142         // +1: Scry 1, then draw a card.
143         // -2: Return target creature to its owner's hand.
144         // -8: You get an emblem with "Whenever an opponent casts their first spell each turn, counter that spell."
145         addCard(Zone.BATTLEFIELD, playerB, "Jace, Unraveler of Secrets");
146         addCounters(1, PhaseStep.DRAW, playerB, "Jace, Unraveler of Secrets", CounterType.LOYALTY, 8);
147 
148         addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
149         // Enchant permanent (Target a permanent as you cast this. This card enters the battlefield attached to that permanent.)
150         // You control enchanted permanent.
151         addCard(Zone.HAND, playerA, "Confiscate"); // Enchantment Aura
152 
153         addCard(Zone.BATTLEFIELD, playerC, "Plains", 2);
154         addCard(Zone.HAND, playerC, "Silvercoat Lion");
155 
156         addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
157         addCard(Zone.HAND, playerD, "Silvercoat Lion");
158 
159         addCard(Zone.BATTLEFIELD, playerC, "Silvercoat Lion");
160 
161         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Confiscate", "Jace, Unraveler of Secrets");
162         activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "-8: You get an emblem with");
163 
164         attack(3, playerC, "Silvercoat Lion", playerB);
165         castSpell(3, PhaseStep.POSTCOMBAT_MAIN, playerC, "Silvercoat Lion");
166 
167         castSpell(5, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
168 
169         setStopAt(5, PhaseStep.END_TURN);
170         execute();
171 
172         assertLife(playerB, 0);
173         assertPermanentCount(playerB, 0);
174         assertGraveyardCount(playerA, "Confiscate", 1);
175         assertPermanentCount(playerA, "Jace, Unraveler of Secrets", 0); // Removed from game because player C left the game
176         assertEmblemCount(playerA, 1);
177         assertPermanentCount(playerC, "Silvercoat Lion", 2); // Emblem does not work yet on player C, because range 1
178         assertGraveyardCount(playerD, "Silvercoat Lion", 1); // Emblem should counter the spell
179     }

181     /**
182      * Situation: I attacked an opponent with some creatures with True
183      * Conviction in play. There were multiple "deals combat damage to a
184      * player"-triggers (Edric, Spymaster of Trest, Daxos of Meletis et al),
185      * then the opponent lost the game during the first strike combat
186      * damage-step . In the second combat damage step the triggers went on the
187      * stack again, although there was no player being dealt damage (multiplayer
188      * game, so the game wasn't over yet). I don't think these abilities should
189      * trigger again here.
190      */
191     @Test
192     public void TestPlayerDiesDuringFirstStrikeDamageStep() {
193         // Creatures you control have double strike and lifelink.
194         addCard(Zone.BATTLEFIELD, playerD, "True Conviction");
195         // Whenever a creature deals combat damage to one of your opponents, its controller may draw a card.
196         addCard(Zone.BATTLEFIELD, playerD, "Edric, Spymaster of Trest");
197         addCard(Zone.BATTLEFIELD, playerD, "Dross Crocodile", 8); // Creature 5/1
198 
199         attack(2, playerD, "Dross Crocodile", playerC);
200 
201         setStopAt(3, PhaseStep.END_TURN);
202         execute();
203 
204         assertLife(playerC, -3);
205         assertLife(playerD, 7);
206 
207         assertHandCount(playerD, 2); // 1 (normal draw) + 1 from True Convition
208         assertPermanentCount(playerC, 0);
209 
210     }

213      * I've encountered a case today where someone conceded on their turn. The
214      * remaining phases were went through as normal, but my Luminarch Ascension
215      * did not trigger during the end step.
216      */
217     // Player order: A -> D -> C -> B
218     @Test
219     public void TestTurnEndTrigger() {
220         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
221         // At the beginning of each opponent's end step, if you didn't lose life this turn, you may put a quest counter on Luminarch Ascension.
222         // {1}{W}: Create a 4/4 white Angel creature token with flying. Activate this ability only if Luminarch Ascension has four or more quest counters on it..
223         addCard(Zone.HAND, playerA, "Luminarch Ascension"); // Enchantment {1}{W}
224 
225         addCard(Zone.HAND, playerC, "Lightning Bolt");
226         addCard(Zone.BATTLEFIELD, playerC, "Mountain", 1);
227 
228         addCard(Zone.HAND, playerD, "Silvercoat Lion");
229         addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
230 
231         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Luminarch Ascension");
232 
233         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
234         castSpell(2, PhaseStep.BEGIN_COMBAT, playerC, "Lightning Bolt", playerD);
235 
236         setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
237         execute();
238 
239         assertPermanentCount(playerA, "Luminarch Ascension", 1);
240         assertGraveyardCount(playerC, "Lightning Bolt", 1);
241 
242         assertLife(playerD, -1);
243         Assert.assertFalse("Player D is no longer in the game", playerD.isInGame());
244 
245         assertCounterCount(playerA, "Luminarch Ascension", CounterType.QUEST, 1); // 1 from turn 2
246     }

275     /**
276      * Pithing Needle keeps the named card's abilities disabled even after the
277      * player controlling the Needle loses the game.
278      * <p>
279      * I saw it happen during a Commander game. A player cast Pithing Needle
280      * targeting my Proteus Staff. After I killed him, I still couldn't activate
281      * the Staff.
282      */
283     @Test
284     public void TestPithingNeedle() {
285         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
286         // As Pithing Needle enters the battlefield, name a card.
287         // Activated abilities of sources with the chosen name can't be activated unless they're mana abilities.
288         addCard(Zone.HAND, playerA, "Pithing Needle"); // Artifact {1}
289         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
290         addCard(Zone.LIBRARY, playerA, "Pillarfield Ox", 1);
291 
292         addCard(Zone.BATTLEFIELD, playerB, "Island", 3);
293         addCard(Zone.BATTLEFIELD, playerB, "Proteus Staff", 1);
294 
295         addCard(Zone.BATTLEFIELD, playerD, "Island", 3);
296         // {2}{U}, {T}: Put target creature on the bottom of its owner's library. That creature's controller reveals cards from the
297         // top of their library until they reveal a creature card. The player puts that card onto the battlefield and the
298         // rest on the bottom of their library in any order. Activate this ability only any time you could cast a sorcery.
299         addCard(Zone.BATTLEFIELD, playerD, "Proteus Staff", 1);
300 
301         addCard(Zone.BATTLEFIELD, playerD, "Eager Cadet", 1);
302         addCard(Zone.LIBRARY, playerD, "Storm Crow", 2);
303 
304         addCard(Zone.BATTLEFIELD, playerC, "Island", 3);
305         addCard(Zone.BATTLEFIELD, playerC, "Proteus Staff", 1);
306         addCard(Zone.BATTLEFIELD, playerC, "Wall of Air", 1);
307         addCard(Zone.LIBRARY, playerC, "Wind Drake", 2);
308 
309         skipInitShuffling();
310 
311         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Pithing Needle");
312         setChoice(playerA, "Proteus Staff");
313 
314         activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerC, "{2}{U}", "Eager Cadet"); // allowed because Needle out of range
315 
316         // Concede the game
317         concede(3, PhaseStep.POSTCOMBAT_MAIN, playerA);
318 
319         activateAbility(4, PhaseStep.PRECOMBAT_MAIN, playerB, "{2}{U}", "Wall of Air"); // allowed because Needle lost game
320 
321         setStopAt(4, PhaseStep.POSTCOMBAT_MAIN);
322         execute();
323 
324         assertLife(playerA, 2);
325         Assert.assertFalse("Player A is no longer in the game", playerA.isInGame());
326 
327         assertPermanentCount(playerA, 0);
328 
329         assertPermanentCount(playerD, "Eager Cadet", 0);
330         assertPermanentCount(playerD, "Storm Crow", 1);
331         assertPermanentCount(playerC, "Wall of Air", 0);
332         assertPermanentCount(playerC, "Wind Drake", 1);
333     }

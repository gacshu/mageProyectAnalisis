121     /**
122      * Xmage throws an error involving an emblem unable to find the initial
123      * source if it has a proc. To reproduce, a Planeswalker was taken from an
124      * original player's control, such as using Scrambleverse to shuffle Jace,
125      * Unraveler of Secrets, to a second player and then the second player uses
126      * Jace's ability to create an emblem ("Whenever an opponent casts their
127      * first spell each turn, counter that spell."). Then the original player
128      * concedes the game and removes the Planeswalker. Once it becomes an
129      * opponent of the original player's turn and that opponent plays a spell,
130      * Xmage throws an error and rollsback the turn.
131      * <p>
132      * I don't have the actual error report on my due to negligence, but what I
133      * can recollect is that the error message was along the lines of "The
134      * emblem cannot find the original source. This turn will be rolled back".
135      * This error message will always appear when an opponent tries to play a
136      * spell. Player order: A -> D -> C -> B
137      */
138     @Test
139     public void TestOtherPlayerPlaneswalkerCreatedEmblem() {
140         // +1: Scry 1, then draw a card.
141         // -2: Return target creature to its owner's hand.
142         // -8: You get an emblem with "Whenever an opponent casts their first spell each turn, counter that spell."
143         addCard(Zone.BATTLEFIELD, playerC, "Jace, Unraveler of Secrets");
144         addCounters(1, PhaseStep.DRAW, playerC, "Jace, Unraveler of Secrets", CounterType.LOYALTY, 8);
145 
146         addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
147         // Enchant permanent (Target a permanent as you cast this. This card enters the battlefield attached to that permanent.)
148         // You control enchanted permanent.
149         addCard(Zone.HAND, playerA, "Confiscate"); // Enchantment Aura
150 
151         addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
152         addCard(Zone.HAND, playerB, "Silvercoat Lion");
153 
154         addCard(Zone.BATTLEFIELD, playerD, "Silvercoat Lion");
155 
156         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Confiscate", "Jace, Unraveler of Secrets");
157         activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "-8: You get an emblem with");
158 
159         attack(2, playerD, "Silvercoat Lion", playerC);
160 
161         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerB, "Silvercoat Lion");
162 
163         setStopAt(3, PhaseStep.BEGIN_COMBAT);
164         execute();
165 
166         assertLife(playerC, 0);
167         assertPermanentCount(playerC, 0);
168         assertGraveyardCount(playerA, "Confiscate", 1);
169         assertPermanentCount(playerA, "Jace, Unraveler of Secrets", 0); // Removed from game because player C left the game
170         assertEmblemCount(playerA, 1);
171         assertGraveyardCount(playerB, "Silvercoat Lion", 1);
172     }

174     /**
175      * Situation: I attacked an opponent with some creatures with True
176      * Conviction in play. There were multiple "deals combat damage to a
177      * player"-triggers (Edric, Spymaster of Trest, Daxos of Meletis et al),
178      * then the opponent lost the game during the first strike combat
179      * damage-step . In the second combat damage step the triggers went on the
180      * stack again, although there was no player being dealt damage (multiplayer
181      * game, so the game wasn't over yet). I don't think these abilities should
182      * trigger again here.
183      */
184     @Test
185     public void TestPlayerDiesDuringFirstStrikeDamageStep() {
186         // Creatures you control have double strike and lifelink.
187         addCard(Zone.BATTLEFIELD, playerD, "True Conviction");
188         // Whenever a creature deals combat damage to one of your opponents, its controller may draw a card.
189         addCard(Zone.BATTLEFIELD, playerD, "Edric, Spymaster of Trest");
190         addCard(Zone.BATTLEFIELD, playerD, "Dross Crocodile", 8); // Creature 5/1
191 
192         attack(2, playerD, "Dross Crocodile", playerC);
193 
194         setStopAt(3, PhaseStep.END_TURN);
195         execute();
196 
197         assertLife(playerC, -3);
198         assertLife(playerD, 7);
199 
200         assertHandCount(playerD, 2); // 1 (normal draw) + 1 from True Convition
201         assertPermanentCount(playerC, 0);
202 
203     }

212     public void TestTurnEndTrigger() {
213         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
214         // At the beginning of each opponent's end step, if you didn't lose life this turn, you may put a quest counter on Luminarch Ascension.
215         // {1}{W}: Create a 4/4 white Angel creature token with flying. Activate this ability only if Luminarch Ascension has four or more quest counters on it..
216         addCard(Zone.HAND, playerA, "Luminarch Ascension"); // Enchantment {1}{W}
217 
218         addCard(Zone.HAND, playerC, "Lightning Bolt");
219         addCard(Zone.BATTLEFIELD, playerC, "Mountain", 1);
220 
221         addCard(Zone.HAND, playerD, "Silvercoat Lion");
222         addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
223 
224         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Luminarch Ascension");
225 
226         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
227         castSpell(2, PhaseStep.BEGIN_COMBAT, playerC, "Lightning Bolt", playerD);
228 
229         setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
230         execute();
231 
232         assertPermanentCount(playerA, "Luminarch Ascension", 1);
233         assertGraveyardCount(playerC, "Lightning Bolt", 1);
234 
235         assertLife(playerD, -1);

237 
238         assertCounterCount(playerA, "Luminarch Ascension", CounterType.QUEST, 1); // 1 from turn 2

243      * "When playing in a multiplayer match against humans, the aura curse
244      * "Curse of Vengeance" is supposed to award cards and life to its caster
245      * when the victim of the spell loses the game. It seems to erroneously
246      * award those cards to the victim of the curse, who, by that point, is
247      * already dead, making the spell almost totally useless."
248      */
249     @Test
250     public void TestCurseOfVengeance() {
251         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
252         // Whenever enchanted player casts a spell, put a spite counter on Curse of Vengeance.
253         // When enchanted player loses the game, you gain X life and draw X cards, where X is the number of spite counters on Curse of Vengeance.
254         addCard(Zone.HAND, playerA, "Curse of Vengeance"); // Enchantment {B}
255 
256         addCard(Zone.HAND, playerC, "Lightning Bolt");
257         addCard(Zone.BATTLEFIELD, playerC, "Mountain", 1);
258 
259         addCard(Zone.HAND, playerD, "Silvercoat Lion", 2);
260         addCard(Zone.BATTLEFIELD, playerD, "Plains", 4);
261 
262         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Curse of Vengeance", playerD);
263         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
264 
265         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion", true);
266         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
267 
268         castSpell(2, PhaseStep.BEGIN_COMBAT, playerC, "Lightning Bolt", playerD);
269 
270         setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
271         execute();
272 
273         assertGraveyardCount(playerC, "Lightning Bolt", 1);
274         assertGraveyardCount(playerA, "Curse of Vengeance", 1);
275 
276         assertLife(playerD, -1);
277         Assert.assertFalse("Player D is no longer in the game", playerD.isInGame());
278 
279         assertHandCount(playerA, 3);
280         assertLife(playerA, 4);
281

420     /**
421      * https://github.com/magefree/mage/issues/6997
422        Some continuous effects should stay in play even after the player that set them leaves the game. 
423        Example: 
424        * Player A: Casts Vorinclex, Voice of Hunger 
425        * Player D: Taps all lands and do stuff (lands shouldn't untap during his next untap step)     
426        * Player C: Kills Player A Player D: Lands untapped normally, though they shouldn't
427        * 
428        * This happened playing commander against 3 AIs. One of the AIs played Vorinclex, I tapped all my lands during my turn to do stuff. 
429        * Next AI killed the one that had Vorinclex. When the game got to my turn, my lands untapped normally.
430      */
431     @Test
432     public void TestContinuousEffectStaysAfterCreatingPlayerLeft() {
433         // Player order: A -> D -> C -> B
434         setStrictChooseMode(true);
435 
436         addCard(Zone.BATTLEFIELD, playerA, "Forest", 8);
437         // Trample
438         // Whenever you tap a land for mana, add one mana of any type that land produced.
439         // Whenever an opponent taps a land for mana, that land doesn't untap during its controller's next untap step.
440         addCard(Zone.HAND, playerA, "Vorinclex, Voice of Hunger"); // Creature 7/6  {6}{G}{G}
441 
442         addCard(Zone.HAND, playerD, "Silvercoat Lion");
443         addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
444 
445         addCard(Zone.HAND, playerC, "Lightning Bolt", 1);
446         addCard(Zone.BATTLEFIELD, playerC, "Mountain", 1);
447 
448         addCard(Zone.HAND, playerB, "Silvercoat Lion", 1);
449         addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
450 
451         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Vorinclex, Voice of Hunger");
452 
453         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
454 
455         setChoice(playerA, "Whenever an opponent taps a land for mana");
456         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerC, "Lightning Bolt", playerA);
457 
458         setStopAt(5, PhaseStep.BEGIN_COMBAT);
459         
460         execute();
461 
462         assertPermanentCount(playerD, "Silvercoat Lion", 1);
463         
464         assertGraveyardCount(playerC, "Lightning Bolt", 1);
465         
466         Assert.assertFalse("Player A is no longer in the game", playerA.isInGame());
467 
468         Assert.assertTrue("Player D is the active player",currentGame.getActivePlayerId().equals(playerD.getId()));
469         
470         assertTappedCount("Plains", true, 2); // Do not untap because of Vorinclex do not untap effect
471

21  public class PrivilegedPositionTest extends CardTestMultiPlayerBase {
22  
23      @Override
24      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
25          Game game = new FreeForAll(MultiplayerAttackOption.MULTIPLE, RangeOfInfluence.ALL, MulliganType.GAME_DEFAULT.getMulligan(0), 40);
26          // Player order: A -> D -> C -> B
27          playerA = createPlayer(game, "PlayerA");
28          playerB = createPlayer(game, "PlayerB");
29          playerC = createPlayer(game, "PlayerC");
30          playerD = createPlayer(game, "PlayerD");
31          return game;
32      }
33  
34      /*
35       * Reported bug: see issue #3328
36       * Players unable to attack Planeswalker with Privileged Position on battlefield.
37       */
38      @Test
39      public void testAttackPlaneswalkerWithHexproofPrivilegedPosition() {
40  
41          /*
42           Privileged Position {2}{G/W}{G/W}{G/W}
43          Enchantment
44          Other permanents you control have hexproof.
45           */
46          String pPosition = "Privileged Position";
47          String sorin = "Sorin, Solemn Visitor"; // planeswalker {2}{W}{B} 4 loyalty
48          String memnite = "Memnite"; // {0} 1/1
49  
50          addCard(Zone.BATTLEFIELD, playerD, pPosition);
51          addCard(Zone.BATTLEFIELD, playerD, sorin);
52          addCard(Zone.BATTLEFIELD, playerA, memnite);
53          addCard(Zone.BATTLEFIELD, playerB, memnite);
54          addCard(Zone.BATTLEFIELD, playerC, memnite);
55  
56          // Player order: A -> D -> C -> B
57          attack(1, playerA, memnite, sorin);
58          attack(3, playerC, memnite, sorin);
59          attack(4, playerB, memnite, sorin);
60  
61          setStopAt(4, PhaseStep.END_COMBAT);
62          execute();
63  
64          assertPermanentCount(playerD, pPosition, 1);
65          assertPermanentCount(playerD, sorin, 1);
66          assertTappedCount(memnite, true, 3);
67          assertLife(playerD, 40);
68          assertCounterCount(sorin, CounterType.LOYALTY, 1);
69          assertAbility(playerD, sorin, HexproofAbility.getInstance(), true);
70      }

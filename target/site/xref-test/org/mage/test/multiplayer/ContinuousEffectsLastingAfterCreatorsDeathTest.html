15      // Player order: A -> D -> C -> B
16      @Test
17      public void testDontUntapNormal() {
18          // Trample
19          // Whenever you tap a land for mana, add one mana of any type that land produced.
20          // Whenever an opponent taps a land for mana, that land doesn't untap during its controller's next untap step.
21          addCard(Zone.BATTLEFIELD, playerA, "Vorinclex, Voice of Hunger"); // Creature {6}{G}{G} 7/6
22  
23          addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
24          addCard(Zone.HAND, playerD, "Silvercoat Lion");
25  
26          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
27  
28          setStopAt(6, PhaseStep.PRECOMBAT_MAIN);
29          execute();
30  
31          assertPermanentCount(playerA, "Vorinclex, Voice of Hunger", 1);
32          assertPermanentCount(playerD, "Silvercoat Lion", 1);
33  
34          Assert.assertTrue("Active player is player D", currentGame.getActivePlayerId().equals(playerD.getId()));
35          assertTappedCount("Plains", true, 2);
36      }

38      @Test
39      public void testDontUntap() {
40          /**
41           * https://github.com/magefree/mage/issues/6997 Some continuous effects
42           * should stay in play even after the player that set them leaves the
43           * game. Example:
44           *
45           * Player A: Casts Vorinclex, Voice of Hunger Player B: Taps all lands
46           * and do stuff (lands shouldn't untap during his next untap step)
47           * Player C: Kills Player A Player B: Lands untapped normally, though
48           * they shouldn't
49           *
50           * This happened playing commander against 3 AIs. One of the AIs played
51           * Vorinclex, I tapped all my lands during my turn to do stuff. Next AI
52           * killed the one that had Vorinclex. When the game got to my turn, my
53           * lands untapped normally.
54           */
55  
56          // Trample
57          // Whenever you tap a land for mana, add one mana of any type that land produced.
58          // Whenever an opponent taps a land for mana, that land doesn't untap during its controller's next untap step.
59          addCard(Zone.BATTLEFIELD, playerA, "Vorinclex, Voice of Hunger"); // Creature {6}{G}{G} 7/6
60  
61          addCard(Zone.BATTLEFIELD, playerD, "Plains", 2);
62          addCard(Zone.HAND, playerD, "Silvercoat Lion");
63  
64          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerD, "Silvercoat Lion");
65  
66          concede(2, PhaseStep.POSTCOMBAT_MAIN, playerA);
67  
68          setStopAt(5, PhaseStep.PRECOMBAT_MAIN);
69          execute();
70  
71          assertPermanentCount(playerA, "Vorinclex, Voice of Hunger", 0);
72          assertPermanentCount(playerD, "Silvercoat Lion", 1);
73  
74          Assert.assertTrue("Active player is player D", currentGame.getActivePlayerId().equals(playerD.getId()));
75          assertTappedCount("Plains", true, 2);
76      }

16      /**
17       * Check that an equipment is not switched again an again between creatures
18       */
19      @Test
20      public void testEquipOnlyOnce() {
21          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
22  
23          // Equipped creature gets +1/+1.
24          // Equip {1}({1}: Attach to target creature you control. Equip only as a sorcery.)
25          addCard(Zone.BATTLEFIELD, playerA, "Fireshrieker", 1);
26          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 2);
27  
28          setStopAt(1, PhaseStep.BEGIN_COMBAT);
29          execute();
30  
31          int tappedLands = 0;
32          for (Permanent permanent : currentGame.getBattlefield().getAllActivePermanents(new FilterLandPermanent(), playerA.getId(), currentGame)) {
33              if (permanent.isTapped()) {
34                  tappedLands++;
35              }
36          }
37          Assert.assertEquals("AI should only used Equipment once", 2, tappedLands);
38      }

40      /**
41       * If the AI on a local server gets control of a Basalt Monolith it will
42       * infinite loop taping for three mana and then using the mana to untap lol.
43       * Seeing the computer durdle troll is quite the hillarious thing
44       */
45      @Test
46      public void testBasaltMonolith() {
47          addCard(Zone.HAND, playerA, "Phyrexian Vault", 1);
48          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
49          // Basalt Monolith doesn't untap during your untap step.
50          // {T}: Add {C}{C}{C}.
51          // {3}: Untap Basalt Monolith.
52          addCard(Zone.BATTLEFIELD, playerA, "Basalt Monolith", 1, true);
53  
54          setStopAt(5, PhaseStep.END_TURN);
55          execute();
56  
57          // {2}, {T}, Sacrifice a creature: Draw a card.
58          assertPermanentCount(playerA, "Phyrexian Vault", 1);
59          assertTapped("Basalt Monolith", true);
60          assertTappedCount("Plains", false, 3);
61      }

64       * AI gets stuck with two Kiora's Followers #1167
65       */
66      @Test
67      public void testKiorasFollower() {
68          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2, true);
69          // {T}: Untap another target permanent.
70          addCard(Zone.BATTLEFIELD, playerA, "Kiora's Follower", 1, true);
71          addCard(Zone.BATTLEFIELD, playerA, "Kiora's Follower", 1);
72  
73          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2);
74  
75          attack(2, playerB, "Silvercoat Lion");
76          attack(2, playerB, "Silvercoat Lion");
77          blockSkip(2, playerA);
78  
79          setStopAt(2, PhaseStep.END_TURN);
80          execute();
81  
82          assertPermanentCount(playerA, "Kiora's Follower", 2);
83          assertPermanentCount(playerB, "Silvercoat Lion", 2);
84          assertLife(playerA, 16);
85          assertTapped("Kiora's Follower", false);

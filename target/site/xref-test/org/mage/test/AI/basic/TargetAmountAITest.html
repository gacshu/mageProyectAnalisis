13      @Test
14      public void test_AI_ChooseTargets() {
15          // Distribute four +1/+1 counters among any number of target creatures.
16          addCard(Zone.HAND, playerA, "Blessings of Nature", 1); // {4}{G}
17          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
18          //
19          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 4); // 2/2
20          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 4); // 2/2
21  
22          // ai must choose by special dialog, not full simulation
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Blessings of Nature");
24  
25          setStopAt(1, PhaseStep.END_TURN);
26          //setStrictChooseMode(true); // ai must choose
27          execute();
28  
29          assertPowerToughness(playerA, "Balduvian Bears", 2 + 4, 2 + 4); // boost one creature (it's just a choose code, so can be different from simulation results)
30      }

32      @Test
33      public void test_AI_SimulateTargets() {
34          // Distribute four +1/+1 counters among any number of target creatures.
35          addCard(Zone.HAND, playerA, "Blessings of Nature", 1); // {4}{G}
36          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
37          //
38          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 4); // 2/2
39          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 4); // 2/2
40  
41          // AI must put creatures on own permanents (all in one creature to boost it)
42          aiPlayPriority(1, PhaseStep.PRECOMBAT_MAIN, playerA);
43  
44          setStopAt(1, PhaseStep.END_TURN);
45          setStrictChooseMode(true);
46          execute();
47  
48          assertPowerToughness(playerA, "Balduvian Bears", 2 + 1, 2 + 1); // boost each possible creatures
49          assertPowerToughness(playerB, "Balduvian Bears", 2, 2); // no boost for enemy
50      }

13      @Test
14      public void test_OpponentMakeChooseInsteadPlayer_User() {
15          // Gain control of target creature of an opponent’s choice they control.
16          addCard(Zone.HAND, playerA, "Evangelize", 1); // {4}{W}
17          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
18          //
19          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 1); // 2/2
20          addCard(Zone.BATTLEFIELD, playerB, "Spectral Bears", 1); // 3/3
21  
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Evangelize"); // do not call direct target setup
23          addTarget(playerA, playerB); // choose target opponent
24          setChoice(playerA, false); // no buyback
25          //
26          addTarget(playerB, "Balduvian Bears"); // give small bear to A
27  
28          setStrictChooseMode(true);
29          setStopAt(1, PhaseStep.END_TURN);
30          execute();
31  
32          assertGraveyardCount(playerA, "Evangelize", 1);
33          assertPermanentCount(playerA, "Balduvian Bears", 1);
34      }

18      // Gideon, Battle-Forged
19      // +2: Up to one target creature an opponent controls attacks Gideon, Battle-Forged during its controller's next turn if able.
20      // +1: Until your next turn, target creature gains indestructible. Untap that creature.
21      // 0: Until end of turn, Gideon, Battle-Forged becomes a 4/4 Human Soldier creature with indestructible that's still a planeswalker. Prevent all damage that would be dealt to him this turn.
22      @Test
23      public void testGideonBattleForged() {
24          // At end of combat, if Kytheon, Hero of Akros and at least two other creatures attacked this combat, exile Kytheon,
25          // then return him to the battlefield transformed under his owner's control.
26          // {2}{W}: Kytheon gains indestructible until end of turn.
27          addCard(Zone.BATTLEFIELD, playerB, "Kytheon, Hero of Akros");
28          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
29          addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox");
30  
31          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
32          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
33  
34          attack(2, playerB, "Kytheon, Hero of Akros");
35          attack(2, playerB, "Silvercoat Lion");
36          attack(2, playerB, "Pillarfield Ox");
37  
38          activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "+1: Until your next turn, target creature gains indestructible. Untap that creature.", "Silvercoat Lion");
39  
40          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Silvercoat Lion");
41  
42          setStopAt(4, PhaseStep.PRECOMBAT_MAIN);
43          execute();
44  
45          assertGraveyardCount(playerA, "Lightning Bolt", 1);
46          assertPermanentCount(playerB, "Gideon, Battle-Forged", 1);
47          assertCounterCount("Gideon, Battle-Forged", CounterType.LOYALTY, 4);
48          assertLife(playerA, 14);
49  
50          assertPermanentCount(playerB, "Silvercoat Lion", 1);
51          assertAbility(playerB, "Silvercoat Lion", IndestructibleAbility.getInstance(), false);
52      }

55       * Reported bug: When Gideon, Champion of Justice uses his +0 ability to become a creature,
56       * he is immediately sent to the grave instead.
57       */
58      @Test
59      public void testGideonChampionOfJusticeSecondAbility() {
60          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
61          /*
62          Gideon, Champion of Justice {2}{W}{W} - 4 Loyalty
63          +1: Put a loyalty counter on Gideon, Champion of Justice for each creature target opponent controls.
64  
65          0: Until end of turn, Gideon, Champion of Justice becomes a Human Soldier creature with power and toughness
66          each equal to the number of loyalty counters on him and gains indestructible. He's still a planeswalker.
67          Prevent all damage that would be dealt to him this turn.
68          LoyaltyAbility ability1 = new LoyaltyAbility(
69  
70          -15: Exile all other permanents.
71           */
72          addCard(Zone.HAND, playerA, "Gideon, Champion of Justice", 1);
73  
74          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2);
75  
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gideon, Champion of Justice", true);
77          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Until end of turn");
78  
79          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: Put a loyalty counter on", playerB);
80  
81          activateAbility(5, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Until end of turn");
82  
83          setStopAt(5, PhaseStep.BEGIN_COMBAT);
84          execute();
85  
86          assertGraveyardCount(playerA, "Gideon, Champion of Justice", 0);
87          assertPermanentCount(playerA, "Gideon, Champion of Justice", 1);
88          assertCounterCount(playerA, "Gideon, Champion of Justice", CounterType.LOYALTY, 7);
89          assertPowerToughness(playerA, "Gideon, Champion of Justice", 7, 7);

91      }

93      /**
94       * When you use Gideon, Battle-Forged (flipped version of Kytheon from Magic
95       * Origins) and use his 0 ability to turn him into a creature and equip the
96       * Stitcher's Graft from EMN, he should have to be sacced at the end of
97       * turn.
98       */
99      @Test
100     public void testGideonBattleForgedSacrifice() {
101         addCard(Zone.BATTLEFIELD, playerB, "Mountain", 2);
102         // At end of combat, if Kytheon, Hero of Akros and at least two other creatures attacked this combat, exile Kytheon,
103         // then return him to the battlefield transformed under his owner's control.
104         // {2}{W}: Kytheon gains indestructible until end of turn.
105         // Gideon, Battle-Forged
106         // +2: Up to one target creature an opponent controls attacks Gideon, Battle-Forged during its controller's next turn if able.
107         // +1: Until your next turn, target creature gains indestructible. Untap that creature.
108         // +0: Until end of turn, Gideon, Battle-Forged becomes a 4/4 Human Soldier creature with indestructible that's still a planeswalker. Prevent all damage that would be dealt to him this turn.
109         addCard(Zone.BATTLEFIELD, playerB, "Kytheon, Hero of Akros");
110         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
111         addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox");
112         // Equipped creature gets +3/+3.
113         // Whenever equipped creature attacks, it doesn't untap during its controller's next untap step.
114         // Whenever Stitcher's Graft becomes unattached from a permanent, sacrifice that permanent.
115         // Equip {2}
116         addCard(Zone.BATTLEFIELD, playerB, "Stitcher's Graft", 1);
117 
118         // transform
119         attack(2, playerB, "Kytheon, Hero of Akros");
120         attack(2, playerB, "Silvercoat Lion");
121         attack(2, playerB, "Pillarfield Ox");
122         checkPermanentCount("after transform", 2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Gideon, Battle-Forged", 1);
123 
124         // become creature and equip
125         activateAbility(4, PhaseStep.PRECOMBAT_MAIN, playerB, "0: Until ");
126         waitStackResolved(4, PhaseStep.PRECOMBAT_MAIN);
127         activateAbility(4, PhaseStep.PRECOMBAT_MAIN, playerB, "Equip {2}", "Gideon, Battle-Forged");
128 
129         attack(4, playerB, "Gideon, Battle-Forged"); // 7 damage
130 
131         setStrictChooseMode(true);
132         setStopAt(5, PhaseStep.PRECOMBAT_MAIN);
133         execute();
134 
135         assertPermanentCount(playerB, "Silvercoat Lion", 1);
136         assertLife(playerA, 7);
137         Permanent equipment = getPermanent("Stitcher's Graft", playerB);
138         Assert.assertTrue("Stitcher's Graft may no longer be equipped", equipment.getAttachedTo() == null);
139         assertPermanentCount(playerB, "Gideon, Battle-Forged", 0);
140         assertGraveyardCount(playerB, "Kytheon, Hero of Akros", 1);
141     }

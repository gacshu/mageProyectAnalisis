2   package org.mage.test.cards.planeswalker;
3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import mage.counters.CounterType;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   *
12   * @author LevelX2
13   */
14  public class UginTest extends CardTestPlayerBase {
15  
16      @Test
17      public void testCard() {
18          // +2: Ugin, the Spirit Dragon deals 3 damage to any target.
19          // -X: Exile each permanent with converted mana cost X or less that's one or more colors.
20          // -10: You gain 7 life, draw 7 cards, then put up to seven permanent cards from your hand onto the battlefield.
21          addCard(Zone.BATTLEFIELD, playerA, "Ugin, the Spirit Dragon"); // starts with 7 Loyality counters
22          // Whenever a creature dies, you may put a quest counter on Quest for the Gravelord.
23          addCard(Zone.BATTLEFIELD, playerA, "Quest for the Gravelord");
24          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
25  
26          addCard(Zone.LIBRARY, playerB, "Forest", 2);
27          addCard(Zone.BATTLEFIELD, playerB, "Forest", 6);
28          // When Nissa, Vastwood Seer enters the battlefield, you may search your library for a basic Forest card, reveal it, put it into your hand, then shuffle your library.
29          // Whenever a land enters the battlefield under your control, if you control seven or more lands, exile Nissa, then return her to the battlefield transformed under her owner's control.
30          // +1: Reveal the top card of your library. If it's a land card, put it onto the battlefield. Otherwise, put it into your hand.
31          // -2: Create a legendary 4/4 green Elemental creature token named Ashaya, the Awoken World.
32          // -7: Untap up to six target lands. They become 6/6 Elemental creatures. They're still lands.
33          addCard(Zone.HAND, playerB, "Nissa, Vastwood Seer");
34  
35          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+2: {this} deals 3 damage", playerB);
36          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
37  
38          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Nissa, Vastwood Seer", true);
39          setChoice(playerB, "Yes");
40          playLand(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Forest");
41          activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "-2: Create Ashaya, the Awoken World, a legendary 4/4 green Elemental creature token.");
42  
43          attack(3, playerA, "Silvercoat Lion");
44          block(3, playerB, "Ashaya, the Awoken World", "Silvercoat Lion");
45  
46          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "-X: Exile each permanent with mana value X or less that's one or more colors");
47          setChoice(playerA, "X=3");
48  
49          setStopAt(3, PhaseStep.END_TURN);
50          execute();
51  
52          assertPermanentCount(playerA, "Ugin, the Spirit Dragon", 1);
53          assertCounterCount("Ugin, the Spirit Dragon", CounterType.LOYALTY, 6);  // 7 + 2 - 3
54  
55          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
56          assertPermanentCount(playerB, "Ashaya, the Awoken World", 0);
57  
58          assertExileCount("Nissa, Vastwood Seer", 1);
59          assertExileCount("Quest for the Gravelord", 1);
60  
61          assertLife(playerA, 20);
62          assertLife(playerB, 17);
63      }
64  }

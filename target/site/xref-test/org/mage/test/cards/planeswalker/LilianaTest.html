17      @Test
18      public void testCreatureGainsZombieAsAdditionalType() {
19          /*
20          Binding Mummy {1}{W}
21          Creature - Zombie 2/2
22          Whenever another Zombie enters the battlefield under your control, you may tap target artifact or creature.
23           */
24          String bMummy = "Binding Mummy";
25  
26          /*
27          Liliana, Death's Majesty {3}{B}{B}
28           Planeswalker — Liliana 5 loyalty
29          [+1] : Create a 2/2 black Zombie creature token. Put the top two cards of your library into your graveyard.
30          [-3] : Return target creature card from your graveyard to the battlefield. That creature is a black Zombie in addition to its other colors and types.
31          [-7] : Destroy all non-Zombie creatures.
32           */
33          String liliannaDM = "Liliana, Death's Majesty";
34  
35          /*
36          Winged Shepherd {5}{W}
37          Creature - Angel 3/3
38          Flying, vigilance
39          Cycling {W}
40           */
41          String wShepherd = "Winged Shepherd";
42  
43          String yOx = "Yoked Ox"; // {W} 0/4
44  
45          addCard(Zone.BATTLEFIELD, playerA, bMummy);
46          addCard(Zone.HAND, playerA, liliannaDM);
47          addCard(Zone.GRAVEYARD, playerA, wShepherd);
48          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
49          addCard(Zone.BATTLEFIELD, playerB, yOx);
50  
51          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, liliannaDM, true);
52          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-3:"); // Liliana -3
53          addTarget(playerA, wShepherd); // returns to battlefield and become zombie on top of other types
54          setChoice(playerA, true); // use Binding Mummy ability
55          addTarget(playerA, yOx); // tap the ox
56  
57          setStopAt(1, PhaseStep.BEGIN_COMBAT);
58          setStrictChooseMode(true);
59          execute();
60  
61          assertPermanentCount(playerA, bMummy, 1);
62          assertPermanentCount(playerA, liliannaDM, 1);
63          assertPermanentCount(playerA, wShepherd, 1);
64          assertPermanentCount(playerB, yOx, 1);
65          assertCounterCount(playerA, liliannaDM, CounterType.LOYALTY, 2);
66          assertType(wShepherd, CardType.CREATURE, SubType.ZOMBIE); // should have subtype zombie on top of angel type
67          assertType(wShepherd, CardType.CREATURE, SubType.ANGEL);
68          assertTapped(yOx, true);
69      }

71      @Test
72      public void testCastingCreaturesFromGraveTriggerDesecratedTomb() {
73  
74          /*
75          Liliana, Untouched by Death {2}{B}{B}
76          Legendary Planeswalker — Liliana
77          +1: Put the top three cards of your library into your graveyard. If at least one of them is a Zombie card, each opponent loses 2 life and you gain 2 life.
78          −2: Target creature gets -X/-X until end of turn, where X is the number of Zombies you control.
79          −3: You may cast Zombie cards from your graveyard this turn.
80           */
81          String liliannaUbD = "Liliana, Untouched by Death";
82  
83          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
84          addCard(Zone.HAND, playerA, liliannaUbD);
85          /*
86           * Carrion Feeder {B}
87           * Creature — Zombie
88           * Carrion Feeder can’t block.
89           * Sacrifice a creature: Put a +1/+1 counter on Carrion Feeder.
90           */
91          addCard(Zone.GRAVEYARD, playerA, "Carrion Feeder");
92  
93          // Whenever one or more creature cards leave your graveyard, create a 1/1 black Bat creature token with flying.
94          addCard(Zone.BATTLEFIELD, playerA, "Desecrated Tomb", 1);
95  
96          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, liliannaUbD, true);
97  
98          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-3:"); // Liliana -3
99          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Carrion Feeder");
100 
101         setStopAt(1, PhaseStep.END_TURN);
102         execute();
103 
104         assertPermanentCount(playerA, liliannaUbD, 1);
105         assertCounterCount(playerA, liliannaUbD, CounterType.LOYALTY, 1);
106         assertPermanentCount(playerA, "Carrion Feeder", 1);
107 
108         assertPermanentCount(playerA, "Bat Token", 1);
109     }

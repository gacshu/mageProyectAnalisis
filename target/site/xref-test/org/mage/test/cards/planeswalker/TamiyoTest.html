21      /**
22       * Reported bug: I activated Tamiyo's +1 ability on a 5/5 Gideon and his 2/2 Knight Ally, but when they both attacked
23       *  and dealt damage I only drew one card when I'm pretty sure I was supposed to draw for each of the two.
24       */
25      @Test
26      public void testFieldResearcherFirstEffectOnGideon() {
27          addCard(Zone.BATTLEFIELD, playerA, "Tamiyo, Field Researcher", 1);
28  
29          /* Gideon, Ally of Zendikar {2}{W}{W} - 4 loyalty
30           * +1: Until end of turn, Gideon, Ally of Zendikar becomes a 5/5 Human Soldier Ally creature with indestructible
31           * that's still a planeswalker. Prevent all damage that would be dealt to him this turn.
32           * 0: Create a 2/2 white Knight Ally creature token.
33          **/
34          addCard(Zone.BATTLEFIELD, playerA, "Gideon, Ally of Zendikar", 1);
35  
36          // put 2/2 knight ally token on battlefield
37          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Create a");
38  
39          // next, activate Gideon to make him a 5/5 human soldier ally creature
40          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: Until end of turn");
41          waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
42          // finally, use Tamiyo +1 on both creatures
43          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: Choose up to two");
44          addTarget(playerA, "Knight Ally Token^Gideon, Ally of Zendikar"); // both token and Gideon as creature
45  
46          // attack with both unblocked
47          attack(3, playerA, "Knight Ally Token");
48          attack(3, playerA, "Gideon, Ally of Zendikar");
49  
50          setStopAt(3, PhaseStep.END_COMBAT);
51          execute();
52  
53          assertLife(playerB, 13); // 5 + 2 damage, 20 - 7 = 13
54          assertPermanentCount(playerA, "Tamiyo, Field Researcher", 1);
55          assertPermanentCount(playerA, "Gideon, Ally of Zendikar", 1);
56          assertPermanentCount(playerA, "Knight Ally Token", 1);
57          assertHandCount(playerA, 3); // two cards drawn from each creature dealing damage + 1 card drawn on turn
58      }

160

162         block(2, playerA, "Sylvan Advocate", "Memnite");
163 
164         castSpell(3, PhaseStep.UPKEEP, playerB, "Hero's Downfall");
165         addTarget(playerB, "Tamiyo, Field Researcher");
166 
167         attack(3, playerA, "Sylvan Advocate"); // should not get extra card

169         setStopAt(3, PhaseStep.END_COMBAT);

196

198 
199         attack(3, playerA, "Pillarfield Ox");
200         attack(3, playerA, "Silvercoat Lion");
201 
202         setStopAt(3, PhaseStep.END_COMBAT);
203         execute();
204

206         assertLife(playerB, 14);

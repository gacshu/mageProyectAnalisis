14  public class AjaniTest extends CardTestPlayerBase {
15  
16      @Test
17      public void CastAjani() {
18          // +1: You gain 2 life.
19          // -1: Put a +1/+1 counter on each creature you control. Those creatures gain vigilance until end of turn.
20          // -6: Put a white Avatar creature token onto the battlefield. It has "This creature's power and toughness are each equal to your life total."
21          addCard(Zone.HAND, playerA, "Ajani Goldmane"); // {2}{W}{W} starts with 4 Loyality counters
22          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
23  
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ajani Goldmane", true);
25          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: You gain 2 life");
26  
27          setStopAt(1, PhaseStep.BEGIN_COMBAT);
28          execute();
29  
30          assertPermanentCount(playerA, "Ajani Goldmane", 1);
31          assertCounterCount("Ajani Goldmane", CounterType.LOYALTY, 5);  // 4 + 1 = 5
32  
33          assertLife(playerA, 22);
34          assertLife(playerB, 20);
35      }
36  
37      @Test
38      public void CastAjaniWithOathOfGideon() {
39          // +1: You gain 2 life.
40          // -1: Put a +1/+1 counter on each creature you control. Those creatures gain vigilance until end of turn.
41          // -6: Put a white Avatar creature token onto the battlefield. It has "This creature's power and toughness are each equal to your life total."
42          addCard(Zone.HAND, playerA, "Ajani Goldmane"); // {2}{W}{W} starts with 4 Loyality counters
43          // When Oath of Gideon enters the battlefield, put two 1/1 Kor Ally creature tokens onto the battlefield.
44          // Each planeswalker you control enters the battlefield with an additional loyalty counter on it.
45          addCard(Zone.HAND, playerA, "Oath of Gideon"); // {2}{W}
46          addCard(Zone.BATTLEFIELD, playerA, "Plains", 7);
47  
48          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Oath of Gideon", true);
49          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ajani Goldmane", true);
50          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: You gain 2 life");
51  
52          setStopAt(1, PhaseStep.BEGIN_COMBAT);
53          execute();
54  
55          assertPermanentCount(playerA, "Kor Ally Token", 2);
56          assertPermanentCount(playerA, "Oath of Gideon", 1);
57          assertPermanentCount(playerA, "Ajani Goldmane", 1);
58          assertCounterCount("Ajani Goldmane", CounterType.LOYALTY, 6);  // 4 + 1 + 1 = 6
59  
60          assertLife(playerA, 22);
61          assertLife(playerB, 20);
62      }

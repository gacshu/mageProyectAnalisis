18  // https://github.com/magefree/mage/issues/5677
19  /*
20  EDH again, I'm playing Nissa, Vastwood Seer as my commander.
21  I got her to 7 loyalty, used her -7 to untap lands and make them into creatures. So far so good.
22  Nissa died, went back to command zone, lands stayed as creatures (as they should).
23  Next turn I replay Nissa, but the moment I try to cast another spell (or activate an ability, 
24  maybe it was directly on replaying her, can't remember exactly), all my 6 creature lands are back to being regular lands.
25  A few turns later, my opponent uses overloaded cyclonic rift, which also bounces Nissa to my hand.
26  Not entirely sure, but I believe it again took at least until priority got passed for my lands to reappear, 
27  maybe even until I tried to activate an ability or cast something.
28  This was not a 1 time thing either.
29  After replaying Nissa (again) and casting another card, my creature lands went back to being regular lands again.
30  Later I used her -2 to kill her, which (after putting another spell on the stack) got me my creature lands back again.
31      */
32      @Test
33      public void NissaVastwoodSeerAnimationTest() {
34          setStrictChooseMode(true);
35          
36          addCard(Zone.LIBRARY, playerA, "Forest");
37          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
38          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
39          
40          // When Nissa, Vastwood Seer enters the battlefield, you may search your library for a basic Forest card, 
41          // reveal it, put it into your hand, then shuffle your library.
42          // Whenever a land enters the battlefield under your control, if you control seven or more lands, 
43          // exile Nissa, then return her to the battlefield transformed under her owner's control.
44          // Nissa, Sage Animist
45          // +1: Reveal the top card of your library. If it's a land card, put it onto the battlefield. Otherwise, put it into your hand.
46          // -2: Create a legendary 4/4 green Elemental creature token named Ashaya, the Awoken World.
47          // -7: Untap up to six target lands. They become 6/6 Elemental creatures. They're still lands.        
48          addCard(Zone.HAND, playerA, "Nissa, Vastwood Seer");
49  
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nissa, Vastwood Seer", true);
51          setChoice(playerA, true);
52          addTarget(playerA, "Forest");
53          
54          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Forest");
55  
56          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "+1: Reveal");
57          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "+1: Reveal");
58          activateAbility(5, PhaseStep.POSTCOMBAT_MAIN, playerA, "+1: Reveal");
59          activateAbility(7, PhaseStep.POSTCOMBAT_MAIN, playerA, "+1: Reveal");
60          activateAbility(9, PhaseStep.POSTCOMBAT_MAIN, playerA, "-7: Untap up to six target");
61          addTarget(playerA, "Forest^Forest^Forest^Forest^Forest^Forest");
62          setStopAt(9, PhaseStep.END_TURN);
63          execute();
64  
65          assertPermanentCount(playerA, "Nissa, Vastwood Seer", 0);
66          assertPermanentCount(playerA, "Nissa, Sage Animist", 0);
67          assertGraveyardCount(playerA, "Nissa, Vastwood Seer", 1);
68  
69          assertPermanentCount(playerA, "Swamp", 1);
70  
71          assertType("Forest", CardType.CREATURE, SubType.ELEMENTAL);        
72          assertPermanentCount(playerA, "Forest", 6);        
73          assertPowerToughness(playerA, "Forest", 6, 6, Filter.ComparisonScope.All);
74          
75          
76          
77          
78      }

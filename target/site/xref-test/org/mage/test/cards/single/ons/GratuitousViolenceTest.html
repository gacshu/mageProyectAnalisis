3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

9    *

12  public class GratuitousViolenceTest extends CardTestPlayerBase {
13      @Test
14      public void testDoublesDamageFromCreatures() {
15          // Enchantment: If a creature you control would deal damage to a creature
16          // or player, it deals double that damage to that creature or player instead.
17          addCard(Zone.BATTLEFIELD, playerA, "Gratuitous Violence");
18          addCard(Zone.BATTLEFIELD, playerA, "Elvish Visionary"); // 1/1
19          
20          attack(1, playerA, "Elvish Visionary");
21          
22          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
23          execute();
24          assertLife(playerB, 18);
25      }
26      
27      @Test
28      public void testIgnoresNonCreatures() {
29          // Legendary Enchantment - Shrine: At the beginning of your upkeep, Honden of Infinite
30          // Rage deals damage to any target equal to the number of Shrines you control.
31          addCard(Zone.BATTLEFIELD, playerA, "Honden of Infinite Rage");
32          addCard(Zone.BATTLEFIELD, playerA, "Gratuitous Violence");
33          
34          addTarget(playerA, playerB);
35          
36          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
37          execute();
38          
39          // Honden should deal 1 damage at upkeep (since playerA only
40          // has one Shrine). GV should not double this.
41          assertLife(playerB, 19);
42      }
43      
44      @Test
45      public void testIgnoresInstants() {
46          addCard(Zone.BATTLEFIELD, playerA, "Gratuitous Violence");
47          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
48          addCard(Zone.HAND, playerA, "Lightning Bolt");
49          
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
51          
52          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
53          execute();
54          assertLife(playerB, 17);
55      }
56  }

13      /*
14        New Perspectives {5}{U}
15        Enchantment
16        When New Perspectives enters the battlefield, draw three cards.
17        As long as you have seven or more cards in hand, you may pay 0 rather than pay cycling costs.
18       */
19      private final String newPerspectives = "New Perspectives";
20      private final String reliquaryTower = "Reliquary Tower"; // only used to not have to discard due to hand size to make testing easier
21      private final String flameJet = "Flame Jet"; // just a cycle card with (2) cycling
22  
23      /**
24       * just a basic test for free cycling!
25       */
26      @Test
27      public void newPerspectives_7Cards_FreeCycling() {
28  
29          removeAllCardsFromHand(playerA);
30          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
31          addCard(Zone.BATTLEFIELD, playerA, reliquaryTower);
32          addCard(Zone.HAND, playerA, newPerspectives);
33          addCard(Zone.HAND, playerA, "Mountain", 3);
34          addCard(Zone.HAND, playerA, flameJet);
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, newPerspectives);
37          activateAbility(1, PhaseStep.BEGIN_COMBAT, playerA, "Cycling");
38          setChoice(playerA, true);
39  
40          setStopAt(1, PhaseStep.END_COMBAT);
41          execute();
42  
43          assertPermanentCount(playerA, newPerspectives, 1);
44          assertGraveyardCount(playerA, flameJet, 1);
45          assertHandCount(playerA, 7); // 4 + 3 (new perspectives ETB) (+ 1 - 1) (cycling)
46      }
47  
48      /**
49       * Fewer than 7 cards in hand, no free cycle!
50       */
51      @Test
52      public void newPerspectives_LessThan7Cards_CyclingNotFree() {
53  
54          removeAllCardsFromHand(playerA);
55          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
56          addCard(Zone.BATTLEFIELD, playerA, reliquaryTower);
57          addCard(Zone.HAND, playerA, newPerspectives);
58          addCard(Zone.HAND, playerA, flameJet);
59  
60          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, newPerspectives);
61          checkPlayableAbility("Can't cycle", 1, PhaseStep.BEGIN_COMBAT, playerA, "Cycling", false);
62  
63          setStopAt(1, PhaseStep.END_COMBAT);
64          execute();
65  
66          assertPermanentCount(playerA, newPerspectives, 1);
67          assertGraveyardCount(playerA, flameJet, 0);
68          assertHandCount(playerA, flameJet, 1);
69          assertHandCount(playerA, 4); // 1 + 3 (new perspectives ETB)
70      }

85          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, newPerspectives);

88          activateAbility(3, PhaseStep.BEGIN_COMBAT, playerA, "Cycling");

92          execute();

18      @Test
19      public void testCard() {
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
21          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
22          // {2}{R}{G}: Until end of turn, Raging Ravine becomes a 3/3 red and green Elemental creature with "Whenever this creature attacks, put a +1/+1 counter on it." It's still a land.
23          addCard(Zone.BATTLEFIELD, playerA, "Raging Ravine");
24          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
25          // Until end of turn, target creature loses all abilities and becomes a blue Frog with base power and toughness 1/1.
26          addCard(Zone.HAND, playerB, "Turn to Frog");
27  
28          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{2}{R}{G}: Until end of turn, {this} becomes a 3/3 red and green Elemental creature with \"Whenever this creature attacks, put a +1/+1 counter on it.\" It's still a land.");
29          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
30          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Turn to Frog", "Raging Ravine");
31          attack(1, playerA, "Raging Ravine");
32  
33          setStopAt(1, PhaseStep.END_COMBAT);
34          execute();
35  
36          assertLife(playerA, 20);
37          assertLife(playerB, 19);
38          assertPowerToughness(playerA, "Raging Ravine", 1, 1, Filter.ComparisonScope.Any);
39          assertCounterCount("Raging Ravine", CounterType.P1P1, 0);
40      }

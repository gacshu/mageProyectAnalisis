2   package org.mage.test.cards.single.lgn;
3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   *
11   * @author cg5
12   */
13  public class InfernalCaretakerTest extends CardTestPlayerBase {
14      
15      @Test
16      /*
17       * Infernal Caretaker {3}{B}
18       * Creature - Human Cleric
19       * Morph {3}{B}
20       * When Infernal Caretaker is turned face up, return all Zombie cards from
21       * all graveyards to their owners' hands.
22       */
23      public void testInfernalCaretaker() {
24          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 10);
25          addCard(Zone.BATTLEFIELD, playerA, "Walking Corpse", 1);
26          addCard(Zone.HAND, playerA, "Infernal Caretaker", 1);
27          addCard(Zone.GRAVEYARD, playerA, "Walking Corpse", 4);
28          addCard(Zone.GRAVEYARD, playerA, "Storm Crow", 4);
29          
30          addCard(Zone.GRAVEYARD, playerB, "Festering Goblin", 4);
31          addCard(Zone.GRAVEYARD, playerB, "Elvish Visionary", 4);
32          
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Infernal Caretaker", true);
34          setChoice(playerA, true); // Cast as a morph
35          
36          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{3}{B}: Turn this face-down permanent face up.");
37          setStopAt(1, PhaseStep.END_TURN);
38          execute();
39          
40          assertPermanentCount(playerA, "Infernal Caretaker", 1);
41          assertPermanentCount(playerA, "Walking Corpse", 1);
42          assertHandCount(playerA, "Walking Corpse", 4);
43          assertHandCount(playerA, "Storm Crow", 0);
44          assertHandCount(playerA, "Festering Goblin", 0);
45          assertHandCount(playerA, "Elvish Visionary", 0);
46          assertGraveyardCount(playerA, 4); // 4 * Storm Crow
47          
48          assertPermanentCount(playerB, "Infernal Caretaker", 0);
49          assertPermanentCount(playerB, "Walking Corpse", 0);
50          assertHandCount(playerB, "Walking Corpse", 0);
51          assertHandCount(playerB, "Storm Crow", 0);
52          assertHandCount(playerB, "Festering Goblin", 4);
53          assertHandCount(playerB, "Elvish Visionary", 0);
54          assertGraveyardCount(playerB, 4); // 4 * Elvish Visionary
55      }
56  }

17   * When you cycle Decree of Pain, all creatures get -2/-2 until end of turn.
18   */
19  public class DecreeOfPainTest  extends CardTestPlayerBase {
20  
21      /**
22       * Reported bug: https://github.com/magefree/mage/issues/9273
23       * You cast Decree of pain while an opponent has Xyris, the Writhing Storm and 9 other creatures on the battlefield.
24       * All 10 creatures die, you draw 10 cards, but Xyris triggers and creates 10 snakes.
25       * This is wrong since the draws occur after the creatures have been destroyed.
26       */
27      @Test
28      @Ignore
29      public void testDrawHappensAfterDestruction(){
30          // Destroy all creatures. They can’t be regenerated. Draw a card for each creature destroyed this way.
31          addCard(Zone.HAND, playerA, "Decree of Pain");
32          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8);
33  
34          // Whenever an opponent draws a card except the first one they draw in each of their draw steps, create a 1/1 green Snake creature token.
35          addCard(Zone.BATTLEFIELD, playerB, "Xyris, the Writhing Storm");
36          addCard(Zone.BATTLEFIELD, playerB, "Aven Envoy", 9);
37  
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Decree of Pain");
39  
40          execute();
41  
42          assertGraveyardCount(playerB, "Xyris, the Writhing Storm", 1);
43          assertGraveyardCount(playerB, "Aven Envoy", 9);
44          assertPermanentCount(playerB, 0);
45          assertHandCount(playerB, 0);
46      }

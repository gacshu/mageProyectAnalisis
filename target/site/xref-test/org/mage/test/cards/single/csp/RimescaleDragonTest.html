1   package org.mage.test.cards.single.csp;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import mage.counters.CounterType;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   * @author JRHerlehy Created on 7/27/17.
11   */
12  public class RimescaleDragonTest extends CardTestPlayerBase {
13  
14      // Flying
15      // {2}{S}: Tap target creature and put an ice counter on it.
16      // Creatures with ice counters on them don't untap during their controllers' untap steps.
17      private final String dragon = "Rimescale Dragon";
18      private final String snowMountain = "Snow-Covered Mountain";
19      private final String thopter = "Ornithopter";
20      private final String terror = "Terror";
21  
22      private final String ability = "{2}{S}: Tap target";
23  
24      @Test
25      public void testActivatedAbility() {
26          this.setupTest();
27  
28          this.setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
29          this.execute();
30  
31          this.assertTapped(thopter, true);
32          this.assertCounterCount(thopter, CounterType.ICE, 1);
33      }
34  
35      @Test
36      public void testStaticAbility() {
37          this.setupTest();
38  
39          this.setStopAt(2, PhaseStep.UPKEEP);
40          this.execute();
41  
42          this.assertTapped(thopter, true);
43          this.assertCounterCount(thopter, CounterType.ICE, 1);
44      }
45  
46      @Test
47      public void testStaticAbilityEnded() {
48          this.setupTest();
49  
50          this.castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, terror, dragon);
51  
52          this.setStopAt(2, PhaseStep.UPKEEP);
53          this.execute();
54  
55          this.assertTapped(thopter, false);
56          this.assertCounterCount(thopter, CounterType.ICE, 1);
57      }
58  
59      private void setupTest() {
60          this.addCard(Zone.BATTLEFIELD, playerA, dragon);
61          this.addCard(Zone.BATTLEFIELD, playerA, snowMountain, 3);
62  
63          this.addCard(Zone.BATTLEFIELD, playerB, thopter);
64          this.addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
65          this.addCard(Zone.HAND, playerB, terror);
66  
67          this.activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, ability, thopter);
68      }
69  }

13      @Test
14      public void test_Exile_PermanentCard() {
15          // Exile target nonland permanent. For as long as that card remains exiled, its owner may cast it without paying its mana cost.
16          addCard(Zone.HAND, playerA, "Release to the Wind"); // {2}{U}
17          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
18          //
19          addCard(Zone.BATTLEFIELD, playerB, "Grizzly Bears", 1);
20  
21          // exile
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Release to the Wind", "Grizzly Bears");
23          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
24          checkExileCount("after exile", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Grizzly Bears", 1);
25          checkPlayableAbility("after exile - non owner can't play 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Grizzly Bears", false);
26  
27          // owner can play
28          checkPlayableAbility("after exile - non owner can't play 2", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Grizzly Bears", false);
29          checkPlayableAbility("after exile - owner can play", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Grizzly Bears", true);
30          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Grizzly Bears");
31  
32          setStrictChooseMode(true);
33          setStopAt(2, PhaseStep.END_TURN);
34          execute();
35  
36          assertPermanentCount(playerB, "Grizzly Bears", 1);
37      }

39      @Test
40      public void test_Exile_ModalDoubleFacesCard() {
41          // Exile target nonland permanent. For as long as that card remains exiled, its owner may cast it without paying its mana cost.
42          addCard(Zone.HAND, playerA, "Release to the Wind"); // {2}{U}
43          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
44          //
45          // Akoum Warrior {5}{R} - creature
46          // Akoum Teeth - land
47          addCard(Zone.HAND, playerA, "Akoum Warrior");
48          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
49  
50          // prepare mdf
51          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {R}", 6);
52          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akoum Warrior");
53          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
54          checkPermanentCount("prepare", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akoum Warrior", 1);
55  
56          // exile mdf creature
57          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Release to the Wind", "Akoum Warrior");
58          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
59          checkExileCount("after exile", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akoum Warrior", 1);
60  
61          // you can cast mdf, but can't play a land
62          checkPlayableAbility("after exile - can play mdf creature", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Akoum Warrior", true);
63          checkPlayableAbility("after exile - can't play mdf land", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Play Akoum Teeth", false);
64  
65          // cast mdf again for free
66          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akoum Warrior");
67  
68          setStrictChooseMode(true);
69          setStopAt(1, PhaseStep.END_TURN);
70          execute();
71  
72          assertPermanentCount(playerA, "Akoum Warrior", 1);
73      }

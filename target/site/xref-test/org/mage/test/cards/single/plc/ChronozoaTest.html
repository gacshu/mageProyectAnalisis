24    // When Chronozoa dies, if it had no time counters on it, put two tokens that are copies of it onto the battlefield.

26    /**
27     * Test that time counters are removed before the draw phase
28     */
29    @Test
30    public void testVanishing() {
31      addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
32      addCard(Zone.HAND, playerA, "Chronozoa");
33  
34      castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chronozoa");
35  
36      setStopAt(5, PhaseStep.DRAW);
37      execute();
38  
39      // Make sure one time counter was removed at beginning of playerA turn num 3
40      assertCounterCount("Chronozoa", CounterType.TIME, 1);
41    }
42  
43    /**
44     * Test that the tokens are put to battlefield if the last time counter is removed
45     */
46    @Test
47    public void testDuplicationEffect() {
48      addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
49      addCard(Zone.HAND, playerA, "Chronozoa");
50  
51      castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chronozoa");
52  
53      setStopAt(9, PhaseStep.PRECOMBAT_MAIN);
54      execute();
55  
56      // The original Chronozoa card should be in graveyard
57      assertGraveyardCount(playerA, 1);
58  
59      final List<Permanent> creatures = currentGame.getBattlefield().getAllActivePermanents(CardType.CREATURE, currentGame);
60      Assert.assertEquals(2, creatures.size());
61  
62      for (final Permanent creature : creatures) {
63        // Make sure the creatures are Chronozoa tokens
64        Assert.assertEquals("Chronozoa", creature.getName());
65        Assert.assertEquals("Chronozoa has to be a token", true, creature instanceof PermanentToken);
66  
67        // Make sure each token has 2 time counters
68        final Counters counters = creature.getCounters(currentGame);
69        Assert.assertEquals(1, counters.size());
70        for(final Counter counter : counters.values()) {
71          Assert.assertEquals(CounterType.TIME.getName(), counter.getName());
72          Assert.assertEquals(2, counter.getCount());
73        }
74      }
75    }
76  
77    @Test
78    public void testChronozoaDestroyedWithTimeCounters() throws Exception {
79      addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
80      addCard(Zone.HAND, playerA, "Chronozoa");
81      addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
82      addCard(Zone.HAND, playerB, "Lightning Bolt");
83  
84      castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chronozoa");
85      // Destroy Chronozoa the same phase it should duplicate -> due to stack Chronozoa is destroyed before duplication
86      castSpell(7, PhaseStep.UPKEEP, playerB, "Lightning Bolt", "Chronozoa");
87  
88      setStopAt(7, PhaseStep.PRECOMBAT_MAIN);
89      execute();
90  
91      // Chronozoa in gy
92      assertGraveyardCount(playerA, 1);
93      // Lightning Bolt in gt
94      assertGraveyardCount(playerB, 1);
95  
96      // Chronozoa shouldn't duplicate
97      final List<Permanent> creatures = currentGame.getBattlefield().getAllActivePermanents(CardType.CREATURE, currentGame);
98      Assert.assertTrue(creatures.isEmpty());
99  
100   }

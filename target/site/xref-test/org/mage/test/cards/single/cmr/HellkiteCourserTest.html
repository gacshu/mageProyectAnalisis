13  
14      @Test
15      public void test_ETB() {
16          // https://github.com/magefree/mage/issues/7198
17  
18          // When Hellkite Courser enters the battlefield, you may put a commander you own from the command
19          // zone onto the battlefield. It gains haste. Return it to the command zone at the
20          // beginning of the next end step.
21          addCard(Zone.HAND, playerA, "Hellkite Courser", 1); // {4}{R}{R}
22          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
23          //
24          addCard(Zone.COMMAND, playerA, "Balduvian Bears", 1); // {1}{G}
25  
26          checkCommandCardCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
27  
28          // cast card and move commander to battle
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hellkite Courser");
30          setChoice(playerA, true); // put commander to battlefield
31          setChoice(playerA, "Balduvian Bears");
32          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
33          checkPermanentCount("after etb", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
34  
35          // try to attack (test haste)
36          attack(1, playerA, "Balduvian Bears", playerB);
37  
38          // keep until end
39          checkPermanentCount("keep after battle", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Balduvian Bears", 1);
40  
41          // return to command zone
42          checkPermanentCount("return on end", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 0);
43          checkCommandCardCount("return on end", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
44  
45          setStrictChooseMode(true);
46          setStopAt(2, PhaseStep.END_TURN);
47          execute();
48      }
49  }

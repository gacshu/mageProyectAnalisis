15      @Test
16      public void test_Emblem_Normal() {
17          removeAllCardsFromHand(playerA);
18  
19          // −6: You get an emblem with "Whenever you cast an Elf spell, it gains haste until end of turn and you draw two cards."
20          addCard(Zone.BATTLEFIELD, playerA, "Tyvar Kell");
21          //
22          addCard(Zone.HAND, playerA, "Arbor Elf", 1); // {G}
23          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
24  
25          // prepare emblem
26          addCounters(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tyvar Kell", CounterType.LOYALTY, 10);
27          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-6:");
28          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
29          checkHandCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1); // 1x elf
30  
31          // cast elf and draw
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf");
33          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
34          checkHandCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2); // 2x from draw
35          checkAbility("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf", HasteAbility.class, true);
36  
37          // check that it can attack
38          attack(1, playerA, "Arbor Elf", playerB);
39  
40          // haste must end on next turn
41          checkAbility("end", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf", HasteAbility.class, false);
42  
43          setStrictChooseMode(true);
44          setStopAt(2, PhaseStep.END_TURN);
45          execute();
46  
47          assertLife(playerB, 20 - 1); // 1x from elf
48      }

50      @Test
51      public void test_Emblem_Blink() {
52          removeAllCardsFromHand(playerA);
53  
54          // −6: You get an emblem with "Whenever you cast an Elf spell, it gains haste until end of turn and you draw two cards."
55          addCard(Zone.BATTLEFIELD, playerA, "Tyvar Kell");
56          //
57          addCard(Zone.HAND, playerA, "Arbor Elf", 1); // {G}
58          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
59          //
60          // Exile target creature you control, then return that card to the battlefield under your control.
61          addCard(Zone.HAND, playerA, "Cloudshift", 1); // {W}
62          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
63  
64          // prepare emblem
65          addCounters(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tyvar Kell", CounterType.LOYALTY, 10);
66          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-6:");
67          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
68          checkHandCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2); // 1x elf + 1x shift
69  
70          // cast elf and draw
71          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf");
72          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
73          checkHandCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 3); // 2x from draw + 1x shift
74          checkAbility("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf", HasteAbility.class, true);
75  
76          // blink elf, so it must lose haste
77          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cloudshift", "Arbor Elf");
78          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
79          checkAbility("blink", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Elf", HasteAbility.class, false);
80  
81          setStrictChooseMode(true);
82          setStopAt(1, PhaseStep.END_TURN);
83          execute();
84      }

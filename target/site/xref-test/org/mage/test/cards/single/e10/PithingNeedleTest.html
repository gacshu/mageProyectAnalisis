1   package org.mage.test.cards.single.e10;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Assert;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestMultiPlayerBase;
8   
9   /**
10   * {@link mage.cards.p.PithingNeedle Pithing Needle}
11   * Artifact
12   * {1}
13   * As Pithing Needle comes into play, name a card.
14   * Activated abilities of sources with the chosen name can't be played unless they're mana abilities.
15   *
16   * @author LevelX2, Alex-Vasile
17   */
18  public class PithingNeedleTest extends CardTestMultiPlayerBase {
19  
20      /**
21       * Test that it works and it respects range of influence
22       */
23      @Test
24      public void TestPithingNeedle() {
25          addCard(Zone.HAND, playerA, "Pithing Needle");
26          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
27          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
28          addCard(Zone.LIBRARY, playerA, "Pillarfield Ox", 1);
29  
30          addCard(Zone.BATTLEFIELD, playerB, "Island", 3);
31          // {2}{U}, {T}: Put target creature on the bottom of its owner's library.
32          //              That creature's controller reveals cards from the top of their library until they reveal a creature card.
33          //              The player puts that card onto the battlefield and the rest on the bottom of their library in any order.
34          //              Activate this ability only any time you could cast a sorcery.
35          addCard(Zone.BATTLEFIELD, playerB, "Proteus Staff", 1);
36  
37          addCard(Zone.BATTLEFIELD, playerC, "Island", 3);
38          addCard(Zone.BATTLEFIELD, playerC, "Proteus Staff", 1);
39          addCard(Zone.BATTLEFIELD, playerC, "Wall of Air", 1);
40          addCard(Zone.LIBRARY, playerC, "Wind Drake", 2);
41  
42          addCard(Zone.BATTLEFIELD, playerD, "Island", 3);
43          addCard(Zone.BATTLEFIELD, playerD, "Eager Cadet", 1);
44          addCard(Zone.LIBRARY, playerD, "Storm Crow", 2);
45  
46          skipInitShuffling();
47  
48          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Pithing Needle");
49          setChoice(playerA, "Proteus Staff");
50  
51          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerC, "{2}{U}", "Eager Cadet"); // allowed because Needle out of range
52  
53          activateAbility(4, PhaseStep.PRECOMBAT_MAIN, playerB, "{2}{U}", "Wall of Air"); // not allowed
54  
55          setStopAt(4, PhaseStep.END_TURN);
56          try {
57              execute();
58              Assert.fail("must throw exception on execute");
59          } catch (Throwable e) {
60              if (!e.getMessage().contains("Can't find ability to activate command: {2}{U}$target=Wall of Air")) {
61                  Assert.fail("Should have thrown an error about PlayerB not being able to use the staff to target Wall of Air, but got:\n" + e.getMessage());
62              }
63          }
64  
65          assertPermanentCount(playerD, "Eager Cadet", 0);
66          assertPermanentCount(playerD, "Storm Crow", 1);
67          assertPermanentCount(playerC, "Wall of Air", 1);
68          assertPermanentCount(playerC, "Wind Drake", 0);
69      }
70  }

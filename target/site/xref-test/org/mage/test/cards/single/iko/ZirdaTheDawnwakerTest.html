14      private static final String zirda = "Zirda, the Dawnwaker";
15      private static final String golem = "Igneous Golem";
16      private static final String mauler = "Barkhide Mauler";
17      private static final String geth = "Geth, Lord of the Vault";
18      private static final String lion = "Silvercoat Lion";
19  
20      @Test
21      public void testReduceToOne() {
22          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
23          addCard(Zone.BATTLEFIELD, playerA, zirda);
24          addCard(Zone.BATTLEFIELD, playerA, golem);
25  
26          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{2}");
27  
28          setStrictChooseMode(true);
29          setStopAt(1, PhaseStep.END_TURN);
30          execute();
31  
32          assertTapped("Swamp", true);
33          assertAbility(playerA, golem, TrampleAbility.getInstance(), true);
34      }
35  
36      @Test
37      public void testCycling() {
38          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
39          addCard(Zone.BATTLEFIELD, playerA, zirda);
40          addCard(Zone.HAND, playerA, mauler);
41  
42          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cycling");
43  
44          setStrictChooseMode(true);
45          setStopAt(1, PhaseStep.END_TURN);
46          execute();
47  
48          assertTapped("Swamp", true);
49          assertGraveyardCount(playerA, mauler, 1);

52      @Test
53      public void testWithGeth() {
54          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
55          addCard(Zone.BATTLEFIELD, playerA, zirda);
56          addCard(Zone.BATTLEFIELD, playerA, geth);
57          addCard(Zone.GRAVEYARD, playerB, lion);
58  
59          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{X}{B}");
60          setChoice(playerA, "X=2");
61          addTarget(playerA, lion);
62  
63          setStrictChooseMode(true);
64          setStopAt(1, PhaseStep.END_TURN);
65          execute();
66  
67          assertPermanentCount(playerA, lion, 1);
68          assertTapped(lion, true);
69          assertGraveyardCount(playerB, 2);
70          assertGraveyardCount(playerB, lion, 0);
71      }
72  
73      @Test
74      public void testWithGeth2() {
75          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
76          addCard(Zone.BATTLEFIELD, playerA, geth);
77          addCard(Zone.GRAVEYARD, playerB, zirda);
78          addCard(Zone.GRAVEYARD, playerB, lion);
79  
80          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{X}{B}");
81          setChoice(playerA, "X=3");
82          addTarget(playerA, zirda);
83  
84          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{X}{B}");
85          setChoice(playerA, "X=2");
86          addTarget(playerA, lion);
87  
88          setStrictChooseMode(true);
89          setStopAt(1, PhaseStep.END_TURN);
90          execute();
91  
92          assertPermanentCount(playerA, zirda, 1);
93          assertPermanentCount(playerA, lion, 1);
94          assertTapped(zirda, true);
95          assertTapped(lion, true);
96          assertGraveyardCount(playerB, 2 + 3);
97          assertGraveyardCount(playerB, zirda, 0);
98          assertGraveyardCount(playerB, lion, 0);
99      }

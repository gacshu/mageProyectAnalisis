14      @Test
15      public void test_ColorlessCreatureMustNotShareColorWithColoredCreatures() {
16          // bug: https://github.com/magefree/mage/issues/6984
17          // Golos got destroyed by spreading plague when a player played a red creature, and all colored creatures got destroyed when golos got played.
18  
19          addCard(Zone.HAND, playerA, "Golos, Tireless Pilgrim", 1); // {5}
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
21          //
22          // Whenever a creature enters the battlefield, destroy all other creatures that share a color with it. They canâ€™t be regenerated.
23          addCard(Zone.BATTLEFIELD, playerA, "Spreading Plague", 1);
24          //
25          addCard(Zone.BATTLEFIELD, playerA, "Goblin Archaeologist", 1); // red creature
26          addCard(Zone.BATTLEFIELD, playerA, "Blisterpod", 1); // devoided creature (colorless)
27          //
28          addCard(Zone.HAND, playerA, "Goblin Arsonist", 1); // {R}
29          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
30  
31          checkPermanentCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Archaeologist", 1);
32          checkPermanentCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Blisterpod", 1);
33  
34          // play colorless creature and do not destroy anything
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Golos, Tireless Pilgrim");
36          setChoice(playerA, "When {this} enters the battlefield"); // two replacement effects
37          setChoice(playerA, false); // no search
38          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
39          checkPermanentCount("after colorless play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Golos, Tireless Pilgrim", 1);
40          checkPermanentCount("after colorless play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Archaeologist", 1);
41          checkPermanentCount("after colorless play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Blisterpod", 1);
42          checkPermanentCount("after colorless play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Arsonist", 0);
43  
44          // play red creature and destroy other red
45          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Arsonist");
46          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
47          checkPermanentCount("after red play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Golos, Tireless Pilgrim", 1);
48          checkPermanentCount("after red play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Archaeologist", 0);
49          checkPermanentCount("after red play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Blisterpod", 1);
50          checkPermanentCount("after red play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Goblin Arsonist", 1);
51  
52          setStrictChooseMode(true);
53          setStopAt(1, PhaseStep.END_TURN);
54          setStrictChooseMode(true);
55          execute();
56      }

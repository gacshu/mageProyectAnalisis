13  public class KeranosGodOfStormsTest extends CardTestPlayerBase {
14  
15      @Test
16      public void testKeranosNormal() {
17          // Reveal the first card you draw on each of your turns. 
18          // Whenever you reveal a land card this way, draw a card. 
19          // Whenever you reveal a nonland card this way, Keranos deals 3 damage to any target.
20          addCard(Zone.BATTLEFIELD, playerB, "Keranos, God of Storms"); // {3}{U}{R}
21          // Look at target player's hand.
22          // Draw a card.
23          addCard(Zone.LIBRARY, playerB, "Silvercoat Lion", 3);
24          skipInitShuffling();
25  
26          addTarget(playerB, playerA); // damage from Keranos trigger
27  
28          setStopAt(2, PhaseStep.END_TURN);
29          execute();
30  
31          assertPermanentCount(playerB, "Keranos, God of Storms", 1);
32  
33          assertHandCount(playerB, "Silvercoat Lion", 1); // 1 Lion from the draw and 1 Lion from Peek
34          
35          assertLife(playerA, 17);
36          assertLife(playerB, 20);
37  
38      }

40      /**
41       * Keranos, God of Storms, will look at the first card drawn after he is
42       * played, not the first card drawn each turn.
43       *
44       * My opponent, draws, plays Keranos, then plays Stroke of Genius. Keranos
45       * triggers on the first card for Stroke of Genius.
46       */
47      @Test
48      public void testKeranosCastAfterFirstDraw() {
49          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
50          addCard(Zone.BATTLEFIELD, playerB, "Island", 5);
51          // Reveal the first card you draw on each of your turns. 
52          // Whenever you reveal a land card this way, draw a card. 
53          // Whenever you reveal a nonland card this way, Keranos deals 3 damage to any target.
54          addCard(Zone.HAND, playerB, "Keranos, God of Storms"); // {3}{U}{R}
55          // Look at target player's hand.
56          // Draw a card.
57          addCard(Zone.HAND, playerB, "Peek");
58          
59          addCard(Zone.LIBRARY, playerB, "Silvercoat Lion", 3);
60          skipInitShuffling();
61  
62          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Keranos, God of Storms");
63          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Peek", playerA); // you won't do damage because it's not the first draw this turn - Draw in draw phase was the first
64  
65          setStopAt(2, PhaseStep.END_TURN);
66          execute();
67  
68          assertPermanentCount(playerB, "Keranos, God of Storms", 1);
69          assertGraveyardCount(playerB, "Peek", 1);
70  
71          assertHandCount(playerB, "Silvercoat Lion", 2); // 1 Lion from the draw and 1 Lion from Peek
72          
73          assertLife(playerA, 20);
74          assertLife(playerB, 20);
75  
76      }

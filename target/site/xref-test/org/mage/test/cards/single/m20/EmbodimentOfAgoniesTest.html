13      @Test
14      public void testHybridAndPhyrexian() {
15          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
16          addCard(Zone.HAND, playerA, "Embodiment of Agonies");
17          addCard(Zone.GRAVEYARD, playerA, "Rakdos Cackler"); // Mana Cost: {B/R}
18          addCard(Zone.GRAVEYARD, playerA, "Surgical Extraction"); // Mana Cost: {B/P}
19          addCard(Zone.GRAVEYARD, playerA, "Tormented Soul"); // Mana Cost: {B}
20          addCard(Zone.GRAVEYARD, playerA, "Gut Shot"); // Mana Cost: {R/P}
21          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt"); // Mana Cost: {R}
22  
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Embodiment of Agonies");
24  
25          setStopAt(1, PhaseStep.END_TURN);
26          execute();
27  
28          // Creature should be 5/5 as there are 5 distinct mana costs in graveyard
29          assertPowerToughness(playerA, "Embodiment of Agonies", 5, 5);

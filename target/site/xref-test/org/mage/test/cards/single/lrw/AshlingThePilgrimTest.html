29          addCard(Zone.BATTLEFIELD, playerA, ashling);

31          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, ashAbility);
32          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, ashAbility);

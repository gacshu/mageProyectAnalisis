9    * @author TheElk801

11  public class SoulReapTest extends CardTestPlayerBase {
12      private static final String reap = "Soul Reap";
13      private static final String lion = "Silvercoat Lion";
14      private static final String rats = "Muck Rats";
15  
16      @Test
17      public void testSpell() {
18          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
19          addCard(Zone.BATTLEFIELD, playerA, lion);
20          addCard(Zone.HAND, playerA, reap);
21          addCard(Zone.HAND, playerA, rats);
22  
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, rats);
24          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, reap, lion);
25  
26          setStrictChooseMode(true);
27          setStopAt(1, PhaseStep.END_TURN);
28          execute();
29  
30          assertPermanentCount(playerA, lion, 0);
31          assertPermanentCount(playerA, rats, 1);
32          assertGraveyardCount(playerA, lion, 1);
33          assertGraveyardCount(playerA, reap, 1);
34          assertLife(playerA, 20 - 3);
35      }
36  
37      @Test
38      public void testNoSpell() {
39          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
40          addCard(Zone.BATTLEFIELD, playerA, lion);

42  
43          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, reap, lion);
44  
45          setStrictChooseMode(true);
46          setStopAt(1, PhaseStep.END_TURN);

49          assertPermanentCount(playerA, lion, 0);

53      }

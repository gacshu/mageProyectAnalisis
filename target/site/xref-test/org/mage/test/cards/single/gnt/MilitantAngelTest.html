21      @Test
22      public void test_AttackedThisTurnOpponentsCount() {
23          // Player order: A -> D -> C -> B
24  
25          // it's testing counter only (no need to test card -- it's same)
26          // When Militant Angel enters the battlefield, create a number of 2/2 white Knight creature tokens
27          // with vigilance equal to the number of opponents you attacked this turn.
28          //addCard(Zone.BATTLEFIELD, playerA, "Militant Angel", 2);
29          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
30          addCard(Zone.BATTLEFIELD, playerA, "Kitesail Corsair", 1);
31  
32          // turn 1: 2x attack -> 1 player = 1 value
33          mustHaveValue("2x->1 = 1", 1, PhaseStep.PRECOMBAT_MAIN, 0);
34          attack(1, playerA, "Balduvian Bears", playerB);
35          attack(1, playerA, "Kitesail Corsair", playerB);
36          mustHaveValue("2x->1 = 1", 1, PhaseStep.POSTCOMBAT_MAIN, 1);
37  
38          // between attacks - no value
39          mustHaveValue("no attacks = 0", 2, PhaseStep.PRECOMBAT_MAIN, 0);
40  
41          // turn 5: 2x attack -> 2 players = 2 value
42          mustHaveValue("2x->2 = 2", 5, PhaseStep.PRECOMBAT_MAIN, 0);
43          attack(5, playerA, "Balduvian Bears", playerB);
44          attack(5, playerA, "Kitesail Corsair", playerC);
45          mustHaveValue("2x->2 = 2", 5, PhaseStep.POSTCOMBAT_MAIN, 2);
46          runCode("watcher must be copyable", 5, PhaseStep.POSTCOMBAT_MAIN, playerA, (info, player, game) -> {
47              PlayersAttackedThisTurnWatcher watcher = game.getState().getWatcher(PlayersAttackedThisTurnWatcher.class);
48              PlayersAttackedThisTurnWatcher newWatcher = watcher.copy();
49              Assert.assertEquals("old watcher", 2, watcher.getAttackedOpponentsCount(player.getId()));
50              Assert.assertEquals("new watcher", 2, newWatcher.getAttackedOpponentsCount(player.getId()));
51          });
52  
53          // between attacks - no value
54          mustHaveValue("no attacks = 0", 6, PhaseStep.PRECOMBAT_MAIN, 0);
55  
56          setStrictChooseMode(true);
57          setStopAt(6, PhaseStep.END_TURN);
58          execute();
59      }
60  
61      private void mustHaveValue(String needInfo, int turnNum, PhaseStep step, int needValue) {
62          runCode(needInfo, turnNum, step, playerA, (info, player, game) -> {
63              assertCounterValue(info, player, game, needValue);
64          });
65      }
66  
67      private void assertCounterValue(String checkName, Player player, Game game, int needValue) {
68          Ability fakeAbility = new SimpleStaticAbility((Effect) null); // dynamic value need ability's controllerId
69          fakeAbility.setControllerId(player.getId());
70          Assert.assertEquals(checkName, needValue, AttackedThisTurnOpponentsCount.instance.calculate(game, fakeAbility, null));
71      }

19      @Test
20      public void test_CheckExiled() {
21          setStrictChooseMode(true);
22          
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
24          
25          addCard(Zone.LIBRARY, playerA, "Silvercoat Lion", 1);
26          addCard(Zone.LIBRARY, playerB, "Pillarfield Ox", 1);
27          
28          skipInitShuffling();
29          // Partner with Pako, Arcane Retriever
30          // You may play noncreature cards from exile with fetch counters on them if you 
31          // exiled them, and you may spend mana as though it were mana of any color to cast those spells.
32          addCard(Zone.BATTLEFIELD, playerA, "Haldan, Avid Arcanist", 1);
33          // Partner with Haldan, Avid Arcanist
34          // Haste
35          // Whenever Pako, Arcane Retriever attacks, exile the top card of each player's library and put a fetch counter on each of them. Put a +1/+1 counter on Pako for each noncreature card exiled this way.        
36          addCard(Zone.BATTLEFIELD, playerA, "Pako, Arcane Retriever", 1); // Creature {3}{R}{G}  (3/3)
37  
38          attack(1, playerA, "Pako, Arcane Retriever");
39                 
40          setStopAt(1, PhaseStep.END_TURN);
41          execute();
42  
43          assertLife(playerA, 20);
44          assertLife(playerB, 17);
45          
46          assertExileCount(playerA, "Silvercoat Lion", 1);
47          assertExileCount(playerB, "Pillarfield Ox", 1);
48  
49          for(Card card :currentGame.getExile().getAllCards(currentGame)) {
50              Assert.assertTrue(card.getName() + " has a fetch counter",card.getCounters(currentGame).getCount(CounterType.FETCH) == 1);
51          }        
52  
53      }

55      @Test
56      public void test_CastExiled() {
57          // setStrictChooseMode(true);
58          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
59          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
60          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
61          
62          addCard(Zone.LIBRARY, playerA, "Lightning Bolt", 1); // Instant 3 damge
63          // Create a 3/3 green Centaur creature token.
64          addCard(Zone.LIBRARY, playerB, "Call of the Conclave", 1); // Sorcery {W}{G} 
65          
66          skipInitShuffling();
67          // Partner with Pako, Arcane Retriever
68          // You may play noncreature cards from exile with fetch counters on them if you 
69          // exiled them, and you may spend mana as though it were mana of any color to cast those spells.
70          addCard(Zone.BATTLEFIELD, playerA, "Haldan, Avid Arcanist", 1);
71          // Partner with Haldan, Avid Arcanist
72          // Haste
73          // Whenever Pako, Arcane Retriever attacks, exile the top card of each player's library and put a fetch counter on each of them. Put a +1/+1 counter on Pako for each noncreature card exiled this way.        
74          addCard(Zone.BATTLEFIELD, playerA, "Pako, Arcane Retriever", 1); // Creature {3}{R}{G}  (3/3)
75  
76          attack(1, playerA, "Pako, Arcane Retriever");        
77  
78          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
79          waitStackResolved(1, PhaseStep.POSTCOMBAT_MAIN);
80          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Call of the Conclave");
81          
82          setStopAt(1, PhaseStep.END_TURN);
83          execute();
84  
85          assertLife(playerA, 20);
86          assertLife(playerB, 12); // 3+2 (attack) + 3 Lighning Bolt
87                 
88          assertGraveyardCount(playerA, "Lightning Bolt", 1);
89          assertGraveyardCount(playerB, "Call of the Conclave", 1);
90      }

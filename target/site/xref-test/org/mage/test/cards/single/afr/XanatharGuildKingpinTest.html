13      @Test
14      public void test_Play() {
15          removeAllCardsFromLibrary(playerA);
16          removeAllCardsFromLibrary(playerB);
17          skipInitShuffling();
18  
19          // At the beginning of your upkeep, choose target opponent.
20          // Until end of turn, that player canâ€™t cast spells, you may look at the top card of their library any time,
21          // you may play the top card of their library, and you may spend mana as though it were mana of any color
22          // to cast spells this way.
23          addCard(Zone.BATTLEFIELD, playerA, "Xanathar, Guild Kingpin");
24          //
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
26          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
27          //
28          addCard(Zone.HAND, playerA, "Thunderbolt");
29          //
30          addCard(Zone.HAND, playerB, "Lightning Bolt");
31          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 2);
32          //
33          addCard(Zone.LIBRARY, playerB, "Mountain");
34          addCard(Zone.LIBRARY, playerB, "Lightning Bolt");
35          addCard(Zone.LIBRARY, playerB, "Grizzly Bears");
36  
37          // activate on opponent
38          addTarget(playerA, playerB);
39  
40          // B can't cast spells
41          checkPlayableAbility("B can't cast", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Lightning Bolt", false);
42          // A can cast own and from B library
43          checkPlayableAbility("A can cast own", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Thunderbolt", true);
44          checkPlayableAbility("A can cast from B", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Grizzly Bears", true);
45  
46          // cast from B and try another one with any color and full stack
47          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}", 2); // pay for {G} as any
48          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears");
49          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", null, "Grizzly Bears");
50          addTarget(playerA, playerB); // bolt
51          checkStackSize("multi cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
52  
53          // B can cast again
54          checkPlayableAbility("B can cast", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Lightning Bolt", true);
55  
56          setStrictChooseMode(true);
57          setStopAt(2, PhaseStep.END_TURN);
58          execute();
59      }

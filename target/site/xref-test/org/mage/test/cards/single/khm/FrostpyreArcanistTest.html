22      private static final String arcanist = "Frostpyre Arcanist";
23      private static final String bolt = "Lightning Bolt";
24  
25      @Test
26      public void testCanFind() {
27          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
28          addCard(Zone.LIBRARY, playerA, bolt);
29          addCard(Zone.GRAVEYARD, playerA, bolt);
30          addCard(Zone.HAND, playerA, arcanist);
31  
32          setStrictChooseMode(true);
33  
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, arcanist);
35          addTarget(playerA, bolt);
36  
37          setStopAt(1, PhaseStep.END_TURN);
38          execute();
39  
40          assertLibraryCount(playerA, bolt, 0);
41          assertHandCount(playerA, bolt, 1);
42      }
43  
44      @Test
45      public void testCantFind() {
46          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
47          addCard(Zone.LIBRARY, playerA, bolt);
48          addCard(Zone.HAND, playerA, arcanist);
49  
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, arcanist);
51          // Let the choice be made automatically since there is no sorcery or instant card in the graveyard.
52  
53          setStopAt(1, PhaseStep.END_TURN);
54          execute();
55  
56          assertLibraryCount(playerA, bolt, 1);
57          assertHandCount(playerA, bolt, 0);
58      }

3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import mage.counters.CounterType;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   * @author escplan9
11   */
12  public class DecimatorBeetleTest extends CardTestPlayerBase {
13  
14      /*    
15  Decimator Beetle {3}{B}{G}
16  Creature - Insect 4/5
17  When Decimator Beetle enters the battlefield, put a -1/-1 counter on target creature you control.
18  Whenever Decimator Beetle attacks, remove a -1/-1 counter from target creature you control and put a -1/-1 counter on up to one target creature defending player controls.
19      */
20      private final String decimator = "Decimator Beetle";
21  
22      @Test
23      public void targetOpponentCreatureWithDecimator() {
24  
25          String grizzly = "Grizzly Bears"; // {1}{G} 2/2
26          String hillGiant = "Hill Giant"; // {3}{R} 3/3
27  
28          addCard(Zone.HAND, playerA, decimator);
29          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
30          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
31          addCard(Zone.BATTLEFIELD, playerA, grizzly);
32          addCard(Zone.BATTLEFIELD, playerB, hillGiant);
33  
34          // put -1/-1 on own creature
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, decimator);
36          addTarget(playerA, grizzly);
37  
38          // remove -1/-1 from own creature and put to defender control
39          attack(3, playerA, decimator);
40          addTarget(playerA, grizzly); // remove
41          addTarget(playerA, hillGiant); // put
42  
43          setStrictChooseMode(true);
44          setStopAt(3, PhaseStep.END_COMBAT);
45          execute();
46  
47          assertPowerToughness(playerA, grizzly, 2, 2); // had -1/-1 counter, but removed on attack
48          assertPowerToughness(playerB, hillGiant, 2, 2); // gets -1/-1 counter from decimator attack ability
49          assertCounterCount(playerA, grizzly, CounterType.M1M1, 0);
50          assertCounterCount(playerB, hillGiant, CounterType.M1M1, 1);
51          assertLife(playerB, 16);
52      }
53  }

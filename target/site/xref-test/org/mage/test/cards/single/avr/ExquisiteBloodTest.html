48       *  Ajani, Inspiring leader does not trigger Exquisite Blood + Defiant Bloodlord #6464
49       */
50      @Test
51      public void triggerCascadeTest() {
52          // +2: You gain 2 life. Put two +1/+1 counters on up to one target creature.
53          // −3: Exile target creature. Its controller gains 2 life.
54          // −10: Creatures you control gain flying and double strike until end of turn.        
55          addCard(Zone.BATTLEFIELD, playerA, "Ajani, Inspiring Leader", 1); // Planeswalker (5)        
56  
57          // Flying
58          // Whenever you gain life, target opponent loses that much life.
59          addCard(Zone.BATTLEFIELD, playerA, "Defiant Bloodlord", 1); // Creature 4/5 {5}{B}{B}        
60          
61          // Whenever an opponent loses life, you gain that much life.
62          addCard(Zone.BATTLEFIELD, playerA, "Exquisite Blood", 1); // Enchantment {4}{B}
63  
64          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+2:", "Defiant Bloodlord");
65          addTarget(playerA, playerB);  // Target opponent of Defiant Bloodlord triggered ability (looping until opponent is dead)
66          addTarget(playerA, playerB);  
67          addTarget(playerA, playerB);  
68          addTarget(playerA, playerB);
69          addTarget(playerA, playerB);
70          addTarget(playerA, playerB);
71          addTarget(playerA, playerB);
72          addTarget(playerA, playerB);
73          addTarget(playerA, playerB);
74          addTarget(playerA, playerB);
75          
76          setStrictChooseMode(true);
77          setStopAt(1, PhaseStep.BEGIN_COMBAT);
78          execute();
79  
80          assertPowerToughness(playerA, "Defiant Bloodlord", 6, 7);
81          assertCounterCount("Ajani, Inspiring Leader", CounterType.LOYALTY, 7);
82  
83          assertLife(playerB, 0); // Player B is dead, game ends
84          assertLife(playerA, 40);
85          
86          
87      }

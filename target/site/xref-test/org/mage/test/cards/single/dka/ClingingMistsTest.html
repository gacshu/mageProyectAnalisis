15  public class ClingingMistsTest extends CardTestPlayerBase {
16  
17      @Test
18      public void testCard() {
19          addCard(Zone.BATTLEFIELD, playerA, "White Knight");
20          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
21          addCard(Zone.HAND, playerA, "Clinging Mists");
22  
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Clinging Mists");
24          attack(1, playerA, "White Knight");
25  
26          setStopAt(1, PhaseStep.END_TURN);
27          execute();
28  
29          assertLife(playerA, 20);
30          assertLife(playerB, 20);
31      }
32  
33      @Test
34      public void testCardExile1() {
35          setLife(playerA, 5);
36          addCard(Zone.BATTLEFIELD, playerA, "Abbey Griffin");
37          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
38          addCard(Zone.HAND, playerA, "Clinging Mists");
39  
40          attack(1, playerA, "Abbey Griffin");
41          castSpell(1, PhaseStep.DECLARE_BLOCKERS, playerA, "Clinging Mists");
42  
43          setStopAt(3, PhaseStep.END_TURN);
44          execute();
45  
46          assertLife(playerA, 5);
47          assertLife(playerB, 20);
48          assertTapped("Abbey Griffin", true);
49      }
50  
51      @Test
52      public void testCardExile2() {
53          setLife(playerA, 5);
54          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
55          /**
56           * Clinging Mists
57           * Instant
58           * Prevent all combat damage that would be dealt this turn.
59           * Fateful hour - If you have 5 or less life, tap all attacking creatures. Those creatures don't untap during their controller's next untap step.
60           *
61           */
62          addCard(Zone.HAND, playerA, "Clinging Mists");
63          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
64          addCard(Zone.BATTLEFIELD, playerB, "Abbey Griffin");
65          addCard(Zone.HAND, playerB, "Lightning Bolt");
66  
67          attack(2, playerB, "Abbey Griffin");
68          castSpell(2, PhaseStep.DECLARE_BLOCKERS, playerA, "Clinging Mists");
69          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Lightning Bolt", playerA);
70  
71          setStopAt(6, PhaseStep.DRAW);
72          execute();
73  
74          assertLife(playerA, 2);
75          assertLife(playerB, 20);
76          assertTapped("Abbey Griffin", false);
77          assertGraveyardCount(playerB, "Lightning Bolt", 1);
78      }

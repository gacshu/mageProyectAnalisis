14      @Test
15      public void test_EonFrolicker_NormalPlay() {
16          // https://github.com/magefree/mage/issues/6780
17  
18          // {2}{U}{U}
19          // When Eon Frolicker enters the battlefield, if you cast it, target opponent takes an extra turn after this one.
20          // Until your next turn, you and planeswalkers you control gain protection from that player.
21          // (You and planeswalkers you control can’t be targeted, dealt damage, or enchanted by anything controlled by that player.)
22          addCard(Zone.HAND, playerA, "Eon Frolicker", 1);
23          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
24          //
25          // Chandra’s Fury deals 4 damage to target player or planeswalker and 1 damage to each creature that player or that planeswalker’s controller controls.
26          addCard(Zone.HAND, playerB, "Chandra's Fury", 1); // {4}{R}
27          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 5);
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Eon Frolicker");
30          // addTarget(playerA, playerB); Autochosen, only target
31  
32          // AI can targets only Eon Frolicker (cause A protected from B)
33          checkPlayableAbility("after", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Cast Chandra's Fury", true);
34          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Chandra's Fury");
35          //addTarget(playerB, playerB);
36  
37          //setStrictChooseMode(true); // AI must choose target for fury (itself)
38          setStopAt(1, PhaseStep.END_TURN);
39          execute();
40  
41          assertGraveyardCount(playerB, "Chandra's Fury", 1);
42          assertLife(playerA, 20);
43          assertLife(playerB, 20 - 4);
44      }
45  }

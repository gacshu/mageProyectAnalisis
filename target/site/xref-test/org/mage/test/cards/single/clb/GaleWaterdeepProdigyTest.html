1   package org.mage.test.cards.single.clb;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;
7   
8   /**
9    * @author Rjayz
10   */
11  public class GaleWaterdeepProdigyTest extends CardTestPlayerBase {
12  
13      @Test
14      public void TestGaleWaterDeepProdigy() {
15          addCard(Zone.BATTLEFIELD, playerA, "Island", 10);
16          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 10);
17  
18          // Whenever you cast an instant or sorcery spell from your hand, you may cast up to one of the other type from your graveyard.
19          // If a spell cast from your graveyard this way would be put into your graveyard, exile it instead.
20          addCard(Zone.BATTLEFIELD, playerA, "Gale, Waterdeep Prodigy", 1);
21  
22          // Draw two cards, sorcery
23          addCard(Zone.HAND, playerA, "Divination");
24          // Deal three damage to any target, instant
25          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt");
26  
27          playerA.getLibrary().clear();
28          addCard(Zone.LIBRARY, playerA, "Island");
29          addCard(Zone.LIBRARY, playerA, "Island");
30          skipInitShuffling();
31  
32          setStrictChooseMode(true);
33  
34          // Cast Divination from hand,
35          // this will trigger Gale's ability and let playerA cast an instant from their graveyard
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Divination", true);
37          // Target Lightning Bolt in graveyard with Gale's ability
38          addTarget(playerA, "Lightning Bolt");
39          // Choose to cast Lightning Bolt when prompted
40          setChoice(playerA, true);
41          // Target opponent with Lightning bolt
42          addTarget(playerA, playerB);
43  
44          execute();
45  
46          // Assert Divination was cast from hand
47          assertHandCount(playerA, 2);
48          assertHandCount(playerA, "Island", 2);
49          assertLibraryCount(playerA, 0);
50          assertGraveyardCount(playerA, 1);
51          assertGraveyardCount(playerA, "Divination", 1);
52  
53          // Assert Lightning Bolt was cast from graveyard,
54          // and afterwards exiled instead of being put back into the graveyard
55          assertExileCount(playerA, 1);
56          assertExileCount(playerA, "Lightning Bolt", 1);
57  
58          // Assert opponent was targeted by Lightning Bolt
59          assertLife(playerB, 17);
60      }
61  }

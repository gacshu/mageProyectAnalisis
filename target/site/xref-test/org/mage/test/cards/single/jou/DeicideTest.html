13      public void targetNonCreatureeGod() {
14          // Exile target enchantment. If the exiled card is a God card, search its controller's graveyard, hand,
15          // and library for any number of cards with the same name as that card and exile them, then that player shuffles their library.
16          addCard(Zone.HAND, playerA, "Deicide");
17  
18          addCard(Zone.BATTLEFIELD, playerB, "Heliod, Sun-Crowned");
19          // add for Devotion {W}{W} x 3
20          addCard(Zone.BATTLEFIELD, playerB, "Crusade", 3);
21  
22          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
23  
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Deicide", "Heliod, Sun-Crowned");
25          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
26          execute();
27  
28          assertExiledCardSubtype("Heliod, Sun-Crowned", SubType.GOD);
29  
30  
31      }

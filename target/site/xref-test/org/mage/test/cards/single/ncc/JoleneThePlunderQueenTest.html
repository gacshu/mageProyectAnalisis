2   package org.mage.test.cards.single.ncc;
3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestCommander4Players;
8   
9   /**
10   *
11   * @author Susucre
12   */
13  public class JoleneThePlunderQueenTest extends CardTestCommander4Players {
14  
15      /*
16          Jolene, the Plunder Queen {2}{R}{G}
17          Legendary Creature — Human Warrior 2/2
18  
19          Whenever a player attacks one or more of your opponents, that attacking player creates a Treasure token.
20          If you would create one or more Treasure tokens, instead create those tokens plus an additional Treasure token.
21          Sacrifice five Treasures: Put five +1/+1 counters on Jolene.
22      */
23      String jolene = "Jolene, the Plunder Queen";
24  
25      /*
26          Elite Vanguard {W}
27          Creature — Human Soldier 2/1
28      */
29      String vanguard = "Elite Vanguard";
30  
31      /*
32          Balduvian Bears {1}{G}
33          Creature — Bear 2/2
34      */
35      String bear = "Balduvian Bears";
36  
37      /**
38       * test with three players:
39       * A with a Jolene and an Elite Vanguard
40       * B, C other players.
41       *
42       * A attacks both B & C, get two treasures.
43       */
44      @Test
45      public void testAttackingTwoOpponents() {
46          addCard(Zone.BATTLEFIELD, playerA, jolene, 1);
47          addCard(Zone.BATTLEFIELD, playerA, vanguard, 1);
48  
49          attack(1, playerA, jolene, playerB);
50          attack(1, playerA, vanguard, playerC);
51          
52          setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
53          execute();
54  
55          // 1 attack trigger, +1 Treasure with the replacement effect.
56          assertPermanentCount(playerA, "Treasure Token", 2);
57      }
58  
59      /**
60       * test with three players:
61       * A with an Elite Vanguard and a Balduvian Bears
62       * B with a Jolene
63       * C other player.
64       *
65       * A attacks both B & C, get one treasure.
66       */
67      @Test
68      public void testAttackingJoleneAndAnotherOpponent() {
69          addCard(Zone.BATTLEFIELD, playerA, vanguard, 1);
70          addCard(Zone.BATTLEFIELD, playerA, bear, 1);
71          addCard(Zone.BATTLEFIELD, playerB, jolene, 1);
72  
73          attack(1, playerA, bear, playerB);
74          attack(1, playerA, vanguard, playerC);
75          
76          setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
77          execute();
78  
79          // 1 attack trigger, not controlling Jolene so no replacement effect.
80          assertPermanentCount(playerA, "Treasure Token", 1);
81      }
82  
83      /**
84       * test with three players:
85       * A with an Elite Vanguard
86       * B with a Jolene
87       * C other player.
88       *
89       * A attacks only B, no trigger, no treasure.
90       */
91      @Test
92      public void testAttackingJoleneOnly() {
93          addCard(Zone.BATTLEFIELD, playerA, vanguard, 1);
94          addCard(Zone.BATTLEFIELD, playerB, jolene, 1);
95  
96          attack(1, playerA, vanguard, playerB);
97          
98          setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
99          execute();
100 
101         // no trigger, no token.
102         assertPermanentCount(playerA, "Treasure Token", 0);
103     }
104 
105     /**
106      * test with three players:
107      * A with a Jolene and an Elite Vanguard
108      * B with a Jolene.
109      * C with a Jolene.
110      *
111      * A attacks both B & C, 3 triggers, 6 treasures.
112      */
113     @Test
114     public void testEveryoneGotAJolene() {
115         addCard(Zone.BATTLEFIELD, playerA, vanguard, 1);
116         addCard(Zone.BATTLEFIELD, playerA, jolene, 1);
117         addCard(Zone.BATTLEFIELD, playerB, jolene, 1);
118         addCard(Zone.BATTLEFIELD, playerC, jolene, 1);
119 
120         attack(1, playerA, vanguard, playerB);
121         attack(1, playerA, jolene, playerC);
122 
123         setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
124         execute();
125 
126         // 3 triggers (1 for each Jolene), +1 Treasure for each with the replacement effect.
127         assertPermanentCount(playerA, "Treasure Token", 6);
128     }
129 }

12  public class SuddenDisappearanceTest extends CardTestPlayerBase {
13  
14      /**
15       * Sudden Disappearance
16       * Sorcery {5}{W}
17       * Exile all nonland permanents target player controls. 
18       * Return the exiled cards to the battlefield under 
19       * their owner's control at the beginning of the next end step
20       */
21      @Test
22      public void testCard() {
23          addCard(Zone.BATTLEFIELD, playerA, "Plains", 6);
24          addCard(Zone.HAND, playerA, "Sudden Disappearance");
25          addCard(Zone.BATTLEFIELD, playerB, "Air Elemental", 1);
26          addCard(Zone.BATTLEFIELD, playerB, "Horned Turtle", 4);
27          addCard(Zone.BATTLEFIELD, playerB, "Altar of the Lost", 1);
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sudden Disappearance", playerB);
30  
31          setStopAt(1, PhaseStep.BEGIN_COMBAT);
32          execute();
33  
34          assertLife(playerA, 20);
35          assertLife(playerB, 20);
36          assertPermanentCount(playerB, "Air Elemental", 0);
37          assertPermanentCount(playerB, "Horned Turtle", 0);
38          assertPermanentCount(playerB, "Altar of the Lost", 0);
39          assertExileCount("Air Elemental", 1);
40          assertExileCount("Horned Turtle", 4);
41          assertExileCount("Altar of the Lost", 1);
42      }

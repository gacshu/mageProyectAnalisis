20      @Test
21      public void testBasicExileCreature() {
22          // Soul Separator {3} Artifact
23          // {5}, {T}, Sacrifice Soul Separator: Exile target creature card from your graveyard.
24          // Create a token that's a copy of that card except it's 1/1, it's a Spirit in addition to its other types, and it has flying.
25          // Create a black Zombie creature token onto the battlefield with power equal to that card's power and toughness equal that card's toughness.
26          addCard(Zone.BATTLEFIELD, playerA, "Soul Separator");
27          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
28          addCard(Zone.GRAVEYARD, playerA, "Sylvan Advocate"); // 2/3 vigilance
29  
30          setStrictChooseMode(true);
31  
32          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{5}");
33          addTarget(playerA, "Sylvan Advocate");
34  
35          setStopAt(1, PhaseStep.BEGIN_COMBAT);
36          execute();
37  
38          assertGraveyardCount(playerA, "Soul Separator", 1);
39          assertExileCount("Sylvan Advocate", 1);
40          assertPermanentCount(playerA, "Sylvan Advocate", 1);
41          assertPermanentCount(playerA, "Zombie Token", 1);
42          assertPowerToughness(playerA, "Zombie Token", 2, 3);
43  
44          Permanent saToken = getPermanent("Sylvan Advocate", playerA);
45          Assert.assertTrue(saToken.getAbilities().contains(FlyingAbility.getInstance()));
46          Assert.assertTrue(saToken.hasSubtype(SubType.SPIRIT, currentGame));
47          Assert.assertTrue(saToken.getAbilities().contains(VigilanceAbility.getInstance()));
48          assertPowerToughness(playerA, "Sylvan Advocate", 1, 1);
49      }

51      // Reported bug: https://github.com/magefree/mage/issues/2174
52      //      Exiled Tree of Perdition with Soul Separator
53      //      The token copy when activated reduced the opponent's life total to 13 (tree toughness) instead of 1 (1/1 token)
54      @Test
55      public void testExileTreeOfPerdition() {
56          // Soul Separator {3} Artifact
57          // {5}, {T}, Sacrifice Soul Separator: Exile target creature card from your graveyard.
58          // Create a token that's a copy of that card except it's 1/1, it's a Spirit in addition to its other types, and it has flying.
59          // Create a black Zombie creature token onto the battlefield with power equal to that card's power and toughness equal that card's toughness.
60          addCard(Zone.BATTLEFIELD, playerA, "Soul Separator");
61          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
62  
63          // Tree of Perdition {3}{B} Creature - Defender (0/13)
64          // {tap}: Exchange target opponent's life total with Tree of Perdition's toughness.
65          addCard(Zone.GRAVEYARD, playerA, "Tree of Perdition");
66  
67          setStrictChooseMode(true);
68  
69          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{5}");
70          addTarget(playerA, "Tree of Perdition");
71          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Exchange");
72          addTarget(playerA, playerB);
73  
74          setStopAt(3, PhaseStep.BEGIN_COMBAT);
75          execute();
76  
77          assertGraveyardCount(playerA, "Soul Separator", 1);
78          assertExileCount("Tree of Perdition", 1);
79          assertPermanentCount(playerA, "Tree of Perdition", 1);
80          assertPermanentCount(playerA, "Zombie Token", 1);
81          assertPowerToughness(playerA, "Zombie Token", 0, 13);
82  
83          Permanent treeToken = getPermanent("Tree of Perdition", playerA);
84          Assert.assertTrue(treeToken.getAbilities().contains(FlyingAbility.getInstance()));
85          Assert.assertTrue(treeToken.hasSubtype(SubType.SPIRIT, currentGame));
86          Assert.assertTrue(treeToken.getAbilities().contains(DefenderAbility.getInstance()));
87  
88          assertLife(playerA, 20);
89          assertLife(playerB, 1);
90          assertPowerToughness(playerA, "Tree of Perdition", 1, 20);
91      }

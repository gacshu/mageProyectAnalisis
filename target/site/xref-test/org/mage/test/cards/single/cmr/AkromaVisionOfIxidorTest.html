13  
14      @Test
15      public void test_MustBoostCreatures() {
16          // https://github.com/magefree/mage/issues/7209
17  
18          // At the beginning of each combat, until end of turn, each other creature you control gets +1/+1 if it has flying, +1/+1 if it has first strike, and so on
19          addCard(Zone.BATTLEFIELD, playerA, "Akroma, Vision of Ixidor", 1);
20          //
21          // Flying
22          addCard(Zone.BATTLEFIELD, playerA, "Shalai, Voice of Plenty", 1); // 3/4
23          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 1); // 2/2
24  
25          checkPT("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akroma, Vision of Ixidor", 6, 6);
26          checkPT("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Shalai, Voice of Plenty", 3, 4);
27          checkPT("before", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Balduvian Bears", 2, 2);
28  
29          // on combat it must get boost for flying
30          waitStackResolved(1, PhaseStep.BEGIN_COMBAT);
31          checkPT("no change for itself", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akroma, Vision of Ixidor", 6, 6);
32          checkPT("boost fly", 1, PhaseStep.BEGIN_COMBAT, playerA, "Shalai, Voice of Plenty", 3 + 1, 4 + 1);
33          checkPT("no change for opponent", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Balduvian Bears", 2, 2);
34  
35          // end boost
36          checkPT("end boost on end turn", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Shalai, Voice of Plenty", 3, 4);
37  
38          setStrictChooseMode(true);
39          setStopAt(2, PhaseStep.END_TURN);
40          execute();
41      }
42  }

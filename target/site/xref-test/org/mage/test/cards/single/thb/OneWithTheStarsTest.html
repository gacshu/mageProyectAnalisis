17  public class OneWithTheStarsTest extends CardTestPlayerBase {
18  
19      private static final String stars = "One with the Stars";
20      private static final String knight = "Dragonsoul Knight";
21      private static final String brute = "Gingerbrute";
22      private static final String shrine = "Honden of Cleansing Fire";
23      private static final String blossom = "Bitterblossom";
24  
25      private void makeRainbowLand(int count) {
26          for (int i = 0; i < count; i++) {
27              addCustomCardWithAbility(
28                      "Rainbow", playerA, new AnyColorManaAbility(),
29                      null, CardType.LAND, "", Zone.BATTLEFIELD
30              );
31          }
32      }
33  
34      @Test
35      public void testDragonsoulKnight() {
36          addCard(Zone.HAND, playerA, stars);
37          addCard(Zone.BATTLEFIELD, playerA, knight);
38          makeRainbowLand(9);
39  
40          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, stars, knight);
41  
42          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{W}{U}");
43  
44          setStopAt(1, PhaseStep.END_TURN);
45          execute();
46  
47          assertType(knight, CardType.ENCHANTMENT, true);
48          assertType(knight, CardType.CREATURE, false);
49          assertNotSubtype(knight, SubType.HUMAN);
50          assertNotSubtype(knight, SubType.KNIGHT);
51          assertNotSubtype(knight, SubType.DRAGON);
52          assertAbility(playerA, knight, FlyingAbility.getInstance(), true);
53          assertAbility(playerA, knight, TrampleAbility.getInstance(), true);
54      }
55  
56      @Test
57      public void testGingerbrute() {

59          addCard(Zone.BATTLEFIELD, playerA, brute);

64          setStopAt(1, PhaseStep.END_TURN);
65          execute();
66

68          assertType(brute, CardType.ARTIFACT, false);
69          assertType(brute, CardType.CREATURE, false);

71          assertNotSubtype(brute, SubType.FOOD);
72      }

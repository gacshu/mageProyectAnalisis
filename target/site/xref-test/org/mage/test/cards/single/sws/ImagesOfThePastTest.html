16  
17          // Images of the Past
18          // Return up to two target creature cards from your graveyard to the battlefield, then exile those creatures.
19          addCard(Zone.HAND, playerA, "Images of the Past"); // {G}{W}
20          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
21          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
22          //
23          addCard(Zone.GRAVEYARD, playerA, "Balduvian Bears");
24          addCard(Zone.GRAVEYARD, playerA, "Silvercoat Lion");
25          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt");
26          //
27          // Healer of the Pride
28          // Whenever another creature enters the battlefield under your control, you gain 2 life.
29          addCard(Zone.BATTLEFIELD, playerA, "Healer of the Pride", 1);
30          //
31          // Syr Konrad, the Grim
32          // Whenever another creature dies, or a creature card is put into a graveyard from anywhere other than
33          // the battlefield, or a creature card leaves your graveyard, Syr Konrad, the Grim deals 1 damage to each opponent.
34          addCard(Zone.BATTLEFIELD, playerA, "Syr Konrad, the Grim", 1);
35  
36  
37          // activate and rise 2x dies triggers (adds 2x Spirit)
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Images of the Past", "Balduvian Bears^Silvercoat Lion");
39          setChoice(playerA, "Whenever another creature dies", 2); // 2x triggers from Syr
40          setChoice(playerA, "Whenever another creature enters"); // 2x triggers from Healer
41          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
42          //
43          checkExileCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
44          checkExileCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Silvercoat Lion", 1);
45          checkExileCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 0);
46          checkGraveyardCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 0);
47          checkGraveyardCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Silvercoat Lion", 0);
48          checkGraveyardCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 1);
49          //
50          // trigger result to check enter battlefield (4x life from Healer)
51          checkLife("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 20 + 4);
52          // trigger result to check grave leave (2x damage from Syr)
53          checkLife("after", 1, PhaseStep.PRECOMBAT_MAIN, playerB, 20 - 2);
54  
55          setStrictChooseMode(true);
56          setStopAt(1, PhaseStep.END_TURN);
57          execute();
58      }

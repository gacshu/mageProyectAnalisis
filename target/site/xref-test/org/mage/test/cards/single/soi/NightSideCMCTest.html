3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * New SOI mechanics change requires the night-side CMC to be equal to the
10   * day-side CMC
11   *
12   * @author escplan9
13   */
14  public class NightSideCMCTest extends CardTestPlayerBase {
15      
16      // Delver of Secrets {U} 1/1
17      // At the beginning of your upkeep, look at the top card of your library. You may reveal that card. If an instant or sorcery card is revealed this way, transform Delver of Secrets.
18      private final String delver = "Delver of Secrets";
19      // Night-side of Delver, 3/2 Flying should have CMC 1 as well
20      private final String insect = "Insectile Aberration";
21  
22      /**
23       * Repeal for X = 1 on Delver flipped "Insectile Aberration"
24       */
25      @Test
26      public void insectileAbberationRepealXis1Test() {
27              
28          skipInitShuffling();
29          addCard(Zone.BATTLEFIELD, playerA, delver);
30          addCard(Zone.LIBRARY, playerA, "Lightning Bolt"); // to transform Delver of Secrets
31          // Instant - {X}{U}
32          // Return target nonland permanent with converted mana cost X to its owner's hand. Draw a card.
33          addCard(Zone.HAND, playerB, "Repeal");
34          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Repeal");
37          setChoice(playerB, "X=1");
38          // Insectile Aberration is auto-chosen since only target
39  
40          setStopAt(1, PhaseStep.BEGIN_COMBAT);
41          execute();
42  
43          assertGraveyardCount(playerB, "Repeal", 1);
44          assertPermanentCount(playerA, insect, 0);
45          assertHandCount(playerA, delver, 1); // day-side of Insectile Abberation returned to hand
46      }
47      
48      /**
49       * Engineered Explosives for Sunburst = 1 on Delver flipped "Insectile Aberration"
50       */
51      @Test
52      public void insectileAbberationEngeeredExplosivesSunburstIs1Test() {
53          
54          /*
55          Engineered Explosives {X}
56         Artifact
57         Sunburst (This enters the battlefield with a charge counter on it for each color of mana spent to cast it.)
58          {2}, Sacrifice Engineered Explosives: Destroy each nonland permanent with converted mana cost equal to the number of charge counters on Engineered Explosives.
59          */
60          String explosives = "Engineered Explosives";
61          
62          skipInitShuffling();
63          addCard(Zone.BATTLEFIELD, playerA, delver);
64          addCard(Zone.LIBRARY, playerA, "Lightning Bolt"); // to transform Delver of Secrets
65          
66          addCard(Zone.HAND, playerB, explosives);
67          addCard(Zone.BATTLEFIELD, playerB, "Island", 3);
68  
69          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, explosives);
70          setChoice(playerB, "X=1");
71          activateAbility(2, PhaseStep.BEGIN_COMBAT, playerB, "{2}"); // activate explosives
72  
73          setStopAt(2, PhaseStep.END_COMBAT);
74          execute();
75  
76          assertGraveyardCount(playerB, explosives, 1);
77          assertPermanentCount(playerA, insect, 0);
78          assertGraveyardCount(playerA, delver, 1);
79      }
80  }

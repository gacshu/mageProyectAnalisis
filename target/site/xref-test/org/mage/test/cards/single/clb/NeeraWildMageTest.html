12  public class NeeraWildMageTest extends CardTestPlayerBase {
13  
14      @Test
15      public void TestNeeraWildMage() {
16          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
17  
18          // Whenever you cast a spell, you may put it on the bottom of its owner's library.
19          // If you do, reveal cards from the top of your library until you reveal a nonland card.
20          // You may cast that card without paying its mana cost.
21          // Then put the rest on the bottom of your library in a random order.
22          // This ability triggers only once each turn.
23          addCard(Zone.BATTLEFIELD, playerA, "Neera, Wild Mage", 1);
24  
25          addCard(Zone.HAND, playerA, "Consider");
26  
27          playerA.getLibrary().clear();
28          addCard(Zone.LIBRARY, playerA, "Mental Note");
29          addCard(Zone.LIBRARY, playerA, "Island");
30          addCard(Zone.LIBRARY, playerA, "Island");
31          skipInitShuffling();
32  
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Consider");
34          // Choose to use Neera's ability
35          setChoice(playerA, true);
36          // Do not cast Mental Note revealed with Neera's ability
37          setChoice(playerA, false);
38  
39          setStopAt(3, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          // Consider was not cast so should be on top of the deck after resolving Neera's ability
43          // On turn 3 it will be drawn by player A
44          assertHandCount(playerA, 1);
45          assertHandCount(playerA, "Consider", 1);
46          assertLibraryCount(playerA, 3);
47          assertLibraryCount(playerA, "Mental Note", 1);
48          assertLibraryCount(playerA, "Island", 2);
49      }
50  }

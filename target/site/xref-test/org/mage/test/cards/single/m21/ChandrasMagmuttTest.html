12      public void pingPlayer(){

14          addCard(Zone.BATTLEFIELD, playerA, "Chandra's Magmutt");
15  
16          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: {this} deals", playerB);
17  
18          setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
19          setStrictChooseMode(true);
20          execute();

26      public void pingPlanesWalker(){

28          addCard(Zone.BATTLEFIELD, playerA, "Chandra's Magmutt");
29          // +1: Put a +1/+1 counter on up to one target creature. It gains indestructible until end of turn.
30          // −2: Whenever one or more nontoken creatures attack this turn, create that many 1/1 white Soldier creature tokens that are tapped and attacking.
31          // −6: You get an emblem with "At the beginning of combat on your turn, create a 1/1 white Soldier creature token, then put a +1/+1 counter on each creature you control."        
32          addCard(Zone.BATTLEFIELD, playerB, "Basri Ket");
33  
34          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: {this} deals", "Basri Ket");
35          setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);

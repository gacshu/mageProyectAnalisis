3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * Defender
10   *
11   * Thing in the Ice enters the battlefield with four ice counters on it.
12   *
13   * Whenever you cast an instant or sorcery spell, remove an ice counter from
14   * Thing in the Ice. Then if it has no ice counters on it, transform it. When
15   * this creature transforms into Awoken Horror, return all non-Horror creatures
16   * to their owners' hands.
17   *
18   * @author escplan9 (Derek Monturo - dmontur1 at gmail dot com)
19   */
20  public class ThingInTheIceTest extends CardTestPlayerBase {
21  
22      /**
23       * Reported bug: When Thing in the Ice transforms, it bounces Clue tokens.
24       *
25       */
26      @Test
27      public void testClueTokens() {
28          // Whenever a land enters the battlefield under your control, investigate. <i>(Create a colorless Clue artifact token onto the battlefield with "{2}, Sacrifice this artifact: Draw a card.")</i>
29          // Whenever you sacrifice a Clue, put a +1/+1 counter on Tireless Tracker.
30          addCard(Zone.BATTLEFIELD, playerA, "Tireless Tracker", 1); // Human, Scout 3/2
31          addCard(Zone.HAND, playerA, "Forest", 1);
32          // Defender
33          // Thing in the Ice enters the battlefield with four ice counters on it.
34          // Whenever you cast an instant or sorcery spell, remove an ice counter from Thing in the Ice. Then if it has no ice counters on it, transform it.
35          // When this creature transforms into Awoken Horrow, return all non-Horror creatures to their owners' hands.
36          addCard(Zone.BATTLEFIELD, playerB, "Thing in the Ice", 1);
37          // Target creature gains haste until end of turn.
38          // Draw a card.
39          addCard(Zone.HAND, playerB, "Expedite", 4);
40          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 4);
41  
42          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Forest"); // creates a clue
43  
44          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Expedite");
45          addTarget(playerB, "Thing in the Ice");
46          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Expedite");
47          addTarget(playerB, "Thing in the Ice");
48          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Expedite");
49          addTarget(playerB, "Thing in the Ice");
50          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Expedite");
51          addTarget(playerB, "Thing in the Ice"); // remove all 4 ice counters to transform it
52  
53          setStopAt(2, PhaseStep.BEGIN_COMBAT);
54          execute();
55  
56          assertPermanentCount(playerA, "Clue Token", 1);
57          assertHandCount(playerA, "Tireless Tracker", 1); // returned to hand
58          assertPermanentCount(playerA, "Tireless Tracker", 0);
59          assertPermanentCount(playerB, "Awoken Horror", 1); // transformed
60          assertGraveyardCount(playerB, "Expedite", 4);
61      }
62  }

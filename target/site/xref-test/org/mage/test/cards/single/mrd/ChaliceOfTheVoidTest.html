13      /**
14       * Theres a Chalice of the Void with 1 counter in play under my control.
15       * Then I cast second chalice with x=1. For spells on the stack the cmc is
16       * the base CMC + X value * {X} in casting costs on top right of card. So
17       * cmc should be 2 in this case, it shouldnt be countered.
18       * http://boardgames.stackexchange.com/questions/7327/what-is-the-converted-mana-cost-of-a-spell-with-x-when-cast-with-the-miracle-m
19       */
20      @Test
21      public void testX1CountsFor2CMC() {
22          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
23          // Chalice of the Void enters the battlefield with X charge counters on it.
24          // Whenever a player casts a spell with converted mana cost equal to the number of charge counters on Chalice of the Void, counter that spell.
25          addCard(Zone.HAND, playerA, "Chalice of the Void", 2);
26  
27          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void", true);
28          setChoice(playerA, "X=1");
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void");
30          setChoice(playerA, "X=1");
31  
32          setStrictChooseMode(true);
33          setStopAt(1, PhaseStep.BEGIN_COMBAT);
34          execute();
35  
36          assertPermanentCount(playerA, "Chalice of the Void", 2);
37      }

39      /*
40      If X=1 the cmc of Chalice on the stack is 2. So it can't be countered by Mental Misstep
41       */
42      @Test
43      public void testCantBeCounteredByMentalMisstep() {
44          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
45          addCard(Zone.HAND, playerA, "Chalice of the Void", 1);
46  
47          addCard(Zone.BATTLEFIELD, playerB, "Island", 1);
48          // Counter target spell with converted mana cost 1.
49          addCard(Zone.HAND, playerB, "Mental Misstep", 1);
50  
51          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void");
52          setChoice(playerA, "X=1");
53          checkPlayableAbility("can't Mental Misstep", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Mental", false);
54  //        castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Mental Misstep", "Chalice of the Void", "Chalice of the Void");
55  
56          setStrictChooseMode(true);
57          setStopAt(1, PhaseStep.BEGIN_COMBAT);
58          execute();
59          // assertAllCommandsUsed(); // cast Mental Misstep must be ignored
60  
61          assertHandCount(playerB, "Mental Misstep", 1); // cannot be cast because no legal target exists
62          assertPermanentCount(playerA, "Chalice of the Void", 1); // was not countered
63  
64      }

66      /*
67          Conflagurate flashed back for X >= 0 should not be countered by chalice.
68       */
69      @Test
70      public void testConflagrateFlashback() {
71          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
72          addCard(Zone.HAND, playerA, "Chalice of the Void", 1);
73  
74          // Conflagrate deals X damage divided as you choose among any number of target creatures and/or players.
75          // Flashback-{R}{R}, Discard X cards.
76          addCard(Zone.GRAVEYARD, playerB, "Conflagrate", 1);
77          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 2);
78          addCard(Zone.HAND, playerB, "Mountain", 1);
79  
80          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void");
81          setChoice(playerA, "X=1");
82  
83          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Flashback {R}{R}");
84          setChoice(playerB, "X=1");
85          addTargetAmount(playerB, playerA, 1);
86          setChoice(playerB, "Mountain"); // discard 1 card
87  
88          setStrictChooseMode(true);
89          setStopAt(2, PhaseStep.BEGIN_COMBAT);
90          execute();
91  
92          assertExileCount(playerB, "Conflagrate", 1);
93          //TODO: Apparently there are two mountains in the graveyard at the end of the test now.
94          assertGraveyardCount(playerB, "Mountain", 1);
95  
96          assertLife(playerA, 19);
97          assertLife(playerB, 20);
98      }

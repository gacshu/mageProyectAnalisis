4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   * {@link mage.cards.i.InfernoTrap Inferno Trap}
11   * {3}{R}
12   * Instant — Trap
13   *
14   * If you’ve been dealt damage by two or more creatures this turn, you may pay {R} rather than pay this spell’s mana cost.
15   * Inferno Trap deals 4 damage to target creature.
16   *
17   * @author LevelX2
18   */
19  public class InfernoTrapTest extends CardTestPlayerBase {
20  
21      @Test
22      public void testTwoDamageStepsCountOnlyAsOneCreature() {
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
24          // Instant {3}{R}
25          addCard(Zone.HAND, playerA, "Inferno Trap");
26  
27          // Flying, double strike 1/1
28          addCard(Zone.BATTLEFIELD, playerB, "Skyhunter Skirmisher");
29  
30          attack(2, playerB, "Skyhunter Skirmisher");
31  
32          checkPlayableAbility("Inferno Trap not reduced", 2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Cast Inferno", false);
33  
34          setStopAt(2, PhaseStep.END_TURN);
35          execute();
36  
37          assertLife(playerA, 18);
38          assertLife(playerB, 20);
39  
40          assertGraveyardCount(playerA, "Inferno Trap", 0);
41          assertGraveyardCount(playerB, "Skyhunter Skirmisher", 0);
42      }
43  
44      @Test
45      public void testPlayByAlternateCost() {
46          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
47          // If you've been dealt damage by two or more creatures this turn, you may pay {R} rather than pay Inferno Trap's mana cost.
48          // Inferno Trap deals 4 damage to target creature.
49          addCard(Zone.HAND, playerA, "Inferno Trap"); // Instant {3}{R}
50  
51          // Flying, double strike
52          addCard(Zone.BATTLEFIELD, playerB, "Skyhunter Skirmisher"); // 1/1
53          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion"); // 2/2
54  
55          attack(2, playerB, "Skyhunter Skirmisher");
56          attack(2, playerB, "Silvercoat Lion");
57  
58          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Inferno Trap", "Skyhunter Skirmisher");
59  
60          setStopAt(2, PhaseStep.END_TURN);
61          execute();
62  
63          assertLife(playerA, 16);
64          assertLife(playerB, 20);
65  
66          assertGraveyardCount(playerA, "Inferno Trap", 1);
67          assertGraveyardCount(playerB, "Skyhunter Skirmisher", 1);
68      }

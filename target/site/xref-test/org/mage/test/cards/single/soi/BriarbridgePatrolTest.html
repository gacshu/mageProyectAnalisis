10  /**
11   * 3G Creature - Human Warrior Whenever Briarbridge Patrol deals damage to one
12   * or more creatures, investigate. (Put a colorless Clue artifact token onto the
13   * battlefield with "2, Sacrifice this artifact: Draw a card.")
14   *
15   * At the beginning of each end step, if you sacrificed three or more clues this
16   * turn, you may put a creature card from your hand onto the battlefield.
17   *
18   * @author escplan9 (Derek Monturo - dmontur1 at gmail dot com)
19   */
20  public class BriarbridgePatrolTest extends CardTestPlayerBase {
21  
22      /**
23       * Deals damage to creature test.
24       */
25      @Test
26      public void dealtDamageToCreatureInvestigate() {
27  
28          addCard(Zone.BATTLEFIELD, playerA, "Briarbridge Patrol", 1);
29          addCard(Zone.BATTLEFIELD, playerB, "Elite Vanguard", 1);
30  
31          attack(1, playerA, "Briarbridge Patrol");
32          block(1, playerB, "Elite Vanguard", "Briarbridge Patrol");
33  
34          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
35          execute();
36  
37          assertPermanentCount(playerA, "Clue Token", 1);
38          assertPermanentCount(playerA, "Briarbridge Patrol", 1);
39          assertPermanentCount(playerB, "Elite Vanguard", 0);
40          assertGraveyardCount(playerB, "Elite Vanguard", 1);
41      }
42  
43      /**
44       * Reported bug: Briarbridge Patrol investigate doesn't trigger from
45       * noncombat damage like Rabid Bite
46       */
47      @Test
48      public void dealtNonCombatDamageToCreatureInvestigate() {
49          // Whenever Briarbridge Patrol deals damage to one or more creatures, investigate (Create a colorless Clue artifact token onto the battlefield with "2, Sacrifice this artifact: Draw a card.").
50          // At the beginning of each end step, if you sacrificed three or more Clues this turn, you may put a creature card from your hand onto the battlefield.
51          addCard(Zone.BATTLEFIELD, playerA, "Briarbridge Patrol", 1);
52          addCard(Zone.BATTLEFIELD, playerB, "Wall of Roots", 1);
53          // {1}{G} Sorcery Target creature you control deals damage equal to its power to target creature you don't control.
54          addCard(Zone.HAND, playerA, "Rabid Bite");
55          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
56  
57          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Rabid Bite", "Briarbridge Patrol");
58          // Wall of Roots is auto-chosen since only target
59          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
60          execute();
61  
62          assertPermanentCount(playerA, "Clue Token", 1);
63          assertPermanentCount(playerA, "Briarbridge Patrol", 1);
64          assertPermanentCount(playerB, "Wall of Roots", 1);
65  
66          Permanent wall = getPermanent("Wall of Roots", playerB);
67          Assert.assertEquals("Wall of Roots should have 3 damage to it", 3, wall.getDamage());
68      }
69  }

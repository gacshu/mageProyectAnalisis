1   package org.mage.test.cards.single.mor;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;
7   
8   /**
9    * {@link mage.cards.s.SharedAnimosity Shared Animosity}
10   * {2}{R}
11   * Enchantment
12   * Whenever a creature you control attacks, it gets +1/+0 until end of turn for each other attacking creature that shares a creature type with it.
13   *
14   * @author Alex-Vasile
15   */
16  public class SharedAnimosityTest extends CardTestPlayerBase {
17  
18      private static final String sharedAnimosity = "Shared Animosity";
19  
20      /**
21       * Reported bug: https://github.com/magefree/mage/issues/9428
22       *      "Shared Animosity Giving a +1/+0 to a single attacking creature,
23       *      Silvar, Devourer of the Free attacked alone and got the +1".
24       */
25      @Test
26      public void attackingAlone() {
27          // 4/2
28          String silvar = "Silvar, Devourer of the Free";
29          addCard(Zone.BATTLEFIELD, playerA, sharedAnimosity);
30          addCard(Zone.BATTLEFIELD, playerA, silvar);
31  
32          attack(1, playerA, silvar);
33  
34          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
35          execute();
36          assertPowerToughness(playerA, silvar, 4, 2); // Should not have gotten a boost
37          assertLife(playerB, 20 - 4);
38      }
39  
40      /**
41       * Attack with 3 creatures.
42       * One shares types with both of the others.
43       * The other two only share a type with the 3rd, but not with each other.
44       */
45      @Test
46      public void attackingWithOthers() {
47          // 4/2
48          // Cat Nightmare
49          String silvar = "Silvar, Devourer of the Free";
50          // 2/2
51          // Cat Soldier
52          String ajanisPridemate = "Ajani's Pridemate";
53          // 1/5
54          // Gloom Pangolin
55          String gloomPangolin = "Gloom Pangolin";
56  
57          addCard(Zone.BATTLEFIELD, playerA, sharedAnimosity);
58          addCard(Zone.BATTLEFIELD, playerA, ajanisPridemate);
59          addCard(Zone.BATTLEFIELD, playerA, gloomPangolin);
60          addCard(Zone.BATTLEFIELD, playerA, silvar);
61  
62          attack(1, playerA, silvar);
63          attack(1, playerA, ajanisPridemate);
64          attack(1, playerA, gloomPangolin);
65  
66          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
67          execute();
68          assertPowerToughness(playerA, silvar, 4+2, 2); // Shares with both the pridemate and the pangolin
69          assertPowerToughness(playerA, ajanisPridemate, 2+1, 2); // Shares only with Silvar
70          assertPowerToughness(playerA, gloomPangolin, 1+1, 5); // Shares only with Silvar
71          assertLife(playerB, 20 - (4+2) - (2+1) - (1+1));
72      }
73  }

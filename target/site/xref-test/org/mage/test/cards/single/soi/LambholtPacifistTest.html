3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Assert;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   * Lambholt Pacifist {1}{G}
11   * 3/3
12   * Creature — Human Shaman Werewolf
13   * Lambholt Pacifist can’t attack unless you control a creature with power 4 or greater.
14   * At the beginning of each upkeep, if no spells were cast last turn, transform Lambholt Pacifist.
15   *
16   * @author escplan9
17   */
18  public class LambholtPacifistTest extends CardTestPlayerBase {
19  
20      private final String lambholt = "Lambholt Pacifist";
21  
22      /**
23       * An uncrewed vehicle does not count as a creature.
24       */
25      @Test
26      public void uncrewedVehicle_LambholtCannotAttack() {
27  
28          /*
29           Heart of Kiran {2}
30           4/4
31           Legendary Artifact - Vehicle
32           Flying, vigilance
33           Crew 3 (Tap any number of creatures you control with total power 3 or more:
34                   This Vehicle becomes an artifact creature until end of turn.)
35           You may remove a loyalty counter from a planeswalker you control rather than pay Heart of Kiran's crew cost.
36           */
37          String heartKiran = "Heart of Kiran";
38          // Just needs something to cast to prevent lambholt transforming
39          String orni = "Ornithopter"; // {0} 0/2 flyer
40          
41          addCard(Zone.HAND, playerA, lambholt);
42          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
43          addCard(Zone.BATTLEFIELD, playerA, heartKiran);
44          addCard(Zone.HAND, playerB, orni);
45          
46          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, lambholt);
47          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, orni);
48  
49          attack(3, playerA, lambholt);
50          
51          setStopAt(3, PhaseStep.END_COMBAT);
52  
53          try {
54              execute();
55          } catch (Throwable e) {
56              if (!e.getMessage().contains("Player PlayerA must have 0 actions but found 1")) {
57                  Assert.fail("Should have had error about not being able to attack, but got:\n" + e.getMessage());
58              }
59          }
60          
61          assertPermanentCount(playerB, orni, 1);
62          assertTapped(lambholt, false);
63          assertLife(playerB, 20);        
64      }
65  }

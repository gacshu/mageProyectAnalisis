15  public class NimDeathmantleTest extends CardTestPlayerBase {
16  
17      @Test
18      public void test_Basic() {
19          setStrictChooseMode(true);
20          
21          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
22          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
23          // Equipped creature gets +2/+2, has intimidate, and is a black Zombie.
24          // Whenever a nontoken creature is put into your graveyard from the battlefield, you may pay {4}. If you do, return that card to the battlefield and attach Nim Deathmantle to it.        
25          addCard(Zone.HAND, playerA, "Nim Deathmantle"); // Artifact Equipment {2}
26  
27          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
28          addCard(Zone.HAND, playerB, "Lightning Bolt");        
29  
30          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nim Deathmantle");
31          
32          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
33          
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Silvercoat Lion");
35          setChoice(playerA, true); // Message: Nim Deathmantle - Pay {4}?
36          
37          setStopAt(1, PhaseStep.BEGIN_COMBAT);
38          execute();
39  
40          assertGraveyardCount(playerB, "Lightning Bolt",1);
41          
42          assertPermanentCount(playerA, "Nim Deathmantle", 1);
43          
44          assertPowerToughness(playerA, "Silvercoat Lion", 4, 4);
45          assertAbility(playerA, "Silvercoat Lion", IntimidateAbility.getInstance(), true);
46          assertType("Silvercoat Lion", CardType.CREATURE, SubType.ZOMBIE);        
47  
48      }
49  }

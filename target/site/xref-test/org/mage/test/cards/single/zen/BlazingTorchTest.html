11  public class BlazingTorchTest extends CardTestPlayerBase {
12  
13      private static final String plains = "Plains";
14      private static final String torch = "Blazing Torch";
15      private static final String lion = "Silvercoat Lion";
16      private static final String masterwork = "Masterwork of Ingenuity";
17      private static final String halo = "Runed Halo";
18  
19      @Test
20      public void testCard() {
21          addCard(Zone.BATTLEFIELD, playerA, plains);
22          addCard(Zone.BATTLEFIELD, playerA, lion);
23          addCard(Zone.BATTLEFIELD, playerA, torch);
24  
25          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip", lion);
26          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T},", playerB);
27  
28          setStrictChooseMode(true);
29          setStopAt(1, PhaseStep.END_TURN);
30          execute();
31  
32          assertLife(playerA, 20);
33          assertLife(playerB, 18);
34          assertPermanentCount(playerA, lion, 1);
35          assertPermanentCount(playerA, torch, 0);
36          assertGraveyardCount(playerA, torch, 1);
37      }
38  
39      @Test
40      public void testCardCopied() {
41          addCard(Zone.BATTLEFIELD, playerA, plains, 2);
42          addCard(Zone.BATTLEFIELD, playerA, lion);
43          // You may have Masterwork of Ingenuity enter the battlefield as a copy of any Equipment on the battlefield.
44          addCard(Zone.HAND, playerA, masterwork);
45          addCard(Zone.BATTLEFIELD, playerB, torch);
46  
47          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, masterwork, true);
48          setChoice(playerA, true); // use copy
49          setChoice(playerA, torch);
50  
51          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip", lion);
52          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
53          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T},", playerB);
54  
55          setStrictChooseMode(true);
56          setStopAt(1, PhaseStep.END_TURN);
57          execute();
58  
59          assertLife(playerA, 20);
60          assertLife(playerB, 18);
61          assertPermanentCount(playerA, lion, 1);
62          assertPermanentCount(playerA, torch, 0);
63          assertPermanentCount(playerB, torch, 1);
64          assertGraveyardCount(playerA, masterwork, 1);
65          assertGraveyardCount(playerB, torch, 0);
66      }
67  
68      @Test
69      public void testRunedHalo() {
70          addCard(Zone.BATTLEFIELD, playerA, plains);
71          addCard(Zone.BATTLEFIELD, playerA, lion);
72          addCard(Zone.BATTLEFIELD, playerA, torch);
73          addCard(Zone.BATTLEFIELD, playerB, plains, 2);
74          addCard(Zone.HAND, playerB, halo);
75  
76          setChoice(playerB, torch);
77          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, halo);
78  
79          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip", lion);
80          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T},", playerB);
81  
82          setStrictChooseMode(true);
83          setStopAt(3, PhaseStep.END_TURN);
84          execute();
85  
86          assertLife(playerA, 20);
87          assertLife(playerB, 20); // Ability resolves as it's from the equipped creature but damage is from torch and is prevented
88          assertPermanentCount(playerA, lion, 1);
89          assertPermanentCount(playerA, torch, 0);
90          assertGraveyardCount(playerA, torch, 1);
91      }

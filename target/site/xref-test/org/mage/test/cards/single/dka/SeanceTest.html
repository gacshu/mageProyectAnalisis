15       * Seance    Enchantment  {2}{W}{W}
16       * 
17       * At the beginning of each upkeep, you may exile target creature card from your graveyard. 
18       * If you do, put a token onto the battlefield that's a copy of that card except it's a 
19       * Spirit in addition to its other types. Exile it at the beginning of the next end step.
20       */
21      @Test
22      public void testCard() {
23          addCard(Zone.BATTLEFIELD, playerA, "Seance");
24          addCard(Zone.GRAVEYARD, playerA, "Craw Wurm");
25  
26          setStopAt(1, PhaseStep.DRAW);
27          execute();
28  
29          assertLife(playerA, 20);
30          assertLife(playerB, 20);

33      }

16      /*
17          Hapatra, Vizier of Poisons BG
18          Legendary Creature - Human Cleric 2/2
19          Whenever Hapatra, Vizier of Poisons deals combat damage to a player, you may put a -1/-1 counter on target creature.
20          Whenever you put one or more -1/-1 counters on a creature, create a 1/1 green Snake creature token with deathtouch.
21       */
22      private final String hapatra = "Hapatra, Vizier of Poisons";
23  
24      @Test
25      public void hapatraCombatDamageToPlayer() {
26  
27          String gBears = "Grizzly Bears";
28          addCard(Zone.BATTLEFIELD, playerA, hapatra);
29          addCard(Zone.BATTLEFIELD, playerB, gBears);
30  
31          attack(3, playerA, hapatra);
32          setChoice(playerA, true); // opt to place -1/-1 counter on creature
33          addTarget(playerA, gBears);
34  
35          setStopAt(3, PhaseStep.END_COMBAT);
36          execute();
37  
38          assertLife(playerB, 18);
39          assertCounterCount(playerB, gBears, CounterType.M1M1, 1);
40          assertPowerToughness(playerB, gBears, 1, 1); // 2/2 with -1/1 counter
41          assertPermanentCount(playerA, "Snake Token", 1);
42          assertAbility(playerA, "Snake Token", DeathtouchAbility.getInstance(), true);
43      }

45      /*
46      Issue #3288.
47      Hapatra, Vizier of Poisons is not triggering off of Infect. Tested with Blight Mamba
48       */
49      @Test
50      public void infectDamageTriggersHapatra() {
51  
52          String bMamba = "Blight Mamba"; // {1}{G} 1/1 Creature - Snake, Infect with {1}{G}:Regen
53          String wOmens = "Wall of Omens"; // {1}{W} 0/4 defender ETB: draw a card
54  
55          addCard(Zone.BATTLEFIELD, playerA, hapatra);
56          addCard(Zone.BATTLEFIELD, playerA, bMamba);
57          addCard(Zone.BATTLEFIELD, playerB, wOmens);
58  
59          attack(3, playerA, bMamba);
60          block(3, playerB, wOmens, bMamba);
61  
62          setStopAt(3, PhaseStep.END_COMBAT);
63          execute();
64  
65          assertLife(playerB, 20);
66          assertCounterCount(playerB, CounterType.POISON, 0);
67          assertPowerToughness(playerB, wOmens, -1, 3); // 0/4 with -1/-1 counter
68          assertCounterCount(playerB, wOmens, CounterType.M1M1, 1);
69          assertPermanentCount(playerA, "Snake Token", 1);
70          assertAbility(playerA, "Snake Token", DeathtouchAbility.getInstance(), true);
71      }

102     /**
103      * Testing fix for issue #5886
104      * Tokens with wither/infect that deal damage were not triggering Hapatra's snake creating ability
105      * @author jgray1206
106      */
107     @Test
108     public void testTokensWithInfectTriggerHapatra() {
109         String concordantCrossroads = "Concordant Crossroads"; //All creatures have haste
110         String krakenHatchling = "Kraken Hatchling"; //Arbitrary creature to defend
111         String triumphOfTheHordes = "Triumph of the Hordes"; //Creatures you control gain infect
112         String sprout = "Sprout"; //Create a 1/1 Saproling creature token
113         
114         addCard(Zone.HAND, playerA, sprout, 1);
115         addCard(Zone.HAND, playerA, triumphOfTheHordes, 1);
116 
117         addCard(Zone.BATTLEFIELD, playerA, "Forest", 6);
118         addCard(Zone.BATTLEFIELD, playerA, hapatra, 1);
119         addCard(Zone.BATTLEFIELD, playerA, concordantCrossroads, 1);
120         addCard(Zone.BATTLEFIELD, playerB, krakenHatchling, 1);
121 
122         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, sprout, true);
123         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, triumphOfTheHordes);
124 
125         attack(1, playerA, "Saproling Token");
126         block(1, playerB, krakenHatchling, "Saproling Token");
127         setStopAt(1, PhaseStep.END_COMBAT);
128         setStrictChooseMode(true);
129         execute();
130 
131         assertPowerToughness(playerB, krakenHatchling, -2, 2);
132         assertCounterCount(playerB, krakenHatchling, CounterType.M1M1, 2);
133         assertPermanentCount(playerA, "Snake Token", 1); //Should have triggered when Saproling added -1/-1 counter
134     }

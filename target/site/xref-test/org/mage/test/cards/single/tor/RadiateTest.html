14      @Test
15      public void test_Play_Manual() {
16          // Choose target instant or sorcery spell that targets only a single permanent or player. Copy that spell
17          // for each other permanent or player the spell could target. Each copy targets a different one of those
18          // permanents and players.
19          addCard(Zone.HAND, playerA, "Radiate", 6); // {3}{R}{R}
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
21          //
22          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
24          //
25          addCard(Zone.BATTLEFIELD, playerA, "Grizzly Bears", 2);
26          addCard(Zone.BATTLEFIELD, playerB, "Kitesail Corsair", 2);
27  
28          // cast bolt and copy spell for each another target
29          setChoice(playerA, TestPlayer.CHOICE_SKIP); // skip stack order
30          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Radiate", "Lightning Bolt", "Lightning Bolt");
32          checkStackSize("before radiate", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
33          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
34          // must have: 2x for corsairs, 2x for bears, 1x for A
35          checkStackSize("after radiate", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1 + 5);
36  
37          setStrictChooseMode(true);
38          setStopAt(1, PhaseStep.END_TURN);
39          execute();
40  
41          assertPermanentCount(playerA, 6); // 6 lands
42          assertPermanentCount(playerB, 0);
43          assertLife(playerA, 20 - 3);
44          assertLife(playerB, 20 - 3);
45      }

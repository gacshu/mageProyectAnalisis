8   /**
9    * {@link mage.cards.t.TreeOfPerdition Tree of Perdition}
10   * {3}{B}
11   * Creature — Plant
12   * Defender
13   * P/T 0/13
14   * {T}: Exchange target opponent’s life total with Tree of Perdition’s toughness.
15   *
16   * @author escplan9 (Derek Monturo - dmontur1 at gmail dot com)
17   */
18  public class TreeOfPerditionTest extends CardTestPlayerBase {
19      
20      /**
21       * Reported bug: https://github.com/magefree/mage/issues/2101
22       *      Tree of Perdition retains toughness change after being bounced and replayed
23       *      Exchanged toughness with opponent's life to become an 0/20, got bounced,
24       *      and when replayed it was still 0/20. It should be a new object and enter as an 0/13.
25      */
26      @Test
27      public void testTreeOfPerditionBouncedAndReplayed() {
28          addCard(Zone.BATTLEFIELD, playerA, "Tree of Perdition");
29          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
30          addCard(Zone.HAND, playerB, "Unsummon", 1); // {U} instant - return target creature to owner hand
31          addCard(Zone.BATTLEFIELD,playerB,"Island",1);
32  
33          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Exchange");
34          // Player B is auto-targeted since they're the only option
35  
36          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Unsummon");
37  
38          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Tree of Perdition");
39  
40          setStopAt(3, PhaseStep.BEGIN_COMBAT);
41          execute();
42  
43          assertLife(playerA, 20);
44          assertLife(playerB, 13);
45          assertPowerToughness(playerA, "Tree of Perdition", 0, 13);
46      }
47      
48     /**
49      * Reported bug:
50      *       Tree of Perdition is gaining both power and toughness equal to opponent's life total
51      *       instead of just toughness equal to it.
52      */
53      @Test
54      public void testTreeOfPerditionOnlyGainsToughnessEqualToLife() {
55          addCard(Zone.BATTLEFIELD, playerA, "Tree of Perdition");
56          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
57  
58          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Exchange");
59          // Player B is auto-targeted since they're the only option
60  
61          setStopAt(1, PhaseStep.BEGIN_COMBAT);
62          execute();
63  
64          assertLife(playerA, 20);
65          assertLife(playerB, 13);
66          assertPowerToughness(playerA, "Tree of Perdition", 0, 20);
67      }

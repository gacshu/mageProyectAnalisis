25      /**
26       * Reported bug: "Archangel Avacyn damages herself when she transforms"
27       */
28      @Test
29      public void basicTransformTest() {
30          // Flash
31          // Flying
32          // Vigilance
33          // When Archangel Avacyn enters the battlefield, creatures you control gain indestructible until end of turn.
34          // When a non-Angel creature you control dies, transform Archangel Avacyn at the beginning of the next upkeep.
35          // Transformed side: Avacyn, the Purifier - Creature 6/5
36          // Flying
37          // When this creature transforms into Avacyn, the Purifier, it deals 3 damage to each other creature and each opponent.
38          addCard(Zone.BATTLEFIELD, playerA, "Archangel Avacyn");
39          addCard(Zone.BATTLEFIELD, playerA, "Wall of Omens"); // 0/4
40          addCard(Zone.HAND, playerA, "Elite Vanguard"); // 2/1
41          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
42          addCard(Zone.BATTLEFIELD, playerB, "Hill Giant"); // 3/3
43          addCard(Zone.BATTLEFIELD, playerB, "Wall of Roots"); // 0/5
44          addCard(Zone.HAND, playerB, "Shock");
45          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 2);
46  
47          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Elite Vanguard");
48          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Shock", "Elite Vanguard");
49          setStopAt(3, PhaseStep.DRAW);
50          execute();
51  
52          assertGraveyardCount(playerB, "Shock", 1);
53          assertPermanentCount(playerA, "Avacyn, the Purifier", 1);
54          assertPermanentCount(playerA, "Wall of Omens", 1);
55          assertGraveyardCount(playerA, "Elite Vanguard", 1);
56          assertPermanentCount(playerB, "Wall of Roots", 1);
57  
58          Permanent avacyn = getPermanent("Avacyn, the Purifier", playerA);
59          Assert.assertEquals("Damage to Avacyn, the Purifier should be 0 not 3", 0, avacyn.getDamage());
60  
61          assertGraveyardCount(playerB, "Hill Giant", 1);
62  
63      }

65      /**
66       * Reported Bug: If more than 1(in my case 3) non-angel creature dies, she
67       * flips various times dealing the 3 damage more than 1 time.
68       */
69      @Test
70      public void TransformOnlyOnceTest() {
71          // Flash
72          // Flying
73          // Vigilance
74          // When Archangel Avacyn enters the battlefield, creatures you control gain indestructible until end of turn.
75          // When a non-Angel creature you control dies, transform Archangel Avacyn at the beginning of the next upkeep.
76          // Transformed side: Avacyn, the Purifier - Creature 6/5
77          // Flying
78          // When this creature transforms into Avacyn, the Purifier, it deals 3 damage to each other creature and each opponent.
79          addCard(Zone.BATTLEFIELD, playerA, "Archangel Avacyn");
80          addCard(Zone.BATTLEFIELD, playerA, "Wall of Roots"); // 0/5
81          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
82          // Seismic Shudder deals 1 damage to each creature without flying.
83          addCard(Zone.HAND, playerA, "Seismic Shudder"); // {1}{R}
84  
85          addCard(Zone.BATTLEFIELD, playerA, "Elite Vanguard", 3); // 2/1
86  
87          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Seismic Shudder");
88          setStopAt(2, PhaseStep.DRAW);
89          execute();
90  
91          assertGraveyardCount(playerA, "Seismic Shudder", 1);
92          assertGraveyardCount(playerA, "Elite Vanguard", 3);
93  
94          assertPermanentCount(playerA, "Wall of Roots", 1);
95  
96          assertLife(playerA, 20);
97          assertLife(playerB, 17);
98      }

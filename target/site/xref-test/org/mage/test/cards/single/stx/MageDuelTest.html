9   /**
10   * {@link mage.cards.m.MageDuel Mage Duel}
11   * {2}{G}
12   * Instant
13   * <p>
14   * This spell costs {2} less to cast if you’ve cast another instant or sorcery spell this turn.
15   * <p>
16   * Target creature you control gets +1/+2 until end of turn.
17   * Then it fights target creature you don’t control.
18   * (Each deals damage equal to its power to the other.)
19   *
20   * @author Alex-Vasile
21   */
22  public class MageDuelTest extends CardTestPlayerBase {
23  
24      private static final String mageDuel = "Mage Duel";
25      // Instant
26      // {W}
27      // You gain hexproof until your next turn. You gain 2 life.
28      private static final String blossomingCalm = "Blossoming Calm";
29      // Sorcery
30      // {W}
31      // You gain 2 life for each creature you control.
32      private static final String festivalOfTrokin = "Festival of Trokin";
33  
34      // Creature you control
35      // 1/1
36      private static final String akroanJailer = "Akroan Jailer";
37      // Creature you don't control
38      // 2/2
39      private static final String lion = "Silvercoat Lion";
40  
41      /**
42       * Test that the spell costs {2}{G} if no other spell was cast
43       */
44      @Test
45      public void testCostReductionNotAlwaysOn() {
46          addCard(Zone.HAND, playerA, mageDuel);
47          addCard(Zone.HAND, playerA, "Forest");
48          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
49          addCard(Zone.BATTLEFIELD, playerA, akroanJailer);
50  
51          addCard(Zone.BATTLEFIELD, playerB, lion);
52  
53          checkPlayableAbility("Can't cast for {1}{G}", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast" + mageDuel, false);
54          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Forest");
55          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, mageDuel);
56  
57          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
58          execute();
59          assertPowerToughness(playerA, akroanJailer, 2, 3);
60          assertPermanentCount(playerA, akroanJailer, 1);
61          assertGraveyardCount(playerB, lion, 1);
62      }

64      /**
65       * Test cost reduction does not apply to itself.
66       * If you have {2}{G} to pay for it, it should still cost {2}{G} rather than {G}
67       */
68      @Test
69      public void testDoesntDiscountItself() {
70          addCard(Zone.HAND, playerA, mageDuel);
71          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
72          addCard(Zone.BATTLEFIELD, playerA, akroanJailer);
73  
74          addCard(Zone.BATTLEFIELD, playerB, lion);
75  
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, mageDuel);
77  
78          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
79          execute();
80          assertPowerToughness(playerA, akroanJailer, 2, 3);
81          assertPermanentCount(playerA, akroanJailer, 1);
82          assertGraveyardCount(playerB, lion, 1);
83          assertManaPool(playerA, ManaType.GREEN, 0);
84          assertTappedCount("Forest", true, 3);
85      }

88       * Test that the cost reduction works with instant spells.
89       */
90      @Test
91      public void testInstantCostReduction() {
92          addCard(Zone.HAND, playerA, mageDuel);
93          addCard(Zone.HAND, playerA, blossomingCalm);
94          addCard(Zone.BATTLEFIELD, playerA, "Plains");
95          addCard(Zone.BATTLEFIELD, playerA, "Forest");
96          addCard(Zone.BATTLEFIELD, playerA, akroanJailer);
97  
98          addCard(Zone.BATTLEFIELD, playerB, lion);
99  
100         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, blossomingCalm);
101         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
102         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, mageDuel);
103 
104         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
105         execute();
106         assertPowerToughness(playerA, akroanJailer, 2, 3);
107         assertPermanentCount(playerA, akroanJailer, 1);
108         assertGraveyardCount(playerB, lion, 1);
109         assertLife(playerA, 20 + 2);

137     /**
138      * Test that it works if a instant/sorcery card is copied and then cast
139      */
140     @Test
141     public void testCopiedCardCast() {
142         String isochronScpeter = "Isochron Scepter";
143         addCard(Zone.HAND, playerA, isochronScpeter);
144         addCard(Zone.HAND, playerA, mageDuel);
145         addCard(Zone.HAND, playerA, blossomingCalm);
146         addCard(Zone.BATTLEFIELD, playerA, "Forest", 1 + 2 + 2);
147         addCard(Zone.BATTLEFIELD, playerA, akroanJailer);
148 
149         addCard(Zone.BATTLEFIELD, playerB, lion);
150 
151         setStrictChooseMode(true);
152 
153         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, isochronScpeter);
154         setChoice(playerA, "Yes");
155         setChoice(playerA, blossomingCalm);
156         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
157 
158         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{2}, {T}:");
159         setChoice(playerA, "Yes");
160         setChoice(playerA, "Yes");
161 
162         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
163         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, mageDuel);
164         addTarget(playerA, akroanJailer);
165         addTarget(playerA, lion);
166 
167         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
168         execute();
169         assertPowerToughness(playerA, akroanJailer, 2, 3);
170         assertPermanentCount(playerA, akroanJailer, 1);
171         assertGraveyardCount(playerB, lion, 1);
172         assertLife(playerA, 20 + 2);
173     }

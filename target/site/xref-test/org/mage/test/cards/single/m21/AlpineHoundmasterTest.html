12      @Test
13      public void searchDog() {
14          // When Alpine Houndmaster enters the battlefield, you may search your library for a card named
15          // Alpine Watchdog and/or a card named Igneous Cur, reveal them, put them into your hand, then shuffle your library.
16          addCard(Zone.HAND, playerA, "Alpine Houndmaster", 1);
17          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
18          addCard(Zone.BATTLEFIELD, playerA, "Plains");
19          addCard(Zone.LIBRARY, playerA, "Alpine Watchdog");
20          addCard(Zone.LIBRARY, playerA, "Igneous Cur");
21  
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Alpine Houndmaster");
23          setChoice(playerA, true);
24          addTarget(playerA, "Alpine Watchdog");
25  
26          setStrictChooseMode(true);
27          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
28          execute();
29  
30          assertHandCount(playerA, 1);

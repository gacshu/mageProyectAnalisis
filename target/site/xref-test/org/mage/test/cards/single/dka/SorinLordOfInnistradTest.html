19      public void testCard() {
20          addCard(Zone.BATTLEFIELD, playerA, "Sorin, Lord of Innistrad");
21  
22          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: Create a 1/1 black Vampire creature token with lifelink.");
23          setStopAt(1, PhaseStep.BEGIN_COMBAT);
24          execute();
25  
26          assertLife(playerA, 20);
27          assertLife(playerB, 20);

33      @Test
34      public void testCard2() {
35          addCard(Zone.BATTLEFIELD, playerA, "Sorin, Lord of Innistrad");
36          addCard(Zone.BATTLEFIELD, playerA, "Sejiri Merfolk");
37  
38          addCounters(1, PhaseStep.UPKEEP, playerA, "Sorin, Lord of Innistrad", CounterType.LOYALTY, 1);
39          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-2: You get an emblem with ");
40          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "-2: You get an emblem with ");
41          setStopAt(3, PhaseStep.BEGIN_COMBAT);
42          execute();
43  
44          assertLife(playerA, 20);

51      @Test
52      public void testCard3() {
53          // Sorin, Lord of Innistrad English
54          //  Planeswalker — Sorin (Loyalty: 3), 2WB
55          //  +1: Create a 1/1 black Vampire creature token with lifelink.
56          //  -2: You get an emblem with "Creatures you control get +1/+0."
57          //  -6: Destroy up to three target creatures and/or other planeswalkers. Return each card put into a graveyard this way to the battlefield under your control.
58          addCard(Zone.BATTLEFIELD, playerA, "Sorin, Lord of Innistrad");
59          // Craw Wurm
60          //  Creature — Wurm 6/4, 4GG
61          addCard(Zone.BATTLEFIELD, playerB, "Craw Wurm");
62          // Angel of Mercy English
63          //  Creature — Angel 3/3, 4W (5)
64          //  Flying
65          //  When Angel of Mercy enters the battlefield, you gain 3 life.
66          addCard(Zone.BATTLEFIELD, playerB, "Angel of Mercy");
67  
68          addCounters(1, PhaseStep.UPKEEP, playerA, "Sorin, Lord of Innistrad", CounterType.LOYALTY, 3);
69          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-6: ", "Craw Wurm^Angel of Mercy");
70          setStopAt(1, PhaseStep.BEGIN_COMBAT);
71          execute();
72  
73          assertPermanentCount(playerA, "Craw Wurm", 1);
74          assertPermanentCount(playerA, "Angel of Mercy", 1);
75  
76          assertLife(playerB, 20);
77          assertLife(playerA, 23);
78          assertPermanentCount(playerA, "Sorin, Lord of Innistrad", 0);
79          assertPermanentCount(playerB, "Craw Wurm", 0);
80          assertPermanentCount(playerB, "Angel of Mercy", 0);
81          assertPermanentCount(playerA, "Craw Wurm", 1);
82          assertPermanentCount(playerA, "Angel of Mercy", 1);
83      }

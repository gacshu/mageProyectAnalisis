37          setChoice(playerA, "Red");

39          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt");
40          addTarget(playerA, "White Knight");
41  
42          setStopAt(1, PhaseStep.END_TURN);
43  
44          try {

12  public class KrarkTheThumblessTest extends CardTestPlayerBase {
13  
14      @Test
15      public void test_MustAbleToCastAgainAfterRemoveToHand() {
16          // https://github.com/magefree/mage/issues/7198
17          // bug: Casting a lightning bolt and failing flip 1 and then winning flip 2. While the copy was on the stack
18          // and the lightning bolt back in hand, i was not allowed to cast the lightning bolt until the copy resolved.
19  
20          // Whenever you cast an instant or sorcery spell, flip a coin. If you lose the flip, return that spell to
21          // its owner's hand. If you win the flip, copy that spell, and you may choose new targets for the copy.
22          addCard(Zone.BATTLEFIELD, playerA, "Krark, the Thumbless", 1);
23          //
24          // You may have Sakashima the Impostor enter the battlefield as a copy of any creature on the battlefield,
25          // except its name is Sakashima the Impostor, it's legendary in addition to its other types...
26          addCard(Zone.HAND, playerA, "Sakashima the Impostor", 1); // {2}{U}{U}
27          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
28          //
29          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
30          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2 * 2);
31  
32          // prepare two legendary creatures (for two flipcoin events)
33          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 4);
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sakashima the Impostor");
35          setChoice(playerA, true); // enter as copy
36          setChoice(playerA, "Krark, the Thumbless"); // copy target
37          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
38  
39          // FIRST BOLT CAST
40          // cast bolt and generate 2 triggers
41          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
42          setChoice(playerA, "Whenever"); // 2x triggers raise
43          checkStackSize("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 3); // bolt + 2x triggers
44          checkStackObject("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast L", 1);
45          checkStackObject("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Whenever you cast", 2);
46          checkHandCardCount("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 0);
47  
48          // resolve first trigger and lose (move spell to hand)
49          setFlipCoinResult(playerA, false);
50          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
51          checkStackSize("after lose trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1); // one trigger
52          checkStackObject("after lose trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast L", 0);
53          checkStackObject("after lose trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Whenever you cast", 1);
54          checkHandCardCount("after lose trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 1); // moved to hand
55  
56          // resolve second trigger and win (copy the spell)
57          setFlipCoinResult(playerA, true);
58          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
59          setChoice(playerA, false); // do not change a target of the copy
60          checkStackSize("after win trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 1); // copied bolt
61          checkStackObject("after lose trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast L", 1);
62          checkStackObject("after win trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Whenever you cast", 0);
63          checkHandCardCount("after win trigger", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 1); // stil in hand
64  
65          // SECOND BOLT CAST
66          // try to cast removed spell from hand
67          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
68          setChoice(playerA, "Whenever"); // 2x triggers raise
69          setFlipCoinResult(playerA, true); // win and copy second bolt
70          setChoice(playerA, false); // keep target
71          setFlipCoinResult(playerA, true); // win and copy second bolt
72          setChoice(playerA, false); // keep target
73  
74          // TOTAL to resolve:
75          // * 1 copied bolt from first bolt cast
76          // * 1 bolt + 2 copied bolts from second bolt cast
77          // total damage: 4 * 3
78  
79          setStrictChooseMode(true);
80          setStopAt(1, PhaseStep.END_TURN);
81          execute();
82  
83          assertLife(playerB, 20 - 4 * 3);
84      }

4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   * {@link mage.cards.d.DaybreakCoronet Daybreak Coronet}
11   * {W}{W}
12   * Enchantment — Aura
13   * Enchant creature with another Aura attached to it
14   * Enchanted creature gets +3/+3 and has first strike, vigilance, and lifelink.
15   * (Damage dealt by the creature also causes its controller to gain that much life.)
16   *
17   * @author LevelX2
18   */
19  public class DaybreakCoronetTest extends CardTestPlayerBase {
20  
21      @Test
22      public void testCantEnchantTargetWithoutAura() {
23          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
24          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
25          addCard(Zone.HAND, playerA, "Daybreak Coronet");
26  
27          checkPlayableAbility("can't cast coronet", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Daybreak", false);
28  
29          setStopAt(1, PhaseStep.BEGIN_COMBAT);
30          execute();
31  
32          assertPermanentCount(playerA, "Daybreak Coronet", 0);
33      }
34  
35      @Test
36      public void testCanBeCastIfEnchantedByAura() {
37          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
38          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
39          addCard(Zone.HAND, playerA, "Holy Strength");
40          addCard(Zone.HAND, playerA, "Daybreak Coronet");
41  
42  
43          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Holy Strength", "Silvercoat Lion");
44          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
45          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Daybreak Coronet", "Silvercoat Lion");
46  
47          setStopAt(1, PhaseStep.BEGIN_COMBAT);
48          execute();
49  
50          assertPermanentCount(playerA, "Daybreak Coronet", 1);
51      }
52  
53      /**
54       * If all other Auras attached to the enchanted creature stop enchanting it,
55       * Daybreak Coronet will be attached to an illegal permanent and will be put into its owner’s graveyard.
56       * (2007-05-01)
57       */
58      @Test
59      public void testTargetGetsIllegal() {
60          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
61          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
62          addCard(Zone.HAND, playerA, "Holy Strength");
63          addCard(Zone.HAND, playerA, "Daybreak Coronet");
64  
65          addCard(Zone.BATTLEFIELD, playerB, "Plains", 1);
66          addCard(Zone.HAND, playerB, "Demystify");
67  
68          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Holy Strength", "Silvercoat Lion");
69          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
70          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Daybreak Coronet", "Silvercoat Lion");
71          
72          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Demystify", "Holy Strength");
73  
74          setStopAt(1, PhaseStep.END_TURN);
75          execute();
76  
77          assertGraveyardCount(playerA, "Holy Strength", 1);
78          assertGraveyardCount(playerB, "Demystify", 1);
79          assertPermanentCount(playerA, "Daybreak Coronet", 0);
80          assertGraveyardCount(playerA, "Daybreak Coronet", 1);
81      }
82  }

11  public class EmeriaShepherdTest extends CardTestPlayerBase {

13      /*
14       * Reported bug: Emeria Shepherd can't bounce Bruna to the table when a plains enters the battlefield.
15       */
16      @Test
17      public void emeriaInteractionWithBruna()
18      {
19          /*
20           *
21           * Emeria Shepherd (5)(W)(W)
22              Flying 4/4 Creature Angel
23              Landfall â€” Whenever a land enters the battlefield under your control, you may return target nonland permanent card from your graveyard to your hand.
24              If that land is a Plains, you may return that nonland permanent card to the battlefield instead.
25           */
26          String emeria = "Emeria Shepherd";
27  
28          /*
29           *  Bruna, The Fading Light (5)(W)(W)
30           *  Legendary Creature - Angel Horror 5/7
31              When you cast Bruna, the Fading Light, you may return target Angel or Human creature card from your graveyard to the battlefield.
32              Flying, vigilance
33              (Melds with Gisela, the Broken Blade.)
34           */
35          String bruna = "Bruna, the Fading Light";
36  
37          addCard(Zone.BATTLEFIELD, playerA, emeria);
38          addCard(Zone.HAND, playerA, "Plains");
39          addCard(Zone.GRAVEYARD, playerA, bruna);
40          addCard(Zone.BATTLEFIELD, playerA, "Plains", 7);
41  
42          setStrictChooseMode(true);
43  
44          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Plains");
45          setChoice(playerA, true); // opt to use Emeria's triggered ability
46          setChoice(playerA, true); // opt to put it onto the battlefield instead of the hand
47          addTarget(playerA, bruna); // target Bruna in grave
48  
49          setStopAt(1, PhaseStep.BEGIN_COMBAT);
50          execute();
51  
52          assertHandCount(playerA, "Plains", 0);
53          assertGraveyardCount(playerA, bruna, 0);
54          assertHandCount(playerA, bruna, 0);
55          assertPermanentCount(playerA, bruna, 1);
56      }
57  }

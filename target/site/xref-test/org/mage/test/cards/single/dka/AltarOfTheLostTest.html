9   /**
10   * {@link mage.cards.a.AltarOfTheLost Altar of the Lost}
11   * {3}
12   * Artifact
13   * Altar of the Lost enters the battlefield tapped.
14   * {T}: Add two mana in any combination of colors.
15   *      Spend this mana only to cast spells with flashback from a graveyard.
16   *
17   * @author BetaSteward
18   */
19  public class AltarOfTheLostTest extends CardTestPlayerBase {
20  
21      /**
22       * Test that the mana can be used to cast the spell from the graveyard using flashback.
23       */
24      @Test
25      public void testFlashback() {
26          addCard(Zone.BATTLEFIELD, playerA, "Altar of the Lost");
27          // Put two 1/1 white Spirit creature tokens with flying onto the battlefield.
28          // Flashback {1}{B}
29          addCard(Zone.GRAVEYARD, playerA, "Lingering Souls");
30  
31          // Add 2 black mana (mana choice in WUBRG order)
32          setChoice(playerA, "X=0");
33          setChoice(playerA, "X=0");
34          setChoice(playerA, "X=2");
35          setChoice(playerA, TestPlayer.CHOICE_SKIP);
36  
37          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Flashback {1}{B}");
38          setStopAt(3, PhaseStep.BEGIN_COMBAT);
39          execute();
40  
41          assertPermanentCount(playerA, "Spirit Token", 2);
42      }
43  
44      /**
45       * Test that the same card cannot be cast from the hand even though it has a flashback ability.
46       */
47      @Test
48      public void testCantCastFromHand() {
49          addCard(Zone.BATTLEFIELD, playerA, "Plains",2 );
50          addCard(Zone.BATTLEFIELD, playerA, "Altar of the Lost");
51          addCard(Zone.HAND, playerA, "Lingering Souls");
52  
53          checkPlayableAbility("mana not available", 3, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lingering", false);
54          setStopAt(3, PhaseStep.BEGIN_COMBAT);
55          execute();
56  
57          assertPermanentCount(playerA, "Spirit Token", 0);
58      }
59  
60  }

38      /**
39       * Test that it works for sorcery, but only the first one.
40       */
41      @Test
42      public void testWorksForFirstOnly() {
43          addCard(Zone.BATTLEFIELD, playerA, anhelo);
44          addCard(Zone.BATTLEFIELD, playerA, mountain, 2);
45          addCard(Zone.BATTLEFIELD, playerA, aetherwindBasker, 2);
46          addCard(Zone.HAND, playerA, lightningBolt, 2);
47  
48          setStrictChooseMode(true);
49  
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, lightningBolt, playerB);
51          setChoice(playerA, "Yes"); // Cast with Casualty
52          setChoice(playerA, aetherwindBasker);
53          setChoice(playerA, "No"); // Don't change targets
54          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, lightningBolt, playerB);
55  
56          setStopAt(1, PhaseStep.END_TURN);
57          execute();
58          assertLife(playerB, 20 - 2*3 - 3);
59          assertPermanentCount(playerA, aetherwindBasker, 1);
60      }

148     /**
149      * Test that a card which already has Casualty will gain a second instance of Casualty and thus let you sacrifice twice in order to get 2 copies.
150      */
151     @Test
152     public void testGainsSecondCasualty() {
153         addCard(Zone.BATTLEFIELD, playerA, anhelo);
154         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
155         addCard(Zone.BATTLEFIELD, playerA, aetherwindBasker, 2);
156         addCard(Zone.HAND, playerA, aLittleChat);
157         addCard(Zone.LIBRARY, playerA, "Desert", 6);
158 
159         setStrictChooseMode(true);
160         skipInitShuffling();
161 
162         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, aLittleChat);
163         setChoice(playerA, "Yes"); // First instance of casualty
164         setChoice(playerA, "Yes"); // Second instance of casualty
165         setChoice(playerA, aetherwindBasker);
166         setChoice(playerA, aetherwindBasker);
167         setChoice(playerA, "When you do"); // Chose which of the two copies to put on the stack first
168         addTarget(playerA, "Desert");
169         addTarget(playerA, "Desert");
170         addTarget(playerA, "Desert");
171 
172         setStopAt(2, PhaseStep.END_TURN);
173         execute();
174         assertHandCount(playerA, 3);
175         assertPermanentCount(playerA, aetherwindBasker, 0);
176     }

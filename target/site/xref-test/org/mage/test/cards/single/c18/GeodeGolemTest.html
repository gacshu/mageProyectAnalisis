13      @Test
14      public void test_Normal() {
15          // Whenever Geode Golem deals combat damage to a player, you may
16          // cast your commander from the command zone without paying its mana cost.
17          addCard(Zone.BATTLEFIELD, playerA, "Geode Golem");
18          //
19          addCard(Zone.COMMAND, playerA, "Grizzly Bears"); // {1}{G}
20          addCard(Zone.BATTLEFIELD, playerA, "Forest", 10);
21          //
22          addCustomEffect_TargetDamage(playerA, 2);
23  
24          checkCommandCardCount("before 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 1);
25  
26          // turn 1 - first cast
27  
28          // attack and cast first time (free)
29          attack(1, playerA, "Geode Golem");
30          setChoice(playerA, true); // cast commander
31          setChoice(playerA, "Grizzly Bears"); // commander choice
32          waitStackResolved(1, PhaseStep.COMBAT_DAMAGE);
33          checkPermanentCount("after 1", 1, PhaseStep.COMBAT_DAMAGE, playerA, "Grizzly Bears", 1);
34          checkPermanentTapped("after 1", 1, PhaseStep.COMBAT_DAMAGE, playerA, "Forest", true, 0);
35          //
36          // remove to command zone (0x tax)
37          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "target damage 2", "Grizzly Bears");
38          setChoice(playerA, true); // move to command zone
39  
40          // turn 3 - second cast (1x tax)
41  
42          attack(3, playerA, "Geode Golem");
43          setChoice(playerA, true); // cast commander
44          setChoice(playerA, "Grizzly Bears"); // commander choice
45          waitStackResolved(3, PhaseStep.COMBAT_DAMAGE);
46          checkPermanentCount("after 2", 3, PhaseStep.COMBAT_DAMAGE, playerA, "Grizzly Bears", 1);
47          checkPermanentTapped("after 2", 3, PhaseStep.COMBAT_DAMAGE, playerA, "Forest", true, 2); // 1x tax
48          //
49          // remove to command zone
50          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "target damage 2", "Grizzly Bears");
51          setChoice(playerA, true); // move to command zone
52  
53          // turn 5 - third cast (2x tax)
54  
55          attack(5, playerA, "Geode Golem");
56          setChoice(playerA, true); // cast commander
57          setChoice(playerA, "Grizzly Bears"); // commander choice
58          waitStackResolved(5, PhaseStep.COMBAT_DAMAGE);
59          checkPermanentCount("after 3", 5, PhaseStep.COMBAT_DAMAGE, playerA, "Grizzly Bears", 1);
60          checkPermanentTapped("after 3", 5, PhaseStep.COMBAT_DAMAGE, playerA, "Forest", true, 2 * 2); // 2x tax
61  
62          setStrictChooseMode(true);
63          setStopAt(5, PhaseStep.END_TURN);
64          execute();
65  
66          assertPermanentCount(playerA, "Grizzly Bears", 1);
67      }

131         addCard(Zone.BATTLEFIELD, playerA, "Geode Golem");
132         //
133         // Birgi, God of Storytelling {2}{R} - creature 3/3
134         // Harnfel, Horn of Bounty {4}{R} - artifact
135         addCard(Zone.COMMAND, playerA, "Birgi, God of Storytelling");
136         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 10);
137         //
138         addCustomEffect_TargetDamage(playerA, 3);
139         addCustomEffect_DestroyTarget(playerA);
140 
141         checkCommandCardCount("before 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Birgi, God of Storytelling", 1);
142 
143         // turn 1 - first cast, LEFT side
144 
145         // attack and cast first time (free)
146         attack(1, playerA, "Geode Golem");
147         setChoice(playerA, true); // cast commander
148         setChoice(playerA, "Birgi, God of Storytelling"); // commander choice
149         setChoice(playerA, "Cast Birgi, God of Storytelling"); // spell choice
150         waitStackResolved(1, PhaseStep.COMBAT_DAMAGE);
151         checkPermanentCount("after 1", 1, PhaseStep.COMBAT_DAMAGE, playerA, "Birgi, God of Storytelling", 1);
152         checkPermanentTapped("after 1", 1, PhaseStep.COMBAT_DAMAGE, playerA, "Mountain", true, 0);
153         //
154         // remove to command zone (0x tax)
155         activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "target damage 3", "Birgi, God of Storytelling");
156         setChoice(playerA, true); // move to command zone
157 
158         // turn 3 - second cast, LEFT side (1x tax)
159 
160         attack(3, playerA, "Geode Golem");
161         setChoice(playerA, true); // cast commander
162         setChoice(playerA, "Birgi, God of Storytelling"); // commander choice
163         setChoice(playerA, "Cast Birgi, God of Storytelling"); // spell choice
164         waitStackResolved(3, PhaseStep.COMBAT_DAMAGE);

166         checkPermanentTapped("after 2", 3, PhaseStep.COMBAT_DAMAGE, playerA, "Mountain", true, 2); // 1x tax
167         //
168         // remove to command zone
169         activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "target damage 3", "Birgi, God of Storytelling");

171 
172         // turn 5 - third cast, RIGHT side (2x tax)
173 
174         attack(5, playerA, "Geode Golem");

176         setChoice(playerA, "Birgi, God of Storytelling"); // commander choice

184         setChoice(playerA, true); // move to command zone
185

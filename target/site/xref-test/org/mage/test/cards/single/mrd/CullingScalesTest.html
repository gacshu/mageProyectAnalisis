12  public class CullingScalesTest extends CardTestPlayerBase {
13      
14      @Test
15      public void testCullingScalesBasic() {
16          addCard(Zone.BATTLEFIELD, playerA, "Mountain"); // CMC = 0, but not a nonland permanent
17          addCard(Zone.BATTLEFIELD, playerA, "Culling Scales"); // CMC = 3
18          addCard(Zone.BATTLEFIELD, playerB, "Siege Rhino"); // CMC = 4
19          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
20          execute();
21          
22          assertPermanentCount(playerA, "Mountain", 1);
23          assertPermanentCount(playerA, "Culling Scales", 0);
24          assertPermanentCount(playerB, "Siege Rhino", 1);
25      }
26      
27      @Test
28      public void testCullingScalesPlusHexproof() {
29          addCard(Zone.BATTLEFIELD, playerB, "Bassara Tower Archer"); // CMC = 2, hexproof
30          addCard(Zone.BATTLEFIELD, playerA, "Culling Scales"); // CMC = 3
31          addCard(Zone.BATTLEFIELD, playerB, "Siege Rhino"); // CMC = 4
32          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
33          execute();
34          
35          // Nothing happens since no valid targets
36          // (the only nonland permanent with the lowest CMC has hexproof)
37          assertPermanentCount(playerB, "Bassara Tower Archer", 1);
38          assertPermanentCount(playerA, "Culling Scales", 1);
39          assertPermanentCount(playerB, "Siege Rhino", 1);
40      }
41      
42      @Test
43      public void testCullingScalesFizzleByMakingLowerCostedPermanent() {
44          // Gatherer ruling: If the targeted permanent doesn't have the lowest converted mana cost
45          // when the ability resolves, the ability is countered and the permanent isn't destroyed.
46          
47          addCard(Zone.HAND, playerB, "Raise the Alarm"); // Make 2 tokens
48          addCard(Zone.BATTLEFIELD, playerB, "Elvish Visionary"); // CMC = 2
49          addCard(Zone.BATTLEFIELD, playerB, "Plains", 5);
50          addCard(Zone.BATTLEFIELD, playerA, "Culling Scales"); // CMC = 3
51  
52          setStrictChooseMode(true);
53          // On upkeep Culling Scales targets Elvish Visionary
54          addTarget(playerA, "Elvish Visionary");
55          
56          // Cast Raise the Alarm in response
57          castSpell(1, PhaseStep.UPKEEP, playerB, "Raise the Alarm", null, "At the beginning of");
58          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
59          execute();
60          
61          // Culling Scales trigger fizzles since the Visionary no longer has the lowest CMC
62          assertPermanentCount(playerB, "Soldier Token", 2);
63          assertPermanentCount(playerB, "Elvish Visionary", 1);
64          assertPermanentCount(playerB, "Plains", 5);
65          assertPermanentCount(playerA, "Culling Scales", 1);
66      }

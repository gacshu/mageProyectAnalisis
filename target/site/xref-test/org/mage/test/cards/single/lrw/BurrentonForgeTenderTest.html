15      @Test
16      public void testPreventDamageFromStack() {
17          // Sacrifice Burrenton Forge-Tender: Prevent all damage a red source of your choice would deal this turn.
18          addCard(Zone.BATTLEFIELD, playerA, "Burrenton Forge-Tender");
19          addCard(Zone.BATTLEFIELD, playerA, "Soldier of the Pantheon");
20  
21          addCard(Zone.HAND, playerB, "Lightning Bolt");
22          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
23  
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Soldier of the Pantheon");
25          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice {this}: Prevent all damage a red source of your choice would deal this turn.", TestPlayer.NO_TARGET, "Cast Lightning Bolt");
26          playerA.addChoice("Lightning Bolt");
27  
28          setStopAt(1, PhaseStep.BEGIN_COMBAT);
29          execute();
30  
31          assertGraveyardCount(playerB, "Lightning Bolt", 1);
32  
33          assertGraveyardCount(playerA, "Burrenton Forge-Tender", 1);
34          assertPermanentCount(playerA, "Soldier of the Pantheon", 1);
35      }

93      @Test
94      public void testPreventDamageFromToken() {
95          addCard(Zone.BATTLEFIELD, playerA, "Burrenton Forge-Tender");
96          addCard(Zone.BATTLEFIELD, playerA, "Soldier of the Pantheon");
97          // Sacrifice Mogg Fanatic: Mogg Fanatic deals 1 damage to any target.
98          addCard(Zone.BATTLEFIELD, playerA, "Mogg Fanatic");
99          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
100         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
101         // Choose one - Return target creature you control and all Auras you control attached to it to their owner's hand; 
102         // or destroy target creature and you lose life equal to its toughness; 
103         // or return target creature card with converted mana cost 1 or less from your graveyard to the battlefield.
104         addCard(Zone.HAND, playerA, "Orzhov Charm");
105 
106         // Kicker {5} (You may pay an additional as you cast this spell.)
107         // Create a tokenonto the battlefield that's a copy of target creature. If Rite of Replication was kicked, put five of those tokens onto the battlefield instead.
108         addCard(Zone.HAND, playerB, "Rite of Replication");
109         addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
110 
111         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Rite of Replication", "Mogg Fanatic");
112         setChoice(playerB, false); // no kicker
113 
114         castSpell(2, PhaseStep.BEGIN_COMBAT, playerA, "Orzhov Charm", "Mogg Fanatic");
115         setModeChoice(playerA, "1");
116 
117         activateAbility(2, PhaseStep.END_COMBAT, playerA, "Sacrifice {this}: Prevent all damage a red source of your choice would deal this turn.");
118         playerA.addChoice("Mogg Fanatic");
119 
120         activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Sacrifice {this}: {this} deals 1 damage to ", "Soldier of the Pantheon");
121 
122         setStopAt(2, PhaseStep.END_TURN);
123         execute();
124 
125         assertPermanentCount(playerB, "Mogg Fanatic", 0);
126         assertGraveyardCount(playerA, "Orzhov Charm", 1);
127         assertHandCount(playerA, "Mogg Fanatic", 1);
128         assertGraveyardCount(playerB, "Rite of Replication", 1);
129 
130         assertGraveyardCount(playerA, "Burrenton Forge-Tender", 1);
131         assertPermanentCount(playerA, "Soldier of the Pantheon", 1);
132     }

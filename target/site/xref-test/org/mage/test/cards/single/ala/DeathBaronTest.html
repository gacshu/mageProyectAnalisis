3   import mage.abilities.Ability;
4   import mage.abilities.keyword.DeathtouchAbility;
5   import mage.constants.PhaseStep;
6   import mage.constants.Zone;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;

10  /**
11   * @author awjackson
12   */
13  public class DeathBaronTest extends CardTestPlayerBase {
14  
15      private static final String baron = "Death Baron";
16      private static final String skeleton = "Drudge Skeletons";
17      private static final String zombie = "Scathe Zombies";
18      private static final String knight = "Black Knight";
19  
20      @Test
21      public void testDoesntNormallyAffectSelf() {
22          addCard(Zone.BATTLEFIELD, playerA, baron);
23          addCard(Zone.BATTLEFIELD, playerA, skeleton);
24          addCard(Zone.BATTLEFIELD, playerA, zombie);
25          addCard(Zone.BATTLEFIELD, playerA, knight);
26  
27          addCard(Zone.BATTLEFIELD, playerB, skeleton);
28          addCard(Zone.BATTLEFIELD, playerB, zombie);
29  
30          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
31          execute();
32  
33          Ability deathtouch = DeathtouchAbility.getInstance();
34  
35          // Death Baron doesn't normally affect itself
36          assertPowerToughness(playerA, baron, 2, 2);
37          assertAbility(playerA, baron, deathtouch, false);
38  
39          assertPowerToughness(playerA, skeleton, 2, 2);
40          assertPowerToughness(playerA, zombie, 3, 3);
41          assertPowerToughness(playerA, knight, 2, 2);
42          assertAbility(playerA, skeleton, deathtouch, true);
43          assertAbility(playerA, zombie, deathtouch, true);
44          assertAbility(playerA, knight, deathtouch, false);
45  
46          assertPowerToughness(playerB, skeleton, 1, 1);
47          assertPowerToughness(playerB, zombie, 2, 2);
48          assertAbility(playerB, skeleton, deathtouch, false);
49          assertAbility(playerB, zombie, deathtouch, false);
50      }
51  
52      @Test
53      public void testBecomeSkeleton() {
54          addCard(Zone.BATTLEFIELD, playerA, baron);
55          addCard(Zone.BATTLEFIELD, playerA, "Amoeboid Changeling");
56  
57          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Target creature gains", baron);
58  
59          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
60          execute();
61  
62          // If you manage to turn it into a Skeleton, however, then it will give itself +1/+1 and deathtouch
63          assertPowerToughness(playerA, baron, 3, 3);
64          assertAbility(playerA, baron, DeathtouchAbility.getInstance(), true);
65      }
66  }

13      // Test that extra turn is added correctly when Emrakul is cast during an opponent's turn.
14      @Test
15      public void test_ExtraTurn_Turn2() {
16          addCard(Zone.BATTLEFIELD, playerB, "Island", 20);
17          // Creature cards you own that aren't on the battlefield have flash.
18          addCard(Zone.HAND, playerB, "Teferi, Mage of Zhalfir");
19          // When you cast Emrakul, you gain control of target opponent during that player's next turn.
20          // After that turn, that player takes an extra turn.
21          addCard(Zone.HAND, playerB, "Emrakul, the Promised End");
22  
23          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Teferi, Mage of Zhalfir",  true);
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Emrakul, the Promised End");
25          addTarget(playerB, playerA);
26  
27          setStrictChooseMode(true);
28          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
29          execute();
30  
31          assertActivePlayer(playerB);
32      }

68      @Test
69      public void test_CostReduction_FromHand() {
70          // {13}
71          // This spell costs {1} less to cast for each card type among cards in your graveyard.
72          // When you cast this spell, you gain control of target opponent during that player’s next turn. After that turn, that player takes an extra turn.
73          addCard(Zone.HAND, playerA, "Emrakul, the Promised End");
74          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 13 - 2);
75          addCard(Zone.GRAVEYARD, playerA, "Balduvian Bears", 1);
76          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt", 1);
77  
78          checkPlayableAbility("can play", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Emrakul, the Promised End", true);
79          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Emrakul, the Promised End");
80          addTarget(playerA, playerB);
81  
82          setStrictChooseMode(true);
83          setStopAt(1, PhaseStep.END_TURN);
84          execute();
85  
86          assertPermanentCount(playerA, "Emrakul, the Promised End", 1);
87      }

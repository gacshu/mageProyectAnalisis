19      /**
20       * Only landfall under your control triggers first ability
21       */
22      @Test
23      public void landfallUnderOwnControlTriggers() {
24          
25          addCard(Zone.BATTLEFIELD, playerA, "Tireless Tracker", 1);
26          addCard(Zone.HAND, playerA, "Forest", 3);
27          addCard(Zone.HAND, playerB, "Wastes", 3);
28          
29          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Forest");
30          playLand(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Wastes");
31          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
32          execute();
33          
34          assertPermanentCount(playerA, "Forest", 1);
35          assertPermanentCount(playerB, "Wastes", 1);
36          assertPermanentCount(playerA, "Clue Token", 1);
37      }

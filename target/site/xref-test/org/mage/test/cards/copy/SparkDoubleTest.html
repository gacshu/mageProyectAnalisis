25      private Permanent findDoubleSparkPermanent(Game game) {
26          for (Permanent perm : game.getBattlefield().getAllActivePermanents()) {
27              if (perm.isCopy()) {
28                  return perm;
29              }
30          }
31          Assert.fail("spark must exist");
32          return null;
33      }

35      @Test
36      public void test_CopyCreatureAndGetOneCounter() {
37          addCard(Zone.BATTLEFIELD, playerA, "Abbey Griffin", 1); // 2/2, fly, vig
38          //
39          addCard(Zone.HAND, playerA, "Spark Double"); // {3}{U}
40          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
41  
42          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Spark Double");
43          setChoice(playerA, true);
44          setChoice(playerA, "Abbey Griffin");
45  
46          setStrictChooseMode(true);
47          setStopAt(1, PhaseStep.BEGIN_COMBAT);
48          execute();
49  
50          assertPermanentCount(playerA, "Abbey Griffin", 2);
51  
52          Permanent spark = findDoubleSparkPermanent(currentGame);
53          Assert.assertEquals("must add 1 counter", 1, spark.getCounters(currentGame).getCount(CounterType.P1P1));
54          //
55          Assert.assertEquals("must copy p/t", 3, spark.getPower().getValue());
56          Assert.assertEquals("must copy p/t", 3, spark.getToughness().getValue());
57          Assert.assertTrue("must copy ability", spark.getAbilities().contains(VigilanceAbility.getInstance()));
58      }

160 
161         /*
162         Spark Double isn’t legendary if it copies a legendary permanent, and this exception is copiable.
163         If something else copies Spark Double later, that copy also won’t be legendary.
164         If you control two or more permanents with the same name but only one is legendary, the “legend rule” doesn’t apply. (2019-05-03)
165 
166         it's applier copy check
167          */
168         //
169         addCard(Zone.HAND, playerA, "Spark Double"); // {3}{U}
170         addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
171         //
172         addCard(Zone.BATTLEFIELD, playerA, "Akroma, Angel of Wrath", 1); // legendary
173         //
174         // Create a 1/1 white Bird creature token with flying, then populate. (Create a token that’s a copy of a creature token you control.)
175         addCard(Zone.HAND, playerA, "Eyes in the Skies"); // {3}{W}
176         addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
177         //
178         // Create a token that’s a copy of target creature you control.
179         addCard(Zone.HAND, playerA, "Quasiduplicate"); // {1}{U}{U}
180         addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
181 
182         // make copy of legendary creature (it's not legendary now)
183         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Spark Double");
184         setChoice(playerA, true);
185         setChoice(playerA, "Akroma, Angel of Wrath");
186         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
187         checkPermanentCount("must have copy", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Akroma, Angel of Wrath", 2);
188

190 //        showBattlefield("before last copy", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
191         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Quasiduplicate");
192         addTarget(playerA, "Akroma, Angel of Wrath[only copy]");
193         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);

238     @Test
239     public void test_SparkCopyEachOther() {
240         // rules:
241         // 706.9e Some replacement effects that generate copy effects include an exception that’s an
242         // additional effect rather than a modification of the affected object’s characteristics.
243         // If another copy effect is applied to that object after applying the copy effect with that
244         // exception, the exception’s effect doesn’t happen.
245         // Example: Altered Ego reads, “You may have Altered Ego enter the battlefield as a copy of any
246         // creature on the battlefield, except it enters with X additional +1/+1 counters on it.” You
247         // choose for it to enter the battlefield as a copy of Clone, which reads “You may have Clone
248         // enter the battlefield as a copy of any creature on the battlefield,” for which no creature
249         // was chosen as it entered the battlefield. If you then choose a creature to copy as you apply
250         // the replacement effect Altered Ego gains by copying Clone, Altered Ego’s replacement effect
251         // won’t cause it to enter the battlefield with any +1/+1 counters on it.
252 
253         addCard(Zone.HAND, playerA, "Spark Double", 2); // {3}{U}
254         addCard(Zone.BATTLEFIELD, playerA, "Island", 4 * 2);
255         //
256         addCard(Zone.BATTLEFIELD, playerA, "Grizzly Bears", 1);
257 
258         // cast first spark
259         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Spark Double");
260         setChoice(playerA, true);
261         setChoice(playerA, "Grizzly Bears");
262         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
263         checkPermanentCount("after 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 2);
264 
265         // cast second spark
266         // rules 706.9e affected, so must get only 1 counter
267         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Spark Double");
268         setChoice(playerA, true);
269         setChoice(playerA, "Grizzly Bears[only copy]");
270         //setChoice(playerA, "Grizzly Bears"); // possible bug: two etb effects
271         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
272         checkPermanentCount("after 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 3);
273 
274         setStrictChooseMode(true);
275         setStopAt(1, PhaseStep.END_TURN);
276         execute();
277     }

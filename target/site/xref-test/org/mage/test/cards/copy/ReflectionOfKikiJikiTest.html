14  public class ReflectionOfKikiJikiTest extends CardTestPlayerBase {
15  
16      /**
17       * Reflection of Kiki-Jiki creates a copy of Humble Defector, activate
18       * Humble defector, token gets sacrificed while under opponents control.
19       */
20      @Test
21      public void testTokenNotSacrificedIfNotControlled() {
22          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
23          // Tap target creature you don't control.
24          // Overload {3}{U}
25          addCard(Zone.HAND, playerA, "Blustersquall", 1);
26  
27          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
28          addCard(Zone.BATTLEFIELD, playerB, "Reflection of Kiki-Jiki", 1);
29          // {T}: Draw two cards. Target opponent gains control of Humble Defector. Activate this ability only during your turn.
30          addCard(Zone.BATTLEFIELD, playerB, "Humble Defector", 1);
31  
32          castSpell(2, PhaseStep.UPKEEP, playerA, "Blustersquall", "Humble Defector"); // Tap nontoken Defector so only the Token can be used later
33  
34          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{1}, {T}: Create a token that's a copy of another target nonlegendary creature you control", "Humble Defector");
35  
36          activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "{T}: Draw two cards. Target opponent gains control");
37  
38          setStopAt(3, PhaseStep.UPKEEP);
39          execute();
40  
41          assertLife(playerA, 20);
42          assertLife(playerB, 20);
43  
44          assertHandCount(playerB, 3); // normal 3 draw from turn two + 2 from Defector
45  
46          assertGraveyardCount(playerA, "Blustersquall", 1);
47          assertPermanentCount(playerB, "Humble Defector", 1);
48          assertPermanentCount(playerA, "Humble Defector", 1);
49  
50      }
51  }

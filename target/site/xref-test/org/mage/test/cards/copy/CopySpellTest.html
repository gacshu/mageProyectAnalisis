28      @Test
29      public void copyChainOfVapor() {
30          // Return target nonland permanent to its owner's hand. Then that permanent's controller may sacrifice a land. If the player does, they may copy this spell and may choose a new target for that copy.
31          addCard(Zone.HAND, playerA, "Chain of Vapor", 1);
32          addCard(Zone.BATTLEFIELD, playerA, "Island", 10);
33  
34          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 10);
35  
36          addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox", 10);
37          addCard(Zone.BATTLEFIELD, playerB, "Island", 10);
38  
39          // start chain from A - return pillar to hand
40          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chain of Vapor", "Pillarfield Ox");
41          // chain 1 - B can return
42          addTarget(playerB, "Island"); // select a land to sacrifice
43          setChoice(playerB, true); // want to copy spell
44          setChoice(playerB, true); // want to change target
45          addTarget(playerB, "Silvercoat Lion"); // new target after copy
46          // chain 2 - A can return
47          addTarget(playerA, "Island"); // select a land to sacrifice
48          setChoice(playerA, true); // want to copy spell
49          setChoice(playerA, true); // want to change target
50          addTarget(playerA, "Pillarfield Ox"); // new target after copy
51          // stop the chain by B
52          addTarget(playerB, TestPlayer.TARGET_SKIP);
53  
54          setStopAt(1, PhaseStep.BEGIN_COMBAT);
55          execute();
56  
57          assertHandCount(playerA, "Silvercoat Lion", 1);
58          assertHandCount(playerB, "Pillarfield Ox", 2);
59          assertPermanentCount(playerA, "Silvercoat Lion", 10 - 1);
60          assertPermanentCount(playerB, "Pillarfield Ox", 10 - 2);
61          assertGraveyardCount(playerA, "Island", 1);
62          assertGraveyardCount(playerB, "Island", 1);
63      }

429         removeAllCardsFromLibrary(playerA);

432         // Spells you cast with converted mana cost 6 or greater have cascade.
433         // Cascade
434         // (When you cast this spell exile cards from the top of your library until you exile a
435         // nonland card whose converted mana cost is less than this spell's converted mana cost. You may cast
436         // that spell without paying its mana cost if its converted mana cost is less than this spell's
437         // converted mana cost. Then put all cards exiled this way that weren't cast on the bottom of
438         // your library in a random order.)
439         addCard(Zone.BATTLEFIELD, playerA, "Imoti, Celebrant of Bounty", 1); // {3}{G}{U}
440         //
441         addCard(Zone.LIBRARY, playerA, "Swamp", 1);
442         addCard(Zone.LIBRARY, playerA, "Lightning Bolt", 1);
443         addCard(Zone.LIBRARY, playerA, "Swamp", 1);
444         addCard(Zone.LIBRARY, playerA, "Lightning Bolt", 1);
445         addCard(Zone.LIBRARY, playerA, "Swamp", 1);
446         //
447         // You may have Spark Double enter the battlefield as a copy of a creature or planeswalker you control,
448         // except it enters with an additional +1/+1 counter on it if it’s a creature, it enters with an
449         // additional loyalty counter on it if it’s a planeswalker, and it isn’t legendary if that
450         // permanent is legendary.
451         addCard(Zone.HAND, playerA, "Spark Double", 1); // {3}{U}
452         addCard(Zone.BATTLEFIELD, playerA, "Island", 4);

573     @Test
574     public void test_CopiedSpellsHasntETB() {
575         // testing:
576         // - x in copied creature spell (copy x)
577         // - copied spells enters as tokens and it hasn't ETB, see rules below
578 
579         // 0/0
580         // Capricopian enters the battlefield with X +1/+1 counters on it.
581         addCard(Zone.HAND, playerA, "Capricopian", 1); // {X}{G}
582         addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
583         //
584         // Grenzo, Dungeon Warden enters the battlefield with X +1/+1 counters on it.
585         addCard(Zone.HAND, playerA, "Grenzo, Dungeon Warden", 1);// {X}{B}{R}
586         addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
587         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
588         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
589         //
590         // Copy target creature spell you control, except it isn't legendary if the spell is legendary.
591         // (A copy of a creature spell becomes a token.)
592         addCard(Zone.HAND, playerA, "Double Major", 2); // {G}{U}
593         addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
594         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
595 
596         // 1. Capricopian
597         // cast and put on stack
598         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 3);
599         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Capricopian");
600         setChoice(playerA, "X=2");
601         // copy of spell
602         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 1);
603         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 1);
604         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Double Major", "Capricopian", "Capricopian");
605 
606         // 2. Grenzo, Dungeon Warden
607         // cast and put on stack
608         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
609         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 2);
610         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}", 1);
611         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {R}", 1);
612         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grenzo, Dungeon Warden");
613         setChoice(playerA, "X=2");
614         // copy of spell
615         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 1);
616         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 1);
617         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Double Major", "Grenzo, Dungeon Warden", "Grenzo, Dungeon Warden");
618 
619         // ETB triggers will not trigger here due not normal cast. From rules:
620         // - The token that a resolving copy of a spell becomes isn’t said to have been “created.” (2021-04-16)
621         // - A nontoken permanent “enters the battlefield” when it’s moved onto the battlefield from another zone.
622         //   A token “enters the battlefield” when it’s created. See rules 403.3, 603.6a, 603.6d, and 614.12.
623         //
624         // So both copies enters without counters:
625         // - Capricopian copy must die
626         // - Grenzo, Dungeon Warden must have default PT
627 
628         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
629         checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Capricopian", 1); // copy dies
630         checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grenzo, Dungeon Warden", 2);
631 
632         setStrictChooseMode(true);
633         setStopAt(1, PhaseStep.END_TURN);
634         execute();
635 
636         // counters checks
637         int originalCounters = currentGame.getBattlefield().getAllActivePermanents().stream()
638                 .filter(p -> p.getName().equals("Grenzo, Dungeon Warden"))
639                 .filter(p -> !p.isCopy())
640                 .mapToInt(p -> p.getCounters(currentGame).getCount(CounterType.P1P1))
641                 .sum();
642         int copyCounters = currentGame.getBattlefield().getAllActivePermanents().stream()
643                 .filter(p -> p.getName().equals("Grenzo, Dungeon Warden"))
644                 .filter(p -> p.isCopy())
645                 .mapToInt(p -> p.getCounters(currentGame).getCount(CounterType.P1P1))
646                 .sum();
647         Assert.assertEquals("original grenzo must have 2x counters", 2, originalCounters);
648         Assert.assertEquals("copied grenzo must have 0x counters", 0, copyCounters);
649     }

779     private void abilitySourceMustBeSame(Card card, String infoPrefix) {
780         Set<UUID> partIds = CardUtil.getObjectParts(card);
781 
782         card.getAbilities(currentGame).forEach(ability -> {
783             // ability can refs to part or main card only
784             if (!partIds.contains(ability.getSourceId())) {
785                 if (ability instanceof MageSingleton) {
786                     // sourceId don't work with MageSingleton abilities
787                     return;
788                 }
789 
790                 Assert.fail(infoPrefix + " - " + "ability source must be same: " + ability);
791             }
792         });
793     }

801     private void cardsMustHaveSameZoneAndZCC(Card originalCard, Card copiedCard, String infoPrefix) {
802         // zcc and zone are not copied, so you don't need it here yet
803         Zone zone1 = currentGame.getState().getZone(originalCard.getId());
804         Zone zone2 = currentGame.getState().getZone(copiedCard.getId());
805         int zcc1 = currentGame.getState().getZoneChangeCounter(originalCard.getId());
806         int zcc2 = currentGame.getState().getZoneChangeCounter(copiedCard.getId());
807         if (zone1 != zone2 || zcc1 != zcc2) {
808             Assert.fail(infoPrefix + " - " + "cards must have same zone and zcc: " + zcc1 + " - " + zone1 + " != " + zcc2 + " - " + zone2);
809         }

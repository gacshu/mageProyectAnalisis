15      @Test
16      public void testCopyCreature() {
17          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
18          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
19  
20          // You may have Phyrexian Metamorph enter the battlefield as a copy of any artifact or creature on the battlefield, except it's an artifact in addition to its other types.
21          addCard(Zone.HAND, playerA, "Phyrexian Metamorph"); // {3}{U/P}
22          addCard(Zone.HAND, playerA, "Cloudshift");
23  
24          //Flying
25          // Vanishing 3 (This permanent enters the battlefield with three time counters on it. At the beginning of your upkeep, remove a time counter from it. When the last is removed, sacrifice it.)
26          // When Aven Riftwatcher enters the battlefield or leaves the battlefield, you gain 2 life.
27          addCard(Zone.BATTLEFIELD, playerB, "Aven Riftwatcher");  // 2/3
28  
29          // When Ponyback Brigade enters the battlefield or is turned face up, put three 1/1 red Goblin creature tokens onto the battlefield.
30          addCard(Zone.BATTLEFIELD, playerB, "Ponyback Brigade");  // 2/2
31  
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Phyrexian Metamorph");
33          setChoice(playerA, "Aven Riftwatcher");
34  
35          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Cloudshift", "Aven Riftwatcher");
36          setChoice(playerA, "Ponyback Brigade");
37  
38          setStopAt(1, PhaseStep.END_TURN);
39          execute();
40  
41          assertLife(playerA, 22);
42          assertLife(playerB, 20);
43  
44          assertGraveyardCount(playerA, "Cloudshift", 1);
45  
46          assertPermanentCount(playerA, "Ponyback Brigade", 1);
47          assertPermanentCount(playerA, "Goblin Token", 3);
48  
49      }

51      /**
52       * An opponent cast Phyrexian Metamorph and cloned another opponent's
53       * Maelstrom Wanderer(his Commander). The first opponent then dealt combat
54       * damage with Brago, King Eternal and chose to flicker several permanents,
55       * including the Phyrexian Metamorph/Maelstrom Wanderer, but that player was
56       * not able to choose a new creature to clone when the Phyrexian Metamorph
57       * re-entered the battlefield.
58       */
59      @Test
60      public void testFlickerWithBrago() {
61          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
62  
63          // You may have Phyrexian Metamorph enter the battlefield as a copy of any artifact or creature on the battlefield, except it's an artifact in addition to its other types.
64          addCard(Zone.HAND, playerA, "Phyrexian Metamorph"); // {3}{U/P}
65  
66          // Flying
67          // When Brago, King Eternal deals combat damage to a player, exile any number of target nonland permanents you control, then return those cards to the battlefield under their owner's control.
68          addCard(Zone.BATTLEFIELD, playerA, "Brago, King Eternal"); // 2/4
69  
70          // Creatures you control have haste.
71          // Cascade, cascade
72          addCard(Zone.BATTLEFIELD, playerB, "Maelstrom Wanderer");  // 7/5
73          // When Ponyback Brigade enters the battlefield or is turned face up, put three 1/1 red Goblin creature tokens onto the battlefield.
74          addCard(Zone.BATTLEFIELD, playerB, "Ponyback Brigade");  // 2/2
75  
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Phyrexian Metamorph");
77          setChoice(playerA, "Maelstrom Wanderer");
78  
79          attack(3, playerA, "Brago, King Eternal");
80          addTarget(playerA, "Maelstrom Wanderer");
81          setChoice(playerA, "Ponyback Brigade");
82  
83          setStopAt(3, PhaseStep.END_COMBAT);
84          execute();
85  
86          assertLife(playerA, 18);
87          assertLife(playerB, 18);
88  
89          assertPermanentCount(playerA, "Ponyback Brigade", 1);
90          assertPermanentCount(playerA, "Goblin Token", 3);
91  
92      }

183     /**
184      * I cast Show and Tell, and put Sheoldred, Whispering One into play and my
185      * opponent put Phyrexian Metamorph into play and they were able to clone my
186      * Sheoldred, Whispering One.
187      *
188      * 6/1/2011 If Phyrexian Metamorph somehow enters the battlefield at the
189      * same time as another permanent (due to Mass Polymorph or Liliana Vess's
190      * third ability, for example), Phyrexian Metamorph can't become a copy of
191      * that permanent. You may only choose a permanent that's already on the
192      * battlefield.
193      *
194      * 400.6. If an object would move from one zone to another, determine what
195      * event is moving the object. If the object is moving to a public zone, all
196      * players look at it to see if it has any abilities that would affect the
197      * move. Then any appropriate replacement effects, whether they come from
198      * that object or from elsewhere, are applied to that event. If any effects
199      * or rules try to do two or more contradictory or mutually exclusive things
200      * to a particular object, that object's controller -- or its owner if it
201      * has no controller -- chooses which effect to apply, and what that effect
202      * does. (Note that multiple instances of the same thing may be mutually
203      * exclusive; for example, two simultaneous "destroy" effects.) Then the
204      * event moves the object.
205      */
206     @Test
207     public void testShowAndTell() {
208         addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
209 
210         // Each player may put an artifact, creature, enchantment, or land card from their hand onto the battlefield.
211         addCard(Zone.HAND, playerA, "Show and Tell"); // SORCERY {2}{U}
212 
213         // Swampwalk
214         // At the beginning of your upkeep, return target creature card from your graveyard to the battlefield.
215         // At the beginning of each opponent's upkeep, that player sacrifices a creature.
216         addCard(Zone.HAND, playerA, "Sheoldred, Whispering One");
217 
218         addCard(Zone.HAND, playerB, "Phyrexian Metamorph");
219 
220         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Show and Tell");
221 
222         setStopAt(1, PhaseStep.BEGIN_COMBAT);
223         execute();
224 
225         assertPermanentCount(playerA, "Sheoldred, Whispering One", 1);
226         assertPermanentCount(playerB, "Sheoldred, Whispering One", 0);
227 
228         assertGraveyardCount(playerB, "Phyrexian Metamorph", 1);
229 
230     }

16      /**
17       * Check that opponent gets a copy of Lightning Bolt
18       */
19      @Test
20      public void testTransform() {
21          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
22          // Whenever a player casts an instant or sorcery spell, each other player copies that spell.
23          // Each of those players may choose new targets for their copy.
24          addCard(Zone.BATTLEFIELD, playerA, "Hive Mind", 1);
25          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
26          setChoice(playerB, true);
27          addTarget(playerB, playerA);
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
30          setStopAt(1, PhaseStep.BEGIN_COMBAT);
31          execute();
32  
33          assertLife(playerB, 17);
34          assertLife(playerA, 17);
35      }

37      /**
38       * The Amulet Bloom deck in Modern has a few wincons, one of them being that
39       * the Bloom player resolves a Hive Mind, then casts a Pact. The Hive Mind
40       * copies it so the opponent also casts a Pact. If the opposing player has a
41       * Chalice set on ZERO, it will counter both copies which it should NOT DO
42       * because Hive Mind puts a copy onto the stack, not 'cast'. So while the
43       * Bloom player's copy is countered, the opponent will still cast and have
44       * to pay during their upkeep.
45       */
46      @Test
47      public void testChaliceOfTtheVoid() {
48          // Whenever a player casts an instant or sorcery spell, each other player copies that spell.
49          // Each of those players may choose new targets for their copy.
50          addCard(Zone.BATTLEFIELD, playerA, "Hive Mind", 1);
51          // Create a 4/4 red Giant creature token onto the battlefield.
52          // At the beginning of your next upkeep, pay {4}{R}. If you don't, you lose the game.
53          addCard(Zone.HAND, playerA, "Pact of the Titan", 1);
54          // Chalice of the Void enters the battlefield with X charge counters on it.
55          // Whenever a player casts a spell with converted mana cost equal to the number of charge counters on Chalice of the Void, counter that spell.
56          addCard(Zone.BATTLEFIELD, playerB, "Chalice of the Void", 1);
57  
58          setChoice(playerB, true);
59  
60          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Pact of the Titan");
61  
62          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
63          execute();
64  
65          assertGraveyardCount(playerA, "Pact of the Titan", 1);
66          assertPermanentCount(playerA, "Giant Token", 0); // was countered by Chalice
67          assertPermanentCount(playerB, "Giant Token", 1); // was not countered by Chalice because it was not cast
68          assertWonTheGame(playerA);
69          assertLostTheGame(playerB);
70          assertLife(playerB, 20);
71          assertLife(playerA, 20);
72      }

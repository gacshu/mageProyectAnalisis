16      /**
17       * Unesh, Criosphinx Sovereign did not have his ETB effect trigger when he
18       * had a token copy of himself made through casting Hour of Eternity. I
19       * think there was another creature too that didn't get the ETB effect
20       * either.
21       */
22      @Test
23      @Ignore
24      public void testTokenTriggeresETBEffect() {
25          // Flying
26          // Sphinx spells you cast cost {2} less to cast.
27          // Whenever Unesh, Criosphinx Sovereign or another Sphinx enters the battlefield
28          // under your control, reveal the top four cards of your library. An opponent seperates
29          // those cards into two piles. Put one pile into your hand and the other into your graveyard.
30          addCard(Zone.GRAVEYARD, playerA, "Unesh, Criosphinx Sovereign", 1); // Sphinx 4/4
31  
32          // Exile X target creature cards from your graveyard. For each card exiled this way,
33          // create a token that's a copy of that card, except it's a 4/4 black Zombie.
34          addCard(Zone.HAND, playerA, "Hour of Eternity"); // Sorcery {X}{X}{U}{U}{U}
35          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
36  
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hour of Eternity", "Unesh, Criosphinx Sovereign");
38          setChoice(playerA, "X=1");
39  
40          setStopAt(1, PhaseStep.BEGIN_COMBAT);
41          execute();
42  
43          assertLife(playerA, 20);
44          assertLife(playerB, 20);
45  
46          assertGraveyardCount(playerA, "Hour of Eternity", 1);
47          assertPermanentCount(playerA, "Unesh, Criosphinx Sovereign", 1);
48          assertGraveyardCount(playerA, "Unesh, Criosphinx Sovereign", 0);
49          assertExileCount(playerA, "Unesh, Criosphinx Sovereign", 1);
50  
51          assertHandCount(playerA, 3);
52          assertGraveyardCount(playerA, 2);
53      }

55      /* https://github.com/magefree/mage/issues/5904
56      I had a Faerie Artisans on the battlefield. My opponent played a Thrashing Brontodon.
57      In response to the Faerie Artisans ability, my opponent sacrificed the Brontodon to destroy an artifact.
58      When the Faerie Artisans trigger resolved, no token was created. */
59      @Test
60      public void testFaerieArtisans() {
61          // Flying
62          // Whenever a nontoken creature enters the battlefield under an opponent's control,
63          // create a token that's a copy of that creature except it's an artifact in addition to its other types.
64          // Then exile all other tokens created with Faerie Artisans.
65          addCard(Zone.BATTLEFIELD, playerA, "Faerie Artisans", 1); // Creature {3}{U}
66          addCard(Zone.BATTLEFIELD, playerA, "Alpha Myr", 1); // Artifact creature 2/1
67  
68          // {1}, Sacrifice Thrashing Brontodon: Destroy target artifact or enchantment.
69          addCard(Zone.HAND, playerB, "Thrashing Brontodon"); // Creature {1}{G}{G}
70          addCard(Zone.BATTLEFIELD, playerB, "Forest", 4);
71  
72          setStrictChooseMode(true);
73  
74          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Thrashing Brontodon", true);
75          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{1}, Sacrifice", "Alpha Myr");
76  
77          setStopAt(2, PhaseStep.BEGIN_COMBAT);
78          execute();
79  
80          assertLife(playerA, 20);
81          assertLife(playerB, 20);
82  
83          assertGraveyardCount(playerB, "Thrashing Brontodon", 1);
84          assertGraveyardCount(playerA, "Alpha Myr", 1);
85          assertPermanentCount(playerA, "Thrashing Brontodon", 1);
86          assertType("Thrashing Brontodon", CardType.ARTIFACT, true);
87      }

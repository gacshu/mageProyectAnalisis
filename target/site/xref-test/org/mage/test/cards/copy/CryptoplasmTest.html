30      /**
31       * I have a Cryptoplasm in play, currently copying a Sigiled Paladin, and I
32       * enchant it with a Followed Footsteps. Next turn the aura triggers (the
33       * Crypto is still copying the same creature) and places a token on the
34       * battlefield, except the token is an untransformed Cryptoplasm, when it
35       * should be a Sigiled Paladin with Cryptoplasm's ability (as per rule
36       * 706.3), since that's what the enchanted creature currently is.
37       *
38       * 6/1/2011 If another creature becomes a copy of Cryptoplasm, it will
39       * become a copy of whatever Cryptoplasm is currently copying (if anything),
40       * plus it will have Cryptoplasm's triggered ability.
41       */
42      @Test
43      public void testFollowedFootsteps() {
44          // First strike
45          // Exalted (Whenever a creature you control attacks alone, that creature gets +1/+1 until end of turn.)
46          addCard(Zone.BATTLEFIELD, playerA, "Sigiled Paladin", 1); //  {W}{W}
47  
48          addCard(Zone.BATTLEFIELD, playerB, "Island", 5);
49          // Enchant creature
50          // At the beginning of your upkeep, put a token that's a copy of enchanted creature onto the battlefield.
51          addCard(Zone.HAND, playerB, "Followed Footsteps", 1); // {3}{U}{U}
52          // At the beginning of your upkeep, you may have Cryptoplasm become a copy of another target creature. If you do, Cryptoplasm gains this ability.
53          addCard(Zone.BATTLEFIELD, playerB, "Cryptoplasm", 1); // {1}{U}{U}
54  
55          // turn 2 - prepare (crypto to paladin, footsteps to crypto)
56          // crypto:  copy as paladin on upkeep
57          setChoice(playerB, true);
58          addTarget(playerB, "Sigiled Paladin");
59          // footsteps: enchant copy of paladin (crypto)
60          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Followed Footsteps");
61          addTarget(playerB, "Sigiled Paladin[only copy]");
62  
63          // turn 4 - ignore crypto ask for new copy
64          setChoice(playerB, false);
65  
66          setStopAt(4, PhaseStep.END_TURN);
67          execute();
68  
69          assertPermanentCount(playerB, "Followed Footsteps", 1);
70          assertPermanentCount(playerB, "Cryptoplasm", 0); // it's a copy
71          assertPermanentCount(playerB, "Sigiled Paladin", 2); // crypto as copy + footstep token as copy
72          assertPermanentCount(playerA, "Sigiled Paladin", 1); // original
73      }

75      /**
76       * I'm at 8 life, opponent (AI) is at 21. I have a Cryptoplasm currently
77       * copying my opponent's Divinity of Pride, the Crypto also has my Followed
78       * Footsteps attached to it. Additionally, I have another Cryptoplasm on the
79       * battlefield also copying the same Divinity, and a Clever Impersonator
80       * that ETB copying the first Cryptoplasm and is currently also a copy of
81       * the Divinity.
82       *
83       * Opponent attacks with their only Divinity of Pride (4/4) and a Serra
84       * Avenger (3/3). I block the Divinity with two of my Divinity copies (the
85       * Clever Impersonator and unenchanted Cryptoplasm) and the Avenger with the
86       * enchanted Divinity (originally a Cryptoplasm). My opponent's Divinity
87       * kills my two copies and dies, and then their Avenger dies and kills the
88       * Divinity blocking it, also sending my Followed Footsteps down with it.
89       *
90       * How does any of that add up? Not only should their Divinity only kill one
91       * of mine since it was a 4/4 and only becomes an 8/8 after dealing its
92       * damage (at which point it should be too late to go back and say the 4
93       * damage are now 8, since it was that exact damage that put them at 25
94       * life), but even more confusing is how the Serra Avenger, which is a 3/3,
95       * somehow kills my 4/4 that had suffered no other damage that turn.
96       *
97       * No other permanents in play at that moment had any influence in this
98       * either, they were only basic lands and a couple of creatures with no
99       * relevant abilities.
100      *
101      * I won't put aside me completely missing something here, but I really
102      * can't see any other explanation to this other than a game bug.
103      */
104     @Test
105     public void testDamageLifelink() {
106         setLife(playerA, 21);
107         setLife(playerB, 8);
108         // First strike
109         // Exalted (Whenever a creature you control attacks alone, that creature gets +1/+1 until end of turn.)
110         addCard(Zone.BATTLEFIELD, playerA, "Divinity of Pride", 1); //  {W}{W}
111 
112         addCard(Zone.BATTLEFIELD, playerB, "Island", 5);
113         // At the beginning of your upkeep, you may have Cryptoplasm become a copy of another target creature. If you do, Cryptoplasm gains this ability.
114         addCard(Zone.BATTLEFIELD, playerB, "Cryptoplasm", 2); // {1}{U}{U}
115         addTarget(playerB, "Divinity of Pride");
116         addTarget(playerB, "Divinity of Pride");
117 
118         attack(3, playerA, "Divinity of Pride");
119         block(3, playerB, "Divinity of Pride:0", "Divinity of Pride");
120         block(3, playerB, "Divinity of Pride:1", "Divinity of Pride");
121 
122         setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
123         execute();
124 
125         assertPermanentCount(playerB, "Cryptoplasm", 0);
126 
127         assertPermanentCount(playerA, "Divinity of Pride", 0);
128         assertPermanentCount(playerB, "Divinity of Pride", 1);
129 
130         assertLife(playerB, 16);
131         assertLife(playerA, 25);
132     }

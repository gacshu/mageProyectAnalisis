15      /**
16       * My opponent ran into an issue with Priest of the Blood Rite being copied
17       * with Flameshadow Conjuring.
18       * Their copy was made and removed correctly at the end of the turn,
19       * but the "lose two life a turn" trigger still happened twice.
20       *
21       * TODO: Seems like there are too much triggered abilities in
22       *       TriggeredAbilities as the TriggeredAbilities get removed from
23       *       PlayerImpl.removeFromBattlefield()
24       */
25      @Test
26      public void testCopyAndItsEffectsRemoved() {
27          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
28          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
29          // Whenever a nontoken creature enters the battlefield under your control, you may pay {R}. If you do, put a token onto the battlefield that's a copy of that creature.
30          // That token gains haste. Exile it at the beginning of the next end step.
31          addCard(Zone.BATTLEFIELD, playerA, "Flameshadow Conjuring", 1);
32  
33          // When Priest of the Blood Rite enters the battlefield, put a 5/5 black Demon creature token with flying onto the battlefield.
34          // At the beginning of your upkeep, you lose 2 life.
35          addCard(Zone.HAND, playerA, "Priest of the Blood Rite", 1); // {3}{B}{B}
36  
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Priest of the Blood Rite");
38          setStopAt(3, PhaseStep.UPKEEP);
39          execute();
40  
41          assertPermanentCount(playerA, "Priest of the Blood Rite", 1);
42          assertPermanentCount(playerA, "Demon Token", 2);
43  
44          assertLife(playerB, 20);
45          assertLife(playerA, 18);
46      }

48      /**
49       * I created a token copy of Wurmcoil Engine and sacrificed it.
50       * This gave me 4 tokens.
51       */
52      @Test
53      public void testWurmcoilEngine() {
54          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 7);
55          // Whenever a nontoken creature enters the battlefield under your control, you may pay {R}. If you do, put a token onto the battlefield that's a copy of that creature.
56          // That token gains haste. Exile it at the beginning of the next end step.
57          addCard(Zone.BATTLEFIELD, playerA, "Flameshadow Conjuring", 1);
58          // Deathtouch, lifelink
59          // When Wurmcoil Engine dies, put a 3/3 colorless Wurm artifact creature token with deathtouch and a 3/3 colorless Wurm artifact creature token with lifelink onto the battlefield.
60          addCard(Zone.HAND, playerA, "Wurmcoil Engine", 1); // 6/6 - {6}
61  
62          addCard(Zone.BATTLEFIELD, playerB, "Plains", 3);
63          // Destroy target attacking creature.
64          addCard(Zone.HAND, playerB, "Kill Shot", 1); // {2}{U}
65  
66          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Wurmcoil Engine");
67          setChoice(playerA, true);
68  
69          attack(1, playerA, "Wurmcoil Engine");
70  
71          castSpell(1, PhaseStep.END_COMBAT, playerB, "Kill Shot", "Wurmcoil Engine");
72  
73          setStopAt(1, PhaseStep.END_TURN);
74          execute();
75  
76          assertGraveyardCount(playerB, "Kill Shot", 1);
77          assertPermanentCount(playerA, "Wurmcoil Engine", 1);
78          assertLife(playerB, 14);
79          assertLife(playerA, 26);
80  
81          assertPermanentCount(playerA, "Phyrexian Wurm Token", 2);
82      }

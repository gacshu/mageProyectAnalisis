59          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 3);
60          // Flamebreak deals 3 damage to each creature without flying and each player. Creatures dealt damage this way can't be regenerated this turn.
61          addCard(Zone.HAND, playerB, "Flamebreak");
62  
63          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Create a token that's a copy of target nonlegendary creature you control, except it has haste. Sacrifice it at the beginning of the next end step.");
64  
65          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Flamebreak");
66          setStopAt(2, PhaseStep.END_TURN);
67          execute();

69          assertLife(playerA, 17);

81      /**
82       * Kiki-Jiki, Mirror Breaker creates a copy of Humble Defector, activate
83       * Humble defector, token gets sacrificed while under opponents control.
84       */
85      @Test
86      public void testTokenNotSacrificedIfNotControlled() {
87          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
88          // Tap target creature you don't control.
89          // Overload {3}{U}
90          addCard(Zone.HAND, playerA, "Blustersquall", 1);
91  
92          addCard(Zone.BATTLEFIELD, playerB, "Kiki-Jiki, Mirror Breaker", 1);
93          // {T}: Draw two cards. Target opponent gains control of Humble Defector. Activate this ability only during your turn.
94          addCard(Zone.BATTLEFIELD, playerB, "Humble Defector", 1);
95  
96          castSpell(2, PhaseStep.UPKEEP, playerA, "Blustersquall", "Humble Defector"); // Tap nontoken Defector so only the Token can be used later
97  
98          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{T}: Create a token that's a copy of target nonlegendary creature you control, except it has haste. Sacrifice it at the beginning of the next end step.");
99  
100         activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "{T}: Draw two cards. Target opponent gains control");
101 
102         setStopAt(3, PhaseStep.UPKEEP);
103         execute();
104 
105         assertLife(playerA, 20);
106         assertLife(playerB, 20);
107 
108         assertHandCount(playerB, 3); // normal 1 draw of turn two + 2 from Defector
109 
110         assertGraveyardCount(playerA, "Blustersquall", 1);
111         assertPermanentCount(playerB, "Humble Defector", 1);
112         assertPermanentCount(playerA, "Humble Defector", 1);
113 
114     }

116     /**
117      *
118      * Kiki-Jiki, Mirror Breaker tokens are not entering with haste...
119      *
120      * I just tried to reproduce this but I was not able--
121      *
122      * In the game in question I used a card (Body Double) to target a card in a
123      * graveyard (don't remember the name) and copy it
124      *
125      * I then used Kiki to copy Body Double that copied a graveyard card of my
126      * opponent---that copy did not have haste...
127      */
128     @Test
129     public void testCopyBodyDouble() {
130         addCard(Zone.GRAVEYARD, playerA, "Silvercoat Lion", 1);
131 
132         addCard(Zone.BATTLEFIELD, playerB, "Island", 5);
133         addCard(Zone.BATTLEFIELD, playerB, "Kiki-Jiki, Mirror Breaker", 1);
134         // You may have Body Double enter the battlefield as a copy of any creature card in a graveyard.
135         addCard(Zone.HAND, playerB, "Body Double", 1); // {4}{U}
136 
137         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Body Double", true);
138         setChoice(playerB, "Silvercoat Lion");
139 
140         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{T}: Create a token that's a copy of target nonlegendary creature you control, except it has haste. Sacrifice it at the beginning of the next end step.");
141 
142         attack(2, playerB, "Silvercoat Lion");
143         setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
144         execute();
145 
146         assertGraveyardCount(playerA, "Silvercoat Lion", 1);
147         assertPermanentCount(playerB, "Silvercoat Lion", 2); // one from Body Double and one from Kiki
148 
149         Permanent kikiCopy = null;
150         for (Permanent permanent : currentGame.getState().getBattlefield().getAllActivePermanents(StaticFilters.FILTER_PERMANENT_CREATURE, currentGame)) {
151             if (permanent.getName().equals("Silvercoat Lion") && (permanent instanceof PermanentToken)) {
152                 kikiCopy = permanent;
153                 break;
154             }
155         }
156         if (kikiCopy != null) {
157             Assert.assertEquals("Has to have haste", kikiCopy.getAbilities(currentGame).containsClass(HasteAbility.class), true);
158         } else {
159             Assert.assertEquals("Silvercoat Lion copied by Kiki is missing", kikiCopy != null, true);
160         }
161 
162         assertLife(playerA, 18);
163         assertLife(playerB, 20);
164 
165     }

15      /**
16       * Eldrazi Mimic also did not copy the last known P/T of a Drowner of Hope
17       * that was killed with the trigger on the stack.
18       *
19       */
20      @Test
21      public void testCopyIfPermanentIsGone() {
22          // Devoid (This card has no color.)
23          // Flying
24          // Whenever you cast a colorless spell, target opponent exiles the top card of their library.
25          addCard(Zone.HAND, playerA, "Thought Harvester", 1); // {3}{U} 2/4
26          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
27          // Whenever another colorless creature enters the battlefield under your control, you may have the base power and toughness of Eldrazi Mimic
28          // become that creature's power and toughness until end of turn.
29          addCard(Zone.BATTLEFIELD, playerA, "Eldrazi Mimic", 1); // 2/1
30  
31          // Destroy target nonartifact, nonblack creature. It can't be regenerated.
32          addCard(Zone.HAND, playerB, "Terror", 1); // {1}{B}
33          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
34  
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Thought Harvester");
36          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Terror", "Thought Harvester");
38  
39          setStopAt(1, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          assertGraveyardCount(playerB, "Terror", 1);
43          assertGraveyardCount(playerA, "Thought Harvester", 1);
44  
45          assertPermanentCount(playerA, "Eldrazi Mimic", 1);
46          assertPowerToughness(playerA, "Eldrazi Mimic", 2, 4);
47  
48      }
49  
50      /**
51       * Eldrazi Mimic ability to change it's health and power on another
52       * creatures entering the battlefield doesn't work after update.
53       */
54      @Test
55      public void testNormalCopy() {
56          addCard(Zone.HAND, playerA, "Composite Golem", 1); // 4/4
57          addCard(Zone.BATTLEFIELD, playerA, "Plains", 6);
58          // Whenever another colorless creature enters the battlefield under your control, you may have the base power and toughness of Eldrazi Mimic
59          // become that creature's power and toughness until end of turn.
60          addCard(Zone.BATTLEFIELD, playerA, "Eldrazi Mimic", 1); // 2/1
61  
62          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Composite Golem");
63          setChoice(playerA, true);
64          setStopAt(1, PhaseStep.BEGIN_COMBAT);
65          execute();
66  
67          assertPowerToughness(playerA, "Composite Golem", 4, 4);
68          assertPowerToughness(playerA, "Eldrazi Mimic", 4, 4);
69  
70      }

106     // copy Nightmare test, check that the P/T setting effect ends
107     // if the clone leaves battlefield
108     @Test
109     public void testCopyNightmare() {
110         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
111         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
112         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 6);
113         addCard(Zone.BATTLEFIELD, playerB, "Forest", 1);
114         // Target creature you control gets +1/+1 and gains hexproof until end of turn. (It can't be the target of spells or abilities your opponents control.)
115         addCard(Zone.HAND, playerB, "Ranger's Guile");
116 
117         addCard(Zone.HAND, playerA, "Clone");
118         // Return target nonland permanent to its owner's hand.
119         addCard(Zone.HAND, playerA, "Disperse");
120 
121         addCard(Zone.BATTLEFIELD, playerB, "Nightmare", 1);
122 
123         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Clone");
124         setChoice(playerA, "Nightmare");
125         castSpell(1, PhaseStep.BEGIN_COMBAT, playerB, "Ranger's Guile", "Nightmare");
126         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Disperse", "Nightmare");
127 
128         setStopAt(2, PhaseStep.UPKEEP);
129         execute();
130 
131         assertPermanentCount(playerB, "Nightmare", 1);
132         assertPowerToughness(playerB, "Nightmare", 6, 6);
133         assertPermanentCount(playerA, "Nightmare", 0);
134         assertHandCount(playerA, "Disperse", 0);
135 
136         FilterCard filter = new FilterCard();
137         filter.add(new NamePredicate("Clone"));
138         Card card = playerA.getHand().getCards(filter, currentGame).iterator().next();
139         if (card != null) {
140             Assert.assertEquals("Power has to be 0 because copy from nightmare P/T ability may no longer be applied", 0, card.getPower().getValue());
141         }
142 
143         Logger.getLogger(CloneTest.class).debug("EXISTING CONTINUOUS EFFECTS:");
144         for (ContinuousEffectsList effectsList : currentGame.getContinuousEffects().allEffectsLists) {
145             for (Object anEffectsList : effectsList) {
146                 ContinuousEffect effect = (ContinuousEffect) anEffectsList;
147                 Logger.getLogger(CloneTest.class).debug("- " + effect.toString());
148             }
149         }

152     /**
153      * When I Clone a creature, and I try to use Vesuvan Doppelganger to my
154      * cloned creature, The Vesuvan disappears and the creature is not created.
155      */
156     @Test
157     public void testCloneAndVesuvanDoppelganger() {
158         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
159         // You may have Clone enter the battlefield as a copy of any creature on the battlefield.
160         addCard(Zone.HAND, playerA, "Clone"); // Creature {3}{U}
161         // You may have Vesuvan Doppelganger enter the battlefield as a copy of any creature on the battlefield
162         // except it doesn't copy that creature's color and it gains "At the beginning of your upkeep,
163         // you may have this creature become a copy of target creature except it doesn't copy that creature's color.
164         // If you do, this creature gains this ability."
165         addCard(Zone.HAND, playerA, "Vesuvan Doppelganger"); // Creature  {3}{U}{U}
166 
167         addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
168         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
169         // {2}, {T} , Sacrifice a creature: Draw a card.
170         addCard(Zone.BATTLEFIELD, playerB, "Phyrexian Vault", 1);
171 
172         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Clone");
173         setChoice(playerA, "Silvercoat Lion");
174 
175         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{2}");
176 
177         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Vesuvan Doppelganger");
178         setChoice(playerA, "Silvercoat Lion");
179 
180         setStopAt(3, PhaseStep.BEGIN_COMBAT);
181         execute();
182 
183         assertGraveyardCount(playerB, "Silvercoat Lion", 1);
184         assertGraveyardCount(playerA, "Vesuvan Doppelganger", 0);
185 
186         assertPermanentCount(playerA, "Silvercoat Lion", 2);
187         assertPermanentCount(playerB, "Silvercoat Lion", 0);
188 
189         assertHandCount(playerB, 2);
190 
191         boolean whiteLion = false;
192         boolean blueLion = false;
193         for (Permanent permanent : currentGame.getBattlefield().getAllActivePermanents(StaticFilters.FILTER_PERMANENT_CREATURE, playerA.getId(), currentGame)) {
194             if (permanent.getColor(currentGame).isWhite()) {
195                 whiteLion = true;
196             }
197             if (permanent.getColor(currentGame).isBlue()) {
198                 blueLion = true;
199             }
200         }
201 
202         Assert.assertTrue("There should be a white and a blue Silvercoat Lion be on the battlefield", blueLion && whiteLion);
203     }

207         addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
208         addCard(Zone.HAND, playerA, "Adaptive Automaton");
209 
210         addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
211         addCard(Zone.HAND, playerB, "Clone");
212 
213         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Adaptive Automaton");
214         setChoice(playerA, "Elf");
215 
216         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Clone");
217         setChoice(playerB, true);
218         setChoice(playerB, "Adaptive Automaton");
219         setChoice(playerB, "Goblin");
220 
221         setStrictChooseMode(true);
222         setStopAt(2, PhaseStep.END_COMBAT);
223         execute();

228

230         Permanent clone = getPermanent("Adaptive Automaton", playerB);
231         Assert.assertFalse("The cloned Adaptive Automaton should not be as Elf", clone.hasSubtype(SubType.ELF, currentGame));
232         Assert.assertTrue("The cloned Adaptive Automaton should be a Goblin", clone.hasSubtype(SubType.GOBLIN, currentGame));

235     /**
236      * Cloning a face-down creature should produce a plain 2/2 creature #3582
237      *
238      * What I expected: Using a clone effect on a face-down creature should
239      * create another 2/2 creature.
240      *
241      * What I observed: The resulting token appears face-up as a copy of the
242      * face-down creature's front face, copying its power/toughness but not
243      * abilities.
244      *
245      * To reproduce: Play a Terastodon, then an Ixidron. Terastodon is turned
246      * face-down. Then play a clone creature (such as Progenitor Mimic). It
247      * appears as a face-up 9/9 Terastodon token, but without any ETB triggers.
248      * The next turn, when it copies itself, it creates a Terastodon that does
249      * have its ETB ability.
250      *
251      * https://blogs.magicjudges.org/articles/2014/09/16/morph-rules-problems-face-down-in-a-face-up-world/
252      * explains that cloning a face-down 2/2 creature should create another 2/2
253      * creature with no abilities (see Face-down Creatures and Clone).
254      */
255 
256     @Test
257     public void testCloningFaceDownCreature() {
258         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
259         // As Ixidron enters the battlefield, turn all other nontoken creatures face down.
260         // Ixidron's power and toughness are each equal to the number of face-down creatures on the battlefield.        
261         addCard(Zone.HAND, playerA, "Ixidron"); // Creature {3}{U}{U}
262 
263         // When Terastodon enters the battlefield, you may destroy up to three target noncreature permanents. 
264         // For each permanent put into a graveyard this way, its controller creates a 3/3 green Elephant creature token.
265         addCard(Zone.BATTLEFIELD, playerA, "Terastodon", 1); // Creature 9/9  {6}{G}{G}
266 
267         // You may have Clone enter the battlefield as a copy of any creature on the battlefield.        
268         addCard(Zone.HAND, playerA, "Clone"); // Creature {3}{U}
269 
270         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ixidron");
271 
272         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Clone");
273         setChoice(playerA, true);
274         setChoice(playerA, EmptyNames.FACE_DOWN_CREATURE.toString());
275 
276         setStrictChooseMode(true);
277         setStopAt(3, PhaseStep.BEGIN_COMBAT);
278         execute();
279 
280 
281         assertPermanentCount(playerA, "Terastodon", 0);
282         
283         assertPermanentCount(playerA, "Ixidron", 1);
284         assertPowerToughness(playerA, "Ixidron", 1, 1);
285 
286         assertPermanentCount(playerA, EmptyNames.FACE_DOWN_CREATURE.toString(), 2);
287         assertPowerToughness(playerA, EmptyNames.FACE_DOWN_CREATURE.toString(), 2, 2, Filter.ComparisonScope.All);
288     }

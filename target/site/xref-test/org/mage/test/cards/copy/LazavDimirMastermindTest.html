30          // Codex Shredder - Artifact
31          // {T}: Target player mills a card.
32          // {5}, {T}, Sacrifice Codex Shredder: Return target card from your graveyard to your hand.
33          addCard(Zone.BATTLEFIELD, playerA, "Codex Shredder", 1);

36          addCard(Zone.LIBRARY, playerB, "Assault Griffin", 5);

49          Assert.assertTrue("Lazav, Dimir Mastermind must have flying", lazav.getAbilities().contains(FlyingAbility.getInstance()));

54       */
55      @Test
56      public void testRatsHaveDeathtouch() {
57          addCard(Zone.BATTLEFIELD, playerA, "Lazav, Dimir Mastermind", 1);
58          addCard(Zone.BATTLEFIELD, playerA, "Gutter Skulk", 1);
59          addCard(Zone.BATTLEFIELD, playerA, "Codex Shredder", 1);
60  
61          // Whenever another nontoken creature dies, you may put a 1/1 black Rat creature token onto the battlefield.
62          // Rats you control have deathtouch.
63          addCard(Zone.LIBRARY, playerB, "Ogre Slumlord", 5);
64          skipInitShuffling();
65  
66          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Target player mills a card.", playerB);
67  
68          setStopAt(1, PhaseStep.END_TURN);
69          execute();
70  
71          assertPermanentCount(playerA, "Lazav, Dimir Mastermind", 1);
72          assertPowerToughness(playerA, "Lazav, Dimir Mastermind", 3, 3);
73          Permanent lazav = getPermanent("Lazav, Dimir Mastermind", playerA.getId());
74          Assert.assertTrue(lazav.hasSubtype(SubType.OGRE, currentGame));
75          Assert.assertTrue(lazav.hasSubtype(SubType.ROGUE, currentGame));
76  
77          Permanent gutterSkulk = getPermanent("Gutter Skulk", playerA.getId());

149     /**
150      * Tests old copy is discarded after reanmiation of Lazav
151      */
152     @Test
153     public void testCopyAfterReanimation() {
154         addCard(Zone.BATTLEFIELD, playerA, "Swamp");
155         // Put target creature card from a graveyard onto the battlefield under your control. You lose life equal to its converted mana cost.
156         addCard(Zone.HAND, playerA, "Reanimate");
157         addCard(Zone.BATTLEFIELD, playerA, "Lazav, Dimir Mastermind", 1);
158         // Codex Shredder - Artifact
159         // {T}: Target player mills a card.
160         // {5}, {T}, Sacrifice Codex Shredder: Return target card from your graveyard to your hand.
161         addCard(Zone.BATTLEFIELD, playerA, "Codex Shredder", 1);
162 
163         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 3);
164         // Flying 3/2
165         addCard(Zone.LIBRARY, playerB, "Assault Griffin", 1);
166         // Target opponent sacrifices a creature. You gain life equal to that creature's toughness.
167         addCard(Zone.HAND, playerB, "Tribute to Hunger");
168 
169         skipInitShuffling();
170 
171         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Target player mills a card.", playerB);
172         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
173 
174         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Tribute to Hunger");
175 
176         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Reanimate", "Lazav, Dimir Mastermind");
177 
178         setStopAt(1, PhaseStep.END_TURN);
179         execute();
180 
181         assertGraveyardCount(playerB, "Tribute to Hunger", 1);
182         assertGraveyardCount(playerA, "Reanimate", 1);
183 
184         assertLife(playerA, 16); // -4 from Reanmiate
185         assertLife(playerB, 22); // +3 from Tribute to Hunger because Lazav is 3/2
186 
187         assertPermanentCount(playerA, "Lazav, Dimir Mastermind", 1);
188         assertPowerToughness(playerA, "Lazav, Dimir Mastermind", 3, 3);
189         Permanent lazav = getPermanent("Lazav, Dimir Mastermind", playerA.getId());
190         Assert.assertFalse(lazav.hasSubtype(SubType.GRIFFIN, currentGame)); // no Griffin type
191         Assert.assertFalse("Lazav, Dimir Mastermind must have flying", lazav.getAbilities().contains(FlyingAbility.getInstance()));
192 
193 
194     }

20      /**
21       * Checking that enters the battlefield abilities of the copied creature
22       * card works.
23       */
24      @Test
25      public void testETBEffect() {
26          // {2}{R}, {T}: Create a token that's a copy of target creature card in your graveyard, except it's an artifact
27          // in addition to its other types. It gains haste. Sacrifice it at the beginning of the next end step.
28          addCard(Zone.BATTLEFIELD, playerA, "Feldon of the Third Path", 1);
29          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
30          //
31          // When Highway Robber enters the battlefield, target opponent loses 2 life and you gain 2 life.
32          addCard(Zone.GRAVEYARD, playerA, "Highway Robber", 1);
33          //
34          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
35          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
36  
37          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerA, "{2}{R}, {T}: Create a token", "Highway Robber");
38          addTarget(playerA, playerB); // opponent to loses 2 life
39          waitStackResolved(2, PhaseStep.PRECOMBAT_MAIN);
40          checkPermanentCount("must have token", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Highway Robber", 1);
41          checkPermanentCount("must have card", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Feldon of the Third Path", 1);
42  
43          // destroy token
44          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Highway Robber");
45          waitStackResolved(2, PhaseStep.PRECOMBAT_MAIN);
46          checkPermanentCount("must haven't token", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Highway Robber", 0);
47  
48          setStrictChooseMode(true);
49          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
50          execute();
51  
52          assertLife(playerA, 22); // +2 from Robber
53          assertLife(playerB, 18); // -2 from Robber
54  
55          // possible bug: triggers from destroyed permanents keeps in game state (e.g. 2 triggers in game state)
56          Assert.assertEquals("game state must have only 1 trigger from original card", 1, currentGame.getState().getTriggers().size());
57      }

59      @Test
60      public void testETB2Effect() {
61          // When Sepulchral Primordial enters the battlefield, for each opponent, you may put up to one
62          // target creature card from that player's graveyard onto the battlefield under your control.
63          addCard(Zone.GRAVEYARD, playerA, "Sepulchral Primordial", 1);
64          addCard(Zone.BATTLEFIELD, playerA, "Feldon of the Third Path", 1);
65          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
66  
67          addCard(Zone.GRAVEYARD, playerB, "Silvercoat Lion", 1);
68  
69          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerA,
70                  "{2}{R}, {T}: Create a token that's a copy of target creature card in your graveyard, except it's an artifact in addition to its other types. It gains haste. Sacrifice it at the beginning of the next end step.",
71                  "Sepulchral Primordial");
72          addTarget(playerA, "Silvercoat Lion"); // target for ETB Sepulchral Primordial
73  
74          setStrictChooseMode(true);
75          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
76          execute();
77  
78          assertPermanentCount(playerA, "Feldon of the Third Path", 1);
79          assertPermanentCount(playerA, "Sepulchral Primordial", 1);
80          assertPermanentCount(playerA, "Silvercoat Lion", 1);
81  
82          assertLife(playerA, 20);
83          assertLife(playerB, 20);
84  
85      }

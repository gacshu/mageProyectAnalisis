17      /**
18       * played a Vesuva on Glimmerpost and since when Vesuva becames Glimmerpost
19       * it is already on the battlefield the ability won't trigger. Then when i
20       * used Vesuva to copy an opponent's Dark Depth i got the land with the 10
21       * counters.
22       *
23       * If Dark Depth says "it enters the battlefield with 10 counter" but Vesuva
24       * is already on the field shouldnt have any counters or the problem was
25       * that Glimmerpost should have trigger?
26       */
27      @Test
28      public void testGlimmerpost() {
29          // When Glimmerpost enters the battlefield, you gain 1 life for each Locus on the battlefield.
30          // {T}: {1} Add.
31          addCard(Zone.HAND, playerA, "Glimmerpost", 1);
32          // You may have Vesuva enter the battlefield tapped as a copy of any land on the battlefield.
33          addCard(Zone.HAND, playerA, "Vesuva", 1);
34  
35          addCard(Zone.HAND, playerB, "Glimmerpost", 1);
36  
37          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Glimmerpost");
38          playLand(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Glimmerpost");
39          playLand(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Vesuva");
40          setChoice(playerA, "Glimmerpost");
41  
42          setStopAt(3, PhaseStep.BEGIN_COMBAT);
43          execute();
44  
45          assertPermanentCount(playerA, "Glimmerpost", 2);
46          assertPermanentCount(playerB, "Glimmerpost", 1);
47  
48          assertLife(playerA, 24); // 20 + 1 + 3
49          assertLife(playerB, 22); // 20 + 2
50      }
51  
52      @Test
53      public void testDarkDepth() {
54          // Dark Depths enters the battlefield with ten ice counters on it.
55          // {3}: Remove an ice counter from Dark Depths.
56          // When Dark Depths has no ice counters on it, sacrifice it. If you do, put a legendary 20/20 black Avatar creature token with flying and "This creature is indestructible" named Marit Lage onto the battlefield.
57          addCard(Zone.BATTLEFIELD, playerB, "Dark Depths", 1);
58  
59          // You may have Vesuva enter the battlefield tapped as a copy of any land on the battlefield.
60          addCard(Zone.HAND, playerA, "Vesuva", 1);
61  
62          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Vesuva");
63          setChoice(playerA, "Dark Depths");
64  
65          setStopAt(1, PhaseStep.BEGIN_COMBAT);
66          execute();
67  
68          assertPermanentCount(playerB, "Dark Depths", 1);
69          assertPermanentCount(playerA, "Dark Depths", 1);
70  
71          Permanent darkDepth = getPermanent("Dark Depths", playerA);
72          if (darkDepth != null) {
73              Assert.assertEquals(10, darkDepth.getCounters(currentGame).getCount("ice"));
74          }
75          assertTappedCount("Dark Depths", true, 1);
76      }

426     /**
427      * Action Game State 1 -----------------> Game State 2 (On 'field) (Move to
428      * GY) (In graveyard)
429      * <p>
430      * LTB abilities such as Persist are expceptional in that they trigger based
431      * on their existence and state of objects before the event (Game State 1,
432      * when the card is on the battlefield) rather than after (Game State 2,
433      * when the card is in the graveyard). It doesn't matter that the LTB
434      * ability doesn't exist in Game State 2. [CR 603.6d]
435      * <p>
436      * 603.6d Normally, objects that exist immediately after an event are
437      * checked to see if the event matched any trigger conditions. Continuous
438      * effects that exist at that time are used to determine what the trigger
439      * conditions are and what the objects involved in the event look like.
440      * However, some triggered abilities must be treated specially.
441      * Leaves-the-battlefield abilities, abilities that trigger when a permanent
442      * phases out, abilities that trigger when an object that all players can
443      * see is put into a hand or library, abilities that trigger specifically
444      * when an object becomes unattached, abilities that trigger when a player
445      * loses control of an object, and abilities that trigger when a player
446      * planeswalks away from a plane will trigger based on their existence, and
447      * the appearance of objects, prior to the event rather than afterward. The
448      * game has to look back in time to determine if these abilities trigger.
449      * <p>
450      * Example: Two creatures are on the battlefield along with an artifact that
451      * has the ability Whenever a creature dies, you gain 1 life. Someone
452      * plays a spell that destroys all artifacts, creatures, and enchantments.
453      * The artifact's ability triggers twice, even though the artifact goes to
454      * its owner's graveyard at the same time as the creatures.
455      */
456     @Test
457     public void testPersist() {
458         addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
459         // When Kitchen Finks enters the battlefield, you gain 2 life.
460         // Persist (When this creature dies, if it had no -1/-1 counters on it, return it to the battlefield under its owner's control with a -1/-1 counter on it.)
461         addCard(Zone.HAND, playerA, "Kitchen Finks");
462 
463         // Destroy target creature an opponent controls. Each other creature that player controls gets -2/-0 until end of turn.
464         addCard(Zone.HAND, playerA, "Public Execution");
465         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 6);
466 
467         addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
468 
469         // You may have Phantasmal Image enter the battlefield as a copy of any creature
470         // on the battlefield, except it's an Illusion in addition to its other types and
471         // it gains "When this creature becomes the target of a spell or ability, sacrifice it."
472         addCard(Zone.HAND, playerB, "Phantasmal Image");
473 
474         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Kitchen Finks");
475 
476         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Phantasmal Image"); // not targeted
477         setChoice(playerB, "Kitchen Finks");
478 
479         castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Public Execution", "Kitchen Finks");
480         setChoice(playerB, "Kitchen Finks");
481 
482         setStopAt(2, PhaseStep.END_TURN);
483         execute();
484 
485         assertGraveyardCount(playerA, "Public Execution", 1);
486 
487         assertLife(playerA, 22);
488         assertLife(playerB, 24);
489 
490         assertPermanentCount(playerA, "Kitchen Finks", 1);
491 
492         assertHandCount(playerB, "Phantasmal Image", 0);
493         assertGraveyardCount(playerB, "Phantasmal Image", 0);
494         assertPermanentCount(playerB, "Kitchen Finks", 1);
495         assertPowerToughness(playerB, "Kitchen Finks", 2, 1);
496     }

16  public class DeceiverOfFormTest extends CardTestPlayerBase {

18      /**
19       * When creatures copy the revealed creature, they do not return to their
20       * original state at the end of turn
21       *
22       */
23      @Test
24      public void testCopyEndsEndOfTurn() {
25          // Vigilance
26          addCard(Zone.LIBRARY, playerA, "Affa Protector", 1); // 1/4
27          // At the beginning of combat on your turn, reveal the top card of your library.
28          // If a creature card is revealed this way, you may have creatures you control other than Deceiver of Form becomes copies of that card until end of turn.
29          // You may put that card on the bottom of your library.
30          addCard(Zone.BATTLEFIELD, playerA, "Deceiver of Form", 1);
31          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 3);
32  
33          attack(1, playerA, "Deceiver of Form");
34          attack(1, playerA, "Affa Protector");
35          attack(1, playerA, "Affa Protector");
36          attack(1, playerA, "Affa Protector");
37  
38          skipInitShuffling();
39  
40          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
41          execute();
42  
43          assertPermanentCount(playerA, "Deceiver of Form", 1);
44          assertPermanentCount(playerA, "Silvercoat Lion", 3);
45          assertPowerToughness(playerA, "Silvercoat Lion", 2, 2, Filter.ComparisonScope.All);
46  
47          assertLife(playerA, 20); // +2 from Robber
48          assertLife(playerB, 9);
49  
50      }

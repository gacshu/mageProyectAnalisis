10  /**
11   *
12   * @author LevelX2
13   */
14  public class HelmOfTheHostTest extends CardTestPlayerBase {
15  
16      /**
17       * If you animate Gideon of the Trials and equip it with Helm of the Host
18       * the nonlegendary copies can't become creatures with the 0 ability. You
19       * can activate it just fine (and it gets put on the stack) but nothing
20       * happens and you can't use another ability.
21       */
22      @Test
23      public void testCopyPlaneswalker() {
24          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
25          //Starting Loyalty: 3
26          // +1: Until your next turn, prevent all damage target permanent would deal.
27          // 0: Until end of turn, Gideon of the Trials becomes a 4/4 Human Soldier creature with indestructible that's still a planeswalker. Prevent all damage that would be dealt to him this turn.
28          // 0: You get an emblem with "As long as you control a Gideon planeswalker, you can't lose the game and your opponent can't win the game."
29          addCard(Zone.BATTLEFIELD, playerA, "Gideon of the Trials", 1);
30          // At the beginning of combat on your turn, create a token that's a copy of equipped creature, except the token isn't legendary if equipped creature is legendary. That token gains haste.
31          // Equip {5}
32          addCard(Zone.BATTLEFIELD, playerA, "Helm of the Host", 1);
33  
34          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Until end of turn");
35          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
36          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip");
37  
38          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Until end of turn");
39          waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
40          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "0: Until end of turn");
41  
42          attack(3, playerA, "Gideon of the Trials");
43          attack(3, playerA, "Gideon of the Trials");
44  
45          setStopAt(4, PhaseStep.PRECOMBAT_MAIN);
46          execute();
47  
48          assertPermanentCount(playerA, "Gideon of the Trials", 2);
49  
50          assertCounterCount("Gideon of the Trials", CounterType.LOYALTY, 3);
51  
52          assertLife(playerB, 12);
53          assertLife(playerA, 20);
54      }
55  }

19      /**
20       * Tests boosting on being blocked
21       */
22      @Test
23      public void testNormal() {
24          addCard(Zone.BATTLEFIELD, playerA, "Elite Vanguard");
25          addCard(Zone.BATTLEFIELD, playerB, "Isao, Enlightened Bushi");
26  
27          attack(2, playerB, "Isao, Enlightened Bushi");
28          block(2, playerA, "Elite Vanguard", "Isao, Enlightened Bushi");
29  
30          setStopAt(2, PhaseStep.END_COMBAT);
31          execute();
32  
33          assertPowerToughness(playerB, "Isao, Enlightened Bushi", 4, 3);
34          assertPermanentCount(playerA, "Elite Vanguard", 0);
35      }
36  
37      /**
38       * Tests boosting on block
39       */
40      @Test
41      public void testBloodlord() {
42          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8);
43          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
44          // Bloodthirst 3 (If an opponent was dealt damage this turn, this creature enters the battlefield with 3 +1/+1 counters)
45          // Whenever you cast a Vampire creature spell, it gains bloodthirst 3
46          addCard(Zone.HAND, playerA, "Bloodlord of Vaasgoth"); // {3}{B}{B}
47          addCard(Zone.HAND, playerA, "Barony Vampire"); // 3/2  {2}{B}
48          addCard(Zone.HAND, playerA, "Lightning Bolt");
49  
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
51          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
52          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Bloodlord of Vaasgoth", true);
53          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Barony Vampire");
54  
55          setStopAt(1, PhaseStep.BEGIN_COMBAT);
56          execute();
57  
58          assertPermanentCount(playerA, "Barony Vampire", 1);
59          Permanent baron = getPermanent("Barony Vampire", playerA);
60          boolean bloodthirstFound = false;
61          for (Ability ability : baron.getAbilities()) {
62              if (ability instanceof BloodthirstAbility) {
63                  bloodthirstFound = true;
64                  break;
65              }
66          }
67          Assert.assertTrue("Baron Vampire is missing the bloodthirst ability", bloodthirstFound);
68          assertPermanentCount(playerA, "Bloodlord of Vaasgoth", 1);
69          assertPowerToughness(playerA, "Bloodlord of Vaasgoth", 6, 6);
70          assertPermanentCount(playerA, "Barony Vampire", 1);
71          assertPowerToughness(playerA, "Barony Vampire", 6, 5);
72      }

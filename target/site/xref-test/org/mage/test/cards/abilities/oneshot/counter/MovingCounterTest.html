15      /**
16       * I'm having an issue when using Bioshift to move only a portion of
17       * counters to another creature. When I attempt to do this, it moves all of
18       * the counters (and in some cases with my Simic deck) kills the creature.
19       */
20      @Test
21      public void testCantBeCounteredNormal() {
22          addCard(Zone.BATTLEFIELD, playerA, "Forest", 6);
23          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
24  
25          // Move any number of +1/+1 counters from target creature onto another target creature with the same controller.
26          addCard(Zone.HAND, playerA, "Bioshift", 1);
27  
28          // Protean Hydra enters the battlefield with X +1/+1 counters on it.
29          // If damage would be dealt to Protean Hydra, prevent that damage and remove that many +1/+1 counters from it.
30          // Whenever a +1/+1 counter is removed from Protean Hydra, put two +1/+1 counters on it at the beginning of the next end step.
31          addCard(Zone.HAND, playerA, "Protean Hydra", 1);
32  
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Protean Hydra");
34          setChoice(playerA, "X=4");
35          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Bioshift", "Protean Hydra^Silvercoat Lion");
36          setChoice(playerA, "X=2");
37  
38          setStopAt(2, PhaseStep.UPKEEP);
39          execute();
40  
41          assertGraveyardCount(playerA, "Bioshift", 1);
42  
43          assertPermanentCount(playerA, "Silvercoat Lion", 1);
44          assertPowerToughness(playerA, "Silvercoat Lion", 4, 4); // added 2 counters
45  
46          assertPermanentCount(playerA, "Protean Hydra", 1);
47          assertPowerToughness(playerA, "Protean Hydra", 6, 6); // started with 4, removed 2, added 4 at end = 6
48  
49      }

51      /**
52       * I'm having an issue when using Bioshift to move only a portion of
53       * counters to another creature. When I attempt to do this, it moves all of
54       * the counters (and in some cases with my Simic deck) kills the creature.
55       */
56      @Test
57      public void testFateTransfer() {
58          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
59          // Noxious Hatchling enters the battlefield with four -1/-1 counters on it.
60          // Wither (This deals damage to creatures in the form of -1/-1 counters.)
61          // Whenever you cast a black spell, remove a -1/-1 counter from Noxious Hatchling.
62          // Whenever you cast a green spell, remove a -1/-1 counter from Noxious Hatchling.
63          addCard(Zone.HAND, playerA, "Noxious Hatchling", 1);// 6/6
64          addCard(Zone.BATTLEFIELD, playerA, "Ruin Processor", 1); // Creature 7/8
65  
66          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
67          // Move all counters from target creature onto another target creature.
68          addCard(Zone.HAND, playerB, "Fate Transfer", 1);
69  
70          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Noxious Hatchling");
71          castSpell(1, PhaseStep.BEGIN_COMBAT, playerB, "Fate Transfer", "Noxious Hatchling^Ruin Processor");
72  
73          setStopAt(1, PhaseStep.END_COMBAT);
74          execute();
75  
76          assertGraveyardCount(playerB, "Fate Transfer", 1);
77  
78          assertPowerToughness(playerA, "Noxious Hatchling", 6, 6);
79  
80          assertPowerToughness(playerA, "Ruin Processor", 3, 4);
81  
82      }

84      /**
85       * The card Leech Bonder (or the token mechanic) doesn't seem to work quite
86       * as intended. If moving a -1/-1 counter from the Leech Bonder onto an
87       * enemy creature with 1/1 the creature stays as a 1/1 with the token being
88       * displayed on it. Going by the rules the creature should have 0/0 and thus
89       * be put into the graveyard.
90       */
91      @Test
92      public void testLeechBonder() {
93          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
94          // Leech Bonder enters the battlefield with two -1/-1 counters on it.
95          // {U}, {untap}: Move a counter from target creature onto another target creature.
96          addCard(Zone.HAND, playerA, "Leech Bonder", 1);// Creature 3/3 - {2}{U}
97  
98          addCard(Zone.BATTLEFIELD, playerB, "Ley Druid", 1); // 1/1
99  
100         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Leech Bonder");
101 
102         attack(3, playerA, "Leech Bonder");
103 
104         activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "{U},", "Leech Bonder");
105         // Ley Druid auto-chosen since only target
106 
107         setStopAt(3, PhaseStep.END_TURN);
108         execute();
109 
110         assertLife(playerB, 19);
111 
112         assertGraveyardCount(playerB, "Ley Druid", 1);
113         assertPowerToughness(playerA, "Leech Bonder", 2, 2);
114     }

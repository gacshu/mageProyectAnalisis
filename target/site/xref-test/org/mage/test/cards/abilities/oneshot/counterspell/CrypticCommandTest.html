18      /**
19       * Test that if command has only one target and that targets is not valid on
20       * resolution, Cryptic Command fizzeles The player does not draw a card
21       */
22      @Test
23      public void testCommand() {
24          addCard(Zone.HAND, playerA, "Thoughtseize");
25          // Counter target spell. If that spell is countered this way, put it into its owner's hand instead of into that player's graveyard.
26          // Draw a card.
27          addCard(Zone.HAND, playerA, "Remand");
28          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
29          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
30  
31          addCard(Zone.HAND, playerB, "Cryptic Command");
32          addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
33  
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Thoughtseize", playerB);
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cryptic Command", "Thoughtseize");
37          setModeChoice(playerB, "1"); // Counter target spell
38          setModeChoice(playerB, "4"); // Draw a card
39  
40          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Remand", "Thoughtseize", "Cast Cryptic Command");
41  
42          setStopAt(1, PhaseStep.CLEANUP);
43          execute();
44  
45          assertLife(playerA, 20);
46          assertLife(playerB, 20);
47  
48          assertGraveyardCount(playerA, "Remand", 1);
49          assertGraveyardCount(playerA, 1);
50          assertGraveyardCount(playerB, "Cryptic Command", 1);
51          assertGraveyardCount(playerB, 1);
52          assertHandCount(playerA, 2); // Thoughtsize + card drawn from Remand
53          assertHandCount(playerB, 0); // Because Cryptic Command has no legal target playerB does not draw a card and has 0 cards in hand
54  
55      }

57      /**
58       * Game is not letting me play Ricochet Trap targetting oponent's Cryptic
59       * Command, modes 1 and 4. It only has one target and should be allowed
60       */
61      @Test
62      public void testCommandChangeTarget() {
63          // Target player reveals their hand. You choose a nonland card from it. That player discards that card. You lose 2 life.
64          addCard(Zone.HAND, playerA, "Thoughtseize");
65          // Counter target spell. If that spell is countered this way, put it into its owner's hand instead of into that player's graveyard.
66          // Draw a card.
67          addCard(Zone.HAND, playerA, "Ricochet Trap");
68          addCard(Zone.HAND, playerA, "Lightning Bolt");
69          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
70          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
71  
72          addCard(Zone.HAND, playerB, "Cryptic Command");
73          addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
74  
75          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Thoughtseize", playerB);
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
77          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cryptic Command", "mode=1Thoughtseize", "Thoughtseize", StackClause.WHILE_ON_STACK);
78          setModeChoice(playerB, "1"); // Counter target spell
79          setModeChoice(playerB, "4"); // Draw a card
80  
81          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ricochet Trap", "Cryptic Command", "Cryptic Command", StackClause.WHILE_ON_STACK);
82          addTarget(playerA, "Lightning Bolt");
83  
84          setStopAt(1, PhaseStep.CLEANUP);
85          execute();
86  
87          assertLife(playerA, 18); // -2 from Thoughtseize
88          assertLife(playerB, 20);
89  
90          assertGraveyardCount(playerA, "Ricochet Trap", 1);
91          assertGraveyardCount(playerA, "Thoughtseize", 1);
92          assertGraveyardCount(playerB, "Cryptic Command", 1);
93          assertGraveyardCount(playerA, "Lightning Bolt", 1);
94  
95          assertHandCount(playerB, 1); // card drawn from Cryptic Command
96  
97      }

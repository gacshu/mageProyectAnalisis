13      /**
14       * Tests miracle alternative cost
15       */
16      @Test
17      public void testMiracleCost() {
18          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
19          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
20          // Put all creatures on the bottom of their owners' libraries.
21          addCard(Zone.LIBRARY, playerA, "Terminus");
22          // Draw a card.
23          addCard(Zone.HAND, playerA, "Think Twice");
24          skipInitShuffling();
25  
26          addCard(Zone.BATTLEFIELD, playerB, "Elite Vanguard", 1);
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Think Twice");
29  
30          setStopAt(1, PhaseStep.BEGIN_COMBAT);
31          execute();
32  
33          // check Terminus was played
34          assertPermanentCount(playerB, "Elite Vanguard", 0);
35      }

60      /**
61       * Test that you can cast a card by miracle if you don't put it back to library before casting
62       */
63      @Test
64      public void testMiracleWillWorkFromHand() {
65          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
66          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
67          addCard(Zone.LIBRARY, playerA, "Plains");
68          addCard(Zone.LIBRARY, playerA, "Forest");
69          addCard(Zone.LIBRARY, playerA, "Thunderous Wrath"); // must be the top most card
70          addCard(Zone.HAND, playerA, "Brainstorm");
71          skipInitShuffling();
72  
73          castSpell(1, PhaseStep.UPKEEP, playerA, "Brainstorm");
74          addTarget(playerA, "Forest");
75          addTarget(playerA, "Plains");
76          addTarget(playerA, playerB);
77  
78          setStopAt(1, PhaseStep.DRAW);
79          execute();
80  
81          assertGraveyardCount(playerA, "Brainstorm", 1);
82          assertHandCount(playerA, "Thunderous Wrath", 0);
83          assertGraveyardCount(playerA, "Thunderous Wrath", 1);
84          assertHandCount(playerA, 0);
85          // check Thunderous Wrath was played
86          assertLife(playerA, 20);
87          assertLife(playerB, 15);
88  
89      }

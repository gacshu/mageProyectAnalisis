4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * @author LevelX2
10   */
11  public class NotOfThisWorldTest extends CardTestPlayerBase {
12  
13      /**
14       * Not of This World doesn't work when trying to counter a triggerd ability
15       * from The Abyss targeting your owned and controlled Ruhan of the Fomori .
16       * At the time I didn't have any mana open, but Ruhan of the Fomori should
17       * qualify for the alternative casting cost (7 less) or Not of This World.
18       */
19      @Test
20      public void testCounterFirstSpell() {
21          // At the beginning of each player's upkeep, destroy target nonartifact creature that player controls of their choice. It can't be regenerated.
22          addCard(Zone.BATTLEFIELD, playerA, "The Abyss", 1);
23          addCard(Zone.BATTLEFIELD, playerB, "Island", 7);
24          //
25          // Counter target spell or ability that targets a permanent you control.
26          // Not of This World costs {7} less to cast if it targets a spell or ability that targets a creature you control with power 7 or greater.
27          addCard(Zone.HAND, playerB, "Not of This World");
28          //
29          // At the beginning of combat on your turn, choose an opponent at random. Ruhan of the Fomori attacks that player this combat if able.
30          addCard(Zone.BATTLEFIELD, playerB, "Ruhan of the Fomori", 1); // 7/7
31  
32          // trigger on upkeep to destroy
33          addTarget(playerB, "Ruhan of the Fomori");
34          // try to counter
35          castSpell(2, PhaseStep.UPKEEP, playerB, "Not of This World", "stack ability (At the beginning of each player's upkeep");
36  
37          setStrictChooseMode(true);
38          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
39          execute();
40  
41          assertHandCount(playerB, "Not of This World", 0);
42          assertGraveyardCount(playerB, "Not of This World", 1);
43          assertPermanentCount(playerB, "Ruhan of the Fomori", 1);
44  
45          assertTapped("Island", false);
46      }
47  
48  }

15      @Test
16      public void testUseOneMode() {
17  
18          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
19          // Escalate {1} (Pay this cost for each mode chosen beyond the first.)
20          // Choose one or more &mdash;
21          // Creatures target player controls gain trample until end of turn.
22          // Savage Alliance deals 2 damage to target creature;
23          // Savage Alliance deals 1 damage to each creature target opponent controls.
24          addCard(Zone.HAND, playerA, "Savage Alliance"); // Instant {2}{R}
25  
26          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Savage Alliance", "mode=2Silvercoat Lion");
29          setModeChoice(playerA, "2");
30  
31          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
32          execute();
33  
34          assertGraveyardCount(playerB, "Silvercoat Lion", 1);
35          assertGraveyardCount(playerA, "Savage Alliance", 1);
36      }

38      @Test
39      public void testGaddockTeegInteraction_ThreeCMC_OneMode() {
40  
41          // Noncreature spells with converted mana cost {4} or greater can't be cast.
42          // Noncreature spells with {X} in their mana costs can't be cast.
43          addCard(Zone.BATTLEFIELD, playerB, "Gaddock Teeg"); // {W}{G} 2/2 Legendary Kithkin
44  
45          // Escalate {1} (Pay this cost for each mode chosen beyond the first.)
46          // Choose one or more —
47          // * Target player discards all the cards in their hand, then draws that many cards.
48          // * Collective Defiance deals 4 damage to target creature.
49          // * Collective Defiance deals 3 damage to target opponent.
50          addCard(Zone.HAND, playerA, "Collective Defiance"); // {1}{R}{R} sorcery
51          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
52  
53          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Collective Defiance", playerB);
54          setModeChoice(playerA, "3"); // deal 3 dmg to opponent
55  
56          setStopAt(1, PhaseStep.BEGIN_COMBAT);
57          execute();
58  
59          assertPermanentCount(playerB, "Gaddock Teeg", 1);
60          assertGraveyardCount(playerA, "Collective Defiance", 1);
61          assertLife(playerB, 17);
62      }

93      @Test
94      public void testSpellQuellerInteraction_ThreeCMC_ThreeModes() {
95  
96          // {1}{W}{U} Flash Flying 2/3 Spirit
97          // When Spell Queller enters the battlefield, exile target spell with converted mana cost 4 or less.
98          // When Spell Queller leaves the battlefield, the exiled card's owner may cast that card without paying its mana cost.
99          addCard(Zone.HAND, playerB, "Spell Queller");
100         addCard(Zone.BATTLEFIELD, playerB, "Wall of Omens"); // {1}{W} 0/4
101         addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
102         addCard(Zone.BATTLEFIELD, playerB, "Plains", 1);
103 
104         // Escalate {1} (Pay this cost for each mode chosen beyond the first.)
105         // Choose one or more —
106         // * Target player discards all the cards in their hand, then draws that many cards.
107         // * Collective Defiance deals 4 damage to target creature.
108         // * Collective Defiance deals 3 damage to target opponent.
109         addCard(Zone.HAND, playerA, "Collective Defiance"); // {1}{R}{R} sorcery
110         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
111 
112         setStrictChooseMode(true);
113 
114         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Collective Defiance", "mode=2Wall of Omens");
115         setModeChoice(playerA, "1"); // opponent discards hand and draws that many
116         setModeChoice(playerA, "2"); // deal 4 dmg to target creature (Wall of Omens)
117         setModeChoice(playerA, "3"); // deal 3 dmg to opponent        
118         addTarget(playerA, playerB); // mode 1
119         addTarget(playerA, playerB); // mode 3
120 
121         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Spell Queller");
122         addTarget(playerB, "Collective Defiance");
123         setStopAt(1, PhaseStep.BEGIN_COMBAT);
124         execute();
125 
126         assertPermanentCount(playerB, "Spell Queller", 1);
127         assertHandCount(playerA, "Collective Defiance", 0);
128         assertExileCount("Collective Defiance", 1);
129         assertGraveyardCount(playerA, "Collective Defiance", 0);
130         assertPermanentCount(playerB, "Wall of Omens", 1);
131         assertLife(playerA, 20);
132         assertLife(playerB, 20);
133     }

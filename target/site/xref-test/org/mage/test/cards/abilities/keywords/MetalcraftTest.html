14  public class MetalcraftTest extends CardTestPlayerBase {
15  
16      /**
17       * Rusted Relic or Blinkmoth nexus is bugged	
18       * Either Relic does not see Blinkmoth as an artifact or it does not turn 
19       * into one when it should.
20       * 
21       */
22      @Test
23      public void testMetalcraftFromBlinkmoth() {
24          addCard(Zone.BATTLEFIELD, playerA, "Darksteel Citadel",1); 
25  
26          // <i>Metalcraft</i> &mdash; {this} is a 5/5 Golem artifact creature as long as you control three or more artifacts
27          addCard(Zone.BATTLEFIELD, playerA, "Rusted Relic", 1); 
28  
29          // {T}: Add {C}to your mana pool.
30          // {1}: Blinkmoth Nexus becomes a 1/1 Blinkmoth artifact creature with flying until end of turn. It's still a land.
31          // {1}, {T}: Target Blinkmoth creature gets +1/+1 until end of turn.        
32          addCard(Zone.BATTLEFIELD, playerA, "Blinkmoth Nexus", 1); 
33          
34          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}:");
35          setStopAt(1, PhaseStep.BEGIN_COMBAT);
36          execute();
37  
38          assertPowerToughness(playerA, "Blinkmoth Nexus", 1, 1);
39          assertPowerToughness(playerA, "Rusted Relic", 5, 5);
40          
41      }

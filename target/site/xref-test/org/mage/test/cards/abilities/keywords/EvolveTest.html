157         // Sudden Disappearance
158         // Sorcery {5}{W}
159         // Exile all nonland permanents target player controls. Return the exiled cards
160         // to the battlefield under their owner's control at the beginning of the next end step.
161         // Battering Krasis (2/1) and Crocanura (1/3) get both a +1/+1 counter each other because they come into play at the same time

169         setChoice(playerA, "Evolve"); // two triggers
170 
171         setStrictChooseMode(true);
172         setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
173         execute();
174

176         assertLife(playerB, 20);
177

182         assertPermanentCount(playerA, "Crocanura", 1);

185         assertPowerToughness(playerA, "Crocanura", 2, 4);

189     /*
190      * Renegade Krasis's ability when trigger it evolves is not triggered under
191      * case.
192      *
193      * I control Renegade Krasis and two Ivy Lane Denizen. (Renegade Krasis and
194      * one of Ivy Lane Denizen have a +1/+1 counter on it)
195      * I cast Adaptive Snapjaw.
196      * When it resolves, there are three abilities on going to stack,
197      * Renegade Krasis's Evolve Ability, two Ivy Lane Denizen ability.
198      * I take two Ivy Lane Denizen on to the stack and then Renegade Krasis's Evolve
199      * Ability, so resolving Renegade Krasis's Evolve Ability is first . (Maybe
200      * Ivy Lane Denizen's target is any.) When Renegade Krasis's Evolve Ability
201      * resolves, +1/+1 counter is placed on it, but doesn't triggers Renegade
202      * Krasis's second ability.
203      */
204     @Test
205     public void testRenegadeKrasis() {
206 
207         // Evolve (Whenever a creature enters the battlefield under your control, if that creature has greater power or toughness
208         // than this creature, put a +1/+1 counter on this creature.)
209         // Whenever Renegade Krasis evolves, put a +1/+1 counter on each other creature you control with a +1/+1 counter on it.
210         addCard(Zone.HAND, playerA, "Renegade Krasis", 1); // {1}{G}{G} - 3/2
211         // Whenever another green creature enters the battlefield under your control, put a +1/+1 counter on target creature.
212         addCard(Zone.BATTLEFIELD, playerA, "Forest", 16);
213 
214         // Whenever another green creature enters the battlefield under your control, put a +1/+1 counter on target creature.
215         addCard(Zone.HAND, playerA, "Ivy Lane Denizen", 2); // {3}{G} - Creature 2/3
216         // Evolve
217         addCard(Zone.HAND, playerA, "Adaptive Snapjaw", 1); // {4}{G} - Creature 6/2
218 
219         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ivy Lane Denizen", true);
220 
221         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Renegade Krasis", true);
222         addTarget(playerA, "Ivy Lane Denizen"); // Ivy target
223 
224         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ivy Lane Denizen", true); // Renegade Krasis evolves
225         addTarget(playerA, "Renegade Krasis"); // Ivy target
226         setChoice(playerA, "Whenever another green creature"); // So +1/+1 counter from Renegade is first put onto Ivy
227 
228         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Adaptive Snapjaw");
229         addTarget(playerA, "Adaptive Snapjaw"); // From Ivy 1
230         addTarget(playerA, "Adaptive Snapjaw"); // From Ivy 2
231         setChoice(playerA, "Evolve"); // Evolve of Renegade Krasis first on the stack, so Adaptive Snapjaw gets +1/+1 from Renegade Krasis ability
232 
233         setStopAt(1, PhaseStep.BEGIN_COMBAT);
234         execute();
235 
236         assertPermanentCount(playerA, "Renegade Krasis", 1);
237         assertPermanentCount(playerA, "Ivy Lane Denizen", 2);
238         assertPermanentCount(playerA, "Adaptive Snapjaw", 1);
239 
240         assertPowerToughness(playerA, "Adaptive Snapjaw", 9, 5); // +2 from Ivys + 1 From Renegade Krasis's Evolve
241         assertPowerToughness(playerA, "Renegade Krasis", 6, 5); // +1 Evolve by Ivy and +1 from Ivy as 2nd Ivy enters +1 Evolve by Snapjaw
242         assertPowerToughness(playerA, "Ivy Lane Denizen", 2, 3, Filter.ComparisonScope.Any);
243         assertPowerToughness(playerA, "Ivy Lane Denizen", 5, 6, Filter.ComparisonScope.Any); // +1 from Other Ivy + 2 from Krasis
244 
245     }

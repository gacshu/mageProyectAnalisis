5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * @author BetaSteward
10   */
11  public class PastInFlamesTest extends CardTestPlayerBase {
12  
13      /**
14       * Past in Flames
15       * Sorcery, 3R (4)
16       * Each instant and sorcery card in your graveyard gains flashback until end
17       * of turn. The flashback cost is equal to its mana cost.
18       * Flashback {4}{R} (You may cast this card from your graveyard for its
19       * flashback cost. Then exile it.)
20       */
21  
22      @Test
23      public void testSpellsGainFlashback() {
24          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
25          addCard(Zone.HAND, playerA, "Past in Flames");
26          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt");
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Past in Flames");
29          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Flashback");
30  
31          setStopAt(1, PhaseStep.END_TURN);
32          execute();
33  
34          assertLife(playerB, 17);
35          assertLife(playerA, 20);
36  
37          assertExileCount("Lightning Bolt", 1);
38          assertGraveyardCount(playerA, "Lightning Bolt", 0);
39      }
40  
41  }

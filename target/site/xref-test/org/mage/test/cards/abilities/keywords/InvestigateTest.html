12   */
13  public class InvestigateTest extends CardTestPlayerBase {
14  
15      @Test
16      public void testBriarbridgePatrol() {
17          // Whenever Briarbridge Patrol deals damage to one or more creatures, investigate (Create a colorless Clue artifact token onto the battlefield with "{2}, Sacrifice this artifact: Draw a card.").
18          // At the beginning of each end step, if you sacrificed three or more Clues this turn, you may put a creature card from your hand onto the battlefield.
19          addCard(Zone.BATTLEFIELD, playerA, "Briarbridge Patrol", 1); // 3/3
20          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
21  
22          addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox", 1); // 2/4
23  
24          attack(1, playerA, "Briarbridge Patrol");
25          block(1, playerB, "Pillarfield Ox", "Briarbridge Patrol");
26  
27          attack(3, playerA, "Briarbridge Patrol");
28          block(3, playerB, "Pillarfield Ox", "Briarbridge Patrol");
29  
30          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "{2},");
31  
32          setStopAt(3, PhaseStep.END_TURN);
33          execute();
34  
35          assertLife(playerA, 20);
36          assertLife(playerB, 20);
37  
38          assertPermanentCount(playerA, "Briarbridge Patrol", 1);
39          assertHandCount(playerA, 2); // 1 from sacrificed Clue and 1 from draw of turn 3
40          assertPermanentCount(playerA, "Clue Token", 1);
41  
42      }
43  }

12  /**
13   * These tests check that {@link mage.abilities.mana.AnyColorLandsProduceManaAbility AnyColorLandsProduceManaAbility}
14   * works properly abilities where Colorless isn't a valid option.
15   * <p>
16   * The test for ensuring that it works with colorless mana is handled in {@link org.mage.test.cards.mana.ReflectingPoolTest}.
17   *
18   * @author Alex-Vasile
19   */
20  public class AnyColorLandsProduceManaAbilityTest extends CardTestPlayerBase {
21  
22      // Any color (but not colorless since that's not a color)
23      private static final String fellwarStone = "Fellwar Stone";
24  
25      /**
26       * Fellwar Stone should not be able to tap for {C}.
27       * <p>
28       * {@link mage.cards.f.FellwarStone Fellwar Stone}
29       * {T}: Add one mana of any color that a land an opponent controls could produce.
30       */
31      @Test
32      public void testColorlessNotAllowed() {
33          addCard(Zone.BATTLEFIELD, playerA, fellwarStone);
34  
35          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
36          addCard(Zone.BATTLEFIELD, playerB, "Desert");
37  
38          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
39          execute();
40  
41          ManaOptions manaOptions = playerA.getAvailableManaTest(currentGame);
42          Assert.assertEquals("mana variations don't fit", 1, manaOptions.size());
43          assertManaOptions("{R}", manaOptions);
44      }
45  }

3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * @author LevelX2

11  public class LightningStormTest extends CardTestPlayerBase {
12  
13      /**
14       * So, this just happened to me. My opponent cast Lightning Storm and while
15       * it was on the stack I couldn't use the ability despite having land in
16       * hand which isn't something I've had an issue with before.
17       * <p>
18       * My opponent had a Leyline of Sanctity in play, so perhaps that was
19       * causing the issue somehow? Does anyone want to try and replicate it?
20       */
21      @Test
22      public void ActivateByBothPlayersTest() {
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
24          // Lightning Storm deals X damage to any target, where X is 3 plus the number of charge counters on it.
25          // Discard a land card: Put two charge counters on Lightning Storm. You may choose a new target for it. Any player may activate this ability but only if Lightning Storm is on the stack.
26          addCard(Zone.HAND, playerA, "Lightning Storm"); // {1}{R}{R}
27  
28          addCard(Zone.HAND, playerA, "Mountain");
29          addCard(Zone.HAND, playerB, "Mountain");
30  
31          // A activate
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Storm", playerB);
33  
34          // B discard and re-target
35          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Discard");
36          setChoice(playerB, true); // change target
37          addTarget(playerB, playerA); // new target
38          setChoice(playerB, "Mountain"); // discard cost
39  
40          // A discard and re-target
41          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Discard");
42          setChoice(playerA, "Mountain");
43          setChoice(playerA, true);
44          addTarget(playerA, playerB);
45  
46          setStrictChooseMode(true);
47          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
48          execute();
49  
50          assertGraveyardCount(playerA, "Lightning Storm", 1);
51          assertGraveyardCount(playerB, "Mountain", 1);
52          assertGraveyardCount(playerA, "Mountain", 1);
53  
54          assertLife(playerA, 20);
55          assertLife(playerB, 20 - 3 - 2 - 2);
56      }

94      @Test
95      public void testChangeControllerForSoulbondCreature() {
96          addCard(Zone.BATTLEFIELD, playerA, "Elite Vanguard");
97          // Soulbond (You may pair this creature with another unpaired creature when either enters the battlefield. They remain paired for as long as you control both of them.)
98          // As long as Trusted Forcemage is paired with another creature, each of those creatures gets +1/+1.
99          addCard(Zone.HAND, playerA, "Trusted Forcemage");
100         addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
101 
102         // Gain control of target creature until end of turn. Untap that creature. It gains haste until end of turn. (It can attack and Tap this turn.)
103         addCard(Zone.HAND, playerB, "Act of Treason");
104         addCard(Zone.BATTLEFIELD, playerB, "Mountain", 3);
105 
106         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Trusted Forcemage");
107         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Act of Treason", "Trusted Forcemage");
108 
109         setStopAt(2, PhaseStep.BEGIN_COMBAT);
110         execute();
111 
112         assertPermanentCount(playerA, "Trusted Forcemage", 0);
113         assertPermanentCount(playerA, "Elite Vanguard", 1);
114         assertPowerToughness(playerA, "Elite Vanguard", 2, 1);
115 
116         assertPermanentCount(playerB, "Trusted Forcemage", 1);
117         assertPowerToughness(playerB, "Trusted Forcemage", 2, 2);

403     /**
404      * Reported bug: Soulbond should use the stack, but unable to use instant speed removal since no trigger occurs
405      */
406     @Test
407     public void testRespondToSoulboundWithRemoval() {
408         // When Palinchron enters the battlefield, untap up to seven lands.
409         // {2}{U}{U}: Return Palinchron to its owner's hand.
410         addCard(Zone.BATTLEFIELD, playerA, "Palinchron"); // 4/5 flying
411 
412         // Soulbond
413         // As long as Deadeye Navigator is paired with another creature, each of those creatures has "{1}{U}: Exile this creature, then return it to the battlefield under your control."
414         addCard(Zone.HAND, playerA, "Deadeye Navigator"); // 5/5
415         addCard(Zone.BATTLEFIELD, playerA, "Island", 8);
416         addCard(Zone.HAND, playerB, "Doom Blade", 1); // {1}{B} instant: Destroy target non-black creature
417         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
418 
419         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Deadeye Navigator");
420         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, 1);
421 
422         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Doom Blade", "Deadeye Navigator");
423 
424         // Deadeye's ability should not be usable since was destroyed before Soulbond trigger resolved
425         checkPlayableAbility("Can't activate", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}{U}:", false);
426         setStopAt(1, PhaseStep.BEGIN_COMBAT);
427         execute();
428 
429         Permanent palinchron = getPermanent("Palinchron", playerA);
430         Assert.assertNull(palinchron.getPairedCard()); // should not be paired
431         assertGraveyardCount(playerA, "Deadeye Navigator", 1);
432         assertGraveyardCount(playerB, "Doom Blade", 1);
433         assertPermanentCount(playerA, "Palinchron", 1);
434     }

15  public class TokenImplActivatedAbilityTest extends CardTestPlayerBase {
16  
17      /**
18       * Check that activated ability of created token works
19       */
20      @Test
21      public void testActivatedManaAbility() {
22          // Green mana doesn't empty from your mana pool as steps and phases end.
23          // Omnath, Locus of Mana gets +1/+1 for each green mana in your mana pool.
24          addCard(Zone.BATTLEFIELD, playerA, "Omnath, Locus of Mana", 1);
25  
26          addCard(Zone.BATTLEFIELD, playerA, "Freyalise, Llanowar's Fury");
27          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+2: Create a 1/1 green Elf Druid creature token with \"{T}: Add {G}.\"");
28          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "+2: Create a 1/1 green Elf Druid creature token with \"{T}: Add {G}.\"");
29          activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}.");
30          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
31          execute();
32  
33          assertPermanentCount(playerA, "Elf Druid Token", 2);
34          assertPermanentCount(playerA, "Freyalise, Llanowar's Fury", 1);
35          Assert.assertEquals("one green mana has to be in the mana pool", 1, playerA.getManaPool().get(ManaType.GREEN));
36      }

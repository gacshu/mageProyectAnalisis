34      /**
35       * I was attacked with an It that Betrays while i had an Academy Rector and
36       * with the annihilator trigger on the stack i cast Cauldron Haze targeting
37       * academy rector then sacrificed her to the annihilator trigger and chose
38       * not to exile her. My persist resolved before the second ability of it
39       * that betrays because i was not the active player, the game log shows:
40       *
41       * 9:18 AM: Ability triggers: Academy Rector [e15] - Persist (When this
42       * creature dies, if it had no -1/-1 counters on it, return it to the
43       * battlefield under its owner's control with a -1/-1 counter on it.)
44       *
45       * 9:19 AM: EllNubNub puts Academy Rector [e15] from graveyard onto the
46       * Battlefield
47       *
48       * 9:20 AM: hellmo puts Academy Rector [e15] from battlefield onto the
49       * Battlefield
50       *
51       * The It that Betrays trigger should have fissled, instead it stole her
52       * from my battlefield and removed the persist counter.
53       */
54      @Test
55      public void testCardItThatBetrays() {
56          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
57  
58          // Choose any number of target creatures. Each of those creatures gains persist until end of turn.
59          // Persist (When this creature dies, if it had no -1/-1 counters on it, return it to the battlefield under its owner's control with a -1/-1 counter on it.)
60          addCard(Zone.HAND, playerA, "Cauldron Haze", 1); // Instant {1}{W/B}
61  
62          // When Academy Rector dies, you may exile it. If you do, search your library for an enchantment card, put that card onto the battlefield, then shuffle your library.
63          addCard(Zone.BATTLEFIELD, playerA, "Academy Rector", 1);
64  
65          // Annihilator 2 (Whenever this creature attacks, defending player sacrifices two permanents.)
66          // Whenever an opponent sacrifices a nontoken permanent, put that card onto the battlefield under your control.
67          addCard(Zone.BATTLEFIELD, playerB, "It That Betrays");
68  
69          setStrictChooseMode(true);
70  
71          attack(2, playerB, "It That Betrays");
72          setChoice(playerA, "Academy Rector"); // Annihilator
73          setChoice(playerA, "Plains"); // Annihilator
74          castSpell(2, PhaseStep.DECLARE_ATTACKERS, playerA, "Cauldron Haze", "Academy Rector", "Annihilator");
75          setChoice(playerB, "Whenever an opponent");  // Put the sacrifice triggers on the stack, order doesn't matter
76          setChoice(playerA, "persist"); // Put the persist trigger on the stack first
77          setChoice(playerA, "No");  // Do not exile the Academy Rector
78  
79          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
80          execute();
81  
82          assertGraveyardCount(playerA, "Cauldron Haze", 1);
83          assertPermanentCount(playerB, "Academy Rector", 0);
84          assertPowerToughness(playerA, "Academy Rector", 0, 1);
85      }

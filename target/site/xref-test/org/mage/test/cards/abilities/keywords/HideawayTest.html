17      /**
18       * 702.74. Hideaway 702.74a Hideaway represents a static ability and a
19       * triggered ability. “Hideaway” means “This permanent enters the
20       * battlefield tapped” and “When this permanent enters the battlefield, look
21       * at the top four cards of your library. Exile one of them face down and
22       * put the rest on the bottom of your library in any order. The exiled card
23       * gains ‘Any player who has controlled the permanent that exiled this card
24       * may look at this card in the exile zone.'”
25       *
26       */
27      /**
28       * Shelldock Isle Land Hideaway (This land enters the battlefield tapped.
29       * When it does, look at the top four cards of your library, exile one face
30       * down, then put the rest on the bottom of your library.) {T}: Add {U} to
31       * your mana pool. {U}, {T}: You may play the exiled card without paying its
32       * mana cost if a library has twenty or fewer cards in it.
33       *
34       */
35      @Test
36      public void testHideaway() {
37          addCard(Zone.HAND, playerA, "Shelldock Isle");
38          addCard(Zone.LIBRARY, playerA, "Silvercoat Lion");
39          skipInitShuffling();
40  
41          this.playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Shelldock Isle");
42          setChoice(playerA, "Silvercoat Lion");
43  
44          setStopAt(1, PhaseStep.BEGIN_COMBAT);
45  
46          setStrictChooseMode(true);
47          execute();
48  
49          assertPermanentCount(playerA, "Shelldock Isle", 1);
50          assertExileCount(playerA, "Silvercoat Lion", 1);
51          for (Card card : currentGame.getExile().getAllCards(currentGame)) {
52              Assert.assertTrue("Exiled card is not face down", card.isFaceDown(currentGame));
53          }
54      }

56      /**
57       * In commander, an opponent cast Ulamog, the Ceaseless Hunger off of
58       * Mosswort Bridge. After it resolved, another opponent exile Ulamog with a
59       * Quarantine Field. Ulamog was shown as exile face down, as it had been
60       * from the Mosswort Bridge.
61       */
62      @Test
63      public void testMosswortBridge() {
64          // Hideaway (This land enters the battlefield tapped. When it does, look at the top four cards of your library, exile one face down, then put the rest on the bottom of your library.)
65          // {T}: Add {G}.
66          // {G}, {T}: You may play the exiled card without paying its mana cost if creatures you control have total power 10 or greater.
67          addCard(Zone.HAND, playerA, "Mosswort Bridge");
68          // When you cast Ulamog, the Ceaseless Hunger, exile two target permanents.
69          // Indestructible
70          // Whenever Ulamog attacks, defending player exiles the top twenty cards of their library.
71          addCard(Zone.LIBRARY, playerA, "Ulamog, the Ceaseless Hunger");
72          skipInitShuffling();
73  
74          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
75          addCard(Zone.BATTLEFIELD, playerA, "Dross Crocodile", 2);// 5/1
76  
77          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Mosswort Bridge");
78          setChoice(playerA, "Ulamog, the Ceaseless Hunger");
79  
80          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{G},");
81          setChoice(playerA, true); // play Ghost Quarter
82  
83          addTarget(playerA, "Dross Crocodile^Dross Crocodile");
84  
85          setStopAt(3, PhaseStep.BEGIN_COMBAT);
86  
87          setStrictChooseMode(true);
88          execute();
89  
90          assertExileCount("Dross Crocodile", 2);
91          assertPermanentCount(playerA, "Mosswort Bridge", 1);
92          assertExileCount(playerA, 2);
93          assertExileCount("Ulamog, the Ceaseless Hunger", 0);
94  
95          assertPermanentCount(playerA, "Ulamog, the Ceaseless Hunger", 1);
96  
97          assertTapped("Mosswort Bridge", true);
98  
99          Permanent permanent = getPermanent("Ulamog, the Ceaseless Hunger", playerA);
100         Card card = currentGame.getCard(permanent.getId());
101         Assert.assertFalse("Previous exiled card may be no longer face down", card.isFaceDown(currentGame));
102     }

313      * Watcher for tomorrow - Watcher of Tomorrow not working when been blinked
314      * by any source, like Ephemerate or Soulherder, still working if dies
315      * blocking.
316      */
317     @Test
318     public void testBlinkWatcherForTomorrow() {
319 
320         /* Hideaway (This permanent enters the battlefield tapped" and
321            "When this permanent enters the battlefield, look at the top four cards of
322            your library. Exile one of them face down and put the rest on the bottom of
323            your library in any order. The exiled card gains 'Any player who has
324            controlled the permanent that exiled this card may look at this card in the
325            exile zone.'") */
326         // When Watcher for Tomorrow leaves the battlefield, put the exiled card into its owner's hand.
327         addCard(Zone.HAND, playerA, "Watcher for Tomorrow"); // Creature 2/1 - {1}{U}
328 
329         // Exile target creature you control, then return it to the battlefield under its owner's control.
330         // Rebound
331         addCard(Zone.HAND, playerA, "Ephemerate"); // Instant {W}
332 
333         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
334         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
335 
336         addCard(Zone.LIBRARY, playerA, "Silvercoat Lion", 4);
337         skipInitShuffling();
338 
339         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Watcher for Tomorrow");
340         setChoice(playerA, "Silvercoat Lion");
341         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Ephemerate", "Watcher for Tomorrow");
342 
343         setStopAt(1, PhaseStep.END_TURN);
344         execute();
345 
346         assertPermanentCount(playerA, "Watcher for Tomorrow", 1);
347         assertExileCount(playerA, "Ephemerate", 1);
348 
349         assertHandCount(playerA, "Silvercoat Lion", 1);
350         assertExileCount(playerA, 2);
351         assertTapped("Watcher for Tomorrow", true);
352     }

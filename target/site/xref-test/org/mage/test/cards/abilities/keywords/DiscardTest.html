14      /*
15       * If Rest in Peace is in play, every card going to the graveyard goes to exile instead.
16       * If a card is discarded while Rest in Peace is on the battlefield, abilities that function
17       * when a card is discarded (such as madness) still work, even though that card never reaches
18       * a graveyard.
19       */
20      @Test
21      public void testRestInPeaceAndCycle() {
22  
23          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
24          addCard(Zone.HAND, playerA, "Tranquil Thicket");
25          addCard(Zone.BATTLEFIELD, playerB, "Rest in Peace", 1);
26  
27          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cycling {G}"); //cycling ability
28          setStopAt(1, PhaseStep.BEGIN_COMBAT);
29          execute();
30  
31          assertLife(playerA, 20);
32          assertLife(playerB, 20);
33  
34          assertExileCount("Tranquil Thicket", 1); //exiled by Rest in Peace
35          assertHandCount(playerA, "Tranquil Thicket", 0); //should be exiled
36          assertHandCount(playerA, 1); // the card drawn by Cycling
37      }

52      /**
53       * With Bazaar of Baghdad, if you use it when you have no cards in hand, you

55       * progress and you lose on time.

57      @Test
58      public void testBazaarOfBaghdad() {
59          // {T}: Draw two cards, then discard three cards.
60          addCard(Zone.BATTLEFIELD, playerA, "Bazaar of Baghdad", 1);
61  
62          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Draw two cards, then discard three cards");
63  
64          setStopAt(1, PhaseStep.BEGIN_COMBAT);
65          execute();
66  
67          assertLife(playerA, 20);
68          assertLife(playerB, 20);
69  
70          assertHandCount(playerA, 0);

75      /**
76       * "When using the enchantment "Liliana's Caress" in a vs. human match, it
77       * seems to count cards that read "you choose a card, enemy discards them"
78       * as the choosing playing making the discard. This means that any spell
79       * that lets you pick what your opponent discards doesn't trigger the Caress
80       * like it should."
81       */
82      @Test
83      public void testLilianasCaress() {
84          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
85          // Whenever an opponent discards a card, that player loses 2 life.
86          addCard(Zone.HAND, playerA, "Liliana's Caress", 1); // ENCHANTMENT {1}{B}
87          // Target opponent reveals their hand. You choose a card from it. That player discards that card.
88          addCard(Zone.HAND, playerA, "Coercion", 1); // SORCERY {2}{B}
89  
90          addCard(Zone.HAND, playerB, "Island", 2);
91  
92          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Liliana's Caress", true);
93          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Coercion", playerB);
94  
95          setStopAt(1, PhaseStep.BEGIN_COMBAT);
96          execute();
97  
98          assertPermanentCount(playerA, "Liliana's Caress", 1);
99  
100         assertGraveyardCount(playerA, "Coercion", 1);
101         assertGraveyardCount(playerB, "Island", 1);
102         assertHandCount(playerB, "Island", 1);
103 
104         assertLife(playerA, 20);
105         assertLife(playerB, 18);
106     }

125      * Test a discard after selecting the cards from another player    
126      */
127     @Test
128     public void GruesomeDiscoveryTest(){
129         // Target player discards two cards.
130         // Morbid - If a creature died this turn, instead that player reveals their hand, 
131         // you choose two cards from it, then that player discards those cards.        
132         addCard(Zone.HAND, playerA, "Gruesome Discovery", 1); // Sorcery {2}{B}{B}
133         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
134         
135         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
136         
137         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
138         addCard(Zone.HAND, playerB, "Silvercoat Lion");
139         addCard(Zone.HAND, playerB, "Aluren");
140         addCard(Zone.HAND, playerB, "Contagion");
141 
142         attack(3, playerA, "Silvercoat Lion");
143         block(3, playerB, "Silvercoat Lion", "Silvercoat Lion");
144         castSpell(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "Gruesome Discovery", playerB);
145 
146         setChoice(playerA, "Aluren^Contagion");
147         
148         // addTarget(playerA, "Aluren");
149         // addTarget(playerA, "Contagion");
150         
151         setStopAt(3, PhaseStep.END_TURN);
152         execute();
153 
154         assertGraveyardCount(playerA, "Silvercoat Lion", 1);
155         assertGraveyardCount(playerB, "Silvercoat Lion", 1);
156         
157         assertGraveyardCount(playerA, "Gruesome Discovery", 1);
158         
159         assertGraveyardCount(playerB, "Aluren", 1);
160         assertGraveyardCount(playerB, "Contagion", 1);
161         assertGraveyardCount(playerB, 3);
162     }

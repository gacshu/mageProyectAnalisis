19      @Test
20      public void testForetellKeyword() {
21          // verify that the foretold card is in the exile zone
22          setStrictChooseMode(true);
23          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);  // Foretell cost 2 mana
24          addCard(Zone.HAND, playerA, "Behold the Multiverse");  // (Instant) Scry 2 and draw 2 cards
25          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fore");
26          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
27          execute();
28  
29          assertExileCount(playerA, "Behold the Multiverse", 1);
30  
31      }

33      @Test
34      public void testForetoldCastSameTurnAsForetold() {
35          // verify that foretold card can't be cast the same turn it was foretold
36          setStrictChooseMode(true);
37          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);  // Foretell cost 2 mana and {1}{U} for foretell cast from exile
38          addCard(Zone.HAND, playerA, "Behold the Multiverse");  // (Instant) Scry 2 and draw 2 cards
39          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fore");
40          checkPlayableAbility("Can't cast turn it was forefold", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Foretell {1}{U}", false);
41          setStopAt(1, PhaseStep.END_TURN);
42          execute();
43  
44          assertExileCount(playerA, "Behold the Multiverse", 1); // still in exile because it can't be cast the same turn
45      }
46  
47      @Test
48      public void testForetoldCastOtherTurnAsForetold() {
49          // verify that foretold card can be cast on a turn other than the one it was foretold
50          setStrictChooseMode(true);
51          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);  // Foretell cost 2 mana and {1}{U} for foretell cast from exile
52          addCard(Zone.HAND, playerA, "Behold the Multiverse");  // (Instant) Scry 2 and draw 2 cards
53          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fore");
54          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Foretell {1}{U}");
55          setStopAt(2, PhaseStep.END_TURN);
56          execute();
57  
58          assertExileCount(playerA, "Behold the Multiverse", 0); // no longer in exile
59          assertGraveyardCount(playerA, "Behold the Multiverse", 1);  // now in graveyard
60          assertHandCount(playerA, 2); // 2 cards drawn
61      }
62      
63      @Test
64      public void testDreamDevourerTrigger() {
65          // Issue #8876
66          setStrictChooseMode(true);
67          addCard(Zone.BATTLEFIELD, playerA, "Dream Devourer", 1); // if a card is foretelled, this gets +2 power
68          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5); // 3 mana for suspend and 2 for foretell
69          addCard(Zone.HAND, playerA, "Sol Talisman", 1);  // Suspend card
70          addCard(Zone.HAND, playerA, "Lightning Bolt", 1); // card to foretell
71          
72          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Susp"); // suspend the Sol Talisman
73          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fore"); // foretell the Lightning Bolt
74          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
75          execute();
76  
77          assertExileCount(playerA, "Sol Talisman", 1); // suspend card in exile
78          assertExileCount(playerA, "Lightning Bolt", 1); // foretold card in exile
79          assertPowerToughness(playerA, "Dream Devourer", 2, 3);  // +2 power boost from trigger due to foretell of Lightning Bolt
80      }

15      /**
16       * Tests to put a token onto the battlefield
17       */
18      @Test
19      public void testOozeFlux() {
20          // Enchantment
21          // {1}{G}, Remove one or more +1/+1 counters from among creatures you control: Put an X/X green Ooze creature token onto the battlefield, where X is the number of +1/+1 counters removed this way.
22          addCard(Zone.BATTLEFIELD, playerA, "Ooze Flux");
23          // Trample
24          // Kalonian Hydra enters the battlefield with four +1/+1 counters on it.
25          // Whenever Kalonian Hydra attacks, double the number of +1/+1 counters on each creature you control.
26          addCard(Zone.BATTLEFIELD, playerA, "Kalonian Hydra");
27          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
28  
29          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}{G},");
30          setChoice(playerA, "X=2"); // Remove how many
31          setChoice(playerA,"Kalonian Hydra");
32          setChoice(playerA, "X=2"); // Remove from Hydra
33  
34          setStopAt(1, PhaseStep.BEGIN_COMBAT);
35          execute();
36  
37          assertPowerToughness(playerA, "Kalonian Hydra", 2, 2);
38          assertPermanentCount(playerA, "Ooze Token", 1);
39          assertPowerToughness(playerA, "Ooze Token", 2, 2);
40  
41      }

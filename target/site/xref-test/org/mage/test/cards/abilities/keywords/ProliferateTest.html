14  public class ProliferateTest extends CardTestPlayerBase {
15  
16      /**
17       * Volt Charge
18       * {2}{R}
19       * Volt Charge deals 3 damage to any target.
20       * Proliferate. (Choose any number of permanents and/or players, then give each another counter of each kind already there.)
21       */
22      @Test
23      public void testCastFromHandMovedToExile() {
24          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
25          addCard(Zone.BATTLEFIELD, playerA, "Chandra, Pyromaster");  // starts with 4 loyality counters
26  
27          addCard(Zone.HAND, playerA, "Volt Charge");
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Volt Charge", playerB);
30          setChoice(playerA, "Chandra, Pyromaster");
31  
32          setStopAt(1, PhaseStep.BEGIN_COMBAT);
33          execute();
34  
35          assertLife(playerB, 17);
36          assertGraveyardCount(playerA, "Volt Charge", 1);
37  
38          assertCounterCount("Chandra, Pyromaster", CounterType.LOYALTY, 5); // 4 + 1 from proliferate
39      }

41      /**
42       * Counters aren't cancelling each other out. Reproducible with any creature
43       * (graft and bloodthirst in my case) with a single +1/+1 counter on it,
44       * with a single -1/-1 placed on it (Grim Affliction, Instill Infection,
45       * etc). The counters should cancel each other out, leaving neither on the
46       * creature, which they don't (though visually there aren't any counters
47       * sitting on the card). Triggering proliferate at any point now (Thrumming
48       * Bird, Steady Progress, etc) will give you the option to add another of
49       * either counter, where you shouldn't have any as an option.
50       */
51      @Test
52      public void testValidTargets() {
53          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
54          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
55          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
56          // Put a +1/+1 counter on target creature.
57          addCard(Zone.HAND, playerA, "Battlegrowth"); // {G}
58          // Proliferate. (You choose any number of permanents and/or players with counters on them, then give each another counter of a kind already there.)
59          // Draw a card.
60          addCard(Zone.HAND, playerA, "Steady Progress"); // {U}{2}
61  
62          addCard(Zone.BATTLEFIELD, playerB, "Sporeback Troll");  // has two +1/+1 counter
63          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 3);
64          // Put a -1/-1 counter on target creature, then proliferate.
65          addCard(Zone.HAND, playerB, "Grim Affliction"); // {B}{2}
66  
67          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Battlegrowth", "Silvercoat Lion");
68  
69          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Grim Affliction", "Silvercoat Lion");
70          // proliferate Sporeback Troll
71  
72          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Steady Progress");
73          // Silvercoat Lion may not be a valid target now
74  
75          setStopAt(2, PhaseStep.BEGIN_COMBAT);
76          execute();
77  
78          assertGraveyardCount(playerA, "Battlegrowth", 1);
79          assertGraveyardCount(playerA, "Steady Progress", 1);
80          assertGraveyardCount(playerB, "Grim Affliction", 1);
81  
82          assertCounterCount("Silvercoat Lion", CounterType.P1P1, 0); // no valid target because no counter
83          assertCounterCount("Sporeback Troll", CounterType.P1P1, 3); // 2 + 1 from proliferate
84          assertPowerToughness(playerB, "Sporeback Troll", 3, 3);
85  
86      }

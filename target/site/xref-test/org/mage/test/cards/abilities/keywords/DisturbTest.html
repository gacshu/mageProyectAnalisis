17      /**
18       * Relevant ruling:
19       *      - When you cast a spell using a card's disturb ability, the card is put onto the stack with its
20       *        back face up. The resulting spell has all the characteristics of that face.
21       *      - The mana value of a spell cast using disturb is determined by the mana cost on the
22       *        front face of the card, no matter what the total cost to cast the spell was.
23       */
24      @Test
25      public void test_SpellAttributesOnStack() {
26          // Disturb {1}{U}
27          // Hook-Haunt Drifter
28          addCard(Zone.GRAVEYARD, playerA, "Baithook Angler", 1); // {1}{U}
29          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
30          //
31          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
32          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
33  
34          checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb", true);
35  
36          // cast with disturb
37          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 2);
38          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb");
39          checkStackObject("on stack", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb", 1);
40          runCode("check stack", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
41              // Stack must contain another card side, so spell/card characteristics must be diff from main side (only mana value is same)
42              Spell spell = (Spell) game.getStack().getFirst();
43              Assert.assertEquals("Hook-Haunt Drifter", spell.getName());
44              Assert.assertEquals(1, spell.getCardType(game).size());
45              Assert.assertEquals(CardType.CREATURE, spell.getCardType(game).get(0));
46              Assert.assertEquals(1, spell.getSubtype(game).size());
47              Assert.assertEquals(SubType.SPIRIT, spell.getSubtype(game).get(0));
48              Assert.assertEquals(1, spell.getPower().getValue());
49              Assert.assertEquals(2, spell.getToughness().getValue());
50              Assert.assertEquals("U", spell.getColor(game).toString());
51  
52              Assert.assertEquals(2, spell.getManaValue()); // {1}{U}
53  
54              Assert.assertEquals("{1}{U}", spell.getSpellAbility().getManaCosts().getText());
55          });
56  
57          // must be transformed
58          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
59          checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 1);
60          checkPT("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 1, 2);
61          checkSubType("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", SubType.SPIRIT, true);
62  
63          // must be exiled on die
64          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Hook-Haunt Drifter");
65          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
66          checkExileCount("after die", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 0);
67          checkExileCount("after die", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Baithook Angler", 1);
68  
69          setStrictChooseMode(true);
70          setStopAt(1, PhaseStep.END_TURN);
71          execute();
72      }

74      /**
75       * Relevant ruling:
76       *      To determine the total cost of a spell, start with the mana cost or alternative cost
77       *      (such as a disturb cost) you're paying, add any cost increases, then apply any cost
78       *      reductions. The mana value of a spell cast using disturb is determined by the mana cost on
79       *      the front face of the card, no matter what the total cost to cast the spell was. (This is
80       *      a special rule that applies only to transforming double faced-cards, including ones with
81       *      disturb.)
82       */
83      @Test
84      public void test_CostModification_CanPlay() {
85          // rules:
86  
87  
88          // Disturb {1}{U}
89          // Hook-Haunt Drifter
90          addCard(Zone.GRAVEYARD, playerA, "Baithook Angler", 1); // {1}{U}
91          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
92  
93          addCustomEffect_SpellCostModification(playerA, -1);
94  
95          checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb", true);
96  
97          // cast with disturb
98          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb");
99          runCode("check stack", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
100             Spell spell = (Spell) game.getStack().getFirst();
101             Assert.assertEquals("mana value must be from main side", 2, spell.getManaValue());
102             Assert.assertEquals("mana cost to pay must be modified", "{U}", spell.getSpellAbility().getManaCostsToPay().getText());
103         });
104         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
105         checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 1);
106

108         setStopAt(1, PhaseStep.END_TURN);

131     /**
132      * Relevant rule:
133      *      If you copy a permanent spell cast this way (perhaps with a card like Double Major), the copy becomes
134      *      a token that's a copy of the card's back face, even though it isn't itself a double-faced card.
135      */
136     @Test
137     public void test_CopySpell() {
138         // Disturb {1}{U}
139         // Hook-Haunt Drifter
140         addCard(Zone.GRAVEYARD, playerA, "Baithook Angler", 1); // {1}{U}
141         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
142 
143         // Copy target creature spell you control, except it isn't legendary if the spell is legendary.
144         addCard(Zone.HAND, playerA, "Double Major", 1); // {G}{U}
145         addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
146         addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
147 
148         checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb", true);
149 
150         // cast with disturb
151         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 2);
152         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb");
153         // prepare copy of spell
154         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Double Major", "Hook-Haunt Drifter", "Hook-Haunt Drifter");
155         checkStackSize("before copy spell", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
156         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA, true);
157         checkStackSize("after copy spell", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
158         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
159         checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 2);
160 
161         setStrictChooseMode(true);
162         setStopAt(1, PhaseStep.END_TURN);
163         execute();
164     }

166     /**
167      * Relevant ruling:
168      *      The back face of each card with disturb has an ability that instructs its controller to exile
169      *      if it would be put into a graveyard from anywhere. This includes going to the graveyard from the
170      *      stack, so if the spell is countered after you cast it using the disturb ability, it will
171      *      be put into exile.
172      */
173     @Test
174     public void test_Counter() {
175         // Disturb {1}{U}
176         // Hook-Haunt Drifter
177         addCard(Zone.GRAVEYARD, playerA, "Baithook Angler", 1); // {1}{U}
178         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
179 
180         // Counter target spell.
181         addCard(Zone.HAND, playerA, "Counterspell", 1); // {U}{U}
182         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
183 
184         checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb", true);
185 
186         // cast with disturb
187         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Hook-Haunt Drifter with Disturb");
188         // counter it
189         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Counterspell", "Hook-Haunt Drifter", "Hook-Haunt Drifter");
190         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
191 
192         checkPermanentCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Hook-Haunt Drifter", 0);
193         checkExileCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Baithook Angler", 1);
194 
195         setStrictChooseMode(true);
196         setStopAt(1, PhaseStep.END_TURN);
197         execute();
198     }

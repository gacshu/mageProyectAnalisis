1   package org.mage.test.cards.abilities.activated;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import mage.game.permanent.Permanent;
6   import org.junit.Assert;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   * {@link mage.cards.l.LinvalaKeeperOfSilence Linvala, Keeper of Silence}
12   * {2}{W}{W}
13   * Legendary Creature — Angel
14   * Flying
15   * Activated abilities of creatures your opponents control can’t be activated.
16   *
17   * Created by Derek M on 3/6/2017.
18   */
19  public class LinvalaKeeperOfSilenceTest extends CardTestPlayerBase {
20  
21      /**
22       * Reported bug: Linvala, Keeper of Silence, played by my opponent,
23       *               did not prevent me from activating abilities with the "tap" symbol.
24       *               Examples included Jagged-Scar Archers and Imperious Perfect;
25       */
26      @Test
27      public void linvalaSilencesActivatedAbilities() {
28          String linvala = "Linvala, Keeper of Silence";
29          String jScarArcher = "Jagged-Scar Archers";
30          String imperiousPerfect = "Imperious Perfect";
31  
32          addCard(Zone.BATTLEFIELD, playerA, linvala);
33  
34          /*
35           * {1}{G}{G} Elf Archer * / *
36           * Jagged-Scar Archers's power and toughness are each equal to the number of Elves you control.
37           * Tap: Jagged-Scar Archers deals damage equal to its power to target creature with flying.
38           */
39          addCard(Zone.BATTLEFIELD, playerB, jScarArcher);
40  
41          /*
42           * {2}{G} Creature - Elf Warrior 2/2 Other
43           * Elf creatures you control get +1/+1.
44           * {G}, Tap: Create a 1/1 green Elf Warrior creature token.
45           */
46          addCard(Zone.BATTLEFIELD, playerB, imperiousPerfect);
47  
48          addCard(Zone.BATTLEFIELD, playerB, "Forest", 2);
49  
50          // Imperious Perfect: create 1/1 elf warrior token
51          checkPlayableAbility("Can't use activated abilities", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "{G}:", false);
52          // Jagged-scar: deal damage to Linvala equal to elves in play
53          checkPlayableAbility("Can't use activated abilities", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Jagged", false);
54  
55          setStopAt(3, PhaseStep.BEGIN_COMBAT);
56          execute();
57  
58          assertPermanentCount(playerA, linvala, 1);
59          assertPowerToughness(playerB, jScarArcher, 3, 3);
60          Permanent p = getPermanent(linvala, playerA);
61          Assert.assertEquals("Linvala should have no damage dealt to her", 0, p.getDamage());
62      }
63  }

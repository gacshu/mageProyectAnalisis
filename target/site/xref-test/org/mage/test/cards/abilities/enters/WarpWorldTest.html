13  public class WarpWorldTest extends CardTestPlayerBase {
14  
15      /**
16       * Hello. I was playing a 4 player EDH game, and after quite a ways into it
17       * I casted Warp World. The game said that Prime Speaker Zegana's ability
18       * triggered for hitting the battlefield, however on all of our screens
19       * nothing was on the battlefield (not even other things that should have
20       * hit from Warp World). I tried to upload the log but it gave me an error
21       * about it possibly being an attack vector. Sorry. Below is at least the
22       * end of the log with hopefully some useful information:
23       */
24      @Test
25      public void testWarpWorld() {
26          // Each player shuffles all permanents they own into their library, then reveals that many cards from the top of their library.
27          // Each player puts all artifact, creature, and land cards revealed this way onto the battlefield, then does the same for enchantment cards,
28          // then puts all cards revealed this way that weren't put onto the battlefield on the bottom of their library.
29          addCard(Zone.HAND, playerA, "Warp World"); // Sorcery {5}{R}{R}{R}
30          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 8);
31  
32          playerA.getLibrary().clear();
33          addCard(Zone.LIBRARY, playerA, "Silvercoat Lion", 8);
34          skipInitShuffling();
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Warp World");
37  
38          setStopAt(1, PhaseStep.BEGIN_COMBAT);
39  
40          execute();
41  
42          assertHandCount(playerA, "Warp World", 0);
43          assertPermanentCount(playerA, 8);
44  
45      }
46  }

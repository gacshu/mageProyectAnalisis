16   * 614.12. Some replacement effects modify how a permanent enters the battlefield. (See rules 614.1c-d.)
17   * Such effects may come from the permanent itself if they affect only that permanent (as opposed to a
18   * general subset of permanents that includes it). They may also come from other sources. To determine
19   * which replacement effects apply and how they apply, check the characteristics of the permanent as it
20   * would exist on the battlefield, taking into account replacement effects that have already modified
21   * how it enters the battlefield (see rule 616.1), continuous effects generated by the resolution of
22   * spells or abilities that changed the permanent's characteristics on the stack (see rule 400.7a), and
23   * continuous effects from the permanent's own static abilities, but ignoring continuous effects from
24   * any other source that would affect it. #
25   *
26   * Example: Orb of Dreams is an artifact that says "Permanents enter the battlefield tapped." It won't
27   * affect itself, so Orb of Dreams enters the battlefield untapped.
28   *
29   */
30  public class OrbOfDreamsTest extends CardTestPlayerBase {
31  
32      @Test
33      public void testOrbNotTappingItself() {
34          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
35          addCard(Zone.HAND, playerA, "Orb of Dreams");
36          addCard(Zone.HAND, playerA, "Razortip Whip");
37  
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Orb of Dreams");
39          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Razortip Whip");
40  
41          setStopAt(1, PhaseStep.END_TURN);
42          execute();
43  
44          assertLife(playerA, 20);
45          assertLife(playerB, 20);
46  
47          assertPermanentCount(playerA, "Orb of Dreams", 1);
48          assertPermanentCount(playerA, "Razortip Whip", 1);
49  
50          Permanent orbOfDreams = getPermanent("Orb of Dreams", playerA.getId());
51          Assert.assertFalse("Orb has tapped itself, but may not", orbOfDreams.isTapped());
52          Permanent razortipWhip = getPermanent("Razortip Whip", playerA.getId());
53          Assert.assertTrue("Razortip Whip must be tapped from Orb of Dreams but isn't", razortipWhip.isTapped());
54      }

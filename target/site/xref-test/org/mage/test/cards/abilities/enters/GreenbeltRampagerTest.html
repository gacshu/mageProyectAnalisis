15      @Test
16      public void testFirstCast() {
17          addCard(Zone.HAND, playerA, "Greenbelt Rampager");
18          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
19  
20          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Greenbelt Rampager");
21  
22          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
23          execute();
24  
25          assertHandCount(playerA, "Greenbelt Rampager", 1);
26          assertPermanentCount(playerA, "Greenbelt Rampager", 0);
27          assertCounterCount(playerA, CounterType.ENERGY, 1);

63      @Test
64      public void testCastNotOwned() {
65          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 10);
66          // Deathtouch
67          // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
68          // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
69          // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
70          addCard(Zone.HAND, playerA, "Gonti, Lord of Luxury"); // Creature {2}{B}{B}
71  
72          // When Greenbelt Rampager enters the battlefield, pay {E}{E}. If you can't, return Greenbelt Rampager to its owner's hand and you get {E}.
73          addCard(Zone.LIBRARY, playerB, "Greenbelt Rampager"); // Creature {G} 3/4
74  
75          skipInitShuffling();
76  
77          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gonti, Lord of Luxury", true);
78          // addTarget(playerA, playerB); playerB is autochosen since only option
79          setChoice(playerA, "Greenbelt Rampager");
80  
81          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Greenbelt Rampager");
82          setStopAt(1, PhaseStep.BEGIN_COMBAT);
83          execute();
84  
85          assertCounterCount(playerA, CounterType.ENERGY, 1);
86          assertPermanentCount(playerA, "Greenbelt Rampager", 0);
87          assertHandCount(playerA, "Greenbelt Rampager", 0);
88          assertHandCount(playerB, "Greenbelt Rampager", 1);
89      }

16      public void testLandAfterFetchUntapped() {
17          addCard(Zone.HAND, playerA, "Verdant Catacombs");
18          addCard(Zone.LIBRARY, playerA, "Forest");
19  
20          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Verdant Catacombs");
21  
22          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}, Pay");
23          addTarget(playerA, "Forest");
24          setStopAt(1, PhaseStep.END_TURN);
25          execute();
26  
27          assertGraveyardCount(playerA, "Verdant Catacombs", 1);
28          assertPermanentCount(playerA, "Forest", 1);
29          assertTapped("Forest", false);
30      }

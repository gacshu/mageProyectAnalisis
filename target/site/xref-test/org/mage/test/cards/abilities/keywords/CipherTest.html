2   package org.mage.test.cards.abilities.keywords;
3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   *
11   * @author LevelX2
12   */
13  public class CipherTest extends CardTestPlayerBase {
14  
15      /**
16       * Produced a copy of the opponents Roil Elemental with Stolen Identity and
17       * used Cipher on that same token. The token's landfall ability then did
18       * trigger normally up to the point where a target creature could be
19       * selected. The selection was logged by XMage, but the effect simply did
20       * not work. The original Roil Elemental controlled by the other player
21       * worked as intended, though.
22       *
23       * Edit: Opponent was AI, if that helps.
24       */
25      @Test
26      public void testStolenIdentity() {
27          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
28          addCard(Zone.HAND, playerA, "Mountain", 1);
29  
30          // Create a token that's a copy of target artifact or creature.
31          // Cipher  (Then you may exile this spell card encoded on a creature you control. Whenever that creature deals combat damage to a player, its controller may cast a copy of the encoded card without paying its mana cost.)
32          addCard(Zone.HAND, playerA, "Stolen Identity"); // Sorcery {4}{U}{U}
33  
34          // Flying
35          // Landfall - Whenever a land enters the battlefield under your control, you may gain control of target creature for as long as you control Roil Elemental.
36          addCard(Zone.BATTLEFIELD, playerB, "Roil Elemental");
37          addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox");
38          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
39  
40          // cast spell, create copy token, exile spell card and encode it to that token of Roil Elemental
41          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Stolen Identity", "Roil Elemental");
42          setChoice(playerA, true); // Cipher activate
43          addTarget(playerA, "Roil Elemental"); // Cipher target for encode
44          checkPermanentCount("playerA must have Roil Elemental", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Roil Elemental", 1);
45          checkPermanentCount("playerB must have Roil Elemental", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Roil Elemental", 1);
46          checkExileCount("Stolen Identity must be in exile zone", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Stolen Identity", 1);
47  
48          // Roil Elemental must activated on new land
49          playLand(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Mountain");
50          setChoice(playerA, true); // activate landfall to control opponent creature
51          addTarget(playerA, "Silvercoat Lion"); // Triggered ability of copied Roil Elemental to gain control
52          checkPermanentCount("must gain control of Lion", 3, PhaseStep.POSTCOMBAT_MAIN, playerA, "Silvercoat Lion", 1);
53          checkPermanentCount("must lose control of Lion", 3, PhaseStep.POSTCOMBAT_MAIN, playerB, "Silvercoat Lion", 0);
54  
55          // on attack must activated ability to free cast
56          attack(5, playerA, "Roil Elemental");
57          setChoice(playerA, true); // activate free cast of encoded card
58          checkPermanentCount("playerA must have 2 Roil Elemental", 5, PhaseStep.POSTCOMBAT_MAIN, playerA, "Roil Elemental", 2);
59          checkPermanentCount("playerB must have Roil Elemental", 5, PhaseStep.POSTCOMBAT_MAIN, playerB, "Roil Elemental", 1);
60  
61          setStopAt(5, PhaseStep.POSTCOMBAT_MAIN);
62          execute();
63  
64          assertLife(playerB, 17); // -3 by Roil
65      }
66  }

13  public class HeroicTest extends CardTestPlayerBase {
14  
15      /**
16       * When casting Dromoka's Command targeting two of my own Heroic creatures,
17       * only one of them triggers. It appears to be the one targeted with mode 4
18       * (fight) rather than the one targeted with mode 3 (+1/+1 counter).
19       * Screenshot attached. Reproducible.
20       */
21      @Test
22      public void testHeroicWithModal() {
23          // Heroic - Whenever you cast a spell that targets Favored Hoplite, put a +1/+1 counter on Favored Hoplite and prevent all damage that would be dealt to it this turn.
24          addCard(Zone.BATTLEFIELD, playerA, "Favored Hoplite", 1); // 1/2
25          // Heroic â€” Whenever you cast a spell that targets Lagonna-Band Trailblazer, put a +1/+1 counter on Lagonna-Band Trailblazer.
26          addCard(Zone.BATTLEFIELD, playerA, "Lagonna-Band Trailblazer"); // 0/4
27  
28          // Mode 3 = Put a +1/+1 counter on target creature
29          // Mode 4 = Target creature you control fights target creature you don't control
30          addCard(Zone.HAND, playerA, "Dromoka's Command", 1); // {G}{W}
31          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
32          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
33  
34          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Dromoka's Command", "mode=3Lagonna-Band Trailblazer^mode=4Favored Hoplite^Silvercoat Lion");
37          // Silvercoat lion will be set by AI as only possible target
38          setModeChoice(playerA, "3");
39          setModeChoice(playerA, "4");
40  
41          setStopAt(1, PhaseStep.BEGIN_COMBAT);
42          execute();
43  
44          assertGraveyardCount(playerA, "Dromoka's Command", 1);
45  
46          assertPowerToughness(playerA, "Favored Hoplite", 2, 3);
47          assertGraveyardCount(playerB, "Silvercoat Lion", 1);
48          assertPowerToughness(playerA, "Lagonna-Band Trailblazer", 2, 6);
49  
50      }

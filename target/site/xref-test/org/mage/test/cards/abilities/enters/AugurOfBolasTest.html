12  public class AugurOfBolasTest extends CardTestPlayerBase {
13  
14      /*
15          Aether Figment   {1}{U}
16          Creature - Illusion
17          1/1
18          Kicker {3} (You may pay an additional as you cast this spell.)
19          Aether Figment can't be blocked.
20          If Aether Figment was kicked, it enters the battlefield with two +1/+1 counters on it.
21       */
22      @Test
23      public void testEnteringWithCounters() {
24          addCard(Zone.LIBRARY, playerA, "Lightning Bolt", 3);
25          skipInitShuffling();
26          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
27          // When Augur of Bolas enters the battlefield, look at the top three cards of your library.
28          // You may reveal an instant or sorcery card from among them and put it into your hand. Put the rest on the bottom of your library in any order.
29          addCard(Zone.HAND, playerA, "Augur of Bolas"); // Creature {1}{U}
30  
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Augur of Bolas");
32  
33          setStopAt(1, PhaseStep.BEGIN_COMBAT);
34          execute();
35  
36          assertPermanentCount(playerA, "Augur of Bolas", 1);
37          assertHandCount(playerA, "Lightning Bolt", 1);
38      }
39  }

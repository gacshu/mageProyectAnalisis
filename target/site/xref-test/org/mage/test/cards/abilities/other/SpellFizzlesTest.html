4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class SpellFizzlesTest extends CardTestPlayerBase {
14  
15      @Test
16      public void testSpellFizzlesWithNoLegalTargets() throws Exception {
17          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
18          // Devoid
19          // Tap up to two target creatures. Those creatures don't untap during their controller's next untap step.
20          // Create a 1/1 colorless Eldrazi Scion creature token onto the battlefield. It has "Sacrifice this creature: Add {C}."
21          addCard(Zone.HAND, playerA, "Adverse Conditions"); // {3}{U}
22  
23          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
24          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
25          addCard(Zone.HAND, playerB, "Lightning Bolt", 1);
26  
27          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Adverse Conditions", "Silvercoat Lion");
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Silvercoat Lion", "Adverse Conditions");
29  
30          setStopAt(1, PhaseStep.BEGIN_COMBAT);
31          execute();
32  
33          assertGraveyardCount(playerA, "Adverse Conditions", 1);
34          assertGraveyardCount(playerB, "Lightning Bolt", 1);
35          assertGraveyardCount(playerB, "Silvercoat Lion", 1);
36          assertPermanentCount(playerA, "Eldrazi Scion Token", 0); // All targets were illegal - spell fizzles
37  
38      }

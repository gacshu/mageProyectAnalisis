14      /* Master Biomancer  {2}{G}{U}
15       * Creature - Elf Wizard
16       * 2/4
17       * Each other creature you control enters the battlefield with a number of additional +1/+1 counters
18       * on it equal to Master Biomancer's power and as a Mutant in addition to its other types.
19       *
20       */
21      @Test
22      public void testCreatureGetsCounters() {
23  
24          // a creature enters the battlefield and gets a counter for each point of power of Master Biomancer
25          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
26          addCard(Zone.BATTLEFIELD, playerA, "Master Biomancer", 1);
27          addCard(Zone.HAND, playerA, "Mindeye Drake");
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Mindeye Drake");
30  
31          setStopAt(1, PhaseStep.BEGIN_COMBAT);
32          execute();
33  
34          assertLife(playerA, 20);
35          assertLife(playerB, 20);
36  
37          assertPermanentCount(playerA, "Master Biomancer", 1);
38          assertPermanentCount(playerA, "Mindeye Drake", 1);
39  
40          assertPowerToughness(playerA, "Master Biomancer", 2, 4);
41          //   P/T = 2/5 + (2 * +1/+1) = 4 / 7
42          assertPowerToughness(playerA, "Mindeye Drake", 4, 7);
43      }

45      @Test
46      public void testCreatureGetsDoubleCountersFromCorpsejackMenace() {
47          // a creature enters the battlefield and gets a counter for each point of power of Master Biomancer
48          // doubled by Corpsejack Menace (when he ist cast, his own ability will not apply)
49          // http://blogs.magicjudges.org/rulestips/2013/03/corpsejack-menace-and-master-biomancer/
50          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
51          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
52          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
53          addCard(Zone.BATTLEFIELD, playerA, "Master Biomancer", 1);
54  
55          // If one or more +1/+1 counters would be placed on a creature you control, twice that many +1/+1 counters are placed on it instead.
56          addCard(Zone.HAND, playerA, "Corpsejack Menace");
57          addCard(Zone.HAND, playerA, "Mindeye Drake");
58  
59          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Corpsejack Menace");
60          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Mindeye Drake");
61  
62          setStopAt(3, PhaseStep.BEGIN_COMBAT);
63          execute();
64  
65          assertLife(playerA, 20);
66          assertLife(playerB, 20);
67  
68          assertPermanentCount(playerA, "Master Biomancer", 1);
69          assertPermanentCount(playerA, "Corpsejack Menace", 1);
70          assertPermanentCount(playerA, "Mindeye Drake", 1);
71  
72          assertPowerToughness(playerA, "Master Biomancer", 2, 4);
73  
74          //   P/T = 4/4 + (2 * +1/+1) = 6 / 6 (own doubling not active yet)
75          assertPowerToughness(playerA, "Corpsejack Menace", 6, 6);
76  
77          //   P/T = 2/5 + 2* (2 * +1/+1) = 6 / 9
78          assertPowerToughness(playerA, "Mindeye Drake", 6, 9);
79      }

81      /**
82       * Progenitor Mimic
83       * Creature - Shapeshifter 0/0
84       * You may have Progenitor Mimic enter the battlefield as a copy of any creature on the battlefield
85       * except it gains "At the beginning of your upkeep, if this creature isn't a token,
86       * put a token onto the battlefield that's a copy of this creature."
87       *
88       * If Progenitor Mimic comes into play, it gets two +1/+1 counters from the Master Biomancer already in play.
89       * It copies the Master Biomancer and is therfore a 4/6 creature.
90       * The Token generated next round from Progenitor Mimic has to get 2 + 4 counters and is therefore a 8/10 creature.
91       */
92      @Test
93      public void testWithProgenitorMimic() {
94          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
95          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
96          // A creature enters the battlefield and gets a counter for each point of power of Master Biomancer
97          addCard(Zone.BATTLEFIELD, playerA, "Master Biomancer", 1);
98          addCard(Zone.HAND, playerA, "Progenitor Mimic");
99  
100         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Progenitor Mimic");
101 
102         setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
103         execute();
104 
105         assertLife(playerA, 20);
106         assertLife(playerB, 20);
107 
108         assertPermanentCount(playerA, "Master Biomancer", 3);
109 
110         // the original Master Biomancer
111         assertPowerToughness(playerA, "Master Biomancer", 2, 4);
112         // the Progenitor Mimic copying the Master Biomancer
113         assertPowerToughness(playerA, "Master Biomancer", 4, 6);
114         // the first token created by the Progenitor Mimic copying the Master Biomancer
115         assertPowerToughness(playerA, "Master Biomancer", 8, 10);
116     }

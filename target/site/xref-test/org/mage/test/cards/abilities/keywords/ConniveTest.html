14      /**
15       * Connive and discard a land.
16       * Creature should not get a +1/+1 counter.
17       */
18      @Test
19      public void conniveDiscardLand() {
20          // P/T : 1/2
21          // {3}: Hypnotic Grifter connives
22          addCard(Zone.BATTLEFIELD, playerA, "Hypnotic Grifter");
23          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
24          addCard(Zone.HAND, playerA, "Plains", 1); // To discard
25  
26          setStrictChooseMode(true);
27  
28          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{3}");
29          setChoice(playerA, "Plains");
30  
31          setStopAt(1, PhaseStep.BEGIN_COMBAT);
32          execute();
33  
34          assertPowerToughness(playerA, "Hypnotic Grifter", 1, 2); // Discarded a land card
35          assertHandCount(playerA, 1); // 1 from drawing at start of turn
36          assertGraveyardCount(playerA, "Plains", 1);
37      }

64      /**
65       * Connive + Madness.
66       * Creature should get a +1/+1 counter and the madness card gets played.
67       */
68      @Test
69      public void conniveMadness() {
70          // P/T : 1/2
71          // {3}: Hypnotic Grifter connives
72          addCard(Zone.BATTLEFIELD, playerA, "Hypnotic Grifter");
73          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
74          // Target opponent loses 3 life and you gain 3 life.
75          // Madness {B}
76          addCard(Zone.HAND, playerA, "Alms of the Vein", 1); // To discard and use mad
77  
78          setStrictChooseMode(true);
79  
80          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{3}");
81          setChoice(playerA, "Alms of the Vein");
82          setChoice(playerA, "Yes"); // Cast for Madness
83          addTarget(playerA, playerB); // Target for Alms of the Vein
84  
85          setStopAt(1, PhaseStep.BEGIN_COMBAT);
86          execute();
87  
88          assertPowerToughness(playerA, "Hypnotic Grifter", 2, 3); // Discarded a non-land
89          assertHandCount(playerA, 1); // 1 from drawing at start of turn
90          assertGraveyardCount(playerA, "Alms of the Vein", 1);
91  
92          assertLife(playerA, 23); // 3 life gained from Alms of the Vein
93          assertLife(playerB, 17); // 3 life lost from Alms of the Vein
94      }

100     @Test

102         // Choose three. You may choose the same mode more than once.
103         //• Until end of turn, target creature loses all abilities and has base power and toughness 1/1.
104         //• Target creature connives. (Draw a card, then discard a card. If you discarded a nonland card, put a +1/+1 counter on that creature.)
105         //• Target player returns a creature card from their graveyard to their hand.
106         addCard(Zone.HAND, playerA, "Obscura Confluence"); // {1}{W}{U}{B}
107         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
108         addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
109         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
110 
111         addCard(Zone.HAND, playerB, "Plains", 3); // 3 land cards for conniving
112         addCard(Zone.BATTLEFIELD, playerB, "Crazed Goblin"); // 1/1 creature that is made to connive
113 
114         setStrictChooseMode(true);
115 
116         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Obscura Confluence");
117         setModeChoice(playerA, "2");
118         setModeChoice(playerA, "2");
119         setModeChoice(playerA, "2");
120         addTarget(playerA, "Crazed Goblin");
121         addTarget(playerA, "Crazed Goblin");
122         addTarget(playerA, "Crazed Goblin");
123         setChoice(playerB, "Plains");

135     /**
136      * Reported bug: https://github.com/magefree/mage/issues/9252
137      * Connive fizzles if the creature that connived leaves the battlefield before connive resolves.
138      *
139      * Ruling:
140      *      If a resolving spell or ability instructs a specific creature to connive but that creature has left the battlefield,
141      *      the creature still connives.
142      *      If you discard a nonland card this way, you won’t put a +1/+1 counter on anything.
143      *      Abilities that trigger “when [that creature] connives” will trigger.
144      *      (2022-04-29)
145      */
146     @Test
147     public void conniveDoesNotFizzle() {
148         // {4}{U}
149         // 3/2
150         // When Psychic Pickpocket enters the battlefield, it connives.
151         // When it connives this way, return up to one target nonland permanent to its owner’s hand.
152         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
153         addCard(Zone.HAND, playerA, "Psychic Pickpocket");
154         addCard(Zone.HAND, playerA, "Swamp");
155 
156         addCard(Zone.HAND, playerB, "Lightning Bolt");
157         addCard(Zone.BATTLEFIELD, playerB, "Mountain");
158         addCard(Zone.BATTLEFIELD, playerB, "Sol Ring"); // Target for Psychic Pickpocket
159 
160         setStrictChooseMode(true);
161 
162         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Psychic Pickpocket");
163         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA, true); // Wait for ETB
164         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Psychic Pickpocket", "When");
165         setChoice(playerA, "Swamp"); // Discard for connive
166         addTarget(playerA, "Sol Ring"); // Target for Psychic Pickpocket
167 
168         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
169 
170         execute();
171 
172         assertHandCount(playerA, 1); // Drew a card at start of turn
173         assertHandCount(playerB, "Sol Ring", 1); // Returned by Psychic Pickpocket's ability
174         assertGraveyardCount(playerA, "Swamp", 1); // Connived
175         assertGraveyardCount(playerA, "Psychic Pickpocket", 1); // Destroyed by Lightning Bolt
176         assertGraveyardCount(playerB, "Lightning Bolt", 1);
177     }

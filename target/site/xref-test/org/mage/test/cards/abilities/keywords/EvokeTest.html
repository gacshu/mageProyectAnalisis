15  public class EvokeTest extends CardTestPlayerBase {
16  
17     /*
18      * Shriekmaw         {4}{B}
19      * Creature â€” Elemental
20      * 3/2
21      * Fear (This creature can't be blocked except by artifact creatures and/or black creatures.)
22      * When Shriekmaw enters the battlefield, destroy target nonartifact, nonblack creature.
23      * Evoke {1}{B}  (You may cast this spell for its evoke cost. If you do, it's sacrificed when it enters the battlefield.)
24  
25      Exhume  {1}{B}
26      Sorcery
27      Each player puts a creature card from their graveyard onto the battlefield.
28  
29      */
30  
31      @Test
32      public void testCreatureComesIntoPlay() {
33          // Check that Lion goes to graveyard from evoke ability
34          // Check that evoke does not trigger again to sacrifice Shriekmaw if it's exhumed
35  
36          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
37          addCard(Zone.HAND, playerA, "Shriekmaw");
38          addCard(Zone.HAND, playerA, "Exhume");
39  
40          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
41  
42          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Shriekmaw");
43          setChoice(playerA, true);
44          // addTarget(playerA, "Silvercoat Lion"); Autochosen, only target
45          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Exhume");
46          setStopAt(1, PhaseStep.END_TURN);
47          execute();
48  
49          assertLife(playerA, 20);
50          assertLife(playerB, 20);
51  
52          assertGraveyardCount(playerA, "Exhume", 1);
53          assertGraveyardCount(playerB, "Silvercoat Lion", 1); // went again to graveyard from shriekmaw's triggered ability
54  
55          assertPermanentCount(playerA, "Shriekmaw", 1);
56      }

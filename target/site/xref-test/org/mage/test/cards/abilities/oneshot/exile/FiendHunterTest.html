27      @Test
28      public void testExileWorks() {
29          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
30          // When Fiend Hunter enters the battlefield, you may exile another target creature.
31          // When Fiend Hunter leaves the battlefield, return the exiled card to the battlefield under its owner's control.
32          addCard(Zone.HAND, playerA, "Fiend Hunter"); // Creature - Human Cleric 1/3  {1}{W}{W}
33  
34          // At the beginning of your upkeep, you lose 1 life and put a 1/1 black Faerie Rogue creature token with flying onto the battlefield.
35          addCard(Zone.BATTLEFIELD, playerB, "Primeval Titan");
36  
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fiend Hunter");
38          // Target autochosen since only one option
39  
40          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
41          execute();
42  
43          assertExileCount("Primeval Titan", 1);
44          assertPermanentCount(playerA, "Fiend Hunter", 1);
45  
46      }

48      @Test
49      public void testExileAndReturnWorks() {
50          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
51          // When Fiend Hunter enters the battlefield, you may exile another target creature.
52          // When Fiend Hunter leaves the battlefield, return the exiled card to the battlefield under its owner's control.
53          addCard(Zone.HAND, playerA, "Fiend Hunter"); // Creature - Human Cleric 1/3  {1}{W}{W}
54          // When Restoration Angel enters the battlefield, you may exile target non-Angel creature you control, then return that card to the battlefield under your control
55          addCard(Zone.HAND, playerA, "Restoration Angel"); // Creature - Angel 3/4  {3}{W}
56  
57          // At the beginning of your upkeep, you lose 1 life and put a 1/1 black Faerie Rogue creature token with flying onto the battlefield.
58          addCard(Zone.BATTLEFIELD, playerB, "Primeval Titan");
59          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
60  
61          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Fiend Hunter");
62          // addTarget(playerA, "Primeval Titan"); Autochosen, only option
63  
64          // When Restoration Angel enters the battlefield, you may exile target non-Angel creature you control, then return that card to the battlefield under your control
65          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Restoration Angel");
66          // addTarget(playerA, "Fiend Hunter"); Autochosen, only option
67  
68          setStopAt(4, PhaseStep.PRECOMBAT_MAIN);
69          execute();
70  
71          assertExileCount("Silvercoat Lion", 1);
72          assertPermanentCount(playerA, "Fiend Hunter", 1);
73          assertPermanentCount(playerA, "Restoration Angel", 1);
74          assertPermanentCount(playerB, "Primeval Titan", 1);
75  
76      }

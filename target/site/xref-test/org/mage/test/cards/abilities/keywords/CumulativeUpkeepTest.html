17      @Test
18      public void basicTest() {
19          setStrictChooseMode(true);
20          
21          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
22          // Flying; fear
23          // Cumulative upkeep {B}
24          addCard(Zone.HAND, playerA, "Phobian Phantasm"); // Creature {1}{B}{B} 3/3
25  
26          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Phobian Phantasm");
27  
28          // Phobian Phantasm - CumulativeUpkeepAbility: Cumulative upkeep {B}
29          setChoice(playerA, true); // Pay {B}?
30          attack(3, playerA, "Phobian Phantasm");
31          checkPermanentCounters("Age counters", 3, PhaseStep.PRECOMBAT_MAIN, playerA, "Phobian Phantasm", CounterType.AGE, 1);
32          
33          setChoice(playerA, true); // Pay {B}{B}?
34          attack(5, playerA, "Phobian Phantasm");
35          checkPermanentCounters("Age counters", 5, PhaseStep.PRECOMBAT_MAIN, playerA, "Phobian Phantasm", CounterType.AGE, 2);
36          
37          setChoice(playerA, false); // Pay {B}{B}{B}?
38          
39          setStopAt(7, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          assertGraveyardCount(playerA, "Phobian Phantasm", 1);
43          
44          assertLife(playerA, 20);
45          assertLife(playerB, 14);
46      }

49      /**
50       I changed control of a Illusions of Grandeur to an AI after cumulative upkeep had triggered but before it resolved.
51       I chose not to pay the upkeep cost and then either the AI sacrificed it or I sacrificed it, neither of which should happen.
52       I can't sacrifice it because it's not under my control. The AI can't sacrifice it because they are not instructed to do so.
53  
54       Here is the reminder text for cumulative upkeep:
55       At the beginning of your upkeep, put an age counter on this permanent, then sacrifice it unless you pay its upkeep cost for each age counter on it.
56       */
57      @Test
58      public void controlChangeTest() {
59          setStrictChooseMode(true);
60          
61          // Whenever Kor Celebrant or another creature enters the battlefield under your control, you gain 1 life.
62          addCard(Zone.HAND, playerB, "Kor Celebrant", 1); // Creature {2}{W}
63          addCard(Zone.BATTLEFIELD, playerB, "Plains", 3);
64          
65          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
66          // Cumulative upkeep {2}
67          // When Illusions of Grandeur enters the battlefield, you gain 20 life.
68          // When Illusions of Grandeur leaves the battlefield, you lose 20 life.
69          addCard(Zone.HAND, playerA, "Illusions of Grandeur"); // Enchantment {3}{U}
70          
71          // At the beginning of your upkeep, you may exchange control of target nonland permanent you control and target nonland permanent an opponent controls with an equal or lesser converted mana cost.        
72          addCard(Zone.HAND, playerA, "Puca's Mischief"); // Enchantment {3}{U}
73  
74          
75          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Illusions of Grandeur");
76               
77          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Kor Celebrant");
78          
79          // Illusions of Grandeur - CumulativeUpkeepAbility: Cumulative upkeep {2}
80          setChoice(playerA, true); // Pay {2}?
81          
82          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Puca's Mischief");
83          
84          setChoice(playerA, "Cumulative upkeep"); // Triggered list (total 2) which trigger goes first on the stack
85          addTarget(playerA, "Illusions of Grandeur"); // Own target permanent of Puca's Mischief
86          addTarget(playerA, "Kor Celebrant"); // Opponent's target permanent of Puca's Mischief
87          
88          setChoice(playerA, true); // At the beginning of your upkeep, you may exchange control of target nonland permanent you control and target nonland permanent an opponent controls with an equal or lesser converted mana cost.
89          setChoice(playerA, false); // Pay {2}{2}?
90          
91          checkPermanentCounters("Age counters", 5, PhaseStep.PRECOMBAT_MAIN, playerB, "Illusions of Grandeur", CounterType.AGE, 2);
92          
93          setStopAt(5, PhaseStep.BEGIN_COMBAT);
94          execute();
95  
96          assertLife(playerA, 40);
97          assertLife(playerB, 21);  
98          
99          assertPermanentCount(playerA, "Kor Celebrant", 1);
100         assertPermanentCount(playerB, "Illusions of Grandeur", 1);
101         
102 
103     }

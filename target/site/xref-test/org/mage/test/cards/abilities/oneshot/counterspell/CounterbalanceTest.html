3   package org.mage.test.cards.abilities.oneshot.counterspell;
4   
5   import mage.constants.PhaseStep;
6   import mage.constants.Zone;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   * Counterbalance
12   * Enchantment, UU
13   * Whenever an opponent casts a spell, you may reveal the top card of your library. If you do, counter that spell
14   * if it has the same converted mana cost as the revealed card.
15   *
16   * @author LevelX2
17   */
18  public class CounterbalanceTest extends CardTestPlayerBase {
19  
20      /**
21       * Test that X mana costs from spells are taken into account to calculate the converted mana costs
22       * of stack objects
23       */
24      @Test
25      public void testCommand() {
26          addCard(Zone.HAND, playerA, "Death Grasp");
27          // Sorcery {X}{W}{B}
28          // Death Grasp deals X damage to any target. You gain X life.
29          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
30          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
31  
32          addCard(Zone.BATTLEFIELD, playerB, "Counterbalance");
33          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
34  
35          addCard(Zone.LIBRARY, playerB, "Desert Twister"); // cmc = 6 ({G}{G}{4} because DeathGrasp = 2 + 4 (of X) = 6
36          skipInitShuffling(); // so the set to top card stays at top
37  
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Death Grasp", "targetPlayer=PlayerB");
39          setChoice(playerA, "X=4");
40  
41          setStopAt(1, PhaseStep.BEGIN_COMBAT);
42          execute();
43  
44          assertLife(playerA, 20);
45          assertLife(playerB, 20);
46  
47          assertGraveyardCount(playerA, "Death Grasp", 1);
48          assertGraveyardCount(playerA, 1);
49          assertGraveyardCount(playerB, 0);
50  
51      }
52  
53      /**
54       * Test that if the top card is a split card, the total of both halves of the split card
55       * count to counter the spell. If one of the split cards halves has the equal casting
56       * cost, the spell is not countered.
57       *
58       */
59  
60      @Test
61      public void testSplitCard() {
62          addCard(Zone.HAND, playerA, "Nessian Courser");
63          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
64  
65          addCard(Zone.BATTLEFIELD, playerB, "Counterbalance");
66          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
67  
68          addCard(Zone.LIBRARY, playerB, "Wear // Tear"); // CMC 3
69          skipInitShuffling(); // so the set to top card stays at top
70  
71          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nessian Courser");
72          setChoice(playerB, true);
73  
74          setStopAt(1, PhaseStep.BEGIN_COMBAT);
75          execute();
76  
77          assertLife(playerA, 20);
78          assertLife(playerB, 20);
79  
80          assertPermanentCount(playerA, "Nessian Courser", 0);
81          assertGraveyardCount(playerA, "Nessian Courser", 1);
82          assertGraveyardCount(playerA, 1);
83          assertGraveyardCount(playerB, 0);
84  
85      }
86  }

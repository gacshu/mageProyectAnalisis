26      private void assertDayNight(boolean daytime) {
27          Assert.assertTrue("It should not be neither day nor night", currentGame.hasDayNight());
28          Assert.assertTrue("It should be " + (daytime ? "day" : "night"), currentGame.checkDayNight(daytime));
29          Assert.assertFalse("It should not be " + (daytime ? "night" : "day"), currentGame.checkDayNight(!daytime));
30      }
31  
32      private void assertRuffianSmasher(boolean daytime) {
33          assertDayNight(daytime);
34          if (daytime) {
35              assertPowerToughness(playerA, ruffian, 2, 5);
36              assertPermanentCount(playerA, smasher, 0);
37          } else {
38              assertPermanentCount(playerA, ruffian, 0);
39              assertPowerToughness(playerA, smasher, 6, 5);
40          }
41      }
42  
43      private void setDayNight(int turn, PhaseStep phaseStep, boolean daytime) {
44          runCode("set game to " + (daytime ? "day" : "night"), turn, phaseStep, playerA, (i, p, game) -> game.setDaytime(daytime));
45      }

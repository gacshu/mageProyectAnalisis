13      /**
14       * Checks that pro black can still be sacrificed (i.e. Geth's Verdict)
15       */
16      @Test
17      public void gethVerdictVersusProtectionFromBlack() {
18          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
19          addCard(Zone.HAND, playerA, "Geth's Verdict"); // {B}{B} target player sacrifices a creature and loses a life
20          addCard(Zone.BATTLEFIELD, playerB, "White Knight"); // {W}{W} pro-black
21  
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Geth's Verdict", playerB);
23          setStopAt(1, PhaseStep.BEGIN_COMBAT);
24          execute();
25  
26          assertPermanentCount(playerB, "White Knight", 0);
27          assertLife(playerB, 19);
28      }

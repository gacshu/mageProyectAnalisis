14  public class ProteanHydraTest extends CardTestPlayerBase {
15  
16      @Test
17      public void testEnteringWithCounters() {
18          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
19          // Protean Hydra enters the battlefield with X +1/+1 counters on it.
20          // If damage would be dealt to Protean Hydra, prevent that damage and remove that many +1/+1 counters from it.
21          // Whenever a +1/+1 counter is removed from Protean Hydra, put two +1/+1 counters on it at the beginning of the next end step.
22          addCard(Zone.HAND, playerA, "Protean Hydra"); // CREATURE - {X}{G}
23  
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Protean Hydra");
25  
26          setStopAt(1, PhaseStep.BEGIN_COMBAT);
27          execute();
28  
29          assertLife(playerA, 20);
30          assertLife(playerB, 20);
31  
32          for (Permanent permanent : currentGame.getBattlefield().getAllPermanents()) {
33              if (permanent.getName().equals("Forest")) {
34                  // check all mana was spent
35                  Assert.assertTrue(permanent.isTapped());
36              }
37          }
38  
39          assertPermanentCount(playerA, "Protean Hydra", 1);
40  
41          Permanent proteanHydra = getPermanent("Protean Hydra", playerA.getId());
42          Assert.assertEquals(4, proteanHydra.getPower().getValue());
43          Assert.assertEquals(4, proteanHydra.getToughness().getValue());
44      }
45  }

25      private void assertDaxosBoost(boolean mustHave) {
26          if (mustHave) {
27              assertPowerToughness(playerA, daxosCard, 5, 5);
28              assertType(daxosCard, CardType.CREATURE, SubType.DEMON);
29              assertAbility(playerA, daxosCard, FlyingAbility.getInstance(), true);
30              assertAbility(playerA, daxosCard, HasteAbility.getInstance(), true);
31          } else {
32              assertPowerToughness(playerA, daxosCard, 0, 0);
33              assertNotSubtype(daxosCard, SubType.DEMON);
34              assertAbility(playerA, daxosCard, FlyingAbility.getInstance(), false);
35              assertAbility(playerA, daxosCard, HasteAbility.getInstance(), false);
36          }
37      }

191     public void playDaxosAndVampire(boolean castVampireDifferentWay) {
192         // 112.10c  If two or more effects add and remove the same ability, in general the most recent one prevails.
193         // 613.7 -- dependacy effects (Mephidross Vampire must ALWAYS wait Daxos effect, not timestamp)
194         addCard(Zone.HAND, playerA, daxosCard, 1);
195         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
196         // Each creature you control is a Vampire in addition to its other creature types
197         addCard(Zone.HAND, playerA, "Mephidross Vampire", 1); // {4}{B}{B}
198         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8);
199         //
200         addCard(Zone.HAND, playerA, "Archetype of Courage", 1); // Enchantment {1}{W}{W} to trigger Daxos
201         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
202 
203         // dax cast
204         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, daxosCard);
205         checkPermanentCount("dax exist", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, daxosCard, 1);
206 
207         // give dax a new type (on next turn) -- effects dependacy
208         if (castVampireDifferentWay) {
209             castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Archetype of Courage"); // make dax to creature
210             waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
211             castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Mephidross Vampire"); // give vampire to creatures
212         } else {
213             // Make sure not white mana is used here to cast the vampire
214             checkManaPool("before B mana", 3, PhaseStep.PRECOMBAT_MAIN, playerA, "B", 0);
215             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
216             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
217             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
218             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
219             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
220             activateManaAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}");
221             checkManaPool("after B mana", 3, PhaseStep.PRECOMBAT_MAIN, playerA, "B", 6);
222             castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Mephidross Vampire"); // give vampire to creatures
223             waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
224             castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Archetype of Courage"); // make dax to creature
225         }
226 
227         checkPT("dax boost", 3, PhaseStep.BEGIN_COMBAT, playerA, daxosCard, 5, 5);
228         checkAbility("dax fly", 3, PhaseStep.BEGIN_COMBAT, playerA, daxosCard, FlyingAbility.class, true);
229         checkSubType("dax is vampire", 3, PhaseStep.BEGIN_COMBAT, playerA, daxosCard, SubType.VAMPIRE, true);
230 
231         setStopAt(3, PhaseStep.END_TURN);
232         execute();
233     }

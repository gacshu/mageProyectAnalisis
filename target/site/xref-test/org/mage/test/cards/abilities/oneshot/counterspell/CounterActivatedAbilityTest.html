12  public class CounterActivatedAbilityTest extends CardTestPlayerBase {
13  
14      /**
15       * The Card 'Sqelch' is bugged. In a game versus a human I tried to counter
16       * a Toloria West activation as well as a Elspeth Sun's Champion activation
17       * to make the game roll back to the begining of the turn and to show a pop
18       * pup with an error message which I cannot post here due to forum
19       * limitations.
20       *
21       */
22      @Test
23      public void testSquelch() {
24          // +1: Create three 1/1 white Soldier creature tokens.
25          // -3: Destroy all creatures with power 4 or greater.
26          // -7: You get an emblem with "Creatures you control get +2/+2 and have flying."
27          addCard(Zone.BATTLEFIELD, playerA, "Elspeth, Sun's Champion", 1);
28  
29          // Counter target activated ability
30          addCard(Zone.HAND, playerB, "Squelch", 1); // Instant {1}{U}
31          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
32  
33          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1");
34  
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Squelch");
36  
37          setStopAt(1, PhaseStep.BEGIN_COMBAT);
38          execute();
39  
40          assertGraveyardCount(playerB, "Squelch", 1);
41          assertPermanentCount(playerA, "Soldier Token", 0);
42  
43      }

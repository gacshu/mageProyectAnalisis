15      /**
16       * Hellspark Elemental (and probably other cards with the unearth ability) -
17       * If I unearth the elemental, attack, and then go to the end of my turn
18       * both the "sacrifice" and "exile" clauses will trigger and the game will
19       * ask me which one I want to put on the stack first. If I choose
20       * "sacrifice" first and "exile" second, all good, the exile part resolves
21       * first and the elemental is exiled, the sacrifice part does nothing
22       * afterwards. But if I choose "exile" first and "sacrifice" second then the
23       * elemental will be sacrificed and placed on my graveyard and after that
24       * the "exile" resolves but does nothing, as I'm guessing it can't "find"
25       * the elemental anymore and so it stays in my graveyard, despite the fact
26       * that because I use its unearth ability it should always be exiled once
27       * leaving the battlefield no matter what. The bug should be easy to
28       * reproduce if following the order I mention above (click the exile part,
29       * so the sacrifice goes on the top of the stack).
30       */
31      @Test
32      public void testUnearthAttackExile() {
33          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
34          // 3/1 - Trample, haste
35          // At the beginning of the end step, sacrifice Hellspark Elemental.
36          // Unearth {1}{R} ({1}{R}: Return this card from your graveyard to the battlefield.
37          // It gains haste. Exile it at the beginning of the next end step or if it would
38          // leave the battlefield. Unearth only as a sorcery.)
39          addCard(Zone.GRAVEYARD, playerA, "Hellspark Elemental", 1);
40  
41          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Unearth");
42  
43          attack(1, playerA, "Hellspark Elemental");
44  
45          setStopAt(2, PhaseStep.UNTAP);
46          execute();
47  
48          assertGraveyardCount(playerA, "Hellspark Elemental", 0);
49          assertLife(playerB, 17);
50  
51          assertPermanentCount(playerA, "Hellspark Elemental", 0);
52          assertExileCount("Hellspark Elemental", 1);
53      }

55      /**
56       * Reported bug: Cards with unearth (e.g. Undead Leotau) are currently
57       * bugged. When you bring a creature back from the graveyard with unearth,
58       * it [should] get exiled at end of turn normally, [but instead] a copy of
59       * the card stays on the battlefield under your control permanently.
60       */
61      @Test
62      public void testUndeadLeotau() {
63  
64          //{R}: Undead Leotau gets +1/-1 until end of turn.
65          // Unearth {2}{B}
66          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
67          addCard(Zone.GRAVEYARD, playerA, "Undead Leotau", 1); // 3/4
68          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Unearth");
69          attack(1, playerA, "Undead Leotau");
70  
71          setStopAt(2, PhaseStep.UNTAP);
72          execute();
73  
74          assertGraveyardCount(playerA, "Undead Leotau", 0);
75          assertLife(playerB, 17);
76  
77          assertPermanentCount(playerA, "Undead Leotau", 0);
78          assertExileCount("Undead Leotau", 1);
79      }

81      /**
82       * At start of the end phase, creatures phased out by Teferi's Veil still
83       * exiled by unearth if they were put to battlefield by unearth.<br>
84       *
85       * http://www.mtgsalvation.com/forums/magic-fundamentals/magic-rulings/magic-rulings-archives/294911-teferis-veil-and-unearth
86       */
87      @Test
88      public void testUnearthWithPhasing() {
89          // Whenever a creature you control attacks, it phases out at end of combat.
90          addCard(Zone.BATTLEFIELD, playerA, "Teferi's Veil", 1);
91          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
92          // Unearth {B} ({B}: Return this card from your graveyard to the battlefield. It gains haste. Exile it at the beginning of the next end step or if it would leave the battlefield. Unearth only as a sorcery.)
93          addCard(Zone.GRAVEYARD, playerA, "Dregscape Zombie", 1); // 2/1
94  
95          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Unearth");
96          attack(1, playerA, "Dregscape Zombie");
97  
98          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
99          execute();
100 
101         assertGraveyardCount(playerA, "Dregscape Zombie", 0);
102         assertLife(playerB, 18);
103         assertExileCount("Dregscape Zombie", 0);
104 
105         assertPermanentCount(playerA, "Dregscape Zombie", 1);
106 
107     }

6   package org.mage.test.cards.abilities.flicker;
7   
8   import mage.constants.PhaseStep;
9   import mage.constants.Zone;
10  import org.junit.Test;
11  import org.mage.test.serverside.base.CardTestPlayerBase;
12  
13  /**
14   *
15   * @author jeffwadsworth
16   */
17  public class GisaGloriousResurrectorBlinkedTest extends CardTestPlayerBase {
18      
19      @Test
20      public void testNormalCastAndRetrieve() {
21          setStrictChooseMode(true);
22          addCard(Zone.BATTLEFIELD, playerB, "Memnite");
23          addCard(Zone.BATTLEFIELD, playerA, "Gisa, Glorious Resurrector");
24          addCard(Zone.HAND, playerA, "Lightning Bolt");
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain");  // mana for Lightning Bolt
26  
27          // The Memnite dies and gets exiled.  It should be put onto the battlefield under playerA's control during the next upkeep
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Memnite");
29  
30          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
31          execute();
32  
33          // should be under playerA control
34          assertPermanentCount(playerA, "Memnite", 1);
35      }
36      
37      @Test
38      public void testBlinkedGisa() {
39          setStrictChooseMode(true);
40          addCard(Zone.BATTLEFIELD, playerB, "Memnite");
41          addCard(Zone.BATTLEFIELD, playerA, "Gisa, Glorious Resurrector");
42          addCard(Zone.HAND, playerA, "Lightning Bolt");
43          addCard(Zone.HAND, playerA, "Flicker");
44          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);  // mana for Lightning Bolt
45          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);  // mana for Flicker
46  
47          // The Memnite dies and gets exiled.
48          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Memnite");
49          
50          // Blink the Gisa, Glorious Resurrector
51          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Flicker", "Gisa, Glorious Resurrector");
52  
53          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
54          execute();
55  
56          // exiled Memnite stays in exile zone
57          assertPermanentCount(playerA, "Memnite", 0);
58      }
59  }

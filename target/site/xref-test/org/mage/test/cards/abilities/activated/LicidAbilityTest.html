20      /**
21       * Activate on another creature
22       */
23      @Test
24      public void BasicUsageTest() {
25          addCard(Zone.BATTLEFIELD, playerA, "Pillarfield Ox");
26          // {R}, {T}: Enraging Licid loses this ability and becomes an Aura enchantment with enchant creature. Attach it to target creature. You may pay {R} to end this effect.
27          // Enchanted creature has haste.
28          addCard(Zone.BATTLEFIELD, playerA, "Enraging Licid");
29          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
30  
31          activateAbility(1, PhaseStep.UPKEEP, playerA, "{R},", "Pillarfield Ox");
32          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
33  
34          execute();
35  
36          assertAbility(playerA, "Pillarfield Ox", HasteAbility.getInstance(), true);
37          assertAbility(playerA, "Enraging Licid", new LicidAbility(new ColoredManaCost(ColoredManaSymbol.R), new ColoredManaCost(ColoredManaSymbol.R)), false);
38          assertType("Enraging Licid", CardType.ENCHANTMENT, true);
39          assertType("Enraging Licid", CardType.CREATURE, false);
40      }

42      /**
43       * Use special action to remove the continuous effect
44       */
45      @Test
46      @Ignore("Test player can't activate special actions yet")
47      public void SpecialActionTest() {
48          addCard(Zone.BATTLEFIELD, playerA, "Pillarfield Ox");
49          // {R}, {T}: Enraging Licid loses this ability and becomes an Aura enchantment with enchant creature. Attach it to target creature. You may pay {R} to end this effect.
50          // Enchanted creature has haste.
51          addCard(Zone.BATTLEFIELD, playerA, "Enraging Licid");
52          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
53  
54          activateAbility(1, PhaseStep.UPKEEP, playerA, "{R},", "Pillarfield Ox");
55          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: End");
56          setStopAt(1, PhaseStep.BEGIN_COMBAT);
57  
58          execute();
59  
60          assertActionsCount(playerA, 0);
61          assertAbility(playerA, "Pillarfield Ox", HasteAbility.getInstance(), false);
62          assertAbility(playerA, "Enraging Licid", new LicidAbility(new ColoredManaCost(ColoredManaSymbol.R), new ColoredManaCost(ColoredManaSymbol.R)), true);
63          assertType("Enraging Licid", CardType.ENCHANTMENT, false);
64          assertType("Enraging Licid", CardType.CREATURE, true);
65      }

67      /**
68       * Licid should die if enchanted creature dies
69       */
70      @Test
71      public void EnchantedCreatureDiesTest() {
72          addCard(Zone.BATTLEFIELD, playerA, "Pillarfield Ox");
73          // {R}, {T}: Enraging Licid loses this ability and becomes an Aura enchantment with enchant creature. Attach it to target creature. You may pay {R} to end this effect.
74          // Enchanted creature has haste.
75          addCard(Zone.BATTLEFIELD, playerA, "Enraging Licid");
76          // Destroy target nonblack creature.
77          addCard(Zone.HAND, playerB, "Doom Blade");
78          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
79          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
80  
81          activateAbility(1, PhaseStep.UPKEEP, playerA, "{R},", "Pillarfield Ox");
82          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Doom Blade", "Pillarfield Ox");
83          setStopAt(1, PhaseStep.BEGIN_COMBAT);
84  
85          execute();
86  
87          assertPermanentCount(playerA, "Enraging Licid", 0);
88  
89          assertGraveyardCount(playerA, "Pillarfield Ox", 1);
90          assertGraveyardCount(playerA, "Enraging Licid", 1);
91      }

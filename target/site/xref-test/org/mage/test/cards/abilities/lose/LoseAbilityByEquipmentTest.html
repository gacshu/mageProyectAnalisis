16  
17      /**
18       * Tests that gaining flying by and after that losing flying by eqipments results in not have flying
19       */
20      @Test
21      public void testGainVsLoseByEquipmentAbility() {
22          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
23          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
24          addCard(Zone.BATTLEFIELD, playerA, "Island", 4);
25  
26          addCard(Zone.HAND, playerA, "Magebane Armor"); // loses Flying
27          addCard(Zone.HAND, playerA, "Cobbled Wings"); // gives Flying
28  
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Magebane Armor");
30          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Cobbled Wings", true);
31          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip {1}", "Silvercoat Lion"); // give Flying
32          activateAbility(3, PhaseStep.POSTCOMBAT_MAIN, playerA, "Equip {2}", "Silvercoat Lion"); // lose Flying
33  
34          setStopAt(3, PhaseStep.END_TURN);
35          execute();
36  
37          assertLife(playerA, 20);
38          Permanent silvercoatLion = getPermanent("Silvercoat Lion", playerA.getId());
39          Assert.assertNotNull(silvercoatLion);
40          Assert.assertEquals("Silvercoat Lion equipments", 2, silvercoatLion.getAttachments().size());
41          Assert.assertEquals("Silvercoat Lion power",4, silvercoatLion.getPower().getValue());
42          Assert.assertEquals("Silvercoat Lion toughness",6, silvercoatLion.getToughness().getValue());
43  
44          // should NOT have flying
45          Assert.assertFalse("Silvercoat Lion has flying but shouldn't have",silvercoatLion.getAbilities().contains(FlyingAbility.getInstance()));
46      }

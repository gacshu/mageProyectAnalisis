15      /**
16       * 702.42. Modular 702.42a Modular represents both a static ability and a
17       * triggered ability. “Modular N” means “This permanent enters the
18       * battlefield with N +1/+1 counters on it” and “When this permanent is put
19       * into a graveyard from the battlefield, you may put a +1/+1 counter on
20       * target artifact creature for each +1/+1 counter on this permanent.”
21       * 702.42b If a creature has multiple instances of modular, each one works
22       * separately.
23       *
24       */
25      /**
26       * Arcbound Bruiser Artifact Creature — Golem 0/0, 5 (5) Modular 3 (This
27       * enters the battlefield with three +1/+1 counters on it. When it dies, you
28       * may put its +1/+1 counters on target artifact creature.)
29       *
30       */
31      /**
32       * Arcbound Hybrid Artifact Creature — Beast 0/0, 4 (4) Haste Modular 2
33       * (This enters the battlefield with two +1/+1 counters on it. When it dies,
34       * you may put its +1/+1 counters on target artifact creature.)
35       */
36      @Test
37      public void testModularEnters() {
38          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
39          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
40          addCard(Zone.HAND, playerA, "Arcbound Bruiser");
41          addCard(Zone.HAND, playerA, "Arcbound Hybrid");
42  
43          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arcbound Bruiser", true);
44          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arcbound Hybrid");
45  
46          setStopAt(1, PhaseStep.BEGIN_COMBAT);
47          execute();
48  
49          assertPermanentCount(playerA, "Arcbound Bruiser", 1);
50          assertPermanentCount(playerA, "Arcbound Hybrid", 1);
51          assertPowerToughness(playerA, "Arcbound Bruiser", 3, 3);
52          assertPowerToughness(playerA, "Arcbound Hybrid", 2, 2);
53  
54      }

56      @Test
57      public void testModularLeaves() {
58          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
59          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
60          addCard(Zone.HAND, playerA, "Arcbound Bruiser");
61          addCard(Zone.HAND, playerA, "Arcbound Hybrid");
62          addCard(Zone.HAND, playerA, "Lightning Bolt");
63  
64          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arcbound Bruiser", true);
65          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arcbound Hybrid", true);
66          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Arcbound Bruiser");
67          setStopAt(3, PhaseStep.BEGIN_COMBAT);
68          setChoice(playerA, true);
69          execute();
70  
71          assertPermanentCount(playerA, "Arcbound Bruiser", 0);
72          assertPermanentCount(playerA, "Arcbound Hybrid", 1);
73          assertGraveyardCount(playerA, "Arcbound Bruiser", 1);
74          assertPowerToughness(playerA, "Arcbound Hybrid", 5, 5);
75  
76      }

78      /**
79       * My Inkmoth Nexus was in creature "form". My Arcbound Ravager died. I
80       * could not transfer his counters on my Inkmoth "creature".
81       */
82      @Test
83      public void testInkmothNexus() {
84          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
85          addCard(Zone.BATTLEFIELD, playerA, "Inkmoth Nexus");
86          addCard(Zone.HAND, playerA, "Arcbound Ravager");
87  
88          addCard(Zone.HAND, playerB, "Lightning Bolt");
89          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
90  
91          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arcbound Ravager");
92          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}: {this} becomes");
93  
94          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Lightning Bolt", "Arcbound Ravager");
95  
96          setStopAt(1, PhaseStep.END_TURN);
97          execute();
98  
99          assertGraveyardCount(playerA, "Arcbound Ravager", 1);
100         assertGraveyardCount(playerB, "Lightning Bolt", 1);
101 
102         assertCounterCount("Inkmoth Nexus", CounterType.P1P1, 1);
103 
104     }

106     /**
107      * If a creature with modular dies due to -1/-1 counters, it still had those counters when it left
108      * and therefore will still transfer them to a creature on the battlefield
109      */
110     @Test
111     public void testMinusCounters() {
112         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
113         addCard(Zone.BATTLEFIELD, playerA, "Arcbound Bruiser");
114         addCard(Zone.BATTLEFIELD, playerA, "Memnite");
115         addCard(Zone.HAND, playerA, "Puncture Blast");
116 
117         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Puncture Blast", "Arcbound Bruiser");
118         setChoice(playerA, true);
119 
120         setStopAt(1, PhaseStep.END_TURN);
121         execute();
122

128     /**
129      * If a creature with modular dies and returns and dies again before any modular triggers resolve,
130      * the modular triggers should use that creature's counters from each time it died
131      * rather than the most recent time it died
132      */
133     @Test
134     public void testReturnedAndKilledAgain() {
135         addCard(Zone.BATTLEFIELD, playerA, "Badlands", 11);
136         addCard(Zone.BATTLEFIELD, playerA, "Arcbound Lancer");
137         addCard(Zone.BATTLEFIELD, playerA, "Memnite");
138         addCard(Zone.HAND, playerA, "Makeshift Mannequin");
139         addCard(Zone.HAND, playerA, "Puncture Blast");
140         addCard(Zone.HAND, playerA, "Flame Slash");
141         addCard(Zone.HAND, playerA, "Murder");
142 
143         // put three -1/-1 counters on lancer, which leaves it with one +1/+1
144         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Puncture Blast", "Arcbound Lancer");
145         setChoice(playerA, true, 2);
146         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
147 
148         // kill lancer with one +1/+1 counter on it
149         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Flame Slash", "Arcbound Lancer");
150         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
151 
152         // in response to modular trigger, return lancer to the battlefield
153         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Makeshift Mannequin", "Arcbound Lancer", "When it dies");
154         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true);
155 
156         // kill lancer again with original modular trigger on the stack
157         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Murder", "Arcbound Lancer", "When it dies");
158 
159         setStopAt(1, PhaseStep.END_TURN);
160         execute();
161 
162         assertGraveyardCount(playerA, "Arcbound Lancer", 1);
163         assertGraveyardCount(playerA, "Puncture Blast", 1);
164         assertGraveyardCount(playerA, "Flame Slash", 1);
165         assertGraveyardCount(playerA, "Murder", 1);
166         // Memnite should have 1 counter for the first trigger and 4 from the second
167         assertCounterCount("Memnite", CounterType.P1P1, 1 + 4);
168     }

91      @Test
92      public void test_PlayConvoke_Manual() {
93          // {2}{R}{R}
94          // Convoke (Your creatures can help cast this spell. Each creature you tap while casting this spell pays for {1} or one mana of that creature’s color.)
95          // Stoke the Flames deals 4 damage to any target.
96          addCard(Zone.HAND, playerA, "Stoke the Flames", 1);
97          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
98          addCard(Zone.BATTLEFIELD, playerA, "Goblin Racketeer", 2); // convoke pay
99  
100         // use special action to pay (need disabled auto-payment and prepared mana pool)
101         disableManaAutoPayment(playerA);
102         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {R}", 2);
103         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Stoke the Flames", playerB);
104         setChoice(playerA, "Red"); // pay 1
105         setChoice(playerA, "Red"); // pay 2
106         setChoice(playerA, "Convoke");
107         addTarget(playerA, "Goblin Racketeer"); // pay 3 as convoke
108         setChoice(playerA, "Convoke");
109         addTarget(playerA, "Goblin Racketeer"); // pay 4 as convoke
110 
111         setStrictChooseMode(true);
112         setStopAt(1, PhaseStep.END_TURN);
113         execute();
114 
115         assertLife(playerB, 20 - 4);
116     }

179     @Test
180     public void test_Other_ConvokeTwoCreatures() {
181         // {1}{W}
182         // Convoke (Your creatures can help cast this spell. Each creature you tap while casting this spell pays for or one mana of that creature's color.)
183         // Target creature gains indestructible until end of turn. (Damage and effects that say "destroy" don't destroy it.)
184         addCard(Zone.HAND, playerA, "Ephemeral Shields");
185         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
186         addCard(Zone.BATTLEFIELD, playerA, "Oreskos Swiftclaw", 1);
187 
188         // AI automaticly use convoke to pay
189         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ephemeral Shields", "Silvercoat Lion");
190         addTarget(playerA, "Silvercoat Lion"); // pay 1 as convoke
191         addTarget(playerA, "Oreskos Swiftclaw"); // pay 2 as convoke
192 
193         setStrictChooseMode(true);
194         setStopAt(1, PhaseStep.END_TURN);
195         execute();
196 
197         assertLife(playerA, 20);
198         assertLife(playerB, 20);
199 
200         assertGraveyardCount(playerA, "Ephemeral Shields", 1);
201         assertPermanentCount(playerA, "Silvercoat Lion", 1);
202         assertPermanentCount(playerA, "Oreskos Swiftclaw", 1);
203     }

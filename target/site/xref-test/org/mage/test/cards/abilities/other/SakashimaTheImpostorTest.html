13      @Test
14      public void copySpellStutterTest() {
15          // Flash, Flying
16          // When Spellstutter Sprite enters the battlefield, counter target spell with converted mana cost X or less, 
17          // where X is the number of Faeries you control.        
18          addCard(Zone.BATTLEFIELD, playerA, "Spellstutter Sprite", 1);
19          addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
20          // You may have Sakashima the Impostor enter the battlefield as a copy of any creature on the battlefield, 
21          // except its name is Sakashima the Impostor, it's legendary in addition to its other types,
22          // and it has "{2}{U}{U}: Return Sakashima the Impostor to its owner's hand at the beginning of the next end step."        
23          addCard(Zone.HAND, playerB, "Sakashima the Impostor", 4);
24  
25          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Sakashima the Impostor");
26          setChoice(playerB, "Spellstutter Sprite");
27  
28          setStopAt(4, PhaseStep.END_TURN);
29          execute();
30  
31          assertPermanentCount(playerA, "Spellstutter Sprite", 1);
32  
33          assertPowerToughness(playerB, "Sakashima the Impostor", 1, 1);
34      }

36      /**
37       * I played Sakashima the Imposter copying an opponents Pawn of Ulamaog.
38       * Sakashima gained the following ability: "Whenever Pawn of Ulamog or
39       * another nontoken creature you control dies, you may create a 0/1
40       * colorless Eldrazi Spawn creature token. It has "Sacrifice this creature:
41       * Add {C}." Then Sakashima died due to combat damage and the ability did
42       * not trigger.
43       *
44       */
45      @Test
46      public void copyDiesTriggeredTest() {
47          // Whenever Pawn of Ulamog or another nontoken creature you control dies, you may create a 0/1 colorless 
48          // Eldrazi Spawn creature token. It has "Sacrifice this creature: Add {C}."        
49          addCard(Zone.BATTLEFIELD, playerA, "Pawn of Ulamog", 1); // Creature 2/2
50          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1); // Creature 2/2
51  
52          addCard(Zone.BATTLEFIELD, playerB, "Island", 4);
53          // You may have Sakashima the Impostor enter the battlefield as a copy of any creature on the battlefield, 
54          // except its name is Sakashima the Impostor, it's legendary in addition to its other types,
55          // and it has "{2}{U}{U}: Return Sakashima the Impostor to its owner's hand at the beginning of the next end step."        
56          addCard(Zone.HAND, playerB, "Sakashima the Impostor", 4);
57  
58          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Sakashima the Impostor");
59          setChoice(playerB, "Pawn of Ulamog");
60  
61          attack(4, playerB, "Sakashima the Impostor");
62          block(4, playerA, "Silvercoat Lion", "Sakashima the Impostor");
63                  
64          setStopAt(4, PhaseStep.POSTCOMBAT_MAIN);
65          execute();
66  
67          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
68          assertGraveyardCount(playerB, "Sakashima the Impostor", 1);        
69          
70          assertPermanentCount(playerA, "Eldrazi Spawn Token", 1);
71          assertPermanentCount(playerB, "Eldrazi Spawn Token", 1);
72      }

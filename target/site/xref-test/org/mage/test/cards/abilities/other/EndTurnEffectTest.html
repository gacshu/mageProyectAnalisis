14      /**
15       * Additional bug: Days Undoing and Sphinx's Tutelage are broken. You
16       * shouldn't get triggers off of Tutelage, since the turn ends, but it has
17       * you resolve them in your cleanup step.
18       * <p>
19       * http://tabakrules.tumblr.com/post/122350751009/days-undoing-has-been-officially-spoiled-on
20       */
21      @Test
22      public void testSpellsAffinity() {
23          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
24  
25          // Whenever you draw a card, target opponent puts the top two cards of their library into their graveyard. If they're both nonland cards that share a color, repeat this process.
26          // {5}{U}: Draw a card, then discard a card.
27          addCard(Zone.BATTLEFIELD, playerA, "Sphinx's Tutelage");
28  
29          // Each player shuffles their hand and graveyard into their library, then draws seven cards. If it's your turn, end the turn.
30          //  (Exile all spells and abilities on the stack, including this card.
31          //   Discard down to your maximum hand size. Damage wears off, and
32          //   "this turn" and "until end of turn" effects end.)
33          addCard(Zone.HAND, playerA, "Day's Undoing"); //Sorcery {2}{U}
34  
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Day's Undoing");
36  
37          setStopAt(2, PhaseStep.UPKEEP);
38          execute();
39  
40          assertExileCount("Day's Undoing", 1);
41  
42          assertHandCount(playerA, 7);
43          assertHandCount(playerB, 7);
44  
45          assertGraveyardCount(playerB, 0); // because the triggers of Sphinx's Tutelage cease to exist
46  
47      }

49      @Test
50      public void testSpellSplitCard() {
51          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
52  
53          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
54  
55          // End the turn.
56          //  (Exile all spells and abilities on the stack, including this card.
57          //   Discard down to your maximum hand size. Damage wears off, and
58          //   "this turn" and "until end of turn" effects end.)
59          addCard(Zone.HAND, playerA, "Time Stop"); //Instant {4}{U}{U}
60  
61          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
62          // Fire
63          // Fire deals 2 damage divided as you choose among one or two target creatures and/or players.
64          // Ice
65          // Tap target permanent. Draw a card.
66          addCard(Zone.HAND, playerB, "Fire // Ice"); // Instant {1}{R} // {1}{U}
67  
68          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Ice", "Silvercoat Lion");
69  
70          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Time Stop", TestPlayer.NO_TARGET, "Ice");
71  
72          setStopAt(2, PhaseStep.UPKEEP);
73          execute();
74  
75          assertHandCount(playerB, "Fire // Ice", 0);
76          assertExileCount(playerA, "Time Stop", 1);
77          assertExileCount(playerB, "Fire // Ice", 1);
78          assertTapped("Silvercoat Lion", false);
79          assertHandCount(playerA, 0);
80          assertHandCount(playerB, 0);
81  
82      }

84      /**
85       * Test to remove a Aftermath card from spell
86       */
87      @Test
88      public void testSpellAftermath() {
89          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
90          // Insult Sorcery {2}{R}
91          // Damage can't be prevented this turn. If a source you control would deal damage this turn, it deals double that damage instead.
92          // Injury Sorcery {2}{R}
93          // Aftermath (Cast this spell only from your graveyard. Then exile it.)
94          // Injury deals 2 damage to target creature and 2 damage to target player.
95          addCard(Zone.GRAVEYARD, playerA, "Insult // Injury");
96  
97          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
98          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 3);
99          // End the turn.(Exile all spells and abilities on the stack. Discard down to your maximum hand size. Damage wears off, and \"this turn\" and \"until end of turn\" effects end.)
100         // At the beginning of your next end step, you lose the game.
101         addCard(Zone.HAND, playerB, "Glorious End"); //Instant {2}{R}
102 
103         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Injury", "Silvercoat Lion");
104         addTarget(playerA, playerB);
105 
106         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Glorious End", TestPlayer.NO_TARGET, "Injury");
107 
108         setStopAt(1, PhaseStep.BEGIN_COMBAT);
109         execute();
110 
111         assertExileCount(playerB, "Glorious End", 1);
112         assertGraveyardCount(playerA, "Insult // Injury", 0);
113         assertExileCount(playerA, "Insult // Injury", 1);
114         assertGraveyardCount(playerB, "Glorious End", 0);
115         assertHandCount(playerA, 0);
116         // TODO Check
117         assertHandCount(playerB, 1); // No idea why playerB has a mountain into hand
118 
119     }

124     @Test
125     public void testSundialOfTheInfinite() {
126         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
127         addCard(Zone.HAND, playerA, "Mountain", 10);
128         // {1}, {T}: End the turn. Activate this ability only during your turn.
129         addCard(Zone.HAND, playerA, "Sundial of the Infinite", 1); // Artifact {2}
130 
131         addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
132         // Destroy target artifact or enchantment.
133         addCard(Zone.HAND, playerB, "Disenchant"); //Instant {1}{W}
134 
135         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sundial of the Infinite");
136         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
137         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Disenchant", "Sundial of the Infinite");
138 
139         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1},", TestPlayer.NO_TARGET, "Disenchant");
140 
141         setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
142         execute();
143 
144         assertExileCount(playerB, "Disenchant", 1);
145 
146         assertPermanentCount(playerA, "Sundial of the Infinite", 1);
147 
148         assertHandCount(playerA, 7); // Discard to maximum hand size
149         assertHandCount(playerB, 1); // 1 card drawn at start of 2nd turn
150     }

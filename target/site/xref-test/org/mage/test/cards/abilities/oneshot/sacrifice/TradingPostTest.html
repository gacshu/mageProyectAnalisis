4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   *
11   * @author LevelX2
12   */
13  public class TradingPostTest extends CardTestPlayerBase {
14  
15      /**
16       * Trading Post doesn't let me sacrifice a creature owned by my opponent,
17       * but controlled by me. I get an error message saying You cannot sacrifice
18       * this creature.
19       */
20      @Test
21      public void testSacrifice() {
22          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
23          // {1}, {T}, Discard a card: You gain 4 life.
24          // {1}, {T}, Pay 1 life: Create a 0/1 white Goat creature token onto the battlefield.
25          // {1}, {T}, Sacrifice a creature: Return target artifact card from your graveyard to your hand.
26          // {1}, {T}, Sacrifice an artifact: Draw a card.
27          addCard(Zone.BATTLEFIELD, playerA, "Trading Post", 1);
28          // Gain control of target creature until end of turn. Untap that creature. It gains haste until end of turn. (It can attack and this turn.)
29          addCard(Zone.HAND, playerA, "Act of Treason"); // Sorcery {2}{R}
30          addCard(Zone.GRAVEYARD, playerA, "Helm of Possession");
31  
32          addCard(Zone.BATTLEFIELD, playerB, "Savannah Lions");
33  
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Act of Treason", "Savannah Lions");
35  
36          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}, {T}, Sacrifice a creature", "Helm of Possession", "Act of Treason", StackClause.WHILE_NOT_ON_STACK);
37          setChoice(playerA, "Savannah Lions");
38  
39          setStopAt(1, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          assertGraveyardCount(playerA, "Act of Treason", 1);
43  
44          assertPermanentCount(playerB, "Savannah Lions", 0);
45          assertGraveyardCount(playerB, "Savannah Lions", 1);
46  
47          assertTapped("Trading Post", true);
48          assertHandCount(playerA, 1);
49  
50      }
51  }

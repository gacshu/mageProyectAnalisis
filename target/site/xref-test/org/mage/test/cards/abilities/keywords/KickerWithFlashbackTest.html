13      // https://github.com/magefree/mage/issues/5389
14      // Burst Lightning {R}
15      // Kicker {4} (You may pay an additional {4} as you cast this spell.)
16      // Burst Lightning deals 2 damage to any target. If this spell was kicked, it deals 4 damage to that permanent or player instead.
17      // Snapcaster Mage {1}{U}
18      // Flash
19      // When Snapcaster Mage enters the battlefield, target instant or sorcery card in your graveyard gains flashback until end of turn.
20      // The flashback cost is equal to its mana cost. (You may cast that card from your graveyard for its flashback cost. Then exile it.)
21      @Test
22      public void test_SimpleKicker() {
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
24          addCard(Zone.HAND, playerA, "Burst Lightning");
25  
26          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Burst Lightning", playerB);
27          setChoice(playerA, true); // use kicker
28  
29          checkLife("after", 1, PhaseStep.END_TURN, playerB, 20 - 4);
30  
31          setStopAt(1, PhaseStep.END_TURN);
32          setStrictChooseMode(true);
33          execute();
34      }

36      @Test
37      public void test_SimpleFlashback() {
38          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
39          addCard(Zone.HAND, playerA, "Snapcaster Mage");
40          //
41          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
42          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt", 1);
43  
44          // add flashback to bolt
45          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}");
46          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}");
47          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Snapcaster Mage");
48          addTarget(playerA, "Lightning Bolt");
49  
50          // cast bolt by flashback
51          activateAbility(1, PhaseStep.BEGIN_COMBAT, playerA, "Flashback");
52          addTarget(playerA, playerB);
53  
54          checkLife("after", 1, PhaseStep.END_TURN, playerB, 20 - 3);
55  
56          setStopAt(1, PhaseStep.END_TURN);
57          setStrictChooseMode(true);
58          execute();
59      }

66          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1 + 4);

72          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Snapcaster Mage");
73          addTarget(playerA, "Burst Lightning");

14  public class KusariGamaTest extends CardTestPlayerBase {
15  
16      // reported bug: trigger occurs but no damage is dealt
17      @Test
18      public void testTriggeredAbilityDealsDamage() {
19  
20          // Kusari-Gama - Artifact Equipment - Equip {3}
21          // Equipped creature has "2: This creature gets +1/+0 until end of turn."
22          // Whenever equipped creature deals damage to a blocking creature, Kusari-Gama deals that much damage to each other creature defending player controls.
23          addCard(Zone.BATTLEFIELD, playerA, "Kusari-Gama");
24          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
25          addCard(Zone.BATTLEFIELD, playerA, "Sylvan Advocate"); // 2/3 vigilance {1}{G}
26  
27          addCard(Zone.BATTLEFIELD, playerB, "Wall of Omens"); // 0/4 {1}{W}
28          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2); // 2/2 {1}{W}
29          addCard(Zone.BATTLEFIELD, playerB, "Hill Giant"); // 3/3 {3}{R}
30  
31          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip {3}", "Sylvan Advocate");
32          attack(1, playerA, "Sylvan Advocate");
33          block(1, playerB, "Wall of Omens", "Sylvan Advocate");
34          setStopAt(1, PhaseStep.END_COMBAT);
35          execute();
36  
37          assertPermanentCount(playerA, "Kusari-Gama", 1);
38          assertPermanentCount(playerB, "Wall of Omens", 1);
39          assertPermanentCount(playerB, "Hill Giant", 1);
40  
41          Permanent wallPerm = getPermanent("Wall of Omens", playerB);
42          Permanent giantPerm = getPermanent("Hill Giant", playerB);
43          Assert.assertEquals("Wall of Omens should have 2 damage dealt to it", 2, wallPerm.getDamage());
44          Assert.assertEquals("Hill Giant should have 2 damage dealt to it", 2, giantPerm.getDamage());
45  
46          assertGraveyardCount(playerB, "Silvercoat Lion", 2);
47      }
48  }

15  public class LeoninShikariTest extends CardTestPlayerBase {
16  
17      /**
18       * Test you can equip during combat
19       */
20      @Test
21      public void testEquipInstant() {
22          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
23          // You may activate equip abilities any time you could cast an instant.
24          addCard(Zone.BATTLEFIELD, playerA, "Leonin Shikari");        
25          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");        
26          // Equipped creature gets +1/+1.        
27          addCard(Zone.BATTLEFIELD, playerA, "Leonin Scimitar");
28  
29          
30          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);        
31          addCard(Zone.HAND, playerB, "Boomerang");
32  
33          activateAbility(1, PhaseStep.BEGIN_COMBAT, playerA, "Equip {1}", "Silvercoat Lion");
34          castSpell(1, PhaseStep.BEGIN_COMBAT, playerB, "Boomerang", "Leonin Scimitar", "Equip");
35          
36          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Leonin Scimitar");
37  
38          setStopAt(1, PhaseStep.END_TURN);
39          execute();
40  
41          assertLife(playerA, 20);
42          assertLife(playerB, 20);
43          
44          assertGraveyardCount(playerB, "Boomerang", 1);
45  
46          assertPermanentCount(playerA, "Leonin Scimitar", 1);
47          assertPowerToughness(playerA, "Silvercoat Lion", 2, 2);
48  
49          Permanent silvercoatLion = getPermanent("Silvercoat Lion", playerA.getId());
50          Assert.assertTrue("Silvercoat Lion may not have any attachments", silvercoatLion.getAttachments().isEmpty());
51          
52          Permanent leoninScimitar = getPermanent("Leonin Scimitar", playerA.getId());
53          Assert.assertTrue(leoninScimitar.getAttachedTo() == null);
54      }

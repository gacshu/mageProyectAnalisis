41      /**
42       * Another bug, was playing Slivers again. I had a Amoeboid Changeling, a
43       * Hibernation Sliver and a Prophet of Kruphix. In response to a boardwipe,
44       * I tapped my Changeling, giving my Prophet Changeling. However, it didn't
45       * gain any Sliver abilities despite having all creature types, including
46       * Sliver, so I couldn't save it with my Hibernation Sliver. I clicked the
47       * Prophet and nothing happened at all.
48       */
49      @Test
50      public void testGainingChangeling() {
51          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
52          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
53          // Untap all creatures and lands you control during each other player's untap step.
54          // You may cast creature cards as though they had flash.
55          addCard(Zone.HAND, playerA, "Prophet of Kruphix");// {3}{G}{U}
56          // Changeling
57          // {T}: Target creature gains all creature types until end of turn.
58          // {T}: Target creature loses all creature types until end of turn.
59          addCard(Zone.BATTLEFIELD, playerA, "Amoeboid Changeling");
60          // All Slivers have "Pay 2 life: Return this permanent to its owner's hand."
61          addCard(Zone.BATTLEFIELD, playerA, "Hibernation Sliver");
62  
63          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Prophet of Kruphix");
64          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T}: Target creature gains", "Prophet of Kruphix");
65  
66          setStopAt(1, PhaseStep.END_TURN);
67          execute();
68  
69          assertTapped("Amoeboid Changeling", true);
70  
71          Permanent prophet = getPermanent("Prophet of Kruphix", playerA);
72          boolean abilityFound = false;
73          for (Ability ability : prophet.getAbilities()) {
74              if (ability.getRule().startsWith("Pay 2 life")) {
75                  abilityFound = true;
76              }
77          }
78          Assert.assertTrue("Prophet of Kruphix has to have the 'Pay 2 life: Return this permanent to its owner's hand.' ability, but has not.", abilityFound);
79  
80      }

128     @Test
129     public void testLoseAllCreatureTypes() {
130         addCard(Zone.BATTLEFIELD, playerA, "Game-Trail Changeling");
131         addCard(Zone.BATTLEFIELD, playerA, "Goblin Chieftain");
132         addCard(Zone.HAND, playerA, "Nameless Inversion");
133         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
134 
135         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nameless Inversion", "Game-Trail Changeling");
136         setStopAt(1, PhaseStep.END_TURN);
137         execute();
138 
139         // Should have no creature types but still have the Changeling ability
140         assertPowerToughness(playerA, "Game-Trail Changeling", 7, 1);
141         assertNotSubtype("Game-Trail Changeling", SubType.SHAPESHIFTER);
142         assertAbility(playerA, "Game-Trail Changeling", HasteAbility.getInstance(), false);
143         assertAbility(playerA, "Game-Trail Changeling", new ChangelingAbility(), true);
144     }

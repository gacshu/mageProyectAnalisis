14  public class EightAndAHalfTailsTest extends CardTestPlayerBase {
15  
16      @Test
17      public void testProtectingPlaneswalker() {
18          setStrictChooseMode(true);
19          
20          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
21  
22          // Activated abilities of artifacts your opponents control can't be activated.
23          // +1: Until your next turn, up to one target noncreature artifact becomes an artifact creature with power and toughness equal to its converted mana cost.
24          // -2: You may choose an artifact card you own from outside the game or in exile, reveal that card, and put it into your hand.
25          addCard(Zone.BATTLEFIELD, playerA, "Karn, the Great Creator"); // Planeswalker (5)
26          
27          // {1}{W}: Target permanent you control gains protection from white until end of turn.
28          // {1}: Target spell or permanent becomes white until end of turn.
29          addCard(Zone.BATTLEFIELD, playerA, "Eight-and-a-Half-Tails"); // Creature
30          
31          // Flying, double strike
32          // Whenever a creature you control deals combat damage to a player, you and that player each gain that much life.
33          // At the beginning of your end step, if you have at least 15 life more than your starting life total, each player Angel of Destiny attacked this turn loses the game.
34          addCard(Zone.BATTLEFIELD, playerB, "Angel of Destiny"); // Creature
35  
36          attack(2, playerB, "Angel of Destiny", "Karn, the Great Creator");
37          activateAbility(2, PhaseStep.DECLARE_ATTACKERS, playerA, "{1}{W}: Target permanent you control gains protection from white until end of turn.");
38          addTarget(playerA, "Karn, the Great Creator");
39          
40          setStopAt(2, PhaseStep.END_COMBAT);
41          execute();
42  
43          assertPermanentCount(playerA, "Karn, the Great Creator", 1);
44          assertCounterCount("Karn, the Great Creator", CounterType.LOYALTY, 5);
45          
46      }

5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    *
10   * @author noxx
11   *
12   *   Whenever another nontoken creature dies, you may draw a card.
13   */
14  public class HarvesterOfSoulsTest extends CardTestPlayerBase {
15  
16      /**
17       * Tests creature on any side would trigger effect
18       * Also tests that tokens don't cause trigger to happen
19       */
20      @Test
21      public void testDisabledEffectOnChangeZone() {
22          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
24          addCard(Zone.HAND, playerA, "Day of Judgment", 1);        
25          addCard(Zone.HAND, playerA, "Thatcher Revolt", 1);
26  
27          // Creature - Demon  5/5   {4}{B}{B}
28          // Deathtouch
29          // Whenever another nontoken creature dies, you may draw a card.
30          addCard(Zone.BATTLEFIELD, playerA, "Harvester of Souls", 1);
31          
32          // Creature - Wurm  6/4   {4}{G}{G}
33          addCard(Zone.BATTLEFIELD, playerA, "Craw Wurm", 1);
34          // Whenever Arrogant Bloodlord blocks or becomes blocked by a creature with power 1 or less, destroy Arrogant Bloodlord at end of combat.
35          addCard(Zone.BATTLEFIELD, playerB, "Arrogant Bloodlord", 1);
36          
37          // Put three 1/1 red Human creature tokens with haste onto the battlefield. Sacrifice those tokens at the beginning of the next end step.
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Thatcher Revolt", true);
39          // Destroy all creatures.
40          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Day of Judgment");
41  
42          setStopAt(1, PhaseStep.BEGIN_COMBAT);
43          execute();
44  
45          assertLife(playerA, 20);
46          assertLife(playerB, 20);
47  
48          assertHandCount(playerA, 2); // draw a card for Harvester and Craw Wurm
49          assertHandCount(playerB, 0);
50      }
51  }

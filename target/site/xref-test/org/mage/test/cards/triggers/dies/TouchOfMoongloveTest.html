4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;
8   
9   /**
10   *
11   * @author LevelX2
12   */
13  public class TouchOfMoongloveTest extends CardTestPlayerBase {
14  
15      /**
16       * I blocked my opponent's Pharika's Disciple with a Cleric of the Forward
17       * Order and Guardian Automaton. They cast Touch of Moonglove on their
18       * Pharika's Disciple and both of my creatures were killed, but I only lost
19       * 2 life instead of 4.(and gained 3 from Guardian Automaton dying).
20       *
21       */
22      @Test
23      public void testDiesAndControllerDamage() {
24          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
25          // When Cleric of the Forward Order enters the battlefield, you gain 2 life for each creature you control named Cleric of the Forward Order.
26          addCard(Zone.HAND, playerA, "Cleric of the Forward Order", 1);
27          // When Guardian Automaton dies, you gain 3 life.
28          addCard(Zone.BATTLEFIELD, playerA, "Guardian Automaton", 1); // 3/3
29  
30          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 1);
31          // Target creature you control gets +1/+0 and gains deathtouch until end of turn.
32          // Whenever a creature dealt damage by that creature this turn dies, its controller loses 2 life.
33          addCard(Zone.HAND, playerB, "Touch of Moonglove"); // {B}
34          addCard(Zone.BATTLEFIELD, playerB, "Pharika's Disciple", 1);
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cleric of the Forward Order");
37          attack(2, playerB, "Pharika's Disciple");
38          block(2, playerA, "Cleric of the Forward Order", "Pharika's Disciple");
39          block(2, playerA, "Guardian Automaton", "Pharika's Disciple");
40  
41          castSpell(2, PhaseStep.DECLARE_BLOCKERS, playerB, "Touch of Moonglove", "Pharika's Disciple");
42  
43          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
44          execute();
45  
46          assertGraveyardCount(playerB, "Touch of Moonglove", 1);
47          assertGraveyardCount(playerA, "Cleric of the Forward Order", 1);
48          assertGraveyardCount(playerB, "Pharika's Disciple", 1);
49  
50          assertGraveyardCount(playerA, "Guardian Automaton", 1);
51  
52          assertLife(playerA, 21); // +2 by Cleric + 2x -2 by Touch +3 by Guardian Automation
53          assertLife(playerB, 20);
54  
55      }

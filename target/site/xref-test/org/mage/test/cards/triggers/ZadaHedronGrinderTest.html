16       * Playing Zada edh, strive cards such as Rouse the Mob do not copy when

19      @Test
20      public void testWithStriveSpell() {
21          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
22  
23          // Whenever you cast an instant or sorcery spell that targets only Zada, Hedron Grinder, copy that spell for each other creature you control that the spell could target. Each copy targets a different one of those creatures.
24          addCard(Zone.BATTLEFIELD, playerA, "Zada, Hedron Grinder", 1); // 3/3
25          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
26  
27          // Strive â€” Rouse the Mob costs {2}{R} more to cast for each target beyond the first.
28          // Any number of target creatures each get +2/+0 and gain trample until end of turn.
29          addCard(Zone.HAND, playerA, "Rouse the Mob", 1); // Instant - {R}
30  
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Rouse the Mob", "Zada, Hedron Grinder");
32  
33          setStopAt(1, PhaseStep.BEGIN_COMBAT);
34          execute();
35  
36          assertLife(playerA, 20);
37          assertLife(playerB, 20);
38  
39          assertGraveyardCount(playerA, "Rouse the Mob", 1);
40  
41          assertPowerToughness(playerA, "Zada, Hedron Grinder", 5, 3);
42          assertAbility(playerA, "Zada, Hedron Grinder", TrampleAbility.getInstance(), true);
43          assertPowerToughness(playerA, "Silvercoat Lion", 4, 2);
44          assertAbility(playerA, "Silvercoat Lion", TrampleAbility.getInstance(), true);
45      }

47      @Test
48      public void testTargetsByTestPlayer() {
49          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
50  
51          // Whenever you cast an instant or sorcery spell that targets only Zada, Hedron Grinder, copy that spell for
52          // each other creature you control that the spell could target. Each copy targets a different one of those creatures.
53          addCard(Zone.BATTLEFIELD, playerA, "Zada, Hedron Grinder", 1); // 3/3
54          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
55          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
56          //
57          // Put a +1/+1 counter on target creature. That creature gains reach until end of turn.
58          addCard(Zone.HAND, playerA, "Arbor Armament", 1); // {G}
59          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
60  
61          // cast
62          setChoice(playerA, TestPlayer.CHOICE_SKIP); // skip stack order
63          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Arbor Armament", "Zada, Hedron Grinder");
64  
65          setStrictChooseMode(true);
66          setStopAt(1, PhaseStep.BEGIN_COMBAT);
67          execute();
68  
69          assertLife(playerA, 20);
70          assertLife(playerB, 20);
71  
72          assertGraveyardCount(playerA, "Arbor Armament", 1);
73          assertPowerToughness(playerA, "Zada, Hedron Grinder", 3 + 1, 3 + 1);
74          assertPowerToughness(playerA, "Silvercoat Lion", 2 + 1, 2 + 1);
75          assertPowerToughness(playerA, "Balduvian Bears", 2 + 1, 2 + 1);
76      }

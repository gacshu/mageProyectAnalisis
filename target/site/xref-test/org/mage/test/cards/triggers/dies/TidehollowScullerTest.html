13      /**
14       * Test if the same Tidehollow Sculler is cast multiple times, the correct
15       * corresponding exiled cards are returned
16       */
17      @Test
18      public void test_CastOneCardFromHandWillBeExiled() {
19          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
20          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
21          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
22          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
23          // Tidehollow Sculler {W}{B}
24          // When Tidehollow Sculler enters the battlefield, target opponent reveals their hand and you choose a nonland card from it. Exile that card.
25          // When Tidehollow Sculler leaves the battlefield, return the exiled card to its owner's hand.
26          addCard(Zone.HAND, playerA, "Tidehollow Sculler", 1); // 2/2
27          addCard(Zone.HAND, playerA, "Lightning Bolt", 1); // {R}
28          //
29          addCard(Zone.HAND, playerB, "Bloodflow Connoisseur", 1);
30  
31          // cast and exile from hand
32          checkHandCardCount("B hand must have blood", 1, PhaseStep.UPKEEP, playerB, "Bloodflow Connoisseur", 1);
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tidehollow Sculler");
34          // addTarget(playerA, playerB); // choose opponent (Autochosen, only target)
35          setChoice(playerA, "Bloodflow Connoisseur"); // card to exile
36          checkHandCardCount("B hand must lost blood", 1, PhaseStep.BEGIN_COMBAT, playerB, "Bloodflow Connoisseur", 0);
37  
38          // destroy and return card to hand
39          checkPermanentCount("A must have tide", 1, PhaseStep.END_COMBAT, playerA, "Tidehollow Sculler", 1);
40          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Tidehollow Sculler");
41          checkPermanentCount("A must lost tide", 1, PhaseStep.END_TURN, playerA, "Tidehollow Sculler", 0);
42          checkHandCardCount("B must return blood", 1, PhaseStep.END_TURN, playerB, "Bloodflow Connoisseur", 1);
43  
44          setStopAt(1, PhaseStep.END_TURN);
45          execute();
46  
47          assertHandCount(playerB, "Bloodflow Connoisseur", 1);
48          assertPermanentCount(playerA, "Tidehollow Sculler", 0);
49      }

51      @Test
52      public void test_CastTwoCardFromHandWillBeExiled() {
53          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
54          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
55          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
56          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
57          // Tidehollow Sculler {W}{B}
58          // When Tidehollow Sculler enters the battlefield, target opponent reveals their hand and you choose a nonland card from it. Exile that card.
59          // When Tidehollow Sculler leaves the battlefield, return the exiled card to its owner's hand.
60          addCard(Zone.HAND, playerA, "Tidehollow Sculler@tide", 2); // 2/2
61          addCard(Zone.HAND, playerA, "Lightning Bolt", 2); // {R}
62          //
63          addCard(Zone.HAND, playerB, "Bloodflow Connoisseur", 1);
64          addCard(Zone.HAND, playerB, "Silvercoat Lion", 1);
65  
66          // turn 1 - A
67          // cast 1 and exile from hand
68          checkHandCardCount("B hand must have blood", 1, PhaseStep.UPKEEP, playerB, "Bloodflow Connoisseur", 1);
69          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tidehollow Sculler");
70          // addTarget(playerA, playerB); // choose opponent (Autochosen, only target)
71          setChoice(playerA, "Bloodflow Connoisseur"); // card to exile
72          checkHandCardCount("B hand must lost blood", 1, PhaseStep.BEGIN_COMBAT, playerB, "Bloodflow Connoisseur", 0);
73          // cast 2 and exile from hand
74          checkHandCardCount("B hand must have lion", 1, PhaseStep.END_COMBAT, playerB, "Silvercoat Lion", 1);
75          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Tidehollow Sculler");
76          // addTarget(playerA, playerB); // choose opponent (Autochosen, only target)
77          setChoice(playerA, "Silvercoat Lion"); // card to exile
78          checkHandCardCount("B hand must lost lion", 1, PhaseStep.END_TURN, playerB, "Silvercoat Lion", 0);
79  
80          // turn 2 - B
81          // destroy 1 and return card to hand
82          checkPermanentCount("A must have 2 tide", 2, PhaseStep.UPKEEP, playerA, "Tidehollow Sculler", 2);
83          checkHandCardCount("B hand must have 0 blood", 2, PhaseStep.UPKEEP, playerB, "Bloodflow Connoisseur", 0);
84          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "@tide.1");
85  //        showHand("B hand", 2, PhaseStep.BEGIN_COMBAT, playerB);
86          checkPermanentCount("A must have 1 tide", 2, PhaseStep.BEGIN_COMBAT, playerA, "Tidehollow Sculler", 1);
87          checkHandCardCount("B hand must have 1 blood", 2, PhaseStep.BEGIN_COMBAT, playerB, "Bloodflow Connoisseur", 1);
88          // destroy 2 and return card to hand
89          checkPermanentCount("A must have 1 tide", 2, PhaseStep.END_COMBAT, playerA, "Tidehollow Sculler", 1);
90          checkHandCardCount("B hand must have 0 lion", 2, PhaseStep.END_COMBAT, playerB, "Silvercoat Lion", 0);
91          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "@tide.2");
92          checkPermanentCount("A must have 0 tide", 2, PhaseStep.END_TURN, playerA, "Tidehollow Sculler", 0);
93          checkHandCardCount("B hand must have 1 lion", 2, PhaseStep.END_TURN, playerB, "Silvercoat Lion", 1);
94  
95          setStopAt(2, PhaseStep.END_TURN);
96          execute();
97      }

4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class ZurTheEnchanterTest extends CardTestPlayerBase {
14  
15      /**
16       * Zur the Enchanter's ability + shroud
17       *
18       * You can reproduce this by attacking with a Zur the Enchanter that has
19       * shroud (Lightning Greaves, Diplomatic Immunity, Greater Auramancy + an
20       * aura on him, etc.) and when his ability triggers searching for an aura
21       * (in this case, Empyrial Armor) and trying to attach it to Zur himself.
22       * The game won't allow you to attach it to him, even though it should,
23       * since the enchantment is put onto the battlefield and not cast, hence, no
24       * targeting is done. The rulings page for Zur itself say it so on Gatherer:
25       *
26       * Shroud shouldn't stop Empyrial Armor from attaching to Zur, only
27       * something like protection from white, for example, would do that.
28       */
29      @Test
30      public void testAuraToBattlefieldDoesNotTarget() {
31          // Flying
32          // Whenever Zur the Enchanter attacks, you may search your library for an enchantment card
33          // with converted mana cost 3 or less and put it onto the battlefield. If you do, shuffle your library.
34          addCard(Zone.BATTLEFIELD, playerB, "Zur the Enchanter"); // 1/4
35  
36          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
37          // Enchant creature
38          // Shroud (This permanent can't be the target of spells or abilities.)
39          // Enchanted creature has shroud.
40          addCard(Zone.HAND, playerB, "Diplomatic Immunity"); // {1}{U}
41          // Enchant creature
42          // Enchanted creature gets +1/+1 for each card in your hand.
43          addCard(Zone.LIBRARY, playerB, "Empyrial Armor", 2);
44  
45          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Diplomatic Immunity", "Zur the Enchanter");
46  
47          attack(2, playerB, "Zur the Enchanter");
48          // setChoice(playerB, "Empyrial Armor"); // No need for this choice since the AI will autosearch and autopick this
49          setChoice(playerB, "Zur the Enchanter");
50  
51          setStopAt(2, PhaseStep.END_TURN);
52          execute();
53  
54          assertPermanentCount(playerB, "Diplomatic Immunity", 1);
55          assertPermanentCount(playerB, "Empyrial Armor", 1);
56          assertPowerToughness(playerB, "Zur the Enchanter", 2, 5);
57      }

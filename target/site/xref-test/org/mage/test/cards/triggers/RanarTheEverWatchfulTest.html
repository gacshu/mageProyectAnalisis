19      @Test
20      public void testRanarWithCurseOfTheSwine() {
21          setStrictChooseMode(true);
22          // Curse of the Swine targets 4 creatures; 2 non-tokens and 2 tokens with Ranar, the Ever-Watchful on the battlefield
23          // Result: 2 boar tokens and 1 spirit token for the playerA.  2 boar tokens for playerB
24          
25          addCard(Zone.BATTLEFIELD, playerA, "Ranar the Ever-Watchful", 1);
26          
27          /*
28          Whenever one or more cards are put into exile from your hand or a spell or ability you control exiles one or 
29          more permanents from the battlefield, create a 1/1 white Spirit creature token with flying.
30          */
31          
32          addCard(Zone.BATTLEFIELD, playerA, "Memnite", 1); // Creature 1/1
33          addCard(Zone.BATTLEFIELD, playerA, "Serra Angel", 1); // Creature 4/4 Flyer
34          addCard(Zone.BATTLEFIELD, playerA, "Bog Imp", 1); // Creature 2/1 Flyer
35          addCard(Zone.BATTLEFIELD, playerB, "Sengir Vampire", 1); // Creature 4/4 Flyer
36          addCard(Zone.BATTLEFIELD, playerA, "Island", 10); // handle mana for Curse of the Swine
37          addCard(Zone.BATTLEFIELD, playerB, "Plains", 10); // handle mana for Call the Calvary
38          addCard(Zone.HAND, playerB, "Call the Cavalry", 1);  // generate 2 tokens for playerB
39          addCard(Zone.HAND, playerA, "Curse of the Swine", 1); // exile target cards and create a boar token for each one exiled
40  
41          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Call the Cavalry");  // create 2 Knight tokens
42          
43          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Curse of the Swine");
44          setChoice(playerA, "X=6");
45          addTarget(playerA, "Memnite");
46          addTarget(playerA, "Serra Angel");
47          addTarget(playerA, "Bog Imp");
48          addTarget(playerA, "Sengir Vampire");
49          addTarget(playerA, "Knight Token");
50          addTarget(playerA, "Knight Token");
51  
52          setStopAt(4, PhaseStep.POSTCOMBAT_MAIN);
53          execute();
54  
55          assertPermanentCount(playerA, "Boar Token", 3); // created token by Curse
56          assertPermanentCount(playerA, "Spirit Token", 1); // created token by Ranar
57          assertPermanentCount(playerB, "Boar Token", 3); // created token by Curse
58          assertPermanentCount(playerB, "Spirit Token", 0);  // no Spirit token for playerB
59      }
60      
61      @Test
62      public void testRanarExiledCardsFromHand() {
63          setStrictChooseMode(true);
64          // A card is exiled from playerA's hand
65          // Result: Due to the card being exiled from playerA's hand, a Spirit token is created for playerA
66          
67          addCard(Zone.BATTLEFIELD, playerA, "Ranar the Ever-Watchful", 1);
68          
69          /*
70          Whenever one or more cards are put into exile from your hand or a spell or ability you control exiles one or 
71          more permanents from the battlefield, create a 1/1 white Spirit creature token with flying.
72          */
73          
74          addCard(Zone.HAND, playerA, "Psychic Theft", 1); // exile instant/sorcery card from target player's hand
75          addCard(Zone.BATTLEFIELD, playerA, "Island", 2); // handle mana for Psychic Theft
76          addCard(Zone.HAND, playerA, "Lightning Bolt", 1); // instant spell card
77          
78          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Psychic Theft", playerA);
79          setChoice(playerA, "Lightning Bolt");
80          
81          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
82          execute();
83  
84          assertPermanentCount(playerA, "Spirit Token", 1); // created token by Ranar
85          assertPermanentCount(playerB, "Spirit Token", 0); // no Spirit tokens for the other player
86      }

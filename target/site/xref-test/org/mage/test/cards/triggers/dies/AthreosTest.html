13      /*
14       * Reported bug: Athreos interaction with Cartel Aristocrat not working.
15       * Athreos ability is not being triggered from Cartel's activated ability to sacrifice a creature.
16       */
17      @Test
18      public void cartelAristrocraftInteractionOpponentDoesNotPayLife() {
19  
20          String athreos = "Athreos, God of Passage";
21          String cAristrocrat = "Cartel Aristocrat";
22          String gBears = "Grizzly Bears";
23  
24          /*
25          Athreos, God of Passage {1}{W}{B}
26          Legendary Enchantment Creature - God 5/4
27          Indestructible
28          As long as your devotion to white and black is less than seven, Athreos isn't a creature.
29          Whenever another creature you own dies, return it to your hand unless target opponent pays 3 life.
30           */
31          addCard(Zone.BATTLEFIELD, playerA, athreos);
32  
33          /*
34          Cartel Aristocrat {W}{B}
35          Creature - Human Advisor 2/2
36          Sacrifice another creature: Cartel Aristocrat gains protection from the color of your choice until end of turn.
37          */
38          addCard(Zone.BATTLEFIELD, playerA, cAristrocrat);
39  
40          addCard(Zone.BATTLEFIELD, playerA, gBears); // {1}{G} 2/2
41  
42          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice");
43          setChoice(playerA, gBears); // sac bears to Cartel activated ability
44          setChoice(playerA, "Red"); // color chosen for Cartel
45  
46          setChoice(playerB, false); // opponent does not pay 3 life
47  
48          setStopAt(1, PhaseStep.BEGIN_COMBAT);
49          execute();
50  
51          assertHandCount(playerA, gBears, 1);
52          assertGraveyardCount(playerA, gBears, 0);
53          assertLife(playerB, 20);
54      }

15      /**
16       * There's currently a bug with Possibility Storm and Zoetic Cavern. The way
17       * it's supposed to work is the P. Storm trigger exiles Zoetic Cavern and
18       * then uses last known information about the spell to determine the type of
19       * card the trigger is looking for(creature in this instance). Instead it's
20       * basing the type solely off what's printed on the card. What happened to
21       * me earlier was the trigger skipped right over an Emrakul and then
22       * revealed a Flooded Strand. I was prompted whether or not I wanted to
23       * "cast" Flooded Strand without paying it's cost. Eventually I clicked yes
24       * and it produced a Game Error that resulted in rollback. I recreated the
25       * error against an AI opponent and copied the code. Can't actually post it
26       * because the filter on this site claims it makes my post look too
27       * "spammy". Here's a screenshot of it instead(in spoiler tag).
28       */
29      @Test
30      public void TestWithZoeticCavern() {
31          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
32          // Whenever a player casts a spell from their hand, that player exiles it, then exiles cards from
33          // the top of their library until they exile a card that shares a card type with it. That
34          // player may cast that card without paying its mana cost. Then they put all cards exiled with
35          // Possibility Storm on the bottom of their library in a random order.
36          addCard(Zone.BATTLEFIELD, playerA, "Possibility Storm", 1);
37  
38          // {T}: Add {C}.
39          // Morph {2}
40          addCard(Zone.HAND, playerA, "Zoetic Cavern");
41  
42          addCard(Zone.LIBRARY, playerA, "Silvercoat Lion");
43          skipInitShuffling();
44  
45          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Zoetic Cavern");
46          setChoice(playerA, true);
47  
48          setStopAt(1, PhaseStep.BEGIN_COMBAT);
49          execute();
50  
51          assertPermanentCount(playerA, "Zoetic Cavern", 0);
52  
53          assertLibraryCount(playerA, "Zoetic Cavern", 1);
54          assertPermanentCount(playerA, "Silvercoat Lion", 1);
55  
56      }

58      /*
59       * Having another Possibility Storm issue(shocking, I know). This time it
60       * occurred when trying to finish off my opponent that was at 3 life.
61       * I cast an Izzet Charm choosing draw 2 discard 2 for mode,
62       * responded to the Possibility Storm trigger with Remand.
63       * Remand's trigger revealed a Pact of Negation I chose not to cast, then the trigger for Izzet Charm
64       * resolved apparently revealing a Cryptic Command. It automatically moved
65       * the spell from exile to my library, apparently because it believed it did
66       * not have a target(no spells remaining on the stack). I've seen this
67       * happen with Mana Leaks that get revealed with no targets, but Cryptic
68       * being modal means it should always be able to be cast. It's worth
69       * mentioning that I've revealed Cryptics off triggers with other spells on
70       * the stack and properly been asked if I wish to cast them. Thanks!
71       */
72      @Test
73      public void TestWithCrypticCommand() {
74          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
75          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
76          // Whenever a player casts a spell from their hand, that player exiles it, then exiles cards from
77          // the top of their library until they exile a card that shares a card type with it. That
78          // player may cast that card without paying its mana cost. Then they put all cards exiled with
79          // Possibility Storm on the bottom of their library in a random order.
80          addCard(Zone.BATTLEFIELD, playerA, "Possibility Storm", 1);
81  
82          // Choose one — Counter target noncreature spell unless its controller pays {2};
83          // or Izzet Charm deals 2 damage to target creature;
84          //  or draw two cards, then discard two cards.
85          addCard(Zone.HAND, playerA, "Izzet Charm"); // {U}{R}
86          // Counter target spell. If that spell is countered this way, put it into its owner's hand instead of into that player's graveyard.
87          // Draw a card.
88          addCard(Zone.HAND, playerA, "Remand");
89  
90          // Choose two -
91          // Counter target spell;
92          // or return target permanent to its owner's hand;
93          // or tap all creatures your opponents control;
94          // or draw a card.
95          addCard(Zone.LIBRARY, playerA, "Cryptic Command");
96          addCard(Zone.LIBRARY, playerA, "Pact of Negation");
97          skipInitShuffling();
98  
99          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
100 
101         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Izzet Charm");
102         setModeChoice(playerA, "3");
103 
104         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Remand", "Izzet Charm", "Whenever a player casts");
105         setChoice(playerA, false); // Don't play  Pact of Negotiation
106 
107         setChoice(playerA, true); // Play Cryptic Command
108         setModeChoice(playerA, "3");
109         setModeChoice(playerA, "4");
110 
111         setStopAt(1, PhaseStep.BEGIN_COMBAT);
112         execute();
113 
114         assertExileCount(playerA, 0);
115         assertGraveyardCount(playerA, "Cryptic Command", 1);
116 
117         assertTapped("Silvercoat Lion", true);
118         assertHandCount(playerA, 1); // from Cryptic Command Draw
119 
120     }

16  public class BecomesBlockedTest extends CardTestPlayerBase {
17      
18      /**
19       * Reported bug: 
20       * There's something wrong with how Rabid Elephant is getting his +2/+2 bonus, 
21       * it doesn't last until end of turn, but seems to be removed right after the blockers step.
22       */
23      @Test
24      public void testRabidElephant() {
25          
26          // {4}{G} 
27          // Whenever Rabid Elephant becomes blocked, it gets +2/+2 until end of turn for each creature blocking it.
28          addCard(Zone.BATTLEFIELD, playerA, "Rabid Elephant", 1); // 3/4
29          
30          addCard(Zone.BATTLEFIELD, playerB, "Savannah Lions", 1); // 2/1
31          addCard(Zone.BATTLEFIELD, playerB, "Hill Giant", 1); // 3/3
32          
33          attack(1, playerA, "Rabid Elephant");
34          block(1, playerB, "Savannah Lions", "Rabid Elephant");
35          block(1, playerB, "Hill Giant", "Rabid Elephant");                
36  
37          // test passes if PhaseStep ends at DECLARE_BLOCKERS
38          //setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
39          setStopAt(1, PhaseStep.COMBAT_DAMAGE);
40          execute();
41          
42          // blocked by 2 creatures, so gets +2/+2 twice, making it 7/8
43          assertPowerToughness(playerA, "Rabid Elephant", 7, 8);
44      }
45  }

18          // When Nazahn, Revered Bladesmith enters the battlefield, search your library for an Equipment card and reveal it.
19          // If you reveal a card named Hammer of Nazahn this way, put it onto the battlefield. Otherwise, put that card into your hand. Then shuffle your library.
20          // Whenever an equipped creature you control attacks, you may tap target creature defending player controls.
21          addCard(Zone.HAND, playerA, "Nazahn, Revered Bladesmith"); // Creature 5/4  {4}{G}{W}
22          // Whenever Hammer of Nazahn or another Equipment enters the battlefiend under your control, you may attach that Equipment to target creature you control.
23          // Equipped creature gets +2/+0 and has indestructible.
24          // Equip {4}
25          addCard(Zone.LIBRARY, playerA, "Hammer of Nazahn");
26          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
27          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
28  
29          // Equipped creature gets +2/+2 and has infect.
30          // Whenever Grafted Exoskeleton becomes unattached from a permanent, sacrifice that permanent.
31          // Equip {2}
32          addCard(Zone.BATTLEFIELD, playerA, "Grafted Exoskeleton", 1);
33  
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nazahn, Revered Bladesmith");
35          // setChoice(playerA, "Hammer of Nazahn"); // Auto-chosen since it's the only equipment in the library

37          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip {2}");

39          setStopAt(3, PhaseStep.BEGIN_COMBAT);
40          execute();

43          assertLife(playerB, 20);

53      /**
54       * I cast Beast Within on a Grafted Exoskeleton (equipped on a Nazahn with
55       * also a Bloodforged Battle-Axe and Hammer of Nazahn), it got destroyed,
56       * but Nazahn didn't get sacrificed. Perhaps of note is the fact that it got
57       * equipped via Hammer of Nazahn's ability. Also I remember this interaction
58       * working correctly in the past, so a recent-ish update must've broken it.
59       */
60      @Test
61      public void testGraftedExoskeletonAndBeastWithinEvent() {
62          // When Nazahn, Revered Bladesmith enters the battlefield, search your library for an Equipment card and reveal it. If you reveal a card named Hammer of Nazahn this way, put it onto the battlefield. Otherwise, put that card into your hand. Then shuffle your library.
63          // Whenever an equipped creature you control attacks, you may tap target creature defending player controls.
64          addCard(Zone.HAND, playerA, "Nazahn, Revered Bladesmith"); // Creature 5/4  {4}{G}{W}
65          // Whenever Hammer of Nazahn or another Equipment enters the battlefiend under your control, you may attach that Equipment to target creature you control.
66          // Equipped creature gets +2/+0 and has indestructible.
67          // Equip {4}
68          addCard(Zone.LIBRARY, playerA, "Hammer of Nazahn");
69          addCard(Zone.BATTLEFIELD, playerA, "Forest", 6);
70          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
71  
72          // Destroy target permanent. Its controller creates a 3/3 green Beast creature token.
73          addCard(Zone.HAND, playerA, "Beast Within"); // Instant {2}{G}
74          // Equipped creature gets +2/+2 and has infect.
75          // Whenever Grafted Exoskeleton becomes unattached from a permanent, sacrifice that permanent.
76          // Equip {2}
77          addCard(Zone.HAND, playerA, "Grafted Exoskeleton", 1); // Artifact Equipment - {4}
78  
79          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Nazahn, Revered Bladesmith");
80          addTarget(playerA, "Hammer of Nazahn");
81          setChoice(playerA, true); // Put the hammer on the battlefield
82          setChoice(playerA, true); // Attach the hammer to a creature
83          addTarget(playerA, "Nazahn, Revered Bladesmith");
84  
85          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Grafted Exoskeleton");
86          setChoice(playerA, true); // Attach the Grafted Exoskeleton to a creature
87          addTarget(playerA, "Nazahn, Revered Bladesmith");
88          
89          castSpell(3, PhaseStep.BEGIN_COMBAT, playerA, "Beast Within", "Grafted Exoskeleton");
90  
91          setStopAt(3, PhaseStep.END_COMBAT);
92          setStrictChooseMode(true);
93          execute();
94  
95          assertLife(playerA, 20);
96          assertLife(playerB, 20);
97  
98          assertPermanentCount(playerA, "Hammer of Nazahn", 1);
99          assertGraveyardCount(playerA, "Beast Within", 1);
100         assertPowerToughness(playerA, "Beast Token", 3, 3);
101         assertGraveyardCount(playerA, "Grafted Exoskeleton", 1);
102         assertGraveyardCount(playerA, "Nazahn, Revered Bladesmith", 1);
103 
104     }

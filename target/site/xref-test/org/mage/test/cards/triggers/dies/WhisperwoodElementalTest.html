9   /**
10   * Whisperwood Elemental - Elemental {3}{G}{G} At the beginning of your end
11   * step, manifest the top card of your library. Sacrifice Whisperwood Elemental:
12   * Until end of turn, face-up, nontoken creatures you control gain "When this
13   * creature dies, manifest the top card of your library."
14   *
15   * @author LevelX2
16   */
17  public class WhisperwoodElementalTest extends CardTestPlayerBase {
18  
19      /**
20       * Tests that the dies triggered ability of silvercoat lion (gained by
21       * sacrificed Whisperwood Elemental) triggers as he dies from Lightning Bolt
22       */
23      @Test
24      public void testDiesTriggeredAbility() {
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
26          addCard(Zone.BATTLEFIELD, playerA, "Whisperwood Elemental", 1);
27          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
28          addCard(Zone.HAND, playerA, "Lightning Bolt");
29  
30          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice {this}: Until end of turn, face-up, nontoken creatures you control gain \"When this creature dies, manifest the top card of your library.");
31          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Silvercoat Lion");
32  
33          setStopAt(1, PhaseStep.END_TURN);
34          execute();
35  
36          assertLife(playerA, 20);
37          assertLife(playerB, 20);
38  
39          assertGraveyardCount(playerA, "Whisperwood Elemental", 1);
40          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
41  
42          // Manifested creature from dying Silvercoat Lion
43          assertPermanentCount(playerA, EmptyNames.FACE_DOWN_CREATURE.toString(), 1);
44      }

4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class MelekIzzetParagonTest extends CardTestPlayerBase {
14  
15      /**
16       * Wenn Melek, Izzet Paragon liegt und man einen Red/Blue Sun's Zenith von
17       * der Bib spielt, wird er nicht kopiert, auch wenn der Effekt auf dem Stack
18       * sichtbar ist.
19       *
20       * Meine Theorie ist, dass die Kopie beim in die Bib mischen den Originalen
21       * nimmt und er daher nicht mehr dem Stack ist um selbst verrechnet zu
22       * werden
23       *
24       */
25      @Test
26      public void testCopyZenith() {
27          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
28          // Play with the top card of your library revealed.
29          // You may cast the top card of your library if it's an instant or sorcery card.
30          // Whenever you cast an instant or sorcery spell from your library, copy it. You may choose new targets for the copy.
31          addCard(Zone.BATTLEFIELD, playerA, "Melek, Izzet Paragon");
32  
33          // Red Sun's Zenith deals X damage to any target.
34          // If a creature dealt damage this way would die this turn, exile it instead.
35          // Shuffle Red Sun's Zenith into its owner's library.
36          addCard(Zone.LIBRARY, playerA, "Red Sun's Zenith"); // {X}{R}
37          skipInitShuffling();
38  
39          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Red Sun's Zenith", playerB);
40          setChoice(playerA, "X=4");
41  
42          setStopAt(1, PhaseStep.BEGIN_COMBAT);
43          execute();
44  
45          assertGraveyardCount(playerA, "Red Sun's Zenith", 0);
46  
47          assertLife(playerA, 20);
48          assertLife(playerB, 12);
49  
50      }
51  }

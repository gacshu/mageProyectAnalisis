15      /**
16       * Sacred Ground {1}{W} Enchantment Whenever a spell or ability an opponent
17       * controls causes a land to be put into your graveyard from the
18       * battlefield, return that card to the battlefield.
19       *
20       *
21       * Destroyed land returns to battlefield
22       */
23      @Test
24      public void testSpellDestroyLandReturnsToBattlefield() {
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
26          // Destroy target land. If that land was nonbasic, Molten Rain deals 2 damage to the land's controller.
27          addCard(Zone.HAND, playerA, "Molten Rain");
28  
29          // Kabira Crossroads Land
30          // Kabira Crossroads enters the battlefield tapped.
31          // When Kabira Crossroads enters the battlefield, you gain 2 life.
32          // {T}: Add {W}to your mana pool.
33          addCard(Zone.HAND, playerB, "Kabira Crossroads");
34          addCard(Zone.BATTLEFIELD, playerB, "Sacred Ground");
35          addCard(Zone.BATTLEFIELD, playerB, "Plains");
36          playLand(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Kabira Crossroads");
37  
38          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Molten Rain", "Kabira Crossroads");
39  
40          setStopAt(3, PhaseStep.BEGIN_COMBAT);
41          execute();
42  
43          assertGraveyardCount(playerA, "Molten Rain", 1);
44          assertPermanentCount(playerB, "Kabira Crossroads", 1);
45  
46          assertLife(playerA, 20);
47          assertLife(playerB, 22); // + 2 * 2 life from  Kabira Crossroads - 2 Life from Molten Rain
48      }

117     /**
118      * I was playing against Sacred Ground. I Molten Rained oponents land and
119      * responded Sacred Ground trigger by exiling it with Surgical Extraction.
120      * Then after that resolved, Sacred Ground ability put the land from exile
121      * onto the battfield! Fix this, please
122      */
123     @Test
124     public void testWithSurgicalExtraction() {
125         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
126         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
127 
128         // Destroy target land.
129         // If that land was nonbasic, Molten Rain deals 2 damage to the land's controller.
130         addCard(Zone.HAND, playerA, "Molten Rain");// Instant {1}{R}{R}
131         // Choose target card in a graveyard other than a basic land card. Search its owner's graveyard,
132         // hand, and library for any number of cards with the same name as that card and exile them.
133         // Then that player shuffles their library.
134         addCard(Zone.HAND, playerA, "Surgical Extraction"); // Instant {B/P}
135 
136         addCard(Zone.BATTLEFIELD, playerB, "Caves of Koilos", 1);
137         /**
138          * Whenever a spell or ability an opponent controls causes a land to be
139          * put into your graveyard from the battlefield, return that card to the
140          * battlefield.
141          */
142         addCard(Zone.BATTLEFIELD, playerB, "Sacred Ground");
143 
144         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Molten Rain", "Caves of Koilos");
145         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, 1);
146         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Surgical Extraction", "Caves of Koilos");
147 
148         setStopAt(1, PhaseStep.BEGIN_COMBAT);
149         execute();
150 
151         assertGraveyardCount(playerA, "Molten Rain", 1);
152         assertGraveyardCount(playerA, "Surgical Extraction", 1);
153         assertExileCount("Caves of Koilos", 1);
154 
155         assertLife(playerA, 18);
156         assertLife(playerB, 18);
157     }

16      /**
17       * The text is rendered incorrect.
18       *
19       * First ability should trigger each end step but is triggering only on
20       * controller's end step.
21       *
22       */
23      @Test
24      public void testTriggersForEachPlayer() {
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
26          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
27          // At the beginning of each end step, if an opponent lost 2 or more life this turn, you may put a quest counter on Bloodchief Ascension. (Damage causes loss of life.)
28          // Whenever a card is put into an opponent's graveyard from anywhere, if Bloodchief Ascension has three or more quest counters on it, you may have that player lose 2 life. If you do, you gain 2 life.
29          addCard(Zone.HAND, playerA, "Bloodchief Ascension"); // Enchantment {B}
30          addCard(Zone.HAND, playerA, "Lightning Bolt", 2);
31  
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Bloodchief Ascension");
33          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
34          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
35          setStopAt(3, PhaseStep.UPKEEP);
36          execute();
37  
38          assertPermanentCount(playerA, "Bloodchief Ascension", 1);
39  
40          assertGraveyardCount(playerA, "Lightning Bolt", 2);
41  
42          assertLife(playerA, 20);
43          assertLife(playerB, 14);
44  
45          assertCounterCount("Bloodchief Ascension", CounterType.QUEST, 2);
46      }

48      /**
49       * Hey, I don't know how to submit bugs but in a game I played today I
50       * sacrificed Child of Alara by casting Bound at the end step of my previous
51       * opponent's turn, then chose Child as one of the cards to return to my
52       * hand.
53       * My graveyard was empty so that was the only card I chose.
54       * Child returned to my hand but it did NOT trigger for some reason.
55       * Nothing was destroyed.
56       */
57      @Test
58      public void testSacrificeChildOfAlara() {
59          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1); //Creature
60  
61          // Trample
62          // When Child of Alara dies, destroy all nonland permanents. They can't be regenerated.
63          addCard(Zone.BATTLEFIELD, playerB, "Icy Manipulator", 1); //Creature
64          addCard(Zone.BATTLEFIELD, playerB, "Child of Alara", 1); //Creature
65          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 1);
66          addCard(Zone.BATTLEFIELD, playerB, "Forest", 4);
67          // Bound
68          //      Sacrifice a creature.
69          //      Return up to X cards from your graveyard to your hand, where X is the number of colors that creature was.
70          //      Exile this card.
71          // Determined
72          //      Other spells you control can't be countered this turn.
73          //      Draw a card.
74          addCard(Zone.HAND, playerB, "Bound // Determined"); // Instant {3}{B}{G} // {G}{U}
75  
76          setStrictChooseMode(true);
77  
78          castSpell(1, PhaseStep.END_TURN, playerB, "Bound");
79          addTarget(playerB, "Child of Alara");
80          addTarget(playerB, "Child of Alara");
81  
82          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
83          execute();
84  
85          assertExileCount(playerB, "Bound // Determined", 1);
86  
87          assertHandCount(playerB, "Child of Alara", 1);
88  
89          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
90          assertGraveyardCount(playerB, "Icy Manipulator", 1);
91      }

16      /**
17       * Test case for issue #6349. Divine Visitation should not replace Treasure tokens created by Smothering Tithe
18       */
19      @Test
20      public void testDivineVisitationDoesNotReplaceNoncreatureTokens() {
21  
22          // If one or more creature tokens would be created under your control,
23          // that many 4/4 white Angel creature tokens with flying and vigilance are created instead.
24          addCard(Zone.BATTLEFIELD, playerA, "Divine Visitation");
25          // Whenever an opponent draws a card, that player may pay {2}.
26          // If the player doesn’t, you create a Treasure token.
27          // (It’s an artifact with “{T}, Sacrifice this artifact: Add one mana of any color.”)
28          addCard(Zone.BATTLEFIELD, playerA, "Smothering Tithe");
29          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
30          // Target player draws three cards.
31          addCard(Zone.HAND, playerA, "Ancestral Recall"); // {U}
32          // Let the opponent draw
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ancestral Recall", playerB);
34          setChoice(playerA, "Whenever an opponent draws a card", 2); // choose order of triggers
35          setChoice(playerB, false, 3); // Decline to pay 2
36          setStopAt(1, PhaseStep.BEGIN_COMBAT);
37  
38          setStrictChooseMode(true);
39          execute();
40  
41          assertHandCount(playerB, 3);
42          assertPermanentCount(playerA, "Treasure Token", 3);
43          assertType("Treasure Token", CardType.ARTIFACT, SubType.TREASURE);
44          assertNotType("Treasure Token", CardType.CREATURE);
45          assertNotSubtype("Treasure Token", SubType.ANGEL);
46          assertPermanentCount(playerA, "Angel Token", 0);
47          assertPermanentCount(playerA, 6);
48          assertGraveyardCount(playerA, 1);
49      }

51      @Test
52      public void testDivineVisitationReplacesCreatureTokens() {
53  
54          // If one or more creature tokens would be created under your control,
55          // that many 4/4 white Angel creature tokens with flying and vigilance are created instead.
56          addCard(Zone.BATTLEFIELD, playerA, "Divine Visitation");
57          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
58  
59          // Create two 1/1 red Goblin creature tokens.
60          addCard(Zone.HAND, playerA, "Dragon Fodder");
61          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Dragon Fodder");
62          setStopAt(1, PhaseStep.BEGIN_COMBAT);
63  
64          setStrictChooseMode(true);
65          execute();
66  
67          assertGraveyardCount(playerA, 1);
68          assertPermanentCount(playerA, 5);
69          assertPermanentCount(playerA, "Goblin Token", 0);
70          assertPermanentCount(playerA, "Angel Token", 2);
71          assertType("Angel Token", CardType.CREATURE, SubType.ANGEL);
72          assertColor(playerA, "Angel Token", ObjectColor.WHITE, true);
73          assertColor(playerA, "Angel Token", ObjectColor.RED, false);
74          assertPowerToughness(playerA, "Angel Token", 4, 4);
75          assertNotSubtype("Angel Token", SubType.GOBLIN);
76  
77      }

79      @Test
80      public void testSacrificeEOT() {
81          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
82          addCard(Zone.BATTLEFIELD, playerA, "Divine Visitation");

86  
87          setStrictChooseMode(true);
88          setStopAt(1, PhaseStep.END_TURN);
89          execute();

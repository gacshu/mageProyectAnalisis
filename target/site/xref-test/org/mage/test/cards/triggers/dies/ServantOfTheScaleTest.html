4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  
14  public class ServantOfTheScaleTest extends CardTestPlayerBase {
15  
16      /**
17       * Tests that the dies triggered ability distributes
18       * the +1/+1 counters on Servant of the Scale
19       */
20      @Test
21      public void testDiesTriggeredAbility() {
22          // Servant of the Scale enters the battlefield with a +1/+1 counter on it.
23          // When Servant of the Scale dies, put X +1/+1 counters on target creature you control, where X is the number of +1/+1 counter on Servant of the Scale.
24          addCard(Zone.HAND, playerA, "Servant of the Scale");
25          addCard(Zone.BATTLEFIELD, playerA, "Forest");
26          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
27  
28          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
29          addCard(Zone.HAND, playerB, "Lightning Bolt");
30  
31          setStrictChooseMode(true);
32  
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Servant of the Scale");
34          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Lightning Bolt", "Servant of the Scale");
35          addTarget(playerA, "Silvercoat Lion");
36  
37          setStopAt(1, PhaseStep.END_TURN);
38          execute();
39  
40          assertLife(playerA, 20);
41          assertLife(playerB, 20);
42  
43          assertGraveyardCount(playerB, "Lightning Bolt", 1);
44          assertGraveyardCount(playerA, "Servant of the Scale", 1);
45  
46          assertPermanentCount(playerA, "Silvercoat Lion", 1);
47          assertPowerToughness(playerA, "Silvercoat Lion", 3,3);
48      }
49  }

12  public class BroodSliverTest extends CardTestPlayerBase {
13  
14      /*
15      Reported bug: It lets the controller of Brood Sliver choose whether or not the token is created, instead of the attacking Sliver's controller.
16       */
17      @Test
18      public void testTokenCreatedBySliverController() {
19  
20          // Brood Sliver {4}{G} 3/3 Sliver
21          // Whenever a Sliver deals combat damage to a player, its controller may put a 1/1 colorless Sliver creature token onto the battlefield.
22          addCard(Zone.BATTLEFIELD, playerB, "Brood Sliver");
23          addCard(Zone.BATTLEFIELD, playerA, "Venom Sliver"); // 1/1 deathtouch granting sliver
24  
25          attack(1, playerA, "Venom Sliver");
26          setChoice(playerA, true); // controller of Venom Sliver dealing damage should get the choice to create token
27          setStopAt(1, PhaseStep.END_COMBAT);
28          execute();
29  
30          assertLife(playerB, 19);
31          assertPermanentCount(playerA, "Sliver Token", 1);
32          assertPermanentCount(playerB, "Sliver Token", 0);
33      }

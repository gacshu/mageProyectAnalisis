4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import mage.filter.Filter;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   *
12   * @author LevelX2
13   */
14  public class VoiceOfResurgenceTest extends CardTestPlayerBase {
15  
16      /**
17       * Voice of Resurgence creates it's token under owner's control upon death
18       * even when it is controlled by an opponent when it dies. Happened with
19       * Treachery.
20       *
21       */
22      @Test
23      public void testDiesTriggeredAbility() {
24          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
25          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
26          // Enchant creature
27          // When Treachery enters the battlefield, untap up to five lands.
28          // You control enchanted creature.
29          addCard(Zone.HAND, playerA, "Treachery"); // {3}{U}{U}
30  
31          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 1);
32          // Whenever an opponent casts a spell during your turn or when Voice of Resurgence dies,
33          // put a green and white Elemental creature token onto the battlefield with "This creature's power and toughness are each equal to the number of creatures you control."
34          addCard(Zone.BATTLEFIELD, playerB, "Voice of Resurgence", 1); // 2/2
35          addCard(Zone.HAND, playerB, "Lightning Bolt");
36  
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Treachery", "Voice of Resurgence");
38  
39          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Lightning Bolt", "Voice of Resurgence");
40  
41          setStopAt(1, PhaseStep.END_TURN);
42          execute();
43  
44          assertLife(playerA, 20);
45          assertLife(playerB, 20);
46  
47          assertGraveyardCount(playerA, "Treachery", 1);
48          assertGraveyardCount(playerB, "Voice of Resurgence", 1);
49  
50          assertPermanentCount(playerA, "Elemental Token", 2); // one from the Lightning Bolt and one from the Voice of Resurgence dying
51          assertPowerToughness(playerA, "Elemental Token", 3, 3, Filter.ComparisonScope.All);
52          assertTappedCount("Island", true, 0);
53      }

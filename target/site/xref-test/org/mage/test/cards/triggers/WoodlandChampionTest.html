12  public class WoodlandChampionTest extends CardTestPlayerBase {
13  
14      // Woodland Champion {1}{G}
15      // Whenever one or more tokens enter the battlefield under your control, put that many +1/+1 counters on Woodland Champion.
16  
17      @Test
18      public void test_TriggerOnTwoTokens() {
19          addCard(Zone.BATTLEFIELD, playerA, "Woodland Champion", 1);
20          // Acorn Harvest {3}{G}
21          // Create two 1/1 green Squirrel creature tokens.
22          addCard(Zone.HAND, playerA, "Acorn Harvest", 6);
23          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
24  
25          checkPermanentCounters("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Woodland Champion", CounterType.P1P1, 0);
26          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Acorn Harvest");
27  
28          checkPermanentCount("after", 1, PhaseStep.BEGIN_COMBAT, playerA, "Squirrel Token", 2);
29          checkPermanentCounters("after", 1, PhaseStep.BEGIN_COMBAT, playerA, "Woodland Champion", CounterType.P1P1, 2);
30  
31          setStrictChooseMode(true);
32          setStopAt(1, PhaseStep.END_TURN);
33          execute();
34      }
35  }

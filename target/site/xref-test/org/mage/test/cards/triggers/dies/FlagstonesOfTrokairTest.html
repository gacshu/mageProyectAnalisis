13  public class FlagstonesOfTrokairTest extends CardTestPlayerBase {
14  
15      /**
16       * If a flagstones of trokair enchated with spreading seas that is then
17       * ghostquartered. Should only fetch 1 land, but instead got 2.
18       */
19      @Test
20      public void testDontTriggerIfIsland() {
21          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
22          // Enchant land
23          // When Spreading Seas enters the battlefield, draw a card.
24          // Enchanted land is an Island.
25          addCard(Zone.HAND, playerA, "Spreading Seas", 1); // Enchantment Aura - {1}{U}
26          // {tap}: Add {W}.
27          // When Flagstones of Trokair is put into a graveyard from the battlefield, you may search your library for a Plains card and put it onto the battlefield tapped. If you do, shuffle your library.
28          addCard(Zone.BATTLEFIELD, playerA, "Flagstones of Trokair", 1);
29          addCard(Zone.LIBRARY, playerA, "Plains", 5);
30          // {T}: Add {C}.
31          // {T}, Sacrifice Ghost Quarter: Destroy target land. Its controller may search their library for a basic land card, put it onto the battlefield, then shuffle their library.
32          addCard(Zone.BATTLEFIELD, playerB, "Ghost Quarter", 1);
33  
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Spreading Seas", "Flagstones of Trokair");
35  
36          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "{T}, Sacrifice", "Flagstones of Trokair");
37          setStopAt(1, PhaseStep.END_TURN);
38          execute();
39  
40          assertGraveyardCount(playerB, "Ghost Quarter", 1);
41          assertGraveyardCount(playerA, "Flagstones of Trokair", 1);
42          assertGraveyardCount(playerA, "Spreading Seas", 1);
43  
44          assertPermanentCount(playerA, 3);
45  
46      }

4   import mage.constants.PhaseStep;

11   * @author LevelX2

13  public class ReturnToHandTest extends CardTestPlayerBase {
14  
15      /*
16          Undiscovered Paradise not being returned to hand on untap step.
17          The land provides mana of any color, but it's never returned to hand.
18       */
19      @Test
20      public void testExilesCreatureAtEndStep() {
21  
22          // {W}: Honor Guard gets +0/+1 until end of turn.
23          addCard(Zone.HAND, playerA, "Honor Guard"); // Creature 1/1
24  
25          // {T}: Add one mana of any color. During your next untap step, as you untap your permanents, return Undiscovered Paradise to its owner's hand.
26          addCard(Zone.BATTLEFIELD, playerA, "Undiscovered Paradise", 1);
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Honor Guard");
29  
30          setStopAt(3, PhaseStep.DRAW);
31          execute();
32  
33          assertPermanentCount(playerA, "Honor Guard", 1);
34          assertPermanentCount(playerA, "Undiscovered Paradise", 0);
35          assertHandCount(playerA, "Undiscovered Paradise", 1);
36  
37      }
38  }

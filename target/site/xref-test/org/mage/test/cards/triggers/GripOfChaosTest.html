15      /**
16       * Reported bug: https://github.com/magefree/mage/issues/6344
17       * I just had a game where we had an interaction between Grip of Chaos, Felidar Guardian, and Panharmonicon.
18       * The cloned Felidar trigger fizzled with valid targets on field because Grip retargeted that
19       * trigger onto Felidar itself, which isn't a valid target.
20       * Grip of Chaos specifically states it only chooses from valid targets when retargeting,
21       * so this is a bug somewhere in that interaction, though whether it only
22       * happens with cloned triggers or if there's a bad interaction between Grip
23       * and Felidar itself isn't clear.
24       */
25      @Test
26      public void noValidTargetsTest() {
27          // Whenever a spell or ability is put onto the stack, if it has a single target, reselect its target at random.
28          addCard(Zone.BATTLEFIELD, playerB, "Grip of Chaos", 1); // Enchantment
29  
30          // If an artifact or creature entering the battlefield causes a triggered ability
31          // of a permanent you control to trigger, that ability triggers an additional time.
32          addCard(Zone.BATTLEFIELD, playerA, "Panharmonicon", 1); // Artifact
33  
34          // When Felidar Guardian enters the battlefield, you may exile another target permanent you control,
35          // then return that card to the battlefield under its owner's control.
36          addCard(Zone.HAND, playerA, "Felidar Guardian"); // Creature {3}{W}
37          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
38          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
39          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
40          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
41  
42          // NOTE: Cannot use setStrictChoiseMode(true) with current test setup.
43          // When both ETB triggers target the same permanent because of Grip of Chaos the second (bottom-most)
44          // Ability will fizzle.
45          // This will cause random errors since setStrictChoiseMode(true) will require the player to make an explicit choice
46          // About whether or not to apply the effect of the ETB trigger.
47          // HOWEVER, when the second ETB fizzles, the "setChoice(playerA, "Yes")" will error out since it was a choice that did not get used.
48  
49          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Felidar Guardian");
50  
51          addTarget(playerA, "Forest");
52          addTarget(playerA, "Mountain");
53  
54          setStopAt(1, PhaseStep.BEGIN_COMBAT);
55  
56          execute();
57  
58          assertPermanentCount(playerA, "Felidar Guardian", 1);
59  
60          int zcc = 0;
61          zcc += getPermanent("Mountain").getZoneChangeCounter(currentGame);
62          zcc += getPermanent("Forest").getZoneChangeCounter(currentGame);
63          zcc += getPermanent("Swamp").getZoneChangeCounter(currentGame);
64          zcc += getPermanent("Plains").getZoneChangeCounter(currentGame);
65          zcc += getPermanent("Panharmonicon").getZoneChangeCounter(currentGame);
66          // If both select the same permanent to exile, one spell fizzles so zcc == 7 otherwise 9
67          if (zcc != 7) {
68              Assert.assertEquals("Sum of zone change counter should be 9", 9, zcc);
69              // creates error if the random targets do select the same target twice zcc is 7 then the second trigger has an invalid target
70          }
71      }

1   package org.mage.test.cards.triggers.dies;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;
7   
8   /**
9    *
10   * @author escplan9
11   */
12  public class ChildOfAlaraTest extends CardTestPlayerBase {
13  
14      @Test
15      public void testChildOfAlaraOblivionRingInteraction() {
16              /*
17              Child of Alara {W}{U}{B}{R}{G}
18               Legendary Creature — Avatar 6/6
19              Trample
20              When Child of Alara dies, destroy all nonland permanents. They can't be regenerated.
21              */
22              String childAlara = "Child of Alara";
23  
24              /*
25              Oblivion Ring {2}{W}
26              Enchantment
27              When Oblivion Ring enters the battlefield, exile another target nonland permanent.
28              When Oblivion Ring leaves the battlefield, return the exiled card to the battlefield under its owner's control.
29              */
30              String oRing = "Oblivion Ring";
31  
32              /*
33              Tamiyo, Field Research {1}{G}{W}{U}
34              Planeswalker - Tamiyo
35              +1: Choose up to two target creatures. Until your next turn, whenever either of those creatures deals combat damage, you draw a card.
36              −2: Tap up to two target nonland permanents. They don't untap during their controller's next untap step.
37              −7: Draw three cards. You get an emblem with "You may cast spells from your hand without paying their mana costs."
38              */
39              String tamiyo = "Tamiyo, Field Researcher";
40  
41              String memnite = "Memnite"; // {0} 1/1
42              String hGiant = "Hill Giant"; // {3}{R} 3/3
43              String dDemolition = "Daring Demolition"; // {2}{B}{B} sorcery: destroy target creature or vehicle
44              
45              addCard(Zone.HAND, playerA, tamiyo);
46              addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
47              addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
48              addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
49              addCard(Zone.BATTLEFIELD, playerA, hGiant);
50              addCard(Zone.BATTLEFIELD, playerA, childAlara);
51              
52              addCard(Zone.HAND, playerB, dDemolition);	
53              addCard(Zone.HAND, playerB, oRing);
54              addCard(Zone.BATTLEFIELD, playerB, "Swamp", 4);
55              addCard(Zone.BATTLEFIELD, playerB, "Plains", 4);
56              addCard(Zone.BATTLEFIELD, playerB, memnite);
57  
58              castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, tamiyo);
59              castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, oRing);
60              addTarget(playerB, tamiyo);
61              waitStackResolved(2, PhaseStep.PRECOMBAT_MAIN);
62              castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, dDemolition, childAlara);
63              
64              setStopAt(2, PhaseStep.BEGIN_COMBAT);
65              execute();
66              
67              assertGraveyardCount(playerB, dDemolition, 1);
68              assertGraveyardCount(playerA, childAlara, 1);
69              assertGraveyardCount(playerB, oRing, 1); // destroyed by child
70              assertGraveyardCount(playerB, memnite, 1);
71              assertGraveyardCount(playerA, hGiant, 1);
72              assertPermanentCount(playerA, tamiyo, 1); // o-ring destroyed returns Tamiyo to battlefield
73      }
74  }

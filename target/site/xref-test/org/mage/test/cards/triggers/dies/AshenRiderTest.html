13  public class AshenRiderTest extends CardTestPlayerBase {
14  
15      /*
16       * Volrath, the Shapestealer and Ashen Rider, Ashen Rider has a counter on it:
17          Turn Volrath into Ashen Rider:
18          Destroy the Volrath (who's the Ashen Rider) with Putrefy:  
19          The death trigger for the Volrath copying Ashen Rider did not trigger.
20       */
21      @Test
22      public void cartelAristrocraftInteractionOpponentDoesNotPayLife() {
23          setStrictChooseMode(true);
24          
25          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);        
26          
27          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);        
28          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);        
29          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);        
30  
31          // Flying
32          // When Ashen Rider enters the battlefield or dies, exile target permanent.
33          addCard(Zone.BATTLEFIELD, playerA, "Ashen Rider"); // Creature {4}{W}{W}{B}{B}
34  
35          // At the beginning of combat on your turn, put a -1/-1 counter on up to one target creature.
36          // {1}: Until your next turn, Volrath, the Shapestealer becomes a copy of target creature with a counter on it, except it's 7/5 and it has this ability.        
37          addCard(Zone.HAND, playerA, "Volrath, the Shapestealer"); // Creature {2}{B}{G}{U}       
38          addTarget(playerA, "Ashen Rider");
39  
40          // Destroy target artifact or creature. It can't be regenerated.
41          addCard(Zone.HAND, playerA, "Putrefy"); // Instant {1}{B}{G}
42  
43  
44          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Volrath, the Shapestealer");
45          
46          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}: Until your next turn");
47          addTarget(playerA,  "Ashen Rider");
48          
49          waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
50          
51          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Putrefy", "Ashen Rider[only copy]");
52  
53          addTarget(playerA, "Silvercoat Lion"); // Dies trigger of Volrath, the Shapestealer copied from Ashen Rider
54  
55          setStopAt(3, PhaseStep.BEGIN_COMBAT);
56          execute();
57  
58          assertPowerToughness(playerA, "Ashen Rider", 4,4);
59          
60          assertGraveyardCount(playerA, "Putrefy", 1);        
61          assertGraveyardCount(playerA, "Volrath, the Shapestealer", 1);
62          
63          assertExileCount(playerB, "Silvercoat Lion", 1);
64          
65      }    
66  }

13      /**
14       * When Brain Maggot enters the battlefield, target opponent reveals their
15       * hand and you choose a nonland card from it. Exile that card until Brain
16       * Maggot leaves the battlefield.
17       */
18      @Test
19      public void testCardFromHandWillBeExiled() {
20          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
21          addCard(Zone.HAND, playerA, "Brain Maggot", 2);
22  
23          addCard(Zone.HAND, playerB, "Bloodflow Connoisseur", 1);
24  
25          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Brain Maggot");
26          // addTarget(playerA, playerB); Autochosen, only target
27          setChoice(playerA, "Bloodflow Connoisseur");
28  
29          setStopAt(1, PhaseStep.BEGIN_COMBAT);
30          execute();
31  
32          assertPermanentCount(playerA, "Brain Maggot", 1);
33          assertExileCount("Bloodflow Connoisseur", 1);
34      }

36      @Test
37      public void testCardFromHandWillBeExiledAndReturn() {
38          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
39          // When Brain Maggot enters the battlefield, target opponent reveals their hand and you choose a nonland card from it. Exile that card until Brain Maggot leaves the battlefield.
40          addCard(Zone.HAND, playerA, "Brain Maggot", 2);
41  
42          addCard(Zone.HAND, playerB, "Bloodflow Connoisseur", 1);
43          addCard(Zone.HAND, playerB, "Lightning Bolt", 1);
44          addCard(Zone.BATTLEFIELD, playerB, "Mountain", 2);
45  
46          // exile
47          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Brain Maggot");
48          // addTarget(playerA, playerB); Autochosen, only target
49          setChoice(playerA, "Bloodflow Connoisseur");
50  //        showExile("exile", 1, PhaseStep.BEGIN_COMBAT, playerB);
51          checkExileCount("blood must be in exile", 1, PhaseStep.BEGIN_COMBAT, playerB, "Bloodflow Connoisseur", 1);
52  
53          // return
54          castSpell(1, PhaseStep.END_COMBAT, playerB, "Lightning Bolt", "Brain Maggot");
55          checkPermanentCount("brain must die", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Brain Maggot", 0);
56          checkExileCount("blood must return from exile", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Bloodflow Connoisseur", 0);
57          checkHandCardCount("blood must be in hand", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Bloodflow Connoisseur", 1);
58  
59          setStopAt(1, PhaseStep.END_TURN);
60          execute();
61  
62          assertGraveyardCount(playerA, "Brain Maggot", 1);
63          assertGraveyardCount(playerB, "Lightning Bolt", 1);
64          assertHandCount(playerB, "Bloodflow Connoisseur", 1);
65          assertExileCount("Bloodflow Connoisseur", 0);
66      }

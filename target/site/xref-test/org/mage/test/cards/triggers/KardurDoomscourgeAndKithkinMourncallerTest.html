19      @Test
20      public void testKDRemovedFromCombatViaRegenerateAbility() {
21          setStrictChooseMode(true);
22          // Kardur, Doomscourge: if an attacking creature dies, each opponent loses 1 life and you gain 1 life
23          addCard(Zone.BATTLEFIELD, playerA, "Kardur, Doomscourge");
24          addCard(Zone.BATTLEFIELD, playerA, "Elvish Archers");
25          addCard(Zone.BATTLEFIELD, playerB, "Serra Angel");
26          addCard(Zone.HAND, playerA, "Regenerate");
27          addCard(Zone.HAND, playerA, "Terror");
28          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
29          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
30  
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Regenerate", "Elvish Archers");
32  
33          attack(1, playerA, "Elvish Archers");
34          block(1, playerB, "Serra Angel", "Elvish Archers");  // regeneration shield used up and EA is removed from combat
35  
36          castSpell(1, PhaseStep.END_COMBAT, playerA, "Terror", "Elvish Archers");  // still within the combat phase, the EA is destroyed/dies
37  
38          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
39          execute();
40  
41          assertGraveyardCount(playerA, "Elvish Archers", 1);
42  
43          // does not fire due to the Elvish Archers not in an attacking state
44          assertLife(playerA, 20);
45          assertLife(playerB, 20);
46  
47      }

49      @Test
50      public void testSuccessfulKDTrigger() {
51          setStrictChooseMode(true);
52          // Kardur, Doomscourge: if an attacking creature dies, each opponent loses 1 life and you gain 1 life
53          addCard(Zone.BATTLEFIELD, playerA, "Kardur, Doomscourge");
54          addCard(Zone.BATTLEFIELD, playerA, "Elvish Archers"); // 2/2 first strike
55          addCard(Zone.BATTLEFIELD, playerB, "Serra Angel");  // 4/4 vigilance
56  
57          attack(1, playerA, "Elvish Archers");
58          block(1, playerB, "Serra Angel", "Elvish Archers");  // Elvish Archer dies causing KD to trigger
59  
60          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
61          execute();
62  
63          assertGraveyardCount(playerA, "Elvish Archers", 1);
64  
65          // successful fire so playerA gains 1 life and playerB loses 1 life
66          assertLife(playerA, 21);
67          assertLife(playerB, 19);
68  
69      }

71      @Test
72      public void testKMTrigger() {
73          setStrictChooseMode(true);
74          // Kithkin Mourncaller: if an elf or kithkin dies, you may draw a card
75          addCard(Zone.BATTLEFIELD, playerA, "Kithkin Mourncaller");
76          addCard(Zone.BATTLEFIELD, playerA, "Elvish Archers");  // 2/1 first strike
77          addCard(Zone.BATTLEFIELD, playerA, "Pearled Unicorn"); // 2/2
78          addCard(Zone.BATTLEFIELD, playerB, "Serra Angel"); // 4/4 vigilance
79          addCard(Zone.BATTLEFIELD, playerB, "Runeclaw Bear");  // 2/2 
80          addCard(Zone.LIBRARY, playerA, "Island", 2);  // used for draw trigger
81  
82          attack(1, playerA, "Elvish Archers");
83          attack(1, playerA, "Pearled Unicorn");
84          block(1, playerB, "Serra Angel", "Elvish Archers");  // Elvish Achers will die and trigger KM
85          block(1, playerB, "Runeclaw Bear", "Pearled Unicorn");  // Pearled Unicorn will die but not trigger KM
86  
87          setChoice(playerA, "Yes");  // accept the drawing of a card from the single trigger (Elvish Archers "elf type")
88          
89          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
90          execute();
91  
92          assertGraveyardCount(playerA, "Elvish Archers", 1);
93          assertGraveyardCount(playerA, "Pearled Unicorn", 1);
94  
95          // successful fire due to dead Elvish Archers (elf) so playerA draws a card
96          assertHandCount(playerA, 1);
97  
98      }

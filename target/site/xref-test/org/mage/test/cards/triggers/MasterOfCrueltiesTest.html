10   *
11   * @author LevelX2
12   */
13  public class MasterOfCrueltiesTest extends CardTestPlayerBase {
14  
15      /*
16      The ability of an Alesha-resurrected Master of Cruelties triggered in an EDH game despite being blocked by a creature.
17       */
18      @Test
19      public void testMasterWasAleshaAnimated() {
20          // First strike
21          // Whenever Alesha, Who Smiles at Death attacks, you may pay {W/B}{W/B}. If you do, return target creature card with power 2 or less from your graveyard to the battlefield tapped and attacking.
22          addCard(Zone.BATTLEFIELD, playerB, "Alesha, Who Smiles at Death"); // 3/2
23  
24          // First strike
25          // Deathtouch
26          // Master of Cruelties can only attack alone.
27          // Whenever Master of Cruelties attacks a player and isn't blocked, that player's life total becomes 1. Master of Cruelties assigns no combat damage this combat.
28          addCard(Zone.GRAVEYARD, playerB, "Master of Cruelties"); // 1/4
29  
30          addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
31  
32          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
33  
34          attack(2, playerB, "Alesha, Who Smiles at Death");
35          setChoice(playerB, true);
36          // Master of Cruelties is autochosen since only target
37  
38          block(2, playerA, "Silvercoat Lion", "Master of Cruelties");
39  
40          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
41          execute();
42  
43          assertPermanentCount(playerB, "Master of Cruelties", 1);
44          assertTapped("Master of Cruelties", true);
45          assertTapped("Alesha, Who Smiles at Death", true);
46  
47          assertLife(playerA, 17);
48          assertLife(playerB, 20);
49  
50          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
51      }
52  }

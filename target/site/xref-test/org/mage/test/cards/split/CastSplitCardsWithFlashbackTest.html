13      @Test
14      public void test_Flashback_Simple() {
15          // {1}{U}
16          // When Snapcaster Mage enters the battlefield, target instant or sorcery card in your graveyard gains flashback until end of turn.
17          addCard(Zone.HAND, playerA, "Snapcaster Mage", 1);
18          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
19          //
20          addCard(Zone.GRAVEYARD, playerA, "Lightning Bolt", 1);
21          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
22  
23          // add flashback
24          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Snapcaster Mage");
25          addTarget(playerA, "Lightning Bolt");
26  
27          // cast as flashback
28          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
29          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Flashback", playerB);
30  
31          setStrictChooseMode(true);
32          setStopAt(1, PhaseStep.BEGIN_COMBAT);
33          execute();
34  
35          assertLife(playerB, 20 - 3);
36          assertExileCount(playerA, "Lightning Bolt", 1);
37      }

82          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);

84          addCard(Zone.BATTLEFIELD, playerB, "Bow of Nylea", 1); // Legendary Enchantment Artifact
85  
86          // play card as normal to change ZCC for split cards (simulate GUI session - ZCC's was bugged and card's parts was able to have different ZCC)

88  
89          // add flashback
90          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);

13      // splice cost applies for one fused spell, not to every part
14      // https://github.com/magefree/mage/issues/6493
15  
16      @Test
17      public void test_ThaliaGuardianOfThraben_CostModification_Fused() {
18          removeAllCardsFromHand(playerA);
19  
20          // Noncreature spells cost {1} more to cast.
21          addCard(Zone.BATTLEFIELD, playerA, "Thalia, Guardian of Thraben", 1);
22          //
23          // Wear {1}{R} Destroy target artifact.
24          // Tear {W} Destroy target enchantment.
25          addCard(Zone.HAND, playerA, "Wear // Tear", 1);
26          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2 + 1);
27          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
28          addCard(Zone.BATTLEFIELD, playerB, "Bident of Thassa", 1); // Legendary Enchantment Artifact
29          addCard(Zone.BATTLEFIELD, playerB, "Bow of Nylea", 1); // Legendary Enchantment Artifact
30  
31          // cast fused
32          // old bug: https://github.com/magefree/mage/issues/6493
33          // * getPlayable checks fused cost (+1 modification)
34          // * activate checks fused cost (+1 modification) andeach card's part (+1 modification)
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "fused Wear // Tear");
36          addTarget(playerA, "Bident of Thassa");
37          addTarget(playerA, "Bow of Nylea");
38  
39          setStrictChooseMode(true);
40          setStopAt(1, PhaseStep.END_TURN);
41          execute();
42  
43          assertGraveyardCount(playerA, "Wear // Tear", 1);
44          assertGraveyardCount(playerB, "Bident of Thassa", 1);
45          assertGraveyardCount(playerB, "Bow of Nylea", 1);
46          assertHandCount(playerA, 0);
47  
48          // must used all mana
49          assertTappedCount("Mountain", true, 3);
50          assertTappedCount("Plains", true, 1);
51      }

53      @Test
54      public void test_ThaliaGuardianOfThraben_CostModification_FusedWithSplice() {
55          removeAllCardsFromHand(playerA);
56  
57          // Noncreature spells cost {1} more to cast.
58          addCard(Zone.BATTLEFIELD, playerA, "Thalia, Guardian of Thraben", 1);
59          //
60          // Draw a card.
61          // Splice onto instant or sorcery {2}{U}
62          addCard(Zone.HAND, playerA, "Everdream", 1);
63          addCard(Zone.BATTLEFIELD, playerA, "Island", 3); // for splice
64          //
65          // Wear {1}{R} Destroy target artifact.
66          // Tear {W} Destroy target enchantment.
67          addCard(Zone.HAND, playerA, "Wear // Tear", 1);
68          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2 + 1);
69          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
70          addCard(Zone.BATTLEFIELD, playerB, "Bident of Thassa", 1); // Legendary Enchantment Artifact
71          addCard(Zone.BATTLEFIELD, playerB, "Bow of Nylea", 1); // Legendary Enchantment Artifact
72  
73          // cast fused with splice
74          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "fused Wear // Tear");
75          setChoice(playerA, true); // use splice
76          addTarget(playerA, "Everdream"); // card to splice
77          addTarget(playerA, "Bident of Thassa"); // target left
78          addTarget(playerA, "Bow of Nylea"); // target right
79  
80          // must used all mana
81          setStrictChooseMode(true);
82          setStopAt(1, PhaseStep.END_TURN);
83          execute();
84  
85          assertGraveyardCount(playerA, "Wear // Tear", 1);
86          assertGraveyardCount(playerB, "Bident of Thassa", 1);
87          assertGraveyardCount(playerB, "Bow of Nylea", 1);
88          assertHandCount(playerA, 2); // splice card + draw effect from spliced
89  
90          // must used all mana
91          assertTappedCount("Island", true, 3);
92          assertTappedCount("Mountain", true, 3);
93          assertTappedCount("Plains", true, 1);
94      }

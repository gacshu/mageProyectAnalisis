11  public class ActOfHeroismTest extends CardTestPlayerBase {
12  
13      // Test that loss of abilities doesn't remove Act of Heroism's "block an additional creature" effect
14      @Test
15      public void testCanBlockMultiple() {
16          // 0/4 Creature - Camel
17          addCard(Zone.BATTLEFIELD, playerA, "Tasseled Dromedary");
18          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
19          addCard(Zone.HAND, playerA, "Act of Heroism");
20  
21          addCard(Zone.BATTLEFIELD, playerB, "Unwavering Initiate");
22          addCard(Zone.BATTLEFIELD, playerB, "Sacred Cat");
23          addCard(Zone.BATTLEFIELD, playerB, "Plains", 8);
24          addCard(Zone.HAND, playerB, "Overwhelming Splendor");
25  
26          // Untap target creature. It gets +2/+2 until end of turn and can block an additional creature this turn
27          castSpell(2, PhaseStep.UPKEEP, playerA, "Act of Heroism", "Tasseled Dromedary");
28  
29          // Creatures enchanted player controls loses all abilities and have base power and toughness 1/1
30          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Overwhelming Splendor", playerA);
31  
32          attack(2, playerB, "Unwavering Initiate");
33          attack(2, playerB, "Sacred Cat");
34  
35          block(2, playerA, "Tasseled Dromedary", "Unwavering Initiate");
36          block(2, playerA, "Tasseled Dromedary", "Sacred Cat");
37  
38          setStopAt(2, PhaseStep.COMBAT_DAMAGE);
39  
40          // Fails if A's creature is unable to block both attackers
41          execute();
42      }
43  }

16      @Test
17      public void test_PreventDamageToGideonOnYourTurn() {

19          addCard(Zone.BATTLEFIELD, playerA, "Alloy Myr"); // 2/2
20  
21          // with redirect
22          checkDamage("turn 1 before", 1, PhaseStep.PRECOMBAT_MAIN, playerB, "Martyrs of Korlis", 0);
23          checkLife("turn 1 before", 1, PhaseStep.PRECOMBAT_MAIN, playerB, 20);
24          attack(1, playerA, "Alloy Myr", playerB);
25          checkDamage("turn 1 after", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Martyrs of Korlis", 2);
26          checkLife("turn 1 after", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, 20);
27  
28          attack(2, playerB, "Martyrs of Korlis", playerA);
29  
30          // without redirect
31          checkDamage("turn 3 before", 3, PhaseStep.PRECOMBAT_MAIN, playerB, "Martyrs of Korlis", 0);
32          checkLife("turn 3 before", 3, PhaseStep.PRECOMBAT_MAIN, playerB, 20);
33          attack(3, playerA, "Alloy Myr", playerB);
34          checkDamage("turn 3 after", 3, PhaseStep.POSTCOMBAT_MAIN, playerB, "Martyrs of Korlis", 0);
35          checkLife("turn 3 after", 3, PhaseStep.POSTCOMBAT_MAIN, playerB, 20 - 2);
36  
37          setStrictChooseMode(true);
38          setStopAt(3, PhaseStep.END_TURN);
39          execute();

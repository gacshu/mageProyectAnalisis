13      /**
14       * Tests BecomesCreatureSourceEffect with unsummon
15       * Wait until Angel's Tomb becomes creature, then (in post combat phase) cast unsummon and cast it again.
16       */
17      @Test
18      public void testUnsummonToAnimatedArtifact() {
19          addCard(Zone.BATTLEFIELD, playerA, "Forest");
20          addCard(Zone.BATTLEFIELD, playerA, "Island");
21          addCard(Zone.BATTLEFIELD, playerA, "Plains", 3);
22  
23          addCard(Zone.BATTLEFIELD, playerA, "Angel's Tomb");
24  
25          addCard(Zone.HAND, playerA, "Llanowar Elves");
26          addCard(Zone.HAND, playerA, "Unsummon");
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Llanowar Elves");
29  
30          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Unsummon", "Angel's Tomb");
31          waitStackResolved(1, PhaseStep.POSTCOMBAT_MAIN);
32          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Angel's Tomb");
33  
34          setStopAt(1, PhaseStep.END_TURN);
35          execute();
36  
37          assertGraveyardCount(playerA, "Unsummon", 1);
38  
39          assertPermanentCount(playerA, "Llanowar Elves", 1);
40          assertPermanentCount(playerA, "Angel's Tomb", 1);
41          assertPowerToughness(playerA, "Angel's Tomb", 0, 0);
42      }

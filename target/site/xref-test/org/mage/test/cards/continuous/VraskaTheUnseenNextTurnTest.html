12  public class VraskaTheUnseenNextTurnTest extends CardTestPlayerBase {
13  
14      @Test
15      public void test_SingleOpponentMustAttack() {
16          // +1: Until your next turn, whenever a creature deals combat damage to Vraska the Unseen, destroy that creature.
17          addCard(Zone.BATTLEFIELD, playerA, "Vraska the Unseen");
18          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 3); // 2/2
19  
20          // 1 - activate
21          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1:");
22          checkPermanentCounters("turn 1", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Vraska the Unseen", CounterType.LOYALTY, 5 + 1);
23          checkPermanentCount("turn 1", 1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Balduvian Bears", 3);
24  
25          // 2 - attack and destroy
26          attack(2, playerB, "Balduvian Bears", "Vraska the Unseen");
27          checkPermanentCounters("turn 2", 2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Vraska the Unseen", CounterType.LOYALTY, 5 + 1 - 2);
28          checkPermanentCount("turn 2", 2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Balduvian Bears", 3 - 1);
29  
30          // 3 - nothing
31          checkPermanentCounters("turn 3", 3, PhaseStep.POSTCOMBAT_MAIN, playerA, "Vraska the Unseen", CounterType.LOYALTY, 5 + 1 - 2);
32          checkPermanentCount("turn 3", 3, PhaseStep.POSTCOMBAT_MAIN, playerB, "Balduvian Bears", 3 - 1);
33  
34          // 4 - attack and DO NOT destroy
35          checkPermanentCounters("turn 4", 4, PhaseStep.POSTCOMBAT_MAIN, playerA, "Vraska the Unseen", CounterType.LOYALTY, 5 + 1 - 2 * 2);
36          attack(4, playerB, "Balduvian Bears", "Vraska the Unseen");
37          checkPermanentCount("turn 4", 4, PhaseStep.POSTCOMBAT_MAIN, playerB, "Balduvian Bears", 3 - 1);
38  
39          setStopAt(4, PhaseStep.END_TURN);
40          setStrictChooseMode(true);
41          execute();
42      }
43  }

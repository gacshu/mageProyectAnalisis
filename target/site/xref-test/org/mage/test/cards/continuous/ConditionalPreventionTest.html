105     @Test
106     public void test_PreventDamageConditionalNotActiveWithOtherEffect() {
107         addCustomCardWithAbility("prevent", playerA, new SimpleStaticAbility(
108                 new ConditionalPreventionEffect(
109                         new PreventAllDamageToAllEffect(Duration.WhileOnBattlefield, StaticFilters.FILTER_PERMANENT),
110                         new PreventAllDamageToPlayersEffect(Duration.WhileOnBattlefield, false),
111                         NotMyTurnCondition.instance,
112                         ""
113                 )
114         ));
115 
116         addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
117         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
118         addCard(Zone.HAND, playerA, "Lightning Bolt", 2);
119 
120         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Balduvian Bears"); // will prevent
121         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerA); // will not prevent
122 
123         setStrictChooseMode(true);
124         setStopAt(1, PhaseStep.END_TURN);
125         execute();
126 
127         assertPermanentCount(playerA, "Balduvian Bears", 0); // not prevented, dies
128         assertLife(playerA, 20); // prevented, no damage
129         assertHandCount(playerA, "Lightning Bolt", 0);
130     }

132     @Test
133     public void test_PreventableCombatDamage() {
134         // Prevent all damage that would be dealt to creatures.
135         addCard(Zone.BATTLEFIELD, playerA, "Bubble Matrix", 1);
136         addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
137         //
138         addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 1);
139 
140         // player A must do damage
141         attack(1, playerA, "Balduvian Bears", playerB);
142 
143         // player B can't do damage (bears must block and safe)
144         attack(4, playerB, "Balduvian Bears", playerA);
145         block(4, playerA, "Balduvian Bears", "Balduvian Bears");
146 
147         setStrictChooseMode(true);
148         setStopAt(4, PhaseStep.END_TURN);
149         execute();
150 
151         assertPermanentCount(playerA, "Balduvian Bears", 1);
152         assertPermanentCount(playerB, "Balduvian Bears", 1);
153         assertLife(playerA, 20);
154         assertLife(playerB, 20 - 2);
155     }

157     @Test
158     public void test_UnpreventableCombatDamage() {
159         // Combat damage that would be dealt by creatures you control can't be prevented.
160         addCard(Zone.BATTLEFIELD, playerB, "Questing Beast", 1);
161         //
162         // Prevent all damage that would be dealt to creatures.
163         addCard(Zone.BATTLEFIELD, playerA, "Bubble Matrix", 1);
164         addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
165         //
166         addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 1);
167 
168         // player A must do damage
169         attack(1, playerA, "Balduvian Bears", playerB);
170 
171         // player B must be prevented by Bubble Matrix, but can't (Questing Beast)
172         // a -> b -- can't do damage (matrix)
173         // b -> a -- can do damage (matrix -> quest)
174         attack(4, playerB, "Balduvian Bears", playerA);
175         block(4, playerA, "Balduvian Bears", "Balduvian Bears");
176 
177         setStrictChooseMode(true);
178         setStopAt(4, PhaseStep.END_TURN);
179         execute();
180 
181         assertPermanentCount(playerA, "Balduvian Bears", 0);
182         assertPermanentCount(playerB, "Balduvian Bears", 1);
183         assertLife(playerA, 20);
184         assertLife(playerB, 20 - 2);
185     }

187     @Test
188     public void test_PreventSomeDamage_Normal() {
189         // Kicker-Sacrifice a land.
190         // Prevent the next 3 damage that would be dealt this turn to any number of target creatures and/or players, divided as you choose.
191         // If Pollen Remedy was kicked, prevent the next 6 damage this way instead.
192         addCard(Zone.HAND, playerA, "Pollen Remedy", 1); // {W}
193         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
194         addCard(Zone.HAND, playerA, "Swamp", 1); // for kicker
195         //
196         addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
197         //
198         addCard(Zone.HAND, playerA, "Lightning Bolt", 3);
199         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
200 
201         // add shield for 3 damage
202         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Pollen Remedy");
203         setChoice(playerA, false); // no kicker
204         addTargetAmount(playerA, "Balduvian Bears", 3);
205         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
206         checkGraveyardCount("shield", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Pollen Remedy", 1);
207 
208         // 6 damage to die (if no shield then can cast only 1 bolt)
209         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Balduvian Bears");
210         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
211         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Balduvian Bears");
212         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
213         checkGraveyardCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", 2);
214         checkGraveyardCount("after", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
215 
216         setStrictChooseMode(true);
217         setStopAt(1, PhaseStep.END_TURN);
218         execute();
219     }

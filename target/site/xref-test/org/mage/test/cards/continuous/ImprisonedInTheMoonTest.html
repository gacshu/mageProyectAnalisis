12  public class ImprisonedInTheMoonTest extends CardTestPlayerBase {
13      
14      /**
15       * Reported bug: Urza land enchanted with Imprisoned In the Moon incorrectly makes it so other Urza lands only tap for 1 <>.
16       * 
17       * Card ruling: https://gatherer.wizards.com/Pages/Card/Details.aspx?multiverseid=414360
18       *   If the enchanted permanent is a land and has land types, it retains those types even though it loses any intrinsic mana abilities associated with them. 
19       *   For example, a Plains enchanted by Imprisoned in the Moon is still a Plains, but it can't tap for {W}, only for {C}.
20       */
21      @Test
22      public void testEnchantUrzaLand() {
23          // Imprisoned in the Moon - Enchantment - Aura - {2}{U}
24          // Enchant creature, land, or planeswalker
25          // Enchanted permanent is a colorless land with "Tap: Add Colorless" and loses all other card types and abilities.
26          addCard(Zone.HAND, playerA, "Imprisoned in the Moon", 1);
27          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
28  
29          addCard(Zone.BATTLEFIELD, playerB, "Urza's Mine", 1);
30          addCard(Zone.BATTLEFIELD, playerB, "Urza's Tower", 1);
31          addCard(Zone.BATTLEFIELD, playerB, "Urza's Power Plant", 1);
32          addCard(Zone.HAND, playerB, "Wurmcoil Engine", 1); // {6} Wurm - 6/6 Deathtouch and Lifelink
33          addCard(Zone.HAND, playerB, "Bonesplitter", 1); // {1} Equipament
34          
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Imprisoned in the Moon");
36          addTarget(playerA, "Urza's Mine");
37          // 6 total mana available - 3 from Tower, 2 from Powerplant, 1 from enchanted Mine
38          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Wurmcoil Engine");
39          checkPlayableAbility("Can't cast Bonesplitter", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Bonesplitter", false);
40  
41          setStopAt(2, PhaseStep.BEGIN_COMBAT);
42          execute();
43                  
44          assertPermanentCount(playerA, "Imprisoned in the Moon", 1);
45          assertPermanentCount(playerB, "Wurmcoil Engine", 1); 
46          assertHandCount(playerB,"Wurmcoil Engine", 0);        
47      }
48  }

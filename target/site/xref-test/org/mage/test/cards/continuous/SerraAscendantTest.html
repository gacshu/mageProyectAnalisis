3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  
14  public class SerraAscendantTest extends CardTestPlayerBase {
15  
16      /**
17       * The game goes on; they play Serra Ascendant on turn one, pass the
18       * turn, you play your newly unbanned Wild Nacatl with a Stomping Ground and
19       * also pass the turn.
20       * On turn 2, they cast a Martyr of Sands and sacrifice it,
21       * revealing 3 white cards to gain 9 life and end up at 29.
22       * They go to the combat phase, declare Serra as an attacker, and you happily block
23       * him, thinking that this is such a bad move from them.
24       * After the damage is dealt, the Serra is still there, bigger than ever.
25       */
26      @Test
27      public void testSilence() {
28          addCard(Zone.HAND, playerA, "Plains", 2);
29          // As long as you have 30 or more life, Serra Ascendant gets +5/+5 and has flying.
30          addCard(Zone.HAND, playerA, "Serra Ascendant");
31          // {1}, Reveal X white cards from your hand, Sacrifice Martyr of Sands: You gain three times X life.
32          addCard(Zone.HAND, playerA, "Martyr of Sands");
33          addCard(Zone.HAND, playerA, "Silvercoat Lion",3);
34  
35          addCard(Zone.HAND, playerB, "Stomping Ground", 1);
36          addCard(Zone.HAND, playerB, "Wild Nacatl", 1);
37  
38          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Plains");
39          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Serra Ascendant");
40          
41          playLand(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Stomping Ground");
42          setChoice(playerB, true);
43          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Wild Nacatl");
44          
45          playLand(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Plains");
46          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Martyr of Sands", true);
47          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}, You may reveal X white cards from your hand");
48          setChoice(playerA, "Silvercoat Lion");
49          setChoice(playerA, "Silvercoat Lion");
50          setChoice(playerA, "Silvercoat Lion");
51          
52          attack(3, playerA, "Serra Ascendant");
53          block(3, playerB, "Wild Nacatl", "Serra Ascendant");
54  
55          setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
56          
57          execute();
58  
59          assertGraveyardCount(playerA, "Martyr of Sands", 1);
60  
61          assertLife(playerB, 18);
62          assertLife(playerA, 30);
63          
64          assertPermanentCount(playerB, "Wild Nacatl", 1);
65      
66          assertPermanentCount(playerA, "Serra Ascendant", 1);
67          assertPowerToughness(playerA, "Serra Ascendant", 6, 6);
68      }
69  }

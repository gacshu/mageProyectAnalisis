11  public class CommanderAsIndomitableCreativityCastTest extends CardTestCommander4Players {
12  
13      // target adjusters test in command zone
14  
15      // Player order: A -> D -> C -> B
16  
17      // https://github.com/magefree/mage/issues/5852
18      // Indomitable Creativity
19      // Destroy X target artifacts and/or creatures. For each permanent destroyed this way,
20      // its controller reveals cards from the top of their library until an artifact or creature card is revealed
21      // and exiles that card. Those players put the exiled cards onto the battlefield, then shuffle their libraries.
22  
23      @Test
24      public void test_CastFromHand() {
25          addCard(Zone.HAND, playerA, "Indomitable Creativity", 1); // {X}{R}{R}{R}
26          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
27          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
28  
29          checkPermanentCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
30  
31          // cast with X=1 and exile
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Indomitable Creativity");
33          setChoice(playerA, "X=1");
34          addTarget(playerA, "Balduvian Bears");
35  
36          checkPermanentCount("after", 1, PhaseStep.BEGIN_COMBAT, playerA, "Balduvian Bears", 0);
37  
38          setStopAt(1, PhaseStep.END_TURN);
39          setStrictChooseMode(true);
40          execute();
41      }
42  
43      @Test
44      public void test_CastFromCommand() {
45          addCard(Zone.COMMAND, playerA, "Indomitable Creativity", 1); // {X}{R}{R}{R}
46          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
47          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
48  
49          checkPermanentCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
50  
51          // cast with X=1 and exile
52          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Indomitable Creativity");
53          setChoice(playerA, "X=1");
54          addTarget(playerA, "Balduvian Bears");
55          setChoice(playerA, true); // return spell as commander
56  
57          checkPermanentCount("after", 1, PhaseStep.BEGIN_COMBAT, playerA, "Balduvian Bears", 0);
58  
59          setStopAt(1, PhaseStep.END_TURN);
60          setStrictChooseMode(true);
61          execute();
62      }

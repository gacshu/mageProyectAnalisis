19      /**
20       * Opalescence plus Enchanted Evening are still not wiping any lands.
21       */
22      @Test
23      public void testLandsAreDestroyed() {
24          // Each other non-Aura enchantment is a creature in addition to its other types 
25          // and has base power and base toughness each equal to its converted mana cost.
26          addCard(Zone.HAND, playerA, "Opalescence", 1); // {2}{W}{W}
27  
28          addCard(Zone.BATTLEFIELD, playerA, "Plains", 9);
29          addCard(Zone.BATTLEFIELD, playerA, "War Horn", 1);
30  
31          // All permanents are enchantments in addition to their other types.
32          addCard(Zone.HAND, playerA, "Enchanted Evening"); //  {3}{W/U}{W/U}
33  
34          addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Opalescence",true);
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Enchanted Evening");
38          
39          setStopAt(1, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          assertPowerToughness(playerA, "Enchanted Evening", 5, 5);
43          assertPowerToughness(playerA, "War Horn", 3, 3);
44  
45          assertPermanentCount(playerA, "Island", 0);
46          assertPermanentCount(playerB, "Plains", 0);
47      }

49      /**
50       * Opalescence is dependent on Enchanted Evening, so it will be applied
51       * after it regardless of timestamp.
52       *
53       * Tokens can also have mana costs, and as a consequence of that, converted
54       * mana costs. A token created with Rite of Replication would have the mana
55       * cost of the creature it targeted. Most tokens do not have mana costs
56       * though.
57       *
58       * Tokens with no mana costs would be 0/0, as you said, and would indeed be
59       * put into owner's graveyard next time State Based Actionas are performed.
60       * Tokens with mana costs would naturally have whatever power and toughness
61       * their CMC indicated.
62       */
63      @Test
64      public void testTokensAreDestroyed() {
65          // Each other non-Aura enchantment is a creature in addition to its other types and has base power and base toughness each equal to its converted mana cost.
66          addCard(Zone.BATTLEFIELD, playerA, "Opalescence", 1);
67          addCard(Zone.BATTLEFIELD, playerA, "Island", 9);
68          // Kicker {5}
69          // Create a tokenthat's a copy of target creature onto the battlefield. If Rite of Replication was kicked, put five of those tokens onto the battlefield instead.
70          addCard(Zone.HAND, playerA, "Rite of Replication", 1); // This token can have a cmc
71          // All permanents are enchantments in addition to their other types.
72          addCard(Zone.HAND, playerA, "Enchanted Evening"); //  {3}{W/U}{W/U}
73  
74          addCard(Zone.BATTLEFIELD, playerB, "Cobblebrute", 1); // 5/2  cmc = 4
75          addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
76          // Put two 1/1 white Soldier creature tokens onto the battlefield.
77          addCard(Zone.HAND, playerB, "Raise the Alarm"); //  Instant {1}{W}
78  
79          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Rite of Replication", "Cobblebrute");
80          setChoice(playerA, false); // no kicker
81          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Raise the Alarm");
82          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Enchanted Evening");
83          setStopAt(1, PhaseStep.END_TURN);
84          execute();
85  
86          assertGraveyardCount(playerA, "Rite of Replication", 1);
87          assertGraveyardCount(playerB, "Raise the Alarm", 1);
88  
89          assertPowerToughness(playerA, "Enchanted Evening", 5, 5);
90  
91          assertPowerToughness(playerB, "Cobblebrute", 4, 4);
92          assertPowerToughness(playerA, "Cobblebrute", 4, 4);
93  
94          assertPermanentCount(playerB, "Soldier Token", 0);
95          assertPermanentCount(playerA, "Island", 0);
96          assertPermanentCount(playerB, "Plains", 0);
97      }

99      @Test
100     public void testYixlidJailerAndNecroticOozeBasic() {
101         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
102         // As long as Necrotic Ooze is on the battlefield, it has all activated abilities of all creature cards in all graveyards
103         addCard(Zone.BATTLEFIELD, playerA, "Necrotic Ooze", 1);
104         // {2}{1},{T}: Tap target creature.
105         addCard(Zone.GRAVEYARD, playerA, "Akroan Jailer", 1);
106         // {T}: Target attacking creature gets +1/+1 until end of turn.
107         addCard(Zone.GRAVEYARD, playerB, "Anointer of Champions", 1);
108 
109         setStopAt(1, PhaseStep.BEGIN_COMBAT);
110         execute();
111 
112         Permanent necroticOoze = getPermanent("Necrotic Ooze", playerA);
113         int numberOfActivatedAbilities = 0;
114         for (Ability ability : necroticOoze.getAbilities(currentGame)) {
115             if (ability.getAbilityType() == AbilityType.ACTIVATED) {
116                 numberOfActivatedAbilities++;
117             }
118         }
119         Assert.assertEquals("Two abilities for Necrotic Ooze", 2, numberOfActivatedAbilities);
120     }

122     @Test
123     public void testYixlidJailerAndNecroticOozeLooseAll() {
124         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
125         // As long as Necrotic Ooze is on the battlefield, it has all activated abilities of all creature cards in all graveyards
126         addCard(Zone.BATTLEFIELD, playerA, "Necrotic Ooze", 1);
127         // {2}{1},{T}: Tap target creature.
128         addCard(Zone.GRAVEYARD, playerA, "Akroan Jailer", 1);
129         // {T}: Target attacking creature gets +1/+1 until end of turn.
130         addCard(Zone.GRAVEYARD, playerB, "Anointer of Champions", 1);
131 
132         // Cards in graveyards lose all abilities.
133         addCard(Zone.HAND, playerA, "Yixlid Jailer", 1); // Creature - {1}{B}
134 
135         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Yixlid Jailer");
136         
137         setStrictChooseMode(true);
138         setStopAt(1, PhaseStep.BEGIN_COMBAT);
139         execute();
140 
141         assertPermanentCount(playerA, "Yixlid Jailer", 1);
142 
143         Permanent necroticOoze = getPermanent("Necrotic Ooze", playerA);
144         int numberOfActivatedAbilities = 0;
145         for (Ability ability : necroticOoze.getAbilities(currentGame)) {
146             if (ability.getAbilityType() == AbilityType.ACTIVATED) {
147                 numberOfActivatedAbilities++;
148             }
149         }
150         Assert.assertEquals("All abilities from cards in graveyard are removed - so no abilities for Necrotic Ooze", 0, numberOfActivatedAbilities);
151     }

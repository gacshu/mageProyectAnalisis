1   package org.mage.test.cards.continuous;
2   
3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;
7   
8   /**
9    * @author JayDi85
10   */
11  public class GoblinGoliathTest extends CardTestPlayerBase {
12  
13      @Test
14      public void test_DoubleDamage() {
15          // test double damage on creature (no), on yourself (no) and on opponent (yes) by bold damage
16  
17          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 20);
18          // {3}{R}, {T}: If a source you control would deal damage to an opponent this turn, it deals double that damage to that player instead.
19          addCard(Zone.BATTLEFIELD, playerA, "Goblin Goliath");
20          addCard(Zone.HAND, playerA, "Lightning Bolt", 4); // 3 damage
21          addCard(Zone.BATTLEFIELD, playerB, "Barktooth Warbeard"); // 6/5
22  
23          // normal damage without ability
24          castSpell(1, PhaseStep.UPKEEP, playerA, "Lightning Bolt", playerB);
25          checkLife("normal damage to opponent", 1, PhaseStep.PRECOMBAT_MAIN, playerB, 20 - 3);
26  
27          // activate double damage
28          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{3}{R}");
29          // cast bolt to player (2x damage)
30          castSpell(1, PhaseStep.BEGIN_COMBAT, playerA, "Lightning Bolt", playerB);
31          checkLife("double damage to opponent", 1, PhaseStep.END_COMBAT, playerB, 20 - 3 - 3 * 2);
32          // cast bolt to creature (1x damage)
33          castSpell(1, PhaseStep.BEGIN_COMBAT, playerA, "Lightning Bolt", "Barktooth Warbeard");
34          checkPermanentCount("normal damage to creature", 1, PhaseStep.END_COMBAT, playerB, "Barktooth Warbeard", 1);
35          // cast bolt to yourself (1x damage)
36          castSpell(1, PhaseStep.BEGIN_COMBAT, playerA, "Lightning Bolt", playerA);
37          checkLife("normal damage to yourself", 1, PhaseStep.END_COMBAT, playerA, 20 - 3);
38  
39          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
40          execute();
41  
42          assertGraveyardCount(playerA, "Lightning Bolt", 4);
43          assertLife(playerA, 20 - 3);
44          assertLife(playerB, 20 - 3 - 3 * 2);
45          assertGraveyardCount(playerB, "Barktooth Warbeard", 0);
46      }
47  }

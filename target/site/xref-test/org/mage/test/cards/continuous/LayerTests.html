19      /**
20       * Conspiracy -> Opalescence -> Enchanted Evening
21       *      Conspiracy is dependent on Opalescence
22       *      Opalescence is dependent on Enchanted Evening
23       *
24       * So, the effects should be applied as follows:
25       *      Enchanted Evening -> Opalescence -> Conspiracy
26       */
27      @Test
28      public void testMultipleLayeredDependency() {
29  
30  
31          addCard(Zone.HAND, playerA, "Conspiracy"); // creatures get chosen subtype
32          addCard(Zone.HAND, playerA, "Opalescence"); // enchantments become creatures P/T equal to CMC
33          addCard(Zone.HAND, playerA, "Enchanted Evening"); // all permanents become enchantments
34  
35          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
36          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
37          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
38          addCard(Zone.BATTLEFIELD, playerA, "Glorious Anthem", 1); // keep lands alive // all creatures +1/+1
39  
40          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Conspiracy", true);
41          setChoice(playerA, "Advisor");
42          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Opalescence", true);
43          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Enchanted Evening");
44  
45          setStrictChooseMode(true);
46          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
47          execute();
48  
49          assertType("Swamp", CardType.LAND, SubType.ADVISOR); // Subtype changed with Conspiracy
50          assertPowerToughness(playerA, "Swamp", 1, 1); // boosted with Glorious Anthem
51          assertType("Enchanted Evening", CardType.ENCHANTMENT, SubType.ADVISOR); // Subtype changed with Conspiracy
52          assertPowerToughness(playerA, "Enchanted Evening", 6, 6); // boosted with Glorious Anthem
53  
54      }

56      /**
57       * Reported bug:
58       *      This came up in a recent EDH game and we had no idea how to progress.
59       *      Player A cast a Humility, then a March of the Machines, and finally a Mycosynth Lattice.
60       *      Does the game get stuck in an endless loop of each card gaining and losing its respective creature-ness and abilities?
61       *      Answer: No, they all die
62       */
63      @Test
64      public void testMycosynthLatticeAndMarchOfTheMachinesAndHumility() {
65          addCard(Zone.HAND, playerA, "Mycosynth Lattice"); // all permanents are artifacts
66          addCard(Zone.HAND, playerA, "March of the Machines"); // artifacts become creatures
67          addCard(Zone.HAND, playerA, "Humility"); // all creatures lose abilities and P/T is 1/1
68  
69          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 10);
70          addCard(Zone.BATTLEFIELD, playerA, "Plains", 10);
71          addCard(Zone.BATTLEFIELD, playerA, "Island", 10);
72  
73          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Humility", true);
74          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "March of the Machines", true);
75          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Mycosynth Lattice");
76  
77          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
78          execute();
79  
80          // everything dies
81          assertPermanentCount(playerA, "Humility", 0);
82          assertPermanentCount(playerA, "March of the Machines", 0);
83          assertPermanentCount(playerA, "Mycosynth Lattice", 0);
84          assertPermanentCount(playerA, "Island", 0);
85      }

87      @Test
88      public void testBloodMoon_UrborgTombOfYawgmothInteraction() {
89          // Blood Moon : Nonbasic lands are Mountains.
90          // Urborg, Tomb of Yawgmoth : Each land is a Swamp in addition to its other types.
91          // Expected behavior:  Urborg loses all abilities and becomes a Mountain.  The Plains does not have subtype Swamp due to this effect.
92          addCard(Zone.BATTLEFIELD, playerA, "Blood Moon");
93          addCard(Zone.BATTLEFIELD, playerA, "Urborg, Tomb of Yawgmoth", 1);  // non-basic land
94          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
95  
96          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
97          execute();
98  
99          assertType("Urborg, Tomb of Yawgmoth", CardType.LAND, SubType.MOUNTAIN);  // Urborg is a Mountain now
100         assertPermanentCount(playerA, "Swamp", 0);  // no Swamp subtypes on the battlefield
101         assertPermanentCount(playerA, "Plains", 1); // the Plains is not affected by the Urborg
102         assertType("Plains", CardType.LAND, SubType.PLAINS);
103

143         // Animate Land: target land is a 3/3 until end of turn and is still a land.
144         // Ovinize: target creature becomes 0/1 and loses all abilities until end of turn.
145         // Urborg, Tomb of Yawgmoth : Each land is a Swamp in addition to its other types.

148         addCard(Zone.HAND, playerA, "Ovinize", 1);

151         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
152 
153         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Animate Land", "Urborg, Tomb of Yawgmoth");
154         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Ovinize", "Urborg, Tomb of Yawgmoth");
155         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
156         execute();

193     @Test
194     public void testExampleFromReddit2021() {
195         // Life and Limb, Humility, and Yavimaya, Cradle of Growth
196         // Result: all lands (including Yavimaya) will be 1/1 green Forest 
197         // lands and Saproling creatures in addition to their other types, 
198         // and have no abilities.
199 
200         /*
201         All Forests and all Saprolings are 1/1 green Saproling creatures 
202         and Forest lands in addition to their other types. (They're affected by summoning sickness.)
203         */
204         addCard(Zone.BATTLEFIELD, playerA, "Life and Limb", 1);
205         
206         /*
207         All creatures lose all abilities and have base power and toughness 1/1.
208         */
209         addCard(Zone.BATTLEFIELD, playerA, "Humility", 1);
210         
211         /*
212         Each land is a Forest in addition to its other land types.
213         */
214         addCard(Zone.BATTLEFIELD, playerA, "Yavimaya, Cradle of Growth", 1);
215         
216         // added some lands to the battlefield
217 
218         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
219         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
220         addCard(Zone.BATTLEFIELD, playerB, "Island", 1);
221 
222         setStrictChooseMode(true);
223         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
224         execute();
225 
226         // all lands are forests in addition to other types
227         assertType("Plains", CardType.CREATURE, SubType.FOREST);
228         assertType("Swamp", CardType.CREATURE, SubType.FOREST);
229         assertType("Island", CardType.CREATURE, SubType.FOREST);

234         assertPowerToughness(playerA, "Swamp", 1, 1);
235         assertPowerToughness(playerB, "Island", 1, 1);
236         assertPowerToughness(playerA, "Yavimaya, Cradle of Growth", 1, 1);
237 
238         // all lands are green
239         assertColor(playerA, "Plains", ObjectColor.GREEN, true);
240         assertColor(playerA, "Swamp", ObjectColor.GREEN, true);
241         assertColor(playerB, "Island", ObjectColor.GREEN, true);

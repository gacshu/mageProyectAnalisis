19      /**
20       * Daxos the Returned 's spirit tokens are not counted as enchantment tokens
21       * and do not count towards experience counters like they should.
22       */
23      @Test
24      public void testCounterAddAndTokenStates() {
25          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8);
26          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
27          // Whenever you cast an enchantment spell, you get an experience counter.
28          // {1}{W}{B}: Create a white and black Spirit enchantment creature token onto the battlefield. It has
29          // "This creature's power and toughness are each equal to the number of experience counters you have."
30          addCard(Zone.BATTLEFIELD, playerA, "Daxos the Returned");
31  
32          // Whenever an opponent draws a card, Underworld Dreams deals 1 damage to that player.
33          addCard(Zone.HAND, playerA, "Underworld Dreams", 2); // {B}{B}{B}
34  
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Underworld Dreams", true);
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Underworld Dreams", true);
37          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{1}{W}{B}");
38          setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
39          execute();
40  
41          assertLife(playerA, 20);
42          assertLife(playerB, 18);
43  
44          assertPermanentCount(playerA, "Underworld Dreams", 2);
45          assertCounterCount(playerA, CounterType.EXPERIENCE, 2);
46          assertPowerToughness(playerA, "Spirit Token", 2, 2, Filter.ComparisonScope.All);
47          assertType("Spirit Token", CardType.ENCHANTMENT, SubType.SPIRIT);
48  
49      }

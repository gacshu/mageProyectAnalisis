15      /**
16       * Additionally, if you play a Mimic naming, for example, "Dwarf", then if
17       * you play a second Mimic also naming "Dwarf", the second Mimic won't enter
18       * with a counter. Going by 616.2 in the Comp Rules, the second one should
19       * be getting a counter as long as you name the same type as the first one.
20       *
21       * 611.3c Continuous effects that modify characteristics of permanents do so
22       * simultaneously with the permanent entering the battlefield. They don't
23       * wait until the permanent is on the battlefield and then change it.
24       * Because such effects apply as the permanent enters the battlefield, they
25       * are applied before determining whether the permanent will cause an
26       * ability to trigger when it enters the battlefield. #
27       *
28       * Example: A permanent with the static ability “All white creatures get
29       * +1/+1” is on the battlefield. A creature spell that would normally create
30       * a 1/1 white creature instead creates a 2/2 white creature. The creature
31       * doesn't enter the battlefield as 1/1 and then change to 2/2.
32       *
33       */
34      @Test
35      public void testMetallicMimic() {
36          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
37  
38          // As Metallic Mimic enters the battlefield, choose a creature type.
39          // Metallic Mimic is the chosen type in addition to its other types.
40          // Each other creature you control of the chosen type enters the battlefield with an additional +1/+1 counter on it.
41          addCard(Zone.HAND, playerA, "Metallic Mimic", 2); // Creature {2}
42  
43          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Metallic Mimic");
44          setChoice(playerA, "Dwarf");
45          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Metallic Mimic");
46          setChoice(playerA, "Dwarf");
47  
48          setStopAt(1, PhaseStep.END_TURN);
49          execute();
50  
51          assertPermanentCount(playerA, "Metallic Mimic", 2);
52          assertPowerToughness(playerA, "Metallic Mimic", 2, 1);
53          assertPowerToughness(playerA, "Metallic Mimic", 3, 2);
54      }

67          addCard(Zone.BATTLEFIELD, playerA, "Bramblewood Paragon", 1);
68  
69          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Metallic Mimic");
70          setChoice(playerA, "Warrior");

72          setStopAt(1, PhaseStep.BEGIN_COMBAT);
73          execute();
74  
75          assertPermanentCount(playerA, "Metallic Mimic", 1);
76          assertPowerToughness(playerA, "Metallic Mimic", 3, 2);
77      }

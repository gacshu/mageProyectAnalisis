80       * Currently, a land hit by Aquitect's Will loses all of its other
81       * abilities, making it a cheap Spreading Seas. It should function like
82       * Urborg, Tomb of Yawgmoth, not Spreading Seas or Blood Moon.

84      @Test
85      public void testLandDoesNotLooseOtherAbilities() {
86          // Put a flood counter on target land.
87          // That land is an Island in addition to its other types for as long as it has a flood counter on it.
88          // If you control a Merfolk, draw a card.
89          addCard(Zone.HAND, playerA, "Aquitect's Will");// Tribal Sorcery{U}
90          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
91          // Forbidding Watchtower enters the battlefield tapped.
92          // {T}: Add {W}.
93          // {1}{W}: Forbidding Watchtower becomes a 1/5 white Soldier creature until end of turn. It's still a land.
94          addCard(Zone.BATTLEFIELD, playerB, "Forbidding Watchtower", 1);
95          addCard(Zone.BATTLEFIELD, playerB, "Plains", 2);
96  
97          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Aquitect's Will", "Forbidding Watchtower");
98  
99          activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{1}{W}:");
100 
101         setStrictChooseMode(true);
102         setStopAt(2, PhaseStep.BEGIN_COMBAT);
103         execute();
104 
105         assertGraveyardCount(playerA, "Aquitect's Will", 1);
106 
107         assertPermanentCount(playerB, "Forbidding Watchtower", 1);
108         assertCounterCount("Forbidding Watchtower", CounterType.FLOOD, 1);
109         assertType("Forbidding Watchtower", CardType.LAND, SubType.ISLAND);
110         assertPowerToughness(playerB, "Forbidding Watchtower", 1, 5);

178        I had Kormus Bell (all swamps are 1/1s) and Urborg, Tomb of Yawgmoth (all lands are swamps) and only basic lands (swamps) otherwise.
179        Opponent had Quicksilver Fountain (put a flood counter on non-island land).
180        In terms of time-stamp order, Urborg was down first, then Kormus Bell, then Quicksilver.
181        When I put a flood counter on a basic swamp, it would become a 0/0 instead of a 1/1 and die.

184     @Test
185     public void testCormusBellAfterUrborg() {
186         // Land - Legendary
187         // Each land is a Swamp in addition to its other land types.
188         addCard(Zone.HAND, playerA, urborgtoy);
189 
190         // All Swamps are 1/1 black creatures that are still lands.
191         addCard(Zone.HAND, playerA, "Kormus Bell"); // Artifact {4}
192 
193         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
194 
195         // At the beginning of each player's upkeep, that player puts a flood counter on target non-Island land they control of their choice.
196         //  That land is an Island for as long as it has a flood counter on it.
197         // At the beginning of each end step, if all lands on the battlefield are Islands, remove all flood counters from them.
198         addCard(Zone.HAND, playerB, "Quicksilver Fountain", 1); // Artifact {3}
199         addCard(Zone.BATTLEFIELD, playerB, "Forest", 3);
200 
201         playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, urborgtoy);
202         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Kormus Bell");
203 
204         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Quicksilver Fountain");
205 
206         addTarget(playerA, "Mountain");
207 
208         setStrictChooseMode(true);
209         setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
210         execute();
211 
212         assertPermanentCount(playerA, urborgtoy, 1);
213         assertPermanentCount(playerA, "Kormus Bell", 1);
214         assertPermanentCount(playerB, "Quicksilver Fountain", 1);
215 
216         assertGraveyardCount(playerA, "Mountain", 0);
217 
218         int islandTypes = 0;
219         int swampTypes = 0;
220         int mountainTypes = 0;
221         int creatures = 0;
222         for (Permanent permanent : currentGame.getBattlefield().getAllActivePermanents(StaticFilters.FILTER_LAND, playerA.getId(), currentGame)) {
223             if (permanent.hasSubtype(SubType.ISLAND, currentGame)) {
224                 islandTypes++;
225             }
226             if (permanent.hasSubtype(SubType.MOUNTAIN, currentGame)) {
227                 mountainTypes++;
228             }
229             if (permanent.hasSubtype(SubType.SWAMP, currentGame)) {
230                 swampTypes++;
231             }
232             if (permanent.isCreature(currentGame)) {
233                 creatures++;
234             }
235         }
236         Assert.assertTrue("1 land has to be of island type", islandTypes == 1);
237         Assert.assertTrue("3 lands have to be of mountain type", mountainTypes == 3);
238         Assert.assertTrue("4 lands have to be of swamp type", swampTypes == 4);
239         Assert.assertTrue("4 lands have to be creatures but there are " + creatures, creatures == 4);
240     }

242     @Test
243     public void testBloodSunWithUrborgtoyAndStormtideLeviathanMan() {
244 
245         addCard(Zone.BATTLEFIELD, playerA, urborgtoy);  // all lands are swamps in addition to their other types
246         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
247         addCard(Zone.BATTLEFIELD, playerA, "Blood Sun");  // all lands lose all abilities except for mana-producing
248         addCard(Zone.BATTLEFIELD, playerA, "Stormtide Leviathan"); // all lands are islands in addition to their other types
249         addCard(Zone.BATTLEFIELD, playerA, "Darksteel Citadel");  // land has indestructible ability
250 
251         setStrictChooseMode(true);
252         setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
253         execute();
254 
255         Permanent darksteel = getPermanent("Darksteel Citadel", playerA.getId());
256         Assert.assertNotNull(darksteel);
257         Assert.assertFalse(darksteel.getAbilities().contains(IndestructibleAbility.getInstance()));  // The ability is removed
258 
259         /*
260         If a continuous effect has started applying in an earlier layer, it will continue to apply in 
261         later layers even if the ability that created that effect has been removed.
262         Urborg ability is applied in the 4th layer.  The Blood Sun works in the 6th.  So the effect still applies to the lands.
263          */
264         assertType(urborgtoy, CardType.LAND, SubType.SWAMP);
265         assertType("Mountain", CardType.LAND, SubType.SWAMP);
266         assertType(urborgtoy, CardType.LAND, SubType.ISLAND);
267         assertType("Mountain", CardType.LAND, SubType.ISLAND);
268     }

29      @Test
30      /*
31       * Vizier was not working with casting creatures off the top of the deck at instant speed
32       * even though they had flash via continues effects (like the one Yeva provides).
33       */
34      public void testCanCastFlashEffectCreaturesOffLibrary() {
35          addCard(Zone.BATTLEFIELD, playerA, "Vizier of the Menagerie");
36           // You may cast green creature cards as though they had flash.
37          addCard(Zone.BATTLEFIELD, playerA, "Yeva, Nature's Herald");
38          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
39          removeAllCardsFromLibrary(playerA);
40          addCard(Zone.LIBRARY, playerA, "Llanowar Elves");
41  
42          castSpell(1, PhaseStep.BEGIN_COMBAT, playerA, "Llanowar Elves");
43  
44          setStopAt(1, PhaseStep.DECLARE_ATTACKERS);
45          execute();
46  
47          assertPermanentCount(playerA, "Llanowar Elves", 1);
48      }

50      @Test
51      /**
52       * The negative case of the above unit test.
53       */
54      public void testCantCastNonFlashEffectedCreaturesOffLibrary() {
55          addCard(Zone.BATTLEFIELD, playerA, "Vizier of the Menagerie");
56          // You may cast green creature cards as though they had flash.
57          addCard(Zone.BATTLEFIELD, playerA, "Yeva, Nature's Herald");
58          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
59          removeAllCardsFromLibrary(playerA);
60          // A non-green creature card should not have Yeva's effect
61          addCard(Zone.LIBRARY, playerA, "Cabal Therapist");
62  
63          checkPlayableAbility("Can't flash Cabal Therapist", 1, PhaseStep.BEGIN_COMBAT, playerA, "Cast Cabal", false);

105     @Test
106     /**
107      * Should be able to cast creatures cards with any mana.
108      */
109     public void testCanCastCreaturesWithAnyMana() {
110         addCard(Zone.BATTLEFIELD, playerA, "Vizier of the Menagerie");
111         addCard(Zone.BATTLEFIELD, playerA, "Forest");
112         //Creature that costs 1 Mountain to cast
113         addCard(Zone.HAND, playerA, "Bloodcrazed Goblin");
114 
115         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Bloodcrazed Goblin");
116 
117         setStopAt(1, PhaseStep.BEGIN_COMBAT);
118         execute();
119 
120         // Should have been able to cast using the one Forest because Vizier's effect
121         assertPermanentCount(playerA, "Bloodcrazed Goblin", 1);
122     }

8   /**
9    * Tests that if a creature was controlled by opponent and died and was cast
10   * again by owner, that continuous effects of this creature work with controller

13   * @author LevelX2

15  public class GainControlDiedCastAgainTest extends CardTestPlayerBase {

17      @Test
18      public void testBoostEffectsWorksForControllerOfElesh() {
19  
20          addCard(Zone.BATTLEFIELD, playerB, "Plains", 4);
21          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 3);
22          /**
23           * Volrath's Stronghold Legendary Land {t}: Add {C}.
24           * {1}{B}, {tap}: Put target creature card from your graveyard on top of
25           * your library.
26           */
27          addCard(Zone.BATTLEFIELD, playerB, "Volrath's Stronghold");
28          /**
29           * Elesh Norn, Grand Cenobite {5}{W}{W} Legendary Creature â€” Praetor 4/7
30           * Vigilance Other creatures you control get +2/+2. Creatures your
31           * opponents control get -2/-2.
32           */
33          addCard(Zone.BATTLEFIELD, playerB, "Elesh Norn, Grand Cenobite");
34          addCard(Zone.BATTLEFIELD, playerB, "Kalonian Tusker"); // simple 3/3
35  
36          /**
37           * Keiga, the Tide Star Legendary Dragon Spirit When Keiga, the Tide
38           * Star dies, gain control of target creature
39           */
40          addCard(Zone.BATTLEFIELD, playerA, "Keiga, the Tide Star");
41          addCard(Zone.BATTLEFIELD, playerA, "Kalonian Tusker");
42  
43          attack(2, playerB, "Elesh Norn, Grand Cenobite");
44          block(2, playerA, "Keiga, the Tide Star", "Elesh Norn, Grand Cenobite");
45          addTarget(playerA, "Elesh Norn, Grand Cenobite");
46  
47          setStrictChooseMode(true);
48          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
49          execute();
50  
51          assertLife(playerA, 20);
52          assertLife(playerB, 20);
53  
54          assertPermanentCount(playerA, "Elesh Norn, Grand Cenobite", 1);
55  
56          assertPowerToughness(playerA, "Kalonian Tusker", 5, 5);
57          assertPowerToughness(playerB, "Kalonian Tusker", 1, 1);
58      }

60      @Test
61      public void testBoostEffectsWorkForController() {
62          addCard(Zone.BATTLEFIELD, playerB, "Plains", 6);
63          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 6);
64          /**
65           * Akroma's Vengeance {4}{W}{W} Sorcery Destroy all artifacts,
66           * creatures, and enchantments.
67           */
68          addCard(Zone.HAND, playerB, "Akroma's Vengeance");
69          /**
70           * Volrath's Stronghold Legendary Land {t}: Add {C}.
71           * {1}{B}, {tap}: Put target creature card from your graveyard on top of
72           * your library.
73           */
74          addCard(Zone.BATTLEFIELD, playerB, "Volrath's Stronghold");
75          /**
76           * Elesh Norn, Grand Cenobite {5}{W}{W} Legendary Creature â€” Praetor
77           * Vigilance Other creatures you control get +2/+2. Creatures your
78           * opponents control get -2/-2.
79           */
80          addCard(Zone.HAND, playerB, "Elesh Norn, Grand Cenobite");
81  
82          addCard(Zone.BATTLEFIELD, playerB, "Darksteel Gargoyle"); // indestructible 3/3
83  
84          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
85          /**
86           * Keiga, the Tide Star {5}{U} Legendary Dragon Spirit 5/5 When Keiga,
87           * the Tide Star dies, gain control of target creature
88           */
89          addCard(Zone.HAND, playerA, "Keiga, the Tide Star");
90          addCard(Zone.BATTLEFIELD, playerA, "Darksteel Gargoyle");
91  
92          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Elesh Norn, Grand Cenobite");
93  
94          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Keiga, the Tide Star");
95  
96          attack(4, playerB, "Elesh Norn, Grand Cenobite");
97          block(4, playerA, "Keiga, the Tide Star", "Elesh Norn, Grand Cenobite");
98          addTarget(playerA, "Elesh Norn, Grand Cenobite");
99  
100         // Destroy all creatures, enchantments and artifacts
101         castSpell(4, PhaseStep.POSTCOMBAT_MAIN, playerB, "Akroma's Vengeance");
102 
103         // Put Elesh Norn back on library
104         activateAbility(5, PhaseStep.END_TURN, playerB, "{1}{B}, {T}: Put target creature card", "Elesh Norn, Grand Cenobite");
105 
106         castSpell(6, PhaseStep.PRECOMBAT_MAIN, playerB, "Elesh Norn, Grand Cenobite");
107         setStopAt(6, PhaseStep.BEGIN_COMBAT);
108         execute();
109 
110         assertLife(playerA, 20);
111         assertLife(playerB, 20);
112 
113         assertPermanentCount(playerA, "Keiga, the Tide Star", 0);
114         assertGraveyardCount(playerA, "Keiga, the Tide Star", 1);
115 
116         assertPermanentCount(playerB, "Elesh Norn, Grand Cenobite", 1);
117         assertGraveyardCount(playerB, "Elesh Norn, Grand Cenobite", 0);
118         assertGraveyardCount(playerB, "Akroma's Vengeance", 1);
119 
120         assertPermanentCount(playerA, "Darksteel Gargoyle", 1);
121         assertPermanentCount(playerB, "Darksteel Gargoyle", 1);
122 
123         assertPowerToughness(playerA, "Darksteel Gargoyle", 1, 1);
124         assertPowerToughness(playerB, "Darksteel Gargoyle", 5, 5);
125     }

3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    * Tests the effect:
10   *   - Target opponent gains control of {this}
11   *
12   * @author noxx
13   */
14  public class TargetOpponentGainsControlTest extends CardTestPlayerBase {
15  
16      @Test
17      public void testPermanentControlEffect() {
18          addCard(Zone.HAND, playerA, "Lightning Bolt", 3);
19          addCard(Zone.BATTLEFIELD, playerA, "Treacherous Pit-Dweller");
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
21  
22          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Treacherous Pit-Dweller");
23  
24          setStopAt(1, PhaseStep.BEGIN_COMBAT);
25          execute();
26  
27          // under opponent's control
28          assertPermanentCount(playerB, "Treacherous Pit-Dweller", 1);
29      }
30  
31      @Test
32      public void testChangeControlEffectFromTwoCards() {
33          addCard(Zone.HAND, playerA, "Lightning Bolt", 3);
34          // Enchant creature
35          // When enchanted creature dies, return that card to the battlefield under your control.
36          addCard(Zone.HAND, playerA, "Unhallowed Pact", 1); // {2}{B}
37          // Undying
38          // When Treacherous Pit-Dweller enters the battlefield from a graveyard, target opponent gains control of it.
39          addCard(Zone.BATTLEFIELD, playerA, "Treacherous Pit-Dweller"); // 4/3
40          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
41          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
42  
43          castSpell(1, PhaseStep.UPKEEP, playerA, "Lightning Bolt", "Treacherous Pit-Dweller"); // comes back with undying
44          
45          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Unhallowed Pact", "Treacherous Pit-Dweller"); 
46          
47          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Treacherous Pit-Dweller"); // Treacherous Pit-Dweller is now 5/4
48          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Treacherous Pit-Dweller");
49  
50          setStopAt(1, PhaseStep.END_TURN);
51          execute();
52  
53          // went to graveyard
54          assertGraveyardCount(playerA, "Unhallowed Pact", 1);
55          assertGraveyardCount(playerA, "Lightning Bolt", 3);
56  
57          // returned back
58          assertGraveyardCount(playerA, "Treacherous Pit-Dweller", 0);
59          assertPermanentCount(playerB, "Treacherous Pit-Dweller", 1); // opponent gets it because ETB of Dweller resolves always last
60      }
61  }

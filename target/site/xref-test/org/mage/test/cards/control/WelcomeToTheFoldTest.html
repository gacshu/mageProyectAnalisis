5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    *  2UU
10      Sorcery
11      Madness XUU (If you discard this card, discard it into exile. When you do, cast it for its madness cost or put it into your graveyard.)
12  
13      Gain control of target creature if its toughness is 2 or less. 
14      * If Welcome to the Fold's madness cost was paid, instead gain control of that creature if its toughness is X or less. 
15   * @author escplan9 (Derek Monturo - dmontur1 at gmail dot com)
16   */
17  public class WelcomeToTheFoldTest extends CardTestPlayerBase {
18      
19      /**
20       * Paid with regular cost, gain control of creatures with toughness <= 2
21       */
22      @Test
23      public void regularCostVariousCreatures() {
24                  
25          addCard(Zone.HAND, playerA, "Welcome to the Fold", 3);
26          addCard(Zone.BATTLEFIELD, playerA, "Island", 12);
27          
28          addCard(Zone.BATTLEFIELD, playerB, "Arashin Cleric", 1); // 1/3
29          addCard(Zone.BATTLEFIELD, playerB, "Jace, Vryn's Prodigy", 1); // 0/2
30          addCard(Zone.BATTLEFIELD, playerB, "Elite Vanguard", 1); // 2/1
31                  
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Welcome to the Fold", true);
33          addTarget(playerA, "Arashin Cleric"); // does not gain control
34          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Welcome to the Fold", true);
35          addTarget(playerA, "Jace, Vryn's Prodigy"); // gains control
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Welcome to the Fold");
37          addTarget(playerA, "Elite Vanguard"); // gains control
38          
39          setStopAt(1, PhaseStep.END_TURN);
40          execute();
41          
42          assertGraveyardCount(playerA, "Welcome to the Fold", 3);
43          assertPermanentCount(playerA, "Arashin Cleric", 0); // should not gain control
44          assertPermanentCount(playerA, "Jace, Vryn's Prodigy", 1); // should gain control
45          assertPermanentCount(playerA, "Elite Vanguard", 1);
46      }

23      public void testPermanentControlEffect() {

27          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);

31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Act of Treason", "Elite Vanguard");

34          setStopAt(2, PhaseStep.BEGIN_COMBAT);
35          execute();
36  
37          assertPermanentCount(playerA, "Elite Vanguard", 1);

40      @Test
41      public void testVillainousWealthExilesCourser() {
42          // Villainous Wealth {X}{B}{G}{U}
43          // Target opponent exiles the top X cards of their library. You may cast any number
44          // of nonland cards with converted mana cost X or less from among them without paying
45          // their mana costs.
46          addCard(Zone.HAND, playerA, "Villainous Wealth");
47          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
48          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
49          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
50  
51          // Courser of Kruphix {1}{G}{G}
52          // Play with the top card of your library revealed.
53          // You may play the top card of your library if it's a land card.
54          // Whenever a land enters the battlefield under your control, you gain 1 life.
55          addCard(Zone.LIBRARY, playerB, "Courser of Kruphix");
56          skipInitShuffling(); // to keep this card on top of library
57  
58          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Villainous Wealth", playerB);
59          setChoice(playerA, "X=3");
60          setChoice(playerA, "Yes"); // Courser of Kruphix is the only option, say Yes to cast for free
61  
62          setStopAt(1, PhaseStep.BEGIN_COMBAT);
63          execute();
64  
65          assertGraveyardCount(playerA, "Villainous Wealth", 1);
66          assertExileCount(playerB, 2);
67          assertExileCount("Courser of Kruphix", 0);
68          assertPermanentCount(playerA, "Courser of Kruphix", 1);
69          Assert.assertTrue("player A should play with top card revealed", playerA.isTopCardRevealed());
70          Assert.assertFalse("player B should play NOT with top card revealed", playerB.isTopCardRevealed());
71      }

114      * My opponent cast Villainous Wealth and took control of my Sylvan Library.
115      * On their next turn, when Sylvan Library's trigger resolved, they kept the two
116      * extra cards without paying life.
117      */
118     @Test
119     public void testVillainousWealthExilesSylvanLibrary() {
120         // Villainous Wealth {X}{B}{G}{U}
121         // Target opponent exiles the top X cards of their library. You may cast any number
122         // of nonland cards with converted mana cost X or less from among them without paying
123         // their mana costs.
124         addCard(Zone.HAND, playerA, "Villainous Wealth");
125         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
126         addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
127         addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
128 
129         // At the beginning of your draw step, you may draw two additional cards.
130         // If you do, choose two cards in your hand drawn this turn.
131         // For each of those cards, pay 4 life or put the card on top of your library.
132         addCard(Zone.LIBRARY, playerB, "Sylvan Library");
133         skipInitShuffling(); // to keep this card on top of library
134 
135         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Villainous Wealth", playerB);
136         setChoice(playerA, "X=3");
137         setChoice(playerA, "Yes"); // Sylvan Library is the only option, say Yes to cast for free
138 
139         setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
140         execute();
141 
142         assertGraveyardCount(playerA, "Villainous Wealth", 1);
143         assertExileCount(playerB, 2);
144         assertExileCount("Sylvan Library", 0);
145         assertPermanentCount(playerA, "Sylvan Library", 1);
146 
147         assertHandCount(playerB, 1);
148         assertHandCount(playerA, 3);
149         assertLife(playerA, 12);
150         assertLife(playerB, 20);
151     }

153     /**
154      * I cast a Villainous Wealth in Vintage Cube, and when it came time to cast
155      * my opponent's cards (Mox Sapphire, Mox Emerald, Brainstorm, Snapcaster
156      * Mage, Fact or Fiction and a Quicken), it rolled back to before I had cast
157      * my spell after Quicken resolved.
158      * I have the error, but the forums won't let me post them.
159      * I did find it was replicatable whenever you try to cast Quicken off a Villainous Wealth.
160      */
161     @Test
162     public void testVillainousWealthAndQuicken() {
163         // Villainous Wealth {X}{B}{G}{U}
164         // Target opponent exiles the top X cards of their library. You may cast any number
165         // of nonland cards with converted mana cost X or less from among them without paying
166         // their mana costs.
167         addCard(Zone.HAND, playerA, "Villainous Wealth"); // {X}{B}{G}{U}
168         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
169         addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
170         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
171 
172         // At the beginning of your draw step, you may draw two additional cards.
173         // If you do, choose two cards in your hand drawn this turn.
174         // For each of those cards, pay 4 life or put the card on top of your library.
175         addCard(Zone.LIBRARY, playerB, "Mox Emerald");
176         // The next sorcery card you cast this turn can be cast as though it had flash.
177         // Draw a card.
178         addCard(Zone.LIBRARY, playerB, "Quicken"); // Instant - {U}
179         addCard(Zone.LIBRARY, playerB, "Mox Sapphire");
180         skipInitShuffling(); // to keep this card on top of library
181 
182         setStrictChooseMode(true);
183 
184         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Villainous Wealth", playerB);
185         setChoice(playerA, "X=3");
186 
187         setChoice(playerA, "Mox Emerald");
188         setChoice(playerA, "Yes");
189 
190         setChoice(playerA, "Mox Sapphire");
191         setChoice(playerA, "Yes");
192 
193         // Quicken is auto-chosen since it's the last of the 3 cards. Only need to say Yes to casting for free.
194         setChoice(playerA, "Yes");
195 
196         setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
197         execute();
198 
199         assertGraveyardCount(playerA, "Villainous Wealth", 1);
200         assertExileCount(playerB, 0);
201         assertPermanentCount(playerA, "Mox Emerald", 1);
202         assertPermanentCount(playerA, "Mox Sapphire", 1);
203         assertGraveyardCount(playerB, "Quicken", 1);
204     }

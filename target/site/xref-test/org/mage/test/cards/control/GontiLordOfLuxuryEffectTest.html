15      /**
16       * Returning to your hand a creature you own but is controlled by an
17       * opponent doesn't let you replay it. Happened after I Aether Tradewinded
18       * my Rashmi that an opponent cast with Gonti, Lord of Luxury (the exile
19       * part could have something to do with this?). Then on my turn I couldn't
20       * replay it.
21       */
22      @Test
23      public void testCanBeCastAgain() {
24          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8);
25          // Deathtouch
26          // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
27          // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
28          // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
29          addCard(Zone.HAND, playerA, "Gonti, Lord of Luxury", 1); // Creature {2}{B}{B}
30  
31          addCard(Zone.LIBRARY, playerB, "Rashmi, Eternities Crafter"); // Creature {2}{G}{U}
32          skipInitShuffling();
33          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
34          addCard(Zone.BATTLEFIELD, playerB, "Forest", 2);
35          // Return target permanent you control and target permanent you don't control to their owners' hands.
36          addCard(Zone.HAND, playerB, "Aether Tradewinds", 1); // Intant {2}{U}
37          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
38  
39          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gonti, Lord of Luxury");
40          // Player B is autochosen since only option
41          setChoice(playerA, "Rashmi, Eternities Crafter");
42          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Rashmi, Eternities Crafter");
43  
44          castSpell(1, PhaseStep.END_TURN, playerB, "Aether Tradewinds", "Silvercoat Lion^Rashmi, Eternities Crafter");
45          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Rashmi, Eternities Crafter");
46  
47          setStopAt(2, PhaseStep.BEGIN_COMBAT);
48          execute();
49  
50          assertGraveyardCount(playerB, "Aether Tradewinds", 1);
51          assertHandCount(playerB, "Silvercoat Lion", 1);
52          assertHandCount(playerB, "Rashmi, Eternities Crafter", 0);
53          assertPermanentCount(playerB, "Rashmi, Eternities Crafter", 1);
54  
55      }

57      /**
58       * Opponent using Gonti, Lord of Luxury took Mirari's Wake out of my library
59       * and cast it. I cast Cyclonic Rift on Mirari's Wake to put it back in my
60       * hand and was unable to recast Mirari's Wake.
61       */
62      @Test
63      public void testCanBeCastAgainCyclonicRift() {
64          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 9);
65          // Deathtouch
66          // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
67          // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
68          // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
69          addCard(Zone.HAND, playerA, "Gonti, Lord of Luxury", 1); // Creature 2/3 {2}{B}{B}
70  
71          // Creatures you control get +1/+1.
72          // Whenever you tap a land for mana, add one mana of any type that land produced.
73          addCard(Zone.LIBRARY, playerB, "Mirari's Wake"); // Enchantment {3}{G}{W}
74          skipInitShuffling();
75  
76          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
77          addCard(Zone.BATTLEFIELD, playerB, "Forest", 2);
78          addCard(Zone.BATTLEFIELD, playerB, "Plains", 1);
79          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
80          // Return target nonland permanent you don't control to its owner's hand.
81          // Overload {6}{U} (You may cast this spell for its overload cost. If you do, change its text by replacing all instances of "target" with "each.")
82          addCard(Zone.HAND, playerB, "Cyclonic Rift", 1); // Intant {1}{U}
83  
84          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gonti, Lord of Luxury");
85          // Player B is autochosen since only option
86          setChoice(playerA, "Mirari's Wake");
87          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Mirari's Wake");
88          castSpell(1, PhaseStep.END_TURN, playerB, "Cyclonic Rift", "Mirari's Wake");
89  
90          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Mirari's Wake");
91  
92          setStopAt(2, PhaseStep.BEGIN_COMBAT);
93          execute();
94  
95          assertPermanentCount(playerA, "Gonti, Lord of Luxury", 1);
96          assertPowerToughness(playerA, "Gonti, Lord of Luxury", 2, 3);
97          assertGraveyardCount(playerB, "Cyclonic Rift", 1);
98  
99          assertPermanentCount(playerB, "Mirari's Wake", 1);
100         assertPowerToughness(playerB, "Silvercoat Lion", 3, 3);
101 
102     }

104     /**
105      * I noticed in a game that when you cast Lingering Souls off of Gonti, Lord
106      * of Luxury and then the lingering souls goes to the graveyard it cannot be
107      * flashed back. The gonti was my opponent's and the lingering souls was
108      * mine for reference.
109      */
110     @Test
111     public void testCanBeCastLaterWithFlashBack() {
112         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 7);
113         // Deathtouch
114         // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
115         // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
116         // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
117         addCard(Zone.HAND, playerA, "Gonti, Lord of Luxury", 1); // Creature 2/3 {2}{B}{B}
118 
119         // Create two 1/1 white Spirit creature tokens with flying.
120         // Flashback {1}{B}
121         addCard(Zone.LIBRARY, playerB, "Lingering Souls"); // Sorcery {2}{W}
122         skipInitShuffling();
123 
124         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 2);
125 
126         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gonti, Lord of Luxury");
127         // Player B is autochosen since only option
128         setChoice(playerA, "Lingering Souls");
129 
130         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lingering Souls");
131 
132         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Flashback");
133 
134         setStopAt(2, PhaseStep.BEGIN_COMBAT);
135         execute();
136 
137         assertPermanentCount(playerA, "Gonti, Lord of Luxury", 1);
138         assertPowerToughness(playerA, "Gonti, Lord of Luxury", 2, 3);
139         assertPermanentCount(playerA, "Spirit Token", 2);
140 
141         assertPermanentCount(playerB, "Spirit Token", 2);
142 
143         assertExileCount("Lingering Souls", 1);
144 
145     }

147     /**
148      * An opponent used a Gonti against me, and took my Ob Nixilis Reignited. He
149      * later played it, and I killed my Ob Nixilis. Then later, using Seasons
150      * Past, I got Ob Nixilis back in my hand. However, the Ob Nixilis was now
151      * uncastable. Has anyone else encountered this?
152      */
153     @Test
154     public void testPlaneswalkerCanBeCastLaterFromHand() {
155         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 9);
156         // Deathtouch
157         // When Gonti, Lord of Luxury enters the battlefield, look at the top four cards of target opponent's library, exile one of them face down,
158         // then put the rest on the bottom of that library in a random order. For as long as that card remains exiled,
159         // you may look at it, you may cast it, and you may spend mana as though it were mana of any type to cast it.
160         addCard(Zone.HAND, playerA, "Gonti, Lord of Luxury", 1); // Creature 2/3 {2}{B}{B}
161 
162         // +1: You draw a card and you lose 1 life.
163         // -3: Destroy target creature.
164         // -8: Target opponent gets an emblem with "Whenever a player draws a card, you lose 2 life."
165         addCard(Zone.LIBRARY, playerB, "Ob Nixilis Reignited"); // Planeswalker [5]  {3}{B}{B}
166         skipInitShuffling();
167 
168         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 3);
169         addCard(Zone.BATTLEFIELD, playerB, "Forest", 3);
170 
171         // Return any number of cards with different converted mana costs from your graveyard to your hand. Put Seasons Past on the bottom of its owner's library.
172         addCard(Zone.HAND, playerB, "Seasons Past", 1); // Sorcery {4}{G}{G}
173 
174         addCard(Zone.BATTLEFIELD, playerB, "Dross Crocodile", 2); // Creature 5/1
175 
176         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gonti, Lord of Luxury");
177         // Player B is autochosen since only option
178         setChoice(playerA, "Ob Nixilis Reignited");
179         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Ob Nixilis Reignited", true);
180         activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "-3:", "Dross Crocodile");
181 
182         attack(2, playerB, "Dross Crocodile", "Ob Nixilis Reignited");
183 
184         castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Seasons Past");
185         setChoice(playerB, "Ob Nixilis Reignited");
186 
187         castSpell(4, PhaseStep.PRECOMBAT_MAIN, playerB, "Ob Nixilis Reignited");
188 
189         setStopAt(4, PhaseStep.BEGIN_COMBAT);
190         execute();
191 
192         assertPermanentCount(playerA, "Gonti, Lord of Luxury", 1);
193         assertPermanentCount(playerB, "Dross Crocodile", 1);
194         assertGraveyardCount(playerB, "Dross Crocodile", 1);
195 
196         assertGraveyardCount(playerB, "Seasons Past", 0);
197 
198         assertHandCount(playerB, "Ob Nixilis Reignited", 0);
199         assertPermanentCount(playerB, "Ob Nixilis Reignited", 1);
200     }

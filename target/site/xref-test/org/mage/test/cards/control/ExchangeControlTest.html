123     /**
124      * First gain control by Act of Treason. Then exchange control with other
125      * opponent's creature.
126      *
127      * Finally second creature should stay under ours control permanently.
128      */
129     @Test
130     public void testInteractionWithOtherChangeControlEffect() {
131         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
132         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
133         addCard(Zone.HAND, playerA, "Switcheroo");
134         addCard(Zone.HAND, playerA, "Act of Treason");
135 
136         // both creatures on opponent's side
137         addCard(Zone.BATTLEFIELD, playerB, "Llanowar Elves");
138         addCard(Zone.BATTLEFIELD, playerB, "Elite Vanguard");
139 
140         // get control
141         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Act of Treason", "Elite Vanguard");
142 
143         // attack
144         attack(1, playerA, "Elite Vanguard");
145 
146         // exchange control after combat
147         castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Switcheroo", "Llanowar Elves^Elite Vanguard");
148 
149         // check the control effect still works on second turn
150         setStopAt(2, PhaseStep.END_TURN);
151         execute();
152 
153         // now it is our creature for ages
154         assertPermanentCount(playerA, "Llanowar Elves", 1);
155         // this one is still on opponent's side
156         assertPermanentCount(playerB, "Elite Vanguard", 1);
157     }

160      * Tests switching controls will affect restriction effect
161      */
162     @Test
163     public void testRestrictionEffect() {
164         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
165         addCard(Zone.HAND, playerA, "Switcheroo");
166 
167         addCard(Zone.BATTLEFIELD, playerA, "War Falcon");
168         addCard(Zone.BATTLEFIELD, playerA, "Elite Vanguard");
169 
170         addCard(Zone.BATTLEFIELD, playerB, "Llanowar Elves");
171 
172         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Switcheroo", "War Falcon^Llanowar Elves");
173 
174         attack(2, playerB, "War Falcon");
175 
176         setStopAt(2, PhaseStep.END_TURN);
177 
178         try {
179             execute();
180             Assert.fail("must throw exception on execute");
181         } catch (Throwable e) {
182             if (!e.getMessage().contains("Player PlayerB must have 0 actions but found 1")) {
183                 Assert.fail("Needed error about PlayerB having too many actions, but got:\n" + e.getMessage());
184             }
185         }
186 
187         // check creatures changes their controllers
188         assertPermanentCount(playerA, "Llanowar Elves", 1);
189         assertPermanentCount(playerB, "War Falcon", 1);
190 
191         // War Falcon can't attack
192         assertLife(playerA, 20);
193     }

195     /**
196      * An control exchanged creature gets an copy effect from an creature with
197      * an activated ability to the by exchange controlled creature. Check that
198      * the activated ability is controlled by the new controller of the copy
199      * target.
200      */
201     @Test
202     public void testExchangeAnCopyEffect() {
203         addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
204         // Gilded Drake {1}{U} Creature - Drake
205         // Flying
206         // When Gilded Drake enters the battlefield, exchange control of Gilded Drake and up to one target
207         // creature an opponent controls. If you don't make an exchange, sacrifice Gilded Drake. This ability
208         // can't be countered except by spells and abilities. (This effect lasts indefinitely.)
209         addCard(Zone.HAND, playerA, "Gilded Drake");
210         // Polymorphous Rush {2}{U} - Instant
211         // Strive — Polymorphous Rush costs {1}{U} more to cast for each target beyond the first.
212         // Choose a creature on the battlefield. Any number of target creatures you control each become a copy of that creature until end of turn.
213         addCard(Zone.HAND, playerA, "Polymorphous Rush");
214         // {U}: Manta Riders gains flying until end of turn.
215         addCard(Zone.BATTLEFIELD, playerB, "Manta Riders");
216         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
217 
218         // exchange control between Gilded Drake and Silvercoat Lion
219         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gilded Drake");
220         addTarget(playerA, "Silvercoat Lion");
221         // Let your Silvercoat Lion now be a copy of the Manta Riders
222         castSpell(1, PhaseStep.BEGIN_COMBAT, playerA, "Polymorphous Rush", "Silvercoat Lion");
223         addTarget(playerA, "Manta Riders");
224 
225         // now use the activated ability to make the "Silvercoat Lions" (that became Mana Riders) flying
226         activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{U}: {this} gains flying until end of turn.");
227 
228         setStopAt(1, PhaseStep.END_TURN);
229         execute();
230 
231         assertLife(playerA, 20);
232         assertLife(playerB, 20);
233 
234         // check creatures change their controllers
235         assertPermanentCount(playerB, "Gilded Drake", 1);
236         assertGraveyardCount(playerA, "Polymorphous Rush", 1);
237 
238         assertPermanentCount(playerB, "Silvercoat Lion", 0);
239         assertPermanentCount(playerA, "Silvercoat Lion", 0);
240 
241         assertPermanentCount(playerB, "Manta Riders", 1);
242         assertPermanentCount(playerA, "Manta Riders", 1);
243 
244         Permanent controlledMantas = getPermanent("Manta Riders", playerA.getId());
245         Assert.assertTrue("Manta Riders should have flying ability", controlledMantas.getAbilities().contains(FlyingAbility.getInstance()));
246 
247     }

250      * Gilded Drake doesn't get sacrificed if the creature its ability targets
251      * is invalid when it enters the battlefield
252      */
253     @Test
254     public void testDrakeSacrificedIfNoExchangeHappens() {
255         addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
256         // Gilded Drake {1}{U} Creature - Drake
257         // Flying
258         // When Gilded Drake enters the battlefield, exchange control of Gilded Drake and up to one target
259         // creature an opponent controls. If you don't make an exchange, sacrifice Gilded Drake. This ability
260         // can't be countered except by spells and abilities. (This effect lasts indefinitely.)
261         addCard(Zone.HAND, playerA, "Gilded Drake");
262 
263         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
264         addCard(Zone.BATTLEFIELD, playerB, "Mountain");
265         addCard(Zone.HAND, playerB, "Lightning Bolt");
266 
267         // exchange control between Gilded Drake and Silvercoat Lion
268         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gilded Drake");
269         addTarget(playerA, "Silvercoat Lion");
270 
271         // Destroy Silvercoat Lion before the exchange resolves so the Drake has to be sacrificed
272         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Silvercoat Lion", "When");
273 
274         setStopAt(1, PhaseStep.BEGIN_COMBAT);
275         execute();
276 
277         assertGraveyardCount(playerB, "Lightning Bolt", 1);
278         assertGraveyardCount(playerB, "Silvercoat Lion", 1);
279         assertGraveyardCount(playerA, "Gilded Drake", 1);
280     }

282     /**
283      * 1. Kiki-Jiki copied Gilded Drake.
284      * 2. Gilded Drake copy is exchanged for another creature.
285      * 3. After the exchange occurs, Gilded Drake is killed by any means.
286      * 4. Exchange creature is returned to previous controller (possible owner) during the next phase.
287      * <p>
288      * See https://github.com/magefree/mage/issues/8742
289      */
290     @Ignore
291     @Test
292     public void testGildedDrakeCopyExchange() {
293         addCard(Zone.BATTLEFIELD, playerA, "Kiki-Jiki, Mirror Breaker");
294         addCard(Zone.HAND, playerA, "Gilded Drake");
295         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
296         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
297 
298         addCard(Zone.HAND, playerA, "Lightning Bolt", 2);
299 
300         addCard(Zone.BATTLEFIELD, playerB, "Dwarven Trader"); // Exchange target
301         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion"); // Exchange target
302 
303         setStrictChooseMode(true);
304         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gilded Drake");
305         addTarget(playerA, "Dwarven Trader"); // Target for Gilded Drake
306         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, true); // Skip the cast and grab the ETB
307 
308         // Copy the drake
309         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}", "Gilded Drake", "When {this}");
310         addTarget(playerA, "Silvercoat Lion"); // Target for Gilded Drake
311 
312         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
313 
314         // Destroy both of the Drakes
315         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt");
316         addTarget(playerA, "Gilded Drake");
317         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
318         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt");
319         addTarget(playerA, "Gilded Drake");
320 
321         setStopAt(1, PhaseStep.END_TURN);
322         execute();
323 
324         assertPermanentCount(playerA, "Kiki-Jiki, Mirror Breaker", 1);
325         assertPermanentCount(playerA, "Dwarven Trader", 1);  // Original's exhange target
326         assertPermanentCount(playerA, "Silvercoat Lion", 1); // Copy's exchange target
327 
328         assertPermanentCount(playerA, "Gilded Drake", 0);
329         assertPermanentCount(playerB, "Gilded Drake", 0);
330     }

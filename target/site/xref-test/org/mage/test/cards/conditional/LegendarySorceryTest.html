13      @Test
14      public void testCastSuccessful() {
15          // (You may cast a legendary sorcery only if you control a legendary creature or planeswalker.)
16          // Exile all nonland permanents that aren't legendary.
17          addCard(Zone.HAND, playerA, "Urza's Ruinous Blast"); // Sorcery Legendary  {4}{W}
18          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
19          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1); // non Legendary
20          // Flying, first strike, vigilance, trample, haste, protection from black and from red
21          addCard(Zone.BATTLEFIELD, playerA, "Akroma, Angel of Wrath", 1); // Legendary
22  
23          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1); // non Legendary
24          // Flying, first strike, vigilance, trample, haste, protection from black and from red
25          addCard(Zone.BATTLEFIELD, playerB, "Akroma, Angel of Wrath", 1); // Legendary
26  
27          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Urza's Ruinous Blast");
28  
29          setStopAt(1, PhaseStep.BEGIN_COMBAT);
30          execute();
31  
32          assertGraveyardCount(playerA, "Urza's Ruinous Blast", 1);
33  
34          assertPermanentCount(playerA, "Plains", 5);
35  
36          assertPermanentCount(playerA, "Silvercoat Lion", 0);
37          assertPermanentCount(playerB, "Silvercoat Lion", 0);
38  
39          assertPermanentCount(playerA, "Akroma, Angel of Wrath", 1);
40          assertPermanentCount(playerB, "Akroma, Angel of Wrath", 1);
41      }

73      @Test
74      public void testCastSuccessfulFromExile() {
75          // First strike
76          // When this enters the battlefield, exile target instant or sorcery card from an opponent's graveyard.
77          // You may cast that card this turn and you may spend mana as though it were mana of any color.
78          // If that card would be put into a graveyard this turn, exile it instead.
79          addCard(Zone.HAND, playerA, "Dire Fleet Daredevil"); // Creature  {1}{R}
80          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
81          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
82          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1); // non Legendary
83          addCard(Zone.BATTLEFIELD, playerA, "Akroma, Angel of Wrath", 1); // Legendary
84  
85          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1); // non Legendary
86          // Flying, first strike, vigilance, trample, haste, protection from black and from red
87          addCard(Zone.BATTLEFIELD, playerB, "Akroma, Angel of Wrath", 1); // Legendary
88          // (You may cast a legendary sorcery only if you control a legendary creature or planeswalker.)
89          // Exile all nonland permanents that aren't legendary.
90          addCard(Zone.GRAVEYARD, playerB, "Urza's Ruinous Blast"); // Sorcery Legendary  {4}{W}
91  
92          setStrictChooseMode(true);
93  
94          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Dire Fleet Daredevil");
95          addTarget(playerA, "Urza's Ruinous Blast");
96  
97          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Urza's Ruinous Blast");
98  
99          setStopAt(1, PhaseStep.END_TURN);
100         execute();
101 
102         assertExileCount(playerB, "Urza's Ruinous Blast", 1);
103 
104         assertHandCount(playerA, "Dire Fleet Daredevil", 0);
105         assertPermanentCount(playerA, "Dire Fleet Daredevil", 0);
106 
107         assertExileCount(playerA, "Dire Fleet Daredevil", 1);
108         assertExileCount(playerA, "Silvercoat Lion", 1);
109         assertExileCount(playerB, "Silvercoat Lion", 1);
110 
111         assertPermanentCount(playerA, "Plains", 5);
112 
113         assertPermanentCount(playerA, "Silvercoat Lion", 0);
114         assertPermanentCount(playerB, "Silvercoat Lion", 0);
115 
116         assertPermanentCount(playerA, "Akroma, Angel of Wrath", 1);
117         assertPermanentCount(playerB, "Akroma, Angel of Wrath", 1);
118     }

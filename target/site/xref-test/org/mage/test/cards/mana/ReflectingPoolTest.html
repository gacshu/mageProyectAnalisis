19      /**
20       * Reflecting Pool does not count Crumbling Vestige as a source of all mana
21       * colors
22       */
23      @Test
24      public void testTriggeredManaAbility() {
25          addCard(Zone.HAND, playerA, "Lightning Bolt", 1); // {R}
26  
27          // {T}: Add one mana of any type that a land you control could produce.
28          addCard(Zone.BATTLEFIELD, playerA, "Reflecting Pool", 1);
29          // Crumbling Vestige enters the battlefield tapped.
30          // When Crumbling Vestige enters the battlefield, add one mana of any color.
31          // {T}: Add {C} to you mana pool.
32          addCard(Zone.HAND, playerA, "Crumbling Vestige", 1);
33  
34          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Crumbling Vestige");
35          setChoice(playerA, "Red");
36          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
37  
38          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
39  
40          setStopAt(1, PhaseStep.BEGIN_COMBAT);
41          execute();
42  
43          assertPermanentCount(playerA, "Crumbling Vestige", 1);
44          assertGraveyardCount(playerA, "Lightning Bolt", 1);
45  
46          assertLife(playerA, 20);
47          assertLife(playerB, 17);
48      }

135     @Test
136     public void testWithDifferentLands() {
137         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
138 
139         // {T}: Add one mana of any type that a land you control could produce.
140         addCard(Zone.BATTLEFIELD, playerA, "Reflecting Pool", 1);
141         // {T}: Add one mana of any color that a land an opponent controls could produce.
142         addCard(Zone.BATTLEFIELD, playerA, "Exotic Orchard", 1);
143         addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
144 
145         // {T}: Add one mana of any color that a land an opponent controls could produce.
146         addCard(Zone.BATTLEFIELD, playerB, "Exotic Orchard", 1);
147         addCard(Zone.BATTLEFIELD, playerB, "Plains", 1);
148 
149         setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
150         execute();
151 
152         ManaOptions options = playerA.getAvailableManaTest(currentGame);
153         Assert.assertEquals("Player A should be able to create only 3 different mana options", 3, options.size());
154         assertManaOptions("{G}{G}{G}", options);
155         assertManaOptions("{G}{G}{W}", options);
156         assertManaOptions("{G}{W}{W}", options);
157 
158         options = playerB.getAvailableManaTest(currentGame);
159         Assert.assertEquals("Player B should be able to create only 2 different mana options", 2, options.size());
160         assertManaOptions("{G}{W}", options);
161         assertManaOptions("{W}{W}", options);
162     }

268     /**
269      * I only control 3 lands, a Triome, a Reflecting Pool, and a Mana
270      * Confluence. The Reflecting Pool is able to tap for colorless, but it
271      * should not be able to.
272      * 
273      * https://blogs.magicjudges.org/rulestips/2012/09/you-have-to-name-a-color-when-you-add-one-mana-of-any-color-to-your-mana-pool/
274      */
275     @Test
276     public void testWithTriomeAndManaConfluence() {
277         // {T}: Add {C}.       
278         // {1}{U}, {T}: Put target artifact card from your graveyard on top of your library.
279         addCard(Zone.BATTLEFIELD, playerA, "Academy Ruins", 1);
280         // {T}, Pay 1 life: Add one mana of any color.        
281         addCard(Zone.BATTLEFIELD, playerA, "Mana Confluence", 1);
282         // {T}: Add one mana of any type that a land you control could produce.        
283         addCard(Zone.BATTLEFIELD, playerA, "Reflecting Pool", 1);
284 
285         setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
286         execute();
287 
288         ManaOptions options = playerA.getAvailableManaTest(currentGame);
289         Assert.assertEquals("Player A should be able to create only 2 different mana options", 2, options.size());
290         assertManaOptions("{C}{C}{Any}", options);
291         assertManaOptions("{C}{Any}{Any}", options);
292     }

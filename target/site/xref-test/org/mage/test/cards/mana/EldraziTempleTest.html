18          // {T}: Add {C}{C}. Spend this mana only to cast colorless Eldrazi spells or activate abilities of colorless Eldrazi.
19          addCard(Zone.BATTLEFIELD, playerA, "Eldrazi Temple", 1);
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
21          // Devoid (This card has no color.)
22          // {2}{R}, Sacrifice another colorless creature: Barrage Tyrant deals damage equal to the sacrificed creature's power to target creature or player.
23          addCard(Zone.HAND, playerA, "Barrage Tyrant", 1); // Creature {4}{R}
24  
25          // If a land is tapped for two or more mana, it produces {C} instead of any other type and amount.
26          // Each spell a player casts costs {1} more to cast for each other spell that player has cast this turn.
27          // addCard(Zone.BATTLEFIELD, playerB, "Damping Sphere", 1);
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Barrage Tyrant");

30          setStopAt(1, PhaseStep.BEGIN_COMBAT);
31          execute();

37      /**
38       * Damping Sphere vs Eldrazi
39       *
40       * My opponent managed to produce 7 mana to cast a All Is Dust with 6 lands
41       * where one was an Eldrazi Temple. This should not be possible if my
42       * reading of Damping Sphere (If a land is tapped for two or more mana, it
43       * produces {C} instead of any other type and amount.) is correct. It should
44       * produce a single colorless mana rather than 2 colorless for the eldrazi
45       * tribal spell
46       */
47      @Test
48      public void TestWithDampingSphere() {
49          // {T}: Add {C}.
50          // {T}: Add {C}{C}. Spend this mana only to cast colorless Eldrazi spells or activate abilities of colorless Eldrazi.
51          addCard(Zone.BATTLEFIELD, playerA, "Eldrazi Temple", 1);
52          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
53          // Devoid (This card has no color.)
54          // {2}{R}, Sacrifice another colorless creature: Barrage Tyrant deals damage equal to the sacrificed creature's power to target creature or player.
55          addCard(Zone.HAND, playerA, "Barrage Tyrant", 1); // Creature {4}{R}
56  
57          // If a land is tapped for two or more mana, it produces {C} instead of any other type and amount.
58          // Each spell a player casts costs {1} more to cast for each other spell that player has cast this turn.
59          addCard(Zone.BATTLEFIELD, playerB, "Damping Sphere", 1);
60          checkPlayableAbility("Damping Sphere effective", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Barrage Tyrant", false);
61  
62          setStopAt(1, PhaseStep.BEGIN_COMBAT);
63          execute();
64      }

66      @Test
67      public void TestUseOfUnconditionalMana() {
68          // {T}: Add {C}.
69          // {T}: Add {C}{C}. Spend this mana only to cast colorless Eldrazi spells or activate abilities of colorless Eldrazi.
70          addCard(Zone.BATTLEFIELD, playerA, "Eldrazi Temple", 1);
71          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
72          // {T}: Add {W}, {B}, or {G}.
73          // {W}{B}{G}, {T}, Sacrifice Abzan Banner: Draw a card.
74          addCard(Zone.HAND, playerA, "Abzan Banner", 1); // Artifact {3}
75  
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Abzan Banner");
77  
78          setStopAt(1, PhaseStep.BEGIN_COMBAT);
79          execute();
80  
81          assertPermanentCount(playerA, "Abzan Banner", 1);

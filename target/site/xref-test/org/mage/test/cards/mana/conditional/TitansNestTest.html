19      @Test
20      public void testTitansNest(){
21          setStrictChooseMode(true);
22          
23          // At the beginning of your upkeep, look at the top card of your library. You may put that card into your graveyard.
24          // Exile a card from your graveyard: Add {C}. Spend this mana only to cast a colored spell without {X} in its mana cost.
25          addCard(Zone.HAND, playerA, "Titans' Nest"); // Enchantment {1}{B}{G}{U}
26          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
27          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
28          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
29          
30          addCard(Zone.GRAVEYARD, playerA, "Grizzly Bears", 2);
31  
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Titans' Nest");                
33          
34          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
35          execute();
36  
37          assertPermanentCount(playerA, "Titans' Nest", 1);
38          
39          ManaOptions manaOptions = playerA.getAvailableManaTest(currentGame);
40          Assert.assertEquals("mana variations don't fit", 1, manaOptions.size());
41          assertManaOptions("{C}{C}[{TitansNestManaCondition}]", manaOptions);        
42      }

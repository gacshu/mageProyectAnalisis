12  public class UndercityReachesTest extends CardTestPlayerBase {
13  
14      @Test
15      public void test_CanTriggerByController() {
16          removeAllCardsFromHand(playerA);
17  
18          // Whenever a creature deals combat damage to a player, its controller may a draw a card
19          addPlane(playerA, Planes.PLANE_UNDERCITY_REACHES);
20          addCard(Zone.BATTLEFIELD, playerA, "Balduvian Bears", 1);
21  
22          attack(1, playerA, "Balduvian Bears");
23          setChoice(playerA, true); // draw
24  
25          setStrictChooseMode(true);
26          setStopAt(1, PhaseStep.END_TURN);
27          execute();
28  
29          assertLife(playerB, 20 - 2);
30          assertHandCount(playerA, 1);
31      }
32  
33      @Test
34      public void test_CanTriggerByOther() {
35          removeAllCardsFromHand(playerB);
36  
37          // Whenever a creature deals combat damage to a player, its controller may a draw a card
38          addPlane(playerA, Planes.PLANE_UNDERCITY_REACHES);
39          addCard(Zone.BATTLEFIELD, playerB, "Balduvian Bears", 1);
40  
41          attack(2, playerB, "Balduvian Bears");
42          setChoice(playerB, true); // draw
43  
44          setStrictChooseMode(true);
45          setStopAt(2, PhaseStep.END_TURN);
46          execute();
47  
48          assertLife(playerA, 20 - 2);
49          assertHandCount(playerB, 1 + 1); // +1 from turn draw, +1 from trigger
50      }
51  }

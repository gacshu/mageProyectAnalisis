3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   /**
9    *
10   * @author BetaSteward
11   */
12  public class ZuberasTest extends CardTestPlayerBase {
13      /*
14       * Ashen-Skin Zubera
15       * Creature — Zubera Spirit 1/2, 1B (2)
16       * When Ashen-Skin Zubera dies, target opponent discards a card for each 
17       * Zubera that died this turn.
18      
19       * Dripping-Tongue Zubera
20       * Creature — Zubera Spirit 1/2, 1G (2)
21       * When Dripping-Tongue Zubera dies, put a 1/1 colorless Spirit creature 
22       * token onto the battlefield for each Zubera that died this turn.
23      
24       * Ember-Fist Zubera
25       * Creature — Zubera Spirit 1/2, 1R (2)
26       * When Ember-Fist Zubera dies, it deals damage to any target
27       * equal to the number of Zubera that died this turn.
28  
29       * Floating-Dream Zubera
30       * Creature — Zubera Spirit 1/2, 1U (2)
31       * When Floating-Dream Zubera dies, draw a card for each Zubera that died 
32       * this turn.
33      
34       * Silent-Chant Zubera
35       * Creature — Zubera Spirit 1/2, 1W (2)
36       * When Silent-Chant Zubera dies, you gain 2 life for each Zubera that died this turn.
37      */
38      
39      // test that creatures damaged by Aggravate attack
40      @Test
41      public void testZuberas() {
42          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
43          addCard(Zone.BATTLEFIELD, playerA, "Ashen-Skin Zubera");
44          addCard(Zone.BATTLEFIELD, playerA, "Dripping-Tongue Zubera");
45          addCard(Zone.BATTLEFIELD, playerA, "Ember-Fist Zubera");
46          addCard(Zone.BATTLEFIELD, playerA, "Floating-Dream Zubera");
47          addCard(Zone.BATTLEFIELD, playerA, "Silent-Chant Zubera");
48          addCard(Zone.HAND, playerA, "Lightning Bolt", 5);
49          addCard(Zone.HAND, playerB, "Island", 3);
50          
51          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Silent-Chant Zubera");
52          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Floating-Dream Zubera");
53          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Ember-Fist Zubera");
54          addTarget(playerA, playerB);
55          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Ashen-Skin Zubera");
56          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Dripping-Tongue Zubera");
57          
58          setStopAt(1, PhaseStep.DECLARE_BLOCKERS);
59          execute();
60                  
61          assertPermanentCount(playerA, "Spirit Token", 1);
62          assertHandCount(playerB, 1);
63          assertHandCount(playerA, 4);
64          assertLife(playerB, 17);
65          assertLife(playerA, 30);
66          
67      }

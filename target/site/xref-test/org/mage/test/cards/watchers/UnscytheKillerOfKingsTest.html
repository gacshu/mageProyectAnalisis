14      /*
15       * Unscythe, Killer of Kings
16       * Legendary Artifact â€” Equipment, UBBR (4)
17       * Equipped creature gets +3/+3 and has first strike.
18       * Whenever a creature dealt damage by equipped creature this turn dies, you 
19       * may exile that card. If you do, put a 2/2 black Zombie creature token onto 
20       * the battlefield.
21       * Equip {2}

25      // test that when creature damaged by equipped creature dies a Zombie token is created
26      @Test
27      public void testDamagedCreatureDies() {
28          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
29          addCard(Zone.BATTLEFIELD, playerA, "Unscythe, Killer of Kings");
30          addCard(Zone.BATTLEFIELD, playerA, "Prodigal Pyromancer");
31          addCard(Zone.BATTLEFIELD, playerB, "Sejiri Merfolk");
32  
33          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Equip", "Prodigal Pyromancer");
34          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T}: {this} deals 1 damage to ", "Sejiri Merfolk");
35  
36          setStopAt(1, PhaseStep.END_TURN);
37          execute();
38  
39          assertPermanentCount(playerB, "Sejiri Merfolk", 0);
40          assertPermanentCount(playerA, "Zombie Token", 1);
41          assertExileCount("Sejiri Merfolk", 1);
42  
43      }
44  
45      // test that when creature damaged by equipped creature dies a Zombie token is created
46      @Test
47      public void testDamagedCreatureDiesAfterEquipped() {
48          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
49          addCard(Zone.BATTLEFIELD, playerA, "Unscythe, Killer of Kings");
50          // {T}: Prodigal Pyromancer deals 1 damage to 
51          addCard(Zone.BATTLEFIELD, playerA, "Prodigal Pyromancer");
52          addCard(Zone.HAND, playerA, "Lightning Bolt");
53          addCard(Zone.BATTLEFIELD, playerB, "Craw Wurm");
54  
55          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: {this} deals 1 damage to ", "Craw Wurm");
56          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Equip", "Prodigal Pyromancer");
57          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Craw Wurm", "Equip", StackClause.WHILE_NOT_ON_STACK);
58  
59          setStopAt(1, PhaseStep.END_TURN);
60          execute();
61  
62          assertPermanentCount(playerB, "Craw Wurm", 0);
63          assertPermanentCount(playerA, "Zombie Token", 1);
64          assertExileCount("Craw Wurm", 1);
65  
66      }

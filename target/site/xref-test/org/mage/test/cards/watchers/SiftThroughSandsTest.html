12  public class SiftThroughSandsTest extends CardTestPlayerBase {
13  
14      /*
15       * Sift Through Sands
16       * Instant — Arcane, 1UU (3)
17       * Draw two cards, then discard a card.
18       * If you've cast a spell named Peer Through Depths and a spell named Reach
19       * Through Mists this turn, you may search your library for a card named
20       * The Unspeakable, put it onto the battlefield, then shuffle your library.
21       *
22       * Peer Through Depths
23       * Instant — Arcane, 1U (2)
24       * Look at the top five cards of your library. You may reveal an instant or
25       * sorcery card from among them and put it into your hand. Put the rest on
26       * the bottom of your library in any order.
27       *
28       * Reach Through Mists
29       * Instant — Arcane, U (1)
30       * Draw a card.
31       *
32       * The Unspeakable
33       * Legendary Creature — Spirit 6/7, 6UUU (9)
34       * Flying, trample
35       * Whenever The Unspeakable deals combat damage to a player, you may return
36       * target Arcane card from your graveyard to your hand.
37       *
38       */
39  
40      @Test
41      public void testTheUnspeakable() {
42          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
43          addCard(Zone.HAND, playerA, "Sift Through Sands");
44          addCard(Zone.HAND, playerA, "Peer Through Depths");
45          addCard(Zone.HAND, playerA, "Reach Through Mists");
46  
47          addCard(Zone.LIBRARY, playerA, "The Unspeakable", 4); // So that at least one will be in library even if you draw 2 with Sift and 1 with Reach
48  
49          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Reach Through Mists");
50          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Peer Through Depths");
51  
52          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Sift Through Sands");
53  
54          setStopAt(1, PhaseStep.END_TURN);
55          execute();
56  
57          assertPermanentCount(playerA, "The Unspeakable", 1);
58          assertHandCount(playerA, 2);
59  
60      }
61  
62      // test that The Unspeakable is not put onto the battlefield
63      @Test
64      public void testNotTheUnspeakable() {
65          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
66          addCard(Zone.HAND, playerA, "Sift Through Sands");
67          addCard(Zone.HAND, playerA, "Peer Through Depths");
68          addCard(Zone.HAND, playerA, "Reach Through Mists");
69          addCard(Zone.LIBRARY, playerA, "The Unspeakable", 4);
70  
71          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Reach Through Mists");
72  
73          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Sift Through Sands");
74  
75          setStopAt(1, PhaseStep.END_TURN);
76          execute();
77  
78          assertPermanentCount(playerA, "The Unspeakable", 0);
79          assertHandCount(playerA, 3);
80  
81      }

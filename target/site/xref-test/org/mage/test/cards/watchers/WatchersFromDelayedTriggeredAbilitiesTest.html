11  public class WatchersFromDelayedTriggeredAbilitiesTest extends CardTestPlayerBase {
12  
13      @Test
14      public void test_PsychicTheft_WatcherFromDelayedAbility() {
15  
16          // Target player reveals their hand. You choose an instant or sorcery card from it and exile that card.
17          // You may cast that card for as long as it remains exiled. At the beginning of the next end step,
18          // if you haven't cast the card, return it to its owner's hand.
19          addCard(Zone.HAND, playerA, "Psychic Theft", 1); // {1}{U}
20          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
21          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
22          //
23          addCard(Zone.HAND, playerB, "Lightning Bolt");
24  
25          // reveal
26          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 2);
27          checkPlayableAbility("can't play bolt", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", false);
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Psychic Theft", playerB);
29          setChoice(playerA, "Lightning Bolt");
30          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
31  
32          // can play until end step
33          checkPlayableAbility("can play bolt on 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", true);
34          checkPlayableAbility("can't play bolt on 2", 2, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", false);
35  
36          setStrictChooseMode(true);
37          setStopAt(2, PhaseStep.END_TURN);
38          execute();
39  
40          assertHandCount(playerB, "Lightning Bolt", 1);
41      }
42  }

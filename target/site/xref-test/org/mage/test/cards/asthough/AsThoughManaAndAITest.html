13      @Test
14      public void test_AutoPaymentMustUseAsThoughMana() {
15          // possible bug: AI auto-payment uses first mana ability from the mana source, so multi-colored lands can be broken
16  
17          // You may spend white mana as though it were red mana.
18          addCard(Zone.BATTLEFIELD, playerA, "Sunglasses of Urza", 1);
19          //
20          // {T}: Add {U} or {W}.
21          addCard(Zone.BATTLEFIELD, playerA, "Sea of Clouds", 1);
22          addCard(Zone.HAND, playerA, "Lightning Bolt", 1); // {R}
23  
24          checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Lightning Bolt", true);
25          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
26  
27          setStrictChooseMode(true);
28          setStopAt(1, PhaseStep.END_TURN);
29          execute();
30  
31          assertLife(playerB, 20 - 3);
32      }

1   package org.mage.test.cards.asthough;

3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBaseWithAIHelps;
7   
8   /**
9    * @author JayDi85
10   */
11  public class PlayLandsFromGraveyardTest extends CardTestPlayerBaseWithAIHelps {
12  
13      @Test
14      public void test_CrucibleOfWorlds() {
15          removeAllCardsFromHand(playerA);
16          removeAllCardsFromLibrary(playerA);
17  
18          // You may play lands from your graveyard.
19          addCard(Zone.HAND, playerA, "Crucible of Worlds"); // {3}
20          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
21          //
22          addCard(Zone.GRAVEYARD, playerA, "Island", 1);
23          //
24          addCard(Zone.GRAVEYARD, playerA, "Balduvian Bears");
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
26  
27          checkGraveyardCount("graveyard before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", 1);
28          checkGraveyardCount("graveyard before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Island", 1);
29          checkPlayableAbility("can't play as creature", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Balduvian Bears", false);
30          checkPlayableAbility("can't play as land", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Play Island", false);
31  
32          // play artifact and apply play from graveyard effect
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Crucible of Worlds");
34          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
35          checkPlayableAbility("can't play as creature", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Balduvian Bears", false);
36          checkPlayableAbility("can play as land", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Play Island", true);
37  
38          // play land
39          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Island");
40  
41          setStrictChooseMode(true);
42          setStopAt(1, PhaseStep.END_TURN);
43          execute();
44  
45          assertPermanentCount(playerA, "Island", 1);
46      }
47  }

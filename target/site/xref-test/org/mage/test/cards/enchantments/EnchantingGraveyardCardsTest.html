19      static final String LIGHTNING_BOLT = "Lightning Bolt";

22      /**
23       * Test that a card in the graveyard can be enchanted
24       */
25      @Test
26      public void testSpellwaeverVoluteNormal() {
27          // Enchant instant card in a graveyard
28          // Whenever you cast a sorcery spell, copy the enchanted instant card. You may cast the copy without paying its mana cost.
29          // If you do, exile the enchanted card and attach Spellweaver Volute to another instant card in a graveyard.
30          addCard(Zone.HAND, playerA, SPELLWEAVER_VOLUTE, 1); // Enchantment Aura {3}{U}{U}
31          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
32  
33          // Lightning Bolt deals 3 damage to any target.
34          addCard(Zone.GRAVEYARD, playerB, "Lightning Bolt", 1); // Instant  {R}
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, SPELLWEAVER_VOLUTE, LIGHTNING_BOLT);
37  
38          setStopAt(1, PhaseStep.BEGIN_COMBAT);
39          execute();
40  
41          assertLife(playerA, 20);
42          assertLife(playerB, 20);
43  
44          assertHandCount(playerA, 0);
45          assertPermanentCount(playerA, SPELLWEAVER_VOLUTE, 1);
46          assertGraveyardCount(playerB, LIGHTNING_BOLT, 1);
47  
48          Permanent spellweaver = getPermanent(SPELLWEAVER_VOLUTE);
49          Card attachedToCard = null;
50          if (spellweaver != null) {
51              attachedToCard = playerB.getGraveyard().get(spellweaver.getAttachedTo(), currentGame);
52          }
53          Assert.assertTrue(SPELLWEAVER_VOLUTE + " has to be attached to Lightning Bolt in graveyard", attachedToCard != null && attachedToCard.getName().equals(LIGHTNING_BOLT));
54      }

56      /**
57       * Test that a card in the graveyard can be enchanted and the enchanted card
58       * switches to a new one
59       */
60      @Test
61      public void testSpellwaeverVoluteAndSorcery() {
62  
63          // Enchant instant card in a graveyard
64          // Whenever you cast a sorcery spell, copy the enchanted instant card. You may cast the copy without paying its mana cost.
65          // If you do, exile the enchanted card and attach Spellweaver Volute to another instant card in a graveyard.
66          addCard(Zone.HAND, playerA, SPELLWEAVER_VOLUTE, 1); // Enchantment Aura {3}{U}{U}
67          addCard(Zone.HAND, playerA, "Cloak of Feathers"); // Sorcery {U}
68          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
69          addCard(Zone.BATTLEFIELD, playerA, "Island", 6);
70          addCard(Zone.GRAVEYARD, playerA, "Aerial Volley", 1); // Instant  {G}
71  
72          // Lightning Bolt deals 3 damage to any target.
73          addCard(Zone.GRAVEYARD, playerB, "Lightning Bolt", 1); // Instant  {R}
74  
75          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, SPELLWEAVER_VOLUTE, LIGHTNING_BOLT, true);
76          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cloak of Feathers", "Silvercoat Lion");
77          setChoice(playerA, true); // play the L. Bold
78          addTarget(playerA, playerB); // Add Target for the L. Bold
79  
80          setStopAt(1, PhaseStep.BEGIN_COMBAT);
81          execute();
82  
83          assertLife(playerA, 20);
84          assertLife(playerB, 17);
85  
86          assertHandCount(playerA, 1);
87          assertGraveyardCount(playerA, "Cloak of Feathers", 1);
88          assertPermanentCount(playerA, SPELLWEAVER_VOLUTE, 1);
89          assertGraveyardCount(playerA, "Aerial Volley", 1);
90          assertExileCount(playerB, LIGHTNING_BOLT, 1);
91          assertAbility(playerA, "Silvercoat Lion", FlyingAbility.getInstance(), true);
92          Permanent spellweaver = getPermanent(SPELLWEAVER_VOLUTE);
93          Card attachedToCard = null;
94          if (spellweaver != null) {
95              attachedToCard = playerA.getGraveyard().get(spellweaver.getAttachedTo(), currentGame);
96          }
97          Assert.assertTrue(SPELLWEAVER_VOLUTE + " has to be attached to Aerial Volley in graveyard", attachedToCard != null && attachedToCard.getName().equals("Aerial Volley"));
98  
99          assertHandCount(playerA, 1);
100 
101     }

151     /**
152      * Test that a card in the graveyard can be enchanted and if the Enchantment
153      * returns to hand, the enchanting ends
154      */
155     @Test
156     public void testSpellwaeverVoluteAndReturnToHand() {
157 
158         // Enchant instant card in a graveyard
159         // Whenever you cast a sorcery spell, copy the enchanted instant card. You may cast the copy without paying its mana cost.
160         // If you do, exile the enchanted card and attach Spellweaver Volute to another instant card in a graveyard.
161         addCard(Zone.HAND, playerA, SPELLWEAVER_VOLUTE, 1); // Enchantment Aura {3}{U}{U}
162         addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
163 
164         // Lightning Bolt deals 3 damage to any target.
165         addCard(Zone.GRAVEYARD, playerB, "Lightning Bolt", 1); // Instant  {R}
166 
167         // Return target permanent to its owner's hand.
168         addCard(Zone.HAND, playerB, "Boomerang", 1); // Instant  {U}{U}
169         addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
170 
171         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, SPELLWEAVER_VOLUTE, LIGHTNING_BOLT);
172 
173         castSpell(1, PhaseStep.BEGIN_COMBAT, playerB, "Boomerang", SPELLWEAVER_VOLUTE);
174 
175         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
176         execute();
177 
178         assertLife(playerA, 20);
179         assertLife(playerB, 20);
180 
181         assertHandCount(playerA, SPELLWEAVER_VOLUTE, 1);
182         assertGraveyardCount(playerB, "Boomerang", 1);
183         assertPermanentCount(playerA, SPELLWEAVER_VOLUTE, 0);
184 
185         for (Card card : playerB.getGraveyard().getCards(currentGame)) {
186             if (card.getName().equals(LIGHTNING_BOLT)) {
187                 Assert.assertTrue(LIGHTNING_BOLT + " may not have any attachments", card.getAttachments().isEmpty());
188 
189             }
190         }
191         assertHandCount(playerA, 1);
192 
193     }

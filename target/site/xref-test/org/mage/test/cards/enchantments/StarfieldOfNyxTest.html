20      /**
21       * I had Starfield of Nyx out. With the upkeep trigger, I brought back a
22       * Cloudform, which was the fifth enchantment. I had another Cloudform, and
23       * Starfield of Nyx not only turned both of them into creatures (it
24       * shouldn't, because they're auras), but it also destroyed them. The
25       * manifests stayed on the battlefield without Flying or Hexproof.
26       */
27      @Test
28      public void testCloudform() {
29          // At the beginning of your upkeep, if you control an artifact, put a 1/1
30          // colorless Thopter artifact creature token with flying onto the battlefield.
31          // Whenever one or more artifact creatures you control deal combat damage to a player, draw a card.
32          addCard(Zone.BATTLEFIELD, playerA, "Thopter Spy Network", 2); // {2}{U}{U}  - added to come to 5 enchantments on the battlefield
33          addCard(Zone.BATTLEFIELD, playerA, "Island", 8);
34          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
35          // At the beginning of your upkeep, you may return target enchantment card from your graveyard to the battlefield.
36          // As long as you control five or more enchantments, each other non-Aura enchantment you control is a creature in
37          // addition to its other types and has base power and base toughness each equal to its converted mana cost.
38          addCard(Zone.HAND, playerA, "Starfield of Nyx"); // "{4}{W}"
39          // When Cloudform enters the battlefield, it becomes an Aura with enchant creature.
40          // Manifest the top card of your library and attach Cloudform to it.
41          // Enchanted creature has flying and hexproof.
42          addCard(Zone.HAND, playerA, "Cloudform"); // {1}{U}{U}
43          addCard(Zone.GRAVEYARD, playerA, "Cloudform");
44  
45          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Starfield of Nyx", true);
46          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cloudform");
47  
48          // addTarget(playerA, "Cloudform"); Autochosen, only target
49  
50          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
51          execute();
52  
53          assertGraveyardCount(playerA, "Thopter Spy Network", 0);
54          assertPowerToughness(playerA, EmptyNames.FACE_DOWN_CREATURE.toString(),
55                  2, 2, Filter.ComparisonScope.All); // the manifested cards
56          assertPermanentCount(playerA, "Starfield of Nyx", 1);
57          assertPowerToughness(playerA, "Thopter Spy Network", 4, 4, Filter.ComparisonScope.All);
58          assertPermanentCount(playerA, "Cloudform", 2);
59      }

61      /**
62       * Regarding Starfield of Nyx: I tried to bring back a Singing Bell Strike
63       * to enchant Silumgar, the Drifting Death. I chose the Aura, but instead of
64       * enchanting Silumgar, it stayed in the graveyard. This shouldn't be the
65       * case, because Auras only target when they're cast as spells.
66       */
67      @Test
68      public void testHexproof() {
69          // At the beginning of your upkeep, if you control an artifact, put a 1/1 colorless
70          // Thopter artifact creature token with flying onto the battlefield.
71          // Whenever one or more artifact creatures you control deal combat damage to a player, draw a card.
72          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
73          // At the beginning of your upkeep, you may return target enchantment card from your graveyard to the battlefield.
74          // As long as you control five or more enchantments, each other non-Aura enchantment you control is a creature in
75          // addition to its other types and has base power and base toughness each equal to its converted mana cost.
76          addCard(Zone.HAND, playerA, "Starfield of Nyx"); // "{4}{W}"
77          // Enchant creature
78          // When Singing Bell Strike enters the battlefield, tap enchanted creature.
79          // Enchanted creature doesn't untap during its controller's untap step.
80          // Enchanted creature has "{6}: Untap this creature."
81          addCard(Zone.GRAVEYARD, playerA, "Singing Bell Strike");
82  
83          addCard(Zone.BATTLEFIELD, playerB, "Silumgar, the Drifting Death", 1);
84  
85          setStrictChooseMode(true);
86  
87          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Starfield of Nyx");
88          setChoice(playerA, "Yes");
89          addTarget(playerA, "Singing Bell Strike");
90          setChoice(playerA, "Silumgar, the Drifting Death");
91  
92          setStopAt(3, PhaseStep.PRECOMBAT_MAIN);
93          execute();
94  
95          assertPermanentCount(playerA, "Starfield of Nyx", 1);
96          assertPermanentCount(playerA, "Singing Bell Strike", 1);
97  
98          Permanent enchantment = getPermanent("Singing Bell Strike", playerA);
99          if (enchantment != null && enchantment.getAttachedTo() != null) {
100             Permanent enchanted = currentGame.getPermanent(enchantment.getAttachedTo());
101             Assert.assertEquals("Silumgar was enchanted", "Silumgar, the Drifting Death", enchanted.getName());
102         } else {
103             Assert.fail("Singing Bell Strike not on the battlefield");
104         }
105     }

107     @Test
108     public void testStarfieldOfNyxLayers() {
109 
110         addCard(Zone.BATTLEFIELD, playerA, "Starfield of Nyx"); // enchantments you control become creatures
111         addCard(Zone.BATTLEFIELD, playerA, "Humility"); // creatures lose all abilities and are 1/1
112         addCard(Zone.BATTLEFIELD, playerA, "Master of the Feast"); // enchantment creature 5/5
113         addCard(Zone.BATTLEFIELD, playerA, "Emrakul, the Aeons Torn"); //15/15 creature
114         addCard(Zone.BATTLEFIELD, playerA, "Crusade", 4); // enchantments to fulfill requirement of Starfield of Nyx
115 
116         setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
117         execute();
118 
119         assertPowerToughness(playerA, "Master of the Feast", 3, 3, Filter.ComparisonScope.All);
120         assertPowerToughness(playerA, "Humility", 4, 4, Filter.ComparisonScope.All);
121         // Humility loses its ability in layer 6.  Layer 7 never gets Humility's effect
122         assertPowerToughness(playerA, "Emrakul, the Aeons Torn", 15, 15, Filter.ComparisonScope.All); // PT not affected
123         Permanent emrakul = getPermanent("Emrakul, the Aeons Torn", playerA.getId());
124         Assert.assertNotNull(emrakul);
125         Assert.assertFalse(emrakul.getAbilities().contains(FlyingAbility.getInstance())); // loses flying though
126 
127     }

129     /**
130      * So Starfield of Nyx in play. Detention Sphere with a Song of the Dryads
131      * under it. Wrath of god on the stack. In response I Parallax Wave Honden
132      * of life's Web, Mirrari's Wake, Sphere of Safety, Aura Shards, and
133      * Enchantress's Presence to save them. Wrath resolves and Song of the
134      * Dryads come back along with the 5 named enchantments. Opp targets
135      * Starfield of Nyx with Song of the Dryads. When song of the dryads
136      * attaches all my enchantments stay creatures but as 1/1's instead of cmc.
137      * I untap draw and cast Humility. All my 1/1 enchantments die while opp's
138      * bruna, light of alabaster keeps her abilities. After mirrari's wake dies
139      * due to this I still have double mana. So yea, something broke big time
140      * there.
141      */
142     @Test
143     public void testStarfieldOfNyxAndSongOfTheDryads() {
144         // Nontoken creatures you control get +1/+1 and have vigilance.
145         addCard(Zone.BATTLEFIELD, playerA, "Always Watching", 5); // Enchantment {1}{W}{W}
146         addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
147         // At the beginning of your upkeep, you may return target enchantment card from your graveyard to the battlefield.
148         // As long as you control five or more enchantments, each other non-Aura enchantment you control is a creature in
149         // addition to its other types and has base power and base toughness each equal to its converted mana cost.
150         addCard(Zone.HAND, playerA, "Starfield of Nyx"); // "{4}{W}"
151 
152         addCard(Zone.BATTLEFIELD, playerB, "Forest", 3);
153         // Enchanted permanent is a colorless Forest land.
154         addCard(Zone.HAND, playerB, "Song of the Dryads"); // Enchant Permanent {2}{G}
155 
156         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Starfield of Nyx");
157         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Song of the Dryads", "Starfield of Nyx");
158 
159         setStopAt(2, PhaseStep.BEGIN_COMBAT);
160 
161         setStrictChooseMode(true);
162         execute();
163 
164         assertPermanentCount(playerA, "Always Watching", 5);
165         assertPermanentCount(playerB, "Song of the Dryads", 1);
166 
167         assertType("Always Watching", CardType.ENCHANTMENT, true);
168         assertType("Always Watching", CardType.CREATURE, false);
169         assertPowerToughness(playerA, "Always Watching", 0, 0, Filter.ComparisonScope.All);
170 
171         assertType("Starfield of Nyx", CardType.LAND, SubType.FOREST);
172     }

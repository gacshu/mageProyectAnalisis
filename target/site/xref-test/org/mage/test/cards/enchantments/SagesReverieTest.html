10   *

12   */
13  public class SagesReverieTest extends CardTestPlayerBase {
14  
15      /**
16       * 21:11: MarioPineda casts Sage's Reverie [26d] targeting face down creature
17       * 21:11: Ability triggers: Sage's Reverie [26d] - When Sage's Reverie [26d] enters the battlefield, draw a card for each aura you control that's attached to a creature.
18       * 21:11: mbvash casts Crackling Doom [b78]
19       * 21:11: MarioPineda loses 2 life
20       * 21:11: MarioPineda sacrificed face down creature
21       * 21:11: mbvash puts Crackling Doom [b78] from stack into their graveyard
22       * 21:11: Cloudform [9cd] is put into graveyard from battlefield
23       * 21:11: Sage's Reverie [26d] is put into graveyard from battlefield
24       * 21:11: MarioPineda draws two cards
25       *
26       * There were two other Auras on the battlefield, and Sage's Reverie made me draw two cards even though the creature it was going to enchant left the battlefield.
27       *
28       * http://www.mtgsalvation.com/forums/magic-fundamentals/magic-rulings/604851-sages-reverie-question
29       */
30      @Test
31      public void testNoCardDrawIfTargetIllegal() {
32          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
33          addCard(Zone.BATTLEFIELD, playerA, "Pillarfield Ox", 1);
34          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
35          // Enchant creature
36          // When Sage's Reverie enters the battlefield, draw a card for each aura you control that's attached to a creature.
37          addCard(Zone.HAND, playerA, "Sage's Reverie", 1); // {3}{W}
38          // Enchant creature
39          // Enchanted creature has lifelink
40          addCard(Zone.HAND, playerA, "Lifelink", 1); // {W}
41  
42          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 3);
43          // Destroy target creature or planeswalker.
44          addCard(Zone.HAND, playerB, "Hero's Downfall"); // {1}{B}{B}
45  
46          setStrictChooseMode(true);
47  
48          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lifelink", "Silvercoat Lion");
49          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Sage's Reverie", "Pillarfield Ox");
50          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Hero's Downfall", "Pillarfield Ox", "Sage's Reverie");
51  
52          setStopAt(1, PhaseStep.END_TURN);
53          execute();
54  
55          assertPermanentCount(playerA, "Lifelink", 1);
56          assertGraveyardCount(playerB, "Hero's Downfall", 1);
57          assertGraveyardCount(playerA, "Pillarfield Ox", 1);
58          assertGraveyardCount(playerA, "Sage's Reverie", 1);
59  
60          assertHandCount(playerA, 0);
61      }

14      /**
15       * I enchanted Bruna with Alexi's Cloak, giving it shroud. Then, it did not
16       * allow me to attach any auras with the triggered ability when I attacked
17       * with Bruna. Here is the ruling from the judges:
18       *
19       * I have a bruna, light of alebaster with shroud. can i still attach auras
20       * to it by the triggered ability? Can Chandra, Torch of Defiance's +1
21       * damage be redirected to a planeswalker? Just asking because it says "each
22       * opponent" Yes. If you're not casting an Aura as an Aura spell, you don't
23       * need to target. See !303.4f 303.4f. If an Aura is entering the
24       * battlefield under a player's control by any means other than by resolving
25       * as an Aura spell, and the effect putting it onto the battlefield doesn't
26       * specify the object or player the Aura will enchant, that player chooses
27       * what it will enchant as the Aura enters the battlefield. The player must
28       * choose a legal object or player according to the Aura's enchant ... ...
29       * ability and any other applicable effects.
30       */
31      @Test
32      public void testOneAttackerDamage() {
33  
34          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
35          // Flash
36          // Enchant creature
37          // Enchanted creature has shroud.
38          addCard(Zone.HAND, playerB, "Alexi's Cloak", 1); //Enchantment {1}{U}
39          // Whenever Bruna, Light of Alabaster attacks or blocks, you may attach to it any number of
40          // Auras on the battlefield and you may put onto the battlefield attached to it any number
41          // of Aura cards that could enchant it from your graveyard and/or hand.
42          addCard(Zone.BATTLEFIELD, playerB, "Bruna, Light of Alabaster", 1); // Creature 5/5 {3}{W}{W}{U}
43          // Enchant creature
44          //Enchanted creature gets +2/+1.
45          addCard(Zone.GRAVEYARD, playerB, "Unholy Strength", 1);
46          // Enchant land
47          // Enchanted land has ": Target creature can't block this turn."
48          addCard(Zone.GRAVEYARD, playerB, "Hostile Realm", 1);
49  
50          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Alexi's Cloak", "Bruna, Light of Alabaster");
51          attack(2, playerB, "Bruna, Light of Alabaster");
52          setChoice(playerB, "Yes");
53          setChoice(playerB, "Yes");
54  
55  
56          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
57          execute();
58  
59          assertPermanentCount(playerB, "Alexi's Cloak", 1);
60          assertPermanentCount(playerB, "Unholy Strength", 1);
61          assertPermanentCount(playerB, "Hostile Realm", 0);
62  
63          assertGraveyardCount(playerB, "Hostile Realm", 1);
64  
65          assertLife(playerA, 13);
66      }

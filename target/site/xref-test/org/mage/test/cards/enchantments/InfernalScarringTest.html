4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class InfernalScarringTest extends CardTestPlayerBase {
14  
15      /**
16       * The 'draw a card, when this creature dies' didn't trigger after i
17       * sacrificed a creature enchanted with infernal scarring (enchanted
18       * creature was Fetid Imp and the sac trigger came from a Nantuko Husk).
19       *
20       */
21      @Test
22      public void testDiesTrigger() {
23          // Sacrifice a creature: Nantuko Husk gets +2/+2 until end of turn.
24          addCard(Zone.BATTLEFIELD, playerA, "Nantuko Husk", 1); // {2}{B}
25          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
26          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
27          // Enchant creature
28          // Enchanted creature gets +2/+0 and has "When this creature dies, draw a card."
29          addCard(Zone.HAND, playerA, "Infernal Scarring"); // {1}{B}
30  
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Infernal Scarring", "Silvercoat Lion");
32  
33          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Sacrifice a creature");
34          setChoice(playerA, "Silvercoat Lion");
35  
36          setStopAt(1, PhaseStep.END_TURN);
37          execute();
38  
39          assertGraveyardCount(playerA, "Silvercoat Lion", 1);
40          assertPowerToughness(playerA, "Nantuko Husk", 4, 4);
41          assertGraveyardCount(playerA, "Infernal Scarring", 1);
42          assertHandCount(playerA, 1);
43      }

4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class CantBeTargetTest extends CardTestPlayerBase {
14  
15      /**
16       * Test that the target of Vines of Vastwood can't
17       * be the target of spells or abilities your opponents control this turn 
18       */
19      @Test
20      public void testVinesofVastwood() {
21          addCard(Zone.HAND, playerA, "Divine Favor");
22          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
23          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
24  
25          addCard(Zone.HAND, playerB, "Vines of Vastwood");
26          addCard(Zone.BATTLEFIELD, playerB, "Forest", 1);
27          
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Divine Favor", "Silvercoat Lion");
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Vines of Vastwood", "Silvercoat Lion", "Divine Favor");
30          setChoice(playerB, false); // don't use kicker
31          setStopAt(1, PhaseStep.BEGIN_COMBAT);
32          execute();
33  
34          assertPowerToughness(playerA, "Silvercoat Lion", 2, 2);
35          assertGraveyardCount(playerA, "Divine Favor", 1);
36  
37          assertLife(playerA, 20);
38          assertLife(playerB, 20);
39  
40      }
41  }

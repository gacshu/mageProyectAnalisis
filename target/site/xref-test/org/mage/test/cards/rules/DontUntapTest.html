16      /**
17       * Test that the attackers blocked by creatures boosted with
18       * Triton Tactics do not untap in their controllers next untap step
19       */
20      @Test
21      public void testTritonTactics() {
22          // Up to two target creatures each get +0/+3 until end of turn. Untap those creatures.
23          // At this turn's next end of combat, tap each creature that was blocked by one of those
24          // creatures this turn and it doesn't untap during its controller's next untap step.        
25          addCard(Zone.HAND, playerA, "Triton Tactics");
26          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
27          // {T}: You gain 1 life.
28          addCard(Zone.BATTLEFIELD, playerA, "Soulmender", 1);
29          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
30          
31          activateAbility(4, PhaseStep.POSTCOMBAT_MAIN, playerA, "{T}: You gain 1 life");
32          
33          attack(4, playerB, "Silvercoat Lion");        
34          castSpell(4, PhaseStep.DECLARE_ATTACKERS, playerA, "Triton Tactics", "Soulmender");
35          block(4, playerA, "Soulmender", "Silvercoat Lion");
36          
37          setStopAt(6, PhaseStep.DRAW);
38          execute();
39  
40          assertGraveyardCount(playerA, "Triton Tactics", 1);
41  
42          assertPowerToughness(playerA, "Soulmender", 1, 1);
43          
44          assertPowerToughness(playerB, "Silvercoat Lion", 2, 2);
45          assertTapped("Silvercoat Lion", true); // Should be stilled tapped in turn 6 because it was blocked in turn 4 with Triton Tactics
46          
47          assertLife(playerA, 21);
48          assertLife(playerB, 20);
49  
50      }

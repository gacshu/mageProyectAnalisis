27  
28          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);

30          addCard(Zone.HAND, playerA, "Jayemdae Tome", 1);

32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Jayemdae Tome"); // {4}

36          try {
37              execute();
38  
39              Assert.fail("must throw exception on execute");

42                  Assert.fail("must not have throw error about bad targets, but got:\n" + e.getMessage());
43              }
44          }

171     /**
172      * Test that panic can only be cast during the correct phase/step
173      */
174     @Test
175     public void testPanic() {
176         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
177 
178         // Cast Panic only during combat before blockers are declared.
179         // Target creature can't block this turn.
180         // Draw a card at the beginning of the next turn's upkeep.
181         addCard(Zone.HAND, playerA, "Panic", 2); // Instant - {R}
182 
183         addCard(Zone.BATTLEFIELD, playerA, "Akroan Conscriptor", 1);
184         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
185 
186         checkPlayableAbility("not in precombat", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Panic", false);
187         attack(1, playerA, "Akroan Conscriptor");
188         castSpell(1, PhaseStep.DECLARE_ATTACKERS, playerA, "Panic", "Silvercoat Lion");
189         checkPlayableAbility("not on declare blockers", 1, PhaseStep.DECLARE_BLOCKERS, playerA, "Cast Panic", false);
190         checkPlayableAbility("not in postcombat", 1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Cast Panic", false);
191 
192         setStopAt(2, PhaseStep.PRECOMBAT_MAIN);
193         execute();
194 
195         assertHandCount(playerA, "Panic", 1);
196         assertHandCount(playerA, 2);
197         assertGraveyardCount(playerA, "Panic", 1);
198     }

200     /**
201      * I "Aether Vialed" (Aether Vial) an Ethersworn Canonist into the
202      * battlefield in response to a colored spell(an Elf). The Canonist entered
203      * the battlefield. Then, my oponente used Abrupt Decay to destroy it and
204      * continue to play spells normaly. Ethersworn Canonist effect should
205      * imediately effect the game as it entered the battlefield, and still count
206      * spells cast earlier in the turn. In other words, my oponent shouldn't be
207      * able to cast anymore colored spells.
208      */
209     @Test
210     public void testEtherswornCanonist() {
211         // Each player who has cast a nonartifact spell this turn can't cast additional nonartifact spells.
212         addCard(Zone.HAND, playerA, "Ethersworn Canonist", 4); // Creaturre - {1}{W}
213 
214         // At the beginning of your upkeep, you may put a charge counter on Aether Vial.
215         // {T}: You may put a creature card with converted mana cost equal to the number of charge counters on Aether Vial from your hand onto the battlefield.
216         addCard(Zone.BATTLEFIELD, playerA, "Aether Vial", 1);
217         // addCounters(1, PhaseStep.UPKEEP, playerA, "Aether Vial", CounterType.CHARGE, 1);
218 
219         addCard(Zone.HAND, playerB, "Llanowar Elves", 1); // Creature {G}
220         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 1);
221         addCard(Zone.BATTLEFIELD, playerB, "Forest", 2);
222 
223         // Abrupt Decay can't be countered.
224         // Destroy target nonland permanent with converted mana cost 3 or less.
225         addCard(Zone.HAND, playerB, "Abrupt Decay", 1); // {B}{G}
226 
227         castSpell(4, PhaseStep.PRECOMBAT_MAIN, playerB, "Llanowar Elves");
228 
229         activateAbility(4, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: You");
230         setChoice(playerB, "Ethersworn Canonist");
231 
232 //        checkPlayableAbility("2nd spell cast", 4, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Abrupt", false);
233         castSpell(4, PhaseStep.POSTCOMBAT_MAIN, playerB, "Abrupt Decay", "Ethersworn Canonist");
234         setStopAt(4, PhaseStep.END_TURN);
235 
236         try {
237             execute();
238 
239             Assert.fail("must throw exception on execute");
240         } catch (Throwable e) {
241             if (!e.getMessage().contains("Cast Abrupt Decay$target=Ethersworn Canonist")) {
242                 Assert.fail("must not have throw error about bad targets, but got:\n" + e.getMessage());
243             }
244         }
245 
246         assertCounterCount(playerA, "Aether Vial", CounterType.CHARGE, 2);
247         assertPermanentCount(playerB, "Llanowar Elves", 1);
248 
249         assertPermanentCount(playerA, "Ethersworn Canonist", 1);
250         assertHandCount(playerB, "Abrupt Decay", 1);
251 
252     }

262         // Your opponents can't cast spells with the chosen name.

264         addCard(Zone.BATTLEFIELD, playerA, "Island", 7);
265 
266         // Destroy all creatures. They can't be regenerated.
267         addCard(Zone.HAND, playerB, "Damnation", 1); // SORCERY {2}{B}{B}
268         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 4);
269 
270         setStrictChooseMode(true);
271 
272         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Alhammarret, High Arbiter");
273         addTarget(playerA, "Damnation");
274 
275         checkPlayableAbility("damnation check", 2, PhaseStep.PRECOMBAT_MAIN, playerB, "Cast Damnation", false);
276         setStopAt(2, PhaseStep.BEGIN_COMBAT);
277         execute();
278 
279         assertPermanentCount(playerA, "Alhammarret, High Arbiter", 1);

16      /**
17       * Venser The Sojourner: -8: You get an emblem with "Whenever you cast a
18       * spell, exile target permanent."
19       */
20      @Test
21      public void testVenserTheSojournerEmblem() {
22          addCard(Zone.BATTLEFIELD, playerA, "Venser, the Sojourner");
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
24          addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
25          addCard(Zone.BATTLEFIELD, playerB, "Elite Vanguard");
26  
27          addCounters(1, PhaseStep.UPKEEP, playerA, "Venser, the Sojourner", CounterType.LOYALTY, 5);
28          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-8: You get an emblem");
29          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
30  
31          setStopAt(1, PhaseStep.END_TURN);
32          execute();
33  
34          assertLife(playerB, 17);
35  
36          assertGraveyardCount(playerA, "Venser, the Sojourner", 1);
37          assertEmblemCount(playerA, 1);
38          // should be exiled using emblem ability
39          assertPermanentCount(playerB, "Elite Vanguard", 0);
40      }

54          addCounters(1, PhaseStep.UPKEEP, playerA, "Sorin, Lord of Innistrad", CounterType.LOYALTY, 1);

59          setStopAt(3, PhaseStep.END_TURN);

96       * Tests "Whenever a card is put into your graveyard from anywhere, you may

98       */
99      @Test
100     public void testTamiyoTheMoonSageSecondEmblem() {
101         addCard(Zone.BATTLEFIELD, playerA, "Tamiyo, the Moon Sage");
102         addCard(Zone.BATTLEFIELD, playerA, "Elite Vanguard");
103         addCard(Zone.BATTLEFIELD, playerB, "Mountain");
104         addCard(Zone.HAND, playerB, "Lightning Bolt");
105 
106         addCounters(1, PhaseStep.UPKEEP, playerA, "Tamiyo, the Moon Sage", CounterType.LOYALTY, 4);
107         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "-8: You get an emblem");
108         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", "Elite Vanguard");
109 
110         setStopAt(2, PhaseStep.BEGIN_COMBAT);
111         execute();
112 
113         assertEmblemCount(playerA, 1);
114         assertPermanentCount(playerA, "Elite Vanguard", 0);
115         assertHandCount(playerA, 1);
116 
117         boolean found = false;
118         for (Card card : playerA.getHand().getCards(currentGame)) {
119             if (card.getName().equals("Elite Vanguard")) {
120                 found = true;
121             }
122         }
123         Assert.assertTrue("Couldn't find a card in hand: Elite Vanguard", found);
124     }

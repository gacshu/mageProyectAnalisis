15      /**
16       * Bane Alley Broker Creature â€” Human Rogue 0/3, 1UB (3) {T}: Draw a card,
17       * then exile a card from your hand face down. You may look at cards exiled
18       * with Bane Alley Broker. {U}{B}, {T}: Return a card exiled with Bane Alley
19       * Broker to its owner's hand.
20       */
21      // test that cards exiled using Bane Alley Broker are face down
22      @Test
23      public void testBaneAlleyBroker() {
24          addCard(Zone.BATTLEFIELD, playerA, "Bane Alley Broker");
25          addCard(Zone.HAND, playerA, "Goblin Roughrider");
26          addCard(Zone.HAND, playerA, "Sejiri Merfolk");
27          addCard(Zone.BATTLEFIELD, playerA, "Island");
28          addCard(Zone.BATTLEFIELD, playerA, "Swamp");
29  
30          setStrictChooseMode(true);
31          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}");
32          addTarget(playerA, "Goblin Roughrider");
33  
34          setStopAt(1, PhaseStep.BEGIN_COMBAT);
35          execute();
36  
37          assertHandCount(playerA, 2);
38          assertHandCount(playerA, "Sejiri Merfolk", 1);
39          assertHandCount(playerA, "Goblin Roughrider", 0);
40  
41          assertExileCount("Goblin Roughrider", 1);
42  
43          for (Card card : currentGame.getExile().getAllCards(currentGame)) {
44              if (card.getName().equals("Goblin Roughrider")) {
45                  Assert.assertTrue("Exiled card is not face down", card.isFaceDown(currentGame));
46              }
47          }
48      }

13  public class MomentousFallTest extends CardTestPlayerBase {

15      /**
16       *  Momentous Fall
17       *  Instant {2}{G}{G}
18       *  As an additional cost to cast Momentous Fall, sacrifice a creature.
19       *  You draw cards equal to the sacrificed creature's power, then you
20       *  gain life equal to its toughness.
21       */
22      @Test
23      public void testSacrificeCostAndLKI() {
24          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
25          addCard(Zone.HAND, playerA, "Momentous Fall");
26  
27          // Geralf's Messenger enters the battlefield tapped.
28          // When Geralf's Messenger enters the battlefield, target opponent loses 2 life.
29          // Undying (When this creature dies, if it had no +1/+1 counters on it, return it to the battlefield under its owner's control with a +1/+1 counter on it.)
30          addCard(Zone.BATTLEFIELD, playerA, "Geralf's Messenger", 1);
31  
32          // Creatures you control get +1/+1.
33          addCard(Zone.BATTLEFIELD, playerA, "Glorious Anthem", 1);
34  
35          // As an additional cost to cast Momentous Fall, sacrifice a creature.
36          // You draw cards equal to the sacrificed creature's power, then you gain life equal to its toughness.
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Momentous Fall");
38  
39          setStopAt(1, PhaseStep.BEGIN_COMBAT);
40          execute();
41  
42          assertPermanentCount(playerA, "Geralf's Messenger", 1);
43          assertCounterCount("Geralf's Messenger", CounterType.P1P1, 1);
44          assertPowerToughness(playerA, "Geralf's Messenger", 5, 4); // +1/+1 counter + Anthem effect
45          assertHandCount(playerA, 4); // +4 cards
46          assertLife(playerA, 23); // +3 life
47      }
48  
49      @Test
50      public void testSacrificeCostForProGreen() {
51          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
52          addCard(Zone.HAND, playerA, "Momentous Fall");
53          addCard(Zone.BATTLEFIELD, playerA, "Mirran Crusader");
54  
55          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Momentous Fall");
56  
57          setStopAt(1, PhaseStep.BEGIN_COMBAT);
58          execute();
59  
60          assertPermanentCount(playerA, "Mirran Crusader", 0);
61          assertHandCount(playerA, 2); // +2 cards
62          assertLife(playerA, 22); // +2 life
63      }

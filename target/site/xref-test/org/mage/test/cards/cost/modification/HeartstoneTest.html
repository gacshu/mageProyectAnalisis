13  public class HeartstoneTest extends CardTestPlayerBase {
14  
15      /**
16       * Heartstone is not reducing the cost of Sigil Tracer 's activated ability
17       */
18      @Test
19      public void testSigilTracer() {
20          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
21          // {1}{U}, Tap two untapped Wizards you control: Copy target instant or sorcery spell. You may choose new targets for the copy.
22          addCard(Zone.BATTLEFIELD, playerA, "Sigil Tracer");
23          // Activated abilities of creatures cost {1} less to activate. This effect can't reduce the amount of mana an ability costs to activate to less than one mana.
24          addCard(Zone.BATTLEFIELD, playerA, "Heartstone");
25          addCard(Zone.BATTLEFIELD, playerA, "Fugitive Wizard");
26  
27          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
28          addCard(Zone.HAND, playerB, "Lightning Bolt");
29  
30          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", playerA);
31          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}{U}", "Lightning Bolt", "Lightning Bolt");
32          setChoice(playerA, true);
33          addTarget(playerA, playerB);
34          setStopAt(1, PhaseStep.BEGIN_COMBAT);
35          execute();
36  
37          assertGraveyardCount(playerB, "Lightning Bolt", 1);
38  
39          assertTapped("Fugitive Wizard", true);
40          assertLife(playerA, 17);
41          assertLife(playerB, 17);
42  
43          assertTappedCount("Island", true, 1);
44  
45      }

11  /**
12   * Reported bugs: https://github.com/magefree/mage/issues/8980
13   *      Cards like Soldevi Excavations, Heart of Yavimaya, and Lake of the Dead
14   *      will sometimes immediately go to the graveyard without asking the controller if they wish to sacrifice a land
15   *      even though they had one available
16   * @author Alex-Vasile
17   */
18  public class SacrificeLandTest extends CardTestPlayerBase {
19  
20      /**
21       * Perhaps it is a rollback issue with saving state
22       */
23      @Test
24      public void testRollback() {
25          // If Soldevi Excavations would enter the battlefield, sacrifice an untapped Island instead.
26          // If you do, put Soldevi Excavations onto the battlefield. If you don't, put it into its owner's graveyard.
27          String soldeviExcavations = "Soldevi Excavations";
28  
29          addCard(Zone.HAND, playerA, soldeviExcavations);
30          addCard(Zone.BATTLEFIELD, playerA, "Island");
31  
32          Random random = new Random();
33          boolean sacFirstLand = random.nextBoolean();
34          boolean sacSecondLand = random.nextBoolean();
35  
36          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, soldeviExcavations);
37          setChoice(playerA, sacFirstLand);
38  
39          rollbackTurns(1, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
40  
41          rollbackAfterActionsStart();
42          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, soldeviExcavations);
43          setChoice(playerA, sacSecondLand);
44          rollbackAfterActionsEnd();
45  
46          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
47          execute();
48  
49          assertPermanentCount(playerA, soldeviExcavations, sacSecondLand ? 1 : 0);
50          assertGraveyardCount(playerA, "Island", sacSecondLand ? 1 : 0);
51      }
52  
53      /**
54       * Perhaps it will come up with playing multiple on seperate turns
55       */
56      @Test
57      public void testMultiple() {
58          // If Soldevi Excavations would enter the battlefield, sacrifice an untapped Island instead.
59          // If you do, put Soldevi Excavations onto the battlefield. If you don't, put it into its owner's graveyard.
60          String soldeviExcavations = "Soldevi Excavations";
61          // Sacrifice a Forest
62          String heartofYavimaya = "Heart of Yavimaya";
63          // Sacrifice a Swamp
64          String lakeOfTheDead = "Lake of the Dead";
65  
66          addCard(Zone.HAND, playerA, soldeviExcavations);
67          addCard(Zone.BATTLEFIELD, playerA, "Island");
68  
69          addCard(Zone.HAND, playerA, heartofYavimaya);
70          addCard(Zone.BATTLEFIELD, playerA, "Forest");
71  
72          addCard(Zone.HAND, playerA, lakeOfTheDead);
73          addCard(Zone.BATTLEFIELD, playerA, "Swamp");
74  
75          playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, soldeviExcavations);
76          setChoice(playerA, "Yes");
77  
78          playLand(3, PhaseStep.PRECOMBAT_MAIN, playerA, heartofYavimaya);
79          setChoice(playerA, "Yes");
80  
81          playLand(5, PhaseStep.PRECOMBAT_MAIN, playerA, lakeOfTheDead);
82          setChoice(playerA, "Yes");
83  
84          setStopAt(5, PhaseStep.PRECOMBAT_MAIN);
85          execute();
86  
87          assertPermanentCount(playerA, soldeviExcavations, 1);
88          assertGraveyardCount(playerA, "Island", 1);
89  
90          assertPermanentCount(playerA, heartofYavimaya, 1);
91          assertGraveyardCount(playerA, "Forest", 1);
92  
93          assertPermanentCount(playerA, lakeOfTheDead, 1);
94          assertGraveyardCount(playerA, "Swamp", 1);
95      }

16      private void testReduce(String sourceCost, int reduceAmount, String needReducedCost) {
17          // load mana by sctrict mode (e.g. for real mana usage in param)
18          ManaCosts<ManaCost> source = new ManaCostsImpl<>();
19          source.load(sourceCost, true);
20          ManaCosts<ManaCost> need = new ManaCostsImpl<>();
21          need.load(needReducedCost, true);
22          ManaCosts<ManaCost> reduced = CardUtil.reduceCost(source, reduceAmount);
23  
24          if (!reduced.getText().equals(need.getText())) {
25              Assert.fail(sourceCost + " after reduction by " + reduceAmount + " must be " + need.getText() + ", but get " + reduced.getText());
26          }
27      }

269     /**
270      * If a spell has an additional cost (optional or mandatory, e.g. Entwine),
271      * Omniscience incorrectly allows you cast the spell as if that cost had
272      * been paid without paying that spell's mana cost. 117.9d If an alternative
273      * cost is being paid to cast a spell, any additional costs, cost increases,
274      * and cost reductions that affect that spell are applied to that
275      * alternative cost. (See rule 601.2f.)
276      */
277     @Test
278     public void testCastingWithCyclonicRiftWithOverload() {
279         // You may cast nonland cards from your hand without paying their mana costs.
280         addCard(Zone.BATTLEFIELD, playerA, "Omniscience");
281         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
282 
283         // Choose one - Barbed Lightning deals 3 damage to target creature; or Barbed Lightning deals 3 damage to target player.
284         // Entwine {2} (Choose both if you pay the entwine cost.)
285         addCard(Zone.HAND, playerA, "Barbed Lightning", 1); // {2}{R}
286 
287         // Creature - 3/3 Swampwalk
288         addCard(Zone.BATTLEFIELD, playerB, "Bog Wraith", 1);
289 
290         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Barbed Lightning");
291         setChoice(playerA, true); // cast without cost
292         setChoice(playerA, true); // pay Entwine
293         addTarget(playerA, "Bog Wraith"); // target form mode 1
294         addTarget(playerA, playerB); // target for mode 2
295 
296         showBattlefield("after", 1, PhaseStep.POSTCOMBAT_MAIN, playerA);
297 
298         setStrictChooseMode(true);
299         setStopAt(1, PhaseStep.END_TURN);
300         execute();
301 
302         assertGraveyardCount(playerA, "Barbed Lightning", 1);
303         assertGraveyardCount(playerB, "Bog Wraith", 1);
304 
305         assertLife(playerA, 20);
306         assertLife(playerB, 17);
307 
308         assertTappedCount("Plains", true, 2); // plains have to be tapped because {2} from Entwine have to be paid
309     }

311     /**
312      * If a spell has an unpayable cost (e.g. Ancestral Vision, which has no mana cost),
313      * Omniscience should allow you to cast that spell without paying its mana cost.
314      * In the case of Ancestral Vision, for example, Xmage only gives you the option to suspend Ancestral Vision.
315      *
316      * 118.6a   If an unpayable cost is increased by an effect or an additional cost is imposed,
317      *          the cost is still unpayable.
318      *          If an alternative cost is applied to an unpayable cost, including an effect that allows a player
319      *          to cast a spell without paying its mana cost, the alternative cost may be paid.
320      */
321     @Test
322     public void testCastingUnpayableCost() {
323         // You may cast nonland cards from your hand without paying their mana costs.
324         addCard(Zone.BATTLEFIELD, playerA, "Omniscience");
325 
326         // Suspend 4-{U}
327         // Target player draws three cards.
328         addCard(Zone.HAND, playerA, "Ancestral Vision", 1);
329 
330         setStrictChooseMode(true);
331         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ancestral Vision", playerA);
332         setChoice(playerA, "Yes");
333 
334         setStopAt(1, PhaseStep.BEGIN_COMBAT);
335         execute();
336 
337         assertGraveyardCount(playerA, "Ancestral Vision", 1);
338

340         assertLife(playerA, 20);

383     @Test
384     public void testJelevaCastingSavageBeatingFromExile() {
385 
386         /*
387         Jeleva, Nephalia's Scourge {1}{U}{B}{R}
388         Legendary Creature - Vampire Wizard 1/3
389         Flying
390         When Jeleva, Nephalia's Scourge enters the battlefield, each player exiles the top X cards of their library, where X is the amount of mana spent to cast Jeleva.
391         Whenever Jeleva attacks, you may cast an instant or sorcery card exiled with it without paying its mana cost.
392          */
393         String jeleva = "Jeleva, Nephalia's Scourge";
394 
395         /*
396         Savage Beating {3}{R}{R}
397         Instant
398         Cast Savage Beating only during your turn and only during combat.
399         Choose one -
400         - Creatures you control gain double strike until end of turn.
401         - Untap all creatures you control. After this phase, there is an additional combat phase.
402         Entwine {1}{R} (Choose both if you pay the entwine cost.)
403          */
404         String savageBeating = "Savage Beating";
405 
406         skipInitShuffling();
407         addCard(Zone.LIBRARY, playerA, savageBeating, 2);
408         addCard(Zone.HAND, playerA, jeleva);
409         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
410         addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
411         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
412 
413         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, jeleva);
414 
415         attack(3, playerA, jeleva);
416         setChoice(playerA, true); // opt to use Jeleva ability
417         setChoice(playerA, savageBeating); // choose to cast Savage Beating for free
418         setChoice(playerA, false); // opt not to pay entwine cost
419         setModeChoice(playerA, "1"); // use first mode of Savage Beating granting double strike
420 
421         setStopAt(3, PhaseStep.END_COMBAT);
422         execute();
423 
424         assertGraveyardCount(playerA, savageBeating, 1);
425         assertTapped(jeleva, true);
426         assertLife(playerB, 18);
427         assertAbility(playerA, jeleva, DoubleStrikeAbility.getInstance(), true);

11  public class GatekeeperOfMalakirTest extends CardTestPlayerBase {
12  
13      @Test
14      public void testCard() {
15          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
16          addCard(Zone.HAND, playerA, "Gatekeeper of Malakir");
17          addCard(Zone.BATTLEFIELD, playerB, "Llanowar Elves");
18  
19          // Kicker {B} (You may pay an additional {B} as you cast this spell.)
20          // When Gatekeeper of Malakir enters the battlefield, if it was kicked, target player sacrifices a creature.
21          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Gatekeeper of Malakir");
22          setChoice(playerA, true); // use kicker
23  
24          setStopAt(1, PhaseStep.BEGIN_COMBAT);
25          execute();
26  
27          assertLife(playerA, 20);
28          assertLife(playerB, 20);
29          assertPermanentCount(playerA, "Gatekeeper of Malakir", 1);
30          assertPermanentCount(playerB, "Llanowar Elves", 0);
31          assertGraveyardCount(playerB, 1);
32      }

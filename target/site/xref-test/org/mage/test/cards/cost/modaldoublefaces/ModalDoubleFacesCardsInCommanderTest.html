71          // Oloro, Ageless Ascetic
72          // At the beginning of your upkeep, if Oloro, Ageless Ascetic is in the command zone, you gain 2 life.
73          addCard(Zone.COMMAND, playerA, "Oloro, Ageless Ascetic");

46      @Test
47      public void test_AncientStoneIdol_AttackingWithSacrifice() {
48          // The total cost to cast a spell is locked in before you pay that cost. For example, if you control five attacking
49          // creatures, including one you can sacrifice to add {C} to your mana pool, Ancient Stone Idol costs {5} to cast.
50          // Then you can sacrifice the creature when you activate mana abilities just before paying the cost, and it still
51          // costs only {5} to cast.
52          // (2018-07-13)
53  
54          // {10}
55          // Flash
56          // This spell costs {1} less to cast for each attacking creature.
57          addCard(Zone.HAND, playerA, "Ancient Stone Idol", 1);
58          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 10 - 4);
59          //
60          // Sacrifice Blood Pet: Add {B}.
61          addCard(Zone.BATTLEFIELD, playerA, "Blood Pet", 2); // give 2 cost reduction + can be sacrificed as 2 mana
62  
63          // before
64          checkPlayableAbility("before attack", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Ancient Stone Idol", false);
65  
66          // prepare for attack
67          attack(1, playerA, "Blood Pet");
68          attack(1, playerA, "Blood Pet");
69  
70          // on attack (must automaticly sacrifice creatures as mana pay)
71          checkPlayableAbility("on attack", 1, PhaseStep.DECLARE_BLOCKERS, playerA, "Cast Ancient Stone Idol", true);
72          castSpell(1, PhaseStep.DECLARE_BLOCKERS, playerA, "Ancient Stone Idol");
73  
74          setStrictChooseMode(true);
75          setStopAt(1, PhaseStep.END_TURN);
76          execute();
77  
78          assertPermanentCount(playerA, "Ancient Stone Idol", 1);
79          assertGraveyardCount(playerA, "Blood Pet", 2);
80      }

107     @Test
108     public void test_TorgaarFamineIncarnate_SacrificeXTargets() {
109         // {6}{B}{B}
110         // As an additional cost to cast this spell, you may sacrifice any number of creatures.
111         // This spell costs {2} less to cast for each creature sacrificed this way.
112         // When Torgaar, Famine Incarnate enters the battlefield, up to one target player's life total becomes half their starting life total, rounded down.
113         addCard(Zone.HAND, playerA, "Torgaar, Famine Incarnate", 1);
114         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 8 - 4 - 2);
115         addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
116         addCard(Zone.HAND, playerA, "Balduvian Bears", 2); // give 4 cost reduction on sacrifice
117 
118         checkPlayableAbility("no cost reduction 1", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Torgaar, Famine Incarnate", false);
119 
120         // prepare creatures for reduce
121         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 4);
122         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears", true);
123         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Balduvian Bears");
124         checkPlayableAbility("no cost reduction 2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Torgaar, Famine Incarnate", false);
125 
126         // can cast on next turn
127         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Torgaar, Famine Incarnate", true);
128         setChoice(playerA, "X=2"); // two creatures sacrifice
129         setChoice(playerA, "Balduvian Bears");
130         setChoice(playerA, "Balduvian Bears");
131         addTarget(playerA, playerB); // target player for half life
132 
133         setStrictChooseMode(true);
134         setStopAt(3, PhaseStep.END_TURN);
135         execute();
136 
137         assertPermanentCount(playerA, "Torgaar, Famine Incarnate", 1);
138         assertLife(playerB, 20 / 2);
139     }

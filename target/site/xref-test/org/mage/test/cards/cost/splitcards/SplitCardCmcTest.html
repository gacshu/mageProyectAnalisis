14      /**
15      * The core of the change is that we're no longer assuming split cards have two sets of characteristics
16      * when they're not on the stack. Some characteristics have multiple pieces of information very naturally
17      * â€”Destined to Lead is an instant sorcery, the same as Ornithopter is an artifact creature. It's black and green
18      * just like Winding Constrictor because its mana cost has B and G in it. Continuing that, the mana cost combines
19      * the components, and a card asking for Destined to Lead's mana cost sees 4BG.
20      *
21      * So now, the converted mana cost question is simple: if Destined to Lead isn't on the stack, it has a converted mana cost
22      * of 6. Destined on the stack is still a black instant with a converted mana cost of 2, and Lead on the stack is still a
23      * green sorcery with a converted mana cost of 4, but Destined to Lead, any time it's not one or the other, is a black and green
24      * instant sorcery with a converted mana cost of 6.
25      */
26  
27      @Test
28      public void testSplitCardCmcInHand() {
29          // Total CMC of Failure // Comply is 3, so should be exiled by Transgress the Mind.
30          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
31          // Devoid
32          // Target player reveals their hand. You may choose a card from it with converted mana cost 3 or greater and exile that card.
33          addCard(Zone.HAND, playerA, "Transgress the Mind");
34          addCard(Zone.HAND, playerB, "Failure // Comply");
35  
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Transgress the Mind", playerB);
37          setStopAt(1, PhaseStep.BEGIN_COMBAT);
38          execute();
39  
40          assertExileCount("Failure // Comply", 1);
41      }

43      @Test
44      public void testSplitCardCmcOnStack() {
45          // Counterbalance revealing Wear // Tear counters a spell with converted mana cost 3, but not 1 or 2.
46          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
47          addCard(Zone.HAND, playerA, "Typhoid Rats"); // Creature 1/1 {B}
48  
49          // Whenever an opponent casts a spell, you may reveal the top card of your library. If you do, counter that spell
50          // if it has the same converted mana cost as the revealed card.
51          addCard(Zone.BATTLEFIELD, playerB, "Counterbalance");
52          addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
53  
54          // Wear {1}{R}
55          // Destroy target artifact.
56          // Tear {W}
57          // Destroy target enchantment.
58          addCard(Zone.LIBRARY, playerB, "Wear // Tear"); // CMC now 3
59          skipInitShuffling(); // so the set to top card stays at top
60  
61          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Typhoid Rats");
62          setChoice(playerB, true); // Reveal to Counterbalance to attempt to counter Typhoid Rats
63  
64          setStopAt(1, PhaseStep.BEGIN_COMBAT);
65          execute();
66  
67          assertLife(playerA, 20);
68          assertLife(playerB, 20);
69  
70          assertPermanentCount(playerA, "Typhoid Rats", 1);
71          assertGraveyardCount(playerA, "Typhoid Rats", 0);
72          assertGraveyardCount(playerA, 0);
73          assertGraveyardCount(playerB, 0);
74      }

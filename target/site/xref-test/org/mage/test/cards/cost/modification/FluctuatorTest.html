18       * Fluctuator makes 'Akroma's Vengeance' cyclic cost reduced to {1} Test it

21      @Test
22      public void testFluctuatorReducedBy2() {
23  
24          // Destroy all artifacts, creatures, and enchantments.
25          // Cycling ({3}, Discard this card: Draw a card.)
26          addCard(Zone.HAND, playerA, "Akroma's Vengeance");
27  
28          // Cycling abilities you activate cost you up to {2} less to activate.
29          addCard(Zone.BATTLEFIELD, playerA, "Fluctuator");
30  
31          // One mana should be enough
32          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
33  
34          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cycling");
35          setChoice(playerA, "2"); // reduce 2 generic mana
36  
37          setStopAt(1, PhaseStep.BEGIN_COMBAT);
38          execute();
39  
40          assertGraveyardCount(playerA, "Akroma's Vengeance", 1);
41          assertHandCount(playerA, 1);
42      }

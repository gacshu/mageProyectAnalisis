14      @Test
15      public void testReturnCardFromSoulfireGrandMaster() {
16          // Total CMC of Failure // Comply is 3, so should be exiled by Transgress the Mind.
17          addCard(Zone.BATTLEFIELD, playerA, "Volcanic Island", 6);
18  
19          // Lifelink
20          // Instant and sorcery spells you control have lifelink.
21          // {2}{U/R}{U/R}: The next time you cast an instant or sorcery spell from your hand this turn, put that card into your hand instead of your graveyard as it resolves.
22          addCard(Zone.BATTLEFIELD, playerA, "Soulfire Grand Master");
23  
24          // Fire - Instant {1}{R}
25          // Fire deals 2 damage divided as you choose among one or two target creatures and/or players.
26          // Ice - Instant {1}{U}
27          // Tap target permanent.
28          // Draw a card.
29          addCard(Zone.HAND, playerA, "Fire // Ice");
30  
31          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{2}{U/R}{U/R}");
32  
33          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Fire", playerB);
34          addTargetAmount(playerA, "targetPlayer=PlayerB", 2);
35  
36          setStopAt(1, PhaseStep.END_TURN);
37          setStrictChooseMode(true);
38          execute();
39  
40          assertLife(playerA, 22);
41          assertLife(playerB, 18);
42  
43          assertHandCount(playerA, "Fire // Ice", 1);
44      }

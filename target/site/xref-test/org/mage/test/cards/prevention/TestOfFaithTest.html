3   package org.mage.test.cards.prevention;
4   
5   import mage.constants.PhaseStep;
6   import mage.constants.Zone;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   * @author LevelX2
12   */
13  public class TestOfFaithTest extends CardTestPlayerBase {
14  
15      @Test
16      public void testOneAttackerOneBlockerUsingFaith() {
17          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
18          addCard(Zone.BATTLEFIELD, playerA, "Soulmender");
19          addCard(Zone.HAND, playerA, "Test of Faith");
20  
21          addCard(Zone.BATTLEFIELD, playerB, "Blur Sliver");
22  
23          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Test of Faith", "Soulmender");
24  
25          attack(2, playerB, "Blur Sliver");
26          block(2, playerA, "Soulmender", "Blur Sliver");
27  
28          setStopAt(2, PhaseStep.END_COMBAT);
29          execute();
30  
31          assertPermanentCount(playerA, "Soulmender", 1);
32          assertPowerToughness(playerA, "Soulmender", 3, 3);
33  
34          assertPermanentCount(playerB, "Blur Sliver", 1);
35      }
36  
37      @Test
38      public void testOneAttackerTwoBlockerOneUsingFaith() {
39          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
40          addCard(Zone.BATTLEFIELD, playerA, "Soulmender");
41          addCard(Zone.BATTLEFIELD, playerA, "Elvish Mystic");
42  
43          addCard(Zone.HAND, playerA, "Test of Faith");
44  
45          addCard(Zone.BATTLEFIELD, playerB, "Kalonian Tusker");
46  
47          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Test of Faith", "Soulmender");
48  
49          attack(2, playerB, "Kalonian Tusker");
50          block(2, playerA, "Elvish Mystic", "Kalonian Tusker");
51          block(2, playerA, "Soulmender", "Kalonian Tusker");
52  
53  
54          setStopAt(2, PhaseStep.END_COMBAT);
55          execute();
56  
57          assertPermanentCount(playerA, "Soulmender", 1);
58          assertPowerToughness(playerA, "Soulmender", 2, 2); // one damage was prevented so Soulmender got +1/+1
59          assertPermanentCount(playerA, "Elvish Mystic", 0);
60  
61          assertPermanentCount(playerB, "Kalonian Tusker", 1); // only 2 damage to Kalonian Tusker so he still lives
62      }
63  
64      @Test
65      public void testOneAttackerTwoBlockerTwoUsingFaith() {
66          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
67          addCard(Zone.BATTLEFIELD, playerA, "Soulmender");
68          addCard(Zone.BATTLEFIELD, playerA, "Elvish Mystic");
69  
70          addCard(Zone.HAND, playerA, "Test of Faith", 2);
71  
72          addCard(Zone.BATTLEFIELD, playerB, "Kalonian Tusker");
73  
74          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Test of Faith", "Soulmender");
75          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Test of Faith", "Elvish Mystic");
76  
77          attack(2, playerB, "Kalonian Tusker");
78          block(2, playerA, "Elvish Mystic", "Kalonian Tusker");
79          block(2, playerA, "Soulmender", "Kalonian Tusker");
80  
81  
82          setStopAt(2, PhaseStep.END_COMBAT);
83          execute();
84  
85          assertPermanentCount(playerA, "Soulmender", 1);
86          assertPowerToughness(playerA, "Soulmender", 2, 2); // one damage was prevented so Soulmender got +1/+1
87          assertPermanentCount(playerA, "Elvish Mystic", 1);
88          assertPowerToughness(playerA, "Elvish Mystic", 3, 3); // two damage were prevented so Elvish Mystic got +2/+2
89  
90          assertPermanentCount(playerB, "Kalonian Tusker", 1); // only 2 damage to Kalonian Tusker so he still lives
91      }

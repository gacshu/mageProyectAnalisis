24       * Tests "If all other elves get the Forestwalk ability and can't be blockt from creatures whose controller has a forest in game"

26  
27      @Test
28      public void testCannotBlockCreatureWithForestwalk() {
29          addCard(Zone.BATTLEFIELD, playerA, "Elvish Champion");
30          addCard(Zone.BATTLEFIELD, playerA, "Arbor Elf");
31          addCard(Zone.BATTLEFIELD, playerA, "Defiant Elf");
32  
33          addCard(Zone.BATTLEFIELD, playerB, "Forest");
34          addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion");
35          addCard(Zone.BATTLEFIELD, playerB, "Canyon Minotaur");
36  
37          attack(3, playerA, "Arbor Elf");
38          attack(3, playerA, "Defiant Elf");
39          block(3, playerB, "Silvercoat Lion", "Arbor Elf");
40          block(3, playerB, "Canyon Minotaur", "Defiant Elf");
41  
42          setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
43          execute();
44  
45          assertPermanentCount(playerA, "Arbor Elf", 1);
46          assertPermanentCount(playerA, "Defiant Elf", 1);
47  
48          assertLife(playerA, 20);
49          assertLife(playerB, 16);
50  
51      }

15      /**
16       * During a commander game both were on the battlefield, and Masumaro's P/T
17       * was not displaying as 1/1.
18       */
19      @Test
20      public void testHumilityAndMasumaro() {
21  
22          // Masumaro, First to Live's power and toughness are each equal to twice the number of cards in your hand.
23          addCard(Zone.BATTLEFIELD, playerB, "Masumaro, First to Live");
24  
25          // Enchantment  {2}{W}{W}
26          // All creatures lose all abilities and are 1/1.
27          addCard(Zone.BATTLEFIELD, playerA, "Humility");
28  
29          addCard(Zone.HAND, playerB, "Plains", 3);
30  
31          setStopAt(1, PhaseStep.PRECOMBAT_MAIN);
32          execute();
33  
34          assertPowerToughness(playerB, "Masumaro, First to Live", 1, 1);
35  
36      }

14      @Test
15      public void testGrindstoneProgenius() {
16          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
17  
18          // Until end of turn, if a creature would enter the battlefield and it wasn't cast, exile it instead.
19          // Draw a card.
20          addCard(Zone.HAND, playerA, "Hallowed Moonlight");
21  
22          addCard(Zone.BATTLEFIELD, playerB, "Plains", 3);
23          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 1);
24          addCard(Zone.HAND, playerB, "Silvercoat Lion");
25          addCard(Zone.GRAVEYARD, playerB, "Pillarfield Ox");
26          // Put a 1/1 colorless Spirit creature token onto the battlefield.
27          // Splice onto Arcane {W}
28          addCard(Zone.HAND, playerB, "Spiritual Visit");
29          // Put target creature card from a graveyard onto the battlefield under your control. You lose life equal to its converted mana cost.
30          addCard(Zone.HAND, playerB, "Reanimate");
31  
32          castSpell(2, PhaseStep.DRAW, playerA, "Hallowed Moonlight");
33  
34          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Spiritual Visit", true);
35          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Reanimate", "Pillarfield Ox");
36          waitStackResolved(2, PhaseStep.PRECOMBAT_MAIN);
37          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Silvercoat Lion");
38  
39          setStopAt(2, PhaseStep.END_TURN);
40          execute();
41  
42          assertGraveyardCount(playerA, "Hallowed Moonlight", 1);
43          assertGraveyardCount(playerB, "Spiritual Visit", 1);
44  
45          assertPermanentCount(playerB, "Spirit Token", 0);
46          assertPermanentCount(playerB, "Silvercoat Lion", 1);
47  
48          assertExileCount(playerB, 1);
49          assertExileCount("Pillarfield Ox", 1);
50  
51      }

53      /**
54       * I cast Rally the Ancestors with many creatures in my graveyard. Opponent
55       * responds with Hallowed Moonlight. After Rally the Ancestors resolves, the
56       * creature cards in my graveyard remain in my graveyard, but are also added
57       * to the exile zone.
58       */
59      @Test
60      public void testRallyTheAncestors() {
61          addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
62          // Until end of turn, if a creature would enter the battlefield and it wasn't cast, exile it instead.
63          // Draw a card.
64          addCard(Zone.HAND, playerA, "Hallowed Moonlight");
65  
66          addCard(Zone.BATTLEFIELD, playerB, "Plains", 6);
67          // Return each creature card with converted mana cost X or less from your graveyard to the battlefield.
68          // Exile those creatures at the beginning of your next upkeep. Exile Rally the Ancestors.
69          addCard(Zone.HAND, playerB, "Rally the Ancestors"); // Instant - {X}{W}{W}
70          addCard(Zone.GRAVEYARD, playerB, "Pillarfield Ox");
71          addCard(Zone.GRAVEYARD, playerB, "Silvercoat Lion");
72  
73          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Rally the Ancestors");
74          setChoice(playerB, "X=4");
75          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Hallowed Moonlight", TestPlayer.NO_TARGET, "Rally the Ancestors");
76  
77          setStopAt(2, PhaseStep.END_TURN);
78          execute();
79  
80          assertGraveyardCount(playerA, "Hallowed Moonlight", 1);
81  
82          assertExileCount("Rally the Ancestors", 1);
83          assertExileCount("Pillarfield Ox", 1);
84          assertExileCount("Silvercoat Lion", 1);
85  
86      }

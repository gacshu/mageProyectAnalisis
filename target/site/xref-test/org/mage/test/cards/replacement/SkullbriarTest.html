4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import mage.counters.CounterType;
7   import org.junit.Test;
8   import org.mage.test.serverside.base.CardTestPlayerBase;
9   
10  /**
11   *
12   * @author Dilnu
13   */
14  public class SkullbriarTest extends CardTestPlayerBase {
15  
16      /**
17       * Skullbriar retains counters even when Humility is on the field.
18       */
19      @Test
20      public void testSkullbriarCloudshift() {
21          // Counters remain on Skullbriar as it moves to any zone other than a player's hand or library.
22          addCard(Zone.BATTLEFIELD, playerB, "Skullbriar, the Walking Grave");
23          addCard(Zone.BATTLEFIELD, playerB, "Plains");
24          addCard(Zone.BATTLEFIELD, playerB, "Forest");
25  
26          // Put a +1/+1 counter on target creature.
27          addCard(Zone.HAND, playerB, "Battlegrowth");
28          // Exile target creature you control, then return that card to the battlefield under your control.
29          addCard(Zone.HAND, playerB, "Cloudshift");
30  
31          castSpell(1, PhaseStep.UPKEEP, playerB, "Battlegrowth", "Skullbriar, the Walking Grave");
32          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cloudshift", "Skullbriar, the Walking Grave");
33  
34          setStopAt(1, PhaseStep.BEGIN_COMBAT);
35          execute();
36  
37          assertCounterCount("Skullbriar, the Walking Grave", CounterType.P1P1, 1);
38      }
39  
40      /**
41       * Skullbriar should not retain counters when Humility is on the field.
42       */
43      @Test
44      public void testHumilityAndSkullbriarCloudshift() {
45          // Counters remain on Skullbriar as it moves to any zone other than a player's hand or library.
46          addCard(Zone.BATTLEFIELD, playerB, "Skullbriar, the Walking Grave");
47  
48          // Enchantment  {2}{W}{W}
49          // All creatures lose all abilities and are 1/1.
50          addCard(Zone.BATTLEFIELD, playerA, "Humility");
51          addCard(Zone.BATTLEFIELD, playerB, "Plains");
52          addCard(Zone.BATTLEFIELD, playerB, "Forest");
53  
54          // Put a +1/+1 counter on target creature.
55          addCard(Zone.HAND, playerB, "Battlegrowth");
56          // Exile target creature you control, then return that card to the battlefield under your control.
57          addCard(Zone.HAND, playerB, "Cloudshift");
58  
59          castSpell(1, PhaseStep.UPKEEP, playerB, "Battlegrowth", "Skullbriar, the Walking Grave");
60          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Cloudshift", "Skullbriar, the Walking Grave");
61  
62          setStopAt(1, PhaseStep.BEGIN_COMBAT);
63          execute();
64  
65          assertCounterCount("Skullbriar, the Walking Grave", CounterType.P1P1, 0);
66      }

16      /**
17       * Tests that 2 of 3 damage is redirected while 1 damage is still dealt to
18       * original target
19       */
20      @Test
21      public void testRedirectTwoDamage() {
22          addCard(Zone.HAND, playerA, "Lightning Bolt");
23          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
24  
25          addCard(Zone.HAND, playerB, "Harm's Way"); // Instant {W}
26          addCard(Zone.BATTLEFIELD, playerB, "Plains");
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Harm's Way", playerA);
30          setChoice(playerB, "Lightning Bolt");
31  
32          setStopAt(1, PhaseStep.END_TURN);
33          execute();
34  
35          assertGraveyardCount(playerA, "Lightning Bolt", 1);
36          assertGraveyardCount(playerB, "Harm's Way", 1);
37  
38          // 2 damage was redirected back
39          assertLife(playerA, 18);
40  
41          // 1 damage was still dealt
42          assertLife(playerB, 19);
43      }

45      /**
46       * Tests redirecting combat damage
47       */
48      @Test
49      public void testRedirectCombatDamage() {
50          addCard(Zone.HAND, playerA, "Harm's Way");
51          addCard(Zone.BATTLEFIELD, playerA, "Plains");
52  
53          addCard(Zone.BATTLEFIELD, playerB, "Craw Wurm");
54  
55          attack(2, playerB, "Craw Wurm");
56          castSpell(2, PhaseStep.DECLARE_BLOCKERS, playerA, "Harm's Way", playerB);
57          setChoice(playerA, "Craw Wurm");
58  
59          setStopAt(2, PhaseStep.END_TURN);
60          execute();
61  
62          // only 4 combat damage
63          assertLife(playerA, 16);
64  
65          // 2 damage is redirected
66          assertLife(playerB, 18);

69      /**
70       * Tests redirecting from triggered ability
71       */
72      @Test
73      public void testRedirectTriggeredAbilityDamage() {
74          addCard(Zone.HAND, playerA, "Lightning Bolt");
75          // The next 2 damage that a source of your choice would deal to you and/or permanents
76          // you control this turn is dealt to any target instead.
77          addCard(Zone.HAND, playerA, "Harm's Way");
78          addCard(Zone.BATTLEFIELD, playerA, "Mountain");
79          addCard(Zone.BATTLEFIELD, playerA, "Plains");
80  
81          // Flying
82          // When Magma Phoenix dies, it deals 3 damage to each creature and each player.
83          addCard(Zone.BATTLEFIELD, playerB, "Magma Phoenix");
84  
85          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Harm's Way", playerB);
86          setChoice(playerA, "Magma Phoenix");
87          /**
88           * When Magma Phoenix dies, Magma Phoenix deals 3 damage to each
89           * creature and each player *
90           */
91          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "Lightning Bolt", "Magma Phoenix");
92  
93          setStopAt(1, PhaseStep.POSTCOMBAT_MAIN);
94          execute();
95  
96          assertLife(playerA, 19); // 3 damage from dying Phoenix -> 2 redirected to playerB so playerA gets only 1 damage
97  
98          assertLife(playerB, 15); // 3 damage from dying Phoenix directly and 2 redirected damage from playerA
99      }

101     /**
102      * Tests that not preventable damage is redirected
103      */
104     @Test
105     public void testRedirectNotPreventableDamage() {
106         addCard(Zone.BATTLEFIELD, playerA, "Mountain");
107         // <i>Ferocious</i> If you control a creature with power 4 or greater, damage can't be prevented this turn.
108         // Wild Slash deals 2 damage to any target.
109         addCard(Zone.HAND, playerA, "Wild Slash"); // {R}
110         addCard(Zone.BATTLEFIELD, playerA, "Serra Angel");
111         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion");
112 
113         // The next 2 damage that a source of your choice would deal to you and/or permanents
114         // you control this turn is dealt to any target instead.
115         addCard(Zone.HAND, playerB, "Harm's Way"); // {W}
116         addCard(Zone.BATTLEFIELD, playerB, "Plains");
117         addCard(Zone.BATTLEFIELD, playerB, "Birds of Paradise");
118 
119         // the 2 damage can't be prevented and have to be redirected to Silvercoat Lion of player A
120         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Wild Slash", "Birds of Paradise");
121         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Harm's Way", "Silvercoat Lion", "Wild Slash");
122         setChoice(playerB, "Wild Slash");
123 
124         setStopAt(1, PhaseStep.BEGIN_COMBAT);
125         execute();
126 
127         assertGraveyardCount(playerA, "Wild Slash", 1);
128         assertGraveyardCount(playerB, "Harm's Way", 1);
129         assertPermanentCount(playerB, "Birds of Paradise", 1);
130         assertGraveyardCount(playerA, "Silvercoat Lion", 1);
131         assertLife(playerA, 20);
132         assertLife(playerB, 20);
133     }

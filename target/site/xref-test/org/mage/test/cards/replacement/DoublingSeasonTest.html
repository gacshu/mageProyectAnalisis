97      /**
98       * Creatures with enter the battlefield triggers are causing a bug when
99       * multiple copies are made simultaneously (ie via Doubling Season +
100      * Kiki-Jiki, Mirror Breaker or Rite of Replication). After the tokens have
101      * entered the battlefield it asks their controller to choose the order that
102      * the triggered abilities on the stack but no window opens to select the
103      * triggers leaving no option to move the game forward (besides rollback and
104      * just not making the tokens). Several attempts with the different
105      * combinations make it *seem to be a general bug about duplicates entering
106      * at the same time and not related to the specific cards.
107      */
108     @Test
109     public void testDoubleRiteOfReplication() {
110         /**
111          * If an effect would put one or more tokens onto the battlefield under
112          * your control, it puts twice that many of those tokens onto the
113          * battlefield instead. If an effect would place one or more counters on
114          * a permanent you control, it places twice that many of those counters
115          * on that permanent instead.
116          */
117 
118         addCard(Zone.BATTLEFIELD, playerA, "Doubling Season");
119         addCard(Zone.BATTLEFIELD, playerA, "Island", 9);
120 
121         // Create a tokenthat's a copy of target creature onto the battlefield. If Rite of Replication was kicked, put five of those tokens onto the battlefield instead.
122         addCard(Zone.HAND, playerA, "Rite of Replication");
123         // When Venerable Monk enters the battlefield, you gain 2 life.
124         addCard(Zone.BATTLEFIELD, playerB, "Venerable Monk", 1);
125 
126         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Rite of Replication", "Venerable Monk");
127         setChoice(playerA, true);
128 
129         setStopAt(1, PhaseStep.BEGIN_COMBAT);
130         execute();
131 
132         assertLife(playerA, 40);
133         assertPermanentCount(playerA, "Venerable Monk", 10);
134 
135     }

161 
162         assertLife(playerB, 10);

166     @Test
167     public void testDoubleRiteOfRagingStormOpponent() {
168         // At the beginning of each player's upkeep, that player creates a 5/1 red Elemental creature token named Lightning Rager.
169         // It has trample, haste, and "At the beginning of the end step, sacrifice this creature."
170         addCard(Zone.HAND, playerA, "Rite of the Raging Storm");// {3}{R}{R}
171         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
172         // If an effect would put one or more tokens onto the battlefield under your control, it puts twice that many of those tokens onto the battlefield instead.
173         // If an effect would place one or more counters on a permanent you control, it places twice that many of those counters on that permanent instead.
174         addCard(Zone.BATTLEFIELD, playerB, "Doubling Season");
175 
176         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Rite of the Raging Storm");
177 
178         attack(2, playerB, "Lightning Rager:0"); // Can't attack
179         attack(2, playerB, "Lightning Rager:1"); // Can't attack
180 
181         setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
182 
183         try {
184             execute();
185 
186             Assert.fail("must throw exception on execute");
187         } catch (Throwable e) {
188             if (!e.getMessage().contains("Player PlayerB must have 0 actions but found 2")) {
189                 Assert.fail("Should have thrown error about cannot attack, but got:\n" + e.getMessage());
190             }
191         }
192 
193         assertPermanentCount(playerA, "Rite of the Raging Storm", 1);
194         assertPermanentCount(playerB, "Lightning Rager", 2);
195     }

198      * Gatherer Ruling: 10/1/2005: Planeswalkers will enter the battlefield with
199      * double the normal amount of loyalty counters. However, if you activate an
200      * ability whose cost has you put loyalty counters on a planeswalker, the
201      * number you put on isn't doubled. This is because those counters are put
202      * on as a cost, not as an effect.
203      */
204     @Test
205     public void testPlaneswalkerLoyalty() {
206         addCard(Zone.BATTLEFIELD, playerA, "Tibalt, the Fiend-Blooded");
207 
208         addCard(Zone.BATTLEFIELD, playerA, "Doubling Season");
209 
210         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "+1: Draw a card, then discard a card at random.");
211 
212         setStopAt(1, PhaseStep.END_TURN);
213         execute();
214 
215         assertLife(playerA, 20);
216         assertLife(playerB, 20);
217 
218         //Should not be doubled
219         assertCounterCount("Tibalt, the Fiend-Blooded", CounterType.LOYALTY, 3);

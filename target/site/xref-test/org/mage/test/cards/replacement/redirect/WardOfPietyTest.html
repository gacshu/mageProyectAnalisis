4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import org.junit.Test;
7   import org.mage.test.serverside.base.CardTestPlayerBase;

9   /**
10   *
11   * @author LevelX2
12   */
13  public class WardOfPietyTest extends CardTestPlayerBase {
14  
15      @Test
16      public void testNonCombatDamageToPlayer() {
17          addCard(Zone.HAND, playerB, "Lightning Bolt");
18          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
19  
20          // Enchant creature
21          // {1}{W}: The next 1 damage that would be dealt to enchanted creature this turn is dealt to any target instead.
22          addCard(Zone.HAND, playerA, "Ward of Piety"); // {1}{W}
23          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion"); // 2/2
24          addCard(Zone.BATTLEFIELD, playerA, "Plains", 6);
25  
26          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ward of Piety", "Silvercoat Lion");
27  
28          activateAbility(1, PhaseStep.BEGIN_COMBAT, playerA, "{1}{W}: The next 1 damage", playerB);
29          activateAbility(1, PhaseStep.BEGIN_COMBAT, playerA, "{1}{W}: The next 1 damage", playerB);
30  
31          castSpell(1, PhaseStep.POSTCOMBAT_MAIN, playerB, "Lightning Bolt", "Silvercoat Lion");
32  
33          setStopAt(1, PhaseStep.END_TURN);
34          execute();
35  
36          assertGraveyardCount(playerB, "Lightning Bolt", 1);
37          assertPermanentCount(playerA, "Ward of Piety", 1);
38          assertPermanentCount(playerA, "Silvercoat Lion", 1);
39  
40          assertLife(playerA, 20);
41          assertLife(playerB, 18);
42  
43      }

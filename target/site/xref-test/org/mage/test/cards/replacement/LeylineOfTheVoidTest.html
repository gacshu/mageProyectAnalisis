15      /**
16       * Leyline of the Void is on the battlefield, Helm of Obedience is used on
17       * an opponent with X=1. Now the Helm states to mill "until a creature card
18       * or X cards are put into the graveyard this way". For each of those milled
19       * cards, Leyline states to "remove [them] from the game instead". In other
20       * words: Leyline + Helm and X of at least 1 exiles the library of one
21       * unlucky opponent. Or should, because right now it doesn't. Right now it's
22       * Helm's originally intended "X or 1st Creature" amount of cards that after
23       * it's effect get exiled.
24       */
25      @Test
26      public void testGrindstoneProgenius() {
27          addCard(Zone.BATTLEFIELD, playerA, "Plains");
28  
29          // If Leyline of the Void is in your opening hand, you may begin the game with it on the battlefield.
30          // If a card would be put into an opponent's graveyard from anywhere, exile it instead.
31          addCard(Zone.BATTLEFIELD, playerA, "Leyline of the Void");
32  
33          // {X}, {T}: Target opponent puts cards from the top of their library into their graveyard until a creature card or X cards are put into that graveyard this way, whichever comes first. If a creature card is put into that graveyard this way, sacrifice Helm of Obedience and put that card onto the battlefield under your control. X can't be 0.
34          addCard(Zone.BATTLEFIELD, playerA, "Helm of Obedience");
35  
36          activateAbility(1, PhaseStep.POSTCOMBAT_MAIN, playerA, "{X}, {T}: Target opponent mills", playerB);
37          setChoice(playerA, "X=1");
38  
39          setStopAt(1, PhaseStep.END_TURN);
40          execute();
41  
42          assertExileCount(playerB, 71); // All cards go to exile replaced from Leyline of the void
43      }

45      /**
46       * Today i casted Ill-gotten Gains in EDH (with a leyline of the veil in
47       * play) and the spell simply discarded both players hands not letting
48       * either of us choose cards to get back, this ended up with me losing the
49       * game as i was going to combo off some cards in yard.
50       */
51      @Test
52      public void testIllgottenGains() {
53          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 4);
54  
55          // If Leyline of the Void is in your opening hand, you may begin the game with it on the battlefield.
56          // If a card would be put into an opponent's graveyard from anywhere, exile it instead.
57          addCard(Zone.BATTLEFIELD, playerA, "Leyline of the Void");
58  
59          // Exile Ill-Gotten Gains.
60          // Each player discards their hand,
61          // then returns up to three cards from their graveyard to their hand.
62          addCard(Zone.HAND, playerA, "Ill-Gotten Gains"); // Sorcery - {2}{B}{B}
63          addCard(Zone.HAND, playerA, "Silvercoat Lion", 4);
64          addCard(Zone.HAND, playerB, "Silvercoat Lion", 4);
65  
66          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ill-Gotten Gains");
67          setChoice(playerA, "Silvercoat Lion^Silvercoat Lion^Silvercoat Lion");
68  
69          setStopAt(1, PhaseStep.END_TURN);
70          execute();
71  
72          assertExileCount(playerB, 4);
73          assertHandCount(playerB, 0);
74  
75          assertExileCount(playerA, 1);
76          assertHandCount(playerA, 3);
77      }

79      @Test
80      public void testMorbidAbility() {
81          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 3);
82          // If Leyline of the Void is in your opening hand, you may begin the game with it on the battlefield.
83          // If a card would be put into an opponent's graveyard from anywhere, exile it instead.
84          addCard(Zone.BATTLEFIELD, playerA, "Leyline of the Void");
85          addCard(Zone.HAND, playerA, "Murder");
86           // Morbid — At the beginning of each end step, if a creature died this turn, you may draw a card.
87          addCard(Zone.BATTLEFIELD, playerB, "Deathreap Ritual");
88          addCard(Zone.BATTLEFIELD, playerB, "Memnite");
89  
90          setStrictChooseMode(true);
91  
92          castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Murder");
93          addTarget(playerA, "Memnite");
94  
95          setStopAt(2, PhaseStep.END_TURN);
96          execute();
97  
98          assertHandCount(playerB, 1); // card drawn for turn
99          assertExileCount(playerB, 1);

103     /*
104     "Leyline of the Void's second ability doesn't affect token permanents that would be put into an opponent's graveyard from the battlefield. 
105     They'll be put into that graveyard as normal (causing any applicable triggered abilities to trigger), then they'll cease to exist."
106     https://gatherer.wizards.com/Pages/Card/Details.aspx?multiverseid=107682
107     
108     */
109     @Test
110     public void testMorbidAbilityWithAwakeningZoneTokens() {
111         // At the beginning of your upkeep, you may put a 0/1 colorless Eldrazi Spawn creature token onto the battlefield. 
112         // It has "Sacrifice this creature: Add mana symbol 1."
113         addCard(Zone.BATTLEFIELD, playerA, "Awakening Zone");
114         // If Leyline of the Void is in your opening hand, you may begin the game with it on the battlefield.
115         // If a card would be put into an opponent's graveyard from anywhere, exile it instead.
116         addCard(Zone.BATTLEFIELD, playerA, "Leyline of the Void");
117         // Morbid — At the beginning of each end step, if a creature died this turn, you may draw a card.
118         addCard(Zone.BATTLEFIELD, playerB, "Deathreap Ritual");
119         
120         setChoice(playerA, true);
121         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice");
122         setChoice(playerB, true);
123         
124         setStopAt(1, PhaseStep.END_TURN);
125         execute();
126         
127         assertPermanentCount(playerA, "Eldrazi Spawn Token", 0);        
128         assertExileCount(playerB, 0);
129         assertHandCount(playerB, 1); 
130     }

8   /**
9    * Pillar of Flame: Pillar of Flame deals 2 damage to any target.
10   * If a creature dealt damage this way would die this turn, exile it instead.
11   *
12   * @author LevelX2
13   */
14  public class PillarOfFlameTest extends CardTestPlayerBase {
15  
16      /**
17       * Tests when cast Pillar of Flame targeting opponent there is no influence
18       * on dying creature of opponent
19       */
20      @Test
21      public void testNotTriggeringExileItInstead() {
22          // Soulbond
23          // As long as Lightning Mauler is paired with another creature, both creatures have haste.
24          addCard(Zone.BATTLEFIELD, playerA, "Lightning Mauler"); // 2/1
25          // Unleash (You may have this creature enter the battlefield with a +1/+1 counter on it. It can't block as long as it has a +1/+1 counter on it.)
26          addCard(Zone.BATTLEFIELD, playerA, "Rakdos Cackler"); // 1/1
27  
28          // Pillar of Flame deals 2 damage to any target.
29          // If a creature dealt damage this way would die this turn, exile it instead.
30          addCard(Zone.HAND, playerA, "Pillar of Flame");
31          // Soulbond
32          // As long as Stonewright is paired with another creature, each of those creatures has "{R}: This creature gets +1/+0 until end of turn."
33          addCard(Zone.HAND, playerA, "Stonewright");
34          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
35  
36          addCard(Zone.BATTLEFIELD, playerB, "Dutiful Thrull");
37  
38          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Pillar of Flame", playerB);
39          waitStackResolved(3, PhaseStep.PRECOMBAT_MAIN);
40          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Stonewright");
41  
42          attack(3, playerA, "Rakdos Cackler");
43          attack(3, playerA, "Lightning Mauler");
44          block(3, playerB, "Dutiful Thrull", "Lightning Mauler");
45  
46          setStopAt(3, PhaseStep.END_TURN);
47          execute();
48  
49          assertLife(playerA, 20);
50          assertLife(playerB, 16);
51  
52          assertPermanentCount(playerA, "Rakdos Cackler", 1);
53          assertPermanentCount(playerA, "Lightning Mauler", 0);
54          assertPermanentCount(playerA, "Stonewright", 1);
55          assertGraveyardCount(playerA, 2);
56  
57          assertGraveyardCount(playerB, 1);
58      }
59  
60      /**
61       * Tests when cast Pillar of Flame targeting creature it goes to exile if
62       * dying later
63       */
64      @Test
65      public void testTriggeringExileItInstead() {
66          addCard(Zone.BATTLEFIELD, playerA, "Lightning Mauler");
67  
68          addCard(Zone.HAND, playerA, "Pillar of Flame");
69          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
70  
71          addCard(Zone.BATTLEFIELD, playerB, "Warmind Infantry");
72  
73          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Pillar of Flame", "Warmind Infantry");
74  
75          attack(3, playerA, "Lightning Mauler");
76          block(3, playerB, "Warmind Infantry", "Lightning Mauler");
77  
78          setStopAt(3, PhaseStep.END_TURN);
79          execute();
80  
81          assertLife(playerA, 20);
82          assertLife(playerB, 20);
83  
84          assertPermanentCount(playerA, "Lightning Mauler", 0);
85          assertGraveyardCount(playerA, 2);
86  
87          assertPermanentCount(playerB, "Warmind Infantry", 0);
88          assertGraveyardCount(playerB, 0);
89      }

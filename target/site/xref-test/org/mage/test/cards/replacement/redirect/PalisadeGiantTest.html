14  public class PalisadeGiantTest extends CardTestPlayerBase {
15  
16      /**
17       * when you have two Palisade Giants in play and your opponent deals you
18       * damage, the damage is applied to both of them rather than allowing you to
19       * choose which replacement effect applies (as it should). I experienced
20       * this multiple times in games vs the A.I.
21       */
22      @Test
23      public void testRedirectDamage() {
24                  
25          // All damage that would be dealt to you or another permanent you control is dealt to Palisade Giant instead.
26          addCard(Zone.BATTLEFIELD, playerA, "Palisade Giant", 2); // Creature 2/7
27  
28          addCard(Zone.HAND, playerB, "Lightning Bolt"); // Instant {R}
29          addCard(Zone.BATTLEFIELD, playerB, "Mountain");
30  
31          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Lightning Bolt", playerA);
32  
33          setChoice(playerA, "Palisade Giant");
34          
35          setStrictChooseMode(true);
36          setStopAt(1, PhaseStep.BEGIN_COMBAT);
37          execute();
38  
39          assertGraveyardCount(playerB, "Lightning Bolt", 1);
40  
41          assertLife(playerA, 20);
42          assertLife(playerB, 20);
43          
44          int damage = 0;
45          for (Permanent permanent : currentGame.getBattlefield().getAllActivePermanents(playerA.getId())) {
46              if (permanent.getName().equals("Palisade Giant")) {
47                  damage += permanent.getDamage();
48              }
49          }
50          Assert.assertEquals("Only 3 damage in sum should be dealt to the Palisade Giants", 3, damage);
51  
52      }
53

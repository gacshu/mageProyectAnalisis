15      @Test
16      public void test_GainedByEnchantment() {
17          addCard(Zone.BATTLEFIELD, playerA, "Underworld Cerberus");
18  
19          addCard(Zone.BATTLEFIELD, playerB, "Sagu Archer");
20          addCard(Zone.HAND, playerB, "Molting Snakeskin");
21          addCard(Zone.BATTLEFIELD, playerB, "Swamp", 4);
22  
23          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Molting Snakeskin", "Sagu Archer");
24          attack(2, playerB, "Sagu Archer");
25          block(2, playerA, "Underworld Cerberus", "Sagu Archer");
26          activateAbility(2, PhaseStep.DECLARE_BLOCKERS, playerB, "{2}{B}: Regenerate {this}.");
27  
28          setStrictChooseMode(true);
29          setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
30          execute();
31  
32          assertLife(playerA, 20);
33          assertLife(playerB, 20);
34  
35          // attacker has to regenerate because of damage
36  
37          assertPermanentCount(playerA, "Underworld Cerberus", 1);
38          assertPermanentCount(playerB, "Sagu Archer", 1);
39          assertPermanentCount(playerB, "Molting Snakeskin", 1);
40      }

276     @Test
277     public void test_Attached_Moved() {
278         // regen shields must be accumulated in permanent, not aura
279 
280         // Enchant creature (Target a creature as you cast this. This card enters the battlefield attached to that creature.)
281         // {G}: Regenerate enchanted creature. (The next time that creature would be destroyed this turn, it isnâ€™t.
282         // Instead tap it, remove all damage from it, and remove it from combat.)
283         addCard(Zone.HAND, playerA, "Regeneration", 1); // {1}{G} aura
284         addCard(Zone.BATTLEFIELD, playerA, "Forest", 2 + 2); // cast + 2 activates
285         //
286         addCard(Zone.BATTLEFIELD, playerA, "Grizzly Bears", 1); // 2/2 permanent 1
287         addCard(Zone.BATTLEFIELD, playerA, "Kitesail Corsair", 1); // 2/1 permanent 2
288         //
289         addCard(Zone.HAND, playerA, "Lightning Bolt", 3);
290         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
291         //
292         // Attach target Aura you control to target creature.
293         addCard(Zone.HAND, playerA, "Aura Finesse", 1); // {U} instant
294         addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
295 
296         // attach to first
297         activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 2);
298         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Regeneration", "Grizzly Bears");
299         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
300 
301         // add multiple regen shields
302         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{G}: Regenerate");
303         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{G}: Regenerate");
304         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
305 
306         // re-attach to second
307         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Aura Finesse");
308         addTarget(playerA, "Regeneration"); // aura to move
309         addTarget(playerA, "Kitesail Corsair"); // attach to permanent 2
310 
311         // damage to one - have regens
312         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Grizzly Bears");
313         setChoice(playerA, "Regeneration"); // two replacement effects
314         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
315         checkPermanentCount("alive", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 1);
316 
317         // damage to second -- die (no regens)
318         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Kitesail Corsair");
319         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
320         checkGraveyardCount("die", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Kitesail Corsair", 1);
321 
322         setStrictChooseMode(true);
323         setStopAt(1, PhaseStep.END_TURN);
324         execute();
325     }

327     @Test
328     public void test_Attached_Sacrifice() {
329         // if you sacrifice aura to enable regen then it must works
330 
331         // Enchant creature
332         // Enchanted creature gets +0/+2.
333         // Sacrifice Carapace: Regenerate enchanted creature.
334         addCard(Zone.HAND, playerA, "Carapace", 1); // {G} aura
335         addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
336         //
337         addCard(Zone.BATTLEFIELD, playerA, "Grizzly Bears", 1); // 2/2
338         //
339         addCard(Zone.HAND, playerA, "Lightning Bolt", 1);
340         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
341 
342         // attach
343         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Carapace", "Grizzly Bears");
344         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA);
345         checkPT("boosted", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 2, 2 + 2);
346 
347         // try to kill (boost lost after regen activate, so need only 1 bolt)
348         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "Grizzly Bears");
349         // activate regen before kill
350         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Sacrifice", TestPlayer.NO_TARGET, "Cast Lightning Bolt", StackClause.WHILE_ON_STACK);
351         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
352 
353         // permanent must be alive
354         checkGraveyardCount("aura sacrificed", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Carapace", 1);
355         checkPermanentCount("permanent alive", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grizzly Bears", 1);
356 
357         setStrictChooseMode(true);
358         setStopAt(1, PhaseStep.END_TURN);
359         execute();

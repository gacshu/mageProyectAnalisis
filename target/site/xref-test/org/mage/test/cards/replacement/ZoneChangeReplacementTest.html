17      // If Darksteel Colossus would be put into a graveyard from anywhere,
18      // reveal Darksteel Colossus and shuffle it into its owner's library instead.
19      @Test
20      public void testFromLibraryZoneChange() {
21          addCard(Zone.LIBRARY, playerA, "Darksteel Colossus");
22          // Tome Scour - Sorcery - {U}
23          // Target player puts the top five cards of their library into their graveyard.
24          addCard(Zone.HAND, playerA, "Tome Scour");
25          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
26          skipInitShuffling();
27  
28          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Tome Scour", playerA);
29  
30          setStopAt(1, PhaseStep.BEGIN_COMBAT);
31          execute();
32  
33          assertGraveyardCount(playerA, "Darksteel Colossus", 0);
34          assertGraveyardCount(playerA, 5); // 4 + Tome Scour
35      }

133     @Test
134     public void testHumilityAndKumano() {
135         // Enchantment  {2}{W}{W}
136         // All creatures lose all abilities and are 1/1.
137         addCard(Zone.BATTLEFIELD, playerA, "Humility");
138 
139         // 3/3
140         // If a creature dealt damage by Kumano's Pupils this turn would die, exile it instead.
141         addCard(Zone.BATTLEFIELD, playerA, "Kumano's Pupils");
142 
143         // Instant {1}{G}
144         // Target creature gets +1/+1 until end of turn.
145         // Draw a card.
146         addCard(Zone.HAND, playerA, "Aggressive Urge");
147         addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
148         // 2/2
149         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 1);
150 
151         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Aggressive Urge", "Kumano's Pupils");
152 
153         attack(2, playerB, "Silvercoat Lion");
154         block(2, playerA, "Kumano's Pupils", "Silvercoat Lion");
155 
156         setStopAt(2, PhaseStep.END_TURN);
157         execute();
158 
159         assertLife(playerA, 20);
160         assertLife(playerB, 20);
161 
162         assertGraveyardCount(playerA, "Aggressive Urge", 1);
163         assertPermanentCount(playerA, "Kumano's Pupils", 1);
164         assertPowerToughness(playerA, "Kumano's Pupils", 2, 2);
165         assertPermanentCount(playerB, "Silvercoat Lion", 0);
166 
167         assertExileCount("Silvercoat Lion", 0);
168         assertGraveyardCount(playerB, "Silvercoat Lion", 1);

205     // A creature gets damage from Kumano's Pupils and returns to hand after.
206     // Then it's cast again. This new permanent instance is destroyed. It may not
207     // got to exile because only previous instance was damgaged by Kumano's Pupils.
208     @Test
209     public void testPermanentNewInstanceAndKumano() {
210         // 3/3
211         // If a creature dealt damage by Kumano's Pupils this turn would die, exile it instead.
212         addCard(Zone.BATTLEFIELD, playerA, "Kumano's Pupils");
213         // 2/4
214         addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox", 1);
215 
216         // Unsummon - Instant {U}
217         // Return target creature to its owner's hand.
218         addCard(Zone.HAND, playerA, "Unsummon");
219         // Terminate - Instant {R}{B}
220         // Destroy target creature. It can't be regenerated.
221         addCard(Zone.HAND, playerA, "Terminate");
222         addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
223         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
224         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
225         addCard(Zone.BATTLEFIELD, playerB, "Plains", 4);
226 
227         attack(2, playerB, "Pillarfield Ox");
228         block(2, playerA, "Kumano's Pupils", "Pillarfield Ox");
229 
230         castSpell(2, PhaseStep.COMBAT_DAMAGE, playerA, "Unsummon", "Pillarfield Ox");
231         castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "Pillarfield Ox");
232         waitStackResolved(2, PhaseStep.POSTCOMBAT_MAIN);
233         castSpell(2, PhaseStep.POSTCOMBAT_MAIN, playerA, "Terminate", "Pillarfield Ox");
234 
235         setStopAt(2, PhaseStep.END_TURN);
236         execute();
237 
238         assertLife(playerA, 20);
239         assertLife(playerB, 20);
240 
241         assertPermanentCount(playerB, "Pillarfield Ox", 0);
242         assertPermanentCount(playerA, "Kumano's Pupils", 1);
243         assertPowerToughness(playerA, "Kumano's Pupils", 3, 3);
244         assertGraveyardCount(playerA, "Unsummon", 1);
245         assertGraveyardCount(playerA, "Terminate", 1);
246 
247         assertExileCount("Pillarfield Ox", 0);
248         assertGraveyardCount(playerB, "Pillarfield Ox", 1);
249     }

252      * Test that a countered spell of a card that goes always to library back
253      * instead of into the graveyard.
254      */
255     @Test
256     public void testCounterAndMoveToLibrary() {
257         addCard(Zone.BATTLEFIELD, playerA, "Plains", 7);
258         // Legacy Weapon - Artifact {7}
259         // {W}{U}{B}{R}{G}: Exile target permanent.
260         // If Legacy Weapon would be put into a graveyard from anywhere, reveal Legacy Weapon and shuffle it into its owner's library instead.
261         addCard(Zone.HAND, playerA, "Legacy Weapon");
262 
263         addCard(Zone.BATTLEFIELD, playerB, "Island", 2);
264         // Counter target spell. At the beginning of your next main phase, add {X}, where X is that spell's converted mana cost.
265         addCard(Zone.HAND, playerB, "Mana Drain");
266         addCard(Zone.HAND, playerB, "Legacy Weapon");
267 
268         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Legacy Weapon");
269         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerB, "Mana Drain", "Legacy Weapon");
270 
271         waitStackResolved(2, PhaseStep.PRECOMBAT_MAIN); // Let the Mana Drain delayed triggered ability resolve
272         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Legacy Weapon");
273         setStopAt(2, PhaseStep.BEGIN_COMBAT);
274         execute();
275 
276         assertLife(playerA, 20);
277         assertLife(playerB, 20);
278 
279         assertHandCount(playerA, "Legacy Weapon", 0);
280         assertPermanentCount(playerA, "Legacy Weapon", 0);
281         assertGraveyardCount(playerA, "Legacy Weapon", 0);
282 
283         assertGraveyardCount(playerB, "Mana Drain", 1);
284         assertPermanentCount(playerB, "Legacy Weapon", 1);
285     }

291     @Test
292     public void testWhipOfErebos() {
293         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
294         // Destroy target nonartifact, nonblack creature. It can't be regenerated.
295         addCard(Zone.HAND, playerA, "Terror");
296 
297         // {2}{B}{B}, {T}: Return target creature card from your graveyard to the battlefield.
298         // It gains haste. Exile it at the beginning of the next end step.
299         // If it would leave the battlefield, exile it instead of putting it anywhere else.

303         addCard(Zone.GRAVEYARD, playerB, "Silvercoat Lion");
304 
305         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{2}{B}{B}, {T}: Return target creature", "Silvercoat Lion");
306 
307         castSpell(2, PhaseStep.BEGIN_COMBAT, playerA, "Terror", "Silvercoat Lion");
308         setStopAt(2, PhaseStep.END_COMBAT);
309         execute();
310 
311         assertLife(playerA, 20);
312         assertLife(playerB, 20);
313 
314         assertGraveyardCount(playerA, "Terror", 1);
315         assertExileCount("Silvercoat Lion", 1);
316     }

319      * Jace, Vryn's Prodigy – Jace, Telepath Unbound
320      *
321      * You can't whip him back with Whip of Erebos , flip him and then keep him.
322      * I think he is considered a new object after being exiled by his own
323      * trigger, so whip shouldn't affect him anymore.
324      */
325     @Test
326     public void testWhipOfErebosTransformPlaneswalker() {
327         // {2}{B}{B}, {T}: Return target creature card from your graveyard to the battlefield.
328         // It gains haste. Exile it at the beginning of the next end step.
329         // If it would leave the battlefield, exile it instead of putting it anywhere else.
330         // Activate this ability only any time you could cast a sorcery.
331         addCard(Zone.BATTLEFIELD, playerB, "Whip of Erebos");
332         addCard(Zone.BATTLEFIELD, playerB, "Swamp", 4);
333         addCard(Zone.GRAVEYARD, playerB, "Swamp", 5);
334         addCard(Zone.GRAVEYARD, playerB, "Jace, Vryn's Prodigy");
335 
336         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{2}{B}{B}, {T}: Return target creature", "Jace, Vryn's Prodigy");
337 
338         // {T}: Draw a card, then discard a card. If there are five or more cards in your graveyard, exile Jace, Vryn's Prodigy, then return him to the battefield transformed under his owner's control.
339         activateAbility(2, PhaseStep.BEGIN_COMBAT, playerB, "{T}:");
340 
341         activateAbility(2, PhaseStep.POSTCOMBAT_MAIN, playerB, "+1: Up to one target creature gets -2/-0 until your next turn");
342         setStopAt(3, PhaseStep.UNTAP);
343         execute();
344 
345         assertLife(playerA, 20);
346         assertLife(playerB, 20);
347 
348         assertPermanentCount(playerB, "Jace, Vryn's Prodigy", 0);
349         assertPermanentCount(playerB, "Jace, Telepath Unbound", 1);
350         assertCounterCount("Jace, Telepath Unbound", CounterType.LOYALTY, 6);

353     /**
354      * I was using Anafenza, the Foremost as Commander. She attacked and traded
355      * with two creatures. I moved Anafenza to the Command Zone, but the
356      * opponent's creatures "when {this} dies" abilities triggered. Since
357      * Anafenza and those creatures all received lethal damage at the same time,
358      * the creatures should have been exiled due to Anafenza's replacement
359      * effect, but I guess since the logic asks if you want to use the Command
360      * Zone replacement effect first, that it doesn't see her leaving the
361      * battlefield at the same time as the other creatures.
362      *
363      * http://blogs.magicjudges.org/rulestips/2015/05/anafenza-vs-deathmist-raptor/
364      */
365     @Test
366     public void testAnafenzaExileInCombat() {
367         // Whenever Anafenza, the Foremost attacks, put a +1/+1 counter on another target tapped creature you control.
368         // If a creature card would be put into an opponent's graveyard from anywhere, exile it instead.
369         addCard(Zone.BATTLEFIELD, playerA, "Anafenza, the Foremost"); // 4/4
370 
371         // Reach (This creature can block creatures with flying.)
372         addCard(Zone.BATTLEFIELD, playerB, "Skyraker Giant"); // 4/3
373 
374         attack(2, playerB, "Skyraker Giant");
375         block(2, playerA, "Anafenza, the Foremost", "Skyraker Giant");
376 
377         setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
378         execute();
379 
380         assertLife(playerA, 20);
381         assertLife(playerB, 20);
382 
383         assertExileCount("Skyraker Giant", 1);
384         assertPermanentCount(playerA, "Anafenza, the Foremost", 0);
385         assertGraveyardCount(playerA, "Anafenza, the Foremost", 1);
386     }

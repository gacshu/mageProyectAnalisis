8   import org.junit.Assert;

10  
11  public class TxtDeckImporterTest {
12  
13      @Test
14      public void testImportWithBlankLineAboveSideboard() {
15          TxtDeckImporter importer = new TxtDeckImporter(false);
16  
17          CardInfo card;
18          DeckCardLists deck = new DeckCardLists();
19  
20          String[] cards = {"Plains", "Forest", "Island"};
21          String[] sideboard = {"Swamp", "Mountain"};
22  
23          for (String c : cards) {
24              card = CardRepository.instance.findPreferredCoreExpansionCard(c);
25              Assert.assertNotNull(String.format("Card %s was null", c), card);
26              deck.getCards().add(new DeckCardInfo(card.getName(), card.getCardNumber(), card.getSetCode()));
27          }
28  
29          for (String s : sideboard) {
30              card = CardRepository.instance.findPreferredCoreExpansionCard(s);
31              Assert.assertNotNull(String.format("Card %s was null", s), card);
32              deck.getSideboard().add(new DeckCardInfo(card.getName(), card.getCardNumber(), card.getSetCode()));
33          }
34  
35          Assert.assertEquals("Deck does not contain 3 cards, found " + deck.getCards().size(), 3, deck.getCards().size());
36          Assert.assertEquals("Sideboard does not contain 2 cards, found " + deck.getSideboard().size(), 2, deck.getSideboard().size());
37  
38          DeckCardLists imported = importer.importDeck("JustLands.txt", false);
39  
40          Assert.assertEquals("Imported deck does not contain 3 cards, found " + imported.getCards().size(), 3, imported.getCards().size());
41          Assert.assertEquals("Imported sideboard does not contain 2 cards, found " + imported.getSideboard().size(), 2, imported.getSideboard().size());
42      }
43  }

14  public class BoostCountTest {
15  
16      @Test
17      public void test_BoostCountSigns() {
18          Assert.assertEquals(CardUtil.getBoostCountAsStr(0, 0), "+0/+0");
19          Assert.assertEquals(CardUtil.getBoostCountAsStr(1, 0), "+1/+0");
20          Assert.assertEquals(CardUtil.getBoostCountAsStr(0, 1), "+0/+1");
21          Assert.assertEquals(CardUtil.getBoostCountAsStr(1, 1), "+1/+1");
22  
23          Assert.assertEquals(CardUtil.getBoostCountAsStr(-1, 0), "-1/-0");
24          Assert.assertEquals(CardUtil.getBoostCountAsStr(0, -1), "-0/-1");
25          Assert.assertEquals(CardUtil.getBoostCountAsStr(-1, 1), "-1/+1");
26          Assert.assertEquals(CardUtil.getBoostCountAsStr(1, -1), "+1/-1");
27      }
28  
29      @Test
30      public void test_DynamicBoostCountSigns() {
31          DynamicValue zero = StaticValue.get(0);
32          DynamicValue plusX = GetXValue.instance;
33          DynamicValue minusX = new SignInversionDynamicValue(plusX);
34  
35          Assert.assertEquals(CardUtil.getBoostCountAsStr(plusX, zero), "+X/+0");
36          Assert.assertEquals(CardUtil.getBoostCountAsStr(zero, plusX), "+0/+X");
37          Assert.assertEquals(CardUtil.getBoostCountAsStr(plusX, plusX), "+X/+X");
38          Assert.assertEquals(CardUtil.getBoostCountAsStr(minusX, zero), "-X/-0");
39          Assert.assertEquals(CardUtil.getBoostCountAsStr(zero, minusX), "-0/-X");
40          Assert.assertEquals(CardUtil.getBoostCountAsStr(minusX, plusX), "-X/+X");
41          Assert.assertEquals(CardUtil.getBoostCountAsStr(plusX, minusX), "+X/-X");
42      }
43  }

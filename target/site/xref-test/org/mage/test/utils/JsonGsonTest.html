14  /**
15   * Gson (Google json) lib uses for scryfall and mtgjson data.
16   * <p>
17   * Tests:
18   * - unknown data parser tests here (JsonParser.parseReader)
19   * - class parser tests in mtgjson verify tests (new Gson().fromJson)
20   */
21  public class JsonGsonTest {
22  
23      @Test
24      public void test_ReadByStreamParser() {
25          String sampleFileName = Paths.get("src", "test", "data", "scryfall-card.json").toString();
26          try {
27              // low level parser for unknown data structure
28              JsonObject json = JsonParser.parseReader(new FileReader(sampleFileName)).getAsJsonObject();
29  
30              // data types
31              Assert.assertEquals("string", "card", JsonUtil.getAsString(json, "object"));
32              JsonArray jsonFaces = JsonUtil.getAsArray(json, "card_faces");
33              Assert.assertEquals("int", 60370, JsonUtil.getAsInt(json, "mtgo_id"));
34              Assert.assertTrue("boolean", JsonUtil.getAsBoolean(json, "highres_image"));
35              Assert.assertEquals("double", 4.0, JsonUtil.getAsDouble(json, "cmc"), 0.0);
36              Assert.assertNotNull("array", jsonFaces);
37  
38              Assert.assertEquals("Card must have 2 faces", 2, jsonFaces.size());
39              Assert.assertEquals("Unknown second side", "Infectious Curse", JsonUtil.getAsString(jsonFaces.get(1).getAsJsonObject(), "name"));
40          } catch (IOException e) {
41              e.printStackTrace();
42              Assert.fail("Can't load sample json file: " + sampleFileName);
43          }
44      }
45  }

22  
23      public static Deck buildRandomDeck(String colors, boolean onlyBasicLands) {
24          return buildRandomDeck(colors, onlyBasicLands, "");
25      }
26  
27      public static Deck buildRandomDeck(String colors, boolean onlyBasicLands, String allowedSets) {
28  
29          List<ColoredManaSymbol> allowedColors = new ArrayList<>();
30          for (int i = 0; i < colors.length(); i++) {
31              char c = colors.charAt(i);
32              allowedColors.add(ColoredManaSymbol.lookup(c));
33          }
34  
35          List<String> allowedList = new ArrayList<>();
36          if (allowedSets != null && !allowedSets.isEmpty()) {
37              String[] codes = allowedSets.split(",");
38              for (String code : codes) {
39                  if (!code.trim().isEmpty()) {
40                      allowedList.add(code.trim());
41                  }
42              }
43          }
44  
45          List<Card> cardPool = Sets.generateRandomCardPool(45, allowedColors, onlyBasicLands, allowedList);
46          return ComputerPlayer.buildDeck(DECK_SIZE, cardPool, allowedColors, onlyBasicLands);
47      }
48  
49      public static DeckCardLists buildRandomDeckAndInitCards(String colors, boolean onlyBasicLands) {
50          return buildRandomDeckAndInitCards(colors, onlyBasicLands, "");
51      }
52  
53      public static DeckCardLists buildRandomDeckAndInitCards(String colors, boolean onlyBasicLands, String allowedSets) {
54          Deck deck = buildRandomDeck(colors, onlyBasicLands, allowedSets);
55  
56          DeckCardLists deckList = new DeckCardLists();
57          for (Card card : deck.getCards()) {
58              CardInfo cardInfo = CardRepository.instance.findCard(card.getExpansionSetCode(), card.getCardNumber());
59              if (cardInfo != null) {
60                  deckList.getCards().add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
61              }
62          }
63          return deckList;
64      }

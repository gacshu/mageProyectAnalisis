3   import mage.constants.PhaseStep;
4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   public class CardUtilTest extends CardTestPlayerBase {
9       // Whenever you cast or copy an instant or sorcery spell, reveal the top card of your library.
10      // If it’s a nonland card, you may cast it by paying {1} rather than paying its mana cost.
11      // If it’s a land card, put it onto the battlefield.
12      private static final String jadzi = "Jadzi, Oracle of Arcavios";
13      // Whenever you discard a nonland card, you may cast it from your graveyard.
14      private static final String oskar = "Oskar, Rubbish Reclaimer";
15      // MDFC where the back side is a land "Akoum Teeth"
16      private static final String akoumWarrior = "Akoum Warrior"; // {5}{R}
17      // Discard your hand, then draw a card for each card you’ve discarded this turn.
18      private static final String changeOfFortune = "Change of Fortune"; // {3}{R}
19      // MDFC where both sides should be playable
20      private static final String birgi = "Birgi, God of Storytelling"; // {2}{R}, frontside of Harnfel
21      private static final String harnfel = "Harnfel, Horn of Bounty"; // {4}{R}, backside of Birgi
22  
23      /**
24       * Test that it will for trigger for discarding a MDFC but will only let you cast the nonland side.
25       */
26      @Test
27      public void cantPlayLandSideOfMDFC() {
28          addCard(Zone.HAND, playerA, changeOfFortune);
29          addCard(Zone.HAND, playerA, akoumWarrior);
30  
31          addCard(Zone.BATTLEFIELD, playerA, oskar);
32          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 10);
33  
34          skipInitShuffling();
35          setStrictChooseMode(true);
36  
37          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, changeOfFortune);
38          setChoice(playerA, "Yes");
39  
40          setStopAt(1, PhaseStep.DECLARE_ATTACKERS);
41          execute();
42          assertPermanentCount(playerA, akoumWarrior, 1);
43      }
44  
45      /**
46       * Test that when both sides of a MDFC card match, we can choose either side.
47       */
48      @Test
49      public void testFrontSideOfMDFC() {
50          addCard(Zone.HAND, playerA, changeOfFortune);
51          addCard(Zone.HAND, playerA, birgi, 2);
52  
53          addCard(Zone.BATTLEFIELD, playerA, oskar);
54          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 12);
55  
56          skipInitShuffling();
57          setStrictChooseMode(true);
58  
59          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, changeOfFortune);
60          setChoice(playerA, "Whenever you discard");
61          setChoice(playerA, "Yes");
62          setChoice(playerA, "Cast " + birgi);
63          setChoice(playerA, "Yes");
64          setChoice(playerA, "Cast " + harnfel);
65  
66          setStopAt(1, PhaseStep.DECLARE_ATTACKERS);
67          execute();
68          assertPermanentCount(playerA, birgi, 1);
69          assertPermanentCount(playerA, harnfel, 1);
70      }
71  
72      /**
73       * Test that with Jadzi, you are able to play the nonland side of a MDFC, and that the alternative cost works properly.
74       */
75      @Test
76      public void testJadziPlayingLandAndCast() {
77          addCard(Zone.BATTLEFIELD, playerA, jadzi);
78          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1+1+1);
79          addCard(Zone.HAND, playerA, "Lightning Bolt", 2);
80          addCard(Zone.LIBRARY, playerA, "Cragcrown Pathway");
81          addCard(Zone.LIBRARY, playerA, akoumWarrior);
82  
83          skipInitShuffling();
84          setStrictChooseMode(true);
85  
86          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
87          setChoice(playerA, "Yes");
88          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", playerB);
89  
90          setStopAt(1, PhaseStep.DECLARE_ATTACKERS);
91          execute();
92          assertPermanentCount(playerA, akoumWarrior, 1);
93          assertPermanentCount(playerA, "Cragcrown Pathway", 1);
94      }

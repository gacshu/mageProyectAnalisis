21  public class ZipFilesReadWriteTest {

23      @Rule
24      public TemporaryFolder tempFolder = new TemporaryFolder();

26      @Test
27      public void test_Read() {
28          // exists
29          TFile fileZip = new TFile(Paths.get("src", "test", "data", "images.zip").toString());
30          Assert.assertTrue(fileZip.exists());
31          TFile fileZipDir = new TFile(Paths.get("src", "test", "data", "images.zip", "SET").toString());
32          Assert.assertTrue(fileZipDir.exists());
33          TFile fileZipFile = new TFile(Paths.get("src", "test", "data", "images.zip", "SET", "image1.png").toString());
34          Assert.assertTrue(fileZipFile.exists());
35  
36          // not exists
37          TFile fileNotZip = new TFile(Paths.get("src", "test", "data", "images-FAIL.zip").toString());
38          Assert.assertFalse(fileNotZip.exists());
39          TFile fileNotZipDir = new TFile(Paths.get("src", "test", "data", "images.zip", "SET-FAIL").toString());
40          Assert.assertFalse(fileNotZipDir.exists());
41          TFile fileNotZipFile = new TFile(Paths.get("src", "test", "data", "images.zip", "SET", "image1-FAIL.png").toString());
42          Assert.assertFalse(fileNotZipFile.exists());
43  
44          // reading
45          Assert.assertEquals(3, fileZipDir.list().length);
46          Assert.assertTrue(Arrays.asList(fileZipDir.list()).contains("image1.png"));
47          Assert.assertTrue(Arrays.asList(fileZipDir.list()).contains("image2.png"));
48          Assert.assertTrue(Arrays.asList(fileZipDir.list()).contains("image3.png"));
49      }
50  
51      @Test
52      public void test_write() {
53          try {
54              String zipPath = tempFolder.newFolder().getAbsolutePath();
55              TFile fileWriteZip = new TFile(Paths.get(zipPath, "temp-images.zip", "DIR", "test.txt").toString());
56              Assert.assertFalse(fileWriteZip.exists());
57  
58              Writer writer = new TFileWriter(fileWriteZip);
59              try {
60                  writer.write("test text");
61                  writer.close();
62                  Assert.assertTrue(fileWriteZip.exists());
63  
64                  TFileReader reader = new TFileReader(fileWriteZip);
65                  BufferedReader br = new BufferedReader(reader);
66                  Assert.assertEquals(br.readLine(), "test text");
67                  reader.close();
68              } finally {
69                  TVFS.umount();
70              }
71          } catch (IOException e) {
72              e.printStackTrace();
73              Assert.fail(e.getMessage());
74          }
75      }
76  }

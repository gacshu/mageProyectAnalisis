26      @Test
27      public void testAutoPay() {
28          testManaToPayVsLand("{R}",                "Blood Crypt", 2, 1); // should use {R}
29          testManaToPayVsLand("{1}{R}",             "Blood Crypt", 2, RedManaAbility.class); // should use {R}
30          testManaToPayVsLand("{R}{B}",             "Blood Crypt", 2, 2); // can't auto choose to pay
31          testManaToPayVsLand("{2}{R}{B}",          "Blood Crypt", 2, 2); // can't auto choose to pay
32          testManaToPayVsLand("{R}{R}{B}{B}",       "Blood Crypt", 2, 2); // can't auto choose to pay
33          testManaToPayVsLand("{R}{G}{W}{W}{U}",    "Blood Crypt", 2, RedManaAbility.class); // should use {R}
34          testManaToPayVsLand("{R}{R}{G}{W}{W}{U}", "Blood Crypt", 2, RedManaAbility.class); // should use {R}
35          testManaToPayVsLand("{R}{R}",             "Blood Crypt", 2, RedManaAbility.class); // should use {R}
36          testManaToPayVsLand("{G}{W}",             "Blood Crypt", 2, 2); // can't auto choose to pay
37          testManaToPayVsLand("{1}{G}{W}",          "Blood Crypt", 2, 1); // should use any but auto choose it
38          testManaToPayVsLand("{2}{G}{W}{U}",       "Blood Crypt", 2, 1); // should use any but auto choose it
39          testManaToPayVsLand("{3}",                "Blood Crypt", 2, 1); // should use any but auto choose it
40  
41          testManaToPayVsLand("{R}{R}{G}{W}{W}{U}", "Watery Grave",   2, 1); // should use {U}
42          testManaToPayVsLand("{R}{R}{G}{W}{W}",    "Steam Vents",    2, 1); // should use {R}
43          testManaToPayVsLand("{R}{R}{G}{B}{U}",    "Temple Garden",  2, 1); // should use {G}
44          testManaToPayVsLand("{W}{W}{G}{B}{U}",    "Sacred Foundry", 2, 1); // should use {W}
45          testManaToPayVsLand("{W}{W}{R}{B}{U}",    "Overgrown Tomb", 2, BlackManaAbility.class); // should use {B}
46          testManaToPayVsLand("{W}{W}{R}{B}{U}",    "Swamp",          1, BlackManaAbility.class);
47          testManaToPayVsLand("{W}{W}{R}{B}{U}",    "Plains",         1, WhiteManaAbility.class);
48  
49          testManaToPayVsLand("{1}{R}",   "Cavern of Souls", 2, 2); // can't auto choose to pay
50          testManaToPayVsLand("{2}",      "Cavern of Souls", 2, 2); // can't auto choose to pay
51  
52          testManaToPayVsLand("{2}", "Eldrazi Temple", 2, 2); // can't auto choose to pay
53  
54          // hybrid mana
55          testManaToPayVsLand("{W/R}{W/R}{W/R}",  "Sacred Foundry",   2, 1); // auto choose for hybrid mana: choose any
56          testManaToPayVsLand("{R}{W/R}",         "Sacred Foundry",   2, RedManaAbility.class); // auto choose for hybrid mana: we should choose {R}
57          testManaToPayVsLand("{G}{W/R}",         "Sacred Foundry",   2, 1); // auto choose for hybrid mana: choose any
58          testManaToPayVsLand("{G}{W/R}{W}",      "Sacred Foundry",   2, WhiteManaAbility.class); // auto choose for hybrid mana: choose {W}
59          testManaToPayVsLand("{W/B}{W/B}",       "Swamp",            1, BlackManaAbility.class);
60  
61          testManaToPayVsLand("{R}",      "Glimmervoid", 1, 1);
62          testManaToPayVsLand("{R}{1}",   "Glimmervoid", 1, 1);
63  
64          // we can't auto choose here:
65          // let say we auto choose {R}, then we have to use it to pay for {R} not {W/R} (as {W/R} is more generic cost)
66          // but in such case what is left to pay is {W/R}{W} and it is possible that we won't have 2 white sources
67          // Example: 1x Sacred Foundry 1x Mountain 1x Mountain
68          // we can pay {W/R}{W}{R} by using Sacred Foundry and choosing {W} then using two Mountains
69          // but if we auto choose {R} then we won't be able to pay the cost at all
70          testManaToPayVsLand("{W/R}{W}{R}", "Sacred Foundry", 2, 2);
71  
72          testManaToPayVsLand("{W/R}{R/G}", "Sacred Foundry", 2, 2); // can't auto choose to pay
73      }

76       * Mana.condenseManaCostString is used to simplify the String representation of a mana cost to make it more readable.
77       */
78      @Test
79      public void testManaCondensing() {
80          Assert.assertEquals("{5}{W}",               ManaUtil.condenseManaCostString("{1}{1}{1}{2}{W}"));
81          Assert.assertEquals("{4}{B}{B}",            ManaUtil.condenseManaCostString("{2}{B}{2}{B}"));
82          Assert.assertEquals("{6}{R}{R}{R}{U}",      ManaUtil.condenseManaCostString("{R}{1}{R}{2}{R}{3}{U}"));
83          Assert.assertEquals("{5}{B}{U}{W}",         ManaUtil.condenseManaCostString("{1}{B}{W}{4}{U}"));
84          Assert.assertEquals("{8}{B}{G}{G}{U}",      ManaUtil.condenseManaCostString("{1}{G}{1}{2}{3}{G}{B}{U}{1}"));
85          Assert.assertEquals("{3}{R}{U}",            ManaUtil.condenseManaCostString("{3}{R}{U}"));
86          Assert.assertEquals("{10}",                 ManaUtil.condenseManaCostString("{1}{2}{3}{4}"));
87          Assert.assertEquals("{B}{G}{R}{U}{W}",      ManaUtil.condenseManaCostString("{B}{G}{R}{U}{W}"));
88          Assert.assertEquals("{R}{R}",               ManaUtil.condenseManaCostString("{R}{R}"));
89          Assert.assertEquals("{U}",                  ManaUtil.condenseManaCostString("{U}"));
90          Assert.assertEquals("{2}",                  ManaUtil.condenseManaCostString("{2}"));
91          Assert.assertEquals("",                     ManaUtil.condenseManaCostString("{}"));
92          Assert.assertEquals("{5}{C}{R}{R}{R}{U}",   ManaUtil.condenseManaCostString("{R}{C}{R}{2}{R}{3}{U}"));
93      }

95      /**
96       * Common way to test ManaUtil.tryToAutoPay
97       *
98       * We get all mana abilities, then try to auto pay and compare to expected1
99       * and expected2 params.
100      *
101      * @param manaToPay Mana that should be paid using land.
102      * @param landName Land to use as mana producer.
103      * @param expected1 The amount of mana abilities the land should have.
104      * @param expected2 The amount of mana abilities that ManaUtil.tryToAutoPay
105      * should be returned after optimization.
106      */
107     private void testManaToPayVsLand(String manaToPay, String landName, int expected1, int expected2) {
108         ManaCost unpaid = new ManaCostsImpl<>(manaToPay);
109         Card card = CardRepository.instance.findCard(landName).getCard();
110         Assert.assertNotNull(card);
111 
112         Map<UUID, ActivatedManaAbilityImpl> useableAbilities = getManaAbilities(card);
113         Assert.assertEquals(expected1, useableAbilities.size());
114 
115         useableAbilities = ManaUtil.tryToAutoPay(unpaid, (LinkedHashMap<UUID, ActivatedManaAbilityImpl>) useableAbilities);
116         Assert.assertEquals(expected2, useableAbilities.size());
117     }

119     /**
120      * Another way to test ManaUtil.tryToAutoPay Here we also check what ability
121      * was auto chosen
122      *
123      * N.B. This method can be used ONLY if we have one ability left that auto
124      * choose mode! That's why we assert the following: Assert.assertEquals(1,
125      * useableAbilities.size());
126      *
127      * We get all mana abilities, then try to auto pay and compare to expected1
128      * and expected2 params.
129      *
130      * @param manaToPay         Mana that should be paid using land.
131      * @param landName          Land to use as mana producer.
132      * @param expected1         The amount of mana abilities the land should have.
133      * @param expectedChosen    The mana ability expected to be chosen.
134      */
135     private void testManaToPayVsLand(String manaToPay, String landName, int expected1, Class<? extends BasicManaAbility> expectedChosen) {
136         ManaCost unpaid = new ManaCostsImpl<>(manaToPay);
137         Card card = CardRepository.instance.findCard(landName).getCard();
138         Assert.assertNotNull(card);
139 
140         Map<UUID, ActivatedManaAbilityImpl> useableAbilities = getManaAbilities(card);
141         Assert.assertEquals(expected1, useableAbilities.size());
142 
143         useableAbilities = ManaUtil.tryToAutoPay(unpaid, (LinkedHashMap<UUID, ActivatedManaAbilityImpl>) useableAbilities);
144         Assert.assertEquals(1, useableAbilities.size());
145         ActivatedManaAbilityImpl ability = useableAbilities.values().iterator().next();
146         Assert.assertTrue("Wrong mana ability has been chosen", expectedChosen.isInstance(ability));
147     }

149     /**
150      * Extracts mana abilities from the card.
151      *
152      * @param card  Card to extract mana abilities from.
153      * @return      Map between the UUID of each ability on the card and the ability.
154      */
155     private Map<UUID, ActivatedManaAbilityImpl> getManaAbilities(Card card) {
156         Map<UUID, ActivatedManaAbilityImpl> useableAbilities = new LinkedHashMap<>();
157         for (Ability ability : card.getAbilities()) {
158             if (ability instanceof ActivatedManaAbilityImpl) {
159                 ability.newId(); // we need to assign id manually as we are not in game
160                 useableAbilities.put(ability.getId(), (ActivatedManaAbilityImpl) ability);
161             }
162         }
163         return useableAbilities;
164     }

16      @Test
17      public void testDoubleStrike() {
18          addCard(Zone.BATTLEFIELD, playerA, "Warren Instigator");
19          addCard(Zone.BATTLEFIELD, playerB, "Merfolk Looter");
20          setLife(playerB, 4);
21  
22          attack(1, playerA, "Warren Instigator");
23          block(1, playerB, "Merfolk Looter", "Warren Instigator");
24          setStopAt(1, PhaseStep.END_COMBAT);
25          execute();
26  
27          Permanent instigator = getPermanent("Warren Instigator", playerA.getId());
28          Assert.assertNotEquals(null, instigator);
29          Assert.assertEquals("Computer didn't attacked with Warren Instigator", true, instigator.isTapped());
30  
31          // should block and die
32          assertPermanentCount(playerB, "Merfolk Looter", 0);
33  
34          // creature is blocked
35          // blocker dies and second strike does nothing
36          assertLife(playerB, 4);
37      }

93      @Test
94      public void testDoubleStrikeTrampleVersusIndestructibleGod() {
95          /**
96           * Heliod, God of the Sun Legendary Enchantment Creature — God 5/6, 3W
97           * Indestructible As long as your devotion to white is less than five,
98           * Heliod isn't a creature. (Each {W} in the mana costs of permanents
99           * you control counts toward your devotion to white.) Other creatures
100          * you control have vigilance. {2}{W}{W}: Create a 2/1 white Cleric
101          * enchantment creature token.
102          */
103         addCard(Zone.BATTLEFIELD, playerA, "Heliod, God of the Sun");
104         addCard(Zone.BATTLEFIELD, playerA, "Terra Eternal"); // only for devotion
105         addCard(Zone.BATTLEFIELD, playerA, "Terra Eternal"); // only for devotion
106 
107         /* Primeval Titan
108          * Creature — Giant 6/6, 4GG
109          * Trample
110          * Whenever Primeval Titan enters the battlefield or attacks, you may search your library for up to two land cards, put them onto the battlefield tapped, then shuffle your library.
111          */
112         addCard(Zone.BATTLEFIELD, playerB, "Primeval Titan");
113         addCard(Zone.BATTLEFIELD, playerB, "Plains", 3);
114         // {W}{W}{2}
115         // If Leyline of Sanctity is in your opening hand, you may begin the game with it on the battlefield.
116         // You have hexproof.
117         addCard(Zone.HAND, playerA, "Leyline of Sanctity", 1);
118         // Enchantment - Aura {W}{2}
119         // Enchant creature
120         // Enchanted creature has double strike.
121         addCard(Zone.HAND, playerB, "Battle Mastery");
122 
123         castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Battle Mastery", "Primeval Titan");
124         attack(2, playerB, "Primeval Titan");
125         block(2, playerA, "Heliod, God of the Sun", "Primeval Titan");
126         setStopAt(2, PhaseStep.END_TURN);
127         execute();
128 
129         // creature is blocked
130         // blocker does not die and second strike can trample through for 6
131         assertLife(playerA, 14);
132         assertLife(playerB, 20);
133 
134         assertPermanentCount(playerA, "Leyline of Sanctity", 1);
135         assertPermanentCount(playerB, "Battle Mastery", 1);
136         // no creatures dies
137         assertPermanentCount(playerA, "Heliod, God of the Sun", 1);
138         assertPowerToughness(playerA, "Heliod, God of the Sun", 5, 6);
139         assertPermanentCount(playerB, "Primeval Titan", 1);
140         assertPowerToughness(playerB, "Primeval Titan", 6, 6);
141 
142     }

147     @Test
148     public void testCombatDamagePhyrexianUnlife() {
149         addCard(Zone.HAND, playerA, "Lightning Bolt", 6);
150         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 6);
151         /**
152          * Phyrexian Unlife {2}{W} Enchantment
153          *
154          * You don't lose the game for having 0 or less life. As long as you
155          * have 0 or less life, all damage is dealt to you as though its source
156          * had infect.
157          */
158         addCard(Zone.BATTLEFIELD, playerA, "Phyrexian Unlife");
159         addCard(Zone.BATTLEFIELD, playerB, "Silvercoat Lion", 2);
160 
161         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
162         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
163         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
164         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
165         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
166         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Lightning Bolt", "targetPlayer=PlayerA");
167 
168         attack(2, playerB, "Silvercoat Lion");
169         attack(2, playerB, "Silvercoat Lion");
170 
171         setStopAt(2, PhaseStep.END_TURN);
172         execute();
173 
174         // No poison counter yet
175         assertCounterCount(playerA, CounterType.POISON, 0);
176         assertPermanentCount(playerB, "Silvercoat Lion", 2);
177 
178         // 4 damage from attack must be dealt as damage (not as poison counter)
179         assertLife(playerA, -2);
180         assertLife(playerB, 20);
181     }

2   package org.mage.test.combat;
3   
4   import mage.constants.PhaseStep;
5   import mage.constants.Zone;
6   import mage.game.permanent.Permanent;
7   import org.junit.Assert;
8   import org.junit.Test;
9   import org.mage.test.serverside.base.CardTestPlayerBase;
10  
11  /**
12   *
13   * @author LevelX2
14   */
15  public class RemoveFromCombatTest extends CardTestPlayerBase {
16  
17      /**
18       * In a test game against the AI, it attacked me with a Stomping Ground
19       * animated by an Ambush Commander and boosted it with the Commander's
20       * second ability. I killed the Commander. The now non-creature land
21       * continued attacking and dealt 3 damage to me.
22       */
23      @Test
24      public void testLeavesCombatIfNoLongerACreature() {
25          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);
26          addCard(Zone.HAND, playerA, "Lightning Blast", 1);
27  
28          addCard(Zone.BATTLEFIELD, playerB, "Forest", 2);
29          addCard(Zone.BATTLEFIELD, playerB, "Elvish Mystic", 1);
30  
31          // Forests you control are 1/1 green Elf creatures that are still lands.
32          // {1}{G},Sacrifice an Elf: Target creature gets +3/+3 until end of turn.
33          addCard(Zone.BATTLEFIELD, playerB, "Ambush Commander", 1);
34          addCard(Zone.BATTLEFIELD, playerB, "Stomping Ground");
35  
36          attack(2, playerB, "Stomping Ground");
37          activateAbility(2, PhaseStep.DECLARE_ATTACKERS, playerB, "{1}{G}", "Stomping Ground");
38          setChoice(playerB, "Elvish Mystic");
39          castSpell(2, PhaseStep.DECLARE_BLOCKERS, playerA, "Lightning Blast", "Ambush Commander");
40          setStopAt(2, PhaseStep.COMBAT_DAMAGE);
41          execute();
42  
43          assertGraveyardCount(playerB, "Elvish Mystic", 1);
44          assertGraveyardCount(playerA, "Lightning Blast", 1);
45          assertGraveyardCount(playerB, "Ambush Commander", 1);
46  
47          assertPowerToughness(playerB, "Stomping Ground", 0, 0);
48  
49          Permanent stompingGround = getPermanent("Stomping Ground", playerB);
50          Assert.assertEquals("Stomping Ground has to be removed from combat", false, stompingGround.isAttacking());
51  
52          assertLife(playerA, 20);
53          assertLife(playerB, 20);
54  
55      }

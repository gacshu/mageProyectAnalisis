24      /**
25       * I was using Anafenza, the Foremost as Commander. She attacked and traded
26       * with two creatures. I moved Anafenza to the Command Zone, but the
27       * opponent's creatures "when {this} dies" abilities triggered. Since
28       * Anafenza and those creatures all received lethal damage at the same time,
29       * the creatures should have been exiled due to Anafenza's replacement
30       * effect, but I guess since the logic asks if you want to use the Command
31       * Zone replacement effect first, that it doesn't see her leaving the
32       * battlefield at the same time as the other creatures.
33       *
34       * http://blogs.magicjudges.org/rulestips/2015/05/anafenza-vs-deathmist-raptor/
35       */
36      @Test
37      public void testAnafenzaExileInCombat() {
38          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
39          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
40          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
41  
42          // When Runed Servitor dies, each player draws a card.
43          addCard(Zone.BATTLEFIELD, playerB, "Runed Servitor", 2);
44  
45          // Whenever Anafenza, the Foremost attacks, put a +1/+1 counter on another target tapped creature you control.
46          // If a creature card would be put into an opponent's graveyard from anywhere, exile it instead.
47          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Anafenza, the Foremost");
48  
49          attack(3, playerA, "Anafenza, the Foremost");
50          block(3, playerB, "Runed Servitor:0", "Anafenza, the Foremost");
51          block(3, playerB, "Runed Servitor:1", "Anafenza, the Foremost");
52  
53          setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
54          execute();
55  
56          assertLife(playerA, 40);
57          assertLife(playerB, 40);
58  
59          assertExileCount("Runed Servitor", 2);
60  
61          assertCommandZoneCount(playerA, "Anafenza, the Foremost", 1);
62          assertGraveyardCount(playerA, "Anafenza, the Foremost", 0);
63  
64          assertHandCount(playerA, 1); // turn 3 draw
65          assertHandCount(playerB, 1); // turn 2 draw
66  
67      }

69      /**
70       * Token don't go to exile because they are no creature cards
71       */
72      @Test
73      public void testAnafenzaExileInCombatOmnathToken() {
74          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
75          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
76          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
77          addCard(Zone.BATTLEFIELD, playerA, "Acidic Slime", 1);
78  
79          addCard(Zone.HAND, playerB, "Forest", 2);
80          // <i>Landfall</i> &mdash; Whenever a land enters the battlefield under your control, put a 5/5 red and green Elemental creature token onto the battlefield.
81          // Whenever Omnath, Locus of Rage or another Elemental you control dies, Omnath deals 3 damage to any target.
82          addCard(Zone.BATTLEFIELD, playerB, "Omnath, Locus of Rage", 1);
83  
84          // Whenever Anafenza, the Foremost attacks, put a +1/+1 counter on another target tapped creature you control.
85          // If a creature card would be put into an opponent's graveyard from anywhere, exile it instead.
86          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Anafenza, the Foremost");
87  
88          playLand(2, PhaseStep.PRECOMBAT_MAIN, playerB, "Forest");
89          playLand(4, PhaseStep.PRECOMBAT_MAIN, playerB, "Forest");
90  
91          attack(5, playerA, "Acidic Slime");
92          block(5, playerB, "Elemental Token:0", "Acidic Slime");
93          attack(5, playerA, "Anafenza, the Foremost");
94          block(5, playerB, "Elemental Token:1", "Anafenza, the Foremost");
95          addTarget(playerB, playerA);
96  
97          setStopAt(5, PhaseStep.POSTCOMBAT_MAIN);
98          execute();
99  
100         assertExileCount(playerA, 0);
101         assertExileCount(playerB, 0);
102 
103         assertPermanentCount(playerB, "Elemental Token", 1);
104 
105         assertGraveyardCount(playerA, "Acidic Slime", 1);
106         assertGraveyardCount(playerA, "Anafenza, the Foremost", 0);
107         assertCommandZoneCount(playerA, "Anafenza, the Foremost", 1);
108 
109         assertHandCount(playerA, 2); // turn 3 + 5 draw
110         assertHandCount(playerB, 2); // turn 2 + 4 draw
111 
112         assertLife(playerA, 37);
113         assertLife(playerB, 40);
114     }

116     // Anafenza + Animated permanents
117     @Test
118     public void testAnafenzaExileAnimatedPermanents() {
119         addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
120         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
121         addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
122 
123         // {0}: Tap all lands you control. Chimeric Idol becomes a 3/3 Turtle artifact creature until end of turn.
124         addCard(Zone.BATTLEFIELD, playerB, "Chimeric Idol");
125         addCard(Zone.BATTLEFIELD, playerB, "Mountain", 3);
126 
127         // Whenever Anafenza, the Foremost attacks, put a +1/+1 counter on another target tapped creature you control.
128         // If a creature card would be put into an opponent's graveyard from anywhere, exile it instead.
129         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Anafenza, the Foremost"); // 4/4
130 
131         activateAbility(2, PhaseStep.PRECOMBAT_MAIN, playerB, "{0}: Tap all lands you control");
132 
133         attack(2, playerB, "Chimeric Idol");
134         block(2, playerA, "Anafenza, the Foremost", "Chimeric Idol");
135 
136         setStopAt(2, PhaseStep.POSTCOMBAT_MAIN);
137         execute();
138 
139         assertLife(playerA, 40);
140         assertLife(playerB, 40);
141 
142         assertExileCount("Chimeric Idol", 1);
143         assertGraveyardCount(playerB, "Chimeric Idol", 0);
144         assertPermanentCount(playerB, "Chimeric Idol", 0);
145 
146         assertTappedCount("Mountain", true, 3);
147 
148     }

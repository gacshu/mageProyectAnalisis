74      /**
75       * I tried playing it in Mairsil the Pretender commander deck and found 2
76       * cases where the creature is not properly destroyed:
77       *
78       * Using Arcanis the Omnipotent ability to return my commander to hand while
79       * trigger is on the stack I got Mairsil to hand then got asked whether I
80       * want to put him in gy so I answered yes assuming it cannot be destroyed
81       * while in my hand. He got put in graveyard straight from my hand.
82       * According to Oracle rules I should get a creature from top of the deck
83       * and still have my commander in hand. After giving my commander undying
84       * with shifting shadow trigger on the stack he got put in gy with undying
85       * not triggering.
86       *
87       * All this points to the card being hardcoded to put the creature into
88       * graveyard instead of simply destroying it
89       */

138     @Test
139     public void useShifitingShadowAndEndlingTest() {
140         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
141         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
142         addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
143         addCard(Zone.LIBRARY, playerA, "Silvercoat Lion", 3);
144         skipInitShuffling();
145         // Enchant creature
146         // Enchanted creature has haste and “At the beginning of your upkeep, destroy this creature. 
147         // Reveal cards from the top of your library until you reveal a creature card.
148         // Put that card onto the battlefield and attach Shifting Shadow to it, 
149         // then put all other cards revealed this way on the bottom of your library in a random order.”
150         addCard(Zone.HAND, playerA, "Shifting Shadow", 1); // {2}{R}
151         // Tap: Draw three cards.
152         // {2}{U}{U}: Return Arcanis the Omnipotent to its owner's hand.        
153         addCard(Zone.HAND, playerA, "Arcanis the Omnipotent", 1); // Creature {3}{U}{U}{U}        
154         // {B}: Endling gains menace until end of turn.
155         // {B}: Endling gains deathtouch until end of turn.
156         // {B}: Endling gains undying until end of turn.
157         // {1}: Endling gets +1/-1 or -1/+1 until end of turn.   
158         addCard(Zone.HAND, playerA, "Endling", 1); // Creature {3}{U}{U}{U}
159 
160         
161         
162         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Mairsil, the Pretender");                
163         setChoice(playerA, true); // Exile a card
164         setChoice(playerA, true); // Exile from Hand
165         setChoice(playerA, "Endling");
166         
167         
168         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Shifting Shadow", "Mairsil, the Pretender");
169 
170         activateAbility(5, PhaseStep.UPKEEP, playerA, "{B}: {this} gains undying", NO_TARGET, "At the beginning of your upkeep");
171         setChoice(playerA, false); // Don't move commander to command zone because can come back by Undying
172         
173         setChoice(playerA, true); // Exile a card (Mairsil comes back from Undying)
174         setChoice(playerA, true); // Exile from hand
175         setChoice(playerA, "Arcanis the Omnipotent");
176         
177         setStopAt(5, PhaseStep.PRECOMBAT_MAIN);
178 
179         setStrictChooseMode(true);
180         execute();
181 
182         assertLife(playerA, 40);
183         assertLife(playerB, 40);
184 
185         assertExileCount("Endling", 1);
186         assertExileCount("Arcanis the Omnipotent", 1);
187 
188         assertCommandZoneCount(playerA, "Mairsil, the Pretender", 0);
189         assertGraveyardCount(playerA, "Mairsil, the Pretender", 0);
190         assertPermanentCount(playerA, "Mairsil, the Pretender", 1); // Returns from Undying
191         assertPowerToughness(playerA, "Mairsil, the Pretender", 5, 5);
192         
193         assertPermanentCount(playerA, "Shifting Shadow", 1);  // Enchants Silvercoat Lion    
194         assertPermanentCount(playerA, "Silvercoat Lion", 1); 
195     }

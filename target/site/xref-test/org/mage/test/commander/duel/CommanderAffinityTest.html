87      @Test
88      public void test_Gained_Affinity() {
89          // bug: Mycosynth Golem did not allow my commander, Karn, Silver Golem, to cost 0 even though I had 7+ artifacts on the board.
90  
91          Ability ability = new SimpleActivatedAbility(Zone.ALL, new CreateTokenEffect(new ArtifactWallToken(), 7), new ManaCostsImpl<>("{R}"));
92          addCustomCardWithAbility("generate tokens", playerA, ability);
93          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
94          //
95          addCard(Zone.COMMAND, playerA, "Karn, Silver Golem", 1); // {5}
96          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5);
97          //
98          // Destroy target artifact.
99          addCard(Zone.HAND, playerA, "Ancient Grudge", 1); // {1}{R}
100         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 2);
101         //
102         // Artifact creature spells you cast have affinity for artifacts. (They cost {1} less to cast for each artifact you control.)
103         addCard(Zone.BATTLEFIELD, playerA, "Mycosynth Golem", 1);
104 
105         // Affinity for artifacts
106         // Artifact creature spells you cast have affinity for artifacts.
107         // addCard(Zone.BATTLEFIELD, playerA, "Mycosynth Golem");
108         // addCard(Zone.HAND, playerA, "Alpha Myr"); // Creature - Myr  2/1
109 
110         checkCommandCardCount("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Karn, Silver Golem", 1);
111         checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Karn, Silver Golem", true);
112 
113         // first cast for 5 and destroy (prepare commander with additional cost)
114         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Karn, Silver Golem");
115         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
116         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ancient Grudge", "Karn, Silver Golem");
117         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
118         checkCommandCardCount("after destroy ", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Karn, Silver Golem", 1);
119         checkPlayableAbility("after destroy", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Karn, Silver Golem", false);
120         setChoice(playerA, true); // move to command zone
121         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
122 
123         // can't do the second cast with additional cost (must pay 2 + 5, but have only R)
124         checkPlayableAbility("after move", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Karn, Silver Golem", false);
125 
126         // generate artifact tokens
127         activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: Create");
128         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
129         checkPermanentCount("after tokens", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Wall Token", 7);
130         checkPlayableAbility("after tokens", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Karn, Silver Golem", true);
131 
132         setStrictChooseMode(true);
133         setStopAt(1, PhaseStep.END_TURN);
134         execute();
135     }

137     @Test
138     public void test_Gained_Convoke() {
139         // bug:
140         // https://github.com/magefree/mage/issues/7171
141         // Commander doesn't get 'convoke' when chief engineer is on the battlefield, so for instance I can't cast Breya by tapping other creatures.
142 
143         // Artifact spells you cast have convoke.
144         addCard(Zone.BATTLEFIELD, playerA, "Chief Engineer", 1);
145         //
146         addCard(Zone.COMMAND, playerA, "Karn, Silver Golem", 1); // {5}
147         addCard(Zone.BATTLEFIELD, playerA, "Mountain", 5 - 2);
148         addCard(Zone.BATTLEFIELD, playerA, "Grizzly Bears", 2);
149 
150         // 3 mana + 2 convoke
151         checkPlayableAbility("before", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Cast Karn, Silver Golem", true);
152         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Karn, Silver Golem");
153         addTarget(playerA, "Grizzly Bears"); // convoke cost
154         addTarget(playerA, "Grizzly Bears"); // convoke cost
155 
156         setStrictChooseMode(true);
157         setStopAt(1, PhaseStep.END_TURN);
158         execute();
159 
160         assertPermanentCount(playerA, "Karn, Silver Golem", 1);
161     }

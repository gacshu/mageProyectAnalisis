14      @Test
15      public void test_TriggerOnSimpleCommander() {
16          // Yorvo, Lord of Garenbrig enters the battlefield with four +1/+1 counters on it.
17          // Whenever another green creature enters the battlefield under your control, put a +1/+1 counter on Yorvo. Then if that creature’s power is greater than Yorvo’s power, put another +1/+1 counter on Yorvo.
18          addCard(Zone.HAND, playerA, "Yorvo, Lord of Garenbrig", 5); // {G}{G}{G}
19          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
20          //
21          addCard(Zone.COMMAND, playerA, "Aggressive Mammoth"); // {3}{G}{G}{G}
22          addCard(Zone.BATTLEFIELD, playerA, "Forest", 6);
23  
24          // prepare yorvo
25          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Yorvo, Lord of Garenbrig");
26          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
27  
28          // cast commander
29          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Aggressive Mammoth");
30          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
31          checkPermanentCounters("must get +2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Yorvo, Lord of Garenbrig", CounterType.P1P1, 4 + 2);
32          checkPermanentCount("must play commander", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Aggressive Mammoth", 1);
33  
34          setStrictChooseMode(true);
35          setStopAt(1, PhaseStep.END_TURN);
36          execute();
37      }

39      @Test
40      public void test_TriggerOnUroTitanOfNaturesWrath() {
41          // Yorvo, Lord of Garenbrig enters the battlefield with four +1/+1 counters on it.
42          // Whenever another green creature enters the battlefield under your control, put a +1/+1 counter on Yorvo. Then if that creature’s power is greater than Yorvo’s power, put another +1/+1 counter on Yorvo.
43          addCard(Zone.HAND, playerA, "Yorvo, Lord of Garenbrig", 5); // {G}{G}{G}
44          addCard(Zone.BATTLEFIELD, playerA, "Forest", 3);
45          //
46          // When Uro enters the battlefield, sacrifice it unless it escaped.
47          // Whenever Uro enters the battlefield or attacks, you gain 3 life and draw a card, then you may put a land card from your hand onto the battlefield.
48          // Escape-{G}{G}{U}{U}, Exile five other cards from your graveyard. (You may cast this card from your graveyard for its escape cost.)
49          addCard(Zone.COMMAND, playerA, "Uro, Titan of Nature's Wrath"); // {1}{G}{U}
50          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
51          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
52  
53          // prepare yorvo
54          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Yorvo, Lord of Garenbrig");
55          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
56  
57          // cast commander
58          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Uro, Titan of Nature's Wrath");
59  
60          // If the entering green creature dies before Yorvo’s triggered ability resolves, use its power as it last
61          // existed on the battlefield to determine whether Yorvo gets a second +1/+1 counter.
62          // (2019-10-04)
63  
64          // order triggers to remove commander first
65          setChoice(playerA, "Whenever {this} enters the battlefield or attacks"); // draw trigger
66          setChoice(playerA, "Whenever another green creature enters the battlefield"); // get counters trigger
67          //setChoice(playerA, "When {this} enters the battlefield, sacrifice it"); // sacrifice trigger must be on top
68  
69          setChoice(playerA, true); // return commander to command zone
70          setChoice(playerA, false); // do not put land to battlefield
71  
72          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
73          checkPermanentCounters("must get +2", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Yorvo, Lord of Garenbrig", CounterType.P1P1, 4 + 2);
74          checkCommandCardCount("return commander", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Uro, Titan of Nature's Wrath", 1);
75  
76          setStrictChooseMode(true);
77          setStopAt(1, PhaseStep.END_TURN);
78          execute();
79      }

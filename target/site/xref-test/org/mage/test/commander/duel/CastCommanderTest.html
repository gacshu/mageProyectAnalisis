32      @Test
33      public void testCastKestiaCommander() {
34          // Bestow {3}{G}{W}{U}
35          // Enchanted creature gets +4/+4.
36          // Whenever an enchanted creature or enchantment creature you control attacks, draw a card.        
37          addCard(Zone.COMMAND, playerA, "Kestia, the Cultivator", 1); // Creature  {1}{G}{W}{U} 4/4
38          
39          addCard(Zone.BATTLEFIELD, playerA, "Forest", 5);
40          addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
41          addCard(Zone.BATTLEFIELD, playerA, "Island", 5);
42  
43          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
44          
45          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Kestia, the Cultivator using bestow", "Silvercoat Lion");
46  
47          setStrictChooseMode(true);
48          
49          setStopAt(1, PhaseStep.BEGIN_COMBAT);
50          execute();
51  
52          assertLife(playerA, 40);
53          assertLife(playerB, 40);
54  
55          
56          assertPermanentCount(playerA, "Kestia, the Cultivator", 1);
57          
58          Permanent kestia = getPermanent("Kestia, the Cultivator", playerA);
59          Assert.assertNotEquals("Kestia may not be an creature", true, kestia.isCreature(currentGame));
60          Assert.assertEquals("Kestia has to be an enchantment", true, kestia.isEnchantment(currentGame));
61          
62          assertPowerToughness(playerA, "Silvercoat Lion", 6, 6);
63      }

66      @Test
67      public void testCastPatronOfTheOrochiCommander() {
68          // Snake offering (You may cast this card any time you could cast an instant 
69          // by sacrificing a Snake and paying the difference in mana costs between this 
70          // and the sacrificed Snake. Mana cost includes color.)
71          // {T}: Untap all Forests and all green creatures. Activate this ability only once each turn.    
72          addCard(Zone.COMMAND, playerA, "Patron of the Orochi", 1); // Creature  {6}{G}{G} 7/7
73          
74          // First strike
75          addCard(Zone.BATTLEFIELD, playerA, "Coiled Tinviper", 1); // Creature Snake {3} 2/1
76          addCard(Zone.BATTLEFIELD, playerA, "Forest", 8); // Cost reduction does not work for getPlayable so you need to have 8 mana available
77         
78          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Patron of the Orochi");
79          setChoice(playerA, true);
80          addTarget(playerA, "Coiled Tinviper");
81  
82          setStrictChooseMode(true);
83          
84          setStopAt(1, PhaseStep.BEGIN_COMBAT);
85          execute();
86  
87          assertLife(playerA, 40);
88          assertLife(playerB, 40);
89         
90          assertPermanentCount(playerA, "Patron of the Orochi", 1);        
91          assertGraveyardCount(playerA, "Coiled Tinviper", 1);
92          assertTappedCount("Forest", false, 3);
93      }

97      @Test
98      public void testCastCommanderAndUnexpectedlyAbsent() {
99          setStrictChooseMode(true);
100         
101         playerA.getLibrary().clear();
102         
103         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 2);
104         addCard(Zone.BATTLEFIELD, playerA, "Plains", 5);
105 
106         // Put target nonland permanent into its owner's library just beneath the top X cards of that library.
107         addCard(Zone.HAND, playerA, "Unexpectedly Absent"); // Instant {X}{W}{W}
108 
109         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ob Nixilis of the Black Oath", true); // {3}{B}{B}
110 
111         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Unexpectedly Absent", "Ob Nixilis of the Black Oath");
112         setChoice(playerA, "X=0");
113         setChoice(playerA, true); // Move Commander to command zone instead library?
114 
115         setStopAt(1, PhaseStep.BEGIN_COMBAT);
116         execute();
117 
118         assertPermanentCount(playerA, "Ob Nixilis of the Black Oath", 0);
119         assertLibraryCount(playerA, "Ob Nixilis of the Black Oath", 0);
120         assertCommandZoneCount(playerA, "Ob Nixilis of the Black Oath", 1);
121 
122         assertLibraryCount(playerA, 0);
123 
124         assertGraveyardCount(playerA, "Unexpectedly Absent", 1);
125 
126         assertLife(playerA, 40);
127         assertLife(playerB, 40);        
128     }

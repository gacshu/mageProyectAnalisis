39      @Test
40      public void testCommanderDamage() {
41          setLife(playerA, 20);
42          setLife(playerB, 20);
43          addCard(Zone.BATTLEFIELD, playerA, "Plains", 6);
44          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
45          // Enchant creature
46          // Enchanted creature gets +4/+4, has flying and first strike, and is an Angel in addition to its other types.
47          // When enchanted creature dies, return Angelic Destiny to its owner's hand.
48          addCard(Zone.HAND, playerA, "Angelic Destiny");
49  
50          addCard(Zone.BATTLEFIELD, playerA, "Teferi, Mage of Zhalfir");
51  
52          // Daxos of Meletis can't be blocked by creatures with power 3 or greater.
53          // Whenever Daxos of Meletis deals combat damage to a player, exile the top card of that player's library.
54          // You gain life equal to that card's converted mana cost. Until end of turn, you may cast that card
55          // and you may spend mana as though it were mana of any color to cast it.
56          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Daxos of Meletis");
57          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
58          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Angelic Destiny", "Daxos of Meletis");
59  
60          attack(3, playerA, "Daxos of Meletis");
61          attack(5, playerA, "Daxos of Meletis");
62          attack(7, playerA, "Daxos of Meletis");
63          attack(9, playerA, "Daxos of Meletis");
64          checkPT("before lost", 9, PhaseStep.PRECOMBAT_MAIN, playerA, "Daxos of Meletis", 6, 6);
65  
66          setStrictChooseMode(true);
67          setStopAt(9, PhaseStep.POSTCOMBAT_MAIN);
68          execute();
69  
70          assertPermanentCount(playerA, "Daxos of Meletis", 1);
71          assertPowerToughness(playerA, "Daxos of Meletis", 6, 6); // no effects removes after game over -- users and tests can get last game state with all affected effects
72  
73          assertWonTheGame(playerA);
74          assertLostTheGame(playerB);
75      }

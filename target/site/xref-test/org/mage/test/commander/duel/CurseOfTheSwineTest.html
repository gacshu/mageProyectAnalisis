14  public class CurseOfTheSwineTest  extends CardTestCommanderDuelBase {
15  
16      @Override
17      protected Game createNewGameAndPlayers() throws GameException, FileNotFoundException {
18          setDecknamePlayerA("CommanderDuel_UW.dck"); // Commander = Daxos of Meletis
19          return super.createNewGameAndPlayers();
20      }
21  
22      /**
23       *  In a Commander game, if a commander is put into the command zone
24       *  instead of being exiled by Curse of the Swine, its controller will
25       *  still get a Boar token.
26       */
27      @Test
28      public void TestCurseOfTheSwine() {
29          addCard(Zone.BATTLEFIELD, playerA, "Island", 2);
30          addCard(Zone.BATTLEFIELD, playerA, "Plains", 1);
31          addCard(Zone.HAND, playerA, "Curse of the Swine");
32  
33          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Daxos of Meletis");
34          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Curse of the Swine");
35          setChoice(playerA, "X=1");
36          // Daxos of Meletis is auto-chosen since only target
37  
38          setStopAt(3, PhaseStep.BEGIN_COMBAT);
39          execute();
40  
41          assertCommandZoneCount(playerA, "Daxos of Meletis", 1);
42          assertExileCount("Daxos of Meletis", 0);
43          assertPermanentCount(playerA, "Boar Token", 1);
44      }
45  }

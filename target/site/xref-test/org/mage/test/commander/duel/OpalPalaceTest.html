12  public class OpalPalaceTest extends CardTestCommanderDuelBase {
13  
14      /**
15       * I cast my commander with Opal Palace's second ability and it did not
16       * receive a +1/+1 counter the first time it was cast (rulings say it should
17       * on the first time cast).
18       */
19      @Test
20      public void testFirstAbility() {
21          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
22          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
23  
24          // {T}: Add {C}.
25          // {1}, {T}: Add one mana of any color in your commander's color identity.
26          // If you spend this mana to cast your commander, it enters the battlefield with a number of +1/+1 counters on it
27          // equal to the number of times it's been cast from the command zone this game.
28          addCard(Zone.BATTLEFIELD, playerA, "Opal Palace", 1);
29  
30          // showHand("hand", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
31          // showCommand("command", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
32          // showAvailableAbilities("abi", 1, PhaseStep.PRECOMBAT_MAIN, playerA);
33          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}");
34          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{1}, {T}");
35          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Ob Nixilis of the Black Oath"); // {3}{B}{B}
36  
37          setStrictChooseMode(true);
38          setStopAt(1, PhaseStep.BEGIN_COMBAT);
39          execute();
40  
41          assertLife(playerA, 40);
42          assertLife(playerB, 40);
43  
44          assertPermanentCount(playerA, "Ob Nixilis of the Black Oath", 1);
45          assertCounterCount("Ob Nixilis of the Black Oath", CounterType.P1P1, 1);
46      }
47  }

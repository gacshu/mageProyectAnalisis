12  public class ExtraTurnTest extends CardTestPlayerBase {
13  
14      // Test that rollback works correctly when extra turn is taken during an opponent turn.
15      @Test
16      public void testThatRollbackWorksCorrectlyWithExtraTurn() {
17          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
18          // The next sorcery card you cast this turn can be cast as though it had flash.
19          addCard(Zone.HAND, playerA, "Quicken");
20          // Take an extra turn after this one.
21          addCard(Zone.HAND, playerA, "Time Walk");
22  
23          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Quicken", true);
24          castSpell(2, PhaseStep.PRECOMBAT_MAIN, playerA, "Time Walk");
25  
26          rollbackTurns(3, PhaseStep.PRECOMBAT_MAIN, playerA, 0);
27  
28          setStopAt(4, PhaseStep.PRECOMBAT_MAIN);
29          execute();
30  
31          assertActivePlayer(playerA);
32      }
33  }

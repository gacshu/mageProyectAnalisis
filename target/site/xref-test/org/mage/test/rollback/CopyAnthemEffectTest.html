4   import mage.constants.Zone;
5   import org.junit.Test;
6   import org.mage.test.serverside.base.CardTestPlayerBase;

8   public class CopyAnthemEffectTest extends CardTestPlayerBase {
9   
10      // Addresses issue #6618 - Creatures get +1/+1 when we cancel the cast of a spell
11  
12      // Creatures of one player get +1/+1 every time someone cancel the cast of a spell.
13      // Looks like the repeating effect was Force Of Virtue's static ability
14      // There was also a Mirrormade cast to copy Force of Virtue
15  
16      // Further investigation shown that the problem could be also reduced to any anthem effect copied by other permanent, then rollbacking
17      @Test
18      public void copyAnthemEffect() {
19          addCard(Zone.BATTLEFIELD, playerA, "Runeclaw Bear"); // 2/2 vanilla creature
20          addCard(Zone.BATTLEFIELD, playerA, "Glorious Anthem"); // creatures you control have +1/+1
21          addCard(Zone.BATTLEFIELD, playerA, "Island", 3);
22  
23          addCard(Zone.HAND, playerA, "Copy Enchantment");
24  
25          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Copy Enchantment");
26          setChoice(playerA, true);
27          setChoice(playerA, "Glorious Anthem");
28  
29          rollbackTurns(3, PhaseStep.UPKEEP, playerA, 0);
30  
31          setStopAt(3, PhaseStep.BEGIN_COMBAT);
32          execute();
33  
34          assertPowerToughness(playerA, "Runeclaw Bear", 4, 4);
35      }
36  
37  }

16      @Test
17      public void testDragonWhelpActivatedFourTimes() {
18          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 4);

20          // 2/3
21          // {R}: Dragon Whelp gets +1/+0 until end of turn. If this ability has been activated four or more times this turn, sacrifice Dragon Whelp at the beginning of the next end step.
22          addCard(Zone.BATTLEFIELD, playerA, "Dragon Whelp", 1); // 2/3
23  
24          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: ");

28          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: ");
29          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: ");

36          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: ");
37          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "{R}: ");

43          execute();

46          assertGraveyardCount(playerA, "Dragon Whelp", 0);
47  
48          assertLife(playerA, 20);
49          assertLife(playerB, 12);

52      @Test
53      public void testBriarbridgePatrol() {
54          setStrictChooseMode(true);
55  
56          // Whenever Briarbridge Patrol deals damage to one or more creatures, investigate (Create a colorless Clue artifact token onto the battlefield with "{2}, Sacrifice this artifact: Draw a card.").
57          // At the beginning of each end step, if you sacrificed three or more Clues this turn, you may put a creature card from your hand onto the battlefield.
58          addCard(Zone.BATTLEFIELD, playerA, "Briarbridge Patrol", 1); // 3/3
59  
60          addCard(Zone.BATTLEFIELD, playerB, "Pillarfield Ox", 1); // 2/4
61  
62          attack(1, playerA, "Briarbridge Patrol");
63          block(1, playerB, "Pillarfield Ox", "Briarbridge Patrol");
64  
65          attack(3, playerA, "Briarbridge Patrol");
66          block(3, playerB, "Pillarfield Ox", "Briarbridge Patrol");
67  
68          rollbackTurns(3, PhaseStep.POSTCOMBAT_MAIN, playerA, 0);
69  
70          rollbackAfterActionsStart();
71          attack(3, playerA, "Briarbridge Patrol");
72          block(3, playerB, "Pillarfield Ox", "Briarbridge Patrol");
73          rollbackAfterActionsEnd();
74  
75          setStopAt(3, PhaseStep.END_TURN);
76          execute();
77  
78          assertLife(playerA, 20);
79          assertLife(playerB, 20);
80  
81          assertPermanentCount(playerA, "Clue Token", 2);
82  
83      }

87          // Create two 1/1 white Bird creature tokens with flying.
88          // Flashbackâ€”Tap three untapped white creatures you control. (You may cast this card from your graveyard for its flashback cost. Then exile it.)
89          addCard(Zone.LIBRARY, playerA, "Battle Screech", 1); // Sorcery {2}{W}{W}
90          skipInitShuffling();
91          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
92          addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
93  
94          castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Battle Screech", true);
95  
96          activateAbility(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Flashback");
97          setChoice(playerA, "Bird Token");
98          setChoice(playerA, "Bird Token");
99          setChoice(playerA, "Silvercoat Lion");
100 
101         setStrictChooseMode(true);
102         setStopAt(3, PhaseStep.POSTCOMBAT_MAIN);
103 
104         execute();
105 
106         // Before rollback
107         assertTappedCount("Plains", true, 4);
108         assertTappedCount("Bird Token", true, 2);
109         assertTappedCount("Silvercoat Lion", true, 1);
110         assertPermanentCount(playerA, "Bird Token", 4);
111         assertHandCount(playerA, 0);
112         assertExileCount(playerA, "Battle Screech", 1);
113 
114         currentGame.rollbackTurns(2);
115 
116         // After rollback to turn 1
117         assertTappedCount("Plains", true, 0);
118         assertTappedCount("Silvercoat Lion", true, 0);
119         assertPermanentCount(playerA, "Bird Token", 0);
120         assertLibraryCount(playerA, "Battle Screech", 1);

124     @Test
125     public void rollbackBuffUntilEndOfTurnTest() {
126         // Target creature gets +1/+1 and gains flying and first strike until end of turn.
127         addCard(Zone.LIBRARY, playerA, "Aerial Maneuver", 1); // Instant {1}{W}
128         skipInitShuffling();
129         addCard(Zone.BATTLEFIELD, playerA, "Plains", 2);
130         addCard(Zone.BATTLEFIELD, playerA, "Silvercoat Lion", 1);
131 
132         castSpell(3, PhaseStep.PRECOMBAT_MAIN, playerA, "Aerial Maneuver", "Silvercoat Lion");
133 
134         setStrictChooseMode(true);
135         setStopAt(3, PhaseStep.BEGIN_COMBAT);
136 
137         execute();
138 
139         // Before rollback
140         assertTappedCount("Plains", true, 2);
141         assertPermanentCount(playerA, "Silvercoat Lion", 1);
142         assertPowerToughness(playerA, "Silvercoat Lion", 3, 3);
143         assertAbility(playerA, "Silvercoat Lion", FlyingAbility.getInstance(), true);
144         assertAbility(playerA, "Silvercoat Lion", FirstStrikeAbility.getInstance(), true);
145         assertHandCount(playerA, 0);
146         assertGraveyardCount(playerA, "Aerial Maneuver", 1);
147 
148         currentGame.rollbackTurns(0);
149 
150         // After rollback to begin turn 3
151         assertTappedCount("Plains", true, 0);
152         assertPermanentCount(playerA, "Silvercoat Lion", 1);
153         assertPowerToughness(playerA, "Silvercoat Lion", 2, 2);
154         assertAbility(playerA, "Silvercoat Lion", FlyingAbility.getInstance(), false);
155         assertAbility(playerA, "Silvercoat Lion", FirstStrikeAbility.getInstance(), false);
156         assertLibraryCount(playerA, "Aerial Maneuver", 1);
157         assertHandCount(playerA, 0);
158     }

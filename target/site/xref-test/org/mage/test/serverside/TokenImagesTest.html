13  public class TokenImagesTest extends CardTestPlayerBase {
14  
15      @Test
16      public void test_TokenMustGetSameSetCodeAsSourceCard() {
17          //{3}{W}, {T}, Sacrifice Memorial to Glory: Create two 1/1 white Soldier creature tokens.
18          addCard(Zone.BATTLEFIELD, playerA, "40K-Memorial to Glory");
19          addCard(Zone.BATTLEFIELD, playerA, "Plains", 4);
20  
21          activateAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{3}{W}, {T}, Sacrifice");
22  
23          setStrictChooseMode(true);
24          setStopAt(1, PhaseStep.BEGIN_COMBAT);
25          execute();
26  
27          assertPermanentCount(playerA, "Soldier Token", 2);
28          currentGame.getBattlefield().getAllPermanents().stream()
29                  .filter(card -> card.getName().equals("Soldier Token"))
30                  .forEach(card -> {
31                      Assert.assertEquals("40K", card.getExpansionSetCode());
32                      Assert.assertEquals("40K", ((PermanentToken) card).getToken().getOriginalExpansionSetCode());
33                  });
34      }

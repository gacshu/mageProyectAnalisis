18      private String prepareCommandFile(String data){
19          try {
20              File commandFile = File.createTempFile("test-commands", ".txt");
21              FileWriter w = new FileWriter(commandFile);
22              w.write(data);
23              w.close();
24              return commandFile.getAbsolutePath();
25          }catch (IOException e) {
26              Assert.fail("Can't create commands file: " + e.getMessage());
27              return null;
28          }
29      }
30  
31      @Test
32      public void testCommands(){
33          addCard(Zone.HAND, playerA, "Razorclaw Bear", 1);
34          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 3);
35  
36          String commandsFile = prepareCommandFile(new StringBuilder()
37                  .append("[group 1]").append('\n') // 1 group
38                  .append("battlefield:Human:Forest:10").append('\n')
39                  .append("//battlefield:Human:Forest:10").append('\n')
40                  .append("battlefield:Human:Island:10").append('\n')
41                  .append("").append('\n')
42                  .append("[@mana add]").append('\n') // 2 group
43                  .append("[group 2]").append('\n') // 3 group
44                  .append("//").append('\n')
45                  .append("hand:Human:Island:10").append('\n') // need that from 3 group
46                  .append("[group 3]").append('\n')
47                  .append("//").append('\n')
48                  .append("hand:Human:Island:5").append('\n')
49                  .append("[group 4]").append('\n') // 4 group
50                  .append("hand:Human:Island:3").append('\n')
51                  .toString()
52                  .replace(":Human:", ":" + playerA.getName() + ":")
53          );
54  
55          setStopAt(1, PhaseStep.BEGIN_COMBAT);
56          execute();
57  
58          setChoice(playerA, "3");
59          SystemUtil.addCardsForTesting(currentGame, commandsFile, playerA);
60  
61          assertHandCount(playerA, "Razorclaw Bear", 1);
62          assertPermanentCount(playerA, "Mountain", 3);
63          assertHandCount(playerA, "Island", 10); // by cheats
64      }

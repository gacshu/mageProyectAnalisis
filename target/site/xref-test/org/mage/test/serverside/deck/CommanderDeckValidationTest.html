14      @Test
15      public void testGristCommander() {
16          // Grist, the Hunger Tide can be your commander as its first ability applies during deck construction.
17          DeckTester deckTester = new DeckTester(new Commander());
18          deckTester.addMaindeck("Forest", 49);
19          deckTester.addMaindeck("Swamp", 50);
20  
21          deckTester.addSideboard("Grist, the Hunger Tide", 1);

23          deckTester.validate("Grist should be legal as a commander");
24      }

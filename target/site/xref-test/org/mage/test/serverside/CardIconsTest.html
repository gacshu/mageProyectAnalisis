20      @Test
21      public void test_CostX_Spells() {
22          // Chalice of the Void enters the battlefield with X charge counters on it.
23          // Whenever a player casts a spell with converted mana cost equal to the number of charge counters on Chalice of the Void, counter that spell.
24          addCard(Zone.HAND, playerA, "Chalice of the Void", 1); // {X}{X}
25          addCard(Zone.BATTLEFIELD, playerA, "Forest", 4);
26  
27          // hand (not visible)
28          runCode("card icons in hand", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
29              GameView gameView = getGameView(player);
30              Assert.assertEquals("must have 1 card in hand", 1, gameView.getHand().values().size());
31              CardView cardView = gameView.getHand().values().stream().findFirst().get();
32              Assert.assertEquals("must have non x cost card icons in hand", 0, cardView.getCardIcons().size());
33          });
34  
35          // cast and put on stack
36          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void");
37          setChoice(playerA, "X=2");
38  
39          // stack (visible)
40          runCode("card icons on stack (spell)", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
41              GameView gameView = getGameView(player);
42              Assert.assertEquals("must have 1 card in stack", 1, gameView.getStack().values().size());
43              CardView cardView = gameView.getStack().values().stream().findFirst().get();
44              Assert.assertEquals("must have x cost card icons in stack", 1, cardView.getCardIcons().size());
45              Assert.assertEquals("x cost text", "x=2", cardView.getCardIcons().get(0).getText());
46          });
47  
48          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
49          checkPermanentCount("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Chalice of the Void", 1);
50  
51          // battlefield (card, not visible)
52          runCode("card icons in battlefield (card)", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
53              GameView gameView = getGameView(player);
54              PlayerView playerView = gameView.getPlayers().get(0);
55              Assert.assertEquals("player", player.getName(), playerView.getName());
56              CardView cardView = playerView.getBattlefield().values().stream().filter(p -> p.getName().equals("Chalice of the Void")).findFirst().orElse(null);
57              Assert.assertNotNull("must have 1 chalice in battlefield", cardView);
58              Assert.assertEquals("must have x cost card icons in battlefield (card)", 1, cardView.getCardIcons().size());
59              Assert.assertEquals("x cost text", "x=2", cardView.getCardIcons().get(0).getText());
60          });
61  
62          setStrictChooseMode(true);
63          setStopAt(1, PhaseStep.END_TURN);
64          execute();
65      }

67      @Test
68      public void test_CostX_Copies() {
69          // Grenzo, Dungeon Warden enters the battlefield with X +1/+1 counters on it.
70          addCard(Zone.HAND, playerA, "Grenzo, Dungeon Warden", 1);// {X}{B}{R}
71          addCard(Zone.BATTLEFIELD, playerA, "Forest", 2);
72          addCard(Zone.BATTLEFIELD, playerA, "Swamp", 1);
73          addCard(Zone.BATTLEFIELD, playerA, "Mountain", 1);
74          //
75          // Copy target creature spell you control, except it isn't legendary if the spell is legendary.
76          addCard(Zone.HAND, playerA, "Double Major", 1); // {G}{U}
77          addCard(Zone.BATTLEFIELD, playerA, "Forest", 1);
78          addCard(Zone.BATTLEFIELD, playerA, "Island", 1);
79  
80          // cast and put on stack
81          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 2);
82          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {B}", 1);
83          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {R}", 1);
84          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grenzo, Dungeon Warden");
85          setChoice(playerA, "X=2");
86  
87          // prepare copy of spell
88          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {G}", 1);
89          activateManaAbility(1, PhaseStep.PRECOMBAT_MAIN, playerA, "{T}: Add {U}", 1);
90          castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Double Major", "Grenzo, Dungeon Warden", "Grenzo, Dungeon Warden");
91          checkStackSize("before copy spell", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
92          waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN, playerA, true);
93          checkStackSize("after copy spell", 1, PhaseStep.PRECOMBAT_MAIN, playerA, 2);
94  
95          // stack (copied spell)
96          runCode("card icons on stack (copied spell)", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
97              GameView gameView = getGameView(player);
98              Assert.assertEquals("must have 2 cards in stack", 2, gameView.getStack().values().size());
99              CardView originalCardView = gameView.getStack().values()
100                     .stream()
101                     .filter(c -> !c.getOriginalCard().isCopy())
102                     .findFirst()
103                     .get();
104             CardView copiedCardView = gameView.getStack().values()
105                     .stream()
106                     .filter(c -> c.getOriginalCard().isCopy())
107                     .findFirst()
108                     .get();
109             Assert.assertNotNull("stack must have original spell", originalCardView);
110             Assert.assertNotNull("stack must have copied spell", copiedCardView);
111             Assert.assertNotEquals("must find two spells on stack", originalCardView.getId(), copiedCardView.getId());
112             Assert.assertEquals("original spell must have x cost card icons", 1, originalCardView.getCardIcons().size());
113             Assert.assertEquals("copied spell must have x cost card icons", 1, copiedCardView.getCardIcons().size());
114             Assert.assertEquals("original x cost text", "x=2", originalCardView.getCardIcons().get(0).getText());
115             Assert.assertEquals("copied x cost text", "x=2", copiedCardView.getCardIcons().get(0).getText());
116         });
117 
118         // must resolve copied creature spell as a token
119         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
120         checkPermanentCount("after cast", 1, PhaseStep.PRECOMBAT_MAIN, playerA, "Grenzo, Dungeon Warden", 2);
121 
122         // battlefield (card and copied card as token)
123         runCode("card icons in battlefield (copied)", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
124             GameView gameView = getGameView(player);
125             PlayerView playerView = gameView.getPlayers().get(0);
126             Assert.assertEquals("player", player.getName(), playerView.getName());
127             // copied spell goes as token to battlefield, not copied card - so must check isToken
128             // original
129             CardView originalCardView = playerView.getBattlefield().values()
130                     .stream()
131                     .filter(p -> p.getName().equals("Grenzo, Dungeon Warden"))
132                     .filter(p -> !p.isToken())
133                     .findFirst()
134                     .orElse(null);
135             Assert.assertNotNull("original card must be in battlefield", originalCardView);
136             Assert.assertEquals("original must have x cost card icons", 1, originalCardView.getCardIcons().size());
137             Assert.assertEquals("original x cost text", "x=2", originalCardView.getCardIcons().get(0).getText());
138             //
139             CardView copiedCardView = playerView.getBattlefield().values()
140                     .stream()
141                     .filter(p -> p.getName().equals("Grenzo, Dungeon Warden"))
142                     .filter(p -> p.isToken())
143                     .findFirst()
144                     .orElse(null);
145             Assert.assertNotNull("copied card must be in battlefield", copiedCardView);
146             Assert.assertEquals("copied must have x cost card icons", 1, copiedCardView.getCardIcons().size());
147             Assert.assertEquals("copied x cost text", "x=0", copiedCardView.getCardIcons().get(0).getText());
148         });
149 
150         setStrictChooseMode(true);
151         setStopAt(1, PhaseStep.END_TURN);
152         execute();
153     }

223     @Test
224     public void test_CostX_MDFC() {
225         // Agadeem's Awakening
226         // Sorcery {X}{B}{B}{B}
227         // Return from your graveyard to the battlefield any number of target creature cards that each have a different converted mana cost X or less.
228         //
229         // Agadeem, the Undercrypt
230         // Land
231         // As Agadeem, the Undercrypt enters the battlefield, you may pay 3 life. If you don't, it enters the battlefield tapped.
232         addCard(Zone.HAND, playerA, "Agadeem's Awakening", 2);
233         addCard(Zone.BATTLEFIELD, playerA, "Swamp", 5);
234 
235         // hand (not visible)
236         runCode("card icons in hand", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
237             GameView gameView = getGameView(player);
238             CardView cardView = gameView.getHand().values().stream()
239                     .filter(c -> c.getName().equals("Agadeem's Awakening"))
240                     .findFirst()
241                     .orElse(null);
242             Assert.assertEquals("main must have non x cost card icons in hand", 0, cardView.getCardIcons().size());
243             Assert.assertEquals("right must have non x cost card icons in hand", 0, cardView.getSecondCardFace().getCardIcons().size());
244         });
245 
246         // play spell and check X
247         castSpell(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Agadeem's Awakening");
248         setChoice(playerA, "X=2");
249         addTarget(playerA, TestPlayer.TARGET_SKIP);
250         // stack (spell - visible)
251         runCode("card icons on stack (visible)", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
252             GameView gameView = getGameView(player);
253             Assert.assertEquals("must have 1 card in stack", 1, gameView.getStack().values().size());
254             CardView cardView = gameView.getStack().values().stream()
255                     .filter(c -> c.getName().equals("Agadeem's Awakening"))
256                     .findFirst()
257                     .orElse(null);
258             Assert.assertEquals("main must have x cost card icons in stack", 1, cardView.getCardIcons().size());
259             Assert.assertNull("right must be null in stack", cardView.getSecondCardFace());
260         });
261         waitStackResolved(1, PhaseStep.PRECOMBAT_MAIN);
262 
263         // play land and check X
264         playLand(1, PhaseStep.PRECOMBAT_MAIN, playerA, "Agadeem, the Undercrypt");
265         setChoice(playerA, false); // not pay life
266         runCode("card icons in battlefield", 1, PhaseStep.PRECOMBAT_MAIN, playerA, (info, player, game) -> {
267             GameView gameView = getGameView(player);
268             PlayerView playerView = gameView.getPlayers().get(0);
269             Assert.assertEquals("player", player.getName(), playerView.getName());
270             CardView cardView = playerView.getBattlefield().values().stream().filter(p -> p.getName().equals("Agadeem, the Undercrypt")).findFirst().orElse(null);
271             Assert.assertNotNull("must have Agadeem, the Undercrypt in battlefield", cardView);
272             Assert.assertEquals("main must have not x cost card icons in battlefield", 0, cardView.getCardIcons().size());
273             Assert.assertNull("second side must be null", cardView.getSecondCardFace());
274         });
275 
276         setStrictChooseMode(true);
277         setStopAt(1, PhaseStep.END_TURN);
278         execute();
279     }

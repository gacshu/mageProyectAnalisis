14      @Test
15      public void writeDeck() throws IOException {
16          ByteArrayOutputStream baos = new ByteArrayOutputStream();
17          DeckCardLists deck = new DeckCardLists();
18          deck.getCards().add(new DeckCardInfo("Forest", "1", "RNA", 2));
19          deck.getCards().add(new DeckCardInfo("Plains", "2", "RNA", 3));
20          deck.getCards().add(new DeckCardInfo("Plains", "2", "RNA", 5)); // must combine
21          deck.getCards().add(new DeckCardInfo("Mountain", "3", "RNA", 1));
22          deck.getSideboard().add(new DeckCardInfo("Island", "1", "RNA", 2));
23          deck.getSideboard().add(new DeckCardInfo("Island", "1", "RNA", 5)); // must combine
24          deck.getSideboard().add(new DeckCardInfo("Mountain", "2", "RNA", 3));
25          DeckExporter exporter = new XmageDeckExporter();
26          exporter.writeDeck(baos, deck);
27          assertEquals("2 [RNA:1] Forest" + System.lineSeparator() +
28                          "8 [RNA:2] Plains" + System.lineSeparator() +
29                          "1 [RNA:3] Mountain" + System.lineSeparator() +
30                          "SB: 7 [RNA:1] Island" + System.lineSeparator() +
31                          "SB: 3 [RNA:2] Mountain" + System.lineSeparator(),
32                  baos.toString());
33      }

2   package mage.watchers.common;
3   
4   import java.util.HashSet;
5   import java.util.Set;
6   import java.util.UUID;
7   import mage.constants.WatcherScope;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.game.events.GameEvent.EventType;
11  import mage.util.CardUtil;
12  import mage.watchers.Watcher;
13  
14  /**
15   * Watcher stores whitch sources did damage to a player
16   * 
17   * @author LevelX
18   */
19  public class PlayerDamagedBySourceWatcher extends Watcher {
20  
21      private final Set<String> damageSourceIds = new HashSet<>();
22  
23      public PlayerDamagedBySourceWatcher() {
24          super(WatcherScope.PLAYER);
25      }
26  
27      @Override
28      public void watch(GameEvent event, Game game) {
29          if (event.getType() == GameEvent.EventType.DAMAGED_PLAYER) {
30              if (event.getTargetId().equals(controllerId)) {
31                  damageSourceIds.add(CardUtil.getCardZoneString(null, event.getSourceId(), game));
32              }
33          }
34      }
35  
36      /**
37       * Checks if the current object with sourceId has damaged the player during the current turn.
38       * The zoneChangeCounter will be taken into account.
39       * 
40       * @param sourceId
41       * @param game
42       * @return 
43       */
44      public boolean hasSourceDoneDamage(UUID sourceId, Game game) {
45          return damageSourceIds.contains(CardUtil.getCardZoneString(null, sourceId, game));
46      }
47  
48      @Override
49      public void reset() {
50          super.reset();
51          damageSourceIds.clear();
52      }
53  }

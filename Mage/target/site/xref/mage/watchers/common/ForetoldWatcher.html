1   package mage.watchers.common;

3   import java.util.HashSet;
4   import java.util.Set;
5   import java.util.UUID;
6   import mage.cards.Card;
7   import mage.constants.WatcherScope;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.util.CardUtil;
11  import mage.watchers.Watcher;
12  
13  /**
14   *
15   * @author jeffwadsworth
16   */
17  public class ForetoldWatcher extends Watcher {
18  
19      // If foretell was activated or a card was Foretold by the controller this turn, this list stores it.  Cleared at the end of the turn.
20      private final Set<UUID> foretellCardsThisTurn = new HashSet<>();
21      private final Set<UUID> foretoldCards = new HashSet<>();
22  
23      public ForetoldWatcher() {
24          super(WatcherScope.GAME);
25      }
26  
27      @Override
28      public void watch(GameEvent event, Game game) {
29          if (event.getType() == GameEvent.EventType.FORETELL) {
30              Card card = game.getCard(event.getTargetId());
31              if (card != null
32                      && controllerId == event.getPlayerId()) {
33                  foretellCardsThisTurn.add(card.getId());
34                  foretoldCards.add(card.getId());
35              }
36          }
37          // Ethereal Valkyrie
38          if (event.getType() == GameEvent.EventType.FORETOLD) {
39              Card card = game.getCard(event.getTargetId());
40              if (card != null) {
41                  // Ethereal Valkyrie does not Foretell the card, it becomes Foretold, so don't add it to the Foretell list
42                  foretoldCards.add(card.getId());
43              }
44          }
45      }
46  
47      public boolean cardWasForetold(UUID sourceId) {
48          return foretoldCards.contains(sourceId);
49      }
50  
51      public int countNumberForetellThisTurn() {
52          return foretellCardsThisTurn.size();
53      }
54  
55      @Override
56      public void reset() {
57          super.reset();
58          foretellCardsThisTurn.clear();
59      }
60  }

1   package mage.watchers.common;
2   
3   import mage.MageObject;
4   import mage.constants.SubType;
5   import mage.constants.WatcherScope;
6   import mage.constants.Zone;
7   import mage.game.Game;
8   import mage.game.events.GameEvent;
9   import mage.game.events.ZoneChangeEvent;
10  import mage.watchers.Watcher;
11  
12  /**
13   * Created by Eric on 9/24/2016.
14   */
15  public class ZuberasDiedWatcher extends Watcher {
16  
17      public int getZuberasDiedThisTurn() {
18          return zuberasDiedThisTurn;
19      }
20  
21      private int zuberasDiedThisTurn = 0;
22  
23      public ZuberasDiedWatcher() {
24          super(WatcherScope.GAME);
25      }
26  
27      @Override
28      public void watch(GameEvent event, Game game) {
29          if (event.getType() == GameEvent.EventType.ZONE_CHANGE && ((ZoneChangeEvent) event).isDiesEvent()) {
30              MageObject card = game.getLastKnownInformation(event.getTargetId(), Zone.BATTLEFIELD);
31              if (card != null && card.hasSubtype(SubType.ZUBERA, game)) {
32                  zuberasDiedThisTurn++;
33              }
34          }
35      }
36  
37      @Override
38      public void reset() {
39          super.reset();
40          zuberasDiedThisTurn = 0;
41      }
42  
43  }

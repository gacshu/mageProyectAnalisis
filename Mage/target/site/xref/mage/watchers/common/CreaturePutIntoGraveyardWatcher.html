1   package mage.watchers.common;

3   import mage.cards.Card;
4   import mage.constants.WatcherScope;
5   import mage.constants.Zone;
6   import mage.game.Game;
7   import mage.game.events.GameEvent;
8   import mage.game.events.ZoneChangeEvent;
9   import mage.watchers.Watcher;
10  
11  import java.util.HashSet;
12  import java.util.Set;
13  import java.util.UUID;
14  
15  /**
16   * @author TheElk801
17   */
18  public class CreaturePutIntoGraveyardWatcher extends Watcher {
19  
20      private final Set<UUID> players = new HashSet<>();
21  
22      public CreaturePutIntoGraveyardWatcher() {
23          super(WatcherScope.GAME);
24      }
25  
26      @Override
27      public void watch(GameEvent event, Game game) {
28          if (event.getType() != GameEvent.EventType.ZONE_CHANGE
29                  || !Zone.GRAVEYARD.match(((ZoneChangeEvent) event).getToZone())) {
30              return;
31          }
32          Card card = game.getCard(event.getTargetId());
33          if (card != null && card.isCreature(game)) {
34              players.add(card.getOwnerId());
35          }
36      }
37  
38      @Override
39      public void reset() {
40          super.reset();
41          players.clear();
42      }
43  
44      public static boolean checkPlayer(UUID playerId, Game game) {
45          return game
46                  .getState()
47                  .getWatcher(CreaturePutIntoGraveyardWatcher.class)
48                  .players
49                  .contains(playerId);
50      }
51  }

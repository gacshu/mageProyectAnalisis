6   import mage.watchers.Watcher;
7   
8   import java.util.HashMap;
9   import java.util.Map;
10  import java.util.UUID;
11  
12  /**
13   * @author TheElk801
14   */
15  
16  public class CardsDrawnThisTurnWatcher extends Watcher {
17  
18      private final Map<UUID, Integer> cardsDrawnThisTurn = new HashMap<>();
19  
20      /**
21       * It's a default game watcher, so no need to init it with card code
22       */
23      public CardsDrawnThisTurnWatcher() {
24          super(WatcherScope.GAME);
25      }
26  
27      @Override
28      public void watch(GameEvent event, Game game) {
29          if (event.getType() == GameEvent.EventType.DREW_CARD) {
30              int cardsDrawn = getCardsDrawnThisTurn(event.getPlayerId());
31              cardsDrawnThisTurn.put(event.getPlayerId(), cardsDrawn + 1);
32          }
33      }
34  
35      public int getCardsDrawnThisTurn(UUID playerId) {
36          return cardsDrawnThisTurn.getOrDefault(playerId, 0);
37      }
38  
39      @Override
40      public void reset() {
41          super.reset();
42          cardsDrawnThisTurn.clear();
43  
44      }
45  }

3   package mage.watchers.common;
4   
5   import java.util.HashMap;
6   import java.util.Map;
7   import mage.MageObjectReference;
8   import mage.constants.WatcherScope;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  import mage.game.permanent.Permanent;
12  import mage.watchers.Watcher;
13  
14  /**
15   *
16   * @author LevelX2
17   */
18  public class NumberOfTimesPermanentTargetedATurnWatcher extends Watcher {
19  
20      private final Map<MageObjectReference, Integer> permanentsTargeted = new HashMap<>();
21  
22      public NumberOfTimesPermanentTargetedATurnWatcher() {
23          super(WatcherScope.GAME);
24      }
25  
26      @Override
27      public void watch(GameEvent event, Game game) {
28          if (event.getType() == GameEvent.EventType.TARGETED) {
29              Permanent permanent = game.getPermanent(event.getTargetId());
30              if (permanent != null) {
31                  MageObjectReference mor = new MageObjectReference(permanent, game);
32                  int amount = 0;
33                  if (permanentsTargeted.containsKey(mor)) {
34                      amount = permanentsTargeted.get(mor);
35                  }
36                  permanentsTargeted.put(mor, ++amount);
37              }
38          }
39      }
40  
41      public boolean notMoreThanOnceTargetedThisTurn(Permanent creature, Game game) {
42          if (permanentsTargeted.containsKey(new MageObjectReference(creature, game))) {
43              return permanentsTargeted.get(new MageObjectReference(creature, game)) < 2;
44          }
45          return true;
46      }
47  
48      @Override
49      public void reset() {
50          super.reset();
51          permanentsTargeted.clear();
52      }
53  }

1   package mage.watchers.common;
2   
3   import mage.constants.WatcherScope;
4   import mage.constants.Zone;
5   import mage.game.Game;
6   import mage.game.events.GameEvent;
7   import mage.game.events.ZoneChangeEvent;
8   import mage.watchers.Watcher;
9   
10  import java.util.HashSet;
11  import java.util.Set;
12  import java.util.UUID;
13  
14  /**
15   * @author TheElk801
16   */
17  public class CreatureEnteredControllerWatcher extends Watcher {
18  
19      private final Set<UUID> players = new HashSet<>();
20  
21      public CreatureEnteredControllerWatcher() {
22          super(WatcherScope.GAME);
23      }
24  
25      @Override
26      public void watch(GameEvent event, Game game) {
27          if (event.getType() == GameEvent.EventType.ZONE_CHANGE) {
28              ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
29              if (zEvent.getToZone() == Zone.BATTLEFIELD
30                      && zEvent.getTarget().isCreature(game)) {
31                  players.add(zEvent.getTarget().getControllerId());
32              }
33          }
34      }
35  
36      @Override
37      public void reset() {
38          players.clear();
39      }
40  
41      public static boolean enteredCreatureForPlayer(UUID playerId, Game game) {
42          return game
43                  .getState()
44                  .getWatcher(CreatureEnteredControllerWatcher.class)
45                  .players
46                  .contains(playerId);
47      }
48  }

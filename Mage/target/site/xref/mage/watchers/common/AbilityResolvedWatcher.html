1   package mage.watchers.common;

3   import mage.abilities.Ability;
4   import mage.constants.WatcherScope;
5   import mage.game.Game;
6   import mage.game.events.GameEvent;
7   import mage.watchers.Watcher;
8   
9   import java.util.HashMap;
10  import java.util.Map;
11  
12  /**
13   * @author TheElk801
14   */
15  public class AbilityResolvedWatcher extends Watcher {
16  
17      private final Map<String, Integer> resolutionMap = new HashMap<>();
18  
19      public AbilityResolvedWatcher() {
20          super(WatcherScope.GAME);
21      }
22  
23      @Override
24      public void watch(GameEvent event, Game game) {
25          if (event.getType() == GameEvent.EventType.RESOLVING_ABILITY) {
26              resolutionMap.merge(event.getTargetId().toString() + game.getState().getZoneChangeCounter(event.getSourceId()), 1, Integer::sum);
27          }
28      }
29  
30      @Override
31      public void reset() {
32          super.reset();
33          resolutionMap.clear();
34      }
35  
36      public static int getResolutionCount(Game game, Ability source) {
37          return game
38                  .getState()
39                  .getWatcher(AbilityResolvedWatcher.class)
40                  .resolutionMap
41                  .getOrDefault(source.getOriginalId().toString() + game.getState().getZoneChangeCounter(source.getSourceId()), 0);
42      }
43  }

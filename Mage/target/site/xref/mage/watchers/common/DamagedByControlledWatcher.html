1   package mage.watchers.common;
2   
3   import mage.MageObjectReference;
4   import mage.constants.WatcherScope;
5   import mage.game.Game;
6   import mage.game.events.GameEvent;
7   import mage.game.permanent.Permanent;
8   import mage.watchers.Watcher;
9   
10  import java.util.HashSet;
11  
12  /**
13   *
14   * @author weirddan455
15   */
16  public class DamagedByControlledWatcher extends Watcher {
17  
18      private final HashSet<MageObjectReference> damagedPermanents = new HashSet<>();
19  
20      public DamagedByControlledWatcher() {
21          super(WatcherScope.PLAYER);
22      }
23  
24      @Override
25      public void watch(GameEvent event, Game game) {
26          if (event.getType() == GameEvent.EventType.DAMAGED_PERMANENT) {
27              Permanent permanent = game.getPermanent(event.getTargetId());
28              if (permanent != null && permanent.isCreature(game)) {
29                  if (controllerId != null && controllerId.equals(game.getControllerId(event.getSourceId()))) {
30                      damagedPermanents.add(new MageObjectReference(event.getTargetId(), game));
31                  }
32              }
33          }
34      }
35  
36      @Override
37      public void reset() {
38          super.reset();
39          damagedPermanents.clear();
40      }
41  
42      public boolean wasDamaged(Permanent permanent, Game game) {
43          return damagedPermanents.contains(new MageObjectReference(permanent, game));
44      }
45  }

1   package mage.watchers.common;

3   import mage.MageObject;
4   import mage.MageObjectReference;
5   import mage.constants.WatcherScope;
6   import mage.game.Game;
7   import mage.game.events.GameEvent;
8   import mage.watchers.Watcher;
9   
10  import java.util.HashSet;
11  import java.util.Set;
12  
13  /**
14   * Watcher stores which sources did damage to anything.
15   *
16   * @author jeffwadsworth
17   */
18  public class SourceDidDamageWatcher extends Watcher {
19  
20      private final Set<MageObjectReference> damageSources = new HashSet<>();
21  
22      public SourceDidDamageWatcher() {
23          super(WatcherScope.GAME);
24      }
25  
26      @Override
27      public void watch(GameEvent event, Game game) {
28          if (event.getType() == GameEvent.EventType.DAMAGED_PERMANENT
29                  || event.getType() == GameEvent.EventType.DAMAGED_PLAYER) {
30              damageSources.add(new MageObjectReference(event.getSourceId(), game));
31          }
32      }
33  
34      public boolean checkSource(MageObject mageObject, Game game) {
35          return damageSources.stream().anyMatch(mor -> mor.refersTo(mageObject, game));
36      }
37  
38      @Override
39      public void reset() {
40          super.reset();
41          damageSources.clear();
42      }
43  }

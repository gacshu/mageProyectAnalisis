3   import mage.constants.WatcherScope;
4   import mage.game.Game;
5   import mage.game.events.GameEvent;
6   import mage.util.CardUtil;
7   import mage.watchers.Watcher;
8   
9   import java.util.HashMap;
10  import java.util.Map;
11  import java.util.UUID;
12  
13  /**
14   * @author TheElk801
15   */
16  public class EndStepCountWatcher extends Watcher {
17  
18      private final Map<UUID, Integer> playerMap = new HashMap<>();
19  
20      public EndStepCountWatcher() {
21          super(WatcherScope.GAME);
22      }
23  
24      @Override
25      public void watch(GameEvent event, Game game) {
26          if (event.getType() == GameEvent.EventType.END_TURN_STEP_PRE) {
27              playerMap.compute(game.getActivePlayerId(), CardUtil::setOrIncrementValue);
28          }
29      }
30  
31      public static int getCount(UUID playerId, Game game) {
32          return game
33                  .getState()
34                  .getWatcher(EndStepCountWatcher.class)
35                  .playerMap
36                  .getOrDefault(playerId, 0);
37      }
38  }

4   import java.util.ArrayDeque;
5   import java.util.Date;
6   import java.util.UUID;
7   import mage.MageObject;
8   import mage.abilities.Ability;
9   import mage.constants.PutCards;
10  import mage.game.Game;
11  import mage.game.events.GameEvent;
12  import org.apache.log4j.Logger;

14  /**
15   *
16   * @author BetaSteward_at_googlemail.com
17   */
18  public class SpellStack extends ArrayDeque<StackObject> {
19  
20      private static final Logger logger = Logger.getLogger(SpellStack.class);
21  
22      protected Date dateLastAdded;
23  
24      public SpellStack() {
25      }
26  
27      public SpellStack(final SpellStack stack) {
28  
29          for (StackObject spell : stack) {
30              this.addLast(spell.copy());
31          }
32      }
33  
34      //resolve top StackObject
35      public void resolve(Game game) {
36          StackObject top = null;
37          try {
38              top = this.peek();
39              top.resolve(game);
40              game.resetControlAfterSpellResolve(top.getId());
41          } finally {
42              if (top != null) {
43                  if (contains(top)) {
44                      logger.warn("StackObject was still on the stack after resolving" + top.getName());
45                      this.remove(top, game);
46                  }
47              }
48          }
49      }
50  
51      public boolean remove(StackObject object, Game game) {
52          for (StackObject spell : this) {
53              if (spell.getId().equals(object.getId())) {
54                  game.getState().setZone(spell.getId(), null);
55                  return super.remove(spell);
56              }
57          }
58          return false;
59      }
60  
61      public boolean counter(UUID objectId, Ability source, Game game) {
62          return counter(objectId, source, game, PutCards.GRAVEYARD);
63      }
64  
65      public boolean counter(UUID objectId, Ability source, Game game, PutCards putCard) {
66          StackObject stackObject = getStackObject(objectId);
67          MageObject sourceObject = game.getObject(source);
68          if (stackObject != null && sourceObject != null) {
69              MageObject targetSourceObject = game.getObject(stackObject.getSourceId());
70              String counteredObjectName, targetSourceName;
71              if (targetSourceObject == null) {
72                  targetSourceName = "[Object not found]";
73              } else {
74                  targetSourceName = game.getObject(stackObject.getSourceId()).getLogName();
75              }
76              if (stackObject instanceof Spell) {
77                  counteredObjectName = targetSourceName;
78              } else {
79                  counteredObjectName = "Ability (" + stackObject.getStackAbility().getRule(targetSourceName) + ") of " + targetSourceName;
80              }
81              if (!game.replaceEvent(GameEvent.getEvent(GameEvent.EventType.COUNTER, objectId, source, stackObject.getControllerId()))) {
82                  if (!(stackObject instanceof Spell)) { // spells are removed from stack by the card movement
83                      this.remove(stackObject, game);
84                  }
85                  stackObject.counter(source, game, putCard);
86                  if (!game.isSimulation()) {
87                      game.informPlayers(counteredObjectName + " is countered by " + sourceObject.getLogName());
88                  }
89                  game.fireEvent(GameEvent.getEvent(GameEvent.EventType.COUNTERED, objectId, source, stackObject.getControllerId()));
90                  return true;
91              } else if (!game.isSimulation()) {
92                  game.informPlayers(counteredObjectName + " could not be countered by " + sourceObject.getLogName());
93              }
94          }
95          return false;
96      }
97  
98      public StackObject getStackObject(UUID id) {
99          for (StackObject stackObject : this) {
100             UUID objectId = stackObject.getId();
101             if (objectId.equals(id)) {
102                 return stackObject;
103             }
104             UUID sourceId = stackObject.getSourceId();
105             if (sourceId.equals(id)) {
106                 return stackObject;
107             }
108         }
109         return null;
110     }
111 
112     public Spell getSpell(UUID id) {
113         return getSpell(id, true);
114     }
115 
116     public Spell getSpell(UUID id, boolean allowCopies) {
117         for (StackObject stackObject : this) {
118             if (stackObject instanceof Spell) {
119                 if (stackObject.getId().equals(id) || stackObject.getSourceId().equals(id)) {
120                     if (allowCopies || !stackObject.isCopy()) {
121                         return (Spell) stackObject;
122                     }
123                 }
124             }
125         }
126         return null;
127     }
128 
129     public SpellStack copy() {
130         return new SpellStack(this);
131     }
132 
133     @Override
134     public void push(StackObject e) {
135         super.push(e);
136         this.dateLastAdded = new Date();
137     }
138 
139     public Date getDateLastAdded() {
140         return dateLastAdded;
141     }
142 
143     @Override
144     public String toString() {
145         return this.size() + (this.isEmpty() ? "" : " (top: " + this.getFirst().toString() + ")");
146     }
147 }

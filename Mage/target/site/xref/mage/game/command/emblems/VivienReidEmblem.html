19  public final class VivienReidEmblem extends Emblem {
20      // -8: You get an emblem with "Creatures you control get +2/+2 and have vigilance, trample, and indestructible.
21  
22      public VivienReidEmblem() {
23          this.setName("Emblem Vivien");
24          this.setExpansionSetCodeForImage("M19");
25          Ability ability = new SimpleStaticAbility(
26                  Zone.COMMAND,
27                  new BoostControlledEffect(
28                          2, 2, Duration.EndOfGame,
29                          StaticFilters.FILTER_PERMANENT_CREATURES,
30                          false
31                  ).setText("creatures you control get +2/+2")
32          );
33          ability.addEffect(new GainAbilityControlledEffect(
34                  VigilanceAbility.getInstance(),
35                  Duration.EndOfGame,
36                  StaticFilters.FILTER_PERMANENT_CREATURES
37          ).setText("and have vigilance"));
38          ability.addEffect(new GainAbilityControlledEffect(
39                  TrampleAbility.getInstance(),
40                  Duration.EndOfGame,
41                  StaticFilters.FILTER_PERMANENT_CREATURES
42          ).setText(", trample"));
43          ability.addEffect(new GainAbilityControlledEffect(
44                  IndestructibleAbility.getInstance(),
45                  Duration.EndOfGame,
46                  StaticFilters.FILTER_PERMANENT_CREATURES
47          ).setText(", and indestructible"));
48          this.getAbilities().add(ability);
49      }
50  }

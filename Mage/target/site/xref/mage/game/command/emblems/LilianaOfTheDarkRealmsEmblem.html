4   import mage.Mana;
5   import mage.abilities.Ability;
6   import mage.abilities.common.SimpleStaticAbility;
7   import mage.abilities.costs.common.TapSourceCost;
8   import mage.abilities.effects.common.continuous.GainAbilityControlledEffect;
9   import mage.abilities.mana.SimpleManaAbility;
10  import mage.constants.Duration;
11  import mage.constants.SubType;
12  import mage.constants.Zone;
13  import mage.filter.common.FilterLandPermanent;
14  import mage.game.command.Emblem;
15  
16  import java.util.Arrays;
17  
18  /**
19   *
20   * @author spjspj
21   */
22  public final class LilianaOfTheDarkRealmsEmblem extends Emblem {
23  
24      private static final FilterLandPermanent filter = new FilterLandPermanent("Swamps");
25  
26      static {
27          filter.add(SubType.SWAMP.getPredicate());
28      }
29  
30      public LilianaOfTheDarkRealmsEmblem() {
31          this.setName("Emblem Liliana");
32          SimpleManaAbility manaAbility = new SimpleManaAbility(Zone.BATTLEFIELD, Mana.BlackMana(4), new TapSourceCost());
33          Ability ability = new SimpleStaticAbility(Zone.COMMAND, new GainAbilityControlledEffect(manaAbility, Duration.WhileOnBattlefield, filter));
34          this.getAbilities().add(ability);
35  
36          availableImageSetCodes = Arrays.asList("M13", "M14");
37      }
38  }

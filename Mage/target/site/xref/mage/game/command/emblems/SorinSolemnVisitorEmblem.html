14   * @author spjspj

16  public final class SorinSolemnVisitorEmblem extends Emblem {
17  
18      /**
19       * Emblem: "At the beginning of each opponent's upkeep, that player
20       * sacrifices a creature."
21       */
22      public SorinSolemnVisitorEmblem() {
23          this.setName("Emblem Sorin");
24          Ability ability = new BeginningOfUpkeepTriggeredAbility(Zone.COMMAND, new SacrificeEffect(StaticFilters.FILTER_PERMANENT_CREATURE, 1, "that player"), TargetController.OPPONENT, false, true);
25          this.getAbilities().add(ability);
26      }
27  }

35      public MomirEmblem() {
36          setName("Emblem Momir Vig, Simic Visionary");
37          setExpansionSetCodeForImage("DIS");
38  
39          // {X}, Discard a card: Create a token that's a copy of a creature card with converted mana cost X chosen at random.
40          // Activate this ability only any time you could cast a sorcery and only once each turn.
41          LimitedTimesPerTurnActivatedAbility ability = new LimitedTimesPerTurnActivatedAbility(Zone.COMMAND, new MomirEffect(), new VariableManaCost(VariableCostType.NORMAL));
42          ability.addCost(new DiscardCardCost());
43          ability.setTiming(TimingRule.SORCERY);
44          this.getAbilities().add(ability);
45      }

48  class MomirEffect extends OneShotEffect {
49  
50      public MomirEffect() {
51          super(Outcome.PutCreatureInPlay);
52      }
53  
54      public MomirEffect(MomirEffect effect) {
55          super(effect);
56          staticText = "Create a token that's a copy of a creature card with mana value X chosen at random";
57      }
58  
59      @Override
60      public MomirEffect copy() {
61          return new MomirEffect(this);
62      }
63  
64      @Override
65      public boolean apply(Game game, Ability source) {
66          int value = source.getManaCostsToPay().getX();
67          if (game.isSimulation()) {
68              // Create dummy token to prevent multiple DB find cards what causes H2 java.lang.IllegalStateException if AI cancels calculation because of time out
69              Token token = new CreatureToken(value, value + 1);
70              token.putOntoBattlefield(1, game, source, source.getControllerId(), false, false);
71              return true;
72          }
73          // should this be random across card names
74          CardCriteria criteria = new CardCriteria().types(CardType.CREATURE).manaValue(value);
75          List<CardInfo> options = CardRepository.instance.findCards(criteria);
76          if (options == null || options.isEmpty()) {
77              game.informPlayers("No random creature card with mana value of " + value + " was found.");
78              return false;
79          }
80  
81          // search for a random non custom set creature
82          EmptyToken token = null;
83          while (!options.isEmpty()) {
84              int index = RandomUtil.nextInt(options.size());
85              ExpansionSet expansionSet = Sets.findSet(options.get(index).getSetCode());
86              if (expansionSet == null || !expansionSet.getSetType().isEternalLegal()) {
87                  options.remove(index);
88              } else {
89                  Card card = options.get(index).getCard();
90                  if (card != null) {
91                      token = new EmptyToken();
92                      CardUtil.copyTo(token).from(card, game);
93                      break;
94                  } else {
95                      options.remove(index);
96                  }
97              }
98          }
99          if (token != null) {
100             token.putOntoBattlefield(1, game, source, source.getControllerId(), false, false);
101             return true;
102         }
103 
104         return false;
105     }
106 }

19  public final class WrennAndRealmbreakerEmblem extends Emblem {
20  
21      // -7: You get an emblem with "You may play lands and cast permanent spells from your graveyard."
22      public WrennAndRealmbreakerEmblem() {
23          this.setName("Emblem Wrenn");
24          this.setExpansionSetCodeForImage("MOM");
25          this.getAbilities().add(new SimpleStaticAbility(Zone.COMMAND, new WrennAndRealmbreakerEmblemEffect()));
26      }
27  }
28  
29  class WrennAndRealmbreakerEmblemEffect extends AsThoughEffectImpl {
30  
31      public WrennAndRealmbreakerEmblemEffect() {
32          super(AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, Duration.WhileOnBattlefield, Outcome.Benefit);
33          staticText = "you may play lands and cast permanent spells from your graveyard";
34      }
35  
36      public WrennAndRealmbreakerEmblemEffect(final WrennAndRealmbreakerEmblemEffect effect) {
37          super(effect);
38      }
39  
40      @Override
41      public boolean apply(Game game, Ability source) {
42          return true;
43      }
44  
45      @Override
46      public WrennAndRealmbreakerEmblemEffect copy() {
47          return new WrennAndRealmbreakerEmblemEffect(this);
48      }
49  
50      @Override
51      public boolean applies(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
52          if (!source.isControlledBy(affectedControllerId)) {
53              return false;
54          }
55          Card card = game.getCard(objectId);
56          return card != null
57                  && card.isPermanent(game)
58                  && card.isOwnedBy(source.getControllerId())
59                  && game.getState().getZone(objectId) == Zone.GRAVEYARD;
60      }
61  }

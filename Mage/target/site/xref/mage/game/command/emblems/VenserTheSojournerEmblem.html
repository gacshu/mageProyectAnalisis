20   *
21   * @author spjspj
22   */
23  public final class VenserTheSojournerEmblem extends Emblem {
24  
25      /**
26       * Emblem: "Whenever you cast a spell, exile target permanent."
27       */
28  
29      public VenserTheSojournerEmblem() {
30          this.setName("Emblem Venser");
31          Ability ability = new VenserTheSojournerSpellCastTriggeredAbility(new ExileTargetEffect(), false);
32          Target target = new TargetPermanent();
33          ability.addTarget(target);
34          this.getAbilities().add(ability);
35  
36          this.setExpansionSetCodeForImage("DDI");
37      }
38  }
39  
40  class VenserTheSojournerSpellCastTriggeredAbility extends TriggeredAbilityImpl {
41  
42  
43      /**
44       * If true, the source that triggered the ability will be set as target to
45       * effect.
46       */
47  
48      public VenserTheSojournerSpellCastTriggeredAbility(Effect effect, boolean optional) {
49          super(Zone.COMMAND, effect, optional);
50      }
51  
52      public VenserTheSojournerSpellCastTriggeredAbility(final VenserTheSojournerSpellCastTriggeredAbility ability) {
53          super(ability);
54      }
55  
56      @Override
57      public boolean checkEventType(GameEvent event, Game game) {
58          return event.getType() == GameEvent.EventType.SPELL_CAST;
59      }
60  
61      @Override
62      public boolean checkTrigger(GameEvent event, Game game) {
63          if (event.getPlayerId().equals(this.getControllerId())) {
64              Spell spell = game.getStack().getSpell(event.getTargetId());
65              return StaticFilters.FILTER_SPELL_A.match(spell, game);
66          }
67          return false;
68      }
69  
70      @Override
71      public String getRule() {
72          return "Whenever you cast a spell, exile target permanent.";
73      }
74  
75      @Override
76      public VenserTheSojournerSpellCastTriggeredAbility copy() {
77          return new VenserTheSojournerSpellCastTriggeredAbility(this);
78      }
79  }

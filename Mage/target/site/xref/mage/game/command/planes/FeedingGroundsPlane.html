37  public class FeedingGroundsPlane extends Plane {
38  
39      private static final String rule = "put X +1/+1 counters on target creature, where X is that creature's mana value";
40  
41      public FeedingGroundsPlane() {
42          this.setPlaneType(Planes.PLANE_FEEDING_GROUNDS);
43          this.setExpansionSetCodeForImage("PCA");
44  
45          // Red spells cost {1} less to cast.  Green spells cost {1} less to cast
46          Ability ability = new SimpleStaticAbility(Zone.COMMAND, new FeedingGroundsEffect());
47          this.getAbilities().add(ability);
48  
49          // Active player can roll the planar die: Whenever you roll {CHAOS}, target red or green creature gets X +1/+1 counters
50          Effect chaosEffect = new AddCountersTargetEffect(CounterType.P1P1.createInstance(), TargetManaValue.instance);
51          Target chaosTarget = new TargetCreaturePermanent(1, 1, StaticFilters.FILTER_PERMANENT_A_CREATURE, false);
52  
53          List<Effect> chaosEffects = new ArrayList<>();
54          chaosEffects.add(chaosEffect);
55          List<Target> chaosTargets = new ArrayList<>();
56          chaosTargets.add(chaosTarget);
57  
58          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
59          chaosAbility.addWatcher(new PlanarRollWatcher());
60          this.getAbilities().add(chaosAbility);
61          chaosAbility.setMayActivate(TargetController.ANY);
62          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
63      }
64  }

66  class FeedingGroundsEffect extends CostModificationEffectImpl {
67  
68      private static final FilterCard filter = new FilterCard("Red spells or Green spells");
69  
70      static {
71          filter.add(Predicates.or(
72                  new ColorPredicate(ObjectColor.RED),
73                  new ColorPredicate(ObjectColor.GREEN)));
74      }
75  
76      private static final String rule = "Red spells cost {1} less to cast. Green spells cost {1} less to cast";
77      private int amount = 1;
78  
79      public FeedingGroundsEffect() {
80          super(Duration.Custom, Outcome.Benefit, CostModificationType.REDUCE_COST);
81          this.amount = 1;
82          this.staticText = rule;
83      }
84  
85      protected FeedingGroundsEffect(FeedingGroundsEffect effect) {
86          super(effect);
87          this.amount = effect.amount;
88      }
89  
90      @Override
91      public void init(Ability source, Game game) {
92          super.init(source, game);
93      }
94  
95      @Override
96      public boolean apply(Game game, Ability source, Ability abilityToModify) {
97          MageObject object = abilityToModify.getSourceObject(game);
98          int reduce = 0;
99          if (object != null) {
100             if (object.getColor(game).isRed()) {
101                 reduce++;
102             }
103             if (object.getColor(game).isGreen()) {
104                 reduce++;
105             }
106         }
107         CardUtil.reduceCost(abilityToModify, reduce);
108         return true;
109     }
110 
111     /**
112      * Overwrite this in effect that inherits from this
113      *
114      * @param card
115      * @param source
116      * @param game
117      * @return
118      */
119     protected boolean selectedByRuntimeData(Card card, Ability source, Game game) {
120         return true;
121     }
122 
123     @Override
124     public boolean applies(Ability abilityToModify, Ability source, Game game) {
125         if (abilityToModify instanceof SpellAbility) {
126             Plane cPlane = game.getState().getCurrentPlane();
127             if (cPlane == null) {
128                 return false;
129             }
130             if (!cPlane.getPlaneType().equals(Planes.PLANE_FEEDING_GROUNDS)) {
131                 return false;
132             }
133 
134             Spell spell = (Spell) game.getStack().getStackObject(abilityToModify.getId());
135             if (spell != null) {
136                 return filter.match(spell, game) && selectedByRuntimeData(spell, source, game);
137             } else {
138                 // used at least for flashback ability because Flashback ability doesn't use stack
139                 Card sourceCard = game.getCard(abilityToModify.getSourceId());
140                 return filter.match(sourceCard, game) && selectedByRuntimeData(sourceCard, source, game);
141             }
142         }
143         return false;
144     }
145 
146     @Override
147     public FeedingGroundsEffect copy() {
148         return new FeedingGroundsEffect(this);
149     }
150 }

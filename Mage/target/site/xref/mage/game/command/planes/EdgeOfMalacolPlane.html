35  public class EdgeOfMalacolPlane extends Plane {
36  
37      public EdgeOfMalacolPlane() {
38          this.setPlaneType(Planes.PLANE_EDGE_OF_MALACOL);
39          this.setExpansionSetCodeForImage("PCA");
40  
41          // If a creature you control would untap during your untap step, put two +1/+1 counters on it instead.
42          SimpleStaticAbility ability = new SimpleStaticAbility(Zone.COMMAND, new EdgeOfMalacolEffect());
43          this.getAbilities().add(ability);
44  
45          // Active player can roll the planar die: Whenever you roll {CHAOS}, untap each creature you control
46          Effect chaosEffect = new UntapAllControllerEffect(new FilterControlledCreaturePermanent(), "untap each creature you control");
47          Target chaosTarget = null;
48  
49          List<Effect> chaosEffects = new ArrayList<Effect>();
50          chaosEffects.add(chaosEffect);
51          List<Target> chaosTargets = new ArrayList<Target>();
52          chaosTargets.add(chaosTarget);
53  
54          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
55          chaosAbility.addWatcher(new PlanarRollWatcher());
56          this.getAbilities().add(chaosAbility);
57          chaosAbility.setMayActivate(TargetController.ANY);
58          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
59      }
60  }

62  class EdgeOfMalacolEffect extends ContinuousRuleModifyingEffectImpl {
63  
64      private static final FilterCreaturePermanent filter = new FilterCreaturePermanent();
65  
66      public EdgeOfMalacolEffect() {
67          super(Duration.Custom, Outcome.Detriment);
68          this.staticText = "If a creature you control would untap during your untap step, put two +1/+1 counters on it instead";
69      }
70  
71      public EdgeOfMalacolEffect(final EdgeOfMalacolEffect effect) {
72          super(effect);
73      }
74  
75      @Override
76      public EdgeOfMalacolEffect copy() {
77          return new EdgeOfMalacolEffect(this);
78      }
79  
80      @Override
81      public boolean apply(Game game, Ability source) {
82          return false;
83      }
84  
85      @Override
86      public boolean checksEventType(GameEvent event, Game game) {
87          return event.getType() == GameEvent.EventType.UNTAP;
88      }
89  
90      @Override
91      public boolean applies(GameEvent event, Ability source, Game game) {
92          // Prevent untap event of creatures of target player
93          if (game.getTurnStepType() == PhaseStep.UNTAP) {
94              Plane cPlane = game.getState().getCurrentPlane();
95              if (cPlane == null) {
96                  return false;
97              }
98              if (!cPlane.getPlaneType().equals(Planes.PLANE_EDGE_OF_MALACOL)) {
99                  return false;
100             }
101             Permanent permanent = game.getPermanent(event.getTargetId());
102             if (filter.match(permanent, game) && Objects.equals(permanent.getControllerId(), game.getActivePlayerId())) {
103                 UUID oldController = source.getControllerId();
104                 source.setControllerId(game.getActivePlayerId());
105                 Effect effect = new AddCountersTargetEffect(CounterType.P1P1.createInstance(2));
106                 effect.setTargetPointer(new FixedTarget(permanent, game));
107                 effect.apply(game, source);
108                 source.setControllerId(oldController);
109                 return true;
110             }
111         }
112         return false;
113     }
114 }

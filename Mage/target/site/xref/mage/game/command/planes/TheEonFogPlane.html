27  public class TheEonFogPlane extends Plane {
28  
29      public TheEonFogPlane() {
30          this.setPlaneType(Planes.PLANE_THE_EON_FOG);
31          this.setExpansionSetCodeForImage("PCA");
32  
33          // All players miss their untap step
34          Ability ability = new SimpleStaticAbility(Zone.COMMAND, new TheEonFogSkipUntapStepEffect(Duration.Custom, true));
35          this.getAbilities().add(ability);
36  
37          // Active player can roll the planar die: Whenever you roll {CHAOS}, untap all permanents you control
38          Effect chaosEffect = new UntapAllControllerEffect(new FilterControlledPermanent());
39          Target chaosTarget = null;
40  
41          List<Effect> chaosEffects = new ArrayList<Effect>();
42          chaosEffects.add(chaosEffect);
43  
44          List<Target> chaosTargets = new ArrayList<Target>();
45          chaosTargets.add(chaosTarget);
46  
47          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
48          chaosAbility.addWatcher(new PlanarRollWatcher());
49          this.getAbilities().add(chaosAbility);
50          chaosAbility.setMayActivate(TargetController.ANY);
51          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
52      }
53  }
54  
55  class TheEonFogSkipUntapStepEffect extends ContinuousRuleModifyingEffectImpl {
56  
57      boolean allPlayers = false;
58  
59      public TheEonFogSkipUntapStepEffect() {
60          super(Duration.WhileOnBattlefield, Outcome.Neutral, false, false);
61          this.allPlayers = false;
62          staticText = "Players skip their untap steps";
63      }
64  
65      public TheEonFogSkipUntapStepEffect(Duration d, boolean allPlayers) {
66          super(d, Outcome.Neutral, false, false);
67          this.allPlayers = allPlayers;
68          staticText = "Players skip their untap steps";
69      }
70  
71      public TheEonFogSkipUntapStepEffect(final TheEonFogSkipUntapStepEffect effect) {
72          super(effect);
73          this.allPlayers = effect.allPlayers;
74      }
75  
76      @Override
77      public TheEonFogSkipUntapStepEffect copy() {
78          return new TheEonFogSkipUntapStepEffect(this);
79      }
80  
81      @Override
82      public boolean applies(GameEvent event, Ability source, Game game) {
83          Plane cPlane = game.getState().getCurrentPlane();
84          if (cPlane == null) {
85              return false;
86          }
87          if (!cPlane.getPlaneType().equals(Planes.PLANE_THE_EON_FOG)) {
88              return false;
89          }
90          return event.getType() == GameEvent.EventType.UNTAP_STEP;
91      }
92  }

31  public class TrugaJunglePlane extends Plane {
32  
33      private static final String rule = "All lands have '{t}: Add one mana of any color";
34  
35      public TrugaJunglePlane() {
36          this.setPlaneType(Planes.PLANE_TRUGA_JUNGLE);
37          this.setExpansionSetCodeForImage("PCA");
38  
39          SimpleStaticAbility ability
40                  = new SimpleStaticAbility(Zone.COMMAND, new ConditionalContinuousEffect(
41                  new GainAbilityAllEffect(new AnyColorManaAbility(), Duration.Custom, StaticFilters.FILTER_LANDS),
42                  new IsStillOnPlaneCondition(this.getName()),
43                  rule));
44          this.getAbilities().add(ability);
45  
46          // Active player can roll the planar die: Whenever you roll {CHAOS}, reveal the top three cards of your libary.  Put all land cards revealed this way into your hand the rest on the bottom of your library in any order.
47          Effect chaosEffect = new RevealLibraryPutIntoHandEffect(3, new FilterLandCard(), Zone.LIBRARY);
48          Target chaosTarget = null;
49  
50          List<Effect> chaosEffects = new ArrayList<Effect>();
51          chaosEffects.add(chaosEffect);
52          List<Target> chaosTargets = new ArrayList<Target>();
53          chaosTargets.add(chaosTarget);
54  
55          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
56          chaosAbility.addWatcher(new PlanarRollWatcher());
57          this.getAbilities().add(chaosAbility);
58          chaosAbility.setMayActivate(TargetController.ANY);
59          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
60      }
61  }

32  public class AstralArenaPlane extends Plane {
33  
34      public AstralArenaPlane() {
35          this.setPlaneType(Planes.PLANE_ASTRAL_ARENA);
36          this.setExpansionSetCodeForImage("PCA");
37  
38          // No more than one creature can attack each turn.  No more than one creature can block each turn.
39          SimpleStaticAbility ability = new SimpleStaticAbility(Zone.COMMAND, new AstralArenaAttackRestrictionEffect());
40          ability.addWatcher(new AttackedThisTurnWatcher());
41          SimpleStaticAbility ability2 = new SimpleStaticAbility(Zone.COMMAND, new AstralArenaBlockRestrictionEffect());
42          ability2.addWatcher(new AttackedThisTurnWatcher());
43          this.getAbilities().add(ability);
44          this.getAbilities().add(ability2);
45  
46          // Active player can roll the planar die: Whenever you roll {CHAOS}, {this} deals 2 damage to each creature
47          Effect chaosEffect = new DamageAllEffect(2, new FilterCreaturePermanent());
48          Target chaosTarget = null;
49  
50          List<Effect> chaosEffects = new ArrayList<Effect>();
51          chaosEffects.add(chaosEffect);
52          List<Target> chaosTargets = new ArrayList<Target>();
53          chaosTargets.add(chaosTarget);
54  
55          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
56          chaosAbility.addWatcher(new PlanarRollWatcher());
57          this.getAbilities().add(chaosAbility);
58          chaosAbility.setMayActivate(TargetController.ANY);
59          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
60      }
61  }

63  class AstralArenaAttackRestrictionEffect extends RestrictionEffect {
64  
65      AstralArenaAttackRestrictionEffect() {
66          super(Duration.WhileOnBattlefield);
67          staticText = "No more than one creature can attack each combat";
68      }
69  
70      AstralArenaAttackRestrictionEffect(final AstralArenaAttackRestrictionEffect effect) {
71          super(effect);
72      }
73  
74      @Override
75      public AstralArenaAttackRestrictionEffect copy() {
76          return new AstralArenaAttackRestrictionEffect(this);
77      }
78  
79      @Override
80      public boolean applies(Permanent permanent, Ability source, Game game) {
81          Plane cPlane = game.getState().getCurrentPlane();
82          if (cPlane == null) {
83              return false;
84          }
85          return cPlane.getPlaneType().equals(Planes.PLANE_ASTRAL_ARENA);
86      }
87  
88      @Override
89      public boolean canAttack(Permanent attacker, UUID defenderId, Ability source, Game game, boolean canUseChooseDialogs) {
90          return game.getCombat().getAttackers().isEmpty();
91      }
92  }

28  public class LetheLakePlane extends Plane {
29  
30      public LetheLakePlane() {
31          this.setPlaneType(Planes.PLANE_LETHE_LAKE);
32          this.setExpansionSetCodeForImage("PCA");
33  
34          // At the beginning of your upkeep, put the top ten cards of your libary into your graveyard
35          Ability ability = new BeginningOfUpkeepTriggeredAbility(Zone.COMMAND, new PutLibraryIntoGraveTargetEffect(10).setText("that player mills 10 cards"), TargetController.ANY, false, true);
36          this.getAbilities().add(ability);
37  
38          // Active player can roll the planar die: Whenever you roll {CHAOS}, target player puts the top ten cards of their library into their graveyard
39          Effect chaosEffect = new MillCardsTargetEffect(10);
40          Target chaosTarget = new TargetPlayer();
41  
42          List<Effect> chaosEffects = new ArrayList<Effect>();
43          chaosEffects.add(chaosEffect);
44  
45          List<Target> chaosTargets = new ArrayList<Target>();
46          chaosTargets.add(chaosTarget);
47  
48          ActivateIfConditionActivatedAbility chaosAbility = new ActivateIfConditionActivatedAbility(Zone.COMMAND, new RollPlanarDieEffect(chaosEffects, chaosTargets), new GenericManaCost(0), MainPhaseStackEmptyCondition.instance);
49          chaosAbility.addWatcher(new PlanarRollWatcher());
50          this.getAbilities().add(chaosAbility);
51          chaosAbility.setMayActivate(TargetController.ANY);
52          this.getAbilities().add(new SimpleStaticAbility(Zone.ALL, new PlanarDieRollCostIncreasingEffect(chaosAbility.getOriginalId())));
53      }
54  }

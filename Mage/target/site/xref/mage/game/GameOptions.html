1   package mage.game;
2   
3   import mage.constants.PhaseStep;
4   import mage.util.Copyable;
5   
6   import java.io.Serializable;
7   import java.util.Collections;
8   import java.util.Set;
9   
10  /**
11   * Game options for Mage game. Mainly used in tests to configure
12   * {@link GameImpl} with specific params.
13   *
14   * @author ayratn
15   */
16  public class GameOptions implements Serializable, Copyable<GameOptions> {
17  
18      private static final GameOptions deinstance = new GameOptions();
19  
20      public static GameOptions getDefault() {
21          return deinstance;
22      }
23  
24      /**
25       * Defines the running mode. There are some exclusions made for test mode.
26       */
27      public boolean testMode = false;
28  
29      /**
30       * Defines the turn number game should stop on. By default, is null meaning
31       * that game shouldn't stop on any specific turn.
32       */
33      public Integer stopOnTurn = null;
34  
35      /**
36       * Stop at the end of the turn if true otherwise stop at the beginning
37       */
38      public PhaseStep stopAtStep = PhaseStep.UNTAP;
39  
40      /**
41       * If true, library won't be shuffled at the beginning of the game
42       */
43      public boolean skipInitShuffling = false;
44  
45      /**
46       * If true, players can rollback turn if all players agree
47       */
48      public boolean rollbackTurnsAllowed = true;
49  
50      /**
51       * Names of users banned from participating in the game
52       */
53      public Set<String> bannedUsers = Collections.emptySet();
54  
55  
56      // PLANECHASE game mode
57      public boolean planeChase = false;
58      // xmage uses increased by 1/3 chances (2/2/9) for chaos/planar result, see 1a9f12f5767ce0beeed26a8ff5c8a8f9490c9c47
59      // if you need combo support with 6-sides rolls then it can be reset to original values
60      public static final int PLANECHASE_PLANAR_DIE_CHAOS_SIDES = 2; // original: 1
61      public static final int PLANECHASE_PLANAR_DIE_PLANAR_SIDES = 2; // original: 1
62      public static final int PLANECHASE_PLANAR_DIE_TOTAL_SIDES = 9; // original: 6
63  
64      public GameOptions() {
65          super();
66      }
67  
68      private GameOptions(final GameOptions options) {
69          this.testMode = options.testMode;
70          this.stopOnTurn = options.stopOnTurn;
71          this.stopAtStep = options.stopAtStep;
72          this.skipInitShuffling = options.skipInitShuffling;
73          this.rollbackTurnsAllowed = options.rollbackTurnsAllowed;
74          this.bannedUsers.addAll(options.bannedUsers);
75          this.planeChase = options.planeChase;
76      }
77  
78      @Override
79      public GameOptions copy() {
80          return new GameOptions(this);
81      }
82  }

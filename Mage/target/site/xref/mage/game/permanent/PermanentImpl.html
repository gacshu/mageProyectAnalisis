263     @Override
264     public List<String> getRules(Game game) {
265         try {
266             List<String> rules = getRules();
267 
268             // info
269             if (info != null) {
270                 for (String data : info.values()) {
271                     rules.add(data);
272                 }
273             }
274 
275             // ability hints
276             List<String> abilityHints = new ArrayList<>();
277             if (HintUtils.ABILITY_HINTS_ENABLE) {
278                 for (Ability ability : getAbilities(game)) {
279                     for (Hint hint : ability.getHints()) {
280                         String s = hint.getText(game, ability);
281                         if (s != null && !s.isEmpty()) {
282                             abilityHints.add(s);
283                         }
284                     }
285                 }
286             }
287 
288             // restrict hints
289             List<String> restrictHints = new ArrayList<>();
290             if (game != null && HintUtils.RESTRICT_HINTS_ENABLE) {
291                 // restrict
292                 for (Map.Entry<RestrictionEffect, Set<Ability>> entry : game.getContinuousEffects().getApplicableRestrictionEffects(this, game).entrySet()) {
293                     for (Ability ability : entry.getValue()) {
294                         if (!entry.getKey().canAttack(game, false) || !entry.getKey().canAttack(this, null, ability, game, false)) {
295                             restrictHints.add(HintUtils.prepareText("Can't attack" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_RESTRICT));
296                         }
297                         if (!entry.getKey().canBlock(null, this, ability, game, false)) {
298                             restrictHints.add(HintUtils.prepareText("Can't block" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_RESTRICT));
299                         }
300                         if (!entry.getKey().canBeUntapped(this, ability, game, false)) {
301                             restrictHints.add(HintUtils.prepareText("Can't untapped" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_RESTRICT));
302                         }
303                         if (!entry.getKey().canUseActivatedAbilities(this, ability, game, false)) {
304                             restrictHints.add(HintUtils.prepareText("Can't use activated abilities" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_RESTRICT));
305                         }
306                         if (!entry.getKey().canTransform(game, false)) {
307                             restrictHints.add(HintUtils.prepareText("Can't transform" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_RESTRICT));
308                         }
309                     }
310                 }
311 
312                 // requirement
313                 for (Map.Entry<RequirementEffect, Set<Ability>> entry : game.getContinuousEffects().getApplicableRequirementEffects(this, false, game).entrySet()) {
314                     for (Ability ability : entry.getValue()) {
315                         if (entry.getKey().mustAttack(game)) {
316                             restrictHints.add(HintUtils.prepareText("Must attack" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
317                         }
318                         if (entry.getKey().mustBlock(game)) {
319                             restrictHints.add(HintUtils.prepareText("Must block" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
320                         }
321                         if (entry.getKey().mustBlockAny(game)) {
322                             restrictHints.add(HintUtils.prepareText("Must block any" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
323                         }
324                         if (entry.getKey().mustBlockAllAttackers(game)) {
325                             restrictHints.add(HintUtils.prepareText("Must block all attackers" + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
326                         }
327 
328                         MageObject object = game.getObject(entry.getKey().mustAttackDefender(ability, game));
329                         if (object != null) {
330                             restrictHints.add(HintUtils.prepareText("Must attack defender " + object.getLogName() + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
331                         }
332                         object = game.getObject(entry.getKey().mustBlockAttacker(ability, game));
333                         if (object != null) {
334                             restrictHints.add(HintUtils.prepareText("Must block attacker " + object.getLogName() + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
335                         }
336                         object = game.getObject(entry.getKey().mustBlockAttackerIfElseUnblocked(ability, game));
337                         if (object != null) {
338                             restrictHints.add(HintUtils.prepareText("Must block attacker if able " + object.getLogName() + addSourceObjectName(game, ability), null, HintUtils.HINT_ICON_REQUIRE));
339                         }
340                     }
341                 }
342 
343                 restrictHints.sort(String::compareTo);
344             }
345 
346             // total hints
347             if (!abilityHints.isEmpty() || !restrictHints.isEmpty()) {
348                 rules.add(HintUtils.HINT_START_MARK);
349                 HintUtils.appendHints(rules, abilityHints);
350                 HintUtils.appendHints(rules, restrictHints);
351             }
352 
353             return rules;
354         } catch (Exception e) {
355             return CardUtil.RULES_ERROR_INFO;
356         }
357     }

379     @Override
380     public void addAbility(Ability ability, UUID sourceId, Game game) {
381         // singleton abilities -- only one instance
382         // other abilities -- any amount of instances
383         if (!abilities.containsKey(ability.getId())) {
384             Ability copyAbility = ability.copy();
385             copyAbility.newId(); // needed so that source can get an ability multiple times (e.g. Raging Ravine)
386             copyAbility.setControllerId(controllerId);
387             copyAbility.setSourceId(objectId);
388             // triggered abilities must be added to the state().triggers
389             // still as long as the prev. permanent is known to the LKI (e.g. Showstopper) so gained dies triggered ability will trigger
390             if (game != null) {
391                 // game is null in cards viewer window (MageBook)
392                 game.getState().addAbility(copyAbility, sourceId, this);
393             }
394             abilities.add(copyAbility);
395             abilities.addAll(ability.getSubAbilities());
396         }
397     }

405     @Override
406     public void removeAbility(Ability abilityToRemove, UUID sourceId, Game game) {
407         if (abilityToRemove == null) {
408             return;
409         }
410 
411         // 112.10b  Effects that remove an ability remove all instances of it.
412         List<Ability> toRemove = new ArrayList<>();
413         abilities.forEach(a -> {
414             if (a.isSameInstance(abilityToRemove)) {
415                 toRemove.add(a);
416             }
417         });
418 
419         // TODO: what about triggered abilities? See addAbility above -- triggers adds to GameState
420         toRemove.forEach(r -> abilities.remove(r));
421     }

610     @Override
611     public boolean transform(Ability source, Game game, boolean ignoreDayNight) {
612         if (!this.isTransformable()
613                 || (!ignoreDayNight && this.checkDayNightBound())
614                 || this.getOtherFace().isInstantOrSorcery()
615                 || !this.checkTransformRestrictionEffects(game)
616                 || (source != null && !source.checkTransformCount(this, game))) {
617             return false;
618         }
619         if (this.transformed) {
620             Card orgCard = this.getMainCard();
621             this.getPower().setModifiedBaseValue(orgCard.getPower().getValue());
622             this.getToughness().setModifiedBaseValue(orgCard.getToughness().getValue());
623         }
624         game.informPlayers(this.getLogName() + " transforms into " + this.getOtherFace().getLogName()
625                 + CardUtil.getSourceLogName(game, source, this.getId()));
626         this.setTransformed(!this.transformed);
627         this.transformCount++;
628         game.applyEffects();
629         this.replaceEvent(EventType.TRANSFORMING, game);
630         game.addSimultaneousEvent(GameEvent.getEvent(EventType.TRANSFORMED, this.getId(), this.getControllerId()));
631         return true;
632     }

655     public boolean phaseIn(Game game, boolean onlyDirect) {
656         if (!phasedIn) {
657             if (!replaceEvent(EventType.PHASE_IN, game)
658                     && (!onlyDirect || !indirectPhase)) {
659                 this.phasedIn = true;
660                 this.indirectPhase = false;
661                 if (!game.isSimulation()) {
662                     game.informPlayers(getLogName() + " phased in");
663                 }
664                 for (UUID attachedId : this.getAttachments()) {
665                     Permanent attachedPerm = game.getPermanent(attachedId);
666                     if (attachedPerm != null) {
667                         attachedPerm.phaseIn(game, false);
668                     }
669                 }
670                 fireEvent(EventType.PHASED_IN, game);
671                 return true;
672             }
673         }
674         return false;
675     }

682     @Override
683     public boolean phaseOut(Game game, boolean indirectPhase) {
684         if (phasedIn) {
685             if (!replaceEvent(EventType.PHASE_OUT, game)) {
686                 for (UUID attachedId : this.getAttachments()) {
687                     Permanent attachedPerm = game.getPermanent(attachedId);
688                     if (attachedPerm != null) {
689                         attachedPerm.phaseOut(game, true);
690                     }

941     /**
942      * @param damageAmount
943      * @param attackerId   id of the permanent or player who make damage (source.getSourceId() in most cases)
944      * @param source
945      * @param game
946      * @param preventable
947      * @param combat
948      * @param markDamage   If true, damage will be dealt later in applyDamage
949      *                     method, uses only in inner markDamage.
950      * @return
951      */
952     private int doDamage(int damageAmount, UUID attackerId, Ability source, Game game, boolean preventable, boolean combat, boolean markDamage, List<UUID> appliedEffects) {
953         int damageDone = 0;
954         if (damageAmount < 1 || !canDamage(game.getObject(attackerId), game)) {
955             return 0;
956         }
957         DamageEvent event = new DamagePermanentEvent(objectId, attackerId, controllerId, damageAmount, preventable, combat);
958         event.setAppliedEffects(appliedEffects);
959         if (game.replaceEvent(event)) {
960             return 0;
961         }
962         int actualDamage = checkProtectionAbilities(event, attackerId, source, game);
963         if (actualDamage < 1) {
964             return 0;
965         }
966         int lethal = getLethalDamage(attackerId, game);
967         MageObject attacker = game.getObject(attackerId);
968         if (this.isCreature(game)) {
969             if (checkWither(event, attacker, game)) {
970                 if (markDamage) {
971                     // mark damage only
972                     markDamage(CounterType.M1M1.createInstance(actualDamage), attacker, true);
973                 } else {
974                     Ability damageSourceAbility = null;
975                     if (attacker instanceof Permanent) {
976                         damageSourceAbility = ((Permanent) attacker).getSpellAbility();
977                     }
978                     // deal damage immediately
979                     addCounters(CounterType.M1M1.createInstance(actualDamage), game.getControllerId(attackerId), damageSourceAbility, game);
980                 }
981             } else {
982                 this.damage = CardUtil.overflowInc(this.damage, actualDamage);
983             }
984         }
985         if (this.isPlaneswalker(game)) {
986             int loyalty = getCounters(game).getCount(CounterType.LOYALTY);
987             int countersToRemove = Math.min(actualDamage, loyalty);
988             if (attacker != null && markDamage) {
989                 markDamage(CounterType.LOYALTY.createInstance(countersToRemove), attacker, false);
990             } else {
991                 removeCounters(CounterType.LOYALTY.getName(), countersToRemove, source, game);
992             }
993         }
994         DamagedEvent damagedEvent = new DamagedPermanentEvent(this.getId(), attackerId, this.getControllerId(), actualDamage, combat);
995         damagedEvent.setExcess(actualDamage - lethal);
996         game.fireEvent(damagedEvent);
997         game.getState().addSimultaneousDamage(damagedEvent, game);
998         damageDone = actualDamage;
999         if (damageDone < 1) {
1000             return 0;
1001         }
1002         UUID sourceControllerId = null;
1003         Abilities sourceAbilities = null;
1004         attacker = game.getPermanentOrLKIBattlefield(attackerId);
1005         if (attacker == null) {
1006             StackObject stackObject = game.getStack().getStackObject(attackerId);
1007             if (stackObject != null) {
1008                 attacker = stackObject.getStackAbility().getSourceObject(game);
1009             } else {
1010                 attacker = game.getObject(attackerId);
1011             }
1012             if (attacker instanceof Spell) {
1013                 sourceAbilities = ((Spell) attacker).getAbilities(game);
1014                 sourceControllerId = ((Spell) attacker).getControllerId();
1015             } else if (attacker instanceof Card) {
1016                 sourceAbilities = ((Card) attacker).getAbilities(game);
1017                 sourceControllerId = ((Card) attacker).getOwnerId();
1018             } else if (attacker instanceof CommandObject) {
1019                 sourceControllerId = ((CommandObject) attacker).getControllerId();
1020                 sourceAbilities = attacker.getAbilities();
1021             } else {
1022                 attacker = null;
1023             }
1024         } else {
1025             sourceAbilities = ((Permanent) attacker).getAbilities(game);
1026             sourceControllerId = ((Permanent) attacker).getControllerId();
1027         }
1028         if (attacker != null && sourceAbilities != null) {
1029             if (sourceAbilities.containsKey(LifelinkAbility.getInstance().getId())) {
1030                 if (markDamage) {
1031                     game.getPermanent(attackerId).markLifelink(damageDone);
1032                 } else {
1033                     Player player = game.getPlayer(sourceControllerId);
1034                     player.gainLife(damageDone, game, source);
1035                 }
1036             }
1037             if (sourceAbilities.containsKey(DeathtouchAbility.getInstance().getId())) {
1038                 deathtouched = true;
1039             }
1040             if (dealtDamageByThisTurn == null) {
1041                 dealtDamageByThisTurn = new HashSet<>();
1042             }
1043             // Unstable ability - Earl of Squirrel
1044             if (sourceAbilities.containsKey(SquirrellinkAbility.getInstance().getId())) {
1045                 Player player = game.getPlayer(sourceControllerId);
1046                 new SquirrelToken().putOntoBattlefield(damageDone, game, source, player.getId());
1047             }
1048             dealtDamageByThisTurn.add(new MageObjectReference(attacker, game));
1049         }
1050         if (attacker == null) {
1051             game.informPlayers(getLogName() + " gets " + damageDone + " damage");
1052         } else {
1053             game.informPlayers(attacker.getLogName() + " deals " + damageDone + " damage to " + getLogName());
1054         }
1055         return damageDone;
1056     }

1424     public boolean canBlock(UUID attackerId, Game game) {
1425         if (tapped && null == game.getState().getContinuousEffects().asThough(this.getId(), AsThoughEffectType.BLOCK_TAPPED, null, this.getControllerId(), game)) {
1426             return false;
1427         }
1428         Permanent attacker = game.getPermanent(attackerId);
1429         if (attacker == null) {
1430             return false;
1431         }
1432         // controller of attacking permanent must be an opponent
1433         if (!game.getPlayer(this.getControllerId()).hasOpponent(attacker.getControllerId(), game)) {
1434             return false;
1435         }
1436         //20101001 - 509.1b
1437         // check blocker restrictions
1438         for (Map.Entry<RestrictionEffect, Set<Ability>> entry : game.getContinuousEffects().getApplicableRestrictionEffects(this, game).entrySet()) {
1439             for (Ability ability : entry.getValue()) {
1440                 if (!entry.getKey().canBlock(attacker, this, ability, game, true)) {
1441                     return false;
1442                 }
1443             }
1444         }
1445         // check also attacker's restriction effects
1446         for (Map.Entry<RestrictionEffect, Set<Ability>> restrictionEntry : game.getContinuousEffects().getApplicableRestrictionEffects(attacker, game).entrySet()) {
1447             for (Ability ability : restrictionEntry.getValue()) {
1448                 if (!restrictionEntry.getKey().canBeBlocked(attacker, this, ability, game, true)) {
1449                     return false;
1450                 }
1451             }
1452         }
1453         return !attacker.hasProtectionFrom(this, game);
1454     }

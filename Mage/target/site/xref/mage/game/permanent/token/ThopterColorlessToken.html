3   import mage.MageInt;
4   import mage.abilities.keyword.FlyingAbility;
5   import mage.constants.CardType;
6   import mage.constants.SubType;
7   import mage.util.RandomUtil;
8   
9   import java.util.Arrays;
10  
11  /**
12   * @author fireshoes
13   */
14  public final class ThopterColorlessToken extends TokenImpl {
15  
16      public ThopterColorlessToken() {
17          super("Thopter Token", "1/1 colorless Thopter artifact creature token with flying");
18          cardType.add(CardType.ARTIFACT);
19          cardType.add(CardType.CREATURE);
20          subtype.add(SubType.THOPTER);
21          power = new MageInt(1);
22          toughness = new MageInt(1);
23  
24          addAbility(FlyingAbility.getInstance());
25  
26          availableImageSetCodes = Arrays.asList("C18", "KLD", "MBS", "ORI", "VMA", "M19", "ZNC",
27                  "KHC", "C21", "MH2", "AFC", "VOC", "NEC", "2XM", "RNA", "NCC", "DDU", "BRO", "ONC");
28      }
29  
30      @Override
31      public void setExpansionSetCodeForImage(String code) {
32          super.setExpansionSetCodeForImage(code);
33  
34          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("C18")) {
35              this.setTokenType(RandomUtil.nextInt(2) + 1);
36          }
37          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("KLD")) {
38              this.setTokenType(RandomUtil.nextInt(3) + 1);
39          }
40          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("ORI")) {
41              this.setTokenType(RandomUtil.nextInt(2) + 1);
42          }
43          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("2XM")) {
44              this.setTokenType(1);
45          }
46          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("DDU")) {
47              this.setTokenType(RandomUtil.nextInt(2) + 1);
48          }
49  
50      }
51  
52      public ThopterColorlessToken(final ThopterColorlessToken token) {
53          super(token);
54      }
55  
56      @Override
57      public ThopterColorlessToken copy() {
58          return new ThopterColorlessToken(this);
59      }
60  
61  }

2   package mage.game.permanent.token;
3   
4   import java.util.Arrays;
5   import java.util.Collections;
6   
7   import mage.MageInt;
8   import mage.abilities.Ability;
9   import mage.abilities.common.DiesSourceTriggeredAbility;
10  import mage.abilities.effects.Effect;
11  import mage.abilities.effects.common.DamageTargetEffect;
12  import mage.constants.CardType;
13  import mage.constants.SubType;
14  import mage.target.common.TargetAnyTarget;
15  
16  /**
17   *
18   * @author fireshoes
19   */
20  public final class DevilToken extends TokenImpl {
21  
22      public DevilToken() {
23          super("Devil Token", "1/1 red Devil creature token with \"When this creature dies, it deals 1 damage to any target.\"");
24          cardType.add(CardType.CREATURE);
25          subtype.add(SubType.DEVIL);
26          color.setRed(true);
27          power = new MageInt(1);
28          toughness = new MageInt(1);
29  
30          // When this creature dies, it deals 1 damage to any target.
31          Effect effect = new DamageTargetEffect(1);
32          effect.setText("it deals 1 damage to any target");
33          Ability ability = new DiesSourceTriggeredAbility(effect);
34          ability.addTarget(new TargetAnyTarget());
35          this.addAbility(ability);
36  
37          availableImageSetCodes = Arrays.asList("SOI", "WAR", "AFR", "MID", "SNC", "NCC", "CLB");
38      }
39  
40      public DevilToken(final DevilToken token) {
41          super(token);
42      }
43  
44      public DevilToken copy() {
45          return new DevilToken(this);
46      }
47  }

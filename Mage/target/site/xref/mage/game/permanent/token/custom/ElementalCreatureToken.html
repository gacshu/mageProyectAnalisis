1   package mage.game.permanent.token.custom;
2   
3   import mage.MageInt;
4   import mage.ObjectColor;
5   import mage.constants.CardType;
6   import mage.constants.SubType;
7   import mage.game.permanent.token.TokenImpl;
8   
9   /**
10   *
11   * @author JayDi85
12   */
13  public final class ElementalCreatureToken extends TokenImpl {
14  
15      public ElementalCreatureToken() {
16          this(0, 0);
17      }
18  
19      public ElementalCreatureToken(int power, int toughness) {
20          this(power, toughness, String.format("%d/%d Elemental creature", power, toughness));
21      }
22  
23      public ElementalCreatureToken(int power, int toughness, String description) {
24          this(power, toughness, description, (ObjectColor) null);
25      }
26  
27      public ElementalCreatureToken(int power, int toughness, String description, ObjectColor color) {
28          super("", description);
29          this.cardType.add(CardType.CREATURE);
30          this.subtype.add(SubType.ELEMENTAL);
31  
32          this.power = new MageInt(power);
33          this.toughness = new MageInt(toughness);
34  
35          if (color != null) {
36              this.color.addColor(color);
37          }
38      }
39  
40      public ElementalCreatureToken(final ElementalCreatureToken token) {
41          super(token);
42      }
43  
44      public ElementalCreatureToken copy() {
45          return new ElementalCreatureToken(this);
46      }
47  }

12  import java.util.Arrays;

15   * @author TheElk801

17  public final class PhyrexianMiteToken extends TokenImpl {
18  
19      public PhyrexianMiteToken() {
20          super("Phyrexian Mite Token", "1/1 colorless Phyrexian Mite artifact creature token with toxic 1 and \"This creature can't block.\"");
21          cardType.add(CardType.ARTIFACT);
22          cardType.add(CardType.CREATURE);
23          subtype.add(SubType.PHYREXIAN);
24          subtype.add(SubType.MITE);
25          power = new MageInt(1);
26          toughness = new MageInt(1);
27  
28          this.addAbility(new ToxicAbility(1));
29          this.addAbility(new SimpleStaticAbility(new CantBlockSourceEffect(Duration.WhileOnBattlefield).setText("this creature can't block")));
30  
31          availableImageSetCodes = Arrays.asList("ONE");
32      }
33  
34      @Override
35      public void setExpansionSetCodeForImage(String code) {
36          super.setExpansionSetCodeForImage(code);
37  
38          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("ONE")) {
39              this.setTokenType(RandomUtil.nextInt(2) + 1);
40          }
41      }
42  
43      public PhyrexianMiteToken(final PhyrexianMiteToken token) {
44          super(token);
45      }
46  
47      @Override
48      public PhyrexianMiteToken copy() {
49          return new PhyrexianMiteToken(this);
50      }
51  }

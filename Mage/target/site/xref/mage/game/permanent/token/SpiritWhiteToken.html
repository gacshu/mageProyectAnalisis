3   import mage.MageInt;
4   import mage.abilities.keyword.FlyingAbility;
5   import mage.constants.CardType;
6   import mage.constants.SubType;
7   
8   import java.util.Arrays;
9   
10  /**
11   * @author nantuko
12   */
13  public final class SpiritWhiteToken extends TokenImpl {
14  
15      public SpiritWhiteToken() {
16          super("Spirit Token", "1/1 white Spirit creature token with flying");
17          cardType.add(CardType.CREATURE);
18          subtype.add(SubType.SPIRIT);
19          color.setWhite(true);
20          power = new MageInt(1);
21          toughness = new MageInt(1);
22  
23          addAbility(FlyingAbility.getInstance());
24  
25          availableImageSetCodes = Arrays.asList("AVR", "C14", "CNS", "DDC", "DDK", "FRF", "ISD", "KTK", "M15", "MM2", "SHM",
26                  "SOI", "EMA", "C16", "MM3", "CMA", "E01", "RAV", "EMN", "M20", "C19", "C20", "CMR", "KHM",
27                  "MID", "VOW", "UMA", "BBD", "IMA", "CM2", "MD1", "DVD", "DDQ", "CN2", "A25", "GK2", "2X2", "ONC");
28      }
29  
30      @Override
31      public void setExpansionSetCodeForImage(String code) {
32          super.setExpansionSetCodeForImage(code);
33          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("AVR")) {
34              setTokenType(2);
35          }
36          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("C16")) {
37              setTokenType(2);
38          }
39          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("EMA")) {
40              setTokenType(2);
41          }
42          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("UMA")) {
43              setTokenType(1);
44          }
45          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("A25")) {
46              setTokenType(2);
47          }
48          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("2X2")) {
49              setTokenType(2);
50          }
51      }
52  
53      public SpiritWhiteToken(final SpiritWhiteToken token) {
54          super(token);
55      }
56  
57      @Override
58      public SpiritWhiteToken copy() {
59          return new SpiritWhiteToken(this);
60      }
61  }

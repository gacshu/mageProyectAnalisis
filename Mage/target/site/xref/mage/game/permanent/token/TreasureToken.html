1   package mage.game.permanent.token;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.common.SacrificeSourceCost;
5   import mage.abilities.costs.common.TapSourceCost;
6   import mage.abilities.effects.mana.AddManaOfAnyColorEffect;
7   import mage.abilities.mana.SimpleManaAbility;
8   import mage.constants.CardType;
9   import mage.constants.SubType;
10  import mage.constants.Zone;
11  import mage.util.RandomUtil;
12  
13  import java.util.Arrays;
14  
15  /**
16   * @author TheElk801
17   */
18  public final class TreasureToken extends TokenImpl {
19  
20      public TreasureToken() {
21          super("Treasure Token", "Treasure token");
22          cardType.add(CardType.ARTIFACT);
23          subtype.add(SubType.TREASURE);
24  
25          // {T}, Sacrifice this artifact: Add one mana of any color.
26          Ability ability = new SimpleManaAbility(Zone.BATTLEFIELD, new AddManaOfAnyColorEffect(), new TapSourceCost());
27          ability.addCost(new SacrificeSourceCost().setText("sacrifice this artifact"));
28          this.addAbility(ability);
29  
30          availableImageSetCodes = Arrays.asList("XLN", "RNA", "M20", "C19", "C20",
31                  "M21", "CMR", "KHM", "STX", "MH2",
32                  "AFR", "VOW", "NEO", "SLD", "2XM",
33                  "SNC", "CLB", "2X2", "DMC", "GN3");
34      }
35  
36      public TreasureToken(final TreasureToken token) {
37          super(token);
38      }
39  
40      public TreasureToken copy() {
41          return new TreasureToken(this);
42      }
43  
44      @Override
45      public void setExpansionSetCodeForImage(String code) {
46          super.setExpansionSetCodeForImage(code);
47  
48          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("XLN")) {
49              this.setTokenType(RandomUtil.nextInt(4) + 1);
50          }
51          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("SNC")) {
52              this.setTokenType(RandomUtil.nextInt(5) + 1);
53          }
54          if (getOriginalExpansionSetCode() != null && getOriginalExpansionSetCode().equals("MH2")) {
55              this.setTokenType(RandomUtil.nextInt(2) + 1);
56          }
57      }
58  }

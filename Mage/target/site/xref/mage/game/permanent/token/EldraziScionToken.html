4   import java.util.ArrayList;
5   import java.util.Arrays;
6   import java.util.List;

8   import mage.MageInt;
9   import mage.Mana;
10  import mage.abilities.costs.common.SacrificeSourceCost;
11  import mage.abilities.mana.SimpleManaAbility;
12  import mage.constants.CardType;
13  import mage.constants.SubType;
14  import mage.constants.Zone;
15  import mage.util.RandomUtil;

17  /**
18   *
19   * @author fireshoes
20   */
21  public final class EldraziScionToken extends TokenImpl {
22  
23      static final private List<String> tokenImageSets = new ArrayList<>();
24  
25      static {
26          tokenImageSets.addAll(Arrays.asList("BFZ", "OGW", "DDR", "2X2"));
27      }
28  
29      public EldraziScionToken() {
30          super("Eldrazi Scion Token", "1/1 colorless Eldrazi Scion creature token with \"Sacrifice this creature: Add {C}.\"");
31          cardType.add(CardType.CREATURE);
32          subtype.add(SubType.ELDRAZI);
33          subtype.add(SubType.SCION);
34          power = new MageInt(1);
35          toughness = new MageInt(1);
36          addAbility(new SimpleManaAbility(Zone.BATTLEFIELD, Mana.ColorlessMana(1), new SacrificeSourceCost()));
37          availableImageSetCodes = tokenImageSets;
38      }
39  
40      @Override
41      public void setExpansionSetCodeForImage(String code) {
42          super.setExpansionSetCodeForImage(code);
43          if (getOriginalExpansionSetCode().equals("BFZ")) {
44              this.setTokenType(RandomUtil.nextInt(3) + 1); // 3 different images
45          }
46          if (getOriginalExpansionSetCode().equals("OGW")) {
47              this.setTokenType(RandomUtil.nextInt(6) + 1); // 6 different images
48          }
49      }
50  
51      public EldraziScionToken(final EldraziScionToken token) {
52          super(token);
53      }
54  
55      @Override
56      public EldraziScionToken copy() {
57          return new EldraziScionToken(this); //To change body of generated methods, choose Tools | Templates.
58      }
59  }

21  public final class SmokeBlessingToken extends TokenImpl {
22  
23      public SmokeBlessingToken() {
24          super(
25                  "Smoke Blessing", "red Aura enchantment token named Smoke Blessing " +
26                          "attached to that creature. Those tokens have enchant creature and " +
27                          "\"When enchanted creature dies, it deals 1 damage to its controller and you create a Treasure token.\""
28          );
29          cardType.add(CardType.ENCHANTMENT);
30          color.setRed(true);
31          subtype.add(SubType.AURA);
32  
33          TargetPermanent auraTarget = new TargetPermanent();
34          Ability ability = new EnchantAbility(auraTarget);
35          ability.addTarget(auraTarget);
36          ability.addEffect(new AttachEffect(Outcome.BoostCreature));
37          this.addAbility(ability);
38  
39          this.addAbility(new DiesAttachedTriggeredAbility(new SmokeBlessingTokenEffect(), "enchanted creature"));
40  
41          availableImageSetCodes = Arrays.asList("NEC");
42      }
43  
44      public SmokeBlessingToken(final SmokeBlessingToken token) {
45          super(token);
46      }
47  
48      public SmokeBlessingToken copy() {
49          return new SmokeBlessingToken(this);
50      }
51  }
52  
53  class SmokeBlessingTokenEffect extends OneShotEffect {
54  
55      SmokeBlessingTokenEffect() {
56          super(Outcome.Benefit);
57          staticText = "it deals 1 damage to its controller and you create a Treasure token";
58      }
59  
60      private SmokeBlessingTokenEffect(final SmokeBlessingTokenEffect effect) {
61          super(effect);
62      }
63  
64      @Override
65      public SmokeBlessingTokenEffect copy() {
66          return new SmokeBlessingTokenEffect(this);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Permanent permanent = (Permanent) getValue("attachedTo");
72          if (permanent != null) {
73              Player player = game.getPlayer(permanent.getControllerId());
74              if (player != null) {
75                  player.damage(1, permanent.getId(), source, game);
76              }
77          }
78          new TreasureToken().putOntoBattlefield(1, game, source);
79          return true;
80      }
81  }

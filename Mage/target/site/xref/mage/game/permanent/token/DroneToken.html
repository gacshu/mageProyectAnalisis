3   import mage.MageInt;
4   import mage.abilities.Ability;
5   import mage.abilities.common.LeavesBattlefieldTriggeredAbility;
6   import mage.abilities.effects.common.GainLifeEffect;
7   import mage.abilities.effects.common.LoseLifeOpponentsEffect;
8   import mage.abilities.keyword.DeathtouchAbility;
9   import mage.constants.CardType;
10  import mage.constants.SubType;
11  
12  /**
13   * @author @stwalsh4118
14   */
15  public class DroneToken extends TokenImpl {
16  
17      public DroneToken() {
18          super("Drone Token", "2/2 colorless Drone artifact creature token with deathtouch and \"When this creature leaves the battlefield, each opponent loses 2 life and you gain 2 life.\"");
19          cardType.add(CardType.CREATURE);
20          cardType.add(CardType.ARTIFACT);
21          subtype.add(SubType.DRONE);
22          power = new MageInt(2);
23          toughness = new MageInt(2);
24          addAbility(DeathtouchAbility.getInstance());
25  
26          Ability ability = new LeavesBattlefieldTriggeredAbility(new LoseLifeOpponentsEffect(2), false).setTriggerPhrase("When this creature leaves the battlefield, ");
27          ability.addEffect(new GainLifeEffect(2).concatBy("and"));
28          addAbility(ability);
29  
30          setOriginalExpansionSetCode("ONE");
31  
32      }
33  
34      private DroneToken(final DroneToken token) {
35          super(token);
36      }
37  
38      @Override
39      public DroneToken copy() {
40          return new DroneToken(this);
41      }
42  }

22  /**
23   * Static permanent on the battlefield. There are possible multiple permanents per one card,
24   * so be carefull for targets (ids are different) and ZCC (zcc is static for permanent).
25   *
26   * @author BetaSteward_at_googlemail.com
27   */
28  @SupportedSourceVersion(SourceVersion.RELEASE_8)
29  public class PermanentCard extends PermanentImpl {
30  
31      protected int maxLevelCounters;
32      // A copy of the origin card that was cast (this is not the original card, so it's possible to change some attribute to this blueprint to change attributes to the permanent if it enters the battlefield with e.g. a subtype)
33      protected Card card;
34      // the number this permanent instance had
35      protected int zoneChangeCounter;
36  
37      public PermanentCard(Card card, UUID controllerId, Game game) {
38          super(card.getId(), card.getOwnerId(), controllerId, card.getName());
39  
40          // usage check: you must put to play only real card's part
41          // if you use it in test code then call CardUtil.getDefaultCardSideForBattlefield for default side
42          // it's a basic check and still allows to create permanent from instant or sorcery
43          boolean goodForBattlefield = true;
44          if (card instanceof ModalDoubleFacesCard) {
45              goodForBattlefield = false;
46          } else if (card instanceof SplitCard) {
47              // fused spells allowed (it uses main card)
48              if (card.getSpellAbility() != null && !card.getSpellAbility().getSpellAbilityType().equals(SpellAbilityType.SPLIT_FUSED)) {
49                  goodForBattlefield = false;
50              }
51          }
52          if (!goodForBattlefield) {
53              throw new IllegalArgumentException("ERROR, can't create permanent card from split or mdf: " + card.getName());
54          }
55  
56          this.card = card;
57          this.zoneChangeCounter = card.getZoneChangeCounter(game); // local value already set to the raised number
58          init(card, game);
59      }
60  
61      private void init(Card card, Game game) {
62          power = card.getPower().copy();
63          toughness = card.getToughness().copy();
64          startingLoyalty = card.getStartingLoyalty();
65          copyFromCard(card, game);
66          // if temporary added abilities to the spell/card exist, you need to add it to the permanent derived from that card
67          Abilities<Ability> otherAbilities = game.getState().getAllOtherAbilities(card.getId());
68          if (otherAbilities != null) {
69              abilities.addAll(otherAbilities);
70          }
71          if (card instanceof LevelerCard) {
72              maxLevelCounters = ((LevelerCard) card).getMaxLevelCounters();
73          }
74          if (card.isTransformable()) {
75              if (game.getState().getValue(TransformAbility.VALUE_KEY_ENTER_TRANSFORMED + getId()) != null
76                      || NightboundAbility.checkCard(this, game)) {
77                  game.getState().setValue(TransformAbility.VALUE_KEY_ENTER_TRANSFORMED + getId(), null);
78                  TransformAbility.transformPermanent(this, getSecondCardFace(), game, null);
79              }
80          }
81      }
82  
83      public PermanentCard(final PermanentCard permanent) {
84          super(permanent);
85          this.card = permanent.card.copy();
86          this.maxLevelCounters = permanent.maxLevelCounters;
87          this.zoneChangeCounter = permanent.zoneChangeCounter;
88      }
89  
90      @Override
91      public void reset(Game game) {
92          // when the permanent is reset, copy all original values from the card
93          // must copy card each reset so that the original values don't get modified
94          copyFromCard(card, game);
95          power.resetToBaseValue();
96          toughness.resetToBaseValue();
97          super.reset(game);
98      }
99  
100     protected void copyFromCard(final Card card, final Game game) {
101         this.name = card.getName();
102         this.abilities.clear();
103         if (this.faceDown) {
104             for (Ability ability : card.getAbilities()) {
105                 if (ability.getWorksFaceDown()) {
106                     this.abilities.add(ability.copy());
107                 }
108             }
109         } else {
110             // copy only own abilities; all dynamic added abilities must be added in the parent call
111             this.abilities = card.getAbilities().copy();
112             this.spellAbility = null; // will be set on first getSpellAbility call if card has one.
113         }
114         this.abilities.setControllerId(this.controllerId);
115         this.abilities.setSourceId(objectId);
116         this.cardType.clear();
117         this.cardType.addAll(card.getCardType());
118         this.color = card.getColor(null).copy();
119         this.frameColor = card.getFrameColor(game).copy();
120         this.frameStyle = card.getFrameStyle();
121         this.manaCost = card.getManaCost().copy();
122         if (card instanceof PermanentCard) {
123             this.maxLevelCounters = ((PermanentCard) card).maxLevelCounters;
124         }
125         this.subtype.copyFrom(card.getSubtype());
126         this.supertype.clear();
127         this.supertype.addAll(card.getSuperType());
128         this.expansionSetCode = card.getExpansionSetCode();
129         this.rarity = card.getRarity();
130         this.cardNumber = card.getCardNumber();
131         this.usesVariousArt = card.getUsesVariousArt();
132 
133         if (card.getSecondCardFace() != null) {
134             this.secondSideCardClazz = card.getSecondCardFace().getClass();
135         }
136         if (card.getMeldsToCard() != null) {
137             this.meldsToClazz = card.getMeldsToCard().getClass();
138         }
139         this.nightCard = card.isNightCard();
140         this.flipCard = card.isFlipCard();
141         this.flipCardName = card.getFlipCardName();
142     }

162     @Override
163     public boolean turnFaceUp(Ability source, Game game, UUID playerId) {
164         if (super.turnFaceUp(source, game, playerId)) {
165             power.setModifiedBaseValue(power.getBaseValue());
166             toughness.setModifiedBaseValue(toughness.getBaseValue());
167             setManifested(false);
168             setMorphed(false);
169             return true;
170         }
171         return false;
172     }
173 
174     @Override
175     public ManaCosts<ManaCost> getManaCost() {
176         if (faceDown) { // face down permanent has always {0} mana costs
177             manaCost.clear();
178             return manaCost;
179         }
180         return super.getManaCost();
181     }
182 
183     @Override
184     public int getManaValue() {
185         if (isTransformed()) {
186             // 711.4b While a double-faced permanent's back face is up, it has only the characteristics of its back face.
187             // However, its converted mana cost is calculated using the mana cost of its front face. This is a change from previous rules.
188             // If a permanent is copying the back face of a double-faced card (even if the card representing that copy
189             // is itself a double-faced card), the converted mana cost of that permanent is 0.
190             return getCard().getManaValue();
191         }
192         if (faceDown) { // game not neccessary
193             return getManaCost().manaValue();
194         }
195         return super.getManaValue();
196 
197     }

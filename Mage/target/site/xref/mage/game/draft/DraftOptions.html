3   import mage.game.tournament.LimitedOptions;
4   
5   import java.io.Serializable;
6   import java.util.Arrays;
7   import java.util.List;
8   
9   /**
10   * @author BetaSteward_at_googlemail.com, JayDi85
11   */
12  public class DraftOptions extends LimitedOptions implements Serializable {
13  
14      protected TimingOption timing;
15  
16      public enum TimingOption {
17  
18          // seconds per card's pick
19          BEGINNER("x2.0", "Beginner (x2.0)", 2.0,
20                  Arrays.asList(150, 140, 130, 120, 110, 100, 90, 80, 45, 35, 30, 25, 20, 15, 10)
21          ),
22          REGULAR("x1.5", "Regular (x1.5)", 1.5,
23                  Arrays.asList(113, 105, 98, 90, 83, 75, 68, 60, 35, 30, 25, 20, 15, 10, 8)
24          ),
25          PROFESSIONAL("x1.0", "Professional (x1.0)", 1.0,
26                  Arrays.asList(75, 70, 65, 60, 55, 50, 45, 40, 30, 25, 20, 15, 12, 10, 7)
27          ),
28          NONE("ERROR", "", 0,
29                  Arrays.asList(1)
30          );
31  
32          private final String shortName;
33          private final String name;
34          private final double customTimeoutFactor; // profi as x1.0, other modes increases by factor from x1.5 to x2.0
35          private final List<Integer> times;
36  
37          TimingOption(String shortName, String name, double customTimeoutFactor, List<Integer> times) {
38              this.shortName = shortName;
39              this.name = name;
40              this.customTimeoutFactor = customTimeoutFactor;
41              this.times = times;
42          }
43  
44          public String getShortName() {
45              return shortName;
46          }
47  
48          public String getName() {
49              return name;
50          }
51  
52          public double getCustomTimeoutFactor() {
53              return customTimeoutFactor;
54          }
55  
56          public int getPickTimeout(int cardNum) {
57              if (cardNum > 15) {
58                  cardNum = 15;
59              }
60  
61              if (times.size() >= cardNum) {
62                  return times.get(cardNum - 1);
63              } else {
64                  return times.get(times.size() - 1);
65              }
66          }
67  
68          @Override
69          public String toString() {
70              return name;
71          }
72      }
73  
74      public TimingOption getTiming() {
75          return timing;
76      }
77  
78      public void setTiming(TimingOption timing) {
79          this.timing = timing;
80      }
81  }

47      private void handlePipers(Player player, Set<Card> commanders) {
48          int piperCount = commanders
49                  .stream()
50                  .filter(CommanderChooseColorAbility::checkCard)
51                  .mapToInt(x -> 1)
52                  .sum();
53          if (piperCount < 1) {
54              return;
55          }
56          FilterMana leftoverColors = new FilterMana();
57          Stream.concat(
58                  player.getLibrary().getCards(this).stream(),
59                  player.getSideboard().getCards(this).stream()
60          ).map(Card::getColorIdentity).forEach(leftoverColors::addAll);
61          FilterMana nonPiperIdentity = new FilterMana();
62          commanders
63                  .stream()
64                  .filter(card -> !CommanderChooseColorAbility.checkCard(card))
65                  .map(Card::getColorIdentity)
66                  .forEach(nonPiperIdentity::addAll);
67          leftoverColors.removeAll(nonPiperIdentity);
68          if (piperCount < leftoverColors.getColorCount()) {
69              throw new UnsupportedOperationException("This deck should not be legal, something went wrong");
70          }
71          Iterator<ObjectColor> iterator = leftoverColors.getColors().listIterator();
72          for (Card commander : commanders) {
73              if (!CommanderChooseColorAbility.checkCard(commander)) {
74                  continue;
75              }
76              ObjectColor color;
77              if (!iterator.hasNext()) {
78                  ChoiceColor choiceColor = new ChoiceColor(
79                          true, "Choose a color for " + commander.getName()
80                  );
81                  player.choose(Outcome.Neutral, choiceColor, this);
82                  color = choiceColor.getColor();
83              } else {
84                  color = iterator.next();
85              }
86              commander.getColor().addColor(color);
87          }
88      }

90      @Override
91      protected void init(UUID choosingPlayerId) {
92          // Karn Liberated calls it to restart game, all data and commanders must be re-initialized
93  
94          // add game mode specific watchers here
95          //state.addWatcher(new CommanderPlaysCountWatcher());
96  
97          // move commanders to command zone
98          for (UUID playerId : state.getPlayerList(startingPlayerId)) {
99              Player player = getPlayer(playerId);
100             if (player == null) {
101                 continue;
102             }
103             // add new commanders
104             Set<Card> commanders = new HashSet<>();
105             for (UUID cardId : player.getSideboard()) {
106                 Card card = this.getCard(cardId);
107                 if (card == null) {
108                     continue;
109                 }
110                 // Check for companions. If it is the only card in the sideboard, it is the commander, not a companion.
111                 if (player.getSideboard().size() > 1 && card.getAbilities(this).stream().anyMatch(CompanionAbility.class::isInstance)) {
112                     continue;
113                 }
114                 commanders.add(card);
115                 addCommander(card, player);
116             }
117 
118             handlePipers(player, commanders);
119 
120             // init commanders
121             for (UUID commanderId : this.getCommandersIds(player, CommanderCardType.ANY, false)) {
122                 Card commander = this.getCard(commanderId);
123                 if (commander != null) {
124                     initCommander(commander, player);
125                 }
126             }
127         }
128 
129         super.init(choosingPlayerId);
130         if (startingPlayerSkipsDraw) {
131             state.getTurnMods().add(new TurnMod(startingPlayerId, PhaseStep.DRAW));
132         }
133     }

135     public void initCommander(Card commander, Player player) {
136         if (!Zone.EXILED.equals(getState().getZone(commander.getId()))) { // Exile check needed for Karn Liberated restart
137             commander.moveToZone(Zone.COMMAND, null, this, true);
138         }
139         commander.getAbilities().setControllerId(player.getId());
140 
141         Ability ability = new SimpleStaticAbility(Zone.COMMAND, new InfoEffect("Commander effects"));
142         initCommanderEffects(commander, player, ability);
143         CommanderInfoWatcher watcher = initCommanderWatcher(commander, checkCommanderDamage);
144         getState().addWatcher(watcher);
145         watcher.addCardInfoToCommander(this);
146         this.getState().addAbility(ability, null);
147     }

153     public void initCommanderEffects(Card commander, Player player, Ability commanderAbility) {
154         // all commander effects must be independent from sourceId or controllerId
155         commanderAbility.addEffect(new CommanderReplacementEffect(commander.getId(), alsoHand, alsoLibrary, false, "Commander"));
156         commanderAbility.addEffect(new CommanderCostModification(commander));
157     }

238     /* 20130711
239      *903.14a A player that's been dealt 21 or more combat damage by the same commander
240      * over the course of the game loses the game. (This is a state-based action. See rule 704.)
241      *
242      */
243     @Override
244     protected boolean checkStateBasedActions() {
245         for (Player player : getPlayers().values()) {
246             for (UUID commanderId : this.getCommandersIds(player, CommanderCardType.COMMANDER_OR_OATHBREAKER, false)) {
247                 CommanderInfoWatcher damageWatcher = getState().getWatcher(CommanderInfoWatcher.class, commanderId);
248                 if (damageWatcher == null) {
249                     continue;
250                 }
251                 for (Map.Entry<UUID, Integer> entrySet : damageWatcher.getDamageToPlayer().entrySet()) {
252                     if (entrySet.getValue() > 20) {
253                         Player opponent = getPlayer(entrySet.getKey());
254                         if (opponent != null && !opponent.hasLost() && player.isInGame()) {
255                             opponent.lost(this);
256                         }
257                     }
258                 }
259             }
260         }
261         return super.checkStateBasedActions();
262     }

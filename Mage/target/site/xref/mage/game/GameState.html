138     public GameState(final GameState state) {
139         this.players = state.players.copy();
140         this.playerList = state.playerList.copy();
141         this.choosingPlayerId = state.choosingPlayerId;
142         this.revealed = state.revealed.copy();
143         this.lookedAt.putAll(state.lookedAt);
144         this.companion = state.companion.copy();
145         this.gameOver = state.gameOver;
146         this.paused = state.paused;
147 
148         this.activePlayerId = state.activePlayerId;
149         this.priorityPlayerId = state.priorityPlayerId;
150         this.playerByOrderId = state.playerByOrderId;
151         this.monarchId = state.monarchId;
152         this.initiativeId = state.initiativeId;
153         this.turn = state.turn.copy();
154 
155         this.stack = state.stack.copy();
156         this.command = state.command.copy();
157         this.isPlaneChase = state.isPlaneChase;
158         this.seenPlanes.addAll(state.seenPlanes);
159         this.designations.addAll(state.designations);
160         this.exile = state.exile.copy();
161         this.battlefield = state.battlefield.copy();
162         this.turnNum = state.turnNum;
163         this.stepNum = state.stepNum;
164         this.extraTurn = state.extraTurn;
165         this.effects = state.effects.copy();
166         for (TriggeredAbility trigger : state.triggered) {
167             this.triggered.add(trigger.copy());
168         }
169         this.triggers = state.triggers.copy();
170         this.delayed = state.delayed.copy();
171         this.specialActions = state.specialActions.copy();
172         this.combat = state.combat.copy();
173         this.turnMods = state.turnMods.copy();
174         this.watchers = state.watchers.copy();
175         for (Map.Entry<String, Object> entry : state.values.entrySet()) {
176             if (entry.getValue() instanceof HashSet) {
177                 this.values.put(entry.getKey(), ((HashSet) entry.getValue()).clone());
178             } else if (entry.getValue() instanceof EnumSet) {
179                 this.values.put(entry.getKey(), ((EnumSet) entry.getValue()).clone());
180             } else if (entry.getValue() instanceof HashMap) {
181                 this.values.put(entry.getKey(), ((HashMap) entry.getValue()).clone());
182             } else if (entry.getValue() instanceof List) {
183                 this.values.put(entry.getKey(), ((List) entry.getValue()).stream().collect(Collectors.toList()));
184             } else {
185                 this.values.put(entry.getKey(), entry.getValue());
186             }
187         }
188         this.zones.putAll(state.zones);
189         this.simultaneousEvents.addAll(state.simultaneousEvents);
190         for (Map.Entry<UUID, CardState> entry : state.cardState.entrySet()) {
191             cardState.put(entry.getKey(), entry.getValue().copy());
192         }
193         for (Map.Entry<UUID, MageObjectAttribute> entry : state.mageObjectAttribute.entrySet()) {
194             mageObjectAttribute.put(entry.getKey(), entry.getValue().copy());
195         }
196         this.zoneChangeCounter.putAll(state.zoneChangeCounter);
197         this.copiedCards.putAll(state.copiedCards);
198         this.permanentOrderNumber = state.permanentOrderNumber;
199         this.applyEffectsCounter = state.applyEffectsCounter;
200         state.usePowerInsteadOfToughnessForDamageLethalityFilters.forEach((uuid, filter)
201                 -> this.usePowerInsteadOfToughnessForDamageLethalityFilters.put(uuid, filter.copy()));
202         this.commandersToStay.addAll(state.commandersToStay);
203         this.hasDayNight = state.hasDayNight;
204         this.isDaytime = state.isDaytime;
205         this.reverseTurnOrder = state.reverseTurnOrder;
206     }

283         this.cardState = state.cardState;
284         this.mageObjectAttribute = state.mageObjectAttribute;

290                 -> this.usePowerInsteadOfToughnessForDamageLethalityFilters.put(uuid, filter.copy()));
291         this.commandersToStay = state.commandersToStay;
292         this.hasDayNight = state.hasDayNight;
293         this.isDaytime = state.isDaytime;
294         this.reverseTurnOrder = state.reverseTurnOrder;
295     }

307     public String getValue(boolean useHidden) {
308         StringBuilder sb = threadLocalBuilder.get();
309 
310         sb.append(turn.getValue(turnNum));
311         sb.append(activePlayerId).append(priorityPlayerId).append(playerByOrderId);
312 
313         for (Player player : players.values()) {
314             sb.append("player").append(player.getLife()).append("hand");
315             if (useHidden) {
316                 sb.append(player.getHand());
317             } else {
318                 sb.append(player.getHand().size());
319             }
320             sb.append("library").append(player.getLibrary().size()).append("graveyard").append(player.getGraveyard());
321         }
322 
323         sb.append("permanents");
324         for (Permanent permanent : battlefield.getAllPermanents()) {
325             sb.append(permanent.getValue(this));
326         }
327 
328         sb.append("spells");
329         for (StackObject spell : stack) {
330             sb.append(spell.getControllerId()).append(spell.getName());
331         }
332 
333         for (ExileZone zone : exile.getExileZones()) {
334             sb.append("exile").append(zone.getName()).append(zone);
335         }
336 
337         sb.append("combat");
338         for (CombatGroup group : combat.getGroups()) {
339             sb.append(group.getDefenderId()).append(group.getAttackers()).append(group.getBlockers());
340         }
341 
342         return sb.toString();
343     }

668     public void applyEffects(Game game) {
669         applyEffectsCounter++;
670         for (Player player : players.values()) {
671             player.reset();
672         }
673         battlefield.reset(game);
674         combat.reset(game);
675         this.reset();
676         effects.apply(game);
677         combat.checkForRemoveFromCombat(game);
678     }

849     public List<GameEvent> createEventGroups(List<GameEvent> events, Game game) {
850 
851         class ZoneChangeData {
852 
853             private final Zone fromZone;
854             private final Zone toZone;
855             private final UUID sourceId;
856             private final UUID playerId;
857             Ability source;
858 
859             public ZoneChangeData(Ability source, UUID sourceId, UUID playerId, Zone fromZone, Zone toZone) {
860                 this.sourceId = sourceId;
861                 this.playerId = playerId;
862                 this.fromZone = fromZone;
863                 this.toZone = toZone;
864                 this.source = source;
865             }
866 
867             @Override
868             public int hashCode() {
869                 return (this.fromZone.ordinal() + 1) * 1
870                         + (this.toZone.ordinal() + 1) * 10
871                         + (this.sourceId != null ? this.sourceId.hashCode() : 0)
872                         + (this.source != null ? this.source.hashCode() : 0);
873             }
874 
875             @Override
876             public boolean equals(Object obj) {
877                 if (obj instanceof ZoneChangeData) {
878                     ZoneChangeData data = (ZoneChangeData) obj;
879                     return this.fromZone == data.fromZone
880                             && this.toZone == data.toZone
881                             && Objects.equals(this.sourceId, data.sourceId)
882                             && Objects.equals(this.source, data.source);
883                 }
884                 return false;
885             }
886         }
887 
888         Map<ZoneChangeData, List<GameEvent>> eventsByKey = new HashMap<>();
889         List<GameEvent> groupEvents = new LinkedList<>();
890         for (GameEvent event : events) {
891             if (event instanceof ZoneChangeEvent) {
892                 ZoneChangeEvent castEvent = (ZoneChangeEvent) event;
893                 ZoneChangeData key = new ZoneChangeData(
894                         castEvent.getSource(),
895                         castEvent.getSourceId(),
896                         castEvent.getPlayerId(),
897                         castEvent.getFromZone(),
898                         castEvent.getToZone());
899                 if (eventsByKey.containsKey(key)) {
900                     eventsByKey.get(key).add(event);
901                 } else {
902                     List<GameEvent> list = new LinkedList<>();
903                     list.add(event);
904                     eventsByKey.put(key, list);
905                 }
906             }
907         }
908         for (Map.Entry<ZoneChangeData, List<GameEvent>> entry : eventsByKey.entrySet()) {
909             Set<Card> movedCards = new LinkedHashSet<>();
910             Set<PermanentToken> movedTokens = new LinkedHashSet<>();
911             for (Iterator<GameEvent> it = entry.getValue().iterator(); it.hasNext(); ) {
912                 GameEvent event = it.next();
913                 ZoneChangeEvent castEvent = (ZoneChangeEvent) event;
914                 UUID targetId = castEvent.getTargetId();
915                 Card card = ZonesHandler.getTargetCard(game, targetId);
916                 if (card instanceof PermanentToken) {
917                     movedTokens.add((PermanentToken) card);
918                 } else if (game.getObject(targetId) instanceof PermanentToken) {
919                     movedTokens.add((PermanentToken) game.getObject(targetId));
920                 } else if (card != null) {
921                     movedCards.add(card);
922                 }
923             }
924             ZoneChangeData eventData = entry.getKey();
925             if (!movedCards.isEmpty() || !movedTokens.isEmpty()) {
926                 ZoneChangeGroupEvent event = new ZoneChangeGroupEvent(
927                         movedCards,
928                         movedTokens,
929                         eventData.sourceId,
930                         eventData.source,
931                         eventData.playerId,
932                         eventData.fromZone,
933                         eventData.toZone);
934                 groupEvents.add(event);
935             }
936         }
937         return groupEvents;
938     }

981      * Abilities that are applied to other objects or applied for a certain time
982      * span
983      *
984      * @param ability
985      * @param sourceId   - if source object can be moved between zones then you
986      *                   must set it here (each game cycle clear all source related triggers)
987      * @param attachedTo
988      */
989     public void addAbility(Ability ability, UUID sourceId, MageObject attachedTo) {
990         if (ability instanceof StaticAbility) {
991             for (UUID modeId : ability.getModes().getSelectedModes()) {
992                 Mode mode = ability.getModes().get(modeId);
993                 for (Effect effect : mode.getEffects()) {
994                     if (effect instanceof ContinuousEffect) {
995                         addEffect((ContinuousEffect) effect, sourceId, ability);
996                     }
997                 }
998             }
999         } else if (ability instanceof TriggeredAbility) {
1000             addTrigger((TriggeredAbility) ability, sourceId, attachedTo);
1001         }
1002 
1003         List<Watcher> watcherList = new ArrayList<>(ability.getWatchers()); // Workaround to prevent ConcurrentModificationException, not clear to me why this is happening now
1004         for (Watcher watcher : watcherList) {
1005             // TODO: Check that watcher for commanderAbility (where attachedTo = null) also work correctly
1006             UUID controllerId = ability.getControllerId();
1007             if (attachedTo instanceof Card) {
1008                 controllerId = ((Card) attachedTo).getOwnerId();
1009             } else if (attachedTo instanceof Controllable) {
1010                 controllerId = ((Controllable) attachedTo).getControllerId();
1011             }
1012             watcher.setControllerId(controllerId);
1013             watcher.setSourceId(attachedTo == null ? ability.getSourceId() : attachedTo.getId());
1014             watchers.add(watcher);
1015         }
1016 
1017         for (Ability sub : ability.getSubAbilities()) {
1018             addAbility(sub, sourceId, attachedTo);
1019         }

1320     /**
1321      * Make full copy of the card and all of the card's parts and put to the
1322      * game.
1323      *
1324      * @param mainCardToCopy
1325      * @param newController
1326      * @param game
1327      * @return
1328      */
1329     public Card copyCard(Card mainCardToCopy, UUID newController, Game game) {
1330         // runtime check
1331         if (!mainCardToCopy.getId().equals(mainCardToCopy.getMainCard().getId())) {
1332             // copyCard allows for main card only, if you catch it then check your targeting code
1333             throw new IllegalArgumentException("Wrong code usage. You can copy only main card.");
1334         }
1335 
1336         // must copy all card's parts
1337         // zcc and zone must be new cause zcc copy logic need card usage info here, but it haven't:
1338         // * reason 1: copied land must be played (+1 zcc), but copied spell must be put on stack and cast (+2 zcc)
1339         // * reason 2: copied card or spell can be used later as blueprint for real copies (see Epic ability)
1340         List<Card> copiedParts = new ArrayList<>();
1341 
1342         // main part (prepare must be called after other parts)
1343         Card copiedCard = mainCardToCopy.copy();
1344         copiedParts.add(copiedCard);
1345 
1346         // other parts
1347         if (copiedCard instanceof SplitCard) {
1348             // left
1349             SplitCardHalf leftOriginal = ((SplitCard) copiedCard).getLeftHalfCard();
1350             SplitCardHalf leftCopied = leftOriginal.copy();
1351             prepareCardForCopy(leftOriginal, leftCopied, newController);
1352             copiedParts.add(leftCopied);
1353             // right
1354             SplitCardHalf rightOriginal = ((SplitCard) copiedCard).getRightHalfCard();
1355             SplitCardHalf rightCopied = rightOriginal.copy();
1356             prepareCardForCopy(rightOriginal, rightCopied, newController);
1357             copiedParts.add(rightCopied);
1358             // sync parts
1359             ((SplitCard) copiedCard).setParts(leftCopied, rightCopied);
1360         } else if (copiedCard instanceof ModalDoubleFacesCard) {
1361             // left
1362             ModalDoubleFacesCardHalf leftOriginal = ((ModalDoubleFacesCard) copiedCard).getLeftHalfCard();
1363             ModalDoubleFacesCardHalf leftCopied = leftOriginal.copy();
1364             prepareCardForCopy(leftOriginal, leftCopied, newController);
1365             copiedParts.add(leftCopied);
1366             // right
1367             ModalDoubleFacesCardHalf rightOriginal = ((ModalDoubleFacesCard) copiedCard).getRightHalfCard();
1368             ModalDoubleFacesCardHalf rightCopied = rightOriginal.copy();
1369             prepareCardForCopy(rightOriginal, rightCopied, newController);
1370             copiedParts.add(rightCopied);
1371             // sync parts
1372             ((ModalDoubleFacesCard) copiedCard).setParts(leftCopied, rightCopied);
1373         } else if (copiedCard instanceof AdventureCard) {
1374             // right
1375             AdventureCardSpell rightOriginal = ((AdventureCard) copiedCard).getSpellCard();
1376             AdventureCardSpell rightCopied = rightOriginal.copy();
1377             prepareCardForCopy(rightOriginal, rightCopied, newController);
1378             copiedParts.add(rightCopied);
1379             // sync parts
1380             ((AdventureCard) copiedCard).setParts(rightCopied);
1381         }
1382 
1383         // main part prepare (must be called after other parts cause it change ids for all)
1384         prepareCardForCopy(mainCardToCopy, copiedCard, newController);
1385 
1386         // add all parts to the game
1387         copiedParts.forEach(card -> {
1388             copiedCards.put(card.getId(), card);
1389             addCard(card);
1390         });
1391 
1392         // copied cards removes from game after battlefield/stack leaves, so remember it here as workaround to fix freeze, see https://github.com/magefree/mage/issues/5437
1393         // TODO: remove that workaround after LKI will be rewritten to support cross-steps/turns data transition and support copied cards
1394         copiedParts.forEach(card -> {
1395             this.setValue(COPIED_CARD_KEY + card.getId(), card.copy());
1396         });
1397 
1398         return copiedCard;
1399     }

1402         copiedCard.assignNewId();
1403         copiedCard.setOwnerId(newController);
1404         copiedCard.setCopy(true, originalCard);
1405     }

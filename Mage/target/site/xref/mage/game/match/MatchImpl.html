76          MatchPlayer mPlayer = getPlayer(playerId);
77          if (mPlayer != null) {
78              if (!hasStarted()) {
79                  return players.remove(mPlayer);
80              }
81              mPlayer.setQuit(true);
82              synchronized (this) {
83                  this.notifyAll();
84              }
85              checkIfMatchEnds();
86              return true;
87          }
88          return false;

111     @Override
112     public boolean hasEnded() {
113         // Some workarounds to end match if for unknown reason the match was not ended regularly
114         if (getGame() == null && isDoneSideboarding()) {
115             checkIfMatchEnds();
116         }
117         if (getGame() != null && getGame().hasEnded()) {
118             for (MatchPlayer matchPlayer : players) {
119                 if (matchPlayer.getPlayer().hasQuit() && !matchPlayer.hasQuit()) {
120                     logger.warn("MatchPlayer was not set to quit matchId " + this.getId() + " - " + matchPlayer.getName());
121                     matchPlayer.setQuit(true);
122                 }
123             }
124             checkIfMatchEnds();
125         }
126         return endTime != null;
127     }

134     @Override
135     public boolean checkIfMatchEnds() {
136         int activePlayers = 0;
137         MatchPlayer matchWinner = null;
138         for (MatchPlayer matchPlayer : players) {
139             if (!matchPlayer.hasQuit()) {
140                 activePlayers++;
141                 matchWinner = matchPlayer;
142             }
143             if (matchPlayer.getWins() >= options.getWinsNeeded()) {
144                 matchPlayer.setMatchWinner(true);
145                 endTime = new Date();
146                 return true;
147             }
148         }
149         if (activePlayers < 2) {
150             if (matchWinner != null) {
151                 matchWinner.setMatchWinner(true);
152             }
153             endTime = new Date();
154             return true;
155         }
156         return false;
157     }

192     protected void initGame(Game game) throws GameException {
193         addGame(); // raises only the number
194         shufflePlayers();
195         for (MatchPlayer matchPlayer : this.players) {
196             if (!matchPlayer.hasQuit() && matchPlayer.getDeck() != null) {
197                 matchPlayer.getPlayer().init(game);
198                 game.loadCards(matchPlayer.getDeck().getCards(), matchPlayer.getPlayer().getId());
199                 game.loadCards(matchPlayer.getDeck().getSideboard(), matchPlayer.getPlayer().getId());
200                 game.addPlayer(matchPlayer.getPlayer(), matchPlayer.getDeck());
201                 // set the priority time left for the match
202                 if (games.isEmpty()) { // first game full time
203                     matchPlayer.getPlayer().setPriorityTimeLeft(options.getPriorityTime());
204                 } else {
205                     if (matchPlayer.getPriorityTimeLeft() > 0) {
206                         matchPlayer.getPlayer().setPriorityTimeLeft(matchPlayer.getPriorityTimeLeft());
207                     }
208                 }
209             } else {
210                 if (matchPlayer.getDeck() == null) {
211                     logger.error("Match: " + this.getId() + " " + matchPlayer.getName() + " has no deck.");
212                 }
213             }
214         }
215         game.setPriorityTime(options.getPriorityTime());
216     }

222     @Override
223     public void endGame() {
224         Game game = getGame();
225         for (MatchPlayer matchPlayer : this.players) {
226             Player player = game.getPlayer(matchPlayer.getPlayer().getId());
227             if (player != null) {
228                 // get the left time from player priority timer
229                 if (game.getPriorityTime() > 0) {
230                     matchPlayer.setPriorityTimeLeft(player.getPriorityTimeLeft());
231                 }
232                 if (player.hasQuit()) {
233                     matchPlayer.setQuit(true);
234                 }
235                 if (player.hasWon()) {
236                     matchPlayer.addWin();
237                 }
238             }
239         }
240         if (game.isADraw()) {
241             addDraw();
242         }
243         checkIfMatchEnds();
244         game.fireGameEndInfo();
245         gamesInfo.add(createGameInfo(game));
246     }

248     @Override
249     public GameInfo createGameInfo(Game game) {
250         StringBuilder playersInfo = new StringBuilder();
251         int counter = 0;
252 
253         for (MatchPlayer matchPlayer : getPlayers()) {
254             if (counter > 0) {
255                 playersInfo.append(" - ");
256             }
257             playersInfo.append(matchPlayer.getName());
258             counter++;
259         }
260 
261         String state;
262         String result;
263         String duelingTime = "";
264         if (game.hasEnded()) {
265             if (game.getEndTime() != null) {
266                 duelingTime = " (" + DateFormat.getDuration((game.getEndTime().getTime() - game.getStartTime().getTime()) / 1000) + ')';
267             }
268             state = "Finished" + duelingTime;
269             result = game.getWinner();
270         } else {
271             if (game.getStartTime() != null) {
272                 duelingTime = " (" + DateFormat.getDuration((new Date().getTime() - game.getStartTime().getTime()) / 1000) + ')';
273             }
274             state = "Dueling" + duelingTime;
275             result = "";
276         }
277         return new GameInfo(0, this.getId(), game.getId(), state, result, playersInfo.toString(), tableId);
278     }

315     @Override
316     public void sideboard() {
317         for (MatchPlayer player : this.players) {
318             if (!player.hasQuit()) {
319                 if (player.getDeck() != null) {
320                     player.setSideboarding();
321                     player.getPlayer().sideboard(this, player.getDeck());
322                 } else {
323                     logger.error("Player " + player.getName() + " has no deck: " + player.getPlayer().getId());
324                 }
325             }
326         }
327         synchronized (this) {
328             while (!isDoneSideboarding()) {
329                 try {
330                     this.wait();
331                 } catch (InterruptedException ex) {
332                 }
333             }
334         }
335     }

371     public void submitDeck(UUID playerId, Deck deck) {
372         MatchPlayer player = getPlayer(playerId);
373         if (player != null) {
374             // make sure the deck name (needed for Tiny Leaders) won't get lost by sideboarding
375             deck.setName(player.getDeck().getName());
376             deck.setDeckHashCode(player.getDeck().getDeckHashCode());
377             player.submitDeck(deck);
378         }
379         synchronized (this) {
380             this.notifyAll();
381         }

384     @Override
385     public boolean updateDeck(UUID playerId, Deck deck) {
386         boolean validDeck = true;
387         MatchPlayer player = getPlayer(playerId);
388         if (player != null) {
389             // Check if the cards included in the deck are the same as in the original deck
390             validDeck = (player.getDeck().getDeckCompleteHashCode() == deck.getDeckCompleteHashCode());
391             if (validDeck == false) {
392                 // clear the deck so the player cheating looses the game
393                 deck.getCards().clear();
394                 deck.getSideboard().clear();
395             }
396             player.updateDeck(deck);
397         }
398         return validDeck;
399     }

401     protected String createGameStartMessage() {
402         StringBuilder sb = new StringBuilder();
403         sb.append("<br/><b>Match score:</b><br/>");
404         for (MatchPlayer mp : this.getPlayers()) {
405             sb.append("   ").append(mp.getPlayer().getLogName());
406             sb.append(" - ").append(mp.getWins()).append(mp.getWins() == 1 ? " win" : " wins");
407             if (mp.hasQuit()) {

410             sb.append("<br/>");
411             if (mp.getDeck() != null) {
412                 sb.append("DeckHash: ").append(mp.getDeck().getDeckHashCode()).append("<br/>");
413             }
414         }
415         if (getDraws() > 0) {
416             sb.append("   Draws: ").append(getDraws()).append("<br/>");
417         }
418         if (options.getRange() != null) {

420         }

487     @Override
488     public MatchProto toProto() {
489         MatchProto.Builder builder = MatchProto.newBuilder()
490                 .setName(this.getName())
491                 .setGameType(this.getOptions().getGameType())
492                 .setDeckType(this.getOptions().getDeckType())
493                 .setGames(this.getNumGames())
494                 .setDraws(this.getDraws())
495                 .setMatchOptions(this.getOptions().toProto())
496                 .setEndTimeMs((this.getEndTime() != null ? this.getEndTime() : new Date()).getTime());
497         for (MatchPlayer matchPlayer : this.getPlayers()) {
498             MatchQuitStatus status = !matchPlayer.hasQuit() ? MatchQuitStatus.NO_MATCH_QUIT
499                     : matchPlayer.getPlayer().hasTimerTimeout() ? MatchQuitStatus.TIMER_TIMEOUT
500                     : matchPlayer.getPlayer().hasIdleTimeout() ? MatchQuitStatus.IDLE_TIMEOUT
501                     : MatchQuitStatus.QUIT;
502             builder.addPlayersBuilder()
503                     .setName(matchPlayer.getName())
504                     .setHuman(matchPlayer.getPlayer().isHuman())
505                     .setQuit(status)
506                     .setWins(matchPlayer.getWins());
507         }
508         return builder.build();
509     }

7   import java.util.*;
8   import java.util.stream.Collectors;

10  /**
11   * @author TheElk801
12   */
13  public class DefenderAttackedEvent extends GameEvent {
14  
15      private final Set<MageObjectReference> morSet = new HashSet<>();
16  
17      public DefenderAttackedEvent(UUID targetId, UUID playerId) {
18          super(EventType.DEFENDER_ATTACKED, targetId, null, playerId);
19  
20      }
21  
22      public static void makeAddEvents(Map<UUID, Set<MageObjectReference>> morMapSet, UUID attackingPlayerId, Game game) {
23          for (Map.Entry<UUID, Set<MageObjectReference>> entry : morMapSet.entrySet()) {
24              DefenderAttackedEvent event = new DefenderAttackedEvent(entry.getKey(), attackingPlayerId);
25              event.morSet.addAll(entry.getValue());
26              game.addSimultaneousEvent(event);
27          }
28      }
29  
30      public List<Permanent> getAttackers(Game game) {
31          return morSet
32                  .stream()
33                  .map(mor -> mor.getPermanent(game))
34                  .filter(Objects::nonNull)
35                  .collect(Collectors.toList());
36      }
37  }

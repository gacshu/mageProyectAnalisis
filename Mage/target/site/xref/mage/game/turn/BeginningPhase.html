3   package mage.game.turn;
4   
5   import java.util.UUID;
6   import mage.constants.TurnPhase;
7   import mage.game.Game;
8   import mage.game.events.GameEvent.EventType;
9   
10  /**
11   *
12   * @author BetaSteward_at_googlemail.com
13   */
14  public class BeginningPhase extends Phase {
15  
16      public BeginningPhase() {
17          this.type = TurnPhase.BEGINNING;
18          this.event = EventType.BEGINNING_PHASE;
19          this.preEvent = EventType.BEGINNING_PHASE_PRE;
20          this.postEvent = EventType.BEGINNING_PHASE_POST;
21          this.steps.add(new UntapStep());
22          this.steps.add(new UpkeepStep());
23          this.steps.add(new DrawStep());
24      }
25  
26      @Override
27      public boolean beginPhase(Game game, UUID activePlayerId) {
28          game.getBattlefield().beginningOfTurn(game);
29          return super.beginPhase(game, activePlayerId);
30      }
31  
32  
33      public BeginningPhase(final BeginningPhase phase) {
34          super(phase);
35      }
36  
37      @Override
38      public BeginningPhase copy() {
39          return new BeginningPhase(this);
40      }
41  
42  }

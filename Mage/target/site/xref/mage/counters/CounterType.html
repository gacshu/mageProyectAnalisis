253     /**
254      * Create instance of counter type with defined amount of counters of the
255      * given type.
256      *
257      * @param amount amount of counters of the given type.
258      * @return
259      */
260     public Counter createInstance(int amount) {
261         switch (this) {
262             case P0P1:
263                 return new BoostCounter(0, 1, amount);
264             case P1P0:
265                 return new BoostCounter(1, 0, amount);
266             case P1P1:
267                 return new BoostCounter(1, 1, amount);
268             case P1P2:
269                 return new BoostCounter(1, 2, amount);
270             case P2P2:
271                 return new BoostCounter(2, 2, amount);
272             case M0M1:
273                 return new BoostCounter(0, -1, amount);
274             case M0M2:
275                 return new BoostCounter(0, -2, amount);
276             case M1M0:
277                 return new BoostCounter(-1, 0, amount);
278             case M1M1:
279                 return new BoostCounter(-1, -1, amount);
280             case M2M1:
281                 return new BoostCounter(-2, -1, amount);
282             case M2M2:
283                 return new BoostCounter(-2, -2, amount);
284             case DEATHTOUCH:
285                 return new AbilityCounter(DeathtouchAbility.getInstance(), amount);
286             case DOUBLE_STRIKE:
287                 return new AbilityCounter(DoubleStrikeAbility.getInstance(), amount);
288             case FIRST_STRIKE:
289                 return new AbilityCounter(FirstStrikeAbility.getInstance(), amount);
290             case FLYING:
291                 return new AbilityCounter(FlyingAbility.getInstance(), amount);
292             case HEXPROOF:
293                 return new AbilityCounter(HexproofAbility.getInstance(), amount);
294             case INDESTRUCTIBLE:
295                 return new AbilityCounter(IndestructibleAbility.getInstance(), amount);
296             case LIFELINK:
297                 return new AbilityCounter(LifelinkAbility.getInstance(), amount);
298             case MENACE:
299                 return new AbilityCounter(new MenaceAbility(), amount);
300             case REACH:
301                 return new AbilityCounter(ReachAbility.getInstance(), amount);
302             case TRAMPLE:
303                 return new AbilityCounter(TrampleAbility.getInstance(), amount);
304             case VIGILANCE:
305                 return new AbilityCounter(VigilanceAbility.getInstance(), amount);
306             default:
307                 return new Counter(name, amount);
308         }
309     }

323     }

325     public CounterPredicate getPredicate() {
326         return predicate;
327     }
328 
329     private static class CounterPredicate implements Predicate<Card> {
330 
331         private final CounterType counter;
332 
333         private CounterPredicate(CounterType counter) {
334             this.counter = counter;
335         }
336 
337         @Override
338         public boolean apply(Card input, Game game) {
339             if (counter == null) {
340                 return !input.getCounters(game).keySet().isEmpty();
341             } else {
342                 return input.getCounters(game).containsKey(counter);
343             }
344         }
345 
346         @Override
347         public String toString() {
348             return "CounterType(" + counter.getName() + ')';
349         }
350     }

3   import mage.abilities.Ability;
4   import mage.counters.CounterType;
5   import mage.filter.FilterPermanent;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   import mage.target.TargetPermanent;
9   
10  /**
11   *
12   * @author TheElk801
13   */
14  public enum VerseCounterAdjuster implements TargetAdjuster {
15      instance;
16  
17      @Override
18      public void adjustTargets(Ability ability, Game game) {
19          Permanent sourcePermanent = game.getPermanentOrLKIBattlefield(ability.getSourceId());
20          if (sourcePermanent != null) {
21              int xValue = sourcePermanent.getCounters(game).getCount(CounterType.VERSE);
22              FilterPermanent permanentFilter = ((TargetPermanent) ability.getTargets().get(0)).getFilter();
23              ability.getTargets().clear();
24              ability.addTarget(new TargetPermanent(0, xValue, permanentFilter, false));
25          }
26      }
27  }

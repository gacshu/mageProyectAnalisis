2   package mage.target.common;
3   
4   import mage.constants.Zone;
5   import mage.abilities.Ability;
6   import mage.cards.Card;
7   import mage.filter.FilterCard;
8   import mage.game.Game;
9   import mage.target.TargetCard;
10  
11  import java.util.UUID;
12  import mage.filter.StaticFilters;
13  import mage.filter.predicate.card.OwnerIdPredicate;
14  
15  /**
16   *
17   * @author BetaSteward_at_googlemail.com
18   */
19  public class TargetDiscard extends TargetCard {
20  
21      private final UUID playerId;
22  
23      public TargetDiscard(UUID playerId) {
24          this(1, 1, StaticFilters.FILTER_CARD, playerId);
25      }
26  
27      public TargetDiscard(FilterCard filter, UUID playerId) {
28          this(1, 1, filter, playerId);
29      }
30  
31      public TargetDiscard(int numTargets, FilterCard filter, UUID playerId) {
32          this(numTargets, numTargets, filter, playerId);
33      }
34  
35      public TargetDiscard(int minNumTargets, int maxNumTargets, FilterCard filter, UUID playerId) {
36          super(minNumTargets, maxNumTargets, Zone.HAND, filter.copy());
37          this.filter.add(new OwnerIdPredicate(playerId));
38          this.playerId = playerId;
39          this.targetName = this.filter.getMessage() + " to discard";
40      }
41  
42      public TargetDiscard(final TargetDiscard target) {
43          super(target);
44          this.playerId = target.playerId;
45      }
46  
47      @Override
48      public boolean canTarget(UUID id, Ability source, Game game) {
49          Card card = game.getPlayer(playerId).getHand().get(id, game);
50          return filter.match(card, source.getControllerId(), game);
51      }
52  
53      @Override
54      public TargetDiscard copy() {
55          return new TargetDiscard(this);
56      }
57  
58  }

18      private boolean emptyDraw;
19      private final Deque<UUID> library = new ArrayDeque<>();
20      private final UUID playerId;

22      public Library(UUID playerId) {
23          this.playerId = playerId;
24      }
25  
26      public Library(final Library lib) {
27          this.emptyDraw = lib.emptyDraw;
28          this.playerId = lib.playerId;
29          for (UUID id : lib.library) {
30              this.library.addLast(id);
31          }
32      }

34      /**
35       * Don't use this directly. Use <player.shuffleLibrary(game)> instead.
36       */
37      public void shuffle() {
38          UUID[] shuffled = library.toArray(new UUID[0]);
39          for (int n = shuffled.length - 1; n > 0; n--) {
40              int r = RandomUtil.nextInt(n + 1);
41              UUID temp = shuffled[n];
42              shuffled[n] = shuffled[r];
43              shuffled[r] = temp;
44          }
45          library.clear();
46          library.addAll(Arrays.asList(shuffled));
47      }

49      /**
50       * Removes the top card of the Library and returns it
51       *
52       * @param game
53       * @return Card
54       * @see Card
55       */
56      public Card removeFromTop(Game game) {
57          UUID cardId = library.pollFirst();
58          Card card = game.getCard(cardId);
59          if (card == null) {
60              emptyDraw = true;
61          }
62          return card;
63      }

65      /**
66       * Removes the bottom card of the Library and returns it
67       *
68       * @param game
69       * @return Card
70       * @see Card
71       */
72      public Card removeFromBottom(Game game) {
73          UUID cardId = library.pollLast();
74          Card card = game.getCard(cardId);
75          if (card == null) {
76              emptyDraw = true;
77          }
78          return card;
79      }

81      /**
82       * Returns the top card of the Library without removing it
83       *
84       * @param game
85       * @return Card
86       * @see Card
87       */
88      public Card getFromTop(Game game) {
89          return game.getCard(library.peekFirst());
90      }

92      /**
93       * Returns the bottommost card of the Library without removing it
94       *
95       * @param game
96       * @return Card
97       * @see Card
98       */
99      public Card getFromBottom(Game game) {
100         return game.getCard(library.pollLast());

103     public void putOnTop(Card card, Game game) {
104         if (card.isOwnedBy(playerId)) {
105             card.setZone(Zone.LIBRARY, game);
106             library.addFirst(card.getId());
107         } else {
108             game.getPlayer(card.getOwnerId()).getLibrary().putOnTop(card, game);
109         }
110     }
111 
112     public void putCardToTopXPos(Card card, int pos, Game game) {
113         if (card != null && pos > -1) {
114             LinkedList<Card> save = new LinkedList<>();
115             int idx = 1;
116             while (hasCards() && idx < pos) {
117                 idx++;
118                 save.add(removeFromTop(game));
119             }
120             putOnTop(card, game);
121             while (!save.isEmpty()) {
122                 putOnTop(save.removeLast(), game);
123             }
124         }
125     }
126 
127     public void putOnBottom(Card card, Game game) {
128         if (card.isOwnedBy(playerId)) {
129             card.setZone(Zone.LIBRARY, game);
130             library.remove(card.getId());
131             library.add(card.getId());
132         } else {
133             game.getPlayer(card.getOwnerId()).getLibrary().putOnBottom(card, game);
134         }
135     }

160     /**
161      * Returns the cards of the library in a list ordered from top to buttom
162      *
163      * @param game
164      * @return
165      */
166     public List<Card> getCards(Game game) {
167         return library.stream().map(game::getCard).filter(Objects::nonNull).collect(Collectors.toList());
168     }
169 
170     public Set<Card> getTopCards(Game game, int amount) {
171         Set<Card> cards = new LinkedHashSet<>();
172         Iterator<UUID> it = library.iterator();
173         int count = 0;
174         while (it.hasNext() && count < amount) {
175             UUID cardId = it.next();
176             Card card = game.getCard(cardId);
177             if (card != null) {
178                 cards.add(card);
179                 ++count;
180             }
181         }
182         return cards;
183     }
184 
185     public Collection<Card> getUniqueCards(Game game) {
186         Map<String, Card> cards = new HashMap<>();
187         for (UUID cardId : library) {
188             Card card = game.getCard(cardId);
189             if (card != null) {
190                 cards.putIfAbsent(card.getName(), card);
191             }
192         }
193         return cards.values();
194     }
195 
196     public int count(FilterCard filter, Game game) {
197         return (int) library.stream().filter(cardId -> filter.match(game.getCard(cardId), game)).count();
198     }

204     public void addAll(Set<Card> cards, Game game) {
205         for (Card card : cards) {
206             card.setZone(Zone.LIBRARY, game);
207             library.add(card.getId());
208         }
209     }

211     public Card getCard(UUID cardId, Game game) {
212         for (UUID card : library) {
213             if (card.equals(cardId)) {
214                 return game.getCard(card);
215             }
216         }
217         return null;
218     }
219 
220     public Card remove(UUID cardId, Game game) {
221         Iterator<UUID> it = library.iterator();
222         while (it.hasNext()) {
223             UUID card = it.next();
224             if (card.equals(cardId)) {
225                 it.remove();
226                 return game.getCard(card);
227             }
228         }
229         return null;
230     }

232     public boolean hasCards() {
233         return size() > 0;
234     }

240     /**
241      * Tests only -- find card position in library
242      * @param cardId
243      * @return 
244      */
245     public int getCardPosition(UUID cardId) {
246         UUID[] list = library.toArray(new UUID[0]);
247         for (int i = 0; i < list.length; i++) {
248             if (list[i].equals(cardId)) {
249                 return i;
250             }
251         }
252         return -1;
253     }

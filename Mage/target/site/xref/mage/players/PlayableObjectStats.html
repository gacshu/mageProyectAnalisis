16  /**
17   * Contains stats with playable abilities for one object
18   *
19   * @author JayDi85
20   */
21  public class PlayableObjectStats implements Serializable, Copyable<PlayableObjectStats> {
22  
23      // map: ability.id -> ability.rules
24      List<PlayableObjectRecord> basicManaAbilities = new ArrayList<>();
25      List<PlayableObjectRecord> basicPlayAbilities = new ArrayList<>();
26      List<PlayableObjectRecord> basicCastAbilities = new ArrayList<>();
27      List<PlayableObjectRecord> other = new ArrayList<>();
28  
29      public PlayableObjectStats() {
30      }
31  
32      public PlayableObjectStats(List<ActivatedAbility> activatedAbilities) {
33          load(activatedAbilities);
34      }
35  
36      public void load(List<ActivatedAbility> activatedAbilities) {
37          this.basicManaAbilities.clear();
38          this.basicPlayAbilities.clear();
39          this.basicCastAbilities.clear();
40          this.other.clear();
41  
42          // split abilities to types (it allows to enable or disable playable icon)
43          for (ActivatedAbility ability : activatedAbilities) {
44              List<PlayableObjectRecord> dest;
45              if (ability instanceof BasicManaAbility) {
46                  dest = this.basicManaAbilities;
47              } else if (ability instanceof PlayLandAbility) {
48                  dest = this.basicPlayAbilities;
49              } else if (ability instanceof SpellAbility && (((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.BASE)) {
50                  dest = this.basicCastAbilities;
51              } else {
52                  dest = this.other;
53              }
54  
55              // collect info about abilities for card icons popup, must be simple online text (html symbols are possible)
56              // some long html tags can be miss (example: ability extra hint) -- that's ok
57              String shortInfo = ability.toString();
58              if (shortInfo.length() > 50) {
59                  shortInfo = shortInfo.substring(0, 50 - 1) + "...";
60              }
61              shortInfo = shortInfo.replace("<br>", " ");
62              shortInfo = shortInfo.replace("\n", " ");
63              dest.add(new PlayableObjectRecord(ability.getId(), shortInfo));
64          }
65      }
66  
67      public PlayableObjectStats(final PlayableObjectStats source) {
68          for (PlayableObjectRecord rec : source.basicManaAbilities) {
69              this.basicManaAbilities.add(rec.copy());
70          }
71          for (PlayableObjectRecord rec : source.basicPlayAbilities) {
72              this.basicPlayAbilities.add(rec.copy());
73          }
74          for (PlayableObjectRecord rec : source.basicCastAbilities) {
75              this.basicCastAbilities.add(rec.copy());
76          }
77          for (PlayableObjectRecord rec : source.other) {
78              this.other.add(rec.copy());
79          }
80      }
81  
82      @Override
83      public PlayableObjectStats copy() {
84          return new PlayableObjectStats(this);
85      }
86  
87      public int getPlayableAmount() {
88          return this.basicManaAbilities.size()
89                  + this.basicPlayAbilities.size()
90                  + this.basicCastAbilities.size()
91                  + this.other.size();
92      }
93  
94      public List<String> getPlayableAbilities() {
95          List<String> all = new ArrayList<>();
96          all.addAll(this.basicManaAbilities.stream().map(PlayableObjectRecord::getValue).sorted().collect(Collectors.toList()));
97          all.addAll(this.basicPlayAbilities.stream().map(PlayableObjectRecord::getValue).sorted().collect(Collectors.toList()));
98          all.addAll(this.basicCastAbilities.stream().map(PlayableObjectRecord::getValue).sorted().collect(Collectors.toList()));
99          all.addAll(this.other.stream().map(PlayableObjectRecord::getValue).sorted().collect(Collectors.toList()));
100         return all;
101     }
102 
103     public int getPlayableImportantAmount() {
104         // return only important abilities (e.g. show it as card icons)
105         return this.other.size();
106     }
107 }

109 class PlayableObjectRecord implements Serializable, Copyable<PlayableObjectRecord> {
110 
111     UUID id;
112     String value;
113 
114     public PlayableObjectRecord(UUID id, String value) {
115         this.id = id;
116         this.value = value;
117     }
118 
119     private PlayableObjectRecord(final PlayableObjectRecord record) {
120         this.id = record.id;
121         this.value = record.value;
122     }
123 
124     public UUID getId() {
125         return id;
126     }
127 
128     public String getValue() {
129         return value;
130     }
131 
132     @Override
133     public PlayableObjectRecord copy() {
134         return new PlayableObjectRecord(this);
135     }
136 }

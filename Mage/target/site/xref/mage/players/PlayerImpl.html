496     @Override
497     public void updateRange(Game game) {
498         // 20100423 - 801.2c
499         // 801.2c The particular players within each player’s range of influence are determined as each turn begins.
500         // BUT it also uses before game start to fill game and card data in starting game events
501         inRange.clear();
502         inRange.add(this.playerId);
503         inRange.addAll(getAllNearPlayers(game, true));
504         inRange.addAll(getAllNearPlayers(game, false));
505     }

507     private Set<UUID> getAllNearPlayers(Game game, boolean needPrevious) {
508         // find all near players (search from current player position)
509         Set<UUID> foundedList = new HashSet<>();
510         PlayerList players = game.getState().getPlayerList(this.playerId);
511         int needAmount = this.getRange().getRange(); // distance to search (0 - ALL range)
512         int foundedAmount = 0;
513         while (needAmount == 0 || foundedAmount < needAmount) {
514             Player foundedPlayer = needPrevious ? players.getPrevious(game) : players.getNext(game, false);
515 
516             // PlayerList is inifine, so stops on repeats
517             if (foundedPlayer == null || foundedPlayer.getId().equals(this.playerId) || foundedList.contains(foundedPlayer.getId())) {
518                 break;
519             }
520             // skip leaved player (no needs cause next/previous code already checks it)
521 
522             foundedList.add(foundedPlayer.getId());
523             foundedAmount++;
524         }
525         return foundedList;
526     }

545     @Override
546     public void controlPlayersTurn(Game game, UUID playerId) {
547         Player player = game.getPlayer(playerId);
548         player.setTurnControlledBy(this.getId());
549         game.informPlayers(getLogName() + " controls the turn of " + player.getLogName());
550         if (!playerId.equals(this.getId())) {
551             this.playersUnderYourControl.add(playerId);
552             if (!player.hasLeft() && !player.hasLost()) {
553                 player.setGameUnderYourControl(false);
554             }
555             DelayedTriggeredAbility ability = new AtTheEndOfTurnStepPostDelayedTriggeredAbility(
556                     new LoseControlOnOtherPlayersControllerEffect(this.getLogName(), player.getLogName()));
557             ability.setSourceId(getId());
558             ability.setControllerId(getId());
559             game.addDelayedTriggeredAbility(ability, null);
560         }
561     }

600     @Override
601     public void setGameUnderYourControl(boolean value, boolean fullRestore) {
602         this.isGameUnderControl = value;
603         if (isGameUnderControl) {
604             if (fullRestore) {
605                 this.turnControllers.clear();
606                 this.turnController = getId();
607             } else {
608                 if (turnControllers.size() > 0) {
609                     this.turnControllers.remove(turnControllers.size() - 1);
610                 }
611                 if (turnControllers.isEmpty()) {
612                     this.turnController = getId();
613                 } else {
614                     this.turnController = turnControllers.get(turnControllers.size() - 1);
615                     isGameUnderControl = false;
616                 }
617             }
618         }

629         if (this.hasLost() || this.hasLeft()) {

631         }

639                     && this.hasOpponent(sourceControllerId, game)

642                     .filter(HexproofBaseAbility.class::isInstance)

787     private boolean doDiscard(Card card, Ability source, Game game, boolean payForCost, boolean fireFinalEvent) {
788         //20100716 - 701.7
789         /* 701.7. Discard #
790          701.7a To discard a card, move it from its owners hand to that players graveyard.
791          701.7b By default, effects that cause a player to discard a card allow the affected
792          player to choose which card to discard. Some effects, however, require a random
793          discard or allow another player to choose which card is discarded.
794          701.7c If a card is discarded, but an effect causes it to be put into a hidden zone
795          instead of into its owners graveyard without being revealed, all values of that
796          cards characteristics are considered to be undefined.
797          TODO:
798          If a card is discarded this way to pay a cost that specifies a characteristic
799          about the discarded card, that cost payment is illegal; the game returns to
800          the moment before the cost was paid (see rule 717, "Handling Illegal Actions").
801          */
802         if (card == null) {
803             return false;
804         }
805         GameEvent gameEvent = GameEvent.getEvent(GameEvent.EventType.DISCARD_CARD, card.getId(), source, playerId);
806         gameEvent.setFlag(!payForCost); // event from effect (1) or from cost (0)
807         if (game.replaceEvent(gameEvent, source)) {
808             return false;
809         }
810         // write info to game log first so game log infos from triggered or replacement effects follow in the game log
811         if (!game.isSimulation()) {
812             game.informPlayers(getLogName() + " discards " + card.getLogName() + CardUtil.getSourceLogName(game, source));
813         }
814         /* If a card is discarded while Rest in Peace is on the battlefield, abilities that function
815          * when a card is discarded (such as madness) still work, even though that card never reaches
816          * a graveyard. In addition, spells or abilities that check the characteristics of a discarded
817          * card (such as Chandra Ablaze's first ability) can find that card in exile. */
818         card.moveToZone(Zone.GRAVEYARD, source, game, false);
819         // So discard is also successful if card is moved to another zone by replacement effect!
820         game.fireEvent(GameEvent.getEvent(GameEvent.EventType.DISCARDED_CARD, card.getId(), source, playerId));
821 
822         if (fireFinalEvent) {
823             game.fireEvent(new DiscardedCardsEvent(source, playerId, 1, new CardsImpl(card)));
824         }
825         return true;
826     }

1148      * @return

1151     public boolean cast(SpellAbility originalAbility, Game game, boolean noMana, ApprovingObject approvingObject) {
1152         if (game == null || originalAbility == null) {
1153             return false;
1154         }
1155 
1156         // Use ability copy to avoid problems with targets and costs on recast (issue https://github.com/magefree/mage/issues/5189).
1157         SpellAbility ability = originalAbility.copy();
1158         ability.setControllerId(getId());
1159         ability.setSourceObjectZoneChangeCounter(game.getState().getZoneChangeCounter(ability.getSourceId()));
1160 
1161         //20091005 - 601.2a
1162         if (ability.getSourceId() == null) {
1163             logger.error("Ability without sourceId turn " + game.getTurnNum() + ". Ability: " + ability.getRule());
1164             return false;
1165         }
1166         Card card = game.getCard(ability.getSourceId());
1167         if (card != null) {
1168             Zone fromZone = game.getState().getZone(card.getMainCard().getId());
1169             GameEvent castEvent = GameEvent.getEvent(GameEvent.EventType.CAST_SPELL,
1170                     ability.getId(), ability, playerId, approvingObject);
1171             castEvent.setZone(fromZone);
1172             if (!game.replaceEvent(castEvent, ability)) {
1173                 int bookmark = game.bookmarkState();
1174                 setStoredBookmark(bookmark); // move global bookmark to current state (if you activated mana before then you can't rollback it)
1175                 card.cast(game, fromZone, ability, playerId);
1176                 Spell spell = game.getStack().getSpell(ability.getId());
1177                 if (spell == null) {
1178                     logger.error("Got no spell from stack. ability: " + ability.getRule());
1179                     return false;
1180                 }
1181                 if (card.isCopy()) {
1182                     spell.setCopy(true, null);
1183                 }
1184                 // Update the zcc to the stack
1185                 ability.setSourceObjectZoneChangeCounter(game.getState().getZoneChangeCounter(ability.getSourceId()));
1186 
1187                 // ALTERNATIVE COST from dynamic effects
1188                 // some effects set sourceId to cast without paying mana costs or other costs
1189                 if (getCastSourceIdWithAlternateMana().contains(ability.getSourceId())) {
1190                     Ability spellAbility = spell.getSpellAbility();
1191                     ManaCosts alternateCosts = getCastSourceIdManaCosts().get(ability.getSourceId());
1192                     Costs<Cost> costs = getCastSourceIdCosts().get(ability.getSourceId());
1193                     if (alternateCosts == null) {
1194                         noMana = true;
1195                     } else {
1196                         spellAbility.getManaCosts().clear();
1197                         spellAbility.getManaCostsToPay().clear();
1198                         spellAbility.getManaCosts().add(alternateCosts.copy());
1199                         spellAbility.getManaCostsToPay().add(alternateCosts.copy());
1200                     }
1201                     spellAbility.getCosts().clear();
1202                     if (costs != null) {
1203                         spellAbility.getCosts().addAll(costs);
1204                     }
1205                 }
1206                 clearCastSourceIdManaCosts(); // TODO: test multiple alternative cost for different cards as same time
1207 
1208                 castEvent = GameEvent.getEvent(GameEvent.EventType.CAST_SPELL,
1209                         spell.getSpellAbility().getId(), spell.getSpellAbility(), playerId, approvingObject);
1210                 castEvent.setZone(fromZone);
1211                 game.fireEvent(castEvent);
1212                 if (spell.activate(game, noMana)) {
1213                     GameEvent castedEvent = GameEvent.getEvent(GameEvent.EventType.SPELL_CAST,
1214                             spell.getSpellAbility().getId(), spell.getSpellAbility(), playerId, approvingObject);
1215                     castedEvent.setZone(fromZone);
1216                     game.fireEvent(castedEvent);
1217                     if (!game.isSimulation()) {
1218                         game.informPlayers(getLogName() + spell.getActivatedMessage(game));
1219                     }
1220                     game.removeBookmark(bookmark);
1221                     resetStoredBookmark(game);
1222                     return true;
1223                 }
1224                 restoreState(bookmark, ability.getRule(), game);
1225             }

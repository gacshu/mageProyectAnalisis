1   package mage.abilities.meta;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.TriggeredAbility;
5   import mage.abilities.TriggeredAbilityImpl;
6   import mage.abilities.effects.Effect;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.util.CardUtil;
11  import mage.watchers.Watcher;
12  
13  import java.util.ArrayList;
14  import java.util.Collections;
15  import java.util.List;
16  import java.util.UUID;
17  import java.util.stream.Collectors;
18  
19  /**
20   * A triggered ability that combines several others and triggers whenever one or
21   * more of them would. The abilities passed in should have null as their effect,
22   * and should have their own targets set if necessary. All other information
23   * will be passed in from changes to this Ability. Note: this does NOT work with
24   * abilities that have intervening if clauses.
25   *
26   * @author noahg
27   */
28  public class OrTriggeredAbility extends TriggeredAbilityImpl {
29  
30      private final String ruleTrigger;
31      private final List<TriggeredAbility> triggeredAbilities = new ArrayList<>();
32  
33      public OrTriggeredAbility(Zone zone, Effect effect, TriggeredAbility... abilities) {
34          this(zone, effect, false, null, abilities);
35      }
36  
37      public OrTriggeredAbility(Zone zone, Effect effect, boolean optional, String ruleTrigger, TriggeredAbility... abilities) {
38          super(zone, effect, optional);
39          this.ruleTrigger = ruleTrigger;
40          Collections.addAll(this.triggeredAbilities, abilities);
41          for (TriggeredAbility ability : triggeredAbilities) {
42              //Remove useless data
43              ability.getEffects().clear();
44          }
45          setTriggerPhrase(generateTriggerPhrase());
46      }
47  
48      public OrTriggeredAbility(OrTriggeredAbility ability) {
49          super(ability);
50          this.ruleTrigger = ability.ruleTrigger;
51          for (TriggeredAbility triggeredAbility : ability.triggeredAbilities) {
52              this.triggeredAbilities.add(triggeredAbility.copy());
53          }
54      }
55  
56      @Override
57      public boolean checkEventType(GameEvent event, Game game) {
58          for (TriggeredAbility ability : triggeredAbilities) {
59              if (ability.checkEventType(event, game)) {
60                  return true;
61              }
62          }
63          return false;
64      }
65  
66      @Override
67      public boolean checkTrigger(GameEvent event, Game game) {
68          boolean toRet = false;
69          for (TriggeredAbility ability : triggeredAbilities) {
70              if (ability.checkEventType(event, game) && ability.checkTrigger(event, game)) {
71                  toRet = true;
72              }
73          }
74          return toRet;
75      }
76  
77      @Override
78      public OrTriggeredAbility copy() {
79          return new OrTriggeredAbility(this);
80      }
81  
82      private String generateTriggerPhrase() {
83          if (ruleTrigger != null && !ruleTrigger.isEmpty()) {
84              return ruleTrigger;
85          }
86          return triggeredAbilities
87                  .stream()
88                  .map(Ability::getRule)
89                  .map(CardUtil::getTextWithFirstCharLowerCase)
90                  .map(s -> s.substring(0, s.length() - 2))
91                  .collect(Collectors.joining(" or ")) + ", ";
92      }
93  
94      @Override
95      public void setControllerId(UUID controllerId) {
96          super.setControllerId(controllerId);
97          for (TriggeredAbility ability : triggeredAbilities) {
98              ability.setControllerId(controllerId);
99          }
100     }
101 
102     @Override
103     public void setSourceId(UUID sourceId) {
104         super.setSourceId(sourceId);
105         for (TriggeredAbility ability : triggeredAbilities) {
106             ability.setSourceId(sourceId);
107         }
108     }
109 
110     @Override
111     public void addWatcher(Watcher watcher) {
112         super.addWatcher(watcher);
113         for (TriggeredAbility ability : triggeredAbilities) {
114             ability.addWatcher(watcher);
115         }
116     }
117 }

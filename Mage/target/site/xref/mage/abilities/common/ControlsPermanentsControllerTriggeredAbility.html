2   package mage.abilities.common;
3   
4   import mage.constants.ComparisonType;
5   import mage.abilities.StateTriggeredAbility;
6   import mage.abilities.effects.Effect;
7   import mage.constants.Zone;
8   import mage.filter.FilterPermanent;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  
12  /**
13   * State triggered ability that triggers if the ability controller controlls the
14   * defined number of permanents.
15   *
16   * @author LevelX2
17   */
18  public class ControlsPermanentsControllerTriggeredAbility extends StateTriggeredAbility {
19  
20      protected final FilterPermanent filter;
21      protected final ComparisonType type;
22      protected final int value;
23  
24      public ControlsPermanentsControllerTriggeredAbility(FilterPermanent filter, Effect effect){
25          this(filter, ComparisonType.MORE_THAN, 0, effect);
26      }
27  
28      public ControlsPermanentsControllerTriggeredAbility(FilterPermanent filter, ComparisonType type, int value, Effect effect) {
29          super(Zone.BATTLEFIELD, effect);
30          this.filter = filter;
31          this.value = value;
32          this.type = type;
33          setTriggerPhrase("When you control " + filter.getMessage() + ", " );
34      }
35  
36      public ControlsPermanentsControllerTriggeredAbility(final ControlsPermanentsControllerTriggeredAbility ability) {
37          super(ability);
38          this.filter = ability.filter;
39          this.type = ability.type;
40          this.value = ability.value;
41      }
42  
43      @Override
44      public ControlsPermanentsControllerTriggeredAbility copy() {
45          return new ControlsPermanentsControllerTriggeredAbility(this);
46      }
47  
48      @Override
49      public boolean checkTrigger(GameEvent event, Game game) {
50          int inputValue = game.getBattlefield().countAll(filter, getControllerId(), game);
51          return ComparisonType.compare(inputValue, type, value);
52      }
53  }

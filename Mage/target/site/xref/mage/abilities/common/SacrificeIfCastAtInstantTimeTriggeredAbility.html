3   package mage.abilities.common;
4   
5   import mage.abilities.TriggeredAbilityImpl;
6   import mage.abilities.common.delayed.AtTheBeginOfNextCleanupDelayedTriggeredAbility;
7   import mage.abilities.effects.common.CreateDelayedTriggeredAbilityEffect;
8   import mage.abilities.effects.common.SacrificeSourceEffect;
9   import mage.constants.Zone;
10  import mage.game.Game;
11  import mage.game.events.GameEvent.EventType;
12  import mage.game.events.GameEvent;
13  import mage.game.stack.Spell;
14  
15  /**
16   *
17   * @author Lonefox
18   */
19  public class SacrificeIfCastAtInstantTimeTriggeredAbility extends TriggeredAbilityImpl {
20  
21      public SacrificeIfCastAtInstantTimeTriggeredAbility() {
22          super(Zone.STACK, new CreateDelayedTriggeredAbilityEffect(new AtTheBeginOfNextCleanupDelayedTriggeredAbility(new SacrificeSourceEffect())));
23      }
24  
25      public SacrificeIfCastAtInstantTimeTriggeredAbility(final SacrificeIfCastAtInstantTimeTriggeredAbility ability) {
26          super(ability);
27      }
28  
29      @Override
30      public SacrificeIfCastAtInstantTimeTriggeredAbility copy() {
31          return new SacrificeIfCastAtInstantTimeTriggeredAbility(this);
32      }
33  
34      @Override
35      public boolean checkEventType(GameEvent event, Game game) {
36          return event.getType() == GameEvent.EventType.SPELL_CAST;
37      }
38  
39      @Override
40      public boolean checkTrigger(GameEvent event, Game game) {
41          // TODO: The sacrifice should occur only if you cast it using its own ability. If you cast it using some
42          // other effect (for instance, if it gained flash from Vedalken Orrery), then it shouldn't be sacrificed.
43          // see https://github.com/magefree/mage/issues/9512
44          Spell spell = game.getStack().getSpell(event.getTargetId());
45          if (spell == null || !spell.getSourceId().equals(getSourceId())) {
46              return false;
47          }
48          // TODO: this is a hack and doesn't handle all other ways a spell could be cast as though it had flash
49          if (Boolean.TRUE.equals(game.getState().getValue("PlayFromNotOwnHandZone" + getSourceId()))) {
50              return false;
51          }
52          return !(game.isMainPhase() && game.isActivePlayer(event.getPlayerId()) && game.getStack().size() == 1);
53      }
54  
55      @Override
56      public String getRule() {
57          return "If you cast it any time a sorcery couldn't have been cast, the controller of the permanent it becomes sacrifices it at the beginning of the next cleanup step.";
58      }
59  }

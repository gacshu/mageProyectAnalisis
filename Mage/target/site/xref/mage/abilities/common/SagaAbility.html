41          this(card, maxChapter, false);

44      public SagaAbility(Card card, SagaChapter maxChapter, boolean readAhead) {
45          super(Zone.ALL, null);
46          this.maxChapter = maxChapter;
47          this.showSacText = card.getSecondCardFace() == null;
48          this.readAhead = readAhead;
49          this.setRuleVisible(true);
50          this.setRuleAtTheTop(true);
51          Ability ability = new EntersBattlefieldAbility(new SagaLoreCountersEffect(readAhead, maxChapter));
52          ability.setRuleVisible(false);
53          card.addAbility(ability);
54      }

57          super(ability);
58          this.maxChapter = ability.maxChapter;
59          this.showSacText = ability.showSacText;
60          this.readAhead = ability.readAhead;
61      }

83      public void addChapterEffect(Card card, SagaChapter fromChapter, SagaChapter toChapter, Effects effects, Target target) {
84          addChapterEffect(card, fromChapter, toChapter, effects, new Targets(target));
85      }
86  
87      public void addChapterEffect(Card card, SagaChapter fromChapter, SagaChapter toChapter, Effects effects, Targets targets) {
88          addChapterEffect(card, fromChapter, toChapter, effects, targets, false);
89      }
90  
91      public void addChapterEffect(Card card, SagaChapter fromChapter, SagaChapter toChapter, Effects effects, Targets targets, boolean optional, Mode... modes) {
92          for (int i = fromChapter.getNumber(); i <= toChapter.getNumber(); i++) {
93              ChapterTriggeredAbility ability = new ChapterTriggeredAbility(null, SagaChapter.getChapter(i), toChapter, optional, readAhead);
94              for (Effect effect : effects) {
95                  if (effect != null) {
96                      ability.addEffect(effect.copy());
97                  }
98              }
99              for (Target target : targets) {
100                 if (target != null) {
101                     ability.addTarget(target.copy());
102                 }
103             }
104             for (Mode mode : modes) {
105                 ability.addMode(mode.copy());
106             }
107             if (i > fromChapter.getNumber()) {
108                 ability.setRuleVisible(false);
109             }
110             card.addAbility(ability);
111         }
112     }

123                 : "<i>(As this Saga enters and after your draw step, add a lore counter.")
124                 + (showSacText ? " Sacrifice after " + maxChapter.toString() + '.' : "") + ")</i> ";
125     }

149 class SagaLoreCountersEffect extends OneShotEffect {
150 
151     private final boolean readAhead;
152     private final SagaChapter maxChapter;
153 
154     SagaLoreCountersEffect(boolean readAhead, SagaChapter maxChapter) {
155         super(Outcome.Benefit);
156         this.readAhead = readAhead;
157         this.maxChapter = maxChapter;
158     }
159 
160     private SagaLoreCountersEffect(final SagaLoreCountersEffect effect) {
161         super(effect);
162         this.readAhead = effect.readAhead;
163         this.maxChapter = effect.maxChapter;
164     }
165 
166     @Override
167     public SagaLoreCountersEffect copy() {
168         return new SagaLoreCountersEffect(this);
169     }
170 
171     @Override
172     public boolean apply(Game game, Ability source) {
173         Permanent permanent = game.getPermanentEntering(source.getSourceId());
174         if (permanent == null) {
175             return false;
176         }
177         if (!readAhead) {
178             return permanent.addCounters(CounterType.LORE.createInstance(), source, game);
179         }
180         Player player = game.getPlayer(source.getControllerId());
181         if (player == null) {
182             return false;
183         }
184         int counters = player.getAmount(
185                 1, maxChapter.getNumber(),
186                 "Choose the number of lore counters to enter with", game
187         );
188         return permanent.addCounters(CounterType.LORE.createInstance(counters), source, game);
189     }

192 class ChapterTriggeredAbility extends TriggeredAbilityImpl {
193 
194     private final SagaChapter chapterFrom, chapterTo;
195     private final boolean readAhead;
196 
197     public ChapterTriggeredAbility(Effect effect, SagaChapter chapterFrom, SagaChapter chapterTo, boolean optional, boolean readAhead) {
198         super(Zone.ALL, effect, optional);
199         this.chapterFrom = chapterFrom;
200         this.chapterTo = chapterTo;
201         this.readAhead = readAhead;
202     }
203 
204     public ChapterTriggeredAbility(final ChapterTriggeredAbility ability) {
205         super(ability);
206         this.chapterFrom = ability.chapterFrom;
207         this.chapterTo = ability.chapterTo;
208         this.readAhead = ability.readAhead;
209     }
210 
211     @Override
212     public boolean checkEventType(GameEvent event, Game game) {
213         return event.getType() == GameEvent.EventType.COUNTERS_ADDED;
214     }
215 
216     @Override
217     public boolean checkTrigger(GameEvent event, Game game) {
218         if (!event.getTargetId().equals(getSourceId())
219                 || !event.getData().equals(CounterType.LORE.getName())) {
220             return false;
221         }
222         Permanent permanent = getSourcePermanentIfItStillExists(game);
223         if (permanent == null) {
224             return false;
225         }
226         int loreCounters = permanent.getCounters(game).getCount(CounterType.LORE);
227         if (readAhead && permanent.getTurnsOnBattlefield() == 0) {
228             return chapterFrom.getNumber() == loreCounters;
229         }
230         return loreCounters - event.getAmount() < chapterFrom.getNumber()
231                 && chapterFrom.getNumber() <= loreCounters;
232     }
233 
234     @Override
235     public ChapterTriggeredAbility copy() {
236         return new ChapterTriggeredAbility(this);
237     }
238 
239     public String getChapterRule() {
240         String rule = super.getRule();
241         return Character.toUpperCase(rule.charAt(0)) + rule.substring(1);
242     }
243 
244     public SagaChapter getChapterFrom() {
245         return chapterFrom;
246     }
247 
248     public SagaChapter getChapterTo() {
249         return chapterTo;
250     }
251 
252     @Override
253     public String getRule() {
254         return Arrays
255                 .stream(SagaChapter.values())
256                 .filter(chapter -> chapterFrom.getNumber() <= chapter.getNumber())
257                 .filter(chapter -> chapter.getNumber() <= chapterTo.getNumber())
258                 .map(SagaChapter::toString)
259                 .reduce((a, b) -> a + ", " + b)
260                 .orElse("")
261                 + " - " + CardUtil.getTextWithFirstCharUpperCase(super.getRule());
262     }

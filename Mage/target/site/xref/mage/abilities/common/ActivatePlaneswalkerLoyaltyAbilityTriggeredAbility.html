1   package mage.abilities.common;
2   
3   import mage.abilities.LoyaltyAbility;
4   import mage.abilities.TriggeredAbilityImpl;
5   import mage.abilities.effects.Effect;
6   import mage.constants.SubType;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.game.permanent.Permanent;
11  import mage.game.stack.StackAbility;
12  
13  public class ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility extends TriggeredAbilityImpl {
14  
15      private final SubType planeswalkerSubType;
16  
17      public ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility(Effect effect, SubType planeswalkerSubType) {
18          super(Zone.BATTLEFIELD, effect, false);
19          this.planeswalkerSubType = planeswalkerSubType;
20          setTriggerPhrase("Whenever you activate a loyalty ability of a " + planeswalkerSubType.getDescription() + " planeswalker, ");
21      }
22  
23      private ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility(final ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility ability) {
24          super(ability);
25          this.planeswalkerSubType = ability.planeswalkerSubType;
26      }
27  
28      @Override
29      public ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility copy() {
30          return new ActivatePlaneswalkerLoyaltyAbilityTriggeredAbility(this);
31      }
32  
33      @Override
34      public boolean checkEventType(GameEvent event, Game game) {
35          return event.getType() == GameEvent.EventType.ACTIVATED_ABILITY;
36      }
37  
38      @Override
39      public boolean checkTrigger(GameEvent event, Game game) {
40          if (!event.getPlayerId().equals(getControllerId())) {
41              return false;
42          }
43          StackAbility stackAbility = (StackAbility) game.getStack().getStackObject(event.getSourceId());
44          if (stackAbility == null || !(stackAbility.getStackAbility() instanceof LoyaltyAbility)) {
45              return false;
46          }
47          Permanent permanent = stackAbility.getSourcePermanentOrLKI(game);
48          if (permanent == null || !permanent.isPlaneswalker(game)
49                  || !permanent.hasSubtype(planeswalkerSubType, game)) {
50              return false;
51          }
52          this.getEffects().setValue("stackObject", stackAbility);
53          return true;
54      }
55  }

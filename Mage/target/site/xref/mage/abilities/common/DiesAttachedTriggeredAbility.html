14  /**
15   * "When enchanted/equipped creature dies" triggered ability
16   *
17   * @author Loki
18   */
19  public class DiesAttachedTriggeredAbility extends TriggeredAbilityImpl {
20  
21      private final String attachedDescription;
22      private final boolean diesRuleText;
23      protected SetTargetPointer setTargetPointer;
24  
25      public DiesAttachedTriggeredAbility(Effect effect, String attachedDescription) {
26          this(effect, attachedDescription, false);
27      }
28  
29      public DiesAttachedTriggeredAbility(Effect effect, String attachedDescription, boolean optional) {
30          this(effect, attachedDescription, optional, true);
31      }
32  
33      public DiesAttachedTriggeredAbility(Effect effect, String attachedDescription, boolean optional, boolean diesRuleText) {
34          this(effect, attachedDescription, optional, diesRuleText, SetTargetPointer.NONE);
35      }
36  
37      public DiesAttachedTriggeredAbility(Effect effect, String attachedDescription, boolean optional,
38                                          boolean diesRuleText, SetTargetPointer setTargetPointer) {
39          super(Zone.ALL, effect, optional); // because the trigger only triggers if the object was attached, it doesn't matter where the Attachment was moved to (e.g. by replacement effect) after the trigger triggered, so Zone.all
40          this.attachedDescription = attachedDescription;
41          this.diesRuleText = diesRuleText;
42          this.setTargetPointer = setTargetPointer;
43          setTriggerPhrase(generateTriggerPhrase());
44      }
45  
46      public DiesAttachedTriggeredAbility(final DiesAttachedTriggeredAbility ability) {
47          super(ability);
48          this.attachedDescription = ability.attachedDescription;
49          this.diesRuleText = ability.diesRuleText;
50          this.setTargetPointer = ability.setTargetPointer;
51      }
52  
53      @Override
54      public DiesAttachedTriggeredAbility copy() {
55          return new DiesAttachedTriggeredAbility(this);
56      }
57  
58      @Override
59      public boolean checkEventType(GameEvent event, Game game) {
60          return event.getType() == GameEvent.EventType.ZONE_CHANGE;
61      }
62  
63      @Override
64      public boolean checkTrigger(GameEvent event, Game game) {
65          if (!((ZoneChangeEvent) event).isDiesEvent()) {
66              return false;
67          }
68          ZoneChangeEvent zEvent = (ZoneChangeEvent) event;
69          boolean triggered = false;
70          if (zEvent.getTarget() != null
71                  && zEvent.getTarget().getAttachments() != null
72                  && zEvent.getTarget().getAttachments().contains(this.getSourceId())) {
73              triggered = true;
74          } else {
75              // If the attachment and attachedTo went to graveyard at the same time, the trigger applies.
76              // If the attachment is removed beforehand, the trigger fails.
77              // IE: A player cast Planar Clensing.  The attachment is Disenchanted in reponse
78              // and successfully removed from the attachedTo.  The trigger fails.
79              Permanent attachment = getSourcePermanentOrLKI(game);
80              Card attachmentCard = game.getCard(getSourceId());
81              if (attachment != null
82                      && zEvent.getTargetId() != null
83                      && attachment.getAttachedTo() != null
84                      && zEvent.getTargetId().equals(attachment.getAttachedTo())) {
85                  Permanent attachedTo = game.getPermanentOrLKIBattlefield(attachment.getAttachedTo());
86                  if (attachedTo != null
87                          && game.getState().getZone(attachedTo.getId()) == (Zone.GRAVEYARD)  // Demonic Vigor
88                          && attachmentCard != null
89                          && attachment.getAttachedToZoneChangeCounter() == attachedTo.getZoneChangeCounter(game)
90                          && attachment.getZoneChangeCounter(game) == attachmentCard.getZoneChangeCounter(game)) {
91                      triggered = true;
92                  }
93              }
94          }
95          if (!triggered) {
96              return false;
97          }
98          if (zEvent.getTarget() == null) {
99              return true;
100         }
101         getEffects().setValue("attachedTo", zEvent.getTarget());
102         getEffects().setValue("zcc", zEvent.getTarget().getZoneChangeCounter(game) + 1); // zone change info from battlefield
103         if (setTargetPointer == SetTargetPointer.ATTACHED_TO_CONTROLLER) {
104             Permanent attachment = game.getPermanentOrLKIBattlefield(getSourceId());
105             if (attachment != null
106                     && attachment.getAttachedTo() != null) {
107                 Permanent attachedTo = (Permanent) game.getLastKnownInformation(attachment.getAttachedTo(),
108                         Zone.BATTLEFIELD, attachment.getAttachedToZoneChangeCounter());
109                 if (attachedTo != null) {
110                     getEffects().setTargetPointer(new FixedTarget(attachedTo.getControllerId()));
111                 }
112             }
113         }
114         // set targetpointer to the creature that died
115         if (setTargetPointer == SetTargetPointer.CARD
116                 || setTargetPointer == SetTargetPointer.PERMANENT) {
117             Permanent attachment = game.getPermanentOrLKIBattlefield(getSourceId());
118             if (attachment != null
119                     && attachment.getAttachedTo() != null) {
120                 Permanent attachedTo = (Permanent) game.getLastKnownInformation(attachment.getAttachedTo(),
121                         Zone.BATTLEFIELD, attachment.getAttachedToZoneChangeCounter());
122                 if (attachedTo != null) {
123                     getEffects().setTargetPointer(new FixedTarget(attachedTo.getId()));
124                 }
125             }
126         }
127         return true;
128     }
129 
130     private String generateTriggerPhrase() {
131         StringBuilder sb = new StringBuilder();
132         if (attachedDescription.startsWith("equipped")) {
133             sb.append("Whenever ");
134         } else {
135             sb.append("When ");
136         }
137         sb.append(attachedDescription);
138         if (diesRuleText) {
139             sb.append(" dies, ");
140         } else {
141             sb.append(" is put into a graveyard, ");
142         }
143         return sb.toString();
144     }

1   package mage.abilities.common;
2   
3   import mage.abilities.TriggeredAbilityImpl;
4   import mage.abilities.effects.Effect;
5   import mage.abilities.effects.common.DestroyTargetEffect;
6   import mage.constants.Zone;
7   import mage.filter.FilterPermanent;
8   import mage.game.Game;
9   import mage.game.events.GameEvent;
10  import mage.game.permanent.Permanent;
11  import mage.target.targetpointer.FixedTarget;
12  
13  public class DestroyPlaneswalkerWhenDamagedTriggeredAbility extends TriggeredAbilityImpl {
14  
15      private final FilterPermanent filter;
16  
17      public DestroyPlaneswalkerWhenDamagedTriggeredAbility() {
18          this((FilterPermanent) null);
19      }
20  
21      public DestroyPlaneswalkerWhenDamagedTriggeredAbility(FilterPermanent filter) {
22          super(Zone.BATTLEFIELD, null);
23          this.filter = filter;
24      }
25  
26      private DestroyPlaneswalkerWhenDamagedTriggeredAbility(final DestroyPlaneswalkerWhenDamagedTriggeredAbility effect) {
27          super(effect);
28          this.filter = effect.filter;
29      }
30  
31      @Override
32      public DestroyPlaneswalkerWhenDamagedTriggeredAbility copy() {
33          return new DestroyPlaneswalkerWhenDamagedTriggeredAbility(this);
34      }
35  
36      @Override
37      public boolean checkEventType(GameEvent event, Game game) {
38          return event.getType() == GameEvent.EventType.DAMAGED_PERMANENT;
39      }
40  
41      @Override
42      public boolean checkTrigger(GameEvent event, Game game) {
43          Permanent permanent = getSourcePermanentIfItStillExists(game);
44          if (permanent == null) {
45              return false;
46          }
47          boolean applies = filter != null ?
48                  permanent.isPlaneswalker(game) && filter.match(permanent, game) : event.getSourceId().equals(getSourceId());
49          if (applies) {
50              Effect effect = new DestroyTargetEffect();
51              effect.setTargetPointer(new FixedTarget(event.getTargetId(), game));
52              this.getEffects().clear();
53              this.addEffect(effect);
54              return true;
55          }
56          return false;
57      }
58  
59      @Override
60      public String getRule() {
61          return "Whenever " + (filter != null ? filter.getMessage() : "this creature") + " deals damage to a planeswalker, destroy that planeswalker.";
62      }
63  }

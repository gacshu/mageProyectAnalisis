35      public TriggeredAbilityImpl(Zone zone, Effect effect, boolean optional) {
36          super(AbilityType.TRIGGERED, zone);
37          setLeavesTheBattlefieldTrigger(false);
38          if (effect != null) {
39              addEffect(effect);
40          }
41          this.optional = optional;

43          // verify check: DoIfCostPaid effect already asks about action (optional), so no needs to ask it again in triggered ability
44          if (effect instanceof DoIfCostPaid) {
45              if (this.optional && ((DoIfCostPaid) effect).isOptional()) {
46                  throw new IllegalArgumentException("DoIfCostPaid effect must have only one optional settings, but it have two (trigger + DoIfCostPaid): " + this.getClass().getSimpleName());
47              }
48          }

133     @Override
134     public boolean resolve(Game game) {
135         if (!checkInterveningIfClause(game)) {
136             return false;
137         }
138         if (isOptional()) {
139             MageObject object = game.getObject(getSourceId());
140             Player player = game.getPlayer(this.getControllerId());
141             if (player == null || object == null
142                     || (doOnlyOnce && checkUsedAlready(game))
143                     || !player.chooseUse(
144                     getEffects().getOutcome(this),
145                     this.getRule(object.getLogName()), this, game
146             )) {
147                 return false;
148             }
149         }
150         //20091005 - 603.4
151         if (!super.resolve(game)) {
152             return false;
153         }
154         if (doOnlyOnce) {
155             game.getState().setValue(CardUtil.getCardZoneString(
156                     "lastTurnUsed" + originalId, sourceId, game
157             ), game.getTurnNum());
158         }
159         return true;
160     }

164         MageObject object = game.getObject(sourceId);
165         StringBuilder sb = new StringBuilder();
166         if (object != null) {
167             sb.append("Ability triggers: ").append(object.getLogName()).append(" - ").append(this.getRule(object.getLogName()));
168         } else {

178     @Override
179     public String getRule() {
180         StringBuilder sb = new StringBuilder();
181         String prefix;
182         if (abilityWord != null) {
183             prefix = abilityWord.formatWord();
184         } else if (flavorWord != null) {
185             prefix = CardUtil.italicizeWithEmDash(flavorWord);
186         } else {
187             prefix = "";
188         }
189         sb.append(prefix);
190 
191         sb.append(triggerPhrase == null ? getTriggerPhrase() : triggerPhrase);
192 
193         String superRule = super.getRule(true);
194         if (!superRule.isEmpty()) {
195             String ruleLow = superRule.toLowerCase(Locale.ENGLISH);
196             if (isOptional()) {
197                 if (ruleLow.startsWith("you ")) {
198                     if (!ruleLow.startsWith("you may")) {
199                         StringBuilder newRule = new StringBuilder(superRule);
200                         newRule.insert(4, "may ");
201                         superRule = newRule.toString();
202                     }
203                 } else if (this.getTargets().isEmpty()
204                         || ruleLow.startsWith("attach")
205                         || ruleLow.startsWith("counter")
206                         || ruleLow.startsWith("destroy")
207                         || ruleLow.startsWith("sacrifice")
208                         || ruleLow.startsWith("exchange")
209                         || ruleLow.startsWith("exile")
210                         || ruleLow.startsWith("gain")
211                         || ruleLow.startsWith("goad")
212                         || ruleLow.startsWith("put")
213                         || ruleLow.startsWith("remove")
214                         || ruleLow.startsWith("return")
215                         || ruleLow.startsWith("tap")
216                         || ruleLow.startsWith("untap")) {
217                     sb.append("you may ");
218                 } else if (!ruleLow.startsWith("its controller may")) {
219                     sb.append("you may have ");
220                 }
221 
222             }
223             sb.append(superRule);
224             if (triggersOnce) {
225                 sb.append(" This ability triggers only once each turn.");
226             }
227             if (doOnlyOnce) {
228                 sb.append(" Do this only once each turn.");
229             }
230         }
231 
232         return sb.toString();
233     }

241     @Override
242     public boolean isInUseableZone(Game game, MageObject source, GameEvent event) {
243 
244         /**
245          * 603.6. Trigger events that involve objects changing zones are called
246          * “zone-change triggers.” Many abilities with zone-change triggers
247          * attempt to do something to that object after it changes zones. During
248          * resolution, these abilities look for the object in the zone that it
249          * moved to. If the object is unable to be found in the zone it went to,
250          * the part of the ability attempting to do something to the object will
251          * fail to do anything. The ability could be unable to find the object
252          * because the object never entered the specified zone, because it left
253          * the zone before the ability resolved, or because it is in a zone that
254          * is hidden from a player, such as a library or an opponent's hand.
255          * (This rule applies even if the object leaves the zone and returns
256          * again before the ability resolves.) The most common zone-change
257          * triggers are enters-the-battlefield triggers and
258          * leaves-the-battlefield triggers.
259          *
260          * from:
261          * http://www.mtgsalvation.com/forums/magic-fundamentals/magic-rulings/magic-rulings-archives/537065-ixidron-and-kozilek
262          * There are two types of triggers that involve the graveyard: dies
263          * triggers (which are a subset of leave-the-battlefield triggers) and
264          * put into the graveyard from anywhere triggers.
265          *
266          * The former triggers trigger based on the game state prior to the move
267          * where the Kozilek permanent is face down and has no abilities. The
268          * latter triggers trigger from the game state after the move where the
269          * Kozilek card is itself and has the ability.
270          */
271         if (event == null || event.getTargetId() == null || !event.getTargetId().equals(getSourceId())) {
272             return super.isInUseableZone(game, source, event);
273         }
274         switch (event.getType()) {
275             case ZONE_CHANGE:
276                 ZoneChangeEvent zce = (ZoneChangeEvent) event;
277                 if (event.getTargetId().equals(getSourceId()) && !zce.getToZone().isPublicZone()) {
278                     // If an ability triggers when the object that has it is put into a hidden zone from a graveyard,
279                     // that ability triggers from the graveyard, (such as Golgari Brownscale),
280                     // Yixlid Jailer will prevent that ability from triggering.
281                     if (zce.getFromZone().match(Zone.GRAVEYARD)) {
282                         if (!CardUtil.cardHadAbility(this, game.getLastKnownInformationCard(getSourceId(), zce.getFromZone()), getSourceId(), game)) {
283                             return false;
284                         }
285                     }
286                 }
287                 if (isLeavesTheBattlefieldTrigger()) {
288                     source = zce.getTarget();
289                 }
290                 break;
291             case DESTROYED_PERMANENT:
292                 if (isLeavesTheBattlefieldTrigger()) {
293                     source = game.getLastKnownInformation(getSourceId(), Zone.BATTLEFIELD);
294                 }
295                 break;
296             case PHASED_OUT:
297             case PHASED_IN:
298                 if (isLeavesTheBattlefieldTrigger()) {
299                     source = game.getLastKnownInformation(getSourceId(), event.getZone());
300                 }
301                 if (this.zone == Zone.ALL || game.getLastKnownInformation(getSourceId(), zone) != null) {
302                     return this.hasSourceObjectAbility(game, source, event);
303                 }
304                 break;
305         }
306         return super.isInUseableZone(game, source, event);
307     }

333     public static boolean isInUseableZoneDiesTrigger(TriggeredAbility source, GameEvent event, Game game) {
334         // Get the source permanent of the ability 
335         MageObject sourceObject = null;
336         if (game.getState().getZone(source.getSourceId()) == Zone.BATTLEFIELD) {
337             sourceObject = game.getPermanent(source.getSourceId());
338         } else {
339             // TODO: multiple calls of ApplyEffects all around the code are breaking a short living lki idea
340             //  (PlayerImpl's call to move to battlefield do the worse thing)
341             //  -
342             //  Original idea: short living LKI must help to find a moment in the inner of resolve
343             //  -
344             //  Example:
345             //   --!---------------!-------------!-----!-----------!
346             //   - !     steps     !  perm zone  ! LKI ! short LKI !
347             //   --!---------------!-------------!-----!-----------!
348             //   - ! resolve start ! battlefield ! no  !   no      !
349             //   - ! step 1        ! battlefield ! no  !   no      ! permanent moving to graveyard by step's command
350             //   - ! step 2        !  graveyard  ! yes !   yes     ! other commands
351             //   - ! step 3        !  graveyard  ! yes !   yes     ! other commands
352             //   - ! raise triggers!  graveyard  ! yes !   yes     ! handle and put triggers that was raised in resolve steps
353             //   - ! resolve end   !  graveyard  ! yes !   no      !
354             //   - ! resolve next  !  graveyard  ! yes !   no      ! resolve next spell
355             //   - ! empty stack   !  graveyard  ! no  !   no      ! no more to resolve
356             //   --!---------------!-------------!-----!-----------!
357             //  -
358             //  - Problem 1: move code (well, not only move) calls ApplyEffects in the middle of the resolve
359             //  - and reset short LKI (after short LKI reset dies trigger will not work)
360             //  - Example: Goblin Welder calls sacrifice and card move in the same effect - but move call do
361             //  - a reset and dies trigger ignored (trigger thinks that permanent already dies)
362             //  -
363             //  - Possible fix:
364             //  - replace ApplyEffects in the move code by game.getState().processAction(game);
365             //  - check and fix many broken (is it was a false positive test or something broken)
366             //sourceObject = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
367             if (game.getShortLivingLKI(source.getSourceId(), Zone.BATTLEFIELD)) {
368                 sourceObject = (Permanent) game.getLastKnownInformation(source.getSourceId(), Zone.BATTLEFIELD);
369             }
370         }
371         if (sourceObject == null) { // source is no permanent
372             sourceObject = game.getObject(source);
373             if (sourceObject == null || sourceObject.isPermanent(game)) {
374                 return false; // No source object found => ability is not valid
375             }
376         }
377 
378         if (!source.hasSourceObjectAbility(game, sourceObject, event)) {
379             return false; // the permanent does currently not have or before it dies the ability so no trigger
380         }
381 
382         // check now it is in graveyard (only if it is no token and was the target itself)
383         if (source.getSourceId().equals(event.getTargetId()) // source is also the target
384                 && !(sourceObject instanceof PermanentToken) // it's no token
385                 && sourceObject.getZoneChangeCounter(game) + 1 == game.getState().getZoneChangeCounter(source.getSourceId())) { // It's in the next zone
386             Zone after = game.getState().getZone(source.getSourceId());
387             if (!Zone.GRAVEYARD.match(after)) { // Zone is not the graveyard
388                 return false; // Moving to graveyard was replaced so no trigger
389             }
390         }
391 
392         return true;
393     }

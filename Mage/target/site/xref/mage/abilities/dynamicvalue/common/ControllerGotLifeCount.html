1   package mage.abilities.dynamicvalue.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.dynamicvalue.DynamicValue;
5   import mage.abilities.effects.Effect;
6   import mage.abilities.hint.Hint;
7   import mage.abilities.hint.ValueHint;
8   import mage.game.Game;
9   import mage.watchers.common.PlayerGainedLifeWatcher;
10  
11  /**
12   * Amount of life the controller got this turn.
13   *
14   * @author LevelX2
15   */
16  public enum ControllerGotLifeCount implements DynamicValue {
17      instance;
18  
19      private static final Hint hint = new ValueHint("Life gained this turn", instance);
20  
21      @Override
22      public int calculate(Game game, Ability sourceAbility, Effect effect) {
23          PlayerGainedLifeWatcher watcher = game.getState().getWatcher(PlayerGainedLifeWatcher.class);
24          if (watcher != null) {
25              return watcher.getLifeGained(sourceAbility.getControllerId());
26          }
27          return 0;
28      }
29  
30      @Override
31      public ControllerGotLifeCount copy() {
32          return instance;
33      }
34  
35      @Override
36      public String toString() {
37          return "X";
38      }
39  
40      @Override
41      public String getMessage() {
42          return "the amount of life you gained this turn";
43      }
44  
45      public static Hint getHint() {
46          return hint;
47      }
48  }

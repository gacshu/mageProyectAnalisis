173     @Override
174     public boolean resolve(Game game) {
175         boolean result = true;
176         //20100716 - 117.12
177         if (checkIfClause(game)) {
178             // Ability has started resolving. Fire event.
179             // Used for abilities counting the number of resolutions like Ashling the Pilgrim.
180             game.fireEvent(new GameEvent(GameEvent.EventType.RESOLVING_ABILITY, this.getOriginalId(), this, this.getControllerId()));
181             if (this instanceof TriggeredAbility) {
182                 for (UUID modeId : this.getModes().getSelectedModes()) {
183                     this.getModes().setActiveMode(modeId);
184                     result = resolveMode(game);
185                 }
186             } else {
187                 result = resolveMode(game);
188             }
189         }
190         return result;
191     }

193     private boolean resolveMode(Game game) {
194         boolean result = true;
195         for (Effect effect : getEffects()) {
196             if (game.inCheckPlayableState() && !(effect instanceof ManaEffect)) {
197                 continue; // Ignored non mana effects - see GameEvent.TAPPED_FOR_MANA
198             }
199             if (effect instanceof OneShotEffect) {
200                 boolean effectResult = effect.apply(game, this);
201                 result &= effectResult;
202                 if (logger.isDebugEnabled() && this.getAbilityType() != AbilityType.MANA && !effectResult) {
203                    
204                         
205                             if (this.getSourceId() != null) {
206                                 MageObject mageObject = game.getObject(this.getSourceId());
207                                 if (mageObject != null) {
208                                     logger.debug("AbilityImpl.resolve: object: " + mageObject.getName());
209                                 }
210                             }
211                             logger.debug("AbilityImpl.resolve: effect returned false -" + effect.getText(this.getModes().getMode()));
212                         
213                     
214                 }
215             } else {
216                 game.addEffect((ContinuousEffect) effect, this);
217             }
218             /**
219              * All restrained trigger events are fired now. To restrain the
220              * events is mainly neccessary because of the movement of multiple
221              * object at once. If the event is fired directly as one object
222              * moved, other objects are not already in the correct zone to check
223              * for their effects. (e.g. Valakut, the Molten Pinnacle)
224              */
225             game.getState().handleSimultaneousEvent(game);
226             game.resetShortLivingLKI();
227             /**
228              * game.applyEffects() has to be done at least for every effect that
229              * moves cards/permanent between zones, or changes control of
230              * objects so Static effects work as intended if dependant from the
231              * moved objects zone it is in Otherwise for example were static
232              * abilities with replacement effects deactivated too late Example:
233              * {@link org.mage.test.cards.replacement.DryadMilitantTest#testDiesByDestroy testDiesByDestroy}
234              */
235             game.applyEffects();
236             game.getState().getTriggers().checkStateTriggers(game);
237         }
238         return result;
239     }

241     @Override
242     public boolean activate(Game game, boolean noMana) {
243         Player controller = game.getPlayer(this.getControllerId());
244         if (controller == null) {
245             return false;
246         }
247         game.applyEffects();
248 
249         MageObject sourceObject = getSourceObject(game);
250         if (getSourceObjectZoneChangeCounter() == 0) {
251             setSourceObjectZoneChangeCounter(game.getState().getZoneChangeCounter(getSourceId()));
252         }
253         setSourcePermanentTransformCount(game);
254 
255         // if ability can be cast for no mana, clear the mana costs now, because additional mana costs must be paid.
256         // For Flashback ability can be set X before, so the X costs have to be restored for the flashbacked ability
257         if (noMana) {
258             if (!this.getManaCostsToPay().getVariableCosts().isEmpty()) {
259                 int xValue = this.getManaCostsToPay().getX();
260                 this.getManaCostsToPay().clear();
261                 VariableManaCost xCosts = new VariableManaCost(VariableCostType.ADDITIONAL);
262                 // no x events - rules from Unbound Flourishing:
263                 // - Spells with additional costs that include X won't be affected by Unbound Flourishing. X must be in the spell's mana cost.
264                 xCosts.setAmount(xValue, xValue, false);
265                 this.getManaCostsToPay().add(xCosts);
266             } else {
267                 this.getManaCostsToPay().clear();
268             }
269         }
270 
271         // fused or spliced spells contain multiple abilities (e.g. fused, left, right)
272         // optional costs and cost modification must be applied only to the first/main ability
273         boolean isMainPartAbility = !CardUtil.isFusedPartAbility(this, game);
274 
275         /* 20220908 - 601.2b
276          * If the player wishes to splice any cards onto the spell (see rule 702.45), they
277          * reveal those cards in their hand.
278          */
279         if (isMainPartAbility && this.abilityType == AbilityType.SPELL) {
280             game.getContinuousEffects().applySpliceEffects(this, game);
281         }
282 
283         // 20130201 - 601.2b
284         // If the spell has alternative or additional costs that will be paid as it's being cast such
285         // as buyback, kicker, or convoke costs (see rules 117.8 and 117.9), the player announces his
286         // or her intentions to pay any or all of those costs (see rule 601.2e).
287         // A player can't apply two alternative methods of casting or two alternative costs to a single spell.
288         if (isMainPartAbility && !activateAlternateOrAdditionalCosts(sourceObject, noMana, controller, game) 
289         		&& getAbilityType() == AbilityType.SPELL
290                     && ((SpellAbility) this).getSpellAbilityType() == SpellAbilityType.FACE_DOWN_CREATURE) {
291                 return false;
292             
293         }
294 
295         // 117.6. Some mana costs contain no mana symbols. This represents an unpayable cost. An ability can
296         // also have an unpayable cost if its cost is based on the mana cost of an object with no mana cost.
297         // Attempting to cast a spell or activate an ability that has an unpayable cost is a legal action.
298         // However, attempting to pay an unpayable cost is an illegal action.
299         //
300         // We apply this now, *AFTER* the user has made the choice to pay an alternative cost for the
301         // spell. You can also still cast a spell with an unplayable cost by... not paying it's mana cost.
302         //if (getAbilityType() == AbilityType.SPELL && getManaCostsToPay().isEmpty() && !noMana) {
303         //    return false;
304         //}
305         if (getAbilityType() == AbilityType.SPELL && (getManaCostsToPay().isEmpty() && getCosts().isEmpty()) && !noMana) {
306             return false;
307         }
308 
309         // 20121001 - 601.2b
310         // If the spell has a variable cost that will be paid as it's being cast (such as an {X} in
311         // its mana cost; see rule 107.3), the player announces the value of that variable.
312         VariableManaCost variableManaCost = handleManaXCosts(game, noMana, controller);
313         String announceString = handleOtherXCosts(game, controller);
314 
315         handlePhyrexianManaCosts(game, controller);
316 
317         /* 20130201 - 601.2b
318          * If the spell is modal the player announces the mode choice (see rule 700.2).
319          */
320         // rules:
321         // You kick a spell as you cast it. You declare whether you're going to pay a kicker cost at the same
322         // time you'd choose a spell's mode, and then you actually pay it at the same time you pay the spell's mana cost.
323         // Kicking a spell is always optional.
324         if (!getModes().choose(game, this)) {
325             return false;
326         }
327 
328         // unit tests only: it allows to add targets/choices by two ways:
329         // 1. From cast/activate command params (process it here)
330         // 2. From single addTarget/setChoice, it's a preffered method for tests (process it in normal choose dialogs like human player)
331         if (controller.isTestsMode() && !controller.addTargets(this, game)) {
332                 return false;
333             
334         }
335 
336         for (UUID modeId : this.getModes().getSelectedModes()) {
337             this.getModes().setActiveMode(modeId);
338 
339             // 20121001 - 601.2c
340             // 601.2c The player announces their choice of an appropriate player, object, or zone for
341             // each target the spell requires. A spell may require some targets only if an alternative or
342             // additional cost (such as a buyback or kicker cost), or a particular mode, was chosen for it;
343             // otherwise, the spell is cast as though it did not require those targets. If the spell has a
344             // variable number of targets, the player announces how many targets they will choose before
345             // they announce those targets. The same target can't be chosen multiple times for any one
346             // instance of the word "target" on the spell. However, if the spell uses the word "target" in
347             // multiple places, the same object, player, or zone can be chosen once for each instance of the
348             // word "target" (as long as it fits the targeting criteria). If any effects say that an object
349             // or player must be chosen as a target, the player chooses targets so that they obey the
350             // maximum possible number of such effects without violating any rules or effects that say that
351             // an object or player can't be chosen as a target. The chosen players, objects, and/or zones
352             // each become a target of that spell. (Any abilities that trigger when those players, objects,
353             // and/or zones become the target of a spell trigger at this point; they'll wait to be put on
354             // the stack until the spell has finished being cast.)
355 
356             if (this.getAbilityType() != AbilityType.TRIGGERED) { // triggered abilities check this already in playerImpl.triggerAbility
357                 adjustTargets(game);
358             }
359 
360             if (!getTargets().isEmpty()) {
361                 Outcome outcome = getEffects().getOutcome(this);
362 
363                 // only activated abilities can be canceled by human user (not triggered)
364                 boolean canCancel = this instanceof ActivatedAbility && controller.isHuman();
365                 if (!getTargets().chooseTargets(outcome, this.controllerId, this, noMana, game, canCancel)) {
366                     // was canceled during targer selection
367                     return false;
368                 }
369             }
370         } // end modes
371 
372         // 20220908 - 601.2e
373         // 601.2e The game checks to see if the proposed spell can legally be cast. If the proposed spell
374         // is illegal, the game returns to the moment before the casting of that spell was proposed
375         // (see rule 727, "Handling Illegal Actions").
376         if (this.getAbilityType() == AbilityType.SPELL) {
377             GameEvent castEvent = GameEvent.getEvent(GameEvent.EventType.CAST_SPELL_LATE, this.getId(), this, getControllerId());
378             castEvent.setZone(game.getState().getZone(CardUtil.getMainCardId(game, sourceId)));
379             if (game.replaceEvent(castEvent, this)) {
380                 return false;
381             }
382         }
383 
384         // this is a hack to prevent mana abilities with mana costs from causing endless loops - pay other costs first
385         if (this instanceof ActivatedManaAbilityImpl && !costs.pay(this, game, this, controllerId, noMana, null)) {
386             logger.debug("activate mana ability failed - non mana costs");
387             return false;
388         }
389 
390         //20101001 - 601.2e
391         if (isMainPartAbility) {
392             adjustCosts(game); // still needed for CostAdjuster objects (to handle some types of dynamic costs)
393             game.getContinuousEffects().costModification(this, game);
394         }
395 
396         UUID activatorId = controllerId;
397         if ((this instanceof ActivatedAbilityImpl) && ((ActivatedAbilityImpl) this).getActivatorId() != null) {
398             activatorId = ((ActivatedAbilityImpl) this).getActivatorId();
399         }
400 
401         //20100716 - 601.2f  (noMana is not used here, because mana costs were cleared for this ability before adding additional costs and applying cost modification effects)
402         if (!manaCostsToPay.pay(this, game, this, activatorId, false, null)) {
403             return false; // cancel during mana payment
404         }
405 
406         //20100716 - 601.2g
407         if (!costs.pay(this, game, this, activatorId, noMana, null)) {
408             logger.debug("activate failed - non mana costs");
409             return false;
410         }
411         // inform about x costs now, so canceled announcements are not shown in the log
412         if ((announceString != null) && (!announceString.equals(""))) {
413             game.informPlayers(announceString);
414         }
415         if (variableManaCost != null) {
416             int xValue = getManaCostsToPay().getX();
417             game.informPlayers(controller.getLogName() + " announces a value of " + xValue + " for " + variableManaCost.getText());
418         }
419         activated = true;
420         return true;
421     }

428     @Override
429     public boolean activateAlternateOrAdditionalCosts(MageObject sourceObject, boolean noMana, Player controller, Game game) {
430         boolean canUseAlternativeCost = true;
431         boolean canUseAdditionalCost = true;
432 
433         if (this instanceof SpellAbility) {
434             // A player can't apply two alternative methods of casting or two alternative costs to a single spell.
435             switch (((SpellAbility) this).getSpellAbilityCastMode()) {
436 
437                 case FLASHBACK:
438                 case MADNESS:
439                 case DISTURB:
440                     // from Snapcaster Mage:
441                     // If you cast a spell from a graveyard using its flashback ability, you can't pay other alternative costs
442                     // (such as that of Foil). (2018-12-07)
443                     canUseAlternativeCost = false;
444                     // You may pay any optional additional costs the spell has, such as kicker costs. You must pay any
445                     // mandatory additional costs the spell has, such as that of Tormenting Voice. (2018-12-07)
446                     canUseAdditionalCost = true;
447                     break;
448                 case NORMAL:
449                 default:
450                     canUseAlternativeCost = true;
451                     canUseAdditionalCost = true;
452                     break;
453             }
454         }
455 
456         boolean alternativeCostUsed = false;
457         if (sourceObject != null && !(sourceObject instanceof Permanent)) {
458             // it's important to apply alternative cost first
459             // example: Omniscience gives free mana as alternative, but Entwine ability adds {2} as additional
460             Abilities<Ability> abilities = CardUtil.getAbilities(sourceObject, game);
461 
462             // 1. ALTERNATIVE COSTS
463             for (Ability ability : abilities) {
464                 // if cast for noMana no Alternative costs are allowed
465                 if (canUseAlternativeCost && !noMana && ability instanceof AlternativeSourceCosts) {
466                     AlternativeSourceCosts alternativeSpellCosts = (AlternativeSourceCosts) ability;
467                     if (alternativeSpellCosts.isAvailable(this, game) 
468                     		&& alternativeSpellCosts.askToActivateAlternativeCosts(this, game)) {
469                             // only one alternative costs may be activated
470                             alternativeCostUsed = true;
471                             break;
472                         
473                     }
474                 }
475             }
476             // controller specific alternate spell costs
477             if (canUseAlternativeCost && !noMana && !alternativeCostUsed
478             		&& this.getAbilityType() == AbilityType.SPELL
479                         // 117.9a Only one alternative cost can be applied to any one spell as it's being cast.
480                         // So an alternate spell ability can't be paid with Omniscience
481                         && ((SpellAbility) this).getSpellAbilityType() != SpellAbilityType.BASE_ALTERNATE) {
482                     for (AlternativeSourceCosts alternativeSourceCosts : controller.getAlternativeSourceCosts()) {
483                         if (alternativeSourceCosts.isAvailable(this, game) &&
484                         		alternativeSourceCosts.askToActivateAlternativeCosts(this, game)) {
485                                 // only one alternative costs may be activated
486                                 alternativeCostUsed = true;
487                                 break;
488                             
489                         }
490                     }
491                 
492             }
493 
494             // 2. ADDITIONAL COST
495             for (Ability ability : abilities) {
496                 if (canUseAdditionalCost && ability instanceof OptionalAdditionalSourceCosts) {
497                     ((OptionalAdditionalSourceCosts) ability).addOptionalAdditionalCosts(this, game);
498                 }
499             }
500         }
501 
502         return alternativeCostUsed;
503     }

505     /**
506      * Handles the setting of non mana X costs
507      *
508      * @param controller
509      * @param game
510      * @return announce message
511      */
512     protected String handleOtherXCosts(Game game, Player controller) {
513         StringBuilder announceString = new StringBuilder();
514         for (VariableCost variableCost : this.costs.getVariableCosts()) {
515             if (!(variableCost instanceof VariableManaCost) && !((Cost) variableCost).isPaid()) {
516                 int xValue = variableCost.announceXValue(this, game);
517                 Cost fixedCost = variableCost.getFixedCostsFromAnnouncedValue(xValue);
518                 if (fixedCost != null) {
519                     costs.add(fixedCost);
520                 }
521                 // set the xcosts to paid
522                 // no x events - rules from Unbound Flourishing:
523                 // - Spells with additional costs that include X won't be affected by Unbound Flourishing. X must be in the spell's mana cost.
524                 variableCost.setAmount(xValue, xValue, false);
525                 ((Cost) variableCost).setPaid();
526                 String message = controller.getLogName() + " announces a value of " + xValue + " (" + variableCost.getActionText() + ')';
527                 announceString.append(message);
528             }
529         }
530         return announceString.toString();
531     }

533     /**
534      * 601.2b If a cost that will be paid as the spell is being cast includes
535      * Phyrexian mana symbols, the player announces whether they intend to pay 2
536      * life or the corresponding colored mana cost for each of those symbols.
537      */
538     private void handlePhyrexianManaCosts(Game game, Player controller) {
539         Iterator<ManaCost> costIterator = manaCostsToPay.iterator();
540         while (costIterator.hasNext()) {
541             ManaCost cost = costIterator.next();
542 
543             if (!cost.isPhyrexian()) {
544                 continue;
545             }
546             PayLifeCost payLifeCost = new PayLifeCost(2);
547             if (payLifeCost.canPay(this, this, controller.getId(), game)
548                     && controller.chooseUse(Outcome.LoseLife, "Pay 2 life instead of " + cost.getText().replace("/P", "") + '?', this, game)) {
549                 costIterator.remove();
550                 costs.add(payLifeCost);
551                 manaCostsToPay.incrPhyrexianPaid();
552             }
553         }
554     }

558         GameEvent xEvent = GameEvent.getEvent(GameEvent.EventType.X_MANA_ANNOUNCE, this.getId(), this, getControllerId(), value);

563     /**
564      * Handles X mana costs and sets manaCostsToPay.
565      *
566      * @param game
567      * @param noMana
568      * @param controller
569      * @return variableManaCost for posting to log later
570      */
571     protected VariableManaCost handleManaXCosts(Game game, boolean noMana, Player controller) {
572         // 20210723 - 601.2b
573         // If the spell has alternative or additional costs that will
574         // be paid as it's being cast such as buyback or kicker costs (see rules 118.8 and 118.9),
575         // the player announces their intentions to pay any or all of those costs (see rule 601.2f).
576         // A player can't apply two alternative methods of casting or two alternative costs to a
577         // single spell. If the spell has a variable cost that will be paid as it's being cast
578         // (such as an {X} in its mana cost; see rule 107.3), the player announces the value of that
579         // variable. If the value of that variable is defined in the text of the spell by a choice
580         // that player would make later in the announcement or resolution of the spell, that player
581         // makes that choice at this time instead of that later time.
582 
583         // TODO: Handle announcing other variable costs here like: RemoveVariableCountersSourceCost
584         VariableManaCost variableManaCost = null;
585         for (ManaCost cost : manaCostsToPay) {
586             if (cost instanceof VariableManaCost) {
587                 if (variableManaCost == null) {
588                     variableManaCost = (VariableManaCost) cost;
589                 } else {
590                     // only one VariableManCost per spell (or is it possible to have more?)
591                     logger.error("Variable mana cost allowes only in one instance per ability: " + this);
592                 }
593             }
594         }
595         if (variableManaCost != null && !variableManaCost.isPaid()) { // should only happen for human players
596                 int xValue;
597                 int xValueMultiplier = handleManaXMultiplier(game, 1);
598                 if (!noMana || variableManaCost.getCostType().canUseAnnounceOnFreeCast()) {
599                     xValue = controller.announceXMana(variableManaCost.getMinX(), variableManaCost.getMaxX(), xValueMultiplier,
600                             "Announce the value for " + variableManaCost.getText(), game, this);
601                     int amountMana = xValue * variableManaCost.getXInstancesCount();
602                     StringBuilder manaString = threadLocalBuilder.get();
603                     if (variableManaCost.getFilter() == null || variableManaCost.getFilter().isGeneric()) {
604                         manaString.append('{').append(amountMana).append('}');
605                     } else {
606                         String manaSymbol = null;
607                         if (variableManaCost.getFilter().isBlack()) {
608                             if (variableManaCost.getFilter().isRed()) {
609                                 manaSymbol = "B/R";
610                             } else {
611                                 manaSymbol = "B";
612                             }
613                         } else if (variableManaCost.getFilter().isRed()) {
614                             manaSymbol = "R";
615                         } else if (variableManaCost.getFilter().isBlue()) {
616                             manaSymbol = "U";
617                         } else if (variableManaCost.getFilter().isGreen()) {
618                             manaSymbol = "G";
619                         } else if (variableManaCost.getFilter().isWhite()) {
620                             manaSymbol = "W";
621                         }
622                         if (manaSymbol == null) {
623                             throw new UnsupportedOperationException("ManaFilter is not supported: " + this);
624                         }
625                         for (int i = 0; i < amountMana; i++) {
626                             manaString.append('{').append(manaSymbol).append('}');
627                         }
628                     }
629                     manaCostsToPay.add(new ManaCostsImpl<>(manaString.toString()));
630                     manaCostsToPay.setX(xValue * xValueMultiplier, amountMana);
631                 }
632                 variableManaCost.setPaid();
633             
634         }
635 
636         return variableManaCost;
637     }

788     @Override
789     public String getRule(boolean all) {
790         StringBuilder sbRule = threadLocalBuilder.get();
791         if (all || this.abilityType != AbilityType.SPELL) { // TODO: Why the override for non-spells?
792             if (!manaCosts.isEmpty()) {
793                 sbRule.append(manaCosts.getText());
794             }
795             if (!costs.isEmpty()) {
796                 if (sbRule.length() > 0) {
797                     sbRule.append(", ");
798                 }
799                 sbRule.append(costs.getText());
800             }
801             if (sbRule.length() > 0) {
802                 sbRule.append(": ");
803             }
804         }
805 
806         String ruleStart = sbRule.toString();
807         String text = getModes().getText();
808         String rule;
809         if (!text.isEmpty()) {
810             if (ruleStart.length() > 1) {
811                 String end = ruleStart.substring(ruleStart.length() - 2).trim();
812                 if (end.isEmpty() || end.equals(":") || end.equals(".")) {
813                     rule = ruleStart + CardUtil.getTextWithFirstCharUpperCase(text);
814                 } else {
815                     rule = ruleStart + text;
816                 }
817             } else {
818                 rule = ruleStart + text;
819             }
820         } else {
821             rule = ruleStart;
822         }
823         String prefix;
824         if (this instanceof TriggeredAbility) {
825             prefix = null;
826         } else if (abilityWord != null) {
827             prefix = abilityWord.formatWord();
828         } else if (flavorWord != null) {
829             prefix = CardUtil.italicizeWithEmDash(flavorWord);
830         } else {
831             prefix = null;
832         }
833         if (prefix != null) {
834             rule = prefix + CardUtil.getTextWithFirstCharUpperCase(rule);
835         }
836         if (appendToRule != null) {
837             rule = rule.concat(appendToRule);
838         }
839         return rule;
840     }

965     protected static boolean canChooseTargetAbility(Ability ability, Modes modes, Game game, UUID controllerId) {
966         int found = 0;
967         for (Mode mode : modes.values()) {
968             boolean validTargets = true;
969             for (Target target : mode.getTargets()) {
970                 UUID abilityControllerId = controllerId;
971                 if (target.getTargetController() != null) {
972                     abilityControllerId = target.getTargetController();
973                 }
974                 if (!target.canChoose(abilityControllerId, ability, game)) {
975                     validTargets = false;
976                     break;
977                 }
978             }
979 
980             if (validTargets) {
981                 found++;
982                 if (modes.isEachModeMoreThanOnce()) {
983                     return true;
984                 }
985                 if (found >= modes.getMinModes()) {
986                     return true;
987                 }
988             }
989         }
990         return false;
991     }

998     @Override
999     public boolean isInUseableZone(Game game, MageObject source, GameEvent event) {
1000         if (!this.hasSourceObjectAbility(game, source, event)) {
1001             return false;
1002         }
1003         if (zone == Zone.COMMAND) {
1004             if (this.getSourceId() == null) { // commander effects
1005                 return true;
1006             }
1007             MageObject object = game.getObject(this.getSourceId());
1008             // emblem/planes are always actual
1009             if (object instanceof Emblem || object instanceof Dungeon || object instanceof Plane) {
1010                 return true;
1011             }
1012         }
1013 
1014         UUID parameterSourceId;
1015         // for singleton abilities like Flying we can't rely on abilities' source because it's only once in continuous effects
1016         // so will use the sourceId of the object itself that came as a parameter if it is not null
1017         if (this instanceof MageSingleton && source != null) {
1018             parameterSourceId = source.getId();
1019         } else {
1020             parameterSourceId = getSourceId();
1021         }
1022         // check against shortLKI for effects that move multiple object at the same time (e.g. destroy all)
1023         if (game.getShortLivingLKI(getSourceId(), getZone())) {
1024             return true;
1025         }
1026         // check against current state
1027         Zone test = game.getState().getZone(parameterSourceId);
1028         return zone.match(test);
1029     }

1032     public boolean hasSourceObjectAbility(Game game, MageObject source, GameEvent event) {
1033         // if source object have this ability
1034         // uses for ability.isInUseableZone
1035         // replacement and other continues effects can be without source, but active (must return true)
1036 
1037         MageObject object = source;
1038         // for singleton abilities like Flying we can't rely on abilities' source because it's only once in continuous effects
1039         // so will use the sourceId of the object itself that came as a parameter if it is not null
1040         if (object == null) {
1041             object = game.getPermanentEntering(getSourceId());
1042             if (object == null) {
1043                 object = game.getObject(getSourceId());
1044             }
1045         }
1046         if (object != null) {
1047             if (object instanceof Permanent) {
1048                 return object.hasAbility(this, game) && ((Permanent) object).isPhasedIn();
1049             } else {
1050                 // cards and other objects
1051                 return object.hasAbility(this, game);
1052             }
1053         }
1054         return true;
1055     }

1133     protected String getMessageText(Game game) {
1134         StringBuilder sb = threadLocalBuilder.get();
1135         MageObject object = game.getObject(this.sourceId);
1136         if (object != null) {
1137             if (object instanceof StackAbility) {
1138                 Card card = game.getCard(((StackAbility) object).getSourceId());
1139                 if (card != null) {
1140                     sb.append(GameLog.getColoredObjectIdName(card));
1141                 } else {
1142                     sb.append(GameLog.getColoredObjectIdName(object));
1143                 }
1144             } else if (object instanceof Spell) {
1145                 Spell spell = (Spell) object;
1146                 String castText = spell.getSpellCastText(game);
1147                 sb.append((castText.startsWith("Cast ") ? castText.substring(5) : castText));
1148                 if (spell.getFromZone() == Zone.GRAVEYARD) {
1149                     sb.append(" from graveyard");
1150                 }
1151                 sb.append(getOptionalTextSuffix(game, spell));
1152             } else {
1153                 sb.append(GameLog.getColoredObjectIdName(object));
1154             }
1155         } else {
1156             sb.append("unknown");
1157         }
1158         if (object instanceof Spell && ((Spell) object).getSpellAbilities().size() > 1) {
1159             if (((Spell) object).getSpellAbility().getSpellAbilityType() == SpellAbilityType.SPLIT_FUSED) {
1160                 Spell spell = (Spell) object;
1161                 int i = 0;
1162                 for (SpellAbility spellAbility : spell.getSpellAbilities()) {
1163                     i++;
1164                     String half;
1165                     if (i == 1) {
1166                         half = " left";
1167                     } else {
1168                         half = " right";
1169                     }
1170                     if (!spellAbility.getTargets().isEmpty()) {
1171                         sb.append(half).append(" half targeting ");
1172                         for (Target target : spellAbility.getTargets()) {
1173                             sb.append(target.getTargetedName(game));
1174                         }
1175                     }
1176                 }
1177             } else {
1178                 Spell spell = (Spell) object;
1179                 int i = 0;
1180                 for (SpellAbility spellAbility : spell.getSpellAbilities()) {
1181                     i++;
1182                     if (i > 1) {
1183                         sb.append(" splicing ");
1184                         if (spellAbility.name.length() > 5 && spellAbility.name.startsWith("Cast ")) {
1185                             sb.append(spellAbility.name.substring(5));
1186                         } else {
1187                             sb.append(spellAbility.name);
1188                         }
1189                     }
1190                     sb.append(getTargetDescriptionForLog(spellAbility.getTargets(), game));
1191                 }
1192             }
1193         } else if (object instanceof Spell && ((Spell) object).getSpellAbility().getModes().size() > 1) {
1194             Modes spellModes = ((Spell) object).getSpellAbility().getModes();
1195             for (UUID selectedModeId : spellModes.getSelectedModes()) {
1196                 Mode selectedMode = spellModes.get(selectedModeId);
1197                 int item = 0;
1198                 for (Mode mode : spellModes.values()) {
1199                     item++;
1200                     if (mode.getId().equals(selectedMode.getId())) {
1201                         sb.append(" (mode ").append(item).append(')');
1202                         sb.append(getTargetDescriptionForLog(selectedMode.getTargets(), game));
1203                         break;
1204                     }
1205                 }
1206             }
1207         } else {
1208             sb.append(getTargetDescriptionForLog(getTargets(), game));
1209         }
1210         return sb.toString();
1211     }

1422     public boolean isSameInstance(Ability ability) {
1423         // same instance (by mtg rules) = same object, ID or class+text (you can't check class only cause it can be different by params/text)
1424         if (ability == null) {
1425             return false;
1426         }
1427 
1428         return (this == ability)
1429                 || (this.getId().equals(ability.getId()))
1430                 || (this.getOriginalId().equals(ability.getOriginalId()))
1431                 || (this.getClass() == ability.getClass() && this.getRule(true).equals(ability.getRule(true)));
1432     }

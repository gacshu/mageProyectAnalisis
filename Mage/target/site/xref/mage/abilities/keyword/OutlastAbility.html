3   package mage.abilities.keyword;
4   
5   import mage.abilities.ActivatedAbilityImpl;
6   import mage.abilities.costs.Cost;
7   import mage.abilities.costs.common.TapSourceCost;
8   import mage.abilities.effects.common.counter.AddCountersSourceEffect;
9   import mage.constants.TimingRule;
10  import mage.constants.Zone;
11  import mage.counters.CounterType;
12  
13  /**
14   *
15   * @author LevelX2
16   */
17  public class OutlastAbility extends ActivatedAbilityImpl {
18  
19      public OutlastAbility(Cost cost) {
20          super(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.P1P1.createInstance()), cost);
21          this.addCost(new TapSourceCost());
22          this.timing = TimingRule.SORCERY;
23      }
24  
25      public OutlastAbility(final OutlastAbility ability) {
26          super(ability);
27      }
28  
29      @Override
30      public OutlastAbility copy() {
31          return new OutlastAbility(this);
32      }
33  
34      @Override
35      public String getRule() {
36          StringBuilder sb = new StringBuilder("Outlast ").append(manaCosts.getText());
37          sb.append(" <i>(").append(getManaCosts().getText()).append(", ").append(getCosts().getText()).append(":  Put a +1/+1 counter on this creature. Outlast only as a sorcery.)</i>");
38          return sb.toString();
39      }
40  
41  }

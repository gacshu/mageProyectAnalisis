1   package mage.abilities.keyword;
2   
3   import mage.abilities.StaticAbility;
4   import mage.abilities.common.AttacksTriggeredAbility;
5   import mage.abilities.common.delayed.AtTheEndOfCombatDelayedTriggeredAbility;
6   import mage.abilities.effects.common.CreateDelayedTriggeredAbilityEffect;
7   import mage.abilities.effects.common.SacrificeSourceEffect;
8   import mage.abilities.effects.common.combat.CantBlockSourceEffect;
9   import mage.constants.Duration;
10  import mage.constants.Zone;
11  
12  /**
13   * @author TheElk801
14   */
15  public class DecayedAbility extends StaticAbility {
16  
17      public DecayedAbility() {
18          super(Zone.BATTLEFIELD, new CantBlockSourceEffect(Duration.WhileOnBattlefield));
19          this.addSubAbility(new AttacksTriggeredAbility(new CreateDelayedTriggeredAbilityEffect(
20                  new AtTheEndOfCombatDelayedTriggeredAbility(new SacrificeSourceEffect())
21          ).setText("sacrifice it at end of combat")).setTriggerPhrase("When {this} attacks, "));
22      }
23  
24      private DecayedAbility(final DecayedAbility ability) {
25          super(ability);
26      }
27  
28      @Override
29      public DecayedAbility copy() {
30          return new DecayedAbility(this);
31      }
32  
33      @Override
34      public String getRule() {
35          return "decayed";
36      }
37  }

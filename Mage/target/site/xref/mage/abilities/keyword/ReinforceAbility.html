4   import mage.abilities.common.SimpleActivatedAbility;
5   import mage.abilities.costs.Cost;
6   import mage.abilities.costs.common.DiscardSourceCost;
7   import mage.abilities.dynamicvalue.DynamicValue;
8   import mage.abilities.dynamicvalue.common.StaticValue;
9   import mage.abilities.effects.common.counter.AddCountersTargetEffect;
10  import mage.constants.Zone;
11  import mage.counters.CounterType;
12  import mage.target.common.TargetCreaturePermanent;
13  import mage.util.CardUtil;
14  
15  public class ReinforceAbility extends SimpleActivatedAbility {
16      private DynamicValue count;
17      private Cost cost;
18  
19      public ReinforceAbility(int count, Cost cost) {
20          this(StaticValue.get(count), cost);
21      }
22  
23      public ReinforceAbility(DynamicValue count, Cost cost) {
24          super(Zone.HAND, new AddCountersTargetEffect(CounterType.P1P1.createInstance(0), count), cost);
25          this.addCost(new DiscardSourceCost());
26          this.addTarget(new TargetCreaturePermanent());
27          this.cost = cost.copy();
28          this.count = count;
29      }
30  
31  
32      public ReinforceAbility(final ReinforceAbility ability) {
33          super(ability);
34          this.cost = ability.cost.copy();
35          this.count = ability.count;
36      }
37  
38      @Override
39      public SimpleActivatedAbility copy() {
40          return new ReinforceAbility(this);
41      }
42  
43      @Override
44      public String getRule() {
45          StringBuilder sb = new StringBuilder("Reinforce ");
46          sb.append(count.toString()).append("&mdash;");
47          sb.append(cost.getText());
48          sb.append(" <i>(").append(cost.getText()).append(", Discard this card: Put ");
49          if (count.toString().equals("1")) {
50              sb.append("a +1/+1 counter");
51          } else {
52              sb.append(CardUtil.numberToText(count.toString())).append(" +1/+1 counters");
53          }
54          sb.append(" on target creature.)</i>");
55          return sb.toString();
56      }
57  }

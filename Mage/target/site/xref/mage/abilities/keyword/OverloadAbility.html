1   package mage.abilities.keyword;

3   import mage.abilities.SpellAbility;
4   import mage.abilities.costs.mana.ManaCosts;
5   import mage.abilities.effects.Effect;
6   import mage.cards.Card;
7   import mage.constants.SpellAbilityType;
8   import mage.constants.TimingRule;
9   
10  /**
11   * 702.94. Overload
12   * <p>
13   * 702.94a. Overload is a keyword that represents two static abilities: one that
14   * functions from any zone in which the spell with overload can be cast and
15   * another that functions while the card is on the stack. Overload [cost] means
16   * "You may choose to pay [cost] rather than pay this spell's mana cost" and "If
17   * you chose to pay this spell's overload cost, change its text by replacing all
18   * instances of the word 'target' with the word 'each.'" Using the overload
19   * ability follows the rules for paying alternative costs in rules 601.2b and
20   * 601.2e-g.
21   * <p>
22   * 702.94b. If a player chooses to pay the overload cost of a spell, that spell
23   * won't require any targets. It may affect objects that couldn't be chosen as
24   * legal targets if the spell were cast without its overload cost being paid.
25   * <p>
26   * 702.94c. Overload's second ability creates a text-changing effect. See rule
27   * 612, "Text-Changing Effects."
28   *
29   * @author LevelX2
30   */
31  public class OverloadAbility extends SpellAbility {
32  
33      public OverloadAbility(Card card, Effect effect, ManaCosts costs) {
34          super(costs, card.getName() + " with overload");
35          this.spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
36          this.addEffect(effect);
37          this.timing = (card.isSorcery(null) ? TimingRule.SORCERY : TimingRule.INSTANT);
38      }
39  
40      public OverloadAbility(final OverloadAbility ability) {
41          super(ability);
42      }
43  
44      @Override
45      public OverloadAbility copy() {
46          return new OverloadAbility(this);
47      }
48  
49      @Override
50      public String getRule(boolean all) {
51          return getRule();
52      }
53  
54      @Override
55      public String getRule() {
56          return "Overload " + getManaCostsToPay().getText() + " <i>(You may cast this spell for its overload cost. If you do, change its text by replacing all instances of \"target\" with \"each.\")</i>";
57      }
58  
59  }

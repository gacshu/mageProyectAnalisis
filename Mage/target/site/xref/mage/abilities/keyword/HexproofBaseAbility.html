17  /**
18   * an abstract base class for hexproof abilities
19   *
20   * @author TheElk801
21   */
22  public abstract class HexproofBaseAbility extends SimpleStaticAbility implements MageSingleton {
23  
24      HexproofBaseAbility() {
25          super(Zone.BATTLEFIELD, null);
26      }
27  
28      public abstract boolean checkObject(MageObject source, Game game);
29  
30      public static Set<HexproofBaseAbility> getFromColor(ObjectColor color) {
31          Set<HexproofBaseAbility> abilities = new HashSet<>();
32          if (color.isWhite()) {
33              abilities.add(HexproofFromWhiteAbility.getInstance());
34          }
35          if (color.isBlue()) {
36              abilities.add(HexproofFromBlueAbility.getInstance());
37          }
38          if (color.isBlack()) {
39              abilities.add(HexproofFromBlackAbility.getInstance());
40          }
41          if (color.isRed()) {
42              abilities.add(HexproofFromRedAbility.getInstance());
43          }
44          if (color.isGreen()) {
45              abilities.add(HexproofFromGreenAbility.getInstance());
46          }
47          return abilities;
48      }
49  
50      public static HexproofBaseAbility getFirstFromColor(ObjectColor color) {
51          if (color.isWhite()) {
52              return HexproofFromWhiteAbility.getInstance();
53          } else if (color.isBlue()) {
54              return HexproofFromBlueAbility.getInstance();
55          } else if (color.isBlack()) {
56              return HexproofFromBlackAbility.getInstance();
57          } else if (color.isRed()) {
58              return HexproofFromRedAbility.getInstance();
59          } else if (color.isGreen()) {
60              return HexproofFromGreenAbility.getInstance();
61          } else {
62              return null;
63          }
64      }
65  
66      public abstract String getCardIconHint(Game game);
67  
68      @Override
69      public List<CardIcon> getIcons(Game game) {
70          if (game == null) {
71              return new ArrayList<>(Collections.singletonList(
72                      HexproofAbilityIcon.instance
73              ));
74          }
75  
76          // dynamic icon (example: colored hexproof)
77          return new ArrayList<>(Collections.singletonList(
78                  HexproofAbilityIcon.createDynamicCardIcon(getCardIconHint(game))
79          ));
80      }
81  }

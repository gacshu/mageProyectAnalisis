24  /**
25   * 702.131. Assist
26   * <p>
27   * 702.131a Assist is a static ability that modifies the rules of paying for the
28   * spell with assist (see rules 601.2g-h). If the total cost to cast a spell
29   * with assist includes a generic mana component, before you activate mana
30   * abilities while casting it, you may choose another player. That player has a
31   * chance to activate mana abilities. Once that player chooses not to activate
32   * any more mana abilities, you have a chance to activate mana abilities. Before
33   * you begin to pay the total cost of the spell, the player you chose may pay
34   * for any amount of the generic mana in the spellâ€™s total cost.
35   *
36   * @author emerald000, JayDi85
37   */
38  public class AssistAbility extends SimpleStaticAbility implements AlternateManaPaymentAbility {
39  
40      private static final FilterPlayer filter = new FilterPlayer("another player");
41  
42      static {
43          filter.add(TargetController.NOT_YOU.getPlayerPredicate());
44      }
45  
46      public AssistAbility() {
47          super(Zone.ALL, null);
48          this.setRuleAtTheTop(true);
49      }
50  
51      public AssistAbility(final AssistAbility ability) {
52          super(ability);
53      }
54  
55      @Override
56      public AssistAbility copy() {
57          return new AssistAbility(this);
58      }
59  
60      @Override
61      public String getRule() {
62          return "Assist <i>(Another player can help pay the generic mana of this spell's cost.)</i>";
63      }
64  
65      @Override
66      public ActivationManaAbilityStep useOnActivationManaAbilityStep() {
67          return ActivationManaAbilityStep.BEFORE;
68      }
69  
70      @Override
71      public void addSpecialAction(Ability source, Game game, ManaCost unpaid) {
72          Player controller = game.getPlayer(source.getControllerId());
73          if (controller != null
74                  && source.getAbilityType() == AbilityType.SPELL
75                  && unpaid.getMana().getGeneric() >= 1
76                  && game.getState().getValue(source.getSourceId().toString() + game.getState().getZoneChangeCounter(source.getSourceId()) + "_assisted") == null) {
77              SpecialAction specialAction = new AssistSpecialAction(unpaid, this);
78              specialAction.setControllerId(source.getControllerId());
79              specialAction.setSourceId(source.getSourceId());
80              Target target = new TargetPlayer(1, 1, true, filter);
81              specialAction.addTarget(target);
82              if (specialAction.canActivate(source.getControllerId(), game).canActivate()) {
83                  game.getState().getSpecialActions().add(specialAction);
84              }
85          }
86      }
87  
88      @Override
89      public ManaOptions getManaOptions(Ability source, Game game, ManaCost unpaid) {
90          ManaOptions options = new ManaOptions();
91          if (unpaid.getMana().getGeneric() == 0) {
92              // nothing to pay
93              return options;
94          }
95  
96          // AI can't use assist (can't ask another player to help), maybe in teammode it can be enabled, but tests must works all the time
97          // Outcome.AIDontUseIt
98          Player controller = game.getPlayer(source.getControllerId());
99          if (controller != null && controller.isComputer()) {
100             return options;
101         }
102 
103         // search opponents who can help with generic pay
104         int opponentCanPayMax = 0;
105         for (UUID opponentId : game.getOpponents(source.getControllerId())) {
106             Player opponent = game.getPlayer(opponentId);
107             if (opponent != null) {
108                 // basic and pool, but no coditional mana
109                 ManaOptions availableMana = opponent.getManaAvailable(game);
110 //                availableMana.addMana(opponent.getManaPool().getMana());
111                 for (Mana mana : availableMana) {
112                     if (mana.count() > 0) {
113                         opponentCanPayMax = Math.max(opponentCanPayMax, mana.count());
114                     }
115                 }
116             }
117         }
118 
119         if (opponentCanPayMax > 0) {
120             options.addMana(Mana.GenericMana(Math.min(unpaid.getMana().getGeneric(), opponentCanPayMax)));
121         }
122 
123         return options;
124     }
125 }

146 class AssistEffect extends OneShotEffect {
147 
148     private final ManaCost unpaid;
149 
150     AssistEffect(ManaCost unpaid) {
151         super(Outcome.Benefit);
152         this.unpaid = unpaid;
153         this.staticText = "Assist <i>(Another player can help pay the generic mana of this spell's cost.)</i>";
154     }
155 
156     AssistEffect(final AssistEffect effect) {
157         super(effect);
158         this.unpaid = effect.unpaid;
159     }
160 
161     @Override
162     public AssistEffect copy() {
163         return new AssistEffect(this);
164     }
165 
166     @Override
167     public boolean apply(Game game, Ability source) {
168         Player controller = game.getPlayer(source.getControllerId());
169         Player targetPlayer = game.getPlayer(this.getTargetPointer().getFirst(game, source));
170         Spell spell = game.getStack().getSpell(source.getSourceId());
171         if (controller != null && spell != null && targetPlayer != null) {
172             // AI can't assist other players, maybe for teammates only (but tests must work as normal)
173             int amountToPay = 0;
174             if (!targetPlayer.isComputer()) {
175                 amountToPay = targetPlayer.announceXMana(0, unpaid.getMana().getGeneric(),
176                         "How much mana to pay as assist for " + controller.getName() + "?", game, source);
177             }
178 
179             if (amountToPay > 0) {
180                 Cost cost = ManaUtil.createManaCost(amountToPay, false);
181                 if (cost.pay(source, game, source, targetPlayer.getId(), false)) {
182                     ManaPool manaPool = controller.getManaPool();
183                     manaPool.addMana(Mana.ColorlessMana(amountToPay), game, source);
184                     manaPool.unlockManaType(ManaType.COLORLESS); // it's unlock mana for one use/click, but it can gives more
185                     game.informPlayers(targetPlayer.getLogName() + " paid {" + amountToPay + "} for " + controller.getLogName());
186                     game.getState().setValue(source.getSourceId().toString() + game.getState().getZoneChangeCounter(source.getSourceId()) + "_assisted", true);
187                 }
188 
189                 // assist must be used before activating mana abilities, so no need to switch step after usage
190                 // (mana and other special abilities can be used after assist)
191                 //spell.setCurrentActivatingManaAbilitiesStep(ActivationManaAbilityStep.NORMAL);
192             }
193             return true;
194         }
195         return false;
196     }

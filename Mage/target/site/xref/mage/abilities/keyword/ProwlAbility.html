1   package mage.abilities.keyword;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.condition.common.ProwlCondition;
5   import mage.abilities.costs.AlternativeSourceCostsImpl;
6   import mage.abilities.hint.common.ProwlHint;
7   import mage.cards.Card;
8   import mage.game.Game;
9   import mage.watchers.common.ProwlWatcher;
10  
11  /**
12   * 702.74. Prowl #
13   * <p>
14   * 702.74a Prowl is a static ability that functions on the stack. "Prowl [cost]"
15   * means "You may pay [cost] rather than pay this spell's mana cost if a player
16   * was dealt combat damage this turn by a source that, at the time it dealt that
17   * damage, was under your control and had any of this spell's creature types."
18   * Paying a spell's prowl cost follows the rules for paying alternative costs in
19   * rules 601.2b and 601.2e-g
20   *
21   * @author LevelX2
22   */
23  public class ProwlAbility extends AlternativeSourceCostsImpl {
24  
25      private static final String PROWL_KEYWORD = "Prowl";
26      private static final String reminderText = "You may cast this for its prowl cost if you dealt combat damage to a "
27              + "player this turn with a creature that shared a creature type with {this}";
28  
29      public ProwlAbility(Card card, String manaString) {
30          super(PROWL_KEYWORD, reminderText, manaString);
31          this.setRuleAtTheTop(true);
32          this.addWatcher(new ProwlWatcher());
33          this.addHint(ProwlHint.instance);
34      }
35  
36      private ProwlAbility(final ProwlAbility ability) {
37          super(ability);
38      }
39  
40      @Override
41      public ProwlAbility copy() {
42          return new ProwlAbility(this);
43      }
44  
45      @Override
46      public boolean isAvailable(Ability source, Game game) {
47          return ProwlCondition.instance.apply(game, source);
48      }
49  }

19  /**
20   * 702.102. Bestow
21   * <p>
22   * 702.102a Bestow represents two static abilities, one that functions while the
23   * card with bestow is on the stack and another that functions both while it's
24   * on stack and while it's on the battlefield. "Bestow [cost]" means "You may
25   * cast this card by paying [cost] rather than its mana cost." and "If you chose
26   * to pay this spell's bestow cost, it becomes an Aura enchantment and gains
27   * enchant creature. These effects last until one of two things happens: this
28   * spell has an illegal target as it resolves and or the permanent this spell
29   * becomes, becomes unattached." Paying a card's bestow cost follows the rules
30   * for paying alternative costs in rules 601.2b and 601.2e-g.
31   * <p>
32   * 702.102b If a spell's controller chooses to pay its bestow cost, that player
33   * chooses a legal target for that Aura spell as defined by its enchant creature
34   * ability and rule 601.2c. See also rule 303.4.
35   * <p>
36   * 702.102c A spell's controller can't choose to pay its bestow cost unless that
37   * player can choose a legal target for that spell after it becomes an Aura
38   * spell.
39   * <p>
40   * 702.102d As an Aura spell with bestow begins resolving, if its target is
41   * illegal, the effect making it an Aura spell ends. It continues resolving as a
42   * creature spell and will be put onto the battlefield under the control of the
43   * spell's controller. This is an exception to rule 608.3a.
44   * <p>
45   * 702.102e If an Aura with bestow is attached to an illegal object or player,
46   * it becomes unattached. This is an exception to rule 704.5n.
47   * <p>
48   * You don't choose whether the spell is going to be an Aura spell or not until
49   * the spell is already on the stack. Abilities that affect when you can cast a
50   * spell, such as flash, will apply to the creature card in whatever zone you're
51   * casting it from. For example, an effect that said you can cast creature
52   * spells as though they have flash will allow you to cast a creature card with
53   * bestow as an Aura spell anytime you could cast an instant.
54   * <p>
55   * On the stack, a spell with bestow is either a creature spell or an Aura
56   * spell. It's never both.
57   * <p>
58   * Unlike other Aura spells, an Aura spell with bestow isn't countered if its
59   * target is illegal as it begins to resolve. Rather, the effect making it an
60   * Aura spell ends, it loses enchant creature, it returns to being an
61   * enchantment creature spell, and it resolves and enters the battlefield as an
62   * enchantment creature.
63   * <p>
64   * Unlike other Auras, an Aura with bestow isn't put into its owner's graveyard
65   * if it becomes unattached. Rather, the effect making it an Aura ends, it loses
66   * enchant creature, and it remains on the battlefield as an enchantment
67   * creature. It can attack (and its {T} abilities can be activated, if it has
68   * any) on the turn it becomes unattached if it's been under your control
69   * continuously, even as an Aura, since your most recent turn began.
70   * <p>
71   * If a permanent with bestow enters the battlefield by any method other than
72   * being cast, it will be an enchantment creature. You can't choose to pay the
73   * bestow cost and have it become an Aura.
74   * <p>
75   * Auras attached to a creature don't become tapped when the creature becomes
76   * tapped. Except in some rare cases, an Aura with bestow remains untapped when
77   * it becomes unattached and becomes a creature.
78   *
79   * @author LevelX2
80   */
81  public class BestowAbility extends SpellAbility {
82  
83      public BestowAbility(Card card, String manaString) {
84          super(new ManaCostsImpl<>(manaString), card.getName() + " using bestow");
85          this.spellAbilityType = SpellAbilityType.BASE_ALTERNATE;
86          this.spellAbilityCastMode = SpellAbilityCastMode.BESTOW;
87          this.timing = TimingRule.SORCERY;
88          TargetPermanent auraTarget = new TargetCreaturePermanent();
89          this.addTarget(auraTarget);
90          this.addEffect(new AttachEffect(Outcome.BoostCreature));
91          Ability ability = new SimpleStaticAbility(Zone.BATTLEFIELD, new BestowEntersBattlefieldEffect());
92          ability.setRuleVisible(false);
93          addSubAbility(ability);
94      }
95  
96      public BestowAbility(final BestowAbility ability) {
97          super(ability);
98      }
99  
100     @Override
101     public BestowAbility copy() {
102         return new BestowAbility(this);
103     }
104 
105     @Override
106     public String getRule(boolean all) {
107         return getRule();
108     }
109 
110     @Override
111     public String getRule() {
112         return "Bestow " + getManaCostsToPay().getText() + " <i>(If you cast this card for its bestow cost, it's an Aura spell with enchant creature. It becomes a creature again if it's not attached to a creature.)</i>";
113     }
114 
115     static public void becomeCreature(Permanent permanent, Game game) {
116         // permanently changes to the object
117         if (permanent != null) {
118             MageObject basicObject = permanent.getBasicMageObject(game);
119             if (basicObject != null) {
120                 game.checkStateAndTriggered();  // Bug #8157
121                 basicObject.getSubtype().remove(SubType.AURA);
122                 basicObject.addCardType(CardType.CREATURE);
123             }
124             permanent.getSubtype().remove(SubType.AURA);
125             permanent.addCardType(CardType.CREATURE);
126         }
127     }
128 
129     static public void becomeAura(Card card) {
130         // permanently changes to the object
131         if (card != null) {
132             card.addSubType(SubType.AURA);
133             card.removeCardType(CardType.CREATURE);
134             card.addCardType(CardType.ENCHANTMENT);
135         }
136     }
137 }

139 class BestowEntersBattlefieldEffect extends ReplacementEffectImpl {
140 
141     public BestowEntersBattlefieldEffect() {
142         super(Duration.WhileOnBattlefield, Outcome.Neutral);
143     }
144 
145     public BestowEntersBattlefieldEffect(final BestowEntersBattlefieldEffect effect) {
146         super(effect);
147     }
148 
149     @Override
150     public boolean checksEventType(GameEvent event, Game game) {
151         return EventType.ENTERS_THE_BATTLEFIELD_SELF == event.getType();
152     }
153 
154     @Override
155     public boolean applies(GameEvent event, Ability source, Game game) {
156         return event.getTargetId().equals(source.getSourceId());
157     }
158 
159     @Override
160     public boolean replaceEvent(GameEvent event, Ability source, Game game) {
161         Permanent bestowPermanent = game.getPermanentEntering(source.getSourceId());
162         if (bestowPermanent == null || !bestowPermanent.hasSubtype(SubType.AURA, game)) {
163             return false;
164         }
165 
166         // change types permanently
167         MageObject basicObject = bestowPermanent.getBasicMageObject(game);
168         if (basicObject != null && !basicObject.getSubtype().contains(SubType.AURA)) {
169             basicObject.addSubType(SubType.AURA);
170             basicObject.removeCardType(CardType.CREATURE);
171         }
172         return false;
173     }
174 
175     @Override
176     public BestowEntersBattlefieldEffect copy() {
177         return new BestowEntersBattlefieldEffect(this);
178     }
179 
180 }

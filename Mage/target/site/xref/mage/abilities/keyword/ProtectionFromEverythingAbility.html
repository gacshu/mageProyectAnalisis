1   package mage.abilities.keyword;
2   
3   import mage.MageObject;
4   import mage.filter.FilterCard;
5   import mage.game.Game;
6   
7   /**
8    * @author TheElk801
9    */
10  public class ProtectionFromEverythingAbility extends ProtectionAbility {
11  
12      private static final FilterCard filter = new FilterCard("everything");
13  
14      public ProtectionFromEverythingAbility() {
15          super(filter);
16      }
17  
18      private ProtectionFromEverythingAbility(final ProtectionFromEverythingAbility ability) {
19          super(ability);
20      }
21  
22      @Override
23      public ProtectionFromEverythingAbility copy() {
24          return new ProtectionFromEverythingAbility(this);
25      }
26  
27      //     2/1/2009: "Protection from everything" means the following: Progenitus can't be blocked,
28      //               Progenitus can't be enchanted or equipped, Progenitus can't be the target of
29      //               spells or abilities, and all damage that would be dealt to Progenitus is prevented.
30      //     2/1/2009: Progenitus can still be affected by effects that don't target it or deal damage
31      //               to it (such as Day of Judgment).
32      @Override
33      public boolean canTarget(MageObject source, Game game) {
34          return false;
35      }
36  }

26  //
27  //    702.95. Scavenge
28  //
29  //    702.95a Scavenge is an activated ability that functions only while the card
30  //    with scavenge is in a graveyard. "Scavenge [cost]" means "[Cost], Exile this
31  //    card from your graveyard: Put a number of +1/+1 counters equal to this card's
32  //    power on target creature. Activate this ability only any time you could cast
33  //    a sorcery."
34  //
35  
36  public class ScavengeAbility extends ActivatedAbilityImpl {
37  
38      public ScavengeAbility(ManaCosts costs) {
39          super(Zone.GRAVEYARD, new ScavengeEffect(), costs);
40          this.timing = TimingRule.SORCERY;
41          this.addCost(new ExileSourceFromGraveCost());
42          this.addTarget(new TargetCreaturePermanent());
43      }
44  
45      public ScavengeAbility(final ScavengeAbility ability) {
46          super(ability);
47      }
48  
49      @Override
50      public ScavengeAbility copy() {
51          return new ScavengeAbility(this);
52      }
53  
54      @Override
55      public String getRule() {
56          return "Scavenge " + getManaCosts().getText() + " <i>(" + getManaCosts().getText() + ", Exile this card from your graveyard: Put a number of +1/+1 counter's equal to this card's power on target creature. Scavenge only as a sorcery.)</i>";
57      }
58  }
59  
60  class ScavengeEffect extends OneShotEffect {
61      ScavengeEffect() {
62          super(Outcome.BoostCreature);
63      }
64  
65      ScavengeEffect(final ScavengeEffect effect) {
66          super(effect);
67      }
68  
69      @Override
70      public boolean apply(Game game, Ability source) {
71          Card card = game.getCard(source.getSourceId());
72          if (card != null) {
73              int count = card.getPower().getValue();
74              if (count > 0) {
75                  Effect effect = new AddCountersTargetEffect(CounterType.P1P1.createInstance(count));
76                  effect.setTargetPointer(getTargetPointer());
77                  return effect.apply(game, source);
78              }
79          }
80  
81          return false;
82      }
83  
84      @Override
85      public ScavengeEffect copy() {
86          return new ScavengeEffect(this);
87      }
88  }

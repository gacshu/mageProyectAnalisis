22  /**
23   * 702.45. Splice
24   *
25   * 702.45a Splice is a static ability that functions while a card is in your
26   * hand. "Splice onto [subtype] [cost]" means "You may reveal this card from
27   * your hand as you cast a [subtype] spell. If you do, copy this card's text box
28   * onto that spell and pay [cost] as an additional cost to cast that spell."
29   * Paying a card's splice cost follows the rules for paying additional costs in
30   * rules 601.2b and 601.2e-g.
31   *
32   * Example: Since the card with splice remains in the player's hand, it can
33   * later be cast normally or spliced onto another spell. It can even be
34   * discarded to pay a "discard a card" cost of the spell it's spliced onto.
35   *
36   * 702.45b You can't choose to use a splice ability if you can't make the
37   * required choices (targets, etc.) for that card's instructions. You can't
38   * splice any one card onto the same spell more than once. If you're splicing
39   * more than one card onto a spell, reveal them all at once and choose the order
40   * in which their instructions will be followed. The instructions on the main
41   * spell have to be followed first.
42   *
43   * 702.45c The spell has the characteristics of the main spell, plus the text
44   * boxes of each of the spliced cards. The spell doesn't gain any other
45   * characteristics (name, mana cost, color, supertypes, card types, subtypes,
46   * etc.) of the spliced cards. Text copied onto the spell that refers to a card
47   * by name refers to the spell on the stack, not the card from which the text
48   * was copied.
49   *
50   * Example: Glacial Ray is a red card with splice onto Arcane that reads,
51   * "Glacial Ray deals 2 damage to any target." Suppose Glacial Ray is spliced
52   * onto Reach Through Mists, a blue spell. The spell is still blue, and Reach
53   * Through Mists deals the damage. This means that the ability can target a
54   * creature with protection from red and deal 2 damage to that creature.
55   *
56   * 702.45d Choose targets for the added text normally (see rule 601.2c). Note
57   * that a spell with one or more targets will be countered if all of its targets
58   * are illegal on resolution.
59   *
60   * 702.45e The spell loses any splice changes once it leaves the stack (for
61   * example, when it's countered, it's exiled, or it resolves).
62   *
63   * Rulings
64   *
65   * You must reveal all of the cards you intend to splice at the same time. Each
66   * individual card can only be spliced once onto a spell. If you have more than
67   * one card with the same name in your hand, you may splice both of them onto
68   * the spell. A card with a splice ability can't be spliced onto itself because
69   * the spell is on the stack (and not in your hand) when you reveal the cards
70   * you want to splice onto it. The target for a card that's spliced onto a spell
71   * may be the same as the target chosen for the original spell or for another
72   * spliced-on card. (A recent change to the targeting rules allows this, but
73   * most other cards are unaffected by the change.) If you splice a targeted card
74   * onto an untargeted spell, the entire spell will be countered if the target
75   * isn't legal when the spell resolves. If you splice an untargeted card onto a
76   * targeted spell, the entire spell will be countered if the target isn't legal
77   * when the spell resolves. A spell is countered on resolution only if *all* of
78   * its targets are illegal (or the spell is countered by an effect).
79   *
80   * @author LevelX2, awjackson
81   */
82  public class SpliceAbility extends SimpleStaticAbility {
83  
84      public static final FilterObject ARCANE = new FilterObject("Arcane");
85      public static final FilterObject INSTANT_OR_SORCERY = new FilterObject("instant or sorcery");
86  
87      static {
88          ARCANE.add(SubType.ARCANE.getPredicate());
89          ARCANE.setLockedFilter(true);
90          INSTANT_OR_SORCERY.add(Predicates.or(
91                  CardType.INSTANT.getPredicate(),
92                  CardType.SORCERY.getPredicate()
93          ));
94          INSTANT_OR_SORCERY.setLockedFilter(true);
95      }
96  
97      private final Costs<Cost> spliceCosts = new CostsImpl<>();
98      private final String rule;
99  
100     public SpliceAbility(FilterObject filter, String manaString) {
101         super(Zone.HAND, new SpliceCardEffectImpl(filter));
102         spliceCosts.add(new ManaCostsImpl<>(manaString));
103         rule = "Splice onto " + filter.getMessage() + ' ' + spliceCosts.getText() + getReminder(filter);
104     }
105 
106     public SpliceAbility(FilterObject filter, Cost cost) {
107         super(Zone.HAND, new SpliceCardEffectImpl(filter));
108         spliceCosts.add(cost);
109         rule = "Splice onto " + filter.getMessage() + "&mdash;" + spliceCosts.getText() + '.' + getReminder(filter);
110     }
111 
112     private SpliceAbility(final SpliceAbility ability) {
113         super(ability);
114         this.spliceCosts.addAll(ability.spliceCosts);
115         this.rule = ability.rule;
116     }
117 
118     @Override
119     public SimpleStaticAbility copy() {
120         return new SpliceAbility(this);
121     }
122 
123     public Costs getSpliceCosts() {
124         return spliceCosts;
125     }
126 
127     private static String getReminder(FilterObject filter) {
128         return " <i>(As you cast an " + filter.getMessage() + " spell, you may reveal this card from your hand and pay its splice cost. If you do, add this card's effects to that spell.)</i>";
129     }
130 
131     @Override
132     public String getRule() {
133         return rule;
134     }
135 }

137 class SpliceCardEffectImpl extends ContinuousEffectImpl implements SpliceCardEffect {
138 
139     private final FilterObject filter;
140 
141     public SpliceCardEffectImpl(FilterObject filter) {
142         super(Duration.WhileOnBattlefield, Outcome.Copy);
143         this.effectType = EffectType.SPLICE;
144         this.filter = filter;
145         staticText = "Splice onto " + filter;
146     }
147 
148     private SpliceCardEffectImpl(final SpliceCardEffectImpl effect) {
149         super(effect);
150         this.filter = effect.filter;
151     }
152 
153     @Override
154     public SpliceCardEffectImpl copy() {
155         return new SpliceCardEffectImpl(this);
156     }
157 
158     @Override
159     public final boolean apply(Game game, Ability source) {
160         return false;
161     }
162 
163     @Override
164     public boolean apply(Game game, Ability source, Ability abilityToModify) {
165         Player controller = game.getPlayer(source.getControllerId());
166         Card spliceCard = game.getCard(source.getSourceId());
167         if (spliceCard != null && controller != null) {
168             Spell spell = game.getStack().getSpell(abilityToModify.getId());
169             if (spell != null) {
170                 SpellAbility splicedAbility = spliceCard.getSpellAbility().copy();
171                 splicedAbility.setSpellAbilityType(SpellAbilityType.SPLICE);
172                 splicedAbility.setSourceId(abilityToModify.getSourceId());
173                 spell.addSpellAbility(splicedAbility);
174                 for (Iterator it = ((SpliceAbility) source).getSpliceCosts().iterator(); it.hasNext();) {
175                     Cost cost = (Cost) it.next();
176                     if (cost instanceof ManaCost) {
177                         spell.getSpellAbility().getManaCostsToPay().add((ManaCost) cost.copy());
178                     } else {
179                         spell.getSpellAbility().getCosts().add(cost.copy());
180                     }
181                 }
182             }
183             return true;
184         }
185         return false;
186     }
187 
188     @Override
189     public boolean applies(Ability abilityToModify, Ability source, Game game) {
190         if (!filter.match(game.getObject(abilityToModify.getSourceId()), game)) {
191             return false;
192         }
193         Card card = game.getCard(source.getSourceId());
194         if (card == null) {
195             return false;
196         }
197         return card.getSpellAbility().canSpliceOnto(abilityToModify, game);
198     }

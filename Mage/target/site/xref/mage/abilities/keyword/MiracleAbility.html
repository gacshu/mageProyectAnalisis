20  /**
21   * 702.92. Miracle
22   * <p>
23   * 702.92a Miracle is a static ability linked to a triggered ability (see rule
24   * 603.10). "Miracle [cost]" means "You may reveal this card from your hand as
25   * you draw it if it's the first card you've drawn this turn. When you reveal
26   * this card this way, you may cast it by paying [cost] rather than its mana
27   * cost."
28   * <p>
29   * 702.92b If a player chooses to reveal a card using its miracle ability, they
30   * play with that card revealed until that card leaves their hand, that ability
31   * resolves, or that ability otherwise leaves the stack.
32   * <p>
33   * You can cast a card for its miracle cost only as the miracle triggered
34   * ability resolves. If you don't want to cast it at that time (or you can't
35   * cast it, perhaps because there are no legal targets available), you won't be
36   * able to cast it later for the miracle cost.
37   * <p>
38   * RULINGS: You still draw the card, whether you use the miracle ability or not.
39   * Any ability that triggers whenever you draw a card, for example, will
40   * trigger. If you don't cast the card using its miracle ability, it will remain
41   * in your hand.
42   * <p>
43   * You can reveal and cast a card with miracle on any turn, not just your own,
44   * if it's the first card you've drawn that turn.
45   * <p>
46   * You don't have to reveal a drawn card with miracle if you don't wish to cast
47   * it at that time.
48   * <p>
49   * You can cast a card for its miracle cost only as the miracle triggered
50   * ability resolves. If you don't want to cast it at that time (or you can't
51   * cast it, perhaps because there are no legal targets available), you won't be
52   * able to cast it later for the miracle cost.
53   * <p>
54   * You cast the card with miracle during the resolution of the triggered
55   * ability. Ignore any timing restrictions based on the card's type.
56   * <p>
57   * It's important to reveal a card with miracle before it is mixed with the
58   * other cards in your hand.
59   * <p>
60   * Multiple card draws are always treated as a sequence of individual card
61   * draws. For example, if you haven't drawn any cards yet during a turn and cast
62   * a spell that instructs you to draw three cards, you'll draw them one at a
63   * time. Only the first card drawn this way may be revealed and cast using its
64   * miracle ability.
65   * <p>
66   * If the card with miracle leaves your hand before the triggered ability
67   * resolves, you won't be able to cast it using its miracle ability.
68   * <p>
69   * You draw your opening hand before any turn begins. Cards you draw for your
70   * opening hand can't be cast using miracle.
71   *
72   * @author noxx, LevelX2
73   */
74  public class MiracleAbility extends TriggeredAbilityImpl {
75  
76      private static final String staticRule = " <i>(You may cast this card for its miracle cost when you draw it if it's the first card you drew this turn.)</i>";
77      private final String ruleText;
78  
79      public MiracleAbility(String miracleCosts) {
80          super(Zone.HAND, new MiracleEffect(miracleCosts), true);
81          addWatcher(new MiracleWatcher());
82          ruleText = "Miracle " + miracleCosts + staticRule;
83      }
84  
85      private MiracleAbility(final MiracleAbility ability) {
86          super(ability);
87          this.ruleText = ability.ruleText;
88      }
89  
90      @Override
91      public MiracleAbility copy() {
92          return new MiracleAbility(this);
93      }
94  
95      @Override
96      public boolean checkEventType(GameEvent event, Game game) {
97          return event.getType() == GameEvent.EventType.MIRACLE_CARD_REVEALED;
98      }
99  
100     @Override
101     public boolean checkTrigger(GameEvent event, Game game) {
102         if (event.getTargetId().equals(this.getSourceId())) {
103             // Refer to the card at the zone it is now (hand)
104             getEffects().setTargetPointer(new FixedTarget(game.getCard(event.getTargetId()), game));
105             return true;
106         }
107         return false;
108     }
109 
110     @Override
111     public String getRule() {
112         return ruleText;
113     }
114 }
115 
116 class MiracleEffect extends OneShotEffect {
117 
118     private final ManaCosts<ManaCost> miracleCosts;
119 
120     public MiracleEffect(String miracleCosts) {
121         super(Outcome.Benefit);
122         this.staticText = "cast this card for its miracle cost";
123         this.miracleCosts = new ManaCostsImpl<>(miracleCosts);
124     }
125 
126     public MiracleEffect(final MiracleEffect effect) {
127         super(effect);
128         this.miracleCosts = effect.miracleCosts;
129     }
130 
131     @Override
132     public MiracleEffect copy() {
133         return new MiracleEffect(this);
134     }
135 
136     @Override
137     public boolean apply(Game game, Ability source) {
138         Player controller = game.getPlayer(source.getControllerId());
139         // use target pointer here, so it's the same card that triggered the event (not gone back to library e.g.)
140         Card card = game.getCard(getTargetPointer().getFirst(game, source));
141         if (controller != null && card != null) {
142             SpellAbility abilityToCast = card.getSpellAbility().copy();
143             ManaCosts<ManaCost> costRef = abilityToCast.getManaCostsToPay();
144             // replace with the new cost
145             costRef.clear();
146             costRef.add(miracleCosts);
147             controller.cast(abilityToCast, game, false, new ApprovingObject(source, game));
148             return true;
149         }
150         return false;
151     }
152 }

2   package mage.abilities.keyword;

4   import java.util.UUID;
5   import mage.abilities.TriggeredAbilityImpl;
6   import mage.abilities.effects.common.counter.AddCountersSourceEffect;
7   import mage.constants.Zone;
8   import mage.counters.CounterType;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  import mage.players.Player;
12  
13  /**
14   * Dethrone triggers whenever a creature with dethrone attacks the player with
15   * the most life or tied for the most life. When the ability resolves, you put a
16   * +1/+1 counter on the creature. This happens before blockers are declared.
17   * Once the ability triggers, it doesn't matter what happens to anybody's life
18   * total. If the defending player doesn't have the most life when the ability
19   * resolves, the creature will still get the +1/+1 counter. Note that dethrone
20   * won't trigger if the creature attacks a Planeswalker. You're going after the
21   * crown, after all, not the royal advisors. If you have the most life, your
22   * dethrone abilities won't trigger, but you may find a few choice ways to avoid
23   * that situation.
24   *
25   * @author LevelX2
26   */
27  public class DethroneAbility extends TriggeredAbilityImpl {
28  
29      public DethroneAbility() {
30          super(Zone.BATTLEFIELD, new AddCountersSourceEffect(CounterType.P1P1.createInstance()), false);
31      }
32  
33      public DethroneAbility(final DethroneAbility ability) {
34          super(ability);
35      }
36  
37      @Override
38      public DethroneAbility copy() {
39          return new DethroneAbility(this);
40      }
41  
42      @Override
43      public boolean checkEventType(GameEvent event, Game game) {
44          return event.getType() == GameEvent.EventType.DECLARED_ATTACKERS;
45      }
46  
47      @Override
48      public boolean checkTrigger(GameEvent event, Game game) {
49          UUID defenderId = game.getCombat().getDefenderId(getSourceId());
50          if (defenderId != null) {
51              Player attackedPlayer = game.getPlayer(defenderId);
52              Player controller = game.getPlayer(getControllerId());
53              if (attackedPlayer != null && controller != null) {
54                  int mostLife = Integer.MIN_VALUE;
55                  for (UUID playerId : game.getState().getPlayersInRange(controller.getId(), game)) {
56                      Player player = game.getPlayer(playerId);
57                      if (player != null) {
58                          if (player.getLife() > mostLife) {
59                              mostLife = player.getLife();
60                          }
61                      }
62                  }
63                  return attackedPlayer.getLife() == mostLife;
64              }
65          }
66          return false;
67      }
68  
69      @Override
70      public String getRule() {
71          return "Dethrone (<i>Whenever this creature attacks the player with the most life or tied for most life, put a +1/+1 counter on it.</i>)";
72      }
73  }

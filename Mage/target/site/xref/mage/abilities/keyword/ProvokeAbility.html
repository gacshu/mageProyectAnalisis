17  /**
18   * 702.38. Provoke 702.38a Provoke is a triggered ability. “Provoke” means
19   * “Whenever this creature attacks, you may choose to have target creature
20   * defending player controls block this creature this combat if able. If you do,
21   * untap that creature.” 702.38b If a creature has multiple instances of
22   * provoke, each triggers separately.
23   *
24   * @author LevelX2
25   *
26   */
27  public class ProvokeAbility extends AttacksTriggeredAbility {
28  
29      public ProvokeAbility() {
30          this("Provoke <i>(Whenever this attacks, you may have target creature defending player controls untap and block it if able.)</i>");
31      }
32  
33      public ProvokeAbility(String text) {
34          super(new UntapTargetEffect(), true, text);
35          this.addEffect(new ProvokeRequirementEffect());
36      }
37  
38      public ProvokeAbility(final ProvokeAbility ability) {
39          super(ability);
40      }
41  
42      @Override
43      public boolean checkTrigger(GameEvent event, Game game) {
44          if (super.checkTrigger(event, game)) {
45              FilterCreaturePermanent filter = new FilterCreaturePermanent("creature defending player controls");
46              UUID defendingPlayerId = game.getCombat().getDefendingPlayerId(sourceId, game);
47              filter.add(new ControllerIdPredicate(defendingPlayerId));
48              this.getTargets().clear();
49              TargetCreaturePermanent target = new TargetCreaturePermanent(filter);
50              this.addTarget(target);
51              return true;
52          }
53          return false;
54      }
55  
56      @Override
57      public ProvokeAbility copy() {
58          return new ProvokeAbility(this);
59      }
60  
61  }
62  
63  class ProvokeRequirementEffect extends RequirementEffect {
64  
65      public ProvokeRequirementEffect() {
66          this(Duration.EndOfTurn);
67      }
68  
69      public ProvokeRequirementEffect(Duration duration) {
70          super(duration);
71          staticText = "and blocks {this} this turn if able";
72      }
73  
74      public ProvokeRequirementEffect(final ProvokeRequirementEffect effect) {
75          super(effect);
76      }
77  
78      @Override
79      public boolean applies(Permanent permanent, Ability source, Game game) {
80          return permanent.getId().equals(targetPointer.getFirst(game, source));
81      }
82  
83      @Override
84      public boolean mustAttack(Game game) {
85          return false;
86      }
87  
88      @Override
89      public boolean mustBlock(Game game) {
90          return true;
91      }
92  
93      @Override
94      public UUID mustBlockAttacker(Ability source, Game game) {
95          return source.getSourceId();
96      }
97  
98      @Override
99      public ProvokeRequirementEffect copy() {
100         return new ProvokeRequirementEffect(this);
101     }
102 
103 }

32  public class EncoreAbility extends ActivatedAbilityImpl {
33  
34      private final String rule;
35  
36      public EncoreAbility(Cost cost) {
37          super(Zone.GRAVEYARD, new EncoreEffect(), cost);
38          this.addCost(new ExileSourceFromGraveCost());
39          this.timing = TimingRule.SORCERY;
40          this.rule = setRule(cost);
41      }
42  
43      public EncoreAbility(final EncoreAbility ability) {
44          super(ability);
45          this.rule = ability.rule;
46      }
47  
48      @Override
49      public EncoreAbility copy() {
50          return new EncoreAbility(this);
51      }
52  
53      @Override
54      public String getRule() {
55          return rule;
56      }
57  
58      private String setRule(Cost cost) {
59          return "Encore " + cost.getText() + " <i>(" + cost.getText() + ", " +
60                  "Exile this card from your graveyard: For each opponent, " +
61                  "create a token copy that attacks that opponent this turn if able. " +
62                  "They gain haste. Sacrifice them at the beginning of the next end step. " +
63                  "Activate only as a sorcery)</i>";
64      }
65  }

67  class EncoreEffect extends OneShotEffect {
68  
69      EncoreEffect() {
70          super(Outcome.PutCreatureInPlay);
71      }
72  
73      private EncoreEffect(final EncoreEffect effect) {
74          super(effect);
75      }
76  
77      @Override
78      public EncoreEffect copy() {
79          return new EncoreEffect(this);
80      }
81  
82      @Override
83      public boolean apply(Game game, Ability source) {
84          Card card = game.getCard(source.getSourceId());
85          if (card == null) {
86              return false;
87          }
88          EmptyToken token = new EmptyToken();
89          CardUtil.copyTo(token).from(card, game);
90          Set<MageObjectReference> addedTokens = new HashSet<>();
91          int opponentCount = OpponentsCount.instance.calculate(game, source, this);
92          if (opponentCount < 1) {
93              return false;
94          }
95          token.putOntoBattlefield(opponentCount, game, source, source.getControllerId());
96          Iterator<UUID> it = token.getLastAddedTokenIds().iterator();
97          while (it.hasNext()) {
98              for (UUID playerId : game.getOpponents(source.getControllerId())) {
99                  if (game.getPlayer(playerId) == null) {
100                     continue;
101                 }
102                 UUID tokenId = it.next();
103                 MageObjectReference mageObjectReference = new MageObjectReference(tokenId, game);
104                 game.addEffect(new EncoreRequirementEffect(
105                         mageObjectReference, playerId
106                 ), source);
107                 game.addEffect(new GainAbilityTargetEffect(
108                         HasteAbility.getInstance(), Duration.Custom
109                 ).setTargetPointer(new FixedTarget(mageObjectReference)), source);
110                 addedTokens.add(mageObjectReference);
111             }
112         }
113         game.addDelayedTriggeredAbility(new AtTheBeginOfNextEndStepDelayedTriggeredAbility(
114                 new EncoreSacrificeEffect(addedTokens)
115         ), source);
116         return true;
117     }
118 }

120 class EncoreRequirementEffect extends RequirementEffect {
121 
122     private final MageObjectReference token;
123     private final UUID playerId;
124 
125     EncoreRequirementEffect(MageObjectReference token, UUID playerId) {
126         super(Duration.EndOfTurn);
127         this.token = token;
128         this.playerId = playerId;
129     }
130 
131     private EncoreRequirementEffect(final EncoreRequirementEffect effect) {
132         super(effect);
133         this.token = effect.token;
134         this.playerId = effect.playerId;
135     }
136 
137     @Override
138     public EncoreRequirementEffect copy() {
139         return new EncoreRequirementEffect(this);
140     }
141 
142     @Override
143     public boolean applies(Permanent permanent, Ability source, Game game) {
144         return token.refersTo(permanent, game);
145     }
146 
147     @Override
148     public UUID mustAttackDefender(Ability source, Game game) {
149         return playerId;
150     }
151 
152     @Override
153     public boolean mustAttack(Game game) {
154         return true;
155     }
156 
157     @Override
158     public boolean mustBlock(Game game) {
159         return false;
160     }
161 }

163 class EncoreSacrificeEffect extends OneShotEffect {
164 
165     private final Set<MageObjectReference> mageObjectReferenceSet = new HashSet<>();
166 
167     EncoreSacrificeEffect(Set<MageObjectReference> mageObjectReferenceSet) {
168         super(Outcome.Benefit);
169         this.mageObjectReferenceSet.addAll(mageObjectReferenceSet);
170         staticText = "sacrifice those tokens";
171     }
172 
173     private EncoreSacrificeEffect(final EncoreSacrificeEffect effect) {
174         super(effect);
175         this.mageObjectReferenceSet.addAll(effect.mageObjectReferenceSet);
176     }
177 
178     @Override
179     public EncoreSacrificeEffect copy() {
180         return new EncoreSacrificeEffect(this);
181     }
182 
183     @Override
184     public boolean apply(Game game, Ability source) {
185         for (MageObjectReference mageObjectReference : mageObjectReferenceSet) {
186             Permanent permanent = mageObjectReference.getPermanent(game);
187             if (permanent != null) {
188                 permanent.sacrifice(source, game);
189             }
190         }
191         return true;
192     }

1   package mage.abilities.costs.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.Cost;
5   import mage.abilities.costs.SacrificeCost;
6   import mage.abilities.costs.UseAttachedCost;
7   import mage.game.Game;
8   import mage.game.permanent.Permanent;
9   
10  import java.util.UUID;
11  
12  /**
13   * @author TheElk801
14   */
15  public class SacrificeAttachmentCost extends UseAttachedCost implements SacrificeCost {
16  
17      public SacrificeAttachmentCost() {
18          super();
19      }
20  
21      public SacrificeAttachmentCost(final SacrificeAttachmentCost cost) {
22          super(cost);
23      }
24  
25      @Override
26      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
27          if (mageObjectReference == null) {
28              return false;
29          }
30          Permanent permanent = game.getPermanent(source.getSourceId());
31          if (permanent == null) {
32              return paid;
33          }
34          for (UUID attachmentId : permanent.getAttachments()) {
35              if (!this.mageObjectReference.refersTo(attachmentId, game)) {
36                  continue;
37              }
38              Permanent attachment = game.getPermanent(attachmentId);
39              paid = attachment != null
40                      && attachment.isControlledBy(controllerId)
41                      && attachment.sacrifice(source, game);
42              if (paid) {
43                  break;
44              }
45          }
46  
47          return paid;
48      }
49  
50      @Override
51      public SacrificeAttachmentCost copy() {
52          return new SacrificeAttachmentCost(this);
53      }
54  
55      @Override
56      public String getText() {
57          return "sacrifice " + this.name;
58      }
59  }

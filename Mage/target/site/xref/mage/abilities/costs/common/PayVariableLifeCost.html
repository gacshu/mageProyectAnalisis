1   package mage.abilities.costs.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.Cost;
5   import mage.abilities.costs.VariableCostImpl;
6   import mage.abilities.costs.VariableCostType;
7   import mage.game.Game;
8   import mage.players.Player;
9   
10  /**
11   * @author LevelX2
12   */
13  public class PayVariableLifeCost extends VariableCostImpl {
14  
15      public PayVariableLifeCost() {
16          this(false);
17      }
18  
19      public PayVariableLifeCost(boolean useAsAdditionalCost) {
20          super(useAsAdditionalCost ? VariableCostType.ADDITIONAL : VariableCostType.NORMAL,
21                  "life to pay");
22          this.text = new StringBuilder(useAsAdditionalCost ? "As an additional cost to cast this spell, pay " : "Pay ")
23                  .append(xText).append(' ').append("life").toString();
24      }
25  
26      public PayVariableLifeCost(final PayVariableLifeCost cost) {
27          super(cost);
28      }
29  
30      @Override
31      public PayVariableLifeCost copy() {
32          return new PayVariableLifeCost(this);
33      }
34  
35      @Override
36      public Cost getFixedCostsFromAnnouncedValue(int xValue) {
37          return new PayLifeCost(xValue);
38      }
39  
40      @Override
41      public int getMaxValue(Ability source, Game game) {
42          int maxValue = 0;
43          Player controller = game.getPlayer(source.getControllerId());
44          if (controller != null) {
45              // Paying 0 life is not considered paying any life, so paying 0 is still allowed
46              if (controller.canPayLifeCost(source)) {
47                  maxValue = controller.getLife();
48              }
49          }
50          return Math.max(0, maxValue);
51      }
52  
53  }

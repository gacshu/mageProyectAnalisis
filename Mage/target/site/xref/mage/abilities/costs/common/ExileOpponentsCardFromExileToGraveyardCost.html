21  public class ExileOpponentsCardFromExileToGraveyardCost extends CostImpl {
22  
23      public ExileOpponentsCardFromExileToGraveyardCost(boolean thatPlayersText) {
24          if (!thatPlayersText) {
25              this.text = "put a card an opponent owns from exile into its owner's graveyard";
26          } else {
27              this.text = "put a card an opponent owns from exile into that player's graveyard";
28          }
29      }
30  
31      public ExileOpponentsCardFromExileToGraveyardCost(ExileOpponentsCardFromExileToGraveyardCost cost) {
32          super(cost);
33      }
34  
35      @Override
36      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
37          Player controller = game.getPlayer(controllerId);
38          if (controller != null) {
39              FilterCard filter = new FilterCard();
40              filter.add(TargetController.OPPONENT.getOwnerPredicate());
41              Target target = new TargetCardInExile(filter);
42              if (controller.chooseTarget(Outcome.Damage, target, ability, game)) {
43                  Card card = game.getCard(target.getFirstTarget());
44                  if (card != null) {
45                      paid = true;
46                      controller.moveCards(card, Zone.GRAVEYARD, ability, game);
47                  }
48              }
49          }
50          return paid;
51      }
52  
53      @Override
54      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
55          Player controller = game.getPlayer(controllerId);
56          if (controller != null) {
57              for (Card card : game.getExile().getAllCards(game)) {
58                  if (controller.hasOpponent(card.getOwnerId(), game)) {
59                      return true;
60                  }
61              }
62          }
63          return false;
64      }
65  
66      @Override
67      public ExileOpponentsCardFromExileToGraveyardCost copy() {
68          return new ExileOpponentsCardFromExileToGraveyardCost(this);
69      }
70  
71  }

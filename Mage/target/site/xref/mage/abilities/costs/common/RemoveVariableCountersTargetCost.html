12  import mage.target.TargetPermanent;
13  
14  import java.util.UUID;

16  /**
17   * @author LevelX
18   */
19  public class RemoveVariableCountersTargetCost extends VariableCostImpl {
20  
21      protected FilterPermanent filter;
22      protected CounterType counterTypeToRemove;
23      protected int minValue;
24  
25      public RemoveVariableCountersTargetCost(FilterPermanent filter) {
26          this(filter, null);
27      }
28  
29      public RemoveVariableCountersTargetCost(FilterPermanent filter, CounterType counterTypeToRemove) {
30          this(filter, counterTypeToRemove, "X", 0);
31      }
32  
33      public RemoveVariableCountersTargetCost(FilterPermanent filter, CounterType counterTypeToRemove, String xText, int minValue) {
34          this(filter, counterTypeToRemove, xText, minValue, null);
35      }
36  
37      public RemoveVariableCountersTargetCost(FilterPermanent filter, CounterType counterTypeToRemove, String xText, int minValue, String text) {
38          super(VariableCostType.NORMAL, xText, new StringBuilder(counterTypeToRemove != null ? counterTypeToRemove.getName() + ' ' : "").append("counters to remove").toString());
39          this.filter = filter;
40          this.counterTypeToRemove = counterTypeToRemove;
41          if (text != null && !text.isEmpty()) {
42              this.text = text;
43          } else {
44              this.text = setText();
45          }
46          this.minValue = minValue;
47      }
48  
49      public RemoveVariableCountersTargetCost(final RemoveVariableCountersTargetCost cost) {
50          super(cost);
51          this.filter = cost.filter;
52          this.counterTypeToRemove = cost.counterTypeToRemove;
53          this.minValue = cost.minValue;
54      }
55  
56      @Override
57      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
58          return paid;
59      }
60  
61      private String setText() {
62          StringBuilder sb = new StringBuilder("Remove ").append(xText);
63          if (counterTypeToRemove != null) {
64              sb.append(' ').append(counterTypeToRemove.getName());
65          }
66          sb.append(" counters from among ").append(filter.getMessage());
67          return sb.toString();
68      }
69  
70      @Override
71      public RemoveVariableCountersTargetCost copy() {
72          return new RemoveVariableCountersTargetCost(this);
73      }
74  
75      @Override
76      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
77          return getMaxValue(source, game) >= minValue;
78      }
79  
80      @Override
81      public int getMinValue(Ability source, Game game) {
82          return minValue;
83      }
84  
85      @Override
86      public int getMaxValue(Ability source, Game game) {
87          int maxValue = 0;
88          for (Permanent permanent : game.getBattlefield().getAllActivePermanents(filter, source.getControllerId(), game)) {
89              if (counterTypeToRemove != null) {
90                  maxValue += permanent.getCounters(game).getCount(counterTypeToRemove);
91              } else {
92                  for (Counter counter : permanent.getCounters(game).values()) {
93                      maxValue += counter.getCount();
94                  }
95              }
96          }
97          return maxValue;
98      }
99  
100     @Override
101     public Cost getFixedCostsFromAnnouncedValue(int xValue) {
102         return new RemoveCounterCost(new TargetPermanent(minValue, Integer.MAX_VALUE, filter, true), counterTypeToRemove, xValue);
103     }
104 
105 }

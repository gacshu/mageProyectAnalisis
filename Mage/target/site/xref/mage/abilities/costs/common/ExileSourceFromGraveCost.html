1   package mage.abilities.costs.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.Cost;
5   import mage.abilities.costs.CostImpl;
6   import mage.cards.Card;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.players.Player;
10  
11  import java.util.UUID;
12  
13  /**
14   * @author magenoxx_at_gmail.com
15   */
16  public class ExileSourceFromGraveCost extends CostImpl {
17  
18      public ExileSourceFromGraveCost() {
19          this.text = "exile {this} from your graveyard";
20      }
21  
22      private ExileSourceFromGraveCost(final ExileSourceFromGraveCost cost) {
23          super(cost);
24      }
25  
26      @Override
27      public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
28          Player controller = game.getPlayer(controllerId);
29          if (controller == null) {
30              return paid;
31          }
32          Card card = game.getCard(source.getSourceId());
33          if (card != null && game.getState().getZone(source.getSourceId()) == Zone.GRAVEYARD) {
34              controller.moveCards(card, Zone.EXILED, source, game);
35              // 117.11. The actions performed when paying a cost may be modified by effects.
36              // Even if they are, meaning the actions that are performed don't match the actions
37              // that are called for, the cost has still been paid.
38              // so return state here is not important because the user indended to exile the target anyway
39              paid = true;
40          }
41          return paid;
42      }
43  
44      @Override
45      public boolean canPay(Ability ability, Ability source, UUID controllerId, Game game) {
46          return game.getCard(source.getSourceId()) != null
47                  && game.getState().getZone(source.getSourceId()) == Zone.GRAVEYARD;
48      }
49  
50      @Override
51      public ExileSourceFromGraveCost copy() {
52          return new ExileSourceFromGraveCost(this);
53      }
54  }

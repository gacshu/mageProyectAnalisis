1   package mage.abilities.costs;

3   import mage.abilities.Ability;
4   import mage.game.Game;
5   
6   /**
7    * @author BetaSteward_at_googlemail.com
8    */
9   public interface VariableCost {
10      /**
11       * Returns the variable amount if already set
12       *
13       * @return
14       */
15      int getAmount();
16  
17      /**
18       * Sets the variable amount
19       *
20       * @param xValue  - value of X
21       * @param xPay    - total value of pays for X (X * xMultiplier * xInstancesCount)
22       * @param isPayed - is that was real payed or just value setup
23       */
24      void setAmount(int xValue, int xPay, boolean isPayed);
25  
26      /**
27       * returns the action text (e.g. "creature cards to exile from your hand", "life to pay")
28       *
29       * @return
30       */
31      String getActionText();
32  
33      /**
34       * Return a min value to announce
35       *
36       * @param source
37       * @param game
38       * @return
39       */
40      int getMinValue(Ability source, Game game);
41  
42      /**
43       * Returns a max value to announce
44       *
45       * @param source
46       * @param game
47       * @return
48       */
49      int getMaxValue(Ability source, Game game);
50  
51      /**
52       * Asks the controller to announce the variable value
53       *
54       * @param source
55       * @param game
56       * @return
57       */
58      int announceXValue(Ability source, Game game);
59  
60      /**
61       * Returns a fixed cost with the announced variable value
62       *
63       * @param xValue
64       * @return
65       */
66      Cost getFixedCostsFromAnnouncedValue(int xValue);
67  
68      VariableCostType getCostType();
69  
70      void setCostType(VariableCostType costType);
71  }

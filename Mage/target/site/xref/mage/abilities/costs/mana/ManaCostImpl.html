114     protected boolean assignColored(Ability ability, Game game, ManaPool pool, ColoredManaSymbol mana, Cost costToPay) {
115         // first check special mana
116         switch (mana) {
117             case W:
118                 if (pool.pay(ManaType.WHITE, ability, sourceFilter, game, costToPay, usedManaToPay)) {
119                     this.payment.increaseWhite();
120                     return true;
121                 }
122                 break;
123             case U:
124                 if (pool.pay(ManaType.BLUE, ability, sourceFilter, game, costToPay, usedManaToPay)) {
125                     this.payment.increaseBlue();
126                     return true;
127                 }
128                 break;
129             case B:
130                 if (pool.pay(ManaType.BLACK, ability, sourceFilter, game, costToPay, usedManaToPay)) {
131                     this.payment.increaseBlack();
132                     return true;
133                 }
134                 break;
135             case R:
136                 if (pool.pay(ManaType.RED, ability, sourceFilter, game, costToPay, usedManaToPay)) {
137                     this.payment.increaseRed();
138                     return true;
139                 }
140                 break;
141             case G:
142                 if (pool.pay(ManaType.GREEN, ability, sourceFilter, game, costToPay, usedManaToPay)) {
143                     this.payment.increaseGreen();
144                     return true;
145                 }
146                 break;
147         }
148         return false;
149     }

151     protected void assignColorless(Ability ability, Game game, ManaPool pool, int mana, Cost costToPay) {
152         int conditionalCount = pool.getConditionalCount(ability, game, null, costToPay);
153         while (mana > payment.count() && (pool.count() > 0 || conditionalCount > 0)) {
154             if (pool.pay(ManaType.COLORLESS, ability, sourceFilter, game, costToPay, usedManaToPay)) {
155                 this.payment.increaseColorless();
156             }
157             break;
158         }
159     }

161     protected boolean assignGeneric(Ability ability, Game game, ManaPool pool, int mana, FilterMana filterMana, Cost costToPay) {
162         int conditionalCount = pool.getConditionalCount(ability, game, filterMana, costToPay);
163         while (mana > payment.count() && (pool.count() > 0 || conditionalCount > 0)) {
164             // try to use different mana to pay (conditional mana will used in pool.pay)
165             // filterMana can be null, uses for spells like "spend only black mana on X"
166 
167             // {C}
168             if ((filterMana == null || filterMana.isColorless()) && pool.pay(
169                     ManaType.COLORLESS, ability, sourceFilter, game, costToPay, usedManaToPay
170             )) {
171                 this.payment.increaseColorless();
172                 continue;
173             }
174 
175             // {B}
176             if ((filterMana == null || filterMana.isBlack()) && pool.pay(
177                     ManaType.BLACK, ability, sourceFilter, game, costToPay, usedManaToPay
178             )) {
179                 this.payment.increaseBlack();
180                 continue;
181             }
182 
183             // {U}
184             if ((filterMana == null || filterMana.isBlue()) && pool.pay(
185                     ManaType.BLUE, ability, sourceFilter, game, costToPay, usedManaToPay
186             )) {
187                 this.payment.increaseBlue();
188                 continue;
189             }
190 
191             // {W}
192             if ((filterMana == null || filterMana.isWhite()) && pool.pay(
193                     ManaType.WHITE, ability, sourceFilter, game, costToPay, usedManaToPay
194             )) {
195                 this.payment.increaseWhite();
196                 continue;
197             }
198 
199             // {G}
200             if ((filterMana == null || filterMana.isGreen()) && pool.pay(
201                     ManaType.GREEN, ability, sourceFilter, game, costToPay, usedManaToPay
202             )) {
203                 this.payment.increaseGreen();
204                 continue;
205             }
206 
207             // {R}
208             if ((filterMana == null || filterMana.isRed()) && pool.pay(
209                     ManaType.RED, ability, sourceFilter, game, costToPay, usedManaToPay
210             )) {
211                 this.payment.increaseRed();
212                 continue;
213             }
214 
215             // nothing to pay
216             break;
217         }
218         return mana > payment.count();
219     }

221     protected boolean isColoredPaid(ColoredManaSymbol mana) {
222         switch (mana) {
223             case B:
224                 return this.payment.getBlack() > 0;
225             case U:
226                 return this.payment.getBlue() > 0;
227             case W:
228                 return this.payment.getWhite() > 0;
229             case G:
230                 return this.payment.getGreen() > 0;
231             case R:
232                 return this.payment.getRed() > 0;
233         }
234         return false;
235     }

237     protected boolean isColorlessPaid(int mana) {

246     @Override
247     public boolean pay(Ability ability, Game game, Ability source, UUID controllerId, boolean noMana, Cost costToPay) {
248         if (noMana) {
249             setPaid();
250             return true;
251         }
252         Player player = game.getPlayer(controllerId);
253         if (!player.getManaPool().isForcedToPay()) {
254             assignPayment(game, ability, player.getManaPool(), costToPay != null ? costToPay : this);
255         }
256         game.getState().getSpecialActions().removeManaActions();
257         while (player.canRespond() && !isPaid()) {
258             ManaCost unpaid = this.getUnpaid();
259             String promptText = ManaUtil.addSpecialManaPayAbilities(ability, game, unpaid);
260             if (player.playMana(ability, unpaid, promptText, game)) {
261                 assignPayment(game, ability, player.getManaPool(), costToPay != null ? costToPay : this);
262             } else {
263                 return false;
264             }
265             game.getState().getSpecialActions().removeManaActions();
266         }
267         return isPaid();
268     }

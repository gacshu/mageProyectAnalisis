1   package mage.abilities.costs.costadjusters;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.CostAdjuster;
5   import mage.abilities.dynamicvalue.common.PermanentsOnBattlefieldCount;
6   import mage.abilities.hint.Hint;
7   import mage.abilities.hint.ValueHint;
8   import mage.constants.SuperType;
9   import mage.filter.FilterPermanent;
10  import mage.filter.common.FilterControlledCreaturePermanent;
11  import mage.game.Game;
12  import mage.util.CardUtil;
13  
14  /**
15   * @author TheElk801
16   */
17  public enum LegendaryCreatureCostAdjuster implements CostAdjuster {
18      instance;
19  
20      private static final FilterPermanent filter = new FilterControlledCreaturePermanent();
21  
22      static {
23          filter.add(SuperType.LEGENDARY.getPredicate());
24      }
25  
26      private static final Hint hint = new ValueHint(
27              "Legendary creatures you control",
28              new PermanentsOnBattlefieldCount(filter)
29      );
30  
31      @Override
32      public void adjustCosts(Ability ability, Game game) {
33          int count = game.getBattlefield().count(
34                  filter, ability.getControllerId(), ability, game
35          );
36          if (count > 0) {
37              CardUtil.reduceCost(ability, count);
38          }
39      }
40  
41      public static Hint getHint() {
42          return hint;
43      }
44  }

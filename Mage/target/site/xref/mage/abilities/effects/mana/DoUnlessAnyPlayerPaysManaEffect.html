3   import mage.MageObject;
4   import mage.Mana;
5   import mage.abilities.Ability;
6   import mage.abilities.Mode;
7   import mage.abilities.costs.Cost;
8   import mage.constants.Outcome;

10  import mage.players.Player;
11  import mage.util.CardUtil;
12  
13  import java.util.List;
14  import java.util.UUID;
15  
16  /**
17   * @author LevelX2
18   */
19  public class DoUnlessAnyPlayerPaysManaEffect extends ManaEffect {
20  
21      private final ManaEffect manaEffect;
22      private final Cost cost;
23      private final String chooseUseText;
24  
25      public DoUnlessAnyPlayerPaysManaEffect(ManaEffect effect, Cost cost, String chooseUseText) {
26          this.manaEffect = effect;
27          this.cost = cost;
28          this.chooseUseText = chooseUseText;
29      }
30  
31      public DoUnlessAnyPlayerPaysManaEffect(final DoUnlessAnyPlayerPaysManaEffect effect) {
32          super(effect);
33          this.manaEffect = (ManaEffect) effect.manaEffect.copy();
34          this.cost = effect.cost.copy();
35          this.chooseUseText = effect.chooseUseText;
36      }
37  
38      @Override
39      public List<Mana> getNetMana(Game game, Ability source) {
40          return manaEffect.getNetMana(game, source);
41      }
42  
43      @Override
44      public Mana produceMana(Game game, Ability source) {
45          Mana mana = new Mana();
46          if (game != null) {
47              Player controller = getPlayer(game, source);
48              MageObject sourceObject = game.getObject(source);
49              String message = CardUtil.replaceSourceName(chooseUseText, sourceObject.getName());
50              for (UUID playerId : game.getState().getPlayersInRange(controller.getId(), game)) {
51                  Player player = game.getPlayer(playerId);
52                  if (player != null && player.canRespond()
53                          && cost.canPay(source, source, player.getId(), game)
54                          && player.chooseUse(Outcome.Detriment, message, source, game)) {
55                      cost.clearPaid();
56                      if (cost.pay(source, game, source, player.getId(), false, null)) {
57                          if (!game.isSimulation()) {
58                              game.informPlayers(player.getLogName() + " pays the cost to prevent the effect");
59                          }
60                          return mana;
61                      }
62                  }
63              }
64          }
65          return manaEffect.produceMana(game, source);
66      }
67  
68      @Override
69      public String getText(Mode mode) {
70          if (!staticText.isEmpty()) {
71              return staticText;
72          }
73          return manaEffect.getText(mode) + " unless any player pays " + cost.getText();
74      }
75  
76      @Override
77      public ManaEffect copy() {
78          return new DoUnlessAnyPlayerPaysManaEffect(this);
79      }
80  
81  }

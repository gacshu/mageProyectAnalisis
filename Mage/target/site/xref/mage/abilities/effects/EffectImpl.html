1   package mage.abilities.effects;
2   
3   import mage.abilities.MageSingleton;
4   import mage.abilities.Mode;
5   import mage.constants.EffectType;
6   import mage.constants.Outcome;
7   import mage.target.targetpointer.FirstTargetPointer;
8   import mage.target.targetpointer.TargetPointer;
9   
10  import java.util.HashMap;
11  import java.util.Map;
12  import java.util.UUID;
13  
14  /**
15   * @author BetaSteward_at_googlemail.com
16   */
17  public abstract class EffectImpl implements Effect {
18  
19      protected UUID id;
20      protected Outcome outcome;
21      protected EffectType effectType;
22  
23      // read related docs about static and dynamic targets in ContinuousEffectImpl.affectedObjectsSet
24      protected TargetPointer targetPointer = new FirstTargetPointer();
25  
26      protected String staticText = "";
27      protected Map<String, Object> values;
28      protected String concatPrefix = ""; // combines multiple effects in text rule
29  
30      public EffectImpl(Outcome outcome) {
31          this.id = UUID.randomUUID();
32          this.outcome = outcome;
33      }
34  
35      public EffectImpl(final EffectImpl effect) {
36          this.id = effect.id;
37          this.outcome = effect.outcome;
38          this.staticText = effect.staticText;
39          this.effectType = effect.effectType;
40          this.targetPointer = effect.targetPointer.copy();
41          this.concatPrefix = effect.concatPrefix;
42          if (effect.values != null) {
43              values = new HashMap<>();
44              Map<String, Object> map = effect.values;
45              for (Map.Entry<String, Object> entry : map.entrySet()) {
46                  values.put(entry.getKey(), entry.getValue());
47              }
48          }
49      }
50  
51      @Override
52      public UUID getId() {
53          return id;
54      }
55  
56      @Override
57      public String getText(Mode mode) {
58          return staticText;
59      }
60  
61      @Override
62      public Effect setText(String staticText) {
63          this.staticText = staticText;
64          return this;
65      }
66  
67      @Override
68      public Outcome getOutcome() {
69          return outcome;
70      }
71  
72      @Override
73      public void setOutcome(Outcome outcome) {
74          this.outcome = outcome;
75      }
76  
77      @Override
78      public EffectType getEffectType() {
79          return effectType;
80      }
81  
82      @Override
83      public Effect setTargetPointer(TargetPointer targetPointer) {
84          this.targetPointer = targetPointer;
85          return this;
86      }
87  
88      @Override
89      public TargetPointer getTargetPointer() {
90          return this.targetPointer;
91      }
92  
93      @Override
94      public void newId() {
95          if (!(this instanceof MageSingleton)) {
96              this.id = UUID.randomUUID();
97          }
98      }
99  
100     @Override
101     public void setValue(String key, Object value) {
102         synchronized (this) {
103             if (values == null) {
104                 values = new HashMap<>();
105             }
106         }
107         values.put(key, value);
108     }
109 
110     @Override
111     public Object getValue(String key) {
112         if (values == null) { // no value was set
113             return null;
114         }
115         return values.get(key);
116     }
117 
118     @Override
119     public Effect concatBy(String concatPrefix) {
120         this.concatPrefix = concatPrefix;
121         return this;
122     }
123 
124     @Override
125     public String getConcatPrefix() {
126         return this.concatPrefix;
127     }
128 
129 }

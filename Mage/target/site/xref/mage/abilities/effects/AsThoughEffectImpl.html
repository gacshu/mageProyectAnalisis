18  public abstract class AsThoughEffectImpl extends ContinuousEffectImpl implements AsThoughEffect {
19  
20      protected AsThoughEffectType type;
21      boolean consumable;
22  
23      public AsThoughEffectImpl(AsThoughEffectType type, Duration duration, Outcome outcome) {
24          this(type, duration, outcome, false);
25      }
26  
27      public AsThoughEffectImpl(AsThoughEffectType type, Duration duration, Outcome outcome, boolean consumable) {
28          super(duration, outcome);
29          this.type = type;
30          this.effectType = EffectType.ASTHOUGH;
31          this.consumable = consumable;
32      }
33  
34      public AsThoughEffectImpl(final AsThoughEffectImpl effect) {
35          super(effect);
36          this.type = effect.type;
37          this.consumable = effect.consumable;
38      }
39  
40      @Override
41      public boolean applies(UUID objectId, Ability affectedAbility, Ability source, Game game, UUID playerId) {
42          // affectedControllerId = player to check (example: you can activate ability from opponent's card)
43          // by default it applies to full object
44          // if your AsThough effect type needs affected ability then override that method
45          return applies(objectId, source, playerId, game);
46      }
47  
48      @Override
49      public AsThoughEffectType getAsThoughEffectType() {
50          return type;
51      }
52  
53      /**
54       * Helper to check that affectedAbility is compatible for alternative cast
55       * modifications by setCastSourceIdWithAlternateMana
56       *
57       * @param cardToCheck
58       * @param affectedAbilityToCheck
59       * @param playerToCheck
60       * @param source
61       * @return
62       */
63      public boolean isAbilityAppliedForAlternateCast(Card cardToCheck, Ability affectedAbilityToCheck, UUID playerToCheck, Ability source) {
64          return cardToCheck != null
65                  && playerToCheck.equals(source.getControllerId())
66                  && cardToCheck.isOwnedBy(source.getControllerId())
67                  && affectedAbilityToCheck instanceof ActivatedAbility
68                  && (affectedAbilityToCheck.getAbilityType() == AbilityType.SPELL
69                  || affectedAbilityToCheck.getAbilityType() == AbilityType.PLAY_LAND);
70      }
71  
72      /**
73       * Internal method to do the necessary to allow the card from objectId to be
74       * cast or played (if it's a land) without paying any mana. Additional costs
75       * (like sacrificing or discarding) have still to be payed. Checks if the
76       * card is of the correct type or in the correct zone have to be done
77       * before.
78       *
79       * @param objectId sourceId of the card to play
80       * @param source source ability that allows this effect
81       * @param affectedControllerId player allowed to play the card
82       * @param game
83       * @return
84       */
85      protected boolean allowCardToPlayWithoutMana(UUID objectId, Ability source, UUID affectedControllerId, Game game) {
86          Player player = game.getPlayer(affectedControllerId);
87          Card card = game.getCard(objectId);
88          if (card == null || player == null) {
89              return false;
90          }
91          if (!card.isLand(game)) {
92              if (card instanceof SplitCard) {
93                  Card leftCard = ((SplitCard) card).getLeftHalfCard();
94                  player.setCastSourceIdWithAlternateMana(leftCard.getId(), null, leftCard.getSpellAbility().getCosts());
95                  Card rightCard = ((SplitCard) card).getRightHalfCard();
96                  player.setCastSourceIdWithAlternateMana(rightCard.getId(), null, rightCard.getSpellAbility().getCosts());
97              } else if (card instanceof ModalDoubleFacesCard) {
98                  Card leftCard = ((ModalDoubleFacesCard) card).getLeftHalfCard();
99                  Card rightCard = ((ModalDoubleFacesCard) card).getRightHalfCard();
100                 // some MDFC's are land.  IE: sea gate restoration
101                 if (!leftCard.isLand(game)) {
102                     player.setCastSourceIdWithAlternateMana(leftCard.getId(), null, leftCard.getSpellAbility().getCosts());
103                 }
104                 if (!rightCard.isLand(game)) {
105                     player.setCastSourceIdWithAlternateMana(rightCard.getId(), null, rightCard.getSpellAbility().getCosts());
106                 }
107             } else if (card instanceof AdventureCard) {
108                 Card creatureCard = card.getMainCard();
109                 Card spellCard = ((AdventureCard) card).getSpellCard();
110                 player.setCastSourceIdWithAlternateMana(creatureCard.getId(), null, creatureCard.getSpellAbility().getCosts());
111                 player.setCastSourceIdWithAlternateMana(spellCard.getId(), null, spellCard.getSpellAbility().getCosts());
112             }
113             player.setCastSourceIdWithAlternateMana(objectId, null, card.getSpellAbility().getCosts());
114         }
115         return true;
116     }
117 
118     @Override
119     public boolean isConsumable() {
120         return consumable;
121     }

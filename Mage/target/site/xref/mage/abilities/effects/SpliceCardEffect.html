8    * on the stack (splice a card).  {@link mage.abilities.Ability Abilities} with this type of effect will be
9    * called once and only once from the {@link mage.abilities.AbilityImpl#activate(mage.game.Game,
10   * boolean) Ability.activate} method before alternative or additional costs are paid.
11   *
12   * @author levelX2
13   */
14  
15  public interface SpliceCardEffect extends ContinuousEffect {
16      /**
17       * Called by the {@link ContinuousEffects#costModification(Ability abilityToModify, Game game) ContinuousEffects.costModification}
18       * method.
19       *
20       * @param game The game for which this effect should be applied.
21       * @param source The source ability of this effect.
22       * @param abilityToModify The {@link mage.abilities.SpellAbility} or {@link Ability} which should be modified.
23       * @return
24       */
25      boolean apply ( Game game, Ability source, Ability abilityToModify );
26  
27      /**
28       * Called by the {@link ContinuousEffects#costModification(mage.abilities.Ability, mage.game.Game) ContinuousEffects.costModification}
29       * method.
30       *
31       * @param abilityToModify The ability to possibly modify.
32       * @param source The source ability of this effect.
33       * @param game The game for which this effect shoul dbe applied.
34       * @return
35       */
36      boolean applies(Ability abilityToModify, Ability source, Game game);
37  }

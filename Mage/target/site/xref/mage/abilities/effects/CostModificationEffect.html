3   package mage.abilities.effects;
4   
5   import mage.abilities.Ability;
6   import mage.constants.CostModificationType;
7   import mage.game.Game;
8   
9   /**
10   * Represents a {@link ContinuousEffect} that will modify the cost of a spell or
11   * ability on the stack.  {@link mage.abilities.Ability Abilities} with this type
12   * of effect will be called once and only once from the
13   * {@link mage.abilities.AbilityImpl#activate(mage.game.Game, boolean) Ability.activate}
14   * method before costs are paid.  After this time the costs are locked in and
15   * should not be modified further.
16   *
17   * @author maurer.it_at_gmail.com
18   */
19  //20101001 - 601.2e/613.10
20  public interface CostModificationEffect extends ContinuousEffect {
21      /**
22       * Called by the {@link ContinuousEffects#costModification(java.util.UUID, mage.abilities.Ability, mage.game.Game) ContinuousEffects.costModification}
23       * method.
24       *
25       * @param game The game for which this effect should be applied.
26       * @param source The source ability of this effect.
27       * @param abilityToModify The {@link mage.abilities.SpellAbility} or {@link Ability} which should be modified.
28       * @return
29       */
30      boolean apply ( Game game, Ability source, Ability abilityToModify );
31  
32      /**
33       * Called by the {@link ContinuousEffects#costModification(mage.abilities.Ability, mage.game.Game) ContinuousEffects.costModification}
34       * method.
35       * 
36       * @param abilityToModify The ability to possibly modify.
37       * @param source The source ability of this effect.
38       * @param game The game for which this effect should be applied.
39       * @return
40       */
41      boolean applies(Ability abilityToModify, Ability source, Game game);
42      
43      /**
44       * Return the type of modification
45       * @return 
46       */
47      CostModificationType getModificationType();
48  }

17   */
18  public interface Effect extends Serializable, Copyable<Effect> {
19  
20      UUID getId();
21  
22      void newId();
23  
24      /**
25       * Some general behaviours for rule text handling: Rule text of effects get
26       * automatically a full stop "." at the end, if not another effect e.g. with
27       * a starting "and" follows. So at least for effects of the framework, that
28       * are used from multiple cards, it's better to set no full stop at the end
29       * of the rule text of an effect. Also the starting letter of an effect is
30       * automatically converted to upper case if the rule text starts with this
31       * text. So There is no need to let the effect text start with upper case,
32       * even if extracted from a filter message. Also here it's important to use
33       * only lower cases for effects located in the framework, so if used for a
34       * triggered abilitiy, the effect text needs to start with lower case after
35       * the comma.
36       *
37       * @param mode the selected mode of the ability (mostly there is only one)
38       * @return
39       */
40      String getText(Mode mode);
41  
42      Effect setText(String staticText);
43  
44      boolean apply(Game game, Ability source);
45  
46      /**
47       * The outcome is used for the AI to decide if an effect does bad or good to
48       * the targets.
49       *
50       * @return
51       */
52      Outcome getOutcome();
53  
54      void setOutcome(Outcome outcome);
55  
56      EffectType getEffectType();
57  
58      Effect setTargetPointer(TargetPointer targetPointer);
59  
60      TargetPointer getTargetPointer();
61  
62      void setValue(String key, Object value);
63  
64      Object getValue(String key);
65  
66      Effect copy();
67  
68      Effect concatBy(String concatPrefix);
69  
70      String getConcatPrefix();
71  
72      /**
73       * Used to check if this is a conditional version without needed any Java reflection hacks or chained instanceof checks.
74       * Only conditions on the activation of the ability are checked, and not on how the effects can be used.
75       * E.g. For a conditional mana ability this will only return the condition under which it can be activated,
76       *      and NOT the conditions on the mane produced.
77       *
78       * Assumed that if the returned condition is null then this is not a conditional version, or that it does not matter.
79       *
80       * @return
81       */
82      public default Condition getCondition() {
83          return null;
84      }
85  }

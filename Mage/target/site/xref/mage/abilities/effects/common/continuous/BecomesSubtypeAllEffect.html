2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.ContinuousEffectImpl;
5   import mage.constants.*;
6   import mage.filter.StaticFilters;
7   import mage.filter.common.FilterCreaturePermanent;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  
11  import java.util.ArrayList;
12  import java.util.Arrays;
13  import java.util.List;
14  
15  /**
16   * @author LevelX2
17   */
18  public class BecomesSubtypeAllEffect extends ContinuousEffectImpl {
19  
20      private final List<SubType> subtypes = new ArrayList<>();
21      private final boolean loseOther; // loses other subtypes
22      private final FilterCreaturePermanent filter;
23  
24      public BecomesSubtypeAllEffect(Duration duration, SubType subtype) {
25          this(duration, Arrays.asList(subtype));
26      }
27  
28      public BecomesSubtypeAllEffect(Duration duration, List<SubType> subtypes) {
29          this(duration, subtypes, StaticFilters.FILTER_PERMANENT_CREATURE, true);
30      }
31  
32      public BecomesSubtypeAllEffect(Duration duration, List<SubType> subtypes, FilterCreaturePermanent filter, boolean loseOther) {
33          super(duration, Layer.TypeChangingEffects_4, SubLayer.NA, Outcome.Detriment);
34          this.subtypes.addAll(subtypes);
35          this.staticText = setText();
36          this.loseOther = loseOther;
37          this.filter = filter;
38      }
39  
40      public BecomesSubtypeAllEffect(final BecomesSubtypeAllEffect effect) {
41          super(effect);
42          this.subtypes.addAll(effect.subtypes);
43          this.loseOther = effect.loseOther;
44          this.filter = effect.filter;
45      }
46  
47      @Override
48      public BecomesSubtypeAllEffect copy() {
49          return new BecomesSubtypeAllEffect(this);
50      }
51  
52      @Override
53      public boolean apply(Game game, Ability source) {
54          boolean flag = false;
55          for (Permanent permanent : game.getBattlefield().getActivePermanents(filter, source.getControllerId(), source, game)) {
56              if (permanent == null) {
57                  continue;
58              }
59              flag = true;
60              if (loseOther) {
61                  permanent.removeAllCreatureTypes(game);
62              }
63              for (SubType subtype : subtypes) {
64                  permanent.addSubType(game, subtype);
65              }
66          }
67          if (!flag && duration == Duration.Custom) {
68              discard();
69          }
70          return true;
71      }
72  
73      private String setText() {
74          StringBuilder sb = new StringBuilder();
75          sb.append("Target creature becomes that type");
76          if (!duration.toString().isEmpty()
77                  && duration != Duration.EndOfGame) {
78              sb.append(' ').append(duration.toString());
79          }
80          return sb.toString();
81      }
82  }

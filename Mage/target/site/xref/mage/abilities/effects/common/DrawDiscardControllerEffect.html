1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.Mode;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.constants.Outcome;
7   import mage.game.Game;
8   import mage.players.Player;
9   import mage.util.CardUtil;
10  
11  /**
12   * @author BetaSteward_at_googlemail.com
13   */
14  public class DrawDiscardControllerEffect extends OneShotEffect {
15  
16      private final int cardsToDraw;
17      private final int cardsToDiscard;
18      private final boolean optional;
19  
20      public DrawDiscardControllerEffect() {
21          this(1, 1);
22      }
23  
24      public DrawDiscardControllerEffect(boolean optional) {
25          this(1, 1, optional);
26      }
27  
28      public DrawDiscardControllerEffect(int cardsToDraw, int cardsToDiscard) {
29          this(cardsToDraw, cardsToDiscard, false);
30      }
31  
32      public DrawDiscardControllerEffect(int cardsToDraw, int cardsToDiscard, boolean optional) {
33          super(Outcome.DrawCard);
34          this.cardsToDraw = cardsToDraw;
35          this.cardsToDiscard = cardsToDiscard;
36          this.optional = optional;
37      }
38  
39      public DrawDiscardControllerEffect(final DrawDiscardControllerEffect effect) {
40          super(effect);
41          this.cardsToDraw = effect.cardsToDraw;
42          this.cardsToDiscard = effect.cardsToDiscard;
43          this.optional = effect.optional;
44      }
45  
46      @Override
47      public DrawDiscardControllerEffect copy() {
48          return new DrawDiscardControllerEffect(this);
49      }
50  
51      @Override
52      public boolean apply(Game game, Ability source) {
53          Player player = game.getPlayer(source.getControllerId());
54          if (player == null) {
55              return false;
56          }
57          if (optional && !player.chooseUse(outcome, "Draw, then discard?", source, game)) {
58              return true;
59          }
60          player.drawCards(cardsToDraw, source, game);
61          player.discard(cardsToDiscard, false, false, source, game);
62          return true;
63      }
64  
65      @Override
66      public String getText(Mode mode) {
67          if (staticText != null && !staticText.isEmpty()) {
68              return staticText;
69          }
70          return (optional ? "you may " : "") +
71                  "draw " +
72                  (cardsToDraw == 1 ? "a" : CardUtil.numberToText(cardsToDraw)) +
73                  " card" +
74                  (cardsToDraw == 1 ? "" : "s") +
75                  (optional ? ". If you do," : ", then") +
76                  " discard " +
77                  (cardsToDiscard == 1 ? "a" : CardUtil.numberToText(cardsToDiscard)) +
78                  " card" +
79                  (cardsToDiscard == 1 ? "" : "s");
80      }
81  }

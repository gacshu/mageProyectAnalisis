2   package mage.abilities.effects.common.combat;
3   
4   import java.util.UUID;
5   import mage.abilities.Ability;
6   import mage.abilities.effects.RequirementEffect;
7   import mage.constants.Duration;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  
11  /**
12   * !! This effect does only support one target.
13   *
14   *
15   * http://tappedout.net/mtg-questions/must-be-blocked-if-able-effect-makes-other-attacking-creatures-essentially-unblockable/
16   *
17   * When you Declare Blockers, you choose an arrangement for your blockers, then
18   * check to see if there are any restrictions or requirements.
19   *
20   * If any restrictions are violated, the block is illegal. (For example, trying
21   * to block with Sightless Ghoul) If any requirements are violated, the least
22   * possible number of requirements must be violated, otherwise the block is
23   * illegal. (For example, your opponent control two creatures that he has cast
24   * Deadly Allure on, but you control only one creature. Blocking either one will
25   * violate a requirement, "This creature must be blocked this turn if able", but
26   * it will also violate the least possible number of requirements, thus it is
27   * legal.) If the block is illegal, the game state backs up and you declare
28   * blockers again. (Note that while you can, in some cases, circumvent
29   * requirements such as "This creature must be blocked" or "This creature must
30   * block any attacking creature" you can never circumvent restrictions: "This
31   * creature can't block" or "Only one creature may block this turn.") Because
32   * you declare ALL your blockers at once, THEN check for
33   * restrictions/requirements, you may block Deadly Allure'd creatures with only
34   * one creature, if you choose. This still works with Lure: This card sets up a
35   * requirement that ALL creatures must block it if able. Any block that violates
36   * more than the minimum number of requirements is still illegal.
37   *
38   * @author LevelX2
39   */
40  public class MustBeBlockedByAtLeastOneTargetEffect extends RequirementEffect {
41  
42      public MustBeBlockedByAtLeastOneTargetEffect() {
43          this(Duration.EndOfTurn);
44      }
45  
46      public MustBeBlockedByAtLeastOneTargetEffect(Duration duration) {
47          super(duration);
48          staticText = "target creature must be blocked this turn if able";
49      }
50  
51      public MustBeBlockedByAtLeastOneTargetEffect(final MustBeBlockedByAtLeastOneTargetEffect effect) {
52          super(effect);
53      }
54  
55      @Override
56      public boolean applies(Permanent permanent, Ability source, Game game) {
57          return permanent.canBlock(this.getTargetPointer().getFirst(game, source), game);
58      }
59  
60      @Override
61      public boolean mustAttack(Game game) {
62          return false;
63      }
64  
65      @Override
66      public boolean mustBlock(Game game) {
67          return false;
68      }
69  
70      @Override
71      public UUID mustBlockAttackerIfElseUnblocked(Ability source, Game game) {
72          return this.getTargetPointer().getFirst(game, source);
73      }
74  
75      @Override
76      public MustBeBlockedByAtLeastOneTargetEffect copy() {
77          return new MustBeBlockedByAtLeastOneTargetEffect(this);
78      }
79  
80  }

2   package mage.abilities.effects.common;
3   
4   import mage.abilities.Ability;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.constants.Outcome;
7   import mage.constants.Zone;
8   import mage.game.ExileZone;
9   import mage.game.Game;
10  import mage.players.Player;
11  import mage.util.CardUtil;
12  
13  /**
14   * @author BetaSteward_at_googlemail.com
15   */
16  public class ReturnFromExileEffect extends OneShotEffect {
17  
18      private final Zone zone;
19      private final boolean tapped;
20  
21      public ReturnFromExileEffect(Zone zone) {
22          this(zone, false);
23      }
24  
25      public ReturnFromExileEffect(Zone zone, String text) {
26          this(zone, false);
27          staticText = text;
28      }
29  
30      public ReturnFromExileEffect(Zone zone, boolean tapped) {
31          super(Outcome.PutCardInPlay);
32          this.zone = zone;
33          this.tapped = tapped;
34          setText();
35      }
36  
37      public ReturnFromExileEffect(final ReturnFromExileEffect effect) {
38          super(effect);
39          this.zone = effect.zone;
40          this.tapped = effect.tapped;
41      }
42  
43      @Override
44      public ReturnFromExileEffect copy() {
45          return new ReturnFromExileEffect(this);
46      }
47  
48      @Override
49      public boolean apply(Game game, Ability source) {
50          ExileZone exile = game.getExile().getExileZone(CardUtil.getExileZoneId(game, source));
51          Player controller = game.getPlayer(source.getControllerId());
52          if (controller == null || exile == null) {
53              return false;
54          }
55          if (zone == Zone.LIBRARY) {
56              return controller.putCardsOnTopOfLibrary(exile, game, source, false);
57          }
58          return controller.moveCards(exile.getCards(game), zone, source, game, tapped, false, true, null);
59      }
60  
61      private void setText() {
62          StringBuilder sb = new StringBuilder();
63          sb.append("return the exiled cards ");
64          switch (zone) {
65              case BATTLEFIELD:
66                  sb.append("to the battlefield under its owner's control");
67                  if (tapped) {
68                      sb.append(" tapped");
69                  }
70                  break;
71              case HAND:
72                  sb.append("to their owner's hand");
73                  break;
74              case GRAVEYARD:
75                  sb.append("to their owner's graveyard");
76                  break;
77          }
78          staticText = sb.toString();
79      }
80  
81  }

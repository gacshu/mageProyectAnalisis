20  /*
21   * 903.9. If mana would be added to a player's mana pool of a color that isn't in the color identity
22   * of that player's commander, that amount of colorless mana is added to that player's mana pool instead.
23   *
24   * Commander rule #4 was removed Jan. 18, 2016 
25   *
26   */
27  @Deprecated
28  public class CommanderManaReplacementEffect extends ReplacementEffectImpl {
29  
30      private final UUID playerId;
31      private final FilterMana commanderMana;
32  
33      public CommanderManaReplacementEffect(UUID playerId, FilterMana commanderMana) {
34          super(Duration.EndOfGame, Outcome.Neutral);
35          staticText = "If mana would be added to a player's mana pool of a color that isn't in the color identity of that player's commander, that amount of colorless mana is added to that player's mana pool instead.";
36          this.playerId = playerId;
37          this.commanderMana = commanderMana;
38      }
39  
40      public CommanderManaReplacementEffect(final CommanderManaReplacementEffect effect) {
41          super(effect);
42          this.playerId = effect.playerId;
43          this.commanderMana = effect.commanderMana;
44      }
45  
46      @Override
47      public CommanderManaReplacementEffect copy() {
48          return new CommanderManaReplacementEffect(this);
49      }
50  
51      @Override
52      public boolean apply(Game game, Ability source) {
53          return true;
54      }
55  
56      @Override
57      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
58          Mana mana = ((ManaEvent) event).getMana();
59          if (mana.getBlack() > 0 && !commanderMana.isBlack()) {
60              for (int i = 0; i < mana.getBlack(); i++) {
61                  mana.increaseGeneric();
62              }
63              mana.setBlack(0);
64          }
65          if (mana.getBlue() > 0 && !commanderMana.isBlue()) {
66              for (int i = 0; i < mana.getBlue(); i++) {
67                  mana.increaseGeneric();
68              }
69              mana.setBlue(0);
70          }
71          if (mana.getGreen() > 0 && !commanderMana.isGreen()) {
72              for (int i = 0; i < mana.getGreen(); i++) {
73                  mana.increaseGeneric();
74              }
75              mana.setGreen(0);
76          }
77          if (mana.getRed() > 0 && !commanderMana.isRed()) {
78              for (int i = 0; i < mana.getRed(); i++) {
79                  mana.increaseGeneric();
80              }
81              mana.setRed(0);
82          }
83          if (mana.getWhite() > 0 && !commanderMana.isWhite()) {
84              for (int i = 0; i < mana.getWhite(); i++) {
85                  mana.increaseGeneric();
86              }
87              mana.setWhite(0);
88          }
89          return false;
90      }
91  
92      @Override
93      public boolean checksEventType(GameEvent event, Game game) {
94          return event.getType() == GameEvent.EventType.ADD_MANA;
95      }
96  
97      @Override
98      public boolean applies(GameEvent event, Ability source, Game game) {
99          return event.getPlayerId().equals(playerId);
100     }

102 }

6   import mage.constants.Outcome;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.stack.Spell;
10  import mage.players.Player;
11  
12  /**
13   * @author nantuko
14   */
15  public class ShuffleSpellEffect extends OneShotEffect implements MageSingleton {
16  
17      private static final ShuffleSpellEffect instance = new ShuffleSpellEffect();
18  
19      public static ShuffleSpellEffect getInstance() {
20          return instance;
21      }
22  
23      private ShuffleSpellEffect() {
24          super(Outcome.Neutral);
25          staticText = "Shuffle {this} into its owner's library";
26      }
27  
28      @Override
29      public boolean apply(Game game, Ability source) {
30          Player controller = game.getPlayer(source.getControllerId());
31          if (controller != null) {
32              // We have to use the spell id because in case of copied spells, the sourceId can be multiple times on the stack
33              Spell spell = game.getStack().getSpell(source.getId());
34              if (spell != null) {
35                  if (controller.moveCards(spell, Zone.LIBRARY, source, game)) {
36                      Player owner = game.getPlayer(spell.getCard().getOwnerId());
37                      if (owner != null) {
38                          owner.shuffleLibrary(source, game);
39                      }
40  
41                  }
42              }
43              return true;
44          }
45          return false;
46      }
47  
48      @Override
49      public ShuffleSpellEffect copy() {
50          return instance;
51      }
52  }

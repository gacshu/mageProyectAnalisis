1   package mage.abilities.effects.common;

3   import mage.abilities.Ability;
4   import mage.abilities.effects.ContinuousEffect;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.abilities.effects.common.continuous.GainControlTargetEffect;
7   import mage.constants.Duration;
8   import mage.constants.Outcome;
9   import static mage.constants.Outcome.Benefit;
10  import mage.game.Game;
11  import mage.game.permanent.Permanent;
12  import mage.players.Player;
13  import mage.target.Target;
14  import mage.target.common.TargetOpponent;
15  import mage.target.targetpointer.FixedTarget;
16  
17  /**
18   * Use this effect only with EntersBattlefieldAbility like abilities
19   *
20   * @author LevelX2
21   */
22  public class EntersBattlefieldUnderControlOfOpponentOfChoiceEffect extends OneShotEffect {
23  
24      public EntersBattlefieldUnderControlOfOpponentOfChoiceEffect() {
25          super(Benefit);
26          staticText = "under the control of an opponent of your choice";
27      }
28  
29      private EntersBattlefieldUnderControlOfOpponentOfChoiceEffect(final EntersBattlefieldUnderControlOfOpponentOfChoiceEffect effect) {
30          super(effect);
31      }
32  
33      @Override
34      public EntersBattlefieldUnderControlOfOpponentOfChoiceEffect copy() {
35          return new EntersBattlefieldUnderControlOfOpponentOfChoiceEffect(this);
36      }
37  
38      @Override
39      public boolean apply(Game game, Ability source) {
40          Player controller = game.getPlayer(source.getControllerId());
41          if (controller == null) {
42              return false;
43          }
44          Target target = new TargetOpponent();
45          target.setNotTarget(true);
46          if (!controller.choose(Outcome.Benefit, target, source, game)) {
47              return false;
48          }
49          Player opponent = game.getPlayer(target.getFirstTarget());
50          if (opponent == null) {
51              return false;
52          }
53          Permanent permanent = game.getPermanentEntering(source.getSourceId());
54          if (permanent != null) {
55              permanent.setOriginalControllerId(opponent.getId()); // permanent was controlled by this player since the existance of this object so original controller has to be set to the first controller
56              permanent.setControllerId(opponent.getId()); // neccessary to set already here because spell caster never controlled the permanent (important for rule 800.4a)
57              game.informPlayers(permanent.getLogName() + " enters the battlefield under the control of " + opponent.getLogName());
58          }
59          ContinuousEffect continuousEffect = new GainControlTargetEffect(
60                  Duration.Custom, true, opponent.getId()
61          );
62          continuousEffect.setTargetPointer(new FixedTarget(
63                  source.getSourceId(), source.getSourceObjectZoneChangeCounter()
64          ));
65          game.addEffect(continuousEffect, source);
66          return true;
67      }
68  }

1   package mage.abilities.effects.common;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.Mode;
6   import mage.abilities.effects.OneShotEffect;
7   import mage.constants.Outcome;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  import mage.target.Target;
11  
12  /**
13   * @author LevelX2
14   */
15  public class FightTargetSourceEffect extends OneShotEffect {
16  
17      public FightTargetSourceEffect() {
18          super(Outcome.Damage);
19      }
20  
21      public FightTargetSourceEffect(final FightTargetSourceEffect effect) {
22          super(effect);
23      }
24  
25      @Override
26      public boolean apply(Game game, Ability source) {
27          MageObject sourceObject = source.getSourceObject(game);
28          if (sourceObject != null) {
29              Permanent sourcePermanent = source.getSourcePermanentIfItStillExists(game);
30              Permanent creature1 = game.getPermanent(getTargetPointer().getFirst(game, source));
31              // 20110930 - 701.10
32              if (creature1 != null && sourcePermanent != null) {
33                  if (creature1.isCreature(game) && sourcePermanent.isCreature(game)) {
34                      return sourcePermanent.fight(creature1, source, game);
35                  }
36              }
37              if (!game.isSimulation()) {
38                  game.informPlayers(sourceObject.getLogName() + ": Fighting effect has been fizzled.");
39              }
40          }
41          return false;
42      }
43  
44      @Override
45      public FightTargetSourceEffect copy() {
46          return new FightTargetSourceEffect(this);
47      }
48  
49      @Override
50      public String getText(Mode mode) {
51          if (staticText != null && !staticText.isEmpty()) {
52              return staticText;
53          }
54          Target target = mode.getTargets().get(0);
55          StringBuilder sb = new StringBuilder("{this} fights ");
56          if (target.getMinNumberOfTargets() == 0 && target.getMaxNumberOfTargets() == 1) {
57              sb.append("up to one ");
58          }
59          if (!target.getTargetName().contains("other")) {
60              sb.append("target ");
61          }
62          sb.append(target.getTargetName());
63          return sb.toString();
64      }
65  }

9   import mage.constants.Zone;
10  import mage.filter.StaticFilters;
11  import mage.game.Game;
12  import mage.players.Player;
13  import mage.target.common.TargetCardInLibrary;
14  
15  /**
16   * @author TheElk801
17   */
18  public class SearchLibraryAndExileTargetEffect extends SearchEffect {
19  
20      public SearchLibraryAndExileTargetEffect(int amount, boolean upTo) {
21          super(new TargetCardInLibrary(upTo ? 0 : amount, amount, amount > 1 ? StaticFilters.FILTER_CARD_CARDS : StaticFilters.FILTER_CARD), Outcome.Exile);
22      }
23  
24      private SearchLibraryAndExileTargetEffect(final SearchLibraryAndExileTargetEffect effect) {
25          super(effect);
26      }
27  
28      @Override
29      public SearchLibraryAndExileTargetEffect copy() {
30          return new SearchLibraryAndExileTargetEffect(this);
31      }
32  
33      @Override
34      public boolean apply(Game game, Ability source) {
35          Player controller = game.getPlayer(source.getControllerId());
36          Player player = game.getPlayer(getTargetPointer().getFirst(game, source));
37          if (controller == null || player == null) {
38              return false;
39          }
40          controller.searchLibrary(target, source, game, player.getId());
41          Cards cards = new CardsImpl();
42          target.getTargets()
43                  .stream()
44                  .map(uuid -> player.getLibrary().getCard(uuid, game))
45                  .forEach(cards::add);
46          controller.moveCards(cards, Zone.EXILED, source, game);
47          player.shuffleLibrary(source, game);
48          return true;
49      }
50  
51      @Override
52      public String getText(Mode mode) {
53          if (staticText != null && !staticText.isEmpty()) {
54              return staticText;
55          }
56          return "search "
57                  + getTargetPointer().describeTargets(mode.getTargets(), "that player")
58                  + "'s library for "
59                  + target.getDescription()
60                  + (target.getMaxNumberOfTargets() > 1 ? " and exile them" : " and exile it")
61                  + ". Then that player shuffles";
62      }
63  }

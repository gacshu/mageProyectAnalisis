23  /**
24   * FAQ 2013/01/11
25   * <p>
26   * 702.97. Cipher
27   * <p>
28   * 702.97a Cipher appears on some instants and sorceries. It represents two
29   * static abilities, one that functions while the spell is on the stack and one
30   * that functions while the card with cipher is in the exile zone. "Cipher"
31   * means "If this spell is represented by a card, you may exile this card
32   * encoded on a creature you control" and "As long as this card is encoded on
33   * that creature, that creature has 'Whenever this creature deals combat damage
34   * to a player, you may copy this card and you may cast the copy without paying
35   * its mana cost.'"
36   * <p>
37   * 702.97b The term "encoded" describes the relationship between the card with
38   * cipher while in the exile zone and the creature chosen when the spell
39   * represented by that card resolves.
40   * <p>
41   * 702.97c The card with cipher remains encoded on the chosen creature as long
42   * as the card with cipher remains exiled and the creature remains on the
43   * battlefield. The card remains encoded on that object even if it changes
44   * controller or stops being a creature, as long as it remains on the
45   * battlefield.
46   * <p>
47   * TODO: Implement Cipher as two static abilities concerning the rules.
48   *
49   * @author LevelX2
50   */
51  public class CipherEffect extends OneShotEffect {
52  
53      public CipherEffect() {
54          super(Outcome.Copy);
55          staticText = "<br><br/>Cipher <i>(Then you may exile this spell card encoded on a creature you control. Whenever that creature deals combat damage to a player, its controller may cast a copy of the encoded card without paying its mana cost.)</i>";
56      }
57  
58      public CipherEffect(final CipherEffect effect) {
59          super(effect);
60      }
61  
62      @Override
63      public boolean apply(Game game, Ability source) {
64          Player controller = game.getPlayer(source.getControllerId());
65          if (controller != null) {
66              TargetControlledCreaturePermanent target = new TargetControlledCreaturePermanent();
67              target.setNotTarget(true);
68              if (target.canChoose(source.getControllerId(), source, game)
69                      && controller.chooseUse(outcome, "Cipher this spell to a creature?", source, game)) {
70                  controller.chooseTarget(outcome, target, source, game);
71                  Card sourceCard = game.getCard(source.getSourceId());
72                  Permanent targetCreature = game.getPermanent(target.getFirstTarget());
73                  if (targetCreature != null && sourceCard != null) {
74                      String ruleText = "you may cast a copy of " + sourceCard.getLogName() + " without paying its mana cost";
75                      Ability ability = new DealsCombatDamageToAPlayerTriggeredAbility(new CipherStoreEffect(source.getSourceId(), ruleText), true);
76                      ContinuousEffect effect = new GainAbilityTargetEffect(ability, Duration.Custom);
77                      effect.setTargetPointer(new FixedTarget(target.getFirstTarget(), game));
78                      game.addEffect(effect, source);
79                      if (!game.isSimulation()) {
80                          game.informPlayers(sourceCard.getLogName() + ": Spell ciphered to " + targetCreature.getLogName());
81                      }
82                      return controller.moveCards(sourceCard, Zone.EXILED, source, game);
83                  } else {
84                      return false;
85                  }
86              }
87              return true;
88          }
89          return false;
90      }
91  
92      @Override
93      public CipherEffect copy() {
94          return new CipherEffect(this);
95      }
96  }

98  class CipherStoreEffect extends OneShotEffect {
99  
100     private final UUID cipherCardId;
101 
102     public CipherStoreEffect(UUID cipherCardId, String ruleText) {
103         super(Outcome.Copy);
104         this.cipherCardId = cipherCardId;
105         staticText = ruleText;
106     }
107 
108     public CipherStoreEffect(final CipherStoreEffect effect) {
109         super(effect);
110         this.cipherCardId = effect.cipherCardId;
111     }
112 
113     @Override
114     public boolean apply(Game game, Ability source) {
115         Player controller = game.getPlayer(source.getControllerId());
116         Card cipherCard = game.getCard(cipherCardId);
117         if (cipherCard != null && controller != null) {
118             Card copyCard = game.copyCard(cipherCard, source, controller.getId());
119             SpellAbility ability = copyCard.getSpellAbility();
120             // remove the cipher effect from the copy
121             ability.getEffects().removeIf(CipherEffect.class::isInstance);
122             controller.cast(ability, game, true, new ApprovingObject(source, game));
123 
124         }
125 
126         return false;
127     }
128 
129     @Override
130     public CipherStoreEffect copy() {
131         return new CipherStoreEffect(this);
132     }
133 }

1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.OneShotEffect;
5   import mage.cards.Cards;
6   import mage.cards.CardsImpl;
7   import mage.constants.Outcome;
8   import mage.constants.Zone;
9   import mage.filter.FilterPermanent;
10  import mage.game.Game;
11  import mage.players.Player;
12  import mage.util.CardUtil;
13  
14  /**
15   * @author LevelX2
16   */
17  public class ExileAllEffect extends OneShotEffect {
18  
19      private final FilterPermanent filter;
20      private final boolean forSource;
21  
22      public ExileAllEffect(FilterPermanent filter) {
23          this(filter, false);
24      }
25  
26      public ExileAllEffect(FilterPermanent filter, boolean forSource) {
27          super(Outcome.Exile);
28          this.filter = filter;
29          this.forSource = forSource;
30          setText();
31      }
32  
33      public ExileAllEffect(final ExileAllEffect effect) {
34          super(effect);
35          this.filter = effect.filter.copy();
36          this.forSource = effect.forSource;
37      }
38  
39      @Override
40      public ExileAllEffect copy() {
41          return new ExileAllEffect(this);
42      }
43  
44      @Override
45      public boolean apply(Game game, Ability source) {
46          Player controller = game.getPlayer(source.getControllerId());
47          if (controller == null) {
48              return false;
49          }
50          Cards cards = new CardsImpl();
51          game.getBattlefield().getActivePermanents(
52                  filter, source.getControllerId(), source, game
53          ).stream().forEach(cards::add);
54          if (forSource) {
55              return controller.moveCardsToExile(cards.getCards(game), source, game, true, CardUtil.getExileZoneId(game, source), CardUtil.getSourceName(game, source));
56          }
57          return controller.moveCards(cards, Zone.EXILED, source, game);
58  
59      }
60  
61      private void setText() {
62          StringBuilder sb = new StringBuilder();
63          sb.append("exile all ").append(filter.getMessage());
64          staticText = sb.toString();
65      }
66  }

21  public class GainSuspendEffect extends ContinuousEffectImpl implements SourceEffect {
22  
23      MageObjectReference mor;
24  
25      public GainSuspendEffect(MageObjectReference mor) {
26          super(Duration.Custom, Layer.AbilityAddingRemovingEffects_6, SubLayer.NA, Outcome.AddAbility);
27          this.mor = mor;
28          staticText = "{this} gains suspend";
29      }
30  
31      public GainSuspendEffect(final GainSuspendEffect effect) {
32          super(effect);
33          this.mor = effect.mor;
34      }
35  
36      @Override
37      public GainSuspendEffect copy() {
38          return new GainSuspendEffect(this);
39      }
40  
41      @Override
42      public boolean apply(Game game, Ability source) {
43          Card card = game.getCard(mor.getSourceId());
44          if (card != null && mor.refersTo(card, game) && game.getState().getZone(card.getId()) == Zone.EXILED) {
45              SuspendAbility.addSuspendTemporaryToCard(card, source, game);
46          } else {
47              discard();
48          }
49          return true;
50      }
51  }

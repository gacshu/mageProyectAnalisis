2   package mage.abilities.effects.common;
3   
4   import mage.constants.Duration;
5   import mage.abilities.Ability;
6   import mage.abilities.Mode;
7   import mage.abilities.dynamicvalue.DynamicValue;
8   import mage.abilities.effects.PreventionEffectImpl;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  
12  /**
13   *
14   * @author BetaSteward_at_googlemail.com
15   */
16  public class PreventDamageToTargetEffect extends PreventionEffectImpl {
17  
18      public PreventDamageToTargetEffect(Duration duration) {
19          this(duration, false);
20      }
21  
22      public PreventDamageToTargetEffect(Duration duration, boolean onlyCombat) {
23          this(duration, Integer.MAX_VALUE, onlyCombat);
24      }
25  
26      public PreventDamageToTargetEffect(Duration duration, int amount) {
27          this(duration, amount, false);
28      }
29  
30      public PreventDamageToTargetEffect(Duration duration, int amount, boolean onlyCombat) {
31          super(duration, amount, onlyCombat);
32      }
33  
34      public PreventDamageToTargetEffect(Duration duration, boolean onlyCombat, boolean consumable, DynamicValue amountToPreventDynamic) {
35          super(duration, 0, onlyCombat, consumable, amountToPreventDynamic);
36      }
37  
38      public PreventDamageToTargetEffect(final PreventDamageToTargetEffect effect) {
39          super(effect);
40      }
41  
42      @Override
43      public PreventDamageToTargetEffect copy() {
44          return new PreventDamageToTargetEffect(this);
45      }
46  
47      @Override
48      public boolean applies(GameEvent event, Ability source, Game game) {
49          return !this.used && super.applies(event, source, game) && event.getTargetId().equals(targetPointer.getFirst(game, source));
50      }
51  
52      @Override
53      public String getText(Mode mode) {
54          if (staticText != null && !staticText.isEmpty()) {
55              return staticText;
56          }
57          StringBuilder sb = new StringBuilder();
58          if (amountToPrevent == Integer.MAX_VALUE) {
59              sb.append("prevent all damage that would be dealt to ");
60          } else {
61              sb.append("prevent the next ").append(amountToPrevent).append(" damage that would be dealt to ");
62          }
63          String targetName = mode.getTargets().get(0).getTargetName();
64          if (targetName.contains("any")) {
65              sb.append(targetName);
66          } else {
67              sb.append("target ").append(targetName);
68          }
69          if (!duration.toString().isEmpty()) {
70              sb.append(' ');
71              if (duration == Duration.EndOfTurn) {
72                  sb.append("this turn");
73              } else {
74                  sb.append(duration.toString());
75              }
76          }
77          return sb.toString();
78      }
79  
80  }

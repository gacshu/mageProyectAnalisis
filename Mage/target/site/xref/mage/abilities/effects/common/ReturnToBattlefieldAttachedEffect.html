1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.OneShotEffect;
5   import mage.cards.Card;
6   import mage.constants.Outcome;
7   import mage.constants.Zone;
8   import mage.game.Game;
9   import mage.game.permanent.Permanent;
10  import mage.players.Player;
11  
12  /**
13   * Used for "Return {this} to the battlefield attached to that creature at the beginning of the next end step" abilities.
14   * E.g.g Gift of Immortality and Next of Kin.
15   *
16   * @author LevelX2, Alex-Vasile
17   */
18  public class ReturnToBattlefieldAttachedEffect extends OneShotEffect {
19  
20      public ReturnToBattlefieldAttachedEffect() {
21          super(Outcome.PutCardInPlay);
22          staticText = "Return {this} to the battlefield attached to that creature at the beginning of the next end step";
23      }
24  
25      public ReturnToBattlefieldAttachedEffect(final ReturnToBattlefieldAttachedEffect effect) {
26          super(effect);
27      }
28  
29      @Override
30      public boolean apply(Game game, Ability source) {
31          Player controller = game.getPlayer(source.getControllerId());
32          Permanent creature = game.getPermanent(getTargetPointer().getFirst(game, source));
33          Card aura = game.getCard(source.getSourceId());
34          if (controller == null
35                  || creature == null
36                  || aura == null || game.getState().getZone(aura.getId()) != Zone.GRAVEYARD) {
37              return false;
38          }
39  
40          game.getState().setValue("attachTo:" + aura.getId(), creature);
41          controller.moveCards(aura, Zone.BATTLEFIELD, source, game);
42          return creature.addAttachment(aura.getId(), source, game);
43      }
44  
45      @Override
46      public ReturnToBattlefieldAttachedEffect copy() {
47          return new ReturnToBattlefieldAttachedEffect(this);
48      }
49  }

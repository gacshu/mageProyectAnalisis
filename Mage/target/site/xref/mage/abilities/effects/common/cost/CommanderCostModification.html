1   package mage.abilities.effects.common.cost;

3   import mage.abilities.Ability;
4   import mage.abilities.common.CastCommanderAbility;
5   import mage.abilities.common.PlayLandAsCommanderAbility;
6   import mage.cards.Card;
7   import mage.constants.CostModificationType;
8   import mage.constants.Duration;
9   import mage.constants.Outcome;
10  import mage.game.Game;
11  
12  /**
13   * @author Plopman
14   */
15      /*
16      903.8. A player may cast a commander they own from the command zone. A commander cast from the
17      command zone costs an additional {2} for each previous time the player casting it has cast it from
18      the command zone that game. This additional cost is informally known as the “commander tax.”
19       */
20  
21  // cast from hand like Remand do not increase commander tax
22  
23  public class CommanderCostModification extends CostModificationEffectImpl {
24  
25      private final Card commander;
26  
27      public CommanderCostModification(Card commander) {
28          super(Duration.Custom, Outcome.Neutral, CostModificationType.INCREASE_COST);
29          this.commander = commander;
30      }
31  
32      public CommanderCostModification(final CommanderCostModification effect) {
33          super(effect);
34          this.commander = effect.commander;
35      }
36  
37      @Override
38      public boolean apply(Game game, Ability source, Ability abilityToModify) {
39          return commander.commanderCost(game, source, abilityToModify);
40      }
41  
42      @Override
43      public boolean applies(Ability abilityToModify, Ability source, Game game) {
44          Card cardToCheck = game.getCard(abilityToModify.getSourceId()); // split/mdf cards support
45          if (cardToCheck == null) {
46              return false;
47          }
48  
49          return commander.getId().equals(cardToCheck.getMainCard().getId())
50                  && (abilityToModify instanceof CastCommanderAbility
51                  || abilityToModify instanceof PlayLandAsCommanderAbility);
52      }
53  
54      @Override
55      public CommanderCostModification copy() {
56          return new CommanderCostModification(this);
57      }
58  }

22  /**
23   * @author TheElk801
24   */
25  public class DraftFromSpellbookEffect extends OneShotEffect {
26  
27      private final List<String> spellbook;
28  
29      public DraftFromSpellbookEffect(List<String> spellbook) {
30          super(Outcome.DrawCard);
31          this.spellbook = spellbook;
32          staticText = "draft a card from {this}'s spellbook";
33      }
34  
35      private DraftFromSpellbookEffect(final DraftFromSpellbookEffect effect) {
36          super(effect);
37          this.spellbook = effect.spellbook;
38      }
39  
40      @Override
41      public DraftFromSpellbookEffect copy() {
42          return new DraftFromSpellbookEffect(this);
43      }
44  
45      @Override
46      public boolean apply(Game game, Ability source) {
47          Player player = game.getPlayer(source.getControllerId());
48          if (player == null) {
49              return false;
50          }
51          Set<String> toSelect = new HashSet<>();
52          while (toSelect.size() < 3) {
53              toSelect.add(RandomUtil.randomFromCollection(spellbook));
54          }
55          Choice choice = new ChoiceImpl(true, ChoiceHintType.CARD);
56          choice.setMessage("Choose a card to draft");
57          choice.setChoices(toSelect);
58          player.choose(outcome, choice, game);
59          String cardName = choice.getChoice();
60          if (cardName == null) {
61              return false;
62          }
63          CardInfo cardInfo = CardRepository
64                  .instance
65                  .findCards(new CardCriteria().nameExact(cardName))
66                  .stream()
67                  .findFirst()
68                  .orElse(null);
69          if (cardInfo == null) {
70              return false;
71          }
72          Set<Card> cards = new HashSet<>();
73          cards.add(cardInfo.getCard());
74          game.loadCards(cards, player.getId());
75          player.moveCards(cards, Zone.HAND, source, game);
76          return true;
77      }
78  }

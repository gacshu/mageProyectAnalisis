1   package mage.abilities.effects.common;
2   
3   import mage.abilities.TriggeredAbilityImpl;
4   import mage.abilities.effects.Effect;
5   import mage.constants.Zone;
6   import mage.game.Game;
7   import mage.game.events.GameEvent;
8   import mage.game.permanent.Permanent;
9   
10  public class CreatureExploresTriggeredAbility extends TriggeredAbilityImpl {
11  
12      public CreatureExploresTriggeredAbility(Effect effect) {
13          super(Zone.BATTLEFIELD, effect, false);
14          setTriggerPhrase("Whenever a creature you control explores, ");
15      }
16  
17      public CreatureExploresTriggeredAbility(final CreatureExploresTriggeredAbility effect) {
18          super(effect);
19      }
20  
21      @Override
22      public boolean checkEventType(GameEvent event, Game game) {
23          return event.getType() == GameEvent.EventType.EXPLORED;
24      }
25  
26      @Override
27      public boolean checkTrigger(GameEvent event, Game game) {
28          Permanent creature = game.getPermanentOrLKIBattlefield(event.getTargetId());
29          if (creature != null) {
30              return creature.isControlledBy(getControllerId());
31          }
32          return false;
33      }
34  
35      @Override
36      public CreatureExploresTriggeredAbility copy() {
37          return new CreatureExploresTriggeredAbility(this);
38      }
39  }

1   package mage.abilities.effects.common.combat;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.RestrictionEffect;
5   import mage.constants.AttachmentType;
6   import mage.constants.Duration;
7   import mage.game.Game;
8   import mage.game.permanent.Permanent;
9   
10  /**
11   * @author North
12   */
13  public class CantBeBlockedAttachedEffect extends RestrictionEffect {
14  
15      public CantBeBlockedAttachedEffect(AttachmentType attachmentType) {
16          super(Duration.WhileOnBattlefield);
17          this.staticText = attachmentType.verb() + " creature can't be blocked";
18      }
19  
20      public CantBeBlockedAttachedEffect(CantBeBlockedAttachedEffect effect) {
21          super(effect);
22      }
23  
24      @Override
25      public CantBeBlockedAttachedEffect copy() {
26          return new CantBeBlockedAttachedEffect(this);
27      }
28  
29      @Override
30      public boolean canBeBlocked(Permanent attacker, Permanent blocker, Ability source, Game game, boolean canUseChooseDialogs) {
31          return false;
32      }
33  
34      @Override
35      public boolean applies(Permanent permanent, Ability source, Game game) {
36          return permanent.getAttachments().contains(source.getSourceId());
37      }
38  }

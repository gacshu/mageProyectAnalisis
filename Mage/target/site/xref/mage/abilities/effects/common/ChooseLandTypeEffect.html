4   import mage.abilities.Ability;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.choices.Choice;
7   import mage.choices.ChoiceImpl;
8   import mage.constants.Outcome;
9   import mage.constants.SubType;
10  import mage.game.Game;
11  import mage.game.permanent.Permanent;
12  import mage.players.Player;
13  import mage.util.CardUtil;
14  
15  import java.util.stream.Collectors;
16  
17  /**
18   * @author fireshoes
19   */
20  public class ChooseLandTypeEffect extends OneShotEffect {
21  
22      public ChooseLandTypeEffect(Outcome outcome) {
23          super(outcome);
24          staticText = "choose a land type";
25      }
26  
27      public ChooseLandTypeEffect(final ChooseLandTypeEffect effect) {
28          super(effect);
29      }
30  
31      @Override
32      public boolean apply(Game game, Ability source) {
33          Player controller = game.getPlayer(source.getControllerId());
34          MageObject mageObject = game.getPermanentEntering(source.getSourceId());
35          if (mageObject == null) {
36              mageObject = game.getObject(source);
37          }
38          if (controller != null && mageObject != null) {
39              Choice typeChoice = new ChoiceImpl(true);
40              typeChoice.setMessage("Choose land type");
41              typeChoice.setChoices(SubType.getLandTypes().stream().map(SubType::toString).collect(Collectors.toSet()));
42              if (controller.choose(outcome, typeChoice, game)) {
43                  if (!game.isSimulation()) {
44                      game.informPlayers(mageObject.getName() + ": " + controller.getLogName() + " has chosen " + typeChoice.getChoice());
45                  }
46                  game.getState().setValue(mageObject.getId() + "_type", SubType.byDescription(typeChoice.getChoice()));
47                  if (mageObject instanceof Permanent) {
48                      ((Permanent) mageObject).addInfo("chosen type", CardUtil.addToolTipMarkTags("Chosen type: " + typeChoice.getChoice()), game);
49                  }
50                  return true;
51              }
52          }
53          return false;
54      }
55  
56      @Override
57      public ChooseLandTypeEffect copy() {
58          return new ChooseLandTypeEffect(this);
59      }
60  
61  }

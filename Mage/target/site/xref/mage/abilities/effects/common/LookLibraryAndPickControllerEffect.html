77          this.numberToPick = numberToPick;
78          this.putPickedCards = putPickedCards;
79          this.filter = (numberToPick > 1) ? StaticFilters.FILTER_CARD_CARDS : StaticFilters.FILTER_CARD_A;
80          this.revealPickedCards = false;
81          this.optional = false;
82          this.upTo = upTo || numberToPick == Integer.MAX_VALUE;
83      }

100     public LookLibraryAndPickControllerEffect(DynamicValue numberOfCards, int numberToPick, FilterCard filter,
101                                               PutCards putPickedCards, PutCards putLookedCards, boolean optional) {
102         super(putPickedCards.getOutcome(), numberOfCards, putLookedCards);
103         this.numberToPick = numberToPick;
104         this.putPickedCards = putPickedCards;
105         this.filter = filter;
106         this.revealPickedCards = !putPickedCards.getZone().isPublicZone();
107         this.optional = optional;
108         this.upTo = (numberToPick > 1);
109     }

111     public LookLibraryAndPickControllerEffect(final LookLibraryAndPickControllerEffect effect) {
112         super(effect);
113         this.numberToPick = effect.numberToPick;
114         this.putPickedCards = effect.putPickedCards;
115         this.filter = effect.filter.copy();
116         this.revealPickedCards = effect.revealPickedCards;
117         this.optional = effect.optional;
118         this.upTo = effect.upTo;
119     }

121     @Override
122     public LookLibraryAndPickControllerEffect copy() {
123         return new LookLibraryAndPickControllerEffect(this);
124     }

126     @Override
127     protected boolean actionWithLookedCards(Game game, Ability source, Player player, Cards cards) {
128         int number = Math.min(numberToPick, cards.count(filter, source.getControllerId(), source, game));
129         if (number < 1
130                 || optional && !player.chooseUse(putPickedCards.getOutcome(), getMayText(), source, game)) {
131             return actionWithPickedCards(game, source, player, new CardsImpl(), cards);
132         }
133         TargetCard target = new TargetCard((upTo ? 0 : number), number, Zone.LIBRARY, filter);
134         target.withChooseHint(getChooseHint());
135         if (!player.chooseTarget(putPickedCards.getOutcome(), cards, target, source, game)) {
136             return actionWithPickedCards(game, source, player, new CardsImpl(), cards);
137         }
138         Cards pickedCards = new CardsImpl(target.getTargets());
139         if (revealPickedCards) {
140             player.revealCards(source, pickedCards, game);
141         }
142         cards.removeAll(pickedCards);
143         return actionWithPickedCards(game, source, player, pickedCards, cards);
144     }
145 
146     protected boolean actionWithPickedCards(Game game, Ability source, Player player, Cards pickedCards, Cards otherCards) {
147         boolean result = putPickedCards.moveCards(player, pickedCards, source, game);
148         result |= putLookedCards.moveCards(player, otherCards, source, game);
149         return result;
150     }
151 
152     protected String getMayText() {
153         boolean plural = numberToPick > 1;
154         StringBuilder sb = new StringBuilder(revealPickedCards ? "Reveal " : "Put ");
155         sb.append(plural ? filter.getMessage() : CardUtil.addArticle(filter.getMessage()));
156         if (revealPickedCards) {
157             sb.append(" and put ");
158             sb.append(plural ? "them" : "it");
159         }
160         sb.append(" ");
161         sb.append(putPickedCards.getMessage(false, plural));
162         return sb.append("?").toString();
163     }
164 
165     protected String getChooseHint() {
166         return "to put " + putPickedCards.getMessage(false, numberToPick > 1);
167     }
168 
169     @Override
170     public String getText(Mode mode) {
171         if (staticText != null && !staticText.isEmpty()) {
172             return staticText;
173         }
174         StringBuilder sb = new StringBuilder(". ");
175         if (optional) {
176             sb.append(revealPickedCards ? "You may reveal " : "You may put ");
177         } else {
178             sb.append(revealPickedCards ? "Reveal " : "Put ");
179         }
180         boolean havePredicates = filter.hasPredicates();
181         boolean plural = numberToPick > 1;
182         if (havePredicates && !plural && !upTo) {
183             sb.append(CardUtil.addArticle(filter.getMessage()));
184         } else if (numberToPick == Integer.MAX_VALUE) {
185             sb.append("any number of ");
186             if (havePredicates) {
187                 sb.append(filter.getMessage());
188             }
189         } else {
190             if (upTo) {
191                 sb.append("up to ");
192             }
193             sb.append(CardUtil.numberToText(numberToPick));
194             sb.append(" ");
195             sb.append(havePredicates ? filter.getMessage() : "of ");
196         }
197         if (havePredicates) {
198             sb.append(" from among ");
199         }
200         sb.append("them ");
201         if (revealPickedCards) {
202             sb.append("and put ");
203             sb.append(plural ? "them " : "it ");
204         }
205         sb.append(putPickedCards.getMessage(false, plural));
206 
207         plural = optional
208                 || upTo
209                 || !(numberOfCards instanceof StaticValue)
210                 || numberOfCards.calculate(null, null, this) - numberToPick != 1;
211 
212         // if remaining text would be "put the other on top of your library", omit it
213         if (!plural && putLookedCards == PutCards.TOP_ANY) {
214             return setText(mode, sb.toString());
215         }
216         sb.append(havePredicates && (optional || upTo) ? ". Put" : " and");
217         sb.append(" the ");
218         sb.append(plural ? "rest " : "other ");
219         if (putPickedCards == PutCards.GRAVEYARD && putLookedCards == PutCards.TOP_ANY) {
220             sb.append("back ");
221         }
222         sb.append(putLookedCards.getMessage(false, plural));
223 
224         // get text frame from super class and inject action text
225         return setText(mode, sb.toString());
226     }

1   package mage.abilities.effects.common.replacement;
2   
3   import mage.MageObjectReference;
4   import mage.abilities.Ability;
5   import mage.abilities.effects.ReplacementEffectImpl;
6   import mage.constants.Duration;
7   import mage.constants.Outcome;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  import mage.game.events.GameEvent.EventType;
12  import mage.game.events.ZoneChangeEvent;
13  import mage.game.permanent.Permanent;
14  import mage.players.Player;
15  
16  /**
17   *
18   * @author LevelX2
19   */
20  public class DiesReplacementEffect extends ReplacementEffectImpl {
21  
22      private final MageObjectReference objectRef;
23  
24      public DiesReplacementEffect(MageObjectReference objectRef, Duration duration) {
25          super(duration, Outcome.Exile);
26          this.objectRef = objectRef;
27          staticText = "If that creature would die " + (duration == Duration.EndOfTurn ? "this turn" : "") + ", exile it instead";
28      }
29  
30      public DiesReplacementEffect(final DiesReplacementEffect effect) {
31          super(effect);
32          this.objectRef = effect.objectRef;
33      }
34  
35      @Override
36      public DiesReplacementEffect copy() {
37          return new DiesReplacementEffect(this);
38      }
39  
40      @Override
41      public boolean apply(Game game, Ability source) {
42          return true;
43      }
44  
45      @Override
46      public boolean replaceEvent(GameEvent event, Ability source, Game game) {
47          Permanent permanent = ((ZoneChangeEvent) event).getTarget();
48          Player controller = game.getPlayer(source.getControllerId());
49          if (controller != null && permanent != null) {
50              return controller.moveCards(permanent, Zone.EXILED, source, game);
51          }
52          return false;
53      }
54  
55      @Override
56      public boolean checksEventType(GameEvent event, Game game) {
57          return event.getType() == GameEvent.EventType.ZONE_CHANGE;
58      }
59  
60      @Override
61      public boolean applies(GameEvent event, Ability source, Game game) {
62          ZoneChangeEvent zce = (ZoneChangeEvent) event;
63          return zce.isDiesEvent()
64                  && objectRef.equals(new MageObjectReference(zce.getTarget(), game));
65      }
66  
67  }

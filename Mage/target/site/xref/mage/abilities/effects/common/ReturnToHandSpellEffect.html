1   package mage.abilities.effects.common;

3   import mage.abilities.Ability;
4   import mage.abilities.MageSingleton;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.cards.Card;
7   import mage.constants.Outcome;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  import mage.players.Player;
11  
12  /**
13   *
14   * @author Loki
15   */
16  public class ReturnToHandSpellEffect extends OneShotEffect implements MageSingleton {
17  
18      private static final ReturnToHandSpellEffect instance = new ReturnToHandSpellEffect();
19  
20      public static ReturnToHandSpellEffect getInstance() {
21          return instance;
22      }
23  
24      private ReturnToHandSpellEffect() {
25          super(Outcome.Benefit);
26          staticText = "Return {this} to its owner's hand";
27      }
28  
29      @Override
30      public boolean apply(Game game, Ability source) {
31          Player controller = game.getPlayer(source.getControllerId());
32          if (controller != null) {
33              Card spellCard = game.getStack().getSpell(source.getSourceId()).getCard();
34              controller.moveCards(spellCard, Zone.HAND, source, game);
35              return true;
36          }
37          return false;
38      }
39  
40      @Override
41      public ReturnToHandSpellEffect copy() {
42          return instance;
43      }
44  }

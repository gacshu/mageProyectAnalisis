2   package mage.abilities.effects.common.search;
3   
4   import mage.MageObject;
5   import mage.abilities.Ability;
6   import mage.abilities.effects.SearchEffect;
7   import mage.cards.Cards;
8   import mage.cards.CardsImpl;
9   import mage.constants.Outcome;
10  import mage.game.Game;
11  import mage.players.Player;
12  import mage.target.common.TargetCardInLibrary;
13  import mage.util.CardUtil;
14  
15  /**
16   * @author BetaSteward_at_googlemail.com
17   */
18  public class SearchLibraryPutOnLibraryEffect extends SearchEffect {
19  
20      private boolean reveal;
21      private boolean forceShuffle;
22  
23      public SearchLibraryPutOnLibraryEffect(TargetCardInLibrary target) {
24          this(target, false, true);
25          setText();
26      }
27  
28      public SearchLibraryPutOnLibraryEffect(TargetCardInLibrary target, boolean reveal, boolean forceShuffle) {
29          super(target, Outcome.DrawCard);
30          this.reveal = reveal;
31          this.forceShuffle = forceShuffle;
32          setText();
33      }
34  
35      public SearchLibraryPutOnLibraryEffect(final SearchLibraryPutOnLibraryEffect effect) {
36          super(effect);
37          this.reveal = effect.reveal;
38          this.forceShuffle = effect.forceShuffle;
39      }
40  
41      @Override
42      public SearchLibraryPutOnLibraryEffect copy() {
43          return new SearchLibraryPutOnLibraryEffect(this);
44      }
45  
46      @Override
47      public boolean apply(Game game, Ability source) {
48          Player controller = game.getPlayer(source.getControllerId());
49          MageObject sourceObject = source.getSourceObject(game);
50          if (controller == null || sourceObject == null) {
51              return false;
52          }
53          if (controller.searchLibrary(target, source, game)) {
54              Cards foundCards = new CardsImpl(target.getTargets());
55              if (reveal && !foundCards.isEmpty()) {
56                  controller.revealCards(sourceObject.getIdName(), foundCards, game);
57              }
58              if (forceShuffle) {
59                  controller.shuffleLibrary(source, game);
60              }
61              controller.putCardsOnTopOfLibrary(foundCards, game, source, reveal);
62              return true;
63          }
64          // shuffle
65          if (forceShuffle) {
66              controller.shuffleLibrary(source, game);
67          }
68          return false;
69      }
70  
71      private void setText() {
72          StringBuilder sb = new StringBuilder();
73          sb.append("search your library for ").append(CardUtil.addArticle(target.getTargetName()));
74          if (reveal) {
75              sb.append(", reveal it");
76          }
77          sb.append(", then shuffle and put that card on top");
78          staticText = sb.toString();
79      }
80  }

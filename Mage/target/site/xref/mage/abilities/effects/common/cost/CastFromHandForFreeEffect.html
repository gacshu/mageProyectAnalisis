1   package mage.abilities.effects.common.cost;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.effects.OneShotEffect;
5   import mage.cards.CardsImpl;
6   import mage.constants.Outcome;
7   import mage.filter.FilterCard;
8   import mage.game.Game;
9   import mage.players.Player;
10  import mage.util.CardUtil;
11  
12  /**
13   * @author fireshoes - Original Code
14   * @author JRHerlehy - Implement as seperate class
15   * <p>
16   * Allows player to choose to cast as card from hand without paying its mana
17   * cost.
18   * </p>
19   */
20  public class CastFromHandForFreeEffect extends OneShotEffect {
21  
22      private final FilterCard filter;
23  
24      public CastFromHandForFreeEffect(FilterCard filter) {
25          super(Outcome.PlayForFree);
26          this.filter = filter;
27          this.staticText = "you may cast " + filter.getMessage() + " from your hand without paying its mana cost";
28      }
29  
30      public CastFromHandForFreeEffect(final CastFromHandForFreeEffect effect) {
31          super(effect);
32          this.filter = effect.filter;
33      }
34  
35      @Override
36      public boolean apply(Game game, Ability source) {
37          Player controller = game.getPlayer(source.getControllerId());
38          if (controller == null) {
39              return false;
40          }
41          return CardUtil.castSpellWithAttributesForFree(controller, source, game, new CardsImpl(controller.getHand()), filter);
42      }
43  
44      @Override
45      public CastFromHandForFreeEffect copy() {
46          return new CastFromHandForFreeEffect(this);
47      }
48  }

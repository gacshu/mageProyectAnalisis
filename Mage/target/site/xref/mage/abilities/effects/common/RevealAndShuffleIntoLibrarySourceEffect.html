14  import mage.game.permanent.Permanent;
15  import mage.game.permanent.PermanentCard;
16  import mage.game.stack.Spell;
17  import mage.players.Player;

20   *
21   * @author LevelX2
22   */
23  
24  public class RevealAndShuffleIntoLibrarySourceEffect extends OneShotEffect {
25      
26      public RevealAndShuffleIntoLibrarySourceEffect() {
27          super(Outcome.Neutral);
28          staticText = "reveal {this} and shuffle it into its owner's library instead";
29      }
30  
31      public RevealAndShuffleIntoLibrarySourceEffect(final RevealAndShuffleIntoLibrarySourceEffect effect) {
32          super(effect);
33      }
34  
35      @Override
36      public boolean apply(Game game, Ability source) {
37          MageObject sourceObject = source.getSourceObjectIfItStillExists(game);
38          Player controller = game.getPlayer(source.getControllerId());
39          if (sourceObject != null && controller != null) {
40              Player owner = null;
41              Cards cards = new CardsImpl();
42              Permanent permanent = null;
43              if (sourceObject instanceof Spell) {
44                  sourceObject = ((Spell)sourceObject).getCard();
45              }
46              if (sourceObject instanceof Permanent) {
47                  permanent = (Permanent) sourceObject;
48                  owner = game.getPlayer(permanent.getOwnerId());
49                  if (sourceObject instanceof PermanentCard) {
50                      cards.add(permanent);
51                  }
52              } else if (sourceObject instanceof Card) {
53                  owner = game.getPlayer(((Card)sourceObject).getOwnerId());
54                  cards.add((Card)sourceObject);
55              }
56              if (owner != null) {
57                  Zone fromZone = game.getState().getZone(sourceObject.getId());
58                  if (!cards.isEmpty()) {
59                      controller.revealCards(sourceObject.getName(), cards, game);
60                  }
61                  if (permanent != null) {
62                      controller.moveCardToLibraryWithInfo(permanent, source, game, fromZone, true, true);
63                  } else {
64                      controller.moveCardToLibraryWithInfo((Card) sourceObject, source, game, fromZone, true, true);
65                  }
66                  if (!cards.isEmpty()) {
67                      controller.shuffleLibrary(source, game);
68                  }
69              }
70              return true;
71          }
72          return false;
73      }
74  
75      @Override
76      public RevealAndShuffleIntoLibrarySourceEffect copy() {
77          return new RevealAndShuffleIntoLibrarySourceEffect(this);
78      }
79  
80  }

1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.costs.common.DiscardCardCost;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.constants.Outcome;
7   import mage.constants.SubType;
8   import mage.filter.FilterCard;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  
12  /**
13   * @author TheElk801
14   */
15  public class LearnEffect extends OneShotEffect {
16  
17      private static final FilterCard filter = new FilterCard("Lesson card");
18  
19      static {
20          filter.add(SubType.LESSON.getPredicate());
21      }
22  
23      private static final String defaultText = "learn. <i>(You may reveal a Lesson card you own from outside the game " +
24              "and put it into your hand, or discard a card to draw a card.)</i>";
25  
26      public LearnEffect() {
27          super(Outcome.Neutral);
28          staticText = defaultText;
29      }
30  
31      private LearnEffect(final LearnEffect effect) {
32          super(effect);
33      }
34  
35      @Override
36      public boolean apply(Game game, Ability source) {
37          if (game.replaceEvent(GameEvent.getEvent(
38                  GameEvent.EventType.LEARN, source.getSourceId(),
39                  source, source.getControllerId()
40          ))) {
41              return false;
42          }
43          return new WishEffect(filter, true).apply(game, source)
44                  || new DoIfCostPaid(
45                  new DrawCardSourceControllerEffect(1), new DiscardCardCost()
46          ).apply(game, source);
47      }
48  
49      @Override
50      public LearnEffect copy() {
51          return new LearnEffect(this);
52      }
53  
54      public static String getDefaultText() {
55          return defaultText;
56      }
57  }

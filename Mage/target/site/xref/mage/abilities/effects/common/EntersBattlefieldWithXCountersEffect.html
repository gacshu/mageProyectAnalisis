17  /**
18   * Use this effect only (I guess) with EntersBattlefieldAbility like abilities
19   *
20   * @author LevelX2
21   */
22  public class EntersBattlefieldWithXCountersEffect extends OneShotEffect {
23  
24      protected final Counter counter;
25  
26      public EntersBattlefieldWithXCountersEffect(Counter counter) {
27          super(Outcome.BoostCreature);
28          this.counter = counter;
29          staticText = "with X " + counter.getName() + " counters on it";
30      }
31  
32      public EntersBattlefieldWithXCountersEffect(final EntersBattlefieldWithXCountersEffect effect) {
33          super(effect);
34          this.counter = effect.counter.copy();
35      }
36  
37      @Override
38      public boolean apply(Game game, Ability source) {
39          Permanent permanent = game.getPermanent(source.getSourceId());
40          if (permanent == null) {
41              if (source.getAbilityType() == AbilityType.STATIC) {
42                  permanent = game.getPermanentEntering(source.getSourceId());
43              }
44          }
45          if (permanent != null) {
46              SpellAbility spellAbility = (SpellAbility) getValue(EntersBattlefieldEffect.SOURCE_CAST_SPELL_ABILITY);
47              if (spellAbility != null
48                      && spellAbility.getSourceId().equals(source.getSourceId())
49                      && permanent.getZoneChangeCounter(game) == spellAbility.getSourceObjectZoneChangeCounter()) {
50                  if (spellAbility.getSourceId().equals(source.getSourceId())) { // put into play by normal cast
51                      int amount = spellAbility.getManaCostsToPay().getX();
52                      if (amount > 0) {
53                          Counter counterToAdd = counter.copy();
54                          counterToAdd.add(amount - counter.getCount());
55                          List<UUID> appliedEffects = (ArrayList<UUID>) this.getValue("appliedEffects");
56                          permanent.addCounters(counterToAdd, source.getControllerId(), source, game, appliedEffects);
57                      }
58                  }
59              }
60          }
61          return true;
62      }
63  
64      @Override
65      public EntersBattlefieldWithXCountersEffect copy() {
66          return new EntersBattlefieldWithXCountersEffect(this);
67      }
68  
69  }

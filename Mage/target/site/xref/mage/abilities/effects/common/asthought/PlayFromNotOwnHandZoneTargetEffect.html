83      @Override
84      public boolean applies(UUID objectId, Ability affectedAbility, Ability source, Game game, UUID playerId) {
85          if (affectedAbility == null) {
86              // ContinuousEffects.asThough already checks affectedAbility, so that error must never be called here
87              // PLAY_FROM_NOT_OWN_HAND_ZONE must applies to affectedAbility only
88              // If you see it then parent conditional effect must override both applies methods to support different
89              // AsThough effect types in one conditional effect
90              throw new IllegalArgumentException("ERROR, can't call applies method on empty affectedAbility");
91          }
92  
93          // invalid targets
94          List<UUID> targets = getTargetPointer().getTargets(game, source);
95          if (targets.isEmpty()) {
96              this.discard();
97              return false;
98          }
99  
100         // invalid zone
101         if (!game.getState().getZone(objectId).match(fromZone)) {
102             return false;
103         }
104 
105         // invalid caster
106         switch (allowedCaster) {
107             case YOU:
108                 if (playerId != source.getControllerId()) {
109                     return false;
110                 }
111                 break;
112             case OPPONENT:
113                 if (!game.getOpponents(source.getControllerId()).contains(playerId)) {
114                     return false;
115                 }
116                 break;
117             case OWNER:
118                 if (playerId != game.getCard(objectId).getOwnerId()) {
119                     return false;
120                 }
121                 break;
122             case ANY:
123                 break;
124         }
125 
126         // targets goes to main card all the time
127         UUID objectIdToCast = CardUtil.getMainCardId(game, objectId);
128         if (!targets.contains(objectIdToCast)) {
129             return false;
130         }
131 
132         // if can't play lands
133         if (!affectedAbility.getAbilityType().isPlayCardAbility()
134                 || onlyCastAllowed && affectedAbility instanceof PlayLandAbility) {
135             return false;
136         }
137 
138         // OK, allow to play
139         if (withoutMana) {
140             allowCardToPlayWithoutMana(objectId, source, playerId, game);
141         }
142         return true;
143     }

145     public static boolean exileAndPlayFromExile(Game game, Ability source, Card card, TargetController allowedCaster,
146                                                 Duration duration, boolean withoutMana, boolean anyColor, boolean onlyCastAllowed) {
147         if (card == null) {
148             return true;
149         }
150         Set<Card> cards = new HashSet<>();
151         cards.add(card);
152         return exileAndPlayFromExile(game, source, cards, allowedCaster, duration, withoutMana, anyColor, onlyCastAllowed);
153     }
154 
155     /**
156      * Exiles the cards and let the allowed player play them from exile for the given duration
157      * Supports:
158      *  - cards (use any side)
159      *  - permanents (use permanent, not permanent's card)
160      *
161      * @param game
162      * @param source
163      * @param cards
164      * @param allowedCaster
165      * @param duration
166      * @param withoutMana
167      * @param anyColor
168      * @param onlyCastAllowed true for rule "cast that card" and false for rule "play that card"
169      * @return
170      */
171     public static boolean exileAndPlayFromExile(Game game, Ability source, Set<Card> cards, TargetController allowedCaster,
172                                                 Duration duration, boolean withoutMana, boolean anyColor, boolean onlyCastAllowed) {
173         if (cards == null || cards.isEmpty()) {
174             return true;
175         }
176         Player controller = game.getPlayer(source.getControllerId());
177         MageObject sourceObject = source.getSourceObject(game);
178         if (controller == null || sourceObject == null) {
179             return false;
180         }
181         UUID exileId = CardUtil.getExileZoneId(
182                 controller.getId().toString()
183                         + "-" + game.getState().getTurnNum()
184                         + "-" + sourceObject.getIdName(), game
185         );
186         String exileName = sourceObject.getIdName() + " free play"
187                 + (Duration.EndOfTurn.equals(duration) ? " on turn " + game.getState().getTurnNum() : "")
188                 + " for " + controller.getName();
189         if (Duration.EndOfTurn.equals(duration)) {
190             game.getExile().createZone(exileId, exileName).setCleanupOnEndTurn(true);
191         }
192         if (!controller.moveCardsToExile(cards, source, game, true, exileId, exileName)) {
193             return false;
194         }
195 
196         // get real cards (if it was called on permanent instead card, example: Release to the Wind)
197         Set<Card> cardsToPlay = cards
198                 .stream()
199                 .map(Card::getMainCard)
200                 .filter(card -> Zone.EXILED.equals(game.getState().getZone(card.getId())))
201                 .collect(Collectors.toSet());
202 
203         ContinuousEffect effect = new PlayFromNotOwnHandZoneTargetEffect(Zone.EXILED, allowedCaster, duration, withoutMana, onlyCastAllowed);
204         effect.setTargetPointer(new FixedTargets(cardsToPlay, game));
205         game.addEffect(effect, source);
206         if (anyColor) {
207             for (Card card : cardsToPlay) {
208                 game.addEffect(new YouMaySpendManaAsAnyColorToCastTargetEffect(duration).setTargetPointer(new FixedTarget(card, game)), source);
209             }
210         }
211         return true;
212     }

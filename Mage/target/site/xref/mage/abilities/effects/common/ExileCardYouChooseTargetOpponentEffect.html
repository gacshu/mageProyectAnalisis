5   import mage.cards.Card;

7   import mage.constants.Zone;
8   import mage.filter.FilterCard;
9   import mage.game.Game;
10  import mage.players.Player;
11  import mage.target.TargetCard;
12  
13  /**
14   * @author LevelX2
15   */
16  public class ExileCardYouChooseTargetOpponentEffect extends OneShotEffect {
17  
18      private final FilterCard filter;
19  
20      public ExileCardYouChooseTargetOpponentEffect(FilterCard filter) {
21          super(Outcome.Discard);
22          this.staticText = "target opponent reveals their hand. You choose "
23                  + filter.getMessage() + " from it and exile that card";
24          this.filter = filter;
25      }
26  
27      private ExileCardYouChooseTargetOpponentEffect(final ExileCardYouChooseTargetOpponentEffect effect) {
28          super(effect);
29          this.filter = effect.filter;
30      }
31  
32      @Override
33      public boolean apply(Game game, Ability source) {
34          Player controller = game.getPlayer(source.getControllerId());
35          Player opponent = game.getPlayer(targetPointer.getFirst(game, source));
36          if (controller == null || opponent == null) {
37              return false;
38          }
39          opponent.revealCards(source, opponent.getHand(), game);
40          if (opponent.getHand().count(filter, game) < 1) {
41              return true;
42          }
43          TargetCard target = new TargetCard(Zone.HAND, filter);
44          controller.choose(Outcome.Exile, opponent.getHand(), target, game);
45          Card card = opponent.getHand().get(target.getFirstTarget(), game);
46          if (card != null) {
47              controller.moveCards(card, Zone.EXILED, source, game);
48          }
49          return true;
50      }
51  
52      @Override
53      public ExileCardYouChooseTargetOpponentEffect copy() {
54          return new ExileCardYouChooseTargetOpponentEffect(this);
55      }
56  }

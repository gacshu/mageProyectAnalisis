1   package mage.abilities.effects.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.Mode;
5   import mage.abilities.effects.OneShotEffect;
6   import mage.constants.Outcome;
7   import mage.game.Game;
8   import mage.game.stack.StackObject;
9   
10  /**
11   * @author BetaSteward_at_googlemail.com
12   */
13  public class ChooseNewTargetsTargetEffect extends OneShotEffect {
14  
15      private final boolean forceChange;
16      private final boolean onlyOneTarget;
17  
18      public ChooseNewTargetsTargetEffect() {
19          this(false, false);
20      }
21  
22      /**
23       * @param forceChange     forces the user to choose another target (only targets
24       *                        with maxtargets = 1 supported)
25       * @param onlyOneTarget   only one target can be selected for the change
26       * @param filterNewTarget restriction to the new target
27       */
28      public ChooseNewTargetsTargetEffect(boolean forceChange, boolean onlyOneTarget) {
29          super(Outcome.Benefit);
30          this.forceChange = forceChange;
31          this.onlyOneTarget = onlyOneTarget;
32      }
33  
34      public ChooseNewTargetsTargetEffect(final ChooseNewTargetsTargetEffect effect) {
35          super(effect);
36          this.forceChange = effect.forceChange;
37          this.onlyOneTarget = effect.onlyOneTarget;
38      }
39  
40      @Override
41      public boolean apply(Game game, Ability source) {
42          StackObject stackObject = game.getStack().getStackObject(source.getFirstTarget());
43          if (stackObject != null) {
44              return stackObject.chooseNewTargets(game, source.getControllerId(), forceChange, onlyOneTarget, null);
45          }
46          return false;
47      }
48  
49      @Override
50      public ChooseNewTargetsTargetEffect copy() {
51          return new ChooseNewTargetsTargetEffect(this);
52      }
53  
54      @Override
55      public String getText(Mode mode) {
56          if (staticText != null && !staticText.isEmpty()) {
57              return staticText;
58          }
59          StringBuilder sb = new StringBuilder();
60          if (forceChange) {
61              sb.append("change the target of target ");
62          } else {
63              sb.append("you may choose new targets for target ");
64          }
65          sb.append(mode.getTargets().get(0).getTargetName());
66          return sb.toString();
67      }
68  }

1   package mage.abilities.effects.common;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.effects.ContinuousRuleModifyingEffectImpl;
6   import mage.constants.Duration;
7   import mage.constants.Outcome;
8   import mage.constants.PhaseStep;
9   import mage.game.Game;
10  import mage.game.events.GameEvent;
11  
12  public class DontUntapInControllersNextUntapStepSourceEffect extends ContinuousRuleModifyingEffectImpl {
13  
14      private int validForTurnNum;
15      
16      public DontUntapInControllersNextUntapStepSourceEffect() {
17          super(Duration.Custom, Outcome.Detriment, false, true);
18          staticText = "{this} doesn't untap during your next untap step";
19          validForTurnNum = 0;
20      }
21  
22      public DontUntapInControllersNextUntapStepSourceEffect(final DontUntapInControllersNextUntapStepSourceEffect effect) {
23          super(effect);
24      }
25  
26      @Override
27      public DontUntapInControllersNextUntapStepSourceEffect copy() {
28          return new DontUntapInControllersNextUntapStepSourceEffect(this);
29      }
30  
31      @Override
32      public boolean apply(Game game, Ability source) {
33          return false;
34      }
35  
36      @Override
37      public String getInfoMessage(Ability source, GameEvent event, Game game) {
38          MageObject mageObject = game.getObject(source);
39          if (mageObject != null) {
40              return "{this} doesn't untap (" + mageObject.getLogName() + ')';
41          }
42          return null;
43      }
44  
45      @Override
46      public boolean checksEventType(GameEvent event, Game game) {
47          return event.getType() == GameEvent.EventType.UNTAP_STEP || event.getType() == GameEvent.EventType.UNTAP;
48      }
49  
50      @Override
51      public boolean applies(GameEvent event, Ability source, Game game) {
52          // the check for turn number is needed if multiple effects are added to prevent untap in next untap step
53          // if we don't check for turn number, every turn only one effect would be used instead of correctly consuming
54          // all existing skip the next untap step effects.
55          
56          // Discard effect if related to a previous turn
57          if (validForTurnNum > 0 && validForTurnNum < game.getTurnNum()) {
58              discard();
59              return false;
60          }
61          // remember the turn of the untap step the effect has to be applied
62          if (event.getType() == GameEvent.EventType.UNTAP_STEP
63                  && game.isActivePlayer(source.getControllerId())) {
64              if (validForTurnNum == game.getTurnNum()) { // the turn has a second untap step but the effect is already related to the first untap step
65                  discard();
66                  return false;                
67              }
68              validForTurnNum = game.getTurnNum();
69          }
70          // skip untap action
71          if (game.getTurnStepType() == PhaseStep.UNTAP
72                  && event.getType() == GameEvent.EventType.UNTAP
73                  && game.isActivePlayer(source.getControllerId())
74                  && event.getTargetId().equals(source.getSourceId())) {
75                  discard();
76              return true;
77          }
78          return false;
79      }
80  
81  }

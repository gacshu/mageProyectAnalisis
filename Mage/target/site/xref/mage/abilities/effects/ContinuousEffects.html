167      * Return effects list ordered by timestamps (timestamps are automaticity
168      * generates from new/old lists on same layer)

170      * @param game
171      * @param timestampGroupName workaround to fix broken timestamps on effect's
172      *                           add/remove between different layers
173      * @return effects list ordered by timestamp
174      */
175     public synchronized List<ContinuousEffect> getLayeredEffects(Game game, String timestampGroupName) {
176         List<ContinuousEffect> layerEffects = new ArrayList<>();
177         for (ContinuousEffect effect : layeredEffects) {
178             switch (effect.getDuration()) {
179                 case WhileOnBattlefield:
180                 case WhileControlled:
181                 case WhileOnStack:
182                 case WhileInGraveyard:
183                     Set<Ability> abilities = layeredEffects.getAbility(effect.getId());
184                     if (!abilities.isEmpty()) {
185                         for (Ability ability : abilities) {
186                             // If e.g. triggerd abilities (non static) created the effect, the ability must not be in usable zone (e.g. Unearth giving Haste effect)
187                             if (!(ability instanceof StaticAbility) || ability.isInUseableZone(game, null, null)) {
188                                 layerEffects.add(effect);
189                                 break;
190                             }
191                         }
192                     } else {
193                         logger.error("No abilities for continuous effect: " + effect);
194                     }
195                     break;
196                 default:
197                     layerEffects.add(effect);
198             }
199         }
200 
201         updateTimestamps(timestampGroupName, layerEffects);
202         layerEffects.sort(Comparator.comparingLong(ContinuousEffect::getOrder));
203         /* debug effects apply order:
204         if (game.getStep() != null) System.out.println("layr - " + game.getTurnNum() + "." + game.getTurnStepType() + ": layers " + layerEffects.size()
205                 + " - " + layerEffects.stream().map(l -> l.getClass().getSimpleName()).collect(Collectors.joining(", "))
206                 + " - " + callName);
207         //*/
208 
209         return layerEffects;
210     }

212     /**
213      * Initially effect timestamp is set when game starts in game.loadCard
214      * method. After that timestamp should be updated whenever effect becomes
215      * "actual" meaning it becomes turned on that is defined by
216      * Ability.#isInUseableZone(Game, boolean) method in
217      * #getLayeredEffects(Game).
218      * <p>
219      * It must be called with different timestamp group name (otherwise sort
220      * order will be changed for add/remove effects, see Urborg and Bloodmoon
221      * test)
222      *
223      * @param layerEffects
224      */
225     private synchronized void updateTimestamps(String timestampGroupName, List<ContinuousEffect> layerEffects) {
226         if (!lastEffectsListOnLayer.containsKey(timestampGroupName)) {
227             lastEffectsListOnLayer.put(timestampGroupName, new ContinuousEffectsList<>());
228         }
229         ContinuousEffectsList<ContinuousEffect> prevs = lastEffectsListOnLayer.get(timestampGroupName);
230         for (ContinuousEffect continuousEffect : layerEffects) {
231             // check if it's new, then set order
232             if (!prevs.contains(continuousEffect)) {
233                 setOrder(continuousEffect);
234             }
235         }
236         prevs.clear();
237         prevs.addAll(layerEffects);
238     }

316                 for (Ability ability : abilities) {
317                     // for replacment effects of static abilities do not use LKI to check if to apply
318                     if (ability.getAbilityType() != AbilityType.STATIC || ability.isInUseableZone(game, null, event)) {
319                         if (effect.getDuration() != Duration.OneUse || !effect.isUsed()) {
320                             if (!game.getScopeRelevant() || effect.hasSelfScope() || !event.getTargetId().equals(ability.getSourceId())) {
321                                 if (effect.applies(event, ability, game)
322                                         && !((PayCostToAttackBlockEffect) effect).isCostless(event, ability, game)) {
323                                     return true;
324                                 }
325                             }
326                         }
327                     }
328                 }
329             }
330         }
331         return false;

340     private Map<ReplacementEffect, Set<Ability>> getApplicableReplacementEffects(GameEvent event, Game game) {
341         Map<ReplacementEffect, Set<Ability>> replaceEffects = new LinkedHashMap<>();
342         if (auraReplacementEffect.checksEventType(event, game) && auraReplacementEffect.applies(event, null, game)) {
343             replaceEffects.put(auraReplacementEffect, null);
344         }
345         // boolean checkLKI = event.getType().equals(EventType.ZONE_CHANGE) || event.getType().equals(EventType.DESTROYED_PERMANENT);
346         //get all applicable transient Replacement effects
347         for (Iterator<ReplacementEffect> iterator = replacementEffects.iterator(); iterator.hasNext(); ) {
348             ReplacementEffect effect = iterator.next();
349             if (!effect.checksEventType(event, game)) {
350                 continue;
351             }
352             if (event.getAppliedEffects() != null && event.getAppliedEffects().contains(effect.getId())) {
353                 if (!(effect instanceof CommanderReplacementEffect)) { // 903.9.
354                     // Effect already applied to this event, ignore it
355                     // TODO: Handle also gained effect that are connected to different abilities.
356                     continue;
357                 }
358             }
359             Set<Ability> abilities = replacementEffects.getAbility(effect.getId());
360             Set<Ability> applicableAbilities = new HashSet<>();
361             for (Ability ability : abilities) {
362                 // for replacment effects of static abilities do not use LKI to check if to apply
363                 if (ability.getAbilityType() != AbilityType.STATIC || ability.isInUseableZone(game, null, event)) {
364                     if (!effect.isUsed()) {
365                         if (!game.getScopeRelevant()
366                                 || effect.hasSelfScope()
367                                 || !event.getTargetId().equals(ability.getSourceId())) {
368                             if (effect.applies(event, ability, game)) {
369                                 applicableAbilities.add(ability);
370                             }
371                         }
372                     }
373                 }
374             }
375             if (!applicableAbilities.isEmpty()) {
376                 replaceEffects.put(effect, applicableAbilities);
377             }
378         }
379 
380         for (Iterator<PreventionEffect> iterator = preventionEffects.iterator(); iterator.hasNext(); ) {
381             PreventionEffect effect = iterator.next();
382             if (!effect.checksEventType(event, game)) {
383                 continue;
384             }
385             if (event.getAppliedEffects() != null && event.getAppliedEffects().contains(effect.getId())) {
386                 // Effect already applied to this event, ignore it
387                 // TODO: Handle also gained effect that are connected to different abilities.
388                 continue;
389             }
390             Set<Ability> abilities = preventionEffects.getAbility(effect.getId());
391             Set<Ability> applicableAbilities = new HashSet<>();
392             for (Ability ability : abilities) {
393                 if (ability.getAbilityType() != AbilityType.STATIC || ability.isInUseableZone(game, null, event)) {
394                     if (effect.getDuration() != Duration.OneUse || !effect.isUsed()) {
395                         if (effect.applies(event, ability, game)) {
396                             applicableAbilities.add(ability);
397                         }
398                     }
399                 }
400             }
401             if (!applicableAbilities.isEmpty()) {
402                 replaceEffects.put(effect, applicableAbilities);
403             }
404         }
405 
406         return replaceEffects;
407     }

409     private boolean checkAbilityStillExists(Ability ability, ContinuousEffect effect, GameEvent event, Game game) {
410         switch (effect.getDuration()) { // effects with fixed duration don't need an object with the source ability (e.g. a silence cast with isochronic Scepter has no more a card object
411             case EndOfCombat:
412             case EndOfGame:
413             case EndOfStep:
414             case EndOfTurn:
415             case OneUse:
416             case Custom:  // custom duration means the effect ends itself if needed
417                 return true;
418         }
419         if (ability.getSourceId() == null) { // commander replacement effect
420             return true;
421         }
422         MageObject object;
423         if (event.getType() == GameEvent.EventType.ZONE_CHANGE
424                 && ((ZoneChangeEvent) event).getFromZone() == Zone.BATTLEFIELD
425                 && event.getTargetId().equals(ability.getSourceId())) {
426             object = ((ZoneChangeEvent) event).getTarget();
427         } else {
428             object = game.getObject(ability.getSourceId());
429         }
430         if (object == null) {
431             return false;
432         }
433         boolean exists = true;
434         if (!object.hasAbility(ability, game)) {
435             exists = false;
436             if (object instanceof PermanentCard) {
437                 PermanentCard permanent = (PermanentCard) object;
438                 if (permanent.isTransformable() && event.getType() == GameEvent.EventType.TRANSFORMED) {
439                     exists = permanent.getCard().hasAbility(ability, game);
440                 }
441             }
442         } else if (object instanceof PermanentCard) {
443             PermanentCard permanent = (PermanentCard) object;
444             if (permanent.isFaceDown(game) && !ability.getWorksFaceDown()) {
445                 return false;
446             }
447         } else if (object instanceof Spell) {
448             Spell spell = (Spell) object;
449             if (spell.isFaceDown(game) && !ability.getWorksFaceDown()) {
450                 return false;
451             }
452         }
453         return exists;
454     }

504     /**
505      * @param objectId        object to check
506      * @param type
507      * @param affectedAbility null if check full object or ability if check only one ability from that object
508      * @param controllerId
509      * @param game
510      * @return sourceId of the permitting effect if any exists otherwise returns null
511      */
512     public ApprovingObject asThough(UUID objectId, AsThoughEffectType type, Ability affectedAbility, UUID controllerId, Game game) {
513 
514         // usage check: effect must apply for specific ability only, not to full object (example: PLAY_FROM_NOT_OWN_HAND_ZONE)
515         if (type.needAffectedAbility() && affectedAbility == null) {
516             throw new IllegalArgumentException("ERROR, you can't call asThough check to whole object, call it with affected ability instead: " + type);
517         }
518 
519         // usage check: effect must apply to full object, not specific ability (example: ATTACK_AS_HASTE)
520         // P.S. In theory a same AsThough effect can be applied to object or to ability, so if you really, really
521         // need it then disable that check or add extra param to AsThoughEffectType like needAffectedAbilityOrFullObject
522         if (!type.needAffectedAbility() && affectedAbility != null) {
523             throw new IllegalArgumentException("ERROR, you can't call AsThough check to affected ability, call it with empty affected ability instead: " + type);
524         }
525 
526         List<AsThoughEffect> asThoughEffectsList = getApplicableAsThoughEffects(type, game);
527         if (!asThoughEffectsList.isEmpty()) {
528             MageObject objectToCheck;
529             if (affectedAbility != null) {
530                 objectToCheck = affectedAbility.getSourceObject(game);
531             } else {
532                 objectToCheck = game.getCard(objectId);
533             }
534 
535             UUID idToCheck;
536             if (!type.needPlayCardAbility() && objectToCheck instanceof SplitCardHalf) {
537                 // each split side uses own characteristics to check for playing, all other cases must use main card
538                 // rules:
539                 // 708.4. In every zone except the stack, the characteristics of a split card are those of its two halves combined.
540                 idToCheck = ((SplitCardHalf) objectToCheck).getMainCard().getId();
541             } else if (!type.needPlayCardAbility() && objectToCheck instanceof AdventureCardSpell) {
542                 // adventure spell uses alternative characteristics for spell/stack, all other cases must use main card
543                 idToCheck = ((AdventureCardSpell) objectToCheck).getMainCard().getId();
544             } else if (!type.needPlayCardAbility() && objectToCheck instanceof ModalDoubleFacesCardHalf) {
545                 // each mdf side uses own characteristics to check for playing, all other cases must use main card
546                 // rules:
547                 // "If an effect allows you to play a land or cast a spell from among a group of cards,
548                 // you may play or cast a modal double-faced card with any face that fits the criteria
549                 // of that effect. For example, if Sejiri Shelter / Sejiri Glacier is in your graveyard
550                 // and an effect allows you to play lands from your graveyard, you could play Sejiri Glacier.
551                 // That effect doesn't allow you to cast Sejiri Shelter."
552                 idToCheck = ((ModalDoubleFacesCardHalf) objectToCheck).getMainCard().getId();
553             } else {
554                 idToCheck = objectId;
555             }
556 
557             Set<ApprovingObject> possibleApprovingObjects = new HashSet<>();
558             for (AsThoughEffect effect : asThoughEffectsList) {
559                 Set<Ability> abilities = asThoughEffectsMap.get(type).getAbility(effect.getId());
560                 for (Ability ability : abilities) {
561                     if (affectedAbility == null) {
562                         // applies to full object (one effect can be used in multiple abilities)
563                         if (effect.applies(idToCheck, ability, controllerId, game)) {
564                             if (effect.isConsumable() && !game.inCheckPlayableState()) {
565                                 possibleApprovingObjects.add(new ApprovingObject(ability, game));
566                             } else {
567                                 return new ApprovingObject(ability, game);
568                             }
569                         }
570                     } else {
571                         // applies to one affected ability
572 
573                         // filter play abilities (no need to check it in every effect's code)
574                         if (type.needPlayCardAbility() && !affectedAbility.getAbilityType().isPlayCardAbility()) {
575                             continue;
576                         }
577 
578                         if (effect.applies(idToCheck, affectedAbility, ability, game, controllerId)) {
579                             if (effect.isConsumable() && !game.inCheckPlayableState()) {
580                                 possibleApprovingObjects.add(new ApprovingObject(ability, game));
581                             } else {
582                                 return new ApprovingObject(ability, game);
583                             }
584                         }
585                     }
586                 }
587             }
588 
589             if (possibleApprovingObjects.size() == 1) {
590                 return possibleApprovingObjects.iterator().next();
591             } else if (possibleApprovingObjects.size() > 1) {
592                 // Select the ability that you use to permit the action                
593                 Map<String, String> keyChoices = new HashMap<>();
594                 for (ApprovingObject approvingObject : possibleApprovingObjects) {
595                     MageObject mageObject = game.getObject(approvingObject.getApprovingAbility().getSourceId());
596                     String choiceKey = approvingObject.getApprovingAbility().getId().toString();

904                 }
905             }

912             }

916             Ability rAbility = null;

930                     } else {

933                         while (it.hasNext() && rAbility == null) {
934                             if (checked == index) {
935                                 rAbility = (Ability) it.next();
936                             } else {
937                                 it.next();
938                                 checked++;
939                             }
940                         }
941                         break;
942                     }
943                 }
944             }

949                 if (Duration.OneUse.equals(rEffect.getDuration())) {
950                     rEffect.discard();
951                 }
952             }
953             if (caught) { // Event was completely replaced -> stop applying effects to it
954                 break;
955             }
956 
957             // add the applied effect to the consumed effects
958             if (rEffect != null) {
959                 if (consumed.containsKey(rEffect.getId())) {
960                     Set<UUID> set = consumed.get(rEffect.getId());
961                     if (rAbility != null) {
962                         set.add(rAbility.getId());

976             // do no longer work correctly because the entering permanents are not yet on the battlefield (before they were).
977             // game.applyEffects();
978         } while (true);

1177     private void applyLayer(List<ContinuousEffect> activeLayerEffects, Layer currentLayer, Game game, String timestampGroupName) {
1178         List<ContinuousEffect> layer = filterLayeredEffects(activeLayerEffects, currentLayer);
1179         // layer is a list of all effects at the current layer
1180         if (!layer.isEmpty()) {
1181             int numberOfEffects = layer.size();
1182             // appliedEffects holds the list of effects currently applied to the layer
1183             Set<UUID> appliedEffects = new HashSet<>();
1184             // waitingEffects holds the list of dependent effects and their independent counterparts
1185             Map<ContinuousEffect, Set<UUID>> waitingEffects = new LinkedHashMap<>();
1186             for (ContinuousEffect effect : layer) {
1187                 if (numberOfEffects > 1) {
1188                     // If an effect is dependent to not applied effects yet of this layer, so wait to apply this effect
1189                     // check to see if any effect is dependent to other cards indirectly due to the independent card being dependent, etc.
1190                     Set<UUID> dependentTo = effect.isDependentTo(layer);
1191                     if (!appliedEffects.containsAll(dependentTo)) {
1192                         waitingEffects.put(effect, dependentTo);
1193                         continue;
1194                     }
1195                 }
1196                 // apply the effect
1197                 applyContinuousEffect(effect, currentLayer, game);
1198                 // add it to the applied effects list
1199                 appliedEffects.add(effect.getId());
1200                 layer = getLayeredEffects(game, timestampGroupName);
1201 
1202                 // check waiting effects to see if it has anything to check
1203                 if (!waitingEffects.isEmpty()) {
1204                     // check if waiting effects can be applied now
1205                     for (Entry<ContinuousEffect, Set<UUID>> entry : waitingEffects.entrySet()) {
1206                         // all dependent to effects are applied now so apply the effect itself
1207                         if (appliedEffects.containsAll(entry.getValue())) {
1208                             applyContinuousEffect(entry.getKey(), currentLayer, game);
1209                             // add it to the applied effects list
1210                             appliedEffects.add(entry.getKey().getId());
1211                             layer = getLayeredEffects(game, timestampGroupName);
1212                         }
1213                     }
1214                 }
1215                 if (numberOfEffects != appliedEffects.size()) {
1216                     for (Entry<ContinuousEffect, Set<UUID>> entry : waitingEffects.entrySet()) {
1217                         // all dependent to effects are applied now so apply the effect itself
1218                         if (appliedEffects.containsAll(entry.getValue())) {
1219                             applyContinuousEffect(entry.getKey(), currentLayer, game);
1220                             // add it to the applied effects list
1221                             appliedEffects.add(entry.getKey().getId());
1222                             layer = getLayeredEffects(game, timestampGroupName);
1223                         }
1224                     }
1225                 }
1226             }
1227         }
1228     }

1241         switch (effect.getDuration()) { // effects with fixed duration don't need an object with the source ability (e.g. a silence cast with isochronic Scepter has no more a card object
1242             case EndOfCombat:
1243             case EndOfGame:

1245             case EndOfTurn:
1246             case OneUse:

1248                 return true;

1250         final Card card = game.getPermanentOrLKIBattlefield(ability.getSourceId());
1251         return effect instanceof BecomesFaceDownCreatureEffect
1252                 || effect == null || card == null || card.hasAbility(ability, game);

1275     public synchronized void addEffect(ContinuousEffect effect, Ability source) {
1276         if (effect == null) {
1277             logger.error("Effect is null: " + source.toString());
1278             return;
1279         } else if (source == null) {
1280             logger.warn("Adding effect without ability : " + effect);
1281         }
1282         switch (effect.getEffectType()) {
1283             case REPLACEMENT:
1284             case REDIRECTION:
1285                 replacementEffects.addEffect((ReplacementEffect) effect, source);
1286                 break;
1287             case PREVENTION:
1288                 preventionEffects.addEffect((PreventionEffect) effect, source);
1289                 break;
1290             case RESTRICTION:
1291                 restrictionEffects.addEffect((RestrictionEffect) effect, source);
1292                 break;
1293             case RESTRICTION_UNTAP_NOT_MORE_THAN:
1294                 restrictionUntapNotMoreThanEffects.addEffect((RestrictionUntapNotMoreThanEffect) effect, source);
1295                 break;
1296             case REQUIREMENT:
1297                 requirementEffects.addEffect((RequirementEffect) effect, source);
1298                 break;
1299             case ASTHOUGH:
1300                 AsThoughEffect newAsThoughEffect = (AsThoughEffect) effect;
1301                 asThoughEffectsMap.computeIfAbsent(newAsThoughEffect.getAsThoughEffectType(), x -> {
1302                     ContinuousEffectsList<AsThoughEffect> list = new ContinuousEffectsList<>();
1303                     allEffectsLists.add(list);
1304                     return list;
1305                 }).addEffect(newAsThoughEffect, source);
1306                 break;
1307             case COSTMODIFICATION:
1308                 costModificationEffects.addEffect((CostModificationEffect) effect, source);
1309                 break;
1310             case SPLICE:
1311                 spliceCardEffects.addEffect((SpliceCardEffect) effect, source);
1312                 break;
1313             case CONTINUOUS_RULE_MODIFICATION:
1314                 continuousRuleModifyingEffects.addEffect((ContinuousRuleModifyingEffect) effect, source);
1315                 break;
1316             default:
1317                 layeredEffects.addEffect(effect, source);
1318                 break;
1319         }
1320     }

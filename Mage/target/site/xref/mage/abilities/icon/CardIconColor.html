3   import java.awt.*;

5   /**
6    * For GUI: custom color modification for card icons (e.g. for special card icons like commander)
7    * <p>
8    * Card icons uses svg, so all color settings must be predefined on app's loading
9    * <p>
10   * TODO: current version uses panels (initCardIconsPanels) to define custom colors,
11   * but it can be removed to single/dynamic icon in the future (example: r/g/b dynamic semaphore icon)
12   *
13   * @author JayDi85
14   */
15  public enum CardIconColor {
16  
17      DEFAULT(),
18      YELLOW(new Color(231, 203, 18), new Color(76, 76, 76), new Color(0, 0, 0)),
19      RED(new Color(255, 31, 75), new Color(76, 76, 76), new Color(229, 228, 228));
20  
21      private final Color fillColor;
22      private final Color strokeColor;
23      private final Color textColor;
24  
25      CardIconColor() {
26          this(null, null, null);
27      }
28  
29      CardIconColor(Color fillColor, Color strokeColor, Color textColor) {
30          this.fillColor = fillColor;
31          this.strokeColor = strokeColor;
32          this.textColor = textColor;
33      }
34  
35      public Color getFillColor() {
36          return fillColor;
37      }
38  
39      public Color getStrokeColor() {
40          return strokeColor;
41      }
42  
43      public Color getTextColor() {
44          return textColor;
45      }
46  }

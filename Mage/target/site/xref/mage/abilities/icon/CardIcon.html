21      /**
22       * Combined info (text + hint)
23       *
24       * @return
25       */
26      default String getCombinedInfo() {

28          if (!getHint().isEmpty()) {
29              res += res.isEmpty() ? "" : " - ";

31          }
32          return res;
33      }

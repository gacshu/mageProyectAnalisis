2   package mage.abilities.mana;
3   
4   import mage.Mana;
5   import mage.abilities.costs.Cost;
6   import mage.abilities.effects.mana.AddManaOfAnyColorEffect;
7   import mage.abilities.effects.mana.BasicManaEffect;
8   import mage.constants.Zone;
9   import mage.game.Game;
10  
11  /**
12   *
13   * @author LevelX2
14   */
15  public class ActivateOncePerTurnManaAbility extends ActivatedManaAbilityImpl {
16  
17      public ActivateOncePerTurnManaAbility(Zone zone, BasicManaEffect effect, Cost cost) {
18          super(zone, effect, cost);
19          this.netMana.add(effect.getManaTemplate());
20          this.maxActivationsPerTurn = 1;
21      }
22  
23      public ActivateOncePerTurnManaAbility(Zone zone, AddManaOfAnyColorEffect effect, Cost cost) {
24          super(zone, effect, cost);
25          this.netMana.add(new Mana(0, 0, 0, 0, 0, 0, effect.getAmount(), 0));
26          this.maxActivationsPerTurn = 1;
27      }
28  
29      public ActivateOncePerTurnManaAbility(ActivateOncePerTurnManaAbility ability) {
30          super(ability);
31      }
32  
33      @Override
34      public boolean activate(Game game, boolean noMana) {
35          if (canActivate(this.controllerId, game).canActivate()) {
36              return super.activate(game, noMana);
37          }
38          return false;
39      }
40  
41      @Override
42      public String getRule() {
43          return super.getRule() + " Activate only once each turn.";
44      }
45  
46      @Override
47      public ActivateOncePerTurnManaAbility copy() {
48          return new ActivateOncePerTurnManaAbility(this);
49      }
50  
51  }

7   import mage.abilities.effects.Effect;
8   import mage.abilities.effects.mana.AddManaOfAnyColorEffect;
9   import mage.abilities.effects.mana.ManaEffect;
10  import mage.constants.Zone;
11  import mage.game.Game;
12  
13  import java.util.ArrayList;
14  import java.util.List;
15  
16  public class AnyColorManaAbility extends ActivatedManaAbilityImpl {
17  
18      public AnyColorManaAbility() {
19          this(new TapSourceCost());
20      }
21  
22      public AnyColorManaAbility(Cost cost) {
23          this(cost, false);
24      }
25  
26      public AnyColorManaAbility(Cost cost, boolean setFlag) {
27          super(Zone.BATTLEFIELD, new AddManaOfAnyColorEffect(1, setFlag), cost);
28          this.netMana.add(new Mana(0, 0, 0, 0, 0, 0, 1, 0));
29      }
30  
31      /**
32       * @param cost
33       * @param netAmount dynamic value used during available mana calculation to
34       *                  set the max possible amount the source can produce
35       * @param setFlag
36       */
37      public AnyColorManaAbility(Cost cost, DynamicValue netAmount, boolean setFlag) {
38          this(Zone.BATTLEFIELD, cost, netAmount, setFlag);
39      }
40  
41      public AnyColorManaAbility(Zone zone, Cost cost, DynamicValue netAmount, boolean setFlag) {
42          super(zone, new AddManaOfAnyColorEffect(1, netAmount, setFlag), cost);
43          this.netMana.add(new Mana(0, 0, 0, 0, 0, 0, 1, 0));
44      }
45  
46      public AnyColorManaAbility(final AnyColorManaAbility ability) {
47          super(ability);
48      }
49  
50      @Override
51      public List<Mana> getNetMana(Game game) {
52          if (game != null && game.inCheckPlayableState()) {
53              List<Mana> dynamicNetMana = new ArrayList<>();
54              for (Effect effect : getEffects()) {
55                  if (effect instanceof ManaEffect) {
56                      List<Mana> effectNetMana = ((ManaEffect) effect).getNetMana(game, this);
57                      if (effectNetMana != null) {
58                          dynamicNetMana.addAll(effectNetMana);
59                      }
60                  }
61              }
62              return dynamicNetMana;
63          }
64          return super.getNetMana(game);
65      }
66  
67      @Override
68      public AnyColorManaAbility copy() {
69          return new AnyColorManaAbility(this);
70      }
71  }

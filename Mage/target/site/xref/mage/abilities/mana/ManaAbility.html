1   package mage.abilities.mana;
2   
3   import java.util.List;
4   import java.util.Set;
5   import mage.Mana;
6   import mage.constants.ManaType;
7   import mage.game.Game;
8   
9   /**
10   *
11   * @author LevelX2
12   */
13  public interface ManaAbility {
14  
15      /**
16       * Used to check the possible mana production to determine which spells
17       * and/or abilities can be used. (player.getPlayable()).
18       *
19       * @param game
20       * @return
21       */
22      List<Mana> getNetMana(Game game);
23      
24      /**
25       * Used to check the possible mana production to determine which spells
26       * and/or abilities can be used. (player.getPlayable()).
27       * Only used for abilities were the poolDependant flag is set
28       *
29       * @param game
30       * @param possibleManaInPool The possible mana already produced by other sources for this calculation option
31       * @return 
32       */
33      List<Mana> getNetMana(Game game, Mana possibleManaInPool);
34  
35      /**
36       * The type of mana a permanent "could produce" is the type of mana that any
37       * ability of that permanent can generate, taking into account any
38       * applicable replacement effects. If the type of mana can’t be defined,
39       * there’s no type of mana that that permanent could produce. The "type" of
40       * mana is its color, or lack thereof (for colorless mana).
41       *
42       * @param game
43       * @return
44       */
45      Set<ManaType> getProducableManaTypes(Game game);
46  
47      /**
48       * Used to check if the ability itself defines mana types it can produce.
49       *
50       * @param game
51       * @return
52       */
53      boolean definesMana(Game game);
54  
55      /**
56       * Set to true if the ability is dependant from the mana pool. E.g. the more
57       * mana the pool contains the more mana the ability can produce (Doubling
58       * Cube). Therefore the use of that ability after other mana abilities does produce more mana.
59       *
60       * @return
61       */
62      boolean isPoolDependant();
63      
64      ManaAbility setPoolDependant(boolean pooleDependant);
65      
66  }

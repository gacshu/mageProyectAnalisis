4   import mage.abilities.TriggeredAbility;
5   import mage.abilities.TriggeredAbilityImpl;

9   import mage.constants.EffectType;
10  import mage.game.Game;
11  import mage.game.events.GameEvent;
12  import mage.util.CardUtil;
13  import mage.watchers.Watcher;
14  
15  import java.util.List;
16  
17  /**
18   * Adds condition to {@link mage.abilities.effects.ContinuousEffect}. Acts as
19   * decorator.
20   *
21   * @author nantuko
22   */
23  public class ConditionalInterveningIfTriggeredAbility extends TriggeredAbilityImpl {
24  
25      protected TriggeredAbility ability;
26      protected Condition condition;
27      protected String abilityText;
28  
29      /**
30       * Triggered ability with a condition. Set the optionality for the trigger
31       * ability itself.
32       *
33       * @param ability
34       * @param condition
35       * @param text      explicit rule text for the ability, if null or empty, the
36       *                  rule text generated by the triggered ability itself is used.
37       */
38      public ConditionalInterveningIfTriggeredAbility(TriggeredAbility ability, Condition condition, String text) {
39          super(ability.getZone(), null);
40          if (ability.isLeavesTheBattlefieldTrigger()) {
41              this.setLeavesTheBattlefieldTrigger(true);
42          }
43          this.ability = ability;
44          this.condition = condition;
45          this.abilityText = text;
46      }
47  
48      public ConditionalInterveningIfTriggeredAbility(final ConditionalInterveningIfTriggeredAbility triggered) {
49          super(triggered);
50          this.ability = triggered.ability.copy();
51          this.condition = triggered.condition;
52          this.abilityText = triggered.abilityText;
53      }
54  
55      @Override
56      public boolean checkInterveningIfClause(Game game) {
57          return condition.apply(game, this);
58      }
59  
60      @Override
61      public ConditionalInterveningIfTriggeredAbility copy() {
62          return new ConditionalInterveningIfTriggeredAbility(this);
63      }
64  
65      @Override
66      public boolean checkEventType(GameEvent event, Game game) {
67          return ability.checkEventType(event, game);
68      }
69  
70      @Override
71      public boolean checkTrigger(GameEvent event, Game game) {
72          ability.setSourceId(this.getSourceId());
73          ability.setControllerId(this.getControllerId());
74          return ability.checkTrigger(event, game);
75      }
76  
77      @Override
78      public String getRule() {
79          if (abilityText == null || abilityText.isEmpty()) {
80              return ability.getRule();
81          }
82          return (flavorWord != null ? CardUtil.italicizeWithEmDash(flavorWord) : "") +
83                  (abilityWord != null ? abilityWord.formatWord() : "") +
84                  abilityText + (abilityText.endsWith(".") || abilityText.endsWith("\"") || abilityText.endsWith(">") ? "" : ".");
85      }
86  
87      @Override
88      public Effects getEffects() {
89          return ability.getEffects();
90      }
91  
92      @Override
93      public void addEffect(Effect effect) {
94          ability.addEffect(effect);
95      }
96  
97      @Override
98      public Modes getModes() {
99          return ability.getModes();
100     }
101 
102     @Override
103     public List<Watcher> getWatchers() {
104         return ability.getWatchers();
105     }
106 
107     @Override
108     public void addWatcher(Watcher watcher) {
109         ability.addWatcher(watcher);
110     }
111 
112     @Override
113     public Effects getEffects(Game game, EffectType effectType) {
114         return ability.getEffects(game, effectType);
115     }
116 
117     @Override
118     public boolean isOptional() {
119         return ability.isOptional();
120     }
121 
122     @Override
123     public void setSourceObjectZoneChangeCounter(int sourceObjectZoneChangeCounter) {
124         ability.setSourceObjectZoneChangeCounter(sourceObjectZoneChangeCounter);
125     }
126 
127     @Override
128     public int getSourceObjectZoneChangeCounter() {
129         return ability.getSourceObjectZoneChangeCounter();
130     }
131 
132     @Override
133     public boolean caresAboutManaColor() {
134         return condition.caresAboutManaColor();
135     }
136 }

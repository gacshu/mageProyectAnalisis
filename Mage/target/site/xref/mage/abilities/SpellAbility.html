58      @Override
59      public boolean canChooseTarget(Game game, UUID playerId) {
60          if (SpellAbilityType.SPLIT_FUSED.equals(getSpellAbilityType())) {
61              Card card = game.getCard(getSourceId());
62              if (card == null) {
63                  return false;
64              }
65              SpellAbility left = ((SplitCard) card).getLeftHalfCard().getSpellAbility();
66              SpellAbility right = ((SplitCard) card).getRightHalfCard().getSpellAbility();
67              return canChooseTargetAbility(left, left.getModes(), game, playerId) && canChooseTargetAbility(right, right.getModes(), game, playerId);
68          }
69          return super.canChooseTarget(game, playerId);
70      }

72      public boolean canSpliceOnto(Ability abilityToModify, Game game) {
73          return canChooseTargetAbility(abilityToModify, getModes(), game, abilityToModify.getControllerId());

77       * 7/5/19 - jgray1206 - Moved null != game.getContinuesEffects()... into this method instead of having it in
78       *                      canActivate. There are abilities that directly use this method that should know when spells
79       *                      can be casted that are affected by the CastAsInstant effect.
80       *                      (i.e. Vizier of the Menagerie and issue #5816)
81       */
82      public boolean spellCanBeActivatedRegularlyNow(UUID playerId, Game game) {
83          MageObject object = game.getObject(sourceId);
84          if (object == null) {
85              return false;
86          }
87  
88          // forced to cast (can be part id or main id)
89          Set<UUID> idsToCheck = new HashSet<>();
90          idsToCheck.add(object.getId());
91          if (object instanceof Card && !(object instanceof AdventureCardSpell)) {
92              idsToCheck.add(((Card) object).getMainCard().getId());
93          }
94          for (UUID idToCheck : idsToCheck) {
95              if (game.getState().getValue("PlayFromNotOwnHandZone" + idToCheck) != null) {
96                  return (Boolean) game.getState().getValue("PlayFromNotOwnHandZone" + idToCheck);  // card like Chandra, Torch of Defiance +1 loyal ability)
97              }
98          }
99  
100         return null != game.getContinuousEffects().asThough(sourceId, AsThoughEffectType.CAST_AS_INSTANT, this, playerId, game) // check this first to allow Offering in main phase
101                 || timing == TimingRule.INSTANT
102                 || object.isInstant(game)
103                 || object.hasAbility(FlashAbility.getInstance(), game)
104                 || game.canPlaySorcery(playerId);
105     }

107     @Override
108     public ActivationStatus canActivate(UUID playerId, Game game) {
109         // spells can be cast from non hand zones, so must use custom check
110         // no super.canActivate() call
111 
112         if (this.spellCanBeActivatedRegularlyNow(playerId, game)) {
113             if (spellAbilityType == SpellAbilityType.SPLIT
114                     || spellAbilityType == SpellAbilityType.SPLIT_AFTERMATH) {
115                 return ActivationStatus.getFalse();
116             }
117 
118             // play from not own hand
119             ApprovingObject approvingObject = game.getContinuousEffects().asThough(getSourceId(), AsThoughEffectType.PLAY_FROM_NOT_OWN_HAND_ZONE, this, playerId, game);
120             if (approvingObject == null) {
121                 Card card = game.getCard(sourceId);
122                 if (!(card != null && card.isOwnedBy(playerId))) {
123                     return ActivationStatus.getFalse();
124                 }
125             }
126 
127             // play restrict
128             // Check if rule modifying events prevent to cast the spell in check playable mode
129             if (game.inCheckPlayableState()) {
130                 Card card = game.getCard(sourceId);
131                 GameEvent castEvent = GameEvent.getEvent(GameEvent.EventType.CAST_SPELL, this.getId(), this, playerId);
132                 castEvent.setZone(card == null ? null : game.getState().getZone(card.getMainCard().getId()));
133                 if (game.getContinuousEffects().preventedByRuleModification(
134                         castEvent, this, game, true)) {
135                     return ActivationStatus.getFalse();
136                 }
137             }
138 
139             // no mana restrict
140             // Alternate spell abilities (Flashback, Overload) can't be cast with no mana to pay option
141             if (getSpellAbilityType() == SpellAbilityType.BASE_ALTERNATE) {
142                 Player player = game.getPlayer(playerId);
143                 if (player != null
144                         && player.getCastSourceIdWithAlternateMana().contains(getSourceId())) {
145                     return ActivationStatus.getFalse();
146                 }
147             }
148 
149             // can pay all costs and choose targets
150             if (costs.canPay(this, this, playerId, game)) {
151                 if (getSpellAbilityType() == SpellAbilityType.SPLIT_FUSED) {
152                     SplitCard splitCard = (SplitCard) game.getCard(getSourceId());
153                     if (splitCard != null) {
154                         // fused can be called from hand only, so not permitting object allows or other zones checks
155                         // see https://www.mtgsalvation.com/forums/magic-fundamentals/magic-rulings/magic-rulings-archives/251926-snapcaster-mage-and-fuse
156                         if (game.getState().getZone(splitCard.getId()) == Zone.HAND) {
157                             return new ActivationStatus(splitCard.getLeftHalfCard().getSpellAbility().canChooseTarget(game, playerId)
158                                     && splitCard.getRightHalfCard().getSpellAbility().canChooseTarget(game, playerId), null);
159                         }
160                     }
161                     return ActivationStatus.getFalse();
162                 } else {
163                     return new ActivationStatus(canChooseTarget(game, playerId), approvingObject);
164                 }
165             }
166         }
167         return ActivationStatus.getFalse();
168     }

203     public SpellAbility copySpell(Card originalCard, Card copiedCard) {
204         // all copied spells must have own copied card
205         Map<UUID, MageObject> mapOldToNew = CardUtil.getOriginalToCopiedPartsMap(originalCard, copiedCard);
206         if (!mapOldToNew.containsKey(this.getSourceId())) {
207             throw new IllegalStateException("Can't find source id after copy: " + originalCard.getName() + " -> " + copiedCard.getName());
208         }
209         UUID copiedSourceId = mapOldToNew.getOrDefault(this.getSourceId(), copiedCard).getId();
210 
211         SpellAbility spell = new SpellAbility(this);
212         spell.newId();
213         spell.setSourceId(copiedSourceId);
214         return spell;
215     }

229     public int getConvertedXManaCost(Card card) {
230         int xMultiplier = 0;
231         int amount = 0;
232         if (card == null) {
233             return 0;
234         }
235 
236         // mana cost instances
237         for (ManaCost manaCost : card.getManaCost()) {
238             if (manaCost instanceof VariableManaCost) {
239                 xMultiplier = ((VariableManaCost) manaCost).getXInstancesCount();
240                 break;
241             }
242         }
243 
244         // mana cost final X value
245         boolean hasNonManaXCost = false;
246         for (Cost cost : getCosts()) {
247             if (cost instanceof VariableCost) {
248                 hasNonManaXCost = true;
249                 amount = ((VariableCost) cost).getAmount();
250                 break;
251             }
252         }
253 
254         if (!hasNonManaXCost) {
255             amount = getManaCostsToPay().getX();
256         }
257         return amount * xMultiplier;
258     }

265     private void setSpellName() {
266         switch (spellAbilityType) {
267             case SPLIT_FUSED:
268                 this.name = "Cast fused " + cardName;
269                 break;
270             default:
271                 this.name = "Cast " + cardName + (this.spellAbilityCastMode != SpellAbilityCastMode.NORMAL ? " using " + spellAbilityCastMode.toString() : "");
272         }

288     /**
289      * Returns a card object with the spell characteristics like calor, types,
290      * subtypes etc. E.g. if you cast a Bestow card as enchantment, the
291      * characteristics don't include the creature type.
292      *
293      * @param game
294      * @return card object with the spell characteristics
295      */
296     public Card getCharacteristics(Game game) {
297         Card spellCharacteristics = game.getSpell(this.getId());
298         if (spellCharacteristics == null) {
299             spellCharacteristics = game.getCard(this.getSourceId());
300         }
301         if (spellCharacteristics != null) {
302             if (getSpellAbilityCastMode() != SpellAbilityCastMode.NORMAL) {
303                 spellCharacteristics = getSpellAbilityCastMode().getTypeModifiedCardObjectCopy(spellCharacteristics, game);
304             }
305         }
306         return spellCharacteristics;
307     }

309     public static SpellAbility getSpellAbilityFromEvent(GameEvent event, Game game) {
310         if (event.getType() != GameEvent.EventType.CAST_SPELL) {
311             return null;
312         }
313 
314         Card card = game.getCard(event.getSourceId());
315         if (card != null) {
316             Optional<Ability> ability = card.getAbilities(game).get(event.getTargetId());
317             if (ability.isPresent() && ability.get() instanceof SpellAbility) {
318                 return (SpellAbility) ability.get();
319             }
320             return card.getSpellAbility();
321         }
322         return null;
323     }

45      /**
46       * Change loyalty cost by amount value
47       *
48       * @param amount
49       */
50      public void increaseLoyaltyCost(int amount) {
51  
52          // loyalty cost modification rules from Carth the Lion
53  
54          // If a planeswalker’s loyalty ability normally has a cost of [+1], Carth’s ability makes it cost [+2] instead.
55          // A cost of [0] would become [+1], and a cost of [-6] would become [-5].
56          // (2021-06-18)
57          //
58          // If you somehow manage to control two Carths (perhaps because of Spark Double), the cost-changing effect is
59          // cumulative. In total, loyalty abilities will cost an additional [+2] to activate.
60          // (2021-06-18)
61          //
62          // The total cost of a planeswalker’s loyalty ability is calculated before any counters are added or removed.
63          // If a loyalty ability normally costs [-3] to activate, you do not remove three counters from it and then
64          // put one counter on it. You remove two counters at one time when you pay the cost.
65          // (2021-06-18)
66          //
67          // If an effect replaces the number of counters that would be placed on a planeswalker, such as that of
68          // Vorinclex, Monstrous Raider, that replacement happens only once, at the time payment is made.
69          // (2021-06-18)
70  
71          // cost modification support only 1 cost item
72          int staticCount = 0;
73          for (Cost cost : costs) {
74              if (cost instanceof PayLoyaltyCost) {
75                  // static cost
76                  PayLoyaltyCost staticCost = (PayLoyaltyCost) cost;
77                  staticCost.setAmount(staticCost.getAmount() + amount);
78                  staticCount++;
79              } else if (cost instanceof PayVariableLoyaltyCost) {
80                  // x cost (after x announce: x cost + static cost)
81                  PayVariableLoyaltyCost xCost = (PayVariableLoyaltyCost) cost;
82                  xCost.setCostModification(xCost.getCostModification() + amount);
83              }
84          }
85          if (staticCount > 1) {
86              throw new IllegalArgumentException(String.format("Loyalty ability must have only 1 static cost, but has %d: %s",
87                      staticCount, this.getRule()));
88          }
89      }
90  }

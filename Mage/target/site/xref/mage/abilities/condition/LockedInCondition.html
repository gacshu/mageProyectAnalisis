2   package mage.abilities.condition;
3   
4   import mage.abilities.Ability;
5   import mage.game.Game;
6   
7   /**
8    * The use of this class must be handled carefully because conditions don't
9    * have a copy method, the condition state is kept when the effect or ability
10   * is copied that uses the condition.
11   * So if you use this class, you have to do it like in ConditionalContinuousEffect,
12   * where always a new FixedCondition(condition.apply(...)) is used if a
13   * LockedInCondition is given.
14   *
15   * Needs probably some redesign, don't like it the way it's done now.
16   *
17   * @author LevelX2
18   */
19  public class LockedInCondition implements Condition {
20  
21      private boolean conditionChecked = false;
22      private boolean result;
23      private final Condition condition;
24  
25      public LockedInCondition ( Condition condition ) {
26          this.condition = condition;
27      }
28  
29      /*
30       * {@inheritDoc}
31       */
32      @Override
33      public boolean apply(Game game, Ability source) {
34          if(!conditionChecked) {
35              result = condition.apply(game, source);
36              conditionChecked = true;
37          }
38          return result;
39      }
40  
41      public Condition getBaseCondition() {
42          return condition;
43      }
44  
45      @Override
46      public boolean caresAboutManaColor() {
47          return condition.caresAboutManaColor();
48      }
49  
50  }

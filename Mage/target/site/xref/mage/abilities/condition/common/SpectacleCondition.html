2   package mage.abilities.condition.common;
3   
4   import mage.abilities.Ability;
5   import mage.abilities.condition.Condition;
6   import mage.abilities.keyword.SpectacleAbility;
7   import mage.constants.AbilityType;
8   import mage.game.Game;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  /**
14   * @author TheElk801
15   */
16  public enum SpectacleCondition implements Condition {
17  
18      instance;
19  
20      @Override
21      public boolean apply(Game game, Ability source) {
22          if (source.getAbilityType() == AbilityType.TRIGGERED) {
23              @SuppressWarnings("unchecked")
24              List<Integer> spectacleActivations = (ArrayList) game.getState().getValue(SpectacleAbility.SPECTACLE_ACTIVATION_VALUE_KEY + source.getSourceId());
25              if (spectacleActivations != null) {
26                  return spectacleActivations.contains(game.getState().getZoneChangeCounter(source.getSourceId()) - 1);
27              }
28              return false;
29          } else {
30              return source instanceof SpectacleAbility;
31          }
32      }
33  }

18  public enum DesertControlledOrGraveyardCondition implements Condition {
19      instance;
20  
21      private static final FilterPermanent filter = new FilterControlledPermanent(SubType.DESERT);
22      private static final FilterCard filter2 = new FilterCard();
23  
24      static {
25          filter2.add(SubType.DESERT.getPredicate());
26      }
27  
28      @Override
29      public boolean apply(Game game, Ability source) {
30          if (game.getBattlefield().contains(filter, source, game, 1)) {
31              return true;
32          }
33          Player player = game.getPlayer(source.getControllerId());
34          return player != null && player.getGraveyard().count(filter2, game) > 0;
35      }
36  
37      @Override
38      public String toString() {
39          return "you control a Desert or there is a Desert card in your graveyard";
40      }
41  
42      public static Hint getHint() {
43          return DesertControlledOrGraveyardHint.instance;
44      }
45  
46      enum DesertControlledOrGraveyardHint implements Hint {
47          instance;
48  
49          @Override
50          public String getText(Game game, Ability ability) {
51              Player player = game.getPlayer(ability.getControllerId());
52  
53              StringBuilder sb = new StringBuilder();
54              boolean controlDesert = game.getBattlefield().contains(filter, ability, game, 1);
55              boolean desertInGraveyard = player != null && player.getGraveyard().count(filter2, game) > 0;
56  
57              sb.append(HintUtils.prepareText("You control a Desert.<br>", null, controlDesert ? HintUtils.HINT_ICON_GOOD : HintUtils.HINT_ICON_BAD));
58              sb.append(HintUtils.prepareText("You have a Desert card in your graveyard.", null, desertInGraveyard ? HintUtils.HINT_ICON_GOOD : HintUtils.HINT_ICON_BAD));
59              return sb.toString();
60          }
61  
62          @Override
63          public Hint copy() {
64              return this;
65          }
66      }

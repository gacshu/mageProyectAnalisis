1   package mage.abilities.condition.common;
2   
3   import mage.MageInt;
4   import mage.MageObject;
5   import mage.abilities.Ability;
6   import mage.abilities.condition.Condition;
7   import mage.filter.StaticFilters;
8   import mage.game.Game;
9   
10  import java.util.Objects;
11  
12  /**
13   * @author TheElk801
14   */
15  public enum CovenCondition implements Condition {
16      instance;
17  
18      @Override
19      public boolean apply(Game game, Ability source) {
20          return game
21                  .getBattlefield()
22                  .getActivePermanents(
23                          StaticFilters.FILTER_CONTROLLED_CREATURE,
24                          source.getControllerId(), source, game
25                  )
26                  .stream()
27                  .filter(Objects::nonNull)
28                  .map(MageObject::getPower)
29                  .mapToInt(MageInt::getValue)
30                  .distinct()
31                  .count() >= 3;
32      }
33  
34      @Override
35      public String toString() {
36          return "if you control three or more creatures with different powers";
37      }
38  }

3   package mage.abilities.condition.common;
4   
5   import mage.abilities.Ability;
6   import mage.abilities.condition.Condition;
7   import mage.constants.PhaseStep;
8   import mage.constants.TurnPhase;
9   import mage.game.Game;
10  import mage.game.permanent.Permanent;
11  import mage.game.turn.Step;
12  
13  import java.util.UUID;
14  
15  /**
16   *
17   * @author spjspj
18   */
19  
20  public class SourceTappedBeforeUntapStepCondition implements Condition {
21  
22      UUID permanentId = null;
23      boolean permanentWasTappedBeforeUntapStep = false;
24      int lastTurnNum = -1;
25      
26  
27  
28      public void setPermanentId(UUID permanentId) {
29          this.permanentId = permanentId;
30      }
31  
32      @Override
33      public boolean apply(Game game, Ability source) {
34          TurnPhase turnPhase = game.getTurn().getPhase().getType();
35          Step step = game.getPhase().getStep();
36          Permanent permanent = game.getBattlefield().getPermanent(permanentId);
37  
38          if (permanent != null) {
39              if (lastTurnNum != game.getTurnNum() && turnPhase == TurnPhase.BEGINNING) {
40                  lastTurnNum = game.getTurnNum();
41                  permanentWasTappedBeforeUntapStep = permanent.isTapped();
42              }
43  
44              if (step.getType() == PhaseStep.UNTAP) {
45                  return permanentWasTappedBeforeUntapStep;
46              } else {
47                  return permanent.isTapped();
48              }
49          }
50          return false;
51      }
52  }

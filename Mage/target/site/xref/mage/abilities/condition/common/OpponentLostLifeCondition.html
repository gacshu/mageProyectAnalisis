2   package mage.abilities.condition.common;
3   
4   import mage.abilities.Ability;
5   import mage.constants.ComparisonType;
6   import mage.abilities.condition.IntCompareCondition;
7   import mage.game.Game;
8   import mage.watchers.common.PlayerLostLifeWatcher;
9   
10  import java.util.UUID;
11  
12  /**
13   * Describes condition when an opponent has lost an amount of life
14   *
15   * @author LevelX2
16   */
17  public class OpponentLostLifeCondition extends IntCompareCondition {
18  
19      public OpponentLostLifeCondition(ComparisonType type, int value) {
20          super(type, value);
21      }
22  
23      @Override
24      protected int getInputValue(Game game, Ability source) {
25          int maxLostLive = 0;
26          PlayerLostLifeWatcher watcher = game.getState().getWatcher(PlayerLostLifeWatcher.class);
27          if (watcher != null) {
28              for (UUID opponentId : game.getOpponents(source.getControllerId())) {
29                  int lostLive = watcher.getLifeLost(opponentId);
30                  if (lostLive > maxLostLive) {
31                      maxLostLive = lostLive;
32                  }
33              }
34          }
35          return maxLostLive;
36      }
37  
38      @Override
39      public String toString() {
40          StringBuilder sb = new StringBuilder("if an opponent lost ");
41          switch (type) {
42              case MORE_THAN:
43                  sb.append(value + 1).append(" or more life this turn");
44                  break;
45              case EQUAL_TO:
46                  sb.append(value).append(" life this turn");
47                  break;
48              case FEWER_THAN:
49                  sb.append(" less than ").append(value).append(" life this turn");
50                  break;
51          }
52          return sb.toString();
53      }
54  }

1   package mage.abilities.condition.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.condition.Condition;
5   import mage.constants.ComparisonType;
6   import mage.filter.FilterPermanent;
7   import mage.filter.predicate.permanent.ControllerIdPredicate;
8   import mage.game.Game;
9   
10  /**
11   * Battlefield checking condition. This condition can decorate other conditions
12   * as well as be used standalone.
13   *
14   * @author nantuko
15   * @author maurer.it_at_gmail.com
16   * @see #Controls(mage.filter.Filter)
17   */
18  public class PermanentsOnTheBattlefieldCondition implements Condition {
19  
20      private final FilterPermanent filter;
21      private final ComparisonType type;
22      private final int count;
23      private final boolean onlyControlled;
24  
25      /**
26       * Applies a filter and delegates creation to
27       * {@link #ControlsPermanent(mage.filter.FilterPermanent, mage.abilities.condition.common.ControlsPermanent.CountType, int)}
28       * with {@link ComparisonType#MORE_THAN}, and 0.
29       *
30       * @param filter
31       */
32      public PermanentsOnTheBattlefieldCondition(FilterPermanent filter) {
33          this(filter, true);
34      }
35  
36      public PermanentsOnTheBattlefieldCondition(FilterPermanent filter, boolean onlyControlled) {
37          this(filter, ComparisonType.MORE_THAN, 0, onlyControlled);
38      }
39  
40      /**
41       * Applies a filter, a {@link ComparisonType}, and count to permanents on
42       * the battlefield when checking the condition during the
43       * {@link #apply(mage.game.Game, mage.abilities.Ability) apply} method
44       * invocation.
45       *
46       * @param filter
47       * @param type
48       * @param count
49       */
50      public PermanentsOnTheBattlefieldCondition(FilterPermanent filter, ComparisonType type, int count) {
51          this(filter, type, count, true);
52      }
53  
54      public PermanentsOnTheBattlefieldCondition(FilterPermanent filter, ComparisonType type, int count, boolean onlyControlled) {
55          this.filter = filter;
56          this.type = type;
57          this.count = count;
58          this.onlyControlled = onlyControlled;
59      }
60  
61      @Override
62      public boolean apply(Game game, Ability source) {
63          FilterPermanent localFilter;
64          if (onlyControlled) {
65              localFilter = filter.copy();
66              localFilter.add(new ControllerIdPredicate(source.getControllerId()));
67          } else {
68              localFilter = filter;
69          }
70          return ComparisonType.compare(game.getBattlefield().count(
71                  localFilter, source.getControllerId(), source, game
72          ), type, count);
73      }
74  
75      @Override
76      public String toString() {
77          return filter.getMessage();
78      }
79  }

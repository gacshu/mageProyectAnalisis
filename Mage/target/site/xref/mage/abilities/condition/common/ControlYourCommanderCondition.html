1   package mage.abilities.condition.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.condition.Condition;
5   import mage.constants.CommanderCardType;
6   import mage.game.Game;
7   import mage.game.permanent.Permanent;
8   
9   import java.util.Collection;
10  import java.util.Objects;
11  
12  /**
13   * Checks if a player controls their commander.
14   *
15   * @author Alex-Vasile
16   */
17  public enum ControlYourCommanderCondition implements Condition {
18      instance;
19  
20      @Override
21      public boolean apply(Game game, Ability source) {
22          return game.getPlayerList()
23                  .stream()
24                  .map(game::getPlayer)
25                  .filter(Objects::nonNull)
26                  .map(player -> game.getCommandersIds(player, CommanderCardType.COMMANDER_OR_OATHBREAKER, true)) // must search all card parts (example: mdf commander on battlefield)
27                  .flatMap(Collection::stream)
28                  .map(game::getPermanent)
29                  .filter(Objects::nonNull)
30                  .map(Permanent::getOwnerId)
31                  .anyMatch(source.getControllerId()::equals);
32      }
33  
34      @Override
35      public String toString() {
36          return "If you control your commander";
37      }
38  }

2   package mage.abilities.condition.common;
3   
4   import mage.abilities.Ability;
5   import mage.constants.ComparisonType;
6   import mage.abilities.condition.IntCompareCondition;
7   import mage.abilities.effects.Effect;
8   import mage.abilities.effects.common.DevourEffect;
9   import mage.abilities.keyword.DevourAbility;
10  import mage.game.Game;
11  import mage.game.permanent.Permanent;
12  
13  /**
14   * Describes condition about how many creatures were devoured
15   *
16   * @author LevelX2
17   */
18  public class DevouredCreaturesCondition extends IntCompareCondition {
19  
20      public DevouredCreaturesCondition(ComparisonType type, int value) {
21          super(type, value);
22      }
23  
24      @Override
25      protected int getInputValue(Game game, Ability source) {
26          int devouredCreatures = 0;
27          Permanent sourcePermanent = game.getPermanent(source.getSourceId());
28          if (sourcePermanent != null) {
29              for (Ability ability : sourcePermanent.getAbilities()) {
30                  if (ability instanceof DevourAbility) {
31                      for (Effect effect: ability.getEffects()) {
32                          if (effect instanceof DevourEffect) {
33                              DevourEffect devourEffect = (DevourEffect) effect;
34                              devouredCreatures = devourEffect.getDevouredCreaturesAmount(game, sourcePermanent.getId());
35                          }
36                      }
37                  }
38              }
39          }
40          return devouredCreatures;
41      }
42  }

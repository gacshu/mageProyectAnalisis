129     protected boolean checkTargetController(UUID playerId, Game game) {
130         switch (mayActivate) {
131             case ANY:
132             case EACH_PLAYER:
133                 return true;
134             case ACTIVE:
135                 return game.getActivePlayerId() == playerId;
136             case NOT_YOU:
137                 return !controlsAbility(playerId, game);
138             case TEAM:
139                 return !game.getPlayer(controllerId).hasOpponent(playerId, game);
140             case OPPONENT:
141                 return game.getPlayer(controllerId).hasOpponent(playerId, game);
142             case OWNER:
143                 Permanent permanent = game.getPermanent(getSourceId());
144                 return permanent != null && permanent.isOwnedBy(playerId);
145             case YOU:
146                 return controlsAbility(playerId, game);
147             case CONTROLLER_ATTACHED_TO:
148                 Permanent enchantment = game.getPermanent(getSourceId());
149                 if (enchantment == null || enchantment.getAttachedTo() == null) {
150                     return false;
151                 }
152                 Permanent enchanted = game.getPermanent(enchantment.getAttachedTo());
153                 return enchanted != null && enchanted.isControlledBy(playerId);
154         }
155         return true;
156     }

158     /**
159      * Activated ability check, not spells. It contains costs and targets legality too.
160      * <p>
161      * WARNING, don't forget to call super.canActivate on override in card's code in most cases.
162      *
163      * @param playerId
164      * @param game
165      * @return
166      */
167     @Override
168     public ActivationStatus canActivate(UUID playerId, Game game) {
169         //20091005 - 602.2
170         if (!(hasMoreActivationsThisTurn(game)
171                 && (condition == null
172                 || condition.apply(game, this)))) {
173             return ActivationStatus.getFalse();
174         }
175 
176         if (!this.checkTargetController(playerId, game)) {
177             return ActivationStatus.getFalse();
178         }
179 
180         // timing check
181         //20091005 - 602.5d/602.5e
182         boolean asInstant;
183         ApprovingObject approvingObject = game.getContinuousEffects()
184                 .asThough(sourceId,
185                         AsThoughEffectType.ACTIVATE_AS_INSTANT,
186                         this,
187                         controllerId,
188                         game);
189         asInstant = approvingObject != null;
190         asInstant |= (timing == TimingRule.INSTANT);
191         if (!asInstant && !game.canPlaySorcery(playerId)) {
192             return ActivationStatus.getFalse();
193         }
194 
195         // targets and costs check
196         if (!costs.canPay(this, this, playerId, game)
197                 || !canChooseTarget(game, playerId)) {
198             return ActivationStatus.getFalse();
199         }
200 
201         // all fine, can be activated
202         // TODO: WTF, must be rework to remove data change in canActivate call
203         //  (it can be called from any place by any player or card).
204         //  game.inCheckPlayableState() can't be a help here cause some cards checking activating status,
205         //  activatorId must be removed
206         this.activatorId = playerId;
207         return new ActivationStatus(true, approvingObject);
208     }

217     protected boolean controlsAbility(UUID playerId, Game game) {
218         if (this.controllerId != null && this.controllerId.equals(playerId)) {
219             return true;
220         }
221         MageObject mageObject = game.getObject(this.sourceId);
222         if (mageObject instanceof CommandObject) {
223             return ((CommandObject) mageObject).isControlledBy(playerId);
224         } else if (game.getState().getZone(this.sourceId) != Zone.BATTLEFIELD) {
225             return ((Card) mageObject).isOwnedBy(playerId);
226         }
227         return false;
228     }

248     protected boolean hasMoreActivationsThisTurn(Game game) {
249         if (getMaxActivationsPerTurn(game) == Integer.MAX_VALUE && maxActivationsPerGame == Integer.MAX_VALUE) {
250             return true;
251         }
252         ActivationInfo activationInfo = getActivationInfo(game);
253         if (activationInfo == null) {
254             return true;
255         }
256         if (activationInfo.totalActivations >= maxActivationsPerGame) {
257             return false;
258         }
259         return activationInfo.turnNum != game.getTurnNum()
260                 || activationInfo.activationCounter < getMaxActivationsPerTurn(game);
261     }

263     @Override
264     public boolean activate(Game game, boolean noMana) {
265         if (!hasMoreActivationsThisTurn(game) || !super.activate(game, noMana)) {
266             return false;
267         }
268         ActivationInfo activationInfo = getActivationInfo(game);
269         if (activationInfo == null) {
270             activationInfo = new ActivationInfo(game.getTurnNum(), 1, 0);
271         } else if (activationInfo.turnNum != game.getTurnNum()) {
272             activationInfo.turnNum = game.getTurnNum();
273             activationInfo.activationCounter = 1;
274         } else {
275             activationInfo.activationCounter++;
276         }
277         activationInfo.totalActivations++;
278         setActivationInfo(activationInfo, game);
279         return true;
280     }

292     protected ActivationInfo getActivationInfo(Game game) {
293         Integer turnNum = (Integer) game.getState()
294                 .getValue(CardUtil.getCardZoneString("activationsTurn" + originalId, sourceId, game));
295         Integer activationCount = (Integer) game.getState()
296                 .getValue(CardUtil.getCardZoneString("activationsCount" + originalId, sourceId, game));
297         Integer totalActivations = (Integer) game.getState()
298                 .getValue(CardUtil.getCardZoneString("totalActivations" + originalId, sourceId, game));
299         if (turnNum == null || activationCount == null || totalActivations == null) {
300             return null;
301         }
302         return new ActivationInfo(turnNum, activationCount, totalActivations);
303     }
304 
305     protected void setActivationInfo(ActivationInfo activationInfo, Game game) {
306         game.getState().setValue(CardUtil
307                 .getCardZoneString("activationsTurn" + originalId, sourceId, game), activationInfo.turnNum);
308         game.getState().setValue(CardUtil
309                 .getCardZoneString("activationsCount" + originalId, sourceId, game), activationInfo.activationCounter);
310         game.getState().setValue(CardUtil
311                 .getCardZoneString("totalActivations" + originalId, sourceId, game), activationInfo.totalActivations);
312     }

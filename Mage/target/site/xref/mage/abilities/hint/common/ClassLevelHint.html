1   package mage.abilities.hint.common;

3   import mage.abilities.Ability;
4   import mage.abilities.hint.Hint;
5   import mage.game.Game;
6   import mage.game.permanent.Permanent;
7   
8   /**
9    * @author TheElk801
10   */
11  public enum ClassLevelHint implements Hint {
12      instance;
13  
14      @Override
15      public String getText(Game game, Ability ability) {
16          Permanent permanent = ability.getSourcePermanentIfItStillExists(game);
17          if (permanent == null) {
18              return null;
19          }
20          return "Class level: " + permanent.getClassLevel();
21      }
22  
23      @Override
24      public ClassLevelHint copy() {
25          return this;
26      }
27  }

1   package mage.abilities.hint.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.hint.Hint;
5   import mage.constants.SubType;
6   import mage.filter.StaticFilters;
7   import mage.game.Game;
8   
9   import java.util.Arrays;
10  import java.util.Collection;
11  import java.util.Comparator;
12  import java.util.List;
13  import java.util.stream.Collectors;
14  
15  /**
16   * @author TheElk801
17   */
18  public enum DomainHint implements Hint {
19      instance;
20      private static final List<String> typesInOrder = Arrays.asList("Plains", "Island", "Swamp", "Mountain", "Forest");
21  
22      @Override
23      public String getText(Game game, Ability ability) {
24          List<String> landTypes = game.getBattlefield()
25                  .getActivePermanents(
26                          StaticFilters.FILTER_CONTROLLED_PERMANENT_LANDS,
27                          ability.getControllerId(), ability, game
28                  ).stream()
29                  .map(permanent -> SubType
30                          .getBasicLands()
31                          .stream()
32                          .filter(subType -> permanent.hasSubtype(subType, game))
33                          .collect(Collectors.toSet()))
34                  .flatMap(Collection::stream)
35                  .distinct()
36                  .map(SubType::getDescription)
37                  .sorted(Comparator.comparing(typesInOrder::indexOf))
38                  .collect(Collectors.toList());
39          return "Basic land types among lands you control: " + landTypes.size()
40                  + (landTypes.size() > 0 ? " (" + String.join(", ", landTypes) + ')' : "");
41      }
42  
43      @Override
44      public DomainHint copy() {
45          return instance;
46      }
47  }

1   package mage.abilities.hint.common;
2   
3   import mage.abilities.Ability;
4   import mage.abilities.hint.Hint;
5   import mage.game.Game;
6   import mage.watchers.common.CastSpellLastTurnWatcher;
7   
8   /**
9    * @author TheElk801
10   */
11  public enum DayNightHint implements Hint {
12      instance;
13  
14      @Override
15      public String getText(Game game, Ability ability) {
16          if (!game.hasDayNight()) {
17              return "It's neither day nor night.";
18          }
19          boolean isDay = game.checkDayNight(true);
20          int spellsThisTurn = game
21                  .getState()
22                  .getWatcher(CastSpellLastTurnWatcher.class)
23                  .getActivePlayerThisTurnCount();
24          StringBuilder sb = new StringBuilder("It's currently ");
25          sb.append(isDay ? "day" : "night");
26          sb.append(", active player has cast ");
27          sb.append(spellsThisTurn);
28          sb.append(" spells this turn. It will ");
29          sb.append((isDay ? spellsThisTurn == 0 : spellsThisTurn >= 2) ? "" : "not");
30          sb.append(" become ");
31          sb.append(isDay ? "night" : "day");
32          sb.append(" next turn.");
33          return sb.toString();
34      }
35  
36      @Override
37      public DayNightHint copy() {
38          return this;
39      }
40  }

3   package mage.abilities.abilityword;
4   
5   import mage.abilities.common.SimpleStaticAbility;
6   import mage.abilities.condition.common.CommanderInPlayCondition;
7   import mage.abilities.decorator.ConditionalContinuousEffect;
8   import mage.abilities.effects.ContinuousEffect;
9   import mage.abilities.effects.Effect;
10  import mage.abilities.effects.Effects;
11  import mage.abilities.effects.common.continuous.BoostSourceEffect;
12  import mage.constants.Duration;
13  import mage.constants.Zone;
14  
15  /**
16   *
17   * @author emerald000
18   */
19  
20  public class LieutenantAbility extends SimpleStaticAbility {
21  
22      public LieutenantAbility(ContinuousEffect effect) {
23          super(Zone.BATTLEFIELD, new ConditionalContinuousEffect(new BoostSourceEffect(2, 2, Duration.WhileOnBattlefield), CommanderInPlayCondition.instance, "<i>Lieutenant</i> &mdash; As long as you control your commander, {this} gets +2/+2"));
24          this.addEffect(new ConditionalContinuousEffect(effect, CommanderInPlayCondition.instance, effect.getText(null)));
25      }
26      
27      public LieutenantAbility(Effects effects) {
28          super(Zone.BATTLEFIELD, new ConditionalContinuousEffect(new BoostSourceEffect(2, 2, Duration.WhileOnBattlefield), CommanderInPlayCondition.instance, "<i>Lieutenant</i> &mdash; As long as you control your commander, {this} gets +2/+2"));
29          for (Effect effect : effects) {
30              this.addEffect(new ConditionalContinuousEffect((ContinuousEffect) effect, CommanderInPlayCondition.instance, effect.getText(null)));
31          }
32      }
33  
34      public LieutenantAbility(final LieutenantAbility ability) {
35          super(ability);
36      }
37  
38      @Override
39      public LieutenantAbility copy() {
40          return new LieutenantAbility(this);
41      }
42  }

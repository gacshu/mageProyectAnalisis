501     public static class SubTypePredicate implements Predicate<MageObject> {
502 
503         private final SubType subtype;
504 
505         private SubTypePredicate(SubType subtype) {
506             this.subtype = subtype;
507         }
508 
509 
510         @Override
511         public boolean apply(MageObject input, Game game) {
512             return input.hasSubtype(subtype, game);
513         }
514 
515         @Override
516         public String toString() {
517             return "Subtype(" + subtype + ')';
518         }
519     }
520 
521     private static final Set<SubType> landTypes = new HashSet<>();
522     private static final Map<SubTypeSet, Set<SubType>> subTypeSetMap = new HashMap<>();
523 
524     static {
525         for (SubTypeSet subTypeSet : SubTypeSet.values()) {
526             subTypeSetMap.put(
527                     subTypeSet,
528                     Arrays.stream(values())
529                             .filter(subType -> subType.getSubTypeSet() == subTypeSet)
530                             .collect(Collectors.toSet())
531             );
532         }
533         landTypes.addAll(subTypeSetMap.get(SubTypeSet.BasicLandType));
534         landTypes.addAll(subTypeSetMap.get(SubTypeSet.NonBasicLandType));
535     }
536 
537     private final SubTypeSet subTypeSet;
538     private final String description;
539     private final boolean customSet;
540     private final SubTypePredicate predicate;
541 
542     SubType(String description, SubTypeSet subTypeSet) {
543         this(description, subTypeSet, false);
544     }
545 
546     SubType(String description, SubTypeSet subTypeSet, boolean customSet) {
547         this.description = description;
548         this.subTypeSet = subTypeSet;
549         this.customSet = customSet;
550         this.predicate = new SubTypePredicate(this);
551     }
552 
553     public String getDescription() {
554         return description;
555     }
556 
557     @Override
558     public String toString() {
559         return description;
560     }
561 
562     public SubTypePredicate getPredicate() {
563         return predicate;
564     }

567     public String getIndefiniteArticle() {
568         if (isVowel(description.charAt(0))) {
569             return "an";
570         } else {
571             return "a";
572         }
573     }
574 
575     private boolean isVowel(char c) {
576         return "AEIOUaeiou".indexOf(c) != -1;
577     }

579     public boolean isCustomSet() {
580         return customSet;
581     }

583     public static SubType fromString(String value) {
584         for (SubType st : SubType.values()) {
585             if (st.toString().equals(value)) {
586                 return st;
587             }
588         }
589 
590         throw new IllegalArgumentException("Can''t find subtype enum value: " + value);
591     }

593     public static SubType byDescription(String subType) {
594         for (SubType s : values()) {
595             if (s.getDescription().equals(subType)) {
596                 return s;
597             }
598         }
599         org.apache.log4j.Logger.getLogger(SubType.class).error("no subtype for " + subType + " exists");
600         return null;
601     }

608         return canGain(null, mageObject);

611     public boolean canGain(Game game, MageObject mageObject) {
612         switch (subTypeSet) {
613             case CreatureType:
614                 return mageObject.isCreature(game) || mageObject.isTribal(game);
615             case BasicLandType:
616             case NonBasicLandType:
617                 return mageObject.isLand(game);
618             case EnchantmentType:
619                 return mageObject.isEnchantment(game);
620             case ArtifactType:
621                 return mageObject.isArtifact(game);
622             case PlaneswalkerType:
623                 return mageObject.isPlaneswalker(game);
624             case SpellType:
625                 return mageObject.isInstantOrSorcery(game);
626         }
627         return false;
628     }

6   public enum Duration {
7       OneUse("", true, true),
8       EndOfGame("for the rest of the game", false, false),
9       WhileOnBattlefield("", false, false),
10      WhileControlled("for as long as you control {this}", true, false),
11      WhileOnStack("", false, true),
12      WhileInGraveyard("", false, false),
13      EndOfTurn("until end of turn", true, true),
14      UntilYourNextTurn("until your next turn", true, true),
15      UntilYourNextEndStep("until your next end step", true, true),
16      UntilEndOfYourNextTurn("until the end of your next turn", true, true),
17      UntilSourceLeavesBattlefield("until {this} leaves the battlefield", true, false), // supported for continuous layered effects
18      EndOfCombat("until end of combat", true, true),
19      EndOfStep("until end of phase step", true, true),
20      Custom("", true, true);
21  
22      private final String text;
23      private final boolean onlyValidIfNoZoneChange; // defines if an effect lasts only if the source has not changed zone since init of the effect
24      private final boolean fixedController; // has the controller of the effect to change, if the controller of the source changes
25  
26      Duration(String text, boolean onlyValidIfNoZoneChange, boolean fixedController) {
27          this.text = text;
28          this.onlyValidIfNoZoneChange = onlyValidIfNoZoneChange;
29          this.fixedController = fixedController;
30      }
31  
32      @Override
33      public String toString() {
34          return text;
35      }
36  
37      public boolean isOnlyValidIfNoZoneChange() {
38          return onlyValidIfNoZoneChange;
39      }
40  
41      public boolean isFixedController() {
42          return fixedController;
43      }
44  }

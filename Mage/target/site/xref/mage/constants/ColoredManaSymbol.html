7   public enum ColoredManaSymbol {

11      R("R", "red", "<font color='red'>red</font>"),
12      G("G", "green", "<font color='green'>green</font>"),
13      O("O", "gold", "<font color='gold'>gold</font>");
14  
15      private final String text;
16      private final String colorName;
17      private final String colorHtmlName;
18  
19      ColoredManaSymbol(String text, String colorName, String colorHtmlName) {

21          this.colorName = colorName;
22          this.colorHtmlName = colorHtmlName;

24

38      public static ColoredManaSymbol lookup(char c) {

40              case 'W':
41                  return W;
42              case 'R':
43                  return R;
44              case 'G':
45                  return G;
46              case 'B':
47                  return B;
48              case 'U':
49                  return U;
50

52                  return O;

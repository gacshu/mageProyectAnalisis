37      public static Choice getChoiceOfManaTypes(Set<ManaType> types, boolean onlyColors) {
38          Choice choice = new ChoiceColor(true);
39          choice.getChoices().clear();
40          choice.setMessage("Pick a mana " + (onlyColors ? "color" : "type"));
41          if (types.contains(ManaType.BLACK)) {
42              choice.getChoices().add("Black");
43          }
44          if (types.contains(ManaType.RED)) {
45              choice.getChoices().add("Red");
46          }
47          if (types.contains(ManaType.BLUE)) {
48              choice.getChoices().add("Blue");
49          }
50          if (types.contains(ManaType.GREEN)) {
51              choice.getChoices().add("Green");
52          }
53          if (types.contains(ManaType.WHITE)) {
54              choice.getChoices().add("White");
55          }
56          if (types.contains(ManaType.COLORLESS) && !onlyColors) {
57              choice.getChoices().add("Colorless");
58          }
59          return choice;
60      }

62      public static List<Mana> getManaListFromManaTypes(Set<ManaType> manaTypes, boolean onlyColors) {
63          List<Mana> netManas = new ArrayList<>();
64          if ((manaTypes.size() == 5 && !manaTypes.contains(ManaType.COLORLESS)) || manaTypes.size() == 6) { // GENERIC should never be returned from getManaTypes
65              netManas.add(Mana.AnyMana(1));
66          } else {
67              if (manaTypes.contains(ManaType.BLACK)) {
68                  netManas.add(Mana.BlackMana(1));
69              }
70              if (manaTypes.contains(ManaType.RED)) {
71                  netManas.add(Mana.RedMana(1));
72              }
73              if (manaTypes.contains(ManaType.BLUE)) {
74                  netManas.add(Mana.BlueMana(1));
75              }
76              if (manaTypes.contains(ManaType.GREEN)) {
77                  netManas.add(Mana.GreenMana(1));
78              }
79              if (manaTypes.contains(ManaType.WHITE)) {
80                  netManas.add(Mana.WhiteMana(1));
81              }
82          }
83          if (!onlyColors && manaTypes.contains(ManaType.COLORLESS)) {
84              netManas.add(Mana.ColorlessMana(1));
85          }
86  
87          return netManas;
88      }

90      public static Set<ManaType> getManaTypesFromManaList(Mana mana) {
91          Set<ManaType> manaTypes = EnumSet.noneOf(ManaType.class);
92          if (mana.getAny() > 0) {
93              manaTypes.add(ManaType.BLACK);
94              manaTypes.add(ManaType.BLUE);
95              manaTypes.add(ManaType.GREEN);
96              manaTypes.add(ManaType.WHITE);
97              manaTypes.add(ManaType.RED);
98          } else {
99              if (mana.getBlack() > 0) {
100                 manaTypes.add(ManaType.BLACK);
101             }
102             if (mana.getBlue() > 0) {
103                 manaTypes.add(ManaType.BLUE);
104             }
105             if (mana.getGreen() > 0) {
106                 manaTypes.add(ManaType.GREEN);
107             }
108             if (mana.getWhite() > 0) {
109                 manaTypes.add(ManaType.WHITE);
110             }
111             if (mana.getRed() > 0) {
112                 manaTypes.add(ManaType.RED);
113             }
114         }
115         if (mana.getColorless() > 0) {
116             manaTypes.add(ManaType.COLORLESS);
117         }
118         return manaTypes;
119     }

121     /**
122      * Utility function to find the ManaType associated with a name without needing to have the if-statements
123      * cluttering up the code.
124      * <p>
125      * Used for things like mapping back to a ManaType after the user chose from several of them.
126      *
127      * @param name  The name of the mana to find
128      * @return      The ManaType representing that mana (or null)
129      */
130     public static ManaType findByName(String name) {
131         switch (name) {
132             case "Black":
133                 return ManaType.BLACK;
134             case "Blue":
135                 return ManaType.BLUE;
136             case "Red":
137                 return ManaType.RED;
138             case "Green":
139                 return ManaType.GREEN;
140             case "White":
141                 return ManaType.WHITE;
142             case "Colorless":
143                 return ManaType.COLORLESS;
144             default:
145                 return null;
146         }
147     }

149     public static Set<ManaType> getManaTypesFromManaList(List<Mana> manaList) {
150         Set<ManaType> manaTypes = new HashSet<>();
151         for (Mana mana : manaList) {
152             manaTypes.addAll(getManaTypesFromManaList(mana));
153         }
154         return manaTypes;
155     }
156     public static Set<ManaType> getTrueManaTypes() {
157         return EnumSet.of(BLACK, BLUE, GREEN, RED, WHITE, COLORLESS);

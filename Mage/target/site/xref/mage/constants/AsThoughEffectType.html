3   /**
4    * Allows to change game rules and logic by special conditionals from ContinuousEffects (example: play card from non hand)
5    * <p>
6    * Can be applied to the game, object or specific ability (affectedAbility param in ContinuousEffects.asThough)
7    * <p>
8    * There are two usage styles possible:
9    * - object specific rules and conditionals (use ContinuousEffects.asThough)
10   * - global rules (use game.getState().getContinuousEffects().getApplicableAsThoughEffects)
11   * Each AsThough type supports only one usage style
12   *
13   * @author North
14   */
15  public enum AsThoughEffectType {
16      ATTACK,
17      ATTACK_AS_HASTE,
18      ACTIVATE_HASTE(true, false),
19      //
20      BLOCK_TAPPED,
21      BLOCK_SHADOW,
22      BLOCK_DRAGON,
23      BLOCK_LANDWALK,
24      BLOCK_PLAINSWALK,
25      BLOCK_ISLANDWALK,
26      BLOCK_SWAMPWALK,
27      BLOCK_MOUNTAINWALK,
28      BLOCK_FORESTWALK,
29      //
30      DAMAGE_NOT_BLOCKED,
31      //
32      // PLAY_FROM_NOT_OWN_HAND_ZONE + CAST_AS_INSTANT:
33      // 1. Do not use dialogs in "applies" method for that type of effect (it calls multiple times and will freeze the game)
34      // 2. All effects in "applies" must checks affectedControllerId/playerId.equals(source.getControllerId()) (if not then all players will be able to play it)
35      // 3. Target must points to mainCard, but checking goes for every card's parts and characteristics from objectId (split, adventure)
36      // 4. You must implement/override an applies method with "Ability affectedAbility" (e.g. check multiple play/cast abilities from all card's parts)
37      // TODO: search all PLAY_FROM_NOT_OWN_HAND_ZONE and CAST_AS_INSTANT effects and add support of mainCard and objectId
38      PLAY_FROM_NOT_OWN_HAND_ZONE(true, true),
39      CAST_AS_INSTANT(true, true),
40      //
41      ACTIVATE_AS_INSTANT(true, false),
42      //
43      SHROUD,
44      HEXPROOF,
45      //
46      PAY_0_ECHO(true, false),
47      LOOK_AT_FACE_DOWN,
48      //
49      // SPEND_OTHER_MANA:
50      // 1. It's uses for mana calcs at any zone, not stack only
51      // 2. Compare zone change counter as "objectZCC <= targetZCC + 1"
52      // 3. Compare zone with original (like exiled) and stack, not stack only
53      // TODO: search all SPEND_ONLY_MANA effects and improve counters compare as SPEND_OTHER_MANA
54      SPEND_OTHER_MANA(true, false),
55      //
56      SPEND_ONLY_MANA,
57      //
58      // ALLOW_FORETELL_ANYTIME:
59      // For Cosmos Charger effect
60      ALLOW_FORETELL_ANYTIME;
61  
62      private final boolean needAffectedAbility; // mark what AsThough check must be called for specific ability, not full object (example: spell check)
63      private final boolean needPlayCardAbility; // mark what AsThough check must be called for play/cast abilities
64  
65      AsThoughEffectType() {
66          this(false, false);
67      }
68  
69      AsThoughEffectType(boolean needAffectedAbility, boolean needPlayCardAbility) {
70          this.needAffectedAbility = needAffectedAbility;
71          this.needPlayCardAbility = needPlayCardAbility;
72      }
73  
74      public boolean needAffectedAbility() {
75          return needAffectedAbility;
76      }
77  
78      public boolean needPlayCardAbility() {
79          return needPlayCardAbility;
80      }
81  }

6   public enum PhaseStep {
7       UNTAP("Untap", 0, "untap step", "UN"),
8       UPKEEP("Upkeep", 1, "upkeep", "UP"), // card texts don't use the word "step" for this phase step
9       DRAW("Draw", 2, "draw step", "DR"),
10      PRECOMBAT_MAIN("Precombat Main", 3, "precombat main step", "M1"),
11      BEGIN_COMBAT("Begin Combat", 4, "begin combat step", "BC"),
12      DECLARE_ATTACKERS("Declare Attackers", 5, "declare attackers step", "DA"),
13      DECLARE_BLOCKERS("Declare Blockers", 6, "declare blockers step", "DB"),
14      FIRST_COMBAT_DAMAGE("First Combat Damage", 7, "first combat damage", "FCD"),
15      COMBAT_DAMAGE("Combat Damage", 8, "combat damage step", "CD"),
16      END_COMBAT("End Combat", 9, "end of combat step", "EC"),
17      POSTCOMBAT_MAIN("Postcombat Main", 10, "postcombat main step", "M2"),
18      END_TURN("End Turn", 11, "end turn step", "ET"),
19      CLEANUP("Cleanup", 12, "cleanup step", "CL");
20  
21      private final String text;
22      private final String stepText;
23      private final String stepShortText; // for chats/logs
24  
25      /**
26       * Index is used for game state scoring system.
27       */
28      private final int index;
29  
30      PhaseStep(String text, int index, String stepText, String stepShortText) {
31          this.text = text;
32          this.index = index;
33          this.stepText = stepText;
34          this.stepShortText = stepShortText;
35      }
36  
37      public boolean isBefore(PhaseStep other) {
38          return this.getIndex() < other.getIndex();
39      }
40  
41      public boolean isAfter(PhaseStep other) {
42          return this.getIndex() > other.getIndex();
43      }
44  
45      public int getIndex() {
46          return index;
47      }
48  
49      @Override
50      public String toString() {
51          return text;
52      }
53  
54      public String getStepText() {
55          return stepText;
56      }
57  
58      public String getStepShortText() {
59          return stepShortText;
60      }
61  }

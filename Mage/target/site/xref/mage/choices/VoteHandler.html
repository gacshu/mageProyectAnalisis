17  public abstract class VoteHandler<T> {
18  
19      protected final Map<UUID, List<T>> playerMap = new HashMap<>();
20      protected VoteHandlerAI<T> aiVoteHint = null;
21  
22      public void doVotes(Ability source, Game game) {
23          doVotes(source, game, null);
24      }
25  
26      public void doVotes(Ability source, Game game, VoteHandlerAI<T> aiVoteHint) {
27          this.aiVoteHint = aiVoteHint;
28          this.playerMap.clear();
29          int stepCurrent = 0;
30          int stepTotal = game.getState().getPlayersInRange(source.getControllerId(), game).size();
31          for (UUID playerId : game.getState().getPlayersInRange(source.getControllerId(), game)) {
32              stepCurrent++;
33              VoteEvent event = new VoteEvent(playerId, source);
34              game.replaceEvent(event);
35              Player player = game.getPlayer(event.getTargetId());
36              Player decidingPlayer = game.getPlayer(event.getPlayerId());
37              if (player == null || decidingPlayer == null) {
38                  continue;
39              }
40              int voteCount = event.getExtraVotes() + event.getOptionalExtraVotes() + 1;
41              for (int i = 0; i < voteCount; i++) {
42                  // Decision for extra choice goes from original player, not from deciding.
43                  // Rules from Illusion of Choice:
44                  // If another player controls Ballot Broker, that player first takes their “normal” vote
45                  // with you choosing the result, then that player decides whether they are taking the
46                  // additional vote. If there is an additional vote, you again choose the result.
47                  // (2016-08-23)
48  
49                  // Outcome.Benefit - AI must use extra vote all the time
50                  if (i > event.getExtraVotes() && !player.chooseUse(
51                          Outcome.Benefit, "Use an extra vote?", source, game
52                  )) {
53                      continue;
54                  }
55  
56                  String stepName = (i > 0 ? "extra step" : "step");
57                  String voteInfo = String.format("Vote, %s %d of %d", stepName, stepCurrent, stepTotal);
58                  T vote;
59                  if (decidingPlayer.isComputer() && this.aiVoteHint != null) {
60                      // ai choose
61                      vote = this.aiVoteHint.makeChoice(this, player, decidingPlayer, source, game);
62                  } else {
63                      // human choose
64                      vote = playerChoose(voteInfo, player, decidingPlayer, source, game);
65                  }
66                  if (vote == null) {
67                      continue;
68                  }
69                  String message = voteInfo + ": " + player.getName() + " voted for " + voteName(vote);
70                  if (!Objects.equals(player, decidingPlayer)) {
71                      message += " (chosen by " + decidingPlayer.getName() + ')';
72                  }
73                  game.informPlayers(message);
74                  this.playerMap.computeIfAbsent(playerId, x -> new ArrayList<>()).add(vote);
75              }
76          }
77  
78          // show final results to players
79  
80          Map<T, Integer> totalVotes = new LinkedHashMap<>();
81          // fill by possible choices
82          this.getPossibleVotes(source, game).forEach(vote -> {
83              totalVotes.putIfAbsent(vote, 0);
84          });
85          // fill by real choices
86          playerMap.entrySet()
87                  .stream()
88                  .flatMap(votesList -> votesList.getValue().stream())
89                  .forEach(vote -> {
90                      totalVotes.compute(vote, CardUtil::setOrIncrementValue);
91                  });
92  
93          Set<T> winners = this.getMostVoted();
94          String totalVotesStr = totalVotes.entrySet()
95                  .stream()
96                  .map(entry -> (winners.contains(entry.getKey()) ? " -win- " : " -lose- ") + voteName(entry.getKey()) + ": " + entry.getValue())
97                  .sorted()
98                  .collect(Collectors.joining("<br>"));
99          game.informPlayers("Vote results:<br>" + totalVotesStr);
100 
101         game.fireEvent(new VotedEvent(source, this));
102     }

104     /**
105      * Return possible votes. Uses for info only (final results).
106      *
107      * @param source
108      * @param game
109      * @return
110      */
111     protected abstract Set<T> getPossibleVotes(Ability source, Game game);
112 
113     /**
114      * Choose dialog for voting. Another player can choose it (example: Illusion of Choice)
115      *
116      * @param player
117      * @param decidingPlayer
118      * @param source
119      * @param game
120      * @return
121      */
122     protected abstract T playerChoose(String voteInfo, Player player, Player decidingPlayer, Ability source, Game game);
123 
124     /**
125      * Show readable choice name
126      *
127      * @param vote
128      * @return
129      */
130     protected abstract String voteName(T vote);
131 
132     public List<T> getVotes(UUID playerId) {
133         return playerMap.computeIfAbsent(playerId, x -> new ArrayList<>());
134     }
135 
136     public int getVoteCount(T vote) {
137         return playerMap
138                 .values()
139                 .stream()
140                 .flatMap(Collection::stream)
141                 .map(vote::equals)
142                 .mapToInt(x -> x ? 1 : 0)
143                 .sum();
144     }
145 
146     public List<UUID> getVotedFor(T vote) {
147         return playerMap
148                 .entrySet()
149                 .stream()
150                 .filter(entry -> entry.getValue() != null && entry.getValue().contains(vote))
151                 .map(Map.Entry::getKey)
152                 .collect(Collectors.toList());
153     }
154 
155     public Set<T> getMostVoted() {
156         Map<T, Integer> map = new HashMap<>();
157         playerMap
158                 .values()
159                 .stream()
160                 .flatMap(Collection::stream)
161                 .forEach(t -> map.compute(t, CardUtil::setOrIncrementValue));
162         int max = map.values().stream().mapToInt(x -> x).max().orElse(0);
163         return map
164                 .entrySet()
165                 .stream()
166                 .filter(e -> e.getValue() >= max)
167                 .map(Map.Entry::getKey)
168                 .collect(Collectors.toSet());
169     }
170 
171     public Set<UUID> getDidntVote(UUID playerId) {
172         if (playerMap.computeIfAbsent(playerId, x -> new ArrayList<>()).isEmpty()) {
173             return playerMap.keySet();
174         }
175         return playerMap
176                 .entrySet()
177                 .stream()
178                 .filter(e -> e.getValue() != null && !e.getValue().isEmpty())
179                 .filter(e -> !e.getValue().stream().allMatch(playerMap.get(playerId)::contains))
180                 .map(Map.Entry::getKey)
181                 .collect(Collectors.toSet());
182     }

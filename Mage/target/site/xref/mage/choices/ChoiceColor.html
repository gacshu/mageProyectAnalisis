1   package mage.choices;

3   import mage.MageObject;
4   import mage.Mana;
5   import mage.ObjectColor;
6   
7   import java.util.ArrayList;
8   import java.util.List;

10  /**
11   * @author BetaSteward_at_googlemail.com, JayDi85
12   */
13  public class ChoiceColor extends ChoiceImpl {
14  
15      private static final List<String> colorChoices = getBaseColors();
16  
17      public static List<String> getBaseColors() {
18          List<String> arr = new ArrayList<>();
19          arr.add("White");
20          arr.add("Blue");
21          arr.add("Black");
22          arr.add("Red");
23          arr.add("Green");
24          return arr;
25      }
26  
27      public ChoiceColor() {
28          this(true);
29      }
30  
31      public ChoiceColor(boolean required) {
32          this(required, "Choose color");
33      }
34  
35      public ChoiceColor(boolean required, String chooseMessage) {
36          this(required, chooseMessage, "");
37      }
38  
39      public ChoiceColor(boolean required, String chooseMessage, MageObject source) {
40          this(required, chooseMessage, source.getIdName());
41      }
42  
43      public ChoiceColor(boolean required, String chooseMessage, String chooseSubMessage) {
44          super(required);
45  
46          this.choices.addAll(colorChoices);
47  
48          this.setMessage(chooseMessage);
49          this.setSubMessage(chooseSubMessage);
50      }
51  
52      public ChoiceColor(final ChoiceColor choice) {
53          super(choice);
54      }
55  
56      @Override
57      public ChoiceColor copy() {
58          return new ChoiceColor(this);
59      }
60  
61      public void removeColorFromChoices(String colorName) {
62          this.choices.remove(colorName);
63      }
64  
65      public ObjectColor getColor() {
66          return getColorFromString(choice);
67      }
68  
69      public static ObjectColor getColorFromString(String colorString) {
70          if (colorString == null) {
71              return null;
72          }
73          ObjectColor color = new ObjectColor();
74          switch (colorString) {
75              case "Black":
76                  color.setBlack(true);
77                  break;
78              case "Blue":
79                  color.setBlue(true);
80                  break;
81              case "Green":
82                  color.setGreen(true);
83                  break;
84              case "Red":
85                  color.setRed(true);
86                  break;
87              case "White":
88                  color.setWhite(true);
89                  break;
90          }
91          return color;
92      }
93  
94      public Mana getMana(int amount) {
95          Mana mana;
96          if (getColor().isBlack()) {
97              mana = Mana.BlackMana(amount);
98          } else if (getColor().isBlue()) {
99              mana = Mana.BlueMana(amount);
100         } else if (getColor().isRed()) {
101             mana = Mana.RedMana(amount);
102         } else if (getColor().isGreen()) {
103             mana = Mana.GreenMana(amount);
104         } else if (getColor().isWhite()) {
105             mana = Mana.WhiteMana(amount);
106         } else {
107             mana = Mana.ColorlessMana(amount);
108         }
109         return mana;
110     }
111 
112     public void increaseMana(Mana mana) {
113         if (getColor().isBlack()) {
114             mana.increaseBlack();
115         } else if (getColor().isBlue()) {
116             mana.increaseBlue();
117         } else if (getColor().isRed()) {
118             mana.increaseRed();
119         } else if (getColor().isGreen()) {
120             mana.increaseGreen();
121         } else if (getColor().isWhite()) {
122             mana.increaseWhite();
123         } else {
124             mana.increaseColorless();
125         }
126     }
127 }

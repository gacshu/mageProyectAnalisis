232     @Override
233     public void setRandomChoice() {
234 
235         if (this.isKeyChoice()) {
236             // key mode
237             String[] vals = this.getKeyChoices().keySet().toArray(new String[0]);
238             if (vals.length > 0) {
239                 int choiceNum = RandomUtil.nextInt(vals.length);
240                 this.setChoiceByKey(vals[choiceNum], false);
241             }
242         } else {
243             // string mode
244             String[] vals = this.getChoices().toArray(new String[0]);
245             if (vals.length > 0) {
246                 int choiceNum = RandomUtil.nextInt(vals.length);
247                 this.setChoice(vals[choiceNum], false);
248             }
249         }
250     }

252     @Override
253     public boolean setChoiceByAnswers(List<String> answers, boolean removeSelectAnswerFromList) {
254         // select by answers
255         if (this.isKeyChoice()) {
256             // keys mode
257             for (String needChoice : answers) {
258                 for (Map.Entry<String, String> currentChoice : this.getKeyChoices().entrySet()) {
259                     if (currentChoice.getKey().equals(needChoice)) {
260                         if (removeSelectAnswerFromList) {
261                             this.setChoiceByKey(needChoice, false);
262                             answers.remove(needChoice);
263                         }
264                         return true;
265                     }
266 
267                 }
268             }
269             // no key answer found, try to match by text starting with
270             for (String needChoice : answers) {
271                 for (Map.Entry<String, String> currentChoice : this.getKeyChoices().entrySet()) {
272                     if (currentChoice.getValue().startsWith(needChoice)) {
273                         if (removeSelectAnswerFromList) {
274                             this.setChoiceByKey(currentChoice.getKey(), false);
275                             answers.remove(needChoice);
276                         }
277                         return true;
278                     }
279                 }
280             }
281         } else {
282             // string mode
283             for (String needChoice : answers) {
284                 for (String currentChoice : this.getChoices()) {
285                     if (currentChoice.equals(needChoice)) {
286                         if (removeSelectAnswerFromList) {
287                             this.setChoice(needChoice, false);
288                             answers.remove(needChoice);
289                         }
290                         return true;
291                     }
292                 }
293             }
294         }
295         return false; // can't find answer
296     }

331     private void protectFromEmptyChoices() {
332         // if there are no choices then required must be disabled to allow user to close a dialog
333         // example: database error on too low memory, see Brain Pry and 500 Mb server
334 
335         // normal situation
336         if (!this.required) {
337             return;
338         }
339 
340         // special checkbox can allow empty choices
341         if (this.specialEnabled && this.specialCanBeEmpty) {
342             return;
343         }
344 
345         if (this.choices.isEmpty() && this.keyChoices.isEmpty()) {
346             // it can be a server problems or wrong card code
347             this.required = false;
348             logger.error("Empty choice dialog in " + this.getClass().getCanonicalName(), new Throwable());
349         }

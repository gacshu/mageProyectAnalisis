8   public class ManaChoice {
9       public static Mana chooseAnyColor(Player player, Game game, int amount) {
10          if (player == null) {
11              return null;
12          }
13          Mana mana = new Mana();
14          for (int i = 0; i < amount; i++) {
15              ChoiceColor choiceColor = new ChoiceColor();
16              if (amount > 1) {
17                  choiceColor.setMessage("Choose color " + (i + 1));
18              }
19              if (!player.choose(Outcome.Benefit, choiceColor, game)) {
20                  return null;
21              }
22              choiceColor.increaseMana(mana);
23          }
24          return mana;
25      }
26  
27      public static Mana chooseTwoDifferentColors(Player player, Game game) {
28          Mana mana = new Mana();
29  
30          if (game == null || player == null) {
31              return mana;
32          }
33  
34          ChoiceColor color1 = new ChoiceColor(true, "Choose color 1");
35          if (!player.choose(Outcome.PutManaInPool, color1, game) || color1.getColor() == null) {
36              return mana;
37          }
38  
39          ChoiceColor color2 = new ChoiceColor(true, "Choose color 2");
40          color2.removeColorFromChoices(color1.getChoice());
41          if (!player.choose(Outcome.PutManaInPool, color2, game) || color2.getColor() == null) {
42              return mana;
43          }
44  
45          if (color1.getColor().equals(color2.getColor())) {
46              game.informPlayers("Player " + player.getName() + " is cheating with mana choices.");
47              return mana;
48          }
49  
50          mana.add(color1.getMana(1));
51          mana.add(color2.getMana(1));
52          return mana;
53      }
54  }

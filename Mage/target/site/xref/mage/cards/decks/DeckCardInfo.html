3   package mage.cards.decks;
4   
5   import mage.util.Copyable;
6   
7   import java.io.Serializable;
8   
9   /**
10   *
11   * @author LevelX2
12   */
13  public class DeckCardInfo implements Serializable, Copyable<DeckCardInfo> {
14  
15      private String cardName;
16      private String setCode;
17      private String cardNum;
18      private int quantity;
19  
20      public DeckCardInfo() {
21          super();
22      }
23  
24      public DeckCardInfo(final DeckCardInfo info) {
25          this.cardName = info.cardName;
26          this.setCode = info.setCode;
27          this.cardNum = info.cardNum;
28          this.quantity = info.quantity;
29      }
30  
31      public DeckCardInfo(String cardName, String cardNum, String setCode) {
32          this(cardName, cardNum, setCode, 1);
33      }
34  
35      public DeckCardInfo(String cardName, String cardNum, String setCode, int quantity) {
36          this.cardName = cardName;
37          this.cardNum = cardNum;
38          this.setCode = setCode;
39          this.quantity = quantity;
40      }
41  
42      public String getCardName() {
43          return cardName;
44      }
45  
46      public String getSetCode() {
47          return setCode;
48      }
49  
50      public String getCardNum() {
51          return cardNum;
52      }
53  
54      public int getQuantity() {
55          return quantity;
56      }
57  
58      public DeckCardInfo increaseQuantity() {
59          quantity++;
60          return this;
61      }
62  
63      public String getCardKey() {
64          return setCode + cardNum;
65      }
66  
67      @Override
68      public DeckCardInfo copy() {
69          return new DeckCardInfo(this);
70      }
71  }

15   * @author github: timhae

17  public class MtgjsonDeckImporter extends JsonDeckImporter {
18  
19      @Override
20      protected void readJson(JsonObject json, DeckCardLists deckList) {
21          JsonObject data = JsonUtil.getAsObject(json, "data");
22          if (data == null) {
23              sbMessage.append("Could not find data in json").append("'\n");
24              return;
25          }
26  
27          // info
28          String deckSet = JsonUtil.getAsString(data, "code");
29          String name = JsonUtil.getAsString(data, "name");
30          if (!name.isEmpty()) {
31              deckList.setName(name);
32          }
33  
34          // mainboard
35          JsonArray mainBoard = JsonUtil.getAsArray(data, "mainBoard");
36          List<mage.cards.decks.DeckCardInfo> mainDeckList = deckList.getCards();
37          addBoardToList(mainBoard, mainDeckList, deckSet);
38  
39          // sideboard
40          JsonArray sideBoard = JsonUtil.getAsArray(data, "sideBoard");
41          List<mage.cards.decks.DeckCardInfo> sideDeckList = deckList.getSideboard();
42          addBoardToList(sideBoard, sideDeckList, deckSet);
43      }
44  
45      private void addBoardToList(JsonArray board, List<mage.cards.decks.DeckCardInfo> list, String deckSet) {
46          if (board == null || board.isEmpty()) {
47              return;
48          }
49  
50          board.forEach(arrayCard -> {
51              JsonObject card = (JsonObject) arrayCard;
52              String name = JsonUtil.getAsString(card, "name");
53              String setCode = JsonUtil.getAsString(card, "setCode");
54              if (setCode.isEmpty()) {
55                  setCode = deckSet;
56              }
57  
58              int num = JsonUtil.getAsInt(card, "count");
59              Optional<CardInfo> cardLookup = getCardLookup().lookupCardInfo(name, setCode);
60              if (!cardLookup.isPresent()) {
61                  sbMessage.append("Could not find card: '").append(name).append("'\n");
62              } else {
63                  CardInfo cardInfo = cardLookup.get();
64                  for (int i = 0; i < num; i++) {
65                      list.add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
66                  }
67              }
68          });
69      }
70  }

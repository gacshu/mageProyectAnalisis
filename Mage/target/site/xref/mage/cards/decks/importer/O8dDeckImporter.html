18      /**
19       * @param fileName
20       * @param errorMessages
21       * @param saveAutoFixedFile not supported for current format
22       * @return
23       */
24      @Override
25      public DeckCardLists importDeck(String fileName, StringBuilder errorMessages, boolean saveAutoFixedFile) {
26          try {
27              Document doc = getXmlDocument(fileName);
28              DeckCardLists decklist = new DeckCardLists();
29  
30              List<Node> mainCards = getNodes(doc, "/deck/section[@name='Main']/card");
31              decklist.setCards(mainCards.stream()
32                      .flatMap(toDeckCardInfo(getCardLookup(), errorMessages))
33                      .collect(Collectors.toList()));
34  
35              List<Node> sideboardCards = getNodes(doc, "/deck/section[@name='Sideboard']/card");
36              decklist.setSideboard(sideboardCards.stream()
37                      .flatMap(toDeckCardInfo(getCardLookup(), errorMessages))
38                      .collect(Collectors.toList()));
39  
40              return decklist;
41          } catch (Exception e) {
42              logger.error("Error loading deck", e);
43              errorMessages.append("There was an error loading the deck.");
44              return new DeckCardLists();
45          }
46      }
47  
48      private static int getQuantityFromNode(Node node) {
49          Node numberNode = node.getAttributes().getNamedItem("qty");
50          if (numberNode == null) {
51              return 1;
52          }
53          try {
54              return Math.min(100, Math.max(1, Integer.parseInt(numberNode.getNodeValue())));
55          } catch (NumberFormatException e) {
56              return 1;
57          }
58      }
59  
60      private static Function<Node, Stream<DeckCardInfo>> toDeckCardInfo(CardLookup lookup, StringBuilder errors) {
61          return node -> {
62              String name = node.getTextContent();
63              Optional<CardInfo> cardInfo = lookup.lookupCardInfo(name);
64              if (cardInfo.isPresent()) {
65                  CardInfo info = cardInfo.get();
66                  return Collections.nCopies(
67                          getQuantityFromNode(node),
68                          new DeckCardInfo(info.getName(), info.getCardNumber(), info.getSetCode())).stream();
69              } else {
70                  errors.append("Could not find card: '").append(name).append("'\n");
71                  return Stream.empty();
72              }
73          };
74      }

11  public class DekDeckImporter extends PlainTextDeckImporter {
12  
13      @Override
14      protected void readLine(String line, DeckCardLists deckList, FixedInfo fixedInfo) {
15  
16          if (line.isEmpty() || line.startsWith("#") || !line.contains("<Cards CatID")) {
17              return;
18          }
19          try {
20              // e.g. <Cards CatID="61202" Quantity="1" Sideboard="false" Name="Vildin-Pack Outcast" />
21              Integer cardCount = Integer.parseInt(extractAttribute(line, "Quantity"));
22              String cardName = extractAttribute(line, "Name");
23              boolean isSideboard = "true".equals(extractAttribute(line, "Sideboard"));
24              CardInfo cardInfo = CardRepository.instance.findPreferredCoreExpansionCard(cardName);
25              if (cardInfo == null) {
26                  sbMessage.append("Could not find card: '").append(cardName).append("' at line ").append(lineCount).append('\n');
27              } else {
28                  for (int i = 0; i < cardCount; i++) {
29                      DeckCardInfo deckCardInfo = new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode());
30                      if (isSideboard) {
31                          deckList.getSideboard().add(deckCardInfo);
32                      } else {
33                          deckList.getCards().add(deckCardInfo);
34                      }
35                  }
36              }
37          } catch (NumberFormatException nfe) {
38              sbMessage.append("Invalid number: ").append(extractAttribute(line, "Quantity")).append(" at line ").append(lineCount).append('\n');
39          }
40  
41      }
42  
43      private String extractAttribute(String line, String name) {
44          String searchString = name + "=\"";
45          int startDelim = line.indexOf(searchString) + searchString.length();
46          int endDelim = line.substring(startDelim).indexOf('\"');
47          return line.substring(startDelim, startDelim + endDelim);
48      }

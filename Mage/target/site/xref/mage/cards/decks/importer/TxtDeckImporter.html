4   import mage.cards.decks.DeckCardInfo;
5   import mage.cards.decks.DeckCardLists;
6   import mage.cards.repository.CardInfo;
7   import mage.cards.repository.CardRepository;
8   
9   import java.util.Arrays;
10  import java.util.HashSet;
11  import java.util.Locale;
12  import java.util.Set;
13  
14  /**
15   * @author BetaSteward_at_googlemail.com
16   */
17  public class TxtDeckImporter extends PlainTextDeckImporter {
18  
19      private static final String[] SET_VALUES = new String[]{"lands", "creatures", "planeswalkers", "other spells", "sideboard cards",
20              "Instant", "Land", "Enchantment", "Artifact", "Sorcery", "Planeswalker", "Creature"};
21      private static final Set<String> IGNORE_NAMES = new HashSet<>(Arrays.asList(SET_VALUES));
22  
23      private boolean sideboard = false;
24      private boolean switchSideboardByEmptyLine = true; // all cards after first empty line will be sideboard (like mtgo format)
25      private boolean wasCardLines = false;
26  
27      public TxtDeckImporter(boolean haveSideboardSection) {
28          if (haveSideboardSection) {
29              switchSideboardByEmptyLine = false;
30          }
31      }
32  
33      @Override
34      protected void readLine(String line, DeckCardLists deckList, FixedInfo fixedInfo) {
35  
36          line = line.trim();
37  
38          // process comment:
39          // skip or force to sideboard
40          String commentString = line.toLowerCase(Locale.ENGLISH);
41          if (commentString.startsWith("//")) {
42              // use start, not contains (card names may contain commands like "Legerdemain")
43  
44              if (commentString.startsWith("//sideboard")) {
45                  sideboard = true;
46              }
47  
48              // skip comment line
49              return;
50          }
51  
52          // remove inner card comments from text line: 2 Blinding Fog #some text (like deckstats format)
53          int commentDelim = line.indexOf('#');
54          if (commentDelim >= 0) {
55              line = line.substring(0, commentDelim).trim();
56          }
57  
58          // ignore all empty lines until real cards starts
59          if (line.isEmpty() && !wasCardLines) {
60              return;
61          }
62  
63          // switch sideboard by empty line
64          if (switchSideboardByEmptyLine && line.isEmpty()) {
65              if (!sideboard) {
66                  sideboard = true;
67              } else {
68                  sbMessage.append("Found empty line at ").append(lineCount).append(", but sideboard already used. Use //sideboard switcher OR use only one empty line to devide your cards.").append('\n');
69              }
70  
71              // skip empty line
72              return;
73          }
74  
75          // single line sideboard card from deckstats.net
76          // SB: 3 Carnage Tyrant
77          boolean singleLineSideBoard = false;
78          if (line.startsWith("SB:")) {
79              line = line.replace("SB:", "").trim();
80              singleLineSideBoard = true;
81          }
82  
83          line = line.replace("\t", " "); // changing tabs to blanks as delimiter
84          int delim = line.indexOf(' ');
85          String lineNum = "";
86          if (delim > 0) {
87              lineNum = line.substring(0, delim).trim();
88              if (IGNORE_NAMES.contains(lineNum)) {
89                  return;
90              }
91          }
92  
93          // amount
94          int cardAmount = 0;
95          boolean haveCardAmout = false;
96          if (!lineNum.isEmpty()) {
97              try {
98                  cardAmount = Integer.parseInt(lineNum.replaceAll("\\D+", ""));
99                  if ((cardAmount <= 0) || (cardAmount >= 100)) {
100                     sbMessage.append("Invalid number (too small or too big): ").append(lineNum).append(" at line ").append(lineCount).append('\n');
101                     return;
102                 }
103                 haveCardAmout = true;
104             } catch (NumberFormatException nfe) {
105                 // card without amount
106             }
107         }
108 
109         String lineName;
110         if (haveCardAmout) {
111             lineName = line.substring(delim).trim();
112         } else {
113             lineName = line.trim();
114             cardAmount = 1;
115         }
116 
117         lineName = CardNameUtil.normalizeCardName(lineName);
118         if (lineName.contains("//") && !lineName.contains(" // ")) {
119             lineName = lineName.replace("//", " // ");
120         }
121         lineName = lineName.replaceFirst("(?<=[^/])\\s*/\\s*(?=[^/])", " // ");
122 
123         if (IGNORE_NAMES.contains(lineName)) {
124             return;
125         }
126 
127         wasCardLines = true;
128 
129         CardInfo cardInfo = CardRepository.instance.findPreferredCoreExpansionCard(lineName);
130         if (cardInfo == null) {
131             sbMessage.append("Could not find card: '").append(lineName).append("' at line ").append(lineCount).append('\n');
132         } else {
133             for (int i = 0; i < cardAmount; i++) {
134                 if (!sideboard && !singleLineSideBoard) {
135                     deckList.getCards().add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
136                 } else {
137                     deckList.getSideboard().add(new DeckCardInfo(cardInfo.getName(), cardInfo.getCardNumber(), cardInfo.getSetCode()));
138                 }
139             }
140         }
141     }

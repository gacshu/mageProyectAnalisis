22      @Override
23      public void writeDeck(PrintWriter out, DeckCardLists deck) {
24          List<DeckCardInfo> deckMain = new ArrayList<>();
25          List<DeckCardInfo> deckSideboard = new ArrayList<>();
26          Map<String, Integer> amount = new HashMap<>();
27  
28          // info
29          if (deck.getName() != null && !deck.getName().isEmpty()) {
30              out.println("NAME:" + deck.getName());
31          }
32          if (deck.getAuthor() != null && !deck.getAuthor().isEmpty()) {
33              out.println("AUTHOR:" + deck.getAuthor());
34          }
35  
36          // main
37          for (DeckCardInfo card : deck.getCards()) {
38              String code = "M@" + card.getCardKey();
39              int curAmount = amount.getOrDefault(code, 0);
40              if (curAmount == 0) {
41                  deckMain.add(card);
42              }
43              amount.put(code, curAmount + card.getQuantity());
44          }
45          // sideboard
46          for (DeckCardInfo card : deck.getSideboard()) {
47              String code = "S@" + card.getCardKey();
48              int curAmount = amount.getOrDefault(code, 0);
49              if (curAmount == 0) {
50                  deckSideboard.add(card);
51              }
52              amount.put(code, curAmount + card.getQuantity());
53          }
54  
55          // cards print
56          for (DeckCardInfo card : deckMain) {
57              out.printf("%d [%s:%s] %s%n", amount.get("M@" + card.getCardKey()), card.getSetCode(), card.getCardNum(), card.getCardName());
58          }
59          for (DeckCardInfo card : deckSideboard) {
60              out.printf("SB: %d [%s:%s] %s%n", amount.get("S@" + card.getCardKey()), card.getSetCode(), card.getCardNum(), card.getCardName());
61          }
62  
63          // layout print
64          if (deck.getCardLayout() != null) {
65              out.print("LAYOUT MAIN:");
66              writeCardLayout(out, deck.getCardLayout());
67              out.println("");
68              out.print("LAYOUT SIDEBOARD:");
69              writeCardLayout(out, deck.getSideboardLayout());
70              out.println("");
71          }
72      }
73  
74      private static void writeCardLayout(PrintWriter out, DeckCardLayout layout) {
75          if (layout == null) {
76              return;
77          }
78          List<List<List<DeckCardInfo>>> cardGrid = layout.getCards();
79          int height = cardGrid.size();
80          int width = (height > 0) ? cardGrid.get(0).size() : 0;
81          out.print("(" + height + ',' + width + ')');
82          out.print(layout.getSettings());
83          out.print("|");
84          for (List<List<DeckCardInfo>> row : cardGrid) {
85              for (List<DeckCardInfo> stack : row) {
86                  out.print("(");
87                  for (int i = 0; i < stack.size(); ++i) {
88                      DeckCardInfo info = stack.get(i);
89                      out.printf("[%s:%s]", info.getSetCode(), info.getCardNum());
90                      if (i != stack.size() - 1) {
91                          out.print(",");
92                      }
93                  }
94                  out.print(")");
95              }
96          }
97      }

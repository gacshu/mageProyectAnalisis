21  public abstract class SplitCard extends CardImpl implements CardWithHalves {

23      protected Card leftHalfCard;
24      protected Card rightHalfCard;

26      public SplitCard(UUID ownerId, CardSetInfo setInfo, CardType[] cardTypes, String costsLeft, String costsRight, SpellAbilityType spellAbilityType) {
27          this(ownerId, setInfo, cardTypes, cardTypes, costsLeft, costsRight, spellAbilityType);
28      }

30      public SplitCard(UUID ownerId, CardSetInfo setInfo, CardType[] typesLeft, CardType[] typesRight, String costsLeft, String costsRight, SpellAbilityType spellAbilityType) {
31          super(ownerId, setInfo, CardType.mergeTypes(typesLeft, typesRight), costsLeft + costsRight, spellAbilityType);
32          String[] names = setInfo.getName().split(" // ");
33          leftHalfCard = new SplitCardHalfImpl(this.getOwnerId(), new CardSetInfo(names[0], setInfo.getExpansionSetCode(), setInfo.getCardNumber(), setInfo.getRarity(), setInfo.getGraphicInfo()), typesLeft, costsLeft, this, SpellAbilityType.SPLIT_LEFT);
34          rightHalfCard = new SplitCardHalfImpl(this.getOwnerId(), new CardSetInfo(names[1], setInfo.getExpansionSetCode(), setInfo.getCardNumber(), setInfo.getRarity(), setInfo.getGraphicInfo()), typesRight, costsRight, this, SpellAbilityType.SPLIT_RIGHT);
35      }
36  
37      public SplitCard(SplitCard card) {
38          super(card);
39          this.leftHalfCard = card.getLeftHalfCard().copy();
40          ((SplitCardHalf) leftHalfCard).setParentCard(this);
41          this.rightHalfCard = card.rightHalfCard.copy();
42          ((SplitCardHalf) rightHalfCard).setParentCard(this);
43      }
44  
45      public void setParts(SplitCardHalf leftHalfCard, SplitCardHalf rightHalfCard) {
46          // for card copy only - set new parts
47          this.leftHalfCard = leftHalfCard;
48          leftHalfCard.setParentCard(this);
49          this.rightHalfCard = rightHalfCard;
50          rightHalfCard.setParentCard(this);
51      }

69      public void setCopy(boolean isCopy, MageObject copiedFrom) {

73      }

75      @Override
76      public boolean moveToZone(Zone toZone, Ability source, Game game, boolean flag, List<UUID> appliedEffects) {
77          if (super.moveToZone(toZone, source, game, flag, appliedEffects)) {
78              game.getState().setZone(getLeftHalfCard().getId(), toZone);
79              game.getState().setZone(getRightHalfCard().getId(), toZone);
80              return true;
81          }
82          return false;
83      }
84  
85      @Override
86      public void setZone(Zone zone, Game game) {
87          super.setZone(zone, game);
88          game.setZone(getLeftHalfCard().getId(), zone);
89          game.setZone(getRightHalfCard().getId(), zone);
90      }
91  
92      @Override
93      public boolean moveToExile(UUID exileId, String name, Ability source, Game game, List<UUID> appliedEffects) {
94          if (super.moveToExile(exileId, name, source, game, appliedEffects)) {
95              Zone currentZone = game.getState().getZone(getId());
96              game.getState().setZone(getLeftHalfCard().getId(), currentZone);
97              game.getState().setZone(getRightHalfCard().getId(), currentZone);
98              return true;
99          }
100         return false;
101     }

103     @Override
104     public boolean removeFromZone(Game game, Zone fromZone, Ability source) {
105         // zone contains only one main card
106         return super.removeFromZone(game, fromZone, source);
107     }
108 
109     @Override
110     public void updateZoneChangeCounter(Game game, ZoneChangeEvent event) {
111         if (isCopy()) { // same as meld cards
112             super.updateZoneChangeCounter(game, event);
113             return;
114         }
115         super.updateZoneChangeCounter(game, event);
116         getLeftHalfCard().updateZoneChangeCounter(game, event);
117         getRightHalfCard().updateZoneChangeCounter(game, event);
118     }
119 
120     @Override
121     public boolean cast(Game game, Zone fromZone, SpellAbility ability, UUID controllerId) {
122         switch (ability.getSpellAbilityType()) {
123             case SPLIT_LEFT:
124                 return this.getLeftHalfCard().cast(game, fromZone, ability, controllerId);
125             case SPLIT_RIGHT:
126                 return this.getRightHalfCard().cast(game, fromZone, ability, controllerId);
127             default:
128                 this.getLeftHalfCard().getSpellAbility().setControllerId(controllerId);
129                 this.getRightHalfCard().getSpellAbility().setControllerId(controllerId);
130                 return super.cast(game, fromZone, ability, controllerId);
131         }
132     }

134     @Override
135     public Abilities<Ability> getAbilities() {
136         Abilities<Ability> allAbilites = new AbilitiesImpl<>();
137         for (Ability ability : super.getAbilities()) {
138             // ignore split abilities
139             // TODO: why it here, for GUI's cleanup in card texts? Maybe it can be removed, see mdf cards
140             if (ability instanceof SpellAbility
141                     && (((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.SPLIT
142                     || ((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.SPLIT_AFTERMATH)) {
143                 continue;
144             }
145             allAbilites.add(ability);
146         }
147         allAbilites.addAll(leftHalfCard.getAbilities());
148         allAbilites.addAll(rightHalfCard.getAbilities());
149         return allAbilites;
150     }

170     public Abilities<Ability> getAbilities(Game game) {
171         Abilities<Ability> allAbilites = new AbilitiesImpl<>();
172         for (Ability ability : super.getAbilities(game)) {
173             // ignore split abilities
174             // TODO: why it here, for GUI's cleanup in card texts? Maybe it can be removed, see mdf cards
175             if (ability instanceof SpellAbility
176                     && (((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.SPLIT
177                     || ((SpellAbility) ability).getSpellAbilityType() == SpellAbilityType.SPLIT_AFTERMATH)) {
178                 continue;
179             }
180             allAbilites.add(ability);

182         allAbilites.addAll(leftHalfCard.getAbilities(game));
183         allAbilites.addAll(rightHalfCard.getAbilities(game));
184         return allAbilites;

187     @Override
188     public List<String> getRules() {
189         List<String> rules = new ArrayList<>();
190         if (getSpellAbility().getSpellAbilityType() == SpellAbilityType.SPLIT_FUSED) {
191             rules.add("--------------------------------------------------------------------------\nFuse (You may cast one or both halves of this card from your hand.)");
192         }
193         return rules;
194     }

198         super.setOwnerId(ownerId);

200         leftHalfCard.getAbilities().setControllerId(ownerId);
201         leftHalfCard.setOwnerId(ownerId);
202         rightHalfCard.getAbilities().setControllerId(ownerId);
203         rightHalfCard.setOwnerId(ownerId);

206     @Override
207     public int getManaValue() {
208         // 202.3d The converted mana cost of a split card not on the stack or of a fused split spell on the
209         // stack is determined from the combined mana costs of its halves. Otherwise, while a split card is
210         // on the stack, the converted mana cost of the spell is determined by the mana cost of the half
211         // that was chosen to be cast. See rule 708, “Split Cards.”
212 
213         // split card and it's halfes contains own mana costs, so no need to rewrite logic
214         return super.getManaValue();
215     }

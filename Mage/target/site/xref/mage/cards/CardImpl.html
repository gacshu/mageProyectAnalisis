63      public CardImpl(UUID ownerId, CardSetInfo setInfo, CardType[] cardTypes, String costs, SpellAbilityType spellAbilityType) {
64          this(ownerId, setInfo.getName());
65          this.rarity = setInfo.getRarity();
66          this.cardNumber = setInfo.getCardNumber();
67          this.expansionSetCode = setInfo.getExpansionSetCode();
68          this.cardType.addAll(Arrays.asList(cardTypes));
69          this.manaCost.load(costs);
70          setDefaultColor();
71          if (this.isLand()) {
72              Ability ability = new PlayLandAbility(name);
73              ability.setSourceId(this.getId());
74              abilities.add(ability);
75          } else {
76              SpellAbility ability = new SpellAbility(manaCost, name, Zone.HAND, spellAbilityType);
77              if (!this.isInstant()) {
78                  ability.setTiming(TimingRule.SORCERY);
79              }
80              ability.setSourceId(this.getId());
81              abilities.add(ability);
82          }
83  
84          CardGraphicInfo graphicInfo = setInfo.getGraphicInfo();
85          if (graphicInfo != null) {
86              this.usesVariousArt = graphicInfo.getUsesVariousArt();
87              if (graphicInfo.getFrameColor() != null) {
88                  this.frameColor = graphicInfo.getFrameColor().copy();
89              }
90              if (graphicInfo.getFrameStyle() != null) {
91                  this.frameStyle = graphicInfo.getFrameStyle();
92              }
93          }
94  
95          this.morphCard = false;
96      }

168     public static Card createCard(Class<?> clazz, CardSetInfo setInfo, List<String> errorList) {
169         String setCode = null;
170         try {
171             Card card;
172             if (setInfo == null) {
173                 Constructor<?> con = clazz.getConstructor(UUID.class);
174                 card = (Card) con.newInstance(new Object[]{null});
175             } else {
176                 setCode = setInfo.getExpansionSetCode();
177                 Constructor<?> con = clazz.getConstructor(UUID.class, CardSetInfo.class);
178                 card = (Card) con.newInstance(null, setInfo);
179             }
180             return card;
181         } catch (Exception e) {
182             String err = "Error loading card: " + clazz.getCanonicalName() + " (" + setCode + ")";
183             if (errorList != null) {
184                 errorList.add(err);
185             }
186 
187             if (e instanceof InvocationTargetException) {
188                 logger.fatal(err, ((InvocationTargetException) e).getTargetException());
189             } else {
190                 logger.fatal(err, e);
191             }
192 
193             return null;
194         }
195     }

240     /**
241      * Gets all current abilities - includes additional abilities added by other
242      * cards or effects. Warning, you can't modify that list.
243      *
244      * @param game
245      * @return A list of {@link Ability} - this collection is not modifiable
246      */
247     @Override
248     public Abilities<Ability> getAbilities(Game game) {
249         if (game == null) {
250             return abilities; // deck editor with empty game
251         }
252 
253         CardState cardState = game.getState().getCardState(this.getId());
254         if (cardState == null) {
255             return abilities;
256         }
257 
258         // collects all abilities
259         Abilities<Ability> all = new AbilitiesImpl<>();
260 
261         // basic
262         if (!cardState.hasLostAllAbilities()) {
263             all.addAll(abilities);
264         }
265 
266         // dynamic
267         all.addAll(cardState.getAbilities());
268 
269         // workaround to add dynamic flashback ability from main card to all parts (example: Snapcaster Mage gives flashback to split card)
270         if (!this.getId().equals(this.getMainCard().getId())) {
271             CardState mainCardState = game.getState().getCardState(this.getMainCard().getId());
272             if (this.getSpellAbility() != null // lands can't be casted (haven't spell ability), so ignore it
273                     && mainCardState != null
274                     && !mainCardState.hasLostAllAbilities()
275                     && mainCardState.getAbilities().containsClass(FlashbackAbility.class)) {
276                 FlashbackAbility flash = new FlashbackAbility(this, this.getManaCost());
277                 flash.setSourceId(this.getId());
278                 flash.setControllerId(this.getOwnerId());
279                 flash.setSpellAbilityType(this.getSpellAbility().getSpellAbilityType());
280                 flash.setAbilityName(this.getName());
281                 all.add(flash);
282             }
283         }
284 
285         return all;
286     }

306     @Override
307     public void addAbility(Ability ability) {
308         ability.setSourceId(this.getId());
309         abilities.add(ability);
310         abilities.addAll(ability.getSubAbilities());
311 
312         // dynamic check: you can't add ability to the PermanentCard, use permanent.addAbility(a, source, game) instead
313         // reason: triggered abilities are not processing here
314         if (this instanceof PermanentCard) {
315             throw new IllegalArgumentException("Wrong code usage. Don't use that method for permanents, use permanent.addAbility(a, source, game) instead.");
316         }
317 
318         // verify check: draw effect can't be rollback after mana usage (example: Chromatic Sphere)
319         // (player can cheat with cancel button to see next card)
320         // verify test will catch that errors
321         if (ability instanceof ActivatedManaAbilityImpl) {
322             ActivatedManaAbilityImpl manaAbility = (ActivatedManaAbilityImpl) ability;
323             String rule = manaAbility.getRule().toLowerCase(Locale.ENGLISH);
324             if (manaAbility.getEffects().stream().anyMatch(e -> e.getOutcome().equals(Outcome.DrawCard))
325                     || rule.contains("reveal ")
326                     || rule.contains("draw ")) {
327                 if (manaAbility.isUndoPossible()) {
328                     throw new IllegalArgumentException("Ability contains draw/reveal effect, but isUndoPossible is true. Ability: "
329                             + ability.getClass().getSimpleName() + "; " + ability.getRule());
330                 }
331             }
332         }
333     }

340     public void replaceSpellAbility(SpellAbility newAbility) {
341         SpellAbility oldAbility = this.getSpellAbility();
342         while (oldAbility != null) {
343             abilities.remove(oldAbility);
344             spellAbility = null;
345             oldAbility = this.getSpellAbility();
346         }
347 
348         if (newAbility != null) {
349             addAbility(newAbility);
350         }
351     }

392     public boolean moveToZone(Zone toZone, Ability source, Game game, boolean flag, List<UUID> appliedEffects) {
393         Zone fromZone = game.getState().getZone(objectId);
394         ZoneChangeEvent event = new ZoneChangeEvent(this.objectId, source, ownerId, fromZone, toZone, appliedEffects);
395         ZoneChangeInfo zoneChangeInfo;
396         if (null != toZone) {
397             switch (toZone) {
398                 case LIBRARY:
399                     zoneChangeInfo = new ZoneChangeInfo.Library(event, flag /* put on top */);
400                     break;
401                 case BATTLEFIELD:
402                     zoneChangeInfo = new ZoneChangeInfo.Battlefield(event, flag /* comes into play tapped */, source);
403                     break;
404                 default:
405                     zoneChangeInfo = new ZoneChangeInfo(event);
406                     break;
407             }
408             return ZonesHandler.moveCard(zoneChangeInfo, game, source);
409         }
410         return false;
411     }

457     @Override
458     public boolean removeFromZone(Game game, Zone fromZone, Ability source) {
459         boolean removed = false;
460         MageObject lkiObject = null;
461         switch (fromZone) {
462             case GRAVEYARD:
463                 removed = game.getPlayer(ownerId).removeFromGraveyard(this, game);
464                 break;
465             case HAND:
466                 removed = game.getPlayer(ownerId).removeFromHand(this, game);
467                 break;
468             case LIBRARY:
469                 removed = game.getPlayer(ownerId).removeFromLibrary(this, game);
470                 break;
471             case EXILED:
472                 if (game.getExile().getCard(getId(), game) != null) {
473                     removed = game.getExile().removeCard(this, game);
474                 }
475                 break;
476             case STACK:
477                 StackObject stackObject;
478                 if (getSpellAbility() != null) {
479                     stackObject = game.getStack().getSpell(getSpellAbility().getId(), false);
480                 } else {
481                     stackObject = game.getStack().getSpell(this.getId(), false);
482                 }
483 
484                 // handle half of Split Cards on stack
485                 if (stackObject == null && (this instanceof SplitCard)) {
486                     stackObject = game.getStack().getSpell(((SplitCard) this).getLeftHalfCard().getId(), false);
487                     if (stackObject == null) {
488                         stackObject = game.getStack().getSpell(((SplitCard) this).getRightHalfCard().getId(), false);
489                     }
490                 }
491 
492                 // handle half of Modal Double Faces Cards on stack
493                 if (stackObject == null && (this instanceof ModalDoubleFacesCard)) {
494                     stackObject = game.getStack().getSpell(((ModalDoubleFacesCard) this).getLeftHalfCard().getId(), false);
495                     if (stackObject == null) {
496                         stackObject = game.getStack().getSpell(((ModalDoubleFacesCard) this).getRightHalfCard().getId(), false);
497                     }
498                 }
499 
500                 if (stackObject == null && (this instanceof AdventureCard)) {
501                     stackObject = game.getStack().getSpell(((AdventureCard) this).getSpellCard().getId(), false);
502                 }
503 
504                 if (stackObject == null) {
505                     stackObject = game.getStack().getSpell(getId(), false);
506                 }
507                 if (stackObject != null) {
508                     removed = game.getStack().remove(stackObject, game);
509                     lkiObject = stackObject;
510                 }
511                 break;
512             case COMMAND:
513                 for (CommandObject commandObject : game.getState().getCommand()) {
514                     if (commandObject.getId().equals(objectId)) {
515                         lkiObject = commandObject;
516                     }
517                 }
518                 if (lkiObject != null) {
519                     removed = game.getState().getCommand().remove(lkiObject);
520                 }
521                 break;
522             case OUTSIDE:
523                 if (isCopy()) { // copied cards have no need to be removed from a previous zone
524                     removed = true;
525                 } else if (game.getPlayer(ownerId).getSideboard().contains(this.getId())) {
526                     game.getPlayer(ownerId).getSideboard().remove(this.getId());
527                     removed = true;
528                 } else if (game.getPhase() == null) {
529                     // E.g. Commander of commander game
530                     removed = true;
531                 } else {
532                     // Unstable - Summon the Pack
533                     removed = true;
534                 }
535                 break;
536             case BATTLEFIELD: // for sacrificing permanents or putting to library
537                 removed = true;
538                 break;
539             default:
540                 MageObject sourceObject = game.getObject(source);
541                 logger.fatal("Invalid from zone [" + fromZone + "] for card [" + this.getIdName()
542                         + "] source [" + (sourceObject != null ? sourceObject.getName() : "null") + ']');
543                 break;
544         }
545         if (removed) {
546             if (fromZone != Zone.OUTSIDE) {
547                 game.rememberLKI(lkiObject != null ? lkiObject.getId() : objectId, fromZone, lkiObject != null ? lkiObject : this);
548             }
549         } else {
550             logger.warn("Couldn't find card in fromZone, card=" + getIdName() + ", fromZone=" + fromZone);
551             // possible reason: you to remove card from wrong zone or card already removed,
552             // e.g. you added copy card to wrong graveyard (see owner) or removed card from graveyard before moveToZone call
553         }
554         return removed;
555     }

578     @Override
579     public boolean turnFaceUp(Ability source, Game game, UUID playerId) {
580         GameEvent event = GameEvent.getEvent(GameEvent.EventType.TURNFACEUP, getId(), source, playerId);
581         if (!game.replaceEvent(event)) {
582             setFaceDown(false, game);
583             for (Ability ability : abilities) { // abilities that were set to not visible face down must be set to visible again
584                 if (ability.getWorksFaceDown() && !ability.getRuleVisible()) {
585                     ability.setRuleVisible(true);
586                 }
587             }
588             game.fireEvent(GameEvent.getEvent(GameEvent.EventType.TURNEDFACEUP, getId(), source, playerId));
589             return true;
590         }
591         return false;
592     }

628     private Card initSecondSideCard(Class<? extends Card> cardClazz) {
629         // must be non strict search in any sets, not one set
630         // example: if set contains only one card side
631         // method used in cards database creating, so can't use repository here
632         ExpansionSet.SetCardInfo info = Sets.findCardByClass(cardClazz, expansionSetCode, cardNumber);
633         if (info == null) {
634             return null;
635         }
636         return createCard(cardClazz, new CardSetInfo(info.getName(), expansionSetCode, info.getCardNumber(), info.getRarity(), info.getGraphicInfo()));
637     }

734     public boolean addCounters(Counter counter, UUID playerAddingCounters, Ability source, Game game, List<UUID> appliedEffects, boolean isEffect, int maxCounters) {
735         if (this instanceof Permanent) {
736             if (!((Permanent) this).isPhasedIn()) {
737                 return false;
738             }
739         }
740 
741         boolean returnCode = true;
742         GameEvent addingAllEvent = GameEvent.getEvent(GameEvent.EventType.ADD_COUNTERS, objectId, source, playerAddingCounters, counter.getName(), counter.getCount());
743         addingAllEvent.setAppliedEffects(appliedEffects);
744         addingAllEvent.setFlag(isEffect);
745         if (!game.replaceEvent(addingAllEvent)) {
746             int amount;
747             if (maxCounters < Integer.MAX_VALUE) {
748                 amount = Integer.min(addingAllEvent.getAmount(), maxCounters - this.getCounters(game).getCount(counter.getName()));
749             } else {
750                 amount = addingAllEvent.getAmount();
751             }
752             boolean isEffectFlag = addingAllEvent.getFlag();
753             int finalAmount = amount;
754             for (int i = 0; i < amount; i++) {
755                 Counter eventCounter = counter.copy();
756                 eventCounter.remove(eventCounter.getCount() - 1);
757                 GameEvent addingOneEvent = GameEvent.getEvent(GameEvent.EventType.ADD_COUNTER, objectId, source, playerAddingCounters, counter.getName(), 1);
758                 addingOneEvent.setAppliedEffects(appliedEffects);
759                 addingOneEvent.setFlag(isEffectFlag);
760                 if (!game.replaceEvent(addingOneEvent)) {
761                     getCounters(game).addCounter(eventCounter);
762                     GameEvent addedOneEvent = GameEvent.getEvent(GameEvent.EventType.COUNTER_ADDED, objectId, source, playerAddingCounters, counter.getName(), 1);
763                     addedOneEvent.setFlag(addingOneEvent.getFlag());
764                     game.fireEvent(addedOneEvent);
765                 } else {
766                     finalAmount--;
767                     returnCode = false;
768                 }
769             }
770             if (finalAmount > 0) {
771                 GameEvent addedAllEvent = GameEvent.getEvent(GameEvent.EventType.COUNTERS_ADDED, objectId, source, playerAddingCounters, counter.getName(), amount);
772                 addedAllEvent.setFlag(isEffectFlag);
773                 game.fireEvent(addedAllEvent);
774             }
775         } else {
776             returnCode = false;
777         }
778         return returnCode;
779     }

781     @Override
782     public void removeCounters(String name, int amount, Ability source, Game game) {
783         int finalAmount = 0;
784         for (int i = 0; i < amount; i++) {
785             if (!getCounters(game).removeCounter(name, 1)) {
786                 break;
787             }
788             GameEvent event = GameEvent.getEvent(GameEvent.EventType.COUNTER_REMOVED, objectId, source, getControllerOrOwner());
789             if (source != null
790                     && source.getControllerId() != null) {
791                 event.setPlayerId(source.getControllerId()); // player who controls the source ability that removed the counter
792             }
793             event.setData(name);
794             game.fireEvent(event);
795             finalAmount++;
796         }
797         GameEvent event = GameEvent.getEvent(GameEvent.EventType.COUNTERS_REMOVED, objectId, source, getControllerOrOwner());
798         if (source != null
799                 && source.getControllerId() != null) {
800             event.setPlayerId(source.getControllerId()); // player who controls the source ability that removed the counter
801         }
802         event.setData(name);
803         event.setAmount(finalAmount);
804         game.fireEvent(event);
805     }

848     @Override
849     public boolean addAttachment(UUID permanentId, Ability source, Game game) {
850         if (permanentId == null
851                 || this.attachments.contains(permanentId)
852                 || permanentId.equals(this.getId())) {
853             return false;
854         }
855         Permanent attachment = game.getPermanent(permanentId);
856         if (attachment == null) {
857             attachment = game.getPermanentEntering(permanentId);
858         }
859         if (attachment == null) {
860             return false;
861         }
862         if (attachment.hasSubtype(SubType.EQUIPMENT, game)
863                 && (attachment.isCreature(game)  // seems strange and perhaps someone knows why this is checked.
864                 && !attachment.getAbilities(game).containsClass(ReconfigureAbility.class))) {
865             return false;
866         }
867         if (attachment.hasSubtype(SubType.FORTIFICATION, game)
868                 && (attachment.isCreature(game) || !this.isLand(game))) {
869             return false;
870         }
871         if (game.replaceEvent(new AttachEvent(objectId, attachment, source))) {
872             return false;
873         }
874         this.attachments.add(permanentId);
875         attachment.attachTo(objectId, source, game);
876         game.fireEvent(new AttachedEvent(objectId, attachment, source));
877         return true;
878     }

880     @Override
881     public boolean removeAttachment(UUID permanentId, Ability source, Game game) {
882         if (this.attachments.contains(permanentId)) {
883             Permanent attachment = game.getPermanent(permanentId);
884             if (attachment != null) {
885                 attachment.unattach(game);
886             }
887             if (!game.replaceEvent(new UnattachEvent(objectId, permanentId, attachment, source))) {
888                 this.attachments.remove(permanentId);
889                 game.fireEvent(new UnattachedEvent(objectId, permanentId, attachment, source));
890                 return true;
891             }
892         }
893         return false;
894     }

906     @Override
907     public boolean hasSubTypeForDeckbuilding(SubType subType) {
908         // own subtype
909         if (this.hasSubtype(subType, null)) {
910             return true;
911         }
912 
913         // gained subtypes from source ability
914         if (this.getAbilities()
915                 .stream()
916                 .filter(SimpleStaticAbility.class::isInstance)
917                 .map(Ability::getAllEffects)
918                 .flatMap(Collection::stream)
919                 .filter(HasSubtypesSourceEffect.class::isInstance)
920                 .map(HasSubtypesSourceEffect.class::cast)
921                 .anyMatch(effect -> effect.hasSubtype(subType))) {
922             return true;
923         }
924 
925         // changeling (any subtype)
926         return subType.getSubTypeSet() == SubTypeSet.CreatureType
927                 && this.getAbilities().containsClass(ChangelingAbility.class);
928     }

930     /**
931      * This is used for disabling auto-payments for any any cards which care about the color
932      * of the mana used to cast it beyond color requirements. E.g. Sunburst, Adamant, Flamespout.
933      * <p>
934      * This is <b>not</b> about which colors are in the mana costs.
935      * <p>
936      * E.g. "Pentad Prism" {2} will return true since it has Sunburst, but "Abbey Griffin" {3}{W} will
937      * return false since the mana spent on the generic cost has no impact on the card.
938      *
939      * @return Whether the given spell cares about the mana color used to pay for it.
940      */
941     public boolean caresAboutManaColor(Game game) {
942         // SunburstAbility
943         if (abilities.containsClass(SunburstAbility.class)) {
944             return true;
945         }
946 
947         // Look at each individual ability
948         //      ConditionalInterveningIfTriggeredAbility (e.g. Ogre Savant)
949         //      Spellability with ManaWasSpentCondition (e.g. Firespout)
950         //      Modular (only Arcbound Wanderer)
951         for (Ability ability : getAbilities(game)) {
952             if (((AbilityImpl) ability).caresAboutManaColor()) {
953                 return true;
954             }
955         }
956 
957         // Only way to get here is if none of the effects on the card care about mana color.
958         return false;
959     }

3   import mage.MageObject;
4   import mage.constants.CardType;
5   import mage.constants.SubType;
6   import mage.filter.predicate.Predicate;
7   import mage.filter.predicate.Predicates;
8   
9   import java.util.Arrays;
10  import java.util.List;
11  import java.util.Objects;
12  import java.util.stream.Collectors;
13  
14  /**
15   * If you add predicate to permanentFilter then it will be applied to planeswalker too
16   *
17   * @author JRHerlehy Created on 4/8/18.
18   */
19  public class FilterCreaturePlayerOrPlaneswalker extends FilterPermanentOrPlayer {
20  
21      public FilterCreaturePlayerOrPlaneswalker() {
22          this("any target");
23      }
24  
25      public FilterCreaturePlayerOrPlaneswalker(String name) {
26          this(name, (SubType) null);
27      }
28  
29      public FilterCreaturePlayerOrPlaneswalker(String name, SubType... andCreatureTypes) {
30          super(name);
31          List<Predicate<MageObject>> allCreaturePredicates = Arrays.stream(andCreatureTypes)
32                  .filter(Objects::nonNull)
33                  .map(SubType::getPredicate)
34                  .collect(Collectors.toList());
35          allCreaturePredicates.add(0, CardType.CREATURE.getPredicate());
36          Predicate<MageObject> planeswalkerPredicate = CardType.PLANESWALKER.getPredicate();
37  
38          this.permanentFilter.add(Predicates.or(
39                  Predicates.and(allCreaturePredicates),
40                  planeswalkerPredicate
41          ));
42      }
43  
44      public FilterCreaturePlayerOrPlaneswalker(final FilterCreaturePlayerOrPlaneswalker filter) {
45          super(filter);
46      }
47  
48      @Override
49      public FilterCreaturePlayerOrPlaneswalker copy() {
50          return new FilterCreaturePlayerOrPlaneswalker(this);
51      }
52  }

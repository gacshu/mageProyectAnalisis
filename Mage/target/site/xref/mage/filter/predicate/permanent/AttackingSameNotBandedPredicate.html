2   package mage.filter.predicate.permanent;
3   
4   import mage.filter.predicate.Predicate;
5   import mage.game.Game;
6   import mage.game.combat.CombatGroup;
7   import mage.game.permanent.Permanent;
8   
9   import java.util.UUID;
10  
11  /**
12   *
13   * @author L_J
14   */
15  public class AttackingSameNotBandedPredicate implements Predicate<Permanent> {
16  
17      private final UUID defenderId;
18  
19      public AttackingSameNotBandedPredicate(UUID defenderId) {
20          this.defenderId = defenderId;
21      }
22  
23      @Override
24      public boolean apply(Permanent input, Game game) {
25          CombatGroup combatGroup = game.getCombat().findGroup(input.getId());
26          return combatGroup != null
27                      && input.isAttacking()
28                      && input.getBandedCards().isEmpty()
29                      && combatGroup.getDefenderId().equals(defenderId);
30      }
31  }

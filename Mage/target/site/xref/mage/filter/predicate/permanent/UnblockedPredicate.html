2   package mage.filter.predicate.permanent;
3   
4   import mage.constants.PhaseStep;
5   import mage.filter.predicate.Predicate;
6   import mage.game.Game;
7   import mage.game.combat.CombatGroup;
8   import mage.game.permanent.Permanent;
9   import mage.game.turn.Step;
10  
11  /**
12   * Checks if an attacking creature is unblocked after the declare blockers step.
13   *
14   * @author LevelX2
15   */
16  public enum UnblockedPredicate implements Predicate<Permanent> {
17      instance;
18  
19      @Override
20      public boolean apply(Permanent input, Game game) {
21          if (input.isAttacking()) {
22              if ((game.getPhase().getStep().getType() == PhaseStep.DECLARE_BLOCKERS
23                      && game.getStep().getStepPart() == Step.StepPart.PRIORITY)
24                      || game.getPhase().getStep().getType() == PhaseStep.FIRST_COMBAT_DAMAGE
25                      || game.getPhase().getStep().getType() == PhaseStep.COMBAT_DAMAGE
26                      || game.getPhase().getStep().getType() == PhaseStep.END_COMBAT) {
27                  CombatGroup combatGroup = game.getCombat().findGroup(input.getId());
28                  if (combatGroup != null) {
29                      return !combatGroup.getBlocked();
30                  }
31              }
32          }
33          return false;
34      }
35  
36      @Override
37      public String toString() {
38          return "Unblocked";
39      }
40  }

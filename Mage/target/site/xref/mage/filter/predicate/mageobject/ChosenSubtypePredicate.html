1   package mage.filter.predicate.mageobject;
2   
3   import mage.MageObject;
4   import mage.abilities.effects.common.ChooseCreatureTypeEffect;
5   import mage.constants.SubType;
6   import mage.filter.predicate.ObjectSourcePlayer;
7   import mage.filter.predicate.ObjectSourcePlayerPredicate;
8   import mage.game.Game;
9   
10  /**
11   * Warning, chosen type assign to original source ability, but after gain you will see another sourceId,
12   * see Traveler's Cloak for workaround to trasfer settings
13   *
14   * @author LoneFox
15   */
16  public enum ChosenSubtypePredicate implements ObjectSourcePlayerPredicate<MageObject> {
17      TRUE(true), FALSE(false);
18  
19      private final boolean value;
20  
21      ChosenSubtypePredicate(boolean value) {
22          this.value = value;
23      }
24  
25      @Override
26      public boolean apply(ObjectSourcePlayer<MageObject> input, Game game) {
27          SubType subType = ChooseCreatureTypeEffect.getChosenCreatureType(input.getSourceId(), game);
28          return input.getObject().hasSubtype(subType, game) == value;
29      }
30  
31      @Override
32      public String toString() {
33          return "Chosen subtype";
34      }
35  }

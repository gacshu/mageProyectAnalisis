1   package mage.filter.predicate.card;
2   
3   import mage.cards.Card;
4   import mage.filter.StaticFilters;
5   import mage.filter.predicate.Predicate;
6   import mage.game.Game;
7   
8   /**
9    * @author Plopman, Alex-Vasile
10   */
11  public class CardManaCostLessThanControlledLandCountPredicate implements Predicate<Card> {
12  
13      private static final String string = "card with mana value less than or equal to the number of lands you control";
14      private static final CardManaCostLessThanControlledLandCountPredicate instance = new CardManaCostLessThanControlledLandCountPredicate();
15  
16      private CardManaCostLessThanControlledLandCountPredicate() { }
17  
18      public static CardManaCostLessThanControlledLandCountPredicate getInstance() {
19          return instance;
20      }
21  
22      @Override
23      public boolean apply(Card input, Game game) {
24          return input.getManaValue() <= game.getBattlefield().getAllActivePermanents(StaticFilters.FILTER_CONTROLLED_PERMANENT_LAND, input.getOwnerId(), game).size();
25      }
26  
27      @Override
28      public String toString() {
29          return string;
30      }
31  }

2   package mage.filter.predicate.card;
3   
4   import java.util.UUID;
5   import mage.cards.Card;
6   import mage.filter.predicate.Predicate;
7   import mage.game.Game;
8   import mage.target.Target;
9   
10  /**
11   *
12   * @author LevelIX
13   */
14  public class CardCanTargetPermanentId implements Predicate<Card> {
15  
16      private final UUID toBeCheckedPermanentId;
17  
18      public CardCanTargetPermanentId(UUID toBeCheckedPermanentId) {
19          this.toBeCheckedPermanentId = toBeCheckedPermanentId;
20      }
21  
22      @Override
23      public boolean apply(Card input, Game game) {
24          for (Target target : input.getSpellAbility().getTargets()) {
25              if (target.canTarget(toBeCheckedPermanentId, input.getSpellAbility(), game)) {
26                  return true;
27              }
28          }
29          return false;
30      }
31  
32      @Override
33      public String toString() {
34          return "CardCanTargetPermanentId(" + toBeCheckedPermanentId + ')';
35      }
36  }

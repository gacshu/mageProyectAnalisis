1   package mage.util.functions;
2   
3   import mage.MageObject;
4   import mage.abilities.Ability;
5   import mage.abilities.keyword.MorphAbility;
6   import mage.cards.Card;
7   import mage.constants.CardType;
8   import mage.constants.SuperType;
9   import mage.game.Game;
10  import mage.game.permanent.PermanentCard;
11  import mage.game.permanent.PermanentToken;
12  import mage.game.permanent.token.Token;
13  import mage.game.stack.Spell;
14  
15  /**
16   * @author nantuko
17   */
18  public class CopyTokenFunction implements Function<Token, Card> {
19  
20      protected Token target;
21  
22      public CopyTokenFunction(Token target) {
23          if (target == null) {
24              throw new IllegalArgumentException("Target can't be null");
25          }
26          this.target = target;
27      }
28  
29      @Override
30      public Token apply(Card source, Game game) {
31          if (target == null) {
32              throw new IllegalArgumentException("Target can't be null");
33          }
34          // A copy contains only the attributes of the basic card or basic Token that's the base of the permanent
35          // else gained abililies would be copied too.
36  
37          MageObject sourceObj = source;
38          if (source instanceof PermanentToken) {
39              sourceObj = ((PermanentToken) source).getToken();
40              // to show the source image, the original values have to be used
41              target.setOriginalExpansionSetCode(((Token) sourceObj).getOriginalExpansionSetCode());
42              target.setOriginalCardNumber(((Token) sourceObj).getOriginalCardNumber());
43              target.setCopySourceCard(((PermanentToken) source).getToken().getCopySourceCard());
44          } else if (source instanceof PermanentCard) {
45              if (((PermanentCard) source).isMorphed() || ((PermanentCard) source).isManifested()) {
46                  MorphAbility.setPermanentToFaceDownCreature(target, game);
47                  return target;
48              } else {
49                  if (((PermanentCard) source).isTransformed() && source.getSecondCardFace() != null) {
50                      sourceObj = ((PermanentCard) source).getSecondCardFace();
51                  } else {
52                      sourceObj = ((PermanentCard) source).getCard();
53                  }
54  
55                  target.setOriginalExpansionSetCode(source.getExpansionSetCode());
56                  target.setOriginalCardNumber(source.getCardNumber());
57                  target.setCopySourceCard((Card) sourceObj);
58              }
59          } else {
60              target.setOriginalExpansionSetCode(source.getExpansionSetCode());
61              target.setOriginalCardNumber(source.getCardNumber());
62              target.setCopySourceCard(source);
63          }
64  
65          // modify all attributes permanently (without game usage)
66          target.setName(sourceObj.getName());
67          target.getColor().setColor(sourceObj.getColor());
68          target.getManaCost().clear();
69          target.getManaCost().add(sourceObj.getManaCost().copy());
70          target.removeAllCardTypes();
71          for (CardType type : sourceObj.getCardType()) {
72              target.addCardType(type);
73          }
74          target.getSubtype().copyFrom(sourceObj.getSubtype());
75          target.getSuperType().clear();
76          for (SuperType type : sourceObj.getSuperType()) {
77              target.addSuperType(type);
78          }
79  
80          target.getAbilities().clear();
81  
82          for (Ability ability0 : sourceObj.getAbilities()) {
83              Ability ability = ability0.copy();
84  
85              // The token is independant from the copy from object so it need a new original Id,
86              // otherwise there are problems to check for created continuous effects to check if
87              // the source (the Token) has still this ability
88              ability.newOriginalId();
89  
90              target.addAbility(ability);
91          }
92  
93          target.setPower(sourceObj.getPower().getBaseValue());
94          target.setToughness(sourceObj.getToughness().getBaseValue());
95          target.setStartingLoyalty(sourceObj.getStartingLoyalty());
96  
97          return target;
98      }
99  
100     public Token from(Card source, Game game) {
101         return from(source, game, null);
102     }
103 
104     public Token from(Card source, Game game, Spell spell) {
105         apply(source, game);
106 
107         // token's ZCC must be synced with original card to keep abilities settings
108         // Example: kicker ability and kicked status
109         if (spell != null) {
110             // copied spell puts to battlefield as token, so that token's ZCC must be synced with spell instead card (card can be moved before resolve)
111             target.setZoneChangeCounter(spell.getZoneChangeCounter(game), game);
112             // Copy starting loyalty from spell (Ob Nixilis, the Adversary)
113             target.setStartingLoyalty(spell.getStartingLoyalty());
114         } else {
115             target.setZoneChangeCounter(source.getZoneChangeCounter(game), game);
116         }
117 
118         return target;
119     }
120 }

5   import mage.cards.Card;
6   import mage.cards.Cards;
7   import mage.cards.CardsImpl;
8   import mage.cards.MeldCard;
9   import mage.game.Game;
10  import mage.game.permanent.PermanentCard;
11  import mage.game.permanent.PermanentMeld;
12  
13  public class ExileUtil {
14    public static Cards returnCardsFromExile(Set<Card> cards, Game game) {
15      Cards cardsToReturn = new CardsImpl();
16      for (Card exiled : cards) {
17        if (exiled instanceof PermanentMeld) {
18          MeldCard meldCard = (MeldCard) ((PermanentCard) exiled).getCard();
19          Card topCard = meldCard.getTopHalfCard();
20          Card bottomCard = meldCard.getBottomHalfCard();
21          if (topCard.getZoneChangeCounter(game) == meldCard.getTopLastZoneChangeCounter()) {
22            cardsToReturn.add(topCard);
23          }
24          if (bottomCard.getZoneChangeCounter(game) == meldCard.getBottomLastZoneChangeCounter()) {
25            cardsToReturn.add(bottomCard);
26          }
27        } else if (exiled.getZoneChangeCounter(game) == game.getState().getZoneChangeCounter(exiled.getId()) - 1) {
28          cardsToReturn.add(exiled);
29        }
30      }
31  
32      return cardsToReturn;
33    }
34  }
